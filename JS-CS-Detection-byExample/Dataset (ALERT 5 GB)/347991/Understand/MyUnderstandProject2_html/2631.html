45     // Converts a list of FilePaths to a list of ui::SelectedFileInfo.
46     std::vector<ui::SelectedFileInfo> FilePathListToSelectedFileInfoList(
47         const std::vector<base::FilePath>& paths) {
48       std::vector<ui::SelectedFileInfo> selected_files;
49       for (size_t i = 0; i < paths.size(); ++i) {
50         selected_files.push_back(
51             ui::SelectedFileInfo(paths[i], paths[i]));

53       return selected_files;
54     }

79     FileSelectHelper::~FileSelectHelper() {
80       // There may be pending file dialogs, we need to tell them that we've gone
81       // away so they don't try and call back to us.
82       if (select_file_dialog_.get())
83         select_file_dialog_->ListenerDestroyed();
84     
85       // Stop any pending directory enumeration, prevent a callback, and free
86       // allocated memory.
87       std::map<int, ActiveDirectoryEnumeration*>::iterator iter;
88       for (iter = directory_enumerations_.begin();
89            iter != directory_enumerations_.end();
90            ++iter) {
91         iter->second->lister_.reset();
92         delete iter->second;
93       }
94     }

110     void FileSelectHelper::FileSelectedWithExtraInfo(
111         const ui::SelectedFileInfo& file,
112         int index,
113         void* params) {
114       if (!render_view_host_)
115         return;
116     
117       const base::FilePath& path = file.local_path;
118       if (dialog_type_ == ui::SelectFileDialog::SELECT_UPLOAD_FOLDER) {
119         StartNewEnumeration(path, kFileSelectEnumerationId, render_view_host_);
120         return;
121       }
122     
123       std::vector<ui::SelectedFileInfo> files;
124       files.push_back(file);
125       NotifyRenderViewHost(render_view_host_, files, dialog_mode_);
126     
127       // No members should be accessed from here on.
128       RunFileChooserEnd();
129     }

166     void FileSelectHelper::StartNewEnumeration(const base::FilePath& path,
167                                                int request_id,
168                                                RenderViewHost* render_view_host) {
169       scoped_ptr<ActiveDirectoryEnumeration> entry(new ActiveDirectoryEnumeration);
170       entry->rvh_ = render_view_host;
171       entry->delegate_.reset(new DirectoryListerDispatchDelegate(this, request_id));
172       entry->lister_.reset(new net::DirectoryLister(path,
173                                                     true,
174                                                     net::DirectoryLister::NO_SORT,
175                                                     entry->delegate_.get()));
176       if (!entry->lister_->Start()) {
177         if (request_id == kFileSelectEnumerationId)
178           FileSelectionCanceled(NULL);
179         else
180           render_view_host->DirectoryEnumerationFinished(request_id,
181                                                          entry->results_);
182       } else {
183         directory_enumerations_[request_id] = entry.release();
184       }
185     }

187     void FileSelectHelper::OnListFile(
188         int id,
189         const net::DirectoryLister::DirectoryListerData& data) {
190       ActiveDirectoryEnumeration* entry = directory_enumerations_[id];
191     
192       // Directory upload returns directories via a "." file, so that
193       // empty directories are included.  This util call just checks
194       // the flags in the structure; there's no file I/O going on.
195       if (data.info.IsDirectory())
196         entry->results_.push_back(data.path.Append(FILE_PATH_LITERAL(".")));
197       else
198         entry->results_.push_back(data.path);
199     }
200     
201     void FileSelectHelper::OnListDone(int id, int error) {
202       // This entry needs to be cleaned up when this function is done.
203       scoped_ptr<ActiveDirectoryEnumeration> entry(directory_enumerations_[id]);
204       directory_enumerations_.erase(id);
205       if (!entry->rvh_)
206         return;
207       if (error) {
208         FileSelectionCanceled(NULL);
209         return;
210       }
211     
212       std::vector<ui::SelectedFileInfo> selected_files =
213           FilePathListToSelectedFileInfoList(entry->results_);
214     
215       if (id == kFileSelectEnumerationId)
216         NotifyRenderViewHost(entry->rvh_, selected_files, dialog_mode_);
217       else
218         entry->rvh_->DirectoryEnumerationFinished(id, entry->results_);
219     
220       EnumerateDirectoryEnd();
221     }

223     scoped_ptr<ui::SelectFileDialog::FileTypeInfo>
224     FileSelectHelper::GetFileTypesFromAcceptType(
225         const std::vector<base::string16>& accept_types) {
226       scoped_ptr<ui::SelectFileDialog::FileTypeInfo> base_file_type(
227           new ui::SelectFileDialog::FileTypeInfo());
228       if (accept_types.empty())
229         return base_file_type.Pass();
230     
231       // Create FileTypeInfo and pre-allocate for the first extension list.
232       scoped_ptr<ui::SelectFileDialog::FileTypeInfo> file_type(
233           new ui::SelectFileDialog::FileTypeInfo(*base_file_type));
234       file_type->include_all_files = true;
235       file_type->extensions.resize(1);
236       std::vector<base::FilePath::StringType>* extensions =
237           &file_type->extensions.back();
238     
239       // Find the corresponding extensions.
240       int valid_type_count = 0;
241       int description_id = 0;
242       for (size_t i = 0; i < accept_types.size(); ++i) {
243         std::string ascii_type = UTF16ToASCII(accept_types[i]);
244         if (!IsAcceptTypeValid(ascii_type))
245           continue;
246     
247         size_t old_extension_size = extensions->size();
248         if (ascii_type[0] == '.') {
249           // If the type starts with a period it is assumed to be a file extension
250           // so we just have to add it to the list.
251           base::FilePath::StringType ext(ascii_type.begin(), ascii_type.end());
252           extensions->push_back(ext.substr(1));
253         } 
254         else {
255           /* TODO(spolu): FixMe */
256           /*
257           if (ascii_type == "image*")
258             description_id = IDS_IMAGE_FILES;
259           else if (ascii_type == "audio*")
260             description_id = IDS_AUDIO_FILES;
261           else if (ascii_type == "video*")
262             description_id = IDS_VIDEO_FILES;
263           */
264           net::GetExtensionsForMimeType(ascii_type, extensions);
265         }
266     
267         if (extensions->size() > old_extension_size)
268           valid_type_count++;
269       }
270     
271       // If no valid extension is added, bail out.
272       if (valid_type_count == 0)
273         return base_file_type.Pass();
274     
275       // Use a generic description "Custom Files" if either of the following is
276       // true:
277       // 1) There're multiple types specified, like "audio/*,video/*"
278       // 2) There're multiple extensions for a MIME type without parameter, like
279       //    "ehtml,shtml,htm,html" for "text/html". On Windows, the select file
280       //    dialog uses the first extension in the list to form the description,
281       //    like "EHTML Files". This is not what we want.
282       /* TODO(spolu): FixMe */
283       /*
284       if (valid_type_count > 1 ||
285           (valid_type_count == 1 && description_id == 0 && extensions->size() > 1))
286         description_id = IDS_CUSTOM_FILES;
287       */
288     
289       if (description_id) {
290         file_type->extension_description_overrides.push_back(
291             l10n_util::GetStringUTF16(description_id));
292       }
293     
294       return file_type.Pass();
295     }

317     void FileSelectHelper::RunFileChooser(RenderViewHost* render_view_host,
318                                           content::WebContents* web_contents,
319                                           const FileChooserParams& params) {
320       DCHECK(!render_view_host_);
321       DCHECK(!web_contents_);
322       render_view_host_ = render_view_host;
323       web_contents_ = web_contents;
324       notification_registrar_.RemoveAll();
325       notification_registrar_.Add(
326           this, content::NOTIFICATION_RENDER_WIDGET_HOST_DESTROYED,
327           content::Source<RenderWidgetHost>(render_view_host_));
328       notification_registrar_.Add(
329           this, content::NOTIFICATION_WEB_CONTENTS_DESTROYED,
330           content::Source<WebContents>(web_contents_));
331     
332       BrowserThread::PostTask(
333           BrowserThread::FILE, FROM_HERE,
334           base::Bind(&FileSelectHelper::RunFileChooserOnFileThread, this, params));
335     
336       // Because this class returns notifications to the RenderViewHost, it is
337       // difficult for callers to know how long to keep a reference to this
338       // instance. We AddRef() here to keep the instance alive after we return
339       // to the caller, until the last callback is received from the file dialog.
340       // At that point, we must call RunFileChooserEnd().
341       AddRef();
342     }

344     void FileSelectHelper::RunFileChooserOnFileThread(
345         const FileChooserParams& params) {
346       select_file_types_ = GetFileTypesFromAcceptType(params.accept_types);
347     
348       BrowserThread::PostTask(
349           BrowserThread::UI, FROM_HERE,
350           base::Bind(&FileSelectHelper::RunFileChooserOnUIThread, this, params));
351     }

353     void FileSelectHelper::RunFileChooserOnUIThread(
354         const FileChooserParams& params) {
355       if (!render_view_host_ || !web_contents_) {
356         // If the renderer was destroyed before we started, just cancel the
357         // operation.
358         RunFileChooserEnd();
359         return;
360       }
361     
362       select_file_dialog_ = ui::SelectFileDialog::Create(this, NULL);
363     
364       dialog_mode_ = params.mode;
365       switch (params.mode) {
366         case FileChooserParams::Open:
367           dialog_type_ = ui::SelectFileDialog::SELECT_OPEN_FILE;
368           break;
369         case FileChooserParams::OpenMultiple:
370           dialog_type_ = ui::SelectFileDialog::SELECT_OPEN_MULTI_FILE;
371           break;
372         case FileChooserParams::UploadFolder:
373           dialog_type_ = ui::SelectFileDialog::SELECT_UPLOAD_FOLDER;
374           break;
375         case FileChooserParams::Save:
376           dialog_type_ = ui::SelectFileDialog::SELECT_SAVEAS_FILE;
377           break;
378         default:
379           // Prevent warning.
380           dialog_type_ = ui::SelectFileDialog::SELECT_OPEN_FILE;
381           NOTREACHED();
382       }
383     
384       base::FilePath default_file_name = params.default_file_name;
385     
386       gfx::NativeWindow owning_window =
387           platform_util::GetTopLevel(render_view_host_->GetView()->GetNativeView());
388     
389       select_file_dialog_->SelectFile(
390           dialog_type_,
391           params.title,
392           default_file_name,
393           select_file_types_.get(),
394           select_file_types_.get() && !select_file_types_->extensions.empty()
395               ? 1
396               : 0,  // 1-based index of default extension to show.
397           base::FilePath::StringType(),
398           owning_window,
399           NULL); //const_cast<content::FileChooserParams*>(&params),
400     
401       select_file_types_.reset();
402     }

404     // This method is called when we receive the last callback from the file
405     // chooser dialog. Perform any cleanup and release the reference we added
406     // in RunFileChooser().
407     void FileSelectHelper::RunFileChooserEnd() {
408       render_view_host_ = NULL;
409       web_contents_ = NULL;
410       Release();
411     }

433     void FileSelectHelper::Observe(int type,
434                                    const content::NotificationSource& source,
435                                    const content::NotificationDetails& details) {
436       switch (type) {
437         case content::NOTIFICATION_RENDER_WIDGET_HOST_DESTROYED: {
438           DCHECK(content::Source<RenderWidgetHost>(source).ptr() ==
439                  render_view_host_);
440           render_view_host_ = NULL;
441           break;
442         }
443     
444         case content::NOTIFICATION_WEB_CONTENTS_DESTROYED: {
445           DCHECK(content::Source<WebContents>(source).ptr() == web_contents_);
446           web_contents_ = NULL;
447           break;
448         }
449     
450         default:
451           NOTREACHED();
452       }
453     }

455     // static
456     bool FileSelectHelper::IsAcceptTypeValid(const std::string& accept_type) {
457       // TODO(raymes): This only does some basic checks, extend to test more cases.
458       // A 1 character accept type will always be invalid (either a "." in the case
459       // of an extension or a "/" in the case of a MIME type).
460       std::string unused;
461       if (accept_type.length() <= 1 ||
462           StringToLowerASCII(accept_type) != accept_type ||
463           TrimWhitespaceASCII(accept_type, base::TRIM_ALL, &unused) != base::TRIM_NONE) {
464         return false;
465       }
466       return true;
467     }
