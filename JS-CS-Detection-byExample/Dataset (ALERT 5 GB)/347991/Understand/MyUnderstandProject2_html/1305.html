72     void 
73     ThrustShellRendererClient::RenderThreadStarted() 
74     {
75       observer_.reset(new ThrustShellRenderProcessObserver());
76       visited_link_slave_.reset(new visitedlink::VisitedLinkSlave());
77       if (!extension_dispatcher_)
78         extension_dispatcher_.reset(new extensions::Dispatcher());
79     #if defined(OS_MACOSX)
80       // We need to call this once before the sandbox was initialized to cache the
81       // value.
82       base::debug::BeingDebugged();
83     #endif
84     
85       RenderThread* thread = RenderThread::Get();
86       thread->AddObserver(visited_link_slave_.get());
87       thread->AddObserver(extension_dispatcher_.get());
88     }
