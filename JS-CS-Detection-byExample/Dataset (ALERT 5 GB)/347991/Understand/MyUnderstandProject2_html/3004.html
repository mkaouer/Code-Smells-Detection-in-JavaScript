23     bool 
24     ThrustMenu::IsCommandIdChecked(
25         int command_id) const 
26     {
27       if(checked_.find(command_id) != checked_.end()) {
28         return checked_.find(command_id)->second;
29       }
30       return false;
31     }
32     
33     bool 
34     ThrustMenu::IsCommandIdEnabled(
35         int command_id) const 
36     {
37       if(enabled_.find(command_id) != enabled_.end()) {
38         return enabled_.find(command_id)->second;
39       }
40       return true;
41     }

43     bool 
44     ThrustMenu::IsCommandIdVisible(

49       }

53     bool 
54     ThrustMenu::GetAcceleratorForCommandId(
55         int command_id,
56         ui::Accelerator* accelerator) 
57     {
58       if(accelerator_.find(command_id) != accelerator_.end()) {
59         return accelerator_util::StringToAccelerator(accelerator_[command_id], accelerator);
60       }
61       return false;
62     }
