37     // Controls the link coloring database. The master controls all writing to the
38     // database as well as disk I/O. There should be only one master.
39     //
40     // This class will defer writing operations to the file thread. This means that
41     // class destruction, the file may still be open since operations are pending on
42     // another thread.
43     class VisitedLinkMaster : public VisitedLinkCommon {
44      public:
45       // Listens to the link coloring database events. The master is given this
46       // event as a constructor argument and dispatches events using it.
47       class Listener {
48        public:
49         virtual ~Listener() {}
50     
51         // Called when link coloring database has been created or replaced. The
52         // argument is the new table handle.
53         virtual void NewTable(base::SharedMemory*) = 0;
54     
55         // Called when new link has been added. The argument is the fingerprint
56         // (hash) of the link.
57         virtual void Add(Fingerprint fingerprint) = 0;
58     
59         // Called when link coloring state has been reset. This may occur when
60         // entire or parts of history were deleted.
61         virtual void Reset() = 0;
62       };
63     
64       VisitedLinkMaster(content::BrowserContext* browser_context,
65                         VisitedLinkDelegate* delegate,
66                         bool persist_to_disk);
67     
68       // In unit test mode, we allow the caller to optionally specify the database
69       // filename so that it can be run from a unit test. The directory where this
70       // file resides must exist in this mode. You can also specify the default
71       // table size to test table resizing. If this parameter is 0, we will use the
72       // defaults.
73       //
74       // In the unit test mode, we also allow the caller to provide a history
75       // service pointer (the history service can't be fetched from the browser
76       // process when we're in unit test mode). This can be NULL to try to access
77       // the main version, which will probably fail (which can be good for testing
78       // this failure mode).
79       //
80       // When |suppress_rebuild| is set, we'll not attempt to load data from
81       // history if the file can't be loaded. This should generally be set for
82       // testing except when you want to test the rebuild process explicitly.
83       VisitedLinkMaster(Listener* listener,
84                         VisitedLinkDelegate* delegate,
85                         bool persist_to_disk,
86                         bool suppress_rebuild,
87                         const base::FilePath& filename,
88                         int32 default_table_size);
89       virtual ~VisitedLinkMaster();
90     
91       // Must be called immediately after object creation. Nothing else will work
92       // until this is called. Returns true on success, false means that this
93       // object won't work.
94       bool Init();
95     
96       base::SharedMemory* shared_memory() { return shared_memory_; }
97     
98       // Adds a URL to the table.
99       void AddURL(const GURL& url);
100     
101       // Adds a set of URLs to the table.
102       void AddURLs(const std::vector<GURL>& url);
103     
104       // See DeleteURLs.
105       class URLIterator {
106        public:
107         // HasNextURL must return true when this is called. Returns the next URL
108         // then advances the iterator. Note that the returned reference is only
109         // valid until the next call of NextURL.
110         virtual const GURL& NextURL() = 0;
111     
112         // Returns true if still has URLs to be iterated.
113         virtual bool HasNextURL() const = 0;
114     
115        protected:
116         virtual ~URLIterator() {}
117       };
118     
119       // Deletes the specified URLs from |rows| from the table.
120       void DeleteURLs(URLIterator* iterator);
121     
122       // Clears the visited links table by deleting the file from disk. Used as
123       // part of history clearing.
124       void DeleteAllURLs();
125     
126       // Returns the Delegate of this Master.
127       VisitedLinkDelegate* GetDelegate();
128     
129     #if defined(UNIT_TEST) || !defined(NDEBUG) || defined(PERF_TEST)
130       // This is a debugging function that can be called to double-check internal
131       // data structures. It will assert if the check fails.
132       void DebugValidate();
133     
134       // Sets a task to execute when the next rebuild from history is complete.
135       // This is used by unit tests to wait for the rebuild to complete before
136       // they continue. The pointer will be owned by this object after the call.
137       void set_rebuild_complete_task(const base::Closure& task) {
138         DCHECK(rebuild_complete_task_.is_null());
139         rebuild_complete_task_ = task;
140       }
141     
142       // returns the number of items in the table for testing verification
143       int32 GetUsedCount() const {
144         return used_items_;
145       }
146     
147       // Returns the listener.
148       VisitedLinkMaster::Listener* GetListener() const {
149         return listener_.get();
150       }
151     
152       // Call to cause the entire database file to be re-written from scratch
153       // to disk. Used by the performance tester.
154       void RewriteFile() {
155         WriteFullTable();
156       }
157     #endif

187       // individual writes for each of them. When the count exceeds this threshold,
188       // we will write the whole table to disk at once instead of individual items.
189       static const size_t kBigDeleteThreshold;

192       void InitMembers();
193     
194       // If a rebuild is in progress, we save the URL in the temporary list.
195       // Otherwise, we add this to the table. Returns the index of the
196       // inserted fingerprint or null_hash_ on failure.
197       Hash TryToAddURL(const GURL& url);

199       // File I/O functions
200       // ------------------
201       // These functions are only called if |persist_to_disk_| is true.
202     
203       // Posts the given task to the blocking worker pool with our options.
204       void PostIOTask(const tracked_objects::Location& from_here,
205                       const base::Closure& task);
206     
207       // Writes the entire table to disk. It will leave the table file open and
208       // the handle to it will be stored in file_.
209       void WriteFullTable();
210     
211       // Try to load the table from the database file. If the file doesn't exist or
212       // is corrupt, this will return failure.
213       bool InitFromFile();
214     
215       // Reads the header of the link coloring database from disk. Assumes the
216       // file pointer is at the beginning of the file and that there are no pending
217       // asynchronous I/O operations.
218       //
219       // Returns true on success and places the size of the table in num_entries
220       // and the number of nonzero fingerprints in used_count. This will fail if
221       // the version of the file is not the current version of the database.
222       bool ReadFileHeader(FILE* hfile, int32* num_entries, int32* used_count,
223                           uint8 salt[LINK_SALT_LENGTH]);
224     
225       // Fills *filename with the name of the link database filename
226       bool GetDatabaseFileName(base::FilePath* filename);
227     
228       // Wrapper around Window's WriteFile using asynchronous I/O. This will proxy
229       // the write to a background thread.
230       void WriteToFile(FILE** hfile, off_t offset, void* data, int32 data_size);
231     
232       // Helper function to schedule and asynchronous write of the used count to
233       // disk (this is a common operation).
234       void WriteUsedItemCountToFile();
235     
236       // Helper function to schedule an asynchronous write of the given range of
237       // hash functions to disk. The range is inclusive on both ends. The range can
238       // wrap around at 0 and this function will handle it.
239       void WriteHashRangeToFile(Hash first_hash, Hash last_hash);
240     
241       // Synchronous read from the file. Assumes there are no pending asynchronous
242       // I/O functions. Returns true if the entire buffer was successfully filled.
243       bool ReadFromFile(FILE* hfile, off_t offset, void* data, size_t data_size);
244     
245       // General table handling
246       // ----------------------
247     
248       // Called to add a fingerprint to the table. If |send_notifications| is true
249       // and the item is added successfully, Listener::Add will be invoked.
250       // Returns the index of the inserted fingerprint or null_hash_ if there was a
251       // duplicate and this item was skippped.
252       Hash AddFingerprint(Fingerprint fingerprint, bool send_notifications);
253     
254       // Deletes all fingerprints from the given vector from the current hash table
255       // and syncs it to disk if there are changes. This does not update the
256       // deleted_since_rebuild_ list, the caller must update this itself if there
257       // is an update pending.
258       void DeleteFingerprintsFromCurrentTable(
259           const std::set<Fingerprint>& fingerprints);
260     
261       // Removes the indicated fingerprint from the table. If the update_file flag
262       // is set, the changes will also be written to disk. Returns true if the
263       // fingerprint was deleted, false if it was not in the table to delete.
264       bool DeleteFingerprint(Fingerprint fingerprint, bool update_file);
265     
266       // Creates a new empty table, call if InitFromFile() fails. Normally, when
267       // |suppress_rebuild| is false, the table will be rebuilt from history,
268       // keeping us in sync. When |suppress_rebuild| is true, the new table will be
269       // empty and we will not consult history. This is used when clearing the
270       // database and for unit tests.
271       bool InitFromScratch(bool suppress_rebuild);
272     
273       // Allocates the Fingerprint structure and length. When init_to_empty is set,
274       // the table will be filled with 0s and used_items_ will be set to 0 as well.
275       // If the flag is not set, these things are untouched and it is the
276       // responsibility of the caller to fill them (like when we are reading from
277       // a file).
278       bool CreateURLTable(int32 num_entries, bool init_to_empty);
279     
280       // A wrapper for CreateURLTable, this will allocate a new table, initialized
281       // to empty. The caller is responsible for saving the shared memory pointer
282       // and handles before this call (they will be replaced with new ones) and
283       // releasing them later. This is designed for callers that make a new table
284       // and then copy values from the old table to the new one, then release the
285       // old table.
286       //
287       // Returns true on success. On failure, the old table will be restored. The
288       // caller should not attemp to release the pointer/handle in this case.
289       bool BeginReplaceURLTable(int32 num_entries);
290     
291       // unallocates the Fingerprint table
292       void FreeURLTable();
293     
294       // For growing the table. ResizeTableIfNecessary will check to see if the
295       // table should be resized and calls ResizeTable if needed. Returns true if
296       // we decided to resize the table.
297       bool ResizeTableIfNecessary();
298     
299       // Resizes the table (growing or shrinking) as necessary to accomodate the
300       // current count.
301       void ResizeTable(int32 new_size);
302     
303       // Returns the desired table size for |item_count| URLs.
304       uint32 NewTableSizeForCount(int32 item_count) const;
305     
306       // Computes the table load as fraction. For example, if 1/4 of the entries are
307       // full, this value will be 0.25
308       float ComputeTableLoad() const {
309         return static_cast<float>(used_items_) / static_cast<float>(table_length_);
310       }
311     
312       // Initializes a rebuild of the visited link database based on the browser
313       // history. This will set table_builder_ while working, and there should not
314       // already be a rebuild in place when called. See the definition for more
315       // details on how this works.
316       //
317       // Returns true on success. Failure means we're not attempting to rebuild
318       // the database because something failed.
319       bool RebuildTableFromDelegate();
320     
321       // Callback that the table rebuilder uses when the rebuild is complete.
322       // |success| is true if the fingerprint generation succeeded, in which case
323       // |fingerprints| will contain the computed fingerprints. On failure, there
324       // will be no fingerprints.
325       void OnTableRebuildComplete(bool success,
326                                   const std::vector<Fingerprint>& fingerprints);
327     
328       // Increases or decreases the given hash value by one, wrapping around as
329       // necessary. Used for probing.
330       inline Hash IncrementHash(Hash hash) {
331         if (hash >= table_length_ - 1)
332           return 0;  // Wrap around.
333         return hash + 1;
334       }
335       inline Hash DecrementHash(Hash hash) {
336         if (hash <= 0)
337           return table_length_ - 1;  // Wrap around.
338         return hash - 1;
339       }

341     #ifndef NDEBUG
342       // Indicates whether any asynchronous operation has ever been completed.
343       // We do some synchronous reads that require that no asynchronous operations
344       // are pending, yet we don't track whether they have been completed. This
345       // flag is a sanity check that these reads only happen before any
346       // asynchronous writes have been fired.
347       bool posted_asynchronous_operation_;
348     #endif

350       // Reference to the browser context that this object belongs to
351       // (it knows the path to where the data is stored)
352       content::BrowserContext* browser_context_;
353     
354       // Client owns the delegate and is responsible for it being valid through
355       // the life time this VisitedLinkMaster.
356       VisitedLinkDelegate* delegate_;
357     
358       // VisitedLinkEventListener to handle incoming events.
359       scoped_ptr<Listener> listener_;
360     
361       // Lazily initialized sequence token for posting file tasks.
362       base::SequencedWorkerPool::SequenceToken sequence_token_;
363     
364       // When non-NULL, indicates we are in database rebuild mode and points to
365       // the class collecting fingerprint information from the history system.
366       // The pointer is owned by this class, but it must remain valid while the
367       // history query is running. We must only delete it when the query is done.
368       scoped_refptr<TableBuilder> table_builder_;
369     
370       // Indicates URLs added and deleted since we started rebuilding the table.
371       std::set<Fingerprint> added_since_rebuild_;
372       std::set<Fingerprint> deleted_since_rebuild_;
373     
374       // TODO(brettw) Support deletion, we need to track whether anything was
375       // deleted during the rebuild here. Then we should delete any of these
376       // entries from the complete table later.
377       // std::vector<Fingerprint> removed_since_rebuild_;
378     
379       // The currently open file with the table in it. This may be NULL if we're
380       // rebuilding and haven't written a new version yet or if |persist_to_disk_|
381       // is false. Writing to the file may be safely ignored in this case. Also
382       // |file_| may be non-NULL but point to a NULL pointer. That would mean that
383       // opening of the file is already scheduled in a background thread and any
384       // writing to the file can also be scheduled to the background thread as it's
385       // guaranteed to be executed after the opening.
386       // The class owns both the |file_| pointer and the pointer pointed
387       // by |*file_|.
388       FILE** file_;
389     
390       // If true, will try to persist the hash table to disk. Will rebuild from
391       // VisitedLinkDelegate::RebuildTable if there are disk corruptions.
392       bool persist_to_disk_;
393     
394       // Shared memory consists of a SharedHeader followed by the table.
395       base::SharedMemory *shared_memory_;
396     
397       // When we generate new tables, we increment the serial number of the
398       // shared memory object.
399       int32 shared_memory_serial_;
400     
401       // Number of non-empty items in the table, used to compute fullness.
402       int32 used_items_;
403     
404       // Testing values -----------------------------------------------------------
405       //
406       // The following fields exist for testing purposes. They are not used in
407       // release builds. It'd be nice to eliminate them in release builds, but we
408       // don't want to change the signature of the object between the unit test and
409       // regular builds. Instead, we just have "default" values that these take
410       // in release builds that give "regular" behavior.
411     
412       // Overridden database file name for testing
413       base::FilePath database_name_override_;
414     
415       // When nonzero, overrides the table size for new databases for testing
416       int32 table_size_override_;
417     
418       // When set, indicates the task that should be run after the next rebuild from
419       // history is complete.
420       base::Closure rebuild_complete_task_;
421     
422       // Set to prevent us from attempting to rebuild the database from global
423       // history if we have an error opening the file. This is used for testing,
424       // will be false in production.
425       bool suppress_rebuild_;
426     
427       DISALLOW_COPY_AND_ASSIGN(VisitedLinkMaster);
428     };

433     #if defined(UNIT_TEST) || defined(PERF_TEST) || !defined(NDEBUG)
434     inline void VisitedLinkMaster::DebugValidate() {
435       int32 used_count = 0;
436       for (int32 i = 0; i < table_length_; i++) {
437         if (hash_table_[i])
438           used_count++;
439       }
440       DCHECK_EQ(used_count, used_items_);
441     }
