32     net::StreamListenSocketFactory* CreateSocketFactory() {
33       const CommandLine& command_line = *CommandLine::ForCurrentProcess();
34       // See if the user specified a port on the command line (useful for
35       // automation). If not, use an ephemeral port by specifying 0.
36       int port = 0;
37       if(command_line.HasSwitch(switches::kRemoteDebuggingPort)) {
38         int temp_port;
39         std::string port_str =
40           command_line.GetSwitchValueASCII(switches::kRemoteDebuggingPort);
41         if(base::StringToInt(port_str, &temp_port) &&
42             temp_port > 0 && temp_port < 65535) {
43           port = temp_port;
44         } 
45         else {
46           DLOG(WARNING) << "Invalid http debugger port number " << temp_port;
47         }
48       }
49       return new net::TCPListenSocketFactory("127.0.0.1", port);
50     }

53     class Target : public content::DevToolsTarget {
54      public:
55       explicit Target(WebContents* web_contents);
56     
57       virtual std::string GetId() const OVERRIDE { return id_; }
58       virtual std::string GetParentId() const OVERRIDE { return std::string(); }
59       virtual std::string GetType() const OVERRIDE { return kTargetTypePage; }
60       virtual std::string GetTitle() const OVERRIDE { return title_; }
61       virtual std::string GetDescription() const OVERRIDE { return std::string(); }
62       virtual GURL GetURL() const OVERRIDE { return url_; }
63       virtual GURL GetFaviconURL() const OVERRIDE { return favicon_url_; }
64       virtual base::TimeTicks GetLastActivityTime() const OVERRIDE {
65         return last_activity_time_;
66       }
67       virtual bool IsAttached() const OVERRIDE {
68         return agent_host_->IsAttached();
69       }
70       virtual scoped_refptr<DevToolsAgentHost> GetAgentHost() const OVERRIDE {
71         return agent_host_;
72       }
73       virtual bool Activate() const OVERRIDE;
74       virtual bool Close() const OVERRIDE;
75     
76      private:
77       scoped_refptr<DevToolsAgentHost> agent_host_;
78       std::string id_;
79       std::string title_;
80       GURL url_;
81       GURL favicon_url_;
82       base::TimeTicks last_activity_time_;
83     };
84     
85     Target::Target(
86         WebContents* web_contents) 
87     {
88       agent_host_ =
89           DevToolsAgentHost::GetOrCreateFor(web_contents->GetRenderViewHost());
90       id_ = agent_host_->GetId();
91       title_ = base::UTF16ToUTF8(web_contents->GetTitle());
92       url_ = web_contents->GetURL();
93       content::NavigationController& controller = web_contents->GetController();
94       content::NavigationEntry* entry = controller.GetActiveEntry();
95       if (entry != NULL && entry->GetURL().is_valid())
96         favicon_url_ = entry->GetFavicon().url;
97       last_activity_time_ = web_contents->GetLastActiveTime();
98     }
99      
100     bool 
101     Target::Activate() const 
102     {
103       RenderViewHost* rvh = agent_host_->GetRenderViewHost();
104       if (!rvh)
105         return false;
106       WebContents* web_contents = WebContents::FromRenderViewHost(rvh);
107       if (!web_contents)
108         return false;
109       web_contents->GetDelegate()->ActivateContents(web_contents);
110       return true;
111     }
112     
113     bool 
114     Target::Close() const 
115     {
116       RenderViewHost* rvh = agent_host_->GetRenderViewHost();
117       if (!rvh)
118         return false;
119       rvh->ClosePage();
120       return true;
121     }

179     void 
180     ThrustShellDevToolsDelegate::EnumerateTargets(TargetCallback callback) {
181       TargetList targets;
182       std::vector<RenderViewHost*> rvh_list =
183         content::DevToolsAgentHost::GetValidRenderViewHosts();
184       for(std::vector<RenderViewHost*>::iterator it = rvh_list.begin();
185           it != rvh_list.end(); ++it) {
186         WebContents* web_contents = WebContents::FromRenderViewHost(*it);
187         if(web_contents) {
188           /* TODO(spolu): FixMe */
189           /*
190           ExoFrame* frame = ExoFrame::ExoFrameForWebContents(web_contents);
191           if(frame->session() == session_) {
192             targets.push_back(new Target(web_contents));
193           }
194           */
195           targets.push_back(new Target(web_contents));
196         }
197       }
198       callback.Run(targets);
199     }
