27     // Writes |message| to stack to show up in minidump, then crashes.
28     void CheckWithMinidump(const std::string& message) {
29       char minidump[1024];
30       base::debug::Alias(&minidump);
31       base::snprintf(
32           minidump, arraysize(minidump), "e::console: %s", message.c_str());
33       CHECK(false) << message;
34     }

36     typedef void (*LogMethod)(v8::Handle<v8::Context> context,
37                               const std::string& message);
38     
39     void BoundLogMethodCallback(const v8::FunctionCallbackInfo<v8::Value>& info) {
40       LogMethod log_method =
41           reinterpret_cast<LogMethod>(info.Data().As<v8::External>()->Value());
42       std::string message;
43       for (int i = 0; i < info.Length(); ++i) {
44         if (i > 0)
45           message += " ";
46         message += *v8::String::Utf8Value(info[i]);
47       }
48       (*log_method)(info.GetIsolate()->GetCallingContext(), message);
49     }
50     
51     void BindLogMethod(v8::Isolate* isolate,
52                        v8::Local<v8::Object> target,
53                        const std::string& name,
54                        LogMethod log_method) {
55       v8::Local<v8::FunctionTemplate> tmpl = v8::FunctionTemplate::New(
56           isolate,
57           &BoundLogMethodCallback,
58           v8::External::New(isolate, reinterpret_cast<void*>(log_method)));
59       target->Set(v8::String::NewFromUtf8(isolate, name.c_str()),
60                   tmpl->GetFunction());
61     }

86     void AddMessage(content::RenderView* render_view,
87                     content::ConsoleMessageLevel level,
88                     const std::string& message) {
89       blink::WebView* web_view = render_view->GetWebView();
90       if (!web_view || !web_view->mainFrame())
91         return;
92       blink::WebConsoleMessage::Level target_level =
93           blink::WebConsoleMessage::LevelLog;
94       switch (level) {
95         case content::CONSOLE_MESSAGE_LEVEL_DEBUG:
96           target_level = blink::WebConsoleMessage::LevelDebug;
97           break;
98         case content::CONSOLE_MESSAGE_LEVEL_LOG:
99           target_level = blink::WebConsoleMessage::LevelLog;
100           break;
101         case content::CONSOLE_MESSAGE_LEVEL_WARNING:
102           target_level = blink::WebConsoleMessage::LevelWarning;
103           break;
104         case content::CONSOLE_MESSAGE_LEVEL_ERROR:
105           target_level = blink::WebConsoleMessage::LevelError;
106           break;
107       }
108       web_view->mainFrame()->addMessageToConsole(
109           blink::WebConsoleMessage(target_level, base::UTF8ToUTF16(message)));
110     }

133     void AddMessage(v8::Handle<v8::Context> context,
134                     content::ConsoleMessageLevel level,
135                     const std::string& message) {
136       if (context.IsEmpty()) {
137         LOG(WARNING) << "Could not log \"" << message << "\": no context given";
138         return;
139       }
140       ScriptContext* sc = ScriptContext::FromV8Context(context);
141       if(!sc) {
142         LOG(WARNING) << "Could not log \"" << message << "\": no script context found";
143         return;
144       }
145       content::RenderFrame* render_frame =
146         content::RenderFrame::FromWebFrame(sc->web_frame());
147       if(!render_frame) {
148         LOG(WARNING) << "Could not log \"" << message << "\": no render frame found";
149         return;
150       }
151       content::RenderView* render_view = render_frame->GetRenderView();
152       if(!render_view) {
153         LOG(WARNING) << "Could not log \"" << message << "\": no render view found";
154         return;
155       }
156       AddMessage(render_view, level, message);
157     }

159     v8::Local<v8::Object> AsV8Object() {
160       v8::Isolate* isolate = v8::Isolate::GetCurrent();
161       v8::EscapableHandleScope handle_scope(isolate);
162       v8::Local<v8::Object> console_object = v8::Object::New(isolate);
163       BindLogMethod(isolate, console_object, "debug", &Debug);
164       BindLogMethod(isolate, console_object, "log", &Log);
165       BindLogMethod(isolate, console_object, "warn", &Warn);
166       BindLogMethod(isolate, console_object, "error", &Error);
167       return handle_scope.Escape(console_object);
168     }
