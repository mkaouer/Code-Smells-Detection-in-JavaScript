23     // ## WebViewGuest
24     //
25     // Browser-side API implementation for the <webview> tag. It's in charge of
26     // maintaining the association between the guest WebContents and the embedder
27     // WebContents. Receives events issued from the guest and realys them to the
28     // embedder.
29     //
30     // A WebViewGuest is also a WebContentsObserver on the guest WebContents of the
31     // associated <webview> tag. WebViewGuest is created on attachment. That is, when a 
32     // guest WebContents is associated with a particular embedder WebContents. 
33     // This happens on either initial navigation or through the use of the New 
34     // Window API, when a new window is attached to a particular <webview>.
35     class WebViewGuest : public content::BrowserPluginGuestDelegate,
36                          public content::NotificationObserver,
37                          public content::WebContentsDelegate,
38                          public content::WebContentsObserver {
39     public:
40     
41       /****************************************************************************/
42       /* STATIC API */
43       /****************************************************************************/
44       static WebViewGuest* Create(int guest_instance_id);
45     
46       static WebViewGuest* FromWebContents(content::WebContents* web_contents);
47     
48       // Returns guestview::kInstanceIDNone if |contents| does not correspond to a
49       // WebViewGuest.
50       static int GetViewInstanceId(content::WebContents* contents);
51       // Parses partition related parameters from |extra_params|.
52       // |storage_partition_id| is the parsed partition ID and |persist_storage|
53       // specifies whether or not the partition is in memory.
54       static void ParsePartitionParam(const base::DictionaryValue* extra_params,
55                                       std::string* storage_partition_id,
56                                       bool* persist_storage);

66       virtual void GuestSizeChanged(const gfx::Size& old_size,
67                                     const gfx::Size& new_size) OVERRIDE FINAL;
68       virtual void RegisterDestructionCallback(
69           const DestructionCallback& callback) OVERRIDE FINAL;
70       virtual void WillAttach(
71           content::WebContents* embedder_web_contents,

73     
74       virtual content::WebContents* CreateNewGuestWindow(
75           const content::WebContents::CreateParams& create_params) OVERRIDE;

81       */
82

84       /* NOTIFICATION_OBSERVER IMPLEMENTATION */
85       /****************************************************************************/

87                            const content::NotificationSource& source,
88                            const content::NotificationDetails& details) OVERRIDE;

90       /****************************************************************************/
91       /* WEBVIEW API */
92       /****************************************************************************/
93       // ### LoadUrl
94       //
95       // Loads the specified url
96       void LoadUrl(const GURL& url);
97     
98       // ### SetZoom
99       //
100       // Set the zoom factor.
101       void SetZoom(double zoom_factor);
102     
103       // ### GetZoom
104       //
105       // Returns the current zoom factor.
106       double GetZoom();
107     
108       // ### Go
109       //
110       // If possible, navigate the guest to |relative_index| entries away from the
111       // current navigation entry.
112       // ```
113       // @relative_index {int}
114       // ```
115       void Go(int relative_index);
116     
117       // ### Reload
118       //
119       // Reload the guest.
120       // ```
121       // @ignore_cache {bool}
122       // ```
123       void Reload(bool ignore_cache);
124     
125       // ### Stop
126       //
127       // Stop loading the guest.
128       void Stop();
129     
130       // ### Terminate
131       //
132       // Terminates the renderer process for this guest webview
133       void Terminate();
134     
135       // ### Find
136       //
137       // Searches for a string in the guesst webview
138       //
139       //```
140       // @request_id  {int} the request_id for this request
141       // @search_text {string} the search text
142       // @options     {WebFindOptions} the find options
143       // ```
144       void Find(int request_id, 
145                 const std::string& search_text,
146                 const blink::WebFindOptions& options);
147     
148       //  ### StopFinding
149       //
150       //  Stops a findin query and specifiy which action to perform
151       //  ```
152       //  @action {StopFindingAction} the action to perform
153       //  ```
154       void StopFinding(content::StopFindAction action);
155     
156       // ### InsertCSS
157       //
158       // Inserts some CSS in the main frame document
159       // ```
160       // @css {string} css text
161       // ```
162       void InsertCSS(const std::string& css);
163     
164       // ### executeScript
165       //
166       // Executes script in the main frame document
167       // ```
168       // @css {string} script text
169       // ```
170       void ExecuteScript(const std::string& script);
171

173       /* PUBLIC API */
174       /****************************************************************************/
175       void Init(content::WebContents* guest_web_contents);
176     
177       // Toggles autosize mode for this GuestView.
178       void SetAutoSize(bool enabled,
179                        const gfx::Size& min_size,

183         return embedder_web_contents_;
184       }
185     
186       // Returns the guest WebContents.
187       content::WebContents* guest_web_contents() const {
188         return guest_web_contents_;
189       }
190     
191       // Returns whether this guest has an associated embedder.
192       bool attached() const { return !!embedder_web_contents_; }
193     
194       // Returns the instance ID of the <*view> element.
195       int view_instance_id() const { return view_instance_id_; }
196     
197       // Returns the instance ID of the guest WebContents.
198       int guest_instance_id() const { return guest_instance_id_; }
199     
200       // Returns the user browser context of the embedder.
201       content::BrowserContext* browser_context() const { return browser_context_; }
202

284       int                                       embedder_render_process_id_;
