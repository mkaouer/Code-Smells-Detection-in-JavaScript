15     static base::DictionaryValue*
16     ValueFromCanonicalCookie(
17         const net::CanonicalCookie& cc)
18     {
19       base::DictionaryValue* cookie_v = new base::DictionaryValue;
20     
21       GURL url(cc.Source());
22       std::string domain_key(net::cookie_util::GetEffectiveDomain(url.scheme(),
23                                                                   url.host()));
24     
25       cookie_v->SetString("domain_key", domain_key);
26       cookie_v->SetString("source", cc.Source());
27       cookie_v->SetString("name", cc.Name());
28       cookie_v->SetString("value", cc.Value());
29       cookie_v->SetString("domain", cc.Domain());
30       cookie_v->SetString("path", cc.Path());
31     
32       cookie_v->SetInteger("creation", cc.CreationDate().ToInternalValue());
33       cookie_v->SetInteger("expiry", cc.ExpiryDate().ToInternalValue());
34       cookie_v->SetInteger("last_access", cc.LastAccessDate().ToInternalValue());
35     
36       cookie_v->SetInteger("secure", cc.IsSecure());
37       cookie_v->SetBoolean("http_only", cc.IsHttpOnly());
38     
39       cookie_v->SetInteger("prority", cc.Priority());
40     
41       return cookie_v;
42     }

44     net::CanonicalCookie*
45     CanonicalCookieFromValue(
46         base::DictionaryValue* cookie_v)
47     {
48       std::string source = "";
49       std::string name = "";
50       std::string value = "";
51       std::string domain = "";
52       std::string path = "";
53       int creation = 0;
54       int expiry = 0;
55       int last_access = 0;
56       bool secure = false;
57       bool http_only = false;
58       int priority = 0;
59       
60       cookie_v->GetString("source", &source);
61       cookie_v->GetString("name", &name);
62       cookie_v->GetString("value", &value);
63       cookie_v->GetString("domain", &domain);
64       cookie_v->GetString("path", &path);
65     
66       cookie_v->GetInteger("creation", &creation);
67       cookie_v->GetInteger("expiry", &expiry);
68       cookie_v->GetInteger("last_access", &last_access);
69     
70       cookie_v->GetBoolean("secure", &secure);
71       cookie_v->GetBoolean("http_only", &http_only);
72     
73       cookie_v->GetInteger("priority", &priority);
74     
75       net::CanonicalCookie* cc = new net::CanonicalCookie(
76           GURL(source),
77           name,
78           value,
79           domain,
80           path,
81           base::Time::FromInternalValue(creation),
82           base::Time::FromInternalValue(expiry),
83           base::Time::FromInternalValue(last_access),
84           secure,
85           http_only,
86           (net::CookiePriority)priority);
87     
88       return cc;
89     }

111     ThrustSessionBinding::ThrustSessionBinding(
112         const unsigned int id, 
113         scoped_ptr<base::DictionaryValue> args)
114       : APIBinding("session", id)
115     {
116       LOG(INFO) << "ThrustSessionBinding Constructor [" << this << "] " << id_;
117     
118       bool off_the_record = true;
119       args->GetBoolean("off_the_record", &off_the_record);
120     
121       std::string path = "dummy_session";
122       args->GetString("path", &path);
123     
124       bool cookie_store = false;
125       args->GetBoolean("cookie_store", &cookie_store);
126     
127       session_.reset(new ThrustSession(this,
128                                        off_the_record, 
129                                        path, 
130                                        !cookie_store));
131       session_->Initialize();
132     }

141     void
142     ThrustSessionBinding::CallLocalMethod(
143         const std::string& method,
144         scoped_ptr<base::DictionaryValue> args,
145         const API::MethodCallback& callback)
146     {
147       std::string err = std::string("");
148       base::DictionaryValue* res = new base::DictionaryValue;
149     
150       LOG(INFO) << "CALL " << method;
151       if(method.compare("off_the_record") == 0) {
152         res->SetBoolean("off_the_record", session_->IsOffTheRecord());
153       }
154       else if(method.compare("visitedlink_add") == 0) {
155         std::string url = "";
156         args->GetString("url", &url);
157         session_->GetVisitedLinkStore()->Add(url);
158       }
159       else if(method.compare("visitedlink_clear") == 0) {
160         session_->GetVisitedLinkStore()->Clear();
161       }
162       else {
163         err = "exo_session_binding:method_not_found";
164       }
165     
166       callback.Run(err, scoped_ptr<base::DictionaryValue>(res).Pass());
167     }

169     void
170     ThrustSessionBinding::CookiesLoadCallback(
171         const LoadedCallback& loaded_callback,
172         const std::string& error,
173         scoped_ptr<base::DictionaryValue> result)
174     {
175       /* Runs on UI thread. */
176       std::vector<net::CanonicalCookie*> ccs;
177     
178       base::ListValue* cookies;
179       if(result->GetList("cookies", &cookies)) {
180         for(size_t i = 0; i < cookies->GetSize(); i++) {
181           base::DictionaryValue* cookie;
182           if(cookies->GetDictionary(i, &cookie)) {
183             ccs.push_back(CanonicalCookieFromValue(cookie));
184           }
185         }
186       }
187     
188       LOG(INFO) << "COOKIE LOAD CALLBACK " << error;
189       LOG(INFO) << ccs.size();
190     
191       content::BrowserThread::PostTask(
192           content::BrowserThread::IO, FROM_HERE,
193           base::Bind(loaded_callback, ccs));
194     }

197     ThrustSessionBinding::CookiesLoad(
198         const LoadedCallback& loaded_callback)
199     {
200       /* Runs on UI thread. */
201       base::DictionaryValue* args = new base::DictionaryValue;
202     
203       this->InvokeRemoteMethod("cookies_load", 
204                                scoped_ptr<base::DictionaryValue>(args).Pass(),
205                                base::Bind(&ThrustSessionBinding::CookiesLoadCallback, 
206                                           this, loaded_callback));
207     }
