59     ThrustWindow::ThrustWindow(
60         ThrustWindowBinding* binding,
61         WebContents* web_contents,
62         const gfx::Size& size,
63         const std::string& title,
64         const std::string& icon_path,
65         const bool has_frame)
66       : WebContentsObserver(web_contents),
67         binding_(binding),
68         is_closed_(false),
69         title_(title),
70         has_frame_(has_frame),
71         inspectable_web_contents_(
72             brightray::InspectableWebContents::Create(web_contents))
73     {
74       web_contents->SetDelegate(this);
75       inspectable_web_contents()->SetDelegate(this);
76       //WebContentsObserver::Observe(web_contents);
77     
78       // Get notified of title updated message.
79       registrar_.Add(this, NOTIFICATION_WEB_CONTENTS_TITLE_UPDATED,
80                      Source<WebContents>(web_contents));
81     
82       /* Set the icon for the window before it gets displayed. We exceptionally */
83       /* allow IO on this thread (UI) to load the PNG image.                    */
84       base::ThreadRestrictions::SetIOAllowed(true);
85       gfx::Image icon;
86       base::FilePath p = base::FilePath::FromUTF8Unsafe(icon_path);
87       // Read the file from disk.
88       std::string file_contents;
89       if(!p.empty() && base::ReadFileToString(p, &file_contents)) {
90         // Decode the bitmap using WebKit's image decoder.
91         const unsigned char* data =
92           reinterpret_cast<const unsigned char*>(file_contents.data());
93         size_t size = file_contents.size();
94         scoped_ptr<SkBitmap> decoded(new SkBitmap());
95     
96         if(!gfx::PNGCodec::Decode(data, size, decoded.get())) {
97           decoded.reset(gfx::JPEGCodec::Decode(data, size));                             
98         }
99         if(decoded) {                                                                   
100           icon_.AddRepresentation(gfx::ImageSkiaRep(
101                 *decoded.release(), 1.0f));     
102         }                                  
103       }
104       base::ThreadRestrictions::SetIOAllowed(false);
105     
106       /*
107       renderer_preferences_util::UpdateFromSystemSettings(
108           web_contents_->GetMutableRendererPrefs());
109       web_contents_->GetRenderViewHost()->SyncRendererPrefs();
110        */
111       PlatformCreateWindow(size);
112     
113       LOG(INFO) << "ThrustWindow Constructor [" << web_contents << "]";
114       s_instances.push_back(this);
115     }
116     
117     ThrustWindow::~ThrustWindow() 
118     {
119       LOG(INFO) << "ThrustWindow Destructor [" << inspectable_web_contents() << "]";
120     
121       CloseImmediately();
122       PlatformCleanUp();
123     
124       for(size_t i = 0; i < s_instances.size(); ++i) {
125         if (s_instances[i] == this) {
126           s_instances.erase(s_instances.begin() + i);
127           break;
128         }
129       }
130     }

150     // static
151     ThrustWindow*
152     ThrustWindow::CreateNew(
153         ThrustWindowBinding* binding,
154         ThrustSession* session,
155         const GURL& root_url,
156         const gfx::Size& size,
157         const std::string& title,
158         const std::string& icon_path,
159         const bool has_frame)
160     {
161       if(session == NULL) {
162         session = ThrustShellBrowserClient::Get()->system_session();
163       }
164       WebContents::CreateParams create_params((BrowserContext*)session);
165       WebContents* web_contents = WebContents::Create(create_params);
166       
167       NavigationController::LoadURLParams params(root_url);
168       params.transition_type = PageTransitionFromInt(
169           PAGE_TRANSITION_TYPED | PAGE_TRANSITION_FROM_ADDRESS_BAR);
170       web_contents->GetController().LoadURLWithParams(params);
171     
172       LOG(INFO) << "ThrustWindow Constructor (web_contents created) [" 
173                 << web_contents << "]";
174     
175       return CreateNew(binding, web_contents, size, title, icon_path, has_frame);
176     }

211     void
212     ThrustWindow::Close()
213     {
214       content::WebContents* web_contents(GetWebContents());
215       if(!web_contents) {
216         CloseImmediately();
217         return;
218       }
219     
220       if(web_contents->NeedToFireBeforeUnload())
221         web_contents->DispatchBeforeUnload(false);
222       else
223         web_contents->Close();
224     }

253     /******************************************************************************/
254     WebContents* 
255     ThrustWindow::OpenURLFromTab(
256         WebContents* source,
257         const content::OpenURLParams& params) 
258     {
259       if(params.disposition != CURRENT_TAB)
260         return NULL;
261     
262       content::NavigationController::LoadURLParams load_url_params(params.url);
263       load_url_params.referrer = params.referrer;
264       load_url_params.transition_type = params.transition;
265       load_url_params.extra_headers = params.extra_headers;
266       load_url_params.should_replace_current_entry =
267           params.should_replace_current_entry;
268       load_url_params.is_renderer_initiated = params.is_renderer_initiated;
269       load_url_params.transferred_global_request_id =
270           params.transferred_global_request_id;
271     
272       source->GetController().LoadURLWithParams(load_url_params);
273       return source;
274     }

390     void 
391     ThrustWindow::Observe(
392         int type,
393         const NotificationSource& source,
394         const NotificationDetails& details) 
395     {
396       if (type == NOTIFICATION_WEB_CONTENTS_TITLE_UPDATED) {
397         std::pair<NavigationEntry*, bool>* title =
398             Details<std::pair<NavigationEntry*, bool>>(details).ptr();
399     
400         if (title->first) {
401           std::string text = base::UTF16ToUTF8(title->first->GetTitle());
402           SetTitle(text);
403         }
404       }
405     }

472     void 
473     ThrustWindow::CreateWebViewGuest(
474         const base::DictionaryValue& params,
475         int* guest_instance_id)
476     {
477     
478       *guest_instance_id = 
479         ThrustShellBrowserClient::Get()->ThrustSessionForBrowserContext(
480             GetWebContents()->GetBrowserContext())->
481           GetNextInstanceID();
482     
483       LOG(INFO) << "ThrustWindow CreateWebViewGuest " << *guest_instance_id;
484     
485       WebViewGuest* guest = WebViewGuest::Create(*guest_instance_id);
486     
487       GURL guest_site(base::StringPrintf("%s://webview",
488                                          content::kGuestScheme));
489       content::SiteInstance* guest_site_instance =
490         content::SiteInstance::CreateForURL(
491             GetWebContents()->GetBrowserContext(), guest_site);
492     
493       WebContents::CreateParams create_params(
494           GetWebContents()->GetBrowserContext(),
495           guest_site_instance);
496       create_params.guest_delegate = guest;
497       WebContents* guest_web_contents =
498           WebContents::Create(create_params);
499     
500       guest->Init(guest_web_contents);
501     }

564     
565       guest->SetAutoSize(enabled, min_size, max_size);
566     }
