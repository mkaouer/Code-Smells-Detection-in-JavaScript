61     // ### ThrustWindow
62     //
63     // This represents an ThrustShell window. The window opens on a `root_url` 
64     // provided at creation. The window exposes only one webcontents with support 
65     // for the <webview> tag.
66     //
67     // The ThrustWindow lives on the BrowserThread::UI thread
68     class ThrustWindow : public brightray::DefaultWebContentsDelegate,
69                          public brightray::InspectableWebContentsDelegate,
70                          public content::WebContentsObserver,
71     #if defined(USE_AURA)
72                          public views::WidgetDelegateView,
73                          public views::WidgetObserver,
74     #elif defined(OS_MACOSX)
75     #endif
76                          public content::NotificationObserver {

81       // ### CreateNew
82       //
83       // Creates a new ThrustWindow with the specified `root_url`
84       // ```
85       // @binding   {ThrustWindowBinding} the binding associated
86       // @session   {ThrustSession} the session to use
87       // @root_url  {GURL} the main document root url
88       // @size      {Size} the initial size of the window
89       // @title     {string} the title to use
90       // @icon_path {string} icon_path (no effect on OSX)
91       // @has_frame {boolean} has a frame
92       // ```
93       static ThrustWindow* CreateNew(
94           ThrustWindowBinding* binding,
95           ThrustSession* session,
96           const GURL& root_url,
97           const gfx::Size& size,
98           const std::string& title,
99           const std::string& icon_path,
100           const bool has_frame);
101     
102       // ### CreateNew
103       //
104       // Creates a new ThrustWindow out of an existing WebContents
105       // ```
106       // @binding      {ThrustWindowBinding} the binding associated
107       // @web_contents {WebContents} the web_contents to use
108       // @size         {Size} the initial size of the window
109       // @title        {string} the title to use
110       // @icon_path    {string} icon_path (no effect on OSX)
111       // @has_frame    {boolean} has a frame
112       // ```
113       static ThrustWindow* CreateNew(
114           ThrustWindowBinding* binding,
115           content::WebContents* web_contents,
116           const gfx::Size& size,
117           const std::string& title,
118           const std::string& icon_path,
119           const bool has_frame);
