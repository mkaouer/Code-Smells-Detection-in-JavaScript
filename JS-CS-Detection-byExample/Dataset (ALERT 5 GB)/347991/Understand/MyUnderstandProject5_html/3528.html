207     bool 
208     ThrustShellBrowserClient::IsHandledURL(
209         const GURL& url) 
210     {
211       if (!url.is_valid())
212         return false;
213       DCHECK_EQ(url.scheme(), base::StringToLowerASCII(url.scheme()));
214       // Keep in sync with ProtocolHandlers added by
215       // ThrustShellURLRequestContextGetter::GetURLRequestContext().
216       /* TODO(spolu): Check in sync */
217       static const char* const kProtocolList[] = {
218           url::kBlobScheme,
219           url::kFileSystemScheme,
220           kChromeUIScheme,
221           kChromeDevToolsScheme,
222           url::kDataScheme,
223           url::kFileScheme,
224       };
225       for (size_t i = 0; i < arraysize(kProtocolList); ++i) {
226         if (url.scheme() == kProtocolList[i])
227           return true;
228       }
229       return false;
230     }

232     void 
233     ThrustShellBrowserClient::RegisterThrustSession(
234         ThrustSession* session)
235     {
236       LOG(INFO) << "Register Session";
237       sessions_.push_back(session);
238     }
239     
240     void 
241     ThrustShellBrowserClient::UnRegisterThrustSession(
242         ThrustSession* session)
243     {
244       std::vector<ThrustSession*>::iterator it;
245       for(it = sessions_.begin(); it != sessions_.end(); it++) {
246         if(*it == session)
247           break;
248       }
249       if(it != sessions_.end()) {
250         LOG(INFO) << "UnRegister Session";
251         sessions_.erase(it);
252       }
253     }
254     
255     ThrustSession*
256     ThrustShellBrowserClient::ThrustSessionForBrowserContext(
257         BrowserContext* browser_context) 
258     {
259       std::vector<ThrustSession*>::iterator it;
260       for(it = sessions_.begin(); it != sessions_.end(); it++) {
261         if(*it == browser_context) {
262           return *it;
263         }
264       }
265       return NULL;
266     }
