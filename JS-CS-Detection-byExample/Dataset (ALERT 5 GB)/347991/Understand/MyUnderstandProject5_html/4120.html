29     ThrustWindowBinding::ThrustWindowBinding(
30         const unsigned int id, 
31         scoped_ptr<base::DictionaryValue> args)
32       : APIBinding("window", id)
33     {
34       LOG(INFO) << "ThrustWindowBinding Constructor [" << this << "] " << id_;
35     
36       std::string root_url = "http://google.com";
37       args->GetString("root_url", &root_url);
38     
39       int width = 800;
40       int height = 600;
41     
42       args->GetInteger("size.width", &width);
43       args->GetInteger("size.height", &height);
44     
45       std::string title = "ThrustShell";
46       args->GetString("title", &title);
47     
48       std::string icon_path = "";
49       args->GetString("icon_path", &icon_path);
50     
51       ThrustSession* session = NULL;
52     
53       int session_id = -1;
54       args->GetInteger("session_id", &session_id);
55     
56       ThrustSessionBinding* sb = 
57         (ThrustSessionBinding*)(API::Get()->GetBinding(session_id));
58       if(sb != NULL) {
59         session = sb->GetSession();
60       }
61       else {
62         session = ThrustShellBrowserClient::Get()->system_session();
63       }
64     
65       window_.reset(ThrustWindow::CreateNew(
66             this,
67             session,
68             GURL(root_url), 
69             gfx::Size(width, height), 
70             title, 
71             icon_path, 
72             true));
73     }

76     {
77       LOG(INFO) << "ThrustWindowBinding Destructor [" << this << "] " << id_;
78       window_.reset();
79     }

82     void
83     ThrustWindowBinding::CallLocalMethod(
84         const std::string& method,
85         scoped_ptr<base::DictionaryValue> args,
86         const API::MethodCallback& callback)
87     {
88       std::string err = std::string("");
89       base::DictionaryValue* res = new base::DictionaryValue;
90     
91       LOG(INFO) << "ThrustWindow call [" << method << "]";
92     
93       /* Methods */
94       if(method.compare("show") == 0) {
95         window_->Show();
96       }
97       else if(method.compare("focus") == 0) {
98         bool focus = true;
99         args->GetBoolean("focus", &focus);
100         window_->Focus(focus);
101       }
102       else if(method.compare("maximize") == 0) {
103         window_->Maximize();
104       }
105       else if(method.compare("unmaximize") == 0) {
106         window_->UnMaximize();
107       }
108       else if(method.compare("minimize") == 0) {
109         window_->Minimize();
110       }
111       else if(method.compare("restore") == 0) {
112         window_->Restore();
113       }
114       else if(method.compare("set_title") == 0) {
115         std::string title = "";
116         args->GetString("title", &title);
117         window_->SetTitle(title);
118       }
119       else if(method.compare("set_fullscreen") == 0) {
120         bool fullscreen;
121         args->GetBoolean("fullscreen", &fullscreen);
122         window_->SetFullscreen(fullscreen);
123       }
124       else if(method.compare("set_kiosk") == 0) {
125         bool kiosk;
126         args->GetBoolean("kiosk", &kiosk);
127         LOG(INFO) << "*************************************** KIOSK " << kiosk;
128         window_->SetKiosk(kiosk);
129       }
130       else if(method.compare("move") == 0) {
131         int x, y;
132         args->GetInteger("x", &x);
133         args->GetInteger("y", &y);
134     
135         window_->Move(x, y);
136       }
137       else if(method.compare("resize") == 0) {
138         int width, height;
139         args->GetInteger("width", &width);
140         args->GetInteger("height", &height);
141     
142         window_->Resize(width, height);
143       }
144       else if(method.compare("close") == 0) {
145         window_->Close();
146       }
147       /* Accessors */
148       else if(method.compare("is_closed") == 0) {
149         res->SetBoolean("closed", window_->IsClosed());
150       }
151       else if(method.compare("size") == 0) {
152         res->SetInteger("size.width", window_->GetSize().width());
153         res->SetInteger("size.height", window_->GetSize().height());
154       }
155       else if(method.compare("position") == 0) {
156         res->SetInteger("position.x", window_->GetPosition().x());
157         res->SetInteger("position.y", window_->GetPosition().y());
158       }
159       else if(method.compare("is_maximized") == 0) {
160         res->SetBoolean("maximized", window_->IsMaximized());
161       }
162       else if(method.compare("is_minimized") == 0) {
163         res->SetBoolean("minimized", window_->IsMinimized());
164       }
165       else if(method.compare("is_fullscreen") == 0) {
166         res->SetBoolean("fullscreen", window_->IsFullscreen());
167       }
168       else if(method.compare("is_kiosk") == 0) {
169         res->SetBoolean("kiosk", window_->IsKiosk());
170       }
171       /* Default */
172       else {
173         err = "thrust_window_binding:method_not_found";
174       }
175     
176       callback.Run(err, scoped_ptr<base::DictionaryValue>(res).Pass());
177     }
