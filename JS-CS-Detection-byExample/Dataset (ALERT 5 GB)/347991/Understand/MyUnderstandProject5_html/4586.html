41     void
42     API::InstallBinding(
43         const std::string& type,
44         APIBindingFactory* factory)
45     {
46       LOG(INFO) << "[API] INSTALL: " << type;
47       factories_[type] = factory;
48     }
49     
50     int
51     API::Create(
52         const std::string type,
53         scoped_ptr<base::DictionaryValue> args)
54     {
55       int target = 0;
56       if(type.length() > 0 && factories_[type]) {
57         /* We call into the binding factory to create the binding. This will */
58         /* trigger the creation of a local object.                           */
59         target = ++next_binding_id_;
60         LOG(INFO) << "[API] CREATE: " << type << " " << target;
61         bindings_[target] = factories_[type]->Create(target, args.Pass());
62       }
63       return target;
64     }
65     
66     void
67     API::Delete(
68         unsigned int target)
69     {
70       if(target > 0 && bindings_[target]) {
71         LOG(INFO) << "[API] DELETE: " << target;
72         /* We remove our scoped_refptr to that binding which should trigger */
73         /* its deletion as soon as all callbacks have been called.          */
74         bindings_.erase(target);
75     
76         /* Finally we remove the remote object. We don't delete it as it */
77         /* is not owned by the API                                       */
78         if(remotes_[target]) {
79           remotes_.erase(target);
80         }
81       }
82     }
83     
84     void 
85     API::CallMethod(
86         unsigned int target,
87         std::string method,
88         scoped_ptr<base::DictionaryValue> args,
89         const API::MethodCallback& callback)
90     {
91       if(target > 0 && bindings_[target]) {
92         LOG(INFO) << "[API] CALL: " << target << " " << method;
93         /* We route the request to the right binding. */
94         bindings_[target]->CallLocalMethod(method, args.Pass(), callback);
95       }
96     }
97     
98     void 
99     API::SetRemote(
100         unsigned int target,
101         APIBindingRemote* remote)
102     {
103       if(target > 0 && bindings_[target]) {
104         remotes_[target] = remote;
105       }
106     }
107     
108     APIBinding*
109     API::GetBinding(
110         unsigned int target)
111     {
112       if(target > 0 && bindings_[target]) {
113         return bindings_[target].get();
114       }
115       return NULL;
116     }
117     
118     APIBindingRemote*
119     API::GetRemote(
120         unsigned int target)
121     {
122       if(target > 0 && bindings_[target] && remotes_[target]) {
123         return remotes_[target];
124       }
125       return NULL;
126     }
