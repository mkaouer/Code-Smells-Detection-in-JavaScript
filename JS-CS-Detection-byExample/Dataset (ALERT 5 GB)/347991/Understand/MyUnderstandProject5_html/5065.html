82     void 
83     ThrustShellRendererClient::RenderThreadStarted() 
84     {
85       observer_.reset(new ThrustShellRenderProcessObserver());
86       visited_link_slave_.reset(new visitedlink::VisitedLinkSlave());
87     #if defined(OS_MACOSX)
88       // We need to call this once before the sandbox was initialized to cache the
89       // value.
90       base::debug::BeingDebugged();
91     #endif
92     
93       RenderThread* thread = RenderThread::Get();
94     
95       thread->RegisterExtension(SafeBuiltins::CreateV8Extension());
96     
97     #include "./extensions/resources/web_view.js.bin"
98       std::string web_view_src(
99           (char*)src_renderer_extensions_resources_web_view_js,
100           src_renderer_extensions_resources_web_view_js_len);
101       source_map_.RegisterSource("webview", web_view_src);
102     
103       thread->AddObserver(observer_.get());
104       thread->AddObserver(visited_link_slave_.get());
105     }

132     void 
133     ThrustShellRendererClient::DidCreateScriptContext(
134         blink::WebFrame* frame, 
135         v8::Handle<v8::Context> v8_context, 
136         int extension_group,
137         int world_id) 
138     {
139       /* We limit the injection of WebViewBindings to the top level RenderFrames */
140       content::RenderFrame* render_frame = 
141         content::RenderFrame::FromWebFrame(frame);
142       if(render_frame != render_frame->GetRenderView()->GetMainRenderFrame()) {
143         return;
144       }
145     
146       LOG(INFO) << "ThrustShellRendererClient::DidCreateScriptContext `" 
147                 << frame->uniqueName().utf8() << "` "
148                 << extension_group << " " 
149                 << world_id;
150       ScriptContext* context = new ScriptContext(v8_context, frame);
151       {
152         scoped_ptr<ModuleSystem> module_system(new ModuleSystem(context,
153                                                                 &source_map_));
154         context->set_module_system(module_system.Pass());
155       }
156       ModuleSystem* module_system = context->module_system();
157     
158       // Enable natives in startup.
159       ModuleSystem::NativesEnabledScope natives_enabled_scope(module_system);
160     
161       /* NOTE: please use the naming convention "foo_natives" for these. */
162       module_system->RegisterNativeHandler("document_natives",
163           scoped_ptr<NativeHandler>(
164               new DocumentBindings(context)));
165       module_system->RegisterNativeHandler("webview_natives",
166           scoped_ptr<NativeHandler>(
167               new WebViewBindings(context)));
168     
169       module_system->Require("webview");
170     }
