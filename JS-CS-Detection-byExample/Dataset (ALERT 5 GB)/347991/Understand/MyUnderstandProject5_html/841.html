29     // Distinguish directories from regular files.
30     bool IsDirectory(const base::FilePath& path) {
31       base::File::Info file_info;
32       return base::GetFileInfo(path, &file_info) ?
33           file_info.is_directory : path.EndsWithSeparator();
34     }

36     void ConvertFilters(const Filters& filters,
37                         std::vector<std::wstring>* buffer,
38                         std::vector<COMDLG_FILTERSPEC>* filterspec) {
39       if (filters.empty()) {
40         COMDLG_FILTERSPEC spec = { L"All Files (*.*)", L"*.*" };
41         filterspec->push_back(spec);
42         return;
43       }
44     
45       buffer->reserve(filters.size() * 2);
46       for (size_t i = 0; i < filters.size(); ++i) {
47         const Filter& filter = filters[i];
48     
49         COMDLG_FILTERSPEC spec;
50         buffer->push_back(base::UTF8ToWide(filter.first));
51         spec.pszName = buffer->back().c_str();
52     
53         std::vector<std::string> extensions(filter.second);
54         for (size_t j = 0; j < extensions.size(); ++j)
55           extensions[j].insert(0, "*.");
56         buffer->push_back(base::UTF8ToWide(JoinString(extensions, ";")));
57         spec.pszSpec = buffer->back().c_str();
58     
59         filterspec->push_back(spec);
60       }
61     }

63     // Generic class to delegate common open/save dialog's behaviours, users need to
64     // call interface methods via GetPtr().
65     template <typename T>
66     class FileDialog {
67      public:
68       FileDialog(const base::FilePath& default_path, const std::string& title,
69                  const Filters& filters, int options) {
70         std::wstring file_part;
71         if (!IsDirectory(default_path))
72           file_part = default_path.BaseName().value();
73     
74         std::vector<std::wstring> buffer;
75         std::vector<COMDLG_FILTERSPEC> filterspec;
76         ConvertFilters(filters, &buffer, &filterspec);
77     
78         dialog_.reset(new T(file_part.c_str(), options, NULL,
79                             filterspec.data(), filterspec.size()));
80     
81         if (!title.empty())
82           GetPtr()->SetTitle(base::UTF8ToUTF16(title).c_str());
83     
84         SetDefaultFolder(default_path);
85       }
86     
87       bool Show(thrust_shell::ThrustWindow* parent_window) {
88         HWND window = parent_window ? 
89           parent_window->GetNativeWindow()->GetHost()->GetAcceleratedWidget() :
90           NULL;
91         /*
92         HWND window = parent_window ? static_cast<atom::NativeWindowViews*>(
93             parent_window)->GetAcceleratedWidget() :
94             NULL;
95         */
96         return dialog_->DoModal(window) == IDOK;
97       }
98     
99       T* GetDialog() { return dialog_.get(); }
100     
101       IFileDialog* GetPtr() const { return dialog_->GetPtr(); }
102     
103      private:
104       // Set up the initial directory for the dialog.
105       void SetDefaultFolder(const base::FilePath file_path) {
106         std::wstring directory = IsDirectory(file_path) ?
107             file_path.value() :
108             file_path.DirName().value();
109     
110         ATL::CComPtr<IShellItem> folder_item;
111         HRESULT hr = SHCreateItemFromParsingName(directory.c_str(),
112                                                  NULL,
113                                                  IID_PPV_ARGS(&folder_item));
114         if (SUCCEEDED(hr))
115           GetPtr()->SetDefaultFolder(folder_item);
116       }
117     
118       scoped_ptr<T> dialog_;
119     
120       DISALLOW_COPY_AND_ASSIGN(FileDialog);
121     };

125     bool ShowOpenDialog(thrust_shell::ThrustWindow* parent_window,
126                         const std::string& title,
127                         const base::FilePath& default_path,
128                         const Filters& filters,
129                         int properties,
130                         std::vector<base::FilePath>* paths) {
131       int options = FOS_FORCEFILESYSTEM | FOS_FILEMUSTEXIST;
132       if (properties & FILE_DIALOG_OPEN_DIRECTORY)
133         options |= FOS_PICKFOLDERS;
134       if (properties & FILE_DIALOG_MULTI_SELECTIONS)
135         options |= FOS_ALLOWMULTISELECT;
136     
137       FileDialog<CShellFileOpenDialog> open_dialog(
138           default_path, title, filters, options);
139       if (!open_dialog.Show(parent_window))
140         return false;
141     
142       ATL::CComPtr<IShellItemArray> items;
143       HRESULT hr = static_cast<IFileOpenDialog*>(open_dialog.GetPtr())->GetResults(
144           &items);
145       if (FAILED(hr))
146         return false;
147     
148       ATL::CComPtr<IShellItem> item;
149       DWORD count = 0;
150       hr = items->GetCount(&count);
151       if (FAILED(hr))
152         return false;
153     
154       paths->reserve(count);
155       for (DWORD i = 0; i < count; ++i) {
156         hr = items->GetItemAt(i, &item);
157         if (FAILED(hr))
158           return false;
159     
160         wchar_t file_name[MAX_PATH];
161         hr = CShellFileOpenDialog::GetFileNameFromShellItem(
162             item, SIGDN_FILESYSPATH, file_name, MAX_PATH);
163         if (FAILED(hr))
164           return false;
165     
166         paths->push_back(base::FilePath(file_name));
167       }
168     
169       return true;
170     }

172     void ShowOpenDialog(thrust_shell::ThrustWindow* parent_window,
173                         const std::string& title,
174                         const base::FilePath& default_path,
175                         const Filters& filters,
176                         int properties,
177                         const OpenDialogCallback& callback) {

188     bool ShowSaveDialog(thrust_shell::ThrustWindow* parent_window,
189                         const std::string& title,
190                         const base::FilePath& default_path,
191                         const Filters& filters,
192                         base::FilePath* path) {
193       FileDialog<CShellFileSaveDialog> save_dialog(
194           default_path, title, filters,
195           FOS_FORCEFILESYSTEM | FOS_PATHMUSTEXIST | FOS_OVERWRITEPROMPT);
196       if (!save_dialog.Show(parent_window))
197         return false;
198     
199       wchar_t buffer[MAX_PATH];
200       HRESULT hr = save_dialog.GetDialog()->GetFilePath(buffer, MAX_PATH);
201       if (FAILED(hr))
202         return false;
203     
204       std::string file_name = base::WideToUTF8(std::wstring(buffer));
205     
206       // Append extension according to selected filter.
207       if (!filters.empty()) {
208         UINT filter_index = 1;
209         save_dialog.GetPtr()->GetFileTypeIndex(&filter_index);
210         const Filter& filter = filters[filter_index - 1];
211     
212         bool matched = false;
213         for (size_t i = 0; i < filter.second.size(); ++i) {
214           if (EndsWith(file_name, filter.second[i], false)) {
215             matched = true;
216             break;;
217           }
218         }
219     
220         if (!matched && !filter.second.empty())
221           file_name += ("." + filter.second[0]);
222       }
223     
224       *path = base::FilePath(base::UTF8ToUTF16(file_name));
225       return true;
226     }
