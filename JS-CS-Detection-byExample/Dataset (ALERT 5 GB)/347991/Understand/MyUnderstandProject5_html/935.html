59     // Fills the given salt structure with some quasi-random values
60     // It is not necessary to generate a cryptographically strong random string,
61     // only that it be reasonably different for different users.
62     void GenerateSalt(uint8 salt[LINK_SALT_LENGTH]) {
63       DCHECK_EQ(LINK_SALT_LENGTH, 8) << "This code assumes the length of the salt";
64       uint64 randval = base::RandUint64();
65       memcpy(salt, &randval, 8);
66     }

74     // Returns true if the write was complete.
75     static bool WriteToFile(FILE* file,
76                             off_t offset,
77                             const void* data,
78                             size_t data_len) {
79       if (fseek(file, offset, SEEK_SET) != 0)
80         return false;  // Don't write to an invalid part of the file.
81     
82       size_t num_written = fwrite(data, 1, data_len, file);
83     
84       // The write may not make it to the kernel (stdlib may buffer the write)
85       // until the next fseek/fclose call.  If we crash, it's easy for our used
86       // item count to be out of sync with the number of hashes we write.
87       // Protect against this by calling fflush.
88       int ret = fflush(file);
89       DCHECK_EQ(0, ret);
90       return num_written == data_len;
91     }

93     // This task executes on a background thread and executes a write. This
94     // prevents us from blocking the UI thread doing I/O. Double pointer to FILE
95     // is used because file may still not be opened by the time of scheduling
96     // the task for execution.
97     void AsyncWrite(FILE** file, int32 offset, const std::string& data) {
98       if (*file)
99         WriteToFile(*file, offset, data.data(), data.size());

122     
123     // How rebuilding from history works
124     // ---------------------------------
125     //
126     // We mark that we're rebuilding from history by setting the table_builder_
127     // member in VisitedLinkMaster to the TableBuilder we create. This builder
128     // will be called on the history thread by the history system for every URL
129     // in the database.
130     //
131     // The builder will store the fingerprints for those URLs, and then marshalls
132     // back to the main thread where the VisitedLinkMaster will be notified. The
133     // master then replaces its table with a new table containing the computed
134     // fingerprints.
135     //
136     // The builder must remain active while the history system is using it.
137     // Sometimes, the master will be deleted before the rebuild is complete, in
138     // which case it notifies the builder via DisownMaster(). The builder will
139     // delete itself once rebuilding is complete, and not execute any callback.
140     class VisitedLinkMaster::TableBuilder
141         : public VisitedLinkDelegate::URLEnumerator {
142      public:
143       TableBuilder(VisitedLinkMaster* master,
144                    const uint8 salt[LINK_SALT_LENGTH]);
145     
146       // Called on the main thread when the master is being destroyed. This will
147       // prevent a crash when the query completes and the master is no longer
148       // around. We can not actually do anything but mark this fact, since the
149       // table will be being rebuilt simultaneously on the other thread.
150       void DisownMaster();
151     
152       // VisitedLinkDelegate::URLEnumerator
153       virtual void OnURL(const GURL& url) OVERRIDE;
154       virtual void OnComplete(bool succeed) OVERRIDE;
155     
156      private:
157       virtual ~TableBuilder() {}
158     
159       // OnComplete mashals to this function on the main thread to do the
160       // notification.
161       void OnCompleteMainThread();
162     
163       // Owner of this object. MAY ONLY BE ACCESSED ON THE MAIN THREAD!
164       VisitedLinkMaster* master_;
165     
166       // Indicates whether the operation has failed or not.
167       bool success_;
168     
169       // Salt for this new table.
170       uint8 salt_[LINK_SALT_LENGTH];
171     
172       // Stores the fingerprints we computed on the background thread.
173       VisitedLinkCommon::Fingerprints fingerprints_;
174     
175       DISALLOW_COPY_AND_ASSIGN(TableBuilder);
176     };

208     VisitedLinkMaster::~VisitedLinkMaster() {
209       if (table_builder_.get()) {
210         // Prevent the table builder from calling us back now that we're being
211         // destroyed. Note that we DON'T delete the object, since the history
212         // system is still writing into it. When that is complete, the table
213         // builder will destroy itself when it finds we are gone.
214         table_builder_->DisownMaster();
215       }
216       FreeURLTable();
217       // FreeURLTable() will schedule closing of the file and deletion of |file_|.
218       // So nothing should be done here.
219     }

248     VisitedLinkMaster::Hash VisitedLinkMaster::TryToAddURL(const GURL& url) {
249       // Extra check that we are not incognito. This should not happen.
250       // TODO(boliu): Move this check to HistoryService when IsOffTheRecord is
251       // removed from BrowserContext.
252       if (browser_context_ && browser_context_->IsOffTheRecord()) {
253         NOTREACHED();
254         return null_hash_;
255       }
256     
257       if (!url.is_valid())
258         return null_hash_;  // Don't add invalid URLs.
259     
260       Fingerprint fingerprint = ComputeURLFingerprint(url.spec().data(),
261                                                       url.spec().size(),
262                                                       salt_);
263       if (table_builder_.get()) {
264         // If we have a pending delete for this fingerprint, cancel it.
265         std::set<Fingerprint>::iterator found =
266             deleted_since_rebuild_.find(fingerprint);
267         if (found != deleted_since_rebuild_.end())
268             deleted_since_rebuild_.erase(found);
269     
270         // A rebuild is in progress, save this addition in the temporary list so
271         // it can be added once rebuild is complete.
272         added_since_rebuild_.insert(fingerprint);
273       }
274     
275       // If the table is "full", we don't add URLs and just drop them on the floor.
276       // This can happen if we get thousands of new URLs and something causes
277       // the table resizing to fail. This check prevents a hang in that case. Note
278       // that this is *not* the resize limit, this is just a sanity check.
279       if (used_items_ / 8 > table_length_ / 10)
280         return null_hash_;  // Table is more than 80% full.
281     
282       return AddFingerprint(fingerprint, true);
283     }

292     void VisitedLinkMaster::AddURL(const GURL& url) {
293       Hash index = TryToAddURL(url);
294       if (!table_builder_.get() && index != null_hash_) {
295         // Not rebuilding, so we want to keep the file on disk up-to-date.
296         if (persist_to_disk_) {
297           WriteUsedItemCountToFile();
298           WriteHashRangeToFile(index, index);
299         }
300         ResizeTableIfNecessary();
301       }
302     }

304     void VisitedLinkMaster::AddURLs(const std::vector<GURL>& url) {
305       for (std::vector<GURL>::const_iterator i = url.begin();
306            i != url.end(); ++i) {
307         Hash index = TryToAddURL(*i);
308         if (!table_builder_.get() && index != null_hash_)
309           ResizeTableIfNecessary();
310       }
311     
312       // Keeps the file on disk up-to-date.
313       if (!table_builder_.get() && persist_to_disk_)
314         WriteFullTable();
315     }

317     void VisitedLinkMaster::DeleteAllURLs() {
318       // Any pending modifications are invalid.
319       added_since_rebuild_.clear();
320       deleted_since_rebuild_.clear();
321     
322       // Clear the hash table.
323       used_items_ = 0;
324       memset(hash_table_, 0, this->table_length_ * sizeof(Fingerprint));
325     
326       // Resize it if it is now too empty. Resize may write the new table out for
327       // us, otherwise, schedule writing the new table to disk ourselves.
328       if (!ResizeTableIfNecessary() && persist_to_disk_)
329         WriteFullTable();
330     
331       listener_->Reset();
332     }

338     void VisitedLinkMaster::DeleteURLs(URLIterator* urls) {
339       if (!urls->HasNextURL())
340         return;
341     
342       listener_->Reset();
343     
344       if (table_builder_.get()) {
345         // A rebuild is in progress, save this deletion in the temporary list so
346         // it can be added once rebuild is complete.
347         while (urls->HasNextURL()) {
348           const GURL& url(urls->NextURL());
349           if (!url.is_valid())
350             continue;
351     
352           Fingerprint fingerprint =
353               ComputeURLFingerprint(url.spec().data(), url.spec().size(), salt_);
354           deleted_since_rebuild_.insert(fingerprint);
355     
356           // If the URL was just added and now we're deleting it, it may be in the
357           // list of things added since the last rebuild. Delete it from that list.
358           std::set<Fingerprint>::iterator found =
359               added_since_rebuild_.find(fingerprint);
360           if (found != added_since_rebuild_.end())
361             added_since_rebuild_.erase(found);
362     
363           // Delete the URLs from the in-memory table, but don't bother writing
364           // to disk since it will be replaced soon.
365           DeleteFingerprint(fingerprint, false);
366         }
367         return;
368       }
369     
370       // Compute the deleted URLs' fingerprints and delete them
371       std::set<Fingerprint> deleted_fingerprints;
372       while (urls->HasNextURL()) {
373         const GURL& url(urls->NextURL());
374         if (!url.is_valid())
375           continue;
376         deleted_fingerprints.insert(
377             ComputeURLFingerprint(url.spec().data(), url.spec().size(), salt_));
378       }
379       DeleteFingerprintsFromCurrentTable(deleted_fingerprints);
380     }

382     // See VisitedLinkCommon::IsVisited which should be in sync with this algorithm
383     VisitedLinkMaster::Hash VisitedLinkMaster::AddFingerprint(
384         Fingerprint fingerprint,
385         bool send_notifications) {
386       if (!hash_table_ || table_length_ == 0) {
387         NOTREACHED();  // Not initialized.
388         return null_hash_;
389       }
390     
391       Hash cur_hash = HashFingerprint(fingerprint);
392       Hash first_hash = cur_hash;
393       while (true) {
394         Fingerprint cur_fingerprint = FingerprintAt(cur_hash);
395         if (cur_fingerprint == fingerprint)
396           return null_hash_;  // This fingerprint is already in there, do nothing.
397     
398         if (cur_fingerprint == null_fingerprint_) {
399           // End of probe sequence found, insert here.
400           hash_table_[cur_hash] = fingerprint;
401           used_items_++;
402           // If allowed, notify listener that a new visited link was added.
403           if (send_notifications)
404             listener_->Add(fingerprint);
405           return cur_hash;
406         }
407     
408         // Advance in the probe sequence.
409         cur_hash = IncrementHash(cur_hash);
410         if (cur_hash == first_hash) {
411           // This means that we've wrapped around and are about to go into an
412           // infinite loop. Something was wrong with the hashtable resizing
413           // logic, so stop here.
414           NOTREACHED();
415           return null_hash_;
416         }
417       }
418     }

420     void VisitedLinkMaster::DeleteFingerprintsFromCurrentTable(
421         const std::set<Fingerprint>& fingerprints) {
422       bool bulk_write = (fingerprints.size() > kBigDeleteThreshold);
423     
424       // Delete the URLs from the table.
425       for (std::set<Fingerprint>::const_iterator i = fingerprints.begin();
426            i != fingerprints.end(); ++i)
427         DeleteFingerprint(*i, !bulk_write);
428     
429       // These deleted fingerprints may make us shrink the table.
430       if (ResizeTableIfNecessary())
431         return;  // The resize function wrote the new table to disk for us.
432     
433       // Nobody wrote this out for us, write the full file to disk.
434       if (bulk_write && persist_to_disk_)
435         WriteFullTable();
436     }

438     bool VisitedLinkMaster::DeleteFingerprint(Fingerprint fingerprint,
439                                               bool update_file) {
440       if (!hash_table_ || table_length_ == 0) {
441         NOTREACHED();  // Not initialized.
442         return false;
443       }
444       if (!IsVisited(fingerprint))
445         return false;  // Not in the database to delete.
446     
447       // First update the header used count.
448       used_items_--;
449       if (update_file && persist_to_disk_)
450         WriteUsedItemCountToFile();
451     
452       Hash deleted_hash = HashFingerprint(fingerprint);
453     
454       // Find the range of "stuff" in the hash table that is adjacent to this
455       // fingerprint. These are things that could be affected by the change in
456       // the hash table. Since we use linear probing, anything after the deleted
457       // item up until an empty item could be affected.
458       Hash end_range = deleted_hash;
459       while (true) {
460         Hash next_hash = IncrementHash(end_range);
461         if (next_hash == deleted_hash)
462           break;  // We wrapped around and the whole table is full.
463         if (!hash_table_[next_hash])
464           break;  // Found the last spot.
465         end_range = next_hash;
466       }
467     
468       // We could get all fancy and move the affected fingerprints around, but
469       // instead we just remove them all and re-add them (minus our deleted one).
470       // This will mean there's a small window of time where the affected links
471       // won't be marked visited.
472       base::StackVector<Fingerprint, 32> shuffled_fingerprints;
473       Hash stop_loop = IncrementHash(end_range);  // The end range is inclusive.
474       for (Hash i = deleted_hash; i != stop_loop; i = IncrementHash(i)) {
475         if (hash_table_[i] != fingerprint) {
476           // Don't save the one we're deleting!
477           shuffled_fingerprints->push_back(hash_table_[i]);
478     
479           // This will balance the increment of this value in AddFingerprint below
480           // so there is no net change.
481           used_items_--;
482         }
483         hash_table_[i] = null_fingerprint_;
484       }
485     
486       if (!shuffled_fingerprints->empty()) {
487         // Need to add the new items back.
488         for (size_t i = 0; i < shuffled_fingerprints->size(); i++)
489           AddFingerprint(shuffled_fingerprints[i], false);
490       }
491     
492       // Write the affected range to disk [deleted_hash, end_range].
493       if (update_file && persist_to_disk_)
494         WriteHashRangeToFile(deleted_hash, end_range);
495     
496       return true;
497     }

499     void VisitedLinkMaster::WriteFullTable() {
500       // This function can get called when the file is open, for example, when we
501       // resize the table. We must handle this case and not try to reopen the file,
502       // since there may be write operations pending on the file I/O thread.
503       //
504       // Note that once we start writing, we do not delete on error. This means
505       // there can be a partial file, but the short file will be detected next time
506       // we start, and will be replaced.
507       //
508       // This might possibly get corrupted if we crash in the middle of writing.
509       // We should pick up the most common types of these failures when we notice
510       // that the file size is different when we load it back in, and then we will
511       // regenerate the table.
512       DCHECK(persist_to_disk_);
513     
514       if (!file_) {
515         file_ = static_cast<FILE**>(calloc(1, sizeof(*file_)));
516         base::FilePath filename;
517         GetDatabaseFileName(&filename);
518         PostIOTask(FROM_HERE, base::Bind(&AsyncOpen, file_, filename));
519       }
520     
521       // Write the new header.
522       int32 header[4];
523       header[0] = kFileSignature;
524       header[1] = kFileCurrentVersion;
525       header[2] = table_length_;
526       header[3] = used_items_;
527       WriteToFile(file_, 0, header, sizeof(header));
528       WriteToFile(file_, sizeof(header), salt_, LINK_SALT_LENGTH);
529     
530       // Write the hash data.
531       WriteToFile(file_, kFileHeaderSize,
532                   hash_table_, table_length_ * sizeof(Fingerprint));
533     
534       // The hash table may have shrunk, so make sure this is the end.
535       PostIOTask(FROM_HERE, base::Bind(&AsyncTruncate, file_));
536     }

538     bool VisitedLinkMaster::InitFromFile() {
539       DCHECK(file_ == NULL);
540       DCHECK(persist_to_disk_);
541     
542       base::FilePath filename;
543       GetDatabaseFileName(&filename);
544       base::ScopedFILE file_closer(base::OpenFile(filename, "rb+"));
545       if (!file_closer.get())
546         return false;
547     
548       int32 num_entries, used_count;
549       if (!ReadFileHeader(file_closer.get(), &num_entries, &used_count, salt_))
550         return false;  // Header isn't valid.
551     
552       // Allocate and read the table.
553       if (!CreateURLTable(num_entries, false))
554         return false;
555       if (!ReadFromFile(file_closer.get(), kFileHeaderSize,
556                         hash_table_, num_entries * sizeof(Fingerprint))) {
557         FreeURLTable();
558         return false;
559       }
560       used_items_ = used_count;
561     
562     #ifndef NDEBUG
563       DebugValidate();
564     #endif
565     
566       file_ = static_cast<FILE**>(malloc(sizeof(*file_)));
567       *file_ = file_closer.release();
568       return true;
569     }

571     bool VisitedLinkMaster::InitFromScratch(bool suppress_rebuild) {
572       int32 table_size = kDefaultTableSize;
573       if (table_size_override_)
574         table_size = table_size_override_;
575     
576       // The salt must be generated before the table so that it can be copied to
577       // the shared memory.
578       GenerateSalt(salt_);
579       if (!CreateURLTable(table_size, true))
580         return false;
581     
582     #ifndef NDEBUG
583       DebugValidate();
584     #endif
585     
586       if (suppress_rebuild && persist_to_disk_) {
587         // When we disallow rebuilds (normally just unit tests), just use the
588         // current empty table.
589         WriteFullTable();
590         return true;
591       }
592     
593       // This will build the table from history. On the first run, history will
594       // be empty, so this will be correct. This will also write the new table
595       // to disk. We don't want to save explicitly here, since the rebuild may
596       // not complete, leaving us with an empty but valid visited link database.
597       // In the future, we won't know we need to try rebuilding again.
598       return RebuildTableFromDelegate();
599     }

601     bool VisitedLinkMaster::ReadFileHeader(FILE* file,
602                                            int32* num_entries,
603                                            int32* used_count,
604                                            uint8 salt[LINK_SALT_LENGTH]) {
605       DCHECK(persist_to_disk_);
606     
607       // Get file size.
608       // Note that there is no need to seek back to the original location in the
609       // file since ReadFromFile() [which is the next call accessing the file]
610       // seeks before reading.
611       if (fseek(file, 0, SEEK_END) == -1)
612         return false;
613       size_t file_size = ftell(file);
614     
615       if (file_size <= kFileHeaderSize)
616         return false;
617     
618       uint8 header[kFileHeaderSize];
619       if (!ReadFromFile(file, 0, &header, kFileHeaderSize))
620         return false;
621     
622       // Verify the signature.
623       int32 signature;
624       memcpy(&signature, &header[kFileHeaderSignatureOffset], sizeof(signature));
625       if (signature != kFileSignature)
626         return false;
627     
628       // Verify the version is up-to-date. As with other read errors, a version
629       // mistmatch will trigger a rebuild of the database from history, which will
630       // have the effect of migrating the database.
631       int32 version;
632       memcpy(&version, &header[kFileHeaderVersionOffset], sizeof(version));
633       if (version != kFileCurrentVersion)
634         return false;  // Bad version.
635     
636       // Read the table size and make sure it matches the file size.
637       memcpy(num_entries, &header[kFileHeaderLengthOffset], sizeof(*num_entries));
638       if (*num_entries * sizeof(Fingerprint) + kFileHeaderSize != file_size)
639         return false;  // Bad size.
640     
641       // Read the used item count.
642       memcpy(used_count, &header[kFileHeaderUsedOffset], sizeof(*used_count));
643       if (*used_count > *num_entries)
644         return false;  // Bad used item count;
645     
646       // Read the salt.
647       memcpy(salt, &header[kFileHeaderSaltOffset], LINK_SALT_LENGTH);
648     
649       // This file looks OK from the header's perspective.
650       return true;
651     }

668     // Initializes the shared memory structure. The salt should already be filled
669     // in so that it can be written to the shared memory
670     bool VisitedLinkMaster::CreateURLTable(int32 num_entries, bool init_to_empty) {
671       // The table is the size of the table followed by the entries.
672       uint32 alloc_size = num_entries * sizeof(Fingerprint) + sizeof(SharedHeader);
673     
674       // Create the shared memory object.
675       shared_memory_ = new base::SharedMemory();
676       if (!shared_memory_)
677         return false;
678     
679       if (!shared_memory_->CreateAndMapAnonymous(alloc_size)) {
680         delete shared_memory_;
681         shared_memory_ = NULL;
682         return false;
683       }
684     
685       if (init_to_empty) {
686         memset(shared_memory_->memory(), 0, alloc_size);
687         used_items_ = 0;
688       }
689       table_length_ = num_entries;
690     
691       // Save the header for other processes to read.
692       SharedHeader* header = static_cast<SharedHeader*>(shared_memory_->memory());
693       header->length = table_length_;
694       memcpy(header->salt, salt_, LINK_SALT_LENGTH);
695     
696       // Our table pointer is just the data immediately following the size.
697       hash_table_ = reinterpret_cast<Fingerprint*>(
698           static_cast<char*>(shared_memory_->memory()) + sizeof(SharedHeader));
699     
700       return true;
701     }
702     
703     bool VisitedLinkMaster::BeginReplaceURLTable(int32 num_entries) {
704       base::SharedMemory *old_shared_memory = shared_memory_;
705       Fingerprint* old_hash_table = hash_table_;
706       int32 old_table_length = table_length_;
707       if (!CreateURLTable(num_entries, true)) {
708         // Try to put back the old state.
709         shared_memory_ = old_shared_memory;
710         hash_table_ = old_hash_table;
711         table_length_ = old_table_length;
712         return false;
713       }
714     
715     #ifndef NDEBUG
716       DebugValidate();
717     #endif
718     
719       return true;
720     }

722     void VisitedLinkMaster::FreeURLTable() {
723       if (shared_memory_) {
724         delete shared_memory_;
725         shared_memory_ = NULL;
726       }
727       if (!persist_to_disk_ || !file_)
728         return;
729       PostIOTask(FROM_HERE, base::Bind(&AsyncClose, file_));
730       // AsyncClose() will close the file and free the memory pointed by |file_|.
731       file_ = NULL;
732     }

734     bool VisitedLinkMaster::ResizeTableIfNecessary() {
735       DCHECK(table_length_ > 0) << "Must have a table";
736     
737       // Load limits for good performance/space. We are pretty conservative about
738       // keeping the table not very full. This is because we use linear probing
739       // which increases the likelihood of clumps of entries which will reduce
740       // performance.
741       const float max_table_load = 0.5f;  // Grow when we're > this full.
742       const float min_table_load = 0.2f;  // Shrink when we're < this full.
743     
744       float load = ComputeTableLoad();
745       if (load < max_table_load &&
746           (table_length_ <= static_cast<float>(kDefaultTableSize) ||
747            load > min_table_load))
748         return false;
749     
750       // Table needs to grow or shrink.
751       int new_size = NewTableSizeForCount(used_items_);
752       DCHECK(new_size > used_items_);
753       DCHECK(load <= min_table_load || new_size > table_length_);
754       ResizeTable(new_size);
755       return true;
756     }
757     
758     void VisitedLinkMaster::ResizeTable(int32 new_size) {
759       DCHECK(shared_memory_ && shared_memory_->memory() && hash_table_);
760       shared_memory_serial_++;
761     
762     #ifndef NDEBUG
763       DebugValidate();
764     #endif
765     
766       base::SharedMemory* old_shared_memory = shared_memory_;
767       Fingerprint* old_hash_table = hash_table_;
768       int32 old_table_length = table_length_;
769       if (!BeginReplaceURLTable(new_size))
770         return;
771     
772       // Now we have two tables, our local copy which is the old one, and the new
773       // one loaded into this object where we need to copy the data.
774       for (int32 i = 0; i < old_table_length; i++) {
775         Fingerprint cur = old_hash_table[i];
776         if (cur)
777           AddFingerprint(cur, false);
778       }
779     
780       // On error unmapping, just forget about it since we can't do anything
781       // else to release it.
782       delete old_shared_memory;
783     
784       // Send an update notification to all child processes so they read the new
785       // table.
786       listener_->NewTable(shared_memory_);
787     
788     #ifndef NDEBUG
789       DebugValidate();
790     #endif
791     
792       // The new table needs to be written to disk.
793       if (persist_to_disk_)
794         WriteFullTable();
795     }

797     uint32 VisitedLinkMaster::NewTableSizeForCount(int32 item_count) const {
798       // These table sizes are selected to be the maximum prime number less than
799       // a "convenient" multiple of 1K.
800       static const int table_sizes[] = {
801           16381,    // 16K  = 16384   <- don't shrink below this table size
802                     //                   (should be == default_table_size)
803           32767,    // 32K  = 32768
804           65521,    // 64K  = 65536
805           130051,   // 128K = 131072
806           262127,   // 256K = 262144
807           524269,   // 512K = 524288
808           1048549,  // 1M   = 1048576
809           2097143,  // 2M   = 2097152
810           4194301,  // 4M   = 4194304
811           8388571,  // 8M   = 8388608
812           16777199,  // 16M  = 16777216
813           33554347};  // 32M  = 33554432
814     
815       // Try to leave the table 33% full.
816       int desired = item_count * 3;
817     
818       // Find the closest prime.
819       for (size_t i = 0; i < arraysize(table_sizes); i ++) {
820         if (table_sizes[i] > desired)
821           return table_sizes[i];
822       }
823     
824       // Growing very big, just approximate a "good" number, not growing as much
825       // as normal.
826       return item_count * 2 - 1;
827     }

839     // See the TableBuilder declaration above for how this works.
840     void VisitedLinkMaster::OnTableRebuildComplete(
841         bool success,
842         const std::vector<Fingerprint>& fingerprints) {
843       if (success) {
844         // Replace the old table with a new blank one.
845         shared_memory_serial_++;
846     
847         // We are responsible for freeing it AFTER it has been replaced if
848         // replacement succeeds.
849         base::SharedMemory* old_shared_memory = shared_memory_;
850     
851         int new_table_size = NewTableSizeForCount(
852             static_cast<int>(fingerprints.size() + added_since_rebuild_.size()));
853         if (BeginReplaceURLTable(new_table_size)) {
854           // Free the old table.
855           delete old_shared_memory;
856     
857           // Add the stored fingerprints to the hash table.
858           for (size_t i = 0; i < fingerprints.size(); i++)
859             AddFingerprint(fingerprints[i], false);
860     
861           // Also add anything that was added while we were asynchronously
862           // generating the new table.
863           for (std::set<Fingerprint>::iterator i = added_since_rebuild_.begin();
864                i != added_since_rebuild_.end(); ++i)
865             AddFingerprint(*i, false);
866           added_since_rebuild_.clear();
867     
868           // Now handle deletions.
869           DeleteFingerprintsFromCurrentTable(deleted_since_rebuild_);
870           deleted_since_rebuild_.clear();
871     
872           // Send an update notification to all child processes.
873           listener_->NewTable(shared_memory_);
874     
875           if (persist_to_disk_)
876             WriteFullTable();
877         }
878       }
879       table_builder_ = NULL;  // Will release our reference to the builder.
880     
881       // Notify the unit test that the rebuild is complete (will be NULL in prod.)
882       if (!rebuild_complete_task_.is_null()) {
883         rebuild_complete_task_.Run();
884         rebuild_complete_task_.Reset();
885       }
886     }

908     void VisitedLinkMaster::WriteHashRangeToFile(Hash first_hash, Hash last_hash) {
909       DCHECK(persist_to_disk_);
910     
911       if (!file_)
912         return;  // See comment on the file_ variable for why this might happen.
913       if (last_hash < first_hash) {
914         // Handle wraparound at 0. This first write is first_hash->EOF
915         WriteToFile(file_, first_hash * sizeof(Fingerprint) + kFileHeaderSize,
916                     &hash_table_[first_hash],
917                     (table_length_ - first_hash + 1) * sizeof(Fingerprint));
918     
919         // Now do 0->last_lash.
920         WriteToFile(file_, kFileHeaderSize, hash_table_,
921                     (last_hash + 1) * sizeof(Fingerprint));
922       } else {
923         // Normal case, just write the range.
924         WriteToFile(file_, first_hash * sizeof(Fingerprint) + kFileHeaderSize,
925                     &hash_table_[first_hash],
926                     (last_hash - first_hash + 1) * sizeof(Fingerprint));
927       }
928     }

930     bool VisitedLinkMaster::ReadFromFile(FILE* file,
931                                          off_t offset,
932                                          void* data,
933                                          size_t data_size) {
934       DCHECK(persist_to_disk_);
935     #ifndef NDEBUG
936       // Since this function is synchronous, we require that no asynchronous
937       // operations could possibly be pending.
938       DCHECK(!posted_asynchronous_operation_);
939     #endif
940     
941       if (fseek(file, offset, SEEK_SET) != 0)
942         return false;
943     
944       size_t num_read = fread(data, 1, data_size, file);
945       return num_read == data_size;
946     }
