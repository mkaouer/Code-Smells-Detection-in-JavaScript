28     namespace extensions {
29     
30     // Wrapper for a v8 context.
31     class ScriptContext {
32      public:
33       ScriptContext(const v8::Handle<v8::Context>& context,
34                     blink::WebFrame* frame);
35       virtual ~ScriptContext();
36     
37       static ScriptContext* 
38         FromV8Context(const v8::Handle<v8::Context>& v8_context);
39     
40       // Clears the WebFrame for this contexts and invalidates the associated
41       // ModuleSystem.
42       void Invalidate();
43     
44       // Returns true if this context is still valid, false if it isn't.
45       // A context becomes invalid via Invalidate().
46       bool is_valid() const { 
47         return !v8_context_.IsEmpty(); 
48       }
49     
50       v8::Handle<v8::Context> v8_context() const {
51         return v8_context_.NewHandle(isolate());
52       }
53     
54       blink::WebFrame* web_frame() const { 
55         return web_frame_; 
56       }
57     
58       void set_module_system(scoped_ptr<ModuleSystem> module_system) {
59         module_system_ = module_system.Pass();
60       }
61     
62       ModuleSystem* module_system() { return module_system_.get(); }
63     
64       SafeBuiltins* safe_builtins() { return &safe_builtins_; }
65     
66       const SafeBuiltins* safe_builtins() const { return &safe_builtins_; }
67     
68       // Returns the RenderView associated with this context. Can return NULL if the
69       // context is in the process of being destroyed.
70       content::RenderView* GetRenderView() const;
71     
72       // Runs |function| with appropriate scopes. Doesn't catch exceptions, callers
73       // must do that if they want.
74       //
75       // USE THIS METHOD RATHER THAN v8::Function::Call WHEREVER POSSIBLE.
76       v8::Local<v8::Value> CallFunction(v8::Handle<v8::Function> function,
77                                         int argc,
78                                         v8::Handle<v8::Value> argv[]) const;
79     
80       // Fires the onunload event on the unload_event module.
81       void DispatchOnUnloadEvent();
82     
83       v8::Isolate* isolate() const { return isolate_; }
84     
85       // Get the URL of this context's web frame.
86       GURL GetURL() const;
87     
88       // Utility to get the URL we will match against for a frame. If the frame has
89       // committed, this is the commited URL. Otherwise it is the provisional URL.
90       static GURL GetDataSourceURLForFrame(const blink::WebFrame* frame);
91     
92       // Returns the first non-about:-URL in the document hierarchy above and
93       // including |frame|. The document hierarchy is only traversed if
94       // |document_url| is an about:-URL and if |match_about_blank| is true.
95       static GURL GetEffectiveDocumentURL(const blink::WebFrame* frame,
96                                           const GURL& document_url,
97                                           bool match_about_blank);
98     
99     
100      protected:
101       // The v8 context the bindings are accessible to.
102       ScopedPersistent<v8::Context> v8_context_;
103     
104      private:
105       // The WebFrame associated with this context. This can be NULL because this
106       // object can outlive is destroyed asynchronously.
107       blink::WebFrame* web_frame_;
108     
109       // Owns and structures the JS that is injected to set up extension bindings.
110       scoped_ptr<ModuleSystem> module_system_;
111     
112       // Contains safe copies of builtin objects like Function.prototype.
113       SafeBuiltins safe_builtins_;
114     
115       v8::Isolate* isolate_;
116     
117       // A static container of all the script contexts.
118       static std::vector<ScriptContext*> s_instances;
119     
120       DISALLOW_COPY_AND_ASSIGN(ScriptContext);
121     };
122     
123     }  // namespace extensions
124     
125     #endif  // EXTENSIONS_RENDERER_SCRIPT_CONTEXT_H_
