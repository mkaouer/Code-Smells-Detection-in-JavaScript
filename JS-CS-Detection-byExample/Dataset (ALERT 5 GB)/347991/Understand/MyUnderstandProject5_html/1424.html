30     bool 
31     IsSwitchEnabled(
32         base::CommandLine* command_line,
33         const char* switch_string,
34         bool* enabled) 
35     {
36       std::string value = command_line->GetSwitchValueASCII(switch_string);
37       if (value == "true")
38         *enabled = true;
39       else if (value == "false")
40         *enabled = false;
41       else
42         return false;
43       return true;
44     }
45      
46     }

90     void 
91     ThrustShellRenderProcessObserver::EnableWebRuntimeFeatures() 
92     {
93       base::CommandLine* command_line = base::CommandLine::ForCurrentProcess();
94       bool b;
95       if(IsSwitchEnabled(command_line, switches::kExperimentalFeatures, &b)) {
96         blink::WebRuntimeFeatures::enableExperimentalFeatures(b);
97       }
98       if(IsSwitchEnabled(command_line, switches::kExperimentalCanvasFeatures, &b)) {
99         blink::WebRuntimeFeatures::enableExperimentalCanvasFeatures(b);
100       }
101       if(IsSwitchEnabled(command_line, switches::kSubpixelFontScaling, &b)) {
102         blink::WebRuntimeFeatures::enableSubpixelFontScaling(b);
103       }
104       if(IsSwitchEnabled(command_line, switches::kOverlayScrollbars, &b)) {
105         blink::WebRuntimeFeatures::enableOverlayScrollbars(b);
106       }
107       if(IsSwitchEnabled(command_line, switches::kOverlayFullscreenVideo, &b)) {
108         blink::WebRuntimeFeatures::enableOverlayFullscreenVideo(b);
109       }
110       if(IsSwitchEnabled(command_line, switches::kSharedWorker, &b)) {
111         blink::WebRuntimeFeatures::enableSharedWorker(b);
112       }
113     }
