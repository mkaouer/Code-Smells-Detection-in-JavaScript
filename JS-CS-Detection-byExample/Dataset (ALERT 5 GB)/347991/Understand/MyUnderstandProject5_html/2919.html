32     ThrustShellDownloadManagerDelegate::~ThrustShellDownloadManagerDelegate()
33     {
34       if (download_manager_) {

51     void 
52     ThrustShellDownloadManagerDelegate::Shutdown() 
53     {
54       // Revoke any pending callbacks. download_manager_ et. al. are no longer safe
55       // to access after this point.

57       download_manager_ = NULL;

60     bool 
61     ThrustShellDownloadManagerDelegate::DetermineDownloadTarget(
62         DownloadItem* download,
63         const DownloadTargetCallback& callback) 
64     {
65       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::UI));
66       // This assignment needs to be here because even at the call to
67       // SetDownloadManager, the system is not fully initialized.
68       if (default_download_path_.empty()) {
69         default_download_path_ = download_manager_->GetBrowserContext()->GetPath().
70             Append(FILE_PATH_LITERAL("Downloads"));
71       }
72     
73       if (!download->GetForcedFilePath().empty()) {
74         callback.Run(download->GetForcedFilePath(),
75                      DownloadItem::TARGET_DISPOSITION_OVERWRITE,
76                      DOWNLOAD_DANGER_TYPE_NOT_DANGEROUS,
77                      download->GetForcedFilePath());
78         return true;
79       }
80     
81       FilenameDeterminedCallback filename_determined_callback =
82           base::Bind(&ThrustShellDownloadManagerDelegate::OnDownloadPathGenerated,
83                      weak_ptr_factory_.GetWeakPtr(),
84                      download->GetId(),
85                      callback);
86     
87       BrowserThread::PostTask(
88           BrowserThread::FILE,
89           FROM_HERE,
90           base::Bind(&ThrustShellDownloadManagerDelegate::GenerateFilename,
91                      download->GetURL(),
92                      download->GetContentDisposition(),
93                      download->GetSuggestedFilename(),
94                      download->GetMimeType(),
95                      default_download_path_,
96                      filename_determined_callback));
97       return true;
98     }

116     void 
117     ThrustShellDownloadManagerDelegate::GenerateFilename(
118         const GURL& url,
119         const std::string& content_disposition,
120         const std::string& suggested_filename,
121         const std::string& mime_type,
122         const base::FilePath& suggested_directory,
123         const FilenameDeterminedCallback& callback)
124     {
125       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::FILE));
126     
127       base::FilePath generated_name = net::GenerateFileName(url,
128                                                             content_disposition,
129                                                             std::string(),
130                                                             suggested_filename,
131                                                             mime_type,
132                                                             "download");
133       if (!base::PathExists(suggested_directory))
134         base::CreateDirectory(suggested_directory);
135     
136       base::FilePath suggested_path(suggested_directory.Append(generated_name));
137       BrowserThread::PostTask(
138           BrowserThread::UI, FROM_HERE, base::Bind(callback, suggested_path));
139     }

141     void 
142     ThrustShellDownloadManagerDelegate::OnDownloadPathGenerated(
143         uint32 download_id,
144         const DownloadTargetCallback& callback,
145         const base::FilePath& suggested_path) 
146     {
147       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::UI));
148       if (suppress_prompting_) {
149         // Testing exit.
150         callback.Run(suggested_path, DownloadItem::TARGET_DISPOSITION_OVERWRITE,
151                      DOWNLOAD_DANGER_TYPE_NOT_DANGEROUS,
152                      suggested_path.AddExtension(FILE_PATH_LITERAL(".crdownload")));
153         return;
154       }
155     
156       ChooseDownloadPath(download_id, callback, suggested_path);
157     }
