29     ThrustWindowBinding::ThrustWindowBinding(
30         const unsigned int id, 
31         scoped_ptr<base::DictionaryValue> args)
32       : APIBinding("window", id)
33     {
34       LOG(INFO) << "ThrustWindowBinding Constructor [" << this << "] " << id_;
35     
36       std::string root_url = "http://google.com";
37       args->GetString("root_url", &root_url);
38     
39       int width = 800;
40       int height = 600;
41     
42       args->GetInteger("size.width", &width);
43       args->GetInteger("size.height", &height);
44     
45       std::string title = "ThrustShell";
46       args->GetString("title", &title);
47     
48       std::string icon_path = "";
49       args->GetString("icon_path", &icon_path);
50     
51       bool has_frame = true;
52       args->GetBoolean("has_frame", &has_frame);
53     
54       ThrustSession* session = NULL;
55     
56       int session_id = -1;
57       args->GetInteger("session_id", &session_id);
58     
59       ThrustSessionBinding* sb = 
60         (ThrustSessionBinding*)(API::Get()->GetBinding(session_id));
61       if(sb != NULL) {
62         session = sb->GetSession();
63       }
64       else {
65         session = ThrustShellBrowserClient::Get()->system_session();
66       }
67     
68       window_.reset(ThrustWindow::CreateNew(
69             this,
70             session,
71             GURL(root_url), 
72             gfx::Size(width, height), 
73             title, 
74             icon_path, 
75             has_frame));
76     }

80       LOG(INFO) << "ThrustWindowBinding Destructor [" << this << "] " << id_;
81       window_.reset();
82     }

85     void
86     ThrustWindowBinding::CallLocalMethod(
87         const std::string& method,
88         scoped_ptr<base::DictionaryValue> args,
89         const API::MethodCallback& callback)
90     {
91       std::string err = std::string("");
92       base::DictionaryValue* res = new base::DictionaryValue;
93     
94       LOG(INFO) << "ThrustWindow call [" << method << "]";
95     
96       /* Methods */
97       if(method.compare("show") == 0) {
98         window_->Show();
99       }
100       else if(method.compare("focus") == 0) {
101         bool focus = true;
102         args->GetBoolean("focus", &focus);
103         window_->Focus(focus);
104       }
105       else if(method.compare("maximize") == 0) {
106         window_->Maximize();
107       }
108       else if(method.compare("unmaximize") == 0) {
109         window_->UnMaximize();
110       }
111       else if(method.compare("minimize") == 0) {
112         window_->Minimize();
113       }
114       else if(method.compare("restore") == 0) {
115         window_->Restore();
116       }
117       else if(method.compare("set_title") == 0) {
118         std::string title = "";
119         args->GetString("title", &title);
120         window_->SetTitle(title);
121       }
122       else if(method.compare("set_fullscreen") == 0) {
123         bool fullscreen;
124         args->GetBoolean("fullscreen", &fullscreen);
125         window_->SetFullscreen(fullscreen);
126       }
127       else if(method.compare("set_kiosk") == 0) {
128         bool kiosk;
129         args->GetBoolean("kiosk", &kiosk);
130         window_->SetKiosk(kiosk);
131       }
132       else if(method.compare("open_devtools") == 0) {
133         window_->OpenDevTools();
134       }
135       else if(method.compare("close_devtools") == 0) {
136         window_->CloseDevTools();
137       }
138       else if(method.compare("move") == 0) {
139         int x, y;
140         args->GetInteger("x", &x);
141         args->GetInteger("y", &y);
142     
143         window_->Move(x, y);
144       }
145       else if(method.compare("resize") == 0) {
146         int width, height;
147         args->GetInteger("width", &width);
148         args->GetInteger("height", &height);
149     
150         window_->Resize(width, height);
151       }
152       else if(method.compare("close") == 0) {
153         window_->Close();
154       }
155       /* Accessors */
156       else if(method.compare("is_closed") == 0) {
157         res->SetBoolean("closed", window_->IsClosed());
158       }
159       else if(method.compare("size") == 0) {
160         res->SetInteger("size.width", window_->GetSize().width());
161         res->SetInteger("size.height", window_->GetSize().height());
162       }
163       else if(method.compare("position") == 0) {
164         res->SetInteger("position.x", window_->GetPosition().x());
165         res->SetInteger("position.y", window_->GetPosition().y());
166       }
167       else if(method.compare("is_maximized") == 0) {
168         res->SetBoolean("maximized", window_->IsMaximized());
169       }
170       else if(method.compare("is_minimized") == 0) {
171         res->SetBoolean("minimized", window_->IsMinimized());
172       }
173       else if(method.compare("is_fullscreen") == 0) {
174         res->SetBoolean("fullscreen", window_->IsFullscreen());
175       }
176       else if(method.compare("is_kiosk") == 0) {
177         res->SetBoolean("kiosk", window_->IsKiosk());
178       }
179       else if(method.compare("is_devtools_opened") == 0) {
180         res->SetBoolean("opened", window_->IsDevToolsOpened());
181       }
182       /* Default */
183       else {
184         err = "thrust_window_binding:method_not_found";
185       }
186     
187       callback.Run(err, scoped_ptr<base::DictionaryValue>(res).Pass());
188     }
