41     void 
42     ThrustShellAccessTokenStore::LoadAccessTokens(
43         const LoadAccessTokensCallbackType& callback) 
44     {
45       AccessTokenSet access_token_set;
46     
47       // Equivelent to access_token_set[kGeolocationProviderUrl].
48       // Somehow base::string16 is causing compilation errors when used in a pair
49       // of std::map on Linux, this can work around it.
50       std::pair<GURL, base::string16> token_pair;
51       token_pair.first = GURL(kGeolocationProviderUrl);
52       access_token_set.insert(token_pair);
53     
54       callback.Run(access_token_set,
55                    ThrustShellBrowserClient::Get()->system_session()->url_request_context_getter());
56     }
