22     // Return key code of the char.
23     ui::KeyboardCode KeyboardCodeFromCharCode(char c, bool* shifted) {
24       *shifted = false;
25       switch (c) {
26         case 8: case 0x7F: return ui::VKEY_BACK;
27         case 9: return ui::VKEY_TAB;
28         case 0xD: case 3: return ui::VKEY_RETURN;
29         case 0x1B: return ui::VKEY_ESCAPE;
30         case ' ': return ui::VKEY_SPACE;
31     
32         case 'a': return ui::VKEY_A;
33         case 'b': return ui::VKEY_B;
34         case 'c': return ui::VKEY_C;
35         case 'd': return ui::VKEY_D;
36         case 'e': return ui::VKEY_E;
37         case 'f': return ui::VKEY_F;
38         case 'g': return ui::VKEY_G;
39         case 'h': return ui::VKEY_H;
40         case 'i': return ui::VKEY_I;
41         case 'j': return ui::VKEY_J;
42         case 'k': return ui::VKEY_K;
43         case 'l': return ui::VKEY_L;
44         case 'm': return ui::VKEY_M;
45         case 'n': return ui::VKEY_N;
46         case 'o': return ui::VKEY_O;
47         case 'p': return ui::VKEY_P;
48         case 'q': return ui::VKEY_Q;
49         case 'r': return ui::VKEY_R;
50         case 's': return ui::VKEY_S;
51         case 't': return ui::VKEY_T;
52         case 'u': return ui::VKEY_U;
53         case 'v': return ui::VKEY_V;
54         case 'w': return ui::VKEY_W;
55         case 'x': return ui::VKEY_X;
56         case 'y': return ui::VKEY_Y;
57         case 'z': return ui::VKEY_Z;
58     
59         case ')': *shifted = true; case '0': return ui::VKEY_0;
60         case '!': *shifted = true; case '1': return ui::VKEY_1;
61         case '@': *shifted = true; case '2': return ui::VKEY_2;
62         case '#': *shifted = true; case '3': return ui::VKEY_3;
63         case '$': *shifted = true; case '4': return ui::VKEY_4;
64         case '%': *shifted = true; case '5': return ui::VKEY_5;
65         case '^': *shifted = true; case '6': return ui::VKEY_6;
66         case '&': *shifted = true; case '7': return ui::VKEY_7;
67         case '*': *shifted = true; case '8': return ui::VKEY_8;
68         case '(': *shifted = true; case '9': return ui::VKEY_9;
69     
70         case ':': *shifted = true; case ';': return ui::VKEY_OEM_1;
71         case '+': *shifted = true; case '=': return ui::VKEY_OEM_PLUS;
72         case '<': *shifted = true; case ',': return ui::VKEY_OEM_COMMA;
73         case '_': *shifted = true; case '-': return ui::VKEY_OEM_MINUS;
74         case '>': *shifted = true; case '.': return ui::VKEY_OEM_PERIOD;
75         case '?': *shifted = true; case '/': return ui::VKEY_OEM_2;
76         case '~': *shifted = true; case '`': return ui::VKEY_OEM_3;
77         case '{': *shifted = true; case '[': return ui::VKEY_OEM_4;
78         case '|': *shifted = true; case '\\': return ui::VKEY_OEM_5;
79         case '}': *shifted = true; case ']': return ui::VKEY_OEM_6;
80         case '"': *shifted = true; case '\'': return ui::VKEY_OEM_7;
81     
82         default: return ui::VKEY_UNKNOWN;
83       }
84     }

88     bool StringToAccelerator(const std::string& description,
89                              ui::Accelerator* accelerator) {
90       if (!base::IsStringASCII(description)) {
91         LOG(ERROR) << "The accelerator string can only contain ASCII characters";
92         return false;
93       }
94       std::string shortcut(StringToLowerASCII(description));
95     
96       std::vector<std::string> tokens;
97       base::SplitString(shortcut, '+', &tokens);
98     
99       // Now, parse it into an accelerator.
100       int modifiers = ui::EF_NONE;
101       ui::KeyboardCode key = ui::VKEY_UNKNOWN;
102       for (size_t i = 0; i < tokens.size(); i++) {
103         // We use straight comparing instead of map because the accelerator tends
104         // to be correct and usually only uses few special tokens.
105         if (tokens[i].size() == 1) {
106           bool shifted = false;
107           key = KeyboardCodeFromCharCode(tokens[i][0], &shifted);
108           if (shifted)
109             modifiers |= ui::EF_SHIFT_DOWN;
110         } else if (tokens[i] == "ctrl" || tokens[i] == "control") {
111           modifiers |= ui::EF_CONTROL_DOWN;
112         } else if (tokens[i] == "cmd" || tokens[i] == "command") {
113           modifiers |= ui::EF_COMMAND_DOWN;
114         } else if (tokens[i] == "commandorcontrol" || tokens[i] == "cmdorctrl") {
115     #if defined(OS_MACOSX)
116           modifiers |= ui::EF_COMMAND_DOWN;
117     #else
118           modifiers |= ui::EF_CONTROL_DOWN;
119     #endif
120         } else if (tokens[i] == "alt") {
121           modifiers |= ui::EF_ALT_DOWN;
122         } else if (tokens[i] == "shift") {
123           modifiers |= ui::EF_SHIFT_DOWN;
124         } else if (tokens[i] == "tab") {
125           key = ui::VKEY_TAB;
126         } else if (tokens[i] == "space") {
127           key = ui::VKEY_SPACE;
128         } else if (tokens[i] == "backspace") {
129           key = ui::VKEY_BACK;
130         } else if (tokens[i] == "delete") {
131           key = ui::VKEY_DELETE;
132         } else if (tokens[i] == "insert") {
133           key = ui::VKEY_INSERT;
134         } else if (tokens[i] == "enter" || tokens[i] == "return") {
135           key = ui::VKEY_RETURN;
136         } else if (tokens[i] == "up") {
137           key = ui::VKEY_UP;
138         } else if (tokens[i] == "down") {
139           key = ui::VKEY_DOWN;
140         } else if (tokens[i] == "left") {
141           key = ui::VKEY_LEFT;
142         } else if (tokens[i] == "right") {
143           key = ui::VKEY_RIGHT;
144         } else if (tokens[i] == "home") {
145           key = ui::VKEY_HOME;
146         } else if (tokens[i] == "end") {
147           key = ui::VKEY_END;
148         } else if (tokens[i] == "pagedown") {
149           key = ui::VKEY_PRIOR;
150         } else if (tokens[i] == "pageup") {
151           key = ui::VKEY_NEXT;
152         } else if (tokens[i] == "esc" || tokens[i] == "escape") {
153           key = ui::VKEY_ESCAPE;
154         } else if (tokens[i] == "volumemute") {
155           key = ui::VKEY_VOLUME_MUTE;
156         } else if (tokens[i] == "volumeup") {
157           key = ui::VKEY_VOLUME_UP;
158         } else if (tokens[i] == "volumedown") {
159           key = ui::VKEY_VOLUME_DOWN;
160         } else if (tokens[i] == "medianexttrack") {
161           key = ui::VKEY_MEDIA_NEXT_TRACK;
162         } else if (tokens[i] == "mediaprevioustrack") {
163           key = ui::VKEY_MEDIA_PREV_TRACK;
164         } else if (tokens[i] == "mediastop") {
165           key = ui::VKEY_MEDIA_STOP;
166         } else if (tokens[i] == "mediaplaypause") {
167           key = ui::VKEY_MEDIA_PLAY_PAUSE;
168         } else if (tokens[i].size() > 1 && tokens[i][0] == 'f') {
169           // F1 - F24.
170           int n;
171           if (base::StringToInt(tokens[i].c_str() + 1, &n)) {
172             key = static_cast<ui::KeyboardCode>(ui::VKEY_F1 + n - 1);
173           } else {
174             LOG(WARNING) << tokens[i] << "is not available on keyboard";
175             return false;
176           }
177         } else {
178           LOG(WARNING) << "Invalid accelerator token: " << tokens[i];
179           return false;
180         }
181       }
182     
183       if (key == ui::VKEY_UNKNOWN) {
184         LOG(WARNING) << "The accelerator doesn't contain a valid key";
185         return false;
186       }
187     
188       *accelerator = ui::Accelerator(key, modifiers);
189       SetPlatformAccelerator(accelerator);
190       return true;
191     }

193     void GenerateAcceleratorTable(AcceleratorTable* table, ui::MenuModel* model) {
194       int count = model->GetItemCount();
195       for (int i = 0; i < count; ++i) {
196         ui::MenuModel::ItemType type = model->GetTypeAt(i);
197         if (type == ui::MenuModel::TYPE_SUBMENU) {
198           ui::MenuModel* submodel = model->GetSubmenuModelAt(i);
199           GenerateAcceleratorTable(table, submodel);
200         } else {
201           ui::Accelerator accelerator;
202           if (model->GetAcceleratorAt(i, &accelerator)) {
203             MenuItem item = { i, model };
204             (*table)[accelerator] = item;
205           }
206         }
207       }
208     }

210     bool TriggerAcceleratorTableCommand(AcceleratorTable* table,
211                                         const ui::Accelerator& accelerator) {
212       if (ContainsKey(*table, accelerator)) {
213         const accelerator_util::MenuItem& item = (*table)[accelerator];
214         item.model->ActivatedAt(item.position);
215         return true;
216       } else {
217         return false;
218       }
