23     // ## WebViewGuest
24     //
25     // Browser-side API implementation for the <webview> tag. It's in charge of
26     // maintaining the association between the guest WebContents and the embedder
27     // WebContents. Receives events issued from the guest and realys them to the
28     // embedder.
29     //
30     // A WebViewGuest is also a WebContentsObserver on the guest WebContents of the
31     // associated <webview> tag. WebViewGuest is created on attachment. That is, when a 
32     // guest WebContents is associated with a particular embedder WebContents. 
33     // This happens on either initial navigation or through the use of the New 
34     // Window API, when a new window is attached to a particular <webview>.
35     class WebViewGuest : public content::BrowserPluginGuestDelegate,
36                          public content::NotificationObserver,
37                          public content::WebContentsDelegate,
38                          public content::WebContentsObserver {
39     public:
40     
41       class Event {
42        public:
43         Event(const std::string& name, scoped_ptr<base::DictionaryValue> args);
44         ~Event();
45     
46         const std::string& name() const { return name_; }
47     
48         scoped_ptr<base::DictionaryValue> GetArguments();
49     
50        private:
51         const std::string name_;
52         scoped_ptr<base::DictionaryValue> args_;
53       };
54     
55       /****************************************************************************/
56       /* STATIC API */
57       /****************************************************************************/
58       static WebViewGuest* Create(int guest_instance_id,
59                                   content::WebContents* guest_web_contents);
60     
61       static WebViewGuest* From(int embedder_process_id, int instance_id);
62       static WebViewGuest* FromWebContents(content::WebContents* web_contents);
63     
64       // Returns guestview::kInstanceIDNone if |contents| does not correspond to a
65       // WebViewGuest.
66       static int GetViewInstanceId(content::WebContents* contents);
67       // Parses partition related parameters from |extra_params|.
68       // |storage_partition_id| is the parsed partition ID and |persist_storage|
69       // specifies whether or not the partition is in memory.
70       static void ParsePartitionParam(const base::DictionaryValue* extra_params,
71                                       std::string* storage_partition_id,
72                                       bool* persist_storage);
73     
74       /****************************************************************************/
75       /* BROWSER_PLUGIN_GUEST_DELEGATE IMPLEMENTATION */
76       /****************************************************************************/
77       virtual void Destroy() OVERRIDE FINAL;
78       virtual void DidAttach() OVERRIDE FINAL;
79       virtual void RegisterDestructionCallback(
80           const DestructionCallback& callback) OVERRIDE FINAL;
81       virtual void WillAttach(
82           content::WebContents* embedder_web_contents,
83           const base::DictionaryValue& extra_params) OVERRIDE FINAL;
84     
85       /****************************************************************************/
86       /* NOTIFICATION_OBSERVER IMPLEMENTATION */
87       /****************************************************************************/
88       virtual void Observe(int type,
89                            const content::NotificationSource& source,
90                            const content::NotificationDetails& details) OVERRIDE;
91     
92       /****************************************************************************/
93       /* WEBVIEW API */
94       /****************************************************************************/
95       // Set the zoom factor.
96       void SetZoom(double zoom_factor);
97     
98       // Returns the current zoom factor.
99       double GetZoom();
100     
101       // If possible, navigate the guest to |relative_index| entries away from the
102       // current navigation entry.
103       void Go(int relative_index);
104     
105       // Reload the guest.
106       void Reload();
107     
108       // Stop loading the guest.
109       void Stop();
110     
111     
112       /****************************************************************************/
113       /* PUBLIC API */
114       /****************************************************************************/
115       content::WebContents* embedder_web_contents() const {
116         return embedder_web_contents_;
117       }
118     
119       // Returns the guest WebContents.
120       content::WebContents* guest_web_contents() const {
121         return guest_web_contents_;
122       }
123     
124       // Returns whether this guest has an associated embedder.
125       bool attached() const { return !!embedder_web_contents_; }
126     
127       // Returns the instance ID of the <*view> element.
128       int view_instance_id() const { return view_instance_id_; }
129     
130       // Returns the instance ID of the guest WebContents.
131       int guest_instance_id() const { return guest_instance_id_; }
132     
133       // Returns the user browser context of the embedder.
134       content::BrowserContext* browser_context() const { return browser_context_; }
135     
136       // Returns the embedder's process ID.
137       int embedder_render_process_id() const { return embedder_render_process_id_; }
138     
139      protected:
140       WebViewGuest(int guest_instance_id,
141                    content::WebContents* guest_web_contents);
142       virtual ~WebViewGuest();
143     
144       // Dispatches an event |event_name| to the embedder with the |event| fields.
145       void DispatchEvent(Event* event);
146     
147      private:
148       class EmbedderWebContentsObserver;
149     
150       void SendQueuedEvents();
151

164           int64 frame_id,

179       virtual void DocumentLoadedInFrame(

186       virtual void UserAgentOverrideSet(const std::string& user_agent) OVERRIDE;
187       */
188     
189       content::WebContents* const               guest_web_contents_;
190       content::WebContents*                     embedder_web_contents_;
191       const std::string                         embedder_extension_id_;
192       int                                       embedder_render_process_id_;
193       content::BrowserContext* const            browser_context_;
194       // |guest_instance_id_| is a profile-wide unique identifier for a guest
195       // WebContents.
196       const int                                 guest_instance_id_;
197       // |view_instance_id_| is an identifier that's unique within a particular
198       // embedder RenderViewHost for a particular <*view> instance.
199       int                                       view_instance_id_;
200       bool                                      initialized_;
201       content::NotificationRegistrar            notification_registrar_;
202       // Stores the current zoom factor.
203       double                                    current_zoom_factor_;
204       // This is a queue of Events that are destined to be sent to the embedder once
205       // the guest is attached to a particular embedder.
206       std::deque<linked_ptr<Event> >            pending_events_;
207       DestructionCallback                       destruction_callback_;
208       // The extra parameters associated with this GuestView passed
209       // in from JavaScript. This will typically be the view instance ID,
210       // the API to use, and view-specific parameters. These parameters
211       // are passed along to new guests that are created from this guest.
212       scoped_ptr<base::DictionaryValue>         extra_params_;
213       scoped_ptr<EmbedderWebContentsObserver>   embedder_web_contents_observer_;
214       // This is used to ensure pending tasks will not fire after this object is
215       // destroyed.
216       base::WeakPtrFactory<WebViewGuest>        weak_ptr_factory_;
217     
218       DISALLOW_COPY_AND_ASSIGN(WebViewGuest);
219     };
