56         public:
57           Remote(APIServer::Client* client,
58                  unsigned int target);
59     
60           virtual void InvokeMethod(const std::string method,
61                                     scoped_ptr<base::DictionaryValue> args,
62                                     const API::MethodCallback& callback) OVERRIDE;

64                                  scoped_ptr<base::DictionaryValue> event) OVERRIDE;
65     
66         private:
67           APIServer::Client*                      client_;
68     
69           unsigned int                            target_;
70     
71           DISALLOW_COPY_AND_ASSIGN(Remote);
72         };
73     
74       private:
75         void PerformAction(scoped_ptr<base::DictionaryValue> action);
76     
77         void SendReply(const unsigned id,
78                        const std::string& error,
79                        scoped_ptr<base::DictionaryValue> result);
80         void SendInvoke(const API::MethodCallback& callback,
81                         unsigned int target,
82                         const std::string method,
83                         scoped_ptr<base::DictionaryValue> args);
84         void SendEvent(unsigned int target,
85                        const std::string type,
86                        scoped_ptr<base::DictionaryValue> event);
87                          
88     
89         APIServer*                                         server_;
90         API*                                               api_;
91         unsigned int                                       action_id_;
92     
93         scoped_ptr<net::StreamListenSocket>                conn_;

95     
96         std::map<unsigned int, scoped_refptr<Remote> >     remotes_;
97         std::map<int, API::MethodCallback >                invokes_;
98     
99         DISALLOW_COPY_AND_ASSIGN(Client);
100       };

104       /****************************************************************************/
105       APIServer(API* api,
106                 const base::FilePath& socket_path);
107     
108       // ### Start
109       // 
110       // Starts the APIServer server on its own thread listening
111       void Start();
112     
113       // ### Stop
114       //
115       // Stops the APIServer and shuts down the server
116       void Stop();
117     
118       /****************************************************************************/
119       /* STREAMLISTENSOCKET::DELEGATE INTERFACE */
120       /****************************************************************************/
121       virtual void DidAccept(net::StreamListenSocket* server,                          
122                              scoped_ptr<net::StreamListenSocket> connection) OVERRIDE;
123       virtual void DidRead(net::StreamListenSocket* connection,
124                            const char* data,
125                            int len) OVERRIDE;
126       virtual void DidClose(net::StreamListenSocket* sock) OVERRIDE;
127     
128     private:
129       /****************************************************************************/
130       /* PRIVATE INTERFACE */
131       /****************************************************************************/
132       bool UserCanConnectCallback(uid_t user_id, gid_t group_id);

135       void StartHandlerThread();

137       void StopHandlerThread();
138       void ThreadInit();
139       void ThreadTearDown();
140

142
