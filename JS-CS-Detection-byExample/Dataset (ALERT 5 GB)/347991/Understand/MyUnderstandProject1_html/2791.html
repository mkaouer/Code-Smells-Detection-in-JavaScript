31     ThrustShellDownloadManagerDelegate::~ThrustShellDownloadManagerDelegate()
32     {

50     void 
51     ThrustShellDownloadManagerDelegate::Shutdown() 
52     {
53       // Revoke any pending callbacks. download_manager_ et. al. are no longer safe
54       // to access after this point.
55       weak_ptr_factory_.InvalidateWeakPtrs();
56       download_manager_ = NULL;

59     bool 
60     ThrustShellDownloadManagerDelegate::DetermineDownloadTarget(
61         DownloadItem* download,
62         const DownloadTargetCallback& callback) 
63     {
64       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::UI));
65       // This assignment needs to be here because even at the call to
66       // SetDownloadManager, the system is not fully initialized.
67       if (default_download_path_.empty()) {
68         default_download_path_ = download_manager_->GetBrowserContext()->GetPath().
69             Append(FILE_PATH_LITERAL("Downloads"));
70       }
71     
72       if (!download->GetForcedFilePath().empty()) {
73         callback.Run(download->GetForcedFilePath(),
74                      DownloadItem::TARGET_DISPOSITION_OVERWRITE,
75                      DOWNLOAD_DANGER_TYPE_NOT_DANGEROUS,
76                      download->GetForcedFilePath());
77         return true;
78       }
79     
80       FilenameDeterminedCallback filename_determined_callback =
81           base::Bind(&ThrustShellDownloadManagerDelegate::OnDownloadPathGenerated,
82                      weak_ptr_factory_.GetWeakPtr(),
83                      download->GetId(),
84                      callback);
85     
86       BrowserThread::PostTask(
87           BrowserThread::FILE,
88           FROM_HERE,
89           base::Bind(&ThrustShellDownloadManagerDelegate::GenerateFilename,
90                      download->GetURL(),
91                      download->GetContentDisposition(),
92                      download->GetSuggestedFilename(),
93                      download->GetMimeType(),
94                      default_download_path_,
95                      filename_determined_callback));
96       return true;
97     }

115     void 
116     ThrustShellDownloadManagerDelegate::GenerateFilename(
117         const GURL& url,
118         const std::string& content_disposition,
119         const std::string& suggested_filename,
120         const std::string& mime_type,
121         const base::FilePath& suggested_directory,
122         const FilenameDeterminedCallback& callback)
123     {
124       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::FILE));
125     
126       base::FilePath generated_name = net::GenerateFileName(url,
127                                                             content_disposition,
128                                                             std::string(),
129                                                             suggested_filename,
130                                                             mime_type,
131                                                             "download");
132       if (!base::PathExists(suggested_directory))
133         base::CreateDirectory(suggested_directory);
134     
135       base::FilePath suggested_path(suggested_directory.Append(generated_name));
136       BrowserThread::PostTask(
137           BrowserThread::UI, FROM_HERE, base::Bind(callback, suggested_path));
138     }

140     void 
141     ThrustShellDownloadManagerDelegate::OnDownloadPathGenerated(
142         uint32 download_id,
143         const DownloadTargetCallback& callback,
144         const base::FilePath& suggested_path) 
145     {
146       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::UI));
147       if (suppress_prompting_) {
148         // Testing exit.
149         callback.Run(suggested_path, DownloadItem::TARGET_DISPOSITION_OVERWRITE,
150                      DOWNLOAD_DANGER_TYPE_NOT_DANGEROUS,
151                      suggested_path.AddExtension(FILE_PATH_LITERAL(".crdownload")));
152         return;
153       }
154     
155       ChooseDownloadPath(download_id, callback, suggested_path);
156     }
