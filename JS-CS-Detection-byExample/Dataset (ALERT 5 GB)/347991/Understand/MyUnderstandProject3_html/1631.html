100     void 
101     EnsureMethodsLoaded() 
102     {
103       static bool attempted_load = false;
104       if (attempted_load)
105         return;
106       attempted_load = true;
107     
108       void* dbusmenu_lib = dlopen("libdbusmenu-glib.so", RTLD_LAZY);
109       if (!dbusmenu_lib)
110         dbusmenu_lib = dlopen("libdbusmenu-glib.so.4", RTLD_LAZY);
111       if (!dbusmenu_lib)
112         return;
113     
114       // DbusmenuMenuItem methods.
115       menuitem_new = reinterpret_cast<dbusmenu_menuitem_new_func>(
116           dlsym(dbusmenu_lib, "dbusmenu_menuitem_new"));
117       menuitem_new_with_id = reinterpret_cast<dbusmenu_menuitem_new_with_id_func>(
118           dlsym(dbusmenu_lib, "dbusmenu_menuitem_new_with_id"));
119       menuitem_get_id = reinterpret_cast<dbusmenu_menuitem_get_id_func>(
120           dlsym(dbusmenu_lib, "dbusmenu_menuitem_get_id"));
121       menuitem_get_children = reinterpret_cast<dbusmenu_menuitem_get_children_func>(
122           dlsym(dbusmenu_lib, "dbusmenu_menuitem_get_children"));
123       menuitem_take_children =
124           reinterpret_cast<dbusmenu_menuitem_get_children_func>(
125               dlsym(dbusmenu_lib, "dbusmenu_menuitem_take_children"));
126       menuitem_child_append = reinterpret_cast<dbusmenu_menuitem_child_append_func>(
127           dlsym(dbusmenu_lib, "dbusmenu_menuitem_child_append"));
128       menuitem_property_set = reinterpret_cast<dbusmenu_menuitem_property_set_func>(
129           dlsym(dbusmenu_lib, "dbusmenu_menuitem_property_set"));
130       menuitem_property_set_variant =
131           reinterpret_cast<dbusmenu_menuitem_property_set_variant_func>(
132               dlsym(dbusmenu_lib, "dbusmenu_menuitem_property_set_variant"));
133       menuitem_property_set_bool =
134           reinterpret_cast<dbusmenu_menuitem_property_set_bool_func>(
135               dlsym(dbusmenu_lib, "dbusmenu_menuitem_property_set_bool"));
136       menuitem_property_set_int =
137           reinterpret_cast<dbusmenu_menuitem_property_set_int_func>(
138               dlsym(dbusmenu_lib, "dbusmenu_menuitem_property_set_int"));
139     
140       // DbusmenuServer methods.
141       server_new = reinterpret_cast<dbusmenu_server_new_func>(
142           dlsym(dbusmenu_lib, "dbusmenu_server_new"));
143       server_set_root = reinterpret_cast<dbusmenu_server_set_root_func>(
144           dlsym(dbusmenu_lib, "dbusmenu_server_set_root"));
145     }

147     ui::MenuModel* 
148     ModelForMenuItem(
149         DbusmenuMenuitem* item) 
150     {
151       return reinterpret_cast<ui::MenuModel*>(
152           g_object_get_data(G_OBJECT(item), "model"));
153     }

155     bool 
156     GetMenuItemID(
157         DbusmenuMenuitem* item, 
158         int *id) 
159     {
160       gpointer id_ptr = g_object_get_data(G_OBJECT(item), "menu-id");
161       if (id_ptr != NULL) {
162         *id = GPOINTER_TO_INT(id_ptr) - 1;
163         return true;
164       }
165     
166       return false;
167     }
168     
169     void 
170     SetMenuItemID(
171         DbusmenuMenuitem* item, 
172         int id) 
173     {
174       DCHECK_GE(id, 0);
175     
176       // Add 1 to the menu_id to avoid setting zero (null) to "menu-id".
177       g_object_set_data(G_OBJECT(item), "menu-id", GINT_TO_POINTER(id + 1));
178     }

182     GlobalMenuBarX11::GlobalMenuBarX11(
183         ThrustWindow* window)
184       : window_(window),
185         xid_(window_->GetNativeWindow()->GetHost()->GetAcceleratedWidget()),
186         server_(NULL) 
187     {
188       EnsureMethodsLoaded();
189       if (server_new)
190         InitServer(xid_);
191     
192       GlobalMenuBarRegistrarX11::GetInstance()->OnWindowMapped(xid_);
193     }
194     
195     GlobalMenuBarX11::~GlobalMenuBarX11() 
196     {
197       if (IsServerStarted())
198         g_object_unref(server_);
199     
200       GlobalMenuBarRegistrarX11::GetInstance()->OnWindowUnmapped(xid_);
201     }

211     void 
212     GlobalMenuBarX11::SetMenu(
213         ui::MenuModel* menu_model) 
214     {
215       if (!IsServerStarted())
216         return;
217     
218       DbusmenuMenuitem* root_item = menuitem_new();
219       menuitem_property_set(root_item, kPropertyLabel, "Root");
220       menuitem_property_set_bool(root_item, kPropertyVisible, true);
221       BuildMenuFromModel(menu_model, root_item);
222     
223       server_set_root(server_, root_item);
224       g_object_unref(root_item);
225     }

228     GlobalMenuBarX11::IsServerStarted() const 
229     {
230       return server_;
231     }

234     GlobalMenuBarX11::InitServer(
235         gfx::AcceleratedWidget xid) 
236     {
237       std::string path = GetPathForWindow(xid);
238       server_ = server_new(path.c_str());

241     void 
242     GlobalMenuBarX11::BuildMenuFromModel(
243         ui::MenuModel* model,
244         DbusmenuMenuitem* parent) 
245     {
246       for (int i = 0; i < model->GetItemCount(); ++i) {
247         DbusmenuMenuitem* item = menuitem_new();
248         menuitem_property_set_bool(item, kPropertyVisible, model->IsVisibleAt(i));
249     
250         ui::MenuModel::ItemType type = model->GetTypeAt(i);
251         if (type == ui::MenuModel::TYPE_SEPARATOR) {
252           menuitem_property_set(item, kPropertyType, kTypeSeparator);
253         } else {
254           std::string label = ui::ConvertAcceleratorsFromWindowsStyle(
255               base::UTF16ToUTF8(model->GetLabelAt(i)));
256           menuitem_property_set(item, kPropertyLabel, label.c_str());
257           menuitem_property_set_bool(item, kPropertyEnabled, model->IsEnabledAt(i));
258     
259           g_object_set_data(G_OBJECT(item), "model", model);
260           SetMenuItemID(item, i);
261     
262           if (type == ui::MenuModel::TYPE_SUBMENU) {
263             menuitem_property_set(item, kPropertyChildrenDisplay, kDisplaySubmenu);
264             g_signal_connect(item, "about-to-show",
265                              G_CALLBACK(OnSubMenuShowThunk), this);
266           } else {
267             ui::Accelerator accelerator;
268             if (model->GetAcceleratorAt(i, &accelerator))
269               RegisterAccelerator(item, accelerator);
270     
271             g_signal_connect(item, "item-activated",
272                              G_CALLBACK(OnItemActivatedThunk), this);
273     
274             if (type == ui::MenuModel::TYPE_CHECK ||
275                 type == ui::MenuModel::TYPE_RADIO) {
276               menuitem_property_set(item, kPropertyToggleType,
277                   type == ui::MenuModel::TYPE_CHECK ? kToggleCheck : kToggleRadio);
278               menuitem_property_set_int(item, kPropertyToggleState,
279                   model->IsItemCheckedAt(i));
280             }
281           }
282         }
283     
284         menuitem_child_append(parent, item);
285         g_object_unref(item);
286       }
287     }

289     void 
290     GlobalMenuBarX11::RegisterAccelerator(
291         DbusmenuMenuitem* item,
292         const ui::Accelerator& accelerator) 
293     {
294       // A translation of libdbusmenu-gtk's menuitem_property_set_shortcut()
295       // translated from GDK types to ui::Accelerator types.
296       GVariantBuilder builder;
297       g_variant_builder_init(&builder, G_VARIANT_TYPE_ARRAY);
298     
299       if (accelerator.IsCtrlDown())
300         g_variant_builder_add(&builder, "s", "Control");
301       if (accelerator.IsAltDown())
302         g_variant_builder_add(&builder, "s", "Alt");
303       if (accelerator.IsShiftDown())
304         g_variant_builder_add(&builder, "s", "Shift");
305     
306       char* name = XKeysymToString(XKeysymForWindowsKeyCode(
307           accelerator.key_code(), false));
308       if (!name) {
309         NOTIMPLEMENTED();
310         return;
311       }
312       g_variant_builder_add(&builder, "s", name);
313     
314       GVariant* inside_array = g_variant_builder_end(&builder);
315       g_variant_builder_init(&builder, G_VARIANT_TYPE_ARRAY);
316       g_variant_builder_add_value(&builder, inside_array);
317       GVariant* outside_array = g_variant_builder_end(&builder);
318     
319       menuitem_property_set_variant(item, kPropertyShortcut, outside_array);
320     }

322     void 
323     GlobalMenuBarX11::OnItemActivated(
324         DbusmenuMenuitem* item,
325         unsigned int timestamp) 
326     {
327       int id;
328       ui::MenuModel* model = ModelForMenuItem(item);
329       if (model && GetMenuItemID(item, &id))
330         model->ActivatedAt(id, 0);
331     }

333     void 
334     GlobalMenuBarX11::OnSubMenuShow(
335         DbusmenuMenuitem* item) 
336     {
337       int id;
338       ui::MenuModel* model = ModelForMenuItem(item);
339       if (!model || !GetMenuItemID(item, &id))
340         return;
341     
342       // Clear children.
343       GList *children = menuitem_take_children(item);
344       g_list_foreach(children, reinterpret_cast<GFunc>(g_object_unref), NULL);
345       g_list_free(children);
346     
347       // Build children.
348       BuildMenuFromModel(model->GetSubmenuModelAt(id), item);
349     }
