23     ObjectBackedNativeHandler::ObjectBackedNativeHandler(ScriptContext* context)
24         : router_data_(context->v8_context()->GetIsolate()),
25           context_(context),
26           object_template_(
27               v8::ObjectTemplate::New(context->v8_context()->GetIsolate())) {
28     }
29     
30     ObjectBackedNativeHandler::~ObjectBackedNativeHandler() {
31       Invalidate();
32     }

34     v8::Handle<v8::Object> ObjectBackedNativeHandler::NewInstance() {
35       return object_template_.NewHandle(v8::Isolate::GetCurrent())->NewInstance();
36     }

38     // static
39     void ObjectBackedNativeHandler::Router(
40         const v8::FunctionCallbackInfo<v8::Value>& args) {
41       v8::HandleScope handle_scope(args.GetIsolate());
42       v8::Handle<v8::Object> data = args.Data().As<v8::Object>();
43     
44       v8::Handle<v8::Value> handler_function_value =
45           data->Get(v8::String::NewFromUtf8(args.GetIsolate(), kHandlerFunction));
46       // See comment in header file for why we do this.
47       if (handler_function_value.IsEmpty() ||
48           handler_function_value->IsUndefined()) {
49         LOG(INFO) << "Extension view no longer exists";
50         return;
51       }
52       DCHECK(handler_function_value->IsExternal());
53       static_cast<HandlerFunction*>(
54           handler_function_value.As<v8::External>()->Value())->Run(args);
55     }
56     
57     void ObjectBackedNativeHandler::RouteFunction(
58         const std::string& name,
59         const HandlerFunction& handler_function) {
60       v8::Isolate* isolate = v8::Isolate::GetCurrent();
61       v8::HandleScope handle_scope(isolate);
62       v8::Context::Scope context_scope(context_->v8_context());
63     
64       v8::Local<v8::Object> data = v8::Object::New(isolate);
65       data->Set(
66           v8::String::NewFromUtf8(isolate, kHandlerFunction),
67           v8::External::New(isolate, new HandlerFunction(handler_function)));
68       v8::Handle<v8::FunctionTemplate> function_template =
69           v8::FunctionTemplate::New(isolate, Router, data);
70       object_template_.NewHandle(isolate)
71           ->Set(isolate, name.c_str(), function_template);
72       router_data_.Append(data);
73     }
74     
75     v8::Isolate* ObjectBackedNativeHandler::GetIsolate() const {
76       return context_->isolate();
77     }
78     
79     void ObjectBackedNativeHandler::Invalidate() {
80       if (!is_valid())
81         return;
82       v8::Isolate* isolate = v8::Isolate::GetCurrent();
83       v8::HandleScope handle_scope(isolate);
84       v8::Context::Scope context_scope(context_->v8_context());
85     
86       for (size_t i = 0; i < router_data_.Size(); i++) {
87         v8::Handle<v8::Object> data = router_data_.Get(i);
88         v8::Handle<v8::Value> handler_function_value =
89             data->Get(v8::String::NewFromUtf8(isolate, kHandlerFunction));
90         CHECK(!handler_function_value.IsEmpty());
91         delete static_cast<HandlerFunction*>(
92             handler_function_value.As<v8::External>()->Value());
93         data->Delete(v8::String::NewFromUtf8(isolate, kHandlerFunction));
94       }
95       router_data_.Clear();
96       object_template_.reset();
97       context_ = NULL;
98       NativeHandler::Invalidate();
99     }
