119     bool 
120     ThrustShellNetworkDelegate::OnCanGetCookies(
121         const net::URLRequest& request,
122         const net::CookieList& cookie_list) 
123     {
124       return true;
125       net::StaticCookiePolicy::Type policy_type = g_accept_all_cookies ?
126           net::StaticCookiePolicy::ALLOW_ALL_COOKIES :
127           net::StaticCookiePolicy::BLOCK_ALL_THIRD_PARTY_COOKIES;
128       net::StaticCookiePolicy policy(policy_type);
129       int rv = policy.CanGetCookies(
130           request.url(), request.first_party_for_cookies());
131       return rv == net::OK;
132     }
