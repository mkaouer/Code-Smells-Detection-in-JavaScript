206     @Test
207     public void testXmlSchema() throws IOException, RuleSetNotFoundException, ParserConfigurationException, SAXException {
208         if (isJdk14) {
209             // ignore failure with jdk 1.4
210             return;
211         }
212 
213 		SAXParserFactory saxParserFactory = SAXParserFactory.newInstance();
214 		saxParserFactory.setValidating(true);
215 		saxParserFactory.setNamespaceAware(true);
216 
217 		// Hope we're using Xerces, or this may not work!
218 		// Note: Features are listed here http://xerces.apache.org/xerces2-j/features.html
219 		saxParserFactory.setFeature("http://xml.org/sax/features/validation", true);
220 		saxParserFactory.setFeature("http://apache.org/xml/features/validation/schema", true);
221 		saxParserFactory.setFeature("http://apache.org/xml/features/validation/schema-full-checking", true);
222 		
223 		boolean allValid = true;
224 		List<String> ruleSetFileNames = getRuleSetFileNames();
225 		for (String fileName : ruleSetFileNames) {
226 	        InputStream inputStream = loadResourceAsStream(fileName);
227 
228 			SAXParser saxParser = saxParserFactory.newSAXParser();
229 			ValidateDefaultHandler validateDefaultHandler = new ValidateDefaultHandler("etc/ruleset_xml_schema.xsd");
230 			saxParser.parse(inputStream, validateDefaultHandler);
231 			allValid = allValid && validateDefaultHandler.isValid();
232 			if (! validateDefaultHandler.isValid()) {
233 				System.err.println("Validation against XML Schema failed for: " + fileName);
234 			}
235 		}
236 		assertTrue("All XML must parse without producing validation messages.", allValid);
237     }

239     @Test
240     public void testDtd() throws IOException, RuleSetNotFoundException, ParserConfigurationException, SAXException  {
241 		SAXParserFactory saxParserFactory = SAXParserFactory.newInstance();
242 		saxParserFactory.setValidating(true);
243 		saxParserFactory.setNamespaceAware(true);
244 
245 		boolean allValid = true;
246 		List<String> ruleSetFileNames = getRuleSetFileNames();
247 		for (String fileName : ruleSetFileNames) {
248 	        InputStream inputStream = loadResourceAsStream(fileName);
249 
250 	        // Read file into memory
251 	        StringBuffer buf = new StringBuffer(64 * 1024);
252 	        BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
253 	        String line;
254 	        while ((line = reader.readLine()) != null) {
255 	        	buf.append(line);
256 	        	buf.append(PMD.EOL);
257 	        }
258 	        reader.close();
259 
260 	        // Remove XML Schema stuff, replace with DTD
261 	        String file = buf.toString();
262 	        file = file.replaceAll("<\\?xml version=\"1.0\"\\?>", "");
263 	        file = file.replaceAll("xmlns=\"http://pmd.sf.net/ruleset/1.0.0\"", "");
264 	        file = file.replaceAll("xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"", "");
265 	        file = file.replaceAll("xsi:schemaLocation=\"http://pmd.sf.net/ruleset/1.0.0 http://pmd.sf.net/ruleset_xml_schema.xsd\"", "");
266 	        file = file.replaceAll("xsi:noNamespaceSchemaLocation=\"http://pmd.sf.net/ruleset_xml_schema.xsd\"", "");
267 
268 	        file = "<?xml version=\"1.0\"?>" + PMD.EOL + "<!DOCTYPE ruleset SYSTEM \"file://" +
269 	            System.getProperty("user.dir") + "/etc/ruleset.dtd\">" + PMD.EOL + file;
270 
271 	        inputStream = new ByteArrayInputStream(file.getBytes());
272 
273 			SAXParser saxParser = saxParserFactory.newSAXParser();
274 			ValidateDefaultHandler validateDefaultHandler = new ValidateDefaultHandler("etc/ruleset.dtd");
275 			saxParser.parse(inputStream, validateDefaultHandler);
276 			allValid = allValid && validateDefaultHandler.isValid();
277 			if (! validateDefaultHandler.isValid()) {
278 				System.err.println("Validation against DTD failed for: " + fileName);
279 			}
280 		}
281 		assertTrue("All XML must parse without producing validation messages.", allValid);
282     }

285     private List<String> getRuleSetFileNames() throws IOException, RuleSetNotFoundException {
286         Properties properties = new Properties();
287         properties.load(ResourceLoader.loadResourceAsStream("rulesets/rulesets.properties"));
288         String fileNames = properties.getProperty("rulesets.testnames");
289         StringTokenizer st = new StringTokenizer(fileNames, ",");
290         List<String> ruleSetFileNames = new ArrayList<String>();
291         while (st.hasMoreTokens()) {
292         	ruleSetFileNames.add(st.nextToken());
293         }
294         return ruleSetFileNames;
295     }

297     private class ValidateDefaultHandler extends DefaultHandler {
298     	private final String validateDocument;
299 		private boolean valid = true;
300 		
301 		public ValidateDefaultHandler(String validateDocument) {
302 			this.validateDocument = validateDocument;
303 		}
304 		public boolean isValid() {
305 			return valid;
306 		}
307 		public void error(SAXParseException e) throws SAXException {
308 			log("Error", e);
309 		}
310 		public void fatalError(SAXParseException e) throws SAXException {
311 			log("FatalError", e);
312 		}
313 		public void warning(SAXParseException e) throws SAXException {
314 			log("Warning", e);
315 		}
316 		private void log(String prefix, SAXParseException e) {
317 			String message = prefix + " at (" + e.getLineNumber() + ", " + e.getColumnNumber() + "): " + e.getMessage();
318 			System.err.println(message);
319 			valid = false;
320 		}
321 		public InputSource resolveEntity(String publicId, String systemId) throws IOException, SAXException {
322 			if ("http://pmd.sf.net/ruleset_xml_schema.xsd".equals(systemId) || systemId.endsWith("ruleset.dtd")) {
323 		        try {
324 					InputStream inputStream = loadResourceAsStream(validateDocument);
325 					return new InputSource(inputStream);
326 				} catch (RuleSetNotFoundException e) {
327 					System.err.println(e.getMessage());
328 					throw new IOException(e.getMessage());
329 				}
330 			} else {
331 				throw new IllegalArgumentException("No clue how to handle: publicId=" + publicId + ", systemId=" + systemId);
332 			}
333 		}
334     }

336     private InputStream loadResourceAsStream(String resource) throws RuleSetNotFoundException {
337     	InputStream inputStream = ResourceLoader.loadResourceAsStream(resource, this.getClass().getClassLoader());
338         if (inputStream == null) {
339             throw new RuleSetNotFoundException("Can't find resource " + resource + "  Make sure the resource is a valid file or URL or is on the CLASSPATH.  Here's the current classpath: " + System.getProperty("java.class.path"));
340         }
341         return inputStream;
342     }
