219     private Object getProperty(PropertyDescriptor descriptor) {
220     	    	
221     	if (descriptor.maxValueCount() > 1) propertyGetError(descriptor, true);
222     	
223     	String rawValue = properties.getProperty(descriptor.name());
224     	
225         return rawValue == null || rawValue.length() == 0 ?
226         	descriptor.defaultValue() :
227         	descriptor.valueFrom(rawValue);
228     }

231     	    	
232     	if (descriptor.maxValueCount() > 1) propertySetError(descriptor, true);
233     	
234     	properties.setProperty(descriptor.name(), descriptor.asDelimitedString(value));
235     }

295     /**
296      * Test if rules are equals. Rules are equals if
297      * 1. they have the same implementation class
298      * 2. they have the same name
299      * 3. they have the same priority
300      * 4. they share the same properties/values
301      */
302     public boolean equals(Object o) {
303         if (o == null) {
304             return false; // trivial
305         }
306 
307         if (this == o) {
308             return true;  // trivial
309         }
310 
311         Rule rule = null;
312         boolean equality = this.getClass().getName().equals(o.getClass().getName());
313 
314         if (equality) {
315             rule = (Rule) o;
316             equality = this.getName().equals(rule.getName())
317                     && this.getPriority() == rule.getPriority()
318                     && this.getProperties().equals(rule.getProperties());
319         }
320 
321         return equality;
322     }

449     public static boolean importsPackage(ASTCompilationUnit node, String packageName) {
450     	
451         List<ASTImportDeclaration> nodes = node.findChildrenOfType(ASTImportDeclaration.class);
452         for (ASTImportDeclaration n: nodes) {
453             if (n.getPackageName().startsWith(packageName)) {
454                 return true;
455             }
456         }
457         return false;
458     }
