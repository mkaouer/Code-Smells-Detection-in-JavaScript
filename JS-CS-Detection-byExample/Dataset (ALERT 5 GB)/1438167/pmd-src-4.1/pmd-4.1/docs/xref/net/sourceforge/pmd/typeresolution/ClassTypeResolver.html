89      public Object visit(ASTCompilationUnit node, Object data) {
90  
91          String className = null;
92          try {
93              importedOnDemand = new ArrayList<String>();
94              className = getClassName(node);
95              populateClassName(node, className);
96          } catch (ClassNotFoundException e) {
97              LOG.log(Level.FINE, "Could not find class " + className);
98          } catch (NoClassDefFoundError e) {
99              LOG.log(Level.WARNING, "Could not find class " + className);
100         } finally {
101             populateImports(node);
102         }
103         return super.visit(node, data);
104     }

106     private String getClassName(ASTCompilationUnit node) {
107         String className = null;
108 
109         ASTClassOrInterfaceDeclaration decl = node.getFirstChildOfType(ASTClassOrInterfaceDeclaration.class);
110         if (decl != null) {
111             ASTPackageDeclaration pkgDecl = node.getFirstChildOfType(ASTPackageDeclaration.class);
112             if (pkgDecl == null) {
113                 className = decl.getImage();
114             } else {
115                 importedOnDemand.add(((ASTName) pkgDecl.jjtGetChild(0)).getImage());
116                 className = ((ASTName) pkgDecl.jjtGetChild(0)).getImage() + "." + decl.getImage();
117             }
118         }
119         return className;
120     }

122     /**
123      * If the outer class wasn't found then we'll get in here
124      * 
125      * @param node
126      */
127     private void populateImports(ASTCompilationUnit node) {
128         List<ASTImportDeclaration> theImportDeclarations = node.findChildrenOfType(ASTImportDeclaration.class);
129         importedClasses = new HashMap<String, String>();
130 
131         // go through the imports
132         for (ASTImportDeclaration anImportDeclaration : theImportDeclarations) {
133             String strPackage = anImportDeclaration.getPackageName();
134             if (anImportDeclaration.isImportOnDemand()) {
135                 importedOnDemand.add(strPackage);
136             } else if (!anImportDeclaration.isImportOnDemand()) {
137                 String strName = anImportDeclaration.getImportedName();
138                 importedClasses.put(strName, strName);
139                 importedClasses.put(strName.substring(strPackage.length() + 1), strName);
140             }
141         }
142 
143         importedClasses.putAll(myJavaLang);
144     }

146     private void populateClassName(ASTCompilationUnit node, String className) throws ClassNotFoundException {
147         if (className != null) {
148             Class c = pmdClassLoader.loadClass(className);
149             node.setType(c);
150             importedClasses = pmdClassLoader.getImportedClasses(className);
151         }
152     }

154     public Object visit(ASTImportDeclaration node, Object data) {
155 
156         ASTName importedType = (ASTName) node.jjtGetChild(0);
157         if(importedType.getType() != null){
158            node.setType(importedType.getType()); 
159         } else {
160             populateType(node, importedType.getImage());
161         }
162         
163         if(node.getType() != null){
164             node.setPackage(node.getType().getPackage());
165         }
166 
167         return data;
168     }

176     public Object visit(ASTName node, Object data) {
177         /*
178          * Only doing this for nodes where getNameDeclaration is null this means
179          * it's not a named node, i.e. Static reference or Annotation Doing this
180          * for memory - TODO: Investigate if there is a valid memory concern or
181          * not
182          */
183         if (node.getNameDeclaration() == null) {
184             String name = node.getImage();
185             if (name.indexOf('.') != -1) {
186                 name = name.substring(0, name.indexOf('.'));
187             }
188             populateType(node, name);
189         }
190         return super.visit(node, data);
191     }

193     public Object visit(ASTVariableDeclaratorId node, Object data) {
194         if (node == null || node.getNameDeclaration() == null) {
195             return super.visit(node, data);
196         }
197         String name = node.getNameDeclaration().getTypeImage();
198         if (name.indexOf('.') != -1) {
199             name = name.substring(0, name.indexOf('.'));
200         }
201         populateType(node, name);
202         return super.visit(node, data);
203     }

205     private void populateType(TypeNode node, String className) {
206 
207         String qualifiedName = className;
208         Class myType = myPrimitiveTypes.get(className);
209         if (myType == null && importedClasses != null) {
210             if (importedClasses.containsKey(className)) {
211                 qualifiedName = importedClasses.get(className);
212             } else if (importedClasses.containsValue(className)) {
213                 qualifiedName = className;
214             }
215             if (qualifiedName != null) {
216                 try {
217                     /*
218                      * TODO - the map right now contains just class names. if we
219                      * use a map of classname/class then we don't have to hit
220                      * the class loader for every type - much faster
221                      */
222                     myType = pmdClassLoader.loadClass(qualifiedName);
223                 } catch (ClassNotFoundException e) {
224                     myType = processOnDemand(qualifiedName);
225                 }
226             }
227         }
228         if (myType != null) {
229             node.setType(myType);
230         }
231     }

245     private Class processOnDemand(String qualifiedName) {
246         for (String entry : importedOnDemand) {
247             try {
248                 return pmdClassLoader.loadClass(entry + "." + qualifiedName);
249             } catch (Throwable e) {
250             }
251         }
252         return null;
