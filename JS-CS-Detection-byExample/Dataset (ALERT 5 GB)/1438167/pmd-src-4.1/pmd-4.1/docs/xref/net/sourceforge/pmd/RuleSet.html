35      /**
36       * Add a new rule to this ruleset
37       *
38       * @param rule the rule to be added
39       */
40      public void addRule(Rule rule) {
41          if (rule == null) {
42              throw new RuntimeException("Null Rule reference added to a RuleSet; that's a bug somewhere in PMD");
43          }
44          rules.add(rule);
45      }

47      /**
48       * Returns the actual Collection of rules in this ruleset
49       *
50       * @return a Collection with the rules. All objects are of type {@link Rule}
51       */
52      public Collection<Rule> getRules() {
53          return rules;
54      }
55  
56      /**
57       * @return true if any rule in the RuleSet needs the DFA layer
58       */
59      public boolean usesDFA() {
60          for (Rule r: rules) {
61              if (r.usesDFA()) {
62                  return true;
63              }
64          }
65          return false;
66      }

68      /**
69       * Returns the Rule with the given name
70       *
71       * @param ruleName the name of the rule to find
72       * @return the rule or null if not found
73       */
74      public Rule getRuleByName(String ruleName) {
75          Rule rule = null;
76          for (Iterator<Rule> i = rules.iterator(); i.hasNext() && (rule == null);) {
77              Rule r = i.next();
78              if (r.getName().equals(ruleName)) {
79                  rule = r;
80              }
81          }
82          return rule;
83      }
84  
85      /**
86       * Add a whole RuleSet to this RuleSet
87       *
88       * @param ruleSet the RuleSet to add
89       */
90      public void addRuleSet(RuleSet ruleSet) {
91          rules.addAll(rules.size(), ruleSet.getRules());
92      }
93  
94      public void apply(List acuList, RuleContext ctx) {
95          long start = System.nanoTime();
96          for (Rule rule: rules) {
97              if (!rule.usesRuleChain()) {
98                  rule.apply(acuList, ctx);
99                  long end = System.nanoTime();
100                 Benchmark.mark(Benchmark.TYPE_RULE, rule.getName(), end - start, 1);
101                 start = end;
102             }
103         }
104     }
