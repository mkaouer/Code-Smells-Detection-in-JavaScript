78      /**
79       * Processes the file read by the reader against the rule set.
80       *
81       * @param reader     input stream reader
82       * @param ruleSets   set of rules to process against the file
83       * @param ctx        context in which PMD is operating. This contains the Renderer and
84       *                   whatnot
85       * @param sourceType the SourceType of the source
86       * @throws PMDException if the input could not be parsed or processed
87       */
88      public void processFile(Reader reader, RuleSets ruleSets, RuleContext ctx,
89                              SourceType sourceType) throws PMDException {
90          try {
91          	if (ruleSets.applies(ctx.getSourceCodeFile())) {
92  	            SourceTypeHandler sourceTypeHandler = SourceTypeHandlerBroker.getVisitorsFactoryForSourceType(sourceType);
93  	            ctx.setSourceType(sourceType);
94  	            Parser parser = sourceTypeHandler.getParser();
95  	            parser.setExcludeMarker(excludeMarker);
96  	            long start = System.nanoTime();
97  	            CompilationUnit rootNode = (CompilationUnit) parser.parse(reader);
98  	            ctx.excludeLines(parser.getExcludeMap());
99  	            long end = System.nanoTime();
100 	            Benchmark.mark(Benchmark.TYPE_PARSER, end - start, 0);
101 	            start = System.nanoTime();
102 	            sourceTypeHandler.getSymbolFacade().start(rootNode);
103 	            end = System.nanoTime();
104 	            Benchmark.mark(Benchmark.TYPE_SYMBOL_TABLE, end - start, 0);
105 	
106 	            Language language = SourceTypeToRuleLanguageMapper.getMappedLanguage(sourceType);
107 	
108 	            if (ruleSets.usesDFA(language)) {
109 	                start = System.nanoTime();
110 	                sourceTypeHandler.getDataFlowFacade().start(rootNode);
111 	                end = System.nanoTime();
112 	                Benchmark.mark(Benchmark.TYPE_DFA, end - start, 0);
113 	            }
114 	
115 	            if (ruleSets.usesTypeResolution(language)) {
116 	                start = System.nanoTime();
117 	                sourceTypeHandler.getTypeResolutionFacade(classLoader).start(rootNode);
118 	                end = System.nanoTime();
119 	                Benchmark.mark(Benchmark.TYPE_TYPE_RESOLUTION, end - start, 0);
120 	            }
121 	
122 	            List<CompilationUnit> acus = new ArrayList<CompilationUnit>();
123 	            acus.add(rootNode);
124 	
125 	            ruleSets.apply(acus, ctx, language);
126         	}
127         } catch (ParseException pe) {
128             throw new PMDException("Error while parsing "
129                     + ctx.getSourceCodeFilename(), pe);
130         } catch (Exception e) {
131             throw new PMDException("Error while processing "
132                     + ctx.getSourceCodeFilename(), e);
133         } finally {
134             try {
135                 reader.close();
136             } catch (IOException e) {}
137         }
138     }

140     /**
141      * Get the SourceType of the source file with given name. This depends on the fileName
142      * extension, and the java version.
143      * <p/>
144      * For compatibility with older code that does not always pass in a correct filename,
145      * unrecognized files are assumed to be java files.
146      *
147      * @param fileName Name of the file, can be absolute, or simple.
148      * @return the SourceType
149      */
150     private SourceType getSourceTypeOfFile(String fileName) {
151         SourceType sourceType = sourceTypeDiscoverer.getSourceTypeOfFile(fileName);
152         if (sourceType == null) {
153             // For compatibility with older code that does not always pass in
154             // a correct filename.
155             sourceType = sourceTypeDiscoverer.getSourceTypeOfJavaFiles();
156         }
157         return sourceType;
158     }

267 	 * Create a ClassLoader which loads classes using a CLASSPATH like String.
268 	 * If the String looks like a URL to a file (e.g. starts with <code>file://</code>)
269 	 * the file will be read with each line representing an entry on the classpath.
270 	 * <p>
271 	 * The ClassLoader used to load the <code>net.sourceforge.pmd.PMD</code> class
272 	 * will be used as the parent ClassLoader of the created ClassLoader.

274 	 * @param classpath The classpath String.
275 	 * @return A ClassLoader
276 	 * @throws IOException
277 	 * @see ClasspathClassLoader
278 	 */

280 		ClassLoader classLoader = PMD.class.getClassLoader();
281 		if (classpath != null) {
282 			classLoader = new ClasspathClassLoader(classpath, classLoader);
283 		}
284 		return classLoader;

287 	private static void doPMD(CommandLineOptions opts){
288         long startFiles = System.nanoTime();
289         SourceFileSelector fileSelector = new SourceFileSelector();
290 
291         fileSelector.setSelectJavaFiles(opts.isCheckJavaFiles());
292         fileSelector.setSelectJspFiles(opts.isCheckJspFiles());
293 
294         List<DataSource> files;
295         if (opts.containsCommaSeparatedFileList()) {
296             files = collectFromCommaDelimitedString(opts.getInputPath(),
297                     fileSelector);
298         } else {
299             files = collectFilesFromOneName(opts.getInputPath(), fileSelector);
300         }
301         long endFiles = System.nanoTime();
302         Benchmark.mark(Benchmark.TYPE_COLLECT_FILES, endFiles - startFiles, 0);
303 
304         SourceType sourceType;
305         if (opts.getTargetJDK().equals("1.3")) {
306             LOG.fine("In JDK 1.3 mode");
307             sourceType = SourceType.JAVA_13;
308         } else if (opts.getTargetJDK().equals("1.5")) {
309             LOG.fine("In JDK 1.5 mode");
310             sourceType = SourceType.JAVA_15;
311         } else if (opts.getTargetJDK().equals("1.6")) {
312             LOG.fine("In JDK 1.6 mode");
313             sourceType = SourceType.JAVA_16;
314         } else if (opts.getTargetJDK().equals("1.7")) {
315             LOG.fine("In JDK 1.7 mode");
316             sourceType = SourceType.JAVA_17;
317         } else {
318             LOG.fine("In JDK 1.4 mode");
319             sourceType = SourceType.JAVA_14;
320         }
321 
322         final ClassLoader classLoader;
323         try {
324         	classLoader = createClasspathClassLoader(opts.getAuxClasspath());
325         } catch (IOException e) {
326         	LOG.log(Level.SEVERE, "Bad -auxclasspath argument", e);
327         	System.out.println(opts.usage());        	
328         	return;
329         }
330         
331         long reportStart, reportEnd;
332         Renderer renderer;
333         Writer w = null;
334 
335         reportStart = System.nanoTime();
336         try {
337             renderer = opts.createRenderer();
338             List<Renderer> renderers = new LinkedList<Renderer>();
339             renderers.add(renderer);
340             if (opts.getReportFile() != null) {
341                 w = new BufferedWriter(new FileWriter(opts.getReportFile()));
342             } else {
343                 w = new OutputStreamWriter(System.out);
344             }
345             renderer.setWriter(w);
346             renderer.start();
347 
348             reportEnd = System.nanoTime();
349             Benchmark.mark(Benchmark.TYPE_REPORTING, reportEnd - reportStart, 0);
350     
351             RuleContext ctx = new RuleContext();
352 
353             try {
354                 long startLoadRules = System.nanoTime();
355                 RuleSetFactory ruleSetFactory = new RuleSetFactory();
356                 ruleSetFactory.setMinimumPriority(opts.getMinPriority());
357     
358                 RuleSets rulesets = ruleSetFactory.createRuleSets(opts.getRulesets());
359                 printRuleNamesInDebug(rulesets);
360                 long endLoadRules = System.nanoTime();
361                 Benchmark.mark(Benchmark.TYPE_LOAD_RULES, endLoadRules - startLoadRules, 0);
362     
363                 processFiles(opts.getCpus(), ruleSetFactory, sourceType, files, ctx, renderers,
364                         opts.stressTestEnabled(),
365                         opts.getRulesets(), opts.shortNamesEnabled(),
366                         opts.getInputPath(), opts.getEncoding(), opts.getExcludeMarker(), classLoader);
367             } catch (RuleSetNotFoundException rsnfe) {
368                 LOG.log(Level.SEVERE, "Ruleset not found", rsnfe);
369                 System.out.println(opts.usage());
370             }
371 
372             reportStart = System.nanoTime();
373             renderer.end();
374             w.write(EOL);
375             w.flush();
376             if (opts.getReportFile() != null) {
377                 w.close();
378                 w = null;
379             }
380         } catch (Exception e) {
381         	String message = e.getMessage();
382         	if (message != null) {
383         		LOG.severe(message);
384         	} else {
385                 LOG.log(Level.SEVERE, "Exception during processing", e);
386         	}
387             
388             LOG.log(Level.FINE, "Exception during processing", e);	//Only displayed when debug logging is on
389  
390             LOG.info(opts.usage());
391         } finally {
392             if (opts.getReportFile() != null && w != null) {
393                 try {
394                     w.close();
395                 } catch (Exception e) {
396                     System.out.println(e.getMessage());
397                 }
398             }
399             reportEnd = System.nanoTime();
400             Benchmark.mark(Benchmark.TYPE_REPORTING, reportEnd - reportStart, 0);
401         }
402     }

404     public static void main(String[] args) {
405         long start = System.nanoTime();
406         final CommandLineOptions opts = new CommandLineOptions(args);
407 
408         final Level logLevel = opts.debugEnabled() ? Level.FINER : Level.INFO;
409         final Handler logHandler = new ConsoleLogHandler();
410         final ScopedLogHandlersManager logHandlerManager = new ScopedLogHandlersManager(logLevel, logHandler);
411         final Level oldLogLevel = LOG.getLevel();

413         try{
414             doPMD(opts);
415         }finally{
416             logHandlerManager.close();
417             LOG.setLevel(oldLogLevel);
418             if (opts.benchmark()) {
419                 long end = System.nanoTime();
420                 Benchmark.mark(Benchmark.TYPE_TOTAL_PMD, end - start, 0);
421                 System.err.println(Benchmark.report());
422             }
423         }

426     private static class PmdRunnable extends PMD implements Callable<Report> {
427         private final ExecutorService executor;
428         private final DataSource dataSource;
429         private final String fileName;
430         private final String encoding;
431         private final String rulesets;
432         private final List<Renderer> renderers;
433 
434         public PmdRunnable(ExecutorService executor, DataSource dataSource, String fileName, SourceType sourceType,
435                 List<Renderer> renderers, String encoding, String rulesets, String excludeMarker, ClassLoader classLoader) {
436             this.executor = executor;
437             this.dataSource = dataSource;
438             this.fileName = fileName;
439             this.encoding = encoding;
440             this.rulesets = rulesets;
441             this.renderers = renderers;
442 
443             setJavaVersion(sourceType);
444             setExcludeMarker(excludeMarker);
445             setClassLoader(classLoader);
446         }
447 
448         public Report call() {
449             PmdThread thread = (PmdThread) Thread.currentThread();
450 
451             RuleContext ctx = thread.getRuleContext();
452             RuleSets rs = thread.getRuleSets(rulesets);
453 
454             Report report = new Report();
455             ctx.setReport(report);
456 
457             ctx.setSourceCodeFilename(fileName);
458             if (LOG.isLoggable(Level.FINE)) {
459                 LOG.fine("Processing " + ctx.getSourceCodeFilename());
460             }
461             for(Renderer r: renderers) {
462                 r.startFileAnalysis(dataSource);
463             }
464 
465             try {
466                 InputStream stream = new BufferedInputStream(dataSource.getInputStream());
467                 processFile(stream, encoding, rs, ctx);
468             } catch (PMDException pmde) {
469                 LOG.log(Level.FINE, "Error while processing file", pmde.getCause());
470 
471                 report.addError(
472                         new Report.ProcessingError(pmde.getMessage(),
473                         fileName));
474             } catch (IOException ioe) {
475                 // unexpected exception: log and stop executor service
476                 LOG.log(Level.FINE, "IOException during processing", ioe);
477 
478                 report.addError(
479                         new Report.ProcessingError(ioe.getMessage(),
480                         fileName));
481 
482                 executor.shutdownNow();
483             } catch (RuntimeException re) {
484                 // unexpected exception: log and stop executor service
485                 LOG.log(Level.FINE, "RuntimeException during processing", re);
486 
487                 report.addError(
488                         new Report.ProcessingError(re.getMessage(),
489                         fileName));
490 
491                 executor.shutdownNow();
492             }
493             return report;
494         }
495 
496     }

498     private static class PmdThreadFactory implements ThreadFactory {

500         private final RuleSetFactory ruleSetFactory;
501         private final RuleContext ctx;
502         private final AtomicInteger counter = new AtomicInteger();
503 
504         public PmdThreadFactory(RuleSetFactory ruleSetFactory, RuleContext ctx) {
505             this.ruleSetFactory = ruleSetFactory;
506             this.ctx = ctx;
507         }
508 
509         public Thread newThread(Runnable r) {
510             PmdThread t = new PmdThread(counter.incrementAndGet(), r, ruleSetFactory, ctx);
511             threadList.add(t);
512             return t;
513         }

515         public List<PmdThread> threadList = Collections.synchronizedList(new LinkedList<PmdThread>());

521         public PmdThread(int id, Runnable r, RuleSetFactory ruleSetFactory, RuleContext ctx) {
522             super(r, "PmdThread " + id);
523             this.id = id;
524             context = new RuleContext(ctx);
525             this.ruleSetFactory = ruleSetFactory;
526         }
527         
528         private int id;
529         private RuleContext context;
530         private RuleSets rulesets;
531         private RuleSetFactory ruleSetFactory;

534             return context;
535         }
536

540                     rulesets = ruleSetFactory.createRuleSets(rsList);
541                 } catch (Exception e) {
542                     e.printStackTrace();
543                 }
544             }
545             return rulesets;
546         }
547 
548         public String toString() {
549             return "PmdThread " + id;
550         }

555      * Do we have proper permissions to use multithreading?
556      */
557     private static final boolean mtSupported;
558     
559     static {
560         boolean error = false;
561         try {
562             /*
563              * ant task ran from Eclipse with jdk 1.5.0 raises an AccessControlException
564              * when shutdown is called. Standalone pmd or ant from command line are fine.
565              * 
566              * With jdk 1.6.0, ant task from Eclipse also works.
567              */
568             ExecutorService executor = Executors.newFixedThreadPool(1);
569             executor.shutdown();
570         } catch (RuntimeException e) {
571             error = true;
572         }
573         mtSupported = !error;
574     }

581     public static void processFiles(int threadCount, RuleSetFactory ruleSetFactory, SourceType sourceType, List<DataSource> files, RuleContext ctx,
582             List<Renderer> renderers, String rulesets, final boolean shortNamesEnabled, final String inputPath,

588     /**
589      * Run PMD on a list of files using multiple threads.
590      *
591      * @throws IOException If one of the files could not be read
592      */
593     public static void processFiles(int threadCount, RuleSetFactory ruleSetFactory, SourceType sourceType, List<DataSource> files, RuleContext ctx,
594             List<Renderer> renderers, boolean stressTestEnabled, String rulesets, final boolean shortNamesEnabled, final String inputPath,
595             String encoding, String excludeMarker, ClassLoader classLoader) {
596 
597         /*
598          * Check if multithreaded is supported. 
599          * ExecutorService can also be disabled if threadCount is not positive, e.g. using the
600          * "-cpus 0" command line option.
601          */
602         boolean useMT = mtSupported && (threadCount > 0);
603 
604         if (stressTestEnabled) {
605             // randomize processing order
606             Collections.shuffle(files);
607         } else {
608             Collections.sort(files, new Comparator<DataSource>() {
609                 public int compare(DataSource d1,DataSource d2) {
610                     String s1 = d1.getNiceFileName(shortNamesEnabled, inputPath);
611                     String s2 = d2.getNiceFileName(shortNamesEnabled, inputPath);
612                     return s1.compareTo(s2);
613                 }
614             });
615         }
616 
617         if (useMT) {
618             RuleSets rs = null;
619             try {
620                 rs = ruleSetFactory.createRuleSets(rulesets);
621             } catch (RuleSetNotFoundException rsnfe) {
622                 // should not happen: parent already created a ruleset
623             }
624             rs.start(ctx);
625 
626             PmdThreadFactory factory = new PmdThreadFactory(ruleSetFactory, ctx);
627             ExecutorService executor = Executors.newFixedThreadPool(threadCount, factory);
628             List<Future<Report>> tasks = new LinkedList<Future<Report>>();
629 
630             for (DataSource dataSource:files) {
631                 String niceFileName = dataSource.getNiceFileName(shortNamesEnabled,
632                         inputPath);
633     
634                 PmdRunnable r = new PmdRunnable(executor, dataSource, niceFileName, sourceType, renderers,
635                                                 encoding, rulesets, excludeMarker, classLoader);
636     
637                 Future<Report> future = executor.submit(r);
638                 tasks.add(future);
639             }
640             executor.shutdown();
641     
642             while (!tasks.isEmpty()) {
643                 Future<Report> future = tasks.remove(0);
644                 Report report = null;
645                 try {
646                     report = future.get();
647                 } catch (InterruptedException ie) {
648                     Thread.currentThread().interrupt();
649                     future.cancel(true);
650                 } catch (ExecutionException ee) {
651                     Throwable t = ee.getCause();
652                     if (t instanceof RuntimeException) {
653                         throw (RuntimeException) t;
654                     } else if (t instanceof Error) {
655                         throw (Error) t;
656                     } else {
657                         throw new IllegalStateException("PmdRunnable exception", t);
658                     }
659                 }
660     
661                 try {
662                     long start = System.nanoTime();
663                     for (Renderer r: renderers) {
664                         r.renderFileReport(report);
665                     }
666                     long end = System.nanoTime();
667                     Benchmark.mark(Benchmark.TYPE_REPORTING, end - start, 1);
668                 } catch (IOException ioe) {
669                 }
670             }
671 
672             try {
673                 rs.end(ctx);

831     /**
832      * Collects the files from the given <code>filename</code>.
833      *
834      * @param filename     the source from which to collect files
835      * @param fileSelector Filtering of wanted source files
836      * @return a list of files found at the given <code>filename</code>
837      * @throws RuntimeException if <code>filename</code> is not found
838      */
839     private static List<DataSource> collect(String filename, SourceFileSelector fileSelector) {
840         File inputFile = new File(filename);
841         if (!inputFile.exists()) {
842             throw new RuntimeException("File " + inputFile.getName()
843                     + " doesn't exist");
844         }
845         List<DataSource> dataSources = new ArrayList<DataSource>();
846         if (!inputFile.isDirectory()) {
847             if (filename.endsWith(".zip") || filename.endsWith(".jar")) {
848                 ZipFile zipFile;
849                 try {
850                     zipFile = new ZipFile(inputFile);
851                     Enumeration e = zipFile.entries();
852                     while (e.hasMoreElements()) {
853                         ZipEntry zipEntry = (ZipEntry) e.nextElement();
854                         if (fileSelector.isWantedFile(zipEntry.getName())) {
855                             dataSources.add(new ZipDataSource(zipFile, zipEntry));
856                         }
857                     }
858                 } catch (IOException ze) {
859                     throw new RuntimeException("Zip file " + inputFile.getName()
860                             + " can't be opened");
861                 }
862             } else {
863                 dataSources.add(new FileDataSource(inputFile));
864             }
865         } else {
866             FileFinder finder = new FileFinder();
867             List<File> files = finder.findFilesFrom(inputFile.getAbsolutePath(),
868                     new SourceFileOrDirectoryFilter(fileSelector), true);
869             for (File f: files) {
870                 dataSources.add(new FileDataSource(f));
871             }
872         }
873         return dataSources;
874     }
