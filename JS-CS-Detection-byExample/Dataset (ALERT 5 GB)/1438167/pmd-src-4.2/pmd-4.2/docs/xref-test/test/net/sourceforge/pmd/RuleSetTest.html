331     protected void verifyRuleSet(RuleSet IUT, int size, Set values) throws Throwable {
332 
333         RuleContext context = new RuleContext();
334         Set<RuleViolation> reportedValues = new HashSet<RuleViolation>();
335         context.setReport(new Report());
336         IUT.apply(makeCompilationUnits(), context);
337 
338         assertEquals("Invalid number of Violations Reported", size, context.getReport().size());
339 
340         Iterator violations = context.getReport().iterator();
341         while (violations.hasNext()) {
342             RuleViolation violation = (RuleViolation) violations.next();
343 
344             reportedValues.add(violation);
345             assertTrue("Unexpected Violation Returned: " + violation, values.contains(violation));
346         }
347 
348         Iterator expected = values.iterator();
349         while (expected.hasNext()) {
350             RuleViolation violation = (RuleViolation) expected.next();
351             assertTrue("Expected Violation not Returned: " + violation, reportedValues.contains(violation));
352         }
353     }

356     protected List makeCompilationUnits() throws Throwable {
357         List<ASTCompilationUnit> RC = new ArrayList<ASTCompilationUnit>();
358         JavaParser parser = TargetJDKVersion.DEFAULT_JDK_VERSION.createParser(new StringReader(javaCode));
359         RC.add(parser.CompilationUnit());
