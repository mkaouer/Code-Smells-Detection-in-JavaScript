26      private static class OverrideParser {
27          private Element ruleElement;
28  
29          public OverrideParser(Element ruleElement) {
30              this.ruleElement = ruleElement;
31          }
32  
33          public void overrideAsNecessary(Rule rule) {
34              if (ruleElement.hasAttribute("name")) {
35                  rule.setName(ruleElement.getAttribute("name"));
36              }
37              if (ruleElement.hasAttribute("message")) {
38                  rule.setMessage(ruleElement.getAttribute("message"));
39              }
40              if (ruleElement.hasAttribute("externalInfoUrl")) {
41                  rule.setExternalInfoUrl(ruleElement.getAttribute("externalInfoUrl"));
42              }
43              for (int i = 0; i < ruleElement.getChildNodes().getLength(); i++) {
44                  Node node = ruleElement.getChildNodes().item(i);
45                  if (node.getNodeType() == Node.ELEMENT_NODE) {
46                      if (node.getNodeName().equals("description")) {
47                          rule.setDescription(parseTextNode(node));
48                      } else if (node.getNodeName().equals("example")) {
49                          rule.addExample(parseTextNode(node));
50                      } else if (node.getNodeName().equals("priority")) {
51                          rule.setPriority(Integer.parseInt(parseTextNode(node)));
52                      } else if (node.getNodeName().equals("properties")) {
53                          Properties p = new Properties();
54                          parsePropertiesNode(p, node);
55                          rule.addProperties(p);
56                      }
57                  }
58              }
59          }
60      }

87       * Create a RuleSets from a list of names.
88       *
89       * @param ruleSetFileNames comma-separated list of rule set files.
90       * @param classLoader the classloader to load the rulesets
91       * @throws RuleSetNotFoundException
92       */
93      public RuleSets createRuleSets(String ruleSetFileNames, ClassLoader classLoader)
94              throws RuleSetNotFoundException {
95          RuleSets ruleSets = new RuleSets();
96  
97          for (StringTokenizer st = new StringTokenizer(ruleSetFileNames, ","); st
98                  .hasMoreTokens();) {
99              RuleSet ruleSet = createSingleRuleSet(st.nextToken().trim(), classLoader);
100             ruleSets.addRuleSet(ruleSet);
101         }
102 
103         return ruleSets;
104     }

130         RuleSet result = new RuleSet();
131         RuleSet[] allRuleSets = ruleSets.getAllRuleSets();
132         for (RuleSet ruleSet: allRuleSets) {
133             result.addRuleSet(ruleSet);
134         }

175     /**
176      * Create a ruleset from an input stream with a specified class loader
177      *
178      * @param inputStream an input stream that contains a ruleset descripion
179      * @param classLoader a class loader used to load rule classes
180      * @return a new ruleset
181      */
182     public RuleSet createRuleSet(InputStream inputStream, ClassLoader classLoader) {
183         try {
184             DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
185             Document doc = builder.parse(inputStream);
186             Element root = doc.getDocumentElement();
187 
188             RuleSet ruleSet = new RuleSet();
189             ruleSet.setName(root.getAttribute("name"));
190             ruleSet.setLanguage(Language.getByName(root.getAttribute("language")));
191 
192             NodeList nodeList = root.getChildNodes();
193             for (int i = 0; i < nodeList.getLength(); i++) {
194                 Node node = nodeList.item(i);
195                 if (node.getNodeType() == Node.ELEMENT_NODE) {
196                     if (node.getNodeName().equals("description")) {
197                         ruleSet.setDescription(parseTextNode(node));
198                     } else if (node.getNodeName().equals("rule")) {
199                         parseRuleNode(ruleSet, node, classLoader);
200                     }
201                 }
202             }
203 
204             return ruleSet;
205         } catch (ClassNotFoundException cnfe) {
206             cnfe.printStackTrace();
207             throw new RuntimeException("Couldn't find that class " + cnfe.getMessage());
208         } catch (InstantiationException ie) {
209             ie.printStackTrace();
210             throw new RuntimeException("Couldn't find that class " + ie.getMessage());
211         } catch (IllegalAccessException iae) {
212             iae.printStackTrace();
213             throw new RuntimeException("Couldn't find that class " + iae.getMessage());
214         } catch (ParserConfigurationException pce) {
215             pce.printStackTrace();
216             throw new RuntimeException("Couldn't find that class " + pce.getMessage());
217         } catch (RuleSetNotFoundException rsnfe) {
218             rsnfe.printStackTrace();
219             throw new RuntimeException("Couldn't find that class " + rsnfe.getMessage());
220         } catch (IOException ioe) {
221             ioe.printStackTrace();
222             throw new RuntimeException("Couldn't find that class " + ioe.getMessage());
223         } catch (SAXException se) {
224             se.printStackTrace();
225             throw new RuntimeException("Couldn't find that class " + se.getMessage());
226         }
227     }

249     /**
250      * Parse a rule node
251      *
252      * @param ruleSet  the ruleset being constructed
253      * @param ruleNode must be a rule element node
254      */
255     private void parseRuleNode(RuleSet ruleSet, Node ruleNode, ClassLoader classLoader)
256             throws ClassNotFoundException, InstantiationException,
257             IllegalAccessException, RuleSetNotFoundException {
258         Element ruleElement = (Element) ruleNode;
259         String ref = ruleElement.getAttribute("ref");
260         if (ref.trim().length() == 0) {
261             parseInternallyDefinedRuleNode(ruleSet, ruleNode, classLoader);
262         } else {
263             parseExternallyDefinedRuleNode(ruleSet, ruleNode);
264         }
265     }

267     /**
268      * Process a rule definition node
269      *
270      * @param ruleSet  the ruleset being constructed
271      * @param ruleNode must be a rule element node
272      */
273     private void parseInternallyDefinedRuleNode(RuleSet ruleSet, Node ruleNode, ClassLoader classLoader)
274             throws ClassNotFoundException, InstantiationException, IllegalAccessException {
275         Element ruleElement = (Element) ruleNode;
276 
277         String attribute = ruleElement.getAttribute("class");
278         Class c = classLoader.loadClass(attribute);
279         Rule rule = (Rule) c.newInstance();
280 
281         rule.setName(ruleElement.getAttribute("name"));
282         rule.setMessage(ruleElement.getAttribute("message"));
283         rule.setRuleSetName(ruleSet.getName());
284         rule.setExternalInfoUrl(ruleElement.getAttribute("externalInfoUrl"));
285 
286         if (ruleElement.hasAttribute("dfa")
287                 && ruleElement.getAttribute("dfa").equals("true")) {
288             rule.setUsesDFA();
289         }
290 
291         if (ruleElement.hasAttribute("typeResolution")
292                 && ruleElement.getAttribute("typeResolution").equals("true")) {
293             rule.setUsesTypeResolution();
294         }
295 
296         for (int i = 0; i < ruleElement.getChildNodes().getLength(); i++) {
297             Node node = ruleElement.getChildNodes().item(i);
298             if (node.getNodeType() == Node.ELEMENT_NODE) {
299                 if (node.getNodeName().equals("description")) {
300                     rule.setDescription(parseTextNode(node));
301                 } else if (node.getNodeName().equals("example")) {
302                     rule.addExample(parseTextNode(node));
303                 } else if (node.getNodeName().equals("priority")) {
304                     rule.setPriority(Integer.parseInt(parseTextNode(node).trim()));
305                 } else if (node.getNodeName().equals("properties")) {
306                     Properties p = new Properties();
307                     parsePropertiesNode(p, node);
308                     for (Iterator j = p.keySet().iterator(); j.hasNext();) {
309                         String key = (String) j.next();
310                         rule.addProperty(key, p.getProperty(key));
311                     }
312                 }
313             }
314         }
315         if (rule.getPriority() <= minPriority) {
316             ruleSet.addRule(rule);
317         }
318     }

321      * Process a reference to a rule
322      *
323      * @param ruleSet  the ruleset being constructucted
324      * @param ruleNode must be a rule element node
325      */
326     private void parseExternallyDefinedRuleNode(RuleSet ruleSet, Node ruleNode)
327             throws RuleSetNotFoundException {
328         Element ruleElement = (Element) ruleNode;
329         String ref = ruleElement.getAttribute("ref");
330         if (ref.endsWith("xml")) {
331             parseRuleNodeWithExclude(ruleSet, ruleElement, ref);
332         } else {
333             parseRuleNodeWithSimpleReference(ruleSet, ruleNode, ref);
334         }
335     }

337     /**
338      * Parse a rule node with a simple reference
339      *
340      * @param ruleSet the ruleset being constructed
341      * @param ref     a reference to a rule
342      */
343     private void parseRuleNodeWithSimpleReference(RuleSet ruleSet, Node ruleNode,
344                                                   String ref) throws RuleSetNotFoundException {
345         RuleSetFactory rsf = new RuleSetFactory();
346 
347         ExternalRuleID externalRuleID = new ExternalRuleID(ref);
348         RuleSet externalRuleSet = rsf.createRuleSet(ResourceLoader
349                 .loadResourceAsStream(externalRuleID.getFilename()));
350         Rule externalRule = externalRuleSet.getRuleByName(externalRuleID.getRuleName());
351         if (externalRule == null) {
352             throw new IllegalArgumentException("Unable to find rule "
353                     + externalRuleID.getRuleName()
354                     + "; perhaps the rule name is mispelled?");
355         }
356 
357         OverrideParser p = new OverrideParser((Element) ruleNode);
358         p.overrideAsNecessary(externalRule);
359 
360         if (externalRule.getPriority() <= minPriority) {
361             ruleSet.addRule(externalRule);
362         }
363     }

365     /**
366      * Parse a reference rule node with excludes
367      *
368      * @param ruleSet     the ruleset being constructed
369      * @param ruleElement must be a rule element
370      * @param ref         the ruleset reference
371      */
372     private void parseRuleNodeWithExclude(RuleSet ruleSet, Element ruleElement, String ref)
373             throws RuleSetNotFoundException {
374         NodeList excludeNodes = ruleElement.getChildNodes();
375         Set<String> excludes = new HashSet<String>();
376         for (int i = 0; i < excludeNodes.getLength(); i++) {
377             if ((excludeNodes.item(i).getNodeType() == Node.ELEMENT_NODE)
378                     && (excludeNodes.item(i).getNodeName().equals("exclude"))) {
379                 Element excludeElement = (Element) excludeNodes.item(i);
380                 excludes.add(excludeElement.getAttribute("name"));
381             }
382         }
383 
384         RuleSetFactory rsf = new RuleSetFactory();
385         RuleSet externalRuleSet = rsf.createRuleSet(ResourceLoader
386                 .loadResourceAsStream(ref));
387         for (Rule rule: externalRuleSet.getRules()) {
388             if (!excludes.contains(rule.getName()) && rule.getPriority() <= minPriority) {
389                 ruleSet.addRule(rule);
390             }
391         }
392     }

394     private static String parseTextNode(Node node) {
395         StringBuffer buffer = new StringBuffer();
396         for (int i = 0; i < node.getChildNodes().getLength(); i++) {
397             Node childNode = node.getChildNodes().item(i);
398             if (childNode.getNodeType() == Node.CDATA_SECTION_NODE
399                     || childNode.getNodeType() == Node.TEXT_NODE) {
400                 buffer.append(childNode.getNodeValue());
401             }
402         }
403         return buffer.toString();
404     }

408      *
409      * @param propertiesNode must be a properties element node

411     private static void parsePropertiesNode(Properties p, Node propertiesNode) {
412         for (int i = 0; i < propertiesNode.getChildNodes().getLength(); i++) {
413             Node node = propertiesNode.getChildNodes().item(i);
414             if (node.getNodeType() == Node.ELEMENT_NODE
415                     && node.getNodeName().equals("property")) {
416                 parsePropertyNode(p, node);
417             }
418         }

421     /**
422      * Parse a property node
423      *
424      * @param propertyNode must be a property element node
425      */
426     private static void parsePropertyNode(Properties p, Node propertyNode) {
427         Element propertyElement = (Element) propertyNode;
428         String name = propertyElement.getAttribute("name");
429         String value = propertyElement.getAttribute("value");
430         // TODO String desc = propertyElement.getAttribute("description");
431         if (value.trim().length() == 0) {
432             for (int i = 0; i < propertyNode.getChildNodes().getLength(); i++) {
433                 Node node = propertyNode.getChildNodes().item(i);
434                 if ((node.getNodeType() == Node.ELEMENT_NODE)
435                         && node.getNodeName().equals("value")) {
436                     value = parseTextNode(node);
437                 }
438             }
439         }
440         if (propertyElement.hasAttribute("pluginname")) {
441             p.setProperty("pluginname", propertyElement.getAttributeNode("pluginname")
442                     .getNodeValue());
443         }
444         p.setProperty(name, value);
445     }
