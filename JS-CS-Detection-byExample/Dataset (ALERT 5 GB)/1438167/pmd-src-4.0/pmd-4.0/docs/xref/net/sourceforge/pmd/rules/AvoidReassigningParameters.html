14  public class AvoidReassigningParameters extends AbstractRule {
15  
16      public Object visit(ASTMethodDeclarator node, Object data) {
17          Map<VariableNameDeclaration, List<NameOccurrence>> params = node.getScope().getVariableDeclarations();
18          for (Map.Entry<VariableNameDeclaration, List<NameOccurrence>> entry: params.entrySet()) {
19              VariableNameDeclaration decl = entry.getKey();
20              List<NameOccurrence> usages = entry.getValue();
21              for (NameOccurrence occ: usages) {
22                  if ((occ.isOnLeftHandSide() || occ.isSelfAssignment()) && 
23                      occ.getNameForWhichThisIsAQualifier() == null &&
24                      (!decl.isArray() || occ.getLocation().jjtGetParent().jjtGetParent().jjtGetNumChildren() == 1))
25                  {
26                      // not an array or no primary suffix to access the array values
27                      addViolation(data, decl.getNode(), decl.getImage());
28                  }
29              }
30          }
31          return super.visit(node, data);
32      }
