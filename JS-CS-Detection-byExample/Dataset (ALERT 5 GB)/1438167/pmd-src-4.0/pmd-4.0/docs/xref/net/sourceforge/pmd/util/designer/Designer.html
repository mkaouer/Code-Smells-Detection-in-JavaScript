154     private class ExceptionNode implements TreeNode {
155 
156     	private Object 			item;    	
157     	private ExceptionNode[] kids;
158     	
159     	public ExceptionNode(Object theItem) {
160     		item = theItem;
161     		
162     		if (item instanceof ParseException) createKids();
163     	}
164     	
165     	// each line in the error message becomes a separate tree node
166     	private void createKids() {
167     		    		
168     		String message = ((ParseException)item).getMessage();    		
169             String[] lines = StringUtil.substringsOf(message, PMD.EOL);
170 
171 			kids = new ExceptionNode[lines.length];
172 			for (int i=0; i<lines.length; i++) {
173 				kids[i] = new ExceptionNode(lines[i]);
174 			}
175     	}
176     	
177 		public int getChildCount() { return kids == null ? 0 : kids.length; }
178 		public boolean getAllowsChildren() {return false; }
179 		public boolean isLeaf() { return kids == null; }
180 		public TreeNode getParent() { return null; }
181 		public TreeNode getChildAt(int childIndex) { return kids[childIndex]; }
182 		public String label() {	return item.toString();	}
183 		
184 		public Enumeration children() {
185 			Enumeration e = new Enumeration() {
186 				int i = 0;
187 				public boolean hasMoreElements() { 
188 					return kids != null && i < kids.length;
189 				}
190 
191 				public Object nextElement() { return kids[i++]; }
192 				};
193 			return e;
194 		}
195 		
196 		public int getIndex(TreeNode node) {
197 			for (int i=0; i<kids.length; i++) {
198 				if (kids[i] == node) return i;
199 			}
200 			return -1;
201 		}
202     }

204     // Tree node that wraps the AST node for the tree widget and
205     // any possible children they may have.
206     private class ASTTreeNode implements TreeNode {
207 
208     	private Node 			node;
209     	private ASTTreeNode 	parent;
210     	private ASTTreeNode[] 	kids;
211     	
212     	public ASTTreeNode(Node theNode) {
213     		node = theNode;
214     		
215     		Node prnt = node.jjtGetParent();
216     		if (prnt != null) parent = new ASTTreeNode(prnt);    		
217     	}
218     	
219 		public int getChildCount() { return node.jjtGetNumChildren(); }
220 		public boolean getAllowsChildren() { return false;	}
221 		public boolean isLeaf() { return node.jjtGetNumChildren() == 0;	}
222 		public TreeNode getParent() { return parent; }
223 		
224 		public Enumeration children() {
225 			
226 			if (getChildCount() > 0) getChildAt(0);	// force it to build kids
227 			
228 			Enumeration e = new Enumeration() {
229 				int i = 0;
230 				public boolean hasMoreElements() { 
231 					return kids != null && i < kids.length;
232 				}
233 				public Object nextElement() { return kids[i++]; }
234 				};
235 			return e;
236 		}
237 
238 		public TreeNode getChildAt(int childIndex) {
239 			
240 			if (kids == null) {
241 				kids = new ASTTreeNode[node.jjtGetNumChildren()];
242     			for (int i=0; i<kids.length; i++) {
243     				kids[i] = new ASTTreeNode(node.jjtGetChild(i));
244     				}
245 				}			
246 			return kids[childIndex];
247 		}
248 
249 		public int getIndex(TreeNode node) {
250 
251 			for (int i=0; i<kids.length; i++) {
252 				if (kids[i] == node) return i;
253 			}
254 			return -1;
255 		}
256     	
257 		public String label() {
258 			if (node instanceof SimpleNode) {
259 				SimpleNode sn = (SimpleNode)node;
260                 if (sn.getLabel() != null) {
261                     return node.toString() + LABEL_IMAGE_SEPARATOR + sn.getLabel();
262                 }
263 				if (sn.getImage() == null) {
264                     return node.toString();
265                 }
266 				return node.toString() + LABEL_IMAGE_SEPARATOR + sn.getImage();
267 			}
268 			return node.toString();
269 		}
270     }

276     private class ASTCellRenderer extends DefaultTreeCellRenderer {
277     	
278     	private ASTTreeNode node;
279 
280     	public Icon getIcon() { return null; }
281     	
282     	public Component getTreeCellRendererComponent(JTree tree, Object value,	boolean sel,boolean expanded,boolean leaf, int row,  boolean hasFocus) {
283 
284     		if (value instanceof ASTTreeNode) {
285     			node = (ASTTreeNode)value;
286     		}
287     		return super.getTreeCellRendererComponent(tree, value, sel, expanded, leaf, row, hasFocus);
288     	}
289     	
290     	// overwrite the image data (if present) in a different colour
291     	public void paint(Graphics g) {
292     		
293     		super.paint(g);
294     		
295     		if (node == null) return;
296     		
297     		String text = node.label();
298     		int separatorPos = text.indexOf(LABEL_IMAGE_SEPARATOR);
299     		if (separatorPos < 0) return;
300     		    		
301     		String label = text.substring(0, separatorPos+1);
302     		String image = text.substring(separatorPos+1);
303 
304     		FontMetrics fm = g.getFontMetrics();
305     		int width = SwingUtilities.computeStringWidth(fm, label);
306     		
307     		g.setColor(IMAGE_TEXT_COLOR);
308     		g.drawString(image, width, fm.getMaxAscent());
309     	}
310     }

315     private class ASTTreeWidget extends JTree {
316     	
317     	public ASTTreeWidget(Object[] items) {
318     		super(items);
319     	}
320     	
321         public String convertValueToText(Object value, boolean selected, boolean expanded, boolean leaf, int row, boolean hasFocus) {
322         	if (value == null) return "";
323         	if (value instanceof ASTTreeNode) {
324         		return ((ASTTreeNode)value).label();
325         	}        	
326         	if (value instanceof ExceptionNode) {
327         		return ((ExceptionNode)value).label();
328         	}        	
329         	return value.toString();
330     	}
331         
332         public void expandAll(boolean expand) {
333             TreeNode root = (TreeNode)getModel().getRoot();
334             expandAll(new TreePath(root), expand);
335         }
336         
337         private void expandAll(TreePath parent, boolean expand) {
338             // Traverse children
339             TreeNode node = (TreeNode)parent.getLastPathComponent();
340             if (node.getChildCount() >= 0) {
341                 for (Enumeration e=node.children(); e.hasMoreElements(); ) {
342                     TreeNode n = (TreeNode)e.nextElement();
343                     TreePath path = parent.pathByAddingChild(n);
344                     expandAll(path, expand);
345                 }
346             }
347         
348             if (expand) {
349                 expandPath(parent);
350             } else {
351                 collapsePath(parent);
352             }
353         }        
354     }

408     private class XPathListener implements ActionListener {
409         public void actionPerformed(ActionEvent ae) {
410             xpathResults.clear();
411             if (xpathQueryArea.getText().length() == 0) {
412                 xpathResults.addElement("XPath query field is empty");
413                 xpathResultList.repaint();
414                 codeEditorPane.requestFocus();
415                 return;
416             }
417             SimpleNode c = getCompilationUnit();
418             try {
419                 XPath xpath = new BaseXPath(xpathQueryArea.getText(), new DocumentNavigator());
420                 for (Iterator iter = xpath.selectNodes(c).iterator(); iter.hasNext();) {
421                     StringBuffer sb = new StringBuffer();
422                     Object obj = iter.next();
423                     if (obj instanceof String) {
424                         System.out.println("Result was a string: " + ((String) obj));
425                     } else if (!(obj instanceof Boolean)) {
426                         // if it's a Boolean and it's 'false', what does that mean?
427                         SimpleNode node = (SimpleNode) obj;
428                         String name = node.getClass().getName().substring(node.getClass().getName().lastIndexOf('.') + 1);
429                         String line = " at line " + node.getBeginLine();
430                         sb.append(name).append(line).append(PMD.EOL);
431                         xpathResults.addElement(sb.toString().trim());
432                     }
433                 }
434                 if (xpathResults.isEmpty()) {
435                     xpathResults.addElement("No matching nodes " + System.currentTimeMillis());
436                 }
437             } catch (ParseException pe) {
438                 xpathResults.addElement(pe.fillInStackTrace().getMessage());
439             } catch (JaxenException je) {
440                 xpathResults.addElement(je.fillInStackTrace().getMessage());
441             }
442             xpathResultList.repaint();
443             xpathQueryArea.requestFocus();
444         }
445     }

605     private static void makeTextComponentUndoable(JTextComponent textConponent) {
606         final UndoManager undoManager = new UndoManager();

660     /**
661      * Returns an unformatted xml string (without the declaration)
662      * 
663      * @throws TransformerException if the XML cannot be converted to a string
664      */
665     private String getXmlString(SimpleNode node) throws TransformerException {
666         StringWriter writer = new StringWriter();
667         
668         Source source = new DOMSource(node.asXml());
669         Result result = new StreamResult(writer);
670         TransformerFactory transformerFactory = TransformerFactory.newInstance();
671         try {
672             transformerFactory.setAttribute("indent-number", 4);   //For java 5
673         } catch (IllegalArgumentException e) {
674             //Running on Java 1.4 which does not support this attribute
675         }
676         Transformer xformer = transformerFactory.newTransformer();
677         xformer.setOutputProperty(OutputKeys.INDENT, "yes");
678         xformer.setOutputProperty("{http://xml.apache.org/xalan}indent-amount", "4");   //For java 1.4
679         xformer.transform(source, result);
680         
681         return writer.toString();
682     }
