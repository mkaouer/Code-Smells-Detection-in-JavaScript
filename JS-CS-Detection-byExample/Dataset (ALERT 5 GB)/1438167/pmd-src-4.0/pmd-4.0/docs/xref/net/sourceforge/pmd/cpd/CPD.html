72      private void addDirectory(String dir, boolean recurse) throws IOException {
73          if (!(new File(dir)).exists()) {
74              throw new FileNotFoundException("Couldn't find directory " + dir);
75          }
76          FileFinder finder = new FileFinder();
77          // TODO - could use SourceFileSelector here
78          add(finder.findFilesFrom(dir, language.getFileFilter(), recurse));
79      }

81      private Set<String> current = new HashSet<String>();
82  
83      private void add(int fileCount, File file) throws IOException {
84  
85          if (skipDuplicates) {
86              // TODO refactor this thing into a separate class
87              String signature = file.getName() + '_' + file.length();
88              if (current.contains(signature)) {
89                  System.err.println("Skipping " + file.getAbsolutePath() + " since it appears to be a duplicate file and --skip-duplicate-files is set");
90                  return;
91              }
92              current.add(signature);
93          }
94  
95          if (!file.getCanonicalPath().equals(new File(file.getAbsolutePath()).getCanonicalPath())) {
96              System.err.println("Skipping " + file + " since it appears to be a symlink");
97              return;
98          }
99  
100         listener.addedFile(fileCount, file);
101         SourceCode sourceCode = new SourceCode(new SourceCode.FileCodeLoader(file, encoding));
102         language.getTokenizer().tokenize(sourceCode, tokens);
103         source.put(sourceCode.getFileName(), sourceCode);
104     }

106     public static Renderer getRendererFromString(String name, String encoding) {
107         if (name.equalsIgnoreCase("text") || name.equals("")) {
108             return new SimpleRenderer();
109         } else if ("xml".equals(name)) {
110             return new XMLRenderer(encoding);
111         }  else if ("csv".equals(name)) {
112             return new CSVRenderer();
113         }
114         try {
115             return (Renderer) Class.forName(name).newInstance();
116         } catch (Exception e) {
117             System.out.println("Can't find class '" + name + "', defaulting to SimpleRenderer.");
118         }
119         return new SimpleRenderer();
120     }

151     public static void main(String[] args) {
152         if (args.length == 0) {
153             usage();
154         }
155 
156         try {
157             boolean skipDuplicateFiles = findBooleanSwitch(args, "--skip-duplicate-files");
158             String pathToFiles = findRequiredStringValue(args, "--files");
159             String languageString = findOptionalStringValue(args, "--language", "java");
160             String formatString = findOptionalStringValue(args, "--format", "text");
161             String encodingString = findOptionalStringValue(args, "--encoding", System.getProperty("file.encoding"));
162             int minimumTokens = Integer.parseInt(findRequiredStringValue(args, "--minimum-tokens"));
163             LanguageFactory f = new LanguageFactory();
164             Language language = f.createLanguage(languageString);
165             Renderer renderer = CPD.getRendererFromString(formatString, encodingString);
166             CPD cpd = new CPD(minimumTokens, language);
167             cpd.setEncoding(encodingString);
168             if (skipDuplicateFiles) {
169                 cpd.skipDuplicates();
170             }
171             cpd.addRecursively(pathToFiles);
172             cpd.go();
173             System.out.println(renderer.render(cpd.getMatches()));
174         } catch (Exception e) {
175             e.printStackTrace();
176         }
177     }
