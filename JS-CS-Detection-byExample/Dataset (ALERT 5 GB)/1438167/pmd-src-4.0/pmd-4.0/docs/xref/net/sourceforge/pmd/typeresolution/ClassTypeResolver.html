99  	/**
100 	 * If the outer class wasn't found then we'll get in here
101 	 * 
102 	 * @param node
103 	 */
104     private void populateImports(ASTCompilationUnit node){
105 		List<ASTImportDeclaration> theImportDeclarations = node.findChildrenOfType(ASTImportDeclaration.class);
106         importedClasses = new HashMap<String, String>();
107 
108 		// go through the imports
109 		for (ASTImportDeclaration anImportDeclaration : theImportDeclarations) {
110             String strPackage = anImportDeclaration.getPackageName();
111             if (anImportDeclaration.isImportOnDemand()) {
112                 importedOnDemand.add(strPackage);
113             } else if (!anImportDeclaration.isImportOnDemand()) {
114                 String strName = anImportDeclaration.getImportedName();
115                 importedClasses.put(strName, strName);
116                 importedClasses.put(strName.substring(strPackage.length() + 1), strName);
117             }
118         }
119 
120 		importedClasses.putAll(myJavaLang);
121 	}

123 	private void populateClassName(ASTCompilationUnit node) throws ClassNotFoundException {
124 		ASTClassOrInterfaceDeclaration decl = node.getFirstChildOfType(ASTClassOrInterfaceDeclaration.class);
125 		if (decl != null) {
126 			ASTPackageDeclaration pkgDecl = node.getFirstChildOfType(ASTPackageDeclaration.class);
127             if (pkgDecl == null) {
128                 className = decl.getImage();
129             } else {
130                 importedOnDemand.add(((ASTName) pkgDecl.jjtGetChild(0)).getImage());
131                 className = ((ASTName) pkgDecl.jjtGetChild(0)).getImage() + "." + decl.getImage();
132             }
133 			pmdClassLoader.loadClass(className);
134 			importedClasses = pmdClassLoader.getImportedClasses(className);
135 		}
136 	}

145     public Object visit(ASTName node, Object data) {
146         /*
147          * Only doing this for nodes where getNameDeclaration is null this
148          * means it's not a named node, i.e. Static reference or Annotation
149          * Doing this for memory - TODO: Investigate if there is a valid memory
150          * concern or not
151          */
152         if (node.getNameDeclaration() == null) {
153             String name = node.getImage();
154             if (name.indexOf('.') != -1) {
155                 name = name.substring(0, name.indexOf('.'));
156             }
157             populateType(node, name);
158         }
159         return super.visit(node, data);
160     }

162     private void populateType(TypeNode node, String className) {
163 
164         String qualifiedName = className;
165         Class myType = myPrimitiveTypes.get(className);
166         if (myType == null && importedClasses != null) {
167             if (importedClasses.containsKey(className)) {
168                 qualifiedName = importedClasses.get(className);
169             } else if (importedClasses.containsValue(className)) {
170                 qualifiedName = className;
171             }
172             if (qualifiedName != null) {
173                 try {
174                     /*
175                      * TODO - the map right now contains just class names. if we
176                      * use a map of classname/class then we don't have to hit
177                      * the class loader for every type - much faster
178                      */
179                     myType = pmdClassLoader.loadClass(qualifiedName);
180                 } catch (ClassNotFoundException e) {
181                     myType = processOnDemand(qualifiedName);
182                 }
183             }
184         }
185         if (myType != null) {
186             node.setType(myType);
187         }
188     }

190     private Class processOnDemand(String qualifiedName) {
191         for(String entry : importedOnDemand){
192             try {
193                 return pmdClassLoader.loadClass(entry + "." + qualifiedName);
194             } catch (Throwable e) {
195             }
196         }
197         return null;
198     }
