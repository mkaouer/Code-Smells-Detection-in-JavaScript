48  /**
49   * This class tests the Statistical Rules in PMD.
50   * <p/>
51   * The idea is, that we fill up 999 datapoints into
52   * the Stat Rule, and then throw random parameters
53   * at it.
54   * <p/>
55   * The three parameters which are checked are:
56   * sigma - # Sigmas over the mean.
57   * topscore - Only the top 5 or so items.
58   * minimum - Only things of score 10 or better
59   * <p/>
60   * When more than one parameter is lumped together, then
61   * we expect the one which would return the fewest to
62   * determine what gets sent back.
63   * <p/>
64   * So, we throw each collection of parameters, where each
65   * one is a different order into the system.  We check the
66   * results off of what the smallest value should be.
67   * <p/>
68   * If you are going to work with StatisticalRule any, please
69   * bump the "NUM_TESTS" number up to something like 128.  That
70   * way you are more likely to identify problems.  It is set low
71   * now to make building and running tests easier (when we aren't
72   * touching the file.)
73   * <p/>
74   * Note also, that when verifying the Sigma, I wasn't quite able
75   * to determine how many results it would return (it would vary
76   * from -2 to 2 of what I expected.)  That is what the delta
77   * parameter on the verify method takes.  If you can figure it
78   * out exactly, (without stealing code from the StatRule) then
79   * feel free to change it and tighten the deltas.
80   */

181     /**
182      * This returns a Random value for Sigma which will
183      * return some values.
184      */
185     public double randomSigma() {
186         return random.nextDouble() * 1.0;
187     }
188 
189     /**
190      * This returns a Random value for Sigma which value
191      * is greater than the parameter.
192      */
193     public double randomSigma(int minimum) {
194         double minSigma = ((POINTS - 1 - minimum) - MEAN) / SIGMA;
195 
196         if ((minSigma <= 0) || (minSigma > 2))
197             return randomSigma();
198 
199         return minSigma + (random.nextDouble() * (2 - minSigma));
200     }
201 
202     /**
203      * This returns the expected number of results when
204      * the Sigma rating is the smallest.
205      */
206     public int expectedSigma(double sigma) {
207         long expectedMin = Math.round(MEAN + (sigma * SIGMA));
208 
209         if (((POINTS - 1) - expectedMin) < 0)
210             return 0;
211         return (POINTS - 1) - (int) expectedMin;
212     }
213 
214     /**
215      * This generates a random minimum value for testing.
216      */
217     public double randomMinimum() {
218         return random.nextDouble() * (POINTS - 1);
219     }
220 
221     /**
222      * This generates a random minimum value for which fewer
223      * results would be returned.
224      */
225     public double randomMinimum(int minimum) {
226         double diffTarget = 1.0 * (POINTS - 1 - minimum);
227         return (random.nextDouble() * minimum) + diffTarget;
228     }

230     /**
231      * This returns the expected number of reports.
232      * <p/>
233      * If the Minimum comes in at 521.569 then we expect
234      * 522, 523, ... 999 will pass.
235      */
236     public int expectedMinimum(double minimum) {
237         Double d = new Double(minimum);
238         return POINTS - 1 - d.intValue();
239     }

796     /**
797      * Verifies what happens when you pass these parameters
798      * into the thing.  DELTA is the amount of error allowed.
799      * Usually DELTA is only used for Sigma, as we really can't
800      * calculate it exactly.
801      */
802 
803     public void verifyResults(double sigma, double minimum, int topScore, int expected, int delta) {
804         try {
805             setUp();
806             if (sigma >= 0) {
807                 IUT.addProperty("sigma", Double.toString(sigma));
808             }
809 
810             if (minimum >= 0) {
811                 IUT.addProperty("minimum", Double.toString(minimum));
812             }
813 
814             if (topScore >= 0) {
815                 IUT.addProperty("topscore", Integer.toString(topScore));
816             }
817 
818             Report report = makeReport(IUT);
819             if (delta == 0) {
