189     protected void verifyRuleSet(RuleSet IUT, int size, Set values) throws Throwable {
190 
191         RuleContext context = new RuleContext();
192         Set<RuleViolation> reportedValues = new HashSet<RuleViolation>();
193         context.setReport(new Report());
194         IUT.apply(makeCompilationUnits(), context);
195 
196         assertEquals("Invalid number of Violations Reported", size, context.getReport().size());
197 
198         Iterator violations = context.getReport().iterator();
199         while (violations.hasNext()) {
200             RuleViolation violation = (RuleViolation) violations.next();
201 
202             reportedValues.add(violation);
203             assertTrue("Unexpected Violation Returned: " + violation, values.contains(violation));
204         }
205 
206         Iterator expected = values.iterator();
207         while (expected.hasNext()) {
208             RuleViolation violation = (RuleViolation) expected.next();
209             assertTrue("Expected Violation not Returned: " + violation, reportedValues.contains(violation));
210         }
211     }
