26      public String getLine(int number) {
27          int count = 1;
28          for (StringTokenizer st = new StringTokenizer(getText(), "\n"); st.hasMoreTokens();) {
29              String tok = st.nextToken();
30              if (count == number) {
31                  return tok;
32              }
33              count++;
34          }
35          throw new RuntimeException("Line number " + number + " not found");
36      }
