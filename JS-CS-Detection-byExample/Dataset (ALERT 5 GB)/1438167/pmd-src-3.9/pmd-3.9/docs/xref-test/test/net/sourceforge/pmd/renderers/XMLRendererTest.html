90      public void testTwoFiles() throws Throwable {
91          Report report = new Report();
92          FooRule rule = new FooRule();
93          runTestFromString(TEST2, rule, report);
94          PMD p = new PMD();
95          p.setJavaVersion(SourceType.JAVA_14);
96          RuleContext ctx = new RuleContext();
97          ctx.setReport(report);
98          ctx.setSourceCodeFilename("bar");
99          RuleSet rules = new RuleSet();
100         rules.addRule(rule);
101         p.processFile(new StringReader(TEST2), rules, ctx);
102         Element root = parseRootElement(report);
103         assertEquals("bar", root.getFirstChild().getNextSibling().getAttributes().getNamedItem("name").getNodeValue());
