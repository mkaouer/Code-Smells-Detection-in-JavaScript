316     private Rule loadFirstRule(String ruleSetName) {
317         RuleSet rs = loadRuleSet(ruleSetName);
318         return ((Rule) (rs.getRules().iterator().next()));
319     }

321     private RuleSet loadRuleSet(String ruleSetName) {
