00216   bool FileStore::get( int num, Message& msg ) const
00217   {
00218     NumToOffset::const_iterator find = m_offsets.find(num);
00219     if(find == m_offsets.end()) return false;
00220     const OffsetSize& offset = find->second;
00221     fseek(m_msgFile, offset.first, SEEK_SET);
00222     char* buffer = new char[offset.second+1];
00223     fread(buffer, sizeof(char), offset.second, m_msgFile);
00224     buffer[offset.second] = 0;
00225     msg.setString(buffer);
