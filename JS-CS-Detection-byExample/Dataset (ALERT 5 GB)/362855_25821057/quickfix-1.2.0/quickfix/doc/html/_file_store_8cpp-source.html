00215   bool FileStore::get( int num, Message& msg ) const
00216   {
00217     NumToOffset::const_iterator find = m_offsets.find(num);
00218     if(find == m_offsets.end()) return false;
00219     const OffsetSize& offset = find->second;
00220     fseek(m_msgFile, offset.first, SEEK_SET);
00221     char* buffer = new char[offset.second+1];
00222     fread(buffer, sizeof(char), offset.second, m_msgFile);
00223     buffer[offset.second] = 0;
00224     msg.setString(buffer);
