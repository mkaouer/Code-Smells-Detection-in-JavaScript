00077 bool MemoryStore::get( int msgSeqNum, std::string& msg ) const
00078 throw( IOException& )
00079 {
00080   Messages::const_iterator find = m_messages.find( msgSeqNum );
00081   if ( find == m_messages.end() ) return false;

00086 void MemoryStore::get( int begin, int end,
00087                        std::vector < std::string > & messages ) const
00088 throw( IOException& )
00089 {
00090   messages.clear();
00091   Messages::const_iterator find = m_messages.find( begin );
00092   for ( ; find != m_messages.end() && find->first <= end; ++find )
00093     messages.push_back( find->second );
