00329   std::string::size_type pos = message.find( "\00135=" );

00331 
00332   std::string::size_type startValue = pos + 4;
00333   std::string::size_type soh = message.find_first_of( '\001', startValue );
00334   if ( soh == std::string::npos ) throw MessageParseError();
