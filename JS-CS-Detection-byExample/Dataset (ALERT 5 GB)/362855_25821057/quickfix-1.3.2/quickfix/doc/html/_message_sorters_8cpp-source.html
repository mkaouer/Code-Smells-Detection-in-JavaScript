00099 {
00100   if(size < 1) return;

00102   
00103   int* fields = new int[ size + 1 ];
00104   fields[ 1 ] = m_delim;
00105   // collect all fields and find the largest field number
00106   int i;
00107   for ( i = 2; i <= size; ++i )
