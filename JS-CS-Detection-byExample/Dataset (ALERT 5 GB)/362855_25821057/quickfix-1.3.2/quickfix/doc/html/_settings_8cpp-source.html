00061 std::istream& operator>>( std::istream& stream, Settings& s )
00062 {
00063   ConfigLexer lexer( &stream );
00064   int lcState = 0;
00065   Settings::Sections::iterator currentSection = s.m_sections.end();
00066   std::string currentName;
00067   while ( 0 != ( lcState = lexer.yylex() ) )
00068   {
00069     switch ( lcState )
00070     {
00071       case ConfigLexer::LC_STATE_SECTION:
00072       currentSection = s.m_sections.insert
00073                        ( s.m_sections.end(),
00074                          Dictionary( lexer.YYText() ) );
00075       break;
00076       case ConfigLexer::LC_STATE_NAME:
00077       currentName = lexer.YYText();
00078       break;
00079       case ConfigLexer::LC_STATE_VALUE:
00080       std::string value = lexer.YYText();
00081       std::string::size_type pos = value.find_last_not_of( ' ' );
00082       if ( pos == std::string::npos ) continue;
00083       value.resize( pos + 1 );
00084       ( *currentSection ).setString( currentName, value );
00085       break;
00086     }
00087   }
00088   return stream;
00089 }
