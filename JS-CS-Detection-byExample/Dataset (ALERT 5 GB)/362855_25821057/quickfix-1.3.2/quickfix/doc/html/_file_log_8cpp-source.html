00075 FileLog::FileLog( std::string path, const SessionID& s )
00076     : m_sessionID( s )
00077 {
00078   file_mkdir( path.c_str(), 0700 );
00079 
00080   if ( path.empty() ) path = ".";
00081   const std::string& begin =
00082     s.getBeginString().getString();
00083   const std::string& sender =
00084     s.getSenderCompID().getString();
00085   const std::string& target =
00086     s.getTargetCompID().getString();
00087 
00088   std::string prefix = path + "/" + begin + "-" + sender + "-" + target + ".";
00089   std::string incomingFileName = prefix + "incoming";
00090   std::string outgoingFileName = prefix + "outgoing";
00091   std::string eventFileName = prefix + "event";
00092 
00093   m_incoming.open( incomingFileName.c_str(), std::ios::out | std::ios::app );
00094   if ( !m_incoming.is_open() ) throw ConfigError( "Could not open incoming file" );
00095   m_outgoing.open( outgoingFileName.c_str(), std::ios::out | std::ios::app );
00096   if ( !m_incoming.is_open() ) throw ConfigError( "Could not open outgoing file" );
00097   m_event.open( eventFileName.c_str(), std::ios::out | std::ios::app );
00098   if ( !m_incoming.is_open() ) throw ConfigError( "Could not open event file" );
00099 }
