00123 void Initiator::initialize() throw ( ConfigError& )
00124 {
00125   std::set < SessionID > sessions = m_settings.getSessions();
00126   std::set < SessionID > ::iterator i;
00127 
00128   if ( !sessions.size() )
00129     throw ConfigError( "No sessions defined" );
00130 
00131   SessionFactory factory( m_application, m_messageStoreFactory,
00132                           m_pLogFactory );
00133 
00134   for ( i = sessions.begin(); i != sessions.end(); ++i )
00135   {
00136     if ( m_settings.get( *i ).getString( "ConnectionType" ) == "initiator" )
00137     {
00138       m_sessions[ *i ] = factory.create( *i, m_settings.get( *i ) );
00139       setConnected( *i, false );
00140     }
00141     if ( !m_sessions.size() )
00142       throw ConfigError( "No sessions defined for initiator" );
00143   }
00144 }
