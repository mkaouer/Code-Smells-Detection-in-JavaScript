00073     {
00074       BeginString beginString;
00075       MsgType msgType;
00076       message.getHeader().getField(beginString);
00077       message.getHeader().getField(msgType);
00078 
00079       if(m_beginString.length() && m_beginString != beginString.getValue())
00080         throw UnsupportedVersion();

00082         throw FieldsOutOfOrder();
