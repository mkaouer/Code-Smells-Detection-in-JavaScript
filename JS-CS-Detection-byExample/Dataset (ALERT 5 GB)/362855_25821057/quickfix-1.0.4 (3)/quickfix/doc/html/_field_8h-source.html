00075   class FieldBase
00076   {
00077     friend class Message;
00078   public:
00079     FieldBase( int field, const std::string& string )
00080       : m_field(field), m_length(0), m_total(0)
00081     {
00082       std::stringstream stream;
00083       stream << field << "=" << string << '\001';
00084 
00085       m_string = string;
00086       m_data = stream.str();
00087       m_length = m_data.length();
00088       std::string::const_iterator iter = m_data.begin();
00089       std::string::const_iterator end = m_data.end();
00090       while( iter != end )
00091         {
00092           m_total += *iter;
00093           ++iter;
00094         }
00095     }
