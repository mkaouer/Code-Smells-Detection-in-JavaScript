22         destroyed
23       };
24     
25       static State stateFlag( State newState = doNotChange )
26       {
27         static State state = notCreated;
28         if ( newState != doNotChange )
29           state = newState;

31       }

40       {
41         Registries::const_iterator foundIt = m_registries.find( name );
42         if ( foundIt == m_registries.end() )
43         {
44           TestFactoryRegistry *factory = new TestFactoryRegistry( name );
45           m_registries.insert( std::pair<const std::string, TestFactoryRegistry*>( name, factory ) );
46           return factory;
47         }
48         return (*foundIt).second;
49       }

51     public:
52       TestFactoryRegistryList()
53       {
54         stateFlag( exist );
55       }
56     
57       ~TestFactoryRegistryList()

59         for ( Registries::iterator it = m_registries.begin(); it != m_registries.end(); ++it )
60           delete (*it).second;

63       }

66       {
