118      * =====================
119      * Layout information class.
120      * This class is derived from CObject and made dynamic using
121      * DECLARE_DYNAMIC.
122      * You can derive your own class from it to provide more information
123      * to your own DoMoveCtrl() function (if you have one).

127     {

130     public:
131     	CSize		m_szCurrent,			// current client size
132     				m_szInitial,			// initial client size
133     				m_szDelta;				// current - initial
134     	UINT		m_nCtrlCnt;				// number of controls (>=0)
135     	CPoint	m_pntScrollPos;		// current scrolling position
136     	bool		m_bUseScrollPos;		// use scroll pos if m_szDelta < 0
137     
138     public:
139     	cdxCDynamicLayoutInfo() : m_bUseScrollPos(false) 
140             {
141             }
142     
143     	cdxCDynamicLayoutInfo(cdxCDynamicWnd *pWnd) : m_bUseScrollPos(false) 
144             {

146             }

149             {
150             }
151

153

491     /*
492      * auto-fill in struct
493      */
494     
495     inline bool cdxCDynamicLayoutInfo::operator=(cdxCDynamicWnd *pWnd)
496     {
497     	if(!pWnd || !pWnd->IsUp())
498     		return false;
499     
500     	m_szCurrent			=	pWnd->GetCurrentClientSize();
501     	m_szInitial			=	pWnd->m_szInitial;
502     	m_szDelta			=	m_szCurrent - m_szInitial;
503     	m_nCtrlCnt			=	pWnd->GetCtrlCount();
504     
505     	if(m_bUseScrollPos == pWnd->m_bUseScrollPos)
506     	{
507     		m_pntScrollPos.x	=	pWnd->Window()->GetScrollPos(SB_HORZ);
508     		m_pntScrollPos.y	=	pWnd->Window()->GetScrollPos(SB_VERT);
509     	}
510     
511     	return true;
512     }

544     	if(!::IsWindow(hwnd))
545     	{
546     		TRACE(_T("*** NOTE[cdxCDynamicWnd::AddSzControl(HWND,HWND,bool)]: Handle 0x%lx is not a valid window.\n"),(DWORD)hwnd);
547     		return false;
548     	}
549     
550     	Position	pos;
551     
552     	if(!m_Map.Lookup(hLikeThis,pos))
553     	{
554     		TRACE(_T("*** NOTE[cdxCDynamicWnd::AddSzControl(HWND,HWND,bool)]: For the 'hLikeThis' handle 0x%lx there hasn't been made an entry for yet.\n"),(DWORD)hLikeThis);
555     		return false;
556     	}
557     
558     	return AddSzControl(hwnd,pos);
559     }

658     /*
659      * translates a "mode" into percentage
660      */
661     
662     inline void cdxCDynamicWnd::_translate(Mode md, SBYTE & b1, SBYTE & b2)
663     {
664     	switch(md)
665     	{
666     		default				:	ASSERT(false);
667     		case	mdNone		:	b1	=	0;		b2	=	0;		break;
668     		case	mdResize		:	b1	=	0;		b2	=	100;	break;
669     		case	mdRepos		:	b1	=	100;	b2	=	100;	break;
670     		case	mdRelative	:	b1	=	50;	b2	=	50;	break;
671     	}
672     }
