198     void 
199     TestAssertTest::testAssertDoubleNonFinite()
200     {
201       double inf = std::numeric_limits<double>::infinity();
202       double nan = std::numeric_limits<double>::quiet_NaN();
203       // test our portable floating-point primitives that detect NaN values
204       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsUnordered( nan ) );
205       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( inf ) );
206       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( -inf ) );
207       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( 1.0 ) );
208       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( 1.5 ) );
209       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( 2.0 ) );
210       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( 2.5 ) );
211       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( 0.0 ) );
212       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( -1.0 ) );
213       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsUnordered( -2.0 ) );
214       // test our portable floating-point primitives that detect finite values
215       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsFinite( 0.0 ) );
216       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsFinite( 0.5 ) );
217       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsFinite( 1.0 ) );
218       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsFinite( 1.5 ) );
219       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsFinite( 2.0 ) );
220       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsFinite( 2.5 ) );
221       CPPUNIT_ASSERT( CPPUNIT_NS::floatingPointIsFinite( -1.5 ) );
222       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsFinite( nan ) );
223       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsFinite( inf ) );
224       CPPUNIT_ASSERT( !CPPUNIT_NS::floatingPointIsFinite( -inf ) );
225       // Infinity tests
226       CPPUNIT_ASSERT( inf == inf );
227       CPPUNIT_ASSERT( -inf == -inf );
228       CPPUNIT_ASSERT( -inf != inf );
229       CPPUNIT_ASSERT( -inf < inf );
230       CPPUNIT_ASSERT( inf > -inf );
231       CPPUNIT_ASSERT_ASSERTION_FAIL( CPPUNIT_ASSERT_DOUBLES_EQUAL( inf, 0.0, 1.0 ) );
232       CPPUNIT_ASSERT_ASSERTION_FAIL( CPPUNIT_ASSERT_DOUBLES_EQUAL( 0.0, inf, 1.0 ) );
233       CPPUNIT_ASSERT_ASSERTION_PASS( CPPUNIT_ASSERT_DOUBLES_EQUAL( inf, inf, 1.0 ) );
234       // NaN tests 
235       CPPUNIT_ASSERT_ASSERTION_FAIL( CPPUNIT_ASSERT_DOUBLES_EQUAL( nan, 0.0, 1.0 ) );
236       CPPUNIT_ASSERT_ASSERTION_FAIL( CPPUNIT_ASSERT_DOUBLES_EQUAL( nan, nan, 1.0 ) );
237       CPPUNIT_ASSERT_ASSERTION_FAIL( CPPUNIT_ASSERT_DOUBLES_EQUAL( nan, inf, 1.0 ) );
238       CPPUNIT_ASSERT_ASSERTION_FAIL( CPPUNIT_ASSERT_DOUBLES_EQUAL( inf, nan, 1.0 ) );
239     }

260     void 
261     TestAssertTest::checkMessageContains( CPPUNIT_NS::Exception *e,
262                                           std::string expected )
263     {
264       std::string actual = e->what();
265       CPPUNIT_ASSERT_MESSAGE( "Expected message not found: " + expected +
266                               ", was: " + actual,
267           std::search( actual.begin(), actual.end(), 
268                        expected.begin(), expected.end() ) != actual.end() );
269     }
