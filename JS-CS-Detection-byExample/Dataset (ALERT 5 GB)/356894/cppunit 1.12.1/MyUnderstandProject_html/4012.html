41     void 
42     TestRunnerDlg::setModel( TestRunnerModel *model,
43                              bool autorunTest )
44     {
45       delete _model;
46     
47       _model = model;
48     
49       // update combo when recent list change
50       connect( _recentTests, SIGNAL( listChanged() ),
51                SLOT( refreshRecentTests() ) );
52     
53       // make selected test in combo the "most recent"
54       connect( _comboTest, SIGNAL( activated(int) ),
55                _recentTests, SLOT( selectTestToRun(int) ) );
56     
57       // refresh the test report counters when a test is selected
58       connect( _recentTests, SIGNAL( testToRunChanged(CPPUNIT_NS::Test *) ),
59                _model, SLOT( resetTestReportCounterFor(CPPUNIT_NS::Test *) ) );
60     
61       // refresh progress bar
62       connect( _model, SIGNAL( numberOfTestCaseChanged(int) ),
63                _progressRun, SLOT( setTotalSteps(int) ) );
64       connect( _model, SIGNAL( numberOfTestCaseRunChanged(int) ),
65                _progressRun, SLOT( setProgress(int) ) );
66       
67       // refresh test report counters
68       connect( _model, SIGNAL( numberOfTestCaseChanged( int ) ),
69                SLOT( setNumberOfTestCase( int ) ) );
70       connect( _model, SIGNAL( numberOfTestCaseRunChanged( int ) ),
71                SLOT( setNumberOfTestCaseRun( int ) ) );
72       connect( _model, SIGNAL( numberOfTestCaseFailureChanged( int ) ),
73                SLOT( setNumberOfTestCaseFailure( int ) ) );
74     
75       // clear failure list
76       connect( _model, SIGNAL( failuresCleared() ),
77                SLOT( clearTestFailureList() ) );
78     
79       // clear failure detail list
80       connect( _model, SIGNAL( failuresCleared() ),
81                SLOT( clearFailureDetail() ) );
82     
83       // add failure to failure list
84       connect( _model, SIGNAL( failureAdded(TestFailureInfo *) ),
85                SLOT( reportFailure(TestFailureInfo*) ) );
86     
87       // show detail on failure selection
88       connect( _listFailures, SIGNAL( selectionChanged(QListViewItem*) ),
89                SLOT( showFailureDetailAt(QListViewItem*) ) );
90     
91       // disable button when running test
92       connect( _model, SIGNAL( testRunStarted( CPPUNIT_NS::Test *, CPPUNIT_NS::TestResult *) ),
93                SLOT( beRunningTest() ) );
94     
95       // enable button when finished running test
96       connect( _model, SIGNAL( testRunFinished() ),
97                SLOT( beCanRunTest() ) );
98     
99       _recentTests->setTestToRun( model->rootTest() );
100       beCanRunTest();
101     
102       if ( autorunTest )
103         runTest();
104     }

178     void 
179     TestRunnerDlg::reportFailure( TestFailureInfo *failure )
180     {
181       QListViewItem *item = new TestFailureListViewItem( failure, 
182                                                          _listFailures );
183       item->setText( indexType, 
184                      failure->isError() ? tr("Error") : tr("Failure") );
185       std::string failedtestName = failure->failedTestName().c_str();
186       item->setText( indexTestName, QString::fromLatin1( failedtestName.c_str() ) );
187     
188       CPPUNIT_NS::Exception *thrownException = failure->thrownException();
189     //2.0  item->setText( indexMessage, thrownException->what() );
190       item->setText( indexMessage, QString(thrownException->what()).stripWhiteSpace() );
191       item->setText( indexFilename, failure->sourceLine().fileName().c_str() );
192       item->setText( indexLineNumber,
193                      QString::number( failure->sourceLine().lineNumber() ) );
194     
195       _listFailures->insertItem( item );
196       _listFailures->triggerUpdate();
197     
198       if ( _listFailures->childCount() == 1 )
199         _listFailures->setSelected( item, TRUE );
200     }

203     void 
204     TestRunnerDlg::showFailureDetailAt( QListViewItem *selection )
205     {
206       TestFailureInfo *failure = ((TestFailureListViewItem*)selection)->failure();
207       
208       QString title = tr("Failure detail for: ");
209       title += QString::fromLatin1( failure->failedTestName().c_str() );
210       _groupFailureDetail->setTitle( title );
211     
212       QString location( failure->sourceLine().fileName().c_str() );
213       location += " (" + 
214                   QString::number( failure->sourceLine().lineNumber() ) +
215                   ")";
216       _labelFailureLocation->setText( location );
217     
218       _editFailureMessage->setText( failure->thrownException()->what() );
219     }
