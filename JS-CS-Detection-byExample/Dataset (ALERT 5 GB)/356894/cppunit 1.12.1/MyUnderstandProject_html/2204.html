63     void 
64     TestRunnerModel::loadSettings(Settings & s)
65     {
66       CWinApp *app = AfxGetApp();
67       ASSERT( app != NULL );
68     
69       int autorun = app->GetProfileInt( _T("CppUnit"),
70                                         _T("AutorunAtStartup"),
71                                         1 );
72       s.autorunOnLaunch = (autorun == 1);
73     
74       s.col_1 = app->GetProfileInt( _T("CppUnit"), _T("Col_1"), 40 );
75       s.col_2 = app->GetProfileInt( _T("CppUnit"), _T("Col_2"), 40 );
76       s.col_3 = app->GetProfileInt( _T("CppUnit"), _T("Col_3"), 40 );
77       s.col_4 = app->GetProfileInt( _T("CppUnit"), _T("Col_4"), 40 );
78     
79       loadHistory();
80     }

83     void 
84     TestRunnerModel::loadHistory()
85     {
86       m_history.clear();
87       int idx = 1;
88       do
89       {
90         CString testName = loadHistoryEntry( idx++ );
91         if ( testName.IsEmpty() )
92           break;
93     
94         try
95         {
96           m_history.push_back( m_rootTest->findTest( toAnsiString(testName ) ) );
97         }
98         catch ( std::invalid_argument &)
99         {
100         }
101       }
102       while ( true );
103     }

116     void 
117     TestRunnerModel::saveSettings( const Settings & s )
118     {
119       CWinApp *app = AfxGetApp();
120       ASSERT( app != NULL );
121     
122       int autorun = s.autorunOnLaunch ? 1 : 0;
123       app->WriteProfileInt( _T("CppUnit"), _T("AutorunAtStartup"), autorun );
124     
125       app->WriteProfileInt( _T("CppUnit"), _T("Col_1"),	 s.col_1 );
126       app->WriteProfileInt( _T("CppUnit"), _T("Col_2"),	 s.col_2 );
127       app->WriteProfileInt( _T("CppUnit"), _T("Col_3"),	 s.col_3 );
128       app->WriteProfileInt( _T("CppUnit"), _T("Col_4"),	 s.col_4 );
129     
130       int idx = 1;
131       for ( History::const_iterator it = m_history.begin(); 
132             it != m_history.end(); 
133             ++it , ++idx )
134       {
135         CPPUNIT_NS::Test *test = *it;
136         saveHistoryEntry( idx, test->getName().c_str() );
137       }
138     }

184     CPPUNIT_NS::Test * 
185     TestRunnerModel::findTestByNameFor( const CString &name, 
186                                         CPPUNIT_NS::Test *test ) const
187     {
188       if ( name == test->getName().c_str() )
189         return test;
190     
191       CPPUNIT_NS::TestSuite *suite = dynamic_cast<CPPUNIT_NS::TestSuite *>(test);
192       if ( suite == NULL )
193         return NULL;
194     
195       const std::vector<CPPUNIT_NS::Test *> &tests = suite->getTests();
196       for ( std::vector<CPPUNIT_NS::Test *>::const_iterator it = tests.begin(); 
197             it != tests.end(); 
198             ++it )
199       {
200         CPPUNIT_NS::Test *testFound = findTestByNameFor( name, *it );
201         if ( testFound != NULL )
202           return testFound;
203       }
204       return NULL;
205     }

209     std::string 
210     TestRunnerModel::toAnsiString( const CString &text )
211     {
212     #ifdef _UNICODE
213       int bufferLength = ::WideCharToMultiByte( CP_THREAD_ACP, 0, 
214                                                 text, text.GetLength(),
215                                                 NULL, 0, NULL, NULL ) +1;
216       char *ansiString = new char[bufferLength];
217       ::WideCharToMultiByte( CP_THREAD_ACP, 0, 
218                              text, text.GetLength(),
219                              ansiString, bufferLength, 
220                                                 NULL,
221                                                 NULL );
222     
223       std::string str( ansiString, bufferLength-1 );
224       delete[] ansiString;
225     
226       return str;
227     #else
228       return std::string( (LPCTSTR)text );
229     #endif
230     }
