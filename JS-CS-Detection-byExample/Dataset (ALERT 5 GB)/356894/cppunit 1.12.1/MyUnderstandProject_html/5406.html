50     Message &
51     Message::operator =( const Message &other )
52     {
53        if ( this != &other )
54        {
55           m_shortDescription = other.m_shortDescription.c_str();
56           m_details.clear();
57           Details::const_iterator it = other.m_details.begin();
58           Details::const_iterator itEnd = other.m_details.end();
59           while ( it != itEnd )
60              m_details.push_back( (*it++).c_str() );
61        }
62     
63        return *this;
64     }
