13      * cdxCDynamicDialog
14      * =================
15      * A new resizable dialog.
16      * This should be the base-class for your normal dialogs.
17      * This class supports:
18      * - A sizing icon
19      * - AutoPositioning (stores last position automatically and stuff)
20      * - Anti-Flickering system.
21      * - And of course, it provides
22      *   the Dynamic child control system DcCS by codex design
23      */
24

26     {
27     	DECLARE_DYNAMIC(cdxCDynamicDialog);
28     
29     public:
30     	enum { flDefault = flAntiFlicker|flSizeIcon };
31     
32     public:
33     	cdxCDynamicDialog(UINT idd = 0, CWnd* pParent = NULL, Freedom fd = fdAll, UINT nFlags = flDefault);
34     	cdxCDynamicDialog(LPCTSTR lpszTemplateName, CWnd* pParent = NULL, Freedom fd = fdAll, UINT nFlags = flDefault);
