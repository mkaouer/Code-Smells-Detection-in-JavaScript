113     TestPlugInRunnerDlg::OnChooseDll() 
114     {
115       CFileDialog dlg( TRUE, "*.dll", "", 0, 
116                        "Test Plug-in (*.dll)|*.dll|All Files (*.*)|*.*||",
117                        this );
118       if ( dlg.DoModal() != IDOK )
119         return;
120     
121       try
122       {
123         loadDll(std::string(dlg.GetPathName()));
124       }

140     std::list<std::string> 
141     TestPlugInRunnerDlg::getCommandLineArguments()
142     {
143       int argc;
144       LPWSTR *argv = ::CommandLineToArgvW( ::GetCommandLineW(), &argc );
145     
146       std::list<std::string> arguments;
147       for( int index = 0; index < argc; index++ ) 
148         arguments.push_back( std::string( CString( argv[index] ) ) );
149     
150       ::GlobalFree( argv );
151     
152       return arguments;
153     }

156     void 
157     TestPlugInRunnerDlg::loadPluginIfNesseccary()
158     {
159       std::list<std::string> argv = getCommandLineArguments();
160       std::list<std::string>::iterator iter = std::find( argv.begin(), 
161                                                          argv.end(), 
162                                                          std::string( "-testsuite" ) );
163     
164       if ( iter == argv.end() ) 
165         return;
166     
167       try 
168       {
169         loadDll( *++iter );
170       } 
171       catch( std::exception &e ) 
172       {
173         AfxMessageBox( e.what() );
174       }
175     }
