11       /*! \brief Traits used by CPPUNIT_ASSERT_EQUAL().
12        *
13        * Here is an example of specialization of that traits:
14        *
15        * \code
16        * template<>
17        * struct assertion_traits<std::string>   // specialization for the std::string type
18        * {
19        *   static bool equal( const std::string& x, const std::string& y )
20        *   {
21        *     return x == y;
22        *   }
23        * 
24        *   static std::string toString( const std::string& x )
25        *   {
26        *     std::string text = '"' + x + '"';    // adds quote around the string to see whitespace
27        *     OStringStream ost;
28        *     ost << text;
29        *     return ost.str();
30        *   }
31        * };
32        * \endcode
33        */
34       template <class T>
35       struct assertion_traits 
36       {  
37           static bool equal( const T& x, const T& y )
38           {
39               return x == y;
40           }
41     
42           static std::string toString( const T& x )
43           {
44               OStringStream ost;
45               ost << x;
46               return ost.str();
47           }
48       };

67                            const T& actual,
68                            long lineNumber,
69                            std::string fileName )
70         {
71           if ( !assertion_traits<T>::equal(expected,actual) ) // lazy toString conversion...
72           {
73             assertNotEqualImplementation( assertion_traits<T>::toString(expected),
74                                           assertion_traits<T>::toString(actual),
75                                           lineNumber, 
76                                           fileName );
77           }

159     /** Asserts that two values are equals.
160      * \ingroup Assertions
161      *
162      * Equality and string representation can be defined with
163      * an appropriate CppUnit::assertion_traits class.
164      *
165      * A diagnostic is printed if actual and expected values disagree.
166      *
167      * Requirement for \a expected and \a actual parameters:
168      * - They are exactly of the same type
169      * - They are serializable into a std::strstream using operator <<.
170      * - They can be compared using operator ==. 
171      *
172      * The last two requirements (serialization and comparison) can be
173      * removed by specializing the CppUnit::assertion_traits.
174      */
175     #define CPPUNIT_ASSERT_EQUAL(expected,actual)                     \
176       ( ::CppUnit::TestAssert::assertEquals( (expected),              \
177                                              (actual),                \
178                                              CPPUNIT_SOURCELINE() ) )
179     
180     /** Asserts that two values are equals, provides additional messafe on failure.
181      * \ingroup Assertions
182      *
183      * Equality and string representation can be defined with
184      * an appropriate assertion_traits class.
185      *
186      * A diagnostic is printed if actual and expected values disagree.
187      * The message is printed in addition to the expected and actual value
188      * to provide additional information.
189      *
190      * Requirement for \a expected and \a actual parameters:
191      * - They are exactly of the same type
192      * - They are serializable into a std::strstream using operator <<.
193      * - They can be compared using operator ==. 
194      *
195      * The last two requirements (serialization and comparison) can be
196      * removed by specializing the CppUnit::assertion_traits.
197      */
198     #define CPPUNIT_ASSERT_EQUAL_MESSAGE(message,expected,actual)     \
199       ( ::CppUnit::TestAssert::assertEquals( (expected),              \
200                                              (actual),                \
201                                              CPPUNIT_SOURCELINE(),    \
202                                              (message) ) )
203     #endif
