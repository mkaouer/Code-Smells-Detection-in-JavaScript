13     /** (Implementation) This class manages all the TestFactoryRegistry.
14      *
15      * Responsible for the life-cycle of the TestFactoryRegistry.
16      * 
17      * TestFactory registry must call wasDestroyed() to indicate that
18      * a given TestRegistry was destroyed, and needDestroy() to
19      * know if a given TestFactory need to be destroyed (was not already
20      * destroyed by another TestFactoryRegistry).
21      */
22     class NamedRegistries
23     {
24     public:
25       ~NamedRegistries();
26     
27       static NamedRegistries &getInstance();
28     
29       TestFactoryRegistry &getRegistry( std::string name );
30     
31       void wasDestroyed( TestFactory *factory );
32     
33       bool needDestroy( TestFactory *factory );
34     
35     private:
36       typedef std::map<std::string, TestFactoryRegistry *> Registries;
37       Registries m_registries;
38     
39       typedef std::set<TestFactory *> Factories;
40       Factories m_factoriesToDestroy;
41       Factories m_destroyedFactories;
42     };
43     
44     
45     NamedRegistries::~NamedRegistries()
46     {
47       Registries::iterator it = m_registries.begin();
48       while ( it != m_registries.end() )
49       {
50         TestFactoryRegistry *registry = (it++)->second;
51         if ( needDestroy( registry ) )
52           delete registry;
53       }
54     }

57     NamedRegistries &
58     NamedRegistries::getInstance()
59     {
60       static NamedRegistries namedRegistries;
61       return namedRegistries;
62     }

65     TestFactoryRegistry &
66     NamedRegistries::getRegistry( std::string name )
67     {
68       Registries::const_iterator foundIt = m_registries.find( name );
69       if ( foundIt == m_registries.end() )
70       {
71         TestFactoryRegistry *factory = new TestFactoryRegistry( name );
72         m_registries.insert( std::make_pair( name, factory ) );
73         m_factoriesToDestroy.insert( factory );
74         return *factory;
75       }
76       return *foundIt->second;
77     }

80     void 
81     NamedRegistries::wasDestroyed( TestFactory *factory )
82     {
83       m_factoriesToDestroy.erase( factory );
84       m_destroyedFactories.insert( factory );
85     }

88     bool 
89     NamedRegistries::needDestroy( TestFactory *factory )
90     {
91       return m_destroyedFactories.count( factory ) == 0;
92     }

102     TestFactoryRegistry::~TestFactoryRegistry()
103     {
104       // The wasDestroyed() and needDestroy() is used to prevent
105       // a double destruction of a factory registry.
106       // registerFactory( "All Tests", getRegistry( "Unit Tests" ) );
107       // => the TestFactoryRegistry "Unit Tests" is owned by both
108       // the "All Tests" registry and the NamedRegistries...
109       NamedRegistries::getInstance().wasDestroyed( this );
110     
111       for ( Factories::iterator it = m_factories.begin(); it != m_factories.end(); ++it )
112       {
113         TestFactory *factory = it->second;
114         if ( NamedRegistries::getInstance().needDestroy( factory ) )
115           delete factory;
116       }
117     }

163     void 
164     TestFactoryRegistry::addTestToSuite( TestSuite *suite )
165     {
166       for ( Factories::iterator it = m_factories.begin(); 
167             it != m_factories.end(); 
168             ++it )
169       {
170         TestFactory *factory = (*it).second;
