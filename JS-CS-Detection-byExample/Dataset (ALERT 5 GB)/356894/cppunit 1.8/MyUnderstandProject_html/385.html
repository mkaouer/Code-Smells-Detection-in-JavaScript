22      */

29          */

38         virtual int                   countTestCases () const     = 0;
39     
40         /*! \brief Returns the test name.
41          * 
42          * Each test has a name.  This name may be used to find the

45         virtual std::string           getName () const            = 0;
46     
47         /*! \brief Description of the test, for diagnostic output.
48          *
49          * The test description will typically include the test name,
