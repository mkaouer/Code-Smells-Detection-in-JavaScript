107     void 
108     CompilerOutputter::printFailureMessage( TestFailure *failure )
109     {
110       m_stream  <<  std::endl;
111       Exception *thrownException = failure->thrownException();
112       if ( thrownException->isInstanceOf( NotEqualException::type() ) )
113         printNotEqualMessage( thrownException );
114       else
115         printDefaultMessage( thrownException );
116       m_stream  <<  std::endl;
117     }

120     void 
121     CompilerOutputter::printNotEqualMessage( Exception *thrownException )
122     {
123       NotEqualException *e = (NotEqualException *)thrownException;
124       m_stream  <<  wrap( "- Expected : " + e->expectedValue() );
125       m_stream  <<  std::endl;
126       m_stream  <<  wrap( "- Actual   : " + e->actualValue() );
127       m_stream  <<  std::endl;
128       if ( !e->additionalMessage().empty() )
129       {
130         m_stream  <<  wrap( e->additionalMessage() );
131         m_stream  <<  std::endl;
132       }
133     }

156     std::string
157     CompilerOutputter::wrap( std::string message )
158     {
159       Lines lines = splitMessageIntoLines( message );
160       std::string wrapped;
161       for ( Lines::iterator it = lines.begin(); it != lines.end(); ++it )
162       {
163         std::string line( *it );
164         const int maxLineLength = 80;
165         int index =0;
166         while ( index < line.length() )
167         {
168           std::string line( line.substr( index, maxLineLength ) );
169           wrapped += line;
170           index += maxLineLength;
171           if ( index < line.length() )
172             wrapped += "\n";
173         }
174         wrapped += '\n';
175       }
176       return wrapped;
177     }

180     CompilerOutputter::Lines 
181     CompilerOutputter::splitMessageIntoLines( std::string message )
182     {
183       Lines lines;
184     
185       std::string::iterator itStart = message.begin();
186       while ( true )
187       {
188         std::string::iterator itEol = std::find( itStart, 
189                                                  message.end(), 
190                                                  '\n' );
191         lines.push_back( message.substr( itStart - message.begin(),
192                                          itEol - itStart ) );
193         if ( itEol == message.end() )
194           break;
195         itStart = itEol +1;
196       }
197       return lines;
198     }
