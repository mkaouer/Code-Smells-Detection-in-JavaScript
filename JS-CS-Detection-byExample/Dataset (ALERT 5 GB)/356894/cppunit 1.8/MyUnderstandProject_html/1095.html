18     extern "C" int APIENTRY
19     DllMain(HINSTANCE hInstance, DWORD dwReason, LPVOID lpReserved)
20     {
21         // Remove this if you use lpReserved
22         UNREFERENCED_PARAMETER(lpReserved);
23     
24         if (dwReason == DLL_PROCESS_ATTACH)
25         {
26             TRACE0("TESTRUNNER.DLL Initializing!\n");
27             
28             // Extension DLL one-time initialization
29             if (!AfxInitExtensionModule(TestRunnerDLL, hInstance))
30                 return 0;
31     
32             // Can't find any other way to have LoadAccelerators working...
33             g_testRunnerResource = TestRunnerDLL.hResource;
34     
35             // Insert this DLL into the resource chain
36             // NOTE: If this Extension DLL is being implicitly linked to by
37             //  an MFC Regular DLL (such as an ActiveX Control)
38             //  instead of an MFC application, then you will want to
39             //  remove this line from DllMain and put it in a separate
40             //  function exported from this Extension DLL.  The Regular DLL
41             //  that uses this Extension DLL should then explicitly call that
42             //  function to initialize this Extension DLL.  Otherwise,
43             //  the CDynLinkLibrary object will not be attached to the
44             //  Regular DLL's resource chain, and serious problems will
45             //  result.
46     
47             new CDynLinkLibrary(TestRunnerDLL);
48         }
49         else if (dwReason == DLL_PROCESS_DETACH)
50         {
51             TRACE0("TESTRUNNER.DLL Terminating!\n");
52             // Terminate the library before destructors are called
53             AfxTermExtensionModule(TestRunnerDLL);
54         }
55         return 1;   // ok
56     }
