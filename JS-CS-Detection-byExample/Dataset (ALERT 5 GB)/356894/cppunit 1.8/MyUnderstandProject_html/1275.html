158     void 
159     TestRunnerDlg::OnRun() 
160     {
161       if ( m_bIsRunning )
162         return;
163     
164       m_selectedTest = m_model->selectedTest();
165     
166       if ( m_selectedTest == 0 )
167         return;
168     
169       freeState(); 
170       reset();
171     
172       beRunning();
173     
174       int numberOfTests = m_selectedTest->countTestCases();
175     
176       m_testsProgress->start( numberOfTests );
177     
178       
179       m_result = new CppUnit::TestResultCollector( new MfcSynchronizationObject() );
180       m_testObserver = new CppUnit::TestResult( new MfcSynchronizationObject() );
181       m_testObserver->addListener( m_result );
182       m_testObserver->addListener( this );
183       m_activeTest = new ActiveTest( m_selectedTest );
184     
185       m_testStartTime = timeGetTime();
186     
187       m_activeTest->run( m_testObserver );
188     
189       m_testEndTime = timeGetTime();
190     }

193     void 
194     TestRunnerDlg::addListEntry( const CppUnit::TestFailure &failure )
195     {
196       CListCtrl *listCtrl = (CListCtrl *)GetDlgItem (IDC_LIST);
197       int currentEntry = m_result->testErrors() + 
198                          m_result->testFailures() -1;
199     
200       ErrorTypeBitmaps errorType;
201       if ( failure.isError() )
202         errorType = errorTypeError;
203       else
204         errorType = errorTypeFailure;
205     
206       ListCtrlSetter setter( *listCtrl );
207       setter.insertLine( currentEntry );
208       setter.addSubItem( failure.isError() ? _T("Error") : _T("Failure"), errorType );
209     
210       // Set test name
211       setter.addSubItem( failure.failedTestName().c_str(), errorType );
212     
213       // Set the asserted text
214       setter.addSubItem( failure.thrownException()->what() );
215     
216       // Set the line number
217       if ( failure.sourceLine().isValid() )
218       {
219         CString lineNumber;
220         lineNumber.Format( _T("%ld"), failure.sourceLine().lineNumber() );
221         setter.addSubItem( lineNumber );
222       }
223       else
224         setter.addSubItem( _T("") );
225     
226       // Set the file name
227       setter.addSubItem( failure.sourceLine().fileName().c_str() );
228     
229     /* In place of the missing detail field...
230       std::string dump = "Test: " + test->getName() + "\n";
231       dump += e->what();
232       dump += "\n";
233       ::OutputDebugString( dump.c_str() );
234     */
235     
236       listCtrl->RedrawItems (currentEntry, currentEntry);
237       listCtrl->UpdateWindow ();
238     }

260     void 
261     TestRunnerDlg::endTest( CppUnit::Test *test )
262     {
263       if ( m_selectedTest == 0 )
264         return;
265     
266       m_testsRun++;
267       updateCountsDisplay();
268       m_testsProgress->step( m_failures == 0  &&  m_errors == 0 );
269     
270       m_testEndTime = timeGetTime();
271     
272       updateCountsDisplay();
273     
274       if ( m_testsRun >= m_selectedTest->countTestCases() )
275         beIdle ();
276     }

394     void 
395     TestRunnerDlg::OnSelchangeComboTest() 
396     {
397       int currentSelection = getHistoryCombo()->GetCurSel ();
398     
399       if ( currentSelection >= 0  &&
400            currentSelection < m_model->history().size() )
401       {
402         CppUnit::Test *selectedTest = m_model->history()[currentSelection];
403         m_model->selectHistoryTest( selectedTest );
404         updateHistoryCombo();
405         beIdle();
406       }
407       else
408         beRunDisabled();
409     }

412     void
413     TestRunnerDlg::updateHistoryCombo()
414     {
415       getHistoryCombo()->LockWindowUpdate();
416     
417       getHistoryCombo()->ResetContent();
418     
419       const TestRunnerModel::History &history = m_model->history();
420       for ( TestRunnerModel::History::const_iterator it = history.begin(); 
421             it != history.end(); 
422             ++it )
423       {
424         CppUnit::Test *test = *it;
425         getHistoryCombo()->AddString( CString(test->getName().c_str()) );
426       }
427     
428       if ( history.size() > 0 )
429       {
430         getHistoryCombo()->SetCurSel( 0 );
431         beIdle();
432       }
433       else
434         beRunDisabled();
435     
436       getHistoryCombo()->UnlockWindowUpdate();
437     }

539     TestRunnerDlg::updateLayoutInfo()
540     {
541       // get dialog in screen co-ordinates
542       RECT dlgBounds;
543       GetWindowRect( &dlgBounds );

545       // Set general margin according to distance of browse button from the right
546       m_margin = dlgBounds.right - getItemWindowRect(IDC_BROWSE_TEST).right;
547       m_listViewDelta = dlgBounds.bottom - getItemWindowRect( IDC_LIST ).bottom;

552     void 
553     TestRunnerDlg::OnSize(UINT nType, int cx, int cy) 
554     {
555       CDialog::OnSize(nType, cx, cy);
556       
557       // Layout controls according to new size	
558       CRect r;
559       CRect dlgBounds = getDialogBounds();
560       
561       int buttonLeft  = 0;
562       int buttonRight = 0;
563       
564       CWnd * pItem = GetDlgItem(IDC_BROWSE_TEST);
565       if (pItem)
566       {
567         pItem->GetWindowRect( &r );
568         // adjust to dialog units
569         r.OffsetRect(-dlgBounds.left, -dlgBounds.top);
570         
571         int buttonWidth = r.Width();
572         buttonLeft  = cx - buttonWidth - m_margin; 
573         buttonRight = buttonLeft + buttonWidth;
574         r.left =  buttonLeft;
575         r.right = buttonRight;
576         
577         pItem->MoveWindow( &r );
578         pItem->Invalidate();
579       }
580     
581       updateTopButtonPosition( ID_RUN, buttonLeft, buttonRight );
582       updateTopButtonPosition( IDC_CHECK_AUTORUN, buttonLeft, buttonRight );
583       updateBottomButtonPosition( IDOK, buttonLeft, buttonRight, cy - m_editDelta );
584       updateBottomButtonPosition( ID_STOP, buttonLeft, buttonRight, cy - m_listViewDelta );
585       
586       pItem = GetDlgItem(IDC_COMBO_TEST);
587       if (pItem)
588       {
589         pItem->GetClientRect( &r );
590         pItem->ClientToScreen( &r );
591         
592         // adjust to dialog units
593         r.OffsetRect(-dlgBounds.left, -dlgBounds.top);
594         r.right = buttonLeft - m_margin;
595         
596         pItem->MoveWindow( &r );
597         pItem->Invalidate();
598       }
599     
600       updateListPosition( buttonLeft );
601       
602       if (m_testsProgress)
603       {
604         m_testsProgress->setWidth(r.right - r.left);
605       }
606       
607       pItem = GetDlgItem(IDC_EDIT_TIME);
608       if (pItem)
609       {
610         pItem->GetWindowRect( &r );
611         
612         // adjust to dialog units
613         r.OffsetRect(-dlgBounds.left, -dlgBounds.top);
614         
615         r.right =  buttonLeft - m_margin;
616         
617         int height = r.Height();
618         // order matters here ( r.top uses new r.bottom )
619         r.bottom = cy - m_editDelta;
620         r.top = r.bottom - height;
621         
622         pItem->MoveWindow( &r );
623         pItem->Invalidate();
624       }
625     }

682     void 
683     TestRunnerDlg::updateListPosition( int xButtonLeft )
684     {
685       CWnd *pItem = GetDlgItem( IDC_LIST );
686       if ( !pItem )
687         return;
688       CRect r;
689       pItem->GetWindowRect( &r );
690       
691       // adjust to dialog units
692       CRect dlgBounds = getDialogBounds();
693       r.OffsetRect(-dlgBounds.left, -dlgBounds.top);
694       
695       r.right = xButtonLeft - m_margin;
696       
697       // resize to fit last column
698       
699       LVCOLUMN lv;
700       lv.mask = LVCF_WIDTH;
701       int width_1_4 = 0;
702       for (int i = 0; i < 4; ++i)
703       {
704         m_listCtrl.GetColumn(i, &lv);
705         width_1_4 += lv.cx;
706       }
707       
708       // the 4 offset is so no horiz scroll bar will appear
709       m_listCtrl.SetColumnWidth(4, r.Width() - width_1_4 - 4); 
710       
711       // order matters here ( r.top uses new r.bottom )
712       r.bottom = getDialogBounds().Height() - m_listViewDelta;
713       
714       pItem->MoveWindow( &r );
715       pItem->Invalidate();
716     }
