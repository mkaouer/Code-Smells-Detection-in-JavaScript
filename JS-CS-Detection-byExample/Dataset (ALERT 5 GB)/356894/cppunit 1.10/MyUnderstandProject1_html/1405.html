36     void 
37     TestRunnerModel::resetTestReportCounterFor( CPPUNIT_NS::Test *testToRun )
38     {
39       if ( isTestRunning() )
40         return;
41     
42       {
43         LockGuard guard( _lock );
44         _numberOfTestCase = testToRun->countTestCases();
45         _numberOfTestCaseRun =0;
46         _numberOfTestCaseFailure =0;
47         _failures.clear();
48       }
49       emit failuresCleared();
50     
51       emit numberOfTestCaseChanged( _numberOfTestCase );
52       emit numberOfTestCaseRunChanged( _numberOfTestCaseRun );
53       emit numberOfTestCaseFailureChanged( _numberOfTestCaseFailure );
54     }

89     void 
90     TestRunnerModel::runTest( CPPUNIT_NS::Test *testToRun )
91     {
92       if ( isTestRunning() )
93         return;
94     
95       resetTestReportCounterFor( testToRun );
96     
97       {
98         LockGuard guard( _lock );
99         delete _result;
100         _result = new CPPUNIT_NS::TestResult();
101         _result->addListener( this );
102       }
103     
104       emit testRunStarted( testToRun, _result );
105     
106       LockGuard guard( _lock );
107       _runnerThread = new TestRunnerThread( testToRun, 
108                                             _result, 
109                                             this, 
110                                             new TestRunnerThreadFinishedEvent() );
111     }

114     bool 
115     TestRunnerModel::isTestRunning()
116     {
117       LockGuard guard( _lock );
118       return _runnerThread != NULL  &&  _runnerThread->running();
119     }

122     void 
123     TestRunnerModel::stopRunningTest()
124     {
125       {
126         LockGuard guard( _lock );
127         if ( _result == NULL )
128           return;
129       }
130       if ( isTestRunning() )
131       {
132         LockGuard guard( _lock );
133         _result->stop();
134       }
135     }

155     // Called from the TestRunnerThread.
156     void 
157     TestRunnerModel::endTest( CPPUNIT_NS::Test *test )
158     {
159       int numberOfTestCaseRun;
160       {
161         LockGuard guard( _lock );
162         numberOfTestCaseRun = ++_numberOfTestCaseRun;
163       }
164     
165       // emit signal asynchronously
166       QThread::postEvent( this, 
167                           new TestRunnerTestCaseRunEvent( numberOfTestCaseRun ) );
168     }
169     
170     
171     // Called from the TestRunnerThread.
172     void 
173     TestRunnerModel::addFailureInfo( TestFailureInfo *failure )
174     {
175       int numberOfTestCaseFailure;
176       {
177         LockGuard guard( _lock );
178         _failures.append( failure );
179         numberOfTestCaseFailure = ++_numberOfTestCaseFailure;
180       }
181     
182       // emit signals asynchronously
183       QThread::postEvent( this, 
184                           new TestRunnerFailureEvent( failure,
185                                                       numberOfTestCaseFailure ) );
186     }

189     bool 
190     TestRunnerModel::event( QEvent *event )
191     {
192       if ( event->type() != QEvent::User )
193         return false;
194     
195       TestRunnerThreadEvent *threadEvent = (TestRunnerThreadEvent *)event;
196       threadEvent->process( this );
197       return true;
198     }
