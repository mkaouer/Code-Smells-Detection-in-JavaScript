16     /*! Constructs a new text runner.
17      * \param outputter used to print text result. Owned by the runner.
18      */
19     TextTestRunner::TextTestRunner( Outputter *outputter ) 
20         : m_outputter( outputter )
21         , m_result( new TestResultCollector() )
22         , m_eventManager( new TestResult() )
23     {
24       if ( !m_outputter )
25         m_outputter = new TextOutputter( m_result, std::cout );
26       m_eventManager->addListener( m_result );
27     }

38     /*! Runs the named test case.
39      *
40      * \param testName Name of the test case to run. If an empty is given, then
41      *                 all added tests are run. The name can be the name of any
42      *                 test in the hierarchy.
43      * \param doWait if \c true then the user must press the RETURN key 
44      *               before the run() method exit.
45      * \param doPrintResult if \c true (default) then the test result are printed
46      *                      on the standard output.
47      * \param doPrintProgress if \c true (default) then TextTestProgressListener is
48      *                        used to show the progress.
49      * \return \c true is the test was successful, \c false if the test
50      *         failed or was not found.
51      */
52     bool
53     TextTestRunner::run( std::string testName,
54                            bool doWait,
55                            bool doPrintResult,
56                            bool doPrintProgress )
57     {
58       TextTestProgressListener progress;
59       if ( doPrintProgress )
60         m_eventManager->addListener( &progress );
61     
62       TestRunner *pThis = this;
63       pThis->run( *m_eventManager, testName );
64     
65       if ( doPrintProgress )
66         m_eventManager->removeListener( &progress );
67     
68       printResult( doPrintResult );
69       wait( doWait );
70     
71       return m_result->wasSuccessful();
72     }

76     TextTestRunner::wait( bool doWait )
77     {
78       if ( doWait ) 
79       {
80         std::cout << "<RETURN> to continue" << std::endl;
81         std::cin.get ();
82       }
83     }
