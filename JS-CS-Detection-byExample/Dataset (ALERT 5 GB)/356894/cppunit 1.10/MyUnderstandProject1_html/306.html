22     /*! \brief Message associated to an Exception.
23      * \ingroup CreatingNewAssertions
24      *  A message is composed of two items:
25      *  - a short description (~20/30 characters)
26      *  - a list of detail strings
27      *
28      *  The short description is used to indicate how the detail strings should be
29      *  interpreted. It usually indicates the failure types, such as
30      *  "assertion failed", "forced failure", "unexpected exception caught",
31      *  "equality assertion failed"... It should not contains new line character (\n).
32      *
33      *  Detail strings are used to provide more information about the failure. It
34      *  can contains the asserted expression, the expected and actual values in an
35      *  equality assertion, some addional messages... Detail strings can contains
36      *  new line characters (\n).
37      */
38     class CPPUNIT_API Message
39     {
40     public:
41       Message();
42     
43       explicit Message( const std::string &shortDescription );
44     
45       Message( const std::string &shortDescription,
46                const std::string &detail1 );
47     
48       Message( const std::string &shortDescription,
49                const std::string &detail1,
50                const std::string &detail2 );
51     
52       Message( const std::string &shortDescription,
53                const std::string &detail1,
54                const std::string &detail2,
55                const std::string &detail3 );
56     
57       /*! \brief Returns the short description.
58        * \return Short description.
59        */
60       const std::string &shortDescription() const;
61     
62       /*! \brief Returns the number of detail string.
63        * \return Number of detail string.
64        */
65       int detailCount() const;
66     
67       /*! \brief Returns the detail at the specified index.
68        * \param index Zero based index of the detail string to return.
69        * \returns Detail string at the specified index.
70        * \exception std::invalid_argument if \a index < 0 or index >= detailCount().
71        */
72       std::string detailAt( int index ) const;
73     
74       /*! \brief Returns a string that represents a list of the detail strings.
75        *
76        * Example:
77        * \code
78        * Message message( "not equal", "Expected: 3", "Actual: 7" );
79        * std::string details = message.details();
80        * // details contains:
81        * // "- Expected: 3\n- Actual: 7\n"  \endcode
82        *
83        * \return A string that is a concatenation of all the detail strings. Each detail
84        *         string is prefixed with '- ' and suffixed with '\n' before being
85        *         concatenated to the other.
86        */
87       std::string details() const;
88     
89       /*! \brief Removes all detail strings.
90        */
91       void clearDetails();
92     
93       /*! \brief Adds a single detail string.
94        * \param detail Detail string to add.
95        */
96       void addDetail( const std::string &detail );
97     
98       /*! \brief Adds two detail strings.
99        * \param detail1 Detail string to add.
100        * \param detail2 Detail string to add.
101        */
102       void addDetail( const std::string &detail1,
103                       const std::string &detail2 );
104     
105       /*! \brief Adds three detail strings.
106        * \param detail1 Detail string to add.
107        * \param detail2 Detail string to add.
108        * \param detail3 Detail string to add.
109        */
110       void addDetail( const std::string &detail1,
111                       const std::string &detail2,
112                       const std::string &detail3 );
113     
114       /*! \brief Adds the detail strings of the specified message.
115        * \param message All the detail strings of this message are added to this one.
116        */
117       void addDetail( const Message &message );
118     
119       /*! \brief Sets the short description.
120        * \param shortDescription New short description.
121        */
122       void setShortDescription( const std::string &shortDescription );
123     
124       /*! \brief Tests if a message is identical to another one.
125        * \param other Message this message is compared to.
126        * \return \c true if the two message are identical, \c false otherwise.
127        */
128       bool operator ==( const Message &other ) const;
129     
130     
131       /*! \brief Tests if a message is different from another one.
132        * \param other Message this message is compared to.
133        * \return \c true if the two message are not identical, \c false otherwise.
134        */
135       bool operator !=( const Message &other ) const;
136     
137     private:
138       std::string m_shortDescription;
139     
140       typedef CppUnitDeque<std::string> Details;
141       Details m_details;
142     };
