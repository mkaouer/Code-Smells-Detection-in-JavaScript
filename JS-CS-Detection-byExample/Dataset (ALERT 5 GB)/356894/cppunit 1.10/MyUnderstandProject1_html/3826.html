60      * initialize window
61      */
62     
63     BOOL cdxCDynamicPropSheet::OnInitDialog() 
64     {
65     	// initialize window & dynamic manager
66     
67     	BOOL	b	=	CPropertySheet::OnInitDialog();
68     	DoInitWindow(*this);
69     
70     	ModifyStyle(0,WS_CLIPSIBLINGS);
71     
72     	ASSERT(GetPageCount() > 0);			// NO pages ??
73     	cdxCDynamicPropPage	*pActive	=	(cdxCDynamicPropPage *)GetActivePage();
74     
75     	ASSERT(pActive && pActive->IsKindOf(RUNTIME_CLASS(cdxCDynamicPropPage)));
76     	AddSzControl(*pActive,mdResize,mdResize);
77     	VERIFY( GetControlPosition(*pActive,m_PagePos) );
78     	m_bHasPos	=	true;
79     
80     	return b;
81     }

83     void cdxCDynamicPropSheet::AddPage( cdxCDynamicPropPage & rPage )
84     {
85     	ASSERT(rPage.m_pSheet == NULL);
86     	rPage.m_pSheet	=	this;
87     	CPropertySheet::AddPage(&rPage);
88     }
89     
90     void cdxCDynamicPropSheet::RemovePage( cdxCDynamicPropPage & rPage )
91     {
92     	ASSERT(rPage.m_pSheet == this);
93     	rPage.m_pSheet	=	NULL;
94     }

96     void cdxCDynamicPropSheet::OnInitPage(cdxCDynamicPropPage & rPage)
97     {
98     	ASSERT(::IsWindow(rPage));
99     
100     	if(m_bHasPos)
101     		AddSzControl(rPage,m_PagePos);
102     }

222     BOOL cdxCDynamicPropPage::OnSetActive() 
223     {
224     	BOOL	bGetsActive	=	CPropertyPage::OnSetActive();
225     	if(bGetsActive && !m_bFirstHit)
226     	{
227     		m_pSheet->OnInitPage(*this);
228     		m_bFirstHit	=	true;
229     	}
230     	if(m_pSheet)
231     		m_pSheet->OnSetActive(*this,bGetsActive);
232     	return bGetsActive;
233     }
234     
235     BOOL cdxCDynamicPropPage::OnKillActive() 
236     {
237     	BOOL	bGetsKilled	=	CPropertyPage::OnKillActive();
238     	if(m_pSheet)
239     		m_pSheet->OnKillActive(*this,bGetsKilled);
240     	return bGetsKilled;
241     }
