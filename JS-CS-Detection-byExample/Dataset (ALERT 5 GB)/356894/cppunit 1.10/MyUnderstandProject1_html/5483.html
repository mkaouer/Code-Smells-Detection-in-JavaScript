23       typedef CppUnit::TestRunner SuperClass;   // work around VC++ bug

26       {

57     std::string

59     {

61       int bufferLength = ::WideCharToMultiByte( CP_THREAD_ACP, 0, 
62                                                 text, text.GetLength(),
63                                                 NULL, 0, NULL, NULL ) +1;
64       char *ansiString = new char[bufferLength];
65       ::WideCharToMultiByte( CP_THREAD_ACP, 0, 
66                              text, text.GetLength(),
67                              ansiString, bufferLength, 
68                                                 NULL,
69                                                 NULL );

81     /*! Runs the specified tests located in the root suite.
82      * \param root Root suite that contains all the test of the DLL.
83      * \param testPaths Array of string that contains the test paths of all the test to run.
84      * \param numberOfPath Number of test paths in \a testPaths. If 0 then \a root suite
85      *                     is run.
86      * \return \c true if the run succeed, \c false if a test failed or if a test
87      *         path was not resolved.
88      */
89     bool 
90     runDllTest( CppUnit::Test *root,
91                 TCHAR *testPaths[],
92                 int numberOfPath )
93     {
94       CppUnit::TestResult controller;
95       CppUnit::TestResultCollector result;
96       controller.addListener( &result );        
97       CppUnit::TextTestProgressListener progress;
98       controller.addListener( &progress );      
99     
100       NotOwningTestRunner runner;
101       if ( numberOfPath == 0 )
102         runner.addTest( root );
103       else
104       {
105         for ( int index =0; index < numberOfPath; ++index )
106         {
107           const TCHAR *testPath = testPaths[index];
108           try
109           {
110             runner.addTest( root->resolveTestPath( testPath).getChildTest() );
111           }
112           catch ( std::invalid_argument & )
113           {
114             TCERR  <<  _T("Failed to resolve test path: ")  <<  testPath  <<  std::endl;
115             return false;
116           }
117         }
118       }
119     
120       runner.run( controller );
121     
122       std::cerr << std::endl;
123     
124       CppUnit::CompilerOutputter outputter( &result, std::cerr );
125       outputter.write();
126     
127       return result.wasSuccessful();
128     }

131     /*! Main
132      * 
133      * Usage: 
134      *
135      * DllPlugInTester.exe dll-filename [testpath1] [testpath2]...
136      *
137      * <em>dll-filename</em> must be the name of the DLL. If the DLL use some other DLL, they
138      * should be in the path or in the same directory as the DLL. The DLL must export
139      * a function named "GetTestPlugInInterface" with the signature
140      * GetTestPlugInInterfaceFunction. Both are defined in:
141      * \code
142      * #include <msvc6/testrunner/TestPlugInInterface.h>
143      * \endcode.
144      *
145      * See examples/msvc6/TestPlugIn for an example of post-build testing.
146      *
147      * If no test path is specified, they all the test of the suite returned by the DLL
148      * are run.
149      *
150      * You can specify as much test path as you which. Only the test specified by the
151      * test paths will be run. Test paths are resolved using Test::resolveTestPath() on
152      * the suite returned by the DLL.
153      * 
154      * If all test succeed and no error happen then the application exit with code 0.
155      * If any error occurs (failed to load dll, failed to resolve test paths) or a 
156      * test fail, the application exit with code 1.
157      */
158     int 
159     _tmain( int argc, 
160             TCHAR* argv[], 
161             TCHAR* envp[] )
162     {
163       const int successReturnCode = 0;
164       const int failureReturnCode = 1;
165     
166       // check command line
167       const TCHAR *applicationName = argv[0];
168       if ( argc < 2 )
169       {
170         TCERR  <<  _T("Usage: ")  <<  std::endl
171                <<  applicationName 
172                <<  " dll-filename [test-path] [test-path]..."
173                <<  std::endl;
174         return failureReturnCode;
175       }
176     
177       // open the dll
178       const TCHAR *dllFileName = argv[1];
179     
180       bool wasSuccessful = false;
181       try
182       {
183         CppUnit::TestPlugInSuite suite( dllFileName );
184         wasSuccessful = runDllTest( &suite, argv+2, argc-2 );
185       }
186       catch ( CppUnit::DynamicLibraryManagerException &e )
187       {
188         TCERR  << "Failed to load test plug-in:"  <<  std::endl
189                << toVariableString( e.what() )  << std::endl;
190       }
191     
192       return wasSuccessful ? successReturnCode : failureReturnCode;
193     }
