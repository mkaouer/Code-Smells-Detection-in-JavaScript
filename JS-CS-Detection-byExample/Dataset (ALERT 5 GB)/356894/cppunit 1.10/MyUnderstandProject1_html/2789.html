61     void
62     XmlOutputter::setRootNode()
63     {
64       XmlElement *rootNode = new XmlElement( "TestRun" );
65       m_xml->setRootElement( rootNode );
66     
67       for ( Hooks::iterator it = m_hooks.begin(); it != m_hooks.end(); ++it )
68         (*it)->beginDocument( m_xml );
69     
70       FailedTests failedTests;
71       fillFailedTestsMap( failedTests );
72     
73       addFailedTests( failedTests, rootNode );
74       addSuccessfulTests( failedTests, rootNode );
75       addStatistics( rootNode );
76     
77       for ( Hooks::iterator itEnd = m_hooks.begin(); itEnd != m_hooks.end(); ++itEnd )
78         (*itEnd)->endDocument( m_xml );
79     }

82     void 
83     XmlOutputter::fillFailedTestsMap( FailedTests &failedTests )
84     {
85       const TestResultCollector::TestFailures &failures = m_result->failures();
86       TestResultCollector::TestFailures::const_iterator itFailure = failures.begin();
87       while ( itFailure != failures.end() )
88       {
89         TestFailure *failure = *itFailure++;
90         failedTests.insert( std::pair<Test* const, TestFailure*>(failure->failedTest(), failure ) );
91       }
92     }

95     void
96     XmlOutputter::addFailedTests( FailedTests &failedTests,
97                                   XmlElement *rootNode )
98     {
99       XmlElement *testsNode = new XmlElement( "FailedTests" );
100       rootNode->addElement( testsNode );
101     
102       const TestResultCollector::Tests &tests = m_result->tests();
103       for ( unsigned int testNumber = 0; testNumber < tests.size(); ++testNumber )
104       {
105         Test *test = tests[testNumber];
106         if ( failedTests.find( test ) != failedTests.end() )
107           addFailedTest( test, failedTests[test], testNumber+1, testsNode );
108       }
109     }

114                                                XmlElement *rootNode )

116       XmlElement *testsNode = new XmlElement( "SuccessfulTests" );
117       rootNode->addElement( testsNode );

120       for ( unsigned int testNumber = 0; testNumber < tests.size(); ++testNumber )
121       {
122         Test *test = tests[testNumber];
123         if ( failedTests.find( test ) == failedTests.end() )
124           addSuccessfulTest( test, testNumber+1, testsNode );
125       }

129     void
130     XmlOutputter::addStatistics( XmlElement *rootNode )
131     {
132       XmlElement *statisticsElement = new XmlElement( "Statistics" );
133       rootNode->addElement( statisticsElement );
134       statisticsElement->addElement( new XmlElement( "Tests", m_result->runTests() ) );
135       statisticsElement->addElement( new XmlElement( "FailuresTotal", 
136                                                      m_result->testFailuresTotal() ) );
137       statisticsElement->addElement( new XmlElement( "Errors", m_result->testErrors() ) );
138       statisticsElement->addElement( new XmlElement( "Failures", m_result->testFailures() ) );
139     
140       for ( Hooks::iterator it = m_hooks.begin(); it != m_hooks.end(); ++it )
141         (*it)->statisticsAdded( m_xml, statisticsElement );
142     }

145     void
146     XmlOutputter::addFailedTest( Test *test,
147                                  TestFailure *failure,
148                                  int testNumber,
149                                  XmlElement *testsNode )
150     {
151       Exception *thrownException = failure->thrownException();
152       
153       XmlElement *testElement = new XmlElement( "FailedTest" );
154       testsNode->addElement( testElement );
155       testElement->addAttribute( "id", testNumber );
156       testElement->addElement( new XmlElement( "Name", test->getName() ) );
157       testElement->addElement( new XmlElement( "FailureType", 
158                                                failure->isError() ? "Error" : 
159                                                                     "Assertion" ) );
160     
161       if ( failure->sourceLine().isValid() )
162         addFailureLocation( failure, testElement );
163     
164       testElement->addElement( new XmlElement( "Message", thrownException->what() ) );
165     
166       for ( Hooks::iterator it = m_hooks.begin(); it != m_hooks.end(); ++it )
167         (*it)->failTestAdded( m_xml, testElement, test, failure );
168     }

171     void
172     XmlOutputter::addFailureLocation( TestFailure *failure,
173                                       XmlElement *testElement )
174     {
175       XmlElement *locationNode = new XmlElement( "Location" );
176       testElement->addElement( locationNode );
177       SourceLine sourceLine = failure->sourceLine();
178       locationNode->addElement( new XmlElement( "File", sourceLine.fileName() ) );
179       locationNode->addElement( new XmlElement( "Line", sourceLine.lineNumber() ) );
180     }
