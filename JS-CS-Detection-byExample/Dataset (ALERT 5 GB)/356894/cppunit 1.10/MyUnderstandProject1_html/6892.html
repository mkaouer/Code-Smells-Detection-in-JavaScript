66     BOOL cdxCDynamicBar::Create(LPCTSTR lpszWindowName, CWnd* pParentWnd,
67             CSize sizeDefault, BOOL bHasGripper, UINT nID,
68             DWORD dwStyle)
69     {
70     	if(!( CSizingControlBar::Create(	lpszWindowName,
71     												pParentWnd,
72     												sizeDefault,
73     												bHasGripper,
74     												nID,
75     												dwStyle|WS_CLIPCHILDREN) ))
76     	{
77     		ASSERT(false);
78     		return FALSE;
79     	}
80     
81     	if(!( m_rDlg.Create(this) ))
82     	{
83     		DestroyWindow();
84     		ASSERT(false);
85     		return FALSE;
86     	}
87     
88     	ASSERT(::IsWindow(m_hWnd));
89     	ASSERT(m_rDlg.IsWindow());
90     	ASSERT(!m_rectBorder.IsRectNull());
91     
92     	// the following code will even be provided by 
93     	m_szMin.cx				=	m_rectBorder.left + m_rectBorder.right;
94     	m_szMin.cy				=	m_rectBorder.top  + m_rectBorder.bottom;
95     	m_szMin					+=	m_rDlg.m_szMin;
96     
97     	return TRUE;
98     }

139     /*
140      * OnNcCalcSize() is used to calculate the optimum
141      * min size for the dialog.
142      */
143     
144     void cdxCDynamicBar::OnNcCalcSize(BOOL bCalcValidRects, NCCALCSIZE_PARAMS FAR* lpncsp) 
145     {
146     	m_rectBorder			=	lpncsp->rgrc[0];		// load initial rectangle
147     
148     	CSizingControlBar::OnNcCalcSize(bCalcValidRects, lpncsp);
149     
150     	ASSERT(m_rectBorder.left <= lpncsp->rgrc[0].left);
151     	ASSERT(m_rectBorder.top <= lpncsp->rgrc[0].top);
152     	ASSERT(m_rectBorder.right >= lpncsp->rgrc[0].right);
153     	ASSERT(m_rectBorder.bottom >= lpncsp->rgrc[0].bottom);
154     
155     	m_rectBorder.left		=	lpncsp->rgrc[0].left - m_rectBorder.left;
156     	m_rectBorder.top		=	lpncsp->rgrc[0].top  - m_rectBorder.top;
157     	m_rectBorder.right	=	m_rectBorder.right  - lpncsp->rgrc[0].right;
158     	m_rectBorder.bottom	=	m_rectBorder.bottom - lpncsp->rgrc[0].bottom;
159     	m_szMin					=	m_rDlg.m_szMin;
160     	m_szMin.cx				+=	m_rectBorder.left + m_rectBorder.right;
161     	m_szMin.cy				+=	m_rectBorder.top  + m_rectBorder.bottom;
162     }
