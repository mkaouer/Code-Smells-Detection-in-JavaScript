29     /*
30      * Standard Controller's Position() routine
31      * This has the same functionality as known from the former
32      * cdxCDynamicControlsManager class.
33      *
34      * One exception is the new "szMin" property which allows
35      * the class to "hide" the control if it becomes too small
36      * (it will be moved outside the client area).
37      */
38     
39     void cdxCDynamicWnd::Position::Apply(HWND hwnd, CRect & rectNewPos, const cdxCDynamicLayoutInfo & li) const
40     {
41     	if(li.m_bUseScrollPos)
42     	{
43     		rectNewPos.left	=	left   - li.m_pntScrollPos.x;
44     		rectNewPos.right	=	right  - li.m_pntScrollPos.x;
45     		rectNewPos.top		=	top    - li.m_pntScrollPos.y;
46     		rectNewPos.bottom	=	bottom - li.m_pntScrollPos.y;
47     
48     		if(li.m_szDelta.cx >= 0)
49     		{
50     			rectNewPos.left	+=	(m_Bytes[X1] * li.m_szDelta.cx) / 100;
51     			rectNewPos.right	+=	(m_Bytes[X2] * li.m_szDelta.cx) / 100;
52     		}
53     		if(li.m_szDelta.cy >= 0)
54     		{
55     			rectNewPos.top		+=	(m_Bytes[Y1] * li.m_szDelta.cy) / 100;
56     			rectNewPos.bottom	+=	(m_Bytes[Y2] * li.m_szDelta.cy) / 100;
57     		}
58     	}
59     	else
60     	{
61     		rectNewPos.left	=	left   + (m_Bytes[X1] * li.m_szDelta.cx) / 100;
62     		rectNewPos.right	=	right  + (m_Bytes[X2] * li.m_szDelta.cx) / 100;
63     		rectNewPos.top		=	top    + (m_Bytes[Y1] * li.m_szDelta.cy) / 100;
64     		rectNewPos.bottom	=	bottom + (m_Bytes[Y2] * li.m_szDelta.cy) / 100;
65     	}
66     
67     	if(rectNewPos.left + m_szMin.cx >= rectNewPos.right)
68     	{
69     		rectNewPos.right	=	-10;
70     		rectNewPos.left	=	rectNewPos.right - m_szMin.cx;
71     	}
72     	if(rectNewPos.top + m_szMin.cy >= rectNewPos.bottom)
73     	{
74     		rectNewPos.bottom	=	-10;
75     		rectNewPos.top		=	rectNewPos.bottom - m_szMin.cy;
76     	}
77     }

118     /*
119      * AddSzControl()
120      * --------------
121      * Add a control that will react on changes to the parent window's size.
122      *		hwnd			-	the child control.
123      *		pos			-	describes what to do at all.
124      *		bReposNow	-	true to immediately make the control change its position
125      *							if necessary, false if not
126      *							In the latter case you may like to call Layout() afterwards.
127      *
128      * returns false if an invalid window has been passed to this funciton.
129      */
130     
131     bool cdxCDynamicWnd::AddSzControl(HWND hwnd, const Position & pos, bool bReposNow)
132     {
133     	if(!IsWindow())
134     	{
135     		ASSERT(IsWindow());
136     		return false;			// NO assert if hwnd is invalid
137     	}
138     
139     	if(!::IsWindow(hwnd))
140     	{
141     		TRACE(_T("*** NOTE[cdxCDynamicWnd::AddSzControl(HWND,const Position &,bool)]: Handle 0x%lx is not a valid window.\n"),(DWORD)hwnd);
142     		return false;
143     	}
144     
145     	m_Map.SetAt(hwnd,pos);
146     
147     	if(bReposNow)
148     		UpdateControlPosition(hwnd);
149     
150     	return true;
151     }

153     /*
154      * AllControls()
155      * -------------
156      * Apply positioning to all controls of the window.
157      *		bytes			-	positioning data
158      *		bOverwrite	-	overwrite any existing positioning data.
159      *		bReposNow	-	true to immediately make the control change its position
160      *							if necessary, false if not
161      *							In the latter case you may like to call Layout() afterwards.
162      */
163     
164     void cdxCDynamicWnd::AllControls(const SBYTES & bytes, bool bOverwrite, bool bReposNow)
165     {
166     	if(!IsWindow())
167     	{
168     		ASSERT(false);
169     		return;
170     	}
171     
172     	Position	pos;
173     	UINT		nCnt	=	0;
174     
175     	for(HWND hwnd = ::GetWindow(m_pWnd->m_hWnd,GW_CHILD); hwnd; hwnd = ::GetNextWindow(hwnd,GW_HWNDNEXT))
176     	{
177     		if(bOverwrite || !m_Map.Lookup(hwnd,pos))
178     			if(AddSzControl(hwnd,bytes,false))
179     				++nCnt;
180     	}
181     
182     	if(nCnt && bReposNow)
183     		Layout();
184     }

186     /*
187      * RemSzControl()
188      * --------------
189      * Removes a control from the internal list.
190      * The control will remain at its initial position if bMoveToInitialPos is false
191      * Returns false if an error occured.
192      */
193     
194     bool cdxCDynamicWnd::RemSzControl(HWND hwnd, bool bMoveToInitialPos)
195     {
196     	if(!::IsWindow(hwnd) || !IsWindow())
197     		return false;
198     
199     	if(bMoveToInitialPos)
200     	{
201     		Position	pos;
202     
203     		if(!m_Map.Lookup(hwnd,pos))
204     			return false;
205     
206     		VERIFY( ::SetWindowPos(hwnd,HWND_TOP,
207     									pos.left,pos.top,pos.Width(),pos.Height(),
208     									SWP_NOACTIVATE|SWP_NOOWNERZORDER|SWP_NOZORDER) );
209     	}
210     
211     	return m_Map.RemoveKey(hwnd) != FALSE;
212     }

215      * UpdateControlPosition()
216      * =======================
217      * Move control to its desired position.
218      * returns false if HWND is not valid.
219      */
220     
221     bool cdxCDynamicWnd::UpdateControlPosition(HWND hwnd)
222     {
223     	if(!IsWindow())
224     	{
225     		ASSERT(IsWindow());
226     		return false;			// NO assert if hwnd is invalid
227     	}
228     
229     	if(!::IsWindow(hwnd))
230     	{
231     		TRACE(_T("*** NOTE[cdxCDynamicWnd::UpdateControlPosition()]: Handle 0x%lx is not a valid window.\n"),(DWORD)hwnd);
232     		return false;
233     	}
234     
235     	cdxCDynamicLayoutInfo	*pli	=	DoCreateLayoutInfo();
236     	ASSERT(pli != NULL);
237     
238     	if(!pli || !pli->IsInitial())
239     	{
240     		try
241     		{
242     			CRect							rectNew;
243     			WINDOWPLACEMENT			wpl;
244     			wpl.length	=	sizeof(WINDOWPLACEMENT);
245     			VERIFY(::GetWindowPlacement(hwnd,&wpl) );
246     
247     			rectNew	=	wpl.rcNormalPosition;
248     
249     			if(DoMoveCtrl(hwnd,::GetDlgCtrlID(hwnd),rectNew,*pli) &&
250     				(rectNew != wpl.rcNormalPosition) )
251     			{
252     				VERIFY( ::SetWindowPos(hwnd,HWND_TOP,
253     												rectNew.left,rectNew.top,rectNew.Width(),rectNew.Height(),
254     												SWP_NOACTIVATE|SWP_NOOWNERZORDER|SWP_NOZORDER) );
255     			}
256     		}
257     		catch(...)
258     		{
259     			delete pli;
260     			throw;
261     		}
262     	}
263     
264     	delete pli;
265     
266     	return true;
267     }

274     /*
275      * Layout()
276      * --------
277      * Iterates through all child windows and calls DoMoveCtrl() for them.
278      * This function is NOT virtual.
279      * To implement your own layout algorithm, please
280      * a) overwrite DoCreateLayoutInfo() to return an object of a class
281      *    derived from cdxCDynamicLayoutInfo.
282      *    You can put any user-data into your object; it will be passed
283      *    on to the DoMoveCtrl() function.
284      * b) overwrite DoMoveCtrl() and implement the layout logic.
285      *    An example can be found in the example project, anytime.
286      */
287     
288     void cdxCDynamicWnd::Layout()
289     {
290     	if(!IsWindow())
291     	{
292     		ASSERT(IsWindow());
293     		return;
294     	}
295     	
296     	// resize stuff
297     
298     	cdxCDynamicLayoutInfo	*pli		=	DoCreateLayoutInfo();
299     
300     	if(!pli)
301     	{
302     		ASSERT(false);		// YOU MUST PROVIDE A LAYOUT INFO BLOCK !
303     		return;
304     	}
305     
306     	try
307     	{
308     		HDWP							hdwp		=	::BeginDeferWindowPos(pli->m_nCtrlCnt);
309     		HWND							hwnd;
310     		bool							bRepeat;
311     		CRect							rectNew;
312     		UINT							id;
313     		WINDOWPLACEMENT			wpl;
314     		DWORD							swpFlags	=	SWP_NOACTIVATE|SWP_NOOWNERZORDER|SWP_NOZORDER|(!(m_nFlags & flSWPCopyBits) ? SWP_NOCOPYBITS : 0);
315     
316     		if(!( hwnd = ::GetWindow(m_pWnd->m_hWnd,GW_CHILD) ))
317     		{
318     			TRACE(_T("*** NOTE[cdxCDynamicWnd::Layout()]: The window at 0x%lx does not have child windows.\n"),(DWORD)m_pWnd->m_hWnd);
319     			return;
320     		}
321     
322     		do
323     		{
324     			bRepeat				=	false;
325     
326     			for(; hwnd; hwnd = ::GetNextWindow(hwnd,GW_HWNDNEXT))
327     			{
328     				wpl.length	=	sizeof(WINDOWPLACEMENT);
329     
330     				if(!::GetWindowPlacement(hwnd,&wpl))
331     				{
332     					ASSERT(false);		// GetWindowPlacement() failed
333     					continue;
334     				}
335     
336     				rectNew	=	wpl.rcNormalPosition;
337     				ASSERT(rectNew.left >= 0);
338     				id			=	::GetDlgCtrlID(hwnd);
339     
340     				if(!DoMoveCtrl(hwnd,id,rectNew,*pli) ||
341     					(rectNew == wpl.rcNormalPosition) )
342     				{
343     					// window doesn't need to be moved
344     					// (position is not been changed)
345     					continue;
346     				}
347     
348     				if(hdwp)
349     				{
350     					if(!( hdwp = ::DeferWindowPos(hdwp,hwnd,HWND_TOP,
351     															rectNew.left,rectNew.top,rectNew.Width(),rectNew.Height(),
352     															swpFlags) ))
353     					{
354     						TRACE(_T("*** ERROR[cdxCDynamicWnd::ReorganizeControls()]: DeferWindowPos() failed ??\n"));
355     						bRepeat	=	true;
356     						break;		// error; we'll repeat the loop by using SetWind�wPos()
357     										// this won't look good, but work :)
358     					}
359     				}
360     				else
361     				{
362     					VERIFY( ::SetWindowPos(hwnd,HWND_TOP,
363     												rectNew.left,rectNew.top,rectNew.Width(),rectNew.Height(),
364     												swpFlags) );
365     				}
366     			}
367     		}
368     		while(bRepeat);
369     
370     		if(hdwp)
371     		{
372     			VERIFY( ::EndDeferWindowPos(hdwp) );
373     		}
374     	}
375     	catch(...)
376     	{
377     		delete pli;
378     		throw;
379     	}
380     
381     	delete pli;
382     }

388     /*
389      * DoMoveCtrl()
390      * ------------
391      * This virtual function is used to calculate a child window's new position
392      * based on the some data (from the cdxCDynamicLayoutInfo object).
393      * This standard routine is made to implement the algorithm as known from
394      * the cdxCDynamicControlsManager.
395      * You can implement your own code if you are not satisfied with the
396      * following function.
397      * If you need global data, overwrite DoCreateLayoutInfo() which will
398      * be called by Layout() and which you can use to collect these data
399      * once for the entire layout process.
400      *
401      * PARAMETERS:
402      *
403      *		hwnd			-	handle of the child control
404      *		id				-	its id
405      *		rectNewPos	-	write the new position here in.
406      *							initially contains the current position
407      *		li				-	Some information on the parent window.
408      *							You can provide extra information here
409      *							by overwriting DoCreateLayoutInfo().
410      *
411      * RETURN CODES:
412      *
413      * return false if you don't want to move the control
414      * return true if you updated the control's position and stored it into "rectNewPos"
415      * If you don't change it, the control will not be moved.
416      *
417      * #### don't move the control by yourself. Layout() will do for you to ensure
418      *      that as little flickering as possible will occur.
419      */
420     
421     bool cdxCDynamicWnd::DoMoveCtrl(HWND hwnd, UINT id, CRect & rectNewPos, const cdxCDynamicLayoutInfo & li)
422     {
423     	Position	pos;
424     
425     	if(!GetControlPosition(hwnd,pos))
426     		return false;
427     
428     	pos.Apply(hwnd,rectNewPos,li);
429     	return true;
430     }

448     /*
449      * DoInitWindow()
450      * --------------
451      * This function sets up the window pointer.
452      * It is recommended that "rWnd" points to an existing CWnd.
453      * However, it doesn't need to exist as long as you
454      * 1) provide a non-zero "szInitial" object.
455      * 2) don't want a size icon.
456      *
457      *	PARAMETERS:
458      *
459      *		rWnd			-	reference to your window ("*this")
460      *							the window must exist (::IsWindow(rWnd.m_hWnd) must be true)
461      *		fd				-	Freedom (in which direction(s) your window shall be sizable BY THE USER):
462      *							Possible values: fdAll, fdHorz, fdVert and fdNone.
463      *							This is only applied to user-actions; resizing + layout may work
464      *							even if the freedom parameter is fdNone (in that case user cannot resize
465      *							your window, but you can).
466      *		flags			-	several flags:
467      *								flSizeIcon		-	creates a size icon
468      *								flAntiFlicker	-	activates anti-flickering stuff
469      *		[szInitial	-	initial client size]
470      */
471     
472     void cdxCDynamicWnd::DoInitWindow(CWnd & rWnd)
473     {
474     	ASSERT(::IsWindow(rWnd.m_hWnd));	// ensure the window exists ...
475     
476     	m_pWnd		=	&rWnd;
477     	DoInitWindow(rWnd,GetCurrentClientSize());
478     }
479     
480     void cdxCDynamicWnd::DoInitWindow(CWnd & rWnd, const CSize & szInitial)
481     {
482     	ASSERT(::IsWindow(rWnd.m_hWnd) && szInitial.cx && szInitial.cy);	// ensure the window exists ...
483     
484     	m_pWnd		=	&rWnd;
485     	m_szInitial	=	szInitial;
486     	m_szMin		=	szInitial;
487     
488     	/*
489     	 * this window will flicker somewhat deadly if you do not have the
490     	 * WS_CLIPCHILDREN style set for you window.
491     	 * You may like to use the following line anywhere
492     	 * to apply it:
493     	 
494     		  CWnd::ModifyStyle(0,WS_CLIPCHILDREN);
495     
496         */
497     
498     #ifdef _DEBUG
499     	if(!(rWnd.GetStyle() & WS_CLIPCHILDREN) && !(m_nFlags & flSWPCopyBits))
500     	{
501     		TRACE(_T("***\n")
502     					_T("*** cdxCDynamicWnd class note: If your window flickers too much, add the WS_CLIPCHILDREN style to it\n")
503     					_T("***                            or try to set the flSWPCopyBits flags !!!\n")
504     					_T("***\n"));
505     	}
506     #endif
507     
508     	//
509     	// now, if a DYNAMIC MAP is been defined,
510     	// we start working with it
511     	//
512     
513     	const __dynEntry	*pEntry,*pLast	=	NULL;
514     	UINT					nInitCnt	=	GetCtrlCount();
515     
516     	if(pLast = __getDynMap(pLast))
517     	{
518     		HWND		hwnd;
519     		SBYTES	bytes;
520     
521     		for(pEntry = pLast; pEntry->type != __end; ++pEntry)
522     		{
523     			if((pEntry->id != DYNAMIC_MAP_DEFAULT_ID)
524     				&& !( hwnd = ::GetDlgItem(m_pWnd->m_hWnd,pEntry->id) ))
525     			{
526     				TRACE(_T("*** NOTE[cdxCDynamicWnd::DoInitWindow()]: Dynamic map initialization: There's no control with the id 0x%lx !\n"),pEntry->id);
527     				continue;
528     			}
529     
530     			switch(pEntry->type)
531     			{
532     				case	__bytes:
533     
534     					bytes[X1]	=	pEntry->b1;
535     					bytes[Y1]	=	pEntry->b2;
536     					bytes[X2]	=	pEntry->b3;
537     					bytes[Y2]	=	pEntry->b4;
538     					break;
539     
540     				case	__modes:
541     					
542     					_translate((Mode)pEntry->b1,bytes[X1],bytes[X2]);
543     					_translate((Mode)pEntry->b2,bytes[Y1],bytes[Y2]);
544     					break;
545     
546     				default:
547     
548     					ASSERT(false);		// never come here !!!!!
549     					break;
550     			}
551     
552     			if(pEntry->id == DYNAMIC_MAP_DEFAULT_ID)
553     				AllControls(bytes,false,false);
554     			else
555     				AddSzControl(hwnd,bytes,M_szNull,false);
556     		}
557     	}
558     
559     	//
560     	// handle creation flags
561     	//
562     
563     	if(m_nFlags & flSizeIcon)
564     	{
565     		m_pSizeIcon	=	new cdxCSizeIconCtrl;
566     		VERIFY( m_pSizeIcon->Create(m_pWnd) );
567     
568     		AddSzControl(m_pSizeIcon->m_hWnd,BotRight,M_szNull,false);
569     		m_pSizeIcon->ShowWindow(SW_SHOW);
570     	}
571     
572     	m_bIsAntiFlickering	=	false;
573     	m_nMyTimerID			=	DEFAULT_TIMER_ID;
574     	m_dwClassStyle			=	::GetClassLong(*m_pWnd,GCL_STYLE) & (CS_VREDRAW|CS_HREDRAW);
575     
576     	OnInitialized();
577     
578     	if(nInitCnt < GetCtrlCount())
579     		Layout();
580     }

642     /*
643      * StartAntiFlickering()
644      * ---------------------
645      * This routine modifies the CS_VREDRAW and CS_HREDRAW CLASS style
646      * flags.
647      * If you don't like this, set "m_nMyTimerID" to 0.
648      *		bIsBotRight	-	true if the window is sized in right, bottom or bot/right direction.
649      */
650     
651     void cdxCDynamicWnd::StartAntiFlickering(bool bIsBotRight)
652     {
653     	if(IsWindow() && m_nMyTimerID)
654     	{
655     		DWORD	dw	=	m_dwClassStyle;
656     		if(bIsBotRight)
657     			dw	&=	~(CS_VREDRAW|CS_HREDRAW);
658     		else
659     			dw	|=	CS_VREDRAW|CS_HREDRAW;
660     
661     		m_pWnd->KillTimer(m_nMyTimerID);
662     		m_pWnd->SetTimer(m_nMyTimerID,120,NULL);
663     
664     		if(!m_bIsAntiFlickering)
665     		{
666     			::SetClassLong(*m_pWnd,GCL_STYLE,dw);
667     			m_bIsAntiFlickering	=	true;
668     		}
669     	}
670     }

672     /*
673      * DoOnTimer()
674      * -----------
675      * Processes the timer associated to my DoOnSizing() routine.
676      * Changes back the class style.
677      */
678     
679     void cdxCDynamicWnd::DoOnTimer(UINT nIDEvent) 
680     {
681     	if(IsWindow() && (nIDEvent == m_nMyTimerID))
682     	{
683     		m_pWnd->KillTimer(m_nMyTimerID);
684     		if(m_bIsAntiFlickering)
685     		{
686     			::SetClassLong(*m_pWnd,GCL_STYLE,m_dwClassStyle);
687     			m_bIsAntiFlickering	=	false;
688     		}
689     	}
690     }

694     /*
695      * DoOnGetMinMaxInfo()
696      * -------------------
697      * fill in MINMAXINFO as requested
698      * Call your CWnd's OnGetMinMaxInfo first !
699      * [changed due to a bug reported by Michel Wassink <mww@mitutoyo.nl>]
700      */
701     
702     void cdxCDynamicWnd::DoOnGetMinMaxInfo(MINMAXINFO FAR* lpMMI)
703     {
704     	if(IsWindow() && !IsDisabled()  &&  IsUp())
705     	{
706     		CSize	szDelta	=	GetBorderSize();
707     
708     		lpMMI->ptMinTrackSize.x	=	m_szMin.cx + szDelta.cx;
709     		lpMMI->ptMinTrackSize.y	=	m_szMin.cy + szDelta.cy;
710     
711     		if(m_Freedom & fdHoriz)
712     		{
713     			if(m_szMax.cx > 0)
714     				lpMMI->ptMaxTrackSize.x	=	m_szMax.cx + szDelta.cx;
715     		}
716     		else
717     			lpMMI->ptMaxTrackSize.x	=	lpMMI->ptMinTrackSize.x;
718     
719     		if(m_Freedom & fdVert)
720     		{
721     			if(m_szMax.cy > 0)
722     				lpMMI->ptMaxTrackSize.y	=	m_szMax.cy + szDelta.cy;
723     		}
724     		else
725     			lpMMI->ptMaxTrackSize.y	=	lpMMI->ptMinTrackSize.y;
726     	}
727     }
