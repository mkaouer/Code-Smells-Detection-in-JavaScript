13     /*
14      * cdxCDynamicDlgBarT
15      * ==================
16      * A resizable dialog bar.
17      * The entire bar stuff is handled using
18      *   CSizingControlBar by Cristi Posea <cristip@dundas.com>
19      *   http://www.codeguru.com/docking/docking_window2.shtml
20      *   titled "Resizable Docking Window 2".
21      * To use it, the following steps must be performed:
22      *
23      * a) Create a new dialog say
24      *       CMyBarDlg
25      *
26      * b) Change its base class from CDialog to cdxCDynamicBarDlg.
27      *
28      * c) In your mainframe, add a member variable
29      *       cdxCDynamicDlgBarT<MyBarDialog> m_wndMyBar;
30      *
31      * e) Add the following code to your CMainFrame::OnCreate()
32      *
33      *    if (!m_wndMyBar.Create(_T("My Bar"), this, CSize(200, 100),
34      *         TRUE, AFX_IDW_CONTROLBAR_FIRST + 32))
35      *    { 
36      *			 TRACE0("Failed to create mybar\n");
37      *			 return -1;      // fail to create
38      *    }
39      *
40      *		m_wndMyBar.SetBarStyle(m_wndMyBar.GetBarStyle() |
41      *			 CBRS_TOOLTIPS | CBRS_FLYBY | CBRS_SIZE_DYNAMIC);
42      *
43      *		m_wndMyBar.EnableDocking(CBRS_ALIGN_ANY);
44      *		EnableDocking(CBRS_ALIGN_ANY);              // <---- needed only once for the frame
45      *		DockControlBar(&m_wndMyBar, AFX_IDW_DOCKBAR_LEFT);
46      *
47      * f) Refer to URL stated above to learn more about the features of the
48      *    CSizingControlBar class.
49      */
