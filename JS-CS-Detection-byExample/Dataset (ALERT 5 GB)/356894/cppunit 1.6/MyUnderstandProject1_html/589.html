74     
75     		// Center icon in client rectangle
76     		int cxIcon = GetSystemMetrics(SM_CXICON);
77     		int cyIcon = GetSystemMetrics(SM_CYICON);
78     		CRect rect;
79     		GetClientRect(&rect);
80     		int x = (rect.Width() - cxIcon + 1) / 2;
81     		int y = (rect.Height() - cyIcon + 1) / 2;

107     void 
108     TestPlugInRunnerDlg::OnChooseDll() 
109     {
110       CFileDialog dlg( TRUE, "*.dll", "", 0, 
111                        "Test Plug-in (*.dll)|*.dll|All Files (*.*)|*.*||",
112                        this );
113       if ( dlg.DoModal() != IDOK )
114         return;
115     
116       try
117       {
118         TestPlugIn *plugIn = new TestPlugIn( std::string( dlg.GetPathName() ) );
119         plugInModel().setPlugIn( plugIn );
120         updateHistoryCombo();
121       }
122       catch ( TestPlugInException &e )
123       {
124         AfxMessageBox( e.what() );
125       }
126     }
