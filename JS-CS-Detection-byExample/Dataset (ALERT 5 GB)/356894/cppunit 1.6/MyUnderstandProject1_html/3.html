28     /** \file
29      * Macros intended to ease the definition of test suites.
30      *
31      * The macros
32      * CPPUNIT_TEST_SUITE(), CPPUNIT_TEST(), and CPPUNIT_TEST_SUITE_END()
33      * are designed to facilitate easy creation of a test suite.
34      * For example,
35      *
36      * \code
37      * #include <cppunit/extensions/HelperMacros.h>
38      * class MyTest : public CppUnit::TestCase {
39      *   CPPUNIT_TEST_SUITE( MyTest );
40      *   CPPUNIT_TEST( testEquality );
41      *   CPPUNIT_TEST( testSetName );
42      *   CPPUNIT_TEST_SUITE_END();
43      * public:
44      *   void testEquality();
45      *   void testSetName();
46      * };
47      * \endcode
48      * 
49      * The effect of these macros is to define two methods in the
50      * class MyTest.  The first method is an auxiliary function
51      * named registerTests that you will not need to call directly.
52      * The second function
53      * \code static CppUnit::TestSuite *suite()\endcode
54      * returns a pointer to the suite of tests defined by the CPPUNIT_TEST()
55      * macros.  
56      *
57      * Rather than invoking suite() directly,
58      * the macro CPPUNIT_TEST_SUITE_REGISTRATION() is
59      * used to create a static variable that automatically
60      * registers its test suite in a global registry.
61      * The registry yields a Test instance containing all the
62      * registered suites.
63      * \code
64      * CPPUNIT_TEST_SUITE_REGISTRATION( MyTest );
65      * CppUnit::Test* tp =
66      *   CppUnit::TestFactoryRegistry::getRegistry().makeTest();
67      * \endcode
68      * 
69      * The test suite macros can even be used with templated test classes.
70      * For example:
71      *
72      * \code
73      * template<typename CharType>
74      * class StringTest : public CppUnit::Testcase {
75      *   CPPUNIT_TEST_SUITE( StringTest );
76      *   CPPUNIT_TEST( testAppend );
77      *   CPPUNIT_TEST_SUITE_END();
78      * public:  
79      *   ...
80      * };
81      * \endcode
82      *
83      * You need to add in an implementation file:
84      *
85      * \code
86      * CPPUNIT_TEST_SUITE_REGISTRATION( String<char> );
87      * CPPUNIT_TEST_SUITE_REGISTRATION( String<wchar_t> );
88      * \endcode
89      */
90

92     /** Begin test suite
93      *
94      * This macro starts the declaration of a new test suite.
95      * Use CPPUNIT_TEST_SUB_SUITE() instead, if you wish to include the
96      * test suite of the parent class.
97      *
98      * \param ATestCaseType Type of the test case class.
99      * \see CPPUNIT_TEST_SUB_SUITE, CPPUNIT_TEST, CPPUNIT_TEST_SUITE_END, CPPUNIT_TEST_SUITE_REGISTRATION.
100      */
101     #define CPPUNIT_TEST_SUITE( ATestCaseType )                               \
102       private:                                                                \
103         typedef ATestCaseType __ThisTestCaseType;                             \
104       public:                                                                 \
105         static void                                                           \
106         registerTests( CppUnit::TestSuite *suite,                             \
107                        CppUnit::TestFactory *factory )                        \
108         {                                                                     \
109           CppUnit::TestSuiteBuilder<__ThisTestCaseType> builder( suite );
110     
111     
112     /** Begin test suite (includes parent suite)
113      * 
114      * This macro may only be used in a class whose parent class
115      * defines a test suite using CPPUNIT_TEST_SUITE() or CPPUNIT_TEST_SUB_SUITE().
116      *
117      * This macro begins the declaration of a test suite, in the same
118      * manner as CPPUNIT_TEST_SUITE().  In addition, the test suite of the
119      * parent is automatically inserted in the test suite being
120      * defined.
121      * 
122      * Here is an example:
123      *
124      * \code
125      * #include <cppunit/extensions/HelperMacros.h>
126      * class MySubTest : public MyTest {
127      *   CPPUNIT_TEST_SUB_SUITE( MySubTest, MyTest );
128      *   CPPUNIT_TEST( testAdd );
129      *   CPPUNIT_TEST( testSub );
130      *   CPPUNIT_TEST_SUITE_END();
131      * public:
132      *   void testAdd();
133      *   void testSub();
134      * };
135      * \endcode
136      *
137      * \param ATestCaseType Type of the test case class.
138      * \param ASuperClass   Type of the parent class.
139      * \see CPPUNIT_TEST_SUITE.
140      */
141     #define CPPUNIT_TEST_SUB_SUITE( ATestCaseType, ASuperClass )  \
142       private:                                                    \
143         typedef ASuperClass __ThisSuperClassType;                 \
144         CPPUNIT_TEST_SUITE( ATestCaseType );                      \
145           __ThisSuperClassType::registerTests( suite, factory )
146     
147     
148     /** Add a method to the suite.
149      * \param testMethod Name of the method of the test case to add to the
150      *                   suite. The signature of the method must be of
151      *                   type: void testMethod();
152      * \see  CPPUNIT_TEST_SUITE.
153      */
154     #define CPPUNIT_TEST( testMethod )                                      \
155           builder.addTestCaller( #testMethod,                               \
156                                  &__ThisTestCaseType::testMethod ,          \
157                                  (__ThisTestCaseType*)factory->makeTest() ) 
158     
159     
160     /** End declaration of the test suite.
161      *
162      * After this macro, member access is set to "private".
163      *
164      * \see  CPPUNIT_TEST_SUITE.
165      * \see  CPPUNIT_TEST_SUITE_REGISTRATION.
166      */
167     #define CPPUNIT_TEST_SUITE_END()                                        \
168           builder.takeSuite();                                              \
169         }                                                                   \
170         static CppUnit::TestSuite *suite()                                  \
171         {                                                                   \
172           CppUnit::TestSuiteBuilder<__ThisTestCaseType>                     \
173               builder __CPPUNIT_SUITE_CTOR_ARGS( ATestCaseType );           \
174           ThisTestCaseFactory factory;                                      \
175           __ThisTestCaseType::registerTests( builder.suite(), &factory );   \
176           return builder.takeSuite();                                       \
177         }                                                                   \
178       private:                                                              \
179         class ThisTestCaseFactory : public CppUnit::TestFactory             \
180         {                                                                   \
181           virtual CppUnit::Test *makeTest()                                 \
182           {                                                                 \
183             return new __ThisTestCaseType();                                \
184           }                                                                 \
185         }

190     /** Decorates the specified string with the line number to obtain a unique name;
191      * @param str String to decorate.
192      */
193     #define __CPPUNIT_MAKE_UNIQUE_NAME( str ) __CPPUNIT_CONCATENATE( str, __LINE__ )

196     /** Register test suite into global registry.
197      *
198      * This macro declares a static variable whose construction
199      * causes a test suite factory to be inserted in a global registry
200      * of such factories.  The registry is available by calling
201      * the static function CppUnit::TestFactoryRegistry::getRegistry().
202      * 
203      * \param ATestCaseType Type of the test case class.
204      * \warning This macro should be used only once per line of code (the line
205      *          number is used to name a hidden static variable).
206      * \see  CPPUNIT_TEST_SUITE, CppUnit::AutoRegisterSuite.
207      */
208     #define CPPUNIT_TEST_SUITE_REGISTRATION( ATestCaseType )                     \
209       static CppUnit::AutoRegisterSuite< ATestCaseType >                    \
210                  __CPPUNIT_MAKE_UNIQUE_NAME(__autoRegisterSuite )
