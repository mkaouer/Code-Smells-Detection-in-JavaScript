33     /** Runs the named test case.
34      *
35      * \param testName Name of the test case to run. If an empty is given, then
36      *                 all added test are run. The name must be the name of
37      *                 of an added test.
38      * \param doWait if \c true then the user must press the RETURN key 
39      *               before the run() method exit.
40      */
41     
42     void 
43     TextTestRunner::run( std::string testName,
44                          bool doWait )
45     {
46       runTestByName( testName );
47       wait( doWait );
48     }

51     void 
52     TextTestRunner::runTestByName( std::string testName )
53     {
54       if ( testName.empty() )
55       {
56         runTest( m_suite );
57       }
58       else
59       { 
60         Test *test = findTestByName( testName );
61         if ( test != NULL )
62           runTest( test );
63         else
64         {
65           std::cout << "Test " << testName << " not found." << std::endl;
66         }
67       }
68     }

71     void 
72     TextTestRunner::wait( bool doWait )
73     {
74       if ( doWait ) 
75       {
76         std::cout << "<RETURN> to continue" << std::endl;
77         std::cin.get ();
78       }
79     }

82     Test * 
83     TextTestRunner::findTestByName( std::string name ) const
84     {
85       for ( std::vector<Test *>::const_iterator it = m_suite->getTests().begin(); 
86             it != m_suite->getTests().end(); 
87             ++it )
88       {
89         Test *test = *it;
90         if ( test->getName() == name )
91           return test;
92       }
93       return NULL;
94     }

97     void 
98     TextTestRunner::runTest( Test *test )
99     {
100       TextTestResult result;
101       test->run( &result );
102       std::cout << result << std::endl;
103     }
