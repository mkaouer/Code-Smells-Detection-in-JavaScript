48     {
49       History::iterator end = 
50           std::remove( m_history.begin(), m_history.end(), test );

69     {

73       int autorun = app->GetProfileInt( "CppUnit",

77

83     void 
84     TestRunnerModel::loadHistory()
85     {
86       m_history.clear();
87       int idx = 1;
88       do
89       {
90         std::string testName = loadHistoryEntry( idx++ );
91         if ( testName.empty() )
92           break;
93     
94         CppUnit::Test *test = findTestByName( testName );
95         if ( test != NULL )
96           m_history.push_back( test );
97       }
98       while ( true );
99     }

114     void 
115     TestRunnerModel::saveSettings()
116     {
117       CWinApp *app = AfxGetApp();
118       ASSERT( app != NULL );
119     
120       int autorun = m_autorunOnLaunch ? 1 : 0;
121       app->WriteProfileInt( "CppUnit", "AutorunAtStartup", autorun );
122     
123       int idx = 1;
124       for ( History::const_iterator it = m_history.begin(); 
125             it != m_history.end(); 
126             ++it , ++idx )
127       {
128         CppUnit::Test *test = *it;
129         saveHistoryEntry( idx, test->getName() );
130       }
131     }

177     CppUnit::Test * 
178     TestRunnerModel::findTestByNameFor( const std::string &name, 
179                                         CppUnit::Test *test ) const
180     {
181       if ( name == test->getName() )
182         return test;
183     
184       CppUnit::TestSuite *suite = dynamic_cast<CppUnit::TestSuite *>(test);
185       if ( suite == NULL )
186         return NULL;
187     
188       const std::vector<CppUnit::Test *> &tests = suite->getTests();
189       for ( std::vector<CppUnit::Test *>::const_iterator it = tests.begin(); 
190             it != tests.end(); 
191             ++it )
192       {
193         CppUnit::Test *testFound = findTestByNameFor( name, *it );
194         if ( testFound != NULL )
195           return testFound;
196       }
197       return NULL;
198     }
