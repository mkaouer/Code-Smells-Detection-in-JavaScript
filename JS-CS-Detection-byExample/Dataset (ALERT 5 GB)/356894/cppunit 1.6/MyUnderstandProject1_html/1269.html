10       /** Automatically register the test suite of the specified type.
11        *
12        * This object will register the test returned by TestCaseType::suite()
13        * when constructed to the test registry.
14        *
15        * This object is intented to be used as a static variable.
16        *
17        * \param TestCaseType Type of the test case which suite is registered.
18        * \see CPPUNIT_TEST_SUITE_REGISTRATION.
19        */
20       template<typename TestCaseType>
21       class AutoRegisterSuite
22       {
23         public:
24           /** Auto-register the suite factory in the global registry.
25            */
26           AutoRegisterSuite()
27           {
28             TestFactory *factory = new TestSuiteFactory<TestCaseType>();
29             TestFactoryRegistry::getRegistry().registerFactory( factory );
30           }
31     
32           /** Auto-register the suite factory in the specified registry.
33            * \param name Name of the registry.
34            */
35           AutoRegisterSuite( const std::string &name )
36           {
37             TestFactory *factory = new TestSuiteFactory<TestCaseType>();
38             TestFactoryRegistry::getRegistry( name ).registerFactory( factory );
39           }
40       };
