16     // Paint the background of the progress bar region
17     void ProgressBar::paintBackground (CDC& dc)
18     {
19         CBrush      brshBackground;
20         CPen        penGray     (PS_SOLID, 1, RGB (128, 128, 128));
21         CPen        penWhite    (PS_SOLID, 1, RGB (255, 255, 255));
22     
23         VERIFY (brshBackground.CreateSolidBrush (::GetSysColor (COLOR_BTNFACE)));
24     
25         dc.FillRect (m_bounds, &brshBackground);
26         
27         CPen    *pOldPen;
28                 
29         pOldPen = dc.SelectObject (&penGray);
30         {
31             dc.MoveTo (m_bounds.left, m_bounds.top);
32             dc.LineTo (m_bounds.left + m_bounds.Width () -1, m_bounds.top);
33     
34             dc.MoveTo (m_bounds.left, m_bounds.top);
35             dc.LineTo (m_bounds.left, m_bounds.top + m_bounds.Height () -1);
36     
37         }
38         dc.SelectObject (&penWhite);
39         {
40             dc.MoveTo (m_bounds.left + m_bounds.Width () -1, m_bounds.top);
41             dc.LineTo (m_bounds.left + m_bounds.Width () -1, m_bounds.top + m_bounds.Height () -1);
42     
43             dc.MoveTo (m_bounds.left, m_bounds.top + m_bounds.Height () -1);
44             dc.LineTo (m_bounds.left + m_bounds.Width () -1, m_bounds.top + m_bounds.Height () -1);
45     
46         }
47         dc.SelectObject (pOldPen);
48     
49     }

52     // Paint the actual status of the progress bar
53     void ProgressBar::paintStatus (CDC& dc)
54     {
55         if (m_progress <= 0)
56             return;
57     
58         CBrush      brshStatus;
59         CRect       rect (m_bounds.left, m_bounds.top, 
60                         m_bounds.left + m_progressX, m_bounds.bottom);
61     
62         COLORREF    statusColor = getStatusColor ();
63     
64         VERIFY (brshStatus.CreateSolidBrush (statusColor));
65     
66         rect.DeflateRect (1, 1);
67         dc.FillRect (rect, &brshStatus);
68     
69     }

82     // Setup the progress bar for execution over a total number of steps
83     void ProgressBar::start (int total)
84     {
85         m_total = total;
86         reset ();
87     }

90     // Take one step, indicating whether it was a successful step
91     void ProgressBar::step (bool successful)
92     {
93         m_progress++;
94     
95         int x = m_progressX;
96     
97         m_progressX = scale (m_progress);
98     
99         if (!m_error && !successful)
100         {
101             m_error = true;
102             x = 1;
103         }
104     
105         paintStep (x, m_progressX);
106     
107     }

110     // Map from steps to display units
111     int ProgressBar::scale (int value)
112     {
113         if (m_total > 0)
114             return max (1, value * (m_bounds.Width () - 1) / m_total);
115     
116         return value;
117     
118     }

122     void ProgressBar::reset ()
123     {
124         m_progressX     = 1;
125         m_progress      = 0;
126         m_error         = false;

129         m_baseWindow->UpdateWindow ();
