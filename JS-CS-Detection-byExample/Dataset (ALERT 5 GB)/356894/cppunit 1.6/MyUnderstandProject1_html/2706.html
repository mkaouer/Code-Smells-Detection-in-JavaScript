34     TestFactoryRegistry &
35     TestFactoryRegistry::getRegistry( const std::string &name )
36     {
37     // No clean-up at the current time => memory leaks.
38     // Need to find a way to solve the folowing issue:
39     // getRegistry().registryFactory( "Functionnal", 
40     //                                getRegistry( "Functionnal" ) );
41     // => the test factory registry "Functionnal" would be
42     // destroyed twice: once by the map below, once by the getRegistry() factory.
43       static NamedRegistries registries;
44     
45       NamedRegistries::const_iterator foundIt = registries.find( name );
46       if ( foundIt == registries.end() )
47       {
48         TestFactoryRegistry *factory = new TestFactoryRegistry( name );
49         registries.insert( std::make_pair( name, factory ) );
50         return *factory;
51       }
52       return *foundIt->second;
53     }
