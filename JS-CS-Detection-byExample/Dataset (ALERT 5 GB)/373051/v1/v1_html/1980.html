341         private enum TransactionMethod {
342             METHOD_BEGIN,
343             METHOD_COMMIT,
344             METHOD_CANCEL
345         }
346     
347         // Starting a transaction on the wrong thread will fail
348         private boolean transactionMethodWrongThread(final TransactionMethod method) throws InterruptedException,
349                 ExecutionException {
350             if (method != TransactionMethod.METHOD_BEGIN) {
351                 testRealm.beginTransaction();
352                 testRealm.createObject(Dog.class); // FIXME: Empty transactions cannot be cancelled
353             }
354             ExecutorService executorService = Executors.newSingleThreadExecutor();
355             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
356                 @Override
357                 public Boolean call() throws Exception {
358                     try {
359                         switch (method) {
360                             case METHOD_BEGIN:
361                                 testRealm.beginTransaction();
362                                 break;
363                             case METHOD_COMMIT:
364                                 testRealm.commitTransaction();
365                                 break;
366                             case METHOD_CANCEL:
367                                 testRealm.cancelTransaction();
368                                 break;
369                         }
370                         return false;
371                     } catch (IllegalStateException ignored) {
372                         return true;
373                     }
374                 }
375             });
376     
377             boolean result = future.get();
378             if (result && method != TransactionMethod.METHOD_BEGIN) {
379                 testRealm.cancelTransaction();
380             }
381             return result;
382         }

384         public void testTransactionWrongThread() throws ExecutionException, InterruptedException {
385             for (TransactionMethod method : TransactionMethod.values()) {
386                 assertTrue(method.toString(), transactionMethodWrongThread(method));
387             }
388         }

869         private void fileCopy(File src, File dst) throws IOException {
870             FileInputStream inStream = new FileInputStream(src);
871             FileOutputStream outStream = new FileOutputStream(dst);
872             FileChannel inChannel = inStream.getChannel();
873             FileChannel outChannel = outStream.getChannel();
874             inChannel.transferTo(0, inChannel.size(), outChannel);
875             inStream.close();
876             outStream.close();
877         }
