197         public void setDate(long columnIndex, Date date) {
198             if (date == null)
199                 throw new IllegalArgumentException("Null Date is not allowed.");
200             long timestamp = date.getTime() / 1000;
201             if (timestamp >= Integer.MAX_VALUE || timestamp <= Integer.MIN_VALUE) {
202                 throw new IllegalArgumentException("Date/timestamp is outside valid range");
203             }
204             nativeSetDate(nativePtr, columnIndex, timestamp);
205         }

252         @Override
253         protected void finalize() {
254             synchronized (context) {
255                 if (nativePtr != 0) {
256                     context.asyncDisposeRow(nativePtr);
257                     nativePtr = 0; // Set to 0 if finalize is called before close() for some reason
258                 }
259             }
260             nativeClose(nativePtr);
261             nativePtr = 0;
262         }
