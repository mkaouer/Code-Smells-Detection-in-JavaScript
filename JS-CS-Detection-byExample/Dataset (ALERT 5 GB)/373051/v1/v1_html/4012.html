105         private boolean containsDot(String s) {
106             return s.indexOf('.') != -1;
107         }
108     
109         private String[] splitString(String s) {
110             int i, j, n;
111     
112             // count the number of .
113             n = 0;
114             for (i = 0; i < s.length(); i++)
115                 if (s.charAt(i) == '.')
116                     n++;
117     
118             // split at .
119             String[] arr = new String[n+1];
120             i = 0;
121             n = 0;
122             j = s.indexOf('.');
123             while (j != -1) {
124                 arr[n] = s.substring(i, j);
125                 i = j+1;
126                 j = s.indexOf('.', i);
127                 n++;
128             }
129             arr[n] = s.substring(s.lastIndexOf('.')+1);
130     
131             return arr;
132         }

134         // TODO: consider another caching strategy so linked classes are included in the cache.
135         private long[] getColumnIndices(String fieldName, ColumnType fieldType) {
136             Table table = getTable().getTable();
137     
138             if (containsDot(fieldName)) {
139                 String[] names = splitString(fieldName); //fieldName.split("\\.");
140                 long[] columnIndices = new long[names.length];
141                 for (int i = 0; i < names.length-1; i++) {
142                     long index = table.getColumnIndex(names[i]);
143                     if (index < 0) {
144                         throw new IllegalArgumentException("Invalid query: " + names[i] + " does not refer to a class.");
145                     }
146                     ColumnType type = table.getColumnType(index);
147                     if (type == ColumnType.LINK || type == ColumnType.LINK_LIST) {
148                         table = table.getLinkTarget(index);
149                         columnIndices[i] = index;
150                     } else {
151                         throw new IllegalArgumentException("Invalid query: " + names[i] + " does not refer to a class.");
152                     }
153                 }
154                 columnIndices[names.length-1] = table.getColumnIndex(names[names.length-1]);
155                 if (fieldType != table.getColumnType(columnIndices[names.length-1])) {
156                     throw new IllegalArgumentException(String.format("Field '%s': type mismatch.", names[names.length-1]));
157                 }
158                 return columnIndices;
159             } else {
160                 if (columns.get(fieldName) == null) {
161                     throw new IllegalArgumentException(String.format("Field '%s' does not exist.", fieldName));
162                 }
163                 if (fieldType != table.getColumnType(columns.get(fieldName))) {
164                     throw new IllegalArgumentException(String.format("Field '%s': type mismatch.", fieldName));
165                 }
166                 return new long[] {columns.get(fieldName)};
167             }
168         }

1123         /**
1124          * Find all objects that fulfill the query conditions and sorted by specific field name.
1125          *
1126          * @param fieldName the field name to sort by.
1127          * @param sortAscending sort ascending if SORT_ORDER_ASCENDING, sort descending if SORT_ORDER_DESCENDING.
1128          * @return A sorted RealmResults containing the objects.
1129          * @throws java.lang.IllegalArgumentException if field name does not exist.
1130          */
1131         public RealmResults<E> findAll(String fieldName, boolean sortAscending) {
1132             TableView tableView = query.findAll();
1133             TableView.Order order = sortAscending ? TableView.Order.ascending : TableView.Order.descending;
1134             Integer columnIndex = columns.get(fieldName);
1135             if (columnIndex == null || columnIndex < 0) {
1136                 throw new IllegalArgumentException(String.format("Field name '%s' does not exist.", fieldName));
1137             }
1138             tableView.sort(columnIndex, order);
1139             return new RealmResults<E>(realm, tableView, clazz);
1140         }
