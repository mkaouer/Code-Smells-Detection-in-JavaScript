17     package io.realm;
18     
19     import io.realm.annotations.RealmClass;
20     import io.realm.internal.Row;
21     
22     /**
23      * In Realm you define your model classes by sub-classing RealmObject and adding fields to be
24      * persisted. You then create your objects within a Realm, and use your custom subclasses instead
25      * of using the RealmObject class directly.
26      * <br>
27      * An annotation processor will create a proxy class for your RealmObject subclass. The getters and
28      * setters should not contain any custom code of logic as they are overridden as part of the annotation
29      * process.
30      * <br>
31      * @see Realm#createObject(Class)
32      */
33     
34     @RealmClass
35     public abstract class RealmObject {
36     
37         protected Row row;
38         protected Realm realm;
39     
40         /**
41          * Removes the object from the Realm it is currently associated to.
42          *
43          * After this method is called the object will be invalid and any operation (read or write)
44          * performed on it will fail with an IllegalStateException
45          */
46         public void removeFromRealm() {
47             if (row == null) {
48                 throw new IllegalStateException("Object malformed: missing object in Realm. Make sure to instantiate RealmObjects with Realm.createObject()");
49             }
50             if (realm == null) {
51                 throw new IllegalStateException("Object malformed: missing Realm. Make sure to instantiate RealmObjects with Realm.createObject()");
52             }
53             row.getTable().moveLastOver(row.getIndex());
54         }
55     }
