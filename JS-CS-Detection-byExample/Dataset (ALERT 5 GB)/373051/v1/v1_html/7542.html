17     package io.realm.processor;
18     
19     import com.squareup.javawriter.JavaWriter;
20     
21     import javax.annotation.processing.ProcessingEnvironment;
22     import javax.lang.model.element.Modifier;
23     import javax.tools.JavaFileObject;
24     import java.io.BufferedWriter;
25     import java.io.IOException;
26     import java.util.*;
27     
28     public class RealmValidationListGenerator {
29         private ProcessingEnvironment processingEnvironment;
30         private Set<String> classesToValidate = new HashSet<String>();
31     
32         private static final String REALM_PACKAGE_NAME = "io.realm";
33         private static final String CLASS_NAME = "ValidationList";
34         private static final String PROXY_CLASS_SUFFIX = "RealmProxy";
35     
36     
37         public RealmValidationListGenerator(ProcessingEnvironment processingEnvironment, Set<String> classesToValidate) {
38             this.processingEnvironment = processingEnvironment;
39             this.classesToValidate = classesToValidate;
40         }
41     
42         public void generate() throws IOException {
43             String qualifiedGeneratedClassName = String.format("%s.%s", REALM_PACKAGE_NAME, CLASS_NAME);
44             JavaFileObject sourceFile = processingEnvironment.getFiler().createSourceFile(qualifiedGeneratedClassName);
45             JavaWriter writer = new JavaWriter(new BufferedWriter(sourceFile.openWriter()));
46             writer.setIndent("    ");
47     
48             writer.emitPackage(REALM_PACKAGE_NAME);
49             writer.emitEmptyLine();
50     
51             writer.emitImports("java.util.Arrays", "java.util.List");
52             writer.emitEmptyLine();
53     
54             // Begin the class definition
55             writer.beginType(
56                     qualifiedGeneratedClassName, // full qualified name of the item to generate
57                     "class",                     // the type of the item
58                     EnumSet.of(Modifier.PUBLIC), // modifiers to apply
59                     null);                       // class to extend
60             writer.emitEmptyLine();
61     
62             writer.beginMethod("List<String>", "getProxyClasses", EnumSet.of(Modifier.PUBLIC, Modifier.STATIC));
63             List<String> entries = new ArrayList<String>();
64             for (String classToValidate : classesToValidate) {
65                 entries.add(String.format("\"%s\"", classToValidate));
66             }
67             String statementSection = joinStringList(entries, ", ");
68             writer.emitStatement("return Arrays.asList(%s)", statementSection);
69             writer.endMethod();
70             writer.emitEmptyLine();
71     
72             writer.endType();
73             writer.close();
74         }
75     
76         public static String joinStringList(List<String> strings, String separator) {
77             StringBuilder stringBuilder = new StringBuilder();
78             ListIterator<String> iterator = strings.listIterator();
79             while (iterator.hasNext()) {
80                 int index = iterator.nextIndex();
81                 String item = iterator.next();
82     
83                 if (index > 0) {
84                     stringBuilder.append(separator);
85                 }
86                 stringBuilder.append(item);
87             }
88             return stringBuilder.toString();
89         }
