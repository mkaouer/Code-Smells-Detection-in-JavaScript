73         // Get the application's 256-bit AES key
74         private byte[] getKey() throws GeneralSecurityException, IOException, java.io.IOException {
75             // As of 4.3, Android has a secure per-application key store, but it can't store symmetric keys
76             // As a result, we use it to store a public/private keypair which is used to encrypt the
77             // symmetric key which is stored in a file in the application context
78     
79             byte[] keyData;
80             try {
81                 File file = new File(getFilesDir(), Realm.DEFAULT_REALM_NAME + ".key");
82                 keyData = new byte[256];
83                 FileInputStream stream = new FileInputStream(file);
84                 try {
85                     int read = stream.read(keyData);
86                     if (read != keyData.length) {
87                         keyData = null;
88                     }
89                 }
90                 finally {
91                     stream.close();
92                 }
93             } catch (java.io.IOException e) {
94                 // Generate a new key if reading the existing one failed for any reason
95                 keyData = null;
96             }
97     
98             KeyPair keyPair = getKeyPair();
99             final Cipher cipher = Cipher.getInstance("RSA/NONE/PKCS1Padding");
100     
101             // We have an existing secret key, so decrypt and return it
102             if (keyData != null) {
103                 cipher.init(Cipher.UNWRAP_MODE, keyPair.getPrivate());
104                 return cipher.unwrap(keyData, "AES", Cipher.SECRET_KEY).getEncoded();
105             }
106     
107             // We need to generate a new secret key
108             keyData = new byte[32];
109             new SecureRandom().nextBytes(keyData);
110     
111             cipher.init(Cipher.WRAP_MODE, keyPair.getPublic());
112     
113             // Save the secret key to the file
114             File file = new File(getFilesDir(), Realm.DEFAULT_REALM_NAME + ".key");
115             FileOutputStream stream = new FileOutputStream(file);
116             try {
117                 stream.write(cipher.wrap(new SecretKeySpec(keyData, "AES")));
118             } finally {
119                 stream.close();
120             }
121     
122             // Delete any existing default Realm since we won't be able to open it with the new key
123             Realm.deleteRealmFile(this);
124     
125             return keyData;
126         }

128         // Get the keypair from the keystore if it exists, or create a new one if not
129         private KeyPair getKeyPair() throws GeneralSecurityException, IOException, java.io.IOException {
130             final KeyStore keyStore = KeyStore.getInstance("AndroidKeyStore");
131             keyStore.load(null);
132     
133             if (!keyStore.containsAlias(TAG)) {
134                 generateKeyPair(TAG);
135             }
136     
137             final KeyStore.PrivateKeyEntry entry = (KeyStore.PrivateKeyEntry)keyStore.getEntry(TAG, null);
138             return new KeyPair(entry.getCertificate().getPublicKey(), entry.getPrivateKey());
139         }
140     
141         // Generate a new long-lived keypair
142         private void generateKeyPair(String alias) throws GeneralSecurityException {
143             final Calendar start = new GregorianCalendar();
144             final Calendar end = new GregorianCalendar();
145             end.add(Calendar.YEAR, 100);
146             final KeyPairGeneratorSpec spec = new KeyPairGeneratorSpec.Builder(this)
147                 .setAlias(alias)
148                 .setSubject(new X500Principal("CN=" + alias))
149                 .setSerialNumber(BigInteger.ONE)
150                 .setStartDate(start.getTime())
151                 .setEndDate(end.getTime())
152                 .build();
153     
154             final KeyPairGenerator gen = KeyPairGenerator.getInstance("RSA", "AndroidKeyStore");
155             gen.initialize(spec);
156             gen.generateKeyPair();
157         }
