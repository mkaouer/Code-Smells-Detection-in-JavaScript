21     template <class T>
22     jbyteArray tbl_GetByteArray(JNIEnv* env, jlong nativeTablePtr, jlong columnIndex, jlong rowIndex)
23     {
24         if (!TBL_AND_INDEX_VALID(env, reinterpret_cast<T*>(nativeTablePtr), columnIndex, rowIndex))
25             return NULL;
26     
27         BinaryData bin = reinterpret_cast<T*>(nativeTablePtr)->get_binary( S(columnIndex), S(rowIndex));
28         if (bin.size() <= MAX_JSIZE) {
29             jbyteArray jresult = env->NewByteArray(static_cast<jsize>(bin.size()));
30             if (jresult)
31                 env->SetByteArrayRegion(jresult, 0, static_cast<jsize>(bin.size()), reinterpret_cast<const jbyte*>(bin.data()));  // throws
32             return jresult;
33         }
34         else {
35             ThrowException(env, IllegalArgument, "Length of ByteArray is larger than an Int.");
36             return NULL;
37         }
38     }

40     template <class M, class T>
41     void tbl_nativeDoByteArray(M doBinary, T* pTable, JNIEnv* env, jlong columnIndex, jlong rowIndex, jbyteArray dataArray)
42     {
43         jbyte* bytePtr = env->GetByteArrayElements(dataArray, NULL);
44         if (!bytePtr) {
45             ThrowException(env, IllegalArgument, "doByteArray");
46             return;
47         }
48         size_t dataLen = S(env->GetArrayLength(dataArray));
49         (pTable->*doBinary)( S(columnIndex), S(rowIndex), BinaryData(reinterpret_cast<char*>(bytePtr), dataLen));
50         env->ReleaseByteArrayElements(dataArray, bytePtr, 0);
51     }

54     template <class M, class T>
55     void tbl_nativeDoBinary(M doBinary, T* pTable, JNIEnv* env, jlong columnIndex, jlong rowIndex, jobject byteBuffer)
56     {
57         BinaryData bin;
58         if (GetBinaryData(env, byteBuffer, bin))
59             (pTable->*doBinary)( S(columnIndex), S(rowIndex), bin);
60     }

63     // insertMixed() or setMixed() value for TableView or Table class

65     template <class M, class T>
66     void tbl_nativeDoMixed(M doMixed, T* pTable, JNIEnv* env, jlong columnIndex, jlong rowIndex, jobject jMixedValue)
67     {
68         DataType valueType = GetMixedObjectType(env, jMixedValue);
69         switch(valueType) {
70         case type_Int:
71             {
72                 jlong longValue = GetMixedIntValue(env, jMixedValue);
73                 (pTable->*doMixed)( S(columnIndex), S(rowIndex), Mixed(static_cast<int64_t>(longValue)));
74                 return;
75             }
76         case type_Float:
77             {
78                 jfloat floatValue = GetMixedFloatValue(env, jMixedValue);
79                 (pTable->*doMixed)( S(columnIndex), S(rowIndex), Mixed(floatValue));
80                 return;
81             }
82         case type_Double:
83             {
84                 jdouble doubleValue = GetMixedDoubleValue(env, jMixedValue);
85                 (pTable->*doMixed)( S(columnIndex), S(rowIndex), Mixed(doubleValue));
86                 return;
87             }
88         case type_Bool:
89             {
90                 jboolean boolValue = GetMixedBooleanValue(env, jMixedValue);
91                 (pTable->*doMixed)( S(columnIndex), S(rowIndex), Mixed(boolValue != 0 ? true : false));
92                 return;
93             }
94         case type_String:
95             {
96                 jstring stringValue = GetMixedStringValue(env, jMixedValue);
97                 JStringAccessor string(env, stringValue); // throws
98                 (pTable->*doMixed)( S(columnIndex), S(rowIndex), StringData(string));
99                 return;
100             }
101         case type_DateTime:
102             {
103                 jlong dateTimeValue = GetMixedDateTimeValue(env, jMixedValue);
104                 DateTime date(dateTimeValue);
105                 (pTable->*doMixed)( S(columnIndex), S(rowIndex), Mixed(date));
106                 return;
107             }
108         case type_Binary:
109             {
110                 jint mixedBinaryType = GetMixedBinaryType(env, jMixedValue);
111                 if (mixedBinaryType == 0) {
112                     jbyteArray dataArray = GetMixedByteArrayValue(env, jMixedValue);
113                     if (!dataArray)
114                         break;
115                     char* data = reinterpret_cast<char*>(env->GetByteArrayElements(dataArray, NULL));
116                     if (!data)
117                         break;
118                     size_t size = S(env->GetArrayLength(dataArray));
119                     (pTable->*doMixed)( S(columnIndex), S(rowIndex), BinaryData(data, size));
120                     env->ReleaseByteArrayElements(dataArray, reinterpret_cast<jbyte*>(data), 0);
121                     return;
122                 }
123                 else if (mixedBinaryType == 1) {
124                     jobject jByteBuffer = GetMixedByteBufferValue(env, jMixedValue);
125                     if (!jByteBuffer)
126                         break;
127                     BinaryData binaryData;
128                     if (GetBinaryData(env, jByteBuffer, binaryData))
129                         (pTable->*doMixed)( S(columnIndex), S(rowIndex), binaryData);
130                     return;
131                 }
132                 break; // failed
133             }
134         case type_Table:
135             {
136                 (pTable->*doMixed)( S(columnIndex), S(rowIndex), Mixed::subtable_tag());
137                 return;
138             }
139         case type_Mixed:
140             break;
141         case type_Link:
142             break;
143         case type_LinkList:
144             break;
145         }
146         TR_ERR((env, "\nERROR: nativeSetMixed() failed.\n"));
147         ThrowException(env, IllegalArgument, "nativeSetMixed()");
148     }

150     template <class R>
151     void row_nativeSetMixed(R* pRow, JNIEnv* env, jlong columnIndex, jobject jMixedValue)
152     {
153         DataType valueType = GetMixedObjectType(env, jMixedValue);
154         switch(valueType) {
155         case type_Int:
156             {
157                 jlong longValue = GetMixedIntValue(env, jMixedValue);
158                 pRow->set_mixed( S(columnIndex), Mixed(static_cast<int64_t>(longValue)));
159                 return;
160             }
161         case type_Float:
162             {
163                 jfloat floatValue = GetMixedFloatValue(env, jMixedValue);
164                 pRow->set_mixed( S(columnIndex), Mixed(floatValue));
165                 return;
166             }
167         case type_Double:
168             {
169                 jdouble doubleValue = GetMixedDoubleValue(env, jMixedValue);
170                 pRow->set_mixed( S(columnIndex), Mixed(doubleValue));
171                 return;
172             }
173         case type_Bool:
174             {
175                 jboolean boolValue = GetMixedBooleanValue(env, jMixedValue);
176                 pRow->set_mixed( S(columnIndex), Mixed(boolValue != 0 ? true : false));
177                 return;
178             }
179         case type_String:
180             {
181                 jstring stringValue = GetMixedStringValue(env, jMixedValue);
182                 JStringAccessor string(env, stringValue); // throws
183                 pRow->set_mixed( S(columnIndex), StringData(string));
184                 return;
185             }
186         case type_DateTime:
187             {
188                 jlong dateTimeValue = GetMixedDateTimeValue(env, jMixedValue);
189                 DateTime date(dateTimeValue);
190                 pRow->set_mixed( S(columnIndex), Mixed(date));
191                 return;
192             }
193         case type_Binary:
194             {
195                 jint mixedBinaryType = GetMixedBinaryType(env, jMixedValue);
196                 if (mixedBinaryType == 0) {
197                     jbyteArray dataArray = GetMixedByteArrayValue(env, jMixedValue);
198                     if (!dataArray)
199                         break;
200                     char* data = reinterpret_cast<char*>(env->GetByteArrayElements(dataArray, NULL));
201                     if (!data)
202                         break;
203                     size_t size = S(env->GetArrayLength(dataArray));
204                     pRow->set_mixed( S(columnIndex), BinaryData(data, size));
205                     env->ReleaseByteArrayElements(dataArray, reinterpret_cast<jbyte*>(data), 0);
206                     return;
207                 }
208                 else if (mixedBinaryType == 1) {
209                     jobject jByteBuffer = GetMixedByteBufferValue(env, jMixedValue);
210                     if (!jByteBuffer)
211                         break;
212                     BinaryData binaryData;
213                     if (GetBinaryData(env, jByteBuffer, binaryData))
214                         pRow->set_mixed( S(columnIndex), binaryData);
215                     return;
216                 }
217                 break; // failed
218             }
219         case type_Table:
220             {
221                 pRow->set_mixed( S(columnIndex), Mixed::subtable_tag());
222                 return;
223             }
224         case type_Mixed:
225             break;
226         case type_Link:
227             break;
228         case type_LinkList:
229             break;
230         }
231         TR_ERR((env, "\nERROR: nativeSetMixed() failed.\n"));
232         ThrowException(env, IllegalArgument, "nativeSetMixed()");
233     }
