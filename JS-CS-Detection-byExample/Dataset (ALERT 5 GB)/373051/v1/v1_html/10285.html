24     /**
25      * This class represents a view of a particular table. We can think of
26      * a tableview is a subset of a table. It contains less than or
27      * equal to the number of entries of a table. A table view is often a
28      * result of a query.
29      *
30      * The view don't copy data from the table, but merely has a list of
31      * row-references into the original table with the real data.
32      *
33      * The class serves as a base class of all table view. It is a raw level table view, users
34      * are not encouraged to use this class. Instead users are advised to use
35      * the generated subclass version of the table.
36      *
37      * Let us assume we are going to keep track of a table to store the employees
38      * of a company or an organization.
39      *
40      * Following is a small example how to use the autogenerated class of the
41      * tableview. For that purpose we will first define the spec of an employee
42      * entity
43      *
44      *  @DefineTable
45      *  public class employee {
46      *      String name;
47      *      long age;
48      *      boolean hired;
49      *      byte[] imageData;
50      *  }
51      *
52      * Once this class is compiled along with TightDB annotation processor
53      * this will produce following classes.
54      *
55      * 1. Employee
56      * 2. EmployeeTable
57      * 3. EmployeeView.
58      *
59      * In this class context our interest is EmployeeView which will be inherited from
60      * this class.
61      *
62      * The generated class will have more specialized method to do operations on TableView.
63      *
64      */
65     public class TableView implements TableOrView, Closeable {
66         protected boolean DEBUG = false; //true;

73          * @param parent A table.
74          * @param nativePtr pointer to table.
75          */

77             this.context = context;

87         @Override
88         public void close(){
89             synchronized (context) {
90                 if (nativePtr != 0) {
91                     nativeClose(nativePtr);
92                     
93                     if (DEBUG) 
94                         System.err.println("==== TableView CLOSE, ptr= " + nativePtr);
95                   
96                     nativePtr = 0;
97                 } 
98             }
99         }
100         
101         protected void finalize() {
102             synchronized (context) {
103                 if (nativePtr != 0) {
104                     context.asyncDisposeTableView(nativePtr);
105                     nativePtr = 0; // Set to 0 if finalize is called before close() for some reason
106                 }
107             }
108         }
