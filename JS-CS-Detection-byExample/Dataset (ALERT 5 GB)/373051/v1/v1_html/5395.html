101         public boolean methodWrongThread(final Method method) throws ExecutionException, InterruptedException {
102             final RealmResults<AllTypes> allTypeses = testRealm.where(AllTypes.class).findAll();
103             ExecutorService executorService = Executors.newSingleThreadExecutor();
104             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
105                 @Override
106                 public Boolean call() throws Exception {
107                     try {
108                         switch (method) {
109                             case METHOD_MIN:
110                                 allTypeses.min(FIELD_FLOAT);
111                                 break;
112                             case METHOD_MAX:
113                                 allTypeses.max(FIELD_FLOAT);
114                                 break;
115                             case METHOD_SUM:
116                                 allTypeses.sum(FIELD_FLOAT);
117                                 break;
118                             case METHOD_AVG:
119                                 allTypeses.average(FIELD_FLOAT);
120                                 break;
121                             case METHOD_SORT:
122                                 allTypeses.sort(FIELD_FLOAT);
123                                 break;
124                             case METHOD_WHERE:
125                                 allTypeses.where();
126                         }
127                         return false;
128                     } catch (IllegalStateException ignored) {
129                         return true;
130                     }
131                 }
132             });
133             return future.get();
134         }

218         public void testAvgGivesCorrectValue() {
219             RealmResults<AllTypes> resultList = testRealm.where(AllTypes.class).findAll();
220             double N = (double) TEST_DATA_SIZE;
221     
222             // Sum of numbers 1 to M: M*(M+1)/2
223             // See setUp() for values of fields
224             // N = TEST_DATA_SIZE
225     
226             // Type: double; a = 3.1415
227             // a, a+1, ..., a+i, ..., a+N-1
228             // sum = 3.1415*N + N*(N-1)/2
229             // average = sum/N = 3.1415+(N-1)/2
230             double average = 3.1415 + (N - 1.0) * 0.5;
231             assertEquals(average, resultList.average(FIELD_DOUBLE), 0.0001);
232     
233             // Type: long
234             // 0, 1, ..., N-1
235             // sum = N*(N-1)/2
236             // average = sum/N = (N-1)/2
237             assertEquals(0.5 * (N - 1), resultList.average(FIELD_LONG), 0.0001);
238     
239             // Type: float; b = 1.234567
240             // b, b+1, ..., b+i, ..., b+N-1
241             // sum = b*N + N*(N-1)/2
242             // average = sum/N = b + (N-1)/2
243             assertEquals(1.234567 + 0.5 * (N - 1.0), resultList.average(FIELD_FLOAT), 0.0001);
244         }
