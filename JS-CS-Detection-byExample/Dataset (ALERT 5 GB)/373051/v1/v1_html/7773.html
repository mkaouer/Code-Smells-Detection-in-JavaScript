17     package io.realm.internal;
18     
19     import java.nio.ByteBuffer;
20     
21     // Make sure numbers match with <realm/column_type.hpp>
22     // FIXME: Add a unit test that verifies the correct correspondence.
23     
24     public enum ColumnType {
25         BOOLEAN(1),
26         INTEGER(0),
27         FLOAT(9),
28         DOUBLE(10),
29         STRING(2),
30         BINARY(4),
31         DATE(7),
32         TABLE(5),
33         MIXED(6),
34         LINK(12),
35         LINK_LIST(13);
36         // When adding above, remember to update size of largest number below
37     
38         private final int nativeValue;
39     
40         // Note that if this array is too small, an
41         // IndexOutOfBoundsException will be thrown during class loading.
42         private static ColumnType[] byNativeValue = new ColumnType[14];
43     
44         static {
45             ColumnType[] columnTypes = values();
46             for (int i=0; i<columnTypes.length; ++i) {
47                 int v = columnTypes[i].nativeValue;
48                 byNativeValue[v] = columnTypes[i];
49             }
50         }
51     
52         private ColumnType(int nativeValue)
53         {
54             this.nativeValue = nativeValue;
55         }
56     
57         public int getValue() {
58             return nativeValue;
59         }
60     
61         public boolean matchObject(Object obj) {
62             switch (this.nativeValue) {
63             case 0: return (obj instanceof Long || obj instanceof Integer || obj instanceof Short ||
64                     obj instanceof Byte);
65             case 1: return (obj instanceof Boolean);
66             case 2: return (obj instanceof String);
67             case 4: return (obj instanceof byte[] || obj instanceof ByteBuffer);
68             case 5: return (obj == null || obj instanceof Object[][]);
69             case 6: return (obj instanceof Mixed ||
70                     obj instanceof Long || obj instanceof Integer ||
71                     obj instanceof Short || obj instanceof Byte || obj instanceof Boolean ||
72                     obj instanceof Float || obj instanceof Double ||
73                     obj instanceof String ||
74                     obj instanceof byte[] || obj instanceof ByteBuffer ||
75                     obj == null || obj instanceof Object[][] ||
76                     obj instanceof java.util.Date);
77             case 7: return (obj instanceof java.util.Date);
78             case 9: return (obj instanceof Float);
79             case 10: return (obj instanceof Double);
80             default: throw new RuntimeException(String.format("Invalid index (%d) in ColumnType.", this.nativeValue));
81             }
82         }
83     
84         static ColumnType fromNativeValue(int value)
85         {
86             if (0 <= value && value < byNativeValue.length) {
87                 ColumnType e = byNativeValue[value];
88                 if (e != null)
89                     return e;
90             }
91             throw new IllegalArgumentException("Invalid native column type");
92         }
93     }
