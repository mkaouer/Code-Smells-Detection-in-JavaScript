31     /**
32      * A RealmResults list contains a list of objects of a given type that matches the query.
33      * The objects are not copied from the Realm to the RealmResults list, but just references the original objects.
34      * This preserves memory and increase speed.
35      * It also implies that any modification to any object in a RealmResults is reflected in the objects in the 
36      * Realm that was queried.
37      * Updates to objects must be done within a transaction and the modified object is persisted to the backing
38      * Realm file during the commit of the transaction.
39      *
40      * @param <E> The class of objects in this list
41      * @see RealmQuery#findAll()
42      * @see Realm#allObjects(Class)
43      */
44     public class RealmResults<E extends RealmObject> extends AbstractList<E> {
45     
46         private Class<E> classSpec;
47         private Realm realm;
48         private TableOrView table = null;
49     
50         public static final boolean SORT_ORDER_ASCENDING = true;
51         public static final boolean SORT_ORDER_DESCENDING = false;
52     
53         private static final String TYPE_MISMATCH = "Field '%s': type mismatch - %s expected.";
54         private long currentTableViewVersion = -1;
55     
56         RealmResults(Realm realm, Class<E> classSpec) {
57             this.realm = realm;
58             this.classSpec = classSpec;
59         }

62             this(realm, classSpec);

75             }

92          */
93         @Override
94         public E get(int rowIndex) {
95             E obj;
96             realm.checkIfValid();
97             TableOrView table = getTable();
98             if (table instanceof TableView) {
99                 obj = realm.get(classSpec, ((TableView)table).getSourceRowIndex(rowIndex));
100             } else {
101                 obj = realm.get(classSpec, rowIndex);
102             }
103     
104             return obj;
105         }

163         /**
164          * Sort (ascending) an existing @{link io.realm.RealmList}.
165          * 
166          * @param fieldName  The field name to sort by. Only fields of type boolean, short, int, long,
167          *                   float, double, Date, and String are supported.
168          * @throws java.lang.IllegalArgumentException if field name does not exist.
169          */
170         public void sort(String fieldName) {
171             this.sort(fieldName, SORT_ORDER_ASCENDING);
172         }
173     
174         /**
175          * Sort existing @{link io.realm.RealmList}.
176          *
177          * @param fieldName      The field name to sort by. Only fields of type boolean, short, int,
178          *                       long, float, double, Date, and String are supported.
179          * @param sortAscending  The direction to sort by; if true ascending, otherwise descending
180          *                       You can use the constants SORT_ORDER_ASCENDING and SORT_ORDER_DESCENDING
181          *                       for readability.
182          * @throws java.lang.IllegalArgumentException if field name does not exist.
183          */
184         public void sort(String fieldName, boolean sortAscending) {
185             realm.checkIfValid();
186             TableOrView table = getTable();
187     
188             if (table instanceof TableView) {
189                 long columnIndex = table.getColumnIndex(fieldName);
190                 if (columnIndex < 0) {
191                     throw new IllegalArgumentException(String.format("Field '%s' does not exist.", fieldName));
192                 }
193                 TableView.Order TVOrder = sortAscending ? TableView.Order.ascending : TableView.Order.descending;
194                 ((TableView) table).sort(columnIndex, TVOrder);
195             }
196             else {
197                 throw new IllegalArgumentException("Only RealmResults can be sorted - please use allObject() to create a RealmResults.");
198             }
199         }

213          * Find the minimum value of a field.
214          *
215          * @param fieldName   The field to look for a minimum on. Only int, float, and double
216          *                    are supported.
217          * @return            The minimum value.
218          * @throws            java.lang.IllegalArgumentException if field is not int, float or double.
219          */
220         public Number min(String fieldName) {
221             realm.checkIfValid();
222             long columnIndex = table.getColumnIndex(fieldName);
223             switch (table.getColumnType(columnIndex)) {
224                 case INTEGER:
225                     return table.minimumLong(columnIndex);
226                 case FLOAT:
227                     return table.minimumFloat(columnIndex);
228                 case DOUBLE:
229                     return table.minimumDouble(columnIndex);
230                 default:
231                     throw new IllegalArgumentException(String.format(TYPE_MISMATCH, fieldName, "int, float or double"));
232             }
233         }

402         private void assertRealmIsStable() {
403             long version = table.sync();
404             if (currentTableViewVersion > -1 && version != currentTableViewVersion) {
405                 throw new ConcurrentModificationException("No outside changes to a Realm is allowed while iterating a RealmResults. Use iterators methods instead.");
406             }
407     
408             currentTableViewVersion = version;
409         }

411         // Custom RealmResults iterator. It ensures that we only iterate on a Realm that hasn't changed.
412         private class RealmResultsIterator implements Iterator<E> {
413     
414             int pos = -1;
415             boolean removeUsed = false;
416     
417             RealmResultsIterator() {
418                 currentTableViewVersion = table.sync();
419             }
420     
421             public boolean hasNext() {
422                 assertRealmIsStable();
423                 return pos + 1 < size();
424             }
425     
426             public E next() {
427                 assertRealmIsStable();
428                 pos++;
429                 removeUsed = false;
430                 if (pos >= size()) {
431                     throw new IndexOutOfBoundsException("Cannot access index " + pos + " when size is " + size() +  ". Remember to check hasNext() before using next().");
432                 }
433                 return get(pos);
434             }
435     
436             public void remove() {
437                 assertRealmIsStable();
438                 if (pos == -1) {
439                     throw new IllegalStateException("Must call next() before calling remove()");
440                 }
441                 if (removeUsed) {
442                     throw new IllegalStateException("Cannot call remove() twice. Must call next() in between");
443                 }
444     
445                 RealmResults.this.remove(pos);
446                 pos--;
447                 removeUsed = true;
448                 currentTableViewVersion = getTable().sync();
449             }
450         }

452         // Custom RealmResults list iterator. It ensures that we only iterate on a Realm that hasn't changed.
453         private class RealmResultsListIterator extends RealmResultsIterator implements ListIterator<E> {
454     
455             RealmResultsListIterator(int start) {
456                 if (start >= 0 && start <= size()) {
457                     pos = start - 1;
458                 } else {
459                     throw new IndexOutOfBoundsException("Starting location must be a valid index: [0, " + (size() - 1) + "]. Yours was " + start);
460                 }
461             }
462     
463             @Override
464             public void add(E object) {
465                 throw new RealmException("Adding elements not supported. Use Realm.createObject() instead.");
466             }
467     
468             @Override
469             public boolean hasPrevious() {
470                 assertRealmIsStable();
471                 return pos > 0;
472             }
473     
474             @Override
475             public int nextIndex() {
476                 assertRealmIsStable();
477                 return pos + 1;
478             }
479     
480             @Override
481             public E previous() {
482                 assertRealmIsStable();
483                 pos--;
484                 if (pos < 0) {
485                     throw new IndexOutOfBoundsException("Cannot access index less than zero. This was " + pos + ". Remember to check hasPrevious() before using previous().");
486                 }
487                 return get(pos);
488             }
489     
490             @Override
491             public int previousIndex() {
492                 assertRealmIsStable();
493                 return pos;
494             }
495     
496             @Override
497             public void set(E object) {
498                 throw new RealmException("Replacing elements not supported.");
499             }
500         }
