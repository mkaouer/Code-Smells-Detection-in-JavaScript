17     package io.realm.examples.realmadapters;
18     
19     import android.content.Context;
20     import android.view.View;
21     import android.view.ViewGroup;
22     import android.widget.ListAdapter;
23     import android.widget.TextView;
24     
25     import io.realm.RealmBaseAdapter;
26     import io.realm.RealmResults;
27     import io.realm.examples.realmadapters.models.TimeStamp;
28     
29     public class MyAdapter extends RealmBaseAdapter<TimeStamp> implements ListAdapter {
30     
31         private static class ViewHolder {
32             TextView timestamp;
33         }
34     
35         public MyAdapter(Context context, int resId, RealmResults<TimeStamp> realmResults, boolean automaticUpdate) {
36             super(context, realmResults, automaticUpdate);
37         }
38     
39         @Override
40         public View getView(int position, View convertView, ViewGroup parent) {
41             ViewHolder viewHolder;
42             if (convertView == null) {
43                 convertView = inflater.inflate(android.R.layout.simple_list_item_1, parent, false);
44                 viewHolder = new ViewHolder();
45                 viewHolder.timestamp = (TextView) convertView.findViewById(android.R.id.text1);
46                 convertView.setTag(viewHolder);
47             } else {
48                 viewHolder = (ViewHolder) convertView.getTag();
49             }
50     
51             TimeStamp item = realmResults.get(position);
52             viewHolder.timestamp.setText(item.getTimeStamp());
53             return convertView;
54         }
55     
56         public RealmResults<TimeStamp> getRealmResults() {
57             return realmResults;
58         }
59     }
