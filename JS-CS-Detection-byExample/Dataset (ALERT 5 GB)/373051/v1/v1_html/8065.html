17     package io.realm.internal;
18     
19     
20     public class SubtableSchema implements TableSchema {
21     
22         private long[] path;
23         private long parentNativePtr;
24     
25         SubtableSchema(long parentNativePtr, long[] path) {
26             this.parentNativePtr = parentNativePtr;
27             this.path = path;
28         }
29     
30         @Override
31         public SubtableSchema getSubtableSchema(long columnIndex) {
32             long[] newPath = new long[this.path.length+1];
33             for (int i = 0; i < this.path.length; i++) {
34                 newPath[i] = path[i];
35             }
36             newPath[this.path.length] = columnIndex;
37             return new SubtableSchema(this.parentNativePtr, newPath);
38         }
39     
40         private void verifyColumnName(String name) {
41             if (name.length() > 63) {
42                 throw new IllegalArgumentException("Column names are currently limited to max 63 characters.");
43             }
44         }
45     
46         @Override
47         public long addColumn(ColumnType type, String name) {
48             verifyColumnName(name);
49             return nativeAddColumn(parentNativePtr, path, type.getValue(), name);
50         }
51     
52         protected native long nativeAddColumn(long nativeTablePtr, long[] path, int type, String name);
53     
54         /**
55          * Remove a column in the table dynamically.
56          */
57         @Override
58         public void removeColumn(long columnIndex) {
59             nativeRemoveColumn(parentNativePtr, path, columnIndex);
60         }
61     
62         protected native void nativeRemoveColumn(long nativeTablePtr, long[] path, long columnIndex);
63     
64         /**
65          * Rename a column in the table.
66          */
67         @Override
68         public void renameColumn(long columnIndex, String newName) {
69             verifyColumnName(newName);
70             nativeRenameColumn(parentNativePtr, path, columnIndex, newName);
71         }
72     
73         protected native void nativeRenameColumn(long nativeTablePtr, long[] path, long columnIndex, String name);
74     }
