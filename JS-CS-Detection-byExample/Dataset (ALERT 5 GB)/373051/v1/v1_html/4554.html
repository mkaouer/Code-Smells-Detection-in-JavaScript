17     package io.realm.internal;
18     
19     public class ReadTransaction extends Group {
20     
21         private final SharedGroup db;
22     
23         ReadTransaction(Context context, SharedGroup db, long nativePointer) {
24             super(context, nativePointer, true); // make Group immutable
25             this.db = db;
26         }
27     
28         public void endRead() {
29             db.endRead();
30         }
31     
32         @Override
33         public void close() {
34             db.endRead();
35         }
36     
37         protected void finalize() {} // Nullify the actions of Group.finalize()
38     }
