163         public void removeOneByOne(boolean atFirst) {
164             Set<Long> ages = new HashSet<Long>();
165             testRealm.beginTransaction();
166             testRealm.clear(Dog.class);
167             for (int i = 0; i < TEST_SIZE; i++) {
168                 Dog dog = testRealm.createObject(Dog.class);
169                 dog.setAge(i);
170                 ages.add((long) i);
171             }
172             testRealm.commitTransaction();
173     
174             assertEquals(TEST_SIZE, testRealm.allObjects(Dog.class).size());
175     
176             RealmResults<Dog> dogs = testRealm.allObjects(Dog.class);
177             for (int i = 0; i < TEST_SIZE; i++) {
178                 testRealm.beginTransaction();
179                 Dog dogToRemove;
180                 if (atFirst) {
181                     dogToRemove = dogs.first();
182                 } else {
183                     dogToRemove = dogs.last();
184                 }
185                 ages.remove(Long.valueOf(dogToRemove.getAge()));
186                 dogToRemove.removeFromRealm();
187     
188                 // object is no longer valid
189                 try {
190                     dogToRemove.getAge();
191                     fail();
192                 }
193                 catch (IllegalStateException ignored) {}
194     
195                 testRealm.commitTransaction();
196     
197                 // and removed from realm and remaining objects are place correctly
198                 RealmResults<Dog> remainingDogs = testRealm.allObjects(Dog.class);
199                 assertEquals(TEST_SIZE - i - 1, remainingDogs.size());
200                 for (Dog dog : remainingDogs) {
201                     assertTrue(ages.contains(Long.valueOf(dog.getAge())));
202                 }
203             }
204         }

211         public boolean methodWrongThread(final boolean callGetter) throws ExecutionException, InterruptedException {
212             Realm realm = Realm.getInstance(getContext());
213             realm.beginTransaction();
214             realm.createObject(AllTypes.class);
215             realm.commitTransaction();
216             final AllTypes allTypes = realm.where(AllTypes.class).findFirst();
217             ExecutorService executorService = Executors.newSingleThreadExecutor();
218             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
219                 @Override
220                 public Boolean call() throws Exception {
221                     try {
222                         if (callGetter) {
223                             allTypes.getColumnFloat();
224                         } else {
225                             allTypes.setColumnFloat(1.0f);
226                         }
227                         return false;
228                     } catch (IllegalStateException ignored) {
229                         return true;
230                     }
231                 }
232             });
233     
234             Boolean result = future.get();
235             realm.close();
236             return result;
237         }

324         public void testDateType() {
325             long testDatesNotValid[] = {Long.MIN_VALUE, Long.MAX_VALUE};
326             long testDatesValid[] = {-1000, 0, 1000};
327             long testDatesLoosePrecision[] = {1, 1001};
328     
329             // test valid dates
330             testRealm.beginTransaction();
331             for (long value : testDatesValid) {
332                 AllTypes allTypes = testRealm.createObject(AllTypes.class);
333                 allTypes.setColumnDate(new Date(value));
334             }
335             testRealm.commitTransaction();
336     
337             int i = 0;
338             for (AllTypes allTypes : testRealm.allObjects(AllTypes.class)) {
339                 assertEquals("Item " + i, new Date(testDatesValid[i]), allTypes.getColumnDate());
340                 i++;
341             }
342     
343             // test valid dates but with precision lost
344             testRealm.beginTransaction();
345             testRealm.clear(AllTypes.class);
346             for (long value : testDatesLoosePrecision) {
347                 AllTypes allTypes = testRealm.createObject(AllTypes.class);
348                 allTypes.setColumnDate(new Date(value));
349             }
350             testRealm.commitTransaction();
351     
352             i = 0;
353             for (AllTypes allTypes : testRealm.allObjects(AllTypes.class)) {
354                 assertFalse("Item " + i, new Date(testDatesLoosePrecision[i]) == allTypes.getColumnDate());
355                 assertEquals("Item " + i, new Date(1000*(testDatesLoosePrecision[i]/1000)), allTypes.getColumnDate());
356                 i++;
357             }
358     
359             // test invalid dates
360             for (long value : testDatesNotValid) {
361                 try {
362                     testRealm.beginTransaction();
363                     testRealm.clear(AllTypes.class);
364                     AllTypes allTypes = testRealm.createObject(AllTypes.class);
365                     allTypes.setColumnDate(new Date(value));
366                     testRealm.commitTransaction();
367                     fail();
368                 } catch (IllegalArgumentException ignored) { testRealm.cancelTransaction(); }
369             }
370         }

372         private Date newDate(int year, int month, int dayOfMonth) {
373             Calendar cal = Calendar.getInstance();
374             cal.set(Calendar.YEAR, year);
375             cal.set(Calendar.MONTH, month);
376             cal.set(Calendar.DAY_OF_MONTH, dayOfMonth);
377             cal.set(Calendar.HOUR, 0);
378             cal.set(Calendar.MINUTE, 0);
379             cal.set(Calendar.MILLISECOND, 0);
380             return cal.getTime();
381         }

383         private void addDate(int year, int month, int dayOfMonth) {
384             Date date = newDate(year, month, dayOfMonth);
385     
386             testRealm.beginTransaction();
387             testRealm.clear(AllTypes.class);
388             AllTypes allTypes = testRealm.createObject(AllTypes.class);
389             allTypes.setColumnDate(date);
390             testRealm.commitTransaction();
391     
392             AllTypes object = testRealm.allObjects(AllTypes.class).first();
393     
394             // Realm does not support millisec precision
395             assertEquals(1000 * (date.getTime() / 1000), 1000 * (object.getColumnDate().getTime() / 1000));
396         }
