52     #define CATCH_FILE(fileName) \
53         catch (InvalidDatabase&) { \
54             ThrowException(env, IllegalArgument, "Invalid Group file format."); \
55         } \
56         catch (util::File::PermissionDenied& e) { \
57             ThrowException(env, IOFailed, string(fileName), string("Permission denied. ") + e.what()); \
58         } \
59         catch (util::File::NotFound&) { \
60             ThrowException(env, FileNotFound, string(fileName));    \
61         } \
62         catch (util::File::AccessError& e) { \
63             ThrowException(env, FileAccessError, string(fileName), e.what()); \
64         }
65     
66     #define CATCH_STD() \
67         catch (std::bad_alloc& e) { \
68             ThrowException(env, OutOfMemory, e.what() + std::string(" in ") + std::string(__FILE__) + \
69                                              std::string(" line ") + std::string(STRINGIZE(__LINE__))); \
70         } \
71         catch (std::exception& e) { \
72             ThrowException(env, Unspecified, e.what() + std::string(" in ") + std::string(__FILE__) + \
73                                              std::string(" line ") + std::string(STRINGIZE(__LINE__))); \
74         } \
75         catch (...) { \
76             TIGHTDB_ASSERT(false); \
77             ThrowException(env, RuntimeError, std::string("Exception in ") + \
78                                               std::string(__FILE__) + std::string(" line ") \
79                                               + std::string(STRINGIZE(__LINE__))); \
80         }
81         /* above (...) is not needed if we only throw exceptions derived from std::exception */

206     template <class T>
207     inline bool TableIsValid(JNIEnv* env, T* objPtr)
208     {
209         bool valid = (objPtr != NULL);
210         if (valid) {
211             // Check if Table is valid
212             if (tightdb::util::SameType<tightdb::Table, T>::value) {
213                 valid = TBL(objPtr)->is_attached();
214             }
215             // TODO: Add check for TableView
216     
217         }
218         if (!valid) {
219             TR_ERR((env, "Table %x is no longer attached!", objPtr));
220             ThrowException(env, TableInvalid, "Table is no longer valid to operate on.");
221         }
222         return valid;
223     }
224     
225     inline bool RowIsValid(JNIEnv* env, Row* rowPtr)
226     {
227         bool valid = (rowPtr != NULL && rowPtr->is_attached());
228         if (!valid) {
229             TR_ERR((env, "Row %x is no longer attached!", rowPtr));
230             ThrowException(env, RowInvalid, "Row/Object is no longer valid to operate on. Was it deleted?");
231         }
232         return valid;
233     }

235     // Requires an attached Table
236     template <class T>
237     bool RowIndexesValid(JNIEnv* env, T* pTable, jlong startIndex, jlong endIndex, jlong range)
238     {
239         size_t maxIndex = pTable->size();
240         if (endIndex == -1)
241             endIndex = maxIndex;
242         if (startIndex < 0) {
243             TR_ERR((env, "startIndex %lld < 0 - invalid!", S(startIndex), 0));
244             ThrowException(env, IndexOutOfBounds, "startIndex < 0.");
245             return false;
246         }
247         if (tightdb::util::int_greater_than(startIndex, maxIndex)) {
248             TR_ERR((env, "startIndex %lld > %lld - invalid!", S(startIndex), maxIndex));
249             ThrowException(env, IndexOutOfBounds, "startIndex > available rows.");
250             return false;
251         }
252     
253         if (tightdb::util::int_greater_than(endIndex, maxIndex)) {
254             TR_ERR((env, "endIndex %lld > %lld - invalid!", S(endIndex), maxIndex));
255             ThrowException(env, IndexOutOfBounds, "endIndex > available rows.");
256             return false;
257         }
258         if (startIndex > endIndex) {
259             TR_ERR((env, "startIndex %lld > endIndex %lld- invalid!", S(startIndex), S(endIndex)));
260             ThrowException(env, IndexOutOfBounds, "startIndex > endIndex.");
261             return false;
262         }
263     
264         if (range != -1 && range < 0) {
265             TR_ERR((env, "range %lld < 0 - invalid!", range));
266             ThrowException(env, IndexOutOfBounds, "range < 0.");
267             return false;
268         }
269     
270         return true;
271     }

273     template <class T>
274     inline bool RowIndexValid(JNIEnv* env, T* pTable, jlong rowIndex, bool offset=false)
275     {
276         if (rowIndex < 0) {
277             ThrowException(env, IndexOutOfBounds, "rowIndex is less than 0.");
278             return false;
279         }
280         size_t size = pTable->size();
281         if (size > 0 && offset)
282             size -= 1;
283         bool rowErr = tightdb::util::int_greater_than_or_equal(rowIndex, size);
284         if (rowErr) {
285             TR_ERR((env, "rowIndex %lld > %lld - invalid!", S(rowIndex), size));
286             ThrowException(env, IndexOutOfBounds, "rowIndex > available rows.");
287         }
288         return !rowErr;
289     }

291     template <class T>
292     inline bool TblRowIndexValid(JNIEnv* env, T* pTable, jlong rowIndex, bool offset=false)
293     {
294         if (tightdb::util::SameType<tightdb::Table, T>::value) {
295             if (!TableIsValid(env, TBL(pTable)))
296                 return false;
297         }
298         return RowIndexValid(env, pTable, rowIndex, offset);
299     }

301     template <class T>
302     inline bool ColIndexValid(JNIEnv* env, T* pTable, jlong columnIndex)
303     {
304         if (columnIndex < 0) {
305             ThrowException(env, IndexOutOfBounds, "columnIndex is less than 0.");
306             return false;
307         }
308         bool colErr = tightdb::util::int_greater_than_or_equal(columnIndex, pTable->get_column_count());
309         if (colErr) {
310             TR_ERR((env, "columnIndex %lld > %lld - invalid!", S(columnIndex), pTable->get_column_count()));
311             ThrowException(env, IndexOutOfBounds, "columnIndex > available columns.");
312         }
313         return !colErr;
314     }

345     template <class T>
346     inline bool TblIndexInsertValid(JNIEnv* env, T* pTable, jlong columnIndex, jlong rowIndex)
347     {
348         if (!TblColIndexValid(env, pTable, columnIndex))
349             return false;
350         bool rowErr = tightdb::util::int_greater_than(rowIndex, pTable->size()+1);
351         if (rowErr) {
352             TR_ERR((env, "rowIndex %lld > %lld - invalid!", rowIndex, pTable->size()));
353             ThrowException(env, IndexOutOfBounds,
354                 "rowIndex " + num_to_string(rowIndex) +
355                 " > available rows " + num_to_string(pTable->size()) + ".");
356         }
357         return !rowErr;
358     }

360     template <class T>
361     inline bool TypeValid(JNIEnv* env, T* pTable, jlong columnIndex, jlong rowIndex, int expectColType, bool allowMixed)
362     {
363         size_t col = static_cast<size_t>(columnIndex);
364         int colType = pTable->get_column_type(col);
365         if (allowMixed) {
366             if (colType == tightdb::type_Mixed) {
367                 size_t row = static_cast<size_t>(rowIndex);
368                 colType = pTable->get_mixed_type(col, row);
369             }
370         }
371         if (colType != expectColType) {
372             TR_ERR((env, "Expected columnType %d, but got %d.", expectColType, pTable->get_column_type(col)));
373             ThrowException(env, IllegalArgument, "ColumnType invalid.");
374             return false;
375         }
376         return true;
377     }

423     // Note: JNI offers methods to convert between modified UTF-8 and
424     // UTF-16. Unfortunately these methods are not appropriate in this
425     // context. The reason is that they use a modified version of
426     // UTF-8 where U+0000 is stored as 0xC0 0x80 instead of 0x00 and
427     // where a character in the range U+10000 to U+10FFFF is stored as
428     // two consecutive UTF-8 encodings of the corresponding UTF-16
429     // surrogate pair. Because TightDB uses proper UTF-8, we need to
430     // do the transcoding ourselves.
431     //
432     // See also http://en.wikipedia.org/wiki/UTF-8#Modified_UTF-8
433     
434     jstring to_jstring(JNIEnv*, tightdb::StringData);
435     
436     class JStringAccessor {
437     public:
438         JStringAccessor(JNIEnv*, jstring);  // throws
439     
440         operator tightdb::StringData() const TIGHTDB_NOEXCEPT
441         {
442             return tightdb::StringData(m_data.get(), m_size);
443         }
444     
445     private:
446         tightdb::util::UniquePtr<char[]> m_data;
447         std::size_t m_size;
448     };

450     class KeyBuffer {
451     public:
452         KeyBuffer(JNIEnv* env, jbyteArray arr)
453         : m_env(env)
454         , m_array(arr)
455         , m_ptr(0)
456         {
457     #ifdef TIGHTDB_ENABLE_ENCRYPTION
458             if (arr) {
459                 if (env->GetArrayLength(m_array) != 32)
460                     ThrowException(env, UnsupportedOperation, "Encryption key must be exactly 32 bytes.");
461                 m_ptr = env->GetByteArrayElements(m_array, NULL);
462             }
463     #else
464             if (arr)
465                 ThrowException(env, UnsupportedOperation,
466                                "Encryption was disabled in the native library at compile time.");
467     #endif
468         }
469     
470         const uint8_t *data() const {
471             return reinterpret_cast<const uint8_t *>(m_ptr);
472         }
473     
474         ~KeyBuffer() {
475             if (m_ptr)
476                 m_env->ReleaseByteArrayElements(m_array, m_ptr, JNI_ABORT);
477         }
478     
479     private:
480         JNIEnv* m_env;
481         jbyteArray m_array;
482         jbyte* m_ptr;
483     };
