35     public class ConcurrencyExampleActivity extends Activity implements View.OnClickListener {
36     
37         @SuppressWarnings("UnusedDeclaration")
38         public static final String TAG = ConcurrencyExampleActivity.class.getName();
39     
40         private Realm realm;
41     
42         @Override
43         protected void onCreate(Bundle savedInstanceState) {
44             super.onCreate(savedInstanceState);
45             setContentView(R.layout.activity_realm_example);
46     
47             findViewById(R.id.insert_record_button).setOnClickListener(this);
48             findViewById(R.id.quit_button).setOnClickListener(new View.OnClickListener() {
49                 @Override
50                 public void onClick(View view) {
51                     stopTests();
52                 }
53             });
54     
55             // Reset the realm data before starting the tests
56             Realm.deleteRealmFile(this);
57     
58             // Acquire a realm object
59             realm = Realm.getInstance(this);
60         }

62         @Override
63         public void onResume() {
64             super.onResume();
65             startTests();
66         }
67     
68         @Override
69         public void onStop() {
70             super.onStop();
71             stopTests();
72         }
73     
74         @Override
75         protected void onDestroy() {
76             super.onDestroy();
77             realm.close(); // Remember to close Realm when done.
78         }

80         private void startTests() {
81             startSpawnTests();
82             //Alternatively can use transaction tests here...
83             //startTransactionTests();
84         }
85     
86         private void stopTests() {
87             stopSpawnTests();
88             //Alternatively can use transaction tests here...
89             //stopTransactionTests();
90         }
91     
92         // Using the screen form the user can inject into the Realm
93         // even if other threads are creating transactions.
94         @Override
95         public void onClick(View v) {
96             String personName = ((TextView) findViewById(R.id.name)).getText().toString();
97             String personAge = ((TextView) findViewById(R.id.age)).getText().toString();
98             String petName = ((TextView) findViewById(R.id.pets_name)).getText().toString();
99     
100             Integer parseAge;
101             try {
102                 parseAge = Integer.parseInt(personAge);
103             } catch (NumberFormatException ignored) {
104                 Log.d(TAG, "Age for a person invalid");
105                 return;
106             }
107     
108             realm.beginTransaction();
109             Person person = realm.createObject(Person.class);
110             person.setName(personName);
111             person.setAge(parseAge);
112     
113             int checkedId = ((RadioGroup) findViewById(R.id.petType)).getCheckedRadioButtonId();
114             if (checkedId == R.id.hasCat) {
115                 Cat cat = realm.createObject(Cat.class);
116                 cat.setName(petName);
117                 RealmList<Cat> cats = person.getCats();
118                 cats.add(cat);
119             } else if (checkedId == R.id.hasDog) {
120                 Dog dog = realm.createObject(Dog.class);
121                 dog.setName(petName);
122                 person.setDog(dog);
123             }
124     
125             realm.commitTransaction();
126     
127         }
128     
129         // The transaction tests are one IntentService running multiple Writes
130         // to a Realm in either a multiple transaction or single transaction loop
131         private void startTransactionTests() {
132             Intent serviceIntent = new Intent(this, TransactionService.class);
133             serviceIntent.putExtra(TransactionService.REALM_TESTTYPE_EXTRA,
134                     TransactionService.TestType.MANY_TRANSACTIONS);
135             serviceIntent.putExtra(TransactionService.ITERATION_COUNT, 10000000);
136             this.startService(serviceIntent);
137         }
138     
139         private void stopTransactionTests() {
140             Intent serviceIntent = new Intent(this, TransactionService.class);
141             this.stopService(serviceIntent);
142         }
143     
144         // The Spawned tests create multiple threads of type Reader or Writer
145         // which loop for a specified count
146         private void startSpawnTests() {
147             Intent serviceIntent = new Intent(this, SpawningService.class);
148             serviceIntent.putExtra(SpawningService.REALM_INSERT_COUNT_EXTRA, 10000000);
149             serviceIntent.putExtra(SpawningService.REALM_READ_COUNT_EXTRA, 10000000);
150             this.startService(serviceIntent);
151         }
152     
153         private void stopSpawnTests() {
154             Intent serviceIntent = new Intent(this, SpawningService.class);
155             this.stopService(serviceIntent);
156         }
157     
158         @SuppressWarnings("UnusedDeclaration")
159         private void restartTests() {
160             stopTransactionTests();
161             startTransactionTests();
162         }
163     }
