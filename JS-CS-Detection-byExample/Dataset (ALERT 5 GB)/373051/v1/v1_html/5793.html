44             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
45                 @Override
46                 public Boolean call() throws Exception {
47                     Realm realm = Realm.getInstance(getContext(), false);

88             ExecutorService executorService = Executors.newSingleThreadExecutor();

134         public void testAutoUpdateRealmResults() throws InterruptedException, ExecutionException {
135             final int TEST_SIZE = 10;
136             final AtomicInteger counter = new AtomicInteger(0);
137             final AtomicBoolean isReady = new AtomicBoolean(false);
138             final Map<Integer, Integer> results = new ConcurrentHashMap<Integer, Integer>();
139     
140             ExecutorService executorService = Executors.newSingleThreadExecutor();
141             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
142                 @Override
143                 public Boolean call() throws Exception {
144                     Looper.prepare();
145                     Realm realm = null;
146                     try {
147                         realm = Realm.getInstance(getContext());
148                         final RealmResults<Dog> dogs = realm.allObjects(Dog.class);
149                         assertEquals(0, dogs.size());
150                         realm.addChangeListener(new RealmChangeListener() {
151                             @Override
152                             public void onChange() {
153                                 int c = counter.incrementAndGet();
154                                 results.put(c, dogs.size());
155                             }
156                         });
157                         isReady.set(true);
158                         Looper.loop();
159                     } finally {
160                         if (realm != null) {
161                             realm.close();
162                         }
163                     }
164                     return true;
165                 }
166             });
167     
168             // Wait until the looper is started
169             while (!isReady.get()) {
170                 Thread.sleep(5);
171             }
172             Thread.sleep(100);
173     
174             Realm realm = Realm.getInstance(getContext());
175             realm.beginTransaction();
176             for (int i = 0; i < TEST_SIZE; i++) {
177                 Dog dog = realm.createObject(Dog.class);
178                 dog.setName("Rex " + i);
179             }
180             realm.commitTransaction();
181             assertEquals(TEST_SIZE, realm.allObjects(Dog.class).size());
182             realm.close();
183     
184             try {
185                 future.get(2, TimeUnit.SECONDS);
186             } catch (TimeoutException ignore) {}
187     
188             assertEquals(1, results.size());
189     
190             assertTrue(results.containsKey(1));
191             assertEquals(TEST_SIZE, results.get(1).intValue());
192     
193             assertEquals(1, counter.get());
194             assertTrue(Realm.realmsCache.get().isEmpty());
195         }
