24     jclass GetClassTableSpec(JNIEnv* env)
25     {
26         static jclass myClass = GetClass(env, "io/realm/internal/TableSpec");
27         return myClass;
28     }

30     jmethodID GetTableSpecMethodID(JNIEnv* env, const char* methodStr, const char* typeStr)
31     {
32         jclass myClass = GetClassTableSpec(env);
33         if (myClass == NULL) {
34             return NULL;
35         }
36         jmethodID myMethod = env->GetMethodID(myClass, methodStr, typeStr);
37         if (myMethod == NULL) {
38             ThrowException(env, NoSuchMethod, "TableSpec", methodStr);
39             return NULL;
40         }
41         return myMethod;
42     }

84     void set_descriptor(JNIEnv* env, Descriptor& desc, jobject jTableSpec)
85     {
86         jlong n = Java_io_realm_TableSpec_getColumnCount(env, jTableSpec);
87         for (jlong i = 0; i != n; ++i) {
88             jstring jColumnName = Java_io_realm_TableSpec_getColumnName(env, jTableSpec, i);
89             JStringAccessor name(env, jColumnName);  // throws
90     
91             jobject jColumnType = Java_io_realm_TableSpec_getColumnType(env, jTableSpec, i);
92             DataType type = GetColumnTypeFromJColumnType(env, jColumnType);
93             DescriptorRef subdesc;
94             desc.add_column(type, name, &subdesc); // Throws
95             if (type == type_Table) {
96                 jobject jNextColumnTableSpec = Java_io_realm_TableSpec_getTableSpec(env, jTableSpec, i);
97                 set_descriptor(env, *subdesc, jNextColumnTableSpec);
98             }
99         }
100     }

102     void get_descriptor(JNIEnv* env, const Descriptor& desc, jobject jTableSpec)
103     {
104         static jmethodID jAddColumnMethodId = GetTableSpecMethodID(env, "addColumn", "(ILjava/lang/String;)V");
105         static jmethodID jAddSubtableColumnMethodId = GetTableSpecMethodID(env, "addSubtableColumn", 
106                                                                                 "(Ljava/lang/String;)Lio/realm/internal/TableSpec;");
107     
108         if (jAddColumnMethodId == NULL || jAddSubtableColumnMethodId == NULL) {
109             return;
110         }
111     
112         size_t n = desc.get_column_count(); // noexcept
113         for (size_t i = 0; i != n; ++i) {
114             DataType type   = desc.get_column_type(i); // noexcept
115             StringData name = desc.get_column_name(i); // noexcept
116             if (type == type_Table) {
117                 jobject jSubTableSpec = env->CallObjectMethod(jTableSpec, jAddSubtableColumnMethodId, 
118                                                               to_jstring(env, name));
119                 ConstDescriptorRef subdesc = desc.get_subdescriptor(i); // Throws
120                 get_descriptor(env, *subdesc, jSubTableSpec);
121             }
122             else {
123                 env->CallVoidMethod(jTableSpec, jAddColumnMethodId, jint(type), to_jstring(env, name));
124             }
125         }
