17     package io.realm;
18     
19     import android.content.Context;
20     import android.view.LayoutInflater;
21     import android.widget.BaseAdapter;
22     
23     public abstract class RealmBaseAdapter<T extends RealmObject> extends BaseAdapter {
24     
25         protected LayoutInflater inflater;
26         protected RealmResults<T> realmResults;
27         protected Context context;
28     
29         public RealmBaseAdapter(Context context, RealmResults<T> realmResults, boolean automaticUpdate) {
30             if (context == null) {
31                 throw new IllegalArgumentException("Context cannot be null");
32             }
33             if (realmResults == null) {
34                 throw new IllegalArgumentException("RealmResults cannot be null");
35             }
36     
37             this.context = context;
38             this.realmResults = realmResults;
39             this.inflater = LayoutInflater.from(context);
40             if (automaticUpdate) {
41                 realmResults.getRealm().addChangeListener(new RealmChangeListener() {
42                     @Override
43                     public void onChange() {
44                         notifyDataSetChanged();
45                     }
46                 });
47             }
48         }
49     
50         @Override
51         public int getCount() {
52             return realmResults.size();
53         }
54     
55         @Override
56         public T getItem(int i) {
57             return realmResults.get(i);
58         }
59     
60         @Override
61         public long getItemId(int i) {
62             return i; // TODO: find better solution once we have unique IDs
63         }
64     
65         /**
66          * Update the RealmResults associated to the Adapter. Useful when the query has been changed.
67          * If the query does not change you might consider using the automaticUpdate feature
68          * @param realmResults the new RealmResults coming from the new query.
69          */
70         public void updateRealmResults(RealmResults<T> realmResults) {
71             this.realmResults = realmResults;
72             notifyDataSetChanged();
73         }
74     }
