17     package io.realm.examples.concurrency.services;
18     
19     import android.app.IntentService;
20     import android.content.Intent;
21     import android.util.Log;
22     
23     import io.realm.Realm;
24     import io.realm.examples.concurrency.model.Person;
25     
26     // Note that an IntentService operates in its own thread-pool facilitated by Android
27     public class TransactionService extends IntentService {
28     
29         public static final String TAG = TransactionService.class.getName();
30     
31         private Boolean mQuitting = false;
32     
33         private int mInsertCount = 0;
34     
35         public static final String REALM_TESTTYPE_EXTRA = "TestTypeExtra";
36         public static final String ITERATION_COUNT      = "TestIterationCount";
37     
38         public enum TestType {
39             MANY_INSERTS_ONE_TRANSACTION,
40             MANY_TRANSACTIONS
41         }
42     
43         private Realm realm = null;
44     
45         public TransactionService() {
46             super(TransactionService.class.getName());
47         }
48     
49         @Override
50         protected void onHandleIntent(Intent intent) {
51             Log.d(TAG, TAG + " is starting");
52     
53             realm = Realm.getInstance(this);
54     
55             TestType type = (TestType)intent.getSerializableExtra(REALM_TESTTYPE_EXTRA);
56             mInsertCount = intent.getIntExtra(ITERATION_COUNT, 0);
57     
58             switch(type) {
59                 case MANY_INSERTS_ONE_TRANSACTION:
60                     doSingleTransaction();
61                     break;
62                 case MANY_TRANSACTIONS:
63                     doSeveralTransactions();
64                     break;
65                 default:
66             }
67     
68             Log.d(TAG, TAG + " has quit");
69         }
70     
71         @Override
72         public void onDestroy() {
73             this.mQuitting = true;
74         }
75     
76         // This method creates mInsertCount injections into the Realm
77         // inside >>ONE transaction<<
78         private void doSingleTransaction() {
79             int iterCount = 0;
80     
81             // NOTE:  If you attempt to make writes from the UI while this loop is running
82             // it will block because of the write transaction being open.
83             realm.beginTransaction();
84             while (iterCount < mInsertCount && !mQuitting) {
85                 if ((iterCount % 1000) == 0) {
86                     Log.d(TAG, "WriteOperation#: " + iterCount + "," + Thread.currentThread().getName());
87                 }
88     
89                 Person person = realm.createObject(Person.class);
90                 person.setName("Foo" + iterCount);
91                 iterCount++;
92             }
93             realm.commitTransaction();
94         }
95     
96         // This method creates mInsertCount injections into the Realm
97         // using >>one transaction for EACH insert<<
98         private void doSeveralTransactions() {
99             int iterCount = 0;
100     
101             while (iterCount < mInsertCount && !mQuitting) {
102                 if ((iterCount % 1000) == 0) {
103                     Log.d(TAG, "WriteOperation#: " + iterCount + "," + Thread.currentThread().getName());
104                 }
105                 realm.beginTransaction();
106                 Person person = realm.createObject(Person.class);
107                 person.setName("Foo" + iterCount);
108                 iterCount++;
109                 realm.commitTransaction();
110             }
111         }
112     
113     }
