23     void arrayToVector(JNIEnv* env, jlongArray path, vector<size_t>& nativePath)
24     {
25         jsize size = env->GetArrayLength(path);
26         nativePath.reserve(size+1);
27     
28         jlong* pathElements = env->GetLongArrayElements(path, 0);
29         for (jsize i = 0; i < size; ++i) {
30             nativePath.push_back(S(pathElements[i]));
31         }
32         env->ReleaseLongArrayElements(path, pathElements, JNI_ABORT);
33     }

35     JNIEXPORT jlong JNICALL Java_io_realm_internal_SubtableSchema_nativeAddColumn
36       (JNIEnv* env, jobject, jlong  nativeTablePtr, jlongArray path, jint colType, jstring name)
37     {
38         if (!TABLE_VALID(env, TBL(nativeTablePtr)))
39             return 0;
40         try {
41             JStringAccessor name2(env, name); // throws
42             vector<size_t> nativePath;
43             arrayToVector(env, path, nativePath);
44             return TBL(nativeTablePtr)->add_subcolumn(nativePath, DataType(colType), name2);
45         }
46         CATCH_STD()
47         return 0;
48     }

52     {
53         if (!TBL_AND_COL_INDEX_VALID(env, TBL(nativeTablePtr), columnIndex))
54             return;

56             vector<size_t> nativePath;
57             arrayToVector(env, path, nativePath);

61     }
