33     JNIEXPORT jlong JNICALL Java_io_realm_internal_SharedGroup_createNative(
34         JNIEnv* env, jobject, jstring file_name, jint durability, jboolean no_create, jboolean enable_replication, jbyteArray keyArray)
35     {
36         const char* file_name_ptr = env->GetStringUTFChars(file_name, 0);
37         if (!file_name_ptr)
38             return 0; // Exception is thrown by GetStringUTFChars()
39     
40         SharedGroup* db = 0;
41         try {
42             if (enable_replication) {
43     #ifdef TIGHTDB_ENABLE_REPLICATION
44                 ThrowException(env, UnsupportedOperation,
45                                "Replication is not currently supported by the Java language binding.");
46     //            db = new SharedGroup(SharedGroup::replication_tag(), *file_name_ptr ? file_name_ptr : 0);
47     #else
48                 ThrowException(env, UnsupportedOperation,
49                                "Replication was disabled in the native library at compile time.");
50     #endif
51             }
52             else {
53                 SharedGroup::DurabilityLevel level;
54                 if (durability == 0)
55                     level = SharedGroup::durability_Full;
56                 else if (durability == 1)
57                     level = SharedGroup::durability_MemOnly;
58                 else if (durability == 2)
59     #ifdef _WIN32
60                     level = SharedGroup::durability_Full;   // For Windows, use Full instead of Async
61     #else
62                     level = SharedGroup::durability_Async;
63     #endif
64                 else {
65                     ThrowException(env, UnsupportedOperation, "Unsupported durability.");
66                     return 0;
67                 }
68     
69                 KeyBuffer key(env, keyArray);
70     #ifdef TIGHTDB_ENABLE_ENCRYPTION
71                 db = new SharedGroup(file_name_ptr, no_create!=0, level, key.data());
72     #else
73                 db = new SharedGroup(file_name_ptr, no_create!=0, level);
74     #endif
75             }
76             return reinterpret_cast<jlong>(db);
77         }
78         catch (SharedGroup::PresumablyStaleLockFile& e) {
79             ThrowException(env, FileAccessError, e.what(), " Presumably a stall .lock file is present.");
80         }
81         catch (SharedGroup::LockFileButNoData& e) {
82             ThrowException(env, FileAccessError, e.what(), "The database file is missing, but a .lock file is present.");
83         }
84         CATCH_FILE(file_name_ptr)
85         CATCH_STD()
86         return 0;
87     }

89     JNIEXPORT jlong JNICALL Java_io_realm_internal_SharedGroup_createNativeWithImplicitTransactions
90       (JNIEnv* env, jobject, jlong native_replication_ptr, jbyteArray keyArray)
91     {
92         try {
93             KeyBuffer key(env, keyArray);
94     #ifdef TIGHTDB_ENABLE_ENCRYPTION
95             SharedGroup* db = new SharedGroup(*reinterpret_cast<tightdb::Replication*>(native_replication_ptr), key.data());
96     #else
97             SharedGroup* db = new SharedGroup(*reinterpret_cast<tightdb::Replication*>(native_replication_ptr));
98     #endif
99     
100             return reinterpret_cast<jlong>(db);
101         }
102         catch (SharedGroup::PresumablyStaleLockFile& e) {
103             ThrowException(env, FileAccessError, e.what(), " Presumably a stall .lock file is present.");
104         }
105         catch (SharedGroup::LockFileButNoData& e) {
106             ThrowException(env, FileAccessError, e.what(), "The database file is missing, but a .lock file is present.");
107         }
108         CATCH_STD()
109         return 0;
110     }
