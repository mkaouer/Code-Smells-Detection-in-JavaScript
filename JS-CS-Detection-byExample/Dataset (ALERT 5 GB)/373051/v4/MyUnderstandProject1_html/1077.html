153         public void testAvgGivesCorrectValue() {
154             RealmResults<AllTypes> resultList = testRealm.where(AllTypes.class).findAll();
155             double N = (double)TEST_DATA_SIZE;
156     
157             // Sum of numbers 1 to M: M*(M+1)/2
158             // See setUp() for values of fields
159             // N = TEST_DATA_SIZE
160     
161             // Type: double; a = 3.1415
162             // a, a+1, ..., a+i, ..., a+N-1
163             // sum = 3.1415*N + N*(N-1)/2
164             // average = sum/N = 3.1415+(N-1)/2
165             double average = 3.1415+(N-1.0)*0.5;
166             assertEquals(average, resultList.average(FIELD_DOUBLE), 0.0001);
167     
168             // Type: long
169             // 0, 1, ..., N-1
170             // sum = N*(N-1)/2
171             // average = sum/N = (N-1)/2
172             assertEquals(0.5*(N-1), resultList.average(FIELD_LONG), 0.0001);
173     
174             // Type: float; b = 1.234567
175             // b, b+1, ..., b+i, ..., b+N-1
176             // sum = b*N + N*(N-1)/2
177             // average = sum/N = b + (N-1)/2
178             assertEquals(1.234567+0.5*(N-1.0), resultList.average(FIELD_FLOAT), 0.0001);
179         }
