17     package io.realm;

19     import android.test.AndroidTestCase;
20     
21     import io.realm.entities.AnnotationTypes;
22     import io.realm.internal.Table;
23     
24     public class RealmAnnotationTest extends AndroidTestCase {
25         protected Realm testRealm;
26     
27         @Override
28         protected void setUp() throws Exception {
29             Realm.deleteRealmFile(getContext());
30             testRealm = Realm.getInstance(getContext());
31             testRealm.beginTransaction();
32             AnnotationTypes object = testRealm.createObject(AnnotationTypes.class);
33             object.setNotIndexString("String 1");
34             object.setIndexString("String 2");
35             object.setIgnoreString("String 3");
36             testRealm.commitTransaction();
37         }
38     
39         public void testIgnore() {
40             Table table = testRealm.getTable(AnnotationTypes.class);
41             assertEquals(-1, table.getColumnIndex("ignoreString"));
42         }
43     
44         public void testIndex() {
45             Table table = testRealm.getTable(AnnotationTypes.class);
46             assertTrue(table.hasIndex(table.getColumnIndex("indexString")));
47             assertFalse(table.hasIndex(table.getColumnIndex("notIndexString")));
48         }
49     }
