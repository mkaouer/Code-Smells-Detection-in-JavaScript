16     package io.realm.internal;
17     
18     import android.annotation.TargetApi;
19     import android.os.Build;
20     
21     import junit.framework.Test;
22     import junit.framework.TestCase;
23     import junit.framework.TestSuite;
24     
25     import java.lang.reflect.Constructor;
26     import java.lang.reflect.InvocationTargetException;
27     import java.lang.reflect.Method;
28     import java.lang.reflect.Modifier;
29     import java.util.ArrayList;
30     import java.util.Collection;
31     
32     public class JNITestSuite extends TestSuite {
33     
34         @TargetApi(Build.VERSION_CODES.JELLY_BEAN)
35         public JNITestSuite(
36                 final Class<? extends TestCase> testCaseClass,
37                 final Collection<Object[]> parameters) {
38     
39             setName(testCaseClass.getName());
40     
41             final Constructor<?>[] constructors = testCaseClass.getConstructors();
42             if (constructors.length != 1) {
43                 addTest(warning(testCaseClass.getName()
44                         + " must have a single public constructor."));
45                 return;
46             }
47     
48             final Collection<String> names = getTestMethods(testCaseClass);
49     
50             final Constructor<?> constructor = constructors[0];
51             final Collection<TestCase> testCaseInstances = new ArrayList<TestCase>();
52             try {
53                 for (final Object[] objects : parameters) {
54                     for (final String name : names) {
55                         TestCase testCase = (TestCase) constructor.newInstance(objects);
56                         testCase.setName(name);
57                         testCaseInstances.add(testCase);
58                     }
59                 }
60             } catch (IllegalArgumentException e) {
61                 addConstructionException(e);
62                 return;
63             } catch (InstantiationException e) {
64                 addConstructionException(e);
65                 return;
66             } catch (IllegalAccessException e) {
67                 addConstructionException(e);
68                 return;
69             } catch (InvocationTargetException e) {
70                 addConstructionException(e);
71                 return;
72             }
73     
74     
75             for (final TestCase testCase : testCaseInstances) {
76                 addTest(testCase);
77     
78             }
79         }
80     
81         private Collection<String> getTestMethods(
82                 final Class<? extends TestCase> testCaseClass) {
83             Class<?> superClass = testCaseClass;
84             final Collection<String> names = new ArrayList<String>();
85             while (Test.class.isAssignableFrom(superClass)) {
86                 Method[] methods = superClass.getDeclaredMethods();
87                 for (int i = 0; i < methods.length; i++) {
88                     addTestMethod(methods[i], names, testCaseClass);
89                 }
90                 superClass = superClass.getSuperclass();
91             }
92             return names;
93         }
94     
95         @TargetApi(Build.VERSION_CODES.JELLY_BEAN)
96         private void addTestMethod(Method m, Collection<String> names, Class<?> theClass) {
97             String name = m.getName();
98             if (names.contains(name))
99                 return;
100             if (!isPublicTestMethod(m)) {
101                 if (isTestMethod(m))
102                     addTest(warning("Test method isn't public: " + m.getName()));
103                 return;
104             }
105             names.add(name);
106         }
107     
108         private boolean isPublicTestMethod(Method m) {
109             return isTestMethod(m) && Modifier.isPublic(m.getModifiers());
110         }
111     
112         private boolean isTestMethod(Method m) {
113             String name = m.getName();
114             Class<?>[] parameters = m.getParameterTypes();
115             Class<?> returnType = m.getReturnType();
116             return parameters.length == 0 && name.startsWith("test") && returnType.equals(Void.TYPE);
117         }
118     
119         @TargetApi(Build.VERSION_CODES.JELLY_BEAN)
120         private void addConstructionException(Exception e) {
121             addTest(warning("Instantiation of a testCase failed "
122                     + e.getClass().getName() + " " + e.getMessage()));
123         }
