46     /**
47      * <p>The Realm class is the storage and transactional manager of your object persistent store. Objects
48      * are created. Objects within a Realm can be queried and read at any time. Creating,
49      * modifying, and deleting objects must be done through transactions.</p>
50      *
51      * <p>The transactions ensure that multiple instances (on multiple threads) can access the objects
52      * in a consistent state with full ACID guaranties.</p>
53      *
54      * <p>If auto-refresh is set the instance of the Realm will be automatically updated when another instance commits a
55      * change (create, modify or delete an object). This feature requires the Realm instance to be residing in a
56      * thread attached to a Looper (the main thread has a Looper by default)</p>
57      *
58      *<p>For normal threads Android provides a utility class that can be used like this:</p>
59      *
60      * <pre>
61      * HandlerThread thread = new HandlerThread("MyThread") {
62      *    @Override
63      *    protected void onLooperPrepared() {
64      *       Realm realm = Realm.getInstance(getContext());
65      *       // This realm will be updated by the event loop
66      *       // on every commit performed by other realm instances
67      *    }
68      * };
69      * thread.start();
70      * </pre>
71      */

126                 }
127                 return true;

139         /**
140          * Set the auto-refresh status of the Realm instance
141          *
142          * Auto-refresh is a feature to allows to automatically update the current realm instance and all its derived objects
143          * (RealmResults and RealmObjects instances) when a commit is performed on a Realm acting on the same file in another thread.
144          * This feature is only available if the realm instance lives is a {@link android.os.Looper} enabled thread.
145          *
146          * @param autoRefresh true will turn auto-refresh on, false will turn it off
147          * @throws java.lang.IllegalStateException if trying to enable auto-refresh in a thread without Looper
148          */
149         public void setAutoRefresh(boolean autoRefresh) {
150             if (autoRefresh && Looper.myLooper() == null) {
151                 throw new IllegalStateException("Cannot set auto-refresh in a Thread without a Looper");
152             }
153     
154             if (autoRefresh && !this.autoRefresh) { // Switch it on
155                 handler = new Handler(new RealmCallback());
156                 handlers.put(handler, id);
157             } else if (!autoRefresh && this.autoRefresh && handler != null) { // Switch it off
158                 handler.removeCallbacksAndMessages(null);
159                 handlers.remove(handler);
160             }
161             this.autoRefresh = autoRefresh;
162         }

358         private static Realm createAndValidate(String absolutePath, byte[] key, boolean validateSchema, boolean autoRefresh) {
359             Map<String, Realm> realms = realmsCache.get();
360             Realm realm = realms.get(absolutePath);
361     
362             if (realm == null) {
363                 realm = new Realm(absolutePath, key, autoRefresh);
364                 realms.put(absolutePath, realm);
365                 realmsCache.set(realms);
366             }
367             if (validateSchema) {
368                 Class<?> validationClass;
369                 try {
370                     validationClass = Class.forName("io.realm.ValidationList");
371                 } catch (ClassNotFoundException e) {
372                     throw new RealmException("Could not find the generated ValidationList class");
373                 }
374                 Method getProxyClassesMethod;
375                 try {
376                     getProxyClassesMethod = validationClass.getMethod("getProxyClasses");
377                 } catch (NoSuchMethodException e) {
378                     throw new RealmException("Could not find the getProxyClasses method in the ValidationList class");
379                 }
380                 List<String> proxyClasses;
381                 try {
382                     //noinspection unchecked
383                     proxyClasses = (List<String>) getProxyClassesMethod.invoke(null);
384                 } catch (IllegalAccessException e) {
385                     throw new RealmException("Could not execute the getProxyClasses method in the ValidationList class");
386                 } catch (InvocationTargetException e) {
387                     throw new RealmException("An exception was thrown in the getProxyClasses method in the ValidationList class");
388                 }
389     
390                 long version = realm.getVersion();
391                 try {
392                     realm.beginTransaction();
393                     if (version == UNVERSIONED) {
394                         realm.setVersion(0);
395                     }
396     
397                     for (String className : proxyClasses) {
398                         String[] splitted = className.split("\\.");
399                         String modelClassName = splitted[splitted.length - 1];
400                         String generatedClassName = "io.realm." + modelClassName + "RealmProxy";
401                         Class<?> generatedClass;
402                         try {
403                             generatedClass = Class.forName(generatedClassName);
404                         } catch (ClassNotFoundException e) {
405                             throw new RealmException("Could not find the generated " + generatedClassName + " class");
406                         }
407     
408                         // if not versioned, create table
409                         if (version == UNVERSIONED) {
410                             Method initTableMethod;
411                             try {
412                                 initTableMethod = generatedClass.getMethod("initTable", new Class[]{ImplicitTransaction.class});
413                             } catch (NoSuchMethodException e) {
414                                 throw new RealmException("Could not find the initTable method in the generated " + generatedClassName + " class");
415                             }
416                             try {
417                                 initTableMethod.invoke(null, realm.transaction);
418                             } catch (IllegalAccessException e) {
419                                 throw new RealmException("Could not execute the initTable method in the " + generatedClassName + " class");
420                             } catch (InvocationTargetException e) {
421                                 throw new RealmException("An exception was thrown in the initTable method in the " + generatedClassName + " class");
422                             }
423                         }
424     
425                         // validate created table
426                         Method validateMethod;
427                         try {
428                             validateMethod = generatedClass.getMethod("validateTable", new Class[]{ImplicitTransaction.class});
429                         } catch (NoSuchMethodException e) {
430                             throw new RealmException("Could not find the validateTable method in the generated " + generatedClassName + " class");
431                         }
432                         try {
433                             validateMethod.invoke(null, realm.transaction);
434                         } catch (IllegalAccessException e) {
435                             throw new RealmException("Could not execute the validateTable method in the " + generatedClassName + " class");
436                         } catch (InvocationTargetException e) {
437                             throw new RealmMigrationNeededException(e.getMessage(), e);
438                         }
439     
440                         // Populate the columnIndices table
441                         Method fieldNamesMethod;
442                         try {
443                             fieldNamesMethod = generatedClass.getMethod("getFieldNames");
444                         } catch (NoSuchMethodException e) {
445                             throw new RealmException("Could not find the getFieldNames method in the generated " + generatedClassName + " class");
446                         }
447                         List<String> fieldNames;
448                         try {
449                             //noinspection unchecked
450                             fieldNames = (List<String>) fieldNamesMethod.invoke(null);
451                         } catch (IllegalAccessException e) {
452                             throw new RealmException("Could not execute the getFieldNames method in the generated " + generatedClassName + " class");
453                         } catch (InvocationTargetException e) {
454                             throw new RealmException("An exception was thrown in the getFieldNames method in the generated " + generatedClassName + " class");
455                         }
456                         Table table = realm.transaction.getTable(TABLE_PREFIX + modelClassName);
457                         for (String fieldName : fieldNames) {
458                             long columnIndex = table.getColumnIndex(fieldName);
459                             if (columnIndex == -1) {
460                                 throw new RealmMigrationNeededException("Column '" + fieldName + "' not found for type '" + modelClassName + "'");
461                             }
462                             Map<String, Long> innerMap = columnIndices.get(modelClassName);
463                             if (innerMap == null) {
464                                 innerMap = new HashMap<String, Long>();
465                             }
466                             innerMap.put(fieldName, columnIndex);
467                             columnIndices.put(modelClassName, innerMap);
468                         }
469                     }
470                 } finally {
471                     realm.commitTransaction();
472                 }
473             }
474     
475             return realm;
476         }

478         /**
479          * Instantiates and adds a new object to the realm
480          *
481          * @param clazz The Class of the object to create
482          * @return The new object
483          * @throws RealmException An object could not be created
484          */
485         public <E extends RealmObject> E createObject(Class<E> clazz) {
486             Table table;
487             table = tables.get(clazz);
488             if (table == null) {
489                 String simpleClassName = simpleClassNames.get(clazz);
490                 if (simpleClassName == null) {
491                     simpleClassName = clazz.getSimpleName();
492                     simpleClassNames.put(clazz, simpleClassName);
493                 }
494                 String generatedClassName = "io.realm." + simpleClassName + "RealmProxy";
495     
496                 Class<?> generatedClass = generatedClasses.get(generatedClassName);
497                 if (generatedClass == null) {
498                     try {
499                         generatedClass = Class.forName(generatedClassName);
500                     } catch (ClassNotFoundException e) {
501                         throw new RealmException("Could not find the generated proxy class");
502                     }
503                     generatedClasses.put(generatedClassName, generatedClass);
504                 }
505     
506                 Method method = initTableMethods.get(generatedClass);
507                 if (method == null) {
508                     try {
509                         method = generatedClass.getMethod("initTable", new Class[]{ImplicitTransaction.class});
510                     } catch (NoSuchMethodException e) {
511                         throw new RealmException("Could not find the initTable() method in generated proxy class");
512                     }
513                     initTableMethods.put(generatedClass, method);
514                 }
515     
516                 try {
517                     table = (Table) method.invoke(null, transaction);
518                     tables.put(clazz, table);
519                 } catch (IllegalAccessException e) {
520                     throw new RealmException("Could not launch the initTable method");
521                 } catch (InvocationTargetException e) {
522                     e.printStackTrace();
523                     throw new RealmException("An exception occurred while running the initTable method");
524                 }
525             }
526     
527             long rowIndex = table.addEmptyRow();
528             return get(clazz, rowIndex);
529         }

535         <E extends RealmObject> E get(Class<E> clazz, long rowIndex) {
536             E result;
537     
538             Table table = tables.get(clazz);
539             if (table == null) {
540                 String simpleClassName = simpleClassNames.get(clazz);
541                 if (simpleClassName == null) {
542                     simpleClassName = clazz.getSimpleName();
543                     simpleClassNames.put(clazz, simpleClassName);
544                 }
545     
546                 table = transaction.getTable(TABLE_PREFIX + simpleClassName);
547                 tables.put(clazz, table);
548             }
549     
550             Row row = table.getRow(rowIndex);
551     
552             Constructor constructor = generatedConstructors.get(clazz);
553             if (constructor == null) {
554                 String simpleClassName = simpleClassNames.get(clazz);
555                 if (simpleClassName == null) {
556                     simpleClassName = clazz.getSimpleName();
557                     simpleClassNames.put(clazz, simpleClassName);
558                 }
559                 String generatedClassName = "io.realm." + simpleClassName + "RealmProxy";
560     
561     
562                 Class<?> generatedClass = generatedClasses.get(generatedClassName);
563                 if (generatedClass == null) {
564                     try {
565                         generatedClass = Class.forName(generatedClassName);
566                     } catch (ClassNotFoundException e) {
567                         throw new RealmException("Could not find the generated proxy class");
568                     }
569                     generatedClasses.put(generatedClassName, generatedClass);
570                 }
571     
572                 constructor = constructors.get(generatedClass);
573                 if (constructor == null) {
574                     try {
575                         constructor = generatedClass.getConstructor();
576                     } catch (NoSuchMethodException e) {
577                         throw new RealmException("Could not find the constructor in generated proxy class");
578                     }
579                     constructors.put(generatedClass, constructor);
580                     generatedConstructors.put(clazz, constructor);
581                 }
582             }
583     
584             try {
585                 // We are know the casted type since we generated the class
586                 //noinspection unchecked
587                 result = (E) constructor.newInstance();
588             } catch (InstantiationException e) {
589                 throw new RealmException("Could not instantiate the proxy class");
590             } catch (IllegalAccessException e) {
591                 throw new RealmException("Could not run the constructor of the proxy class");
592             } catch (InvocationTargetException e) {
593                 e.printStackTrace();
594                 throw new RealmException("An exception occurred while instantiating the proxy class");
595             }
596             result.row = row;
597             result.realm = this;
598             return result;
599         }

688         /**
689          * Starts a write transaction, this must be closed with {@link io.realm.Realm#commitTransaction()}
690          * or aborted by @{link io.realm.Realm#cancelTransaction()}. Write transactions are used to
691          * atomically create, update and delete objects within a realm.
692          * <br>
693          * Before beginning the write transaction, @{link io.realm.Realm#beginTransaction()} updates the
694          * realm in the case of pending updates from other threads.
695          * <br>
696          * Notice: it is not possible to nest write transactions. If you start a write
697          * transaction within a write transaction an exception is thrown.
698          * <br>
699          * @throws java.lang.IllegalStateException If already in a write transaction.
700          *
701          */
702         public void beginTransaction() {
703             transaction.promoteToWrite();
704         }
705     
706         /**
707          * All changes since @{link io.realm.Realm#beginTransaction()} are persisted to disk and the
708          * realm reverts back to being read-only. An event is sent to notify all other realm instances
709          * that a change has occurred. When the event is received, the other realms will get their
710          * objects and @{link io.realm.RealmResults} updated to reflect
711          * the changes from this commit.
712          * 
713          * @throws java.lang.IllegalStateException If the write transaction is in an invalid state.
714          */
715         public void commitTransaction() {
716             transaction.commitAndContinueAsRead();
717     
718             for (Map.Entry<Handler, Integer> handlerIntegerEntry : handlers.entrySet()) {
719                 Handler handler = handlerIntegerEntry.getKey();
720                 int realmId = handlerIntegerEntry.getValue();
721                 if (
722                         realmId == id                                // It's the right realm
723                         && !handler.hasMessages(REALM_CHANGED)       // The right message
724                         && handler.getLooper().getThread().isAlive() // The receiving thread is alive
725                 ) {
726                     handler.sendEmptyMessage(REALM_CHANGED);
727                 }
728             }
729         }
730     
731         /**
732          * Revert all writes (created, updated, or deleted objects) made in the current write
733          * transaction and end the transaction.
734          * <br>
735          * The realm reverts back to read-only.
736          * <br>
737          * Calling this when not in a write transaction will throw an exception.
738          *
739          * @throws java.lang.IllegalStateException    If the write transaction is an invalid state or
740         *                                             not in a write transaction.
741         */
742          public void cancelTransaction() {
743              transaction.rollbackAndContinueAsRead();
744          }

789         static public void migrateRealmAtPath(String realmPath, byte[] key, RealmMigration migration, boolean autoUpdate) {
790             Realm realm = Realm.createAndValidate(realmPath, key, false, autoUpdate);
791             realm.beginTransaction();
792             realm.setVersion(migration.execute(realm, realm.getVersion()));
793             realm.commitTransaction();
794     
795             Map<String, Realm> realms = realmsCache.get();
796             realms.put(realmPath, new Realm(realmPath, key, autoUpdate));
797             realmsCache.set(realms);
798         }

813         /**
814          * Delete the Realm file from the filesystem for a custom named Realm.
815          * The realm must be unused and closed before calling this method.
816          *
817          * @param context  an Android @{{@link android.content.Context}.
818          * @param fileName the name of the custom Realm (i.e. "myCustomRealm.realm").
819          * @return false if a file could not be deleted. The failing file will be logged.
820          */
821         public static boolean deleteRealmFile(Context context, String fileName) {
822             boolean result = true;
823             File writableFolder = context.getFilesDir();
824             List<File> filesToDelete = Arrays.asList(
825                     new File(writableFolder, fileName),
826                     new File(writableFolder, fileName + ".lock"));
827             for (File fileToDelete : filesToDelete) {
828                 if (fileToDelete.exists()) {
829                     boolean deleteResult = fileToDelete.delete();
830                     if (!deleteResult) {
831                         result = false;
832                         Log.w(TAG, "Could not delete the file " + fileToDelete);
833                     }
834                 }
835             }
836             return result;
837         }
