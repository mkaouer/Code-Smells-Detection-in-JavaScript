91         public void testNotificationsPlusSelfReceive() {
92             final AtomicInteger counter = new AtomicInteger(0);
93             final Queue<Handler> handlers = new ConcurrentLinkedQueue<Handler>();
94     
95             Thread listenerThread = new Thread() {
96                 @Override
97                 public void run() {
98                     Looper.prepare();
99                     Realm realm = Realm.getInstance(getContext());
100                     realm.addChangeListener(new RealmChangeListener() {
101                         @Override
102                         public void onChange() {
103                             counter.incrementAndGet();
104                             Looper.myLooper().quit();
105                         }
106                     });
107                     Looper.loop();
108                 }
109             };
110             listenerThread.start();
111     
112             Thread writerThread = new Thread() {
113     
114                 @Override
115                 public void run() {
116                     Looper.prepare();
117                     Realm realm = Realm.getInstance(getContext());
118                     realm.addChangeListener(new RealmChangeListener() {
119                         @Override
120                         public void onChange() {
121                             counter.incrementAndGet();
122                         }
123                     });
124                     Handler handler = new Handler();
125                     handlers.add(handler);
126                     Looper.loop();
127                 }
128             };
129             writerThread.start();
130             while (handlers.isEmpty()) {
131                 try {
132                     Thread.sleep(10);
133                 } catch (InterruptedException e) {
134                     fail();
135                 }
136             }
137             Handler handler = handlers.poll();
138             handler.post(new Runnable() {
139                 @Override
140                 public void run() {
141                     Realm realm = Realm.getInstance(getContext());
142                     realm.beginTransaction();
143                     Dog dog = realm.createObject(Dog.class);
144                     dog.setName("Rex");
145                     realm.commitTransaction();
146                     Looper.myLooper().quit();
147                 }
148             });
149     
150             try {
151                 writerThread.join();
152                 listenerThread.join(2000);
153             } catch (InterruptedException e) {
154                 fail();
155             }
156     
157             assertEquals(2, counter.get());
158         }

184         public void testSetAutoRefreshOnHandlerThread() {
185             final AtomicBoolean done = new AtomicBoolean(false);
186             HandlerThread thread = new HandlerThread("TestThread") {
187                 @Override
188                 protected void onLooperPrepared() {
189                     Realm realm = Realm.getInstance(getContext());
190                     assertTrue(realm.isAutoRefresh());
191                     realm.setAutoRefresh(false);

198             thread.start();
