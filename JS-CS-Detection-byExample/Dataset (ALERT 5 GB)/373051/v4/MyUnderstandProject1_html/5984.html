24     /**
25      * This class is a base class for all TightDB tables. The class supports all low
26      * level methods (define/insert/delete/update) a table has. All the native
27      * communications to the TightDB C++ library are also handled by this class.
28      *
29      * A user who wants to create a table of his choice will automatically inherit
30      * from this class by the realm-class generator.
31      *
32      * As an example, let's create a table to keep records of an employee in a
33      * company.
34      *
35      * For this purpose we will create a class named "employee" with an Entity
36      * annotation as follows.
37      *
38      *      @DefineTable
39      *      public class employee {
40      *          String name;
41      *          long age;
42      *          boolean hired;
43      *          byte[] imageData;
44      *      }
45      *
46      * The realm class generator will generate classes relevant to the employee:
47      *
48      * 1. Employee.java:  Represents one employee of the employee table i.e., a single row. Getter/setter
49      *                    methods are declared from which you will be able to set/get values
50      *                    for a particular employee.
51      * 2. EmployeeTable.java:  Represents the class for storing a collection of employee i.e., a table
52      *                    of rows. The class is inherited from the TableBase class as described above.
53      *                    It has all the high level methods to manipulate Employee objects from the table.
54      * 3. EmployeeView.java: Represents view of the employee table i.e., result set of queries.
55      *
56      *
57      */
58

77         /**
78          * Construct a Table base object. It can be used to register columns in this
79          * table. Registering into table is allowed only for empty tables. It
80          * creates a native reference of the object and keeps a reference to it.
81          */
82         public Table() {
83             this.parent = null; // No parent in free-standing table
84             this.context = new Context();
85             // Native methods work will be initialized here. Generated classes will
86             // have nothing to do with the native functions. Generated Java Table
87             // classes will work as a wrapper on top of table.
88             this.nativePtr = createNative();
89             if (nativePtr == 0)
90                 throw new java.lang.OutOfMemoryError("Out of native memory.");
91             if (DEBUG) {
92                 tableNo = ++TableCount;
93                 System.err.println("====== New Tablebase " + tableNo + " : ptr = " + nativePtr);
94             }
95         }

104                 tableNo = ++TableCount;
105                 System.err.println("===== New Tablebase(ptr) " + tableNo + " : ptr = " + nativePtr);
106             }

109         // If close() is called, no penalty is paid for delayed disposal
110         // via the context
111         @Override
112         public void close() {
113             synchronized (context) {
114                 if (nativePtr != 0) {
115                     nativeClose(nativePtr);
116                     
117                     if (DEBUG) {
118                         TableCount--;
119                         System.err.println("==== CLOSE " + tableNo + " ptr= " + nativePtr + " remaining " + TableCount);
120                     }
121                     
122                     nativePtr = 0;
123                 }   
124             }
125         }

129         @Override

131             synchronized (context) {
132                 if (nativePtr != 0) {

134                     context.asyncDisposeTable(nativePtr, isRoot);
135                     nativePtr = 0; // Set to 0 if finalize is called before close() for some reason
136                 }
137             }

407         /**
408          * Inserts a row at the specified row index. Shifts the row currently at that row index and any subsequent rows down (adds one to their row index).
409          * @param rowIndex
410          * @param values
411          */
412         public void addAt(long rowIndex, Object... values) {
413             checkImmutable();
414     
415             // Check index
416             long size = size();
417             if (rowIndex > size) {
418                 throw new IllegalArgumentException("rowIndex " + String.valueOf(rowIndex) +
419                         " must be <= table.size() " + String.valueOf(size) + ".");
420             }
421     
422             // Check values types
423             int columns = (int)getColumnCount();
424             if (columns != values.length) {
425                 throw new IllegalArgumentException("The number of value parameters (" +
426                         String.valueOf(values.length) +
427                         ") does not match the number of columns in the table (" +
428                         String.valueOf(columns) + ").");
429             }
430             ColumnType colTypes[] = new ColumnType[columns];
431             for (int columnIndex = 0; columnIndex < columns; columnIndex++) {
432                 Object value = values[columnIndex];
433                 ColumnType colType = getColumnType(columnIndex);
434                 colTypes[columnIndex] = colType;
435                 if (!colType.matchObject(value)) {
436                     //String representation of the provided value type
437                     String providedType;
438                     if (value == null)
439                         providedType = "null";
440                     else
441                         providedType = value.getClass().toString();
442     
443                     throw new IllegalArgumentException("Invalid argument no " + String.valueOf(1 + columnIndex) +
444                             ". Expected a value compatible with column type " + colType + ", but got " + providedType + ".");
445                 }
446             }
447     
448             // Insert values
449             for (long columnIndex = 0; columnIndex < columns; columnIndex++) {
450                 Object value = values[(int)columnIndex];
451                 switch (colTypes[(int)columnIndex]) {
452                 case BOOLEAN:
453                     nativeInsertBoolean(nativePtr, columnIndex, rowIndex, (Boolean)value);
454                     break;
455                 case INTEGER:
456                     nativeInsertLong(nativePtr, columnIndex, rowIndex, ((Number)value).longValue());
457                     break;
458                 case FLOAT:
459                     nativeInsertFloat(nativePtr, columnIndex, rowIndex, ((Float)value).floatValue());
460                     break;
461                 case DOUBLE:
462                     nativeInsertDouble(nativePtr, columnIndex, rowIndex, ((Double)value).doubleValue());
463                     break;
464                 case STRING:
465                     nativeInsertString(nativePtr, columnIndex, rowIndex, (String)value);
466                     break;
467                 case DATE:
468                     nativeInsertDate(nativePtr, columnIndex, rowIndex, ((Date)value).getTime()/1000);
469                     break;
470                 case MIXED:
471                     nativeInsertMixed(nativePtr, columnIndex, rowIndex, Mixed.mixedValue(value));
472                     break;
473                 case BINARY:
474                     nativeInsertByteArray(nativePtr, columnIndex, rowIndex, (byte[])value);
475                     break;
476                 case TABLE:
477                     nativeInsertSubtable(nativePtr, columnIndex, rowIndex);
478                     insertSubtableValues(rowIndex, columnIndex, value);
479                     break;
480                 default:
481                     throw new RuntimeException("Unexpected columnType: " + String.valueOf(colTypes[(int)columnIndex]));
482                 }
483             }
484             //Insert done. Use native, no need to check for immutable again here
485             nativeInsertDone(nativePtr);
486     
487         }

489         private void insertSubtableValues(long rowIndex, long columnIndex, Object value) {
490             if (value != null) {
491                 // insert rows in subtable recursively
492                 Table subtable = getSubtableDuringInsert(columnIndex, rowIndex);
493                 int rows = ((Object[])value).length;
494                 for (int i=0; i<rows; ++i) {
495                     Object rowArr = ((Object[])value)[i];
496                     subtable.addAt(i, (Object[])rowArr);
497                 }
498             }
499         }

550         /**
551          * Replaces the row at the specified position with the specified row.
552          * @param rowIndex
553          * @param values
554          */
555         public void set(long rowIndex, Object... values) {
556             checkImmutable();
557     
558             // Check index
559             long size = size();
560             if (rowIndex >= size) {
561                 throw new IllegalArgumentException("rowIndex " + String.valueOf(rowIndex) +
562                         " must be < table.size() " + String.valueOf(size) + ".");
563             }
564     
565             // Verify number of 'values'
566             int columns = (int)getColumnCount();
567             if (columns != values.length) {
568                 throw new IllegalArgumentException("The number of value parameters (" +
569                         String.valueOf(values.length) +
570                         ") does not match the number of columns in the table (" +
571                         String.valueOf(columns) + ").");
572             }
573             // Verify type of 'values'
574             ColumnType colTypes[] = new ColumnType[columns];
575             for (int columnIndex = 0; columnIndex < columns; columnIndex++) {
576                 Object value = values[columnIndex];
577                 ColumnType colType = getColumnType(columnIndex);
578                 colTypes[columnIndex] = colType;
579                 if (!colType.matchObject(value)) {
580                     throw new IllegalArgumentException("Invalid argument no " + String.valueOf(1 + columnIndex) +
581                             ". Expected a value compatible with column type " + colType + ", but got " + value.getClass() + ".");
582                 }
583             }
584     
585             // Now that all values are verified, we can remove the row and insert it again.
586             // TODO: Can be optimized to only set the values (but clear any subtables)
587             remove(rowIndex);
588             addAt(rowIndex, values);
589         }
