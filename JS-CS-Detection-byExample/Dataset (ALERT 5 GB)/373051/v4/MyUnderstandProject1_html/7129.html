17     package io.realm.internal;
18     
19     public class ImplicitTransaction extends Group {
20     
21         private final SharedGroup parent;
22     
23         public ImplicitTransaction(Context context, SharedGroup sharedGroup, long nativePtr) {
24             super(context, nativePtr, true);
25             parent = sharedGroup;
26         }
27     
28         public void advanceRead() {
29             parent.advanceRead();
30         }
31     
32         public void promoteToWrite() {
33             if (immutable) {
34                 immutable = false;
35                 parent.promoteToWrite();
36             } else {
37                 throw new IllegalStateException("Nested transactions are not allowed. Use commitTransaction() after each beginTransaction().");
38             }
39         }
40     
41         public void commitAndContinueAsRead() {
42             parent.commitAndContinueAsRead();
43             immutable = true;
44         }
45     
46         public void endRead() {
47             parent.endRead();
48         }
49     
50         public void rollbackAndContinueAsRead() {
51             if (!immutable) {
52                 parent.rollbackAndContinueAsRead();
53                 immutable = true;
54             } else {
55                 throw new IllegalStateException("Cannot cancel a non-write transaction.");
56             }
57         }
58     
59         protected void finalize() {} // Nullify the actions of Group.finalize()
60     
61     }
