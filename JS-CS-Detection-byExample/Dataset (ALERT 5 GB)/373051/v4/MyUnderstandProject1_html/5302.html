16     package io.realm.internal;
17     
18     import junit.framework.Test;
19     import junit.framework.TestCase;
20     
21     import java.util.Arrays;
22     import java.util.Collection;
23     
24     public class JNIParameterizedTestExample extends TestCase {
25     
26         private int input_one;
27         private int input_two;
28         private int expected_value;
29     
30         //set up parameters for tests
31         public static Collection<Object[]> parameters() {
32             return Arrays.asList(
33                     new Object[]{2, 1, 3},
34                     new Object[]{1, 1, 2},
35                     new Object[]{-2, 1, -1}
36             );
37         }
38     
39         // {2, 1, 2} will be set to 2= input_one, 1= input_two and 3=expected_value
40         public JNIParameterizedTestExample(int input_one, int input_two, int expected_value) {
41             this.input_one = input_one;
42             this.input_two = input_two;
43             this.expected_value = expected_value;
44         }
45     
46         // The test suite is set to method name must start with 'test',return type void and empty parameters.
47         public void testIntAdd() {
48             final int actual_value = input_one + input_two;
49             System.out.println(expected_value + "+" + actual_value);
50             assertEquals(expected_value, actual_value);
51         }
52     
53         //Returns the test class and parameters
54         public static Test suite() {
55             return new JNITestSuite(JNIParameterizedTestExample.class, parameters());
56     
57         }
58     
59     
60     }
