33     JNIEXPORT jlong JNICALL Java_io_realm_internal_Group_createNative__Ljava_lang_String_2I(
34         JNIEnv* env, jobject, jstring jFileName, jint mode, jbyteArray keyArray)
35     {
36         TR((env, "Group::createNative(file): "));
37         const char* fileNameCharPtr = env->GetStringUTFChars(jFileName, NULL);
38         if (fileNameCharPtr == NULL)
39             return 0;        // Exception is thrown by GetStringUTFChars()
40     
41         Group* pGroup = 0;
42         try {
43             Group::OpenMode openmode;
44             switch (mode) {
45             case 0: openmode = Group::mode_ReadOnly; break;
46             case 1: openmode = Group::mode_ReadWrite; break;
47             case 2: openmode = Group::mode_ReadWriteNoCreate; break;
48             default:
49                 TR((env, "Invalid mode: %d\n", mode));
50                 ThrowException(env, IllegalArgument, "Group(): Invalid mode parameter.");
51                 return 0;
52             }
53     
54             KeyBuffer key(env, keyArray);
55     #ifdef TIGHTDB_ENABLE_ENCRYPTION
56             pGroup = new Group(fileNameCharPtr, key.data(), openmode);
57     #else
58             pGroup = new Group(fileNameCharPtr, openmode);
59     #endif
60     
61             TR((env, "%x\n", pGroup));
62             return reinterpret_cast<jlong>(pGroup);
63         }
64         CATCH_FILE(fileNameCharPtr)
65         CATCH_STD()
66     
67         // Failed - cleanup
68         if (pGroup)
69             delete pGroup;
70         return 0;
71     }

73     JNIEXPORT jlong JNICALL Java_io_realm_internal_Group_createNative___3B(
74         JNIEnv* env, jobject, jbyteArray jData)
75     {
76         TR((env, "Group::createNative(byteArray): "));
77         // Copy the group buffer given
78         jsize byteArrayLength = env->GetArrayLength(jData);
79         if (byteArrayLength == 0)
80             return 0;
81         jbyte* buf = static_cast<jbyte*>(malloc(S(byteArrayLength)*sizeof(jbyte)));
82         if (!buf) {
83             ThrowException(env, OutOfMemory, "copying the group buffer.");
84             return 0;
85         }
86         env->GetByteArrayRegion(jData, 0, byteArrayLength, buf);
87     
88         TR((env, " %d bytes.", byteArrayLength));
89         Group* pGroup = 0;
90         try {
91             pGroup = new Group(BinaryData(reinterpret_cast<char*>(buf), S(byteArrayLength)), true);
92             TR((env, " groupPtr: %x\n", pGroup));
93             return reinterpret_cast<jlong>(pGroup);
94         }
95         CATCH_FILE("memory-buffer")
96         CATCH_STD()
97     
98         // Failed - cleanup
99         if (buf)
100             free(buf);
101         return 0;
102     }

168     JNIEXPORT void JNICALL Java_io_realm_internal_Group_nativeWriteToFile(
169         JNIEnv* env, jobject, jlong nativeGroupPtr, jstring jFileName, jbyteArray keyArray)
170     {
171         const char* fileNameCharPtr = env->GetStringUTFChars(jFileName, NULL);
172         if (fileNameCharPtr) {
173             KeyBuffer key(env, keyArray);
174             try {
175     #ifdef TIGHTDB_ENABLE_ENCRYPTION
176                 G(nativeGroupPtr)->write(fileNameCharPtr, key.data());
177     #else
178                 G(nativeGroupPtr)->write(fileNameCharPtr);
179     #endif
180             }
181             CATCH_FILE(fileNameCharPtr)
182             CATCH_STD()
183     
184             env->ReleaseStringUTFChars(jFileName, fileNameCharPtr);
185         }
186         // (exception is thrown by GetStringUTFChars if it fails.)
187     }

189     JNIEXPORT jbyteArray JNICALL Java_io_realm_internal_Group_nativeWriteToMem(
190         JNIEnv* env, jobject, jlong nativeGroupPtr)
191     {
192         TR((env, "nativeWriteToMem(%x)\n", nativeGroupPtr));
193         BinaryData buffer;
194         char* bufPtr = 0;
195         try {
196             buffer = G(nativeGroupPtr)->write_to_mem(); // throws
197             bufPtr = const_cast<char*>(buffer.data());
198             // Copy the data to Java array, so Java owns it.
199             jbyteArray jArray = 0;
200             if (buffer.size() <= MAX_JSIZE) {
201                 jsize jlen = static_cast<jsize>(buffer.size());
202                 jArray = env->NewByteArray(jlen);
203                 if (jArray)
204                     // Copy data to Byte[]
205                     env->SetByteArrayRegion(jArray, 0, jlen, reinterpret_cast<const jbyte*>(bufPtr));
206                     // SetByteArrayRegion() may throw ArrayIndexOutOfBoundsException - logic error
207             }
208             if (!jArray) {
209                 ThrowException(env, IndexOutOfBounds, "Group too big to copy and write.");
210             }
211             free(bufPtr);
212             return jArray;
213         }
214         CATCH_STD()
215         if (bufPtr)
216             free(bufPtr);
217         return 0;
218     }

220     JNIEXPORT jobject JNICALL Java_io_realm_internal_Group_nativeWriteToByteBuffer(
221         JNIEnv* env, jobject, jlong nativeGroupPtr)
222     {
223         TR((env, "nativeWriteToByteBuffer(%x)\n", nativeGroupPtr));
224         BinaryData buffer;
225         try {
226             buffer = G(nativeGroupPtr)->write_to_mem();
227             if (util::int_less_than_or_equal(buffer.size(), MAX_JLONG)) {
228                 return env->NewDirectByteBuffer(const_cast<char*>(buffer.data()), static_cast<jlong>(buffer.size()));
229                 // Data is now owned by the Java DirectByteBuffer - so we must not free it.
230             }
231             else {
232                 ThrowException(env, IndexOutOfBounds, "Group too big to write.");
233                 return NULL;
234             }
235         }
236         CATCH_STD()
237         return NULL;
238     }

248         JNIEnv* env, jobject, jlong nativeGroupPtr)
249     {
250         Group* grp = G(nativeGroupPtr);
251     
252         try {
253             // Write group to string in JSON format
254             std::ostringstream ss;
255             ss.sync_with_stdio(false); // for performance
256             grp->to_json(ss);
257             const std::string str = ss.str();
258             return env->NewStringUTF(str.c_str());
259         } CATCH_STD()
