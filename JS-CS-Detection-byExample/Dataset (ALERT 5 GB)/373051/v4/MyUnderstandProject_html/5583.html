42         private void checkNativePtrNotZero() {
43             if (this.nativePtr == 0)
44                 // FIXME: It is wrong to assume that a null pointer means 'out
45                 // of memory'. An out of memory condition in
46                 // createNative() must be handled by having createNative()
47                 // throw OutOfMemoryError.
48                 throw new OutOfMemoryError("Out of native memory.");
49         }

51         public Group() {
52             this.immutable = false;
53             this.context = new Context();
54             this.nativePtr = createNative();
55             checkNativePtrNotZero();
56             
57         }

59         protected native long createNative();

61         public enum OpenMode {
62             // Below values must match the values in realm::group::OpenMode in C++
63             READ_ONLY(0),
64             READ_WRITE(1),
65             READ_WRITE_NO_CREATE(2);
66             private int value;
67             private OpenMode(int value) {
68                 this.value = value;
69             }
70         };
71     
72         public Group(String filepath, OpenMode mode) {
73             if (mode.equals(OpenMode.READ_ONLY))
74                 this.immutable = true;
75             else
76                 this.immutable = false;
77             
78             this.context = new Context();
79             this.nativePtr = createNative(filepath, mode.value);
80             checkNativePtrNotZero();
81         }

85         public Group(String filepath) {
86             this(filepath, OpenMode.READ_ONLY);
87         }

89         public Group(File file) {
90             this(file.getAbsolutePath(), file.canWrite() ? OpenMode.READ_WRITE : OpenMode.READ_ONLY);
91         }

94         public Group(byte[] data) {

126         // If close() is called, no penalty is paid for delayed disposal
127         // via the context
128         public void close() {
129             synchronized (context) {
130                 if (nativePtr != 0) {
131                     nativeClose(nativePtr);
132                     nativePtr = 0;
133                 }
134             }
135         }
136     
137         protected static native void nativeClose(long nativeGroupPtr);
138         
139         protected void finalize() {
140             synchronized (context) {
141                 if (nativePtr != 0) {
142                     context.asyncDisposeGroup(nativePtr);
143                     nativePtr = 0; // Set to 0 if finalize is called before close() for some reason
144                 }
145             }
146         }

153             if (nativePtr == 0)

187         public String getTableName(int index) {
188             verifyGroupIsValid();
189             long cnt = size();
190             if (index < 0 || index >= cnt) {
191                 throw new IndexOutOfBoundsException(
192                         "Table index argument is out of range. possible range is [0, "
193                         + (cnt - 1) + "]");
194             }
195             return nativeGetTableName(nativePtr, index);
196         }

200         /**
201          * Returns a table with the specified name.
202          *
203          * @param name
204          *            The name of the table.
205          * @return The table if it exists, otherwise create it.
206          */
207         public Table getTable(String name) {
208             verifyGroupIsValid();
209             if (name == null || name.equals(""))
210                 throw new IllegalArgumentException("Invalid name. Name must be a non-empty String.");
211             if (immutable)
212                 if (!hasTable(name))
213                     throwImmutable();
214             
215             // Execute the disposal of abandoned realm objects each time a new realm object is created
216             context.executeDelayedDisposal();
217             long nativeTablePointer = nativeGetTableNativePtr(nativePtr, name);
218             try {
219                 // Copy context reference from parent
220                 return new Table(context, this, nativeTablePointer);
221             }
222             catch (RuntimeException e) {
223                 Table.nativeClose(nativeTablePointer);
224                 throw e;
225             }
226         }
227     
228         protected native long nativeGetTableNativePtr(long nativeGroupPtr, String name);

303         public boolean equals(Object other) {
304             if (other == null)
305                 return false;
306             if (other == this)
307                 return true;
