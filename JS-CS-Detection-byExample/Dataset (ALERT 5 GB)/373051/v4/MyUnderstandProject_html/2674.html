196         public void testSortByLong() {
197             RealmResults<AllTypes> resultList = testRealm.where(AllTypes.class).findAll();
198             RealmResults<AllTypes> sortedList = resultList.sort(FIELD_LONG, RealmResults.SORT_ORDER_DECENDING);
199             assertEquals("Should have same size", resultList.size(), sortedList.size());
200             assertEquals(TEST_DATA_SIZE, sortedList.size());
201             assertEquals("First excepted to be last", resultList.first().getColumnLong(), sortedList.last().getColumnLong());
202     
203             RealmResults<AllTypes> reverseList = sortedList.sort(FIELD_LONG, RealmResults.SORT_ORDER_ASCENDING);
204             assertEquals(TEST_DATA_SIZE, reverseList.size());
205             assertEquals("First excepted to be first", resultList.first().getColumnLong(), reverseList.first().getColumnLong());
206             assertEquals("Last excepted to be last", resultList.last().getColumnLong(), reverseList.last().getColumnLong());
207     
208             RealmResults<AllTypes> reserveSortedList = reverseList.sort(FIELD_LONG, RealmResults.SORT_ORDER_DECENDING);
209             assertEquals(TEST_DATA_SIZE, reserveSortedList.size());

228         public void testSortByBoolean() {
229             RealmResults<AllTypes> resultList = testRealm.where(AllTypes.class).findAll();
230             RealmResults<AllTypes> sortedList = resultList.sort(FIELD_BOOLEAN, RealmResults.SORT_ORDER_DECENDING);
231             assertEquals("Should have same size", resultList.size(), sortedList.size());
232             assertEquals(TEST_DATA_SIZE, sortedList.size());
233             assertEquals("Last expected to be false", false, sortedList.last().isColumnBoolean());
234             assertEquals("First expected to be true", true, sortedList.first().isColumnBoolean());
235             assertEquals("Expected to be true", true, sortedList.get(147).isColumnBoolean());
236             assertEquals("Expected to be false", false, sortedList.get(368).isColumnBoolean());
237     
238             RealmResults<AllTypes> reverseList = sortedList.sort(FIELD_BOOLEAN, RealmResults.SORT_ORDER_ASCENDING);
239             assertEquals(TEST_DATA_SIZE, reverseList.size());
240             assertEquals("Last expected to be true", true, reverseList.last().isColumnBoolean());
241             assertEquals("First expected to be false", false, reverseList.first().isColumnBoolean());
242             assertEquals("Expected to be false", false, reverseList.get(147).isColumnBoolean());
243             assertEquals("Expected to be true", true, reverseList.get(368).isColumnBoolean());

245             RealmResults<AllTypes> reserveSortedList = reverseList.sort(FIELD_BOOLEAN, RealmResults.SORT_ORDER_DECENDING);
