46     /**
47      * The Realm class is the storage and transactional manager of your object persistent store. Objects
48      * are created. Objects within a Realm can be queried and read at any time. Creating,
49      * modifying, and deleting objects must be done through transactions.
50      * <p/>
51      * The transactions ensure that multiple instances (on multiple threads) can access the objects
52      * in a consistent state with full ACID guaranties.
53      * <p/>
54      * The instances of a Realm will be automatically updated when one instance commits a
55      * change (create, modify or delete an object).
56      */

92         private Realm(String absolutePath, byte[] key) {
93             this.sharedGroup = new SharedGroup(absolutePath, true, key);
94             this.transaction = sharedGroup.beginImplicitTransaction();
95             this.id = absolutePath.hashCode();
96             if (!looperThread.isAlive()) {
97                 looperThread.start();
98             }
99     
100             if (Looper.myLooper() == null) {
101                 Looper.prepare();
102             }
103             handler = new Handler() {
104                 @Override
105                 public void handleMessage(Message message) {
106                     if (message.what == LooperThread.REALM_CHANGED) {
107                         if (autoRefresh) {
108                             transaction.advanceRead();
109                         }
110                         sendNotifications();
111                     }
112                 }
113             };
114             if (Looper.myLooper() == null) {
115                 Looper.loop();
116             }
117             LooperThread.handlers.put(handler, id);
118         }

121         protected void finalize() throws Throwable {

234         private static Realm createAndValidate(String absolutePath, byte[] key, boolean validateSchema) {
235             Map<String, Realm> realms = realmsCache.get();
236             Realm realm = realms.get(absolutePath);
237     
238             if (realm == null) {
239                 realm = new Realm(absolutePath, key);
240                 realms.put(absolutePath, realm);
241                 realmsCache.set(realms);
242             }
243             if (validateSchema) {
244                 Class<?> validationClass;
245                 try {
246                     validationClass = Class.forName("io.realm.ValidationList");
247                 } catch (ClassNotFoundException e) {
248                     throw new RealmException("Could not find the generated ValidationList class");
249                 }
250                 Method getProxyClassesMethod;
251                 try {
252                     getProxyClassesMethod = validationClass.getMethod("getProxyClasses");
253                 } catch (NoSuchMethodException e) {
254                     throw new RealmException("Could not find the getProxyClasses method in the ValidationList class");
255                 }
256                 List<String> proxyClasses;
257                 try {
258                     //noinspection unchecked
259                     proxyClasses = (List<String>) getProxyClassesMethod.invoke(null);
260                 } catch (IllegalAccessException e) {
261                     throw new RealmException("Could not execute the getProxyClasses method in the ValidationList class");
262                 } catch (InvocationTargetException e) {
263                     throw new RealmException("An exception was thrown in the getProxyClasses method in the ValidationList class");
264                 }
265     
266                 long version = realm.getVersion();
267                 try {
268                     realm.beginTransaction();
269                     if (version == UNVERSIONED) {
270                         realm.setVersion(0);
271                     }
272     
273                     for (String className : proxyClasses) {
274                         String[] splitted = className.split("\\.");
275                         String modelClassName = splitted[splitted.length - 1];
276                         String generatedClassName = "io.realm." + modelClassName + "RealmProxy";
277                         Class<?> generatedClass;
278                         try {
279                             generatedClass = Class.forName(generatedClassName);
280                         } catch (ClassNotFoundException e) {
281                             throw new RealmException("Could not find the generated " + generatedClassName + " class");
282                         }
283     
284                         // if not versioned, create table
285                         if (version == UNVERSIONED) {
286                             Method initTableMethod;
287                             try {
288                                 initTableMethod = generatedClass.getMethod("initTable", new Class[]{ImplicitTransaction.class});
289                             } catch (NoSuchMethodException e) {
290                                 throw new RealmException("Could not find the initTable method in the generated " + generatedClassName + " class");
291                             }
292                             try {
293                                 initTableMethod.invoke(null, realm.transaction);
294                             } catch (IllegalAccessException e) {
295                                 throw new RealmException("Could not execute the initTable method in the " + generatedClassName + " class");
296                             } catch (InvocationTargetException e) {
297                                 throw new RealmException("An exception was thrown in the initTable method in the " + generatedClassName + " class");
298                             }
299                         }
300     
301                         // validate created table
302                         Method validateMethod;
303                         try {
304                             validateMethod = generatedClass.getMethod("validateTable", new Class[]{ImplicitTransaction.class});
305                         } catch (NoSuchMethodException e) {
306                             throw new RealmException("Could not find the validateTable method in the generated " + generatedClassName + " class");
307                         }
308                         try {
309                             validateMethod.invoke(null, realm.transaction);
310                         } catch (IllegalAccessException e) {
311                             throw new RealmException("Could not execute the validateTable method in the " + generatedClassName + " class");
312                         } catch (InvocationTargetException e) {
313                             throw new RealmMigrationNeededException(e.getMessage(), e);
314                         }
315     
316                         // Populate the columnIndices table
317                         Method fieldNamesMethod;
318                         try {
319                             fieldNamesMethod = generatedClass.getMethod("getFieldNames");
320                         } catch (NoSuchMethodException e) {
321                             throw new RealmException("Could not find the getFieldNames method in the generated " + generatedClassName + " class");
322                         }
323                         List<String> fieldNames;
324                         try {
325                             //noinspection unchecked
326                             fieldNames = (List<String>) fieldNamesMethod.invoke(null);
327                         } catch (IllegalAccessException e) {
328                             throw new RealmException("Could not execute the getFieldNames method in the generated " + generatedClassName + " class");
329                         } catch (InvocationTargetException e) {
330                             throw new RealmException("An exception was thrown in the getFieldNames method in the generated " + generatedClassName + " class");
331                         }
332                         Table table = realm.transaction.getTable(TABLE_PREFIX + modelClassName);
333                         for (String fieldName : fieldNames) {
334                             long columnIndex = table.getColumnIndex(fieldName);
335                             if (columnIndex == -1) {
336                                 throw new RealmMigrationNeededException("Column '" + fieldName + "' not found for type '" + modelClassName + "'");
337                             }
338                             Map<String, Long> innerMap = columnIndices.get(modelClassName);
339                             if (innerMap == null) {
340                                 innerMap = new HashMap<String, Long>();
341                             }
342                             innerMap.put(fieldName, columnIndex);
343                             columnIndices.put(modelClassName, innerMap);
344                         }
345                     }
346                 } finally {
347                     realm.commitTransaction();
348                 }
349             }
350     
351             return realm;
352         }

354         /**
355          * Instantiates and adds a new object to the realm
356          *
357          * @param clazz The Class of the object to create
358          * @return The new object
359          * @throws RealmException An object could not be created
360          */
361         public <E extends RealmObject> E createObject(Class<E> clazz) {
362             Table table;
363             table = tables.get(clazz);
364             if (table == null) {
365                 String simpleClassName = simpleClassNames.get(clazz);
366                 if (simpleClassName == null) {
367                     simpleClassName = clazz.getSimpleName();
368                     simpleClassNames.put(clazz, simpleClassName);
369                 }
370                 String generatedClassName = "io.realm." + simpleClassName + "RealmProxy";
371     
372                 Class<?> generatedClass = generatedClasses.get(generatedClassName);
373                 if (generatedClass == null) {
374                     try {
375                         generatedClass = Class.forName(generatedClassName);
376                     } catch (ClassNotFoundException e) {
377                         throw new RealmException("Could not find the generated proxy class");
378                     }
379                     generatedClasses.put(generatedClassName, generatedClass);
380                 }
381     
382                 Method method = initTableMethods.get(generatedClass);
383                 if (method == null) {
384                     try {
385                         method = generatedClass.getMethod("initTable", new Class[]{ImplicitTransaction.class});
386                     } catch (NoSuchMethodException e) {
387                         throw new RealmException("Could not find the initTable() method in generated proxy class");
388                     }
389                     initTableMethods.put(generatedClass, method);
390                 }
391     
392                 try {
393                     table = (Table) method.invoke(null, transaction);
394                     tables.put(clazz, table);
395                 } catch (IllegalAccessException e) {
396                     throw new RealmException("Could not launch the initTable method");
397                 } catch (InvocationTargetException e) {
398                     e.printStackTrace();
399                     throw new RealmException("An exception occurred while running the initTable method");
400                 }
401             }
402     
403             long rowIndex = table.addEmptyRow();
404             return get(clazz, rowIndex);
405         }

411         <E extends RealmObject> E get(Class<E> clazz, long rowIndex) {
412             E result;
413     
414             Table table = tables.get(clazz);
415             if (table == null) {
416                 String simpleClassName = simpleClassNames.get(clazz);
417                 if (simpleClassName == null) {
418                     simpleClassName = clazz.getSimpleName();
419                     simpleClassNames.put(clazz, simpleClassName);
420                 }
421     
422                 table = transaction.getTable(TABLE_PREFIX + simpleClassName);
423                 tables.put(clazz, table);
424             }
425     
426             Row row = table.getRow(rowIndex);
427     
428             Constructor constructor = generatedConstructors.get(clazz);
429             if (constructor == null) {
430                 String simpleClassName = simpleClassNames.get(clazz);
431                 if (simpleClassName == null) {
432                     simpleClassName = clazz.getSimpleName();
433                     simpleClassNames.put(clazz, simpleClassName);
434                 }
435                 String generatedClassName = "io.realm." + simpleClassName + "RealmProxy";
436     
437     
438                 Class<?> generatedClass = generatedClasses.get(generatedClassName);
439                 if (generatedClass == null) {
440                     try {
441                         generatedClass = Class.forName(generatedClassName);
442                     } catch (ClassNotFoundException e) {
443                         throw new RealmException("Could not find the generated proxy class");
444                     }
445                     generatedClasses.put(generatedClassName, generatedClass);
446                 }
447     
448                 constructor = constructors.get(generatedClass);
449                 if (constructor == null) {
450                     try {
451                         constructor = generatedClass.getConstructor();
452                     } catch (NoSuchMethodException e) {
453                         throw new RealmException("Could not find the constructor in generated proxy class");
454                     }
455                     constructors.put(generatedClass, constructor);
456                     generatedConstructors.put(clazz, constructor);
457                 }
458             }
459     
460             try {
461                 // We are know the casted type since we generated the class
462                 //noinspection unchecked
463                 result = (E) constructor.newInstance();
464             } catch (InstantiationException e) {
465                 throw new RealmException("Could not instantiate the proxy class");
466             } catch (IllegalAccessException e) {
467                 throw new RealmException("Could not run the constructor of the proxy class");
468             } catch (InvocationTargetException e) {
469                 e.printStackTrace();
470                 throw new RealmException("An exception occurred while instantiating the proxy class");
471             }
472             result.realmSetRow(row);
473             result.setRealm(this);
474             return result;
475         }

565         /**
566          * Starts a write transaction, this must be closed with commitTransaction() or aborted
567          * by cancelTransaction(). Write transactions are used to atomically create, update and delete
568          * objects within a realm.
569          *
570          * Before beginning the write transaction, beginTransaction() updates the
571          * realm in the case of pending updates from other threads.
572          *
573          * Notice: it is not possible to nest write transactions. If you start a write
574          * transaction within a write transaction an exception is thrown.
575          *
576          * @throws java.lang.IllegalStateException If already in a write transaction.
577          *
578          */
579         public void beginTransaction() {
580             transaction.promoteToWrite();
581         }
582     
583         /**
584          * All changes since beginTransaction() are persisted to disk and the realm reverts back to being read-only,
585          * An event is sent to notify all other realm instances that a change has occured.
586          * When the event is received, the other realms will get their objects and RealmResults updated to reflect
587          * the changes from this commit.
588          * 
589          * @throws java.lang.IllegalStateException If the write transaction is in an invalid state.
590          */
591         public void commitTransaction() {
592             transaction.commitAndContinueAsRead();
593     
594             Message message = Message.obtain();
595             message.arg1 = LooperThread.REALM_CHANGED;
596             message.arg2 = id;
597             if (looperThread.handler != null) {
598                 looperThread.handler.sendMessage(message);
599             } else {
600                 Log.i(TAG, "The LooperThread is not up and running yet. Commit message not sent");
601             }
602         }
603     
604         /**
605          * Revert all writes (created, updated, or deleted objects) made in the current write
606          * transaction and end the transaction.
607          *
608          * The realm reverts back to read-only.
609          *
610          * Calling this when not in a write transaction will throw an exception.
611          *
612          * @throws java.lang.IllegalStateException    If the write transaction is an invalid state or
613         *                                             not in a write transaction.
614         */
615          public void cancelTransaction() {
616              transaction.rollbackAndContinueAsRead();
617          }

652         static public void migrateRealmAtPath(String realmPath, byte[] key, RealmMigration migration) {
653             Realm realm = Realm.createAndValidate(realmPath, key, false);
654             realm.beginTransaction();
655             realm.setVersion(migration.execute(realm, realm.getVersion()));
656             realm.commitTransaction();
657     
658             Map<String, Realm> realms = realmsCache.get();
659             realms.put(realmPath, new Realm(realmPath, key));
660             realmsCache.set(realms);
661         }

678          * The realm must be unused and closed before calling this method.
679          *
680          * @param context  an Android context.
681          * @param fileName the name of the custom Realm (i.e. "myCustomRealm.realm").
682          * @return false if a file could not be deleted. The failing file will be logged.
683          */
684         public static boolean deleteRealmFile(Context context, String fileName) {
685             boolean result = true;
686             File writableFolder = context.getFilesDir();
687             List<File> filesToDelete = Arrays.asList(
688                     new File(writableFolder, fileName),
689                     new File(writableFolder, fileName + ".lock"));
690             for (File fileToDelete : filesToDelete) {
691                 if (fileToDelete.exists()) {
692                     boolean deleteResult = fileToDelete.delete();
693                     if (!deleteResult) {
694                         result = false;
695                         Log.w(TAG, "Could not delete the file " + fileToDelete);
696                     }
697                 }
698             }
699             return result;
700         }
