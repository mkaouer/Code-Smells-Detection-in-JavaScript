16     package io.realm;
17     
18     import android.test.AndroidTestCase;
19     import android.util.Log;
20     
21     import io.realm.entities.Dog;
22     import io.realm.internal.android.LooperThread;
23     
24     public class NotificationsTest extends AndroidTestCase {
25         public void testMessageToDeadThread() {
26             Realm realm = Realm.getInstance(getContext());
27     
28             // Number of handlers before
29             final int handlersBefore = LooperThread.handlers.size();
30     
31             // Make sure the Looper Thread is alive
32             LooperThread looperThread = LooperThread.getInstance();
33             assertTrue(looperThread.isAlive());
34     
35             Thread thread = new Thread() {
36                 @Override
37                 public void run() {
38                     Realm r = Realm.getInstance(getContext());
39                     assertFalse(handlersBefore == LooperThread.handlers.size());
40                     r.addChangeListener(new RealmChangeListener() {
41                         @Override
42                         public void onChange() {
43                             Log.i("Notification Test", "Notification Received");
44                         }
45                     });
46                 }
47             };
48             thread.start();
49     
50             try {
51                 thread.join();
52             } catch (InterruptedException e) {
53                 fail();
54             }
55             assertFalse(thread.isAlive()); // Make sure the thread is dead
56             realm.beginTransaction();
57             Dog dog = realm.createObject(Dog.class);
58             dog.setName("Rex");
59             realm.commitTransaction();
60     
61             // Give some time to log the exception
62             try {
63                 Thread.sleep(500);
64             } catch (InterruptedException e) {
65                 fail();
66             }
67     
68             assertEquals(0, looperThread.exceptions.size());
69         }
70     }
