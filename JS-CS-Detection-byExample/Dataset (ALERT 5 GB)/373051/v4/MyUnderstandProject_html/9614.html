31     void ThrowException(JNIEnv* env, ExceptionKind exception, std::string classStr, std::string itemStr)
32     {
33         std::string message;
34         jclass jExceptionClass = NULL;
35     
36         TR_ERR((env, "\njni: ThrowingException %d, %s, %s.\n", exception, classStr.c_str(), itemStr.c_str()));
37     
38         switch (exception) {
39             case ClassNotFound:
40                 jExceptionClass = env->FindClass("java/lang/ClassNotFoundException");
41                 message = "Class '" + classStr + "' could not be located.";
42                 break;
43     
44             case NoSuchField:
45                 jExceptionClass = env->FindClass("java/lang/NoSuchFieldException");
46                 message = "Field '" + itemStr + "' could not be located in class io.realm." + classStr;
47                 break;
48     
49             case NoSuchMethod:
50                 jExceptionClass = env->FindClass("java/lang/NoSuchMethodException");
51                 message = "Method '" + itemStr + "' could not be located in class io.realm." + classStr;
52                 break;
53     
54             case IllegalArgument:
55                 jExceptionClass = env->FindClass("java/lang/IllegalArgumentException");
56                 message = "Illegal Argument: " + classStr;
57                 break;
58     
59             case TableInvalid:
60                 jExceptionClass = env->FindClass("java/lang/IllegalStateException");
61                 message = "Illegal State: " + classStr;
62                 break;
63     
64             case IOFailed:
65                 jExceptionClass = env->FindClass("io/realm/exceptions/RealmIOException");
66                 message = "Failed to open " + classStr + ". " + itemStr;
67                 break;
68     
69             case FileNotFound:
70                 jExceptionClass = env->FindClass("io/realm/exceptions/RealmIOException");
71                 message = "File not found: " + classStr + ".";
72                 break;
73     
74             case FileAccessError:
75                 jExceptionClass = env->FindClass("io/realm/exceptions/RealmIOException");
76                 message = "Failed to access: " + classStr + ". " + itemStr;
77                 break;
78     
79             case IndexOutOfBounds:
80                 jExceptionClass = env->FindClass("java/lang/ArrayIndexOutOfBoundsException");
81                 message = classStr;
82                 break;
83     
84             case UnsupportedOperation:
85                 jExceptionClass = env->FindClass("java/lang/UnsupportedOperationException");
86                 message = classStr;
87                 break;
88     
89             case OutOfMemory:
90                 jExceptionClass = env->FindClass("io/realm/internal/OutOfMemoryError");
91                 message = classStr + " " + itemStr;
92                 break;
93     
94             case Unspecified:
95                 jExceptionClass = env->FindClass("java/lang/RuntimeException");
96                 message = "Unspecified exception. " + classStr;
97                 break;
98     
99             case RuntimeError:
100                 jExceptionClass = env->FindClass("java/lang/RuntimeException");
101                 message = classStr;
102                 break;
103         }
104         if (jExceptionClass != NULL)
105             env->ThrowNew(jExceptionClass, message.c_str());
106         else {
107             TR_ERR((env, "\nERROR: Couldn't throw exception.\n"));
108         }
109     
110         env->DeleteLocalRef(jExceptionClass);
111     }

113     jclass GetClass(JNIEnv* env, const char* classStr)
114     {
115         jclass localRefClass = env->FindClass(classStr);
116         if (localRefClass == NULL) {
117             ThrowException(env, ClassNotFound, classStr);
118             return NULL;
119         }
120     
121         jclass myClass = reinterpret_cast<jclass>( env->NewGlobalRef(localRefClass) );
122         env->DeleteLocalRef(localRefClass);
123         return myClass;
124     }

126     void jprint(JNIEnv *env, char *txt)
127     {
128     #if 1
129         static_cast<void>(env);
130         fprintf(stderr, " -- JNI: %s", txt);  fflush(stderr);
131     #else
132         static jclass myClass = GetClass(env, "io/realm/internal/util");
133         static jmethodID myMethod = env->GetStaticMethodID(myClass, "javaPrint", "(Ljava/lang/String;)V");
134         if (myMethod)
135             env->CallStaticVoidMethod(myClass, myMethod, env->NewStringUTF(txt));
136     #endif
137     }

150     bool GetBinaryData(JNIEnv* env, jobject jByteBuffer, tightdb::BinaryData& bin)
151     {
152         const char* data = static_cast<char*>(env->GetDirectBufferAddress(jByteBuffer));
153         if (!data) {
154             ThrowException(env, IllegalArgument, "ByteBuffer is invalid");
155             return false;
156         }
157         jlong size = env->GetDirectBufferCapacity(jByteBuffer);
158         if (size < 0) {
159             ThrowException(env, IllegalArgument, "Can't get BufferCapacity.");
160             return false;
161         }
162         bin = BinaryData(data, S(size));
163         return true;
164     }

173     // This assumes that 'jchar' is an integral type with at least 16
174     // non-sign value bits, that is, an unsigned 16-bit integer, or any
175     // signed or unsigned integer with more than 16 bits.
176     struct JcharTraits {
177         static jchar to_int_type(jchar c)  TIGHTDB_NOEXCEPT { return c; }
178         static jchar to_char_type(jchar i) TIGHTDB_NOEXCEPT { return i; }
179     };
180     
181     struct JStringCharsAccessor {
182         JStringCharsAccessor(JNIEnv* e, jstring s):
183             m_env(e), m_string(s), m_data(e->GetStringChars(s,0)), m_size(get_size(e,s)) {}
184         ~JStringCharsAccessor()
185         {
186             m_env->ReleaseStringChars(m_string, m_data);
187         }
188         const jchar* data() const TIGHTDB_NOEXCEPT { return m_data; }
189         size_t size() const TIGHTDB_NOEXCEPT { return m_size; }
190     
191     private:
192         JNIEnv* const m_env;
193         const jstring m_string;
194         const jchar* const m_data;
195         const size_t m_size;
196     
197         static size_t get_size(JNIEnv* e, jstring s)
198         {
199             size_t size;
200             if (int_cast_with_overflow_detect(e->GetStringLength(s), size))
201                 throw runtime_error("String size overflow");
202             return size;
203         }
204     };
205     
206     } // anonymous namespace

209     jstring to_jstring(JNIEnv* env, StringData str)
210     {
211         // For efficiency, if the incoming UTF-8 string is sufficiently
212         // small, we will attempt to store the UTF-16 output into a stack
213         // allocated buffer of static size. Otherwise we will have to
214         // dynamically allocate the output buffer after calculating its
215         // size.
216     
217         const size_t stack_buf_size = 48;
218         jchar stack_buf[stack_buf_size];
219         UniquePtr<jchar[]> dyn_buf;
220     
221         const char* in_begin = str.data();
222         const char* in_end   = str.data() + str.size();
223         jchar* out_begin = stack_buf;
224         jchar* out_curr  = stack_buf;
225         jchar* out_end   = stack_buf + stack_buf_size;
226     
227         typedef Utf8x16<jchar, JcharTraits> Xcode;
228     
229         if (str.size() <= stack_buf_size) {
230             if (!Xcode::to_utf16(in_begin, in_end, out_curr, out_end))
231                 goto bad_utf8;
232             if (in_begin == in_end)
233                 goto transcode_complete;
234         }
235     
236         {
237             const char* in_begin2 = in_begin;
238             size_t size = Xcode::find_utf16_buf_size(in_begin2, in_end);
239             if (in_begin2 != in_end) 
240                 goto bad_utf8;
241             if (int_add_with_overflow_detect(size, stack_buf_size))
242                 throw runtime_error("String size overflow");
243             dyn_buf.reset(new jchar[size]);
244             out_curr = copy(out_begin, out_curr, dyn_buf.get());
245             out_begin = dyn_buf.get();
246             out_end   = dyn_buf.get() + size;
247             if (!Xcode::to_utf16(in_begin, in_end, out_curr, out_end))
248                 goto bad_utf8;
249             TIGHTDB_ASSERT(in_begin == in_end);
250         }
251     
252       transcode_complete:
253         {
254             jsize out_size;
255             if (int_cast_with_overflow_detect(out_curr - out_begin, out_size))
256                 throw runtime_error("String size overflow");
257     
258             return env->NewString(out_begin, out_size);
259         }
260     
261       bad_utf8:
262         throw runtime_error("Bad UTF-8 encoding");
263     }

266     JStringAccessor::JStringAccessor(JNIEnv* env, jstring str)
267     {
268         // For efficiency, if the incoming UTF-16 string is sufficiently
269         // small, we will choose an UTF-8 output buffer whose size (in
270         // bytes) is simply 4 times the number of 16-bit elements in the
271         // input. This is guaranteed to be enough. However, to avoid
272         // excessive over allocation, this is not done for larger input
273         // strings.
274     
275         JStringCharsAccessor chars(env, str);
276     
277         typedef Utf8x16<jchar, JcharTraits> Xcode;
278         size_t max_project_size = 48;
279         TIGHTDB_ASSERT(max_project_size <= numeric_limits<size_t>::max()/4);
280         size_t buf_size;
281         if (chars.size() <= max_project_size) {
282             buf_size = chars.size() * 4;
283         }
284         else {
285             const jchar* begin = chars.data();
286             const jchar* end   = begin + chars.size();
287             buf_size = Xcode::find_utf8_buf_size(begin, end);
288         }
289         m_data.reset(new char[buf_size]);  // throws
290         {
291             const jchar* in_begin = chars.data();
292             const jchar* in_end   = in_begin + chars.size();
293             char* out_begin = m_data.get();
294             char* out_end   = m_data.get() + buf_size;
295             if (!Xcode::to_utf8(in_begin, in_end, out_begin, out_end))
296                 throw runtime_error("Bad UTF-16 encoding");
297             TIGHTDB_ASSERT(in_begin == in_end);
298             m_size = out_begin - m_data.get();
299         }
300     }
