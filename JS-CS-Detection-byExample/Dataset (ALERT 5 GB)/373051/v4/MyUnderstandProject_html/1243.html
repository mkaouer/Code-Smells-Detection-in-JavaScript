17     package io.realm.processor;
18     
19     import javax.annotation.processing.ProcessingEnvironment;
20     import javax.tools.Diagnostic;
21     import java.io.*;
22     import java.net.HttpURLConnection;
23     import java.net.URL;
24     import java.nio.charset.Charset;
25     
26     public class RealmVersionChecker {
27     
28         public static final String UPDATE_FILE_NAME = "realm_version_check.timestamp";
29         public static final String REALM_ANDROID_DOWNLOAD_URL = "http://static.realm.io/downloads/java/latest";
30     
31         private static final String VERSION_URL = "http://static.realm.io/update/java?";
32         private static final String REALM_VERSION = "0.72.0"; //TODO: The version value should be pulled from a build file
33         private static final int READ_TIMEOUT = 2000;
34         private static final int CONNECT_TIMEOUT = 4000;
35     
36         private ProcessingEnvironment processingEnvironment;
37     
38         public RealmVersionChecker(ProcessingEnvironment processingEnvironment) {
39             this.processingEnvironment = processingEnvironment;
40         }
41     
42         private void launchRealmCheck() {
43             long lastRealmUpdate = readRealmStat();
44     
45             if ((lastRealmUpdate + (24 * 60 * 60 * 1000)) < System.currentTimeMillis()) {
46                 updateLastRealmStat();
47     
48                 //Check Realm version server
49                 String latestVersionStr = checkLatestVersion();
50                 if (!latestVersionStr.equals(REALM_VERSION)) {
51                     printMessage("Version " + latestVersionStr + " of Realm is now available: " + REALM_ANDROID_DOWNLOAD_URL);
52                 }
53             }
54         }
55     
56         public void executeRealmVersionUpdate() {
57             Thread backgroundThread = new Thread(new Runnable() {
58                 @Override
59                 public void run() {
60                     launchRealmCheck();
61                 }
62             });
63             backgroundThread.start();
64             try {
65                 backgroundThread.join(CONNECT_TIMEOUT + READ_TIMEOUT);
66             } catch(InterruptedException e) {
67                 // We ignore this exception on purpose not to break the build system if this class fails
68             }
69         }
70     
71         private Long readRealmStat() {
72             InputStream inputStream;
73             BufferedReader bufferedReader;
74             String lastVersion;
75             try {
76                 inputStream = new FileInputStream(UPDATE_FILE_NAME);
77                 bufferedReader = new BufferedReader(new InputStreamReader(inputStream, Charset.forName("UTF-8")));
78                 lastVersion = bufferedReader.readLine();
79                 inputStream.close();
80             } catch (FileNotFoundException e) {
81                 return 0L;
82             } catch (IOException e) {
83                 return 0L;
84             }
85     
86             try {
87                 return Long.parseLong(lastVersion);
88             } catch (NumberFormatException e) {
89                 return 0L;
90             }
91         }
92     
93         private void updateLastRealmStat() {
94             PrintWriter writer;
95             try {
96                 writer = new PrintWriter(UPDATE_FILE_NAME, "UTF-8");
97                 writer.println(System.currentTimeMillis());
98                 writer.close();
99             } catch (IOException e) {
100                 // We ignore this exception on purpose not to break the build system if this class fails
101             }
102         }
103     
104         private String checkLatestVersion() {
105             String result = REALM_VERSION;
106             try {
107                 URL url = new URL(VERSION_URL +REALM_VERSION);
108                 HttpURLConnection conn = (HttpURLConnection)url.openConnection();
109                 conn.setConnectTimeout(CONNECT_TIMEOUT);
110                 conn.setReadTimeout(READ_TIMEOUT);
111                 BufferedReader rd = new BufferedReader(new InputStreamReader(conn.getInputStream()));
112                 result = rd.readLine();
113                 rd.close();
114             } catch (IOException e) {
115                 // We ignore this exception on purpose not to break the build system if this class fails
116             } 
117             return result;
118         }
119     
120         private void printMessage(String message) {
121             processingEnvironment.getMessager().printMessage(Diagnostic.Kind.OTHER, message);
122         }
123     }
