17     package io.realm.processor;
18     
19     import io.realm.annotations.Ignore;
20     import io.realm.annotations.Index;
21     import io.realm.annotations.RealmClass;
22     
23     import javax.annotation.processing.AbstractProcessor;
24     import javax.annotation.processing.RoundEnvironment;
25     import javax.annotation.processing.SupportedAnnotationTypes;
26     import javax.lang.model.SourceVersion;
27     import javax.lang.model.element.*;
28     import javax.tools.Diagnostic;
29     import java.io.IOException;
30     import java.util.ArrayList;
31     import java.util.HashSet;
32     import java.util.List;
33     import java.util.Set;
34     
35     
36     @SupportedAnnotationTypes({"io.realm.annotations.RealmClass", "io.realm.annotations.Ignore", "io.realm.annotations.Index"})
37     public class RealmProcessor extends AbstractProcessor {
38         Set<String> classesToValidate = new HashSet<String>();
39         boolean done = false;
40     
41         @Override public SourceVersion getSupportedSourceVersion() {
42             return SourceVersion.latestSupported();
43         }
44     
45         @Override
46         public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
47             RealmVersionChecker updateChecker = new RealmVersionChecker(processingEnv);
48             updateChecker.executeRealmVersionUpdate();
49     
50             for (Element classElement : roundEnv.getElementsAnnotatedWith(RealmClass.class)) {
51                 String className;
52                 String packageName;
53                 List<VariableElement> fields = new ArrayList<VariableElement>();
54                 List<VariableElement> indexedFields = new ArrayList<VariableElement>();
55                 List<String> ignoredFields = new ArrayList<String>();
56                 List<String> expectedGetters = new ArrayList<String>();
57                 List<String> expectedSetters = new ArrayList<String>();
58                 List<ExecutableElement> methods = new ArrayList<ExecutableElement>();
59     
60                 // Check the annotation was applied to a Class
61                 if (!classElement.getKind().equals(ElementKind.CLASS)) {
62                     error("The RealmClass annotation can only be applied to classes", classElement);
63                 }
64                 TypeElement typeElement = (TypeElement) classElement;
65                 className = typeElement.getSimpleName().toString();
66     
67                 if (typeElement.toString().endsWith(".RealmObject") || typeElement.toString().endsWith("RealmProxy")) {
68                     continue;
69                 }
70     
71                 note("Processing class " + className);
72     
73                 classesToValidate.add(typeElement.toString());
74     
75                 // Get the package of the class
76                 Element enclosingElement = typeElement.getEnclosingElement();
77                 if (!enclosingElement.getKind().equals(ElementKind.PACKAGE)) {
78                     error("The RealmClass annotation does not support nested classes", classElement);
79                 }
80     
81                 TypeElement parentElement = (TypeElement) processingEnv.getTypeUtils().asElement(typeElement.getSuperclass());
82                 if (!parentElement.toString().endsWith(".RealmObject")) {
83                     error("A RealmClass annotated object must be derived from RealmObject", classElement);
84                 }
85     
86                 PackageElement packageElement = (PackageElement) enclosingElement;
87                 packageName = packageElement.getQualifiedName().toString();
88     
89                 for (Element element : typeElement.getEnclosedElements()) {
90                     ElementKind elementKind = element.getKind();
91                     if (elementKind.equals(ElementKind.FIELD)) {
92                         VariableElement variableElement = (VariableElement) element;
93                         String fieldName = variableElement.getSimpleName().toString();
94                         if (variableElement.getAnnotation(Ignore.class) != null) {
95                             // The field has the @Ignore annotation. No need to go any further.
96                             ignoredFields.add(fieldName);
97                             continue;
98                         }
99     
100                         if (variableElement.getAnnotation(Index.class) != null) {
101                             // The field has the @Index annotation. It's only valid for:
102                             // * String
103                             String elementTypeCanonicalName = variableElement.asType().toString();
104                             if (elementTypeCanonicalName.equals("java.lang.String")) {
105                                 indexedFields.add(variableElement);
106                             } else {
107                                 error("@Index is only applicable to String fields - got " + element);
108                                 return true;
109                             }
110                         }
111     
112                         if (!variableElement.getModifiers().contains(Modifier.PRIVATE)) {
113                             error("The fields of the model must be private", variableElement);
114                         }
115     
116                         fields.add(variableElement);
117                         expectedGetters.add(fieldName);
118                         expectedSetters.add(fieldName);
119                     } else if (elementKind.equals(ElementKind.METHOD)) {
120                         ExecutableElement executableElement = (ExecutableElement) element;
121                         methods.add(executableElement);
122                     }
123                 }
124     
125                 for (ExecutableElement executableElement : methods) {
126                     if (!executableElement.getModifiers().contains(Modifier.PUBLIC)) {
127                         error("The methods of the model must be public", executableElement);
128                     }
129     
130                     String methodName = executableElement.getSimpleName().toString();
131                     String computedFieldName = methodName.startsWith("is")?lowerFirstChar(methodName.substring(2)):lowerFirstChar(methodName.substring(3));
132                     if (methodName.startsWith("get") || methodName.startsWith("is")) {
133                         boolean found = false;
134                         for (VariableElement field : fields) {
135                             if (field.getSimpleName().toString().equals(computedFieldName)) {
136                                 found = true;
137                             }
138                         }
139                         if (ignoredFields.contains(computedFieldName)) {
140                             found = true;
141                         }
142                         if (!found) {
143                             error(String.format("No field named %s for the getter %s", computedFieldName, methodName), executableElement);
144                         }
145                         expectedGetters.remove(computedFieldName);
146                     } else if (methodName.startsWith("set")) {
147                         boolean found = false;
148                         for (VariableElement field : fields) {
149                             if (field.getSimpleName().toString().equals(computedFieldName)) {
150                                 found = true;
151                             }
152                         }
153                         if (ignoredFields.contains(computedFieldName)) {
154                             found = true;
155                         }
156                         if (!found) {
157                             error(String.format("No field named %s for the setter %s", computedFieldName, methodName), executableElement);
158                         }
159                         expectedSetters.remove(computedFieldName);
160                     } else {
161                         error("Only getters and setters should be defined in model classes", executableElement);
162                     }
163                 }
164     
165                 for (String expectedGetter : expectedGetters) {
166                     error("No getter found for field " + expectedGetter);
167                 }
168                 for (String expectedSetter : expectedSetters) {
169                     error("No setter found for field " + expectedSetter);
170                 }
171     
172                 RealmProxyClassGenerator sourceCodeGenerator =
173                         new RealmProxyClassGenerator(processingEnv, className, packageName, fields, indexedFields);
174                 try {
175                     sourceCodeGenerator.generate();
176                 } catch (IOException e) {
177                     error(e.getMessage(), classElement);
178                 } catch (UnsupportedOperationException e) {
179                     error(e.getMessage(), classElement);
180                 }
181             }
182     
183             if (!done) {
184                 RealmValidationListGenerator validationGenerator = new RealmValidationListGenerator(processingEnv, classesToValidate);
185                 try {
186                     validationGenerator.generate();
187                     done = true;
188                 } catch (IOException e) {
189                     error(e.getMessage());
190                 }
191             }
192     
193             return true;
194         }
195     
196         private static String lowerFirstChar(String input) {
197             return input.substring(0, 1).toLowerCase() + input.substring(1);
198         }
199     
200         private void error(String message, Element element) {
201             processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, message, element);
202         }
203     
204         private void error(String message) {
205             processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, message);
206         }
207     
208         private void note(String message) {
209             processingEnv.getMessager().printMessage(Diagnostic.Kind.NOTE, message);
210         }
211     }
