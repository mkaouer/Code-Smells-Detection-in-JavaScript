47         public SharedGroup(String databaseFile) {
48             context = new Context();
49             this.nativePtr = createNative(databaseFile, Durability.FULL.value, false, false, null);
50             checkNativePtrNotZero();

52         public SharedGroup(String databaseFile, boolean enableImplicitTransactions, byte[] key) {
53             if (enableImplicitTransactions) {
54                 nativeTransactLogRegistryPtr = nativeCreateTransactLogRegistry(databaseFile);
55                 nativeReplicationPtr = nativeCreateReplication(databaseFile);
56                 nativePtr = createNativeWithImplicitTransactions(nativeReplicationPtr, key);
57                 implicitTransactionsEnabled = true;
58             } else {
59                 nativePtr = createNative(databaseFile, Durability.FULL.value, false, false, key);
60             }
61             context = new Context();
62             checkNativePtrNotZero();
63         }

110         public ImplicitTransaction beginImplicitTransaction() {
111             if (activeTransaction) {
112                 throw new IllegalStateException(
113                         "Can't beginImplicitTransaction() during another active transaction");
114             }
115             long nativeGroupPtr = nativeBeginImplicit(nativePtr);
116             ImplicitTransaction transaction = new ImplicitTransaction(context, this, nativeGroupPtr);
117             activeTransaction = true;
118             return transaction;
119         }
120     
121         private native long nativeBeginImplicit(long nativePtr);
122     
123         public WriteTransaction beginWrite() {
124             if (activeTransaction)
125                 throw new IllegalStateException(
126                         "Can't beginWrite() during another active transaction");
127             // FIXME: throw from nativeMethod in case of error
128     
129             long nativeWritePtr = nativeBeginWrite(nativePtr);
130             try {
131                 // Copy context reference from parent
132                 WriteTransaction t = new WriteTransaction(context, this, nativeWritePtr);
133                 activeTransaction = true;
134                 return t;
135             }
136             catch (RuntimeException e) {
137                 Group.nativeClose(nativeWritePtr);
138                 throw e;
139             }
140         }
141     
142         public ReadTransaction beginRead() {
143             if (activeTransaction)
144                 throw new IllegalStateException(
145                         "Can't beginRead() during another active transaction");
146             // FIXME: throw from nativeMethod in case of error
147     
148             long nativeReadPtr = nativeBeginRead(nativePtr);
149             try {
150                 // Copy context reference from parent
151                 ReadTransaction t = new ReadTransaction(context, this, nativeReadPtr);
152                 activeTransaction = true;
153                 return t;

161         void endRead() {
162             if (isClosed())
163                 throw new IllegalStateException(
164                         "Can't endRead() on closed group. ReadTransaction is invalid.");
165             nativeEndRead(nativePtr);
166             activeTransaction = false;
167         }
168     
169         public void close() {
170             if (activeTransaction)
171                 throw new IllegalStateException(
172                         "Can't close the SharedGroup during an active transaction");
173     
174                 synchronized (context) {
175                 if (nativePtr != 0) {
176                     nativeClose(nativePtr);
177                     nativePtr = 0;
178                     if (implicitTransactionsEnabled) {
179                         if (nativeTransactLogRegistryPtr != 0) {
180                             nativeCloseTransactRegistryLog(nativeTransactLogRegistryPtr);
181                             nativeTransactLogRegistryPtr = 0;
182                         }
183                         if (nativeReplicationPtr != 0) {
184                             nativeCloseReplication(nativeReplicationPtr);
185                             nativeReplicationPtr = 0;
186                         }
187                     }
188                 }
189             }
190         }
191     
192         protected void finalize() {
193             synchronized (context) {
194                 if (nativePtr != 0) {
195                     context.asyncDisposeSharedGroup(nativePtr); 
196                     nativePtr = 0; // Set to 0 if finalize is called before close() for some reason
197                     if (implicitTransactionsEnabled) {
198                         if (nativeTransactLogRegistryPtr != 0) {
199                             nativeCloseTransactRegistryLog(nativeTransactLogRegistryPtr);
200                             nativeTransactLogRegistryPtr = 0;
201                         }
202                         if (nativeReplicationPtr != 0) {
203                             nativeCloseReplication(nativeReplicationPtr);
204                             nativeReplicationPtr = 0;
205                         }
206                     }
207                 }
208             }
209         }
210     
211         void commit() {
212             if (isClosed())
213                 throw new IllegalStateException(
214                         "Can't commit() on closed group. WriteTransaction is invalid.");
215             nativeCommit(nativePtr);
216             activeTransaction = false;
217         }
218     
219         void rollback() {
220             if (isClosed())
221                 throw new IllegalStateException(
222                         "Can't rollback() on closed group. WriteTransaction is invalid.");
223             nativeRollback(nativePtr);
224             activeTransaction = false;
225         }
