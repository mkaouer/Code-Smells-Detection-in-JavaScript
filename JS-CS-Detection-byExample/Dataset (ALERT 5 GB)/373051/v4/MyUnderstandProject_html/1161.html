35         private void realmExamples() {
36     
37             // These operations are small enough that
38             // we can generally safely run them on the UI thread.
39     
40             // Open the default realm ones for the UI thread.
41             Realm realm = Realm.getInstance(this);
42     
43             basicCRUD(realm);
44             basicQuery(realm);
45     
46             // More complex operations can be executed on another thread.
47             new AsyncTask<Void, Void, String>() {
48                 @Override
49                 protected String doInBackground(Void... voids) {
50                     String info = null;
51                     info = complexReadWrite();
52                     info += complexQuery();
53                     return info;
54                 }
55     
56                 @Override
57                 protected void onPostExecute(String result) {
58                     showStatus(result);
59                 }
60             }.execute();
61     
62         }

64         private void basicCRUD(Realm realm) {
65             showStatus("Perform basic Create/Read/Update/Delete (CRUD) operations...");
66     
67             // All writes must be wrapped in a transaction to facilitate safe multi threading
68             realm.beginTransaction();
69     
70             // Add a person
71             Person person = realm.createObject(Person.class);
72             person.setName("Young Person");
73             person.setAge(14);
74     
75             // When the write transaction is committed, all changes a synced to disk.
76             realm.commitTransaction();
77     
78             // Find the first person (no query conditions) and read a field
79             person = realm.where(Person.class).findFirst();
80             showStatus(person.getName() + ":" + person.getAge());
81     
82             // Update person in a write transaction
83             realm.beginTransaction();
84             person.setName("Senior Person");
85             person.setAge(99);
86             showStatus(person.getName() + " got older: " + person.getAge());
87             realm.commitTransaction();
88     
89             // Delete all persons
90             realm.beginTransaction();
91             realm.allObjects(Person.class).clear();
92             realm.commitTransaction();
93         }

95         private void basicQuery(Realm realm) {
96             showStatus("\nPerforming basic Query operation...");
97             showStatus("Number of persons: " + realm.allObjects(Person.class).size());
98     
99             RealmResults<Person> results = realm.where(Person.class).equalTo("age", 99).findAll();
100     
101             showStatus("Size of result set: " + results.size());
102         }

105         private String complexReadWrite() {
106             String status = "\nPerforming complex Read/Write operation...";
107     
108             // Open the default realm. All threads must use it's own reference to the realm.
109             // Those can not be transferred across threads.
110             Realm realm = Realm.getInstance(this);
111     
112             // Add ten persons in one write transaction
113             realm.beginTransaction();
114             Dog fido = realm.createObject(Dog.class);
115             fido.setName("fido");
116             for (int i = 0; i < 10; i++) {
117                 Person person = realm.createObject(Person.class);
118                 person.setName("Person no. " + i);
119                 person.setAge(i);
120                 person.setDog(fido);
121     
122                 // The field tempReference is annotated with @Ignore.
123                 // This means setTempReference sets the Person tempReference
124                 // field directly. The tempReference is NOT saved as part of
125                 // the RealmObject:
126                 person.setTempReference(42);
127     
128                 for (int j = 0; j < i; j++) {
129                     Cat cat = realm.createObject(Cat.class);
130                     cat.setName("Cat_" + j);
131                     person.getCats().add(cat);
132                 }
133             }
134             realm.commitTransaction();
135     
136             // Implicit read transactions allow you to access your objects
137             status += "\nNumber of persons: " + realm.allObjects(Person.class).size();
138     
139             // Iterate over all objects
140             for (Person pers : realm.allObjects(Person.class)) {
141                 String dogName;
142                 if (pers.getDog() == null) {
143                     dogName = "None";
144                 } else {
145                     dogName = pers.getDog().getName();
146                 }
147                 status += "\n" + pers.getName() + ":" + pers.getAge() + " : " + dogName + " : " + pers.getCats().size();
148     
149                 // The field tempReference is annotated with @Ignore
150                 // Though we initially set its value to 42, it has
151                 // not been saved as part of the Person RealmObject:
152                 assert(pers.getTempReference() == 0);
153             }
154     
155             // Sorting
156             RealmResults<Person> sortedPersons = realm.allObjects(Person.class).sort("age", false);
157             assert(realm.allObjects(Person.class).last().getName() == sortedPersons.first().getName());
158             status += "\nSorting " + sortedPersons.last().getName() + " == " + realm.allObjects(Person.class).first().getName();
159     
160             return status;
161         }
162     
163         private String complexQuery() {
164             String status = "\n\nPerforming complex Query operation...";
165     
166             Realm realm = Realm.getInstance(this);
167             status += "\nNumber of persons: " + realm.allObjects(Person.class).size();
168     
169             // Find all persons where age between 7 and 9 and name begins with "Person".
170             RealmResults<Person> results = realm.where(Person.class)
171                     .between("age", 7, 9)       // Notice implicit "and" operation
172                     .beginsWith("name", "Person").findAll();
173             status += "\nSize of result set: " + results.size();
174             return status;
175         }
