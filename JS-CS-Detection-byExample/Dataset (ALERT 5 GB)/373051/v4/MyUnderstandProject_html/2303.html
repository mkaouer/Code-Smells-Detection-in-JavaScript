17     package io.realm.examples.concurrency.services;
18     
19     import android.app.Service;
20     import android.content.Intent;
21     import android.os.IBinder;
22     
23     import java.util.ArrayList;
24     import java.util.List;
25     
26     import io.realm.examples.concurrency.threads.KillableThread;
27     import io.realm.examples.concurrency.threads.RealmReader;
28     import io.realm.examples.concurrency.threads.RealmWriter;
29     
30     public class SpawningService extends Service {
31     
32         public static final String TAG = SpawningService.class.getName();
33     
34         public static final String REALM_INSERTCOUNT_EXTRA = "RealmInsertCountExtra";
35         public static final String REALM_READCOUNT_EXTRA   = "RealmReadCountExtra";
36     
37         private List<KillableThread> allThreads = null;
38     
39         public void onCreate() {
40             super.onCreate();
41         }
42     
43         @Override
44         public int onStartCommand(Intent intent, int flags, int startId) {
45     
46             int insertCount = intent.getIntExtra(REALM_INSERTCOUNT_EXTRA, 0);
47             int readCount   = intent.getIntExtra(REALM_READCOUNT_EXTRA, 0);
48     
49             try {
50                 allThreads = new ArrayList<KillableThread>();
51                 RealmWriter wT = new RealmWriter(this);
52                 wT.setInsertCount(insertCount);
53                 allThreads.add(wT);
54                 wT.start();
55                 RealmReader rT = new RealmReader(this);
56                 rT.setReadCount(readCount);
57                 allThreads.add(rT);
58                 rT.start();
59             } catch (Exception e) {
60                 e.printStackTrace();
61                 quit();
62             }
63     
64             //Service should stay active as long as Activity is active
65             return START_STICKY;
66         }
67     
68         @Override
69         public void onDestroy() {
70             quit();
71         }
72     
73         @Override
74         public IBinder onBind(Intent intent) {
75             return null;
76         }
77     
78         private void quit() {
79             for (KillableThread t : allThreads) {
80                 t.terminate();
81             }
82         }
83     
84     }
