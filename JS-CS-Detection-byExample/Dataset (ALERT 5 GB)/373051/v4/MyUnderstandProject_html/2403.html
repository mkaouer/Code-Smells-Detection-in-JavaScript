47     JNIEXPORT jstring JNICALL Java_io_realm_internal_Util_nativeTestcase(
48         JNIEnv *env, jclass, jint testcase, jboolean dotest, jlong)
49     {
50         string expect;
51     
52         switch (ExceptionKind(testcase)) {
53             case ClassNotFound:
54                 expect = "java.lang.ClassNotFoundException: Class 'parm1' could not be located.";
55                 if (dotest)
56                     ThrowException(env, ClassNotFound, "parm1", "parm2");
57                 break;
58             case NoSuchField:
59                 expect = "java.lang.NoSuchFieldException: Field 'parm2' could not be located in class io.realm.parm1";
60                 if (dotest)
61                     ThrowException(env, NoSuchField, "parm1", "parm2");
62                 break;
63             case NoSuchMethod:
64                 expect = "java.lang.NoSuchMethodException: Method 'parm2' could not be located in class io.realm.parm1";
65                 if (dotest)
66                     ThrowException(env, NoSuchMethod, "parm1", "parm2");
67                 break;
68             case IllegalArgument:
69                 expect = "java.lang.IllegalArgumentException: Illegal Argument: parm1";
70                 if (dotest)
71                     ThrowException(env, IllegalArgument, "parm1", "parm2");
72                 break;
73             case IOFailed:
74                 expect = "io.realm.exceptions.RealmIOException: Failed to open parm1. parm2";
75                 if (dotest)
76                     ThrowException(env, IOFailed, "parm1", "parm2");
77                 break;
78             case FileNotFound:
79                 expect = "io.realm.exceptions.RealmIOException: File not found: parm1.";
80                 if (dotest)
81                     ThrowException(env, FileNotFound, "parm1", "parm2");
82                 break;
83             case FileAccessError:
84                 expect = "io.realm.exceptions.RealmIOException: Failed to access: parm1. parm2";
85                 if (dotest)
86                     ThrowException(env, FileAccessError, "parm1", "parm2");
87                 break;
88             case IndexOutOfBounds:
89                 expect = "java.lang.ArrayIndexOutOfBoundsException: parm1";
90                 if (dotest)
91                     ThrowException(env, IndexOutOfBounds, "parm1", "parm2");
92                 break;
93             case TableInvalid:
94                 expect = "java.lang.IllegalStateException: Illegal State: parm1";
95                 if (dotest)
96                     ThrowException(env, TableInvalid, "parm1", "parm2");
97                 break;
98             case UnsupportedOperation:
99                 expect = "java.lang.UnsupportedOperationException: parm1";
100                 if (dotest)
101                     ThrowException(env, UnsupportedOperation, "parm1", "parm2");
102                 break;
103                     ThrowException(env, OutOfMemory, "parm1", "parm2");
104             case OutOfMemory:
105                 expect = "io.realm.internal.OutOfMemoryError: parm1 parm2";
106                 if (dotest)
107                     ThrowException(env, OutOfMemory, "parm1", "parm2");
108                 break;
109             case Unspecified:
110                 expect = "java.lang.RuntimeException: Unspecified exception. parm1";
111                 if (dotest)
112                     ThrowException(env, Unspecified, "parm1", "parm2");
113                 break;
114             case RuntimeError:
115                 expect = "java.lang.RuntimeException: parm1";
116                 if (dotest)
117                     ThrowException(env, RuntimeError, "parm1", "parm2");
118                 break;
119         }
120         if (dotest) {
121             return NULL;
122         }
123         return to_jstring(env, expect);
124     }
