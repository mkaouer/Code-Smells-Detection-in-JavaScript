198         public void generate() throws IOException, UnsupportedOperationException {
199             String qualifiedGeneratedClassName = String.format("%s.%s%s", REALM_PACKAGE_NAME, className, PROXY_SUFFIX);
200             JavaFileObject sourceFile = processingEnvironment.getFiler().createSourceFile(qualifiedGeneratedClassName);
201             JavaWriter writer = new JavaWriter(new BufferedWriter(sourceFile.openWriter()));
202     
203             Elements elementUtils = processingEnvironment.getElementUtils();
204             Types typeUtils = processingEnvironment.getTypeUtils();
205     
206             TypeMirror realmObject = elementUtils.getTypeElement("io.realm.RealmObject").asType();
207             DeclaredType realmList = typeUtils.getDeclaredType(elementUtils.getTypeElement("io.realm.RealmList"), typeUtils.getWildcardType(null, null));
208     
209             // Set source code indent to 4 spaces
210             writer.setIndent("    ");
211     
212             writer.emitPackage(REALM_PACKAGE_NAME)
213                     .emitEmptyLine();
214     
215             writer.emitImports(
216                     "io.realm.internal.ColumnType",
217                     "io.realm.internal.Table",
218                     "io.realm.internal.ImplicitTransaction",
219                     "io.realm.internal.Row",
220                     "io.realm.internal.LinkView",
221                     "io.realm.RealmList",
222                     "io.realm.RealmObject",
223                     "java.util.*",
224                     packageName + ".*")
225                     .emitEmptyLine();
226     
227             // Begin the class definition
228             writer.beginType(
229                     qualifiedGeneratedClassName, // full qualified name of the item to generate
230                     "class",                     // the type of the item
231                     EnumSet.of(Modifier.PUBLIC), // modifiers to apply
232                     className)                   // class to extend
233                     .emitEmptyLine();
234     
235             // Accessors
236             ListIterator<VariableElement> iterator = fields.listIterator();
237             while (iterator.hasNext()) {
238                 int columnNumber = iterator.nextIndex();
239                 VariableElement field = iterator.next();
240     
241                 String fieldName = field.getSimpleName().toString();
242                 String fieldTypeCanonicalName = field.asType().toString();
243     
244                 if (JAVA_TO_REALM_TYPES.containsKey(fieldTypeCanonicalName)) {
245                     /**
246                      * Primitives and boxed types
247                      */
248                     String realmType = JAVA_TO_REALM_TYPES.get(fieldTypeCanonicalName);
249                     String castingType = CASTING_TYPES.get(fieldTypeCanonicalName);
250     
251                     // Getter
252                     writer.emitAnnotation("Override");
253                     String getterPrefix = fieldTypeCanonicalName.equals("boolean")?"is":"get";
254                     writer.beginMethod(fieldTypeCanonicalName, getterPrefix + capitaliseFirstChar(fieldName), EnumSet.of(Modifier.PUBLIC));
255                     writer.emitStatement(
256                             "return (%s) realmGetRow().get%s(Realm.columnIndices.get(\"%s\").get(\"%s\"))",
257                             fieldTypeCanonicalName, realmType, className, fieldName);
258                     writer.endMethod();
259                     writer.emitEmptyLine();
260     
261                     // Setter
262                     writer.emitAnnotation("Override");
263                     writer.beginMethod("void", "set" + capitaliseFirstChar(fieldName), EnumSet.of(Modifier.PUBLIC), fieldTypeCanonicalName, "value");
264                     writer.emitStatement(
265                             "realmGetRow().set%s(Realm.columnIndices.get(\"%s\").get(\"%s\"), (%s) value)",
266                             realmType, className, fieldName, castingType);
267                     writer.endMethod();
268                 } else if (typeUtils.isAssignable(field.asType(), realmObject)) {
269                     /**
270                      * Links
271                      */
272     
273                     // Getter
274                     writer.emitAnnotation("Override");
275                     writer.beginMethod(fieldTypeCanonicalName, "get" + capitaliseFirstChar(fieldName), EnumSet.of(Modifier.PUBLIC));
276                     writer.beginControlFlow("if (realmGetRow().isNullLink(Realm.columnIndices.get(\"%s\").get(\"%s\")))", className, fieldName);
277                     writer.emitStatement("return null");
278                     writer.endControlFlow();
279                     writer.emitStatement(
280                             "return realm.get(%s.class, realmGetRow().getLink(Realm.columnIndices.get(\"%s\").get(\"%s\")))",
281                             fieldTypeCanonicalName, className, fieldName);
282                     writer.endMethod();
283                     writer.emitEmptyLine();
284     
285                     // Setter
286                     writer.emitAnnotation("Override");
287                     writer.beginMethod("void", "set" + capitaliseFirstChar(fieldName), EnumSet.of(Modifier.PUBLIC), fieldTypeCanonicalName, "value");
288                     writer.beginControlFlow("if (value == null)");
289                     writer.emitStatement("realmGetRow().nullifyLink(Realm.columnIndices.get(\"%s\").get(\"%s\"))", className, fieldName);
290                     writer.endControlFlow();
291                     writer.emitStatement("realmGetRow().setLink(Realm.columnIndices.get(\"%s\").get(\"%s\"), value.realmGetRow().getIndex())", className, fieldName);
292                     writer.endMethod();
293                 } else if (typeUtils.isAssignable(field.asType(), realmList)) {
294                     /**
295                      * LinkLists
296                      */
297                     String genericCanonicalType = ((DeclaredType) field.asType()).getTypeArguments().get(0).toString();
298                     String genericType;
299                     if (genericCanonicalType.contains(".")) {
300                         genericType = genericCanonicalType.substring(genericCanonicalType.lastIndexOf('.') + 1);
301                     } else {
302                         genericType = genericCanonicalType;
303                     }
304     
305                     // Getter
306                     writer.emitAnnotation("Override");
307                     writer.beginMethod(fieldTypeCanonicalName, "get" + capitaliseFirstChar(fieldName), EnumSet.of(Modifier.PUBLIC));
308                     writer.emitStatement(
309                             "return new RealmList(%s.class, realmGetRow().getLinkList(Realm.columnIndices.get(\"%s\").get(\"%s\")), realm)",
310                             genericType, className, fieldName);
311                     writer.endMethod();
312                     writer.emitEmptyLine();
313     
314                     // Setter
315                     writer.emitAnnotation("Override");
316                     writer.beginMethod("void", "set" + capitaliseFirstChar(fieldName), EnumSet.of(Modifier.PUBLIC), fieldTypeCanonicalName, "value");
317                     writer.emitStatement("LinkView links = realmGetRow().getLinkList(Realm.columnIndices.get(\"%s\").get(\"%s\"))", className, fieldName);
318                     writer.beginControlFlow("if (value == null)");
319                     writer.emitStatement("return"); // TODO: delete all the links instead
320                     writer.endControlFlow();
321                     writer.beginControlFlow("for (RealmObject linkedObject : (RealmList<? extends RealmObject>) value)");
322                     writer.emitStatement("links.add(linkedObject.realmGetRow().getIndex())");
323                     writer.endControlFlow();
324                     writer.endMethod();
325                 } else {
326                     throw new UnsupportedOperationException(
327                             String.format("Type %s of field %s is not supported", fieldTypeCanonicalName, fieldName));
328                 }
329                 writer.emitEmptyLine();
330             }
331     
332             /**
333              * initTable method
334              */
335             writer.beginMethod(
336                     "Table", // Return type
337                     "initTable", // Method name
338                     EnumSet.of(Modifier.PUBLIC, Modifier.STATIC), // Modifiers
339                     "ImplicitTransaction", "transaction"); // Argument type & argument name
340     
341             writer.beginControlFlow("if(!transaction.hasTable(\"" + TABLE_PREFIX + this.className + "\"))");
342             writer.emitStatement("Table table = transaction.getTable(\"%s%s\")", TABLE_PREFIX, this.className);
343     
344             // For each field generate corresponding table index constant
345             for (VariableElement field : fields) {
346                 String fieldName = field.getSimpleName().toString();
347                 String fieldTypeCanonicalName = field.asType().toString();
348                 String fieldTypeName;
349                 if (fieldTypeCanonicalName.contains(".")) {
350                     fieldTypeName = fieldTypeCanonicalName.substring(fieldTypeCanonicalName.lastIndexOf('.') + 1);
351                 } else {
352                     fieldTypeName = fieldTypeCanonicalName;
353                 }
354     
355                 if (JAVA_TO_REALM_TYPES.containsKey(fieldTypeCanonicalName)) {
356                     writer.emitStatement("table.addColumn(%s, \"%s\")",
357                             JAVA_TO_COLUMN_TYPES.get(fieldTypeCanonicalName),
358                             fieldName);
359                 } else if (typeUtils.isAssignable(field.asType(), realmObject)) {
360                     writer.beginControlFlow("if (!transaction.hasTable(\"%s%s\"))", TABLE_PREFIX, fieldTypeName);
361                     writer.emitStatement("%s%s.initTable(transaction)", fieldTypeName, PROXY_SUFFIX);
362                     writer.endControlFlow();
363                     writer.emitStatement("table.addColumnLink(ColumnType.LINK, \"%s\", transaction.getTable(\"%s%s\"))",
364                             fieldName, TABLE_PREFIX, fieldTypeName);
365                 } else if (typeUtils.isAssignable(field.asType(), realmList)) {
366                     String genericCanonicalType = ((DeclaredType) field.asType()).getTypeArguments().get(0).toString();
367                     String genericType;
368                     if (genericCanonicalType.contains(".")) {
369                         genericType = genericCanonicalType.substring(genericCanonicalType.lastIndexOf('.') + 1);
370                     } else {
371                         genericType = genericCanonicalType;
372                     }
373                     writer.beginControlFlow("if (!transaction.hasTable(\"%s%s\"))", TABLE_PREFIX, genericType);
374                     writer.emitStatement("%s%s.initTable(transaction)", genericType, PROXY_SUFFIX);
375                     writer.endControlFlow();
376                     writer.emitStatement("table.addColumnLink(ColumnType.LINK_LIST, \"%s\", transaction.getTable(\"%s%s\"))",
377                             fieldName, TABLE_PREFIX, genericType);
378                 }
379             }
380     
381             for (VariableElement field : fieldsToIndex) {
382                 String fieldName = field.getSimpleName().toString();
383                 writer.emitStatement("table.setIndex(table.getColumnIndex(\"%s\"))", fieldName);
384             }
385     
386             writer.emitStatement("return table");
387             writer.endControlFlow();
388             writer.emitStatement("return transaction.getTable(\"%s%s\")", TABLE_PREFIX, this.className);
389             writer.endMethod();
390             writer.emitEmptyLine();
391     
392             /**
393              * validateTable method
394              */
395             writer.beginMethod(
396                     "void", // Return type
397                     "validateTable", // Method name
398                     EnumSet.of(Modifier.PUBLIC, Modifier.STATIC), // Modifiers
399                     "ImplicitTransaction", "transaction"); // Argument type & argument name
400     
401             writer.beginControlFlow("if(transaction.hasTable(\"" + TABLE_PREFIX + this.className + "\"))");
402             writer.emitStatement("Table table = transaction.getTable(\"%s%s\")", TABLE_PREFIX, this.className);
403     
404             // verify number of columns
405             writer.beginControlFlow("if(table.getColumnCount() != " + fields.size() + ")");
406             writer.emitStatement("throw new IllegalStateException(\"Column count does not match\")");
407             writer.endControlFlow();
408     
409             // create type dictionary for lookup
410             writer.emitStatement("Map<String, ColumnType> columnTypes = new HashMap<String, ColumnType>()");
411             writer.beginControlFlow("for(long i = 0; i < " + fields.size() + "; i++)");
412             writer.emitStatement("columnTypes.put(table.getColumnName(i), table.getColumnType(i))");
413             writer.endControlFlow();
414     
415             // For each field verify there is a corresponding column
416             ListIterator<VariableElement> fieldsIterator = fields.listIterator();
417             while (fieldsIterator.hasNext()) {
418                 int columnNumber = fieldsIterator.nextIndex();
419                 VariableElement field = fieldsIterator.next();
420     
421                 String fieldName = field.getSimpleName().toString();
422                 String fieldTypeCanonicalName = field.asType().toString();
423                 String fieldTypeName;
424                 if (fieldTypeCanonicalName.contains(".")) {
425                     fieldTypeName = fieldTypeCanonicalName.substring(fieldTypeCanonicalName.lastIndexOf('.') + 1);
426                 } else {
427                     fieldTypeName = fieldTypeCanonicalName;
428                 }
429     
430                 if (JAVA_TO_REALM_TYPES.containsKey(fieldTypeCanonicalName)) {
431                     // make sure types align
432                     writer.beginControlFlow("if (!columnTypes.containsKey(\"%s\"))", fieldName);
433                     writer.emitStatement("throw new IllegalStateException(\"Missing column '%s'\")", fieldName);
434                     writer.endControlFlow();
435                     writer.beginControlFlow("if (columnTypes.get(\"%s\") != %s)", fieldName, JAVA_TO_COLUMN_TYPES.get(fieldTypeCanonicalName));
436                     writer.emitStatement("throw new IllegalStateException(\"Invalid type '%s' for column '%s'\")",
437                             fieldTypeName, fieldName);
438                     writer.endControlFlow();
439                 } else if (typeUtils.isAssignable(field.asType(), realmObject)) { // Links
440                     writer.beginControlFlow("if (!columnTypes.containsKey(\"%s\"))", fieldName);
441                     writer.emitStatement("throw new IllegalStateException(\"Missing column '%s'\")", fieldName);
442                     writer.endControlFlow();
443                     writer.beginControlFlow("if (columnTypes.get(\"%s\") != ColumnType.LINK)", fieldName);
444                     writer.emitStatement("throw new IllegalStateException(\"Invalid type '%s' for column '%s'\")",
445                             fieldTypeName, fieldName);
446                     writer.endControlFlow();
447                     writer.beginControlFlow("if (!transaction.hasTable(\"%s%s\"))", TABLE_PREFIX, fieldTypeName);
448                     writer.emitStatement("throw new IllegalStateException(\"Missing table '%s%s' for column '%s'\")",
449                             TABLE_PREFIX, fieldTypeName, fieldName);
450                     writer.endControlFlow();
451                     // TODO: Replace with a proper comparison
452     //                writer.emitStatement("Table table_%d = transaction.getTable(\"%s%s\")", columnNumber, TABLE_PREFIX, fieldTypeName);
453     //                writer.beginControlFlow("if (table.getLinkTarget(%d).equals(table_%d))", columnNumber, columnNumber);
454     //                writer.emitStatement("throw new IllegalStateException(\"Mismatching link tables for column '%s'\")",
455     //                        fieldName);
456     //                writer.endControlFlow();
457                 } else if (typeUtils.isAssignable(field.asType(), realmList)) { // Link Lists
458                     String genericCanonicalType = ((DeclaredType) field.asType()).getTypeArguments().get(0).toString();
459                     String genericType;
460                     if (genericCanonicalType.contains(".")) {
461                         genericType = genericCanonicalType.substring(genericCanonicalType.lastIndexOf('.') + 1);
462                     } else {
463                         genericType = genericCanonicalType;
464                     }
465                     writer.beginControlFlow("if(!columnTypes.containsKey(\"%s\"))", fieldName);
466                     writer.emitStatement("throw new IllegalStateException(\"Missing column '%s'\")", fieldName);
467                     writer.endControlFlow();
468                     writer.beginControlFlow("if(columnTypes.get(\"%s\") != ColumnType.LINK_LIST)", fieldName);
469                     writer.emitStatement("throw new IllegalStateException(\"Invalid type '%s' for column '%s'\")",
470                             genericType, fieldName);
471                     writer.endControlFlow();
472                     writer.beginControlFlow("if (!transaction.hasTable(\"%s%s\"))", TABLE_PREFIX, genericType);
473                     writer.emitStatement("throw new IllegalStateException(\"Missing table '%s%s' for column '%s'\")",
474                             TABLE_PREFIX, genericType, fieldName);
475                     writer.endControlFlow();
476                     // TODO: Replace with a proper comparison
477     //                writer.emitStatement("Table table_%d = transaction.getTable(\"%s%s\")", columnNumber, TABLE_PREFIX, genericType);
478     //                writer.beginControlFlow("if (table.getLinkTarget(%d).equals(table_%d))", columnNumber, columnNumber);
479     //                writer.emitStatement("throw new IllegalStateException(\"Mismatching link list tables for column '%s'\")",
480     //                        fieldName);
481     //                writer.endControlFlow();
482                 }
483             }
484             writer.endControlFlow();
485             writer.endMethod();
486             writer.emitEmptyLine();
487     
488             /**
489              * getFieldNames method
490              */
491             writer.beginMethod("List<String>", "getFieldNames", EnumSet.of(Modifier.PUBLIC, Modifier.STATIC));
492             List<String> entries = new ArrayList<String>();
493             for (VariableElement field : fields) {
494                 String fieldName = field.getSimpleName().toString();
495                 entries.add(String.format("\"%s\"", fieldName));
496             }
497             String statementSection = joinStringList(entries, ", ");
498             writer.emitStatement("return Arrays.asList(%s)", statementSection);
499             writer.endMethod();
500             writer.emitEmptyLine();
501     
502             /**
503              * toString method
504              */
505             writer.emitAnnotation("Override");
506             writer.beginMethod("String", "toString", EnumSet.of(Modifier.PUBLIC));
507             writer.emitStatement("StringBuilder stringBuilder = new StringBuilder(\"%s = [\")", className);
508             for (VariableElement field : fields) {
509                 String fieldName = field.getSimpleName().toString();
510                 String fieldTypeCanonicalName = field.asType().toString();
511                 String getterPrefix = fieldTypeCanonicalName.equals("boolean")?"is":"get";
512                 writer.emitStatement("stringBuilder.append(\"{%s:\")", fieldName);
513                 writer.emitStatement("stringBuilder.append(%s%s())", getterPrefix, capitaliseFirstChar(fieldName));
514                 writer.emitStatement("stringBuilder.append(\"} \")", fieldName);
515             }
516             writer.emitStatement("stringBuilder.append(\"]\")");
517             writer.emitStatement("return stringBuilder.toString()");
518             writer.endMethod();
519             writer.emitEmptyLine();
520     
521             /**
522              * hashCode method
523              */
524             writer.emitAnnotation("Override");
525             writer.beginMethod("int", "hashCode", EnumSet.of(Modifier.PUBLIC));
526             writer.emitStatement("int result = 17");
527             int counter = 0;
528             for (VariableElement field : fields) {
529                 String fieldName = field.getSimpleName().toString();
530                 String fieldTypeCanonicalName = field.asType().toString();
531                 if (HASHCODE.containsKey(fieldTypeCanonicalName)) {
532                     for (String statement : HASHCODE.get(fieldTypeCanonicalName)) {
533                         if (statement.contains("%d") && statement.contains("%s")) {
534                             // This statement introduces a temporary variable
535                             writer.emitStatement(statement, counter, capitaliseFirstChar(fieldName));
536                         } else if(statement.contains("%d")) {
537                             // This statement uses the temporary variable
538                             writer.emitStatement(statement, counter, counter);
539                         } else if (statement.contains("%s")) {
540                             // This is a normal statement with only one assignment
541                             writer.emitStatement(statement, capitaliseFirstChar(fieldName));
542                         } else {
543                             // This should never happen
544                             throw new AssertionError();
545                         }
546                     }
547                 } else {
548                     // Links and Link lists
549                     writer.emitStatement("%s temp_%d = get%s()", fieldTypeCanonicalName, counter, capitaliseFirstChar(fieldName));
550                     writer.emitStatement("result = 31 * result + (temp_%d != null ? temp_%d.hashCode() : 0)", counter, counter);
551                 }
552                 counter++;
553             }
554             writer.emitStatement("return result");
555             writer.endMethod();
556             writer.emitEmptyLine();
557     
558             /**
559              * equals method
560              */
561             String proxyClassName = className + PROXY_SUFFIX;
562             writer.emitAnnotation("Override");
563             writer.beginMethod("boolean", "equals", EnumSet.of(Modifier.PUBLIC), "Object", "o");
564             writer.emitStatement("if (this == o) return true");
565             writer.emitStatement("if (o == null || getClass() != o.getClass()) return false");
566             writer.emitStatement("%s a%s = (%s)o", proxyClassName, className, proxyClassName);  // FooRealmProxy aFoo = (FooRealmProxy)o
567     
568             for (VariableElement field : fields) {
569                 String fieldName = field.getSimpleName().toString();
570                 String capFieldName = capitaliseFirstChar(fieldName);
571                 String fieldTypeCanonicalName = field.asType().toString();
572                 if (HOW_TO_EQUAL.containsKey(fieldTypeCanonicalName)) {
573                     switch (HOW_TO_EQUAL.get(fieldTypeCanonicalName)) {
574                         case EQUALS_DIRECT: // if (getField() != aFoo.getField()) return false
575                             String getterPrefix = fieldTypeCanonicalName.equals("boolean") ? "is" : "get";
576                             writer.emitStatement("if (%s%s() != a%s.%s%s()) return false", getterPrefix, capFieldName, className, getterPrefix, capFieldName);
577                             break;
578                         case EQUALS_NULL: // if (getField() != null = !getField().equals(aFoo.getField()) : aFoo.getField() != null) return false
579                             writer.emitStatement("if (get%s() != null ? !get%s().equals(a%s.get%s()) : a%s.get%s() != null) return false",
580                                     capFieldName,
581                                     capFieldName, className, capFieldName,
582                                     className, capFieldName);
583                             break;
584                         case EQUALS_ARRAY: // if (!Arrays.equals(getField(), aFoo.getField())) return false
585                             writer.emitStatement("if (!Arrays.equals(get%s(), a%s.get%s())) return false",
586                                     capFieldName,
587                                     className, capFieldName);
588                             break;
589                         case EQUALS_COMPARE: // if (
590                             writer.emitStatement("if (%s.compare(get%s(), a%s.get%s()) != 0) return false",
591                                     capitaliseFirstChar(fieldTypeCanonicalName), capitaliseFirstChar(fieldName), className,
592                                     capitaliseFirstChar(fieldName));
593                             break;
594                     }
595                 }
596                 else if (typeUtils.isAssignable(field.asType(), realmObject) || typeUtils.isAssignable(field.asType(), realmList)) {
597                     writer.emitStatement("if (get%s() != null ? !get%s().equals(a%s.get%s()) : a%s.get%s() != null) return false",
598                             capFieldName,
599                             capFieldName, className, capFieldName,
600                              className, capFieldName);
601                 }
602             }
603             writer.emitStatement("return true");
604             writer.endMethod();
605             writer.emitEmptyLine();
606     
607             // End the class definition
608             writer.endType();
609             writer.close();
610         }

612         private static String capitaliseFirstChar(String input) {
613             return input.substring(0, 1).toUpperCase() + input.substring(1);
614         }

616         public static String joinStringList(List<String> strings, String separator) {
617             StringBuilder stringBuilder = new StringBuilder();
618             ListIterator<String> iterator = strings.listIterator();
619             while (iterator.hasNext()) {
620                 int index = iterator.nextIndex();
621                 String item = iterator.next();
622     
623                 if (index > 0) {
624                     stringBuilder.append(separator);
625                 }
626                 stringBuilder.append(item);
627             }
628             return stringBuilder.toString();
629         }
