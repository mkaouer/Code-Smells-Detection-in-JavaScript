17     package io.realm.examples.realmgridview;
18     
19     import android.content.Context;
20     import android.view.LayoutInflater;
21     import android.view.View;
22     import android.view.ViewGroup;
23     import android.widget.BaseAdapter;
24     import android.widget.TextView;
25     
26     import java.util.List;
27     
28     // This adapter is strictly to interface with the GridView and doesn't
29     // particular show much interesting Realm functionality.
30     
31     // Alternatively from this example,
32     // a developer could update the getView() to pull items from the Realm.
33     
34     public class CityAdapter extends BaseAdapter {
35     
36         public static final String TAG = RealmExampleActivity.class.getName();
37     
38         private LayoutInflater inflater;
39     
40         private List<City> cities = null;
41     
42         public CityAdapter(Context context) {
43             inflater = (LayoutInflater) context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
44         }
45     
46         public void setData(List<City> details) {
47             this.cities = details;
48         }
49     
50         @Override
51         public int getCount() {
52             if (cities == null) {
53                 return 0;
54             }
55             return cities.size();
56         }
57     
58         @Override
59         public Object getItem(int position) {
60             if (cities == null || cities.get(position) == null) {
61                 return null;
62             }
63             return cities.get(position);
64         }
65     
66         @Override
67         public long getItemId(int i) {
68             return i;
69         }
70     
71         @Override
72         public View getView(int position, View currentView, ViewGroup parent) {
73             if (currentView == null) {
74                 currentView = inflater.inflate(R.layout.city_listitem, parent, false);
75             }
76     
77             City city = cities.get(position);
78     
79             if (city != null) {
80                 ((TextView) currentView.findViewById(R.id.name)).setText(city.getName());
81                 ((TextView) currentView.findViewById(R.id.votes)).setText(Long.toString(city.getVotes()));
82             }
83     
84             return currentView;
85         }
86     }
