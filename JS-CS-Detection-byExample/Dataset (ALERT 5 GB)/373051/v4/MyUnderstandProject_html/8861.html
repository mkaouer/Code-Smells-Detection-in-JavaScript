69     #define GET_ARRAY() \
70         jboolean isCopy; \
71         jsize arr_len = env->GetArrayLength(columnIndexes); \
72         jlong *arr = env->GetLongArrayElements(columnIndexes, &isCopy);

74     
75     #define RELEASE_ARRAY()     \
76         if (isCopy == JNI_TRUE) \
77             env->ReleaseLongArrayElements(columnIndexes, arr, 0);

79     TableRef getTableLink(jlong nativeQueryPtr, jlong *arr, jsize arr_len) {
80         TableRef tbl = Q(nativeQueryPtr)->get_table();
81         for (int i=0; i<arr_len-1; i++) {
82             tbl->link(size_t(arr[i]));
83         }
84         return tbl;
85     }
86     
87     template <typename coretype, typename cpptype, typename javatype>
88     Query numeric_link_equal(TableRef tbl, jlong columnIndex, javatype value) {
89         return tbl->column<coretype>(size_t(columnIndex)) == cpptype(value);
90     }
91     
92     template <typename coretype, typename cpptype, typename javatype>
93     Query numeric_link_notequal(TableRef tbl, jlong columnIndex, javatype value) {
94         return tbl->column<coretype>(size_t(columnIndex)) != cpptype(value);
95     }
96     
97     template <typename coretype, typename cpptype, typename javatype>
98     Query numeric_link_greater(TableRef tbl, jlong columnIndex, javatype value) {
99         return tbl->column<coretype>(size_t(columnIndex)) > cpptype(value);
100     }
101     
102     template <typename coretype, typename cpptype, typename javatype>
103     Query numeric_link_greaterequal(TableRef tbl, jlong columnIndex, javatype value) {
104         return tbl->column<coretype>(size_t(columnIndex)) >= cpptype(value);
105     }
106     
107     template <typename coretype, typename cpptype, typename javatype>
108     Query numeric_link_less(TableRef tbl, jlong columnIndex, javatype value) {
109         return tbl->column<coretype>(size_t(columnIndex)) < cpptype(value);
110     }
111     
112     template <typename coretype, typename cpptype, typename javatype>
113     Query numeric_link_lessequal(TableRef tbl, jlong columnIndex, javatype value) {
114         return tbl->column<coretype>(size_t(columnIndex)) <= cpptype(value);
115     }

120     JNIEXPORT void JNICALL Java_io_realm_internal_TableQuery_nativeEqual__J_3JJ(
121         JNIEnv* env, jobject, jlong nativeQueryPtr, jlongArray columnIndexes, jlong value)
122     {
123         GET_ARRAY()
124         try {
125             if (arr_len == 1) {
126                 if (!QUERY_COL_TYPE_VALID(env, nativeQueryPtr, arr[0], type_Int))
127                     return;
128                 Q(nativeQueryPtr)->equal(S(arr[0]), static_cast<int64_t>(value));
129             }
130             else {
131                 TableRef tbl = getTableLink(nativeQueryPtr, arr, arr_len);
132                 Q(nativeQueryPtr)->and_query(numeric_link_equal<Int, int64_t, jlong>(tbl, arr[arr_len-1], value));
133             }
134         } CATCH_STD()
135         RELEASE_ARRAY()
136     }

831         JNIEnv* env, jobject, jlong nativeQueryPtr, jlong fromTableRow)
832     {
833         Query* pQuery = Q(nativeQueryPtr);
834         Table* pTable = Ref2Ptr(pQuery->get_table());
835         if (!QUERY_VALID(env, pQuery))
836             return -1;
837         // It's valid to go 1 past the end index
838         if ((fromTableRow < 0) || (S(fromTableRow) > pTable->size())) {
839             // below check will fail with appropriate exception
840             (void) ROW_INDEX_VALID(env, pTable, fromTableRow);
841             return -1;
842         }
843     
844         try {
845             size_t r = pQuery->find( S(fromTableRow) );
846             return (r == not_found) ? jlong(-1) : jlong(r);
847         } CATCH_STD()
848         return -1;
849     }
