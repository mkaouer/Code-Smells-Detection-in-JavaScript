20     jint java_lang_List_size(JNIEnv* env, jobject jList)
21     {
22         // WARNING: do not cache these methods, list class may be different based on the object jlist
23         jclass jListClass = env->GetObjectClass(jList);
24         if (jListClass == NULL)
25             return 0;
26         jmethodID jListSizeMethodId = env->GetMethodID(jListClass, "size", "()I");
27         if (jListSizeMethodId == NULL) {
28             ThrowException(env, NoSuchMethod, "jList", "size");
29             return 0;
30         }
31         return env->CallIntMethod(jList, jListSizeMethodId);
32     }

34     jobject java_lang_List_get(JNIEnv* env, jobject jList, jint index)
35     {
36         // WARNING: do not cache these methods/classes, list class may be different based on the object jlist
37         jclass jListClass = env->GetObjectClass(jList);
38          if (jListClass == NULL)
39             return NULL;
40         jmethodID jListGetMethodId = env->GetMethodID(jListClass, "get", "(I)Ljava/lang/Object;");
41         if (jListGetMethodId == NULL) {
42             ThrowException(env, NoSuchMethod, "jList", "get");
43             return NULL;
44         }
45         return env->CallObjectMethod(jList, jListGetMethodId, index);
46     }
