21     static jfieldID GetFieldIDColumnType(JNIEnv* env, const char* methodStr, const char* typeStr)
22     {
23         static jclass myClass = GetClass(env, "io/realm/internal/ColumnType");
24         if (myClass == NULL)
25             return NULL;
26     
27         jfieldID myField = env->GetFieldID(myClass, methodStr, typeStr);
28         if (myField== NULL) {
29             ThrowException(env, NoSuchField, "ColumnType", methodStr);
30             return NULL;
31         }
32         return myField;
33     }

35     DataType GetColumnTypeFromJColumnType(JNIEnv* env, jobject jColumnType)
36     {
37         static jfieldID jValueFieldId = GetFieldIDColumnType(env, "nativeValue", "I");
38         if (jValueFieldId == NULL)
39             return DataType(0);
40     
41         jint columnType = env->GetIntField(jColumnType, jValueFieldId);
42         return static_cast<DataType>(columnType);
43     }
