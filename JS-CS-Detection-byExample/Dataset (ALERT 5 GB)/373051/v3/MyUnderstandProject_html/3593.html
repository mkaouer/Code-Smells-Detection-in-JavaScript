181         public void testMustAllowDoubleCommitAndRollback() {
182             SharedGroup db = new SharedGroup(testFile, SharedGroup.Durability.FULL, null);
183             {
184                 WriteTransaction trans = db.beginWrite();
185                 Table tbl = trans.getTable("EmployeeTable");
186                 tbl.addColumn(ColumnType.STRING, "name");
187                 tbl.addColumn(ColumnType.INTEGER, "number");
188     
189                 // allow commit before any changes
190                 assertEquals(0, tbl.size());
191                 tbl.add("Hello", 1);
192                 trans.commit();
193             }
194             {
195                 WriteTransaction trans = db.beginWrite();
196                 Table tbl = trans.getTable("EmployeeTable");
197                 // allow double rollback
198                 tbl.add("Hello", 2);
199                 assertEquals(2, tbl.size());
200                 trans.rollback();
201                 trans.rollback();
202                 trans.rollback();
203                 trans.rollback();
204             }
205             {
206                 ReadTransaction trans = db.beginRead();
207                 Table tbl = trans.getTable("EmployeeTable");
208                 assertEquals(1, tbl.size());
209                 trans.endRead();
210             }
