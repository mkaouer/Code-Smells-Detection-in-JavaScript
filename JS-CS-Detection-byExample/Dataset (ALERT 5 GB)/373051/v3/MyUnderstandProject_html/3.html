17     #include <sstream>
18     #include <string>
19     
20     #include "util.hpp"
21     #include "io_realm_internal_Version.h"
22     #include <tightdb/version.hpp>
23     
24     static int tightdb_jni_version = 23;
25     
26     
27     using namespace tightdb;
28     
29     JNIEXPORT jint JNICALL Java_io_realm_internal_Version_nativeGetAPIVersion(JNIEnv*, jclass)
30     {
31         return tightdb_jni_version;
32     }
33     
34     JNIEXPORT jstring JNICALL Java_io_realm_internal_Version_nativeGetVersion(JNIEnv *env, jclass)
35     {
36         try {
37             return to_jstring(env, Version::get_version());
38         }
39         CATCH_STD();
40         return NULL;
41     }
42     
43     JNIEXPORT jboolean JNICALL Java_io_realm_internal_Version_nativeHasFeature(JNIEnv *env, jclass, jint feature)
44     {
45         switch (feature) {
46             case 0:
47                 return Version::has_feature(feature_Debug);
48             case 1:
49                 return Version::has_feature(feature_Replication);
50             default: {
51                 std::ostringstream ss;
52                 ss << "Unknown feature code: " << feature;
53                 ThrowException(env, RuntimeError, ss.str());
54             }
55         }
56         return false;
57     }
58     
59     JNIEXPORT jboolean JNICALL Java_io_realm_internal_Version_nativeIsAtLeast(JNIEnv *, jclass,
60         jint major, jint minor, jint patch)
61     {
62         return Version::is_at_least(major, minor, patch);
63     }
