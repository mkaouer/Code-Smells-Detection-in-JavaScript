31     void ThrowException(JNIEnv* env, ExceptionKind exception, std::string classStr, std::string itemStr)
32     {
33         std::string message;
34         jclass jExceptionClass = NULL;
35     
36         TR_ERR((env, "\njni: ThrowingException %d, %s, %s.\n", exception, classStr.c_str(), itemStr.c_str()));
37     
38         switch (exception) {
39             case ClassNotFound:
40                 jExceptionClass = env->FindClass("java/lang/ClassNotFoundException");
41                 message = "Class '" + classStr + "' could not be located.";
42                 break;
43     
44             case NoSuchField:
45                 jExceptionClass = env->FindClass("java/lang/NoSuchFieldException");
46                 message = "Field '" + itemStr + "' could not be located in class io.realm." + classStr;
47                 break;
48     
49             case NoSuchMethod:
50                 jExceptionClass = env->FindClass("java/lang/NoSuchMethodException");
51                 message = "Method '" + itemStr + "' could not be located in class io.realm." + classStr;
52                 break;
53     
54             case IllegalArgument:
55                 jExceptionClass = env->FindClass("java/lang/IllegalArgumentException");
56                 message = "Illegal Argument: " + classStr;
57                 break;
58     
59             case TableInvalid:
60                 jExceptionClass = env->FindClass("java/lang/IllegalStateException");
61                 message = "Illegal State: " + classStr;
62                 break;
63     
64             case IOFailed:
65                 jExceptionClass = env->FindClass("io/realm/exceptions/RealmIOException");
66                 message = "Failed to open " + classStr + ". " + itemStr;
67                 break;
68     
69             case FileNotFound:
70                 jExceptionClass = env->FindClass("io/realm/exceptions/RealmIOException");
71                 message = "File not found: " + classStr + ".";
72                 break;
73     
74             case FileAccessError:
75                 jExceptionClass = env->FindClass("io/realm/exceptions/RealmIOException");
76                 message = "Failed to access: " + classStr + ". " + itemStr;
77                 break;
78     
79             case IndexOutOfBounds:
80                 jExceptionClass = env->FindClass("java/lang/ArrayIndexOutOfBoundsException");
81                 message = classStr;
82                 break;
83     
84             case UnsupportedOperation:
85                 jExceptionClass = env->FindClass("java/lang/UnsupportedOperationException");
86                 message = classStr;
87                 break;
88     
89             case OutOfMemory:
90                 jExceptionClass = env->FindClass("io/realm/internal/OutOfMemoryError");
91                 message = classStr + " " + itemStr;
92                 break;
93     
94             case Unspecified:
95                 jExceptionClass = env->FindClass("java/lang/RuntimeException");
96                 message = "Unspecified exception. " + classStr;
97                 break;
98     
99             case RuntimeError:
100                 jExceptionClass = env->FindClass("java/lang/RuntimeException");
101                 message = classStr;
102                 break;
103     
104             case RowInvalid:
105                 jExceptionClass = env->FindClass("java/lang/IllegalStateException");
106                 message = "Illegal State: " + classStr;
107                 break;
108         }
109         if (jExceptionClass != NULL)
110             env->ThrowNew(jExceptionClass, message.c_str());
111         else {
112             TR_ERR((env, "\nERROR: Couldn't throw exception.\n"));
113         }
114     
115         env->DeleteLocalRef(jExceptionClass);
116     }

118     jclass GetClass(JNIEnv* env, const char* classStr)
119     {
120         jclass localRefClass = env->FindClass(classStr);
121         if (localRefClass == NULL) {
122             ThrowException(env, ClassNotFound, classStr);
123             return NULL;
124         }
125     
126         jclass myClass = reinterpret_cast<jclass>( env->NewGlobalRef(localRefClass) );
127         env->DeleteLocalRef(localRefClass);
128         return myClass;
129     }

131     void jprint(JNIEnv *env, char *txt)
132     {
133     #if 1
134         static_cast<void>(env);
135         fprintf(stderr, " -- JNI: %s", txt);  fflush(stderr);
136     #else
137         static jclass myClass = GetClass(env, "io/realm/internal/util");
138         static jmethodID myMethod = env->GetStaticMethodID(myClass, "javaPrint", "(Ljava/lang/String;)V");
139         if (myMethod)
140             env->CallStaticVoidMethod(myClass, myMethod, env->NewStringUTF(txt));
141     #endif
142     }

155     bool GetBinaryData(JNIEnv* env, jobject jByteBuffer, tightdb::BinaryData& bin)
156     {
157         const char* data = static_cast<char*>(env->GetDirectBufferAddress(jByteBuffer));
158         if (!data) {
159             ThrowException(env, IllegalArgument, "ByteBuffer is invalid");
160             return false;
161         }
162         jlong size = env->GetDirectBufferCapacity(jByteBuffer);
163         if (size < 0) {
164             ThrowException(env, IllegalArgument, "Can't get BufferCapacity.");
165             return false;
166         }
167         bin = BinaryData(data, S(size));
168         return true;
169     }

178     // This assumes that 'jchar' is an integral type with at least 16
179     // non-sign value bits, that is, an unsigned 16-bit integer, or any
180     // signed or unsigned integer with more than 16 bits.
181     struct JcharTraits {
182         static jchar to_int_type(jchar c)  TIGHTDB_NOEXCEPT { return c; }
183         static jchar to_char_type(jchar i) TIGHTDB_NOEXCEPT { return i; }
184     };
185     
186     struct JStringCharsAccessor {
187         JStringCharsAccessor(JNIEnv* e, jstring s):
188             m_env(e), m_string(s), m_data(e->GetStringChars(s,0)), m_size(get_size(e,s)) {}
189         ~JStringCharsAccessor()
190         {
191             m_env->ReleaseStringChars(m_string, m_data);
192         }
193         const jchar* data() const TIGHTDB_NOEXCEPT { return m_data; }
194         size_t size() const TIGHTDB_NOEXCEPT { return m_size; }
195     
196     private:
197         JNIEnv* const m_env;
198         const jstring m_string;
199         const jchar* const m_data;
200         const size_t m_size;
201     
202         static size_t get_size(JNIEnv* e, jstring s)
203         {
204             size_t size;
205             if (int_cast_with_overflow_detect(e->GetStringLength(s), size))
206                 throw runtime_error("String size overflow");
207             return size;
208         }
209     };
210     
211     } // anonymous namespace

214     jstring to_jstring(JNIEnv* env, StringData str)
215     {
216         // For efficiency, if the incoming UTF-8 string is sufficiently
217         // small, we will attempt to store the UTF-16 output into a stack
218         // allocated buffer of static size. Otherwise we will have to
219         // dynamically allocate the output buffer after calculating its
220         // size.
221     
222         const size_t stack_buf_size = 48;
223         jchar stack_buf[stack_buf_size];
224         UniquePtr<jchar[]> dyn_buf;
225     
226         const char* in_begin = str.data();
227         const char* in_end   = str.data() + str.size();
228         jchar* out_begin = stack_buf;
229         jchar* out_curr  = stack_buf;
230         jchar* out_end   = stack_buf + stack_buf_size;
231     
232         typedef Utf8x16<jchar, JcharTraits> Xcode;
233     
234         if (str.size() <= stack_buf_size) {
235             if (!Xcode::to_utf16(in_begin, in_end, out_curr, out_end))
236                 goto bad_utf8;
237             if (in_begin == in_end)
238                 goto transcode_complete;
239         }
240     
241         {
242             const char* in_begin2 = in_begin;
243             size_t size = Xcode::find_utf16_buf_size(in_begin2, in_end);
244             if (in_begin2 != in_end) 
245                 goto bad_utf8;
246             if (int_add_with_overflow_detect(size, stack_buf_size))
247                 throw runtime_error("String size overflow");
248             dyn_buf.reset(new jchar[size]);
249             out_curr = copy(out_begin, out_curr, dyn_buf.get());
250             out_begin = dyn_buf.get();
251             out_end   = dyn_buf.get() + size;
252             if (!Xcode::to_utf16(in_begin, in_end, out_curr, out_end))
253                 goto bad_utf8;
254             TIGHTDB_ASSERT(in_begin == in_end);
255         }
256     
257       transcode_complete:
258         {
259             jsize out_size;
260             if (int_cast_with_overflow_detect(out_curr - out_begin, out_size))
261                 throw runtime_error("String size overflow");
262     
263             return env->NewString(out_begin, out_size);
264         }
265     
266       bad_utf8:
267         throw runtime_error("Bad UTF-8 encoding");
268     }

271     JStringAccessor::JStringAccessor(JNIEnv* env, jstring str)
272     {
273         // For efficiency, if the incoming UTF-16 string is sufficiently
274         // small, we will choose an UTF-8 output buffer whose size (in
275         // bytes) is simply 4 times the number of 16-bit elements in the
276         // input. This is guaranteed to be enough. However, to avoid
277         // excessive over allocation, this is not done for larger input
278         // strings.
279     
280         JStringCharsAccessor chars(env, str);
281     
282         typedef Utf8x16<jchar, JcharTraits> Xcode;
283         size_t max_project_size = 48;
284         TIGHTDB_ASSERT(max_project_size <= numeric_limits<size_t>::max()/4);
285         size_t buf_size;
286         if (chars.size() <= max_project_size) {
287             buf_size = chars.size() * 4;
288         }
289         else {
290             const jchar* begin = chars.data();
291             const jchar* end   = begin + chars.size();
292             buf_size = Xcode::find_utf8_buf_size(begin, end);
293         }
294         m_data.reset(new char[buf_size]);  // throws
295         {
296             const jchar* in_begin = chars.data();
297             const jchar* in_end   = in_begin + chars.size();
298             char* out_begin = m_data.get();
299             char* out_end   = m_data.get() + buf_size;
300             if (!Xcode::to_utf8(in_begin, in_end, out_begin, out_end))
301                 throw runtime_error("Bad UTF-16 encoding");
302             TIGHTDB_ASSERT(in_begin == in_end);
303             m_size = out_begin - m_data.get();
304         }
305     }
