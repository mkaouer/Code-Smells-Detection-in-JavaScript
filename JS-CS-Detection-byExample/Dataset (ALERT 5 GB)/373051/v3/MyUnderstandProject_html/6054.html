23     jclass GetClassMixed(JNIEnv* env)
24     {
25         static jclass jMixedClass = GetClass(env, "io/realm/internal/Mixed");
26         return jMixedClass;
27     }

29     jmethodID GetMixedMethodID(JNIEnv* env, const char* methodStr, const char* typeStr)
30     {
31         jclass myClass = GetClassMixed(env);
32         if (myClass == NULL)
33             return NULL;
34     
35         jmethodID myMethod = env->GetMethodID(myClass, methodStr, typeStr);
36         if (myMethod == NULL) {
37             ThrowException(env, NoSuchMethod, "mixed", methodStr);
38             return NULL;
39         }
40         return myMethod;
41     }

43     DataType GetMixedObjectType(JNIEnv* env, jobject jMixed)
44     {
45         // Call Java "Mixed.getType"
46         static jmethodID jGetTypeMethodId = GetMixedMethodID(env, "getType", "()Lio/realm/internal/ColumnType;");
47         if (jGetTypeMethodId == NULL)
48             return DataType(0);
49     
50         // ???TODO optimize
51         jobject jColumnType = env->CallObjectMethod(jMixed, jGetTypeMethodId);
52         return static_cast<DataType>(GetColumnTypeFromJColumnType(env, jColumnType));
53     }

55     jobject CreateJMixedFromMixed(JNIEnv* env, Mixed& mixed)
56     {
57         jclass jMixedClass = GetClassMixed(env);
58         if (jMixedClass == NULL)
59             return NULL;
60     
61         TR((env, "CreateJMixedFromMixed(type %d)\n", mixed.get_type()));
62         switch (mixed.get_type()) {
63         case type_Int:
64         {
65             jmethodID consId = GetMixedMethodID(env, "<init>", "(J)V");
66             if (consId)
67                 return env->NewObject(jMixedClass, consId, mixed.get_int());
68         }
69         case type_Float:
70         {
71             jmethodID consId = GetMixedMethodID(env, "<init>", "(F)V");
72             if (consId)
73                 return env->NewObject(jMixedClass, consId, mixed.get_float());
74         }
75         case type_Double:
76         {
77             jmethodID consId = GetMixedMethodID(env, "<init>", "(D)V");
78             if (consId)
79                 return env->NewObject(jMixedClass, consId, mixed.get_double());
80         }
81         case type_String:
82         {
83             jmethodID consId = GetMixedMethodID(env, "<init>", "(Ljava/lang/String;)V");
84             if (consId)
85                 return env->NewObject(jMixedClass, consId, to_jstring(env, mixed.get_string()));
86         }
87         case type_Bool:
88         {
89             jmethodID consId = GetMixedMethodID(env, "<init>", "(Z)V");
90             if (consId)
91                 return env->NewObject(jMixedClass, consId, mixed.get_bool());
92         }
93         case type_DateTime:
94             {
95                 time_t timeValue = mixed.get_datetime().get_datetime();
96                 jclass jDateClass = env->FindClass("java/util/Date");
97                 if (jDateClass == NULL) {
98                     ThrowException(env, ClassNotFound, "Date");
99                     return NULL;
100                 }
101                 jmethodID jDateConsId = env->GetMethodID(jDateClass, "<init>", "(J)V");
102                 if (jDateConsId == NULL) {
103                     ThrowException(env, NoSuchMethod, "Date", "<init>");
104                     return NULL;
105                 }
106                 jobject jDate = env->NewObject(jDateClass, jDateConsId, static_cast<jlong>(timeValue));
107                 jmethodID consId = GetMixedMethodID(env, "<init>", "(Ljava/util/Date;)V");
108                 if (consId)
109                     return env->NewObject(jMixedClass, consId, jDate);
110             }
111         case type_Binary:
112             {
113                 BinaryData binaryData = mixed.get_binary();
114                 jmethodID consId = GetMixedMethodID(env, "<init>", "(Ljava/nio/ByteBuffer;)V");
115                 if (consId) {
116                     jobject jByteBuffer = env->NewDirectByteBuffer(const_cast<char*>(binaryData.data()), binaryData.size());
117                     return env->NewObject(jMixedClass, consId, jByteBuffer);
118                 }
119             }
120         case type_Table:
121             {
122                 // param input: Table* t.
123                 TR((env, "   --Mixed(type_Table)\n"));
124                 jmethodID consId = GetMixedMethodID(env, "<init>", "(Lio/realm/internal/ColumnType;)V");
125     
126                 jobject jColumnType = NULL; // GetJColumnTypeFromColumnType(env, type_Table);
127                 if (consId)
128                     return env->NewObject(jMixedClass, consId, jColumnType);
129             }
130         case type_Mixed:
131             break;
132         case type_Link:
133             break;
134         case type_LinkList:
135             break;
136         }
137     
138         return NULL;
139     }
