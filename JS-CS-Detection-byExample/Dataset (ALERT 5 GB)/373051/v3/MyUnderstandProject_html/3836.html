17     package io.realm.internal;
18     
19     public class LinkView {
20     
21         private final Context context;
22         private final long nativeLinkViewPtr;
23         private final Table parent;
24         private final long columnIndexInParent;
25     
26         public LinkView(Context context, Table parent, long columnIndexInParent, long nativeLinkViewPtr) {
27             this.context = context;
28             this.parent = parent;
29             this.columnIndexInParent = columnIndexInParent;
30             this.nativeLinkViewPtr = nativeLinkViewPtr;
31         }
32     
33         protected static native void nativeClose(long nativeLinkViewPtr);
34     
35         public Row get(long pos) {
36             long nativeRowPtr = nativeGetRow(nativeLinkViewPtr, pos);
37             return new Row(context, parent.getLinkTarget(columnIndexInParent), nativeRowPtr);
38         }
39         private native long nativeGetRow(long nativeLinkViewPtr, long pos);
40     
41         public long getTargetRowIndex(long pos) {
42             return nativeGetTargetRowIndex(nativeLinkViewPtr, pos);
43         }
44         private native long nativeGetTargetRowIndex(long nativeLinkViewPtr, long pos);
45     
46         public void add(long rowIndex) {
47             nativeAdd(nativeLinkViewPtr, rowIndex);
48         }
49         private native void nativeAdd(long nativeLinkViewPtr, long rowIndex);
50     
51         public void insert(long pos, long rowIndex) {
52             nativeInsert(nativeLinkViewPtr, pos, rowIndex);
53         }
54         private native void nativeInsert(long nativeLinkViewPtr, long pos, long rowIndex);
55     
56         public void set(long pos, long rowIndex) {
57             nativeSet(nativeLinkViewPtr, pos, rowIndex);
58         }
59         private native void nativeSet(long nativeLinkViewPtr, long pos, long rowIndex);
60     
61         public void move(long oldPos, long newPos) {
62             nativeMove(nativeLinkViewPtr, oldPos, newPos);
63         }
64         private native void nativeMove(long nativeLinkViewPtr, long oldPos, long newPos);
65     
66         public void remove(long pos) {
67             nativeRemove(nativeLinkViewPtr, pos);
68         }
69         private native void nativeRemove(long nativeLinkViewPtr, long pos);
70     
71         public void clear() {
72             nativeClear(nativeLinkViewPtr);
73         }
74         private native void nativeClear(long nativeLinkViewPtr);
75     
76         public long size() {
77             return nativeSize(nativeLinkViewPtr);
78         }
79         private native long nativeSize(long nativeLinkViewPtr);
80     
81         public boolean isEmpty() {
82             return nativeIsEmpty(nativeLinkViewPtr);
83         }
84         private native boolean nativeIsEmpty(long nativeLinkViewPtr);
85     }
