17     package io.realm.examples.concurrency.threads;
18     
19     import android.content.Context;
20     import android.util.Log;
21     
22     import io.realm.Realm;
23     import io.realm.examples.concurrency.model.Dog;
24     import io.realm.examples.concurrency.model.Person;
25     
26     public class RealmWriter extends Thread implements KillableThread {
27     
28         public static final String TAG = RealmWriter.class.getName();
29     
30         private Context context;
31         private boolean mRunning = true;
32         private int mInsertCount = 0;
33     
34         public RealmWriter(Context context) {
35             this.context = context;
36         }
37     
38         public void run() {
39             Realm realm = Realm.getInstance(context, false);
40     
41             int count = 0;
42     
43             while (count < mInsertCount && mRunning) {
44                 realm.beginTransaction();
45     
46                 Person person = realm.createObject(Person.class);
47                 person.setName("Foo" + count);
48                 person.setAge(count % 20 + (50 - 20));
49     
50                 //Add a dog to every 50th person
51                 if (count % 50 == 0) {
52                     Dog dog = realm.createObject(Dog.class);
53                     dog.setName("Foo" + count + "Fido");
54                     person.setDog(dog);
55                 }
56     
57                 count++;
58                 realm.commitTransaction();
59     
60                 if ((count % 1000) == 0) {
61                     Log.d(TAG, "WriteOperation#: " + count + "," + Thread.currentThread().getName());
62                 }
63             }
64         }
65     
66         @Override
67         public void terminate() {
68             mRunning = false;
69         }
70     
71         @SuppressWarnings("UnusedDeclaration")
72         public int getInsertCount() {
73             return mInsertCount;
74         }
75     
76         public void setInsertCount(int count) {
77             this.mInsertCount = count;
78         }
79     }
