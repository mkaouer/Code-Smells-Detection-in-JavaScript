39     {
40         TableRef pTable = pQuery->get_table();

67     // helper functions and macros
68     #define GET_ARRAY() \
69         jsize arr_len = env->GetArrayLength(columnIndexes); \
70         jlong *arr = env->GetLongArrayElements(columnIndexes, NULL);

73     #define RELEASE_ARRAY() \
74         env->ReleaseLongArrayElements(columnIndexes, arr, 0);

76     TableRef getTableLink(jlong nativeQueryPtr, jlong *arr, jsize arr_len) {
77         TableRef tbl = Q(nativeQueryPtr)->get_table();
78         for (int i=0; i<arr_len-1; i++) {
79             tbl->link(size_t(arr[i]));
80         }
81         return tbl;
82     }
83     
84     template <typename coretype, typename cpptype, typename javatype>
85     Query numeric_link_equal(TableRef tbl, jlong columnIndex, javatype value) {
86         return tbl->column<coretype>(size_t(columnIndex)) == cpptype(value);
87     }
88     
89     template <typename coretype, typename cpptype, typename javatype>
90     Query numeric_link_notequal(TableRef tbl, jlong columnIndex, javatype value) {
91         return tbl->column<coretype>(size_t(columnIndex)) != cpptype(value);
92     }
93     
94     template <typename coretype, typename cpptype, typename javatype>
95     Query numeric_link_greater(TableRef tbl, jlong columnIndex, javatype value) {
96         return tbl->column<coretype>(size_t(columnIndex)) > cpptype(value);
97     }
98     
99     template <typename coretype, typename cpptype, typename javatype>
100     Query numeric_link_greaterequal(TableRef tbl, jlong columnIndex, javatype value) {
101         return tbl->column<coretype>(size_t(columnIndex)) >= cpptype(value);
102     }
103     
104     template <typename coretype, typename cpptype, typename javatype>
105     Query numeric_link_less(TableRef tbl, jlong columnIndex, javatype value) {
106         return tbl->column<coretype>(size_t(columnIndex)) < cpptype(value);
107     }
108     
109     template <typename coretype, typename cpptype, typename javatype>
110     Query numeric_link_lessequal(TableRef tbl, jlong columnIndex, javatype value) {
111         return tbl->column<coretype>(size_t(columnIndex)) <= cpptype(value);
112     }

117     JNIEXPORT void JNICALL Java_io_realm_internal_TableQuery_nativeEqual__J_3JJ(
118         JNIEnv* env, jobject, jlong nativeQueryPtr, jlongArray columnIndexes, jlong value)
119     {
120         GET_ARRAY()
121         try {
122             if (arr_len == 1) {
123                 if (!QUERY_COL_TYPE_VALID(env, nativeQueryPtr, arr[0], type_Int))
124                     return;
125                 Q(nativeQueryPtr)->equal(S(arr[0]), static_cast<int64_t>(value));
126             }
127             else {
128                 TableRef tbl = getTableLink(nativeQueryPtr, arr, arr_len);
129                 Q(nativeQueryPtr)->and_query(numeric_link_equal<Int, int64_t, jlong>(tbl, arr[arr_len-1], value));
130             }
131         } CATCH_STD()
132         RELEASE_ARRAY()
133     }

829     {
830         Query* pQuery = Q(nativeQueryPtr);
831         Table* pTable = Ref2Ptr(pQuery->get_table());
832         if (!QUERY_VALID(env, pQuery))
833             return -1;
834         // It's valid to go 1 past the end index
835         if ((fromTableRow < 0) || (S(fromTableRow) > pTable->size())) {
836             // below check will fail with appropriate exception
837             (void) ROW_INDEX_VALID(env, pTable, fromTableRow);
838             return -1;
839         }
840     
841         try {
842             size_t r = pQuery->find( S(fromTableRow) );
843             return (r == not_found) ? jlong(-1) : jlong(r);
844         } CATCH_STD()
845         return -1;
846     }
