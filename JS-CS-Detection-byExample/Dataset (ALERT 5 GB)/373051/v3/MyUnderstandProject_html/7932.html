33             if (DEBUG)

40         public void close() {
41             synchronized (context) {
42                 if (nativePtr != 0) {
43                     nativeClose(nativePtr);  
44     
45                     if (DEBUG)
46                         System.err.println("++++ Query CLOSE, ptr= " + nativePtr);
47     
48                     nativePtr = 0;
49                 }
50             }
51         }
52     
53         protected static native void nativeClose(long nativeQueryPtr);
54     
55         protected void finalize() {
56             synchronized (context) {
57                 if (nativePtr != 0) {
58                     context.asyncDisposeQuery(nativePtr); 
59                     nativePtr = 0; // Set to 0 if finalize is called before close() for some reason
60                 }
61             }
62         }

64         /**
65          * Checks in core if query syntax is valid. Throws exception, if not.
66          */
67         private void validateQuery(){
68             if (! queryValidated) { // If not yet validated, check if syntax is valid
69                 String invalidMessage = nativeValidateQuery(nativePtr);
70                 if (invalidMessage.equals(""))
71                     queryValidated = true; // If empty string error message, query is valid
72                 else
73                     throw new UnsupportedOperationException(invalidMessage);
74             }
75         }
76     
77         protected native String nativeValidateQuery(long nativeQueryPtr);

462             validateQuery();
463     
464             // Execute the disposal of abandoned realm objects each time a new realm object is created
465             context.executeDelayedDisposal();
466             long nativeViewPtr = nativeFindAll(nativePtr, start, end, limit);
467             try {
468                 return new TableView(this.context, this.parent, nativeViewPtr);
469             } catch (RuntimeException e) {
470                 TableView.nativeClose(nativeViewPtr);
471                 throw e;
472             }
473         }
