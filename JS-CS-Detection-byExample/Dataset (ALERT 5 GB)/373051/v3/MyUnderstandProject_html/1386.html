17     package io.realm.internal;
18     
19     public class Version {
20     
21         static final int CORE_MIN_MAJOR = 0;
22         static final int CORE_MIN_MINOR = 1;
23         static final int CORE_MIN_PATCH = 6;
24         static final int REQUIRED_JNI_VERSION = 23;
25     
26         public enum Feature {
27             Feature_Debug(0),
28             Feature_Replication(1);
29     
30             private Feature(int nativeValue)
31             {
32                 this.nativeFeature = nativeValue;
33             }
34     
35             @SuppressWarnings("unused")
36             private final int nativeFeature;
37         }
38     
39         public static String getCoreVersion() {
40             return nativeGetVersion();
41         }
42     
43         public static String getVersion() {
44             // Currently Core version and Java version is the same
45             return getCoreVersion();
46         }
47     
48         public static boolean hasFeature(Feature feature) {
49             return nativeHasFeature(feature.ordinal());
50         }
51     
52         public static boolean coreLibVersionCompatible(boolean throwIfNot) {
53             String errTxt = "";
54             boolean compatible = nativeIsAtLeast(CORE_MIN_MAJOR, CORE_MIN_MINOR, CORE_MIN_PATCH);
55             if (!compatible) {
56                 errTxt = "Version mismatch between realm.jar ("
57                         + CORE_MIN_MAJOR + "." + CORE_MIN_MINOR + "." + CORE_MIN_PATCH
58                         + ") and native core library (" + getCoreVersion() + ")";
59                 if (throwIfNot)
60                     throw new RuntimeException(errTxt);
61                 System.err.println(errTxt);
62                 return false;
63             }
64     
65             compatible = (nativeGetAPIVersion() == REQUIRED_JNI_VERSION);
66             if (!compatible) {
67                 errTxt = "Native lib API is version " + nativeGetAPIVersion()
68                          + " != " +  REQUIRED_JNI_VERSION + " which is expected by the jar.";
69                 if (throwIfNot)
70                     throw new RuntimeException(errTxt);
71                 System.err.println(errTxt);                    
72             }
73             return compatible;
74         }
75     
76         static native String nativeGetVersion();
77         static native boolean nativeHasFeature(int feature);    
78         static native boolean nativeIsAtLeast(int major, int minor, int patch);    
79         static native int nativeGetAPIVersion();
80     }
