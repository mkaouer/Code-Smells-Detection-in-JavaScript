46     /**
47      * <p>The Realm class is the storage and transactional manager of your object persistent store. Objects
48      * are created. Objects within a Realm can be queried and read at any time. Creating,
49      * modifying, and deleting objects must be done through transactions.</p>
50      *
51      * <p>The transactions ensure that multiple instances (on multiple threads) can access the objects
52      * in a consistent state with full ACID guaranties.</p>
53      *
54      * <p>If auto-refresh is set the instance of the Realm will be automatically updated when another instance commits a
55      * change (create, modify or delete an object). This feature requires the Realm instance to be residing in a
56      * thread attached to a Looper (the main thread has a Looper by default)</p>
57      *
58      *<p>For normal threads Android provides a utility class that can be used like this:</p>
59      *
60      * <pre>
61      * HandlerThread thread = new HandlerThread("MyThread") {
62      *    @Override
63      *    protected void onLooperPrepared() {
64      *       Realm realm = Realm.getInstance(getContext());
65      *       // This realm will be updated by the event loop
66      *       // on every commit performed by other realm instances
67      *    }
68      * };
69      * thread.start();
70      * </pre>
71      */

139         /**
140          * Set the auto-refresh status of the Realm instance
141          *
142          * Auto-refresh is a feature to allows to automatically update the current realm instance and all its derived objects
143          * (RealmResults and RealmObjects instances) when a commit is performed on a Realm acting on the same file in another thread.
144          * This feature is only available if the realm instance lives is a {@link android.os.Looper} enabled thread.
145          *
146          * @param autoRefresh true will turn auto-refresh on, false will turn it off
147          * @throws java.lang.IllegalStateException if trying to enable auto-refresh in a thread without Looper
148          */
149         public void setAutoRefresh(boolean autoRefresh) {
150             if (autoRefresh && Looper.myLooper() == null) {
151                 throw new IllegalStateException("Cannot set auto-refresh in a Thread without a Looper");
152             }
153     
154             if (autoRefresh && !this.autoRefresh) { // Switch it on
155                 handler = new Handler(new RealmCallback());
156                 handlers.put(handler, id);
157             } else if (!autoRefresh && this.autoRefresh && handler != null) { // Switch it off
158                 handler.removeCallbacksAndMessages(null);
159                 handlers.remove(handler);
160             }
161             this.autoRefresh = autoRefresh;
162         }

358         private static Realm createAndValidate(String absolutePath, byte[] key, boolean validateSchema, boolean autoRefresh) {
359             Map<String, Realm> realms = realmsCache.get();
360             Realm realm = realms.get(absolutePath);
361     
362             if (realm != null) {
363                 return realm;
364             }
365     
366             realm = new Realm(absolutePath, key, autoRefresh);
367             realms.put(absolutePath, realm);
368             realmsCache.set(realms);
369     
370             if (validateSchema) {
371                 Class<?> validationClass;
372                 try {
373                     validationClass = Class.forName("io.realm.ValidationList");
374                 } catch (ClassNotFoundException e) {
375                     throw new RealmException("Could not find the generated ValidationList class");
376                 }
377                 Method getProxyClassesMethod;
378                 try {
379                     getProxyClassesMethod = validationClass.getMethod("getProxyClasses");
380                 } catch (NoSuchMethodException e) {
381                     throw new RealmException("Could not find the getProxyClasses method in the ValidationList class");
382                 }
383                 List<String> proxyClasses;
384                 try {
385                     //noinspection unchecked
386                     proxyClasses = (List<String>) getProxyClassesMethod.invoke(null);
387                 } catch (IllegalAccessException e) {
388                     throw new RealmException("Could not execute the getProxyClasses method in the ValidationList class");
389                 } catch (InvocationTargetException e) {
390                     throw new RealmException("An exception was thrown in the getProxyClasses method in the ValidationList class");
391                 }
392     
393                 long version = realm.getVersion();
394                 try {
395                     realm.beginTransaction();
396                     if (version == UNVERSIONED) {
397                         realm.setVersion(0);
398                     }
399     
400                     for (String className : proxyClasses) {
401                         String[] splitted = className.split("\\.");
402                         String modelClassName = splitted[splitted.length - 1];
403                         String generatedClassName = "io.realm." + modelClassName + "RealmProxy";
404                         Class<?> generatedClass;
405                         try {
406                             generatedClass = Class.forName(generatedClassName);
407                         } catch (ClassNotFoundException e) {
408                             throw new RealmException("Could not find the generated " + generatedClassName + " class");
409                         }
410     
411                         // if not versioned, create table
412                         if (version == UNVERSIONED) {
413                             Method initTableMethod;
414                             try {
415                                 initTableMethod = generatedClass.getMethod("initTable", new Class[]{ImplicitTransaction.class});
416                             } catch (NoSuchMethodException e) {
417                                 throw new RealmException("Could not find the initTable method in the generated " + generatedClassName + " class");
418                             }
419                             try {
420                                 initTableMethod.invoke(null, realm.transaction);
421                             } catch (IllegalAccessException e) {
422                                 throw new RealmException("Could not execute the initTable method in the " + generatedClassName + " class");
423                             } catch (InvocationTargetException e) {
424                                 throw new RealmException("An exception was thrown in the initTable method in the " + generatedClassName + " class");
425                             }
426                         }
427     
428                         // validate created table
429                         Method validateMethod;
430                         try {
431                             validateMethod = generatedClass.getMethod("validateTable", new Class[]{ImplicitTransaction.class});
432                         } catch (NoSuchMethodException e) {
433                             throw new RealmException("Could not find the validateTable method in the generated " + generatedClassName + " class");
434                         }
435                         try {
436                             validateMethod.invoke(null, realm.transaction);
437                         } catch (IllegalAccessException e) {
438                             throw new RealmException("Could not execute the validateTable method in the " + generatedClassName + " class");
439                         } catch (InvocationTargetException e) {
440                             throw new RealmMigrationNeededException(e.getMessage(), e);
441                         }
442     
443                         // Populate the columnIndices table
444                         Method fieldNamesMethod;
445                         try {
446                             fieldNamesMethod = generatedClass.getMethod("getFieldNames");
447                         } catch (NoSuchMethodException e) {
448                             throw new RealmException("Could not find the getFieldNames method in the generated " + generatedClassName + " class");
449                         }
450                         List<String> fieldNames;
451                         try {
452                             //noinspection unchecked
453                             fieldNames = (List<String>) fieldNamesMethod.invoke(null);
454                         } catch (IllegalAccessException e) {
455                             throw new RealmException("Could not execute the getFieldNames method in the generated " + generatedClassName + " class");
456                         } catch (InvocationTargetException e) {
457                             throw new RealmException("An exception was thrown in the getFieldNames method in the generated " + generatedClassName + " class");
458                         }
459                         Table table = realm.transaction.getTable(TABLE_PREFIX + modelClassName);
460                         for (String fieldName : fieldNames) {
461                             long columnIndex = table.getColumnIndex(fieldName);
462                             if (columnIndex == -1) {
463                                 throw new RealmMigrationNeededException("Column '" + fieldName + "' not found for type '" + modelClassName + "'");
464                             }
465                             Map<String, Long> innerMap = columnIndices.get(modelClassName);
466                             if (innerMap == null) {
467                                 innerMap = new HashMap<String, Long>();
468                             }
469                             innerMap.put(fieldName, columnIndex);
470                             columnIndices.put(modelClassName, innerMap);
471                         }
472                     }
473                 } finally {
474                     realm.commitTransaction();
475                 }
476             }
477     
478             return realm;
479         }

481         /**
482          * Instantiates and adds a new object to the realm
483          *
484          * @param clazz The Class of the object to create
485          * @return The new object
486          * @throws RealmException An object could not be created
487          */
488         public <E extends RealmObject> E createObject(Class<E> clazz) {
489             Table table;
490             table = tables.get(clazz);
491             if (table == null) {
492                 String simpleClassName = simpleClassNames.get(clazz);
493                 if (simpleClassName == null) {
494                     simpleClassName = clazz.getSimpleName();
495                     simpleClassNames.put(clazz, simpleClassName);
496                 }
497                 String generatedClassName = "io.realm." + simpleClassName + "RealmProxy";
498     
499                 Class<?> generatedClass = generatedClasses.get(generatedClassName);
500                 if (generatedClass == null) {
501                     try {
502                         generatedClass = Class.forName(generatedClassName);
503                     } catch (ClassNotFoundException e) {
504                         throw new RealmException("Could not find the generated proxy class");
505                     }
506                     generatedClasses.put(generatedClassName, generatedClass);
507                 }
508     
509                 Method method = initTableMethods.get(generatedClass);
510                 if (method == null) {
511                     try {
512                         method = generatedClass.getMethod("initTable", new Class[]{ImplicitTransaction.class});
513                     } catch (NoSuchMethodException e) {
514                         throw new RealmException("Could not find the initTable() method in generated proxy class");
515                     }
516                     initTableMethods.put(generatedClass, method);
517                 }
518     
519                 try {
520                     table = (Table) method.invoke(null, transaction);
521                     tables.put(clazz, table);
522                 } catch (IllegalAccessException e) {
523                     throw new RealmException("Could not launch the initTable method");
524                 } catch (InvocationTargetException e) {
525                     e.printStackTrace();
526                     throw new RealmException("An exception occurred while running the initTable method");
527                 }
528             }
529     
530             long rowIndex = table.addEmptyRow();
531             return get(clazz, rowIndex);
532         }

538         <E extends RealmObject> E get(Class<E> clazz, long rowIndex) {
539             E result;
540     
541             Table table = tables.get(clazz);
542             if (table == null) {
543                 String simpleClassName = simpleClassNames.get(clazz);
544                 if (simpleClassName == null) {
545                     simpleClassName = clazz.getSimpleName();
546                     simpleClassNames.put(clazz, simpleClassName);
547                 }
548     
549                 table = transaction.getTable(TABLE_PREFIX + simpleClassName);
550                 tables.put(clazz, table);
551             }
552     
553             Row row = table.getRow(rowIndex);
554     
555             Constructor constructor = generatedConstructors.get(clazz);
556             if (constructor == null) {
557                 String simpleClassName = simpleClassNames.get(clazz);
558                 if (simpleClassName == null) {
559                     simpleClassName = clazz.getSimpleName();
560                     simpleClassNames.put(clazz, simpleClassName);
561                 }
562                 String generatedClassName = "io.realm." + simpleClassName + "RealmProxy";
563     
564     
565                 Class<?> generatedClass = generatedClasses.get(generatedClassName);
566                 if (generatedClass == null) {
567                     try {
568                         generatedClass = Class.forName(generatedClassName);
569                     } catch (ClassNotFoundException e) {
570                         throw new RealmException("Could not find the generated proxy class");
571                     }
572                     generatedClasses.put(generatedClassName, generatedClass);
573                 }
574     
575                 constructor = constructors.get(generatedClass);
576                 if (constructor == null) {
577                     try {
578                         constructor = generatedClass.getConstructor();
579                     } catch (NoSuchMethodException e) {
580                         throw new RealmException("Could not find the constructor in generated proxy class");
581                     }
582                     constructors.put(generatedClass, constructor);
583                     generatedConstructors.put(clazz, constructor);
584                 }
585             }
586     
587             try {
588                 // We are know the casted type since we generated the class
589                 //noinspection unchecked
590                 result = (E) constructor.newInstance();
591             } catch (InstantiationException e) {
592                 throw new RealmException("Could not instantiate the proxy class");
593             } catch (IllegalAccessException e) {
594                 throw new RealmException("Could not run the constructor of the proxy class");
595             } catch (InvocationTargetException e) {
596                 e.printStackTrace();
597                 throw new RealmException("An exception occurred while instantiating the proxy class");
598             }
599             result.row = row;
600             result.realm = this;
601             return result;
602         }

613         /**
614          * Returns a typed RealmQuery, which can be used to query for specific objects of this type
615          *
616          * @param clazz The class of the object which is to be queried for
617          * @return A typed RealmQuery, which can be used to query for specific objects of this type

691         /**
692          * Starts a write transaction, this must be closed with {@link io.realm.Realm#commitTransaction()}
693          * or aborted by @{link io.realm.Realm#cancelTransaction()}. Write transactions are used to
694          * atomically create, update and delete objects within a realm.
695          * <br>
696          * Before beginning the write transaction, @{link io.realm.Realm#beginTransaction()} updates the
697          * realm in the case of pending updates from other threads.
698          * <br>
699          * Notice: it is not possible to nest write transactions. If you start a write
700          * transaction within a write transaction an exception is thrown.
701          * <br>
702          * @throws java.lang.IllegalStateException If already in a write transaction.
703          *
704          */
705         public void beginTransaction() {
706             transaction.promoteToWrite();
707         }
708     
709         /**
710          * All changes since @{link io.realm.Realm#beginTransaction()} are persisted to disk and the
711          * realm reverts back to being read-only. An event is sent to notify all other realm instances
712          * that a change has occurred. When the event is received, the other realms will get their
713          * objects and @{link io.realm.RealmResults} updated to reflect
714          * the changes from this commit.
715          * 
716          * @throws java.lang.IllegalStateException If the write transaction is in an invalid state.
717          */
718         public void commitTransaction() {
719             transaction.commitAndContinueAsRead();
720     
721             for (Map.Entry<Handler, Integer> handlerIntegerEntry : handlers.entrySet()) {
722                 Handler handler = handlerIntegerEntry.getKey();
723                 int realmId = handlerIntegerEntry.getValue();
724                 if (
725                         realmId == id                                // It's the right realm
726                         && !handler.hasMessages(REALM_CHANGED)       // The right message
727                         && handler.getLooper().getThread().isAlive() // The receiving thread is alive
728                         && !handler.equals(this.handler)             // Don't notify yourself
729                 ) {
730                     handler.sendEmptyMessage(REALM_CHANGED);
731                 }
732             }
733             sendNotifications();
734         }
735     
736         /**
737          * Revert all writes (created, updated, or deleted objects) made in the current write
738          * transaction and end the transaction.
739          * <br>
740          * The realm reverts back to read-only.
741          * <br>
742          * Calling this when not in a write transaction will throw an exception.
743          *
744          * @throws java.lang.IllegalStateException    If the write transaction is an invalid state or
745         *                                             not in a write transaction.
746         */
747          public void cancelTransaction() {
748              transaction.rollbackAndContinueAsRead();
749          }

816         /**
817          * Delete the Realm file from the filesystem for a custom named Realm.
818          * The realm must be unused and closed before calling this method.
819          *
820          * @param context  an Android @{{@link android.content.Context}.
821          * @param fileName the name of the custom Realm (i.e. "myCustomRealm.realm").
822          * @return false if a file could not be deleted. The failing file will be logged.
823          */
824         public static boolean deleteRealmFile(Context context, String fileName) {
825             boolean result = true;
826             File writableFolder = context.getFilesDir();
827             List<File> filesToDelete = Arrays.asList(
828                     new File(writableFolder, fileName),
829                     new File(writableFolder, fileName + ".lock"));
830             for (File fileToDelete : filesToDelete) {
831                 if (fileToDelete.exists()) {
832                     boolean deleteResult = fileToDelete.delete();
833                     if (!deleteResult) {
834                         result = false;
835                         Log.w(TAG, "Could not delete the file " + fileToDelete);
836                     }
837                 }
838             }
839             return result;
840         }
