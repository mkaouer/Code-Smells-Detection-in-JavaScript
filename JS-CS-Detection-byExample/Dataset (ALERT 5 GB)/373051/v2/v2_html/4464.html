114         public boolean methodWrongThread(final boolean callGetter) throws ExecutionException, InterruptedException {
115             Realm realm = Realm.getInstance(getContext());
116             realm.beginTransaction();
117             realm.createObject(AllTypes.class);
118             realm.commitTransaction();
119             final AllTypes allTypes = realm.where(AllTypes.class).findFirst();
120             ExecutorService executorService = Executors.newSingleThreadExecutor();
121             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
122                 @Override
123                 public Boolean call() throws Exception {
124                     try {
125                         if (callGetter) {
126                             allTypes.getColumnFloat();
127                         } else {
128                             allTypes.setColumnFloat(1.0f);
129                         }
130                         return false;
131                     } catch (IllegalStateException ignored) {
132                         return true;
133                     }
134                 }
135             });
136     
137             return future.get();
138         }
