17     package io.realm.examples.intro.model;
18     
19     import io.realm.RealmList;
20     import io.realm.RealmObject;
21     import io.realm.annotations.Ignore;
22     
23     // Your model just have to extend RealmObject.
24     // This will inherit an annotation which produces proxy getters and setters for all fields.
25     
26     public class Person extends RealmObject {
27     
28         // All fields are by default persisted.
29         private String name;
30         private int age;
31     
32         // Other objects in a one-to-one relation must also subclass RealmObject
33         private Dog dog;
34     
35         // One-to-many relations is simply a RealmList of the objects which also subclass RealmObject
36         private RealmList<Cat> cats;
37     
38         // You can instruct Realm to ignore a field and not persist it.
39         @Ignore
40         private int tempReference;
41     
42         // The standard getters and setters your IDE generates are fine.
43         // Realm will overload them and code inside them is ignored.
44         // So if you prefer you can also just have empty abstract methods.
45     
46         public String getName() {
47             return name;
48         }
49     
50         public void setName(String name) {
51             this.name = name;
52         }
53     
54         public int getAge() {
55             return age;
56         }
57     
58         public void setAge(int age) {
59             this.age = age;
60         }
61     
62         public Dog getDog() {
63             return dog;
64         }
65     
66         public void setDog(Dog dog) {
67             this.dog = dog;
68         }
69     
70         public RealmList<Cat> getCats() {
71             return cats;
72         }
73     
74         public void setCats(RealmList<Cat> cats) {
75             this.cats = cats;
76         }
77     
78         public int getTempReference() {
79             return tempReference;
80         }
81     
82         public void setTempReference(int tempReference) {
83             this.tempReference = tempReference;
84         }
85     }
