73     JNIEXPORT void JNICALL Java_io_realm_internal_Table_nativePivot
74     (JNIEnv *env, jobject, jlong dataTablePtr, jlong stringCol, jlong intCol, jint operation, jlong resultTablePtr)
75     {
76         Table* dataTable = TBL(dataTablePtr);
77         Table* resultTable = TBL(resultTablePtr);
78         Table::AggrType pivotOp;
79         switch (operation) {
80             case 0:
81                 pivotOp = Table::aggr_count;
82                 break;
83             case 1:
84                 pivotOp = Table::aggr_sum;
85                 break;
86             case 2:
87                 pivotOp = Table::aggr_avg;
88                 break;
89             case 3:
90                 pivotOp = Table::aggr_min;
91                 break;
92             case 4:
93                 pivotOp = Table::aggr_max;
94                 break;
95             default:
96                 ThrowException(env, UnsupportedOperation, "No pivot operation specified.");
97                 return;
98         }
99     
100         try {
101             dataTable->aggregate(S(stringCol), S(intCol), pivotOp, *resultTable);
102         } CATCH_STD()
103     }

160     JNIEXPORT jobject JNICALL Java_io_realm_internal_Table_nativeGetTableSpec(
161         JNIEnv* env, jobject, jlong nativeTablePtr)
162     {
163         if (!TABLE_VALID(env, TBL(nativeTablePtr)))
164             return 0;
165     
166         TR((env, "nativeGetTableSpec(table %x)\n", nativeTablePtr));
167         static jmethodID jTableSpecConsId = GetTableSpecMethodID(env, "<init>", "()V");
168         if (jTableSpecConsId) {
169             try {
170                 // Create a new TableSpec object in Java
171                 const Table* pTable = TBL(nativeTablePtr);
172                 ConstDescriptorRef desc = pTable->get_descriptor(); // noexcept
173                 jobject jTableSpec = env->NewObject(GetClassTableSpec(env), jTableSpecConsId);
174                 if (jTableSpec) {
175                     get_descriptor(env, *desc, jTableSpec); // Throws
176                     return jTableSpec;
177                 }
178             }
179             CATCH_STD()
180         }
181         return 0;
182     }
