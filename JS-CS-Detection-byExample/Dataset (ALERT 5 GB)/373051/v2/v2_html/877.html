30     /**
31      * A RealmResults list contains a list of objects of a given type that matches the query.
32      * The objects are not copied from the Realm to the RealmResults list, but just references the original objects.
33      * This preserves memory and increase speed.
34      * It also implies that any modification to any object in a RealmResults is reflected in the objects in the 
35      * Realm that was queried.
36      * Updates to objects must be done within a transaction and the modified object is persisted to the backing
37      * Realm file during the commit of the transaction.
38      *
39      * @param <E> The class of objects in this list
40      * @see RealmQuery#findAll()
41      * @see Realm#allObjects(Class)
42      */
43     public class RealmResults<E extends RealmObject> extends AbstractList<E> {
44     
45         private Class<E> classSpec;
46         private Realm realm;
47         private TableOrView table = null;
48     
49         public static final boolean SORT_ORDER_ASCENDING = true;
50         public static final boolean SORT_ORDER_DECENDING = false;
51     
52         private static final String TYPE_MISMATCH = "Field '%s': type mismatch - %s expected.";
53     
54         RealmResults(Realm realm, Class<E> classSpec) {
55             this.realm = realm;
56             this.classSpec = classSpec;
57         }
58     
59         RealmResults(Realm realm, TableOrView table, Class<E> classSpec) {
60             this(realm, classSpec);
61             this.table = table;
62         }
63     
64         Realm getRealm() {
65             return realm;
66         }
67     
68         TableOrView getTable() {
69             if (table == null) {
70                 return realm.getTable(classSpec);
71             } else {
72                 return table;
73             }
74         }

76         Map<String, Class<?>> cache = new HashMap<String, Class<?>>();

78         /**
79          * Returns a typed @{link io.realm.RealmQuery}, which can be used to query for specific
80          * objects of this type.
81          *
82          * @return A typed RealmQuery
83          * @see io.realm.RealmQuery
84          */
85         public RealmQuery<E> where() {
86             realm.assertThread();
87             return new RealmQuery<E>(this, classSpec);
88         }

90         /**
91          * {@inheritDoc}
92          */
93         @Override
94         public E get(int rowIndex) {
95             E obj;
96             realm.assertThread();
97             TableOrView table = getTable();
98             if (table instanceof TableView) {
99                 obj = realm.get(classSpec, ((TableView)table).getSourceRowIndex(rowIndex));
100             } else {
101                 obj = realm.get(classSpec, rowIndex);
102             }
103     
104             return obj;
105         }

117          * @return The last object.

119         public E last() {

123         // Sorting
124     
125         /**
126          * Get a sorted (ascending) RealmList from an existing @{link io.realm.RealmList}.
127          * Only fields of type boolean, short, int, long, float, double, Date, and String are supported.
128          * 
129          * @param fieldName  The field name to sort by.
130          * @return           A sorted RealmResults list
131          */
132         public RealmResults<E> sort(String fieldName) {
133             return sort(fieldName, SORT_ORDER_ASCENDING);
134         }
135     
136         /**
137          * Get a sorted RealmList from an existing @{link io.realm.RealmList}.
138          * Only fields of type boolean, short, int, long, float, double, Date, and String are supported.
139          *
140          * @param fieldName      The field name to sort by.
141          * @param sortAscending  The direction to sort by; if true ascending, otherwise descending
142          *                       You can use the constants SORT_ORDER_ASCENDING and SORT_ORDER_DECENDING
143          *                       for readability.
144          * @return               A sorted RealmResults list.
145          */
146         public RealmResults<E> sort(String fieldName, boolean sortAscending) {
147             TableView sorted;
148     
149             realm.assertThread();
150             TableOrView table = getTable();
151             long columnIndex = table.getColumnIndex(fieldName);
152             TableView.Order TVOrder = sortAscending ? TableView.Order.ascending : TableView.Order.descending;
153     
154             if (table instanceof TableView) {
155                 TableView v = (TableView)table;
156                 sorted = v.where().findAll();
157                 sorted.sort(columnIndex, TVOrder);
158             }
159             else {
160                 Table t = (Table)table;
161                 sorted = t.getSortedView(columnIndex, TVOrder);
162             }
163     
164             return new RealmResults<E>(realm, sorted, classSpec);
165         }

178         /**
179          * Find the minimum value of a field.
180          *
181          * @param fieldName   The field to look for a minimum on. Only int, float, and double
182          *                    are supported.
183          * @return            The minimum value.
184          * @throws            java.lang.IllegalArgumentException if field is not int, float or double.
185          */
186         public Number min(String fieldName) {
187             realm.assertThread();
188             long columnIndex = table.getColumnIndex(fieldName);
189             switch (table.getColumnType(columnIndex)) {
190                 case INTEGER:
191                     return table.minimumLong(columnIndex);
192                 case FLOAT:
193                     return table.minimumFloat(columnIndex);
194                 case DOUBLE:
195                     return table.minimumDouble(columnIndex);
196                 default:
197                     throw new IllegalArgumentException(String.format(TYPE_MISMATCH, fieldName, "int, float or double"));
198             }
199         }
