17     package io.realm;
18     
19     import java.util.AbstractList;
20     
21     import io.realm.internal.LinkView;
22     import io.realm.internal.TableQuery;
23     
24     /**
25      * RealmList is used in one-to-many relationships.
26      *
27      * @param <E> The class of objects in this list
28      */
29     
30     public class RealmList<E extends RealmObject> extends AbstractList<E> {
31     
32         private Class<E> clazz;
33         private LinkView view;
34         private Realm realm;
35     
36         RealmList(Class<E> clazz, LinkView view, Realm realm) {
37             this.clazz = clazz;
38             this.view = view;
39             this.realm = realm;
40         }
41     
42         /**
43          * {@inheritDoc}
44          */
45         @Override
46         public void add(int location, E object) {
47             view.insert(location, object.row.getIndex());
48         }
49     
50         /**
51          * {@inheritDoc}
52          */
53         @Override
54         public boolean add(E object) {
55             view.add(object.row.getIndex());
56             return true;
57         }
58     
59         /**
60          * {@inheritDoc}
61          */
62         @Override
63         public E set(int location, E object) {
64             view.set(location, object.row.getIndex());
65             return object;
66         }
67     
68         /**
69          * Moves an object from one position to another
70          * @param oldPos
71          * @param newPos
72          */
73         public void move(int oldPos, int newPos) {
74             view.move(oldPos, newPos);
75         }
76     
77         /**
78          * {@inheritDoc}
79          */
80         @Override
81         public void clear() {
82             view.clear();
83         }
84     
85         /**
86          * {@inheritDoc}
87          */
88         @Override
89         public E remove(int location) {
90             view.remove(location);
91             return null;
92         }
93     
94         /**
95          * {@inheritDoc}
96          */
97         @Override
98         public E get(int i) {
99             return realm.get(clazz, view.getTargetRowIndex(i));
100         }
101     
102         /**
103          * Find the first object.
104          *
105          * @return The first object
106          */
107         public E first() {
108             if (!view.isEmpty()) {
109                 return get(0);
110             }
111             return null;
112         }
113     
114         /**
115          * Find the last object.
116          *
117          * @return The last object
118          */
119         public E last() {
120             if (!view.isEmpty()) {
121                 return get(size()-1);
122             }
123             return null;
124         }
125     
126         /**
127          * {@inheritDoc}
128          */
129         @Override
130         public int size() {
131             return ((Long)view.size()).intValue();
132         }
133     
134         /**
135          * Returns a RealmQuery, which can be used to query for specific objects of this class
136          *
137          * @return A RealmQuery object
138          * @see io.realm.RealmQuery
139          */
140         public RealmQuery<E> where() {
141             TableQuery query = this.view.where();
142             return new RealmQuery<E>(this.realm, query, clazz);
143         }
144     }
