130     JNIEXPORT jbyteArray JNICALL Java_io_realm_internal_Row_nativeGetByteArray
131       (JNIEnv* env, jobject, jlong nativeRowPtr, jlong columnIndex)
132     {
133         if (!ROW_AND_COL_INDEX_AND_TYPE_VALID(env, ROW(nativeRowPtr), columnIndex, type_Binary))
134             return 0;
135     
136         BinaryData bin = ROW(nativeRowPtr)->get_binary( S(columnIndex) );
137         if (bin.size() <= MAX_JSIZE) {
138             jbyteArray jresult = env->NewByteArray(static_cast<jsize>(bin.size()));
139             if (jresult)
140                 env->SetByteArrayRegion(jresult, 0, static_cast<jsize>(bin.size()), reinterpret_cast<const jbyte*>(bin.data()));  // throws
141             return jresult;
142         }
143         else {
144             ThrowException(env, IllegalArgument, "Length of ByteArray is larger than an Int.");
145             return NULL;
146         }
147     }

269     JNIEXPORT void JNICALL Java_io_realm_internal_Row_nativeSetByteArray
270       (JNIEnv* env, jobject, jlong nativeRowPtr, jlong columnIndex, jbyteArray value)
271     {
272         if (!ROW_AND_COL_INDEX_AND_TYPE_VALID(env, ROW(nativeRowPtr), columnIndex, type_Binary))
273             return;
274     
275         jbyte* bytePtr = env->GetByteArrayElements(value, NULL);
276         if (!bytePtr) {
277             ThrowException(env, IllegalArgument, "doByteArray");
278             return;
279         }
280         size_t dataLen = S(env->GetArrayLength(value));
281         ROW(nativeRowPtr)->set_binary( S(columnIndex), BinaryData(reinterpret_cast<char*>(bytePtr), dataLen));
282         env->ReleaseByteArrayElements(value, bytePtr, 0);
283     }
