328         private enum TransactionMethod {
329             METHOD_BEGIN,
330             METHOD_COMMIT,
331             METHOD_CANCEL
332         }
333     
334         // Starting a transaction on the wrong thread will fail
335         public boolean transactionMethodWrongThread(final TransactionMethod method) throws InterruptedException, ExecutionException {
336             final Realm realm = Realm.getInstance(getContext());
337     
338             if (method != TransactionMethod.METHOD_BEGIN) {
339                 realm.beginTransaction();
340                 Dog dog = realm.createObject(Dog.class); // FIXME: Empty transactions cannot be cancelled
341             }
342             ExecutorService executorService = Executors.newSingleThreadExecutor();
343             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
344                 @Override
345                 public Boolean call() throws Exception {
346                     try {
347                         switch (method) {
348                             case METHOD_BEGIN:
349                                 realm.beginTransaction();
350                                 break;
351                             case METHOD_COMMIT:
352                                 realm.commitTransaction();
353                                 break;
354                             case METHOD_CANCEL:
355                                 realm.cancelTransaction();
356                                 break;
357                         }
358                         return false;
359                     } catch (IllegalStateException ignored) {
360                         return true;
361                     }
362                 }
363             });
364     
365             boolean result = future.get();
366             if (result && method != TransactionMethod.METHOD_BEGIN) {
367                 realm.cancelTransaction();
368             }
369             return result;
370         }

372         public void testTransactionWrongThread() throws ExecutionException, InterruptedException {
373             for (TransactionMethod method : TransactionMethod.values()) {
374                 assertTrue(method.toString(), transactionMethodWrongThread(method));
375             }
376         }
