59         public static boolean osIsWindows()
60         {
61             String os = System.getProperty("os.name").toLowerCase(Locale.getDefault());
62             return (os.indexOf("win") >= 0);
63         }

65         public static byte[] serialize(Serializable value) {
66             try {
67                 ByteArrayOutputStream mem = new ByteArrayOutputStream();
68                 ObjectOutputStream output = new ObjectOutputStream(mem);
69                 output.writeObject(value);
70                 output.close();
71                 return mem.toByteArray();
72             } catch (Exception e) {
73                 throw new RuntimeException("Cannot serialize the object!", e);
74             }
75         }
76     
77         public static Serializable deserialize(ByteBuffer buf) {
78             return deserialize(buf.array());
79         }
80     
81         public static Serializable deserialize(byte[] value) {
82             try {
83                 ByteArrayInputStream mem = new ByteArrayInputStream(value);
84                 ObjectInputStream output = new ObjectInputStream(mem);
85                 Object obj = output.readObject();
86                 output.close();
87                 return (Serializable) obj;
88             } catch (Exception e) {
89                 throw new RuntimeException("Cannot deserialize the object!", e);
90             }
91         }

97         /**
98          * Guarantee gc is done.
99          */
100         public static void gcGuaranteed(){
101             Object obj = new Object();
102             WeakReference<Object> ref = new WeakReference<Object>(obj);
103             obj = null;
104             while(ref.get()!=null)
105                 System.gc();
106         }
107     
108         /**
109          * Guarantee gc is done after JVM shutdown.
110          */
111         public static void gcOnExit(){
112             Runtime.getRuntime().addShutdownHook(new Thread(){
113                 @Override
114                 public void run(){
115                     gcGuaranteed();
116                 }
117             });
118         }
119     
120         private static void initTightDB() {
121             // Guarantee gc is done on JVM exit to clean up any native resources
122             gcOnExit();
123         }

125         public static void loadLibrary() {
126             if (libraryIsLoaded.get())
127                 // only load library once
128                 return;
129     
130             initTightDB();
131     
132             String jnilib;
133             if (osIsWindows()) {
134                 jnilib = loadLibraryWindows();
135             }
136             else {
137                 String debug = System.getenv("TIGHTDB_JAVA_DEBUG");
138                 if (debug == null || debug.isEmpty()) {
139                     jnilib = "tightdb-jni";
140                 }
141                 else {
142                     jnilib = "tightdb-jni-dbg";
143                 }
144                 System.loadLibrary(jnilib);
145             }
146             libraryIsLoaded.set(true);
147     
148             Version.coreLibVersionCompatible(true);
149         }

166             }
167             else {
168                 jnilib = loadCorrectLibrary("tightdb_jni32", "tightdb_jni64");
169                 if (jnilib == null) {
170                     System.err.println("Searched java.library.path=" + System.getProperty("java.library.path"));
171                     throw new RuntimeException("Couldn't load the TightDB JNI library 'tightdb_jni32.dll or tightdb_jni64.dll" +
172                                                "'. Please include the directory to the library in java.library.path.");
173                 }
174             }
175             return jnilib;
176         }

178         private static String loadCorrectLibrary(String... libraryCandidateNames) {
179             for (String libraryCandidateName : libraryCandidateNames) {
180                 try {
181                     System.loadLibrary(libraryCandidateName);
182                     return libraryCandidateName;
183                 } catch (Throwable e) {
184                 }
185             }
186             return null;
187         }
