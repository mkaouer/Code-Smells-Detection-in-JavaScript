17     package io.realm.examples.realmmigrationexample.model;

19     import io.realm.Realm;
20     import io.realm.RealmMigration;
21     import io.realm.internal.ColumnType;
22     import io.realm.internal.Table;

24     /***************************** NOTE: *********************************************
25      * The API for migration is currently using internal lower level classes that will
26      * be replaced by a new API very soon! Until then you will have to explore and use
27      * below example as inspiration.
28      *********************************************************************************

31     
32     public class Migration implements RealmMigration {
33         @Override
34         public long execute(Realm realm, long version) {
35     
36             /*
37                 // Version 0
38                 class Person
39                     String fullName;
40                     String lastName;
41                     int    age;
42     
43                 // Version 1
44                 class Person
45                     String fullName;        // combine firstName and lastName into single field
46                     int age;
47             */
48     
49             // Migrate from version 0 to version 1
50             if (version == 0) {
51                 Table personTable = realm.getTable(Person.class);
52     
53                 long fistNameIndex = getIndexForProperty(personTable, "firstName");
54                 long lastNameIndex = getIndexForProperty(personTable, "lastName");
55                 long fullNameIndex = personTable.addColumn(ColumnType.STRING, "fullName");
56                 for (int i = 0; i < personTable.size(); i++) {
57                     personTable.setString(fullNameIndex, i, personTable.getString(fistNameIndex, i) + " " +
58                             personTable.getString(lastNameIndex, i));
59                 }
60                 personTable.removeColumn(getIndexForProperty(personTable, "firstName"));
61                 personTable.removeColumn(getIndexForProperty(personTable, "lastName"));
62                 version++;
63             }
64     
65             /*
66                 // Version 2
67                     class Pet                   // add a new model class
68                         String name;
69                         String type;
70     
71                     class Person
72                         String fullName;
73                         int age;
74                         RealmList<Pet> pets;    // add an array property
75     
76             */
77             // Migrate from version 1 to version 2
78             if (version == 1) {
79                 Table personTable = realm.getTable(Person.class);
80                 Table petTable = realm.getTable(Pet.class);
81                 petTable.addColumn(ColumnType.STRING, "name");
82                 petTable.addColumn(ColumnType.STRING, "type");
83                 long petsIndex = personTable.addColumnLink(ColumnType.LINK_LIST, "pets", petTable);
84                 long fullNameIndex = getIndexForProperty(personTable, "fullName");
85     
86                 for (int i = 0; i < personTable.size(); i++) {
87                     if (personTable.getString(fullNameIndex, i).equals("JP McDonald")) {
88                         personTable.getRow(i).getLinkList(petsIndex).add(petTable.add("Jimbo", "dog"));
89                     }
90                 }
91                 version++;
92             }
93     
94             /*
95                 // Version 3
96                     class Pet
97                         String name;
98                         int type;               // type becomes int
99     
100                     class Person
101                         String fullName;
102                         RealmList<Pet> pets;    // age and pets re-ordered
103                         int age;
104             */
105             // Migrate from version 2 to version 3
106             if (version == 2) {
107                 Table petTable = realm.getTable(Pet.class);
108                 long oldTypeIndex = getIndexForProperty(petTable, "type");
109                 long typeIndex = petTable.addColumn(ColumnType.INTEGER, "type");
110                 for (int i = 0; i < petTable.size(); i++) {
111                     String type = petTable.getString(oldTypeIndex, i);
112                     if (type.equals("dog")) {
113                         petTable.setLong(typeIndex, i, 1);
114                     }
115                     else if (type.equals("cat")) {
116                         petTable.setLong(typeIndex, i, 2);
117                     }
118                     else if (type.equals("hamster")) {
119                         petTable.setLong(typeIndex, i, 3);
120                     }
121                 }
122                 petTable.removeColumn(oldTypeIndex);
123                 version++;
124             }
125             return version;
126         }
127     
128         private long getIndexForProperty(Table table, String name) {
129             for (int i = 0; i < table.getColumnCount(); i++) {
130                 if (table.getColumnName(i).equals(name)) {
131                     return i;
132                 }
133             }
134             return -1;
135         }
136     }
