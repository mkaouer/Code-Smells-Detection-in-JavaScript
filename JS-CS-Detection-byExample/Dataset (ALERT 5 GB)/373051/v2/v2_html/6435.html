37     // Calculate Private Working Set
38     // Source: http://www.codeproject.com/KB/cpp/XPWSPrivate.aspx
39     
40     int Compare( const void * Val1, const void * Val2 )
41     {
42         if ( *(PDWORD)Val1 == *(PDWORD)Val2 )
43         return 0;
44     
45         return *(PDWORD)Val1 > *(PDWORD)Val2 ? 1 : -1;
46     }

48     DWORD dWorkingSetPages[ 1024 * 128 ]; // hold the working set
49                     // information get from QueryWorkingSet()
50     DWORD dPageSize = 0x1000;
51     
52     DWORD CalculateWSPrivate(DWORD processID)
53     {
54         DWORD dSharedPages = 0;
55         DWORD dPrivatePages = 0;
56         DWORD dPageTablePages = 0;
57     
58         HANDLE hProcess = OpenProcess( PROCESS_QUERY_INFORMATION |
59             PROCESS_VM_READ, FALSE, processID );
60     
61         if ( !hProcess )
62         return 0;
63     
64         __try
65         {
66             if ( !QueryWorkingSet(hProcess, dWorkingSetPages, sizeof(dWorkingSetPages)) )
67             __leave;
68     
69             DWORD dPages = dWorkingSetPages[0];
70     
71             qsort( &dWorkingSetPages[1], dPages, sizeof(DWORD), Compare );
72     
73             for ( DWORD i = 1; i <= dPages; i++ )
74             {
75                 DWORD dCurrentPageStatus = 0;
76                 DWORD dCurrentPageAddress;
77                 DWORD dNextPageAddress;
78                 DWORD dNextPageFlags;
79                 DWORD dPageAddress = dWorkingSetPages[i] & 0xFFFFF000;
80                 DWORD dPageFlags = dWorkingSetPages[i] & 0x00000FFF;
81     
82                 while ( i <= dPages ) // iterate all pages
83                 {
84                     dCurrentPageStatus++;
85     
86                     if ( i == dPages ) //if last page
87                     break;
88     
89                     dCurrentPageAddress = dWorkingSetPages[i] & 0xFFFFF000;
90                     dNextPageAddress = dWorkingSetPages[i+1] & 0xFFFFF000;
91                     dNextPageFlags = dWorkingSetPages[i+1] & 0x00000FFF;
92     
93                     //decide whether iterate further or exit
94                     //(this is non-contiguous page or have different flags)
95                     if ( (dNextPageAddress == (dCurrentPageAddress + dPageSize))
96                 && (dNextPageFlags == dPageFlags) )
97                     {
98                         i++;
99                     }
100                     else
101                     break;
102                 }
103     
104                 if ( (dPageAddress < 0xC0000000) || (dPageAddress > 0xE0000000) )
105                 {
106                     if ( dPageFlags & 0x100 ) // this is shared one
107                     dSharedPages += dCurrentPageStatus;
108     
109                     else // private one
110                     dPrivatePages += dCurrentPageStatus;
111                 }
112                 else
113                 dPageTablePages += dCurrentPageStatus; //page table region
114             }
115     
116             DWORD dTotal = dPages * 4;
117             DWORD dShared = dSharedPages * 4;
118             DWORD WSPrivate = dTotal - dShared;
119     
120             return WSPrivate;
121         }
122         __finally
123         {
124             CloseHandle( hProcess );
125         }
126         return -1;
127     }
128     
129     } // anonymous namespace
130     
131     size_t GetMemUsage()
132     {
133         return CalculateWSPrivate(GetCurrentProcessId());
134     }

142     size_t GetMemUsage()
143     {
144         struct task_basic_info t_info;
145     
146         mach_msg_type_number_t t_info_count = TASK_BASIC_INFO_COUNT;
147     
148         if (KERN_SUCCESS != task_info(mach_task_self(), TASK_BASIC_INFO, (task_info_t)&t_info, &t_info_count)) return -1;
149     
150         // resident size is in t_info.resident_size;
151         // virtual size is in t_info.virtual_size;
152         return t_info.resident_size;
153     }
