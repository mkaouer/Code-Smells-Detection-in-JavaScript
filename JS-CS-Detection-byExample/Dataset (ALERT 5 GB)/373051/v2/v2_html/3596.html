63         public void testNotifications() throws InterruptedException, ExecutionException {
64             ExecutorService executorService = Executors.newFixedThreadPool(2);
65     
66             final AtomicInteger changed = new AtomicInteger(0);
67     
68             List<Callable<Void>> callables = new ArrayList<Callable<Void>>();
69             callables.add(new Callable<Void>() {
70                 @Override
71                 public Void call() throws Exception {
72                     Looper.prepare();
73                     Realm realm = Realm.getInstance(getContext()); // This does not sent a message to itself [0]
74                     realm.addChangeListener(new RealmChangeListener() {
75                         @Override
76                         public void onChange() {
77                             changed.incrementAndGet();
78                         }
79                     });
80                     Looper.loop();
81                     return null;
82                 }
83             });
84             callables.add(new Callable<Void>() {
85                 @Override
86                 public Void call() throws Exception {
87                     Realm realm = Realm.getInstance(getContext(), false); // This will notify the other thread [1]
88                     realm.beginTransaction();
89                     Dog dog = realm.createObject(Dog.class);
90                     dog.setName("Rex");
91                     realm.commitTransaction(); // This will notify the other thread[2]
92                     return null;
93                 }
94             });
95     
96             executorService.invokeAll(callables, 2, TimeUnit.SECONDS);
97     
98             assertEquals(2, changed.get());

179         public void testAutoUpdateRealmResults() throws InterruptedException, ExecutionException {
180             final int TEST_SIZE = 10;
181             final AtomicInteger counter = new AtomicInteger(0);
182             final AtomicBoolean isReady = new AtomicBoolean(false);
183             final Map<Integer, Integer> results = new ConcurrentHashMap<Integer, Integer>();
184     
185             ExecutorService executorService = Executors.newSingleThreadExecutor();
186             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
187                 @Override
188                 public Boolean call() throws Exception {
189                     Looper.prepare();
190                     Realm.deleteRealmFile(getContext());
191                     final Realm realm = Realm.getInstance(getContext());
192                     final RealmResults<Dog> dogs = realm.allObjects(Dog.class);
193                     assertEquals(0, dogs.size());
194                     realm.addChangeListener(new RealmChangeListener() {
195                         @Override
196                         public void onChange() {
197                             int c = counter.incrementAndGet();
198                             results.put(c, dogs.size());
199                         }
200                     });
201                     isReady.set(true);
202                     Looper.loop();
203                     return true;
204                 }
205             });
206     
207             // Wait until the looper is started
208             while (!isReady.get()) {
209                 Thread.sleep(5);
210             }
211             Thread.sleep(100);
212     
213             Realm realm = Realm.getInstance(getContext());
214             realm.beginTransaction();
215             for (int i = 0; i < TEST_SIZE; i++) {
216                 Dog dog = realm.createObject(Dog.class);
217                 dog.setName("Rex " + i);
218             }
219             realm.commitTransaction();
220             assertEquals(TEST_SIZE, realm.allObjects(Dog.class).size());
221     
222             try {
223                 future.get(2, TimeUnit.SECONDS);
224             } catch (TimeoutException ignore) {}
225     
226             assertEquals(2, results.size());
227     
228             assertTrue(results.containsKey(1));
229             assertEquals(0, results.get(1).intValue());
230             assertTrue(results.containsKey(2));
231             assertEquals(10, results.get(2).intValue());
232     
233             assertEquals(2, counter.get());
234         }
