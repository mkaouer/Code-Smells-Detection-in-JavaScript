17     package io.realm;
18     
19     /**
20      * RealmMigration is used to allow user defined migration of one model to a another.
21      *
22      * @see Realm#migrateRealmAtPath(String, byte[], RealmMigration)
23      * @see Realm#migrateRealmAtPath(String, RealmMigration)
24      */
25     public interface RealmMigration {
26     
27         /**
28          * Implement this method in your subclass to perform migration
29          * @param realm The Realm on which to perform the migration
30          * @param version The version of the Realm at the start of the migration
31          * @return The version of the Realm after executing the migration
32          */
33         public long execute(Realm realm, long version);
34     }
