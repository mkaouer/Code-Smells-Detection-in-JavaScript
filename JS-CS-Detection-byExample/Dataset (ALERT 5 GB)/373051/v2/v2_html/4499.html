133         public void testShouldStoreValuesOfMixedType() throws Throwable {
134             for (int i = 0; i < mixedDataList.size(); i++) {
135                 for (int j = 0; j < mixedDataList.size(); j++) {
136                     for (int k = 0; k < mixedDataList.size(); k++) {
137     
138                         Table table = new Table();
139                         table.addColumn(ColumnType.MIXED, "mix");
140     
141                         table.add(mixedDataList.get(i).value);
142     
143                         checkMixedCell(table, 0, 0, mixedDataList.get(i).type, mixedDataList.get(i).value);
144     
145                         table.setMixed(0, 0, Mixed.mixedValue(mixedDataList.get(j).value));
146     
147                         checkMixedCell(table, 0, 0, mixedDataList.get(j).type, mixedDataList.get(j).value);
148     
149                         table.setMixed(0, 0, Mixed.mixedValue(mixedDataList.get(k).value));
150     
151                         checkMixedCell(table, 0, 0, mixedDataList.get(k).type, mixedDataList.get(k).value);
152                         table.close();
153                     }
154                 }
155             }
156         }

158         private void checkMixedCell(Table table, long col, long row, ColumnType columnType, Object value) throws IllegalMixedTypeException {
159             ColumnType mixedType = table.getMixedType(col, row);
160             assertEquals(columnType, mixedType);
161     
162             Mixed mixed = table.getMixed(col, row);
163             if (columnType == ColumnType.BINARY) {
164                 if (mixed.getBinaryType() == Mixed.BINARY_TYPE_BYTE_ARRAY) {
165                     // NOTE: We never get here because we always "get" a ByteBuffer.
166                     byte[] bin = mixed.getBinaryByteArray();
167                     assertEquals(Mixed.mixedValue(value), bin);
168                 } else {
169                     ByteBuffer binBuf = mixed.getBinaryValue();
170                     // TODO: Below is sort of hack to compare the content of the
171                     // buffers, since you always will get a ByteBuffer from a Mixed.
172                     ByteBuffer valueBuf = ByteBuffer.wrap((byte[]) value);
173                     if (!binBuf.equals(valueBuf))
174                         System.out.println("***failed");
175                     assertEquals(Mixed.mixedValue(valueBuf), Mixed.mixedValue(binBuf));
176                 }
177             } else {
178                 assertEquals(value, mixed.getValue());
179             }
180         }
