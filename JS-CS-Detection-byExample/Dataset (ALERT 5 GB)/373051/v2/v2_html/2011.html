121         public void checkColumnsTest(Table table) {
122             // Check columns
123             long columns = 9;
124             assertEquals(columns, table.getColumnCount());
125             for (long i = 0; i < columns; i++) {
126                 String name = "" + i;
127                 assertEquals(name, table.getColumnName(i));
128                 assertEquals(i, table.getColumnIndex(name));
129             }
130         }

132         public void renameColumnsTest(Table table) {
133             // Test renameColumn():
134             long columns = 9;
135             for (long i = 0; i < columns; i++)
136                 table.renameColumn(i, "New " + i);
137             for (long i = 0; i < columns; i++)
138                 assertEquals("New " + i, table.getColumnName(i));
139         }
140     
141         public void removeColumnsTest(Table table) {
142             // Test removeColumn():
143             long columns = 9;
144             table.removeColumn(1);
145             assertEquals(columns - 1, table.getColumnCount());
146             assertEquals("New 0", table.getColumnName(0));
147             for (long i = 1; i < columns - 1; i++)
148                 assertEquals("New " + (i + 1), table.getColumnName(i));
149             // remove first
150             table.removeColumn(0);
151             assertEquals(columns - 2, table.getColumnCount());
152             for (long i = 0; i < columns - 2; i++)
153                 assertEquals("New " + (i + 2), table.getColumnName(i));
154             // remove last
155             table.removeColumn(columns - 3);
156             assertEquals(columns - 3, table.getColumnCount());
157             for (long i = 0; i < columns - 3; i++)
158                 assertEquals("New " + (i + 2), table.getColumnName(i));
159             // remove all but "New 4"
160             table.removeColumn(0);
161             table.removeColumn(0);
162             assertEquals(columns - 5, table.getColumnCount());
163             for (long i = 0; i < columns - 6; i++)
164                 table.removeColumn(1);
165             assertEquals(1, table.getColumnCount());
166             assertEquals("New 4", table.getColumnName(0));
167             assertEquals("44", table.getString(0, 4));
168         }
