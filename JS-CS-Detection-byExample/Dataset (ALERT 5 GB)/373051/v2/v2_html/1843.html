17     package io.realm.processor;
18     
19     import javax.annotation.processing.ProcessingEnvironment;
20     import javax.tools.Diagnostic;
21     import java.io.*;
22     import java.net.HttpURLConnection;
23     import java.net.URL;
24     import java.nio.charset.Charset;
25     
26     public class RealmVersionChecker {
27     
28         public static final String REALM_ANDROID_DOWNLOAD_URL = "http://static.realm.io/downloads/java/latest";
29         private static final String VERSION_URL = "http://static.realm.io/update/java?";
30         private static final String REALM_VERSION = "0.74.0"; //TODO: The version value should be pulled from a build file
31         private static final int READ_TIMEOUT = 2000;
32         private static final int CONNECT_TIMEOUT = 4000;
33     
34         private ProcessingEnvironment processingEnvironment;
35     
36         public RealmVersionChecker(ProcessingEnvironment processingEnvironment) {
37             this.processingEnvironment = processingEnvironment;
38         }
39     
40         private void launchRealmCheck() {
41             //Check Realm version server
42             String latestVersionStr = checkLatestVersion();
43             if (!latestVersionStr.equals(REALM_VERSION)) {
44                 printMessage("Version " + latestVersionStr + " of Realm is now available: " + REALM_ANDROID_DOWNLOAD_URL);
45             }
46         }
47     
48         public void executeRealmVersionUpdate() {
49             Thread backgroundThread = new Thread(new Runnable() {
50                 @Override
51                 public void run() {
52                     launchRealmCheck();
53                 }
54             });
55             backgroundThread.start();
56             try {
57                 backgroundThread.join(CONNECT_TIMEOUT + READ_TIMEOUT);
58             } catch(InterruptedException e) {
59                 // We ignore this exception on purpose not to break the build system if this class fails
60             }
61         }
62     
63         private String checkLatestVersion() {
64             String result = REALM_VERSION;
65             try {
66                 URL url = new URL(VERSION_URL + REALM_VERSION);
67                 HttpURLConnection conn = (HttpURLConnection)url.openConnection();
68                 conn.setConnectTimeout(CONNECT_TIMEOUT);
69                 conn.setReadTimeout(READ_TIMEOUT);
70                 BufferedReader rd = new BufferedReader(new InputStreamReader(conn.getInputStream()));
71                 result = rd.readLine();
72                 rd.close();
73             } catch (IOException e) {
74                 // We ignore this exception on purpose not to break the build system if this class fails
75             }
76             return result;
77         }
78     
79         private void printMessage(String message) {
80             processingEnvironment.getMessager().printMessage(Diagnostic.Kind.OTHER, message);
81         }
82     }
