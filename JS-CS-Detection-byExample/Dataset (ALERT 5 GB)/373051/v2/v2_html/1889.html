70         // Get the application's 256-bit AES key
71         private byte[] getKey() throws GeneralSecurityException, IOException, java.io.IOException {
72             // As of 4.3, Android has a secure per-application key store, but it can't store symmetric keys
73             // As a result, we use it to store a public/private keypair which is used to encrypt the
74             // symmetric key which is stored in a file in the application context
75     
76             byte[] keyData;
77             try {
78                 File file = new File(getFilesDir(), Realm.DEFAULT_REALM_NAME + ".key");
79                 keyData = new byte[256];
80                 FileInputStream stream = new FileInputStream(file);
81                 try {
82                     int read = stream.read(keyData);
83                     if (read != keyData.length) {
84                         keyData = null;
85                     }
86                 }
87                 finally {
88                     stream.close();
89                 }
90             } catch (java.io.IOException e) {
91                 // Generate a new key if reading the existing one failed for any reason
92                 keyData = null;
93             }
94     
95             KeyPair keyPair = getKeyPair();
96             final Cipher cipher = Cipher.getInstance("RSA/NONE/PKCS1Padding");
97     
98             // We have an existing secret key, so decrypt and return it
99             if (keyData != null) {
100                 cipher.init(Cipher.UNWRAP_MODE, keyPair.getPrivate());
101                 return ((SecretKey)cipher.unwrap(keyData, "AES", Cipher.SECRET_KEY)).getEncoded();
102             }
103     
104             // We need to generate a new secret key
105             keyData = new byte[32];
106             new SecureRandom().nextBytes(keyData);
107     
108             cipher.init(Cipher.WRAP_MODE, keyPair.getPublic());
109     
110             // Save the secret key to the file
111             File file = new File(getFilesDir(), Realm.DEFAULT_REALM_NAME + ".key");
112             FileOutputStream stream = new FileOutputStream(file);
113             try {
114                 stream.write(cipher.wrap(new SecretKeySpec(keyData, "AES")));
115             }
116             finally {
117                 stream.close();
118             }
119     
120             // Delete any existing default Realm since we won't be able to open it with the new key
121             Realm.deleteRealmFile(this);
122     
123             return keyData;
124         }

126         // Get the keypair from the keystore if it exists, or create a new one if not
127         private KeyPair getKeyPair() throws GeneralSecurityException, IOException, java.io.IOException {
128             final KeyStore keyStore = KeyStore.getInstance("AndroidKeyStore");
129             keyStore.load(null);
130     
131             if (!keyStore.containsAlias(TAG)) {
132                 generateKeyPair(TAG);
133             }
134     
135             final KeyStore.PrivateKeyEntry entry = (KeyStore.PrivateKeyEntry)keyStore.getEntry(TAG, null);
136             return new KeyPair(entry.getCertificate().getPublicKey(), entry.getPrivateKey());
137         }
138     
139         // Generate a new long-lived keypair
140         private void generateKeyPair(String alias) throws GeneralSecurityException {
141             final Calendar start = new GregorianCalendar();
142             final Calendar end = new GregorianCalendar();
143             end.add(Calendar.YEAR, 100);
144             final KeyPairGeneratorSpec spec = new KeyPairGeneratorSpec.Builder(this)
145                 .setAlias(alias)
146                 .setSubject(new X500Principal("CN=" + alias))
147                 .setSerialNumber(BigInteger.ONE)
148                 .setStartDate(start.getTime())
149                 .setEndDate(end.getTime())
150                 .build();
151     
152             final KeyPairGenerator gen = KeyPairGenerator.getInstance("RSA", "AndroidKeyStore");
153             gen.initialize(spec);
154             gen.generateKeyPair();
155         }
