33     JNIEXPORT jlong JNICALL Java_io_realm_internal_Group_createNative__Ljava_lang_String_2I(
34         JNIEnv* env, jobject, jstring jFileName, jint mode, jbyteArray keyArray)
35     {
36         TR((env, "Group::createNative(file): "));
37     
38         Group* pGroup = 0;
39         StringData file_name;
40         try {
41             JStringAccessor file_name_tmp(env, jFileName); // throws
42             file_name = StringData(file_name_tmp);
43             Group::OpenMode openmode;
44             switch (mode) {
45             case 0: openmode = Group::mode_ReadOnly; break;
46             case 1: openmode = Group::mode_ReadWrite; break;
47             case 2: openmode = Group::mode_ReadWriteNoCreate; break;
48             default:
49                 TR((env, "Invalid mode: %d\n", mode));
50                 ThrowException(env, IllegalArgument, "Group(): Invalid mode parameter.");
51                 return 0;
52             }
53     
54             KeyBuffer key(env, keyArray);
55     #ifdef TIGHTDB_ENABLE_ENCRYPTION
56             pGroup = new Group(file_name, key.data(), openmode);
57     #else
58             pGroup = new Group(file_name, openmode);
59     #endif
60     
61             TR((env, "%x\n", pGroup));
62             return reinterpret_cast<jlong>(pGroup);
63         }
64         CATCH_FILE(file_name)
65         CATCH_STD()
66     
67         // Failed - cleanup
68         if (pGroup)
69             delete pGroup;
70         return 0;
71     }

73     JNIEXPORT jlong JNICALL Java_io_realm_internal_Group_createNative___3B(
74         JNIEnv* env, jobject, jbyteArray jData)
75     {
76         TR((env, "Group::createNative(byteArray): "));
77         // Copy the group buffer given
78         jsize byteArrayLength = env->GetArrayLength(jData);
79         if (byteArrayLength == 0)
80             return 0;
81         jbyte* buf = static_cast<jbyte*>(malloc(S(byteArrayLength)*sizeof(jbyte)));
82         if (!buf) {
83             ThrowException(env, OutOfMemory, "copying the group buffer.");
84             return 0;
85         }
86         env->GetByteArrayRegion(jData, 0, byteArrayLength, buf);
87     
88         TR((env, " %d bytes.", byteArrayLength));
89         Group* pGroup = 0;
90         try {
91             pGroup = new Group(BinaryData(reinterpret_cast<char*>(buf), S(byteArrayLength)), true);
92             TR((env, " groupPtr: %x\n", pGroup));
93             return reinterpret_cast<jlong>(pGroup);
94         }
95         CATCH_FILE("memory-buffer")
96         CATCH_STD()
97     
98         // Failed - cleanup
99         if (buf)
100             free(buf);
101         return 0;
102     }

186     JNIEXPORT jbyteArray JNICALL Java_io_realm_internal_Group_nativeWriteToMem(
187         JNIEnv* env, jobject, jlong nativeGroupPtr)
188     {
189         TR((env, "nativeWriteToMem(%x)\n", nativeGroupPtr));
190         BinaryData buffer;
191         char* bufPtr = 0;
192         try {
193             buffer = G(nativeGroupPtr)->write_to_mem(); // throws
194             bufPtr = const_cast<char*>(buffer.data());
195             // Copy the data to Java array, so Java owns it.
196             jbyteArray jArray = 0;
197             if (buffer.size() <= MAX_JSIZE) {
198                 jsize jlen = static_cast<jsize>(buffer.size());
199                 jArray = env->NewByteArray(jlen);
200                 if (jArray)
201                     // Copy data to Byte[]
202                     env->SetByteArrayRegion(jArray, 0, jlen, reinterpret_cast<const jbyte*>(bufPtr));
203                     // SetByteArrayRegion() may throw ArrayIndexOutOfBoundsException - logic error
204             }
205             if (!jArray) {
206                 ThrowException(env, IndexOutOfBounds, "Group too big to copy and write.");
207             }
208             free(bufPtr);
209             return jArray;
210         }
211         CATCH_STD()
212         if (bufPtr)
213             free(bufPtr);
214         return 0;
215     }

217     JNIEXPORT jobject JNICALL Java_io_realm_internal_Group_nativeWriteToByteBuffer(
218         JNIEnv* env, jobject, jlong nativeGroupPtr)
219     {
220         TR((env, "nativeWriteToByteBuffer(%x)\n", nativeGroupPtr));
221         BinaryData buffer;
222         try {
223             buffer = G(nativeGroupPtr)->write_to_mem();
224             if (util::int_less_than_or_equal(buffer.size(), MAX_JLONG)) {
225                 return env->NewDirectByteBuffer(const_cast<char*>(buffer.data()), static_cast<jlong>(buffer.size()));
226                 // Data is now owned by the Java DirectByteBuffer - so we must not free it.
227             }
228             else {
229                 ThrowException(env, IndexOutOfBounds, "Group too big to write.");
230                 return NULL;
231             }
232         }
233         CATCH_STD()
234         return NULL;
235     }

244     JNIEXPORT jstring JNICALL Java_io_realm_internal_Group_nativeToJson(
245         JNIEnv* env, jobject, jlong nativeGroupPtr)
246     {
247         Group* grp = G(nativeGroupPtr);
248     
249         try {
250             // Write group to string in JSON format
251             std::ostringstream ss;
252             ss.sync_with_stdio(false); // for performance
253             grp->to_json(ss);
254             const std::string str = ss.str();
255             return to_jstring(env, str);
256         } CATCH_STD()
257         return 0;
