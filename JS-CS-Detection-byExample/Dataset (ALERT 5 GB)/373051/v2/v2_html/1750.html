17     package io.realm.examples.realmgridview;
18     
19     import android.app.Activity;
20     import android.os.Bundle;
21     import android.view.View;
22     import android.widget.AdapterView;
23     import android.widget.GridView;
24     
25     import com.google.gson.JsonArray;
26     import com.google.gson.JsonElement;
27     import com.google.gson.JsonParser;
28     
29     import java.io.IOException;
30     import java.io.InputStream;
31     import java.io.InputStreamReader;
32     import java.util.ArrayList;
33     import java.util.List;
34     
35     import io.realm.Realm;
36     import io.realm.RealmResults;
37     
38     public class RealmExampleActivity extends Activity implements AdapterView.OnItemClickListener {
39     
40         public static final String TAG = RealmExampleActivity.class.getName();
41     
42         private GridView mGridView;
43         private CityAdapter mAdapter;
44     
45         @Override
46         protected void onCreate(Bundle savedInstanceState) {
47             super.onCreate(savedInstanceState);
48             setContentView(R.layout.activity_realm_example);
49         }
50     
51         @Override
52         public void onResume() {
53             super.onResume();
54     
55             // Load from file "cities.json" first time
56             if(mAdapter == null) {
57                 List<City> cities = loadCities();
58     
59                 //This is the GridView adapter
60                 mAdapter = new CityAdapter(this);
61                 mAdapter.setData(cities);
62     
63                 //This is the GridView which will display the list of cities
64                 mGridView = (GridView) findViewById(R.id.cities_list);
65                 mGridView.setAdapter(mAdapter);
66                 mGridView.setOnItemClickListener(RealmExampleActivity.this);
67                 mAdapter.notifyDataSetChanged();
68                 mGridView.invalidate();
69             }
70         }
71     
72         public List<City> loadCities() {
73             List<City> items = new ArrayList<City>();
74     
75             // In this case we're loading from local assets.
76             // NOTE: could alternatively easily load from network
77             InputStream stream = null;
78             try {
79                 stream = getAssets().open("cities.json");
80             } catch (IOException e) {
81                 return null;
82             }
83     
84             JsonParser parser = new JsonParser();
85             JsonArray jsonArray = parser.parse(new InputStreamReader(stream)).getAsJsonArray();
86     
87             // Clear the realm from last time
88             Realm.deleteRealmFile(this);
89     
90             // Store the retrieved items to the Realm
91             Realm realm = Realm.getInstance(this);
92     
93             // Open a transaction to store items into the realm
94             realm.beginTransaction();
95             for (JsonElement e : jsonArray) {
96                 // Create a realm capable object
97                 City realmCity = realm.createObject(City.class);
98                 realmCity.setName(e.getAsJsonObject().get("name").getAsString());
99                 realmCity.setVotes(e.getAsJsonObject().get("votes").getAsInt());
100                 // Minor optimization to keep the new cities in a list
101                 // so it doesn't have to be reloaded the first time
102                 items.add(realmCity);
103             }
104             realm.commitTransaction();
105     
106             return items;
107         }
108     
109         public void updateCities() {
110             Realm realm = Realm.getInstance(this);
111     
112             // Pull all the cities from the realm
113             RealmResults<City> cities = realm.where(City.class).findAll();
114     
115             // Put these items in the Adapter
116             mAdapter.setData(cities);
117             mAdapter.notifyDataSetChanged();
118             mGridView.invalidate();
119         }
120     
121         @Override
122         public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
123             City modifiedCity = (City)mAdapter.getItem(position);
124     
125             // Update the realm object affected by the user
126             Realm realm = Realm.getInstance(this);
127     
128             // Acquire the list of realm cities matching the name of the clicked City.
129             City city = realm.where(City.class).equalTo("name", modifiedCity.getName()).findFirst();
130     
131             // Create a transaction to increment the vote count for the selected City in the realm
132             realm.beginTransaction();
133             city.setVotes(city.getVotes() + 1);
134             realm.commitTransaction();
135     
136             updateCities();
137         }
138     }
