80             final RealmResults<AllTypes> allTypeses = testRealm.where(AllTypes.class).findAll();
81             ExecutorService executorService = Executors.newSingleThreadExecutor();
82             Future<Boolean> future = executorService.submit(new Callable<Boolean>() {
83                 @Override
84                 public Boolean call() throws Exception {
85                     try {
86                         switch (method) {
87                             case METHOD_MIN:
88                                 allTypeses.min(FIELD_FLOAT);
89                                 break;
90                             case METHOD_MAX:
91                                 allTypeses.max(FIELD_FLOAT);
92                                 break;
93                             case METHOD_SUM:
94                                 allTypeses.sum(FIELD_FLOAT);
95                                 break;
96                             case METHOD_AVG:
97                                 allTypeses.average(FIELD_FLOAT);
98                                 break;
99                             case METHOD_SORT:
100                                 allTypeses.sort(FIELD_FLOAT);
101                                 break;
102                             case METHOD_WHERE:
103                                 allTypeses.where();
104                         }
105                         return false;
106                     } catch (IllegalStateException ignored) {
107                         return true;
108                     }
109                 }
110             });
111     
112             return future.get();
113         }

198         public void testAvgGivesCorrectValue() {
199             RealmResults<AllTypes> resultList = testRealm.where(AllTypes.class).findAll();
200             double N = (double)TEST_DATA_SIZE;
201     
202             // Sum of numbers 1 to M: M*(M+1)/2
203             // See setUp() for values of fields
204             // N = TEST_DATA_SIZE
205     
206             // Type: double; a = 3.1415
207             // a, a+1, ..., a+i, ..., a+N-1
208             // sum = 3.1415*N + N*(N-1)/2
209             // average = sum/N = 3.1415+(N-1)/2
210             double average = 3.1415+(N-1.0)*0.5;
211             assertEquals(average, resultList.average(FIELD_DOUBLE), 0.0001);
212     
213             // Type: long
214             // 0, 1, ..., N-1
215             // sum = N*(N-1)/2
216             // average = sum/N = (N-1)/2
217             assertEquals(0.5*(N-1), resultList.average(FIELD_LONG), 0.0001);
218     
219             // Type: float; b = 1.234567
220             // b, b+1, ..., b+i, ..., b+N-1
221             // sum = b*N + N*(N-1)/2
222             // average = sum/N = b + (N-1)/2
223             assertEquals(1.234567+0.5*(N-1.0), resultList.average(FIELD_FLOAT), 0.0001);
224         }
