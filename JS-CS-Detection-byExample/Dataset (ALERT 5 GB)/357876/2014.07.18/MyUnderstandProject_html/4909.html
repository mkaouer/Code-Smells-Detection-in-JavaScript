118         @Override
119         public void onCreate(Bundle savedInstanceState) {
120             super.onCreate(savedInstanceState, R.layout.viewpager_activity);
121     
122             // set title in code, as the activity needs a hard coded title due to the intent filters
123             setTitle(res.getString(R.string.trackable));
124     
125             // get parameters
126             final Bundle extras = getIntent().getExtras();
127             final Uri uri = getIntent().getData();
128     
129             // try to get data from extras
130             if (extras != null) {
131                 geocode = extras.getString(Intents.EXTRA_GEOCODE);
132                 name = extras.getString(Intents.EXTRA_NAME);
133                 guid = extras.getString(Intents.EXTRA_GUID);
134                 id = extras.getString(Intents.EXTRA_ID);
135             }
136     
137             // try to get data from URI
138             if (geocode == null && guid == null && id == null && uri != null) {
139                 geocode = ConnectorFactory.getTrackableFromURL(uri.toString());
140     
141                 final String uriHost = uri.getHost().toLowerCase(Locale.US);
142                 if (uriHost.contains("geocaching.com")) {
143                     geocode = uri.getQueryParameter("tracker");
144                     guid = uri.getQueryParameter("guid");
145                     id = uri.getQueryParameter("id");
146     
147                     if (StringUtils.isNotBlank(geocode)) {
148                         geocode = geocode.toUpperCase(Locale.US);
149                         guid = null;
150                         id = null;
151                     } else if (StringUtils.isNotBlank(guid)) {
152                         geocode = null;
153                         guid = guid.toLowerCase(Locale.US);
154                         id = null;
155                     } else if (StringUtils.isNotBlank(id)) {
156                         geocode = null;
157                         guid = null;
158                         id = id.toLowerCase(Locale.US);
159                     } else {
160                         showToast(res.getString(R.string.err_tb_details_open));
161                         finish();
162                         return;
163                     }
164                 } else if (uriHost.contains("coord.info")) {
165                     final String uriPath = uri.getPath().toLowerCase(Locale.US);
166                     if (StringUtils.startsWith(uriPath, "/tb")) {
167                         geocode = uriPath.substring(1).toUpperCase(Locale.US);
168                         guid = null;
169                         id = null;
170                     } else {
171                         showToast(res.getString(R.string.err_tb_details_open));
172                         finish();
173                         return;
174                     }
175                 }
176             }
177     
178             // no given data
179             if (geocode == null && guid == null && id == null) {
180                 showToast(res.getString(R.string.err_tb_display));
181                 finish();
182                 return;
183             }
184     
185             String message;
186             if (StringUtils.isNotBlank(name)) {
187                 message = Html.fromHtml(name).toString();
188             } else if (StringUtils.isNotBlank(geocode)) {
189                 message = geocode;
190             } else {
191                 message = res.getString(R.string.trackable);
192             }
193             waitDialog = ProgressDialog.show(this, message, res.getString(R.string.trackable_details_loading), true, true);
194     
195             createViewPager(0, null);
196             final LoadTrackableThread thread = new LoadTrackableThread(loadTrackableHandler, geocode, guid, id);
197             thread.start();
198         }

259         private class LoadTrackableThread extends Thread {
260             final private Handler handler;
261             final private String geocode;
262             final private String guid;
263             final private String id;
264     
265             public LoadTrackableThread(Handler handlerIn, String geocodeIn, String guidIn, String idIn) {
266                 handler = handlerIn;
267                 geocode = geocodeIn;
268                 guid = guidIn;
269                 id = idIn;
270             }
271     
272             @Override
273             public void run() {
274                 if (StringUtils.isNotEmpty(geocode)) {
275                     trackable = DataStore.loadTrackable(geocode);
276     
277                     if (trackable == null || trackable.isLoggable()) {
278                         // iterate over the connectors as some codes may be handled by multiple connectors
279                         for (final TrackableConnector trackableConnector : ConnectorFactory.getTrackableConnectors()) {
280                             if (trackableConnector.canHandleTrackable(geocode)) {
281                                 trackable = trackableConnector.searchTrackable(geocode, guid, id);
282                                 if (trackable != null) {
283                                     break;
284                                 }
285                             }
286                         }
287                     }
288                 }
289                 // fall back to GC search by GUID
290                 if (trackable == null) {
291                     trackable = TravelBugConnector.getInstance().searchTrackable(geocode, guid, id);
292                 }
293                 handler.sendMessage(Message.obtain());
294             }
295         }

313                     final HtmlImage imgGetter = new HtmlImage(trackable.getGeocode(), false, 0, false);
314     
315                     final BitmapDrawable image = imgGetter.getDrawable(url);
316                     final Message message = handler.obtainMessage(0, image);

319                     Log.e("TrackableActivity.TrackableIconThread.run: ", e);
320                 }
321             }

376         public class DetailsViewCreator extends AbstractCachingPageViewCreator<ScrollView> {
377     
378             @InjectView(R.id.goal_box) protected LinearLayout goalBox;
379             @InjectView(R.id.goal) protected TextView goalTextView;
380             @InjectView(R.id.details_box) protected LinearLayout detailsBox;

383             @InjectView(R.id.details_list) protected LinearLayout detailsList;

386             @Override
387             public ScrollView getDispatchedView() {
388                 view = (ScrollView) getLayoutInflater().inflate(R.layout.trackable_details_view, null);
389                 ButterKnife.inject(this, view);
390     
391                 final CacheDetailsCreator details = new CacheDetailsCreator(TrackableActivity.this, detailsList);
392     
393                 // action bar icon
394                 if (StringUtils.isNotBlank(trackable.getIconUrl())) {
395                     final TrackableIconHandler iconHandler = new TrackableIconHandler(((TextView) findViewById(R.id.actionbar_title)));
396                     final TrackableIconThread iconThread = new TrackableIconThread(trackable.getIconUrl(), iconHandler);
397                     iconThread.start();
398                 }
399     
400                 // trackable name
401                 registerForContextMenu(details.add(R.string.trackable_name, StringUtils.isNotBlank(trackable.getName()) ? Html.fromHtml(trackable.getName()).toString() : res.getString(R.string.trackable_unknown)));
402     
403                 // trackable type
404                 String tbType;
405                 if (StringUtils.isNotBlank(trackable.getType())) {
406                     tbType = Html.fromHtml(trackable.getType()).toString();
407                 } else {
408                     tbType = res.getString(R.string.trackable_unknown);
409                 }
410                 details.add(R.string.trackable_type, tbType);
411     
412                 // trackable geocode
413                 registerForContextMenu(details.add(R.string.trackable_code, trackable.getGeocode()));
414     
415                 // trackable owner
416                 final TextView owner = details.add(R.string.trackable_owner, res.getString(R.string.trackable_unknown));
417                 if (StringUtils.isNotBlank(trackable.getOwner())) {
418                     owner.setText(Html.fromHtml(trackable.getOwner()), TextView.BufferType.SPANNABLE);
419                     owner.setOnClickListener(new UserActionsClickListener(trackable));
420                 }
421     
422                 // trackable spotted
423                 if (StringUtils.isNotBlank(trackable.getSpottedName()) ||
424                         trackable.getSpottedType() == Trackable.SPOTTED_UNKNOWN ||
425                         trackable.getSpottedType() == Trackable.SPOTTED_OWNER) {
426                     boolean showTimeSpan = true;
427                     StringBuilder text;
428     
429                     if (trackable.getSpottedType() == Trackable.SPOTTED_CACHE) {
430                         text = new StringBuilder(res.getString(R.string.trackable_spotted_in_cache) + ' ' + Html.fromHtml(trackable.getSpottedName()).toString());
431                     } else if (trackable.getSpottedType() == Trackable.SPOTTED_USER) {
432                         text = new StringBuilder(res.getString(R.string.trackable_spotted_at_user) + ' ' + Html.fromHtml(trackable.getSpottedName()).toString());
433                     } else if (trackable.getSpottedType() == Trackable.SPOTTED_UNKNOWN) {
434                         text = new StringBuilder(res.getString(R.string.trackable_spotted_unknown_location));
435                     } else if (trackable.getSpottedType() == Trackable.SPOTTED_OWNER) {
436                         text = new StringBuilder(res.getString(R.string.trackable_spotted_owner));
437                     } else {
438                         text = new StringBuilder("N/A");
439                         showTimeSpan = false;
440                     }
441     
442                     // days since last spotting
443                     if (showTimeSpan && trackable.getLogs() != null) {
444                         for (final LogEntry log : trackable.getLogs()) {
445                             if (log.type == LogType.RETRIEVED_IT || log.type == LogType.GRABBED_IT || log.type == LogType.DISCOVERED_IT || log.type == LogType.PLACED_IT) {
446                                 final int days = log.daysSinceLog();
447                                 text.append(" (").append(res.getQuantityString(R.plurals.days_ago, days, days)).append(')');
448                                 break;
449                             }
450                         }
451                     }
452     
453                     final TextView spotted = details.add(R.string.trackable_spotted, text.toString());
454                     spotted.setClickable(true);
455                     if (Trackable.SPOTTED_CACHE == trackable.getSpottedType()) {
456                         spotted.setOnClickListener(new View.OnClickListener() {

463                                     // for geokrety we only know the cache geocode

470                         });
471                     } else if (Trackable.SPOTTED_USER == trackable.getSpottedType()) {
472                         spotted.setOnClickListener(new UserNameClickListener(trackable, Html.fromHtml(trackable.getSpottedName()).toString()));
473                     } else if (Trackable.SPOTTED_OWNER == trackable.getSpottedType()) {
474                         spotted.setOnClickListener(new UserNameClickListener(trackable, Html.fromHtml(trackable.getOwner()).toString()));

531                         }
532                     });
533     
534                     imageView.addView(trackableImage);
535                 }
536                 return view;
537             }
