98                 @Override

101                     File newPathDir = new File(pathText);
102                     if (newPathDir.exists() && newPathDir.isDirectory()) {
103                         currentDir = newPathDir;

120         /**
121          * Return the directory containing a given path, or a sensible default.
122          *
123          * @param path the path to get the enclosing directory from, can be null or empty
124          * @return the directory containing <code>path</code>, or a sensible default if none
125          */
126         private static File dirContaining(final String path) {
127             return StringUtils.contains(path, File.separatorChar) ?
128                     new File(StringUtils.substringBeforeLast(path, Character.toString(File.separatorChar))) :
129                     Environment.getExternalStorageDirectory();
130         }

132         private void fill(File dir) {
133             lastPosition = -1;
134             resetOkButton();
135             EditText path = (EditText) findViewById(R.id.simple_dir_chooser_path);
136             path.setText(this.getResources().getString(R.string.simple_dir_chooser_current_path) + " " + dir.getAbsolutePath());
137             final File[] dirs = dir.listFiles(new DirOnlyFilenameFilter());
138             List<Option> listDirs = new ArrayList<Option>();
139             try {
140                 for (File currentDir : dirs) {
141                     listDirs.add(new Option(currentDir.getName(), currentDir.getAbsolutePath(), currentDir.canWrite()));
142                 }
143             } catch (RuntimeException e) {
144             }
145             Collections.sort(listDirs, Option.NAME_COMPARATOR);
146             if (dir.getParent() != null) {
147                 listDirs.add(0, new Option(PARENT_DIR, dir.getParent(), false));
148             }
149             this.adapter = new FileArrayAdapter(this, R.layout.simple_dir_item, listDirs);
150             this.setListAdapter(adapter);
151         }

160         public class FileArrayAdapter extends ArrayAdapter<Option> {
161     
162             private Context context;
163             private int id;
164             private List<Option> items;
165     
166             public FileArrayAdapter(Context context, int simpleDirItemResId, List<Option> objects) {
167                 super(context, simpleDirItemResId, objects);
168                 this.context = context;
169                 this.id = simpleDirItemResId;
170                 this.items = objects;
171             }
172     
173             @Override
174             public Option getItem(int index) {
175                 return items.get(index);
176             }
177     
178             @Override
179             public View getView(int position, View convertView, ViewGroup parent) {
180                 View v = convertView;
181                 if (v == null) {
182                     LayoutInflater vi = (LayoutInflater) context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
183                     v = vi.inflate(id, null);
184                 }
185     
186                 final Option option = items.get(position);
187                 if (option != null) {
188                     TextView t1 = (TextView) v.findViewById(R.id.TextView01);
189                     if (t1 != null) {
190                         t1.setOnClickListener(new OnTextViewClickListener(position));
191                         t1.setText(option.getName());
192                     }
193                     CheckBox check = (CheckBox) v.findViewById(R.id.CheckBox);
194                     if (check != null) {
195                         if (!chooseForWriting || option.isWriteable()) {
196                             check.setOnClickListener(new OnCheckBoxClickListener(position));
197                             check.setChecked(option.isChecked());
198                             check.setEnabled(true);
199                         } else {
200                             check.setEnabled(false);
201                         }
202                     }
203                 }
204                 return v;
205             }
206         }

208         public class OnTextViewClickListener implements OnClickListener {
209             private int position;
210     
211             OnTextViewClickListener(int position) {
212                 this.position = position;
213             }
214     
215             @Override
216             public void onClick(View arg0) {
217                 Option option = adapter.getItem(position);
218                 if (option.getName().equals(PARENT_DIR)) {
219                     currentDir = new File(option.getPath());
220                     fill(currentDir);
221                 } else {
222                     File dir = new File(option.getPath());
223                     if (dir.list(new DirOnlyFilenameFilter()).length > 0) {
224                         currentDir = dir;
225                         fill(currentDir);
226                     }
227                 }
228             }
229         }

231         public class OnCheckBoxClickListener implements OnClickListener {
232             private int position;
233     
234             OnCheckBoxClickListener(int position) {
235                 this.position = position;
236             }
237     
238             @Override
239             public void onClick(View arg0) {
240                 Option lastOption = (lastPosition > -1) ? adapter.getItem(lastPosition) : null;
241                 Option currentOption = adapter.getItem(position);
242                 if (lastOption != null) {
243                     lastOption.setChecked(false);
244                 }
245                 if (currentOption != lastOption) {
246                     currentOption.setChecked(true);
247                     lastPosition = position;
248                 } else {
249                     lastPosition = -1;
250                 }
251                 final boolean enabled = currentOption.isChecked() && !currentOption.getName().equals(PARENT_DIR);
252                 okButton.setEnabled(enabled);
253                 okButton.setVisibility(enabled ? View.VISIBLE : View.INVISIBLE);
254                 adapter.notifyDataSetChanged();
255             }
256         }
