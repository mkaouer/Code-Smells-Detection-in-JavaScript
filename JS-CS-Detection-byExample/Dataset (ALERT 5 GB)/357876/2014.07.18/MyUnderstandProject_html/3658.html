205         /**
206          * Gather missing information for new Geocache object from the stored Geocache object.
207          * This is called in the new Geocache parsed from website to set information not yet
208          * parsed.
209          *
210          * @param other
211          *            the other version, or null if non-existent
212          * @return true if this cache is "equal" to the other version
213          */
214         public boolean gatherMissingFrom(final Geocache other) {
215             if (other == null) {
216                 return false;
217             }
218             if (other == this) {
219                 return true;
220             }
221     
222             updated = System.currentTimeMillis();
223             // if parsed cache is not yet detailed and stored is, the information of
224             // the parsed cache will be overwritten
225             if (!detailed && other.detailed) {
226                 detailed = other.detailed;
227                 detailedUpdate = other.detailedUpdate;
228                 // boolean values must be enumerated here. Other types are assigned outside this if-statement
229                 reliableLatLon = other.reliableLatLon;
230                 finalDefined = other.finalDefined;
231             }
232     
233             if (premiumMembersOnly == null) {
234                 premiumMembersOnly = other.premiumMembersOnly;
235             }
236             if (found == null) {
237                 found = other.found;
238             }
239             if (disabled == null) {
240                 disabled = other.disabled;
241             }
242             if (favorite == null) {
243                 favorite = other.favorite;
244             }
245             if (archived == null) {
246                 archived = other.archived;
247             }
248             if (onWatchlist == null) {
249                 onWatchlist = other.onWatchlist;
250             }
251             if (logOffline == null) {
252                 logOffline = other.logOffline;
253             }
254             if (visitedDate == 0) {
255                 visitedDate = other.visitedDate;
256             }
257             if (listId == StoredList.TEMPORARY_LIST_ID) {
258                 listId = other.listId;
259             }
260             if (StringUtils.isBlank(geocode)) {
261                 geocode = other.geocode;
262             }
263             if (StringUtils.isBlank(cacheId)) {
264                 cacheId = other.cacheId;
265             }
266             if (StringUtils.isBlank(guid)) {
267                 guid = other.guid;
268             }
269             cacheType = UncertainProperty.getMergedProperty(cacheType, other.cacheType);
270             if (StringUtils.isBlank(name)) {
271                 name = other.name;
272             }
273             if (StringUtils.isBlank(ownerDisplayName)) {
274                 ownerDisplayName = other.ownerDisplayName;
275             }
276             if (StringUtils.isBlank(ownerUserId)) {
277                 ownerUserId = other.ownerUserId;
278             }
279             if (hidden == null) {
280                 hidden = other.hidden;
281             }
282             if (!detailed && StringUtils.isBlank(getHint())) {
283                 hint = other.getHint();
284             }
285             if (size == null || CacheSize.UNKNOWN == size) {
286                 size = other.size;
287             }
288             if (difficulty == 0) {
289                 difficulty = other.difficulty;
290             }
291             if (terrain == 0) {
292                 terrain = other.terrain;
293             }
294             if (direction == null) {
295                 direction = other.direction;
296             }
297             if (distance == null) {
298                 distance = other.distance;
299             }
300             if (StringUtils.isBlank(getLocation())) {
301                 location = other.getLocation();
302             }
303             coords = UncertainProperty.getMergedProperty(coords, other.coords);
304             // don't use StringUtils.isBlank here. Otherwise we cannot recognize a note which was deleted on GC
305             if (personalNote == null) {
306                 personalNote = other.personalNote;
307             } else if (other.personalNote != null && !personalNote.equals(other.personalNote)) {
308                 final PersonalNote myNote = new PersonalNote(this);
309                 final PersonalNote otherNote = new PersonalNote(other);
310                 final PersonalNote mergedNote = myNote.mergeWith(otherNote);
311                 personalNote = mergedNote.toString();
312             }
313             if (!detailed && StringUtils.isBlank(getShortDescription())) {
314                 shortdesc = other.getShortDescription();
315             }
316             if (StringUtils.isBlank(getDescription())) {
317                 description = other.getDescription();
318             }
319             // FIXME: this makes no sense to favor this over the other. 0 should not be a special case here as it is
320             // in the range of acceptable values. This is probably the case at other places (rating, votes, etc.) too.
321             if (favoritePoints == 0) {
322                 favoritePoints = other.favoritePoints;
323             }
324             if (rating == 0) {
325                 rating = other.rating;
326             }
327             if (votes == 0) {
328                 votes = other.votes;
329             }
330             if (myVote == 0) {
331                 myVote = other.myVote;
332             }
333             if (!detailed && attributes.isEmpty()) {
334                 if (other.attributes != null) {
335                     attributes.addAll(other.attributes);
336                 }
337             }
338             if (waypoints.isEmpty()) {
339                 this.setWaypoints(other.waypoints, false);
340             }
341             else {
342                 final ArrayList<Waypoint> newPoints = new ArrayList<Waypoint>(waypoints);
343                 Waypoint.mergeWayPoints(newPoints, other.waypoints, false);
344                 this.setWaypoints(newPoints, false);
345             }
346             if (spoilers == null) {
347                 spoilers = other.spoilers;
348             }
349             if (inventory == null) {
350                 // If inventoryItems is 0, it can mean both
351                 // "don't know" or "0 items". Since we cannot distinguish
352                 // them here, only populate inventoryItems from
353                 // old data when we have to do it for inventory.
354                 inventory = other.inventory;
355                 inventoryItems = other.inventoryItems;
356             }
357             if (logCounts.isEmpty()) {
358                 logCounts = other.logCounts;
359             }
360     
361             // if cache has ORIGINAL type waypoint ... it is considered that it has modified coordinates, otherwise not
362             userModifiedCoords = false;
363             for (final Waypoint wpt : waypoints) {
364                 if (wpt.getWaypointType() == WaypointType.ORIGINAL) {
365                     userModifiedCoords = true;
366                     break;
367                 }
368             }
369     
370             if (!reliableLatLon) {
371                 reliableLatLon = other.reliableLatLon;
372             }
373     
374             return isEqualTo(other);
375         }

377         /**
378          * Compare two caches quickly. For map and list fields only the references are compared !
379          *
380          * @param other
381          *            the other cache to compare this one to
382          * @return true if both caches have the same content
383          */
384         @SuppressWarnings("deprecation")
385         @SuppressFBWarnings("FE_FLOATING_POINT_EQUALITY")
386         private boolean isEqualTo(final Geocache other) {
387             return detailed == other.detailed &&
388                     StringUtils.equalsIgnoreCase(geocode, other.geocode) &&
389                     StringUtils.equalsIgnoreCase(name, other.name) &&
390                     cacheType == other.cacheType &&
391                     size == other.size &&
392                     ObjectUtils.equals(found, other.found) &&
393                     ObjectUtils.equals(premiumMembersOnly, other.premiumMembersOnly) &&
394                     difficulty == other.difficulty &&
395                     terrain == other.terrain &&
396                     (coords != null ? coords.equals(other.coords) : null == other.coords) &&
397                     reliableLatLon == other.reliableLatLon &&
398                     ObjectUtils.equals(disabled, other.disabled) &&
399                     ObjectUtils.equals(archived, other.archived) &&
400                     listId == other.listId &&
401                     StringUtils.equalsIgnoreCase(ownerDisplayName, other.ownerDisplayName) &&
402                     StringUtils.equalsIgnoreCase(ownerUserId, other.ownerUserId) &&
403                     StringUtils.equalsIgnoreCase(getDescription(), other.getDescription()) &&
404                     StringUtils.equalsIgnoreCase(personalNote, other.personalNote) &&
405                     StringUtils.equalsIgnoreCase(getShortDescription(), other.getShortDescription()) &&
406                     StringUtils.equalsIgnoreCase(getLocation(), other.getLocation()) &&
407                     ObjectUtils.equals(favorite, other.favorite) &&
408                     favoritePoints == other.favoritePoints &&
409                     ObjectUtils.equals(onWatchlist, other.onWatchlist) &&
410                     (hidden != null ? hidden.equals(other.hidden) : null == other.hidden) &&
411                     StringUtils.equalsIgnoreCase(guid, other.guid) &&
412                     StringUtils.equalsIgnoreCase(getHint(), other.getHint()) &&
413                     StringUtils.equalsIgnoreCase(cacheId, other.cacheId) &&
414                     (direction != null ? direction.equals(other.direction) : null == other.direction) &&
415                     (distance != null ? distance.equals(other.distance) : null == other.distance) &&
416                     rating == other.rating &&
417                     votes == other.votes &&
418                     myVote == other.myVote &&
419                     inventoryItems == other.inventoryItems &&
420                     attributes == other.attributes &&
421                     waypoints == other.waypoints &&
422                     spoilers == other.spoilers &&
423                     inventory == other.inventory &&
424                     logCounts == other.logCounts &&
425                     ObjectUtils.equals(logOffline, other.logOffline) &&
426                     finalDefined == other.finalDefined;
427         }

433         public boolean canBeAddedToCalendar() {
434             // is event type?
435             if (!isEventCache()) {
436                 return false;
437             }
438             // has event date set?
439             if (hidden == null) {
440                 return false;
441             }
442             // is not in the past?
443             final Calendar cal = Calendar.getInstance();
444             cal.setTime(new Date());
445             cal.set(Calendar.HOUR_OF_DAY, 0);
446             cal.set(Calendar.MINUTE, 0);
447             cal.set(Calendar.SECOND, 0);
448             cal.set(Calendar.MILLISECOND, 0);
449             return hidden.compareTo(cal.getTime()) >= 0;
450         }

478             final boolean status = DataStore.saveLogOffline(geocode, date.getTime(), logType, log);
479     
480             final Resources res = fromActivity.getResources();

482                 ActivityMixin.showToast(fromActivity, res.getString(R.string.info_log_saved));

790         @Override
791         public String getNameForSorting() {
792             if (null == nameForSorting) {
793                 nameForSorting = name;
794                 // pad each number part to a fixed size of 6 digits, so that numerical sorting becomes equivalent to string sorting
795                 MatcherWrapper matcher = new MatcherWrapper(NUMBER_PATTERN, nameForSorting);
796                 int start = 0;
797                 while (matcher.find(start)) {
798                     final String number = matcher.group();
799                     nameForSorting = StringUtils.substring(nameForSorting, 0, matcher.start()) + StringUtils.leftPad(number, 6, '0') + StringUtils.substring(nameForSorting, matcher.start() + number.length());
800                     start = matcher.start() + Math.max(6, number.length());
801                     matcher = new MatcherWrapper(NUMBER_PATTERN, nameForSorting);
802                 }
803             }
804             return nameForSorting;
805         }

1185         /**
1186          * @param waypoint
1187          *            Waypoint to add to the cache
1188          * @param saveToDatabase
1189          *            Indicates whether to add the waypoint to the database. Should be false if
1190          *            called while loading or building a cache
1191          * @return <code>true</code> if waypoint successfully added to waypoint database
1192          */
1193         public boolean addOrChangeWaypoint(final Waypoint waypoint, boolean saveToDatabase) {
1194             waypoint.setGeocode(geocode);
1195     
1196             if (waypoint.getId() < 0) { // this is a new waypoint
1197                 assignUniquePrefix(waypoint);
1198                 waypoints.add(waypoint);
1199                 if (waypoint.isFinalWithCoords()) {
1200                     finalDefined = true;
1201                 }
1202             } else { // this is a waypoint being edited
1203                 final int index = getWaypointIndex(waypoint);
1204                 if (index >= 0) {
1205                     Waypoint oldWaypoint = waypoints.remove(index);
1206                     waypoint.setPrefix(oldWaypoint.getPrefix());
1207                     //migration
1208                     if (StringUtils.isBlank(waypoint.getPrefix())
1209                             || StringUtils.equalsIgnoreCase(waypoint.getPrefix(), Waypoint.PREFIX_OWN)) {
1210                         assignUniquePrefix(waypoint);
1211                     }
1212                 }
1213                 waypoints.add(waypoint);
1214                 // when waypoint was edited, finalDefined may have changed
1215                 resetFinalDefined();
1216             }
1217             return saveToDatabase && DataStore.saveWaypoint(waypoint.getId(), geocode, waypoint);
1218         }

1220         /*
1221          * Assigns a unique two-digit (compatibility with gc.com)
1222          * prefix within the scope of this cache.
1223          */
1224         private void assignUniquePrefix(Waypoint waypoint) {
1225             // gather existing prefixes
1226             Set<String> assignedPrefixes = new HashSet<String>();
1227             for (Waypoint wp : waypoints) {
1228                 assignedPrefixes.add(wp.getPrefix());
1229             }
1230     
1231             for (int i = OWN_WP_PREFIX_OFFSET; i < 100; i++) {
1232                 String prefixCandidate = StringUtils.leftPad(String.valueOf(i), 2, '0');
1233                 if (!assignedPrefixes.contains(prefixCandidate)) {
1234                     waypoint.setPrefix(prefixCandidate);
1235                     break;
1236                 }
1237             }
1238     
1239         }

1300          */
1301         public boolean deleteWaypoint(final Waypoint waypoint) {
1302             if (waypoint == null) {
1303                 return false;
1304             }
1305             if (waypoint.getId() < 0) {
1306                 return false;
1307             }
1308             if (waypoint.isUserDefined()) {

1310                 waypoints.remove(index);
1311                 DataStore.deleteWaypoint(waypoint.getId());
1312                 DataStore.removeCache(geocode, EnumSet.of(RemoveFlag.REMOVE_CACHE));
1313                 // Check status if Final is defined
1314                 if (waypoint.isFinalWithCoords()) {
1315                     resetFinalDefined();
1316                 }
1317                 return true;
1318             }
1319             return false;
1320         }

1459         public void checkFields() {
1460             if (StringUtils.isBlank(getGeocode())) {
1461                 Log.w("geo code not parsed correctly for " + geocode);
1462             }
1463             if (StringUtils.isBlank(getName())) {
1464                 Log.w("name not parsed correctly for " + geocode);
1465             }
1466             if (StringUtils.isBlank(getGuid())) {
1467                 Log.w("guid not parsed correctly for " + geocode);
1468             }
1469             if (getTerrain() == 0.0) {
1470                 Log.w("terrain not parsed correctly for " + geocode);

1516         public static void storeCache(Geocache origCache, String geocode, int listId, boolean forceRedownload, CancellableHandler handler) {
1517             try {
1518                 Geocache cache = null;
1519                 // get cache details, they may not yet be complete
1520                 if (origCache != null) {
1521                     SearchResult search = null;
1522                     // only reload the cache if it was already stored or doesn't have full details (by checking the description)
1523                     if (origCache.isOffline() || StringUtils.isBlank(origCache.getDescription())) {
1524                         search = searchByGeocode(origCache.getGeocode(), null, listId, false, handler);
1525                     }
1526                     if (search != null) {
1527                         cache = search.getFirstCacheFromResult(LoadFlags.LOAD_CACHE_OR_DB);
1528                     } else {
1529                         cache = origCache;
1530                     }
1531                 } else if (StringUtils.isNotBlank(geocode)) {
1532                     final SearchResult search = searchByGeocode(geocode, null, listId, forceRedownload, handler);
1533                     if (search != null) {
1534                         cache = search.getFirstCacheFromResult(LoadFlags.LOAD_CACHE_OR_DB);
1535                     }
1536                 }
1537     
1538                 if (cache == null) {
1539                     if (handler != null) {
1540                         handler.sendMessage(Message.obtain());
1541                     }
1542     
1543                     return;
1544                 }
1545     
1546                 if (CancellableHandler.isCancelled(handler)) {
1547                     return;
1548                 }
1549     
1550                 final HtmlImage imgGetter = new HtmlImage(cache.getGeocode(), false, listId, true);
1551     
1552                 // store images from description
1553                 if (StringUtils.isNotBlank(cache.getDescription())) {
1554                     Html.fromHtml(cache.getDescription(), imgGetter, null);
1555                 }
1556     
1557                 if (CancellableHandler.isCancelled(handler)) {
1558                     return;
1559                 }
1560     
1561                 // store spoilers
1562                 if (CollectionUtils.isNotEmpty(cache.getSpoilers())) {
1563                     for (final Image oneSpoiler : cache.getSpoilers()) {
1564                         imgGetter.getDrawable(oneSpoiler.getUrl());
1565                     }
1566                 }
1567     
1568                 if (CancellableHandler.isCancelled(handler)) {
1569                     return;
1570                 }
1571     
1572                 // store images from logs
1573                 if (Settings.isStoreLogImages()) {
1574                     for (final LogEntry log : cache.getLogs()) {
1575                         if (log.hasLogImages()) {
1576                             for (final Image oneLogImg : log.getLogImages()) {
1577                                 imgGetter.getDrawable(oneLogImg.getUrl());
1578                             }
1579                         }
1580                     }
1581                 }
1582     
1583                 if (CancellableHandler.isCancelled(handler)) {
1584                     return;
1585                 }
1586     
1587                 cache.setListId(listId);
1588                 DataStore.saveCache(cache, EnumSet.of(SaveFlag.SAVE_DB));
1589     
1590                 if (CancellableHandler.isCancelled(handler)) {
1591                     return;
1592                 }
1593     
1594                 StaticMapsProvider.downloadMaps(cache);
1595     
1596                 imgGetter.waitForBackgroundLoading(handler);
1597     
1598                 if (handler != null) {
1599                     handler.sendMessage(Message.obtain());
1600                 }
1601             } catch (final Exception e) {
1602                 Log.e("Geocache.storeCache", e);
1603             }
1604         }

1606         public static SearchResult searchByGeocode(final String geocode, final String guid, final int listId, final boolean forceReload, final CancellableHandler handler) {
1607             if (StringUtils.isBlank(geocode) && StringUtils.isBlank(guid)) {
1608                 Log.e("Geocache.searchByGeocode: No geocode nor guid given");
1609                 return null;
1610             }
1611     
1612             if (!forceReload && listId == StoredList.TEMPORARY_LIST_ID && (DataStore.isOffline(geocode, guid) || DataStore.isThere(geocode, guid, true, true))) {
1613                 final SearchResult search = new SearchResult();
1614                 final String realGeocode = StringUtils.isNotBlank(geocode) ? geocode : DataStore.getGeocodeForGuid(guid);
1615                 search.addGeocode(realGeocode);
1616                 return search;
1617             }
1618     
1619             // if we have no geocode, we can't dynamically select the handler, but must explicitly use GC
1620             if (geocode == null && guid != null) {
1621                 return GCConnector.getInstance().searchByGeocode(null, guid, handler);
1622             }
1623     
1624             final IConnector connector = ConnectorFactory.getConnector(geocode);
1625             if (connector instanceof ISearchByGeocode) {
1626                 return ((ISearchByGeocode) connector).searchByGeocode(geocode, guid, handler);
1627             }
1628             return null;
1629         }

1635         /**
1636          * guess an event start time from the description
1637          *
1638          * @return start time in minutes after midnight
1639          */
1640         public String guessEventTimeMinutes() {
1641             if (!isEventCache()) {
1642                 return null;
1643             }
1644     
1645             final String hourLocalized = CgeoApplication.getInstance().getString(R.string.cache_time_full_hours);
1646             ArrayList<Pattern> patterns = new ArrayList<Pattern>();
1647     
1648             // 12:34
1649             patterns.add(Pattern.compile("\\b(\\d{1,2})\\:(\\d\\d)\\b"));
1650             if (StringUtils.isNotBlank(hourLocalized)) {
1651                 // 17 - 20 o'clock
1652                 patterns.add(Pattern.compile("\\b(\\d{1,2})(?:\\.00)?" + "\\s*(?:-|[a-z]+)\\s*" + "(?:\\d{1,2})(?:\\.00)?" + "\\s+" + Pattern.quote(hourLocalized), Pattern.CASE_INSENSITIVE));
1653                 // 12 o'clock, 12.00 o'clock
1654                 patterns.add(Pattern.compile("\\b(\\d{1,2})(?:\\.00)?\\s+" + Pattern.quote(hourLocalized), Pattern.CASE_INSENSITIVE));
1655             }
1656     
1657             final String searchText = getShortDescription() + ' ' + getDescription();
1658             for (Pattern pattern : patterns) {
1659                 final MatcherWrapper matcher = new MatcherWrapper(pattern, searchText);
1660                 while (matcher.find()) {
1661                     try {
1662                         final int hours = Integer.parseInt(matcher.group(1));
1663                         int minutes = 0;
1664                         if (matcher.groupCount() >= 2) {
1665                             minutes = Integer.parseInt(matcher.group(2));
1666                         }
1667                         if (hours >= 0 && hours < 24 && minutes >= 0 && minutes < 60) {
1668                             return String.valueOf(hours * 60 + minutes);
1669                         }
1670                     } catch (final NumberFormatException e) {
1671                         // cannot happen, but static code analysis doesn't know
1672                     }
1673                 }
1674             }
1675             return null;
1676         }

1738         private void addLocalSpoilersTo(final List<Image> spoilers) {
1739             if (StringUtils.length(geocode) >= 2) {
1740                 final String suffix = StringUtils.right(geocode, 2);

1745                 final File[] files = finalDir.listFiles();
1746                 if (files != null) {
1747                     for (final File image : files) {
1748                         spoilers.add(new Image("file://" + image.getAbsolutePath(), image.getName()));

1814         public LogType getDefaultLogType() {
1815             if (isEventCache()) {
1816                 final Date eventDate = getHiddenDate();
1817                 boolean expired = DateUtils.isPastEvent(this);
1818     
1819                 if (hasOwnLog(LogType.WILL_ATTEND) || expired || (eventDate != null && DateUtils.daysSince(eventDate.getTime()) == 0)) {
1820                     return hasOwnLog(LogType.ATTENDED) ? LogType.NOTE : LogType.ATTENDED;
1821                 }
1822                 return LogType.WILL_ATTEND;
1823             }
1824             if (isFound()) {
1825                 return LogType.NOTE;
1826             }
1827             if (getType() == CacheType.WEBCAM) {
1828                 return LogType.WEBCAM_PHOTO_TAKEN;
1829             }
1830             return LogType.FOUND_IT;
1831         }
