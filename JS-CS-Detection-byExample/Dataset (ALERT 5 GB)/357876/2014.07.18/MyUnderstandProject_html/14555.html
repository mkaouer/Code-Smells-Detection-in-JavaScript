556                         showToast(R.string.err_dwld_details_failed);

565                         finishActivity();
566                         return;
567                     }
568     
569                     updateStatusMsg(activity.getResources().getString(R.string.cache_dialog_loading_details_status_render));
570     
571                     // Data loaded, we're ready to show it!
572                     activity.notifyDataSetChanged();
573                 }
574             }

696         private class AttributeViewBuilder {
697             private ViewGroup attributeIconsLayout; // layout for attribute icons
698             private ViewGroup attributeDescriptionsLayout; // layout for attribute descriptions
699             private boolean attributesShowAsIcons = true; // default: show icons
700             /**
701              * If the cache is from a non GC source, it might be without icons. Disable switching in those cases.
702              */
703             private boolean noAttributeIconsFound = false;
704             private int attributeBoxMaxWidth;
705     
706             public void fillView(final LinearLayout attributeBox) {
707                 // first ensure that the view is empty
708                 attributeBox.removeAllViews();
709     
710                 // maximum width for attribute icons is screen width - paddings of parents
711                 attributeBoxMaxWidth = Compatibility.getDisplayWidth();
712                 ViewParent child = attributeBox;
713                 do {
714                     if (child instanceof View) {
715                         attributeBoxMaxWidth -= ((View) child).getPaddingLeft() + ((View) child).getPaddingRight();
716                     }
717                     child = child.getParent();
718                 } while (child != null);
719     
720                 // delete views holding description / icons
721                 attributeDescriptionsLayout = null;
722                 attributeIconsLayout = null;
723     
724                 attributeBox.setOnClickListener(new View.OnClickListener() {
725                     @Override
726                     public void onClick(View v) {
727                         // toggle between attribute icons and descriptions
728                         toggleAttributeDisplay(attributeBox, attributeBoxMaxWidth);
729                     }
730                 });
731     
732                 // icons or text?
733                 //
734                 // also show icons when noAttributeImagesFound == true. Explanation:
735                 //  1. no icons could be found in the first invocation of this method
736                 //  2. user refreshes cache from web
737                 //  3. now this method is called again
738                 //  4. attributeShowAsIcons is false but noAttributeImagesFound is true
739                 //     => try to show them now
740                 if (attributesShowAsIcons || noAttributeIconsFound) {
741                     showAttributeIcons(attributeBox, attributeBoxMaxWidth);
742                 } else {
743                     showAttributeDescriptions(attributeBox);
744                 }
745             }
746     
747             /**
748              * lazy-creates the layout holding the icons of the caches attributes
749              * and makes it visible
750              */
751             private void showAttributeIcons(LinearLayout attribBox, int parentWidth) {
752                 if (attributeIconsLayout == null) {
753                     attributeIconsLayout = createAttributeIconsLayout(parentWidth);
754                     // no matching icons found? show text
755                     if (noAttributeIconsFound) {
756                         showAttributeDescriptions(attribBox);
757                         return;
758                     }
759                 }
760                 attribBox.removeAllViews();
761                 attribBox.addView(attributeIconsLayout);
762                 attributesShowAsIcons = true;
763             }
764     
765             /**
766              * lazy-creates the layout holding the descriptions of the caches attributes
767              * and makes it visible
768              */
769             private void showAttributeDescriptions(LinearLayout attribBox) {
770                 if (attributeDescriptionsLayout == null) {
771                     attributeDescriptionsLayout = createAttributeDescriptionsLayout();
772                 }
773                 attribBox.removeAllViews();
774                 attribBox.addView(attributeDescriptionsLayout);
775                 attributesShowAsIcons = false;
776             }
777     
778             /**
779              * toggle attribute descriptions and icons
780              */
781             private void toggleAttributeDisplay(LinearLayout attribBox, int parentWidth) {
782                 // Don't toggle when there are no icons to show.
783                 if (noAttributeIconsFound) {
784                     return;
785                 }
786     
787                 // toggle
788                 if (attributesShowAsIcons) {
789                     showAttributeDescriptions(attribBox);
790                 } else {
791                     showAttributeIcons(attribBox, parentWidth);
792                 }
793             }

795             private ViewGroup createAttributeIconsLayout(int parentWidth) {
796                 final LinearLayout rows = new LinearLayout(CacheDetailActivity.this);
797                 rows.setLayoutParams(new LayoutParams(LayoutParams.MATCH_PARENT, LayoutParams.WRAP_CONTENT));
798                 rows.setOrientation(LinearLayout.VERTICAL);
799     
800                 LinearLayout attributeRow = newAttributeIconsRow();
801                 rows.addView(attributeRow);
802     
803                 noAttributeIconsFound = true;
804     
805                 for (final String attributeName : cache.getAttributes()) {
806                     // check if another attribute icon fits in this row
807                     attributeRow.measure(0, 0);
808                     final int rowWidth = attributeRow.getMeasuredWidth();
809                     final FrameLayout fl = (FrameLayout) getLayoutInflater().inflate(R.layout.attribute_image, null);
810                     final ImageView iv = (ImageView) fl.getChildAt(0);
811                     if ((parentWidth - rowWidth) < iv.getLayoutParams().width) {
812                         // make a new row
813                         attributeRow = newAttributeIconsRow();
814                         rows.addView(attributeRow);
815                     }
816     
817                     final boolean strikethru = !CacheAttribute.isEnabled(attributeName);
818                     final CacheAttribute attrib = CacheAttribute.getByRawName(CacheAttribute.trimAttributeName(attributeName));
819                     if (attrib != null) {
820                         noAttributeIconsFound = false;
821                         Drawable d = res.getDrawable(attrib.drawableId);
822                         iv.setImageDrawable(d);
823                         // strike through?
824                         if (strikethru) {
825                             // generate strikethru image with same properties as attribute image
826                             final ImageView strikethruImage = new ImageView(CacheDetailActivity.this);
827                             strikethruImage.setLayoutParams(iv.getLayoutParams());
828                             d = res.getDrawable(R.drawable.attribute__strikethru);
829                             strikethruImage.setImageDrawable(d);
830                             fl.addView(strikethruImage);
831                         }
832                     } else {
833                         final Drawable d = res.getDrawable(R.drawable.attribute_unknown);
834                         iv.setImageDrawable(d);
835                     }
836     
837                     attributeRow.addView(fl);
838                 }
839     
840                 return rows;
841             }

851             private ViewGroup createAttributeDescriptionsLayout() {
852                 final LinearLayout descriptions = (LinearLayout) getLayoutInflater().inflate(
853                         R.layout.attribute_descriptions, null);
854                 final TextView attribView = (TextView) descriptions.getChildAt(0);
855     
856                 final StringBuilder buffer = new StringBuilder();
857                 for (String attributeName : cache.getAttributes()) {
858                     final boolean enabled = CacheAttribute.isEnabled(attributeName);
859                     // search for a translation of the attribute
860                     CacheAttribute attrib = CacheAttribute.getByRawName(CacheAttribute.trimAttributeName(attributeName));
861                     if (attrib != null) {
862                         attributeName = attrib.getL10n(enabled);
863                     }
864                     if (buffer.length() > 0) {
865                         buffer.append('\n');
866                     }
867                     buffer.append(attributeName);
868                 }
869     
870                 attribView.setText(buffer);
871     
872                 return descriptions;
873             }

1577        /**
1578          * Load the description in the background.
1579         * @param descriptionString the HTML description as retrieved from the connector
1580         * @param descriptionView the view to fill
1581         * @param loadingIndicatorView the loading indicator view, will be hidden when completed
1582         */
1583         private void loadDescription(final String descriptionString, final IndexOutOfBoundsAvoidingTextView descriptionView, final View loadingIndicatorView) {
1584             // The producer produces successives (without then with images) versions of the description.
1585             final Observable<Spanned> producer = Observable.create(new OnSubscribe<Spanned>() {
1586                 @Override
1587                 public void call(final Subscriber<? super Spanned> subscriber) {
1588                     try {
1589                         // Fast preview: parse only HTML without loading any images
1590                         final HtmlImageCounter imageCounter = new HtmlImageCounter();
1591                         final UnknownTagsHandler unknownTagsHandler = new UnknownTagsHandler();
1592                         Spanned description = Html.fromHtml(descriptionString, imageCounter, unknownTagsHandler);
1593                         addWarning(unknownTagsHandler, description);
1594                         subscriber.onNext(description);
1595     
1596                         if (imageCounter.getImageCount() > 0) {
1597                             // Complete view: parse again with loading images - if necessary ! If there are any images causing problems the user can see at least the preview
1598                             description = Html.fromHtml(descriptionString, new HtmlImage(cache.getGeocode(), true, cache.getListId(), false), unknownTagsHandler);
1599                             addWarning(unknownTagsHandler, description);
1600                             subscriber.onNext(description);
1601                         }
1602     
1603                         subscriber.onCompleted();
1604                     } catch (final Exception e) {
1605                         Log.e("loadDescription", e);
1606                         subscriber.onError(e);
1607                     }
1608                 }
1609     
1610                 // If description has an HTML construct which may be problematic to render, add a note at the end of the long description.
1611                 // Technically, it may not be a table, but a pre, which has the same problems as a table, so the message is ok even though
1612                 // sometimes technically incorrect.
1613                 private void addWarning(final UnknownTagsHandler unknownTagsHandler, final Spanned description) {
1614                     if (unknownTagsHandler.isProblematicDetected()) {
1615                         final int startPos = description.length();
1616                         final IConnector connector = ConnectorFactory.getConnector(cache);
1617                         final Spanned tableNote = Html.fromHtml(res.getString(R.string.cache_description_table_note, "<a href=\"" + cache.getUrl() + "\">" + connector.getName() + "</a>"));
1618                         ((Editable) description).append("\n\n").append(tableNote);
1619                         ((Editable) description).setSpan(new StyleSpan(Typeface.ITALIC), startPos, description.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);
1620                     }
1621                 }
1622             });
1623     
1624             AndroidObservable.bindActivity(this, producer).subscribe(new Observer<Spanned>() {
1625                 @Override
1626                 public void onCompleted() {
1627                     if (null != loadingIndicatorView) {
1628                         loadingIndicatorView.setVisibility(View.GONE);
1629                     }
1630                 }
1631     
1632                 @Override
1633                 public void onError(final Throwable throwable) {
1634                     showToast(res.getString(R.string.err_load_descr_failed));
1635                 }
1636     
1637                 @Override
1638                 public void onNext(final Spanned description) {
1639                     if (StringUtils.isNotBlank(descriptionString)) {
1640                         try {
1641                             descriptionView.setText(description, TextView.BufferType.SPANNABLE);
1642                         } catch (final Exception e) {
1643                             // On 4.1, there is sometimes a crash on measuring the layout: https://code.google.com/p/android/issues/detail?id=35412
1644                             Log.e("Android bug setting text: ", e);
1645                             // remove the formatting by converting to a simple string
1646                             descriptionView.setText(description.toString());
1647                         }
1648                         descriptionView.setMovementMethod(AnchorAwareLinkMovementMethod.getInstance());
1649                         fixTextColor(descriptionString);
1650                         descriptionView.setVisibility(View.VISIBLE);
1651                         registerForContextMenu(descriptionView);
1652                     }
1653                 }
1654     
1655                 /**
1656                  * Handle caches with black font color in dark skin and white font color in light skin
1657                  * by changing background color of the view
1658                  *
1659                  * @param text
1660                  *            to be checked
1661                  */
1662                 private void fixTextColor(final String text) {
1663                     int backcolor;
1664                     if (Settings.isLightSkin()) {
1665                         backcolor = color.white;
1666     
1667                         for (final Pattern pattern : LIGHT_COLOR_PATTERNS) {
1668                             final MatcherWrapper matcher = new MatcherWrapper(pattern, text);
1669                             if (matcher.find()) {
1670                                 descriptionView.setBackgroundResource(color.darker_gray);
1671                                 return;
1672                             }
1673                         }
1674                     } else {
1675                         backcolor = color.black;
1676     
1677                         for (final Pattern pattern : DARK_COLOR_PATTERNS) {
1678                             final MatcherWrapper matcher = new MatcherWrapper(pattern, text);
1679                             if (matcher.find()) {
1680                                 descriptionView.setBackgroundResource(color.darker_gray);
1681                                 return;
1682                             }
1683                         }
1684                     }
1685                     descriptionView.setBackgroundResource(backcolor);
1686                 }
1687             }, Schedulers.io());
