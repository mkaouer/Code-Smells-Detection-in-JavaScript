15         /** Calculate from a list of positions (x/y) the coords */
16         public static UTFGridPosition getPositionInGrid(List<UTFGridPosition> positions) {
17             int minX = GRID_MAXX;
18             int maxX = 0;
19             int minY = GRID_MAXY;
20             int maxY = 0;
21             for (UTFGridPosition pos : positions) {
22                 minX = Math.min(minX, pos.x);
23                 maxX = Math.max(maxX, pos.x);
24                 minY = Math.min(minY, pos.y);
25                 maxY = Math.max(maxY, pos.y);
26             }
27             return new UTFGridPosition((minX + maxX) / 2, (minY + maxY) / 2);
28         }
