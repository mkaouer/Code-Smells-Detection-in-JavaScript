270         private class LoadDataThread extends Thread {
271     
272             public LoadDataThread() {
273                 super("Load data for logging trackable");
274                 if (guid == null) {
275                     showToast(res.getString(R.string.err_tb_forgot_saw));
276     
277                     finish();
278                 }
279             }
280     
281             @Override
282             public void run() {
283                 final Parameters params = new Parameters();
284     
285                 showProgressHandler.sendEmptyMessage(0);
286                 gettingViewstate = true;
287                 attempts++;
288     
289                 try {
290                     if (StringUtils.isNotBlank(guid)) {
291                         params.put("wid", guid);
292                     } else {
293                         loadDataHandler.sendEmptyMessage(0);
294                         return;
295                     }
296     
297                     final String page = Network.getResponseData(Network.getRequest("http://www.geocaching.com/track/log.aspx", params));
298     
299                     viewstates = GCLogin.getViewstates(page);
300     
301                     final List<LogType> typesPre = GCParser.parseTypes(page);
302                     if (CollectionUtils.isNotEmpty(typesPre)) {
303                         possibleLogTypes.clear();
304                         possibleLogTypes.addAll(typesPre);
305                     }
306                 } catch (Exception e) {
307                     Log.e("LogTrackableActivity.LoadDataThread.run", e);
308                 }
309     
310                 loadDataHandler.sendEmptyMessage(0);
311             }
312         }

314         private class PostLogThread extends Thread {
315             final private Handler handler;
316             final private String tracking;
317             final private String log;
318     
319             public PostLogThread(final Handler handlerIn, final String trackingIn, final String logIn) {
320                 super("Post trackable log");
321                 handler = handlerIn;
322                 tracking = trackingIn;
323                 log = logIn;
324             }
325     
326             @Override
327             public void run() {
328                 final StatusCode status = postLogFn(tracking, log);
329                 handler.sendMessage(handler.obtainMessage(0, status));
330             }
331         }
332     
333         public StatusCode postLogFn(String tracking, String log) {
334             try {
335                 final StatusCode status = GCParser.postLogTrackable(guid, tracking, viewstates, typeSelected, date.get(Calendar.YEAR), (date.get(Calendar.MONTH) + 1), date.get(Calendar.DATE), log);
336     
337                 if (status == StatusCode.NO_ERROR && Settings.isUseTwitter() &&
338                         Settings.isTwitterLoginValid() &&
339                         tweetCheck.isChecked() && tweetBox.getVisibility() == View.VISIBLE) {
340                     Twitter.postTweetTrackable(geocode, new LogEntry(0, typeSelected, log));
341                 }
342     
343                 return status;
344             } catch (Exception e) {
345                 Log.e("LogTrackableActivity.postLogFn", e);
346             }
347     
348             return StatusCode.LOG_POST_ERROR;
349         }
