29         /**
30          * The place for the implementation to prove that the new version of replaceWhitespace is faster than
31          * BaseUtils.replaceWhitespace()
32          *
33          * @param data
34          * @return
35          */
36         public static String replaceWhitespaceManually(final String data) {
37             final int length = data.length();
38             final char[] chars = new char[length];
39             data.getChars(0, length, chars, 0);
40             int resultSize = 0;
41             boolean lastWasWhitespace = true;
42             for (char c : chars) {
43                 if (c == ' ' || c == '\n' || c == '\r' || c == '\t') {
44                     if (!lastWasWhitespace) {
45                         chars[resultSize++] = ' ';
46                     }
47                     lastWasWhitespace = true;
48                 } else {
49                     chars[resultSize++] = c;
50                     lastWasWhitespace = false;
51                 }
52             }
53             return String.valueOf(chars, 0, resultSize);
54         }
55     
56         public static String replaceWhitespaceStringUtils(final String data) {
57             return StringUtils.join(StringUtils.split(data, " \n\r\t"), " ");
