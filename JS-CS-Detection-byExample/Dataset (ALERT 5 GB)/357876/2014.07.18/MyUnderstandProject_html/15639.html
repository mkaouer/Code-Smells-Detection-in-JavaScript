9      * abstract super implementation for all cache comparators

11      */
12     public abstract class AbstractCacheComparator implements CacheComparator {
13     
14         @Override
15         public final int compare(final Geocache cache1, final Geocache cache2) {
16             try {
17                 final boolean canCompare1 = canCompare(cache1);
18                 final boolean canCompare2 = canCompare(cache2);
19                 if (!canCompare1) {
20                     return canCompare2 ? 1 : fallbackToGeocode(cache1, cache2);
21                 }
22                 return canCompare2 ? compareCaches(cache1, cache2) : -1;
23             } catch (final Exception e) {
24                 Log.e("AbstractCacheComparator.compare", e);
25                 // This may violate the Comparator interface if the exception is not systematic.
26                 return fallbackToGeocode(cache1, cache2);
27             }
28         }
29     
30         private static int fallbackToGeocode(final Geocache cache1, final Geocache cache2) {
31             return StringUtils.defaultString(cache1.getGeocode()).compareToIgnoreCase(StringUtils.defaultString(cache2.getGeocode()));
32         }
33     
34         /**
35          * Check necessary preconditions (like missing fields) before running the comparison itself.
36          * Caches not filling the conditions will be placed last, sorted by Geocode.
37          *
38          * The default returns <code>true</code> and can be overridden if needed in child classes.
39          *
40          * @param cache
41          * @return <code>true</code> if the cache holds the necessary data to be compared meaningfully
42          */
43         @SuppressWarnings("static-method")
44         protected boolean canCompare(final Geocache cache) {
45             return true;
46         }

48         /**
49          * Compares two caches. Logging and exception handling is implemented outside this method already.
50          * <p/>
51          * A cache is smaller than another cache if it is desirable to show it first when presented to the user.
52          * For example, a highly rated cache must be considered smaller than a poorly rated one.
53          *
54          * @param cache1
55          * @param cache2
56          * @return an integer < 0 if cache1 is less than cache2, 0 if they are equal, and > 0 if cache1 is greater than
57          *         cache2.

59         protected abstract int compareCaches(final Geocache cache1, final Geocache cache2);
