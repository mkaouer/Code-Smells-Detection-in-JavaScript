3     import java.io.FilterInputStream;
4     import java.io.IOException;
5     import java.io.InputStream;
6     
7     /**
8      * Filter input stream that doesn't forward close() calls. Needed to parse multiple XML documents by SAX XML parser from
9      * one input stream (e.g. ZipInputStream) because SAX parser closes stream.
10      */
11     public class NoCloseInputStream extends FilterInputStream {
12         private static ClosedInputStream closedInputStream = new ClosedInputStream();
13     
14         public NoCloseInputStream(InputStream in) {
15             super(in);
16         }
17     
18         @Override
19         public void close() throws IOException {
20             in = closedInputStream;
21         }
22     
23     
24         private static class ClosedInputStream extends InputStream {
25             @Override
26             public int read() throws IOException {
27                 throw new IOException("Stream already closed.");
28             }
29         }
30     }
