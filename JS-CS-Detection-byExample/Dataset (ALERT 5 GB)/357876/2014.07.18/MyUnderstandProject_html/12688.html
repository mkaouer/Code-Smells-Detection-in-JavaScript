44         /**
45          * Adds the cache to the Android-calendar if it is an event.
46          * 
47          * @param entry
48          */
49         private void selectCalendarForAdding(final CalendarEntry entry) {
50             final String[] projection = new String[] { "_id", "displayName" };
51             final Uri calendarProvider = Compatibility.getCalendarProviderURI();
52     
53             final Cursor cursor = managedQuery(calendarProvider, projection, "selected=1", null, null);
54     
55             if (cursor == null || cursor.getCount() <= 0) {
56                 showToast(R.string.event_fail);
57                 finish();
58                 return;
59             }
60     
61             final SparseArray<String> calendars = new SparseArray<String>();
62             cursor.moveToFirst();
63     
64             final int indexId = cursor.getColumnIndex("_id");
65             final int indexName = cursor.getColumnIndex("displayName");
66     
67             do {
68                 final String idString = cursor.getString(indexId);
69                 if (idString != null) {
70                     try {
71                         int id = Integer.parseInt(idString);
72                         final String calName = cursor.getString(indexName);
73     
74                         if (id > 0 && calName != null) {
75                             calendars.put(id, calName);
76                         }
77                     } catch (NumberFormatException e) {
78                         Log.e(LOG_TAG, "CalendarActivity.selectCalendarForAdding", e);
79                     }
80                 }
81             } while (cursor.moveToNext());
82     
83             if (calendars.size() == 0) {
84                 showToast(R.string.event_fail);
85                 finish();
86                 return;
87             }
88     
89             final String[] items = new String[calendars.size()];
90             for (int i = 0; i < calendars.size(); i++) {
91                 items[i] = calendars.valueAt(i);
92             }
93     
94             final AlertDialog.Builder builder = new AlertDialog.Builder(this);
95             builder.setTitle(R.string.calendars);
96             builder.setItems(items, new DialogInterface.OnClickListener() {
97                 @Override
98                 public void onClick(DialogInterface dialog, int item) {
99                     final int calendarId = calendars.keyAt(item);
100                     new AddEntry(entry, CalendarActivity.this, calendarId).addEntryToCalendar();
101                     finish();
102                 }
103             });
104             builder.setOnCancelListener(new DialogInterface.OnCancelListener() {
105                 @Override
106                 public void onCancel(DialogInterface dialog) {
107                     finish();
108                 }
109             });
110             builder.create().show();
111         }
