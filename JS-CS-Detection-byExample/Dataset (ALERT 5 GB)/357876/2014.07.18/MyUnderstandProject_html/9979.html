53         // This class implements an all-purpose image getter that can also be used as a ImageGetter interface
54         // when displaying caches. An instance mainly has three possible use cases:
55         //  - If onlySave is true, getDrawable() will return null immediately and will queue the image retrieval
56         //    and saving in the loading subject. Downloads will start in parallel when the blocking
57         //    waitForBackgroundLoading() method is called, and they can be cancelled through the given handler.
58         //  - If onlySave is false and the instance is called through fetchDrawable(), then an observable for the
59         //    given URL will be returned. This observable will emit the local copy of the image if it is present,
60         //    regardless of its freshness, then if needed an updated fresher copy after retrieving it from the network.
61         //  - If onlySave is false and the instance is used as an ImageGetter, only the final version of the
62         //    image will be returned.

128         // Caches are loaded from disk on a computation scheduler to avoid using more threads than cores while decoding
129         // the image. Downloads happen on downloadScheduler, in parallel with image decoding.
130         public Observable<BitmapDrawable> fetchDrawable(final String url) {
131     
132             if (StringUtils.isBlank(url) || ImageUtils.containsPattern(url, BLOCKED)) {
133                 return Observable.from(getTransparent1x1Image(resources));
134             }
135     
136             // Explicit local file URLs are loaded from the filesystem regardless of their age. The IO part is short
137             // enough to make the whole operation on the computation scheduler.
138             if (FileUtils.isFileUrl(url)) {
139                 return Observable.defer(new Func0<Observable<? extends BitmapDrawable>>() {
140                     @Override
141                     public Observable<? extends BitmapDrawable> call() {
142                         final Bitmap bitmap = loadCachedImage(FileUtils.urlToFile(url), true).getLeft();
143                         return bitmap != null ? Observable.from(ImageUtils.scaleBitmapToFitDisplay(bitmap)) : Observable.<BitmapDrawable>empty();
144                     }
145                 }).subscribeOn(RxUtils.computationScheduler);
146             }
147     
148             final boolean shared = url.contains("/images/icons/icon_");
149             final String pseudoGeocode = shared ? SHARED : geocode;
150     
151             return Observable.create(new OnSubscribe<BitmapDrawable>() {
152                 @Override
153                 public void call(final Subscriber<? super BitmapDrawable> subscriber) {
154                     subscription.add(subscriber);
155                     subscriber.add(RxUtils.computationScheduler.schedule(new Action1<Inner>() {
156                         @Override
157                         public void call(final Inner inner) {
158                             final Pair<BitmapDrawable, Boolean> loaded = loadFromDisk();
159                             final BitmapDrawable bitmap = loaded.getLeft();
160                             if (loaded.getRight()) {
161                                 subscriber.onNext(bitmap);
162                                 subscriber.onCompleted();
163                                 return;
164                             }
165                             if (bitmap != null && !onlySave) {
166                                 subscriber.onNext(bitmap);
167                             }
168                             subscriber.add(downloadScheduler.schedule(new Action1<Inner>() {
169                                 @Override
170                                 public void call(final Inner inner) {
171                                     downloadAndSave(subscriber);
172                                 }
173                             }));
174                         }
175                     }));
176                 }
177     
178                 private Pair<BitmapDrawable, Boolean> loadFromDisk() {
179                     final Pair<Bitmap, Boolean> loadResult = loadImageFromStorage(url, pseudoGeocode, shared);
180                     final Bitmap bitmap = loadResult.getLeft();
181                     return new ImmutablePair<BitmapDrawable, Boolean>(bitmap != null ?
182                             ImageUtils.scaleBitmapToFitDisplay(bitmap) :
183                             null,
184                             loadResult.getRight()
185                     );
186                 }

188                 private void downloadAndSave(final Subscriber<? super BitmapDrawable> subscriber) {
189                     final File file = LocalStorage.getStorageFile(pseudoGeocode, url, true, true);
190                     if (url.startsWith("data:image/")) {
191                         if (url.contains(";base64,")) {
192                             ImageUtils.decodeBase64ToFile(StringUtils.substringAfter(url, ";base64,"), file);
193                         } else {
194                             Log.e("HtmlImage.getDrawable: unable to decode non-base64 inline image");
195                             subscriber.onCompleted();
196                             return;
197                         }
198                     } else {
199                         if (subscriber.isUnsubscribed() || downloadOrRefreshCopy(url, file)) {
200                             // The existing copy was fresh enough or we were unsubscribed earlier.
201                             subscriber.onCompleted();
202                             return;
203                         }
204                     }
205                     if (onlySave) {
206                         subscriber.onCompleted();
207                     } else {
208                         RxUtils.computationScheduler.schedule(new Action1<Inner>() {
209                             @Override
210                             public void call(final Inner inner) {
211                                 final Pair<BitmapDrawable, Boolean> loaded = loadFromDisk();
212                                 final BitmapDrawable image = loaded.getLeft();
213                                 if (image != null) {
214                                     subscriber.onNext(image);
215                                 } else {
216                                     subscriber.onNext(returnErrorImage ?
217                                             new BitmapDrawable(resources, BitmapFactory.decodeResource(resources, R.drawable.image_not_loaded)) :
218                                             getTransparent1x1Image(resources));
219                                 }
220                                 subscriber.onCompleted();
221                             }
222                         });
223                     }
224                 }
225             });
226         }

236         /**
237          * Download or refresh the copy of <code>url</code> in <code>file</code>.
238          *
239          * @param url the url of the document
240          * @param file the file to save the document in
241          * @return <code>true</code> if the existing file was up-to-date, <code>false</code> otherwise
242          */
243         private boolean downloadOrRefreshCopy(final String url, final File file) {
244             final String absoluteURL = makeAbsoluteURL(url);
245     
246             if (absoluteURL != null) {
247                 try {
248                     final HttpResponse httpResponse = Network.getRequest(absoluteURL, null, file);
249                     if (httpResponse != null) {
250                         final int statusCode = httpResponse.getStatusLine().getStatusCode();
251                         if (statusCode == 200) {
252                             LocalStorage.saveEntityToFile(httpResponse, file);
253                         } else if (statusCode == 304) {
254                             if (!file.setLastModified(System.currentTimeMillis())) {
255                                 makeFreshCopy(file);
256                             }
257                             return true;
258                         }
259                     }
260                 } catch (Exception e) {
261                     Log.e("HtmlImage.downloadOrRefreshCopy", e);
262                 }
263             }
264             return false;
265         }

267         /**
268          * Make a fresh copy of the file to reset its timestamp. On some storage, it is impossible
269          * to modify the modified time after the fact, in which case a brand new file must be
270          * created if we want to be able to use the time as validity hint.
271          *
272          * See Android issue 1699.
273          *
274          * @param file the file to refresh
275          */
276         private static void makeFreshCopy(final File file) {
277             final File tempFile = new File(file.getParentFile(), file.getName() + "-temp");
278             if (file.renameTo(tempFile)) {
279                 LocalStorage.copy(tempFile, file);
280                 FileUtils.deleteIgnoringFailure(tempFile);
281             }
282             else {
283                 Log.e("Could not reset timestamp of file " + file.getAbsolutePath());
284             }
285         }

291         /**
292          * Load an image from primary or secondary storage.
293          *
294          * @param url the image URL
295          * @param pseudoGeocode the geocode or the shared name
296          * @param forceKeep keep the image if it is there, without checking its freshness
297          * @return <code>true</code> if the image was there and is fresh enough, <code>false</code> otherwise
298          */
299         @NonNull
300         private Pair<Bitmap, Boolean> loadImageFromStorage(final String url, final String pseudoGeocode, final boolean forceKeep) {
301             try {
302                 final File file = LocalStorage.getStorageFile(pseudoGeocode, url, true, false);
303                 final Pair<Bitmap, Boolean> image = loadCachedImage(file, forceKeep);
304                 if (image.getRight() || image.getLeft() != null) {
305                     return image;
306                 }
307                 final File fileSec = LocalStorage.getStorageSecFile(pseudoGeocode, url, true);
308                 return loadCachedImage(fileSec, forceKeep);
309             } catch (Exception e) {
310                 Log.w("HtmlImage.loadImageFromStorage", e);
311             }
312             return new ImmutablePair<Bitmap, Boolean>(null, false);
313         }

315         @Nullable
316         private String makeAbsoluteURL(final String url) {
317             // Check if uri is absolute or not, if not attach the connector hostname
318             // FIXME: that should also include the scheme
319             if (Uri.parse(url).isAbsolute()) {
320                 return url;
321             }
322     
323             final String host = ConnectorFactory.getConnector(geocode).getHost();
324             if (StringUtils.isNotEmpty(host)) {
325                 final StringBuilder builder = new StringBuilder("http://");
326                 builder.append(host);
327                 if (!StringUtils.startsWith(url, "/")) {
328                     // FIXME: explain why the result URL would be valid if the path does not start with
329                     // a '/', or signal an error.
330                     builder.append('/');
331                 }
332                 builder.append(url);
333                 return builder.toString();
334             }
335     
336             return null;
337         }

339         /**
340          * Load a previously saved image.
341          *
342          * @param file the file on disk
343          * @param forceKeep keep the image if it is there, without checking its freshness
344          * @return a pair with <code>true</code> if the image was there and is fresh enough or <code>false</code> otherwise,
345          *         and the image (possibly <code>null</code> if the first component is <code>false</code> and the image
346          *         could not be loaded, or if the first component is <code>true</code> and <code>onlySave</code> is also
347          *         <code>true</code>)
348          */
349         @NonNull
350         private Pair<Bitmap, Boolean> loadCachedImage(final File file, final boolean forceKeep) {
351             if (file.exists()) {
352                 final boolean freshEnough = listId >= StoredList.STANDARD_LIST_ID || file.lastModified() > (new Date().getTime() - (24 * 60 * 60 * 1000)) || forceKeep;
353                 if (onlySave) {
354                     return new ImmutablePair<Bitmap, Boolean>(null, true);
355                 }
356                 final BitmapFactory.Options bfOptions = new BitmapFactory.Options();
357                 bfOptions.inTempStorage = new byte[16 * 1024];
358                 bfOptions.inPreferredConfig = Bitmap.Config.RGB_565;
359                 setSampleSize(file, bfOptions);
360                 final Bitmap image = BitmapFactory.decodeFile(file.getPath(), bfOptions);
361                 if (image == null) {
362                     Log.e("Cannot decode bitmap from " + file.getPath());
363                     return new ImmutablePair<Bitmap, Boolean>(null, false);
364                 }
365                 return new ImmutablePair<Bitmap, Boolean>(image,
366                         freshEnough);
367             }
368             return new ImmutablePair<Bitmap, Boolean>(null, false);
369         }

371         private void setSampleSize(final File file, final BitmapFactory.Options bfOptions) {
372             //Decode image size only
373             BitmapFactory.Options options = new BitmapFactory.Options();
374             options.inJustDecodeBounds = true;
375     
376             BufferedInputStream stream = null;
377             try {
378                 stream = new BufferedInputStream(new FileInputStream(file));
379                 BitmapFactory.decodeStream(stream, null, options);
380             } catch (FileNotFoundException e) {
381                 Log.e("HtmlImage.setSampleSize", e);
382             } finally {
383                 IOUtils.closeQuietly(stream);
384             }
385     
386             int scale = 1;
387             if (options.outHeight > maxHeight || options.outWidth > maxWidth) {
388                 scale = Math.max(options.outHeight / maxHeight, options.outWidth / maxWidth);
389             }
390             bfOptions.inSampleSize = scale;
391         }
