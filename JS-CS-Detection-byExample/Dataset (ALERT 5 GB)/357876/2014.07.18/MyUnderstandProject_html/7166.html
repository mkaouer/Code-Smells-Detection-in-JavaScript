288         /**
289          * Fire up a directory chooser on click on the preference.
290          *
291          * @see #onActivityResult() for processing of the selected directory
292          *
293          * @param dct
294          *            type of directory to be selected
295          */
296         private void startDirChooser(final DirChooserType dct) {
297     
298             final String startDirectory = Settings.getString(dct.keyId, dct.defaultValue);
299     
300             try {
301                 final Intent dirChooser = new Intent(FileManagerIntents.ACTION_PICK_DIRECTORY);
302                 if (StringUtils.isNotBlank(startDirectory)) {
303                     dirChooser.setData(Uri.fromFile(new File(startDirectory)));
304                 }
305                 dirChooser.putExtra(FileManagerIntents.EXTRA_TITLE,
306                         getString(R.string.simple_dir_chooser_title));
307                 dirChooser.putExtra(FileManagerIntents.EXTRA_BUTTON_TEXT,
308                         getString(android.R.string.ok));
309                 startActivityForResult(dirChooser, dct.requestCode);
310             } catch (android.content.ActivityNotFoundException ex) {
311                 // OI file manager not available
312                 final Intent dirChooser = new Intent(this, SimpleDirChooser.class);
313                 dirChooser.putExtra(Intents.EXTRA_START_DIR, startDirectory);
314                 dirChooser.putExtra(SimpleDirChooser.EXTRA_CHOOSE_FOR_WRITING, dct.writeMode);
315                 startActivityForResult(dirChooser, dct.requestCode);
316             }
317         }

486         @Override
487         protected void onActivityResult(final int requestCode, final int resultCode, final Intent data) {
488             super.onActivityResult(requestCode, resultCode, data);
489             if (resultCode != RESULT_OK) {
490                 return;
491             }
492     
493             for (DirChooserType dct : DirChooserType.values()) {
494                 if (requestCode == dct.requestCode) {
495                     setChosenDirectory(dct, data);
496                     return;
497                 }
498             }
499     
500             switch (requestCode) {
501                 case R.string.pref_mapDirectory:
502                     if (data.hasExtra(Intents.EXTRA_MAP_FILE)) {
503                         final String mapFile = data.getStringExtra(Intents.EXTRA_MAP_FILE);
504                         File file = new File(mapFile);
505                         if (!file.isDirectory()) {
506                             Settings.setMapFile(mapFile);
507                             if (!Settings.isValidMapFile(Settings.getMapFile())) {
508                                 ActivityMixin.showToast(this, R.string.warn_invalid_mapfile);
509                             } else {
510                                 // Ensure map source preference is updated accordingly.
511                                 // TODO: There should be a better way to find and select the map source for a map file
512                                 Integer mapSourceId = mapFile.hashCode();
513                                 ListPreference mapSource = (ListPreference) getPreference(R.string.pref_mapsource);
514                                 mapSource.setValue(mapSourceId.toString());
515                                 VALUE_CHANGE_LISTENER.onPreferenceChange(mapSource, mapSourceId);
516                             }
517                         } else {
518                             Settings.setMapFileDirectory(mapFile);
519                         }
520                     }
521                     initMapSourcePreference();
522                     getPreference(R.string.pref_mapDirectory).setSummary(StringUtils.defaultString(Settings.getMapFileDirectory()));
523                     break;
524                 case R.string.pref_fakekey_ocde_authorization:
525                 case R.string.pref_fakekey_ocpl_authorization:
526                 case R.string.pref_fakekey_ocnl_authorization:
527                 case R.string.pref_fakekey_ocus_authorization:
528                 case R.string.pref_fakekey_ocro_authorization:
529                     OCPreferenceKeys key = OCPreferenceKeys.getByAuthId(requestCode);
530                     if (key != null) {
531                         setOCAuthTitle(key);
532                         redrawScreen(key.prefScreenId);
533                     }
534                     break;
535                 case R.string.pref_fakekey_twitter_authorization:
536                     setTwitterAuthTitle();
537                     redrawScreen(R.string.preference_screen_twitter);
538                     break;
539                 default:
540                     throw new IllegalArgumentException();
541             }
542         }

544         /**
545          * A preference value change listener that updates the preference's summary
546          * to reflect its new value.
547          */
548         private static final Preference.OnPreferenceChangeListener VALUE_CHANGE_LISTENER = new Preference.OnPreferenceChangeListener() {
549             @Override
550             public boolean onPreferenceChange(final Preference preference, final Object value) {
551                 String stringValue = value.toString();
552     
553                 if (isPreference(preference, R.string.pref_mapsource)) {
554                     // reset the cached map source
555                     MapSource mapSource;
556                     try {
557                         final int mapSourceId = Integer.parseInt(stringValue);
558                         mapSource = MapProviderFactory.getMapSource(mapSourceId);
559                     } catch (final NumberFormatException e) {
560                         Log.e("SettingsActivity.onPreferenceChange: bad source id '" + stringValue + "'");
561                         mapSource = null;
562                     }
563                     // If there is no corresponding map source (because some map sources were
564                     // removed from the device since) then use the first one available.
565                     if (mapSource == null) {
566                         mapSource = MapProviderFactory.getAnyMapSource();
567                         if (mapSource == null) {
568                             // There are no map source. There is little we can do here, except log an error and
569                             // return to avoid triggering a null pointer exception.
570                             Log.e("SettingsActivity.onPreferenceChange: no map source available");
571                             return true;
572                         }
573                     }
574                     Settings.setMapSource(mapSource);
575                     preference.setSummary(mapSource.getName());
576                 } else if (isPreference(preference, R.string.pref_connectorOCActive)
577                         || isPreference(preference, R.string.pref_connectorOCPLActive)
578                         || isPreference(preference, R.string.pref_connectorOCNLActive)
579                         || isPreference(preference, R.string.pref_connectorOCUSActive)
580                         || isPreference(preference, R.string.pref_connectorOCROActive)
581                         || isPreference(preference, R.string.pref_connectorGCActive)
582                         || isPreference(preference, R.string.pref_connectorOXActive)
583                         || isPreference(preference, R.string.pref_connectorECActive)) {
584                     // update summary
585                     boolean boolVal = ((Boolean) value).booleanValue();
586                     String summary = getServiceSummary(boolVal);
587                     if (OCPreferenceKeys.isOCPreference(preference.getKey())) {
588                         OCPreferenceKeys prefKey = OCPreferenceKeys.getByKey(preference.getKey());
589                         preference.getPreferenceManager().findPreference(getKey(prefKey.prefScreenId)).setSummary(summary);
590                     } else if (isPreference(preference, R.string.pref_connectorGCActive)) {
591                         preference.getPreferenceManager().findPreference(getKey(R.string.preference_screen_gc)).setSummary(summary);
592                     } else if (isPreference(preference, R.string.pref_connectorOXActive)) {
593                         preference.getPreferenceManager().findPreference(getKey(R.string.preference_screen_ox)).setSummary(summary);
594                     } else if (isPreference(preference, R.string.pref_connectorECActive)) {
595                         preference.getPreferenceManager().findPreference(getKey(R.string.preference_screen_ec)).setSummary(summary);
596                     }
597                     redrawScreen(preference.getPreferenceManager().findPreference(getKey(R.string.preference_screen_services)));
598                     // reset log-in status if connector activation was changed
599                     CgeoApplication.getInstance().forceRelog();
600                 } else if (preference instanceof ListPreference) {
601                     // For list preferences, look up the correct display value in
602                     // the preference's 'entries' list.
603                     ListPreference listPreference = (ListPreference) preference;
604                     int index = listPreference.findIndexOfValue(stringValue);
605     
606                     // Set the summary to reflect the new value.
607                     preference.setSummary(
608                             index >= 0
609                                     ? listPreference.getEntries()[index]
610                                     : null);
611                 } else if (isPreference(preference, R.string.pref_fakekey_preference_backup_info)) {
612                     final String text;
613                     if (DatabaseBackupUtils.hasBackup()) {
614                         text = preference.getContext().getString(R.string.init_backup_last) + " "
615                                 + DatabaseBackupUtils.getBackupDateTime();
616                     } else {
617                         text = preference.getContext().getString(R.string.init_backup_last_no);
618                     }
619                     preference.setSummary(text);
620                 } else {
621                     // For all other preferences, set the summary to the value's
622                     // simple string representation.
623                     preference.setSummary(stringValue);
624                 }
625                 // TODO: do not special case geocaching.com here
626                 if ((isPreference(preference, R.string.pref_username) && !stringValue.equals(Settings.getUsername())) || (isPreference(preference, R.string.pref_password) && !stringValue.equals(Settings.getGcCredentials().getRight()))) {
627                     // reset log-in if gc user or password is changed
628                     if (GCLogin.getInstance().isActualLoginStatus()) {
629                         GCLogin.getInstance().logout();
630                     }
631                     CgeoApplication.getInstance().forceRelog();
632                 }
633                 return true;
634             }
635         };

637         /**
638          * Binds a preference's summary to its value. More specifically, when the
639          * preference's value is changed, its summary (line of text below the
640          * preference title) is updated to reflect the value. The summary is also
641          * immediately updated upon calling this method. The exact display format is
642          * dependent on the type of preference.
643          *
644          * @see #VALUE_CHANGE_LISTENER
645          */
646         private static void bindSummaryToValue(final Preference preference, final Object value) {
647             // Set the listener to watch for value changes.
648             if (preference == null) {
649                 return;
650             }
651             preference.setOnPreferenceChangeListener(VALUE_CHANGE_LISTENER);
652     
653             // Trigger the listener immediately with the preference's
654             // current value.
655             VALUE_CHANGE_LISTENER.onPreferenceChange(preference, value);
656         }
657     
658         /**
659          * auto-care for the summary of the preference of string type with this key
660          *
661          * @param key
662          */
663         private void bindSummaryToStringValue(final int key) {
664     
665             Preference pref = getPreference(key);
666             if (pref == null) {
667                 return;
668             }
669     
670             String value = PreferenceManager
671                     .getDefaultSharedPreferences(pref.getContext())
672                     .getString(pref.getKey(), "");
673     
674             bindSummaryToValue(pref, value);
675         }
