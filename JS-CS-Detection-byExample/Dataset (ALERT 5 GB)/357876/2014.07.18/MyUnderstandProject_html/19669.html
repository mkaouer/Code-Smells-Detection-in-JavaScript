23     package com.mapswithme.maps.api;
24     
25     import android.app.Activity;
26     import android.app.Dialog;
27     import android.content.Intent;
28     import android.net.Uri;
29     import android.view.View;
30     import android.view.Window;
31     
32     import com.mapwithme.maps.api.R;
33     
34     public class DownloadMapsWithMeDialog extends Dialog implements android.view.View.OnClickListener
35     {
36     
37       public DownloadMapsWithMeDialog(Activity activity)
38       {
39         super(activity);
40     
41         requestWindowFeature(Window.FEATURE_NO_TITLE);
42         setContentView(R.layout.dlg_install_mwm);
43     
44         findViewById(R.id.btn_lite).setOnClickListener(this);
45         findViewById(R.id.btn_pro).setOnClickListener(this);
46     
47         setOwnerActivity(activity);
48       }
49     
50     
51       public void onDownloadButtonClicked(String url)
52       {
53         Intent i = new Intent(Intent.ACTION_VIEW, Uri.parse(url));
54         getContext().startActivity(i);
55         dismiss();
56       }
57     
58     
59       @Override
60       public void onClick(View v)
61       {
62         String url = getContext().getString(v.getId() == R.id.btn_lite ? R.string.url_lite : R.string.url_pro);
63         onDownloadButtonClicked(url);
64       }
65     }
