154         protected void buildDetailsContextMenu(final ContextMenu menu, final CharSequence clickedItemText, final CharSequence fieldTitle, final boolean copyOnly) {
155             menu.setHeaderTitle(fieldTitle);
156             getMenuInflater().inflate(R.menu.details_context, menu);
157             menu.findItem(R.id.menu_translate_to_sys_lang).setVisible(!copyOnly);
158             if (!copyOnly) {
159                 if (clickedItemText.length() > TranslationUtils.TRANSLATION_TEXT_LENGTH_WARN) {
160                     showToast(res.getString(R.string.translate_length_warning));
161                 }
162                 menu.findItem(R.id.menu_translate_to_sys_lang).setTitle(res.getString(R.string.translate_to_sys_lang, Locale.getDefault().getDisplayLanguage()));
163             }
164             final boolean localeIsEnglish = StringUtils.equals(Locale.getDefault().getLanguage(), Locale.ENGLISH.getLanguage());
165             menu.findItem(R.id.menu_translate_to_english).setVisible(!copyOnly && !localeIsEnglish);
166         }

168         protected boolean onClipboardItemSelected(final MenuItem item, final CharSequence clickedItemText) {
169             switch (item.getItemId()) {
170                 // detail fields
171                 case R.id.menu_copy:
172                     ClipboardUtils.copyToClipboard(clickedItemText);
173                     showToast(res.getString(R.string.clipboard_copy_ok));
174                     return true;
175                 case R.id.menu_translate_to_sys_lang:
176                     TranslationUtils.startActivityTranslate(this, Locale.getDefault().getLanguage(), HtmlUtils.extractText(clickedItemText));
177                     return true;
178                 case R.id.menu_translate_to_english:
179                     TranslationUtils.startActivityTranslate(this, Locale.ENGLISH.getLanguage(), HtmlUtils.extractText(clickedItemText));
180                     return true;
181                 case R.id.menu_cache_share_field:
182                     final Intent intent = new Intent(Intent.ACTION_SEND);
183                     intent.setType("text/plain");
184                     intent.putExtra(Intent.EXTRA_TEXT, clickedItemText.toString());
185                     startActivity(Intent.createChooser(intent, res.getText(R.string.cache_share_field)));
186                     return true;
187                 default:
188                     return false;
189             }
190         }
