878         // class: update location
879         private static class UpdateLoc extends GeoDirHandler {
880             // use the following constants for fine tuning - find good compromise between smooth updates and as less updates as possible
881     
882             // minimum time in milliseconds between position overlay updates
883             private static final long MIN_UPDATE_INTERVAL = 500;
884             // minimum change of heading in grad for position overlay update
885             private static final float MIN_HEADING_DELTA = 15f;
886             // minimum change of location in fraction of map width/height (whatever is smaller) for position overlay update
887             private static final float MIN_LOCATION_DELTA = 0.01f;
888     
889             Location currentLocation = new Location("");
890             boolean locationValid = false;
891             float currentHeading;
892     
893             private long timeLastPositionOverlayCalculation = 0;
894             /**
895              * weak reference to the outer class
896              */
897             private final WeakReference<CGeoMap> map;
898     
899             public UpdateLoc(CGeoMap map) {
900                 this.map = new WeakReference<CGeoMap>(map);
901             }
902     
903             @Override
904             public void updateGeoDir(final IGeoData geo, final float dir) {
905                 if (geo.isPseudoLocation()) {
906                     locationValid = false;
907                 } else {
908                     locationValid = true;
909     
910                     currentLocation = geo.getLocation();
911                     currentHeading = DirectionProvider.getDirectionNow(dir);
912                     repaintPositionOverlay();
913                 }
914             }
915     
916             /**
917              * Repaint position overlay but only with a max frequency and if position or heading changes sufficiently.
918              */
919             void repaintPositionOverlay() {
920                 final long currentTimeMillis = System.currentTimeMillis();
921                 if (currentTimeMillis > timeLastPositionOverlayCalculation + MIN_UPDATE_INTERVAL) {
922                     timeLastPositionOverlayCalculation = currentTimeMillis;
923     
924                     try {
925                         CGeoMap cgeoMapRef = map.get();
926                         if (cgeoMapRef != null) {
927                             if (cgeoMapRef.mapView != null) {
928                                 if (cgeoMapRef.overlayPositionAndScale == null) {
929                                     cgeoMapRef.overlayPositionAndScale = cgeoMapRef.mapView.createAddPositionAndScaleOverlay(cgeoMapRef.activity);
930                                 }
931     
932                                 boolean needsRepaintForDistance = needsRepaintForDistance();
933                                 boolean needsRepaintForHeading = needsRepaintForHeading();
934     
935                                 if (needsRepaintForDistance) {
936                                     if (cgeoMapRef.followMyLocation) {
937                                         cgeoMapRef.centerMap(new Geopoint(currentLocation));
938                                     }
939                                 }
940     
941                                 if (needsRepaintForDistance || needsRepaintForHeading) {
942                                     cgeoMapRef.overlayPositionAndScale.setCoordinates(currentLocation);
943                                     cgeoMapRef.overlayPositionAndScale.setHeading(currentHeading);
944                                     cgeoMapRef.mapView.repaintRequired(cgeoMapRef.overlayPositionAndScale);
945                                 }
946                             }
947                         }
948                     } catch (RuntimeException e) {
949                         Log.w("Failed to update location.");
950                     }
951                 }
952             }
953     
954             boolean needsRepaintForHeading() {
955                 final CGeoMap cgeoMapRef = map.get();
956                 if (cgeoMapRef == null) {
957                     return false;
958                 }
959                 return Math.abs(AngleUtils.difference(currentHeading, cgeoMapRef.overlayPositionAndScale.getHeading())) > MIN_HEADING_DELTA;
960             }
961     
962             boolean needsRepaintForDistance() {
963                 if (!locationValid) {
964                     return false;
965                 }
966     
967                 final CGeoMap cgeoMapRef = map.get();
968                 if (cgeoMapRef == null) {
969                     return false;
970                 }
971                 final Location lastLocation = cgeoMapRef.overlayPositionAndScale.getCoordinates();
972     
973                 float dist = Float.MAX_VALUE;
974                 if (lastLocation != null) {
975                     dist = currentLocation.distanceTo(lastLocation);
976                 }
977     
978                 final float[] mapDimension = new float[1];
979                 if (cgeoMapRef.mapView.getWidth() < cgeoMapRef.mapView.getHeight()) {
980                     final double span = cgeoMapRef.mapView.getLongitudeSpan() / 1e6;
981                     Location.distanceBetween(currentLocation.getLatitude(), currentLocation.getLongitude(), currentLocation.getLatitude(), currentLocation.getLongitude() + span, mapDimension);
982                 } else {
983                     final double span = cgeoMapRef.mapView.getLatitudeSpan() / 1e6;
984                     Location.distanceBetween(currentLocation.getLatitude(), currentLocation.getLongitude(), currentLocation.getLatitude() + span, currentLocation.getLongitude(), mapDimension);
985                 }
986     
987                 return dist > (mapDimension[0] * MIN_LOCATION_DELTA);
988             }
989         }

1006         /**
1007          * loading timer Triggers every 250ms and checks for viewport change and starts a {@link LoadRunnable}.
1008          */
1009         private Subscription startLoadTimer() {
1010             return Schedulers.newThread().schedulePeriodically(new Action1<Scheduler.Inner>() {
1011                 @Override
1012                 public void call(Scheduler.Inner inner) {
1013                     try {
1014                         if (mapView != null) {
1015                             // get current viewport
1016                             final Viewport viewportNow = mapView.getViewport();
1017                             // Since zoomNow is used only for local comparison purposes,
1018                             // it is ok to use the Google Maps compatible zoom level of OSM Maps
1019                             final int zoomNow = mapView.getMapZoomLevel();
1020     
1021                             // check if map moved or zoomed
1022                             //TODO Portree Use Rectangle inside with bigger search window. That will stop reloading on every move
1023                             final boolean moved = markersInvalidated || (isLiveEnabled && !downloaded) || (viewport == null) || zoomNow != zoom ||
1024                                     (mapMoved(viewport, viewportNow) && (cachesCnt <= 0 || CollectionUtils.isEmpty(caches) || !viewport.includes(viewportNow)));
1025     
1026                             // update title on any change
1027                             if (moved || !viewportNow.equals(viewport)) {
1028                                 displayHandler.sendEmptyMessage(UPDATE_TITLE);
1029                             }
1030                             zoom = zoomNow;
1031     
1032                             // save new values
1033                             if (moved) {
1034                                 markersInvalidated = false;
1035     
1036                                 long currentTime = System.currentTimeMillis();
1037     
1038                                 if (1000 < (currentTime - loadThreadRun)) {
1039                                     viewport = viewportNow;
1040                                     loadExecutor.execute(new LoadRunnable(viewport));
1041                                 }
1042                             }
1043                         }
1044     
1045                     } catch (Exception e) {
1046                         Log.w("CGeoMap.startLoadtimer.start", e);
1047                     }
1048                 }
1049             }, 250, 250, TimeUnit.MILLISECONDS);
1050         }

1065          * Worker thread that loads caches and waypoints from the database and then spawns the {@link DownloadRunnable}.
1066          * started by {@link LoadTimer}
1067          */

1069         private class LoadRunnable extends DoRunnable {
1070     
1071             public LoadRunnable(final Viewport viewport) {
1072                 super(viewport);
1073             }
1074     
1075             @Override
1076             public void run() {
1077                 try {
1078                     showProgressHandler.sendEmptyMessage(SHOW_PROGRESS);
1079                     loadThreadRun = System.currentTimeMillis();
1080     
1081                     SearchResult searchResult;
1082                     if (mapMode == MapMode.LIVE) {
1083                         searchResult = isLiveEnabled ? new SearchResult() : new SearchResult(DataStore.loadStoredInViewport(viewport, Settings.getCacheType()));
1084                     } else {
1085                         // map started from another activity
1086                         searchResult = searchIntent != null ? new SearchResult(searchIntent) : new SearchResult();
1087                         if (geocodeIntent != null) {
1088                             searchResult.addGeocode(geocodeIntent);
1089                         }
1090                     }
1091                     // live mode search result
1092                     if (isLiveEnabled) {
1093                         searchResult.addSearchResult(DataStore.loadCachedInViewport(viewport, Settings.getCacheType()));
1094                     }
1095     
1096                     downloaded = true;
1097                     Set<Geocache> cachesFromSearchResult = searchResult.getCachesFromSearchResult(LoadFlags.LOAD_WAYPOINTS);
1098                     // update the caches
1099                     // new collection type needs to remove first
1100                     caches.removeAll(cachesFromSearchResult);
1101                     caches.addAll(cachesFromSearchResult);
1102     
1103                     final boolean excludeMine = Settings.isExcludeMyCaches();
1104                     final boolean excludeDisabled = Settings.isExcludeDisabledCaches();
1105                     if (mapMode == MapMode.LIVE) {
1106                         CGeoMap.filter(caches);
1107                     }
1108                     countVisibleCaches();
1109                     if (cachesCnt < Settings.getWayPointsThreshold() || geocodeIntent != null) {
1110                         // we don't want to see any stale waypoints
1111                         waypoints.clear();
1112                         if (isLiveEnabled || mapMode == MapMode.LIVE
1113                                 || mapMode == MapMode.COORDS) {
1114                             //All visible waypoints
1115                             CacheType type = Settings.getCacheType();
1116                             Set<Waypoint> waypointsInViewport = DataStore.loadWaypoints(viewport, excludeMine, excludeDisabled, type);
1117                             waypoints.addAll(waypointsInViewport);
1118                         }
1119                         else {
1120                             //All waypoints from the viewed caches
1121                             for (Geocache c : caches.getAsList()) {
1122                                 waypoints.addAll(c.getWaypoints());
1123                             }
1124                         }
1125                     }
1126                     else {
1127                         // we don't want to see any stale waypoints when above threshold
1128                         waypoints.clear();
1129                     }
1130     
1131                     //render
1132                     displayExecutor.execute(new DisplayRunnable(viewport));
1133     
1134                     if (isLiveEnabled) {
1135                         downloadExecutor.execute(new DownloadRunnable(viewport));
1136                     }
1137                     lastSearchResult = searchResult;
1138                 } finally {
1139                     showProgressHandler.sendEmptyMessage(HIDE_PROGRESS); // hide progress
1140                 }
1141             }
1142         }

1155             @Override
1156             public void run() {

1158                     showProgressHandler.sendEmptyMessage(SHOW_PROGRESS); // show progress

1160                         if (tokens == null) {
1161                             tokens = GCLogin.getInstance().getMapTokens();
1162                             if (noMapTokenHandler != null && (StringUtils.isEmpty(tokens.getUserSession()) || StringUtils.isEmpty(tokens.getSessionToken()))) {
1163                                 tokens = null;

1177                     caches.removeAll(DataStore.loadCaches(filteredCodes, LoadFlags.LOAD_CACHE_ONLY));
1178                     DataStore.removeCaches(filteredCodes, EnumSet.of(RemoveFlag.REMOVE_CACHE));

1181                     caches.addAll(result);

1184                     //render
1185                     displayExecutor.execute(new DisplayRunnable(viewport));
1186     
1187                 } catch (ThreadDeath e) {

1192                 }

1325         private class LoadDetails extends Thread {
1326     
1327             final private CancellableHandler handler;
1328             final private List<String> geocodes;
1329             final private int listId;
1330     
1331             public LoadDetails(final CancellableHandler handler, final List<String> geocodes, final int listId) {
1332                 this.handler = handler;
1333                 this.geocodes = geocodes;
1334                 this.listId = listId;
1335             }
1336     
1337             public void stopIt() {
1338                 handler.cancel();
1339             }
1340     
1341             @Override
1342             public void run() {
1343                 if (CollectionUtils.isEmpty(geocodes)) {
1344                     return;
1345                 }
1346     
1347                 for (final String geocode : geocodes) {
1348                     try {
1349                         if (handler.isCancelled()) {
1350                             break;
1351                         }
1352     
1353                         if (!DataStore.isOffline(geocode, null)) {
1354                             Geocache.storeCache(null, geocode, listId, false, handler);
1355                         }
1356                     } catch (Exception e) {
1357                         Log.e("CGeoMap.LoadDetails.run", e);
1358                     } finally {
1359                         // one more cache over
1360                         detailProgress++;
1361                         handler.sendEmptyMessage(UPDATE_PROGRESS);
1362                     }
1363                 }
1364     
1365                 // we're done
1366                 handler.sendEmptyMessage(FINISHED_LOADING_DETAILS);
1367             }
1368         }

1581         private LayerDrawable createCacheItem(final Geocache cache, final int hashcode) {
1582             // Set initial capacities to the maximum of layers and insets to avoid dynamic reallocation
1583             final ArrayList<Drawable> layers = new ArrayList<Drawable>(9);
1584             final ArrayList<int[]> insets = new ArrayList<int[]>(8);
1585     
1586             // background: disabled or not
1587             final Drawable marker = getResources().getDrawable(cache.getMapMarkerId());
1588             layers.add(marker);
1589             final int resolution = marker.getIntrinsicWidth() > 40 ? (marker.getIntrinsicWidth() > 50 ? 2 : 1) : 0;
1590             // reliable or not
1591             if (!cache.isReliableLatLon()) {
1592                 insets.add(INSET_RELIABLE[resolution]);
1593                 layers.add(getResources().getDrawable(R.drawable.marker_notreliable));
1594             }
1595             // cache type
1596             layers.add(getResources().getDrawable(cache.getType().markerId));
1597             insets.add(INSET_TYPE[resolution]);
1598             // own
1599             if (cache.isOwner()) {
1600                 layers.add(getResources().getDrawable(R.drawable.marker_own));
1601                 insets.add(INSET_OWN[resolution]);
1602                 // if not, checked if stored
1603             } else if (cache.getListId() > 0) {
1604                 layers.add(getResources().getDrawable(R.drawable.marker_stored));
1605                 insets.add(INSET_OWN[resolution]);
1606             }
1607             // found
1608             if (cache.isFound()) {
1609                 layers.add(getResources().getDrawable(R.drawable.marker_found));
1610                 insets.add(INSET_FOUND[resolution]);
1611                 // if not, perhaps logged offline
1612             } else if (cache.isLogOffline()) {
1613                 layers.add(getResources().getDrawable(R.drawable.marker_found_offline));
1614                 insets.add(INSET_FOUND[resolution]);
1615             }
1616             // user modified coords
1617             if (cache.hasUserModifiedCoords()) {
1618                 layers.add(getResources().getDrawable(R.drawable.marker_usermodifiedcoords));
1619                 insets.add(INSET_USERMODIFIEDCOORDS[resolution]);
1620             }
1621             // personal note
1622             if (cache.getPersonalNote() != null) {
1623                 layers.add(getResources().getDrawable(R.drawable.marker_personalnote));
1624                 insets.add(INSET_PERSONALNOTE[resolution]);
1625             }
1626     
1627             final LayerDrawable ld = new LayerDrawable(layers.toArray(new Drawable[layers.size()]));
1628     
1629             int index = 1;
1630             for (final int[] inset : insets) {
1631                 ld.setLayerInset(index++, inset[0], inset[1], inset[2], inset[3]);
1632             }
1633     
1634             overlaysCache.put(hashcode, ld);
1635             return ld;
1636         }
