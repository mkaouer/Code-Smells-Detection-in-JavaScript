38         @Override
39         public View getView(View convertView, ViewGroup parent) {
40             setOnPreferenceClickListener(new ClickListener());
41             return super.getView(convertView, parent);
42         }
43     
44         private final class ClickListener implements OnPreferenceClickListener {
45             @Override
46             public boolean onPreferenceClick(final Preference preference) {
47                 WebView htmlView = new WebView(preference.getContext());
48                 htmlView.loadDataWithBaseURL(null, createCapabilitiesMessage(), "text/html", "utf-8", null);
49                 AlertDialog.Builder builder = new AlertDialog.Builder(preference.getContext());
50                 builder.setView(htmlView)
51                         .setIcon(android.R.drawable.ic_dialog_info)
52                         .setTitle(R.string.settings_features)
53                         .setPositiveButton(R.string.err_none, new DialogInterface.OnClickListener() {
54                             @Override
55                             public void onClick(DialogInterface dialog, int id) {
56                                 dialog.cancel();
57                             }
58                         });
59                 builder.create().show();
60                 return false;
61             }
62         }
63     
64         public String createCapabilitiesMessage() {
65             // TODO: this needs a better key for the connectors
66             IConnector connector = ConnectorFactory.getConnector(connectorCode + "1234");
67             StringBuilder builder = new StringBuilder("<p>"
68                     + TextUtils.htmlEncode(CgeoApplication.getInstance().getString(R.string.feature_description)) + "</p><ul>");
69     
70             for (String capability : connector.getCapabilities()) {
71                 builder.append("<li>").append(TextUtils.htmlEncode(capability)).append("</li>");
72             }
73     
74             builder.append("</ul>");
75             return builder.toString();
76         }
77     
78         @Override
79         protected void processAttributeValues(TypedArray values) {
80             connectorCode = values.getString(0);
81         }
82     
83         @Override
84         protected int[] getAttributeNames() {
85             return new int[] { R.attr.connector };
86         }
