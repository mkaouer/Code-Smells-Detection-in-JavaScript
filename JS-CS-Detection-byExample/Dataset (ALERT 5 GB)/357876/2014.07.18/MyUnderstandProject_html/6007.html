94         /**
95          * Log the whole content of a string into "/sdcard/cgeo-debug.log".
96          * <br/>
97          * Sometimes, the string we want to work on while debugging or developing a new feature is too long to
98          * be fully stored in Android logcat. This method will log the content of the string in a file named
99          * "/sdcard/cgeo-debug.log". The file will be reset at every run, and if called several times during a run,
100          * the contents will be appended to one another.
101          * <br/>
102          * <strong>This method should never be called in production.</strong>
103          *
104          * @param msg the message to log, or to add to the log if other messages have been stored in the same run
105          */
106         public static synchronized void logToFile(final String msg) {
107             final File file = new File(Environment.getExternalStorageDirectory(), "cgeo-debug.log");
108             if (first) {
109                 first = false;
110                 FileUtils.delete(file);
111             }
112             Writer writer = null;
113             try {
114                 writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(file, true), CharEncoding.UTF_8));
115                 writer.write(msg);
116             } catch (final IOException e) {
117                 Log.e("logToFile: cannot write to " + file, e);
118             } finally {
119                 IOUtils.closeQuietly(writer);
120             }
121         }
