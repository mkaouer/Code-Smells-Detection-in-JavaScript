91         private static String[] getGeocodes(final List<Geocache> caches) {
92             final ArrayList<String> allGeocodes = new ArrayList<String>(caches.size());
93             for (final Geocache geocache : caches) {
94                 allGeocodes.add(geocache.getGeocode());

96             return allGeocodes.toArray(new String[allGeocodes.size()]);
97         }

99         protected class ExportTask extends AsyncTaskWithProgress<String, File> {
100             private final Activity activity;
101     
102             /**
103              * Instantiates and configures the task for exporting field notes.
104              *
105              * @param activity
106              *            optional: Show a progress bar and toasts
107              */
108             public ExportTask(final Activity activity) {
109                 super(activity, getProgressTitle());
110                 this.activity = activity;
111             }
112     
113             private File getExportFile() {
114                 final SimpleDateFormat fileNameDateFormat = new SimpleDateFormat("yyyy-MM-dd", Locale.US);
115                 final Date now = new Date();
116                 return FileUtils.getUniqueNamedFile(Settings.getGpxExportDir() + File.separatorChar + "export_" + fileNameDateFormat.format(now) + ".gpx");
117             }
118     
119             @Override
120             protected File doInBackgroundInternal(String[] geocodes) {
121                 // quick check for being able to write the GPX file
122                 if (!Environment.getExternalStorageState().equals(Environment.MEDIA_MOUNTED)) {
123                     return null;
124                 }
125     
126                 final List<String> allGeocodes = new ArrayList<String>(Arrays.asList(geocodes));
127     
128                 setMessage(CgeoApplication.getInstance().getResources().getQuantityString(R.plurals.cache_counts, allGeocodes.size(), allGeocodes.size()));
129     
130                 final File exportFile = getExportFile();
131                 BufferedWriter writer = null;
132                 try {
133                     final File exportLocation = new File(Settings.getGpxExportDir());
134                     FileUtils.mkdirs(exportLocation);
135     
136                     writer = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(exportFile), CharEncoding.UTF_8));
137                     new GpxSerializer().writeGPX(allGeocodes, writer, new GpxSerializer.ProgressListener() {
138     
139                         @Override
140                         public void publishProgress(final int countExported) {
141                             ExportTask.this.publishProgress(countExported);
142                         }
143                     });
144                 } catch (final IOException e) {
145                     Log.e("GpxExport.ExportTask export", e);
146     
147                     if (writer != null) {
148                         try {
149                             writer.close();
150                         } catch (final IOException e1) {
151                             // Ignore double error
152                         }
153                     }
154                     // delete partial GPX file on error
155                     if (exportFile.exists()) {
156                         FileUtils.deleteIgnoringFailure(exportFile);
157                     }
158     
159                     return null;
160                 }
161     
162                 return exportFile;
163             }
164     
165             @Override
166             protected void onPostExecuteInternal(final File exportFile) {
167                 if (null != activity) {
168                     if (exportFile != null) {
169                         ActivityMixin.showToast(activity, getName() + ' ' + getString(R.string.export_exportedto) + ": " + exportFile.toString());
170                         if (Settings.getShareAfterExport()) {
171                             final Intent shareIntent = new Intent();
172                             shareIntent.setAction(Intent.ACTION_SEND);
173                             shareIntent.putExtra(Intent.EXTRA_STREAM, Uri.fromFile(exportFile));
174                             shareIntent.setType("application/xml");
175                             activity.startActivity(Intent.createChooser(shareIntent, getString(R.string.export_gpx_to)));
176                         }
177                     } else {
178                         ActivityMixin.showToast(activity, getString(R.string.export_failed));
179                     }
180                 }
181             }
