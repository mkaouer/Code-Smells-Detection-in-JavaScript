504         private class Poster extends AsyncTaskWithProgress<String, StatusCode> {
505     
506             public Poster(final Activity activity, final String progressMessage) {
507                 super(activity, null, progressMessage, true);
508             }
509     
510             @Override
511             protected StatusCode doInBackgroundInternal(final String[] logTexts) {
512                 final String log = logTexts[0];
513                 final String logPwd = logTexts.length > 1 ? logTexts[1] : null;
514                 try {
515                     final LogResult logResult = loggingManager.postLog(cache, typeSelected, date, log, logPwd, trackables);
516     
517                     if (logResult.getPostLogResult() == StatusCode.NO_ERROR) {
518                         // update geocache in DB
519                         if (typeSelected == LogType.FOUND_IT || typeSelected == LogType.ATTENDED) {
520                             cache.setFound(true);
521                             cache.setVisitedDate(new Date().getTime());
522                         }
523                         DataStore.saveChangedCache(cache);
524     
525                         // update logs in DB
526                         ArrayList<LogEntry> newLogs = new ArrayList<LogEntry>(cache.getLogs());
527                         final LogEntry logNow = new LogEntry(date.getTimeInMillis(), typeSelected, log);
528                         newLogs.add(0, logNow);
529                         DataStore.saveLogsWithoutTransaction(cache.getGeocode(), newLogs);
530     
531                         // update offline log in DB
532                         cache.clearOfflineLog();
533     
534                         if (typeSelected == LogType.FOUND_IT) {
535                             if (tweetCheck.isChecked() && tweetBox.getVisibility() == View.VISIBLE) {
536                                 Twitter.postTweetCache(geocode, logNow);
537                             }
538                         }
539                         if (GCVote.isValidRating(rating)) {
540                             GCVote.setRating(cache, rating);
541                         }
542     
543                         if (StringUtils.isNotBlank(imageUri.getPath())) {
544                             ImageResult imageResult = loggingManager.postLogImage(logResult.getLogId(), imageCaption, imageDescription, imageUri);
545                             final String uploadedImageUrl = imageResult.getImageUri();
546                             if (StringUtils.isNotEmpty(uploadedImageUrl)) {
547                                 logNow.addLogImage(new Image(uploadedImageUrl, imageCaption, imageDescription));
548                                 DataStore.saveLogsWithoutTransaction(cache.getGeocode(), newLogs);
549                             }
550                             return imageResult.getPostResult();
551                         }
552                     }
553     
554                     return logResult.getPostLogResult();
555                 } catch (RuntimeException e) {
556                     Log.e("VisitCacheActivity.Poster.doInBackgroundInternal", e);
557                 }
558     
559                 return StatusCode.LOG_POST_ERROR;
560             }
561     
562             @Override
563             protected void onPostExecuteInternal(final StatusCode status) {
564                 if (status == StatusCode.NO_ERROR) {
565                     showToast(res.getString(R.string.info_log_posted));
566                     // No need to save the log when quitting if it has been posted.
567                     text = currentLogText();
568                     finish();
569                 } else if (status == StatusCode.LOG_SAVED) {
570                     showToast(res.getString(R.string.info_log_saved));
571                     finish();
572                 } else {
573                     showToast(status.getErrorString(res));
574                 }
575             }
576         }

578         private void saveLog(final boolean force) {
579             final String log = currentLogText();
580     
581             // Do not erase the saved log if the user has removed all the characters
582             // without using "Clear". This may be a manipulation mistake, and erasing
583             // again will be easy using "Clear" while retyping the text may not be.
584             if (force || (StringUtils.isNotEmpty(log) && !StringUtils.equals(log, text))) {
585                 cache.logOffline(this, log, date, typeSelected);
586             }
587             text = log;
588         }

673         private void selectImage() {
674             Intent selectImageIntent = new Intent(this, ImageSelectActivity.class);
675             selectImageIntent.putExtra(ImageSelectActivity.EXTRAS_CAPTION, imageCaption);
676             selectImageIntent.putExtra(ImageSelectActivity.EXTRAS_DESCRIPTION, imageDescription);
677             selectImageIntent.putExtra(ImageSelectActivity.EXTRAS_URI_AS_STRING, imageUri.toString());
678     
679             startActivityForResult(selectImageIntent, SELECT_IMAGE);
680         }
681     
682         @Override
683         protected void onActivityResult(int requestCode, int resultCode, Intent data) {
684             if (requestCode == SELECT_IMAGE) {
685                 if (resultCode == RESULT_OK) {
686                     imageCaption = data.getStringExtra(ImageSelectActivity.EXTRAS_CAPTION);
687                     imageDescription = data.getStringExtra(ImageSelectActivity.EXTRAS_DESCRIPTION);
688                     imageUri = Uri.parse(data.getStringExtra(ImageSelectActivity.EXTRAS_URI_AS_STRING));
689                 } else if (resultCode != RESULT_CANCELED) {
690                     // Image capture failed, advise user
691                     showToast(getResources().getString(R.string.err_select_logimage_failed));
692                 }
693                 updateImageButton();
694     
695             }
696         }
697     
698         private void updateImageButton() {
699             final Button imageButton = (Button) findViewById(R.id.image_btn);
700             if (cache.supportsLogImages()) {
701                 imageButton.setVisibility(View.VISIBLE);
702                 imageButton.setText(StringUtils.isNotBlank(imageUri.getPath()) ?
703                     res.getString(R.string.log_image_edit) : res.getString(R.string.log_image_attach));
704             } else {
705                 imageButton.setVisibility(View.GONE);
706             }
707         }
