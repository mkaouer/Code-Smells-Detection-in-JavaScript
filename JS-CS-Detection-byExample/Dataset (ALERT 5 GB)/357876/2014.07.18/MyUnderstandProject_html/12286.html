62          * Reads and scales an image file to the device display size.
63          *
64          * @param filename
65          *            The image file to read and scale
66          * @return Bitmap The scaled image or Null if source image can't be read
67          */
68         @Nullable
69         public static Bitmap readAndScaleImageToFitDisplay(@NonNull final String filename) {
70             Point displaySize = Compatibility.getDisplaySize();
71             // Restrict image size to 800 x 800 to prevent OOM on tablets
72             final int maxWidth = Math.min(displaySize.x - 25, MAX_DISPLAY_IMAGE_XY);
73             final int maxHeight = Math.min(displaySize.y - 25, MAX_DISPLAY_IMAGE_XY);
74             final Bitmap image = readDownsampledImage(filename, maxWidth, maxHeight);
75             if (image == null) {
76                 return null;
77             }
78             final BitmapDrawable scaledImage = scaleBitmapTo(image, maxWidth, maxHeight);
79             return scaledImage.getBitmap();
80         }

82         /**
83          * Scales a bitmap to the given bounds if it is larger, otherwise returns the original bitmap.
84          *
85          * @param image
86          *            The bitmap to scale
87          * @return BitmapDrawable The scaled image
88          */
89         @NonNull
90         private static BitmapDrawable scaleBitmapTo(@NonNull final Bitmap image, final int maxWidth, final int maxHeight) {
91             final CgeoApplication app = CgeoApplication.getInstance();
92             Bitmap result = image;
93             int width = image.getWidth();
94             int height = image.getHeight();
95     
96             if (width > maxWidth || height > maxHeight) {
97                 final double ratio = Math.min((double) maxHeight / (double) height, (double) maxWidth / (double) width);
98                 width = (int) Math.ceil(width * ratio);
99                 height = (int) Math.ceil(height * ratio);
100                 result = Bitmap.createScaledBitmap(image, width, height, true);
101             }
102     
103             final BitmapDrawable resultDrawable = new BitmapDrawable(app.getResources(), result);
104             resultDrawable.setBounds(new Rect(0, 0, width, height));
105             return resultDrawable;
106         }

108         /**
109          * Store a bitmap to file.
110          *
111          * @param bitmap
112          *            The bitmap to store
113          * @param format
114          *            The image format
115          * @param quality
116          *            The image quality
117          * @param pathOfOutputImage
118          *            Path to store to
119          */
120         public static void storeBitmap(final Bitmap bitmap, final Bitmap.CompressFormat format, final int quality, final String pathOfOutputImage) {
121             try {
122                 FileOutputStream out = new FileOutputStream(pathOfOutputImage);
123                 BufferedOutputStream bos = new BufferedOutputStream(out);
124                 bitmap.compress(format, quality, bos);
125                 bos.flush();
126                 bos.close();
127             } catch (IOException e) {
128                 Log.e("ImageHelper.storeBitmap", e);
129             }
130         }

132         /**
133          * Scales an image to the desired bounds and encodes to file.
134          *
135          * @param filePath
136          *            Image to read
137          * @param maxXY
138          *            bounds
139          * @return filename and path, <tt>null</tt> if something fails
140          */
141         @Nullable
142         public static String readScaleAndWriteImage(@NonNull final String filePath, final int maxXY) {
143             if (maxXY <= 0) {
144                 return filePath;
145             }
146             Bitmap image = readDownsampledImage(filePath, maxXY, maxXY);
147             if (image == null) {
148                 return null;
149             }
150             final BitmapDrawable scaledImage = scaleBitmapTo(image, maxXY, maxXY);
151             final File tempImageFile = ImageUtils.getOutputImageFile();
152             if (tempImageFile == null) {
153                 Log.e("ImageUtils.readScaleAndWriteImage: unable to write scaled image");
154                 return null;
155             }
156             final String uploadFilename = tempImageFile.getPath();
157             storeBitmap(scaledImage.getBitmap(), Bitmap.CompressFormat.JPEG, 75, uploadFilename);
158             return uploadFilename;
159         }

161         /**
162          * Reads and scales an image file with downsampling in one step to prevent memory consumption.
163          *
164          * @param filePath
165          *            The file to read
166          * @param maxX
167          *            The desired width
168          * @param maxY
169          *            The desired height
170          * @return Bitmap the image or null if file can't be read
171          */
172         @Nullable
173         public static Bitmap readDownsampledImage(@NonNull final String filePath, final int maxX, final int maxY) {
174             int orientation = ExifInterface.ORIENTATION_NORMAL;
175             try {
176                 final ExifInterface exif = new ExifInterface(filePath);
177                 orientation = exif.getAttributeInt(ExifInterface.TAG_ORIENTATION, ExifInterface.ORIENTATION_NORMAL);
178             } catch (IOException e) {
179                 Log.e("ImageUtils.readDownsampledImage", e);
180             }
181             final BitmapFactory.Options sizeOnlyOptions = new BitmapFactory.Options();
182             sizeOnlyOptions.inJustDecodeBounds = true;
183             BitmapFactory.decodeFile(filePath, sizeOnlyOptions);
184             final int myMaxXY = Math.max(sizeOnlyOptions.outHeight, sizeOnlyOptions.outWidth);
185             final int maxXY = Math.max(maxX, maxY);
186             final int sampleSize = myMaxXY / maxXY;
187             final BitmapFactory.Options sampleOptions = new BitmapFactory.Options();
188             if (sampleSize > 1) {
189                 sampleOptions.inSampleSize = sampleSize;
190             }
191             final Bitmap decodedImage = BitmapFactory.decodeFile(filePath, sampleOptions);
192             if (decodedImage != null) {
193                 for (int i = 0; i < ORIENTATIONS.length; i++) {
194                     if (orientation == ORIENTATIONS[i]) {
195                         final Matrix matrix = new Matrix();
196                         matrix.postRotate(ROTATION[i]);
197                         return Bitmap.createBitmap(decodedImage, 0, 0, decodedImage.getWidth(), decodedImage.getHeight(), matrix, true);
198                     }
199                 }
200             }
201             return decodedImage;
202         }
