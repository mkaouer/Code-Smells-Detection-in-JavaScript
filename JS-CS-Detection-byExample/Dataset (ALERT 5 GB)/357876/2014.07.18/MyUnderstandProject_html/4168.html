23     package com.mapswithme.maps.api;
24     
25     import android.content.Context;
26     import android.content.Intent;
27     
28     public class MWMResponse
29     {
30       private MWMPoint mPoint;
31       private double   mZoomLevel;
32     
33       /**
34        *
35        * @return point, for which user requested more information in MapsWithMe application.
36        */
37       public MWMPoint getPoint()     { return mPoint; }
38       public boolean  hasPoint()     { return mPoint != null; }
39       public double   getZoomLevel() { return mZoomLevel; }
40     
41       @Override
42       public String toString()
43       {
44         return "MWMResponse [SelectedPoint=" + mPoint + "]";
45       }
46     
47       /**
48        * Factory method to extract response data from intent.
49        *
50        * @param context
51        * @param intent
52        * @return
53        */
54       public static MWMResponse extractFromIntent(Context context, Intent intent)
55       {
56         final MWMResponse response = new MWMResponse();
57         // parse point
58         final double lat = intent.getDoubleExtra(Const.EXTRA_MWM_RESPONSE_POINT_LAT, INVALID_LL);
59         final double lon = intent.getDoubleExtra(Const.EXTRA_MWM_RESPONSE_POINT_LON, INVALID_LL);
60         final String name = intent.getStringExtra(Const.EXTRA_MWM_RESPONSE_POINT_NAME);
61         final String id = intent.getStringExtra(Const.EXTRA_MWM_RESPONSE_POINT_ID);
62     
63         // parse additional info
64         response.mZoomLevel = intent.getDoubleExtra(Const.EXTRA_MWM_RESPONSE_ZOOM, 9);
65     
66         if (lat != INVALID_LL && lon != INVALID_LL)
67           response.mPoint = new MWMPoint(lat, lon, name, id);
68         else
69           response.mPoint = null;
70     
71         return response;
72       }
73     
74       private final static double INVALID_LL =  Double.MIN_VALUE;
75     
76       private MWMResponse() {}
77     }
