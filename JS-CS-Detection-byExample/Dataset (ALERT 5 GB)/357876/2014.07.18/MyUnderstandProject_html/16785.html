68         private void assertWaypointsParsed(final String note, final int expectedWaypoints) {
69     
70             recordMapStoreFlags();
71     
72             try {
73                 setMapStoreFlags(false, false);
74     
75                 Geocache cache = new Geocache();
76                 final String geocode = "Test" + System.nanoTime();
77                 cache.setGeocode(geocode);
78                 cache.setWaypoints(new ArrayList<Waypoint>(), false);
79                 for (int i = 0; i < 2; i++) {
80                     cache.setPersonalNote(note);
81                     cache.parseWaypointsFromNote();
82                     final List<Waypoint> waypoints = cache.getWaypoints();
83                     assertThat(waypoints).isNotNull();
84                     assertThat(waypoints).hasSize(expectedWaypoints);
85                     final Waypoint waypoint = waypoints.get(0);
86                     assertThat(waypoint.getCoords()).isEqualTo(new Geopoint("N51 13.888 E007 03.444"));
87                     //            assertThat(waypoint.getNote()).isEqualTo("Test");
88                     assertThat(waypoint.getName()).isEqualTo(CgeoApplication.getInstance().getString(R.string.cache_personal_note) + " 1");
89                     cache.store(StoredList.TEMPORARY_LIST_ID, null);
90                 }
91                 removeCacheCompletely(geocode);
92             } finally {
93                 restoreMapStoreFlags();
94             }
95         }

239         public static void testGuessEventTime() {
240             assertTime("text 14:20 text", 14, 20);
241             assertNoTime("text 30:40 text");
242             assertNoTime("text 14:90 text");
243             final String timeHours = CgeoApplication.getInstance().getString(R.string.cache_time_full_hours);
244             assertTime("text 16 " + timeHours, 16, 0);
245             assertTime("text 16 " + StringUtils.lowerCase(timeHours), 16, 0);
246             assertTime("text 16:00 " + timeHours, 16, 0);
247             assertTime("text 16.00 " + timeHours, 16, 0);
248             assertTime("text 14:20.", 14, 20);
249             assertTime("<b>14:20</b>", 14, 20);
250             assertTime("<u><em>Uhrzeit:</em></u> 17-20 " + timeHours + "</span></strong>", 17, 00);
251             assertTime("von 11 bis 13 " + timeHours, 11, 00);
252             assertTime("from 11 to 13 " + timeHours, 11, 00);
253         }
