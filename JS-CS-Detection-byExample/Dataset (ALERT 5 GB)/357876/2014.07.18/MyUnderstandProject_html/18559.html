104         /**
105          * insert text into the EditText at the current cursor position
106          *
107          * @param editText
108          * @param insertText
109          * @param moveCursor
110          *            place the cursor after the inserted text
111          */
112         public static void insertAtPosition(final EditText editText, final String insertText, final boolean moveCursor) {
113             int selectionStart = editText.getSelectionStart();
114             int selectionEnd = editText.getSelectionEnd();
115             int start = Math.min(selectionStart, selectionEnd);
116             int end = Math.max(selectionStart, selectionEnd);
117     
118             final String content = editText.getText().toString();
119             String completeText;
120             if (start > 0 && !Character.isWhitespace(content.charAt(start - 1))) {
121                 completeText = " " + insertText;
122             } else {
123                 completeText = insertText;
124             }
125     
126             editText.getText().replace(start, end, completeText);
127             int newCursor = moveCursor ? start + completeText.length() : start;
128             editText.setSelection(newCursor);
129         }
