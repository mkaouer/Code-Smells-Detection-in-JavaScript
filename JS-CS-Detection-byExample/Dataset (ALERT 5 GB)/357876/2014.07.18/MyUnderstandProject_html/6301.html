32     class FieldnoteExport extends AbstractExport {
33         private static final File exportLocation = new File(Environment.getExternalStorageDirectory().getAbsolutePath() + "/field-notes");

36         protected FieldnoteExport() {

40         @Override
41         public void export(final List<Geocache> cachesList, final Activity activity) {
42             final Geocache[] caches = cachesList.toArray(new Geocache[cachesList.size()]);
43             if (null == activity) {
44                 // No activity given, so no user interaction possible.
45                 // Start export with default parameters.
46                 new ExportTask(null, false, false).execute(caches);
47             } else {
48                 // Show configuration dialog
49                 getExportOptionsDialog(caches, activity).show();
50             }
51         }

53         private Dialog getExportOptionsDialog(final Geocache[] caches, final Activity activity) {
54             final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
55     
56             // AlertDialog has always dark style, so we have to apply it as well always
57             final View layout = View.inflate(new ContextThemeWrapper(activity, R.style.dark), R.layout.fieldnote_export_dialog, null);
58             builder.setView(layout);
59     
60             final CheckBox uploadOption = (CheckBox) layout.findViewById(R.id.upload);
61             uploadOption.setChecked(Settings.getFieldNoteExportUpload());
62             final CheckBox onlyNewOption = (CheckBox) layout.findViewById(R.id.onlynew);
63             onlyNewOption.setChecked(Settings.getFieldNoteExportOnlyNew());
64     
65             if (Settings.getFieldnoteExportDate() > 0) {
66                 onlyNewOption.setText(getString(R.string.export_fieldnotes_onlynew) + " (" + Formatter.formatDateTime(Settings.getFieldnoteExportDate()) + ')');
67             }
68     
69             builder.setPositiveButton(R.string.export, new DialogInterface.OnClickListener() {
70     
71                 @Override
72                 public void onClick(final DialogInterface dialog, final int which) {
73                     final boolean upload = uploadOption.isChecked();
74                     final boolean onlyNew = onlyNewOption.isChecked();
75                     Settings.setFieldNoteExportUpload(upload);
76                     Settings.setFieldNoteExportOnlyNew(onlyNew);
77                     dialog.dismiss();
78                     new ExportTask(activity, upload, onlyNew).execute(caches);
79                 }
80             });
81     
82             return builder.create();
83         }

85         private class ExportTask extends AsyncTaskWithProgress<Geocache, Boolean> {
86             private final Activity activity;
87             private final boolean upload;
88             private final boolean onlyNew;
89             private File exportFile;
90     
91             private static final int STATUS_UPLOAD = -1;
92     
93             /**
94              * Instantiates and configures the task for exporting field notes.
95              *
96              * @param activity
97              *            optional: Show a progress bar and toasts
98              * @param upload
99              *            Upload the Field Note to geocaching.com
100              * @param onlyNew
101              *            Upload/export only new logs since last export
102              */
103             public ExportTask(final Activity activity, final boolean upload, final boolean onlyNew) {
104                 super(activity, getProgressTitle(), getString(R.string.export_fieldnotes_creating), true);
105                 this.activity = activity;
106                 this.upload = upload;
107                 this.onlyNew = onlyNew;
108             }
109     
110             @Override
111             protected Boolean doInBackgroundInternal(final Geocache[] caches) {
112                 // export field notes separately for each connector, so the file can be uploaded to the respective site afterwards
113                 for (IConnector connector : ConnectorFactory.getConnectors()) {
114                     if (connector instanceof FieldNotesCapability) {
115                         exportFieldNotes((FieldNotesCapability) connector, caches);
116                     }
117                 }
118                 return true;
119             }
120     
121             private boolean exportFieldNotes(final FieldNotesCapability connector, Geocache[] caches) {
122                 final FieldNotes fieldNotes = new FieldNotes();
123                 try {
124                     int i = 0;
125                     for (final Geocache cache : caches) {
126                         if (ConnectorFactory.getConnector(cache).equals(connector) && cache.isLogOffline()) {
127                             final LogEntry log = DataStore.loadLogOffline(cache.getGeocode());
128                             if (!onlyNew || log.date > Settings.getFieldnoteExportDate()) {
129                                 fieldNotes.add(cache, log);
130                             }
131                         }
132                         publishProgress(++i);
133                     }
134                 } catch (final Exception e) {
135                     Log.e("FieldnoteExport.ExportTask generation", e);
136                     return false;
137                 }
138                 fieldNotesCount += fieldNotes.size();
139     
140                 exportFile = fieldNotes.writeToDirectory(exportLocation);
141                 if (exportFile == null) {
142                     return false;
143                 }
144     
145                 if (upload) {
146                     publishProgress(STATUS_UPLOAD);
147                     if (!connector.uploadFieldNotes(exportFile)) {
148                         return false;
149                     }
150                 }
151     
152                 return true;
153             }
154     
155             @Override
156             protected void onPostExecuteInternal(final Boolean result) {
157                 if (null != activity) {
158                     if (result) {
159                         Settings.setFieldnoteExportDate(System.currentTimeMillis());
160     
161                         ActivityMixin.showToast(activity, getName() + ' ' + getString(R.string.export_exportedto) + ": " + exportFile.toString());
162     
163                         if (upload) {
164                             ActivityMixin.showToast(activity, getString(R.string.export_fieldnotes_upload_success));
165                         }
166                     } else {
167                         ActivityMixin.showToast(activity, getString(R.string.export_failed));
168                     }
169                 }
170             }
171     
172             @Override
173             protected void onProgressUpdateInternal(final int status) {
174                 if (null != activity) {
175                     setMessage(getString(STATUS_UPLOAD == status ? R.string.export_fieldnotes_uploading : R.string.export_fieldnotes_creating) + " (" + fieldNotesCount + ')');
176                 }
177             }
178         }
