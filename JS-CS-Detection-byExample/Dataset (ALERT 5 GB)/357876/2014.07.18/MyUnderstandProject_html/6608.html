151         public void onResume() {
152             super.onResume();
153             final Uri uri = getIntent().getData();
154             if (uri != null) {
155                 final String verifier = uri.getQueryParameter("oauth_verifier");
156                 if (StringUtils.isNotBlank(verifier)) {
157                     exchangeTokens(verifier);

165         private void requestToken() {
166     
167             final Parameters params = new Parameters();
168             params.put("oauth_callback", callback);
169             final String method = "GET";
170             OAuth.signOAuth(host, pathRequest, method, https, params, null, null, consumerKey, consumerSecret);
171             final HttpResponse response = Network.getRequest(getUrlPrefix() + host + pathRequest, params);
172     
173             if (Network.isSuccess(response)) {
174                 final String line = Network.getResponseData(response);
175     
176                 int status = STATUS_ERROR;
177                 if (StringUtils.isNotBlank(line)) {
178                     assert line != null;
179                     final MatcherWrapper paramsMatcher1 = new MatcherWrapper(paramsPattern1, line);
180                     if (paramsMatcher1.find()) {
181                         OAtoken = paramsMatcher1.group(1);
182                     }
183                     final MatcherWrapper paramsMatcher2 = new MatcherWrapper(paramsPattern2, line);
184                     if (paramsMatcher2.find()) {
185                         OAtokenSecret = paramsMatcher2.group(1);
186                     }
187     
188                     if (StringUtils.isNotBlank(OAtoken) && StringUtils.isNotBlank(OAtokenSecret)) {
189                         setTempTokens(OAtoken, OAtokenSecret);
190                         try {
191                             final Parameters paramsBrowser = new Parameters();
192                             paramsBrowser.put("oauth_token", OAtoken);
193                             final String encodedParams = EntityUtils.toString(new UrlEncodedFormEntity(paramsBrowser));
194                             startActivity(new Intent(Intent.ACTION_VIEW, Uri.parse(getUrlPrefix() + host + pathAuthorize + "?" + encodedParams)));
195                             status = STATUS_SUCCESS;
196                         } catch (ParseException e) {
197                             Log.e("OAuthAuthorizationActivity.requestToken", e);
198                         } catch (IOException e) {
199                             Log.e("OAuthAuthorizationActivity.requestToken", e);
200                         }
201                     }
202                 }
203     
204                 requestTokenHandler.sendEmptyMessage(status);
205             } else {
206                 final String extErrMsg = getExtendedErrorMsg(response);
207                 if (StringUtils.isNotBlank(extErrMsg)) {
208                     final Message msg = requestTokenHandler.obtainMessage(STATUS_ERROR_EXT_MSG, extErrMsg);
209                     requestTokenHandler.sendMessage(msg);
210                 } else {
211                     requestTokenHandler.sendEmptyMessage(STATUS_ERROR);
212                 }
213             }
214         }

216         private void changeToken(final String verifier) {
217     
218             int status = NOT_AUTHENTICATED;
219     
220             try {
221                 final Parameters params = new Parameters("oauth_verifier", verifier);
222     
223                 final String method = "POST";
224                 OAuth.signOAuth(host, pathAccess, method, https, params, OAtoken, OAtokenSecret, consumerKey, consumerSecret);
225                 final String line = StringUtils.defaultString(Network.getResponseData(Network.postRequest(getUrlPrefix() + host + pathAccess, params)));
226     
227                 OAtoken = "";
228                 OAtokenSecret = "";
229     
230                 final MatcherWrapper paramsMatcher1 = new MatcherWrapper(paramsPattern1, line);
231                 if (paramsMatcher1.find()) {
232                     OAtoken = paramsMatcher1.group(1);
233                 }
234                 final MatcherWrapper paramsMatcher2 = new MatcherWrapper(paramsPattern2, line);
235                 if (paramsMatcher2.find() && paramsMatcher2.groupCount() > 0) {
236                     OAtokenSecret = paramsMatcher2.group(1);
237                 }
238     
239                 if (StringUtils.isBlank(OAtoken) && StringUtils.isBlank(OAtokenSecret)) {
240                     OAtoken = "";
241                     OAtokenSecret = "";
242                     setTokens(null, null, false);
243                 } else {
244                     setTokens(OAtoken, OAtokenSecret, true);
245                     status = AUTHENTICATED;
246                 }
247             } catch (Exception e) {
248                 Log.e("OAuthAuthorizationActivity.changeToken", e);
249             }
250     
251             changeTokensHandler.sendEmptyMessage(status);
252         }
