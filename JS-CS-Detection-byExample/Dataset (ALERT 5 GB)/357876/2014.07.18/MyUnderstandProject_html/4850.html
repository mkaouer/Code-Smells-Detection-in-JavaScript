48             private void installFromMarket(Activity activity) {
49                 try {
50                     // allow also opening pure http URLs in addition to market packages
51                     final String url = (packageName.startsWith("http:")) ? packageName : "market://details?id=" + packageName;
52                     final Intent marketIntent = new Intent(Intent.ACTION_VIEW, Uri.parse(url));
53                     marketIntent.addFlags(Intent.FLAG_ACTIVITY_NO_HISTORY | Intent.FLAG_ACTIVITY_CLEAR_WHEN_TASK_RESET);
54                     activity.startActivity(marketIntent);
55     
56                 } catch (RuntimeException e) {
57                     // market not available in standard emulator
58                 }
59             }
60         }
