64         @Override
65         public final void onCreate(final Bundle savedInstanceState) {
66             super.onCreate(savedInstanceState);
67             final Intent intent = getIntent();
68     
69             // search suggestion for a cache
70             if (Intents.ACTION_GEOCACHE.equals(intent.getAction())) {
71                 CacheDetailActivity.startActivity(this, intent.getStringExtra(SearchManager.QUERY));
72                 finish();
73                 return;
74             }
75     
76             // search suggestion for a trackable
77             if (Intents.ACTION_TRACKABLE.equals(intent.getAction())) {
78                 TrackableActivity.startActivity(this, null, intent.getStringExtra(SearchManager.QUERY), null);
79                 finish();
80                 return;
81             }
82     
83             // search query
84             if (Intent.ACTION_SEARCH.equals(intent.getAction())) {
85                 hideKeyboard();
86                 final String query = intent.getStringExtra(SearchManager.QUERY);
87                 final boolean keywordSearch = intent.getBooleanExtra(Intents.EXTRA_KEYWORD_SEARCH, true);
88     
89                 if (instantSearch(query, keywordSearch)) {
90                     setResult(RESULT_OK);
91                 } else {
92                     // send intent back so query string is known
93                     setResult(RESULT_CANCELED, intent);
94                 }
95                 finish();
96     
97                 return;
98             }
99     
100             setTheme();
101             setContentView(R.layout.search_activity);
102     
103             // set title in code, as the activity needs a hard coded title due to the intent filters
104             setTitle(res.getString(R.string.search));
105     
106             ButterKnife.inject(this);
107             init();
108         }

123         /**
124          * Performs a search for query either as geocode, trackable code or keyword
125          *
126          * @param query
127          *            String to search for
128          * @param keywordSearch
129          *            Set to true if keyword search should be performed if query isn't GC or TB
130          * @return true if a search was performed, else false
131          */
132         private boolean instantSearch(final String nonTrimmedQuery, final boolean keywordSearch) {
133             final String query = StringUtils.trim(nonTrimmedQuery);
134     
135             // first check if this was a scanned URL
136             String geocode = ConnectorFactory.getGeocodeFromURL(query);
137     
138             // otherwise see if this is a pure geocode
139             if (StringUtils.isEmpty(geocode)) {
140                 geocode = StringUtils.upperCase(StringUtils.trim(query));
141             }
142     
143             final IConnector connector = ConnectorFactory.getConnector(geocode);
144             if (connector instanceof ISearchByGeocode) {
145                 CacheDetailActivity.startActivity(this, geocode.toUpperCase(Locale.US));
146                 return true;
147             }
148     
149             // Check if the query is a TB code
150             TrackableConnector trackableConnector = ConnectorFactory.getTrackableConnector(geocode);
151     
152             // check if the query contains a TB code
153             if (trackableConnector == ConnectorFactory.UNKNOWN_TRACKABLE_CONNECTOR) {
154                 final String tbCode = ConnectorFactory.getTrackableFromURL(query);
155                 if (StringUtils.isNotBlank(tbCode)) {
156                     trackableConnector = ConnectorFactory.getTrackableConnector(tbCode);
157                     geocode = tbCode;
158                 }
159             }
160     
161             if (trackableConnector != ConnectorFactory.UNKNOWN_TRACKABLE_CONNECTOR) {
162                 final Intent trackablesIntent = new Intent(this, TrackableActivity.class);
163                 trackablesIntent.putExtra(Intents.EXTRA_GEOCODE, geocode.toUpperCase(Locale.US));
164                 startActivity(trackablesIntent);
165                 return true;
166             }
167     
168             if (keywordSearch) { // keyword fallback, if desired by caller
169                 CacheListActivity.startActivityKeyword(this, query.trim());
170                 return true;
171             }
172     
173             return false;
174         }

297         private void findByCoordsFn() {
298             final String latText = StringUtils.trim(buttonLatitude.getText().toString());
299             final String lonText = StringUtils.trim(buttonLongitude.getText().toString());
300     
301             if (StringUtils.isEmpty(latText) || StringUtils.isEmpty(lonText)) {
302                 final IGeoData geo = app.currentGeo();
303                 if (geo.getCoords() != null) {
304                     buttonLatitude.setText(geo.getCoords().format(GeopointFormatter.Format.LAT_DECMINUTE));
305                     buttonLongitude.setText(geo.getCoords().format(GeopointFormatter.Format.LON_DECMINUTE));
306                 }
307             } else {
308                 try {
309                     CacheListActivity.startActivityCoordinates(this, new Geopoint(latText, lonText));
310                 } catch (final Geopoint.ParseException e) {
311                     showToast(res.getString(e.resource));
312                 }
313             }
314         }
