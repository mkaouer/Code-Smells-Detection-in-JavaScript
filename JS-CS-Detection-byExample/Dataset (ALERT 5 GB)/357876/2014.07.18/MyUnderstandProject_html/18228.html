31         /**
32          * Checks whether a launch intent is available or if the package is just installed
33          * This function is relatively costly, so if you know that the package in question has
34          * a launch intent, use isLaunchable() instead.
35          *
36          * @param packageName
37          * @return
38          */
39         public static boolean isInstalled(final String packageName) {
40             return isLaunchable(packageName) || hasPackageInstalled(packageName);
41         }
42     
43         /**
44          * This will find installed applications even without launch intent (e.g. the streetview plugin).
45          */
46         private static boolean hasPackageInstalled(final String packageName) {
47             final List<PackageInfo> packs = CgeoApplication.getInstance().getPackageManager().getInstalledPackages(0);
48             for (final PackageInfo packageInfo : packs) {
49                 if (packageName.equals(packageInfo.packageName)) {
50                     return true;
51                 }
52             }
53             return false;
54         }
55     
56         /**
57          * This will find applications, which can be launched.
58          */
59         public static Intent getLaunchIntent(final String packageName) {
60             if (packageName == null) {
61                 return null;
62             }
63             final PackageManager packageManager = CgeoApplication.getInstance().getPackageManager();
64             try {
65                 // This can throw an exception where the exception type is only defined on API Level > 3
66                 // therefore surround with try-catch
67                 return packageManager.getLaunchIntentForPackage(packageName);
68             } catch (final Exception e) {
69                 return null;
70             }
71         }

73         public static boolean isIntentAvailable(final String intent) {
74             return isIntentAvailable(intent, null);
75         }

77         /**
78          * Indicates whether the specified action can be used as an intent. This
79          * method queries the package manager for installed packages that can
80          * respond to an intent with the specified action. If no suitable package is
81          * found, this method returns false.
82          * 
83          * @param action
84          *            The Intent action to check for availability.
85          * @param uri
86          *            The Intent URI to check for availability.
87          * 
88          * @return True if an Intent with the specified action can be sent and
89          *         responded to, false otherwise.
90          */
91         public static boolean isIntentAvailable(final String action, final Uri uri) {
92             final PackageManager packageManager = CgeoApplication.getInstance().getPackageManager();
93             final Intent intent;
94             if (uri == null) {
95                 intent = new Intent(action);
96             } else {
97                 intent = new Intent(action, uri);
98             }
99             final List<ResolveInfo> list = packageManager.queryIntentActivities(intent,
100                     PackageManager.MATCH_DEFAULT_ONLY);
101             return CollectionUtils.isNotEmpty(list);
102         }
