33             public StoreCacheHandler(final int progressMessage) {

36     
37             @Override
38             public void handleRegularMessage(Message msg) {
39                 if (UPDATE_LOAD_PROGRESS_DETAIL == msg.what && msg.obj instanceof String) {
40                     updateStatusMsg((String) msg.obj);
41                 } else {
42                     init();
43                 }
44             }
45     
46             private void updateStatusMsg(final String msg) {
47                 progress.setMessage(res.getString(progressMessage)
48                         + "\n\n"
49                         + msg);
50             }
51         }

104         private class StoreCacheClickListener implements View.OnClickListener {
105             @Override
106             public void onClick(View arg0) {
107                 if (progress.isShowing()) {
108                     showToast(res.getString(R.string.err_detail_still_working));
109                     return;
110                 }
111     
112                 if (Settings.getChooseList()) {
113                     // let user select list to store cache in
114                     new StoredList.UserInterface(CachePopup.this).promptForListSelection(R.string.list_title,
115                             new Action1<Integer>() {
116                                 @Override
117                                 public void call(final Integer selectedListId) {
118                                     storeCache(selectedListId);
119                                 }
120                             }, true, StoredList.TEMPORARY_LIST_ID);
121                 } else {
122                     storeCache(StoredList.TEMPORARY_LIST_ID);
123                 }
124             }
125     
126             protected void storeCache(final int listId) {
127                 final StoreCacheHandler storeCacheHandler = new StoreCacheHandler(R.string.cache_dialog_offline_save_message);
128                 progress.show(CachePopup.this, res.getString(R.string.cache_dialog_offline_save_title), res.getString(R.string.cache_dialog_offline_save_message), true, storeCacheHandler.cancelMessage());
129                 Schedulers.io().schedule(new Action1<Inner>() {
130                     @Override
131                     public void call(final Inner inner) {
132                         cache.store(listId, storeCacheHandler);
133                         invalidateOptionsMenuCompatible();
134                     }
135                 });
136             }
137         }

139         private class RefreshCacheClickListener implements View.OnClickListener {
140             @Override
141             public void onClick(View arg0) {
142                 if (progress.isShowing()) {
143                     showToast(res.getString(R.string.err_detail_still_working));
144                     return;
145                 }
146     
147                 if (!Network.isNetworkConnected(getApplicationContext())) {
148                     showToast(getString(R.string.err_server));
149                     return;
150                 }
151     
152                 final StoreCacheHandler refreshCacheHandler = new StoreCacheHandler(R.string.cache_dialog_offline_save_message);
153                 progress.show(CachePopup.this, res.getString(R.string.cache_dialog_refresh_title), res.getString(R.string.cache_dialog_refresh_message), true, refreshCacheHandler.cancelMessage());
154                 cache.refresh(cache.getListId(), refreshCacheHandler, Schedulers.io());
155             }
156         }
157     
158         private class DropCacheClickListener implements View.OnClickListener {
159             @Override
160             public void onClick(View arg0) {
161                 if (progress.isShowing()) {
162                     showToast(res.getString(R.string.err_detail_still_working));
163                     return;
164                 }
165     
166                 final DropCacheHandler dropCacheHandler = new DropCacheHandler();
167                 progress.show(CachePopup.this, res.getString(R.string.cache_dialog_offline_drop_title), res.getString(R.string.cache_dialog_offline_drop_message), true, null);
168                 cache.drop(dropCacheHandler, Schedulers.io());
169             }
170         }
