17     public class ScaleDrawer {
18         private static final double SCALE_WIDTH_FACTOR = 1.0 / 2.5;
19     
20         private Paint scale = null;
21         private Paint scaleShadow = null;
22         private BlurMaskFilter blur = null;
23         private float pixelDensity = 0;
24     
25         public ScaleDrawer(Activity activity) {
26             DisplayMetrics metrics = new DisplayMetrics();
27             activity.getWindowManager().getDefaultDisplay().getMetrics(metrics);
28             pixelDensity = metrics.density;
29         }
30     
31         static private double keepSignificantDigit(final double distance) {
32             final double scale = Math.pow(10, Math.floor(Math.log10(distance)));
33             return scale * Math.floor(distance / scale);
34         }
35     
36         void drawScale(Canvas canvas, MapViewImpl mapView) {
37             final double span = mapView.getLongitudeSpan() / 1e6;
38             final GeoPointImpl center = mapView.getMapViewCenter();
39     
40             final int bottom = mapView.getHeight() - 14; // pixels from bottom side of screen
41     
42             final Geopoint leftCoords = new Geopoint(center.getLatitudeE6() / 1e6, center.getLongitudeE6() / 1e6 - span / 2);
43             final Geopoint rightCoords = new Geopoint(center.getLatitudeE6() / 1e6, center.getLongitudeE6() / 1e6 + span / 2);
44     
45             final ImmutablePair<Double, String> scaled = Units.scaleDistance(leftCoords.distanceTo(rightCoords) * SCALE_WIDTH_FACTOR);
46     
47             final double distanceRound = keepSignificantDigit(scaled.left);
48             final double pixels = Math.round((mapView.getWidth() * SCALE_WIDTH_FACTOR / scaled.left) * distanceRound);
49     
50             if (blur == null) {
51                 blur = new BlurMaskFilter(3, BlurMaskFilter.Blur.NORMAL);
52             }
53     
54             if (scaleShadow == null) {
55                 scaleShadow = new Paint();
56                 scaleShadow.setAntiAlias(true);
57                 scaleShadow.setStrokeWidth(4 * pixelDensity);
58                 scaleShadow.setMaskFilter(blur);
59                 scaleShadow.setTextSize(14 * pixelDensity);
60                 scaleShadow.setTypeface(Typeface.DEFAULT_BOLD);
61             }
62     
63             if (scale == null) {
64                 scale = new Paint();
65                 scale.setAntiAlias(true);
66                 scale.setStrokeWidth(2 * pixelDensity);
67                 scale.setTextSize(14 * pixelDensity);
68                 scale.setTypeface(Typeface.DEFAULT_BOLD);
69             }
70     
71             if (mapView.needsInvertedColors()) {
72                 scaleShadow.setColor(0xFF000000);
73                 scale.setColor(0xFFFFFFFF);
74             } else {
75                 scaleShadow.setColor(0xFFFFFFFF);
76                 scale.setColor(0xFF000000);
77             }
78     
79             final String formatString = distanceRound >= 1 ? "%.0f" : "%.1f";
80     
81             canvas.drawLine(10, bottom, 10, (bottom - (8 * pixelDensity)), scaleShadow);
82             canvas.drawLine((int) (pixels + 10), bottom, (int) (pixels + 10), (bottom - (8 * pixelDensity)), scaleShadow);
83             canvas.drawLine(8, bottom, (int) (pixels + 12), bottom, scaleShadow);
84             canvas.drawText(String.format(formatString, distanceRound) + " " + scaled.right, (float) (pixels - (10 * pixelDensity)), (bottom - (10 * pixelDensity)), scaleShadow);
85     
86             canvas.drawLine(11, bottom, 11, (bottom - (6 * pixelDensity)), scale);
87             canvas.drawLine((int) (pixels + 9), bottom, (int) (pixels + 9), (bottom - (6 * pixelDensity)), scale);
88             canvas.drawLine(10, bottom, (int) (pixels + 10), bottom, scale);
89             canvas.drawText(String.format(formatString, distanceRound) + " " + scaled.right, (float) (pixels - (10 * pixelDensity)), (bottom - (10 * pixelDensity)), scale);
90         }
