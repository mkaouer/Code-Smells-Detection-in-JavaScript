57                 contacts.addAll(getContacts(nickName, ContactsContract.Contacts.CONTENT_URI, BaseColumns._ID, ContactsContract.Contacts.DISPLAY_NAME, true));
58             }

60             if (contacts.isEmpty()) {
61                 showToast(getString(R.string.contact_not_found, nickName));
62                 finish();

102         private List<Pair<Integer, String>> getContacts(final @NonNull String searchName, Uri uri, final @NonNull String idColumnName, final @NonNull String selectionColumnName, boolean like) {
103             final String[] projection = new String[] { idColumnName, selectionColumnName };
104             final String selection = selectionColumnName + (like ? " LIKE" : " =") + " ? COLLATE NOCASE";
105             final String[] selectionArgs = new String[] { like ? "%" + searchName + "%" : searchName };
106             Cursor cursor = null;
107     
108             List<Pair<Integer, String>> result = new ArrayList<Pair<Integer, String>>();
109             try {
110                 cursor = getContentResolver().query(uri, projection, selection, selectionArgs, null);
111                 while (cursor != null && cursor.moveToNext()) {
112                     int foundId = cursor.getInt(0);
113                     String foundName = cursor.getString(1);
114                     result.add(new Pair<Integer, String>(Integer.valueOf(foundId), foundName));
115                 }
116             } catch (Exception e) {
117                 Log.e(LOG_TAG, "ContactsActivity.getContactId", e);
118             } finally {
119                 if (cursor != null) {
120                     cursor.close();
121                 }
122             }
123             return result;
124         }

133         private static String getParameter(final Uri uri, final String paramKey) {
134             try {
135                 final String param = uri.getQueryParameter(paramKey);
136                 if (param == null) {
137                     return StringUtils.EMPTY;
138                 }
139                 return URLDecoder.decode(param, CharEncoding.UTF_8).trim();
140             } catch (UnsupportedEncodingException e) {
141                 Log.e(LOG_TAG, "ContactsActivity.getParameter", e);
142             }
143             return StringUtils.EMPTY;
144         }
