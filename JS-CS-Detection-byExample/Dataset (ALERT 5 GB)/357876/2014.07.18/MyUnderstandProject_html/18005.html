20         public static void testDownloadStaticMaps() {
21             final double lat = 52.354176d;
22             final double lon = 9.745685d;
23             String geocode = "GCTEST1";
24     
25             boolean backupStore = Settings.isStoreOfflineMaps();
26             boolean backupStoreWP = Settings.isStoreOfflineWpMaps();
27             TestSettings.setStoreOfflineMaps(true);
28             TestSettings.setStoreOfflineWpMaps(true);
29             try {
30                 Geopoint gp = new Geopoint(lat + 0.25d, lon + 0.25d);
31                 Geocache cache = new Geocache();
32                 cache.setGeocode(geocode);
33                 cache.setCoords(gp);
34                 cache.setCacheId(String.valueOf(1));
35     
36                 Waypoint theFinal = new Waypoint("Final", WaypointType.FINAL, false);
37                 Geopoint finalGp = new Geopoint(lat + 0.25d + 1, lon + 0.25d + 1);
38                 theFinal.setCoords(finalGp);
39                 theFinal.setId(1);
40                 cache.addOrChangeWaypoint(theFinal, false);
41     
42                 Waypoint trailhead = new Waypoint("Trail head", WaypointType.TRAILHEAD, false);
43                 Geopoint trailheadGp = new Geopoint(lat + 0.25d + 2, lon + 0.25d + 2);
44                 trailhead.setCoords(trailheadGp);
45                 trailhead.setId(2);
46                 cache.addOrChangeWaypoint(trailhead, false);
47     
48                 // make sure we don't have stale downloads
49                 deleteCacheDirectory(geocode);
50                 assertThat(StaticMapsProvider.hasStaticMap(cache)).isFalse();
51                 assertThat(StaticMapsProvider.hasStaticMapForWaypoint(geocode, theFinal)).isFalse();
52                 assertThat(StaticMapsProvider.hasStaticMapForWaypoint(geocode, trailhead)).isFalse();
53     
54                 // download
55                 StaticMapsProvider.downloadMaps(cache);
56     
57                 try {
58                     Thread.sleep(10000);
59                 } catch (InterruptedException e) {
60                     fail();
61                 }
62     
63                 // check download
64                 assertThat(StaticMapsProvider.hasStaticMap(cache)).isTrue();
65                 assertThat(StaticMapsProvider.hasStaticMapForWaypoint(geocode, theFinal)).isTrue();
66                 assertThat(StaticMapsProvider.hasStaticMapForWaypoint(geocode, trailhead)).isTrue();
67     
68                 // waypoint static maps hashcode dependent
69                 trailhead.setCoords(new Geopoint(lat + 0.24d + 2, lon + 0.25d + 2));
70                 assertThat(StaticMapsProvider.hasStaticMapForWaypoint(geocode, trailhead)).isFalse();
71             } finally {
72                 TestSettings.setStoreOfflineWpMaps(backupStoreWP);
73                 TestSettings.setStoreOfflineMaps(backupStore);
74                 deleteCacheDirectory(geocode);
75             }
76         }
