16         public Viewport(final @NonNull ICoordinates point1, final @NonNull ICoordinates point2) {
17             final Geopoint gp1 = point1.getCoords();
18             final Geopoint gp2 = point2.getCoords();
19             this.bottomLeft = new Geopoint(Math.min(gp1.getLatitude(), gp2.getLatitude()),
20                     Math.min(gp1.getLongitude(), gp2.getLongitude()));
21             this.topRight = new Geopoint(Math.max(gp1.getLatitude(), gp2.getLatitude()),
22                     Math.max(gp1.getLongitude(), gp2.getLongitude()));
23             this.center = new Geopoint((gp1.getLatitude() + gp2.getLatitude()) / 2,
24                     (gp1.getLongitude() + gp2.getLongitude()) / 2);
25         }

27         public Viewport(final @NonNull ICoordinates center, final double latSpan, final double lonSpan) {
28             this.center = center.getCoords();
29             final double centerLat = this.center.getLatitude();
30             final double centerLon = this.center.getLongitude();
31             final double latHalfSpan = Math.abs(latSpan) / 2;
32             final double lonHalfSpan = Math.abs(lonSpan) / 2;
33             bottomLeft = new Geopoint(centerLat - latHalfSpan, centerLon - lonHalfSpan);
34             topRight = new Geopoint(centerLat + latHalfSpan, centerLon + lonHalfSpan);
35         }

66         /**
67          * Check whether a point is contained in this viewport.
68          *
69          * @param point
70          *            the coordinates to check
71          * @return true if the point is contained in this viewport, false otherwise or if the point contains no coordinates

73         public boolean contains(final @NonNull ICoordinates point) {
74             final Geopoint coords = point.getCoords();
75             return coords != null
76                     && coords.getLongitudeE6() >= bottomLeft.getLongitudeE6()
77                     && coords.getLongitudeE6() <= topRight.getLongitudeE6()
78                     && coords.getLatitudeE6() >= bottomLeft.getLatitudeE6()
79                     && coords.getLatitudeE6() <= topRight.getLatitudeE6();
80         }

125         /**
126          * Return the smallest viewport containing all the given points.
127          *
128          * @param points
129          *            a set of points. Point with null coordinates (or null themselves) will be ignored
130          * @return the smallest viewport containing the non-null coordinates, or null if no coordinates are non-null
131          */
132         static public @Nullable
133         Viewport containing(final Collection<? extends ICoordinates> points) {
134             boolean valid = false;
135             double latMin = Double.MAX_VALUE;
136             double latMax = -Double.MAX_VALUE;
137             double lonMin = Double.MAX_VALUE;
138             double lonMax = -Double.MAX_VALUE;
139             for (final ICoordinates point : points) {
140                 if (point != null) {
141                     final Geopoint coords = point.getCoords();
142                     if (coords != null) {
143                         valid = true;
144                         final double latitude = coords.getLatitude();
145                         final double longitude = coords.getLongitude();
146                         latMin = Math.min(latMin, latitude);
147                         latMax = Math.max(latMax, latitude);
148                         lonMin = Math.min(lonMin, longitude);
149                         lonMax = Math.max(lonMax, longitude);
150                     }
151                 }
152             }
153             if (!valid) {
154                 return null;
155             }
156             return new Viewport(new Geopoint(latMin, lonMin), new Geopoint(latMax, lonMax));
157         }
