76         private static void showOfflineMenu(final Geocache cache, final Activity activity) {
77             final LogEntry currentLog = DataStore.loadLogOffline(cache.getGeocode());
78             final LogType currentLogType = currentLog == null ? null : currentLog.type;
79     
80             final List<LogType> logTypes = cache.getPossibleLogTypes();
81             final ArrayList<LogTypeEntry> list = new ArrayList<LogTypeEntry>();
82             for (LogType logType : logTypes) {
83                 list.add(new LogTypeEntry(logType, null, logType == currentLogType));
84             }
85             if (cache.isLogOffline()) {
86                 list.add(new LogTypeEntry(null, SpecialLogType.CLEAR_LOG, false));
87             }
88             list.add(new LogTypeEntry(null, SpecialLogType.LOG_CACHE, false));
89     
90             final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
91             builder.setTitle(R.string.cache_menu_visit_offline);
92     
93             final ArrayAdapter<LogTypeEntry> adapter = new ArrayAdapter<LogTypeEntry>(activity, android.R.layout.select_dialog_item, list);
94     
95             builder.setAdapter(adapter, new DialogInterface.OnClickListener() {
96                 @Override
97                 public void onClick(DialogInterface dialog, int item) {
98                     final LogTypeEntry logTypeEntry = adapter.getItem(item);
99                     if (logTypeEntry.logType == null) {
100                         switch (logTypeEntry.specialLogType) {
101                             case LOG_CACHE:
102                                 cache.logVisit(activity);
103                                 break;
104     
105                             case CLEAR_LOG:
106                                 cache.clearOfflineLog();
107                                 break;
108                         }
109                     } else {
110                         cache.logOffline(activity, logTypeEntry.logType);
111                     }
112                 }
113             });
114     
115             builder.create().show();
116     
117         }

119         public static void onPrepareOptionsMenu(Menu menu, Geocache cache) {
120             if (cache == null) {
121                 return;
122             }
123             final MenuItem itemLog = menu.findItem(R.id.menu_log_visit);
124             itemLog.setVisible(cache.supportsLogging() && !Settings.getLogOffline());
125     
126             final MenuItem itemOffline = menu.findItem(R.id.menu_log_visit_offline);
127             itemOffline.setVisible(cache.supportsLogging() && Settings.getLogOffline());
128         }

130         public static void addMenuItems(Activity activity, Menu menu, Geocache cache) {
