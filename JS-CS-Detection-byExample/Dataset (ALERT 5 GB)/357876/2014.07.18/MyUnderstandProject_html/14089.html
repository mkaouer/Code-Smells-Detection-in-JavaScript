98         /**
99          * Get the guessed file extension of an URL. A file extension can contain up-to 4 characters in addition to the dot.
100          *
101          * @param url
102          *            the relative or absolute URL
103          * @return the file extension, including the leading dot, or the empty string if none could be determined
104          */
105         static String getExtension(final String url) {
106             String urlExt;
107             if (url.startsWith("data:")) {
108                 // "data:image/png;base64,i53â€¦" -> ".png"
109                 urlExt = StringUtils.substringAfter(StringUtils.substringBefore(url, ";"), "/");
110             } else {
111                 // "http://example.com/foo/bar.png" -> ".png"
112                 urlExt = StringUtils.substringAfterLast(url, ".");
113             }
114             return urlExt.length() >= 1 && urlExt.length() <= 4 ? "." + urlExt : "";
115         }

186         /**
187          * Save an HTTP response to a file.
188          *
189          * @param response
190          *            the response whose entity content will be saved
191          * @param targetFile
192          *            the target file, which will be created if necessary
193          * @return true if the operation was successful, false otherwise, in which case the file will not exist
194          */
195         public static boolean saveEntityToFile(final HttpResponse response, final File targetFile) {
196             if (response == null) {
197                 return false;
198             }
199     
200             try {
201                 final boolean saved = saveToFile(response.getEntity().getContent(), targetFile);
202                 saveHeader(HEADER_ETAG, saved ? response : null, targetFile);
203                 saveHeader(HEADER_LAST_MODIFIED, saved ? response : null, targetFile);
204                 return saved;
205             } catch (IOException e) {
206                 Log.e("LocalStorage.saveEntityToFile", e);
207             }
208     
209             return false;
210         }

212         private static void saveHeader(final String name, @Nullable final HttpResponse response, final File baseFile) {
213             final Header header = response != null ? response.getFirstHeader(name) : null;
214             final File file = filenameForHeader(baseFile, name);
215             if (header == null) {
216                 FileUtils.deleteIgnoringFailure(file);
217             } else {
218                 try {
219                     saveToFile(new ByteArrayInputStream(header.getValue().getBytes("UTF-8")), file);
220                 } catch (final UnsupportedEncodingException e) {
221                     // Do not try to display the header in the log message, as our default encoding is
222                     // likely to be UTF-8 and it will fail as well.
223                     Log.e("LocalStorage.saveHeader: unable to decode header", e);
224                 }
225             }
226         }

229             return new File(baseFile.getAbsolutePath() + "-" + name);

232         /**
233          * Get the saved header value for this file.
234          *
235          * @param baseFile
236          *            the name of the cached resource
237          * @param name
238          *            the name of the header ("etag" or "last-modified")
239          * @return the cached value, or <tt>null</tt> if none has been cached
240          */
241         @Nullable
242         public static String getSavedHeader(final File baseFile, final String name) {
243             try {
244                 final File file = filenameForHeader(baseFile, name);
245                 final Reader f = new InputStreamReader(new FileInputStream(file), "UTF-8");
246                 try {
247                     // No header will be more than 256 bytes
248                     final char[] value = new char[256];
249                     final int count = f.read(value);
250                     return new String(value, 0, count);
251                 } finally {
252                     f.close();
253                 }
254             } catch (final FileNotFoundException e) {
255                 // Do nothing, the file does not exist
256             } catch (final Exception e) {
257                 Log.w("could not read saved header " + name + " for " + baseFile, e);
258             }
259             return null;
260         }

262         /**
263          * Save a stream to a file.
264          * <p/>
265          * If the response could not be saved to the file due, for example, to a network error, the file will not exist when
266          * this method returns.
267          *
268          * @param inputStream
269          *            the stream whose content will be saved
270          * @param targetFile
271          *            the target file, which will be created if necessary
272          * @return true if the operation was successful, false otherwise
273          */
274         public static boolean saveToFile(final InputStream inputStream, final File targetFile) {
275             if (inputStream == null) {
276                 return false;
277             }
278     
279     
280             try {
281                 try {
282                     final File tempFile = File.createTempFile("download", null, targetFile.getParentFile());
283                     final FileOutputStream fos = new FileOutputStream(tempFile);
284                     final boolean written = copy(inputStream, fos);
285                     fos.close();
286                     if (written) {
287                         return tempFile.renameTo(targetFile);
288                     }
289                     FileUtils.deleteIgnoringFailure(tempFile);
290                     return false;
291                 } finally {
292                     IOUtils.closeQuietly(inputStream);
293                 }
294             } catch (IOException e) {
295                 Log.e("LocalStorage.saveToFile", e);
296                 FileUtils.deleteIgnoringFailure(targetFile);
297             }
298             return false;
299         }

301         /**
302          * Copy a file into another. The directory structure of target file will be created if needed.
303          *
304          * @param source
305          *            the source file
306          * @param destination
307          *            the target file
308          * @return true if the copy happened without error, false otherwise
309          */
310         public static boolean copy(final File source, final File destination) {
311             FileUtils.mkdirs(destination.getParentFile());
312     
313             InputStream input = null;
314             OutputStream output = null;
315             boolean copyDone = false;
316     
317             try {
318                 input = new BufferedInputStream(new FileInputStream(source));
319                 output = new BufferedOutputStream(new FileOutputStream(destination));
320                 copyDone = copy(input, output);
321                 // close here already to catch any issue with closing
322                 input.close();
323                 output.close();
324             } catch (FileNotFoundException e) {
325                 Log.e("LocalStorage.copy: could not copy file", e);
326                 return false;
327             } catch (IOException e) {
328                 Log.e("LocalStorage.copy: could not copy file", e);
329                 return false;
330             } finally {
331                 // close here quietly to clean up in all situations
332                 IOUtils.closeQuietly(input);
333                 IOUtils.closeQuietly(output);
334             }
335     
336             return copyDone;
337         }

339         public static boolean copy(final InputStream input, final OutputStream output) {
340             try {
341                 int length;
342                 final byte[] buffer = new byte[4096];
343                 while ((length = input.read(buffer)) > 0) {
344                     output.write(buffer, 0, length);
345                 }
346                 // Flushing is only necessary if the stream is not immediately closed afterwards.
347                 // We rely on all callers to do that correctly outside of this method
348             } catch (IOException e) {
349                 Log.e("LocalStorage.copy: error when copying data", e);
350                 return false;
351             }
352     
353             return true;
354         }

365         public static boolean deleteDirectory(@NonNull final File dir) {
366             final File[] files = dir.listFiles();
367     
368             // Although we are called on an existing directory, it might have been removed concurrently
369             // in the meantime, for example by the user or by another cleanup task.
370             if (files != null) {
371                 for (final File file : files) {
372                     if (file.isDirectory()) {
373                         deleteDirectory(file);
374                     } else {
375                         FileUtils.delete(file);
376                     }
377                 }
378             }
379     
380             return FileUtils.delete(dir);
381         }

394                 return;

396             for (final File file : filesToDelete) {

400                     }
401                 } catch (Exception e) {
402                     Log.e("LocalStorage.deleteFilesPrefix", e);
403                 }

427         /**
428          * Get all storages available on the device.
429          * Will include paths like /mnt/sdcard /mnt/usbdisk /mnt/ext_card /mnt/sdcard/ext_card
430          */
431         public static List<File> getStorages() {
432     
433             String extStorage = Environment.getExternalStorageDirectory().getAbsolutePath();
434             List<File> storages = new ArrayList<File>();
435             storages.add(new File(extStorage));
436             File file = new File(FILE_SYSTEM_TABLE_PATH);
437             if (file.canRead()) {
438                 Reader fr = null;
439                 BufferedReader br = null;
440                 try {
441                     fr = new InputStreamReader(new FileInputStream(file), "UTF-8");
442                     br = new BufferedReader(fr);
443                     String s = br.readLine();
444                     while (s != null) {
445                         if (s.startsWith("dev_mount")) {
446                             String[] tokens = StringUtils.split(s);
447                             if (tokens.length >= 3) {
448                                 String path = tokens[2]; // mountpoint
449                                 if (!extStorage.equals(path)) {
450                                     File directory = new File(path);
451                                     if (directory.exists() && directory.isDirectory()) {
452                                         storages.add(directory);
453                                     }
454                                 }
455                             }
456                         }
457                         s = br.readLine();
458                     }
459                 } catch (IOException e) {
460                     Log.e("Could not get additional mount points for user content. " +
461                             "Proceeding with external storage only (" + extStorage + ")");
462                 } finally {
463                     IOUtils.closeQuietly(fr);
464                     IOUtils.closeQuietly(br);
465                 }
466             }
467             return storages;
468         }
