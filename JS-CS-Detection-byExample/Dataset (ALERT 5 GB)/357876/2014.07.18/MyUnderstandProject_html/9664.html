23     package com.mapswithme.maps.api;
24     
25     import android.app.Activity;
26     import android.app.PendingIntent;
27     import android.content.Context;
28     import android.content.Intent;
29     import android.content.pm.ActivityInfo;
30     import android.net.Uri;
31     
32     
33     public final class MapsWithMeApi
34     {
35     
36       /**
37        * Most detailed level, buildings and trees are seen.
38        */
39       public static final double ZOOM_MAX = 19;
40       /**
41        * Least detailed level, continents are seen.
42        */
43       public static final double ZOOM_MIN = 1;
44     
45     
46       public static void showMapsWithMeUrl(Activity caller, PendingIntent pendingIntent, double zoomLevel, String url)
47       {
48         final Uri uri = Uri.parse(url);
49         final String latlon[] = uri.getQueryParameter("ll").split(",");
50         final double lat = Double.parseDouble(latlon[0]);
51         final double lon = Double.parseDouble(latlon[1]);
52         final String name = uri.getQueryParameter("n");
53         final String id = uri.getQueryParameter("id");
54     
55         showPointsOnMap(caller, name, zoomLevel, pendingIntent, new MWMPoint(lat, lon, name, id));
56       }
57     
58       public static void sendRequest(Activity caller, MwmRequest request)
59       {
60         final Intent mwmIntent = request.toIntent(caller);
61     
62         if (isMapsWithMeInstalled(caller))
63         {
64           // Match activity for intent
65           final ActivityInfo aInfo = caller.getPackageManager().resolveActivity(mwmIntent, 0).activityInfo;
66           mwmIntent.setClassName(aInfo.packageName, aInfo.name);
67           caller.startActivity(mwmIntent);
68         }
69         else
70           (new DownloadMapsWithMeDialog(caller)).show();
71       }
72     
73       /**
74        * Shows single point on the map.
75        *
76        * @param caller
77        * @param lat
78        * @param lon
79        * @param name
80        */
81       public static void showPointOnMap(Activity caller, double lat, double lon, String name)
82       {
83         showPointsOnMap(caller, (String) null, (PendingIntent) null, new MWMPoint(lat, lon, name));
84       }
85     
86       /**
87        * Shows single point on the map using specified zoom level in range from
88        * {@link MapsWithMeApi#ZOOM_MIN} to {@link MapsWithMeApi#ZOOM_MAX}.
89        *
90        * @param caller
91        * @param lat
92        * @param lon
93        * @param name
94        * @param zoomLevel
95        */
96       public static void showPointOnMap(Activity caller, double lat, double lon, String name, double zoomLevel)
97       {
98         showPointsOnMap(caller, (String) null, zoomLevel, (PendingIntent) null, new MWMPoint(lat, lon, name));
99       }
100     
101       /**
102        * Shows set of points on the map.
103        *
104        * @param caller
105        * @param title
106        * @param points
107        */
108       public static void showPointsOnMap(Activity caller, String title, MWMPoint... points)
109       {
110         showPointsOnMap(caller, title, null, points);
111       }
112     
113       /**
114        * Shows set of points on the maps and allows MapsWithMeApplication to send
115        * {@link PendingIntent} provided by client application.
116        *
117        * @param caller
118        * @param title
119        * @param pendingIntent
120        * @param points
121        */
122       public static void showPointsOnMap(Activity caller, String title, PendingIntent pendingIntent, MWMPoint... points)
123       {
124         showPointsOnMap(caller, title, -1, pendingIntent, points);
125       }
126     
127       private static void showPointsOnMap(Activity caller, String title, double zoomLevel, PendingIntent pendingIntent,
128           MWMPoint... points)
129       {
130         final MwmRequest request = new MwmRequest()
131                                         .setTitle(title)
132                                         .setZoomLevel(zoomLevel)
133                                         .setPendingIntent(pendingIntent)
134                                         .setPoints(points);
135         sendRequest(caller, request);
136       }
137     
138       public static void pickPoint(Activity caller, String title, PendingIntent pi)
139       {
140         final MwmRequest request = new MwmRequest()
141                                         .setTitle(title)
142                                         .setPickPointMode(true)
143                                         .setPendingIntent(pi);
144         sendRequest(caller, request);
145       }
146     
147       /**
148        * Detects if any version (Lite, Pro) of MapsWithMe, which supports API calls
149        * are installed on the device.
150        *
151        * @param context
152        * @return
153        */
154       public static boolean isMapsWithMeInstalled(Context context)
155       {
156         final Intent intent = new Intent(Const.ACTION_MWM_REQUEST);
157         return context.getPackageManager().resolveActivity(intent, 0) != null;
158       }
