289         public static ImmutablePair<String, String> getCredentials(final @NonNull ICredentials connector) {
290             final String username = getString(connector.getUsernamePreferenceKey(), null);
291             final String password = getString(connector.getPasswordPreferenceKey(), null);
292     
293             if (StringUtils.isBlank(username) || StringUtils.isBlank(password)) {
294                 return new ImmutablePair<String, String>(StringUtils.EMPTY, StringUtils.EMPTY);
295             }
296     
297             return new ImmutablePair<String, String>(username, password);
298         }

602         @NonNull
603         public static synchronized MapSource getMapSource() {
604             if (mapSource != null) {
605                 return mapSource;
606             }
607             final int id = getConvertedMapId();
608             mapSource = MapProviderFactory.getMapSource(id);
609             if (mapSource != null) {
610                 // don't use offline maps if the map file is not valid
611                 if ((!(mapSource instanceof OfflineMapSource)) || (isValidMapFile())) {
612                     return mapSource;
613                 }
614             }
615             // fallback to first available map
616             return MapProviderFactory.getDefaultSource();
617         }

619         private final static int GOOGLEMAP_BASEID = 30;
620         private final static int MAP = 1;
621         private final static int SATELLITE = 2;
622     
623         private final static int MFMAP_BASEID = 40;
624         private final static int MAPNIK = 1;
625         private final static int CYCLEMAP = 3;
626         private final static int OFFLINE = 4;
627     
628         /**
629          * convert old preference ids for maps (based on constant values) into new hash based ids
630          *
631          * @return
632          */
633         private static int getConvertedMapId() {
634             final int id = Integer.parseInt(getString(R.string.pref_mapsource,
635                     String.valueOf(MAP_SOURCE_DEFAULT)));
636             switch (id) {
637                 case GOOGLEMAP_BASEID + MAP:
638                     return GoogleMapProvider.GOOGLE_MAP_ID.hashCode();
639                 case GOOGLEMAP_BASEID + SATELLITE:
640                     return GoogleMapProvider.GOOGLE_SATELLITE_ID.hashCode();
641                 case MFMAP_BASEID + MAPNIK:
642                     return MapsforgeMapProvider.MAPSFORGE_MAPNIK_ID.hashCode();
643                 case MFMAP_BASEID + CYCLEMAP:
644                     return MapsforgeMapProvider.MAPSFORGE_CYCLEMAP_ID.hashCode();
645                 case MFMAP_BASEID + OFFLINE: {
646                     final String mapFile = Settings.getMapFile();
647                     if (StringUtils.isNotEmpty(mapFile)) {
648                         return mapFile.hashCode();
649                     }
650                     break;
651                 }
652                 default:
653                     break;
654             }
655             return id;
656         }

892             File directory = new File(Settings.getCustomRenderThemeBaseFolder());
893             List<File> result = new ArrayList<File>();
894             FileUtils.listDir(result, directory, new ExtensionsBasedFileSelector(new String[] { "xml" }), null);
895     
896             return result.toArray(new File[result.size()]);
897         }

899         private static class ExtensionsBasedFileSelector extends FileSelector {
900             private final String[] extensions;
901             public ExtensionsBasedFileSelector(String[] extensions) {
902                 this.extensions = extensions;
903             }
904             @Override
905             public boolean isSelected(File file) {
906                 String filename = file.getName();
907                 for (String ext : extensions) {
908                     if (StringUtils.endsWithIgnoreCase(filename, ext)) {
909                         return true;
910                     }
911                 }
912                 return false;
913             }
914             @Override
915             public boolean shouldEnd() {
916                 return false;
917             }
918         }
