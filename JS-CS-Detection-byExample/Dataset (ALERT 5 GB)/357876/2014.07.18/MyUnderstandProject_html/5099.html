61         private static void downloadMap(final String geocode, final int zoom, final String mapType, final String markerUrl, final String prefix, final String shadow, final String latlonMap, final int width, final int height, final Parameters waypoints) {
62             final Parameters params = new Parameters(
63                     "center", latlonMap,
64                     "zoom", String.valueOf(zoom),
65                     "size", String.valueOf(limitSize(width)) + 'x' + String.valueOf(limitSize(height)),
66                     "maptype", mapType,
67                     "markers", "icon:" + markerUrl + '|' + shadow + latlonMap,
68                     "sensor", "false");
69             if (waypoints != null) {
70                 params.addAll(waypoints);
71             }
72             final HttpResponse httpResponse = Network.getRequest(GOOGLE_STATICMAP_URL, params);
73     
74             if (httpResponse == null) {
75                 Log.e("StaticMapsProvider.downloadMap: httpResponse is null");
76                 return;
77             }
78             if (httpResponse.getStatusLine().getStatusCode() != 200) {
79                 Log.d("StaticMapsProvider.downloadMap: httpResponseCode = " + httpResponse.getStatusLine().getStatusCode());
80                 return;
81             }
82             final File file = getMapFile(geocode, prefix, true);
83             if (LocalStorage.saveEntityToFile(httpResponse, file)) {
84                 // Delete image if it has no contents
85                 final long fileSize = file.length();
86                 if (fileSize < MIN_MAP_IMAGE_BYTES) {
87                     FileUtils.deleteIgnoringFailure(file);
88                 }
89             }
90         }

94         }

96         public static void downloadMaps(final Geocache cache) {
97             if ((!Settings.isStoreOfflineMaps() && !Settings.isStoreOfflineWpMaps()) || StringUtils.isBlank(cache.getGeocode())) {
98                 return;
99             }
100             int edge = guessMaxDisplaySide();
101     
102             if (Settings.isStoreOfflineMaps() && cache.getCoords() != null) {
103                 storeCachePreviewMap(cache);
104                 storeCacheStaticMap(cache, edge, false);
105             }
106     
107             // clean old and download static maps for waypoints if one is missing
108             if (Settings.isStoreOfflineWpMaps()) {
109                 for (final Waypoint waypoint : cache.getWaypoints()) {
110                     if (!hasAllStaticMapsForWaypoint(cache.getGeocode(), waypoint)) {
111                         refreshAllWpStaticMaps(cache, edge);
112                     }
113                 }
114     
115             }
116         }

138         private static void storeWaypointStaticMap(final String geocode, final int edge, final Waypoint waypoint, final boolean waitForResult) {
139             if (geocode == null) {
140                 Log.e("storeWaypointStaticMap - missing input parameter geocode");
141                 return;
142             }
143             if (waypoint == null) {
144                 Log.e("storeWaypointStaticMap - missing input parameter waypoint");
145                 return;
146             }
147             if (waypoint.getCoords() == null) {
148                 return;
149             }
150             String wpLatlonMap = waypoint.getCoords().format(Format.LAT_LON_DECDEGREE_COMMA);
151             String wpMarkerUrl = getWpMarkerUrl(waypoint);
152             if (!hasAllStaticMapsForWaypoint(geocode, waypoint)) {
153                 // download map images in separate background thread for higher performance
154                 downloadMaps(geocode, wpMarkerUrl, WAYPOINT_PREFIX + waypoint.getId() + '_' + waypoint.getStaticMapsHashcode() + "_", wpLatlonMap, edge, null, waitForResult);
155             }
156         }

163         private static void storeCacheStaticMap(final Geocache cache, final int edge, final boolean waitForResult) {
164             final String latlonMap = cache.getCoords().format(Format.LAT_LON_DECDEGREE_COMMA);
165             final Parameters waypoints = new Parameters();
166             for (final Waypoint waypoint : cache.getWaypoints()) {
167                 if (waypoint.getCoords() == null) {
168                     continue;
169                 }
170                 final String wpMarkerUrl = getWpMarkerUrl(waypoint);
171                 waypoints.put("markers", "icon:" + wpMarkerUrl + '|' + waypoint.getCoords().format(Format.LAT_LON_DECDEGREE_COMMA));
172             }
173             // download map images in separate background thread for higher performance
174             final String cacheMarkerUrl = getCacheMarkerUrl(cache);
175             downloadMaps(cache.getGeocode(), cacheMarkerUrl, "", latlonMap, edge, waypoints, waitForResult);
176         }

191         private static void downloadMaps(final String geocode, final String markerUrl, final String prefix, final String latlonMap, final int edge,
192                 final Parameters waypoints, final boolean waitForResult) {
193             if (waitForResult) {
194                 downloadDifferentZooms(geocode, markerUrl, prefix, latlonMap, edge, waypoints);
195             }
196             else {
197                 final Runnable currentTask = new Runnable() {
198                     @Override
199                     public void run() {
200                         downloadDifferentZooms(geocode, markerUrl, prefix, latlonMap, edge, waypoints);
201                     }
202                 };
203                 try {
204                     POOL.add(currentTask, 20, TimeUnit.SECONDS);
205                 } catch (InterruptedException e) {
206                     Log.e("StaticMapsProvider.downloadMaps error adding task", e);
207                 }
208             }
209         }

211         private static String getCacheMarkerUrl(final Geocache cache) {
212             StringBuilder url = new StringBuilder(MARKERS_URL);
213             url.append("marker_cache_").append(cache.getType().id);
214             if (cache.isFound()) {
215                 url.append("_found");
216             } else if (cache.isDisabled()) {
217                 url.append("_disabled");
218             }
219             url.append(".png");
220             return url.toString();
221         }

228         public static void removeWpStaticMaps(final Waypoint waypoint, final String geocode) {
229             if (waypoint == null) {
230                 return;
231             }
232             int waypointId = waypoint.getId();
233             int waypointMapHash = waypoint.getStaticMapsHashcode();
234             for (int level = 1; level <= MAPS_LEVEL_MAX; level++) {
235                 final File mapFile = StaticMapsProvider.getMapFile(geocode, WAYPOINT_PREFIX + waypointId + "_" + waypointMapHash + '_' + level, false);
236                 if (!FileUtils.delete(mapFile)) {
237                     Log.e("StaticMapsProvider.removeWpStaticMaps failed for " + mapFile.getAbsolutePath());
238                 }
239             }
240         }

242         /**
243          * Check if at least one map file exists for the given cache.
244          *
245          * @param cache
246          * @return <code>true</code> if at least one map file exists; <code>false</code> otherwise
247          */
248         public static boolean hasStaticMap(@NonNull final Geocache cache) {
249            final String geocode = cache.getGeocode();
250             if (StringUtils.isBlank(geocode)) {
251                 return false;
252             }
253             for (int level = 1; level <= MAPS_LEVEL_MAX; level++) {
254                 File mapFile = StaticMapsProvider.getMapFile(geocode, String.valueOf(level), false);
255                 if (mapFile.exists()) {
256                     return true;
257                 }
258             }
259             return false;
260         }

272             for (int level = 1; level <= MAPS_LEVEL_MAX; level++) {
273                 File mapFile = StaticMapsProvider.getMapFile(geocode, WAYPOINT_PREFIX + waypointId + "_" + waypointMapHash + "_" + level, false);
274                 if (mapFile.exists()) {
275                     return true;
276                 }
277             }

281         /**
282          * Checks if all map files exist for the given geocode and waypoint ID.
283          *
284          * @param geocode
285          * @param waypoint
286          * @return <code>true</code> if all map files exist; <code>false</code> otherwise
287          */
288         public static boolean hasAllStaticMapsForWaypoint(final String geocode, final Waypoint waypoint) {
289             int waypointId = waypoint.getId();
290             int waypointMapHash = waypoint.getStaticMapsHashcode();
291             for (int level = 1; level <= MAPS_LEVEL_MAX; level++) {
292                 File mapFile = StaticMapsProvider.getMapFile(geocode, WAYPOINT_PREFIX + waypointId + "_" + waypointMapHash + "_" + level, false);
293                 boolean mapExists = mapFile.exists();
294                 if (!mapExists) {
295                     return false;
296                 }
297             }
298             return true;
299         }
