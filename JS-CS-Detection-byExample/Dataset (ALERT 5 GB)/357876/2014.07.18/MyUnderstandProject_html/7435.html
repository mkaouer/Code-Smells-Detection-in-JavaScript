186         private final static String SEQUENCE_GCID = "0123456789ABCDEFGHJKMNPQRTVWXYZ";
187         private final static long GC_BASE31 = 31;
188         private final static long GC_BASE16 = 16;

190         /**
191          * Convert GCCode (geocode) to (old) GCIds
192          *
193          * Based on http://www.geoclub.de/viewtopic.php?f=111&t=54859&start=40
194          * see http://support.groundspeak.com/index.php?pg=kb.printer.friendly&id=1#p221
195          */
196         public static long gccodeToGCId(final String gccode) {
197             long base = GC_BASE31;
198             final String geocodeWO = gccode.substring(2).toUpperCase(Locale.US);
199     
200             if ((geocodeWO.length() < 4) || (geocodeWO.length() == 4 && SEQUENCE_GCID.indexOf(geocodeWO.charAt(0)) < 16)) {
201                 base = GC_BASE16;
202             }
203     
204             long gcid = 0;
205             for (int p = 0; p < geocodeWO.length(); p++) {
206                 gcid = base * gcid + SEQUENCE_GCID.indexOf(geocodeWO.charAt(p));
207             }
208     
209             if (base == GC_BASE31) {
210                 gcid += Math.pow(16, 4) - 16 * Math.pow(31, 3);
211             }
212             return gcid;
213         }
