82       {
83         final Intent mwmIntent = new Intent(Const.ACTION_MWM_REQUEST);
84     
85         // url
86         final String mwmUrl = createMwmUrl(context, mTitle, mZoomLevel, mPoints).toString();

92         // pick point
93         mwmIntent.putExtra(Const.EXTRA_PICK_POINT, mPickPoint);
94         // custom button name
95         mwmIntent.putExtra(Const.EXTRA_CUSTOM_BUTTON_NAME, mCustomButtonName);
96     
97         final boolean hasIntent = mPendingIntent != null;
98         mwmIntent.putExtra(Const.EXTRA_HAS_PENDING_INTENT, hasIntent);
99         if (hasIntent)
100           mwmIntent.putExtra(Const.EXTRA_CALLER_PENDING_INTENT, mPendingIntent);
101     
102         addCommonExtras(context, mwmIntent);
103     
104         return mwmIntent;
105       }

119       private static StringBuilder createMwmUrl(Context context, String title, double zoomLevel, List<MWMPoint> points)
120       {
121         final StringBuilder urlBuilder = new StringBuilder("mapswithme://map?");
122         // version
123         urlBuilder.append("v=").append(Const.API_VERSION).append("&");
124         // back url, always not null
125         urlBuilder.append("backurl=").append(getCallbackAction(context)).append("&");
126         // title
127         appendIfNotNull(urlBuilder, "appname", title);
128         // zoom
129         appendIfNotNull(urlBuilder, "z", isValidZoomLevel(zoomLevel) ? String.valueOf(zoomLevel) : null);
130     
131         // points
132         for (final MWMPoint point : points)
133         {
134           if (point != null)
135           {
136             urlBuilder.append("ll=").append(String.format(Locale.US, "%f,%f&", point.getLat(), point.getLon()));
137     
138             appendIfNotNull(urlBuilder, "n", point.getName());
139             appendIfNotNull(urlBuilder, "id", point.getId());
140           }
141         }
142     
143         return urlBuilder;
144       }

160       private static StringBuilder appendIfNotNull(StringBuilder builder, String key, String value)
161       {
162         if (value != null)
163           builder.append(key).append("=").append(Uri.encode(value)).append("&");
164     
165         return builder;
166       }
