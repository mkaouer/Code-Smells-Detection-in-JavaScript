88         final ConnectableObservable<IGeoData> worker = new ConnectableObservable<IGeoData>(this) {
89             private int debugSessionCounter = 0;
90     
91             private final Object lock = new Object();
92             private int count = 0;
93     
94             final private GpsStatus.Listener gpsStatusListener = new GpsStatusListener();
95             final private Listener networkListener = new Listener(LocationManager.NETWORK_PROVIDER, netLocation);
96             final private Listener gpsListener = new Listener(LocationManager.GPS_PROVIDER, gpsLocation);
97     
98             @Override
99             public Subscription connect() {
100                 final CompositeSubscription subscription = new CompositeSubscription();
101                 AndroidSchedulers.handlerThread(handlerThread.getHandler()).schedule(new Action1<Inner>() {
102                     @Override
103                     public void call(final Inner inner) {
104                         synchronized(lock) {
105                             if (count++ == 0) {
106                                 Log.d("GeoDataProvider: starting the GPS and network listeners" + " (" + ++debugSessionCounter + ")");
107                                 geoManager.addGpsStatusListener(gpsStatusListener);
108                                 for (final Listener listener : new Listener[] { networkListener, gpsListener }) {
109                                     try {
110                                         geoManager.requestLocationUpdates(listener.locationProvider, 0, 0, listener);
111                                     } catch (final Exception e) {
112                                         Log.w("There is no location provider " + listener.locationProvider);
113                                     }
114                                 }
115                             }
116                         }
117     
118                         subscription.add(Subscriptions.create(new Action0() {
119                             @Override
120                             public void call() {
121                                 AndroidSchedulers.handlerThread(handlerThread.getHandler()).schedule(new Action1<Inner>() {
122                                     @Override
123                                     public void call(final Inner inner) {
124                                         synchronized (lock) {
125                                             if (--count == 0) {
126                                                 Log.d("GeoDataProvider: stopping the GPS and network listeners" + " (" + debugSessionCounter + ")");
127                                                 geoManager.removeUpdates(networkListener);
128                                                 geoManager.removeUpdates(gpsListener);
129                                                 geoManager.removeGpsStatusListener(gpsStatusListener);
130                                             }
131                                         }
132                                     }
133                                 }, 2500, TimeUnit.MILLISECONDS);
134                             }
135                         }));
136                     }
137                 });
138                 return subscription;
139             }
140         };

142         private IGeoData findInitialLocation() {
143             final Location initialLocation = new Location(LAST_LOCATION_PSEUDO_PROVIDER);
144             try {
145                 // Try to find a sensible initial location from the last locations known to Android.
146                 final Location lastGpsLocation = geoManager.getLastKnownLocation(LocationManager.GPS_PROVIDER);
147                 final Location lastNetworkLocation = geoManager.getLastKnownLocation(LocationManager.NETWORK_PROVIDER);
148     
149                 // If both providers are non-null, take the most recent one
150                 if (lastGpsLocation != null && lastNetworkLocation != null) {
151                     if (lastGpsLocation.getTime() >= lastNetworkLocation.getTime()) {
152                         copyCoords(initialLocation, lastGpsLocation);
153                     } else {
154                         copyCoords(initialLocation, lastNetworkLocation);
155                     }
156                 } else if (lastGpsLocation != null) {
157                     copyCoords(initialLocation, lastGpsLocation);
158                 } else if (lastNetworkLocation != null) {
159                     copyCoords(initialLocation, lastNetworkLocation);
160                 } else {
161                     Log.i("GeoDataProvider: no last known location available");
162                 }
163             } catch (final Exception e) {
164                 // This error is non-fatal as its only consequence is that we will start with a dummy location
165                 // instead of a previously known one.
166                 Log.e("GeoDataProvider: error when retrieving last known location", e);
167             }
168             // Start with an historical GeoData just in case someone queries it before we get
169             // a chance to get any information.
170             return new GeoData(initialLocation, false, 0, 0, true);
171         }

209         private final class GpsStatusListener implements GpsStatus.Listener {
210     
211             @Override
212             public void onGpsStatusChanged(final int event) {
213                 boolean changed = false;
214                 switch (event) {
215                     case GpsStatus.GPS_EVENT_FIRST_FIX:
216                     case GpsStatus.GPS_EVENT_SATELLITE_STATUS: {
217                         final GpsStatus status = geoManager.getGpsStatus(null);
218                         int visible = 0;
219                         int fixed = 0;
220                         for (final GpsSatellite satellite : status.getSatellites()) {
221                             if (satellite.usedInFix()) {
222                                 fixed++;
223                             }
224                             visible++;
225                         }
226                         if (visible != satellitesVisible || fixed != satellitesFixed) {
227                             satellitesVisible = visible;
228                             satellitesFixed = fixed;
229                             changed = true;
230                         }
231                         break;
232                     }
233                     case GpsStatus.GPS_EVENT_STARTED:
234                         if (!gpsEnabled) {
235                             gpsEnabled = true;
236                             changed = true;
237                         }
238                         break;
239                     case GpsStatus.GPS_EVENT_STOPPED:
240                         if (gpsEnabled) {
241                             gpsEnabled = false;
242                             satellitesFixed = 0;
243                             satellitesVisible = 0;
244                             changed = true;
245                         }
246                         break;
247                     default:
248                         throw new IllegalStateException();
249                 }
250     
251                 if (changed) {
252                     selectBest();
253                 }
254             }
255         }

257         private LocationData best() {
258             if (gpsLocation.isRecent() || !netLocation.isValid()) {
259                 return gpsLocation.isValid() ? gpsLocation : null;
260             }
261             if (!gpsLocation.isValid()) {
262                 return netLocation;
263             }
264             return gpsLocation.timestamp > netLocation.timestamp ? gpsLocation : netLocation;
265         }
266     
267         private void selectBest() {
268             assign(best());
269         }
270     
271         private void assign(final LocationData locationData) {
272             if (locationData == null) {
273                 return;
274             }
275     
276             // We do not necessarily get signalled when satellites go to 0/0.
277             final int visible = gpsLocation.isRecent() ? satellitesVisible : 0;
278             final boolean pseudoLocation = StringUtils.equals(locationData.location.getProvider(), LAST_LOCATION_PSEUDO_PROVIDER);
279             final IGeoData current = new GeoData(locationData.location, gpsEnabled, visible, satellitesFixed, pseudoLocation);
280             subject.onNext(current);
281         }
