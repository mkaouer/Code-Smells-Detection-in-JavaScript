145         private class SearchFilesThread extends Thread {
146     
147             private final FileListSelector selector = new FileListSelector();
148     
149             public void notifyEnd() {
150                 selector.setShouldEnd(true);
151             }
152     
153             @Override
154             public void run() {
155                 final List<File> list = new ArrayList<File>();
156     
157                 try {
158                     if (Environment.getExternalStorageState().equals(Environment.MEDIA_MOUNTED)) {
159                         boolean loaded = false;
160                         for (final File dir : getExistingBaseFolders()) {
161                             FileUtils.listDir(list, dir, selector, changeWaitDialogHandler);
162                             if (!list.isEmpty()) {
163                                 loaded = true;
164                                 break;
165                             }
166                         }
167                         if (!loaded) {
168                             changeWaitDialogHandler.sendMessage(Message.obtain(changeWaitDialogHandler, MSG_SEARCH_WHOLE_SD_CARD, Environment.getExternalStorageDirectory().getName()));
169                             listDirs(list, LocalStorage.getStorages(), selector, changeWaitDialogHandler);
170                         }
171                     } else {
172                         Log.w("No external media mounted.");
173                     }
174                 } catch (Exception e) {
175                     Log.e("AbstractFileListActivity.loadFiles.run", e);
176                 }
177     
178                 changeWaitDialogHandler.sendMessage(Message.obtain(changeWaitDialogHandler, 0, "loaded directories"));
179     
180                 files.addAll(list);
181                 Collections.sort(files, new Comparator<File>() {
182     
183                     @Override
184                     public int compare(File lhs, File rhs) {
185                         return lhs.getName().compareToIgnoreCase(rhs.getName());
186                     }
187                 });
188     
189                 loadFilesHandler.sendMessage(Message.obtain(loadFilesHandler));
190             }
191     
192             private void listDirs(List<File> list, List<File> directories, FileListSelector selector, Handler feedbackHandler) {
193                 for (final File dir : directories) {
194                     FileUtils.listDir(list, dir, selector, feedbackHandler);
195                 }
196             }
197         }

199         /**
200          * Check if a filename belongs to the AbstractFileListActivity. This implementation checks for file extensions.
201          * Subclasses may override this method to filter out specific files.
202          *
203          * @param filename
204          * @return <code>true</code> if the filename belongs to the list
205          */
206         protected boolean filenameBelongsToList(final String filename) {
207             for (String ext : extensions) {
208                 if (StringUtils.endsWithIgnoreCase(filename, ext)) {
209                     return true;
210                 }
211             }
212             return false;
213         }
214     
215         protected List<File> getExistingBaseFolders() {
216             ArrayList<File> result = new ArrayList<File>();
217             for (final File dir : getBaseFolders()) {
218                 if (dir.exists() && dir.isDirectory()) {
219                     result.add(dir);
220                 }
221             }
222             return result;
223         }

233         private void setExtensions(final String[] extensionsIn) {
234             extensions = extensionsIn;
235             for (int i = 0; i < extensions.length; i++) {
236                 final String extension = extensions[i];
237                 if (StringUtils.isEmpty(extension) || extension.charAt(0) != '.') {
238                     extensions[i] = "." + extension;
239                 }
240             }
241         }
