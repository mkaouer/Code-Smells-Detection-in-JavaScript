99          */
100         protected String parseDescription() {
101             final StringBuilder description = new StringBuilder();
102             description.append(getUrl());
103             if (getShortDesc().length() > 0) {
104                 // remove images in short description
105                 final Spanned spanned = Html.fromHtml(getShortDesc(), null, null);
106                 String text = spanned.toString();
107                 final ImageSpan[] spans = spanned.getSpans(0, spanned.length(), ImageSpan.class);
108                 for (int i = spans.length - 1; i >= 0; i--) {
109                     text = text.substring(0, spanned.getSpanStart(spans[i])) + text.substring(spanned.getSpanEnd(spans[i]));
110                 }
111                 if (text.length() > 0) {
112                     description.append("\n\n");
113                     description.append(text);
114                 }
115             }
116     
117             if (getPersonalNote().length() > 0) {
118                 description.append("\n\n").append(Html.fromHtml(getPersonalNote()).toString());
119             }
120     
121             return description.toString();
122         }
