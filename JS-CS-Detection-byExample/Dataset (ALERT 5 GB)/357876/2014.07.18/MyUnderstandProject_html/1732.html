62         static {
63             final HashMap<String, CacheType> mappingId = new HashMap<String, CacheType>();
64             final HashMap<String, CacheType> mappingPattern = new HashMap<String, CacheType>();
65             final HashMap<String, CacheType> mappingGuid = new HashMap<String, CacheType>();
66             for (CacheType ct : values()) {
67                 mappingId.put(ct.id, ct);
68                 mappingPattern.put(ct.pattern.toLowerCase(Locale.US), ct);
69                 mappingGuid.put(ct.guid, ct);
70             }
71             // add old mystery type for GPX file compatibility
72             mappingPattern.put("Mystery Cache".toLowerCase(Locale.US), MYSTERY);
73     
74             FIND_BY_ID = Collections.unmodifiableMap(mappingId);
75             FIND_BY_PATTERN = Collections.unmodifiableMap(mappingPattern);
76             FIND_BY_GUID = Collections.unmodifiableMap(mappingGuid);
77         }
