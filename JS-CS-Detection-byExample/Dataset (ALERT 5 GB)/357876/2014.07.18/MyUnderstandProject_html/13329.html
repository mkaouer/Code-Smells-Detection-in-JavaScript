43         /**
44          * Display a list of caches / waypoints in Locus
45          *
46          * @param objectsToShow
47          *            which caches/waypoints to show
48          * @param withCacheWaypoints
49          *            Whether to give waypoints of caches to Locus or not
50          * @param activity
51          */
52         protected static boolean showInLocus(final List<?> objectsToShow, final boolean withCacheWaypoints, final boolean export,
53                 final Activity activity) {
54             if (objectsToShow == null || objectsToShow.isEmpty()) {
55                 return false;
56             }
57     
58             final boolean withCacheDetails = objectsToShow.size() < 200;
59             final PointsData pd = new PointsData("c:geo");
60             for (Object o : objectsToShow) {
61                 Point p = null;
62                 // get icon and Point
63                 if (o instanceof Geocache) {
64                     p = getCachePoint((Geocache) o, withCacheWaypoints, withCacheDetails);
65                 } else if (o instanceof Waypoint) {
66                     p = getWaypointPoint((Waypoint) o);
67                 }
68                 if (p != null) {
69                     pd.addPoint(p);
70                 }
71             }
72     
73             if (pd.getPoints().isEmpty()) {
74                 return false;
75             }
76     
77             if (pd.getPoints().size() <= 1000) {
78                 DisplayData.sendData(activity, pd, export);
79             } else {
80                 final ArrayList<PointsData> data = new ArrayList<PointsData>();
81                 data.add(pd);
82                 DisplayData.sendDataCursor(activity, data,
83                         "content://" + LocusDataStorageProvider.class.getCanonicalName().toLowerCase(Locale.US),
84                         export);
85             }
86     
87             return true;
88         }

90         /**
91          * This method constructs a <code>Point</code> for displaying in Locus
92          *
93          * @param cache
94          * @param withWaypoints
95          *            whether to give waypoints to Locus or not
96          * @param withCacheDetails
97          *            whether to give cache details (description, hint) to Locus or not
98          *            should be false for all if more then 200 Caches are transferred
99          * @return null, when the <code>Point</code> could not be constructed
100          */
101         private static Point getCachePoint(Geocache cache, boolean withWaypoints, boolean withCacheDetails) {
102             if (cache == null || cache.getCoords() == null) {
103                 return null;
104             }
105     
106             // create one simple point with location
107             final Location loc = new Location("cgeo");
108             loc.setLatitude(cache.getCoords().getLatitude());
109             loc.setLongitude(cache.getCoords().getLongitude());
110     
111             final Point p = new Point(cache.getName(), loc);
112             final PointGeocachingData pg = new PointGeocachingData();
113             p.setGeocachingData(pg);
114     
115             // set data in Locus' cache
116             pg.cacheID = cache.getGeocode();
117             pg.available = !cache.isDisabled();
118             pg.archived = cache.isArchived();
119             pg.premiumOnly = cache.isPremiumMembersOnly();
120             pg.name = cache.getName();
121             pg.placedBy = cache.getOwnerDisplayName();
122             final Date hiddenDate = cache.getHiddenDate();
123             if (hiddenDate != null) {
124                 pg.hidden = ISO8601DATE.format(hiddenDate);
125             }
126             int locusId = toLocusType(cache.getType());
127             if (locusId != NO_LOCUS_ID) {
128                 pg.type = locusId;
129             }
130             locusId = toLocusSize(cache.getSize());
131             if (locusId != NO_LOCUS_ID) {
132                 pg.container = locusId;
133             }
134             if (cache.getDifficulty() > 0) {
135                 pg.difficulty = cache.getDifficulty();
136             }
137             if (cache.getTerrain() > 0) {
138                 pg.terrain = cache.getTerrain();
139             }
140             pg.found = cache.isFound();
141     
142             if (withWaypoints && cache.hasWaypoints()) {
143                 pg.waypoints = new ArrayList<PointGeocachingDataWaypoint>();
144                 for (Waypoint waypoint : cache.getWaypoints()) {
145                     if (waypoint == null || waypoint.getCoords() == null) {
146                         continue;
147                     }
148                     PointGeocachingDataWaypoint wp = new PointGeocachingDataWaypoint();
149                     wp.code = waypoint.getGeocode();
150                     wp.name = waypoint.getName();
151                     String locusWpId = toLocusWaypoint(waypoint.getWaypointType());
152                     if (locusWpId != null) {
153                         wp.type = locusWpId;
154                     }
155                     wp.lat = waypoint.getCoords().getLatitude();
156                     wp.lon = waypoint.getCoords().getLongitude();
157                     pg.waypoints.add(wp);
158                 }
159             }
160     
161             // Other properties of caches. When there are many caches to be displayed
162             // in Locus, using these properties can lead to Exceptions in Locus.
163             // Should not be used if caches count > 200
164     
165             if (withCacheDetails) {
166                 pg.shortDescription = cache.getShortDescription();
167                 pg.longDescription = cache.getDescription();
168                 pg.encodedHints = cache.getHint();
169             }
170     
171             return p;
172         }

174         /**
175          * This method constructs a <code>Point</code> for displaying in Locus
176          *
177          * @param waypoint
178          * @return null, when the <code>Point</code> could not be constructed
179          */
180         private static Point getWaypointPoint(Waypoint waypoint) {
181             if (waypoint == null || waypoint.getCoords() == null) {
182                 return null;
183             }
184     
185             // create one simple point with location
186             final Location loc = new Location("cgeo");
187             loc.setLatitude(waypoint.getCoords().getLatitude());
188             loc.setLongitude(waypoint.getCoords().getLongitude());
189     
190             final Point p = new Point(waypoint.getName(), loc);
191             p.setDescription("<a href=\"" + waypoint.getUrl() + "\">"
192                     + waypoint.getGeocode() + "</a>");
193     
194             return p;
195         }

215                 case CITO:

217                 case WEBCAM:

219                 case VIRTUAL:
