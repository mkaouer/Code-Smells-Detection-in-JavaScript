82         /**
83          * Load images into a view.
84          *
85          * @param parentView a view to load the images into
86          * @param images the images to load
87          * @param offline <tt>true</tt> if the images must be stored for offline use
88          * @return a subscription which, when unsubscribed, interrupts the loading and clears up resources
89          */
90         public Subscription loadImages(final View parentView, final Collection<Image> images, final boolean offline) {
91             // Start with a fresh subscription because of this method can be called several times if the
92             // englobing activity is stopped/restarted.
93             final CompositeSubscription subscriptions = new CompositeSubscription(Subscriptions.create(new Action0() {
94                 @Override
95                 public void call() {
96                     removeAllViews();
97                 }
98             }));
99     
100             imagesView = (LinearLayout) parentView.findViewById(R.id.spoiler_list);
101     
102             final HtmlImage imgGetter = new HtmlImage(geocode, true, offline ? StoredList.STANDARD_LIST_ID : StoredList.TEMPORARY_LIST_ID, false);
103     
104             for (final Image img : images) {
105                 final LinearLayout rowView = (LinearLayout) inflater.inflate(R.layout.cache_image_item, null);
106                 assert(rowView != null);
107     
108                 if (StringUtils.isNotBlank(img.getTitle())) {
109                     ((TextView) rowView.findViewById(R.id.title)).setText(Html.fromHtml(img.getTitle()));
110                     rowView.findViewById(R.id.titleLayout).setVisibility(View.VISIBLE);
111                 }
112     
113                 if (StringUtils.isNotBlank(img.getDescription())) {
114                     final TextView descView = (TextView) rowView.findViewById(R.id.description);
115                     descView.setText(Html.fromHtml(img.getDescription()), TextView.BufferType.SPANNABLE);
116                     descView.setVisibility(View.VISIBLE);
117                 }
118     
119                 final ImageView imageView = (ImageView) inflater.inflate(R.layout.image_item, null);
120                 assert(imageView != null);
121                 subscriptions.add(AndroidObservable.bindActivity(activity, imgGetter.fetchDrawable(img.getUrl())).subscribe(new Action1<BitmapDrawable>() {
122                     @Override
123                     public void call(final BitmapDrawable image) {
124                         display(imageView, image, img, rowView);
125                     }
126                 }));
127                 rowView.addView(imageView);
128                 imagesView.addView(rowView);
129             }
130     
131             return subscriptions;
132         }

134         private void display(final ImageView imageView, final BitmapDrawable image, final Image img, final LinearLayout view) {
135             if (image != null) {
136                 bitmaps.add(image.getBitmap());
137     
138                 final Rect bounds = image.getBounds();
139     
140                 imageView.setImageResource(R.drawable.image_not_loaded);
141                 imageView.setClickable(true);
142                 imageView.setOnClickListener(new View.OnClickListener() {
143                     @Override
144                     public void onClick(View arg0) {
145                         viewImageInStandardApp(img, image);
146                     }
147                 });
148                 activity.registerForContextMenu(imageView);
149                 imageView.setImageDrawable(image);
150                 imageView.setScaleType(ImageView.ScaleType.CENTER_CROP);
151                 imageView.setLayoutParams(new LinearLayout.LayoutParams(bounds.width(), bounds.height()));
152     
153                 view.findViewById(R.id.progress_bar).setVisibility(View.GONE);
154     
155                 imageView.setId(image.hashCode());
156                 images.put(imageView.getId(), img);
157     
158                 view.invalidate();
159             }
160         }

162         private void removeAllViews() {
163             for (final Bitmap b : bitmaps) {
164                 b.recycle();
165             }
166             bitmaps.clear();
167             images.clear();
168     
169             imagesView.removeAllViews();
170         }

197         private static String mimeTypeForUrl(final String url) {
198             return StringUtils.defaultString(MimeTypeMap.getSingleton().getMimeTypeFromExtension(MimeTypeMap.getFileExtensionFromUrl(url)), "image/*");
199         }

201         private static File saveToTemporaryJPGFile(final BitmapDrawable image) throws FileNotFoundException {
202             final File file = LocalStorage.getStorageFile(null, "temp.jpg", false, true);
203             BufferedOutputStream stream = null;
204             try {
205                 stream = new BufferedOutputStream(new FileOutputStream(file));
206                 image.getBitmap().compress(CompressFormat.JPEG, 100, stream);
207             } finally {
208                 IOUtils.closeQuietly(stream);
209             }
210             file.deleteOnExit();
211             return file;
212         }

214         private void viewImageInStandardApp(final Image img, final BitmapDrawable image) {
215             try {
216                 final Intent intent = new Intent().setAction(android.content.Intent.ACTION_VIEW);
217                 final File file = img.isLocalFile() ? img.localFile() : LocalStorage.getStorageFile(geocode, img.getUrl(), true, true);
218                 if (file.exists()) {
219                     intent.setDataAndType(Uri.fromFile(file), mimeTypeForUrl(img.getUrl()));
220                 } else {
221                     intent.setDataAndType(Uri.fromFile(saveToTemporaryJPGFile(image)), "image/jpeg");
222                 }
223                 activity.startActivity(intent);
224             } catch (Exception e) {
225                 Log.e("ImagesList.viewImageInStandardApp", e);
226             }
227         }
