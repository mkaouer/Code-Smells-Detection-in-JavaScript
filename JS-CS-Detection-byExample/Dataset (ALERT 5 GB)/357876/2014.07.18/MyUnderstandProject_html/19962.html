210             // map GPX-Attribute-Id to baseName
211             public static String getBaseName(final int id) {
212                 if (id < 0) {
213                     return null;
214                 }
215                 // get String out of array
216                 if (CACHE_ATTRIBUTES.length <= id) {
217                     return null;
218                 }
219                 final int stringId = CACHE_ATTRIBUTES[id];
220                 if (stringId == -1) {
221                     return null; // id not found
222                 }
223                 // get text for string
224                 String stringName;
225                 try {
226                     stringName = CgeoApplication.getInstance().getResources().getResourceName(stringId);
227                 } catch (final NullPointerException e) {
228                     return null;
229                 }
230                 if (stringName == null) {
231                     return null;
232                 }
233                 // cut out baseName
234                 final MatcherWrapper m = new MatcherWrapper(BASENAME_PATTERN, stringName);
235                 if (!m.matches()) {
236                     return null;
237                 }
238                 return m.group(1);
239             }

241             // @return  baseName + "_yes" or "_no" e.g. "food_no" or "uv_yes"
242             public static String getInternalId(final int attributeId, final boolean active) {
243                 final String baseName = CacheAttributeTranslator.getBaseName(attributeId);
244                 if (baseName == null) {

257         static Date parseDate(String inputUntrimmed) throws ParseException {
258             String input = inputUntrimmed.trim();
259             // remove milliseconds to reduce number of needed patterns
260             final MatcherWrapper matcher = new MatcherWrapper(PATTERN_MILLISECONDS, input);
261             input = matcher.replaceFirst("");
262             if (input.contains("Z")) {
263                 return formatSimpleZ.parse(input);
264             }
265             if (StringUtils.countMatches(input, ":") == 3) {
266                 final String removeColon = input.substring(0, input.length() - 3) + input.substring(input.length() - 2);
267                 return formatTimezone.parse(removeColon);
268             }
269             return formatSimple.parse(input);
270         }

355                 private void addWaypointToCache() {
356                     fixCache(cache);
357     
358                     if (cache.getName().length() > 2 || StringUtils.isNotBlank(parentCacheCode)) {
359                         if (StringUtils.isBlank(parentCacheCode)) {
360                             if (StringUtils.containsIgnoreCase(scriptUrl, "extremcaching")) {
361                                 parentCacheCode = cache.getName().substring(2);
362                             }
363                             else {
364                                 parentCacheCode = "GC" + cache.getName().substring(2).toUpperCase(Locale.US);
365                             }
366                         }
367                         // lookup cache for waypoint in already parsed caches
368                         final Geocache cacheForWaypoint = DataStore.loadCache(parentCacheCode, LoadFlags.LOAD_CACHE_OR_DB);
369                         if (cacheForWaypoint != null) {
370                             final Waypoint waypoint = new Waypoint(cache.getShortDescription(), convertWaypointSym2Type(sym), false);
371                             if (wptUserDefined) {
372                                 waypoint.setUserDefined();
373                             }
374                             waypoint.setId(-1);
375                             waypoint.setGeocode(parentCacheCode);
376                             waypoint.setPrefix(cacheForWaypoint.getWaypointPrefix(cache.getName()));
377                             waypoint.setLookup("---");
378                             // there is no lookup code in gpx file
379                             waypoint.setCoords(cache.getCoords());
380                             waypoint.setNote(cache.getDescription());
381                             waypoint.setVisited(wptVisited);
382                             final ArrayList<Waypoint> mergedWayPoints = new ArrayList<Waypoint>();
383                             mergedWayPoints.addAll(cacheForWaypoint.getWaypoints());
384     
385                             final ArrayList<Waypoint> newPoints = new ArrayList<Waypoint>();
386                             newPoints.add(waypoint);
387                             Waypoint.mergeWayPoints(newPoints, mergedWayPoints, true);
388                             cacheForWaypoint.setWaypoints(newPoints, false);
389                             DataStore.saveCache(cacheForWaypoint, EnumSet.of(SaveFlag.SAVE_DB));
390                             showProgressMessage(progressHandler, progressStream.getProgress());
391                         }
392                     }

938         static WaypointType convertWaypointSym2Type(final String sym) {
939             if ("parking area".equalsIgnoreCase(sym)) {
940                 return WaypointType.PARKING;
941             }
942             if ("stages of a multicache".equalsIgnoreCase(sym)) {
943                 return WaypointType.STAGE;
944             }
945             if ("question to answer".equalsIgnoreCase(sym)) {
946                 return WaypointType.PUZZLE;
947             }
948             if ("trailhead".equalsIgnoreCase(sym)) {
949                 return WaypointType.TRAILHEAD;
950             }
951             if ("final location".equalsIgnoreCase(sym)) {
952                 return WaypointType.FINAL;
953             }
954             // renamed waypoint types
955             if ("Physical Stage".equalsIgnoreCase(sym)) {
956                 return WaypointType.STAGE;
957             }
958             if ("Virtual Stage".equalsIgnoreCase(sym)) {
959                 return WaypointType.PUZZLE;
960             }
961             // this is not fully correct, but lets also look for localized waypoint types
962             for (final WaypointType waypointType : WaypointType.ALL_TYPES_EXCEPT_OWN_AND_ORIGINAL) {
963                 final String localized = waypointType.getL10n();
964                 if (StringUtils.isNotEmpty(localized)) {
965                     if (localized.equalsIgnoreCase(sym)) {
966                         return waypointType;
967                     }
968                 }
969             }
970             return WaypointType.WAYPOINT;
971         }

973         private void findGeoCode(final String input) {
974             if (input == null || StringUtils.isNotBlank(cache.getGeocode())) {
975                 return;
976             }
977             final String trimmed = input.trim();
978             final MatcherWrapper matcherGeocode = new MatcherWrapper(PATTERN_GEOCODE, trimmed);
979             if (matcherGeocode.find()) {
980                 final String geocode = matcherGeocode.group(1);
981                 // a geocode should not be part of a word
982                 if (geocode.length() == trimmed.length() || Character.isWhitespace(trimmed.charAt(geocode.length()))) {
983                     if (ConnectorFactory.canHandle(geocode)) {
984                         cache.setGeocode(geocode);
985                     }
986                 }
987             }
988         }

1020         private void createNoteFromGSAKUserdata() {
1021             if (StringUtils.isBlank(cache.getPersonalNote())) {

1023                 for (final String anUserData : userData) {
1024                     if (StringUtils.isNotBlank(anUserData)) {

1026                     }

1029                 if (StringUtils.isNotBlank(note)) {
1030                     cache.setPersonalNote(note);

1035         private boolean isValidForImport() {
1036             if (StringUtils.isBlank(cache.getGeocode())) {
1037                 return false;
1038             }
1039             if (cache.getCoords() == null) {
1040                 return false;
1041             }
1042             return ((type == null && sym == null)
1043                     || StringUtils.contains(type, "geocache")
1044                     || StringUtils.contains(sym, "geocache")
1045                     || StringUtils.containsIgnoreCase(sym, "waymark"));
1046         }
