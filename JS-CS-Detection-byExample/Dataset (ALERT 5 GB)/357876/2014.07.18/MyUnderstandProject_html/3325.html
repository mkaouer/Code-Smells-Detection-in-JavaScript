8     /**
9      * sorts caches by distance to given position
10      *
11      */
12     public class DistanceComparator extends AbstractCacheComparator {
13     
14         final private Geopoint coords;
15         final private List<Geocache> list;
16         private boolean cachedDistances;
17     
18         public DistanceComparator(final Geopoint coords, List<Geocache> list) {
19             this.coords = coords;
20             this.list = list;
21         }
22     
23         /**
24          * calculate all distances only once to avoid costly re-calculation of the same distance during sorting
25          */
26         private void calculateAllDistances() {
27             if (cachedDistances) {
28                 return;
29             }
30             for (final Geocache cache : list) {
31                 if (cache.getCoords() != null) {
32                     cache.setDistance(coords.distanceTo(cache.getCoords()));
33                 }
34             }
35             cachedDistances = true;
36         }
37     
38         @Override
39         protected int compareCaches(final Geocache cache1, final Geocache cache2) {
40             calculateAllDistances();
41             final Float distance1 = cache1.getDistance();
42             final Float distance2 = cache2.getDistance();
43             if (distance1 == null) {
44                 return distance2 == null ? 0 : 1;
45             }
46             return distance2 == null ? -1 : Float.compare(distance1, distance2);
47         }
