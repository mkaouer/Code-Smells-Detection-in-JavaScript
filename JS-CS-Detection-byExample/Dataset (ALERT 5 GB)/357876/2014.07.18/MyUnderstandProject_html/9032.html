37     /**
38      * <p>A utility class which helps ease integration with Barcode Scanner via {@link Intent}s. This is a simple
39      * way to invoke barcode scanning and receive the result, without any need to integrate, modify, or learn the
40      * project's source code.</p>
41      *
42      * <h2>Initiating a barcode scan</h2>
43      *
44      * <p>To integrate, create an instance of {@code IntentIntegrator} and call {@link #initiateScan()} and wait
45      * for the result in your app.</p>
46      *
47      * <p>It does require that the Barcode Scanner (or work-alike) application is installed. The
48      * {@link #initiateScan()} method will prompt the user to download the application, if needed.</p>
49      *
50      * <p>There are a few steps to using this integration. First, your {@link Activity} must implement
51      * the method {@link Activity#onActivityResult(int, int, Intent)} and include a line of code like this:</p>
52      *
53      * <pre>{@code
54      * public void onActivityResult(int requestCode, int resultCode, Intent intent) {
55      *   IntentResult scanResult = IntentIntegrator.parseActivityResult(requestCode, resultCode, intent);
56      *   if (scanResult != null) {
57      *     // handle scan result
58      *   }
59      *   // else continue with any other code you need in the method
60      *   ...
61      * }
62      * }</pre>
63      *
64      * <p>This is where you will handle a scan result.</p>
65      *
66      * <p>Second, just call this in response to a user action somewhere to begin the scan process:</p>
67      *
68      * <pre>{@code
69      * IntentIntegrator integrator = new IntentIntegrator(yourActivity);
70      * integrator.initiateScan();
71      * }</pre>
72      *
73      * <p>Note that {@link #initiateScan()} returns an {@link AlertDialog} which is non-null if the
74      * user was prompted to download the application. This lets the calling app potentially manage the dialog.
75      * In particular, ideally, the app dismisses the dialog if it's still active in its {@link Activity#onPause()}
76      * method.</p>
77      * 
78      * <p>You can use {@link #setTitle(String)} to customize the title of this download prompt dialog (or, use
79      * {@link #setTitleByID(int)} to set the title by string resource ID.) Likewise, the prompt message, and
80      * yes/no button labels can be changed.</p>
81      *
82      * <p>Finally, you can use {@link #addExtra(String, Object)} to add more parameters to the Intent used
83      * to invoke the scanner. This can be used to set additional options not directly exposed by this
84      * simplified API.</p>
85      * 
86      * <p>By default, this will only allow applications that are known to respond to this intent correctly
87      * do so. The apps that are allowed to response can be set with {@link #setTargetApplications(List)}.
88      * For example, set to {@link #TARGET_BARCODE_SCANNER_ONLY} to only target the Barcode Scanner app itself.</p>
89      *
90      * <h2>Sharing text via barcode</h2>
91      *
92      * <p>To share text, encoded as a QR Code on-screen, similarly, see {@link #shareText(CharSequence)}.</p>
93      *
94      * <p>Some code, particularly download integration, was contributed from the Anobiit application.</p>
95      *
96      * <h2>Enabling experimental barcode formats</h2>
97      *
98      * <p>Some formats are not enabled by default even when scanning with {@link #ALL_CODE_TYPES}, such as
99      * PDF417. Use {@link #initiateScan(java.util.Collection)} with
100      * a collection containing the names of formats to scan for explicitly, like "PDF_417", to use such
101      * formats.</p>
102      *
103      * @author Sean Owen
104      * @author Fred Lin
105      * @author Isaac Potoczny-Jones
106      * @author Brad Drehmer
107      * @author gcstang
108      */

237       /**
238        * Initiates a scan only for a certain set of barcode types, given as strings corresponding
239        * to their names in ZXing's {@code BarcodeFormat} class like "UPC_A". You can supply constants
240        * like {@link #PRODUCT_CODE_TYPES} for example.
241        *
242        * @return the {@link AlertDialog} that was shown to the user prompting them to download the app
243        *   if a prompt was needed, or null otherwise
244        */
245       public final AlertDialog initiateScan(Collection<String> desiredBarcodeFormats) {
246         Intent intentScan = new Intent(BS_PACKAGE + ".SCAN");
247         intentScan.addCategory(Intent.CATEGORY_DEFAULT);
248     
249         // check which types of codes to scan for
250         if (desiredBarcodeFormats != null) {
251           // set the desired barcode types
252           StringBuilder joinedByComma = new StringBuilder();
253           for (String format : desiredBarcodeFormats) {
254             if (joinedByComma.length() > 0) {
255               joinedByComma.append(',');
256             }
257             joinedByComma.append(format);
258           }
259           intentScan.putExtra("SCAN_FORMATS", joinedByComma.toString());
260         }
261     
262         String targetAppPackage = findTargetAppPackage(intentScan);
263         if (targetAppPackage == null) {
264           return showDownloadDialog();
265         }
266         intentScan.setPackage(targetAppPackage);
267         intentScan.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
268         intentScan.addFlags(Intent.FLAG_ACTIVITY_CLEAR_WHEN_TASK_RESET);
269         attachMoreExtras(intentScan);
270         startActivityForResult(intentScan, REQUEST_CODE);
271         return null;
272       }

288       private String findTargetAppPackage(Intent intent) {
289         PackageManager pm = activity.getPackageManager();
290         List<ResolveInfo> availableApps = pm.queryIntentActivities(intent, PackageManager.MATCH_DEFAULT_ONLY);
291         if (availableApps != null) {
292           for (ResolveInfo availableApp : availableApps) {
293             String packageName = availableApp.activityInfo.packageName;
294             if (targetApplications.contains(packageName)) {
295               return packageName;
296             }
297           }
298         }
299         return null;
300       }

302       private AlertDialog showDownloadDialog() {
303         AlertDialog.Builder downloadDialog = new AlertDialog.Builder(activity);
304         downloadDialog.setTitle(title);
305         downloadDialog.setMessage(message);
306         downloadDialog.setPositiveButton(buttonYes, new DialogInterface.OnClickListener() {
307           @Override
308           public void onClick(DialogInterface dialogInterface, int i) {
309             String packageName = targetApplications.get(0);
310             Uri uri = Uri.parse("market://details?id=" + packageName);
311             Intent intent = new Intent(Intent.ACTION_VIEW, uri);
312             try {
313               activity.startActivity(intent);
314             } catch (ActivityNotFoundException anfe) {
315               // Hmm, market is not installed
316               Log.w(TAG, "Google Play is not installed; cannot install " + packageName);
317             }
318           }
319         });
320         downloadDialog.setNegativeButton(buttonNo, new DialogInterface.OnClickListener() {
321           @Override
322           public void onClick(DialogInterface dialogInterface, int i) {}
323         });
324         return downloadDialog.show();
325       }

328       /**
329        * <p>Call this from your {@link Activity}'s
330        * {@link Activity#onActivityResult(int, int, Intent)} method.</p>
331        *
332        * @return null if the event handled here was not related to this class, or
333        *  else an {@link IntentResult} containing the result of the scan. If the user cancelled scanning,
334        *  the fields will be null.
335        */
336       public static IntentResult parseActivityResult(int requestCode, int resultCode, Intent intent) {
337         if (requestCode == REQUEST_CODE) {
338           if (resultCode == Activity.RESULT_OK) {
339             String contents = intent.getStringExtra("SCAN_RESULT");
340             String formatName = intent.getStringExtra("SCAN_RESULT_FORMAT");
341             byte[] rawBytes = intent.getByteArrayExtra("SCAN_RESULT_BYTES");
342             int intentOrientation = intent.getIntExtra("SCAN_RESULT_ORIENTATION", Integer.MIN_VALUE);
343             Integer orientation = intentOrientation == Integer.MIN_VALUE ? null : intentOrientation;
344             String errorCorrectionLevel = intent.getStringExtra("SCAN_RESULT_ERROR_CORRECTION_LEVEL");
345             return new IntentResult(contents,
346                                     formatName,
347                                     rawBytes,
348                                     orientation,
349                                     errorCorrectionLevel);
350           }
351           return new IntentResult();
352         }
353         return null;
354       }

365       /**
366        * Shares the given text by encoding it as a barcode, such that another user can
367        * scan the text off the screen of the device.
368        *
369        * @param text the text string to encode as a barcode
370        * @param type type of data to encode. See {@code com.google.zxing.client.android.Contents.Type} constants.
371        * @return the {@link AlertDialog} that was shown to the user prompting them to download the app
372        *   if a prompt was needed, or null otherwise
373        */
374       public final AlertDialog shareText(CharSequence text, CharSequence type) {
375         Intent intent = new Intent();
376         intent.addCategory(Intent.CATEGORY_DEFAULT);
377         intent.setAction(BS_PACKAGE + ".ENCODE");
378         intent.putExtra("ENCODE_TYPE", type);
379         intent.putExtra("ENCODE_DATA", text);
380         String targetAppPackage = findTargetAppPackage(intent);
381         if (targetAppPackage == null) {
382           return showDownloadDialog();
383         }
384         intent.setPackage(targetAppPackage);
385         intent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
386         intent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_WHEN_TASK_RESET);
387         attachMoreExtras(intent);
388         activity.startActivity(intent);
389         return null;
390       }

392       private static List<String> list(String... values) {
393         return Collections.unmodifiableList(Arrays.asList(values));
394       }

396       private void attachMoreExtras(Intent intent) {
397         for (Map.Entry<String,Object> entry : moreExtras.entrySet()) {
398           String key = entry.getKey();
399           Object value = entry.getValue();
400           // Kind of hacky
401           if (value instanceof Integer) {
402             intent.putExtra(key, (Integer) value);
403           } else if (value instanceof Long) {
404             intent.putExtra(key, (Long) value);
405           } else if (value instanceof Boolean) {
406             intent.putExtra(key, (Boolean) value);
407           } else if (value instanceof Double) {
408             intent.putExtra(key, (Double) value);
409           } else if (value instanceof Float) {
410             intent.putExtra(key, (Float) value);
411           } else if (value instanceof Bundle) {
412             intent.putExtra(key, (Bundle) value);
413           } else {
414             intent.putExtra(key, value.toString());
415           }
416         }
417       }
