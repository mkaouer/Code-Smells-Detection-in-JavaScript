91         public static LogResult postLog(final Geocache cache, final LogType logType, final Calendar date, final String log, boolean isRetry) {
92             final Parameters params = new Parameters("cache_id", cache.getGeocode());
93             params.add("type", logType.type);
94             params.add("log", log);
95             params.add("date", LOG_DATE_FORMAT.format(date.getTime()));
96             params.add("sid", ecLogin.getSessionId());
97     
98             final String uri = API_HOST + "log.php";
99             final HttpResponse response = Network.postRequest(uri, params);
100     
101             if (response == null) {
102                 return new LogResult(StatusCode.LOG_POST_ERROR_EC, "");
103             }
104             if (!isRetry && response.getStatusLine().getStatusCode() == 403) {
105                 if (ecLogin.login() == StatusCode.NO_ERROR) {
106                     apiRequest(uri, params, true);
107                 }
108             }
109             if (response.getStatusLine().getStatusCode() != 200) {
110                 return new LogResult(StatusCode.LOG_POST_ERROR_EC, "");
111             }
112     
113             final String data = Network.getResponseDataAlways(response);
114             if (!StringUtils.isBlank(data) && StringUtils.contains(data, "success")) {
115                 if (logType == LogType.FOUND_IT || logType == LogType.ATTENDED) {
116                     ecLogin.setActualCachesFound(ecLogin.getActualCachesFound() + 1);
117                 }
118                 final String uid = StringUtils.remove(data, "success:");
119                 return new LogResult(StatusCode.NO_ERROR, uid);
120             }
121     
122             return new LogResult(StatusCode.LOG_POST_ERROR_EC, "");
123         }

134         private static HttpResponse apiRequest(final String uri, final Parameters params, final boolean isRetry) {
135             // add session and cgeo marker on every request
136             if (!isRetry) {
137                 params.add("cgeo", "1");
138                 params.add("sid", ecLogin.getSessionId());
139             }
140     
141             final HttpResponse response = Network.getRequest(API_HOST + uri, params);
142             if (response == null) {
143                 return null;
144             }
145     
146             // retry at most one time
147             if (!isRetry && response.getStatusLine().getStatusCode() == 403) {
148                 if (ecLogin.login() == StatusCode.NO_ERROR) {
149                     return apiRequest(uri, params, true);
150                 }
151             }
152             if (response.getStatusLine().getStatusCode() != 200) {
153                 return null;
154             }
155             return response;
156         }

158         private static Collection<Geocache> importCachesFromGPXResponse(final HttpResponse response) {
159             if (response == null) {
160                 return Collections.emptyList();
161             }
162     
163             try {
164                 return new GPX10Parser(StoredList.TEMPORARY_LIST_ID).parse(response.getEntity().getContent(), null);
165             } catch (Exception e) {
166                 Log.e("Error importing gpx from extremcaching.com", e);
167                 return Collections.emptyList();
168             }
169         }

171         private static List<Geocache> importCachesFromJSON(final HttpResponse response) {
172             if (response != null) {
173                 try {
174                     final String data = Network.getResponseDataAlways(response);
175                     if (StringUtils.isBlank(data) || StringUtils.equals(data, "[]")) {
176                         return Collections.emptyList();
177                     }
178                     final JSONArray json = new JSONArray(data);
179                     final int len = json.length();
180                     final List<Geocache> caches = new ArrayList<Geocache>(len);
181                     for (int i = 0; i < len; i++) {
182                         final Geocache cache = parseCache(json.getJSONObject(i));
183                         if (cache != null) {
184                             caches.add(cache);
185                         }
186                     }
187                     return caches;
188                 } catch (final JSONException e) {
189                     Log.w("JSONResult", e);
190                 }
191             }
192     
193             return Collections.emptyList();
194         }

196         private static Geocache parseCache(final JSONObject response) {
197             final Geocache cache = new Geocache();
198             cache.setReliableLatLon(true);
199             try {
200                 cache.setGeocode("EC" + response.getString("cache_id"));
201                 cache.setName(response.getString("title"));
202                 cache.setCoords(new Geopoint(response.getString("lat"), response.getString("lon")));
203                 cache.setType(getCacheType(response.getString("type")));
204                 cache.setDifficulty((float) response.getDouble("difficulty"));
205                 cache.setTerrain((float) response.getDouble("terrain"));
206                 cache.setSize(CacheSize.getById(response.getString("size")));
207                 cache.setFound(response.getInt("found") == 1);
208                 DataStore.saveCache(cache, EnumSet.of(SaveFlag.SAVE_CACHE));
209             } catch (final JSONException e) {
210                 Log.e("ECApi.parseCache", e);
211                 return null;
212             }
213             return cache;
214         }
