10     /**
11      * BlockingThreadPool restricts the amount of parallel threads executing Runnables.
12      */
13     public class BlockingThreadPool {
14         /** The queue holding the Runnable. **/
15         private BlockingQueue<Runnable> queue = null;
16         /** The Executor. **/
17         private ThreadPoolExecutor executor;
18     
19         /**
20          * Creates a ThreadPool with a given maximum of parallel threads running.
21          * Idle threads will be stopped until new threads are added.
22          *
23          * @param poolSize
24          *            Maximum amout of parallel Threads
25          * @param priority
26          *            The Thread priority e.g. Thread.MIN_PRIORITY
27          */
28         public BlockingThreadPool(int poolSize, int priority) {
29             ThreadFactory threadFactory = new PriorityThreadFactory(priority);
30             this.queue = new ArrayBlockingQueue<Runnable>(poolSize, true);
31             this.executor = new ThreadPoolExecutor(0, poolSize, 5, TimeUnit.SECONDS, this.queue);
32             this.executor.setThreadFactory(threadFactory);
33         }
34     
35         /**
36          * Add a runnable to the pool. This will start the core threads in the underlying
37          * executor and try to add the Runnable to the pool. This method waits until timeout
38          * if no free thread is available.
39          *
40          * @param task
41          *            The Runnable to add to the pool
42          * @param timeout
43          *            The timeout to wait for a free thread
44          * @param unit
45          *            The timeout unit
46          * @return true/false successful added
47          * @throws InterruptedException
48          *             Operation was interrupted
49          */
50         public boolean add(Runnable task, int timeout, TimeUnit unit) throws InterruptedException {
51             this.executor.setCorePoolSize(this.executor.getMaximumPoolSize());
52             this.executor.prestartAllCoreThreads();
53             boolean successfull = this.queue.offer(task, timeout, unit);
54             this.executor.setCorePoolSize(0);
55             return successfull;
56         }
