89         @SuppressWarnings("unchecked")
90         private static <T extends IConnector> Collection<T> getMatchingConnectors(final Class<T> clazz) {
91             final List<T> matching = new ArrayList<T>();
92             for (final IConnector connector : CONNECTORS) {
93                 if (clazz.isInstance(connector)) {
94                     matching.add((T) connector);
95                 }
96             }
97             return Collections.unmodifiableCollection(matching);
98         }

130         public static boolean canHandle(final @Nullable String geocode) {
131             if (geocode == null) {
132                 return false;
133             }
134             if (isInvalidGeocode(geocode)) {
135                 return false;
136             }
137             for (final IConnector connector : CONNECTORS) {
138                 if (connector.canHandle(geocode)) {
139                     return true;
140                 }
141             }
142             return false;
143         }

155         public static TrackableConnector getTrackableConnector(String geocode) {
156             for (final TrackableConnector connector : TRACKABLE_CONNECTORS) {

162         }

164         public static @NonNull
165         IConnector getConnector(final String geocodeInput) {
166             // this may come from user input
167             final String geocode = StringUtils.trim(geocodeInput);
168             if (geocode == null) {
169                 return UNKNOWN_CONNECTOR;
170             }
171             if (isInvalidGeocode(geocode)) {
172                 return UNKNOWN_CONNECTOR;
173             }
174             for (final IConnector connector : CONNECTORS) {
175                 if (connector.canHandle(geocode)) {
176                     return connector;
177                 }
178             }
179             // in case of errors, take UNKNOWN to avoid null checks everywhere
180             return UNKNOWN_CONNECTOR;
181         }

183         private static boolean isInvalidGeocode(final String geocode) {
184             return StringUtils.isBlank(geocode) || !Character.isLetterOrDigit(geocode.charAt(0));
185         }

197         public static String getGeocodeFromURL(final String url) {
198             for (final IConnector connector : CONNECTORS) {
199                 final String geocode = connector.getGeocodeFromUrl(url);
200                 if (StringUtils.isNotBlank(geocode)) {
201                     return geocode;
202                 }
203             }
204             return null;
205         }

217         public static String getTrackableFromURL(final String url) {
