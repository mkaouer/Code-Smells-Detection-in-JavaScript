27         void rememberTrailPosition(Location coordinates) {
28             if (coordinates.getAccuracy() >= 50f) {
29                 return;
30             }
31             if (coordinates.getLatitude() == 0.0 && coordinates.getLongitude() == 0.0) {
32                 return;

40             if (historyRecent.distanceTo(coordinates) <= MINIMUM_DISTANCE_METERS) {
41                 return;
42             }
43     
44             history.add(coordinates);
45     
46             // avoid running out of memory
47             final int itemsToRemove = getHistory().size() - MAX_POSITIONS;
48             if (itemsToRemove > 0) {
49                 for (int i = 0; i < itemsToRemove; i++) {
50                     getHistory().remove(0);
51                 }
52             }
53         }
