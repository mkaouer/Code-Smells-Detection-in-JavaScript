33                     dialog.dismiss();

52                 dialog.setProgressStyle(style);
53                 dialog.show();
54             }
55         }
56     
57         private void createProgressDialog(final Context context, final String title, final String message, final Message cancelMessage) {
58             dialog = hideAbsolute ? new CustomProgressDialog(context) : new ProgressDialog(context);
59             dialog.setTitle(title);
60             dialog.setMessage(message);
61             if (cancelMessage != null) {
62                 dialog.setCancelable(true);
63                 dialog.setCancelMessage(cancelMessage);
64                 dialog.setButton(DialogInterface.BUTTON_NEGATIVE, context.getResources().getString(android.R.string.cancel), cancelMessage);
65             } else {
66                 dialog.setCancelable(false);
67             }
68             dialog.setProgress(0);
69             dialog.setCanceledOnTouchOutside(false);
70             dialog.getWindow().addFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON);
71             this.progress = 0;
72         }
73     
74         public synchronized void setMessage(final String message) {
75             if (dialog != null && dialog.isShowing()) {
76                 dialog.setMessage(message);
77             }
78         }
79     
80         public synchronized boolean isShowing() {
81             return dialog != null && dialog.isShowing();
82         }
83     
84         public synchronized void setMaxProgressAndReset(final int max) {
85             if (isShowing()) {
86                 final int modMax = max / this.progressDivider;
87                 dialog.setMax(modMax);
88                 dialog.setProgress(0);
89             }
90             this.progress = 0;
91         }
92     
93         public synchronized void setProgress(final int progress) {
94             final int modProgress = progress / this.progressDivider;
95             if (isShowing()) {
96                 dialog.setProgress(modProgress);
97             }
98             this.progress = modProgress;
99         }
100     
101         public synchronized int getProgress() {
102             if (dialog != null) {
103                 dialog.getProgress();
104             }
105             return this.progress;
106         }
107     
108         public synchronized void setProgressDivider(final int progressDivider) {
109             this.progressDivider = progressDivider;
