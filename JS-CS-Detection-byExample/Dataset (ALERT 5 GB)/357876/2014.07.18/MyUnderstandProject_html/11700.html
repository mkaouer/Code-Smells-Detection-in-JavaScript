10     public class UnknownTagsHandler implements TagHandler {
11     
12         private enum ListType {
13             Ordered, Unordered
14         }
15     
16         private static final int UNDEFINED_POSITION = -1;
17     
18         private int countCells = 0;
19         private int strikePos = UNDEFINED_POSITION;
20         private boolean problematicDetected = false;
21         private int listIndex = 0;
22         private ListType listType = ListType.Unordered;

24         @Override
25         public void handleTag(boolean opening, String tag, Editable output,
26                 XMLReader xmlReader) {
27             if (tag.equalsIgnoreCase("strike") || tag.equals("s")) {
28                 handleStrike(opening, output);
29             } else if (tag.equalsIgnoreCase("table")) {
30                 handleProblematic();
31             } else if (tag.equalsIgnoreCase("td")) {
32                 handleTd(opening, output);
33             } else if (tag.equalsIgnoreCase("tr")) {
34                 handleTr(opening, output);
35             } else if (tag.equalsIgnoreCase("pre")) {
36                 handleProblematic();
37             } else if (tag.equalsIgnoreCase("ol")) {
38                 handleOl(opening);
39             } else if (tag.equalsIgnoreCase("li")) {
40                 handleLi(opening, output);
41             }
42         }
43     
44         private void handleStrike(boolean opening, Editable output) {
45             final int length = output.length();
46             if (opening) {
47                 strikePos = length;
48             } else {
49                 if (strikePos > UNDEFINED_POSITION) {
50                     output.setSpan(new StrikethroughSpan(), strikePos, length, Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);
51                     strikePos = UNDEFINED_POSITION;
52                 }
53             }
54         }
55     
56         public boolean isProblematicDetected() {
57             return problematicDetected;
58         }
59     
60         private void handleProblematic() {
61             problematicDetected = true;
62         }
63     
64         private void handleTd(boolean opening, Editable output) {
65             // insert bar for each table column, see https://en.wikipedia.org/wiki/Box-drawing_characters
66             if (opening) {
67                 if (countCells++ > 0) {
68                     output.append('â”†');
69                 }
70             }
71         }

73         private void handleTr(boolean opening, Editable output) {
74             // insert new line for each table row
75             if (opening) {
76                 output.append('\n');
77                 countCells = 0;
78             }
79         }

81         // Ordered lists are handled in a simple manner. They are rendered as Arabic numbers starting at 1

83         private void handleOl(boolean opening) {
84             if (opening) {

96                 } else {
