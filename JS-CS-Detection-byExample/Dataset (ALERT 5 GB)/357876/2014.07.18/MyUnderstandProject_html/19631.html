32         public static void listDir(List<File> result, File directory, FileSelector chooser, Handler feedBackHandler) {
33     
34             if (directory == null || !directory.isDirectory() || !directory.canRead()
35                     || result == null
36                     || chooser == null) {
37                 return;
38             }
39     
40             final File[] files = directory.listFiles();
41     
42             if (ArrayUtils.isNotEmpty(files)) {
43                 for (File file : files) {
44                     if (chooser.shouldEnd()) {
45                         return;
46                     }
47                     if (!file.canRead()) {
48                         continue;
49                     }
50                     String name = file.getName();
51                     if (file.isFile()) {
52                         if (chooser.isSelected(file)) {
53                             result.add(file); // add file to list
54                         }
55                     } else if (file.isDirectory()) {
56                         if (name.charAt(0) == '.') {
57                             continue; // skip hidden directories
58                         }
59                         if (name.length() > 16) {
60                             name = name.substring(0, 14) + 'â€¦';
61                         }
62                         if (feedBackHandler != null) {
63                             feedBackHandler.sendMessage(Message.obtain(feedBackHandler, 0, name));
64                         }
65     
66                         listDir(result, file, chooser, feedBackHandler); // go deeper
67                     }
68                 }
69             }
70         }

78         /**
79          * Create a unique non existing file named like the given file name. If a file with the given name already exists,
80          * add a number as suffix to the file name.<br>
81          * Example: For the file name "file.ext" this will return the first file of the list
82          * <ul>
83          * <li>file.ext</li>
84          * <li>file_2.ext</li>
85          * <li>file_3.ext</li>
86          * </ul>
87          * which does not yet exist.
88          */
89         public static File getUniqueNamedFile(final String baseNameAndPath) {
90             String extension = StringUtils.substringAfterLast(baseNameAndPath, ".");
91             String pathName = StringUtils.substringBeforeLast(baseNameAndPath, ".");
92             int number = 1;
93             while (new File(getNumberedFileName(pathName, extension, number)).exists()) {
94                 number++;
95             }
96             return new File(getNumberedFileName(pathName, extension, number));
97         }
98     
99         private static String getNumberedFileName(String pathName, String extension, int number) {
100             return pathName + (number > 1 ? "_" + Integer.toString(number) : "") + "." + extension;
101         }
