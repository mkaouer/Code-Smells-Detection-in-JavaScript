169         @Override
170         protected final cgeo.geocaching.activity.AbstractViewPagerActivity.PageViewCreator createViewCreator(final Page page) {
171             switch (page) {
172                 case VERSION:

174                 case HELP:

197         private String getRawResourceString(final int resourceId) {
198             String result;
199             Scanner scanner = null;
200             try {
201                 final InputStream ins = res.openRawResource(resourceId);
202                 scanner = new Scanner(ins, CharEncoding.UTF_8);
203                 result = scanner.useDelimiter("\\A").next();
204                 IOUtils.closeQuietly(ins);
205             } finally {
206                 if (scanner != null) {
207                     scanner.close();
208                 }
209             }
210             return result;
211         }
