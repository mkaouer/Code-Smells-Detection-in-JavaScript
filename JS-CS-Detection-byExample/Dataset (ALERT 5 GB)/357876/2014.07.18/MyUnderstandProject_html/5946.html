12      * Handles the JSON error response from OKAPI
13      */
14     public class OkapiError {
15     
16         /**
17          * List of detected errors OKAPI might return
18          */
19         public enum OkapiErrors {
20             NO_ERROR,
21             UNSPECIFIED,
22             INVALID_TIMESTAMP,
23             INVALID_TOKEN;
24         }
25     
26         @NonNull private final OkapiErrors state;
27         @NonNull private final String message;
28     
29         public OkapiError(@Nullable JSONObject data) {
30     
31             // A null-response is by definition an error (some exception occurred somewhere in the flow)
32             if (data == null) {
33                 state = OkapiErrors.UNSPECIFIED;
34                 message = StringUtils.EMPTY;
35                 return;
36             }
37             // Second possibility: we get an error object as return (@see http://opencaching.pl/okapi/introduction.html#errors)
38             if (data.has("error")) {
39                 String localmessage = null;
40                 OkapiErrors localstate = OkapiErrors.UNSPECIFIED;
41                 try {
42                     JSONObject error = data.getJSONObject("error");
43                     // Check reason_stack element to look for the specific oauth problems we want to report back
44                     if (error.has("reason_stack")) {
45                         String reason = error.getString("reason_stack");
46                         if (StringUtils.contains(reason, "invalid_oauth_request")) {
47                             if (StringUtils.contains(reason, "invalid_timestamp")) {
48                                 localstate = OkapiErrors.INVALID_TIMESTAMP;
49                             } else if (StringUtils.contains(reason, "invalid_token")) {
50                                 localstate = OkapiErrors.INVALID_TOKEN;
51                             }
52                         }
53                     }
54                     // Check if we can extract a message as well
55                     if (error.has("developer_message")) {
56                         localmessage = error.getString("developer_message");
57                         assert localmessage != null; // by virtue of defaultString
58                     }
59                 } catch (JSONException ex) {
60                     Log.d("OkapiError: Failed to parse JSON", ex);
61                     localstate = OkapiErrors.UNSPECIFIED;
62                 }
63                 state = localstate;
64                 message = StringUtils.defaultString(localmessage);
65                 return;
66             }
67     
68             // Third possibility: some other response, everything is fine!
69             state = OkapiErrors.NO_ERROR;
70             message = StringUtils.EMPTY;
71         }
72     
73         public boolean isError() {
74             return state != OkapiErrors.NO_ERROR;
75         }
76     
77         @NonNull
78         public OkapiErrors getResult() {
79             return state;
80         }
81     
82         @NonNull
83         public String getMessage() {
84             return message;
85         }
