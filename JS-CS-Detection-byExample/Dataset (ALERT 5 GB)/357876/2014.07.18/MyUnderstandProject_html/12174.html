12     /**
13      * filters caches by popularity ratio (favorites per find in %).
14      */
15     class PopularityRatioFilter extends AbstractFilter {
16         private final int minRatio;
17         private final int maxRatio;
18     
19         public PopularityRatioFilter(String name, final int minRatio, final int maxRatio) {
20             super(name);
21             this.minRatio = minRatio;
22             this.maxRatio = maxRatio;
23         }
24     
25         @Override
26         public boolean accepts(final Geocache cache) {
27     
28             int finds;
29             int favorites;
30             float ratio;
31     
32             finds = getFindsCount(cache);
33     
34             // prevent division by zero
35             if (finds == 0) {
36                 return false;
37             }
38     
39             favorites = cache.getFavoritePoints();
40             ratio = ((float) favorites / (float) finds) * 100.0f;
41     
42             return (ratio > minRatio) && (ratio <= maxRatio);
43         }
44     
45         private static int getFindsCount(Geocache cache) {
46             if (cache.getLogCounts().isEmpty()) {
47                 cache.setLogCounts(DataStore.loadLogCounts(cache.getGeocode()));
48             }
49             Integer logged = cache.getLogCounts().get(LogType.FOUND_IT);
50             if (logged != null) {
51                 return logged;
52             }
53             return 0;
54         }
55     
56         public static class Factory implements IFilterFactory {
57     
58             private static final int[] RATIOS = { 10, 20, 30, 40, 50, 75 };
59     
60             @Override
61             public List<IFilter> getFilters() {
62                 final List<IFilter> filters = new ArrayList<IFilter>(RATIOS.length);
63                 for (int i = 0; i < RATIOS.length; i++) {
64                     final int minRange = RATIOS[i];
65                     final int maxRange = Integer.MAX_VALUE;
66                     final String name = "> " + minRange + " " + CgeoApplication.getInstance().getResources().getString(R.string.percent_favorite_points);
67                     filters.add(new PopularityRatioFilter(name, minRange, maxRange));
68                 }
69                 return filters;
70             }
