12     /**
13      * Synchronized set wrapper for the LeastRecentlyUsedMap.
14      *
15      * This code is heavily based on the HashSet code that represents Map as a Set.
16      * Unfortunately HashSet does not allow to use a custom Map as its Storage.
17      * Therefore overriding removeEldestEntry() is impossible for a normal LinkedHashSet.
18      *
19      * Synchronization is added to guard against concurrent modification. Iterator
20      * access has to be guarded externally or the synchronized getAsList method can be used
21      * to get a clone for iteration.
22      */
23     public class LeastRecentlyUsedSet<E> extends AbstractSet<E>
24             implements Cloneable, java.io.Serializable {
25     
26         private static final long serialVersionUID = -1942301031191419547L;
27     
28         private transient LeastRecentlyUsedMap<E, Object> map;
29         private static final Object PRESENT = new Object();
30     
31         public LeastRecentlyUsedSet(int maxEntries, int initialCapacity, float loadFactor) {
32             // because we don't use any Map.get() methods from the Set, BOUNDED and LRU_CACHE have the exact same Behaviour
33             // So we use LRU_CACHE mode because it should perform a bit better (as it doesn't re-add explicitly)
34             map = new LeastRecentlyUsedMap.LruCache<E, Object>(maxEntries, initialCapacity, loadFactor);
35         }
36     
37         public LeastRecentlyUsedSet(int maxEntries) {
38             map = new LeastRecentlyUsedMap.LruCache<E, Object>(maxEntries);
39         }
40     
41         /**
42          * Copy of the HashSet code if iterator()
43          * Iterator access has to be synchronized externally!
44          *
45          * @see HashSet
46          */
47         @NonNull
48         @Override
49         public Iterator<E> iterator() {
50             return map.keySet().iterator();
51         }
52     
53         /**
54          * Synchronized access to set size
55          * Copy of the HashSet code if size()
56          *
57          * @see HashSet
58          */
59         @Override
60         public synchronized int size() {
61             return map.size();
62         }
63     
64         /**
65          * Synchronized check of set emptiness
66          * Copy of the HashSet code if isEmpty()
67          *
68          * @see HashSet
69          */
70         @Override
71         public synchronized boolean isEmpty() {
72             return map.isEmpty();
73         }
74     
75         /**
76          * Synchronized check for containment
77          * Copy of the HashSet code if contains()
78          *
79          * @see HashSet
80          */
81         @Override
82         public synchronized boolean contains(Object o) {
83             return map.containsKey(o);
84         }
85     
86         /**
87          * Synchronized addition of an item
88          * Copy of the HashSet code if add()
89          *
90          * @see HashSet
91          */
92         @Override
93         public synchronized boolean add(E e) {
94             if (e == null) {
95                 throw new IllegalArgumentException("LeastRecentlyUsedSet cannot take null element");
96             }
97             return map.put(e, PRESENT) == null;
98         }
99     
100         /**
101          * Synchronized removal of a contained item
102          * Copy of the HashSet code if remove()
103          *
104          * @see HashSet
105          */
106         @Override
107         public synchronized boolean remove(Object o) {
108             return map.remove(o) == PRESENT;
109         }
110     
111         /**
112          * Synchronized removal of all elements contained in another collection.
113          */
114         @Override
115         public synchronized boolean removeAll(final Collection<?> c) {
116             boolean changed = false;
117             for (final Object o: c) {
118                 changed |= remove(o);
119             }
120             return changed;
121         }
122     
123         /**
124          * Synchronized clearing of the set
125          * Copy of the HashSet code if clear()
126          *
127          * @see HashSet
128          */
129         @Override
130         public synchronized void clear() {
131             map.clear();
132         }
133     
134         /**
135          * (synchronized) Clone of the set
136          * Copy of the HashSet code if clone()
137          *
138          * @see HashSet
139          */
140         @Override
141         @SuppressWarnings("unchecked")
142         public Object clone() throws CloneNotSupportedException {
143             try {
144                 synchronized (this) {
145                     final LeastRecentlyUsedSet<E> newSet = (LeastRecentlyUsedSet<E>) super.clone();
146                     newSet.map = (LeastRecentlyUsedMap<E, Object>) map.clone();
147                     return newSet;
148                 }
149             } catch (CloneNotSupportedException e) {
150                 throw new InternalError();
151             }
152         }
153     
154         /**
155          * Creates a clone as a list in a synchronized fashion.
156          *
157          * @return List based clone of the set
158          */
159         public synchronized List<E> getAsList() {
160             return new ArrayList<E>(this);
161         }

163         /**
164          * Serialization version of HashSet with the additional parameters for the custom Map
165          *
166          * @see HashSet
167          */
168         private void writeObject(java.io.ObjectOutputStream s)
169                 throws java.io.IOException {
170             // Write out any hidden serialization magic
171             s.defaultWriteObject();
172     
173             // Write out HashMap capacity and load factor
174             s.writeInt(map.initialCapacity);
175             s.writeFloat(map.loadFactor);
176             s.writeInt(map.getMaxEntries());
177     
178             // Write out size
179             s.writeInt(map.size());
180     
181             // Write out all elements in the proper order.
182             for (final E e : map.keySet()) {
183                 s.writeObject(e);
184             }
185         }
186     
187         /**
188          * Serialization version of HashSet with the additional parameters for the custom Map
189          *
190          * @see HashSet
191          */
192         @SuppressWarnings("unchecked")
193         private void readObject(java.io.ObjectInputStream s)
194                 throws java.io.IOException, ClassNotFoundException {
195             // Read in any hidden serialization magic
196             s.defaultReadObject();
197     
198             // Read in HashMap capacity and load factor and create backing HashMap
199             final int capacity = s.readInt();
200             final float loadFactor = s.readFloat();
201             final int maxEntries = s.readInt();
202     
203             map = new LeastRecentlyUsedMap.LruCache<E, Object>(maxEntries, capacity, loadFactor);
204     
205             // Read in size
206             final int size = s.readInt();
207     
208             // Read in all elements in the proper order.
209             for (int i = 0; i < size; i++) {
210                 E e = (E) s.readObject();
211                 map.put(e, PRESENT);
212             }
213         }
