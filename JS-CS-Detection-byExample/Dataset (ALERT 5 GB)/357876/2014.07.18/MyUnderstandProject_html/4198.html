121         private static String formatAddress(final Address address) {
122             final ArrayList<String> addressParts = new ArrayList<String>();
123     
124             final String countryName = address.getCountryName();
125             if (countryName != null) {
126                 addressParts.add(countryName);
127             }
128             final String locality = address.getLocality();
129             if (locality != null) {
130                 addressParts.add(locality);
131             } else {
132                 final String adminArea = address.getAdminArea();
133                 if (adminArea != null) {
134                     addressParts.add(adminArea);
135                 }
136             }
137             return StringUtils.join(addressParts, ", ");
138         }

427         protected void selectGlobalTypeFilter() {
428             final List<CacheType> cacheTypes = new ArrayList<CacheType>();
429     
430             //first add the most used types
431             cacheTypes.add(CacheType.ALL);
432             cacheTypes.add(CacheType.TRADITIONAL);
433             cacheTypes.add(CacheType.MULTI);
434             cacheTypes.add(CacheType.MYSTERY);
435     
436             // then add all other cache types sorted alphabetically
437             List<CacheType> sorted = new ArrayList<CacheType>();
438             sorted.addAll(Arrays.asList(CacheType.values()));
439             sorted.removeAll(cacheTypes);
440     
441             Collections.sort(sorted, new Comparator<CacheType>() {
442     
443                 @Override
444                 public int compare(final CacheType left, final CacheType right) {
445                     return left.getL10n().compareToIgnoreCase(right.getL10n());
446                 }
447             });
448     
449             cacheTypes.addAll(sorted);
450     
451             int checkedItem = cacheTypes.indexOf(Settings.getCacheType());
452             if (checkedItem < 0) {
453                 checkedItem = 0;
454             }
455     
456             String[] items = new String[cacheTypes.size()];
457             for (int i = 0; i < cacheTypes.size(); i++) {
458                 items[i] = cacheTypes.get(i).getL10n();
459             }
460     
461             Builder builder = new AlertDialog.Builder(this);
462             builder.setTitle(R.string.menu_filter);
463             builder.setSingleChoiceItems(items, checkedItem, new DialogInterface.OnClickListener() {
464     
465                 @Override
466                 public void onClick(final DialogInterface dialog, final int position) {
467                     CacheType cacheType = cacheTypes.get(position);
468                     Settings.setCacheType(cacheType);
469                     setFilterTitle();
470                     dialog.dismiss();
471                 }
472     
473             });
474             builder.create().show();
475         }

635         private class CountBubbleUpdateThread extends Thread {
636             private Handler countBubbleHandler = new Handler() {
637     
638                 @Override
639                 public void handleMessage(final Message msg) {
640                     try {
641                         if (countBubbleCnt == 0) {
642                             countBubble.setVisibility(View.GONE);
643                         } else {
644                             countBubble.setText(Integer.toString(countBubbleCnt));
645                             countBubble.bringToFront();
646                             countBubble.setVisibility(View.VISIBLE);
647                         }
648                     } catch (Exception e) {
649                         Log.w("MainActivity.countBubbleHander", e);
650                     }
651                 }
652             };
653     
654             @Override
655             public void run() {
656                 if (app == null) {
657                     return;
658                 }
659     
660                 int checks = 0;
661                 while (!DataStore.isInitialized()) {
662                     try {
663                         sleep(500);
664                         checks++;
665                     } catch (Exception e) {
666                         Log.e("MainActivity.CountBubbleUpdateThread.run", e);
667                     }
668     
669                     if (checks > 10) {
670                         return;
671                     }
672                 }
673     
674                 countBubbleCnt = DataStore.getAllCachesCount();
675     
676                 countBubbleHandler.sendEmptyMessage(0);
677             }
678         }

680         private class CleanDatabaseThread extends Thread {
681     
682             @Override
683             public void run() {
684                 if (app == null) {
685                     return;
686                 }
687                 if (cleanupRunning) {
688                     return;
689                 }
690     
691                 boolean more = false;
692                 if (version != Settings.getVersion()) {
693                     Log.i("Initializing hard cleanup - version changed from " + Settings.getVersion() + " to " + version + ".");
694     
695                     more = true;
696                 }
697     
698                 cleanupRunning = true;
699                 DataStore.clean(more);
700                 cleanupRunning = false;
701     
702                 if (version > 0) {
703                     Settings.setVersion(version);
704                 }
705             }
706         }
