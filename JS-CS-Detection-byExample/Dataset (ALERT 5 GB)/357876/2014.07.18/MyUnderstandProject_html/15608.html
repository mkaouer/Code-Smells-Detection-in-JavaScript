16     /**
17      * Cache for Caches. Every cache is stored in memory while c:geo is active to
18      * speed up the app and to minimize network request - which are slow.
19      */
20     public class CacheCache {
21     
22         private static final int MAX_CACHED_CACHES = 1000;
23         final private LeastRecentlyUsedMap<String, Geocache> cachesCache;
24     
25         public CacheCache() {
26             cachesCache = new LeastRecentlyUsedMap.LruCache<String, Geocache>(MAX_CACHED_CACHES);
27             cachesCache.setRemoveHandler(new CacheRemoveHandler());
28         }
29     
30         public synchronized void removeAllFromCache() {
31             cachesCache.clear();
32         }
33     
34         /**
35          * @param geocode
36          *            Geocode of the cache to remove from the cache
37          */
38         public void removeCacheFromCache(final String geocode) {
39             if (StringUtils.isBlank(geocode)) {
40                 throw new IllegalArgumentException("geocode must not be empty");
41             }
42             synchronized(this) {
43                 cachesCache.remove(geocode);
44             }
45         }
46     
47         /**
48          * "Store" a cache in the CacheCache. If the cache is already in the CacheCache the cache gets replaced.
49          *
50          * @param cache
51          *            Cache
52          *
53          */
54         public void putCacheInCache(final Geocache cache) {
55             if (cache == null) {
56                 throw new IllegalArgumentException("cache must not be null");
57             }
58             if (StringUtils.isBlank(cache.getGeocode())) {
59                 throw new IllegalArgumentException("geocode must not be empty");
60             }
61             synchronized(this) {
62                 cache.addStorageLocation(StorageLocation.CACHE);
63                 cachesCache.put(cache.getGeocode(), cache);
64             }
65         }
66     
67         /**
68          * @param geocode
69          *            Geocode of the cache to retrieve from the cache
70          * @return cache if found, null else
71          */
72         public Geocache getCacheFromCache(final String geocode) {
73             if (StringUtils.isBlank(geocode)) {
74                 throw new IllegalArgumentException("geocode must not be empty");
75             }
76             synchronized(this) {
77                 return cachesCache.get(geocode);
78             }
79         }
80     
81         public synchronized Set<String> getInViewport(final Viewport viewport, final CacheType cacheType) {
82             final Set<String> geocodes = new HashSet<String>();
83             for (final Geocache cache : cachesCache.values()) {
84                 if (cache.getCoords() == null) {
85                     // FIXME: this kludge must be removed, it is only present to help us debug the cases where
86                     // caches contain null coordinates.
87                     Log.e("CacheCache.getInViewport: got cache with null coordinates: " + cache.getGeocode());
88                     continue;
89                 }
90                 if (cacheType.contains(cache) && viewport.contains(cache)) {
91                     geocodes.add(cache.getGeocode());
92                 }
93             }
94             return geocodes;
95         }
96     
97         @Override
98         public synchronized String toString() {
99             return StringUtils.join(cachesCache.keySet(), ' ');
100         }
101     
102         private static class CacheRemoveHandler implements RemoveHandler<Geocache> {
103     
104             @Override
105             public void onRemove(final Geocache removed) {
106                 // FIXME: as above, we sometimes get caches with null coordinates, that may then provoke
107                 // a NullPointerException down the invocation chain.
108                 if (removed.getCoords() != null) {
109                     Tile.cache.removeFromTileCache(removed);
110                 }
111             }
112         }
