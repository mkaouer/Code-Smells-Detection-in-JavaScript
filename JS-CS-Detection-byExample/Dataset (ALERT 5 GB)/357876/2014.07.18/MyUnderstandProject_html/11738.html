204         private float[] pathTo(final Geopoint target) {
205             float[] results = new float[2];
206             android.location.Location.distanceBetween(latitude, longitude, target.latitude, target.longitude, results);

210         /**
211          * Calculates distance to given Geopoint in km.
212          *
213          * @param point
214          *            target
215          * @return distance in km
216          * @throws GeopointException
217          *             if there is an error in distance calculation
218          */
219         public float distanceTo(@NonNull final ICoordinates point) {
220             final Geopoint otherCoords = point.getCoords();
221             if (DISTANCE_BROKEN) {
222                 return (float) (getDistance(latitude, longitude, otherCoords.latitude, otherCoords.longitude) / 1000);
223             }
224             return pathTo(otherCoords)[0] / 1000;
225         }

227         /**
228          * Calculates bearing to given Geopoint in degree.
229          *
230          * @param point
231          *            target
232          * @return bearing in degree, in the [0,360[ range
233          */
234         public float bearingTo(final ICoordinates point) {
235             // Android library returns a bearing in the [-180;180] range
236             final float bearing = pathTo(point.getCoords())[1];
237             return bearing < 0 ? bearing + 360 : bearing;
238         }

240         /**
241          * Calculates geopoint from given bearing and distance.
242          *
243          * @param bearing
244          *            bearing in degree
245          * @param distance
246          *            distance in km
247          * @return the projected geopoint
248          */
249         public Geopoint project(final double bearing, final double distance) {
250             final double rlat1 = latitude * DEG_TO_RAD;
251             final double rlon1 = longitude * DEG_TO_RAD;
252             final double rbearing = bearing * DEG_TO_RAD;
253             final double rdistance = distance / EARTH_RADIUS;
254     
255             final double rlat = Math.asin(Math.sin(rlat1) * Math.cos(rdistance) + Math.cos(rlat1) * Math.sin(rdistance) * Math.cos(rbearing));
256             final double rlon = rlon1 + Math.atan2(Math.sin(rbearing) * Math.sin(rdistance) * Math.cos(rlat1), Math.cos(rdistance) - Math.sin(rlat1) * Math.sin(rlat));
257     
258             return new Geopoint(rlat * RAD_TO_DEG, rlon * RAD_TO_DEG);
259         }

555         /**
556          * Gets distance in meters (workaround for 4.2.1 JIT bug).
557          */
558         public static double getDistance(double lat1, double lon1, double lat2, double lon2) {
559             double earthRadius = 6372.8; // for haversine use R = 6372.8 km instead of 6371 km
560             double dLat = toRadians(lat2 - lat1);
561             double dLon = toRadians(lon2 - lon1);
562             double a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
563                     Math.cos(toRadians(lat1)) * Math.cos(toRadians(lat2)) *
564                     Math.sin(dLon / 2) * Math.sin(dLon / 2);
565             //double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
566             //return R * c * 1000;
567             // simplify haversine:
568             return (2 * earthRadius * 1000 * Math.asin(Math.sqrt(a)));
569         }
