14         protected final WeakReference<Progress> progressDialogRef;
15     
16         public SimpleHandler(final AbstractActivity activity, final Progress progress) {
17             activityRef = new WeakReference<AbstractActivity>(activity);
18             progressDialogRef = new WeakReference<Progress>(progress);
19         }
20     
21         @Override
22         public void handleMessage(final Message msg) {
23             final AbstractActivity activity = activityRef.get();
24             if (activity != null && msg.getData() != null && msg.getData().getString(MESSAGE_TEXT) != null) {
25                 activity.showToast(msg.getData().getString(MESSAGE_TEXT));
26             }
27             final Progress progressDialog = progressDialogRef.get();
28             if (progressDialog != null) {
29                 progressDialog.dismiss();
30             }
31         }
