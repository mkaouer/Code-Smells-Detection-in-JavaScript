165         public void saveImageInfo(boolean saveInfo) {
166             if (saveInfo) {
167                 final String filename = writeScaledImage(imageUri.getPath());
168                 if (filename != null) {
169                     imageUri = Uri.parse(filename);
170                     final Intent intent = new Intent();
171                     syncEditTexts();
172                     intent.putExtra(EXTRAS_CAPTION, imageCaption);
173                     intent.putExtra(EXTRAS_DESCRIPTION, imageDescription);
174                     intent.putExtra(EXTRAS_URI_AS_STRING, imageUri.toString());
175                     intent.putExtra(EXTRAS_SCALE, scaleChoiceIndex);
176                     setResult(RESULT_OK, intent);
177                 } else {
178                     showToast(res.getString(R.string.err_select_logimage_failed));
179                     setResult(RESULT_CANCELED);
180                 }
181             } else {
182                 setResult(RESULT_CANCELED);
183             }
184     
185             finish();
186         }

194         private void selectImageFromCamera() {
195             // create Intent to take a picture and return control to the calling application
196             Intent intent = new Intent(MediaStore.ACTION_IMAGE_CAPTURE);
197     
198             imageUri = ImageUtils.getOutputImageFileUri(); // create a file to save the image
199             if (imageUri == null) {
200                 showFailure();
201                 return;
202             }
203             intent.putExtra(MediaStore.EXTRA_OUTPUT, imageUri); // set the image file name
204     
205             // start the image capture Intent
206             startActivityForResult(intent, SELECT_NEW_IMAGE);
207         }

216         @Override
217         protected void onActivityResult(int requestCode, int resultCode, Intent data) {
218             if (resultCode == RESULT_CANCELED) {
219                 // User cancelled the image capture
220                 showToast(getResources().getString(R.string.info_select_logimage_cancelled));
221                 return;
222             }
223     
224             if (resultCode != RESULT_OK) {
225                 // Image capture failed, advise user
226                 showFailure();
227                 return;
228             }
229     
230             // null is an acceptable result if the image has been placed in the imageUri file by the
231             // camera application.
232             if (data != null) {
233                 final Uri selectedImage = data.getData();
234                 if (Build.VERSION.SDK_INT < VERSION_CODES.KITKAT) {
235                     String[] filePathColumn = { MediaColumns.DATA };
236     
237                     Cursor cursor = null;
238                     try {
239                         cursor = getContentResolver().query(selectedImage, filePathColumn, null, null, null);
240                         if (cursor == null) {
241                             showFailure();
242                             return;
243                         }
244                         cursor.moveToFirst();
245     
246                         int columnIndex = cursor.getColumnIndex(filePathColumn[0]);
247                         String filePath = cursor.getString(columnIndex);
248                         if (StringUtils.isBlank(filePath)) {
249                             showFailure();
250                             return;
251                         }
252                         imageUri = Uri.parse(filePath);
253                     } catch (Exception e) {
254                         Log.e("ImageSelectActivity.onActivityResult", e);
255                         showFailure();
256                     } finally {
257                         if (cursor != null) {
258                             cursor.close();
259                         }
260                     }
261     
262                     Log.d("SELECT IMAGE data = " + data.toString());
263                 } else {
264                     InputStream input = null;
265                     OutputStream output = null;
266                     try {
267                         input = getContentResolver().openInputStream(selectedImage);
268                         final File outputFile = ImageUtils.getOutputImageFile();
269                         output = new FileOutputStream(outputFile);
270                         LocalStorage.copy(input, output);
271                         imageUri = Uri.fromFile(outputFile);
272                     } catch (FileNotFoundException e) {
273                         Log.e("ImageSelectActivity.onStartResult", e);
274                     } finally {
275                         IOUtils.closeQuietly(input);
276                         IOUtils.closeQuietly(output);
277                     }
278                 }
279             }
280     
281             if (requestCode == SELECT_NEW_IMAGE) {
282                 showToast(getResources().getString(R.string.info_stored_image) + "\n" + imageUri);
283             }
284     
285             loadImagePreview();
286         }

288         /**
289          * Scales and writes the scaled image.
290          *
291          * @param filePath
292          * @return the scaled image path, or <tt>null</tt> if the image cannot be decoded
293          */
294         @Nullable
295         private String writeScaledImage(@Nullable final String filePath) {
296             if (filePath == null) {
297                 return null;
298             }
299             scaleChoiceIndex = scaleView.getSelectedItemPosition();
300             final int maxXY = getResources().getIntArray(R.array.log_image_scale_values)[scaleChoiceIndex];
301             return ImageUtils.readScaleAndWriteImage(filePath, maxXY);
302         }

308         private void loadImagePreview() {
309             if (imageUri == null || imageUri.getPath() == null) {
310                 return;
311             }
312             if (!new File(imageUri.getPath()).exists()) {
313                 Log.i("Image does not exist");
314                 return;
315             }
316     
317             final Bitmap bitmap = ImageUtils.readAndScaleImageToFitDisplay(imageUri.getPath());
318             imagePreview.setImageBitmap(bitmap);
319             imagePreview.setVisibility(View.VISIBLE);
320         }
