18     abstract class AbstractUserClickListener implements View.OnClickListener {
19     
20         private final List<UserAction> userActions;

22         public AbstractUserClickListener(final @NonNull List<UserAction> userActions) {
23             this.userActions = userActions;
24         }
25     
26         @Override
27         public void onClick(View view) {
28             if (view == null) {
29                 return;
30             }
31     
32             showUserActionsDialog(getUserName(view), view);
33         }
34     
35         protected abstract String getUserName(View view);
36     
37         /**
38          * Opens a dialog to do actions on an user name
39          */
40         private void showUserActionsDialog(final String userName, final View view) {
41             if (userActions.isEmpty()) {
42                 return;
43             }
44     
45             final AbstractActivity activity = (AbstractActivity) view.getContext();
46             final Resources res = activity.getResources();
47     
48             ArrayList<String> labels = new ArrayList<String>(userActions.size());
49             for (UserAction action : userActions) {
50                 labels.add(res.getString(action.displayResourceId));
51             }
52             final CharSequence[] items = labels.toArray(new String[labels.size()]);
53     
54             final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
55             builder.setTitle(res.getString(R.string.user_menu_title) + " " + userName);
56             builder.setItems(items, new DialogInterface.OnClickListener() {
57                 @Override
58                 public void onClick(DialogInterface dialog, int item) {
59                     userActions.get(item).run(new Context(userName, activity));
60                 }
61             });
62             final AlertDialog alert = builder.create();
63             alert.show();
64         }
