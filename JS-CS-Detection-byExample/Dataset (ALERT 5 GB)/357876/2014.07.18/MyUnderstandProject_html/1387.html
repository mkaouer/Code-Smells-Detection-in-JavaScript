38         /**
39          * Parses a pair of coordinates (latitude and longitude) out of a String.
40          * Accepts following formats and combinations of it:
41          * X DD
42          * X DDÂ°
43          * X DDÂ° MM
44          * X DDÂ° MM.MMM
45          * X DDÂ° MM SS
46          *
47          * as well as:
48          * DD.DDDDDDD
49          *
50          * Both . and , are accepted, also variable count of spaces (also 0)
51          *
52          * @param text
53          *            the string to parse
54          * @return an Geopoint with parsed latitude and longitude
55          * @throws Geopoint.ParseException
56          *             if lat or lon could not be parsed
57          */
58         public static Geopoint parse(final String text) {
59             final ResultWrapper latitudeWrapper = parseHelper(text, LatLon.LAT);
60             // cut away the latitude part when parsing the longitude
61             final ResultWrapper longitudeWrapper = parseHelper(text.substring(latitudeWrapper.matcherPos + latitudeWrapper.matcherLength), LatLon.LON);
62     
63             if (longitudeWrapper.matcherPos - (latitudeWrapper.matcherPos + latitudeWrapper.matcherLength) >= 10) {
64                 throw new Geopoint.ParseException("Distance between latitude and longitude text is to large.", LatLon.LON);
65             }
66     
67             final double lat = latitudeWrapper.result;
68             final double lon = longitudeWrapper.result;
69             if (!Geopoint.isValidLatitude(lat)) {
70                 throw new Geopoint.ParseException(text, LatLon.LAT);
71             }
72             if (!Geopoint.isValidLongitude(lon)) {
73                 throw new Geopoint.ParseException(text, LatLon.LON);
74             }
75             return new Geopoint(lat, lon);
76         }

78         /**
79          * Helper for coordinates-parsing
80          * 
81          * @param text
82          * @param latlon
83          * @return
84          */
85         private static ResultWrapper parseHelper(final String text, final LatLon latlon) {
86     
87             MatcherWrapper matcher = new MatcherWrapper(PATTERN_BAD_BLANK, text);
88             String replaceSpaceAfterComma = matcher.replaceAll("$1.$2");
89     
90             final Pattern pattern = LatLon.LAT == latlon ? PATTERN_LAT : PATTERN_LON;
91             matcher = new MatcherWrapper(pattern, replaceSpaceAfterComma);
92     
93             try {
94                 return new ResultWrapper(Double.valueOf(replaceSpaceAfterComma), 0, text.length());
95             } catch (NumberFormatException e1) {
96                 // fall through to advanced parsing
97             }
98     
99             if (matcher.find()) {
100                 final double sign = matcher.group(1).equalsIgnoreCase("S") || matcher.group(1).equalsIgnoreCase("W") ? -1.0 : 1.0;
101                 final double degree = Integer.valueOf(matcher.group(2)).doubleValue();
102     
103                 double minutes = 0.0;
104                 double seconds = 0.0;
105     
106                 if (null != matcher.group(3)) {
107                     minutes = Integer.valueOf(matcher.group(3)).doubleValue();
108     
109                     if (null != matcher.group(4)) {
110                         seconds = Double.parseDouble("0." + matcher.group(4)) * 60.0;
111                     } else if (null != matcher.group(5)) {
112                         seconds = Double.parseDouble(matcher.group(5).replace(",", "."));
113                     }
114                 }
115     
116                 return new ResultWrapper(sign * (degree + minutes / 60.0 + seconds / 3600.0), matcher.start(), matcher.group().length());
117     
118             }
119     
120             // Nothing found with "N 52...", try to match string as decimal degree parts (i.e. multiple doubles)
121             try {
122                 final String[] items = StringUtils.split(text.trim());
123                 if (items.length > 0 && items.length <= 2) {
124                     final int index = (latlon == LatLon.LON ? items.length - 1 : 0);
125                     final String textPart = items[index];
126                     final int pos = (latlon == LatLon.LON ? text.lastIndexOf(textPart) : text.indexOf(textPart));
127                     return new ResultWrapper(Double.parseDouble(textPart), pos, textPart.length());
128                 }
129             } catch (NumberFormatException e) {
130                 // The right exception will be raised below.
131             }
132     
133             throw new Geopoint.ParseException("Could not parse coordinates as " + latlon + ": \"" + text + "\"", latlon);
134         }
