17      * GeoData and Direction handler.
18      * <p>
19      * To use this class, override {@link #updateGeoDir(IGeoData, float)}. You need to start the handler using
20      * {@link #start(int)}. A good place to do so might be the {@code onResume} method of the Activity. Stop the Handler
21      * accordingly in {@code onPause}.
22      *
23      * The direction is always relative to the top of the device (natural direction), and that it must
24      * be fixed using {@link DirectionProvider#getDirectionNow(float)}. When the direction is derived from the GPS,
25      * it is altered so that the fix can still be applied as if the information came from the compass.

50         public void updateDirection(final float direction) {

66         private static Observable<Float> fixedDirection() {
67             return app.directionObservable().map(new Func1<Float, Float>() {
68                 @Override
69                 public Float call(final Float direction) {
70                     final IGeoData geoData = app.currentGeo();
71                     return fixDirection(geoData, direction);
72                 }
73             });

75         }

77         private static float fixDirection(final IGeoData geoData, final float direction) {
78             final boolean useGPSBearing = !Settings.isUseCompass() || geoData.getSpeed() > 5;
79             return useGPSBearing ? DirectionProvider.reverseDirectionNow(geoData.getBearing()) : direction;
80         }

82         /**
83          * Register the current GeoDirHandler for GeoData and direction information (if the
84          * preferences allow it).
85          */
86         public Subscription start(final int flags) {
87             final CompositeSubscription subscriptions = new CompositeSubscription();
88             if ((flags & UPDATE_GEODATA) != 0) {
89                 subscriptions.add(app.geoDataObservable().observeOn(AndroidSchedulers.mainThread()).subscribe(new Action1<IGeoData>() {
90                     @Override
91                     public void call(final IGeoData geoData) {
92                         updateGeoData(geoData);
93                     }
94                 }));
95             }
96             if ((flags & UPDATE_DIRECTION) != 0) {
97                 subscriptions.add(fixedDirection().observeOn(AndroidSchedulers.mainThread()).subscribe(new Action1<Float>() {
98                     @Override
99                     public void call(final Float direction) {
100                         updateDirection(direction);
101                     }
102                 }));
103             }
104             if ((flags & UPDATE_GEODIR) != 0) {
105                 subscriptions.add(Observable.combineLatest(app.geoDataObservable(), app.directionObservable(), new Func2<IGeoData, Float, ImmutablePair<IGeoData, Float>>() {
106                     @Override
107                     public ImmutablePair<IGeoData, Float> call(final IGeoData geoData, final Float direction) {
108                         return ImmutablePair.of(geoData, fixDirection(geoData, direction));
109                     }
110                 }).observeOn(AndroidSchedulers.mainThread()).subscribe(new Action1<ImmutablePair<IGeoData, Float>>() {
111                     @Override
112                     public void call(final ImmutablePair<IGeoData, Float> geoDir) {
113                         updateGeoDir(geoDir.left, geoDir.right);
114                     }
115                 }));
116             }
117             return subscriptions;
118         }
