336          * Attempt to recreate the database if opening has failed

338          * @param dbHelper dbHelper to use to reopen the database

340         private static void recreateDatabase(final DbHelper dbHelper) {
341             final File dbPath = databasePath();
342             final File corruptedPath = new File(LocalStorage.getStorage(), dbPath.getName() + ".corrupted");
343             if (LocalStorage.copy(dbPath, corruptedPath)) {
344                 Log.i("DataStore.init: renamed " + dbPath + " into " + corruptedPath);
345             } else {
346                 Log.e("DataStore.init: unable to rename corrupted database");
347             }
348             try {
349                 database = dbHelper.getWritableDatabase();
350             } catch (Exception f) {
351                 Log.e("DataStore.init: unable to recreate database and open it for R/W", f);
352             }
353         }

866          *
867          * @param db
868          *            the read-write database to use
869          */
870         private static void removeObsoleteCacheDirectories(final SQLiteDatabase db) {
871             final File[] files = LocalStorage.getStorage().listFiles();
872             if (ArrayUtils.isNotEmpty(files)) {
873                 final Pattern oldFilePattern = Pattern.compile("^[GC|TB|EC|GK|O][A-Z0-9]{4,7}$");
874                 final SQLiteStatement select = db.compileStatement("select count(*) from " + dbTableCaches + " where geocode = ?");
875                 final ArrayList<File> toRemove = new ArrayList<File>(files.length);
876                 for (final File file : files) {
877                     if (file.isDirectory()) {
878                         final String geocode = file.getName();
879                         if (oldFilePattern.matcher(geocode).find()) {
880                             select.bindString(1, geocode);
881                             if (select.simpleQueryForLong() == 0) {
882                                 toRemove.add(file);
883                             }
884                         }
885                     }
886                 }
887     
888                 // Use a background thread for the real removal to avoid keeping the database locked
889                 // if we are called from within a transaction.
890                 new Thread(new Runnable() {
891                     @Override
892                     public void run() {
893                         for (final File dir : toRemove) {
894                             Log.i("Removing obsolete cache directory for " + dir.getName());
895                             LocalStorage.deleteDirectory(dir);
896                         }
897                     }
898                 }).start();
899             }

969             } catch (final Exception e) {
970                 Log.e("DataStore.isThere", e);
971             }
972

975                 return false;
976             }
977

979                 // we want to check time for detailed cache, but data are older than 3 hours

981             }

984                 // we want to check time for short cache, but data are older than 3 hours
985                 return false;
986             }
987     
988             // we have some cache
989             return true;

1082          * @param saveFlags
1083          *
1084          */
1085         public static void saveCaches(Collection<Geocache> caches, EnumSet<LoadFlags.SaveFlag> saveFlags) {
1086             if (CollectionUtils.isEmpty(caches)) {
1087                 return;
1088             }
1089             final ArrayList<String> cachesFromDatabase = new ArrayList<String>();
1090             final HashMap<String, Geocache> existingCaches = new HashMap<String, Geocache>();
1091     
1092             // first check which caches are in the memory cache
1093             for (Geocache cache : caches) {
1094                 final String geocode = cache.getGeocode();
1095                 final Geocache cacheFromCache = cacheCache.getCacheFromCache(geocode);
1096                 if (cacheFromCache == null) {
1097                     cachesFromDatabase.add(geocode);
1098                 }
1099                 else {
1100                     existingCaches.put(geocode, cacheFromCache);
1101                 }
1102             }
1103     
1104             // then load all remaining caches from the database in one step
1105             for (Geocache cacheFromDatabase : loadCaches(cachesFromDatabase, LoadFlags.LOAD_ALL_DB_ONLY)) {
1106                 existingCaches.put(cacheFromDatabase.getGeocode(), cacheFromDatabase);
1107             }
1108     
1109             final ArrayList<Geocache> toBeStored = new ArrayList<Geocache>();
1110             // Merge with the data already stored in the CacheCache or in the database if
1111             // the cache had not been loaded before, and update the CacheCache.
1112             // Also, a DB update is required if the merge data comes from the CacheCache
1113             // (as it may be more recent than the version in the database), or if the
1114             // version coming from the database is different than the version we are entering
1115             // into the cache (that includes absence from the database).
1116             for (Geocache cache : caches) {
1117                 final String geocode = cache.getGeocode();
1118                 final Geocache existingCache = existingCaches.get(geocode);
1119                 final boolean dbUpdateRequired = !cache.gatherMissingFrom(existingCache) || cacheCache.getCacheFromCache(geocode) != null;
1120                 cache.addStorageLocation(StorageLocation.CACHE);
1121                 cacheCache.putCacheInCache(cache);
1122     
1123                 // Only save the cache in the database if it is requested by the caller and
1124                 // the cache contains detailed information.
1125                 if (saveFlags.contains(SaveFlag.SAVE_DB) && cache.isDetailed() && dbUpdateRequired) {
1126                     toBeStored.add(cache);
1127                 }
1128             }
1129     
1130             for (Geocache geocache : toBeStored) {
1131                 storeIntoDatabase(geocache);
1132             }
1133         }

1566                 return new HashSet<Geocache>();
1567             }

1585                     loadFlags.contains(LoadFlag.LOAD_SPOILERS) ||
1586                     loadFlags.contains(LoadFlag.LOAD_LOGS) ||
1587                     loadFlags.contains(LoadFlag.LOAD_INVENTORY) ||
1588                     loadFlags.contains(LoadFlag.LOAD_OFFLINE_LOG)) {
1589     
1590                 final Set<Geocache> cachesFromDB = loadCachesFromGeocodes(remaining, loadFlags);
1591                 result.addAll(cachesFromDB);
1592                 for (final Geocache cache : cachesFromDB) {
1593                     remaining.remove(cache.getGeocode());
1594                 }
1595             }
1596     
1597             if (loadFlags.contains(LoadFlag.LOAD_CACHE_AFTER)) {
1598                 for (String geocode : new HashSet<String>(remaining)) {
1599                     Geocache cache = cacheCache.getCacheFromCache(geocode);
1600                     if (cache != null) {
1601                         result.add(cache);
1602                         remaining.remove(cache.getGeocode());
1603                     }
1604                 }
1605             }
1606     
1607             if (remaining.size() >= 1) {
1608                 Log.d("DataStore.loadCaches(" + remaining.toString() + ") returned no results");
1609             }
1610             return result;
1611         }

1613         /**
1614          * Load caches.
1615          *
1616          * @param geocodes
1617          * @param loadFlags
1618          * @return Set of loaded caches. Never null.
1619          */
1620         private static Set<Geocache> loadCachesFromGeocodes(final Set<String> geocodes, final EnumSet<LoadFlag> loadFlags) {
1621             if (CollectionUtils.isEmpty(geocodes)) {
1622                 return Collections.emptySet();
1623             }
1624     
1625             // do not log the entire collection of geo codes to the debug log. This can be more than 100 KB of text for large lists!
1626             init();
1627     
1628             final StringBuilder query = new StringBuilder(QUERY_CACHE_DATA);
1629             if (loadFlags.contains(LoadFlag.LOAD_OFFLINE_LOG)) {
1630                 query.append(',').append(dbTableLogsOffline).append(".log");
1631             }
1632     
1633             query.append(" FROM ").append(dbTableCaches);
1634             if (loadFlags.contains(LoadFlag.LOAD_OFFLINE_LOG)) {
1635                 query.append(" LEFT OUTER JOIN ").append(dbTableLogsOffline).append(" ON ( ").append(dbTableCaches).append(".geocode == ").append(dbTableLogsOffline).append(".geocode) ");
1636             }
1637     
1638             query.append(" WHERE ").append(dbTableCaches).append('.');
1639             query.append(DataStore.whereGeocodeIn(geocodes));
1640     
1641             Cursor cursor = database.rawQuery(query.toString(), null);
1642             try {
1643                 final Set<Geocache> caches = new HashSet<Geocache>();

1645     
1646                 while (cursor.moveToNext()) {
1647                     Geocache cache = DataStore.createCacheFromDatabaseContent(cursor);

1650                         cache.setAttributes(loadAttributes(cache.getGeocode()));

1654                         final List<Waypoint> waypoints = loadWaypoints(cache.getGeocode());

1661                         final List<Image> spoilers = loadSpoilers(cache.getGeocode());

1666                         final Map<LogType, Integer> logCounts = loadLogCounts(cache.getGeocode());
1667                         if (MapUtils.isNotEmpty(logCounts)) {
1668                             cache.getLogCounts().clear();
1669                             cache.getLogCounts().putAll(logCounts);
1670                         }
1671                     }
1672

1679                                 cache.getInventory().clear();
1680                             }

1690                     }
1691                     cache.addStorageLocation(StorageLocation.DATABASE);
1692                     cacheCache.putCacheInCache(cache);
