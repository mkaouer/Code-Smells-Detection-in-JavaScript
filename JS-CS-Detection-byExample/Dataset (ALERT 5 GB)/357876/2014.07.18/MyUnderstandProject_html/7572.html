17     /**
18      * Field Notes are simple plain text files, but poorly documented. Syntax:<br>
19      * <code>GCxxxxx,yyyy-mm-ddThh:mm:ssZ,Found it,"logtext"</code>
20      */
21     class FieldNotes {
22     
23         private static final SynchronizedDateFormat FIELD_NOTE_DATE_FORMAT = new SynchronizedDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'", TimeZone.getTimeZone("UTC"), Locale.US);
24     
25         private int size = 0;
26         private final StringBuilder buffer = new StringBuilder();
27     
28         void add(final Geocache cache, final LogEntry log) {
29             size++;
30             buffer.append(cache.getGeocode())
31                     .append(',')
32                     .append(FIELD_NOTE_DATE_FORMAT.format(new Date(log.date)))
33                     .append(',')
34                     .append(StringUtils.capitalize(log.type.type))
35                     .append(",\"")
36                     .append(StringUtils.replaceChars(log.log, '"', '\''))
37                     .append("\"\n");
38         }
39     
40         public String getContent() {
41             return buffer.toString();
42         }
43     
44         File writeToDirectory(File exportLocation) {
45             if (!LocalStorage.isExternalStorageAvailable()) {
46                 return null;
47             }
48     
49             FileUtils.mkdirs(exportLocation);
50     
51             final SimpleDateFormat fileNameDateFormat = new SimpleDateFormat("yyyyMMddHHmmss", Locale.US);
52             final File exportFile = new File(exportLocation.toString() + '/' + fileNameDateFormat.format(new Date()) + ".txt");
53     
54             if (!FileUtils.writeFileUTF16(exportFile, getContent())) {
55                 return null;
56             }
57     
58             return exportFile;
59         }
60     
61         public int size() {
62             return size;
63         }
64     
65     }
