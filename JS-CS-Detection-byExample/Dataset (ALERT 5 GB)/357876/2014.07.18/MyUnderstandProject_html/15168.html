22     public abstract class AbstractLoggingActivity extends AbstractActivity {
23     
24         @Override
25         public boolean onCreateOptionsMenu(final Menu menu) {
26             getMenuInflater().inflate(R.menu.abstract_logging_activity, menu);
27     
28             final SubMenu menuLog = menu.findItem(R.id.menu_templates).getSubMenu();
29             for (final LogTemplate template : LogTemplateProvider.getTemplates()) {
30                 menuLog.add(0, template.getItemId(), 0, template.getResourceId());
31             }
32     
33             final SubMenu menuSmilies = menu.findItem(R.id.menu_smilies).getSubMenu();
34             for (final Smiley smiley : GCSmiliesProvider.getSmilies()) {
35                 menuSmilies.add(0, smiley.getItemId(), 0, smiley.text);
36             }
37     
38             return true;
39         }
40     
41         @Override
42         public boolean onPrepareOptionsMenu(Menu menu) {
43             final boolean signatureAvailable = StringUtils.isNotBlank(Settings.getSignature());
44             menu.findItem(R.id.menu_signature).setVisible(signatureAvailable);
45     
46             boolean smileyVisible = false;
47             final Geocache cache = getLogContext().getCache();
48             if (cache != null && ConnectorFactory.getConnector(cache).equals(GCConnector.getInstance())) {
49                 smileyVisible = true;
50             }
51             final Trackable trackable = getLogContext().getTrackable();
52             if (trackable != null && ConnectorFactory.getConnector(trackable).equals(TravelBugConnector.getInstance())) {
53                 smileyVisible = true;
54             }
55     
56             menu.findItem(R.id.menu_smilies).setVisible(smileyVisible);
57     
58             return true;
59         }
60     
61         @Override
62         public boolean onOptionsItemSelected(MenuItem item) {
63             final int id = item.getItemId();
64     
65             if (id == R.id.menu_signature) {
66                 insertIntoLog(LogTemplateProvider.applyTemplates(Settings.getSignature(), getLogContext()), true);
67                 return true;
68             }
69     
70             final LogTemplate template = LogTemplateProvider.getTemplate(id);
71             if (template != null) {
72                 insertIntoLog(template.getValue(getLogContext()), true);
73                 return true;
74             }
75     
76             final Smiley smiley = GCSmiliesProvider.getSmiley(id);
77             if (smiley != null) {
78                 insertIntoLog("[" + smiley.text + "]", true);
79                 return true;
80             }
81     
82             return false;
83         }
84     
85         protected abstract LogContext getLogContext();
86     
87         protected void insertIntoLog(String newText, final boolean moveCursor) {
88             final EditText log = (EditText) findViewById(R.id.log);
89             ActivityMixin.insertAtPosition(log, newText, moveCursor);
90         }
