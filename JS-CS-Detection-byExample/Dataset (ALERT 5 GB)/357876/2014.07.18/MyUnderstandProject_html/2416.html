8     /**
9      * Standard handler implementation which uses a weak reference to its activity. This avoids that activities stay in
10      * memory due to references from the handler to the activity (see Android Lint warning "HandlerLeak")
11      *
12      * Create static private subclasses of this handler class in your activity.
13      *
14      * @param <ActivityType>
15      */
16     public abstract class WeakReferenceHandler<ActivityType extends Activity> extends Handler {
17     
18         private final WeakReference<ActivityType> activityRef;
19     
20         protected WeakReferenceHandler(final ActivityType activity) {
21             this.activityRef = new WeakReference<ActivityType>(activity);
22         }
23     
24         protected ActivityType getActivity() {
25             return activityRef.get();
26         }
27     }
