74         @Override
75         protected StatusCode login(boolean retry) {
76             final ImmutablePair<String, String> credentials = Settings.getGcCredentials();
77             final String username = credentials.left;
78             final String password = credentials.right;
79     
80             if (StringUtils.isEmpty(username) || StringUtils.isEmpty(password)) {
81                 clearLoginInfo();
82                 Log.e("Login.login: No login information stored");
83                 return StatusCode.NO_LOGIN_INFO_STORED;
84             }
85     
86             setActualStatus(CgeoApplication.getInstance().getString(R.string.init_login_popup_working));
87             HttpResponse loginResponse = Network.getRequest("https://www.geocaching.com/login/default.aspx");
88             String loginData = Network.getResponseData(loginResponse);
89             if (loginResponse != null && loginResponse.getStatusLine().getStatusCode() == 503 && TextUtils.matches(loginData, GCConstants.PATTERN_MAINTENANCE)) {
90                 return StatusCode.MAINTENANCE;
91             }
92     
93             if (StringUtils.isBlank(loginData)) {
94                 Log.e("Login.login: Failed to retrieve login page (1st)");
95                 return StatusCode.CONNECTION_FAILED; // no login page
96             }
97     
98             if (getLoginStatus(loginData)) {
99                 Log.i("Already logged in Geocaching.com as " + username + " (" + Settings.getGCMemberStatus() + ')');
100                 if (switchToEnglish(loginData) && retry) {
101                     return login(false);
102                 }
103                 return StatusCode.NO_ERROR; // logged in
104             }
105     
106             Cookies.clearCookies();
107             Settings.setCookieStore(null);
108     
109             final Parameters params = new Parameters(
110                     "__EVENTTARGET", "",
111                     "__EVENTARGUMENT", "",
112                     "ctl00$ContentBody$tbUsername", username,
113                     "ctl00$ContentBody$tbPassword", password,
114                     "ctl00$ContentBody$cbRememberMe", "on",
115                     "ctl00$ContentBody$btnSignIn", "Login");
116             final String[] viewstates = GCLogin.getViewstates(loginData);
117             if (isEmpty(viewstates)) {
118                 Log.e("Login.login: Failed to find viewstates");
119                 return StatusCode.LOGIN_PARSE_ERROR; // no viewstates
120             }
121             GCLogin.putViewstates(params, viewstates);
122     
123             loginResponse = Network.postRequest("https://www.geocaching.com/login/default.aspx", params);
124             loginData = Network.getResponseData(loginResponse);
125     
126             if (StringUtils.isBlank(loginData)) {
127                 Log.e("Login.login: Failed to retrieve login page (2nd)");
128                 // FIXME: should it be CONNECTION_FAILED to match the first attempt?
129                 return StatusCode.COMMUNICATION_ERROR; // no login page
130             }
131             assert loginData != null;  // Caught above
132     
133             if (getLoginStatus(loginData)) {
134                 Log.i("Successfully logged in Geocaching.com as " + username + " (" + Settings.getGCMemberStatus() + ')');
135     
136                 if (switchToEnglish(loginData) && retry) {
137                     return login(false);
138                 }
139                 Settings.setCookieStore(Cookies.dumpCookieStore());
140     
141                 return StatusCode.NO_ERROR; // logged in
142             }
143     
144             if (loginData.contains("Your username/password combination does not match.")) {
145                 Log.i("Failed to log in Geocaching.com as " + username + " because of wrong username/password");
146                 return StatusCode.WRONG_LOGIN_DATA; // wrong login
147             }
148     
149             if (loginData.contains("You must validate your account before you can log in.")) {
150                 Log.i("Failed to log in Geocaching.com as " + username + " because account needs to be validated first");
151                 return StatusCode.UNVALIDATED_ACCOUNT;
152             }
153     
154             Log.i("Failed to log in Geocaching.com as " + username + " for some unknown reason");
155             if (retry) {
156                 switchToEnglish(loginData);
157                 return login(false);
158             }
159     
160             return StatusCode.UNKNOWN_ERROR; // can't login
161         }

163         public StatusCode logout() {
164             final HttpResponse logoutResponse = Network.getRequest("https://www.geocaching.com/login/default.aspx?RESET=Y&redir=http%3a%2f%2fwww.geocaching.com%2fdefault.aspx%3f");
165             final String logoutData = Network.getResponseData(logoutResponse);
166             if (logoutResponse != null && logoutResponse.getStatusLine().getStatusCode() == 503 && TextUtils.matches(logoutData, GCConstants.PATTERN_MAINTENANCE)) {
167                 return StatusCode.MAINTENANCE;
168             }
169     
170             resetLoginStatus();
171     
172             return StatusCode.NO_ERROR;
173         }

176         /**
177          * Check if the user has been logged in when he retrieved the data.
178          *
179          * @param page
180          * @return <code>true</code> if user is logged in, <code>false</code> otherwise
181          */
182         public boolean getLoginStatus(@Nullable final String page) {
183             if (StringUtils.isBlank(page)) {
184                 Log.e("Login.checkLogin: No page given");
185                 return false;
186             }
187             assert page != null;
188     
189             if (TextUtils.matches(page, GCConstants.PATTERN_MAP_LOGGED_IN)) {
190                 return true;
191             }
192     
193             setActualStatus(CgeoApplication.getInstance().getString(R.string.init_login_popup_ok));
194     
195             // on every page except login page
196             setActualLoginStatus(TextUtils.matches(page, GCConstants.PATTERN_LOGIN_NAME));
197             if (isActualLoginStatus()) {
198                 setActualUserName(TextUtils.getMatch(page, GCConstants.PATTERN_LOGIN_NAME, true, "???"));
199                 int cachesCount = 0;
200                 try {
201                     cachesCount = Integer.parseInt(TextUtils.getMatch(page, GCConstants.PATTERN_CACHES_FOUND, true, "0").replaceAll("[,.]", ""));
202                 } catch (final NumberFormatException e) {
203                     Log.e("getLoginStatus: bad cache count", e);
204                 }
205                 setActualCachesFound(cachesCount);
206                 Settings.setGCMemberStatus(TextUtils.getMatch(page, GCConstants.PATTERN_MEMBER_STATUS, true, null));
207                 if (page.contains(GCConstants.MEMBER_STATUS_RENEW)) {
208                     Settings.setGCMemberStatus(GCConstants.MEMBER_STATUS_PM);
209                 }
210                 return true;
211             }
212     
213             // login page
214             setActualLoginStatus(TextUtils.matches(page, GCConstants.PATTERN_LOGIN_NAME_LOGIN_PAGE));
215             if (isActualLoginStatus()) {
216                 setActualUserName(Settings.getUsername());
217                 // number of caches found is not part of this page
218                 return true;
219             }
220     
221             setActualStatus(CgeoApplication.getInstance().getString(R.string.init_login_popup_failed));
222             return false;
223         }

225         /**
226          * Ensure that the web site is in English.
227          *
228          * @param previousPage the content of the last loaded page
229          * @return <code>true</code> if a switch was necessary and succesfully performed (non-English -> English)
230          */
231         private boolean switchToEnglish(String previousPage) {
232             if (previousPage != null && previousPage.contains(ENGLISH)) {
233                 Log.i("Geocaching.com language already set to English");
234                 // get find count
235                 getLoginStatus(Network.getResponseData(Network.getRequest("http://www.geocaching.com/email/")));
236             } else {
237                 final String page = Network.getResponseData(Network.getRequest(LANGUAGE_CHANGE_URI));
238                 getLoginStatus(page);
239                 if (page == null) {
240                     Log.e("Failed to read viewstates to set geocaching.com language");
241                 }
242                 final Parameters params = new Parameters(
243                         "__EVENTTARGET", "ctl00$uxLocaleList$uxLocaleList$ctl00$uxLocaleItem", // switch to english
244                         "__EVENTARGUMENT", "");
245                 GCLogin.transferViewstates(page, params);
246                 final HttpResponse response = Network.postRequest(LANGUAGE_CHANGE_URI, params, new Parameters("Referer", LANGUAGE_CHANGE_URI));
247                 if (Network.isSuccess(response)) {
248                     Log.i("changed language on geocaching.com to English");
249                     return true;
250                 }
251                 Log.e("Failed to set geocaching.com language to English");
252             }
253             return false;
254         }

299         public static Date parseGcCustomDate(final String input, final String format) throws ParseException {
300             if (StringUtils.isBlank(input)) {
301                 throw new ParseException("Input is null", 0);
302             }
303     
304             final String trimmed = input.trim();
305     
306             if (GC_CUSTOM_DATE_FORMATS.containsKey(format)) {
307                 try {
308                     return GC_CUSTOM_DATE_FORMATS.get(format).parse(trimmed);
309                 } catch (final ParseException e) {
310                 }
311             }
312     
313             for (final SimpleDateFormat sdf : GC_CUSTOM_DATE_FORMATS.values()) {
314                 try {
315                     return sdf.parse(trimmed);
316                 } catch (final ParseException e) {
317                 }
318             }
319     
320             throw new ParseException("No matching pattern", 0);
321         }

354         /**
355          * read all viewstates from page
356          *
357          * @return String[] with all view states
358          */
359         public static String[] getViewstates(String page) {
360             // Get the number of viewstates.
361             // If there is only one viewstate, __VIEWSTATEFIELDCOUNT is not present
362     
363             if (page == null) { // no network access
364                 return null;
365             }
366     
367             int count = 1;
368             final MatcherWrapper matcherViewstateCount = new MatcherWrapper(GCConstants.PATTERN_VIEWSTATEFIELDCOUNT, page);
369             if (matcherViewstateCount.find()) {
370                 try {
371                     count = Integer.parseInt(matcherViewstateCount.group(1));
372                 } catch (final NumberFormatException e) {
373                     Log.e("getViewStates", e);
374                 }
375             }
376     
377             final String[] viewstates = new String[count];
378     
379             // Get the viewstates
380             final MatcherWrapper matcherViewstates = new MatcherWrapper(GCConstants.PATTERN_VIEWSTATES, page);
381             while (matcherViewstates.find()) {
382                 final String sno = matcherViewstates.group(1); // number of viewstate
383                 int no;
384                 if (StringUtils.isEmpty(sno)) {
385                     no = 0;
386                 } else {
387                     try {
388                         no = Integer.parseInt(sno);
389                     } catch (final NumberFormatException e) {
390                         Log.e("getViewStates", e);
391                         no = 0;
392                     }
393                 }
394                 viewstates[no] = matcherViewstates.group(2);
395             }
396     
397             if (viewstates.length != 1 || viewstates[0] != null) {
398                 return viewstates;
399             }
400             // no viewstates were present
401             return null;
402         }

404         /**
405          * put viewstates into request parameters
406          */
407         public static void putViewstates(final Parameters params, final String[] viewstates) {
408             if (ArrayUtils.isEmpty(viewstates)) {
409                 return;
410             }
411             params.put("__VIEWSTATE", viewstates[0]);
412             if (viewstates.length > 1) {
413                 for (int i = 1; i < viewstates.length; i++) {
414                     params.put("__VIEWSTATE" + i, viewstates[i]);
415                 }
416                 params.put("__VIEWSTATEFIELDCOUNT", String.valueOf(viewstates.length));
417             }
418         }

449         /**
450          * GET HTTP request. Do the request a second time if the user is not logged in
451          *
452          * @param uri
453          * @param params
454          * @return
455          */
456         @Nullable
457         public String getRequestLogged(@NonNull final String uri, @Nullable final Parameters params) {
458             final HttpResponse response = Network.getRequest(uri, params);
459             final String data = Network.getResponseData(response, canRemoveWhitespace(uri));
460     
461             // A page not found will not be found if the user logs in either
462             if (Network.isPageNotFound(response) || getLoginStatus(data)) {
463                 return data;
464             }
465     
466             if (login() == StatusCode.NO_ERROR) {
467                 return Network.getResponseData(Network.getRequest(uri, params), canRemoveWhitespace(uri));
468             }
469     
470             Log.w("Working as guest.");
471             return data;
472         }
