68         @Override
69         public void displayZoomControls(boolean takeFocus) {
70             try {
71                 // Push zoom controls to the right
72                 FrameLayout.LayoutParams zoomParams = new FrameLayout.LayoutParams(WRAP_CONTENT, WRAP_CONTENT);
73                 zoomParams.gravity = Gravity.RIGHT;
74                 // The call to retrieve the zoom buttons controller is undocumented and works so far on all devices
75                 // supported by Google Play, but fails at least on one Jolla.
76                 final ZoomButtonsController controller = (ZoomButtonsController) MethodUtils.invokeMethod(this, "getZoomButtonsController");
77                 controller.getZoomControls().setLayoutParams(zoomParams);
78     
79                 super.displayZoomControls(takeFocus);
80             } catch (NoSuchMethodException e) {
81                 Log.w("GoogleMapView.displayZoomControls: unable to explicitly place the zoom buttons");
82             } catch (Exception e) {
83                 Log.e("GoogleMapView.displayZoomControls", e);
84             }
85         }

161         private class GestureListener extends SimpleOnGestureListener {
162             @Override
163             public boolean onDoubleTap(MotionEvent e) {
164                 getController().zoomInFixing((int) e.getX(), (int) e.getY());
165                 if (onDragListener != null) {
166                     onDragListener.onDrag();
167                 }
168                 return true;
169             }
