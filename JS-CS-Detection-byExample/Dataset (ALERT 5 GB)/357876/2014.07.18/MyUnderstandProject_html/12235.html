31     
32         @Override
33         public View getView(final int position, final View rowView, final ViewGroup parent) {
34             if (position > getCount()) {
35                 Log.w("FileSelectionListAdapter.getView: Attempt to access missing item #" + position);
36                 return null;
37             }
38     
39             File file = getItem(position);
40     
41             View v = rowView;
42     
43             ViewHolder holder;
44             if (v == null) {
45                 v = inflater.inflate(R.layout.mapfile_item, null);
46                 holder = new ViewHolder(v);
47             } else {
48                 holder = (ViewHolder) v.getTag();
49             }
50     
51             String currentFile = parentView.getCurrentFile();
52             if (currentFile != null && file.equals(new File(currentFile))) {
53                 holder.filename.setTypeface(holder.filename.getTypeface(), Typeface.BOLD);
54             } else {
55                 holder.filename.setTypeface(holder.filename.getTypeface(), Typeface.NORMAL);
56             }
57     
58             final TouchListener touchLst = new TouchListener(file);
59             v.setOnClickListener(touchLst);
60     
61             holder.filepath.setText(file.getParent());
62             holder.filename.setText(file.getName());
63     
64             return v;
65         }
66     
67         private class TouchListener implements View.OnClickListener {
68             private File file = null;
69     
70             public TouchListener(File fileIn) {
71                 file = fileIn;
72             }
73     
74             // tap on item
75             @Override
76             public void onClick(View view) {
77                 parentView.setCurrentFile(file.toString());

79             }
80         }
