51         private static class Rot13Encryption {
52             private boolean plaintext = false;
53     
54             char getNextEncryptedCharacter(final char c) {
55                 int result = c;
56                 if (result == '[') {
57                     plaintext = true;
58                 } else if (result == ']') {
59                     plaintext = false;
60                 } else if (!plaintext) {
61                     int capitalized = result & 32;
62                     result &= ~capitalized;
63                     result = ((result >= 'A') && (result <= 'Z') ? ((result - 'A' + 13) % 26 + 'A') : result)
64                             | capitalized;
65                 }
66                 return (char) result;
67             }
68         }

86         public static String md5(String text) {
87             try {
88                 final MessageDigest digest = MessageDigest.getInstance("MD5");
89                 digest.update(text.getBytes(CharEncoding.UTF_8), 0, text.length());
90                 return new BigInteger(1, digest.digest()).toString(16);
91             } catch (NoSuchAlgorithmException e) {
92                 Log.e("CryptUtils.md5", e);
93             } catch (UnsupportedEncodingException e) {
94                 Log.e("CryptUtils.md5", e);
95             }

97             return StringUtils.EMPTY;

132         public static String base64Encode(byte[] in) {
133             int iLen = in.length;
134             int oDataLen = (iLen * 4 + 2) / 3; // output length without padding
135             int oLen = ((iLen + 2) / 3) * 4; // output length including padding
136             char[] out = new char[oLen];
137             int ip = 0;
138             int op = 0;
139     
140             while (ip < iLen) {
141                 int i0 = in[ip++] & 0xff;
142                 int i1 = ip < iLen ? in[ip++] & 0xff : 0;
143                 int i2 = ip < iLen ? in[ip++] & 0xff : 0;
144                 int o0 = i0 >>> 2;
145                 int o1 = ((i0 & 3) << 4) | (i1 >>> 4);
146                 int o2 = ((i1 & 0xf) << 2) | (i2 >>> 6);
147                 int o3 = i2 & 0x3F;
148                 out[op++] = base64map1[o0];
149                 out[op++] = base64map1[o1];
150                 out[op] = op < oDataLen ? base64map1[o2] : '=';
151                 op++;
152                 out[op] = op < oDataLen ? base64map1[o3] : '=';
153                 op++;
154             }
