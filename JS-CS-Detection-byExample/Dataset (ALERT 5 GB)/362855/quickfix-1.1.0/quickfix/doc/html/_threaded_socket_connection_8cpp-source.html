00150   {
00151     std::string msg;
00152     while(!m_deleted)
00153     {
00154       try
00155       {
00156         while(readMessage(msg))
00157         {
00158           if( !m_pSession )
00159           {
00160             if(!setSession(msg))
00161             { disconnect(); continue; }
00162           }
00163           m_pSession->next(msg);
00164         }
00165 
00166         if(m_pSession)

00179       }

00188 
00189     if(m_pSession)
