00063   {
00064     ConfigLexer lexer(&stream);
00065     int lcState = 0;
00066     Settings::Sections::iterator currentSection = s.m_sections.end();
00067     std::string currentName;
00068     while (0 != (lcState = lexer.yylex()))
00069       {
00070         switch(lcState)
00071           {
00072           case ConfigLexer::LC_STATE_SECTION:
00073             currentSection = s.m_sections.insert
00074               (s.m_sections.end(),
00075                Dictionary(lexer.YYText()));
00076             break;
00077           case ConfigLexer::LC_STATE_NAME:
00078             currentName = lexer.YYText();
00079             break;
00080           case ConfigLexer::LC_STATE_VALUE:
00081             (*currentSection).setString(currentName,lexer.YYText());
00082             break;
00083           }
00084       }
00085     return stream;
00086   }
