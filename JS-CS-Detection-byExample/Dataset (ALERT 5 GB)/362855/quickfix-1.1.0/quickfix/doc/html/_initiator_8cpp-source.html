00064   Initiator::Initiator( Application& application,
00065                         MessageStoreFactory& factory,
00066                         const SessionSettings& settings )
00067     throw( ConfigError& )
00068     : m_application(application), m_factory(factory), m_settings(settings)
00069   {
00070     std::set<SessionID> sessions = settings.getSessions();
00071     std::set<SessionID>::iterator i;
00072 
00073     if(!sessions.size())
00074       throw ConfigError("No sessions defined");
00075 
00076     for(i = sessions.begin(); i != sessions.end(); ++i)
00077       {
00078         if(settings.get(*i).getString("ConnectionType") == "initiator")
00079           {
00080             Dictionary dict = settings.get(*i);
00081             std::string url;
00082             if(dict.has("DataDictionary"))
00083             url = dict.getString("DataDictionary");
00084 
00085             HeartBtInt heartBtInt =
00086               HeartBtInt((int)m_settings.get(*i).getLong("HeartBtInt"));
00087 
00088             UtcTimeOnly startTime;
00089             UtcTimeOnly endTime;
00090             try
00091             {
00092               startTime = UtcTimeOnlyConvertor::convert
00093                 (settings.get(*i).getString("StartTime"));
00094               endTime = UtcTimeOnlyConvertor::convert
00095                 (settings.get(*i).getString("EndTime"));
00096             }
00097             catch(FieldConvertError& e)
00098             {
00099               throw ConfigError(e.what());
00100             }
00101 
00102             m_sessions[*i] =
00103                new Session(m_application, m_factory, *i, 
00104                           url.length() ? url : DataDictionary(),
00105                           startTime, endTime, heartBtInt);
00106             m_application.onCreate(*i);
00107             setConnected(*i, false);
00108           }
00109         if(!m_sessions.size())
00110           throw ConfigError("No sessions defined for initiator");
00111       }
00112   }
