00064   Acceptor::Acceptor( Application& application,
00065                       MessageStoreFactory& factory,
00066                       const SessionSettings& settings )
00067     throw( ConfigError& )
00068     : m_application(application), m_settings(settings), m_factory(factory)
00069   {
00070     std::set<SessionID> sessions = settings.getSessions();
00071     std::set<SessionID>::iterator i;
00072 
00073     if(!sessions.size())
00074       throw ConfigError("No sessions defined");
00075 
00076     for(i = sessions.begin(); i != sessions.end(); ++i)
00077       {
00078         if(settings.get(*i).getString("ConnectionType") == "acceptor")
00079           {
00080             Dictionary dict = settings.get(*i);
00081             std::string url;
00082             if(dict.has("DataDictionary"))
00083               url = dict.getString("DataDictionary");
00084 
00085             UtcTimeOnly startTime;
00086             UtcTimeOnly endTime;
00087             try
00088             {
00089               startTime = UtcTimeOnlyConvertor::convert
00090                 (settings.get(*i).getString("StartTime"));
00091               endTime = UtcTimeOnlyConvertor::convert
00092                 (settings.get(*i).getString("EndTime"));
00093             }
00094             catch(FieldConvertError& e)
00095             {
00096               throw ConfigError(e.what());
00097             }
00098 
00099             m_sessions[*i] =
00100               new Session(m_application, m_factory, *i, 
00101                           url.length() ? url : DataDictionary(),
00102                           startTime, endTime);
00103             m_application.onCreate(*i);
00104           }
00105         if(!m_sessions.size())
00106           throw ConfigError("No sessions defined for acceptor");
00107       }

00120   ( const std::string& msg, Session::Responder& responder )
00121   {
00122     Message message;
00123     if( !message.setStringHeader(msg) )
00124       return 0;
00125 
00126     BeginString beginString;
00127     SenderCompID clSenderCompID;
00128     TargetCompID clTargetCompID;
00129     MsgType msgType;
00130     try
00131       {
00132         message.getHeader().getField(beginString);
00133         message.getHeader().getField(clSenderCompID);
00134         message.getHeader().getField(clTargetCompID);
00135         message.getHeader().getField(msgType);
00136         if( msgType != "A" ) return 0;
00137 
00138         SenderCompID senderCompID(clTargetCompID);
00139         TargetCompID targetCompID(clSenderCompID);
00140         SessionID sessionID(beginString, senderCompID, targetCompID);
00141 
00142         Sessions::iterator i = m_sessions.find(sessionID);
00143         if(i != m_sessions.end())
00144           {
00145             i->second->setResponder(&responder);
00146             return i->second;
00147           }
00148       }
00149     catch( FieldNotFound& ) {}
00150     return 0;
