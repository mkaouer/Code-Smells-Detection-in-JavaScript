00148 void ThreadedSocketConnection::readQueue()
00149 {
00150   std::string msg;
00151   while ( !m_deleted )
00152   {
00153     try
00154     {
00155       while ( readMessage( msg ) )
00156       {
00157         if ( !m_pSession )
00158         {
00159           if ( !setSession( msg ) )
00160           { disconnect(); continue; }
00161         }
00162         m_pSession->next( msg );
00163       }
00164 
00165       if ( m_pSession )
00166         m_pSession->next();
00167 
00168       if ( !m_queue.size() )
00169         m_queue.wait();
00170 
00171       char* buffer = 0;
00172       while ( m_queue.pop( buffer ) )
00173       {
00174         if ( buffer == 0 ) throw std::exception();
00175         m_parser.addToStream( buffer );
00176         delete [] buffer;
00177       }
00178     }
00179     catch ( RecvFailed& )
00180     {
00181       m_pSession->disconnect();
00182     }
00183     catch ( InvalidMessage& )
00184     {
00185       if ( !m_pSession->isLoggedOn() )
00186         m_pSession->disconnect();
00187     }
00188     catch ( std::exception& )
00189     { break; }
00190   }
00191 
00192   if ( m_pSession )
00193     m_pSession->disconnect();

00198 bool ThreadedSocketConnection::setSession( const std::string& msg )
00199 {
00200   m_pSession = Session::lookupSession( msg, true );
00201   if ( !m_pSession ) return false;
00202   SessionID sessionID = m_pSession->getSessionID();
00203   // see if the session frees up within 5 seconds
00204   for ( int i = 1; i <= 5; ++i )
00205   {
00206     m_pSession = Session::registerSession( sessionID );
00207     if ( m_pSession ) break;
00208     process_sleep( 1 );
00209   }
00210   if ( !m_pSession ) return false;
00211   m_pSession->setResponder( this );
00212   return true;
