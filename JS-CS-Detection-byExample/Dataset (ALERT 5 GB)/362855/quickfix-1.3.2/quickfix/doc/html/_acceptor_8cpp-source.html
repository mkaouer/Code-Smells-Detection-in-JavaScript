00122 void Acceptor::initialize() throw ( ConfigError& )
00123 {
00124   std::set < SessionID > sessions = m_settings.getSessions();
00125   std::set < SessionID > ::iterator i;
00126 
00127   if ( !sessions.size() )
00128     throw ConfigError( "No sessions defined" );
00129 
00130   SessionFactory factory( m_application, m_messageStoreFactory,
00131                           m_pLogFactory );
00132 
00133   for ( i = sessions.begin(); i != sessions.end(); ++i )
00134   {
00135     if ( m_settings.get( *i ).getString( CONNECTION_TYPE ) == "acceptor" )
00136       m_sessions[ *i ] = factory.create( *i, m_settings.get( *i ) );
00137   }
00138   if ( !m_sessions.size() )
00139     throw ConfigError( "No sessions defined for acceptor" );
00140 }

00149 Session* Acceptor::getSession
00150 ( const std::string& msg, Session::Responder& responder )
00151 {
00152   Message message;
00153   if ( !message.setStringHeader( msg ) )
00154     return 0;
00155 
00156   BeginString beginString;
00157   SenderCompID clSenderCompID;
00158   TargetCompID clTargetCompID;
00159   MsgType msgType;
00160   try
00161   {
00162     message.getHeader().getField( beginString );
00163     message.getHeader().getField( clSenderCompID );
00164     message.getHeader().getField( clTargetCompID );
00165     message.getHeader().getField( msgType );
00166     if ( msgType != "A" ) return 0;
00167 
00168     SenderCompID senderCompID( clTargetCompID );
00169     TargetCompID targetCompID( clSenderCompID );
00170     SessionID sessionID( beginString, senderCompID, targetCompID );
00171 
00172     Sessions::iterator i = m_sessions.find( sessionID );
00173     if ( i != m_sessions.end() )
00174     {
00175       i->second->setResponder( &responder );
00176       return i->second;
00177     }
00178   }
00179   catch ( FieldNotFound& ) {}
00180   return 0;
00181 }
