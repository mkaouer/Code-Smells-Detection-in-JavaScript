00111 inline timeval* SocketMonitor::getTimeval()
00112 {
00113   if ( !m_timeout )
00114     return 0;
00115 #ifdef SELECT_MODIFIES_TIMEVAL
00116   if ( !m_timeval.tv_sec && !m_timeval.tv_usec && m_timeout )
00117     m_timeval.tv_sec = m_timeout;

00119 #else
00120 double elapsed = ( double ) ( clock() - m_ticks ) / ( double ) CLOCKS_PER_SEC;
00121   if ( elapsed >= m_timeout || elapsed == 0.0 )
00122   {
00123     m_ticks = clock();

00128   {
00129     m_timeval.tv_sec = 0;
00130     m_timeval.tv_usec = ( long ) ( ( m_timeout - elapsed ) * 1000000 );
