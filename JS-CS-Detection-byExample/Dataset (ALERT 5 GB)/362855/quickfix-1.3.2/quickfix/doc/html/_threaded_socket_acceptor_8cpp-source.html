00103 bool ThreadedSocketAcceptor::onStart( const SessionSettings& s )
00104 {
00105   m_port = ( short ) s.get().getLong( "SocketAcceptPort" );
00106   m_socket = socket_createAcceptor( m_port );
00107 
00108   int socket = 0;
00109   while ( ( !m_stop && ( socket = socket_accept( m_socket ) ) >= 0 ) )
00110   {
00111     ThreadedSocketConnection * pConnection =
00112       new ThreadedSocketConnection( socket, getApplication() );
00113 
00114     ThreadPair* pair = new ThreadPair( this, pConnection );
00115 
00116     int thread;
00117     if ( !thread_spawn( &socketThread, pair, thread ) )
00118       delete pair;
00119     addThread( socket, thread );
00120   }
00121   return true;
00122 }
