00118         SenderCompID senderCompID(clTargetCompID);

00120         SessionID sessionID(beginString, senderCompID, targetCompID);
00121 
00122         Sessions::iterator i = m_sessions.find(sessionID);
00123         if(i != m_sessions.end())
00124           {
