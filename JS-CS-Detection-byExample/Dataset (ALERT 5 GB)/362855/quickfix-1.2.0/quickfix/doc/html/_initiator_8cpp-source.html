00064   Initiator::Initiator( Application& application,
00065                         MessageStoreFactory& factory,
00066                         const SessionSettings& settings )
00067     throw( ConfigError& )
00068     : m_application(application), m_factory(factory), m_settings(settings)
00069   {
00070     std::set<SessionID> sessions = settings.getSessions();
00071     std::set<SessionID>::iterator i;
00072 
00073     if(!sessions.size())
00074       throw ConfigError("No sessions defined");
00075 
00076     for(i = sessions.begin(); i != sessions.end(); ++i)
00077       {
00078         if(settings.get(*i).getString("ConnectionType") == "initiator")
00079           {
00080             SessionFactory factory(*i, m_application, m_factory, settings.get(*i));
00081             m_sessions[*i] = factory.create();
00082             m_application.onCreate(*i);
00083             setConnected(*i, false);
00084           }
00085         if(!m_sessions.size())
00086           throw ConfigError("No sessions defined for initiator");
00087       }
00088   }
