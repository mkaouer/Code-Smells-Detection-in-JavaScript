00148   void ThreadedSocketConnection::readQueue()
00149   {
00150     std::string msg;
00151     while(!m_deleted)
00152     {
00153       try
00154       {
00155         while(readMessage(msg))
00156         {
00157           if( !m_pSession )
00158           {
00159             if(!setSession(msg))
00160             { disconnect(); continue; }
00161           }
00162           m_pSession->next(msg);
00163         }
00164 
00165         if(m_pSession)
00166                 m_pSession->next();
00167                 
00168         if(!m_queue.size())
00169           m_queue.wait();
00170       
00171         char* buffer = 0;
00172         while(m_queue.pop(buffer))
00173         {
00174           if(buffer == 0) throw std::exception();
00175           m_parser.addToStream(buffer);
00176           delete [] buffer;
00177         }
00178       }
00179       catch( InvalidMessage& )
00180       {
00181         if(!m_pSession->isLoggedOn())
00182           m_pSession->disconnect();
00183       }
00184       catch( std::exception& )
00185       { break; }
00186     }
00187 
00188     if(m_pSession)
00189     {
00190       m_application.onLogout(m_pSession->getSessionID());
00191       m_pSession->disconnect();
00192     }

00198   {
00199     m_pSession = Session::lookupSession(msg, true);
00200     if(!m_pSession) return false;
00201     SessionID sessionID = m_pSession->getSessionID();
00202     // see if the session frees up within 5 seconds
00203     for(int i = 1; i <= 5; ++i)
00204     {      
00205       m_pSession = Session::registerSession(sessionID);
