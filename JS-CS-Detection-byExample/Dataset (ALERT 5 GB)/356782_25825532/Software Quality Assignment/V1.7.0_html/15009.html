61         public void test01ByteBitmapWithDataWith2ndBitMapBitOn() throws Exception
62         {
63     
64             ISOComponent c = new ISOBitMap(1);
65             int consumed = oneByte.unpack(c, DataWith2ndBitMapBitOn,0);
66             assertEquals("1 bytes bitmap: unpack of 1 byte bitmap should consume 2 bytes in characters",2,consumed);
67             assertEquals("1 bytes bitmap: 1 byte bitmap should result in a bitmap holding fields up to 8",8,((BitSet)c.getValue()).length() - 1);
68     
69             outBytes = oneByte.pack(c);
70             assertEquals("1 bytes bitmap: pack of 1 byte bitmap must produce a result 2 bytes long",2,outBytes.length);
71             assertEquals("1 bytes bitmap: pack of unpacked value should be the same as original", ISOUtil.hexString(DataWith2ndBitMapBitOn,0,2),ISOUtil.hexString(outBytes));
72         }
