32         public void testGetBytes() throws Exception {
33             ISOMsg m = new ISOMsg("0800");
34             m.set (3, "000000");
35             m.set (52, "CAFEBABE".getBytes());
36             m.set ("63.2.3", "Field 63.2.3");
37             m.set ("63.2.4", m.getBytes (52));
38     
39             assertEquals ("000000", m.getString (3));
40             assertEquals ("000000", new String(m.getBytes(3)));
41             assertEquals ("CAFEBABE", new String(m.getBytes(52)));
42             assertEquals ("Field 63.2.3", m.getString ("63.2.3"));
43             assertEquals ("CAFEBABE", new String(m.getBytes("63.2.4")));
44         }

46         public void testFPath() throws Exception {
47             ISOMsg m = new ISOMsg("0100");
48             
49             String mynull = null;
50             
51             assertFalse(m.hasField("63"));
52             assertFalse(m.hasField(63));
53             assertFalse(m.hasField("63.2"));
54             assertFalse(m.hasField("63.2.3"));
55             
56             m.set("63.2.3","value3");
57             m.set("63.2.4","value4");
58             
59             /*
60              * Check null processing on set matches existing processing (setting to null same as unset).
61              * 
62              * Please note m.set(fPathString,null) is amiguous and cannot be distinguished from
63              *  m.set(fieldNumberString, byte[]), so take care.
64              */
65             m.set("63.2.5",mynull);  // null
66             m.set("63.2.6",m.getString("99"));  // null as well
67             assertFalse(m.hasField("63.2.5"));
68             assertFalse(m.hasField("63.2.6"));
69                     
70             assertEquals(true,m.hasField("63.2.3"));
71             assertEquals("value3", m.getString("63.2.3"));
72             assertEquals(true,m.hasField("63.2.4"));
73             assertEquals("value4", m.getString("63.2.4"));
74             
75             assertFalse(m.hasField("63.2.999"));
76             assertFalse(m.hasField("63.2.4.999"));
77                     
78             m.unset("63.2.3");
79             
80             assertFalse(m.hasField("63.2.3"));
81             
82             assertEquals(true,m.hasField("63.2"));
83             m.unset("63.2.4");  // Removal of last remaining field triggers removal of immediate parent.
84             assertFalse(m.hasField("63.2"));
85             
86             m.unset("63.2.4.999");  // No problem removing non-existant fields.
87             
88             assertFalse(m.hasField("99"));
89             assertFalse(m.hasField(99));
90             m.set("99.99","value99");
91             m.set("99.98",mynull);
92             
93             assertEquals(true, m.hasField("99.99"));
94             assertEquals("value99", m.getString("99.99"));
95             
96             assertFalse(m.hasField("100"));
97             assertFalse(m.hasField(100));
98             m.set("100.100",mynull);
99             assertFalse(m.hasField("100.100"));
100             assertFalse(m.hasField("100"));  // Not added unnecessarily
101             assertFalse(m.hasField(100));  // Still not added unnecessarily
102             
103     
104         }

106         public void testFPathISOComponent() throws Exception {
107             ISOMsg m = new ISOMsg("0100");
108             
109             String mynull = null;
110             
111             ISOField f100 = new ISOField(100,"value100");
112             
113             assertFalse(m.hasField("100"));
114             assertFalse(m.hasField(100));
115             m.set(f100);
116             assertEquals(true, m.hasField("100"));
117             assertEquals(true, m.hasField(100));
118             assertEquals(f100,m.getComponent("100"));
119             
120             assertFalse(m.hasField("101"));
121             assertFalse(m.hasField(101));
122             m.set("101.101","value101");
123             assertEquals(true, m.hasField("101.101"));
124             assertEquals(true, m.hasField("101"));
125             assertEquals(true, m.hasField(101));
126             assertEquals(101,m.getComponent("101.101").getKey());
127             assertEquals("value101", m.getComponent("101.101").getValue());
128             m.set("101.102","value102");
129             assertEquals(true, m.hasField("101.102"));
130             assertEquals(102,m.getComponent("101.102").getKey());
131             assertEquals("value102", m.getComponent("101.102").getValue());
132     
133             ISOMsg f102 = new ISOMsg(102);
134             f102.set("1","value1");
135             f102.set("2","value2");
136             
137             assertFalse(m.hasField("102"));
138             m.set(f102);
139             assertEquals(true, m.hasField("102.1"));
140             assertEquals(true, m.hasField("102.2"));
141             
142             ISOMsg copy = new ISOMsg();
143             copy.set("101",m.getComponent("101"));
144             assertEquals(true, copy.hasField("101"));
145             assertEquals(true, copy.hasField(101));
146             assertEquals(true, copy.hasField("101.101"));
147             assertEquals(true, copy.hasField("101.102"));
148             assertEquals("value101",copy.getString("101.101"));
149             assertEquals("value102",copy.getString("101.102"));
150             
151             copy.set("102.1", m.getComponent("102.1"));
152             assertEquals(true, copy.hasField("102"));
153             assertEquals(true, copy.hasField(102));
154             assertEquals(true, copy.hasField("102.1"));
155             assertEquals("value1",copy.getString("102.1"));
156             assertFalse(copy.hasField("102.2"));
157             assertNull(copy.getString("102.2"));
158             
159             
160         }

162         public void testFPathISOEcho() throws Exception {
163             ISOMsg m = new ISOMsg("0100");
164             
165             ISOField f100 = new ISOField(100,"value100");
166             m.set(f100);
167             
168             m.set("101.101","value101");
169             m.set("101.102","value102");
170     
171             ISOMsg f102 = new ISOMsg(102);
172             f102.set("1","value1");
173             f102.set("2","value2");
174             m.set(f102);
175             
176             ISOMsg f103 = new ISOMsg(103);
177             f103.set("97","value1");
178             f103.set("98","value2");
179             f103.set("99.1","99.1");
180             m.set(f103);
181             
182             
183             m.set(10,"10");
184             m.set(11,"11");
185             m.set(12,"12");
186             m.set(13,"13");
187             m.set(14,"14");
188             m.set(15,"15");
189             m.set(16,"16");
190             
191             String[] echoList = new String[] {"10","11","16","14","101.102","103","911.911","999"};
192             
193             ISOMsg copyMsg = new ISOMsg();
194             
195             for (String fpath : echoList) {
196                 copyMsg.set(fpath, m.getComponent(fpath));
197             }
198             
199             /*
200              * Check fields in echoMsg
201              */
202             assertEquals(true,copyMsg.hasField("10"));
203             assertEquals(true,copyMsg.hasField(10));
204             assertEquals(true,copyMsg.hasField("11"));
205             assertEquals(true,copyMsg.hasField(11));
206             assertEquals(true,copyMsg.hasField("14"));
207             assertEquals(true,copyMsg.hasField(14));
208             assertEquals(true,copyMsg.hasField("16"));
209             assertEquals(true,copyMsg.hasField(16));
210             assertEquals(true,copyMsg.hasField("101"));
211             assertEquals(true,copyMsg.hasField(101));
212             assertEquals(true,copyMsg.hasField("101.102"));
213             assertEquals(true,copyMsg.hasField("103"));
214             assertEquals(true,copyMsg.hasField(103));       // Whole of 103 copied
215             assertEquals(true,copyMsg.hasField("103.97"));  // Whole of 103 copied
216             assertEquals(true,copyMsg.hasField("103.98"));  // Whole of 103 copied
217             
218             
219             /*
220              * Check fields echod but *not* present in m are not somehow present in copyMsg
221              */
222             assertFalse(copyMsg.hasField("911"));
223             assertFalse(copyMsg.hasField(911));
224             assertFalse(copyMsg.hasField("911.911"));
225             assertFalse(copyMsg.hasField("999"));
226             assertFalse(copyMsg.hasField(999));
227             
228             /*
229              * Check fields present in m but not required to be copied are not present in copyMsg
230              */
231             assertFalse(copyMsg.hasField("12"));
232             assertFalse(copyMsg.hasField(12));
233             assertFalse(copyMsg.hasField("13"));
234             assertFalse(copyMsg.hasField(13));
235             assertFalse(copyMsg.hasField("15"));
236             assertFalse(copyMsg.hasField(15));
237             assertFalse(copyMsg.hasField("100"));
238             assertFalse(copyMsg.hasField(100));
239             assertFalse(copyMsg.hasField("101.101"));
240             assertFalse(copyMsg.hasField("102"));
241             assertFalse(copyMsg.hasField(102));
242             assertFalse(copyMsg.hasField("102.1"));
243             assertFalse(copyMsg.hasField("102.2"));
244             
245             
246                     
247             
248         }
