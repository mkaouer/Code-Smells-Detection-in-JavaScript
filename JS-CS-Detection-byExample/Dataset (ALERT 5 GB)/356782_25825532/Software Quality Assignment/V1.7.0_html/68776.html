19     package org.jpos.transaction.participant;
20     
21     import org.jpos.core.ConfigurationException;
22     import org.jpos.transaction.GroupSelector;
23     import org.jpos.util.LogEvent;
24     import org.jpos.util.Logger;
25     
26     import java.io.Serializable;
27     
28     /**
29      * @author  AMarques
30      */
31     public class BSHGroupSelector extends BSHTransactionParticipant implements GroupSelector {
32         
33         protected BSHMethod selectMethod;
34         
35         public void setConfiguration(org.jdom.Element e) throws ConfigurationException {
36             super.setConfiguration(e);
37             try {
38                 selectMethod = BSHMethod.createBshMethod(e.getChild("select"));
39             } catch (Exception ex) {
40                 throw new ConfigurationException(ex.getMessage(), ex);
41             }
42         }    
43         
44         public String select(long id, java.io.Serializable context) {
45             LogEvent ev = new LogEvent(this, "select");
46             String result = null;
47             if (selectMethod != null) {
48                 try {
49                     result = (String) executeMethod(selectMethod, id, context, ev, "result");
50                 } catch (Exception ex) {
51                     ev.addMessage(ex);
52                 }
53             } 
54             if (result == null) {
55                 result = defaultSelect(id, context);
56             }
57             ev.addMessage("result", result);
58             Logger.log(ev);
59             return result;
60         }
61         
62         public String defaultSelect(long id, Serializable context) {
63             return "";
64         }
65             
66     }
