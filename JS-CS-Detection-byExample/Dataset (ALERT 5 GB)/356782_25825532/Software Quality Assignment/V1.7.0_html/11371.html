19     package org.jpos.ui.factory;
20     
21     import org.jdom.Element;
22     import org.jpos.iso.ISOChannel;
23     import org.jpos.iso.ISOUtil;
24     import org.jpos.iso.gui.ISOChannelPanel;
25     import org.jpos.iso.gui.ISOMeter;
26     import org.jpos.ui.UI;
27     import org.jpos.ui.UIFactory;
28     import org.jpos.util.NameRegistrar;
29     
30     import javax.swing.*;
31     import java.util.Observable;
32     
33     /**
34      * @author Alejandro Revilla
35      *
36      * Creates an ISOMeter component
37      * i.e:
38      * <pre>
39      *  &lt;iso-meter idref="id" scroll="true|false" refresh="nnn"/&gt
40      * </pre>
41      * @see org.jpos.ui.UIFactory
42      */
43     
44     public class ISOMeterFactory implements UIFactory {
45         public JComponent create (UI ui, Element e) {
46             ISOChannelPanel icp = null;
47             try {
48                 Object obj = (Object) 
49                     NameRegistrar.get (e.getAttributeValue ("idref"));
50     
51                 if (obj instanceof ISOChannel) {
52                     icp = new ISOChannelPanel ((ISOChannel) obj, e.getText ());
53                 } else if (obj instanceof Observable) {
54                     icp = new ISOChannelPanel (e.getText());
55                     ((Observable)obj).addObserver (icp);
56                 }
57                 ISOMeter meter = icp.getISOMeter ();
58                 if ("false".equals (e.getAttributeValue ("scroll")))
59                     meter.setScroll (false);
60     
61                 String protect = e.getAttributeValue ("protect");
62                 if (protect != null)
63                     icp.setProtectFields (ISOUtil.toIntArray (protect));
64                 String wipe = e.getAttributeValue ("wipe");
65                 if (wipe != null)
66                     icp.setWipeFields (ISOUtil.toIntArray (wipe));
67     
68                 String refresh = e.getAttributeValue ("refresh");
69                 if (refresh != null)
70                     meter.setRefresh (Integer.parseInt (refresh));
71             } catch (Exception ex) {
72                 ex.printStackTrace ();
73                 return new JLabel (ex.getMessage());
74             }
75             return icp;
76         }
77     }
