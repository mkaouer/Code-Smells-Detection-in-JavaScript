19     package org.jpos.iso;
20     
21     import java.io.IOException;
22     import java.io.InputStream;
23     import java.util.*;
24     
25     /**
26      * ISO Currency Conversion package
27      *
28      * @author vsalaman@gmail.com
29      * @author Jonathan.O'Connor@xcom.de
30      * @version $Id$
31      * @see "http://www.evertype.com/standards/iso4217/iso4217-en.html"
32      *      "http://www.iso.org/iso/en/prods-services/popstds/currencycodeslist.html"
33      */
34     public class ISOCurrency
35     {
36         private static Map<String, Currency> currencies;
37         private static final Object mutex = new Object();
38     
39         // Avoid creation of instances.
40         private ISOCurrency()
41         {
42         }
43     
44         static
45         {
46             synchronized (mutex)
47             {
48                 currencies = new HashMap<String, Currency>();
49             }
50             addBundle(ISOCurrency.class.getName());
51             loadPropertiesFromClasspath("META-INF/org/jpos/config/ISOCurrency.properties");
52         }
53     
54         @SuppressWarnings({"EmptyCatchBlock"})
55         public static void loadPropertiesFromClasspath(String base)
56         {
57             InputStream in=loadResourceAsStream(base);
58             try
59             {
60                 if(in!=null)
61                 {
62                     addBundle(new PropertyResourceBundle(in));
63                 }
64             }
65             catch (IOException e)
66             {
67             }
68             finally
69             {
70                 if(in!=null)
71                 {
72                     try
73                     {
74                         in.close();
75                     }
76                     catch (IOException e)
77                     {
78                     }
79                 }
80             }
81         }
82     
83         /**
84          * Converts from an ISO Amount (12 digit string) to a double taking in
85          * consideration the number of decimal digits according to currency
86          *
87          * @param isoamount - The ISO amount to be converted (eg. ISOField 4)
88          * @param currency  - The ISO currency to be converted (eg. ISOField 49)
89          * @return result - A double representing the converted field
90          * @throws IllegalArgumentException if we fail to convert the amount
91          */
92         public static double convertFromIsoMsg(String isoamount, String currency) throws IllegalArgumentException
93         {
94             Currency c = findCurrency(currency);
95             return c.parseAmountFromISOMsg(isoamount);
96         }
97     
98         public static void addBundle(ResourceBundle r)
99         {
100             synchronized (mutex)
101             {
102                 Enumeration en = r.getKeys();
103                 while (en.hasMoreElements())
104                 {
105                     String alphaCode = (String) en.nextElement();
106                     String[] tmp = r.getString(alphaCode).split(" ");
107                     String isoCode = tmp[0];
108                     int numDecimals = Integer.parseInt(tmp[1]);
109                     addCurrency(alphaCode, isoCode, numDecimals);
110                 }
111             }
112         }
113     
114         public static void addBundle(String bundleName)
115         {
116             ResourceBundle r = ResourceBundle.getBundle(bundleName);
117             addBundle(r);
118         }
119     
120         /**
121          * Converts an amount to an ISO Amount taking in consideration
122          * the number of decimal digits according to currency
123          *
124          * @param amount   - The amount to be converted
125          * @param currency - The ISO currency to be converted (eg. ISOField 49)
126          * @return result - An iso amount representing the converted field
127          * @throws IllegalArgumentException if we fail to convert the amount
128          */
129         public static String convertToIsoMsg(double amount, String currency) throws IllegalArgumentException
130         {
131             return findCurrency(currency).formatAmountForISOMsg(amount);
132         }
133     
134         public static Object[] decomposeComposedCurrency(String incurr) throws IllegalArgumentException
135         {
136             final String[] strings = incurr.split(" ");
137             if (strings.length != 2)
138             {
139                 throw new IllegalArgumentException("Invalid parameter: " + incurr);
140             }
141             return new Object[]{strings[0], Double.valueOf(strings[1])};
142         }
143     
144         public static String getIsoCodeFromAlphaCode(String alphacode) throws IllegalArgumentException
145         {
146             try
147             {
148                 Currency c = findCurrency(alphacode);
149                 return ISOUtil.zeropad(Integer.toString(c.getIsoCode()), 3);
150             }
151             catch (ISOException e)
152             {
153                 throw new IllegalArgumentException("Failed getIsoCodeFromAlphaCode/ zeropad failed?", e);
154             }
155         }
156     
157         public static Currency getCurrency(int code) throws ISOException
158         {
159             final String isoCode = ISOUtil.zeropad(Integer.toString(code), 3);
160             return findCurrency(isoCode);
161         }
162     
163         public static Currency getCurrency(String code) throws ISOException
164         {
165             final String isoCode = ISOUtil.zeropad(code, 3);
166             return findCurrency(isoCode);
167         }
168     
169         private static InputStream loadResourceAsStream(String name)
170         {
171             InputStream in = null;
172     
173             ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
174             if (contextClassLoader != null)
175             {
176                 in = contextClassLoader.getResourceAsStream(name);
177             }
178             if (in == null)
179             {
180                 in = ISOCurrency.class.getClassLoader().getResourceAsStream(name);
181             }
182             return in;
183         }
184     
185         /**
186          * Should be called like this: put("ALL", "008", 2);
187          * Note: the second parameter is zero padded to three digits
188          *
189          * @param alphaCode   An alphabetic code such as USD
190          * @param isoCode     An ISO code such as 840
191          * @param numDecimals the number of implied decimals
192          */
193         private static void addCurrency(String alphaCode, String isoCode, int numDecimals)
194         {
195             // to allow a clean replacement from a more specific resource bundle we
196             // require clearing instead of overriding.
197             if(currencies.containsKey(alphaCode) || currencies.containsKey(isoCode))
198             {
199                 currencies.remove(alphaCode);
200                 currencies.remove(isoCode);
201             }
202             Currency ccy = new Currency(alphaCode, Integer.parseInt(isoCode), numDecimals);
203             currencies.put(alphaCode, ccy);
204             currencies.put(isoCode, ccy);
205         }
206     
207         private static Currency findCurrency(String currency)
208         {
209             final Currency c = currencies.get(currency.toUpperCase());
210             if (c == null)
211             {
212                 throw new IllegalArgumentException("Currency with key '" + currency + "' was not found");
213             }
214             return c;
215         }
216     }
