19     package org.jpos.iso;
20     
21     import junit.framework.TestCase;
22     
23     /**
24      * @author joconnor
25      */
26     public class LeftPadderTest extends TestCase {
27         private LeftPadder padder;
28         
29         protected void setUp() {
30             padder = LeftPadder.ZERO_PADDER;
31         }
32         
33         public void testPaddingNeeded() throws Exception {
34             assertEquals("000123", padder.pad("123", 6));
35         }
36     
37         public void testNoPaddingNeeded() throws Exception {
38             assertEquals("123", padder.pad("123", 3));
39         }
40     
41         public void testPadLengthTooShort() throws Exception {
42             try {
43                 padder.pad("123", 2);
44                 fail("Padding a string longer than the available width should throw an exception");
45             } catch (Exception asIExpected) {
46             }
47         }
48     
49         public void testUnpad() throws Exception {
50             assertEquals("123", padder.unpad("000123"));
51         }
52     
53         public void testUnpadAllPadding() throws Exception {
54             assertEquals("", padder.unpad("000"));
55         }
56     
57         public void testReversability() throws Exception {
58             String origin = "Abc";
59             assertEquals(origin, padder.unpad(padder.pad(origin, 6)));
60         }
61     }
