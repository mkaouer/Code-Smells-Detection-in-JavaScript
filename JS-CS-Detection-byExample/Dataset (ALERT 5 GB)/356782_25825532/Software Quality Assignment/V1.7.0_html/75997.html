19     package org.jpos.q2.cli;
20     
21     import jline.ANSIBuffer;
22     import org.jpos.q2.CLI;
23     import org.jpos.util.LogEvent;
24     import org.jpos.util.LogListener;
25     import org.jpos.util.Logger;
26     import org.jpos.util.NameRegistrar;
27     
28     import java.io.PrintStream;
29     import java.util.Date;
30     import java.util.Iterator;
31     import java.util.Map;
32     
33     public class TAIL implements CLI.Command, LogListener {
34         PrintStream p;
35         CLI cli;
36         boolean ansi;
37             
38         public void exec (CLI cli, String[] args) throws Exception {
39             this.p = cli.getOutputStream();
40             this.cli = cli;
41             this.ansi = cli.getConsoleReader().getTerminal().isANSISupported();
42             if (args.length == 1) {
43                 usage(cli);
44                 return;
45             }
46             for (int i=1; i<args.length; i++) {
47                 try {
48                     Logger logger = (Logger) NameRegistrar.get ("logger." + args[i]);
49                     logger.addListener (this);
50                 } catch (NameRegistrar.NotFoundException e) {
51                     cli.println ("Logger " + args[i] + " not found -- ignored.");
52                 }
53             }
54             cli.getConsoleReader().readCharacter(new char[] { 'q', 'Q' });
55             for (int i=1; i<args.length; i++) {
56                 try {
57                     Logger logger = (Logger) NameRegistrar.get ("logger." + args[i]);
58                     logger.removeListener (this);
59                 } catch (NameRegistrar.NotFoundException e) { }
60             }
61         }
62         public void usage (CLI cli) {
63             cli.println ("Usage: tail [log-name] [log-name] ...");
64             showLoggers (cli);
65         }
66         public synchronized LogEvent log (LogEvent ev) {
67             if (p != null) {
68                 Date d = new Date (System.currentTimeMillis());
69                 ANSIBuffer ab = new ANSIBuffer();
70                 ab.setAnsiEnabled (ansi);
71                 cli.println (
72                     ab.bold (
73                         ev.getSource().getLogger().getName() + 
74                         ": " + ev.getRealm() + " " + d.toString() +"." + d.getTime() % 1000 
75                     ).toString (ansi)
76                 );
77                 ev.dump (p, " ");
78                 p.flush();
79             }
80             return ev;
81         }
82         private void showLoggers (CLI cli) {
83             NameRegistrar nr = NameRegistrar.getInstance();
84             int maxw = 0;
85             Iterator iter = nr.getMap().entrySet().iterator();
86             StringBuffer sb = new StringBuffer ("available loggers:");
87             while (iter.hasNext()) {
88                 Map.Entry entry = (Map.Entry) iter.next ();
89                 String key = (String) entry.getKey();
90                 if (key.startsWith ("logger.") && entry.getValue() instanceof Logger) {
91                     sb.append (' ');
92                     sb.append (key.substring(7));
93                 }
94             }
95             cli.println (sb.toString());
96         }
97     }
