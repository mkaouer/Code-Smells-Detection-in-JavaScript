19     package org.jpos.iso;
20     
21     /**
22      * ISOFieldPackager Binary LLNUM
23      *
24      * @author apr@cs.com.uy
25      * @version $Id$
26      * @see ISOComponent
27      */
28     public class IFB_LLNUM extends ISOStringFieldPackager {
29         public IFB_LLNUM() {
30             super(NullPadder.INSTANCE, BCDInterpreter.RIGHT_PADDED, BcdPrefixer.LL);
31         }
32         /**
33          * @param len - field len
34          * @param description symbolic descrption
35          */
36         public IFB_LLNUM(int len, String description, boolean isLeftPadded) {
37             super(len, description, NullPadder.INSTANCE,
38                     isLeftPadded ? BCDInterpreter.LEFT_PADDED : BCDInterpreter.RIGHT_PADDED,
39                     BcdPrefixer.LL);
40             checkLength(len, 99);
41         }
42         
43         public IFB_LLNUM(int len, String description, boolean isLeftPadded, boolean fPadded) {
44             super(len, description, NullPadder.INSTANCE,
45                     isLeftPadded ? BCDInterpreter.LEFT_PADDED : 
46                         (fPadded ? BCDInterpreter.RIGHT_PADDED_F : BCDInterpreter.RIGHT_PADDED),
47                     BcdPrefixer.LL);
48             checkLength(len, 99);
49         }
50     
51         public void setLength(int len)
52         {
53             checkLength(len, 99);
54             super.setLength(len);
55         }
56     
57         /** Must override ISOFieldPackager method to set the Interpreter correctly */
58         public void setPad (boolean pad)
59         {
60             setInterpreter(pad ? BCDInterpreter.LEFT_PADDED : BCDInterpreter.RIGHT_PADDED);
61             this.pad = pad;
62         }
63     }
