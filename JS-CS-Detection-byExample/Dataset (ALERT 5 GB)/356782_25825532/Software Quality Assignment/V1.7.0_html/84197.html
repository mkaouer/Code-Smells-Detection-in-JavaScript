22     import org.jpos.core.ConfigurationException;
23     import org.jpos.core.ReConfigurable;
24     import org.jpos.iso.*;

30     /**
31      * Space based MUX implementation
32      * @author Alejandro Revilla
33      * @version $Revision$ $Date$
34      * @see org.jpos.iso.MUX
35      */
36     public class SpaceMUX extends SimpleLogSource
37            implements MUX, ReConfigurable, SpaceListener
38     {
39         LocalSpace sp;
40         Configuration cfg;
41         String to, from, unhandled, name;
42         public SpaceMUX () {
43             super ();
44             sp = TransientSpace.getSpace ();
45         }
46         /**
47          * @param to    output queue
48          * @param from  input  queue
49          * @param unhandled optional unhandled queue (may be null)
50          */
51         public SpaceMUX (String to, String from, String unhandled) {
52             this ();
53             this.to        = to;
54             this.from      = from;
55             this.unhandled = unhandled;
56             sp.addListener (from, this);
57         }
58     
59         /**
60          * reads "to", "from" and optional "unhandled" properties
61          * @param cfg 
62          */
63         public void setConfiguration (Configuration cfg) 
64             throws ConfigurationException
65         {
66             this.cfg  = cfg;
67             to        = get ("to");
68             from      = get ("from");
69             unhandled = cfg.get ("unhandled");  // can be null
70     
71             sp.addListener (from, this);
72         }
73         /**
74          * @param m message to send
75          * @param timeout amount of time in millis to wait for a response
76          * @return response or null
77          */
78         public ISOMsg request (ISOMsg m, long timeout) throws ISOException {
79             String key = getKey (m);
80             String req = key + ".req";
81             sp.out (req, m);
82             sp.out (to, m);
83     
84             ISOMsg resp = (ISOMsg) sp.in (key, timeout);
85     
86             if (resp == null && sp.inp (req) == null) {
87                 // possible race condition, retry for a few extra seconds
88                 resp = (ISOMsg) sp.in (key, 10000);
89             }
90             return resp;
91         }
92         protected String getFrom () {
93             return from;
94         }
95         protected String getKey (ISOMsg m) throws ISOException {
96             return getFrom () + "." +
97                (m.hasField(41)?ISOUtil.zeropad((String)m.getValue(41),16) : "")
98                + (m.hasField (11) ?
99                     ISOUtil.zeropad((String) m.getValue(11),6) :
100                     Long.toString (System.currentTimeMillis()));
101         }
102         public void notify (Object k, Object value) {
103             Object obj = sp.inp (k);
104             if (obj instanceof ISOMsg) {
105                 ISOMsg m = (ISOMsg) obj;
106                 try {
107                     String key = getKey (m);
108                     String req = key + ".req";
109                     if (sp.inp (req) != null) {
110                         sp.out (key, m);
111                         return;
112                     }
113                 } catch (ISOException e) { 
114                     Logger.log (new LogEvent (this, "notify", e));
115                 }
116                 if (unhandled != null)
117                     sp.out (unhandled, m, 120000);
118             }
119         }
120         private String get (String prop) throws ConfigurationException
121         {
122             String value = cfg.get (prop);
123             if (value == null)
124                 throw new ConfigurationException ("null property "+prop);
125             return value;
126         }
127         /**
128          * associates this MUX with a name using NameRegistrar
129          * @param name name to register
130          * @see NameRegistrar
131          */
132         public void setName (String name) {
133             this.name = name;
134             NameRegistrar.register ("mux."+name, this);
135         }
136         /**
137          * @return this ISOMUX's name ("" if no name was set)
138          */
139         public String getName() {
140             return this.name;
141         }
142         /**
143          * @return MUX instance with given name.
144          * @throws NameRegistrar.NotFoundException;
145          * @see NameRegistrar
146          */
147         public static MUX getMUX (String name)
148             throws NameRegistrar.NotFoundException
149         {
150             return (MUX) NameRegistrar.get ("mux."+name);
151         }
152         public boolean isConnected () {
153             return true;
154         }
155         public void request (ISOMsg m, long timeout, ISOResponseListener r, Object handBack) 
156             throws ISOException 
157         {
158             throw new ISOException ("Not implemented");
159         }
160     }
