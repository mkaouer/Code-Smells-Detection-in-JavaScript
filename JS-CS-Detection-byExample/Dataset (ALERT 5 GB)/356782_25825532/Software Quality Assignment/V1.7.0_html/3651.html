21     /**
22      * EBCDIC version of IFB_LLLNUM
23      * Uses a 2 EBCDIC byte length field
24      *
25      * @author julien.moebs@paybox.net
26      * @version $Id$
27      * @see ISOFieldPackager
28      * @see ISOComponent
29      */
30     
31     
32     public class IFEB_LLLNUM extends ISOFieldPackager {
33         public IFEB_LLLNUM () {
34             super();
35         }
36         /**
37          * @param len - field len
38          * @param description symbolic descrption
39          */
40         public IFEB_LLLNUM (int len, String description) {
41             super(len, description);
42         }
43         /**
44          * @param c - a component
45          * @return packed component
46          * @exception ISOException
47          */
48         public byte[] pack(ISOComponent c) throws ISOException {
49             boolean odd = false;
50             int len;
51             String s = (String) c.getValue();
52             
53             
54             
55             if ((len=s.length()) > getLength() || len>99)   // paranoia settings
56                 throw new ISOException(
57                 "invalid len "+len +" packing IFEB_LLLNUM field " + c.getKey());
58             
59             
60             //System.out.println("String s = (String) c.getValue(); "+s);
61             
62             
63             
64             // if odd length
65             if ( (len%2)==1 ) {
66                 odd = true;
67                 len = (len/2)+1;
68             } else {
69                 odd = false;
70                 len = len/2;
71             }
72             
73             //System.out.println("len= "+ len +" s.length()= "+len);
74             
75             String fieldLength = ISOUtil.zeropad(Integer.toString(len), 3);
76             
77             byte [] EBCDIClength = ISOUtil.asciiToEbcdic(fieldLength);
78     
79             // bcd stuff
80             byte[] bcd = ISOUtil.str2bcd(s, false);
81             if(odd)
82                 bcd[len-1] = (byte) (bcd[len-1] | 0xf);
83             
84     
85             //System.out.println("bcd2str "+ISOUtil.bcd2str(bcd, 0, bcd.length*2, false) );
86             
87             byte[] b   = new byte[bcd.length + 3];
88             
89             b[0] = EBCDIClength[0];
90             b[1] = EBCDIClength[1];
91             b[2] = EBCDIClength[2];
92             System.arraycopy(bcd, 0, b, 3, bcd.length);
93             
94             return b;
95         }
96         /**
97          * @param c - the Component to unpack
98          * @param b - binary image
99          * @param offset - starting offset within the binary image
100          * @return consumed bytes
101          * @exception ISOException
102          */
103         public int unpack(ISOComponent c, byte[] b, int offset)
104         throws ISOException {
105             boolean pad = false;
106             
107             
108             int len = ((b[offset] & 0x0f) * 100)  + ((b[offset+1] & 0x0f) * 10) + (b[offset+2] & 0x0f);
109             
110             int tempLen = len*2;
111     
112             //System.out.println("len "+ len +"len*2 "+tempLen);
113             
114             
115             // odd handling
116             byte testByte = b[ (offset+3+len-1) ];
117             
118             if( (testByte | 0xf0)== 0xff) {
119                 // odd length
120                 tempLen--;
121             }
122             
123             
124             // bcd line
125             //System.out.println("ISOUtil.bcd2str(b, offset+2, len, pad) "+ISOUtil.bcd2str(b, offset+2, tempLen, pad));
126             
127             c.setValue(ISOUtil.bcd2str(b, offset+3, tempLen, pad));
128             
129             //c.setValue(ISOUtil.ebcdicToAscii(b, offset+2, len));
130             
131             return len+3;
132         }
133         
134         public int getMaxPackedLength() {
135             return getLength()+3;
136         }
137     }
