19     package org.jpos.iso;
20     
21     
22     /**
23      * An interface for padding and unpadding strings and byte arrays.
24      * 
25      * @author joconnor
26      * @version $Revision$ $Date$
27      */
28     public interface Padder
29     {
30         /**
31     	 * Returns a padded string upto a maximum length. If the data is longer
32     	 * than maxLength, then the data is truncated.
33     	 * 
34     	 * @param data
35     	 *            The string to pad.
36     	 * @param maxLength
37     	 *            The maximum length of the padded string.
38     	 * @return A padded string.
39     	 */
40         String pad(String data, int maxLength) throws ISOException;
41     
42         /**
43     	 * Removes the padding from a padded string.
44     	 * 
45     	 * @param paddedData
46     	 *            The string to unpad.
47     	 * @return The unpadded string.
48     	 */
49         String unpad(String paddedData) throws ISOException;
50     }
