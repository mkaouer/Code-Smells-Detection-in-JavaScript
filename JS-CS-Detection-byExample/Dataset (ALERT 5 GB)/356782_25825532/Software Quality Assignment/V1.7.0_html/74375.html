19     package org.jpos.iso;
20     
21     import org.jpos.iso.ISOFilter.VetoException;
22     import org.jpos.util.LogEvent;
23     
24     import java.util.Collection;
25     import java.util.Iterator;
26     import java.util.Observable;
27     import java.util.Vector;
28     
29     /**
30      * Filtered Channel Base
31      * 
32      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
33      * @version $Revision$ $Date$
34      * @see FilteredChannel
35      */
36     
37     public abstract class FilteredBase extends Observable 
38         implements FilteredChannel, Cloneable
39     {
40         protected Vector incomingFilters, outgoingFilters;
41     
42         public FilteredBase () {
43             super();
44             incomingFilters = new Vector();
45             outgoingFilters = new Vector();
46         }
47     
48         /**
49          * @param filter filter to add
50          * @param direction ISOMsg.INCOMING, ISOMsg.OUTGOING, 0 for both
51          */
52         private void addFilter (ISOFilter filter, int direction) {
53             switch (direction) {
54                 case ISOMsg.INCOMING :
55                     incomingFilters.add (filter);
56                     break;
57                 case ISOMsg.OUTGOING :
58                     outgoingFilters.add (filter);
59                     break;
60                 case 0 :
61                     incomingFilters.add (filter);
62                     outgoingFilters.add (filter);
63                     break;
64             }
65         }
66         /**
67          * @param filter incoming filter to add
68          */
69         public void addIncomingFilter (ISOFilter filter) {
70             addFilter (filter, ISOMsg.INCOMING);
71         }
72         /**
73          * @param filter outgoing filter to add
74          */
75         public void addOutgoingFilter (ISOFilter filter) {
76             addFilter (filter, ISOMsg.OUTGOING);
77         }
78     
79         /**
80          * @param filter filter to add (both directions, incoming/outgoing)
81          */
82         public void addFilter (ISOFilter filter) {
83             addFilter (filter, 0);
84         }
85         /**
86          * @param filter filter to remove
87          * @param direction ISOMsg.INCOMING, ISOMsg.OUTGOING, 0 for both
88          */
89         private void removeFilter (ISOFilter filter, int direction) {
90             switch (direction) {
91                 case ISOMsg.INCOMING :
92                     incomingFilters.remove (filter);
93                     break;
94                 case ISOMsg.OUTGOING :
95                     outgoingFilters.remove (filter);
96                     break;
97                 case 0 :
98                     incomingFilters.remove (filter);
99                     outgoingFilters.remove (filter);
100                     break;
101             }
102         }
103         /**
104          * @param filter filter to remove (both directions)
105          */
106         public void removeFilter (ISOFilter filter) {
107             removeFilter (filter, 0);
108         }
109         /**
110          * @param filter incoming filter to remove
111          */
112         public void removeIncomingFilter (ISOFilter filter) {
113             removeFilter (filter, ISOMsg.INCOMING);
114         }
115         /**
116          * @param filter outgoing filter to remove
117          */
118         public void removeOutgoingFilter (ISOFilter filter) {
119             removeFilter (filter, ISOMsg.OUTGOING);
120         }
121         protected ISOMsg applyOutgoingFilters (ISOMsg m, LogEvent evt) 
122             throws VetoException
123         {
124             Iterator iter  = outgoingFilters.iterator();
125             while (iter.hasNext()) {
126                 m.setDirection(ISOMsg.OUTGOING);
127                 m = ((ISOFilter) iter.next()).filter (this, m, evt);
128             }
129             m.setDirection(ISOMsg.OUTGOING);
130             setChanged ();
131             notifyObservers (m);
132             return m;
133         }
134         protected ISOMsg applyIncomingFilters (ISOMsg m, LogEvent evt) 
135             throws VetoException
136         {
137             Iterator iter  = incomingFilters.iterator();
138             while (iter.hasNext()) {
139                 m.setDirection(ISOMsg.INCOMING);
140                 m = ((ISOFilter) iter.next()).filter (this, m, evt);
141             }
142             m.setDirection(ISOMsg.INCOMING);
143             setChanged ();
144             notifyObservers (m);
145             return m;
146         }
147         public Collection getIncomingFilters() {
148             return incomingFilters;
149         }
150         public Collection getOutgoingFilters() {
151             return outgoingFilters;
152         }
153         public void setIncomingFilters (Collection filters) {
154             incomingFilters = new Vector (filters);
155         }
156         public void setOutgoingFilters (Collection filters) {
157             outgoingFilters = new Vector (filters);
158         }
159         
160         public Object clone(){
161           try {
162             return (FilteredBase)super.clone();
163           } catch (CloneNotSupportedException e) {
164             throw new InternalError();
165           }
166         }
167     }
