19     package org.jpos.q2.cli;
20     
21     import org.jpos.q2.CLI;
22     
23     import java.io.InputStream;
24     
25     public class MAN implements CLI.Command {
26         public void exec (CLI cli, String[] args) throws Exception {
27             if (args.length < 2) {
28                 cli.println ("What manual page do you want?");
29                 return;
30             }
31             String command = args[1];
32             InputStream is = MAN.class.getResourceAsStream(
33                 command.toUpperCase()+".man"
34             );
35             if (is != null) {
36                 byte[] b = new byte[is.available()];
37                 is.read (b);
38                 cli.print (new String(b, "ISO8859_1"));
39             } else {
40                 cli.println ("No manual entry for " + command);
41             }
42         }
43     }
