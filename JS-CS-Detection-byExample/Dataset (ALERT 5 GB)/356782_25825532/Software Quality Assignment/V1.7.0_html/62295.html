19     package org.jpos.iso;
20     
21     /**
22      * Implements EBCDIC Interpreter for signed numerics.
23      * (see http://publib.boulder.ibm.com/infocenter/wmbhelp/v6r0m0/index.jsp?topic=/com.ibm.etools.mft.doc/ad06900_.htm)
24      * Strings are converted to and from EBCDIC bytes.
25      * Negatives will be prepended with "-"
26      * Unsigned numbers are interpreted as positive
27      * 
28      * @author nsmith@mxgroup.net
29      * @version $Revision$ $Date$
30      */
31     public class SignedEbcdicNumberInterpreter implements Interpreter
32     {
33         /** An instance of this Interpreter. Only one needed for the whole system */
34         public static final SignedEbcdicNumberInterpreter INSTANCE = new SignedEbcdicNumberInterpreter();
35     
36         public void interpret(String data, byte[] targetArray, int offset) {
37             boolean negative = data.startsWith("-");
38             if (negative) {
39                 ISOUtil.asciiToEbcdic(data.substring(1), targetArray, offset);
40                 targetArray[offset + data.length() - 2] = (byte) (targetArray[offset + data.length() - 2] & 0xDF); 
41             } else {
42                 ISOUtil.asciiToEbcdic(data, targetArray, offset);
43             }
44         }
45     
46         public String uninterpret(byte[] rawData, int offset, int length) {
47             boolean negative = ((byte)(rawData[offset + length - 1] & 0xF0)) == ((byte)0xD0);
48             rawData[offset + length - 1] = ((byte)(rawData[offset + length - 1] | 0xF0));
49             return (negative ? "-" : "") + ISOUtil.ebcdicToAscii(rawData, offset, length);
50         }
51     
52         public int getPackedLength(int nDataUnits)
53         {
54             return nDataUnits;
55         }
56     }
