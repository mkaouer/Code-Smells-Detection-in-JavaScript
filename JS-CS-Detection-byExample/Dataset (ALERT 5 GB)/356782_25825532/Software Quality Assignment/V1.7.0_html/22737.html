19     package org.jpos.iso;
20     
21     
22     /**
23      * Implements the Padder interface for padding strings and byte arrays on the
24      * Right.
25      * 
26      * @author joconnor
27      * @version $Revision$ $Date$
28      */
29     public class RightPadder implements Padder
30     {
31         /**
32     	 * A padder for padding spaces on the right. This is very common in
33     	 * alphabetic fields.
34     	 */
35         public static final RightPadder SPACE_PADDER = new RightPadder(' ');
36     
37         private char pad;
38     
39         /**
40     	 * Creates a Right Padder with a specific pad character.
41     	 * 
42     	 * @param pad
43     	 *            The padding character. For binary padders, the pad character
44     	 *            is truncated to lower order byte.
45     	 */
46         public RightPadder(char pad)
47         {
48             this.pad = pad;
49         }
50     
51         /**
52     	 * @see org.jpos.iso.Padder#pad(java.lang.String, int, char)
53     	 */
54         public String pad(String data, int maxLength) throws ISOException
55         {
56             StringBuffer padded = new StringBuffer(maxLength);
57             int len = data.length();
58             if (len > maxLength)
59             {
60                 throw new ISOException("Data is too long. Max = " + maxLength);
61             } else
62             {
63                 padded.append(data);
64                 for (int i = maxLength - len; i > 0; i--)
65                 {
66                     padded.append(pad);
67                 }
68             }
69             return padded.toString();
70         }
71     
72         /**
73     	 * @see org.jpos.iso.Padder#unpad(java.lang.String, char)
74     	 */
75         public String unpad(String paddedData)
76         {
77             int len = paddedData.length();
78             for (int i = len; i > 0; i--)
79             {
80                 if (paddedData.charAt(i - 1) != pad)
81                 {
82                     return paddedData.substring(0, i);
83                 }
84             }
85             return "";
86         }
87     }
