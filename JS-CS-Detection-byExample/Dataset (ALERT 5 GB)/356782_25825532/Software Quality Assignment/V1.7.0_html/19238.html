137         /**
138          * construct key to match request with responses
139          * @param   m   request/response
140          * @return      key (default terminal(41) + tracenumber(11))
141          */
142         protected String getKey(ISOMsg m) throws ISOException {
143             return (m.hasField(41)?ISOUtil.zeropad((String)m.getValue(41),16) : "")
144                + (m.hasField (traceNumberField) ?
145                     ISOUtil.zeropad((String) m.getValue(traceNumberField),6) :
146                     Long.toString (System.currentTimeMillis()));
147         }

149         /**
150          * get rid of expired requests
151          */
152         private void purgeRxQueue() {
153             Enumeration e = rxQueue.keys();
154             while (e.hasMoreElements()) {
155                 Object key = e.nextElement();
156                 ISORequest r = (ISORequest) rxQueue.get(key);
157                 if (r != null && r.isExpired()) {
158                     rxQueue.remove(key);
159                     cnt[RX_EXPIRED]++;
160                 }
161             }
162         }

247         private class Receiver implements Runnable, LogSource {
248             Runnable parent;
249             protected Receiver(Runnable p) {
250                 parent = p;
251             }
252             public void run() {
253                 int i = 0;
254                 while (!terminate || !rxQueue.isEmpty() || !txQueue.isEmpty()) {
255                     if (i++ % 250 == 1) 
256                         Logger.log (new LogEvent (this, "mux", parent));
257                     if (channel.isConnected()) {
258                         try {
259                             ISOMsg d = channel.receive();
260                             cnt[RX]++;
261                             String k = getKey(d);
262                             ISORequest r = (ISORequest) rxQueue.get(k);
263                             boolean forward = true;
264                             if (r != null) {
265                                 rxQueue.remove(k);
266                                 synchronized (r) {
267                                     if (r.isExpired()) {
268                                         if ((++cnt[RX_EXPIRED]) % 10 == 0)
269                                             purgeRxQueue();
270                                     }
271                                     else {
272                                         r.setResponse(d);
273                                         forward = false;
274                                     }
275                                 }
276                             }
277                             if (forward) {
278                                 if (requestListener != null) {
279                                     requestListener.process(muxInstance, d);
280                                     cnt[RX_FORWARDED]++;
281                                 }
282                                 else 
283                                     cnt[RX_UNKNOWN]++;
284                             }
285                         } catch (Throwable e) {
286                             if (!terminate) {
287                                 channel.setUsable(false);
288                                 if (!(e instanceof EOFException))
289                                     Logger.log (new LogEvent (this, "muxreceiver", e));
290                                 synchronized(parent) {
291                                     parent.notify();
292                                 }
293                             }
294                         }
295                     }
296                     else {
297                         try {
298                             synchronized(rx) {
299                                 rx.wait();
300                             }
301                         } catch (InterruptedException e) { 
302                             Logger.log (new LogEvent (this, "muxreceiver", e));
303                         }
304                     }
305                 }
306                 Logger.log (new LogEvent (this, "muxreceiver", "terminate"));
307             }

309             public String getRealm () {
310                 return realm;
311             }
312             public Logger getLogger() {
313                 return logger;
314             }
315         }

317         private void doTransmit() throws ISOException, IOException {
318             while (txQueue.size() > 0) {
319                 Object o = txQueue.firstElement();
320                 ISOMsg m = null;
321     
322                 if (o instanceof ISORequest) {
323                     ISORequest r = (ISORequest) o;
324                     if (r.isExpired()) 
325                         cnt[TX_EXPIRED]++;
326                     else {
327                         m = r.getRequest();
328                         rxQueue.put (getKey(m), r);
329                         r.setTransmitted ();
330                         synchronized(rx) {
331                             rx.notify(); // required by ChannelPool
332                         }
333                     }
334                 } else if (o instanceof ISOMsg) {
335                     m = (ISOMsg) o;
336                 }
337                 if (m != null) {
338                     try {
339                         channel.send(m);
340                         cnt[TX]++;
341                     } catch (ISOException e) {
342                         Logger.log (new LogEvent (this, "error", e));
343                     }
344                 }
345                 txQueue.removeElement(o);
346                 txQueue.trimToSize();
347             }
348         }
349         public void run () {
350             tx = Thread.currentThread();
351     
352             int rxPriority = rx.getPriority();      // Bug#995787
353             if (rxPriority < Thread.MAX_PRIORITY) {
354                                                     // OS/400 V4R4 JVM 
355                 rx.setPriority (rxPriority+1);      // Thread problem
356                                                     // (Vincent.Greene@amo.com)
357             }
358             rx.start();
359             boolean firstTime = true;
360             while (!terminate || !txQueue.isEmpty()) {
361                 try {
362                     if (channel.isConnected()) {
363                         doTransmit();
364                     }
365                     else if (doConnect) {
366                         if (firstTime) {
367                             firstTime = !firstTime;
368                             channel.connect();
369                         }
370                         else {
371                             Thread.sleep(5000);
372                             channel.reconnect();
373                         }
374                         cnt[CONNECT]++;
375                         synchronized(rx) {
376                             rx.notify();
377                         }
378                     } else {
379                         // nothing to do ...
380                         try {
381                             Thread.sleep (5000);
382                         } catch (InterruptedException ex) { }
383                     }
384                     synchronized(this) {
385                         if (!terminate && 
386                              channel.isConnected() && 
387                              txQueue.size() == 0)
388                         {
389                             this.wait();
390                         }
391                     }
392                 } catch (ConnectException e) {
393                     if (channel instanceof ClientChannel) {
394                         ClientChannel cc = (ClientChannel) channel;
395                         Logger.log (new LogEvent (this, "connection-refused", 
396                             cc.getHost()+":"+cc.getPort())
397                         );
398                     }
399                     try {
400                         Thread.sleep (1000);
401                     } catch (InterruptedException ex) { }
402                 } catch (Exception e) {
403                     Logger.log (new LogEvent (this, "mux", e));
404                     try {
405                         Thread.sleep (1000);
406                     } catch (InterruptedException ex) { }
407                 }
408             }
409             // Wait for the receive queue to empty out before shutting down
410             while (!rxQueue.isEmpty()) {
411                 try {
412                     Thread.sleep(5000); // Wait for the receive queue to clear.
413                     purgeRxQueue();     // get rid of expired stuff
414                 } catch (InterruptedException e) {
415                     break;
416                 }
417             }
418             // By closing the channel, we force the receive thread to terminate
419             try {
420                 channel.disconnect();
421             } catch (IOException e) { }
422     
423             synchronized(rx) {
424                 rx.notify();
425             }
426             try {
427                 rx.join ();
428             } catch (InterruptedException e) { }
429             Logger.log (new LogEvent (this, "mux", "terminate"));
430         }
431         /**
432          * queue an ISORequest
433          */
434         synchronized public void queue(ISORequest r) {
435             txQueue.addElement(r);
436             this.notify();
437         }
438         /**
439          * send a message over channel, usually a
440          * response from an ISORequestListener
441          */
442         synchronized public void send(ISOMsg m) {
443             txQueue.addElement(m);
444             this.notify();
445         }

447         private void terminate(boolean hard) {
448             LogEvent evt = new LogEvent (this, "mux", 
449                 "<terminate type=\"" + (hard ? "hard" : "soft") +"\"/>");
450             evt.addMessage (this);
451             Logger.log (evt);
452     
453             terminate = true;
454             synchronized(this) {
455                 if (hard) {
456                     txQueue.removeAllElements();
457                     rxQueue.clear();
458                 }
459                 this.notify();
460             }
461         }

463         /**
464          * terminate MUX
465          * @param wait Time to wait before forcing shutdown
466          */
467         public void terminate (int wait) {
468             terminate(false);
469             tx.interrupt();
470             rx.interrupt();
471             try {
472                 tx.join(wait);
473                 if (tx.isAlive()) {
474                     terminate(true);
475                     tx.join();
476                 }
477             } catch (InterruptedException e) { }
478         }

533         public void setConnect (boolean connect) {
534             this.doConnect = connect;
535             if (!connect && isConnected()) {
536                 channel.setUsable(false);
537                 try {
538                     channel.disconnect();
539                 } catch (IOException e) { 
540                     Logger.log (new LogEvent(this, "set-connect", e));
541                 }
542                 synchronized(this) {
543                     this.notify();
544                 }
545             }
