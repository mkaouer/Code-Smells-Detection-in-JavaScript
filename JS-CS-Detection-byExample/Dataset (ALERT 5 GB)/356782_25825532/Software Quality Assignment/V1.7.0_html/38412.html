19     package org.jpos.iso.channel;
20     
21     import org.jpos.iso.*;
22     
23     import java.io.BufferedReader;
24     import java.io.EOFException;
25     import java.io.IOException;
26     import java.io.InputStreamReader;
27     import java.net.ServerSocket;
28     import java.net.Socket;
29     import java.util.Vector;
30     
31     /**
32      * Implements an ISOChannel suitable to be used to connect to an X.25 PAD. 
33      * It waits a limited amount of time to decide when a packet is ready
34      * to be unpacked.
35      *
36      * This channel is based on PADChannel version 1.4. The new version
37      * seems to have some problems dealing with ETXs (we're working on it).
38      * Use this version _only_ if you have problems with current PADChannel
39      * as it will be deprecated some time in the future.
40      *
41      * @author  <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
42      * @version $Id$
43      *
44      * @see ISOMsg
45      * @see ISOException
46      * @see ISOChannel
47      */
48     public class X25Channel extends BaseChannel {
49         BufferedReader reader = null;
50         protected byte[] header;
51         /**
52          * No-args constructor
53          */
54         public X25Channel () {
55             super();
56         }
57         /**
58          * Constructs client ISOChannel
59          * @param host  server TCP Address
60          * @param port  server port number
61          * @param p     an ISOPackager
62          * @see ISOPackager
63          */
64         public X25Channel (String host, int port, ISOPackager p) {
65             super(host, port, p);
66         }
67         /**
68          * Construct server ISOChannel
69          * @param p     an ISOPackager
70          * @see ISOPackager
71          * @exception IOException
72          */
73         public X25Channel (ISOPackager p) throws IOException {
74             super(p);
75         }
76         /**
77          * constructs a server ISOChannel associated with a Server Socket
78          * @param p     an ISOPackager
79          * @param serverSocket where to accept a connection
80          * @exception IOException
81          * @see ISOPackager
82          */
83         public X25Channel (ISOPackager p, ServerSocket serverSocket) 
84             throws IOException
85         {
86             super(p, serverSocket);
87         }
88         /**
89          * @return a byte array with the received message
90          * @exception IOException
91          */
92         protected byte[] streamReceive() throws IOException {
93             int c, k=0, len = 1;
94             Vector v = new Vector();
95     
96             c = serverIn.read();
97             if (c == -1)
98                 throw new EOFException ("connection closed");
99             byte[] b = new byte[1];
100             b[0] = (byte) c;
101             v.addElement (b);
102     
103             // Wait for packets until timeout
104             while ((c = serverIn.available()) > 0) {
105                 b = new byte[c];
106                 if (serverIn.read (b) != c)
107                     throw new EOFException ("connection closed");
108                 v.addElement (b);
109                 len += c;
110                 try {
111                     Thread.sleep (50);
112                 } catch (InterruptedException e) { }
113             }
114     
115             byte[] d = new byte[len];
116             for (int i=0; i<v.size(); i++) {
117                 b = (byte[]) v.elementAt(i);
118                 System.arraycopy (b, 0, d, k, b.length);
119                 k += b.length;
120             }
121             return d;
122         }
123         protected void connect (Socket socket) throws IOException {
124             super.connect (socket);
125             reader = new BufferedReader (new InputStreamReader (serverIn));
126         }
127         public void disconnect () throws IOException {
128             super.disconnect ();
129             reader = null;
130         }
131         protected int getHeaderLength() { 
132             return header != null ? header.length : 0;
133         }
134         public void setHeader (byte[] header) {
135             this.header = header;
136         }
137         /**
138          * @param header Hex representation of header
139          */
140         public void setHeader (String header) {
141             setHeader (
142                 ISOUtil.hex2byte (header.getBytes(), 0, header.getBytes().length / 2)
143             );
144         }
145         public byte[] getHeader () {
146             return header;
147         }
148         protected void sendMessageHeader(ISOMsg m, int len) throws IOException { 
149             if (m.getHeader() != null)
150                 serverOut.write(m.getHeader());
151             else if (header != null) 
152                 serverOut.write(header);
153         }
154     }
