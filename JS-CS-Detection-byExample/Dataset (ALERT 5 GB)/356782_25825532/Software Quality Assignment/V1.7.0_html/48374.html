30         
31         IFB_BITMAP oneByte ;
32         IFB_BITMAP eightBytes ;

34         IFB_BITMAP twentyfourBytes;
35         byte[] inBytes, eightByteBitMapIn24Bytes, sixteenByteBitMapIn24Bytes, testbytes, outBytes;

37         public void setUp() {
38             oneByte = new IFB_BITMAP(1,"1 byte bitmap");
39             eightBytes = new IFB_BITMAP(8,"8 byte bitmap");
40             sixteenBytes = new IFB_BITMAP(16,"16 byte bitmap");
41             twentyfourBytes = new IFB_BITMAP(24,"24 byte bitmap");
42             
43             //                                             Next Bitmap?    Next Bitmap?
44             //                                             V               V               
45             inBytes =                    ISOUtil.hex2byte("8181421FF12418F18F81421FF12418F18F81421FF12418F1");
46             eightByteBitMapIn24Bytes =   ISOUtil.hex2byte("7181421FF12418F11881421FF12418F18F81421FF12418F1");
47             sixteenByteBitMapIn24Bytes = ISOUtil.hex2byte("8181421FF12418F11881421FF12418F18F81421FF12418F1");
48     
49             
50         }
51         
52         public void test01ByteBitmap() throws Exception
53         {
54     
55             ISOComponent c = new ISOBitMap(1);
56             int consumed = oneByte.unpack(c, inBytes,0);
57             assertEquals("1 byte should be consumed irrespective of 2nd, 3rd or any bitmap indicators",1,consumed);
58             assertEquals("1 byte can only result in a bitmap holding fields up to 8",8,((BitSet)c.getValue()).length() - 1);
59             
60             outBytes = oneByte.pack(c);
61             assertEquals("1 byte bitmap pack should reflect unpack",ISOUtil.hexString(inBytes,0,1),ISOUtil.hexString(outBytes));
62         }
63         
64         public void test08ByteBitmap() throws Exception
65         {
66     
67             ISOComponent c = new ISOBitMap(1);
68             int consumed = eightBytes.unpack(c, inBytes,0);
69             assertEquals("8 bytes should be consumed irrespective of 2nd and 3rd bitmap indicators",8,consumed);
70             assertEquals("8 byte can only result in a bitmap holding fields up to 64",64,((BitSet)c.getValue()).length() - 1);
71             
72             outBytes = eightBytes.pack(c);
73             assertEquals("8 byte bitmap pack should reflect unpack",ISOUtil.hexString(inBytes,0,8),ISOUtil.hexString(outBytes));
74             
75             try {
76                 outBytes = oneByte.pack(c);
77                 fail("Pack of bitmap with fields outside of 1 byte range should result in ISOException");
78             } catch (Exception e) {
79                 // expected.
80                 assertEquals("Bitmap can only hold fields numbered up to 8 in the 1 bytes available.",e.getMessage());
81             }
82         }
