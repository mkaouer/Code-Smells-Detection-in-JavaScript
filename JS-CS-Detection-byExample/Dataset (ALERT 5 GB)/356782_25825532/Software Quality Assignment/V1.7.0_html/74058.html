19     package org.jpos.transaction.gui;
20     
21     import org.jpos.ui.UIFactory;
22     import org.jpos.ui.UI;
23     import org.jpos.transaction.TransactionManager;
24     import org.jpos.util.NameRegistrar;
25     import org.jdom.Element;
26     
27     import javax.swing.*;
28     
29     public class TMMonitorFactory implements UIFactory {
30         public JComponent create(UI ui, Element e) {
31             try {
32                 TransactionManager tm =
33                     (TransactionManager) NameRegistrar.get (e.getAttributeValue ("transaction-manager"));
34                 return new TMMonitor (ui, tm);
35             } catch (NameRegistrar.NotFoundException ex) {
36                 return new JTextArea (ex.toString());
37             }
38         }
39     }
