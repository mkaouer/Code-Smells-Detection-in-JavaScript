21     import junit.framework.*;
22     import org.jpos.q2.Q2;
23     import org.jpos.space.Space;
24     import org.jpos.space.SpaceFactory;
25     
26     public class TransactionManagerTestCase extends TestCase {
27         Q2 q2;
28         Space sp;
29         public static String QUEUE = "TXNMGRTEST";
30     
31         public void setUp () throws Exception {
32             // sp = SpaceFactory.getSpace("tspace:txnmgrtest");
33             sp = SpaceFactory.getSpace();
34             q2 = new Q2("../test/org/jpos/transaction");
35             q2.start();
36         }
37         public void testSimpleTransaction() {
38             Context ctx = new Context();
39             ctx.put ("volatile", "the quick brown fox");
40             ctx.put ("persistent", "jumped over the lazy dog", true);
41             sp.out (QUEUE, ctx);
42         }
43         public void testRetryTransaction() {
44             Context ctx = new Context();
45             ctx.put ("RETRY", new Integer(10), true);
46             sp.out (QUEUE, ctx);
47         }
48         public void tearDown() throws Exception {
49             Thread.sleep (3000); // let the thing run
50             q2.stop();
51         }
52     }
