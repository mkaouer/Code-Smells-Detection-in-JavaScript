19     package org.jpos.iso;
20     
21     /**
22      * Implements BCD Interpreter. Numeric Strings (consisting of chars '0'..'9' are converted
23      * to and from BCD bytes. Thus, "1234" is converted into 2 bytes: 0x12, 0x34.
24      * 
25      * @author joconnor
26      * @version $Revision$ $Date$
27      */
28     public class BCDInterpreter implements Interpreter
29     {
30         /** This BCDInterpreter sometimes adds a 0-nibble to the left. */
31         public static final BCDInterpreter LEFT_PADDED = new BCDInterpreter(true, false);
32         /** This BCDInterpreter sometimes adds a 0-nibble to the right. */
33         public static final BCDInterpreter RIGHT_PADDED = new BCDInterpreter(false, false);
34         /** This BCDInterpreter sometimes adds a F-nibble to the right. */
35         public static final BCDInterpreter RIGHT_PADDED_F = new BCDInterpreter(false, true);
36         /** This BCDInterpreter sometimes adds a F-nibble to the left. */
37         public static final BCDInterpreter LEFT_PADDED_F = new BCDInterpreter(true, true);
38     
39         private boolean leftPadded;
40         private boolean fPadded;
41     
42         /** Kept private. Only two instances are possible. */
43         private BCDInterpreter(boolean leftPadded, boolean fPadded) {
44             this.leftPadded = leftPadded;
45             this.fPadded = fPadded;
46         }
47     
48         /**
49     	 * (non-Javadoc)
50     	 * 
51     	 * @see org.jpos.iso.Interpreter#interpret(java.lang.String)
52     	 */
53         public void interpret(String data, byte[] b, int offset)
54         {
55             ISOUtil.str2bcd(data, leftPadded, b, offset);
56             // if (fPadded && !leftPadded && data.length()%2 == 1)
57             //   b[b.length-1] |= (byte)(b[b.length-1] << 4) == 0 ? 0x0F : 0x00;
58             int paddedSize = data.length() >> 1;
59             if (fPadded && data.length()%2 == 1)
60                 if (leftPadded)
61                     b[offset] |= (byte) 0xF0;
62                 else
63                     b[offset+paddedSize] |= (byte) 0x0F;
64         }
65     
66         /**
67     	 * (non-Javadoc)
68     	 * 
69     	 * @see org.jpos.iso.Interpreter#uninterpret(byte[])
70     	 */
71         public String uninterpret(byte[] rawData, int offset, int length)
72         {
73             return ISOUtil.bcd2str (rawData, offset, length, leftPadded);
74         }
75     
76         /**
77     	 * Each numeric digit is packed into a nibble, so 2 digits per byte, plus the
78          * possibility of padding.
79     	 * 
80     	 * @see org.jpos.iso.Interpreter#getPackedLength(int)
81     	 */
82         public int getPackedLength(int nDataUnits)
83         {
84             return (nDataUnits + 1) / 2;
85         }
86     }
