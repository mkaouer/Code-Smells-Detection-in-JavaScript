112         public void run () {
113             started = true;
114             Thread.currentThread().setName ("Q2-"+getInstanceId().toString());
115             try {
116                 /*
117                 * The following code determines whether a MBeanServer exists 
118                 * already. If so then the first one in the list is used. 
119                 * I have not yet find a way to interrogate the server for 
120                 * information other than MBeans so to pick a specific one 
121                 * would be difficult.
122                 */
123                 ArrayList mbeanServerList = 
124                     MBeanServerFactory.findMBeanServer(null);
125                 if (mbeanServerList.size() == 0) {
126                     server  = MBeanServerFactory.createMBeanServer (JMX_NAME);
127                 } else {
128                     server = (MBeanServer) mbeanServerList.get(0);
129                 }
130                 final ObjectName loaderName = new ObjectName (Q2_CLASS_LOADER);
131                 try {
132                     loader = (QClassLoader) java.security.AccessController.doPrivileged(
133                         new java.security.PrivilegedAction() {
134                             public Object run() {
135                                 return new QClassLoader (server, libDir, loaderName, mainClassLoader);
136                             }
137                         }
138                     );
139                     server.registerMBean (loader, loaderName);
140                     loader = loader.scan(false);
141                 } catch (Throwable t) {
142                     if (log != null)
143                         log.error ("initial-scan", t);
144                     else
145                         t.printStackTrace();
146                 }
147                 factory = new QFactory (loaderName, this);
148                 initSystemLogger ();
149                 addShutdownHook ();
150                 q2Thread = Thread.currentThread ();
151                 q2Thread.setContextClassLoader (loader);
152                 if (cli != null)
153                     cli.start();
154                 initConfigDecorator();
155                 for (int i=1; !shutdown; i++) {
156                     try {
157                         boolean forceNewClassLoader = scan ();
158                         QClassLoader oldClassLoader = loader;
159                         loader = loader.scan (forceNewClassLoader);
160                         if (loader != oldClassLoader) {
161                             oldClassLoader = null; // We want't this to be null so it gets GCed.
162                             System.gc();  // force a GC
163                             log.info (
164                             "new classloader ["
165                             + Integer.toString(loader.hashCode(),16)
166                             + "] has been created"
167                             );
168                         }
169                         deploy ();
170                         checkModified ();
171                         relax (SCAN_INTERVAL);
172                         if (i % (3600000 / SCAN_INTERVAL) == 0)
173                             logVersion();
174                     } catch (Throwable t) {
175                         log.error ("start", t);
176                         relax ();
177                     }
178                 }
179                 undeploy ();
180                 try {
181                     server.unregisterMBean (loaderName);
182                 } catch (InstanceNotFoundException e) {
183                     log.error (e);
184                 }
185                 if(decorator!=null)
186                 {
187                     decorator.uninitialize();
188                 }
189                 if (exit && !shuttingDown)
190                     System.exit (0);
191             } catch (Exception e) {
192                 log.error (e);
193                 System.exit (1);
194             }
195         }

236         private boolean scan () {
237                 boolean rc = false;
238                 File file[] = deployDir.listFiles (this);
239                 // Arrays.sort (file); --apr not required - we use TreeMap
240                 if (file == null) {
241                     // Shutting down might be best, how to trigger from within?
242                     throw new Error("Deploy directory \""+deployDir.getAbsolutePath()+"\" is not available");
243                 } else {
244                     for (File f : file) {
245                         if (register(f))
246                             rc = true;
247                     }
248                 }
249                 return rc;
250         }

252         private void deploy () {
253             List startList = new ArrayList ();
254             Iterator iter = dirMap.entrySet().iterator();
255             try {
256                 while (iter.hasNext() && !shutdown) {
257                     Map.Entry entry = (Map.Entry) iter.next();
258                     File   f        = (File)   entry.getKey ();
259                     QEntry qentry   = (QEntry) entry.getValue ();
260                     long deployed   = qentry.getDeployed ();
261                     if (deployed == 0) {
262                         if (deploy (f)) {
263                             if (qentry.isQBean ())
264                                 startList.add (qentry.getInstance());
265                             qentry.setDeployed (f.lastModified ());
266                         } else {
267                             // deploy failed, clean up.
268                             iter.remove();
269                         }
270                     } else if (deployed != f.lastModified ()) {
271                         undeploy (f);
272                         iter.remove ();
273                         loader.forceNewClassLoaderOnNextScan();
274                     }
275                 }
276                 iter = startList.iterator();
277                 while (iter.hasNext ()) {
278                     start ((ObjectInstance) iter.next ());
279                 }
280             }
281             catch (Exception e){
282                 log.error ("deploy", e);
283             }
284         }

286         private void undeploy () {
287             Object[] set = dirMap.entrySet().toArray ();
288             int l = set.length;
289     
290             while (l-- > 0) {
291                 Map.Entry entry = (Map.Entry) set[l];
292                 File   f  = (File) entry.getKey ();
293                 undeploy (f);
294             }
295         }

297         private void addShutdownHook () {
298             Runtime.getRuntime().addShutdownHook (
299                 new Thread ("Q2-ShutdownHook") {
300                     public void run () {
301                         shuttingDown = true;
302                         shutdown = true;
303                         if (q2Thread != null) {
304                             log.info ("shutting down (hook)");
305                             try {
306                                 q2Thread.join (SHUTDOWN_TIMEOUT);
307                             } catch (InterruptedException e) { 
308                             } catch (NullPointerException e) {
309                                 // on thin Q2 systems where shutdown is very fast, 
310                                 // q2Thread can become null between the upper if and
311                                 // the actual join. Not a big deal so we ignore the
312                                 // exception.
313                             }
314                         }
315                     }
316                 }
317             );
318         }

320         private void checkModified () {
321             Iterator iter = dirMap.entrySet().iterator();
322             while (iter.hasNext()) {
323                 Map.Entry entry = (Map.Entry) iter.next();
324                 File   f        = (File)   entry.getKey ();
325                 QEntry qentry   = (QEntry) entry.getValue ();
326                 if (qentry.isQBean() && qentry.isQPersist()) {
327                     ObjectName name = qentry.getObjectName ();
328                     if (getState (name) == QBean.STARTED && isModified (name)) {
329                         qentry.setDeployed (persist (f, name));
330                     }
331                 }
332             }
333         }

357         private long persist (File f, ObjectName name) {
358             long deployed = f.lastModified ();
359             try {
360                 Element e = (Element) server.getAttribute (name, "Persist");
361                 if (e != null) {
362                     XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
363                     Document doc = new Document ();
364                     e.detach();
365                     doc.setRootElement (e);
366                     File tmp = new File (f.getAbsolutePath () + ".tmp");
367                     FileWriter writer = new FileWriter (tmp);
368                     out.output (doc, writer);
369                     writer.close ();
370                     f.delete();
371                     tmp.renameTo (f);
372                     deployed = f.lastModified ();
373                 }
374             } catch (Exception ex) {
375                 log.warn ("persist", ex);
376             }
377             return deployed;
378         }

381             QEntry qentry = (QEntry) dirMap.get (f);
382             try {
383                 if (log != null)
384                     log.trace ("undeploying:" + f.getCanonicalPath());
385     
386                 Object obj      = qentry.getObject ();
387                 ObjectName name = qentry.getObjectName ();
388                 factory.destroyQBean (this, name, obj);

392                 getLog().warn ("undeploy", e);

396         private boolean register (File f) {
397             boolean rc = false;
398             if (f.isDirectory()) {
399                 File file[] = f.listFiles (this);
400                 for (int i=0; i<file.length; i++) {
401                     if (register (file[i]))
402                         rc = true;
403                 }
404             } else if (dirMap.get (f) == null) {
405                 dirMap.put (f, new QEntry ());
406                 rc = true;
407             }
408             return rc;
409         }

411         private boolean deploy (File f) {
412             try {
413                 if (log != null)
414                     log.info ("deploy:" + f.getCanonicalPath());
415                 QEntry qentry = (QEntry) dirMap.get (f);
416                 SAXBuilder builder = createSAXBuilder();
417                 Document doc;
418                 if(decorator!=null && !f.getName().equals(LOGGER_CONFIG))
419                 {
420                     doc=decrypt(builder.build(new StringReader(decorator.decorateFile(f))));
421                 }
422                 else
423                 {
424                     doc=decrypt(builder.build(f));
425                 }
426     
427                 Element rootElement = doc.getRootElement();
428                 String iuuid = rootElement.getAttributeValue ("instance");
429                 if (iuuid != null) {
430                     UUID uuid = UUID.fromString(iuuid);
431                     if (!uuid.equals (getInstanceId())) {
432                         deleteFile (f, iuuid);
433                         return false;
434                     }
435                 }
436                 Object obj = factory.instantiate (this, rootElement);
437                 qentry.setObject (obj);
438     
439                 ObjectInstance instance = factory.createQBean (
440                     this, doc.getRootElement(), obj
441                 );
442                 qentry.setInstance (instance);
443             } 
444             catch (InstanceAlreadyExistsException e) {
445                /*
446                 * Ok, the file we tried to deploy, holds an object
447                 *  that already has been deployed.
448                 *  
449                 * Rename it out of the way.
450                 * 
451                 */
452                 tidyFileAway(f,DUPLICATE_EXTENSION);
453                 getLog().warn ("deploy", e);
454                 return false;
455             }
456             catch (Exception e) {
457                 getLog().warn ("deploy", e);
458                 tidyFileAway(f,ERROR_EXTENSION);
459                 // This will also save deploy error repeats...
460                 return false;
461             } 
462             catch (Error e) {
463                 getLog().warn ("deploy", e);
464                 tidyFileAway(f,ENV_EXTENSION);
465                 // This will also save deploy error repeats...
466                 return false;
467             }
468             return true ;
469         }

542         private void parseCmdLine (String[] args) {
543             CommandLineParser parser = new PosixParser ();
544     
545             Options options = new Options ();
546             options.addOption ("v","version", false, "Q2's version");
547             options.addOption ("d","deploydir", true, "Deployment directory");
548             options.addOption ("r","recursive", false, "Deploy subdirectories recursively");
549             options.addOption ("h","help", false, "Usage information");
550             options.addOption ("C","config", true, "Configuration bundle");
551             options.addOption ("e","encrypt", true, "Encrypt configuration bundle");
552             options.addOption ("i","cli", false, "Command Line Interface");
553             options.addOption ("c","command", true, "Command to execute");
554     
555             try {
556                 CommandLine line = parser.parse (options, args);
557                 if (line.hasOption ("v")) {
558                     displayVersion();
559                     System.exit (0);
560                 } 
561                 if (line.hasOption ("h")) {
562                     HelpFormatter helpFormatter = new HelpFormatter ();
563                     helpFormatter.printHelp ("Q2", options);
564                     System.exit (0);
565                 } 
566                 if (line.hasOption ("c")) {
567                     cli = new CLI(this, line.getOptionValue("c"), line.hasOption("i"));
568                 } else if (line.hasOption ("i")) 
569                     cli = new CLI(this, null, true);
570     
571                 String dir = DEFAULT_DEPLOY_DIR;
572                 if (line.hasOption ("d")) {
573                     dir = line.getOptionValue ("d");
574                 }
575                 recursive = line.hasOption ("r");
576                 this.deployDir  = new File (dir);
577                 if (line.hasOption ("C"))
578                     deployBundle (new File (line.getOptionValue ("C")), false);
579                 if (line.hasOption ("e"))
580                     deployBundle (new File (line.getOptionValue ("e")), true);
581             } catch (MissingArgumentException e) {
582                 System.out.println ("ERROR: " + e.getMessage());
583                 System.exit (1);
584             } catch (Exception e) {
585                 e.printStackTrace ();
586                 System.exit (1);
587             }
588         }
589         private void deployBundle (File bundle, boolean encrypt) 
590             throws JDOMException, IOException, 
591                     ISOException, GeneralSecurityException
592         {
593             SAXBuilder builder = createSAXBuilder();
594             Document doc = builder.build (bundle);
595             Iterator iter = doc.getRootElement().getChildren ().iterator ();
596             for (int i=1; iter.hasNext (); i ++) {
597                 Element e = (Element) iter.next();
598                 deployElement (e, String.format ("%02d_%s.xml",i, e.getName()), encrypt, true);
599             }
600         }
601         public void deployElement (Element e, String fileName, boolean encrypt, boolean isTransient)
602             throws ISOException, IOException, GeneralSecurityException
603         {
604             e = ((Element) e.clone ());
605     
606             XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
607             Document doc = new Document ();
608             doc.setRootElement (e);
609             File qbean = new File (deployDir, fileName);
610             if (isTransient) {
611                 e.setAttribute("instance", getInstanceId().toString());
612                 qbean.deleteOnExit();
613             }
614             FileWriter writer = new FileWriter (qbean);
615             if (encrypt)
616                 doc = encrypt (doc);
617             out.output (doc, writer);
618             writer.close ();
619         }

621         private byte[] dodes (byte[] data, int mode) 
622            throws GeneralSecurityException
623         {
624             Cipher cipher = Cipher.getInstance ("DES");
625             cipher.init (mode, new SecretKeySpec(getKey(), "DES"));
626             return cipher.doFinal (data);
627         }
628         protected byte[] getKey() {
629             return "CAFEBABE".getBytes();
630         }
631         protected Document encrypt (Document doc)
632             throws GeneralSecurityException, IOException
633         {
634             ByteArrayOutputStream os = new ByteArrayOutputStream ();
635             OutputStreamWriter writer = new OutputStreamWriter (os);
636             XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
637             out.output (doc, writer);
638             writer.close ();
639     
640             byte[] crypt = dodes (os.toByteArray(), Cipher.ENCRYPT_MODE);
641     
642             Document secureDoc = new Document ();
643             Element root = new Element (PROTECTED_QBEAN);
644             secureDoc.setRootElement (root);
645             Element secureData = new Element ("data");
646             root.addContent (secureData);
647     
648             secureData.setText (
649                 ISOUtil.hexString (crypt)
650             );
651             return secureDoc;
652         }
653     
654         protected Document decrypt (Document doc) 
655             throws GeneralSecurityException, IOException, JDOMException
656         {
657             Element root = doc.getRootElement ();
658             if (PROTECTED_QBEAN.equals (root.getName ())) {
659                 Element data = root.getChild ("data");
660                 if (data != null) {
661                     ByteArrayInputStream is = new ByteArrayInputStream (
662                         dodes (
663                             ISOUtil.hex2byte (data.getTextTrim()),
664                             Cipher.DECRYPT_MODE)
665                     );
666                     SAXBuilder builder = createSAXBuilder();
667                     doc = builder.build (is);
668                 }
669             }
670             return doc;
671         }

687         private void initConfigDecorator()
688         {
689             InputStream in=Q2.class.getClassLoader().getResourceAsStream("META-INF/org/jpos/config/Q2-decorator.properties");
690             try
691             {
692                 if(in!=null)
693                 {
694                     PropertyResourceBundle bundle=new PropertyResourceBundle(in);
695                     String ccdClass=bundle.getString("config-decorator-class");
696                     if(log!=null) log.info("Initializing config decoration provider: "+ccdClass);
697                     decorator= (ConfigDecorationProvider) Q2.class.getClassLoader().loadClass(ccdClass).newInstance();
698                     decorator.initialize(getDeployDir());
699                 }
700             }
701             catch (IOException e)
702             {
703             }
704             catch (Exception e)
705             {
706                 if(log!=null) log.error(e);
707                 else
708                 {
709                     e.printStackTrace();
