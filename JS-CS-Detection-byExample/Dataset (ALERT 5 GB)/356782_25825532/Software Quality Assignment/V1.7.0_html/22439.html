19     package org.jpos.space;
20     
21     public class SpaceTap implements SpaceListener {
22         LocalSpace ssp;
23         LocalSpace dsp;
24         Object key;
25         Object tapKey;
26         long tapTimeout;
27     
28         /**
29          * @param sp space
30          * @param key key to monitor
31          * @param tapKey key to use when copying 
32          * @param tapTimeout copy timeout in millis
33          */
34         public SpaceTap (LocalSpace sp, Object key, Object tapKey, long tapTimeout) {
35             this (sp, sp, key, tapKey, tapTimeout);
36         }
37         /**
38          * @param ssp source space
39          * @param dsp destination space
40          * @param key key to monitor
41          * @param tapKey key to use when copying 
42          * @param tapTimeout copy timeout in millis
43          */
44         public SpaceTap (LocalSpace ssp, LocalSpace dsp, Object key, Object tapKey, long tapTimeout) {
45             super();
46             this.ssp = ssp;
47             this.dsp = dsp;
48             this.key = key;
49             this.tapKey = tapKey;
50             this.tapTimeout = tapTimeout;
51             if (key.equals (tapKey) && ssp == dsp)
52                 throw new IllegalArgumentException ("Possible deadlock - key equals tap-key within same space");
53             ssp.addListener (key, this);
54         }
55         public void notify (Object key, Object value) {
56             dsp.out (tapKey, value, tapTimeout);
57         }
58        
59         public void close() {
60             if (ssp != null) {
61                 ssp.removeListener (key, this);
62                 ssp = null;
63             }
64         }
65     }
