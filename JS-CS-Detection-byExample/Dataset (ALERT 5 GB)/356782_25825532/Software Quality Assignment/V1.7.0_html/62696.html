19     package org.jpos.space;
20     
21     import org.jpos.core.Configuration;
22     import org.jpos.core.ConfigurationException;
23     import org.jpos.core.Configurable;
24     
25     import javax.naming.InitialContext;
26     import javax.naming.NamingException;
27     import java.io.Serializable;
28     import java.rmi.RemoteException;
29     import java.rmi.registry.LocateRegistry;
30     import java.rmi.registry.Registry;
31     import java.rmi.server.*;
32     import java.util.Set;
33     
34     /**
35      * RMI Space Proxy 
36      * @author Alejandro Revilla
37      * @author Niclas Hedhman
38      * @version $Revision$ $Date$
39      * @since 1.4.9
40      */
41     public class SpaceProxy implements RemoteSpace, Configurable {
42         Space sp;
43         Configuration cfg;
44         private RemoteRef ref;
45         private RemoteStub stub;
46         public SpaceProxy () throws RemoteException {
47             super();
48             sp = SpaceFactory.getSpace ();
49             startService ();
50         }
51         public SpaceProxy (String spaceUri) throws RemoteException {
52             super ();
53             sp = SpaceFactory.getSpace (spaceUri);
54             startService ();
55         }
56     
57         private void startService () throws RemoteException 
58         {
59             try {
60                 LocateRegistry.createRegistry (Registry.REGISTRY_PORT);
61             } catch (ExportException e) {
62                 // registry already exists
63             }
64             stub = UnicastRemoteObject.exportObject (this);
65             ref  = stub.getRef();
66         }
67     
68         public void out (Serializable key, Serializable value) 
69             throws RemoteException
70         {
71             sp.out (key, value);
72         }
73         public void out (Serializable key, Serializable value, long timeout)
74             throws RemoteException
75         {
76             sp.out (key, value, timeout);
77         }
78         public Serializable in (Serializable key)
79             throws RemoteException
80         {
81             return (Serializable) sp.in (key);
82         }
83         public Serializable rd  (Serializable key)
84             throws RemoteException
85         {
86             return (Serializable) sp.rd (key);
87         }
88         public Serializable in  (Serializable key, long timeout)
89             throws RemoteException
90         {
91             return (Serializable) sp.in (key, timeout);
92         }
93         public Serializable rd  (Serializable key, long timeout)
94             throws RemoteException
95         {
96             return (Serializable) sp.rd (key, timeout);
97         }
98         public Serializable inp (Serializable key)
99             throws RemoteException
100         {
101             return (Serializable) sp.inp (key);
102         }
103         public Serializable rdp (Serializable key)
104             throws RemoteException
105         {
106             return (Serializable) sp.rdp (key);
107         }
108         public void shutdown() {
109             try {
110                 if (UnicastRemoteObject.unexportObject (this, false)) 
111                     return;
112                 Thread.sleep (5000);
113                 UnicastRemoteObject.unexportObject (this, true);
114             } catch (Exception e) {
115                 // nothing to do .. we're shutting down ...
116             }
117         }
118         public void setConfiguration (Configuration cfg) 
119             throws ConfigurationException
120         {
121             this.cfg = cfg;
122             try {
123                 InitialContext ctx = new InitialContext ();
124                 ctx.rebind (cfg.get ("name"), stub);
125             } catch (NamingException e) {
126                 throw new ConfigurationException (e);
127             }
128         }
129         public Set getKeySet () {
130             if (sp instanceof LocalSpace) 
131                 return ((LocalSpace)sp).getKeySet ();
132             else
133                 return null;
134         }
135         public String toString() {
136             if (ref == null)
137                 return getClass().getName() + "[<unexported>]";
138             else
139                 return getClass().getName() + "[" + ref.remoteToString() + "]";
140         }
141         public int hashCode() {
142             if (ref == null )
143                 return super.hashCode();
144             else
145                 return ref.remoteHashCode();
146         }
147         public boolean equals (Object obj) {
148             if (obj instanceof RemoteObject) {
149                 if (ref == null) {
150                     return obj == this;
151                 } else {
152                     RemoteRef otherRef = ((RemoteObject)obj).getRef();
153                     return ref.remoteEquals (otherRef);
154                 }
155             } else if (obj != null) {
156                 return obj.equals (this);
157             } else {
158                 return false;
159             }
160         }
161     }
