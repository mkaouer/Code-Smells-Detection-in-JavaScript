31         public void testSimpleOut() throws Exception {

37         public void testOutRdpInpRdp() throws Exception {
38             Object o = new Boolean (true);
39             String k = "testOutRdpInpRdp_Key";
40             sp.out (k, o);
41             assertTrue (o.equals (sp.rdp (k)));
42             assertTrue (o.equals (sp.rd  (k)));
43             assertTrue (o.equals (sp.rd  (k, 1000)));
44             assertTrue (o.equals (sp.inp (k)));
45             assertTrue (sp.rdp (k) == null);
46             assertTrue (sp.rd  (k, 100) == null);
47         }
48         public void testMultiKeyLoad() throws Exception {

64         }
65         public void testNoAutoCommit () throws Exception {
66             String s = "The quick brown fox jumped over the lazy dog";
67             Profiler prof = new Profiler ();
68             synchronized (sp) {
69                 sp.setAutoCommit (false);
70                 for (int i=0; i<COUNT; i++) {
71                     sp.out ("testNoAutoCommit_Key" + Integer.toString (i), s);
72                     if (i % 100 == 0)
73                         prof.checkPoint ("out " + i);
74                 }
75                 prof.checkPoint ("pre-commit");
76                 sp.commit();
77                 sp.setAutoCommit (true);
78                 prof.checkPoint ("commit");
79             }
80             // prof.dump (System.err, "NoAutoCommit out >");
81             prof = new Profiler ();
82             synchronized (sp) {
83                 sp.setAutoCommit (false);
84                 for (int i=0; i<COUNT; i++) {
85                     assertTrue (s.equals (sp.in ("testNoAutoCommit_Key" + Integer.toString (i))));
86                     if (i % 100 == 0)
87                         prof.checkPoint ("in " + i);
88                 }
89                 prof.checkPoint ("pre-commit");
90                 sp.commit();
91                 sp.setAutoCommit (true);
92                 prof.checkPoint ("commit");
93             }
94             // prof.dump (System.err, "NoAutoCommit in  >");
95         }

115         public void testTemplate () throws Exception {

117             sp.out (key, "Value 1");
118             sp.out (key, "Value 2");
119             sp.out (key, "Value 3");

121             String k2r = (String)sp.rdp (new MD5Template (key, "Value 2"));

141         public void testExist() {
142             sp.out ("KEYA", Boolean.TRUE);
143             sp.out ("KEYB", Boolean.TRUE);
144     
145             assertTrue (
146                 "existAny ([KEYA])",
147                 sp.existAny (new String[] { "KEYA" })
148             );
149     
150             assertTrue (
151                 "existAny ([KEYB])",
152                 sp.existAny (new String[] { "KEYB" })
153             );
154             assertTrue (
155                 "existAny ([KEYA,KEYB])",
156                 sp.existAny (new String[] { "KEYA", "KEYB" })
157             );
158             assertFalse (
159                 "existAny ([KEYC,KEYD])",
160                 sp.existAny (new String[] { "KEYC", "KEYD" })
161             );
162         }
163         public void testExistWithTimeout() {
164             assertFalse (
165                 "existAnyWithTimeout ([KA,KB])",
166                 sp.existAny (new String[] { "KA", "KB" })
167             );
168             assertFalse (
169                 "existAnyWithTimeout ([KA,KB], delay)",
170                 sp.existAny (new String[] { "KA", "KB" }, 1000L)
171             );
172             new Thread() {
173                 public void run() {
174                     ISOUtil.sleep (1000L);
175                     sp.out ("KA", Boolean.TRUE);
176                 }
177             }.start();
178             long now = System.currentTimeMillis();
179             assertTrue (
180                 "existAnyWithTimeout ([KA,KB], delay)",
181                 sp.existAny (new String[] { "KA", "KB" }, 2000L)
182             );
183             long elapsed = System.currentTimeMillis() - now;
184             assertTrue ( "delay was > 1000", elapsed > 900L);
185             assertNotNull ("Entry should not be null", sp.inp ("KA"));
186         }
