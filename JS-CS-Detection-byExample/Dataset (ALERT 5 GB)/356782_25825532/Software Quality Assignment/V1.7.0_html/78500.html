67             this.tag = tag;

77         /*
78          * @return tag + length + value of the TLV Message
79          */
80         public byte[] getTLV() {
81             String hexVal = Integer.toHexString(tag);
82             byte[] bTag = ISOUtil.hex2byte(hexVal);
83             byte[] bLen = getL();
84             if (value != null) {
85                 int tLength = bTag.length + bLen.length + value.length;
86                 byte[] out = new byte[tLength];
87                 System.arraycopy(bTag, 0, out, 0, bTag.length);
88                 System.arraycopy(bLen, 0, out, bTag.length, bLen.length);
89                 System.arraycopy(value, 0, out, bTag.length + bLen.length,
90                         value.length);
91                 return out;
92             } else {//Length can be 0
93                 int tLength = bTag.length + bLen.length;
94                 byte[] out = new byte[tLength];
95                 System.arraycopy(bTag, 0, out, 0, bTag.length);
96                 System.arraycopy(bLen, 0, out, bTag.length, bLen.length);
97                 return out;
98     
99             }
100         }
101     
102         /**
103          * Value up to 127 can be encoded in single byte and multiple bytes are
104          * required for length bigger than 127
105          * 
106          * @return encoded length
107          */
108         public byte[] getL() {
109     
110             if (value == null)
111                 return new byte[1];
112             int ix = 0;
113             int tmp = value.length;
114             int bytes = 0;
115             // if Length is greater less than 127
116             //set the 8bit as 0 indicating next 7 bits is the length
117             //of the message
118             //if length is more than 127 then, set the first bit as 1 indicating
119             //next 7 bits will indicate the length of following bytes used for
120             // length
121     
122             while (tmp != 0) {
123                 tmp = tmp >> 8;
124                 bytes++;
125             }
126     
127             /* If value can be encoded on one byte */
128             if (bytes <= 1 && value.length <= 127) {
129                 byte[] rBytes = new byte[bytes];
130                 rBytes[0] = (byte) value.length;
131                 return rBytes;
132             }
133             //else {
134             /* Value to be encoded on multiple bytes */
135             //we need 1 byte to indicate the length
136             byte[] rBytes = new byte[1 + bytes];
137             rBytes[0] = (byte) (0x80 | bytes);
138     
139             int mask = 0xFF;
140             tmp = value.length;
141             while (ix < bytes) {
142                 /* Mask off 8 bits of the value at a time */
143                 rBytes[(bytes - ix)] = (byte) (tmp & mask);
144                 bytes--;
145                 /* Shift value right 8 bits, effectively removing them */
146                 tmp = (tmp >> 8);
147             }
148             //}
149             return rBytes;
150         }
151         
152         /*
153          * @return value 
154          */
155         public String getStringValue() {
156             return new String(ISOUtil.hexString(value));
157         }
158     }
