37     /** A TransactionParticipant whichï¿½s prepare, commit and abort methods can be 
38      *  specified through beanshell scripts. <BR>
39      *
40      *  To indicate what code to execute for any of the methods just add an element 
41      *  named 'prepare', 'commit' or 'abort' contained in that of the participant. <BR>
42      *
43      *  See BSHMethod for details on the syntax of these elemets. The value to return 
44      *  in the prepare method should be stored in the script variable named "result".
45      *  None of these tags are mandatory. <BR>
46      *
47      *  You can subclass BSHTransactionParticipant and override the default... 
48      *  methods. That way you can provide default behaviour for a participant and 
49      *  override it at deploy time through scripts. 
50      * 
51      * @see BSHMethod
52      * @author  AMarques
53      */
54     public class BSHTransactionParticipant extends SimpleLogSource 
55         implements TransactionParticipant, AbortParticipant, XmlConfigurable 
56     {
57         
58         protected BSHMethod prepareMethod;
59         protected BSHMethod prepareForAbortMethod;
60         protected BSHMethod commitMethod;
61         protected BSHMethod abortMethod;
62     
63         boolean trace;
64         
65         /** Creates a new instance of BSHTransactionParticipant */
66         public BSHTransactionParticipant() {
67             super();
68         }
69         
70         public void abort(long id, java.io.Serializable context) {
71             LogEvent ev = new LogEvent(this, "abort");
72             if (abortMethod != null) {
73                 try {
74                     executeMethod(abortMethod, id, context, ev, "");
75                 } catch (Exception ex) {
76                     ev.addMessage(ex);
77                 }
78             } else {
79                 defaultAbort(id, context, ev);
80             }
81             if (trace)
82                 Logger.log(ev);
83         }
84         
85         protected void defaultAbort(long id, Serializable context, LogEvent ev) {}
86         
87         public void commit(long id, java.io.Serializable context) {
88             LogEvent ev = new LogEvent(this, "commit");
89             if (commitMethod != null) {
90                 try {
91                     executeMethod(commitMethod, id, context, ev, "");
92                 } catch (Exception ex) {
93                     ev.addMessage(ex);
94                 }
95             } else {
96                 defaultCommit(id, context, ev);
97             }
98             if (trace)
99                 Logger.log(ev);
100         }
101         
102         protected void defaultCommit(long id, Serializable context, LogEvent ev) {}
103         
104         public int prepare(long id, java.io.Serializable context) {
105             LogEvent ev = new LogEvent(this, "prepare");
106             int result = ABORTED | READONLY;
107             if (prepareMethod != null) {
108                 try {
109                     result = ((Integer) executeMethod(prepareMethod, id, context, ev, "result")).intValue();
110                 } catch (Exception ex) {
111                     ev.addMessage(ex);
112                 }
113             } else {
114                 result = defaultPrepare(id, context, ev);
115             }
116             ev.addMessage("result", Integer.toBinaryString(result));
117             if (trace)
118                 Logger.log(ev);
119             return result;
120         }
121     
122         public int prepareForAbort(long id, java.io.Serializable context) {
123             LogEvent ev = new LogEvent(this, "prepare-for-abort");
124             int result = ABORTED | READONLY;
125             if (prepareForAbortMethod != null) {
126                 try {
127                     result = ((Integer) executeMethod(prepareForAbortMethod, id, context, ev, "result")).intValue();
128                 } catch (Exception ex) {
129                     ev.addMessage(ex);
130                 }
131             } 
132             ev.addMessage("result", Integer.toBinaryString(result));
133             if (trace)
134                 Logger.log(ev);
135             return result;
136         }
137         
138         protected int defaultPrepare(long id, Serializable context, LogEvent ev) {
139             return PREPARED | READONLY;
140         }
141         
142         public void setConfiguration(Element e) throws ConfigurationException {
143     	try {
144                 prepareMethod = BSHMethod.createBshMethod(e.getChild("prepare"));
145                 prepareForAbortMethod = BSHMethod.createBshMethod(e.getChild("prepare-for-abort"));
146                 commitMethod = BSHMethod.createBshMethod(e.getChild("commit"));
147                 abortMethod = BSHMethod.createBshMethod(e.getChild("abort"));
148                 trace = "yes".equals (e.getAttributeValue ("trace"));
149             } catch (Exception ex) {
150                 throw new ConfigurationException(ex.getMessage(), ex);
151             }
152         }
153         
154         protected Object executeMethod(BSHMethod m, long id, Serializable context, LogEvent evt, String resultName) 
155         throws EvalError, FileNotFoundException, IOException {
156             Map params = new HashMap();
157             params.put("context", context);
158             params.put("id", new Long(id));
159             params.put("evt", evt);
160             params.put("self", this);
161             return m.execute(params, resultName);
162         }
163     }
