35         /** Tests for equality of array elements */
36         public static void assertEquals(byte[] expected, byte[] was) {
37             if (expected.length != was.length) {
38                 for (int i = 0; i < expected.length && i < was.length; i++) {
39                     Assert.assertEquals("Arrays different lengths. Non-equal objects at index " + i, expected[i], was[i]);
40                 }
41                 // Following always fails, but gives a nice error message
42                 Assert.assertEquals("Wrong size array", expected.length, was.length);
43             } else {
44                 for (int i= 0; i < expected.length; i++) {
45                     Assert.assertEquals("Non-equal objects at index " + i, expected[i], was[i]);
46                 }
47             }
48         }
