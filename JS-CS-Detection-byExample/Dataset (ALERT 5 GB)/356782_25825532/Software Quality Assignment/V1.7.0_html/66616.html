19     package org.jpos.util;
20     
21     import java.io.*;
22     
23     public class Serializer {
24         public static byte[] serialize (Object obj) throws IOException {
25             ByteArrayOutputStream baos = new ByteArrayOutputStream();
26             ObjectOutputStream os = new ObjectOutputStream(baos);
27             os.writeObject(obj);
28             return baos.toByteArray();
29         }
30         public static Object deserialize (byte[] b) throws IOException, ClassNotFoundException {
31             ByteArrayInputStream bais = new ByteArrayInputStream(b);
32             ObjectInputStream is = new ObjectInputStream(bais);
33             return is.readObject();
34         }
35     }
