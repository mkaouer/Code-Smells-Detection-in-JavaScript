40     package org.jpos.iso;

42     /**
43      * AsciiPrefixer constructs a prefix for ASCII messages.
44      * 
45      * @author joconnor
46      * @version $Revision$ $Date$
47      */
48     public class AsciiPrefixer implements Prefixer
49     {
50         /**
51          * A length prefixer for upto 9 chars. The length is encoded with 1 ASCII
52          * char representing 1 decimal digit.
53          */
54         public static final AsciiPrefixer L = new AsciiPrefixer(1);
55         /**
56     	 * A length prefixer for upto 99 chars. The length is encoded with 2 ASCII
57     	 * chars representing 2 decimal digits.
58     	 */
59         public static final AsciiPrefixer LL = new AsciiPrefixer(2);
60         /**
61     	 * A length prefixer for upto 999 chars. The length is encoded with 3 ASCII
62     	 * chars representing 3 decimal digits.
63     	 */
64         public static final AsciiPrefixer LLL = new AsciiPrefixer(3);
65         /**
66     	 * A length prefixer for upto 9999 chars. The length is encoded with 4
67     	 * ASCII chars representing 4 decimal digits.
68     	 */
69         public static final AsciiPrefixer LLLL = new AsciiPrefixer(4);
70         /**
71          * A length prefixer for upto 99999 chars. The length is encoded with 5
72          * ASCII chars representing 5 decimal digits.
73          */
74         public static final AsciiPrefixer LLLLL = new AsciiPrefixer(5);
75     
76         /**
77          * A length prefixer for upto 999999 chars. The length is encoded with 6
78          * ASCII chars representing 6 decimal digits.
79          */
80         public static final AsciiPrefixer LLLLLL = new AsciiPrefixer(6);
81     
82         //private static final LeftPadder PADDER = LeftPadder.ZERO_PADDER;
83         //private static final AsciiInterpreter INTERPRETER = AsciiInterpreter.INSTANCE;
84     
85         /** The number of digits allowed to express the length */
86         private int nDigits;
87         
88         public AsciiPrefixer(int nDigits)
89         {
90             this.nDigits = nDigits;
91         }
92     
93         /*
94     	 * (non-Javadoc)
95     	 * 
96     	 * @see org.jpos.iso.Prefixer#encodeLength(int, byte[])
97     	 */
98         public void encodeLength(int length, byte[] b) throws ISOException
99         {
100             int n = length;
101             // Write the string backwards - I don't know why I didn't see this at first.
102             for (int i = nDigits - 1; i >= 0; i--)
103             {
104                 b[i] = (byte)(n % 10 + '0');
105                 n /= 10;
106             }
107             if (n != 0)
108             {
109                 throw new ISOException("invalid len "+ length + ". Prefixing digits = " + nDigits);
110             }
111         }
112     
113         /*
114     	 * (non-Javadoc)
115     	 * 
116     	 * @see org.jpos.iso.Prefixer#decodeLength(byte[], int)
117     	 */
118         public int decodeLength(byte[] b, int offset)
119         {
120             int len = 0;
121             for (int i = 0; i < nDigits; i++)
122             {
123                 len = len * 10 + b[offset + i] - (byte)'0';
124             }
125             return len;
126         }
127     
128         /*
129     	 * (non-Javadoc)
130     	 * 
131     	 * @see xcom.traxbahn.iso.Prefixer#getLengthInBytes()
132     	 */
133         public int getPackedLength()
134         {
135             return nDigits;
136         }
137     }
