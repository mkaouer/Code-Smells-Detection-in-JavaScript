50     package org.jpos.iso;
51     
52     import java.io.Externalizable;
53     import java.io.IOException;
54     import java.io.InputStream;
55     import java.io.ObjectInput;
56     import java.io.ObjectOutput;
57     import java.io.PrintStream;
58     
59     import org.jpos.iso.packager.XMLPackager;
60     
61     /**
62      * implements <b>Leaf</b> for binary fields
63      *
64      * See the
65      * <a href="API_users_guide.html">API User's Guide</a>
66      * for details.
67      *
68      * @author apr@cs.com.uy
69      * @version $Id$
70      * @see ISOComponent
71      */
72     public class ISOBinaryField 
73         extends ISOComponent 
74         implements Cloneable, Externalizable
75     {
76         protected int fieldNumber;
77         protected byte[] value;
78     
79         /**
80          * No args constructor 
81          * <font size="-1">(required by Externalizable support on ISOMsg)</font>
82          */
83         public ISOBinaryField () {
84             fieldNumber = -1;
85         }
86         /**
87          * @param n - the FieldNumber
88          */
89         public ISOBinaryField(int n) {
90             fieldNumber = n;
91         }
92         /**
93          * @param n - fieldNumber
94          * @param v - fieldValue
95          */
96         public ISOBinaryField(int n, byte[] v) {
97             fieldNumber = n;
98             value = v;
99         }
100         /**
101          * @param n - fieldNumber
102          * @param v - fieldValue
103          * @param offset - starting offset
104          * @param len    - field length
105          */
106         public ISOBinaryField(int n, byte[] v, int offset, int len) {
107             byte[] b = new byte[len];
108             System.arraycopy (v, offset, b, 0, len);
109             fieldNumber = n;
110             value = b;
111         }
112         /**
113          * changes this Component field number<br>
114          * Use with care, this method does not change
115          * any reference held by a Composite.
116          * @param fieldNumber new field number
117          */
118         public void setFieldNumber (int fieldNumber) {
119             this.fieldNumber = fieldNumber;
120         }
121         /**
122          * not available on Leaf - always throw ISOException
123          * @exception ISOException
124          */
125         public byte[] pack() throws ISOException {
126             throw new ISOException ("Not available on Leaf");
127         }
128         /**
129          * not available on Leaf - always throw ISOException
130          * @exception ISOException
131          */
132         public int unpack(byte[] b) throws ISOException {
133             throw new ISOException ("Not available on Leaf");
134         }
135         /**
136          * not available on Leaf - always throw ISOException
137          * @exception ISOException
138          */
139         public void unpack(InputStream in) throws ISOException {
140             throw new ISOException ("Not available on Leaf");
141         }
142         /**
143          * @return Object representing this field number
144          */
145         public Object getKey() {
146             return new Integer(fieldNumber);
147         }
148         /**
149          * @return Object representing this field value
150          */
151         public Object getValue() {
152             return value;
153         }
154         /**
155          * @param obj - Object representing this field value
156          * @exception ISOException
157          */
158         public void setValue (Object obj) throws ISOException {
159             if (obj instanceof String)
160                 value = ((String) obj).getBytes();
161             else
162                 value = (byte[]) obj;
163         }
164         /**
165          * @return byte[] representing this field
166          */
167         public byte[] getBytes() {
168             return value;
169         }
170         /**
171          * dump this field to PrintStream. The output is sorta
172          * XML, intended to be easily parsed.
173          * @param p - print stream
174          * @param indent - optional indent string
175          */
176         public void dump (PrintStream p, String indent) {
177             p.println (indent +"<"+XMLPackager.ISOFIELD_TAG + " " +
178                 XMLPackager.ID_ATTR +"=\"" +fieldNumber +"\" "+
179                 XMLPackager.VALUE_ATTR +"=\"" +this.toString() + "\" " +
180                 XMLPackager.TYPE_ATTR +"=\"" + XMLPackager.TYPE_BINARY + "\"/>"
181             );
182         }
183         public String toString() {
184             return ISOUtil.hexString(value);
185         }
186         public void writeExternal (ObjectOutput out) throws IOException {
187             out.writeShort (fieldNumber);
188             out.writeShort (value.length);
189             out.write (value);
190         }
191         public void readExternal  (ObjectInput in) 
192             throws IOException, ClassNotFoundException
193         {
194             fieldNumber = in.readShort ();
195             value = new byte[in.readShort()];
196             in.readFully (value);
197         }
198     }
