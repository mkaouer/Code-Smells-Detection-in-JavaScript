8     package org.jpos.q2.qbean;
9     
10     import org.jdom.Element;
11     import org.jpos.q2.QBeanSupport;
12     import org.python.util.PythonInterpreter;
13     
14     /**
15      * Jython Interpreter QBean.
16      * 
17      * @author Alwyn Schoeman
18      * @version $Revision$ $Date$
19      *
20      */
21     public class Jython extends QBeanSupport implements Runnable {
22         PythonInterpreter jython;
23         public void initService() {
24             jython = new PythonInterpreter();
25         }
26         
27         public void startService() {
28             new Thread(this).start();
29         }
30         
31         public void run() {
32             Element config = getPersist();
33             jython.exec(config.getText());
34         }
35     }
