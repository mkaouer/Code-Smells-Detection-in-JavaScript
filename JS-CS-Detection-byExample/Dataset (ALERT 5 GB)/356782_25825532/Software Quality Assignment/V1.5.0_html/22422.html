138         public void start () 
139             throws MalformedObjectNameException,
140                    InstanceAlreadyExistsException,
141                    NotCompliantMBeanException,
142                    MBeanRegistrationException
143         {
144             /*
145              * The following code determines whether a MBeanServer exists already.
146              * If so then the first one in the list is used.  I have not yet find a way to
147              * interrogate the server for information other than MBeans so to pick a
148              * specific one would be difficult.
149              */
150             ArrayList mbeanServerList = MBeanServerFactory.findMBeanServer(null);
151             if (mbeanServerList.size() == 0) {
152                 server  = MBeanServerFactory.createMBeanServer (JMX_NAME);
153             } else {
154                 server = (MBeanServer) mbeanServerList.get(0);
155             }
156             ObjectName loaderName = new ObjectName (Q2_CLASS_LOADER);
157             try {
158                 loader = new QClassLoader (server, libDir, loaderName);
159                 server.registerMBean (loader, loaderName);
160                 loader = loader.scan();
161             } catch (Throwable t) {
162                 log.error ("initial-scan", t);
163             }
164             factory = new QFactory (loaderName, this);
165             initSystemLogger ();
166             addShutdownHook ();
167             q2Thread = Thread.currentThread ();
168             q2Thread.setContextClassLoader (loader);
169             while (!shutdown) {
170                 try {
171                     loader = loader.scan ();
172                     scan ();
173                     deploy ();
174                     checkModified ();
175                     relax (SCAN_INTERVAL);
176                 } catch (Throwable t) {
177                     log.error ("start", t);
178                     relax ();
179                 }
180             }
181             q2Thread = null;
182             undeploy ();
183             try {
184                 server.unregisterMBean (loaderName);
185             } catch (InstanceNotFoundException e) {
186                 log.error (e);
187             }
188             if (!shuttingDown)
189                 System.exit (0);
190         }
191         public void shutdown () {
192             shutdown = true;
193             q2Thread.interrupt ();
194         }

210         private void scan () {
211             File file[] = deployDir.listFiles (this);
212             // Arrays.sort (file); --apr not required - we use TreeMap
213             for (int i=0; i<file.length; i++) {
214                 register (file[i]);
215             }
216         }

218         private void deploy () {
219             List startList = new ArrayList ();
220             Iterator iter = dirMap.entrySet().iterator();
221             try {
222                 while (iter.hasNext()) {
223                     Map.Entry entry = (Map.Entry) iter.next();
224                     File   f        = (File)   entry.getKey ();
225                     QEntry qentry   = (QEntry) entry.getValue ();
226                     long deployed   = qentry.getDeployed ();
227                     if (deployed == 0) {
228                         if (deploy (f)) {
229                             if (qentry.isQBean ())
230                                 startList.add (qentry.getInstance());
231                             qentry.setDeployed (f.lastModified ());
232                         } else {
233                             // deploy failed, clean up.
234                             iter.remove();
235                         }
236                     } else if (deployed != f.lastModified ()) {
237                         undeploy (f);
238                         iter.remove ();
239                     }
240                 }
241                 iter = startList.iterator();
242                 while (iter.hasNext ()) {
243                     start ((ObjectInstance) iter.next ());
244                 }
245             }
246             catch (Exception e){
247                 log.warn ("deploy", e);
248             }
249         }

251         private void undeploy () {
252             Object[] set = dirMap.entrySet().toArray ();
253             int l = set.length;
254     
255             while (l-- > 0) {
256                 Map.Entry entry = (Map.Entry) set[l];
257                 File   f  = (File) entry.getKey ();
258                 undeploy (f);
259             }
260         }

262         private void addShutdownHook () {
263             Runtime.getRuntime().addShutdownHook (
264                 new Thread () {
265                     public void run () {
266                         shuttingDown = true;
267                         shutdown = true;
268                         log.info ("shutting down");
269                         if (q2Thread != null) {
270                             try {
271                                 q2Thread.join (SHUTDOWN_TIMEOUT);
272                             } catch (InterruptedException e) { }
273                         }
274                     }
275                 }
276             );
277         }

279         private void checkModified () {
280             Iterator iter = dirMap.entrySet().iterator();
281             while (iter.hasNext()) {
282                 Map.Entry entry = (Map.Entry) iter.next();
283                 File   f        = (File)   entry.getKey ();
284                 QEntry qentry   = (QEntry) entry.getValue ();
285                 if (qentry.isQBean() && qentry.isQPersist()) {
286                     ObjectName name = qentry.getObjectName ();
287                     if (getState (name) == QBean.STARTED && isModified (name)) {
288                         qentry.setDeployed (persist (f, name));
289                     }
290                 }
291             }
292         }

294         private int getState (ObjectName name) {
295             int status = -1;
296             if (name != null) {
297                 try {
298                     status = (
299                         (Integer) server.getAttribute (name, "State")
300                     ).intValue();
301                 } catch (Exception e) {
302                     log.warn ("getState", e);
303                 }
304             }
305             return status;
306         }
307         private boolean isModified (ObjectName name) {
308             boolean modified = false;
309             if (name != null) {
310                 try {
311                     modified = (
312                         (Boolean) server.getAttribute (name, "Modified")
313                     ).booleanValue();
314                 } catch (Exception e) {
315                     // Okay to fail
316                 }
317             }
318             return modified;
319         }
320         private long persist (File f, ObjectName name) {
321             long deployed = f.lastModified ();
322             try {
323                 Element e = (Element) server.getAttribute (name, "Persist");
324                 if (e != null) {
325                     XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
326                     Document doc = new Document ();
327                     e.detach();
328                     doc.setRootElement (e);
329                     File tmp = new File (f.getAbsolutePath () + ".tmp");
330                     FileWriter writer = new FileWriter (tmp);
331                     out.output (doc, writer);
332                     writer.close ();
333                     f.delete();
334                     tmp.renameTo (f);
335                     deployed = f.lastModified ();
336                 }
337             } catch (Exception ex) {
338                 log.warn ("persist", ex);
339             }
340             return deployed;
341         }

343         private void undeploy (File f) {
344             QEntry qentry = (QEntry) dirMap.get (f);
345             try {
346                 if (log != null)
347                     log.trace ("undeploying:" + f.getName());
348     
349                 Object obj      = qentry.getObject ();
350                 ObjectName name = qentry.getObjectName ();
351                 factory.destroyQBean (this, name, obj);
352                 if (log != null)
353                     log.info ("undeployed:" + f.getName());
354             } catch (Exception e) {
355                 getLog().warn ("undeploy", e);
356             }
357         }

364         private boolean deploy (File f) {
365             try {
366                 if (log != null)
367                     log.info ("deploy:" + f.getName());
368                 QEntry qentry = (QEntry) dirMap.get (f);
369                 SAXBuilder builder = new SAXBuilder ();
370                 Document doc = decrypt (builder.build (f));
371     
372                 Object obj = factory.instantiate (this, doc.getRootElement ());
373                 qentry.setObject (obj);
374     
375                 ObjectInstance instance = factory.createQBean (
376                     this, doc.getRootElement(), obj
377                 );
378                 qentry.setInstance (instance);
379             } 
380             catch (InstanceAlreadyExistsException e) {
381                /*
382                 * Ok, the file we tried to deploy, holds an object
383                 *  that already has been deployed.
384                 *  
385                 * Rename it out of the way.
386                 * 
387                 */
388                 tidyFileAway(f,DUPLICATE_EXTENSION);
389                 getLog().warn ("deploy", e);
390                 return false;
391             }
392             catch (Exception e) {
393                 getLog().warn ("deploy", e);
394                 tidyFileAway(f,ERROR_EXTENSION);
395                 // This will also save deploy error repeats...
396                 return false;
397             } 
398             return true ;
399         }

416         private void initSystemLogger () {
417             File loggerConfig = new File (deployDir, LOGGER_CONFIG);
418             if (loggerConfig.canRead()) {
419                 hasSystemLogger = true;
420                 try {
421                     register (loggerConfig);
422                     deploy ();
423                 } catch (Exception e) {
424                     getLog().warn ("init-system-logger", e);
425                 }
426             }
427             getLog().info ("Q2 started, deployDir="+deployDir.getAbsolutePath());
428         }

442         private void parseCmdLine (String[] args) {
443             CommandLineParser parser = new PosixParser ();
444     
445             Options options = new Options ();
446             options.addOption ("v","version", false, "Version of Q2");
447             options.addOption ("d","deploydir", true, "Deployment directory");
448             options.addOption ("h","help", false, "Usage information");
449             options.addOption ("c","config", true, "Configuration bundle");
450             options.addOption ("e","encrypt", true, "Encrypt configuration bundle");
451     
452             try {
453                 CommandLine line = parser.parse (options, args);
454                 if (line.hasOption ("v")) {
455                     System.out.println ("Q2 version: " + Q2_VERSION);
456                     System.exit (0);
457                 } 
458                 if (line.hasOption ("h")) {
459                     HelpFormatter helpFormatter = new HelpFormatter ();
460                     helpFormatter.printHelp ("Q2", options);
461                     System.exit (0);
462                 } 
463                 String dir = DEFAULT_DEPLOY_DIR;
464                 if (line.hasOption ("d")) {
465                     dir = line.getOptionValue ("d");
466                 }
467                 this.deployDir  = new File (dir);
468                 if (line.hasOption ("c"))
469                     deployBundle (new File (line.getOptionValue ("c")), false);
470                 if (line.hasOption ("e"))
471                     deployBundle (new File (line.getOptionValue ("e")), true);
472             } catch (Exception e) {
473                 e.printStackTrace ();
474                 System.exit (1);
475             }
476         }
477         private void deployBundle (File bundle, boolean encrypt) 
478             throws JDOMException, IOException, 
479                     ISOException, GeneralSecurityException
480         {
481             SAXBuilder builder = new SAXBuilder ();
482             Document doc = builder.build (bundle);
483             Iterator iter = doc.getRootElement().getChildren ().iterator ();
484             for (int i=0; iter.hasNext (); i += 5) {
485                 deployElement ((Element) iter.next (), i, encrypt);
486             }
487         }
488         private void deployElement (Element e, int i, boolean encrypt) 
489             throws ISOException, IOException, GeneralSecurityException
490         {
491             e = ((Element) e.clone ());
492     
493             XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
494             Document doc = new Document ();
495             doc.setRootElement (e);
496             File f = new File (deployDir, e.getName ());
497             File qbean = new File (
498                 deployDir, ISOUtil.zeropad (Integer.toString (i),3) + "_" 
499                 + e.getName () + ".xml"
500             );
501             FileWriter writer = new FileWriter (qbean);
502             if (encrypt)
503                 doc = encrypt (doc);
504             out.output (doc, writer);
505             writer.close ();
506         }

508         private byte[] dodes (byte[] data, int mode) 
509            throws GeneralSecurityException
510         {
511             Cipher cipher = Cipher.getInstance ("DES");
512             cipher.init (mode, new SecretKeySpec(getKey(), "DES"));
513             return cipher.doFinal (data);
514         }
515         protected byte[] getKey() {
516             return "CAFEBABE".getBytes();
517         }
518         protected Document encrypt (Document doc)
519             throws GeneralSecurityException, IOException
520         {
521             ByteArrayOutputStream os = new ByteArrayOutputStream ();
522             OutputStreamWriter writer = new OutputStreamWriter (os);
523             XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
524             out.output (doc, writer);
525             writer.close ();
526     
527             byte[] crypt = dodes (os.toByteArray(), Cipher.ENCRYPT_MODE);
528     
529             Document secureDoc = new Document ();
530             Element root = new Element (PROTECTED_QBEAN);
531             secureDoc.setRootElement (root);
532             Element secureData = new Element ("data");
533             root.addContent (secureData);
534     
535             secureData.setText (
536                 ISOUtil.hexString (crypt)
537             );
538             return secureDoc;
539         }
540     
541         protected Document decrypt (Document doc) 
542             throws GeneralSecurityException, IOException, JDOMException
543         {
544             Element root = doc.getRootElement ();
545             if (PROTECTED_QBEAN.equals (root.getName ())) {
546                 Element data = root.getChild ("data");
547                 if (data != null) {
548                     ByteArrayInputStream is = new ByteArrayInputStream (
549                         dodes (
550                             ISOUtil.hex2byte (data.getTextTrim()),
551                             Cipher.DECRYPT_MODE)
552                     );
553                     SAXBuilder builder = new SAXBuilder ();
554                     doc = builder.build (is);
555                 }
556             }
557             return doc;
558         }

560         private void tidyFileAway (File f, String extension) {
561             File rename = new File(f.getAbsolutePath()+"."+extension);
562             while (rename.exists()){
563                 rename = new File(rename.getAbsolutePath()+"."+extension);
564             }
565             getLog().warn("Tidying "+f.getAbsolutePath()+" out of the way, by adding ."+extension,"It will be called: "+rename.getAbsolutePath()+" see log above for detail of problem.");
566             f.renameTo(rename);
567         }
