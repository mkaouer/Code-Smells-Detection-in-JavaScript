7     package org.jpos.space;

9     import java.io.Serializable;
10     import java.rmi.RemoteException;
11     import java.rmi.registry.LocateRegistry;
12     import java.rmi.registry.Registry;
13     import java.rmi.server.ExportException;
14     import java.rmi.server.UnicastRemoteObject;
15     import java.rmi.server.RemoteRef;
16     import java.rmi.server.RemoteStub;
17     import java.rmi.server.RemoteObject;
18     import java.util.Set;
19     
20     import javax.naming.InitialContext;
21     import javax.naming.NamingException;
22     
23     import org.jpos.core.Configuration;
24     import org.jpos.core.ConfigurationException;
25     import org.jpos.core.ReConfigurable;

27     /**
28      * RMI Space Proxy 
29      * @author Alejandro Revilla
30      * @author Niclas Hedhman
31      * @version $Revision$ $Date$
32      * @since 1.4.9
33      */
34     public class SpaceProxy implements RemoteSpace, ReConfigurable {
35         Space sp;
36         Configuration cfg;
37         private RemoteRef ref;
38         private RemoteStub stub;
39         public SpaceProxy () throws RemoteException {
40             super();
41             sp = SpaceFactory.getSpace ();
42             startService ();
43         }
44         public SpaceProxy (String spaceUri) throws RemoteException {
45             super ();
46             sp = SpaceFactory.getSpace (spaceUri);
47             startService ();
48         }
49     
50         private void startService () throws RemoteException 
51         {
52             try {
53                 LocateRegistry.createRegistry (Registry.REGISTRY_PORT);
54             } catch (ExportException e) {
55                 // registry already exists
56             }
57             stub = UnicastRemoteObject.exportObject (this);
58             ref  = stub.getRef();
59         }
60     
61         public void out (Serializable key, Serializable value) 
62             throws RemoteException
63         {
64             sp.out (key, value);
65         }
66         public void out (Serializable key, Serializable value, long timeout)
67             throws RemoteException
68         {
69             sp.out (key, value, timeout);
70         }
71         public Serializable in (Serializable key)
72             throws RemoteException
73         {
74             return (Serializable) sp.in (key);
75         }
76         public Serializable rd  (Serializable key)
77             throws RemoteException
78         {
79             return (Serializable) sp.rd (key);
80         }
81         public Serializable in  (Serializable key, long timeout)
82             throws RemoteException
83         {
84             return (Serializable) sp.in (key, timeout);
85         }
86         public Serializable rd  (Serializable key, long timeout)
87             throws RemoteException
88         {
89             return (Serializable) sp.rd (key, timeout);
90         }
91         public Serializable inp (Serializable key)
92             throws RemoteException
93         {
94             return (Serializable) sp.inp (key);
95         }
96         public Serializable rdp (Serializable key)
97             throws RemoteException
98         {
99             return (Serializable) sp.rdp (key);
100         }
101         public void shutdown() {
102             try {
103                 if (UnicastRemoteObject.unexportObject (this, false)) 
104                     return;
105                 Thread.sleep (5000);
106                 UnicastRemoteObject.unexportObject (this, true);
107             } catch (Exception e) {
108                 // nothing to do .. we're shutting down ...
109             }
110         }
111         public void setConfiguration (Configuration cfg) 
112             throws ConfigurationException
113         {
114             this.cfg = cfg;
115             try {
116                 InitialContext ctx = new InitialContext ();
117                 ctx.rebind (cfg.get ("name"), stub);
118             } catch (NamingException e) {
119                 throw new ConfigurationException (e);
120             }
121         }
122         public Set getKeySet () {
123             if (sp instanceof LocalSpace) 
124                 return ((LocalSpace)sp).getKeySet ();
125             else
126                 return null;
127         }
128         public String toString() {
129             if (ref == null)
130                 return getClass().getName() + "[<unexported>]";
131             else
132                 return getClass().getName() + "[" + ref.remoteToString() + "]";
133         }
134         public int hashCode() {
135             if (ref == null )
136                 return super.hashCode();
137             else
138                 return ref.remoteHashCode();
139         }
140         public boolean equals (Object obj) {
141             if (obj instanceof RemoteObject) {
142                 if (ref == null) {
143                     return obj == this;
144                 } else {
145                     RemoteRef otherRef = ((RemoteObject)obj).getRef();
146                     return ref.remoteEquals (otherRef);
147                 }
148             } else if (obj != null) {
149                 return obj.equals (this);
150             } else {
151                 return false;
152             }
153         }
154     }
