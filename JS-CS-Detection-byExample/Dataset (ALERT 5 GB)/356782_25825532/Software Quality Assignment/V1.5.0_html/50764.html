77     /**
78      * Reference implementation of a user task
79      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
80      * @version $Revision$ $Date$
81      */
82     public class Sender 
83         extends SimpleLogSource
84         implements Runnable, ReConfigurable, SenderMBean
85     {
86         ISOMUX mux;
87         File message;
88         long initialDelay;
89         long delay;
90         int  waitForResponse;
91         long lastRoundTrip = -1;
92         Logger logger;
93         String realm;
94         ISOPackager packager;
95         Sequencer seq;
96         Configuration cfg;
97         boolean disabled = false;
98         String lastResponseCode = null;
99         Thread thisThread;
100     
101         public Sender() {
102             super();
103         }
104     
105         public void setConfiguration (Configuration cfg) 
106             throws ConfigurationException
107         {
108             this.cfg = cfg;
109             try {
110                 mux = ISOMUX.getMUX (cfg.get ("mux"));
111                 message = new File (cfg.get ("message"));
112                 initialDelay    = cfg.getLong ("initial-delay");
113                 waitForResponse = cfg.getInt  ("wait-for-response");
114                 delay           = cfg.getLong ("delay");
115                 packager        = new XMLPackager();
116     
117                 String seqName  = cfg.get ("sequencer", null);
118                 if (seqName != null) {
119                     seq = (Sequencer) NameRegistrar.get (
120                         "sequencer."+cfg.get("sequencer")
121                     );
122                 } else if (seq == null) {
123                     seq = new VolatileSequencer();
124                 }
125             } catch (NameRegistrar.NotFoundException e) {
126                 throw new ConfigurationException (e);
127             } catch (ISOException e) {
128                 throw new ConfigurationException (e);
129             }
130         }
131     
132         private void applyProps (ISOMsg m) throws ISOException {
133             for (int i=0; i<128; i++) {
134                 if (m.hasField(i) && m.getValue(i) instanceof String) {
135                     String value = (String) m.getValue(i);
136                     if (value.equalsIgnoreCase ("$date") )
137                         m.set (new ISOField (i, ISODate.getDateTime(new Date())));
138                     else if ((value.toLowerCase().startsWith ("$date") ) && (value.indexOf("GMT") > 0)) {
139                         String zoneID = value.substring(value.indexOf("GMT"));
140                         m.set (new ISOField (i, ISODate.getDateTime(new Date(), TimeZone.getTimeZone(zoneID))));
141                     }                    
142                     else if (value.charAt (0) == '$')
143                         m.set (new ISOField (i,
144                           ISOUtil.zeropad (
145                             Integer.toString(seq.get (value.substring(1))),6)
146                           )
147                         );
148                     else if (value.charAt (0) == '=') {
149                         String p = cfg.get (value.substring(1), null);
150                         if (p != null)
151                             m.set (new ISOField (i, p));
152                     }
153                 }
154             }
155         }
156     
157         public void run () {
158             thisThread = Thread.currentThread ();
159             if (initialDelay > 0)
160                 try {
161                     Thread.sleep (initialDelay);
162                 } catch (InterruptedException e) { }
163     
164             do {
165                 LogEvent evt = new LogEvent (this, "sender-run");
166                 try {
167                     sendOne (evt);
168                 } catch (Throwable e) {
169                     evt.addMessage (e);
170                 } finally  {
171                     Logger.log (evt);
172                 }
173                 if (delay > 0)
174                     try {
175                         Thread.sleep (delay);
176                     } catch (InterruptedException e) { }
177             } while (!cfg.getBoolean ("one-shot"));
178         }
179     
180         public void sendOne(LogEvent evt) throws IOException {
181             FileInputStream fis = new FileInputStream (message);
182             try {
183                 byte[] b = new byte[fis.available()];
184                 fis.read (b);
185                 ISOMsg m = new ISOMsg ();
186                 m.setPackager (packager);
187                 m.unpack (b);
188                 applyProps (m);
189                 evt.addMessage (m);
190                 ISORequest req = new ISORequest (m);
191                 long start = System.currentTimeMillis ();
192                 mux.queue (req);
193                 if (waitForResponse > 0) {
194                     ISOMsg resp = req.getResponse (waitForResponse);
195                     if (resp != null) {
196                         lastRoundTrip = System.currentTimeMillis () - start;
197     
198                         StringBuffer sb = new StringBuffer (resp.getMTI());
199                         if (resp.hasField (39)) {
200                             sb.append (' ');
201                             sb.append (resp.getString (39));
202                         }
203                         if (resp.hasField (38)) {
204                             sb.append (' ');
205                             sb.append (resp.getString (38));
206                         }
207                         lastResponseCode = sb.toString ();
208     
209                         evt.addMessage (resp);
210                     }
211                     else {
212                         lastRoundTrip = -1;
213                         lastResponseCode = "<timed-out>";
214                     }
215                 }
216             } catch (ISOException e) {
217                 evt.addMessage (e);
218             } finally {
219                 fis.close();
220             }
221         }
222     
223         public void forceSend () {
224             LogEvent evt = new LogEvent (this, "force-send");
225             try {
226                 if (!disabled)
227                     sendOne (evt);
228             } catch (IOException e) {
229                 evt.addMessage (e);
230             }
231             Logger.log (evt);
232         }
233         public long getLastRoundTrip () {
234             return lastRoundTrip;
235         }
236         public void setDisabled (boolean disabled) {
237             if (disabled != this.disabled) {
238                 Logger.log (
239                     new LogEvent (this, "sender", 
240                         "sender task has been " + 
241                             (disabled ? "disabled" : "enabled")
242                             )
243                 );
244             }
245             this.disabled = disabled;
246         }
247         public boolean isDisabled () {
248             return disabled;
249         }
250         public void setDelay (long delay) {
251             this.delay = delay;
252             thisThread.interrupt ();
253         }
254         public long getDelay () {
255             return delay;
256         }
257         public String getLastResponseCode () {
258             return lastResponseCode;
259         }
260     }
