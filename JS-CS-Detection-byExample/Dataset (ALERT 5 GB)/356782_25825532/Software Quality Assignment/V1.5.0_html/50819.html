76         public void init () {
77             if (state == -1) {
78                 setModified (false);
79                 try {
80                     initService();
81                     state = QBean.STOPPED;
82                 } catch (Throwable t) {
83                     log.warn ("init", t);
84                 }
85             }
86         }
87         public synchronized void start() {
88             if (state != QBean.DESTROYED && 
89                 state != QBean.STOPPED   && 
90                 state != QBean.FAILED)
91                return;
92     
93             this.state = QBean.STARTING;
94     
95             try {
96                startService();
97             } catch (Throwable t) {
98                state = QBean.FAILED;
99                log.warn ("start", t);
100                return;
101             }
102             state = QBean.STARTED;
103         }
104         public synchronized void stop () {
105             if (state != QBean.STARTED)
106                return;
107             state = QBean.STOPPING;
108             try {
109                stopService();
110             } catch (Throwable t) {
111                state = QBean.FAILED;
112                log.warn ("stop", t);
113                return;
114             }
115             state = QBean.STOPPED;
116         }
117         public void destroy () {
118             if (state == QBean.DESTROYED)
119                return;
120             if (state != QBean.STOPPED)
121                stop();
122     
123             try {
124                destroyService();
125             }
126             catch (Throwable t) {
127                log.warn ("destroy", t);
128             }
129             state = QBean.DESTROYED;
130         }
131         public void shutdownQ2 () {
132             getServer().shutdown ();
133         }
134         public int getState () {
135             return state;
136         }

138             return server.getLoader().getURLs();
139         }

168         protected void destroyService() throws Exception {}

170         protected Element createElement (String name, Class mbeanClass) {
171             Element e = new Element (name);
172             Element classPath = persist != null ?
173                 persist.getChild ("classpath") : null;
174             if (classPath != null)
175                 e.addContent (classPath);
176             e.setAttribute ("class", getClass().getName());
177             if (!e.getName().equals (getName ()))
178                 e.setAttribute ("name", getName());
179             String loggerName = getLogger();
180             if (loggerName != null)
181                 e.setAttribute ("logger", loggerName);
182     
183             try {
184                 BeanInfo info = Introspector.getBeanInfo (mbeanClass);
185                 PropertyDescriptor[] desc = info.getPropertyDescriptors();
186                 for (int i=0; i<desc.length; i++) {
187                     if (desc[i].getWriteMethod() != null) {
188                         Method read = desc[i].getReadMethod();
189                         Object obj  = read.invoke (this, new Object[] { } );
190                         String type = read.getReturnType().getName();
191                         if ("java.lang.String".equals (type))
192                             type = null;
193     
194                         addAttr (e, desc[i].getName(), obj, type);
195                     }
196                 }
197             } catch (Exception ex) {
198                 log.warn ("get-persist", ex);
199             } 
200             return e;
201         }
202         protected void addAttr (Element e, String name, Object obj, String type) {
203             String value = obj == null ? "null" : obj.toString();
204             Element attr = new Element ("attr");
205             attr.setAttribute ("name", name);
206             if (type != null)
207                 attr.setAttribute ("type", type);
208             attr.setText (value);
209             e.addContent (attr);
210         }
211         protected Iterator getAttrs () {
212             return getPersist().getChildren ("attr").iterator();
213         }
214         protected Iterator getAttrs (String parent) {
215             return getPersist().getChild (parent).
216                 getChildren ("attr").iterator();
217         }
218         protected void setAttr (Iterator attrs, String name, Object obj) {
219             String value = obj == null ? "null" : obj.toString ();
220             while (attrs.hasNext ()) {
221                 Element e = (Element) attrs.next ();
222                 if (name.equals (e.getAttributeValue ("name")))  {
223                     e.setText (value);
224                     break;
225                 }
226             }
227         }
228         protected Iterator getProperties (String parent) {
229             return getPersist().getChild (parent).
230                    getChildren ("property").iterator();
231         }
232         protected void setProperty (Iterator props, String name, String value) {
233             while (props.hasNext()) {
234                 Element e = (Element) props.next();
235                 if (name.equals (e.getAttributeValue ("name"))) {
236                     e.setAttribute ("value", value);
237                     break;
238                 }
239             }
240         }
241         protected String getProperty (Iterator props, String name) {
242             while (props.hasNext()) {
243                 Element e = (Element) props.next();
244                 if (name.equals (e.getAttributeValue ("name"))) {
245                     return e.getAttributeValue ("value");
246                 }
247             }
248             return null;
249         }
