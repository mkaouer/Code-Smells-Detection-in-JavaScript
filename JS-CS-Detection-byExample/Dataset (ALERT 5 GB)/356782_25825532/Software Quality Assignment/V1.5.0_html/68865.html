50     package org.jpos.util;
51     
52     import java.io.PrintStream;
53     
54     /**
55      * Periodically dumps Thread and memory usage
56      * @author apr@cs.com.uy
57      * @version $Id$
58      * @see Logger
59      */
60     public class SystemMonitor implements Runnable, LogSource, Loggeable
61     {
62         private Logger logger = null;
63         private String realm  = null;
64         private int sleepTime = 0;
65         private int delay     = 0;
66         private Thread thread = null;
67     
68         /**
69          * noargs constructor
70          */
71         public SystemMonitor () {
72             super();
73         }
74         /**
75          * @param sleepTime sleep
76          * @param logger current logger
77          * @param realm  instance realm
78          */
79         public SystemMonitor (int sleepTime, Logger logger, String realm) {
80             setLogger (logger, realm);
81             this.sleepTime = sleepTime;
82             startThread();
83         }
84         
85         private void startThread() {
86             if (thread != null)
87                 thread.interrupt();
88             else if (sleepTime > 0) {
89                 thread = new Thread(this);
90                 thread.setName ("SystemMonitor");
91                 thread.setPriority (Thread.MIN_PRIORITY);
92                 thread.start();
93             }
94         }
95     
96         /**
97          * @param sleepTime new sleepTime;
98          */
99         public void setSleepTime (int sleepTime) {
100             this.sleepTime = sleepTime;
101             startThread();
102         }
103     
104         void dumpThreads (ThreadGroup g, PrintStream p, String indent) {
105             Thread[] list = new Thread[g.activeCount()+5];
106             int nthreads = g.enumerate(list);
107             for (int i=0; i<nthreads; i++) 
108                 p.println (indent + list[i]);
109         }
110     
111         public void showThreadGroup (ThreadGroup g, PrintStream p, String indent) {
112             if (g.getParent() != null)
113                 showThreadGroup (g.getParent(), p, indent + "  ");
114             else
115                 dumpThreads (g, p, indent + "    ");
116         }
117     
118         public void run() {
119             for (;;) {
120                 Logger.log (new LogEvent (this, "SystemMonitor", this));
121                 try {
122                     long expected = System.currentTimeMillis() + sleepTime;
123                     Thread.sleep (sleepTime);
124                     delay = (int) (System.currentTimeMillis() - expected);
125                 } catch (InterruptedException e) { }
126             }
127         }
128         public void dump (PrintStream p, String indent) {
129             String newIndent = indent + "  ";
130             Runtime r = Runtime.getRuntime();
131             p.println (indent + "<memory>");
132             p.println (newIndent+" freeMemory="+r.freeMemory());
133             p.println (newIndent+"totalMemory="+r.totalMemory());
134             p.println (newIndent+"inUseMemory="+(r.totalMemory()-r.freeMemory()));
135             p.println (indent + "</memory>");
136             p.println (indent + "<threads>");
137             p.println (newIndent+"      delay="+delay+" ms");
138             p.println (newIndent+"    threads="+Thread.activeCount());
139             showThreadGroup (Thread.currentThread().getThreadGroup(), p, newIndent);
140             p.println (indent + "</threads>");
141             NameRegistrar.getInstance().dump (p, indent);
142         }
143         public void setLogger (Logger logger, String realm) {
144             this.logger = logger;
145             this.realm  = realm;
146         }
147         public String getRealm () {
148             return realm;
149         }
150         public Logger getLogger() {
151             return logger;
152         }
153     }
