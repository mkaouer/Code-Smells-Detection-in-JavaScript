69         public ISOServer(int port, ServerChannel clientSide, ThreadPool pool) {
70             super();
71             this.port = port;
72             this.clientSideChannel = clientSide;
73             this.clientSideChannelClass = clientSide.getClass();
74             this.clientPackager = clientSide.getPackager();
75             if (clientSide instanceof FilteredChannel) {
76                 FilteredChannel fc = (FilteredChannel) clientSide;
77                 this.clientOutgoingFilters = fc.getOutgoingFilters();
78                 this.clientIncomingFilters = fc.getIncomingFilters();
79             }
80             this.pool = (pool == null) ?  
81                 new ThreadPool (1, DEFAULT_MAX_THREADS) : pool;
82             listeners = new Vector();
83             name = "";
84             channels = new Stack();
85             cnt = new int[SIZEOF_CNT];
86         }

92         protected class Session implements Runnable, LogSource {
93             ServerChannel channel;
94             String realm;
95             protected Session(ServerChannel channel) {
96                 this.channel = channel;
97                 realm = ISOServer.this.getRealm() + ".session";
98             }
99             public void run() {
100                 if (channel instanceof BaseChannel) {
101                     LogEvent ev = new LogEvent (this, "session-start");
102                     Socket socket = ((BaseChannel)channel).getSocket ();
103                     realm = realm + socket.getInetAddress();
104                     try {
105                         checkPermission (socket, ev);
106                     } catch (ISOException e) {
107                         try {
108                             int delay = 1000 + new Random().nextInt (4000);
109                             ev.addMessage (e.getMessage());
110                             ev.addMessage ("delay=" + delay);
111                             ISOUtil.sleep (delay);
112                             socket.close ();
113                         } catch (IOException ioe) {
114                             ev.addMessage (ioe);
115                         }
116                         return;
117                     } finally {
118                         Logger.log (ev);
119                     }
120                 }
121                 try {
122                     for (;;) {
123                         ISOMsg m = channel.receive();
124                         Iterator iter = listeners.iterator();
125                         while (iter.hasNext())
126                             if (((ISORequestListener)iter.next()).process
127                                 (channel, m)) 
128                                 break;
129                     }
130                 } catch (EOFException e) {
131                     Logger.log (new LogEvent (this, "session-warning", "<eof/>"));
132                 } catch (SocketException e) {
133                     if (!shutdown) 
134                         Logger.log (new LogEvent (this, "session-warning", e));
135                 } catch (InterruptedIOException e) {
136                     // nothing to log
137                 } catch (Throwable e) { 
138                     Logger.log (new LogEvent (this, "session-error", e));
139                 } 
140                 try {
141                     channel.disconnect();
142                 } catch (IOException ex) { 
143                     Logger.log (new LogEvent (this, "session-error", ex));
144                 }
145                 Logger.log (new LogEvent (this, "session-end"));
146             }
147             public void setLogger (Logger logger, String realm) { 
148             }
149             public String getRealm () {
150                 return realm;
151             }
152             public Logger getLogger() {
153                 return ISOServer.this.getLogger();
154             }
155             public void checkPermission (Socket socket, LogEvent evt) 
156                 throws ISOException 
157             {
158                 if (allow != null && allow.length > 0) {
159                     String ip = socket.getInetAddress().getHostAddress ();
160                     for (int i=0; i<allow.length; i++) {
161                         if (ip.equals (allow[i])) {
162                             evt.addMessage ("access granted, ip=" + ip);
163                             return;
164                         }
165                     }
166                     throw new ISOException ("access denied, ip=" + ip);
167                 }
168             }
169         }

189         public void shutdown () {
190             shutdown = true;
191             new Thread () {
192                 public void run () {
193                     shutdownServer ();
194                     shutdownChannels ();
195                 }
196             }.start();

198         private void shutdownServer () {
199             try {
200                 if (serverSocket != null)
201                     serverSocket.close ();
202             } catch (IOException e) {
203                 Logger.log (new LogEvent (this, "shutdown", e));
204             }
205         }
206         private void shutdownChannels () {
207             Iterator iter = channels.iterator();
208             while (iter.hasNext()) {
209                 ISOChannel c = (ISOChannel) ((WeakReference) iter.next()).get ();
210                 if (c != null) {
211                     try {
212                         c.disconnect ();
213                     } catch (IOException e) {
214                         Logger.log (new LogEvent (this, "shutdown", e));
215                     }
216                 }
217             }
218         }
219         private void purgeChannels () {
220             Iterator iter = channels.iterator();
221             while (iter.hasNext()) {
222                 WeakReference ref = (WeakReference) iter.next();
223                 ISOChannel c = (ISOChannel) ref.get ();
224                 if (c == null || (!c.isConnected()))
225                     iter.remove ();
226             }
227         }

229         public void run() {
230             ServerChannel  channel;
231             while  (!shutdown) {
232                 try {
233                     serverSocket = socketFactory != null ?
234                             socketFactory.createServerSocket(port) :
235                             (new ServerSocket (port));
236                     
237                     Logger.log (new LogEvent (this, "iso-server", 
238                         "listening on port "+port));
239                     while (!shutdown) {
240                         try {
241                             channel = (ServerChannel) 
242                                 clientSideChannelClass.newInstance();
243                             channel.setPackager (clientPackager);
244                             if (channel instanceof LogSource) {
245                                 ((LogSource)channel) .
246                                     setLogger (getLogger(), realmChannel);
247                             }
248                             if (clientSideChannel instanceof BaseChannel) {
249                                 ((BaseChannel)channel).setHeader (
250                                     ((BaseChannel)clientSideChannel).getHeader());
251                                 ((BaseChannel)channel).setTimeout (
252                                     ((BaseChannel)clientSideChannel).getTimeout());
253                             }
254                             setFilters (channel);
255                             if (channel instanceof Observable)
256                                 ((Observable)channel).addObserver (this);
257                             for (int i=0; pool.getAvailableCount() <= 0; i++) {
258                                 ISOUtil.sleep (250);
259                                 if (i % 240 == 0) {
260                                     LogEvent evt = new LogEvent (this, "warn");
261                                     evt.addMessage (
262                                         "pool exahusted " + serverSocket.toString()
263                                     );
264                                     evt.addMessage (pool);
265                                     Logger.log (evt);
266                                 }
267                             }
268                             channel.accept (serverSocket);
269                             if ((cnt[CONNECT]++) % 100 == 0)
270                                 purgeChannels ();
271                             channels.push (new WeakReference (channel));
272                             setChanged ();
273                             notifyObservers (channel);
274                             pool.execute (createSession(channel));
275                         } catch (SocketException e) {
276                             if (!shutdown)
277                                 Logger.log (new LogEvent (this, "iso-server", e));
278                         } catch (IOException e) {
279                             Logger.log (new LogEvent (this, "iso-server", e));
280                             relax();
281                         } catch (InstantiationException e) {
282                             Logger.log (new LogEvent (this, "iso-server", e));
283                             relax();
284                         } catch (IllegalAccessException e) {
285                             Logger.log (new LogEvent (this, "iso-server", e));
286                             relax();
287                         }
288                     }
289                 } catch (Throwable e) {
290                     Logger.log (new LogEvent (this, "iso-server", e));
291                     relax();
292                 }
293             }
294         }

296         private void setFilters (ISOChannel channel) {
297             if (clientOutgoingFilters != null)
298                 ((FilteredChannel)channel) .
299                     setOutgoingFilters (clientOutgoingFilters);
300             if (clientIncomingFilters != null)
301                 ((FilteredChannel)channel) .
302                     setIncomingFilters (clientIncomingFilters);
303         }
304         private void relax() {
305             try {
306                 Thread.sleep (5000);
307             } catch (InterruptedException e) { }
308         }
