50     package org.jpos.iso.filter;
51     import org.jpos.core.ConfigurationException;
52     import org.jpos.core.NodeConfigurable;
53     import org.jpos.iso.ISOChannel;
54     import org.jpos.iso.ISOException;
55     import org.jpos.iso.ISOFilter;
56     import org.jpos.iso.ISOMsg;
57     import org.jpos.util.JepUtil;
58     import org.jpos.util.LogEvent;
59     import org.w3c.dom.Node;
60     import org.w3c.dom.NodeList;
61     
62     /**
63      * Using JEP to compute expressions
64      *
65      * @author <a href="mailto:tzymail@163.com">Zhiyu Tang
66      * @version $Revision$ $Date$
67      * @see org.jpos.iso.ISOFilter
68      */
69     public class CalculateFilter
70         implements ISOFilter,NodeConfigurable
71     {
72         String condition[];
73         String field[];
74         String statement[];
75         int condNum;
76         JepUtil jeputil;
77     
78         public CalculateFilter () {
79             super();
80             condition = null;
81             statement = null;
82             field = null;
83             condNum = 0;
84             jeputil = new JepUtil();
85         }
86     
87     
88        public void setConfiguration (Node node)
89             throws ConfigurationException
90         {
91             NodeList nodes = node.getChildNodes();
92             condNum = nodes.getLength();
93             int i, j = 0;
94             for( i=0 ; i < condNum; i++ )
95             {
96                 if (nodes.item(i).getNodeName().equals ("calculate"))
97                 j++;
98             }
99     
100             condition = new String[j];
101             field = new String[j];
102             statement = new String[j];
103             j = 0;
104             for( i=0 ; i < condNum; i++ )
105             {
106                 if (nodes.item(i).getNodeName().equals ("calculate")) {
107                     condition[j] = nodes.item(i).getAttributes().getNamedItem ("switch").getNodeValue();
108                     field[j] = nodes.item(i).getAttributes().getNamedItem ("field").getNodeValue();
109                     statement[j] = nodes.item(i).getAttributes().getNamedItem ("statement").getNodeValue();
110                     j++;
111                }
112             }
113             condNum = j;
114         }
115     
116        private void calculateStatement( LogEvent evt , ISOMsg m )
117             throws ISOException
118         {
119             int i = 0;
120             for( i = 0 ; i < condNum ; i++ )
121             {
122     
123                 if( jeputil.getResultBoolean( m , condition[i] ) )
124                  {
125                   String valueString = jeputil.getResult( m , statement[i] );
126                   evt.addMessage( "<calculate field=\""+field[i]+"\" value=\""+valueString+"\"/>" );
127                   m.set( Integer.valueOf( field[i]).intValue() , valueString );
128                  }
129             }
130         }
131     
132         public ISOMsg filter (ISOChannel channel, ISOMsg m, LogEvent evt)
133             throws VetoException
134         {
135             try {
136               calculateStatement( evt , m );
137             } catch (ISOException e) {
138                evt.addMessage (e);
139               throw new VetoException (e);
140             }
141             return m;
142         }
143     }
