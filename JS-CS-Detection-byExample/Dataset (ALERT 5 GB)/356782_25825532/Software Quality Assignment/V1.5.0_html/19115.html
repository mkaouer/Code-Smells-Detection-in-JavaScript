76     /**
77      * packs/unpacks ISOMsgs into XML representation
78      *
79      * @author apr@cs.com.uy
80      * @version $Id$
81      * @see ISOPackager
82      */
83     public class XMLPackager extends DefaultHandler
84                              implements ISOPackager, LogSource
85     {
86         protected Logger logger = null;
87         protected String realm = null;
88         private ByteArrayOutputStream out;
89         private PrintStream p;
90         private XMLReader reader = null;
91         private Stack stk;
92     
93         public static final String ISOMSG_TAG    = "isomsg";
94         public static final String ISOFIELD_TAG  = "field";
95         public static final String ID_ATTR       = "id";
96         public static final String VALUE_ATTR    = "value";
97         public static final String TYPE_ATTR     = "type";
98         public static final String TYPE_BINARY   = "binary";
99         public static final String TYPE_BITMAP   = "bitmap";
100     
101         public XMLPackager() throws ISOException {
102             super();
103             out = new ByteArrayOutputStream();
104             p   = new PrintStream(out);
105             stk = new Stack();
106             try {
107                 reader = XMLReaderFactory.createXMLReader(
108                     System.getProperty( "sax.parser",
109                                         "org.apache.xerces.parsers.SAXParser")
110                 );
111                 reader.setFeature ("http://xml.org/sax/features/validation",false);
112                 reader.setContentHandler(this);
113                 reader.setErrorHandler(this);
114             } catch (Exception e) {
115                 throw new ISOException (e.toString());
116             }
117         }
118         public byte[] pack (ISOComponent c) throws ISOException {
119             LogEvent evt = new LogEvent (this, "pack");
120             try {
121                 if (!(c instanceof ISOMsg))
122                     throw new ISOException ("cannot pack "+c.getClass());
123                 ISOMsg m = (ISOMsg) c;
124                 byte[] b;
125                 synchronized (this) {
126                     m.setDirection(0);  // avoid "direction=xxxxxx" in XML msg
127                     m.dump (p, "");
128                     b = out.toByteArray();
129                     out.reset();
130                 }
131                 if (logger != null)
132                     evt.addMessage (m);
133                 return b;
134             } catch (ISOException e) {
135                 evt.addMessage (e);
136                 throw e;
137             } finally {
138                 Logger.log(evt);
139             }
140         }
141     
142         public synchronized int unpack (ISOComponent c, byte[] b) 
143             throws ISOException
144         {
145             LogEvent evt = new LogEvent (this, "unpack");
146             try {
147                 if (!(c instanceof ISOMsg))
148                     throw new ISOException 
149                         ("Can't call packager on non Composite");
150     
151                 while (!stk.empty())    // purge from possible previous error
152                     stk.pop();
153     
154                 InputSource src = new InputSource (new ByteArrayInputStream(b));
155                 reader.parse (src);
156                 if (stk.empty())
157                     throw new ISOException ("error parsing");
158     
159                 ISOMsg m = (ISOMsg) c;
160                 m.merge ((ISOMsg) stk.pop());
161     
162                 if (logger != null) 
163                     evt.addMessage (m);
164                 return b.length;
165             } catch (ISOException e) {
166                 evt.addMessage (e);
167                 throw e;
168             } catch (IOException e) {
169                 evt.addMessage (e);
170                 throw new ISOException (e.toString());
171             } catch (SAXException e) {
172                 evt.addMessage (e);
173                 throw new ISOException (e.toString());
174             } finally {
175                 Logger.log (evt);
176             }
177         }
178     
179         public synchronized void unpack (ISOComponent c, InputStream in) 
180             throws ISOException, IOException
181         {
182             LogEvent evt = new LogEvent (this, "unpack");
183             try {
184                 if (!(c instanceof ISOMsg))
185                     throw new ISOException 
186                         ("Can't call packager on non Composite");
187     
188                 while (!stk.empty())    // purge from possible previous error
189                     stk.pop();
190     
191                 reader.parse (new InputSource (in));
192                 if (stk.empty())
193                     throw new ISOException ("error parsing");
194     
195                 ISOMsg m = (ISOMsg) c;
196                 m.merge ((ISOMsg) stk.pop());
197     
198                 if (logger != null) 
199                     evt.addMessage (m);
200             } catch (ISOException e) {
201                 evt.addMessage (e);
202                 throw e;
203             } catch (SAXException e) {
204                 evt.addMessage (e);
205                 throw new ISOException (e.toString());
206             } finally {
207                 Logger.log (evt);
208             }
209         }
210     
211         public void startElement 
212             (String ns, String name, String qName, Attributes atts)
213             throws SAXException
214         {
215             int fieldNumber = -1;
216             try {
217                 String id       = atts.getValue(ID_ATTR);
218                 if (id != null) {
219                     try {
220                         fieldNumber = Integer.parseInt (id);
221                     } catch (NumberFormatException ex) { }
222                 }
223                 if (name.equals (ISOMSG_TAG)) {
224                     if (fieldNumber >= 0) {
225                         if (stk.empty())
226                             throw new SAXException ("inner without outter");
227     
228                         ISOMsg inner = new ISOMsg(fieldNumber);
229                         ((ISOMsg)stk.peek()).set (inner);
230                         stk.push (inner);
231                     } else {
232                         stk.push (new ISOMsg(0));
233                     }
234                 } else if (name.equals (ISOFIELD_TAG)) {
235                     ISOMsg m     = (ISOMsg) stk.peek();
236                     String value = atts.getValue(VALUE_ATTR);
237                     String type  = atts.getValue(TYPE_ATTR);
238                     if (id == null || value == null)
239                         throw new SAXException ("invalid field");   
240                     if (TYPE_BINARY.equals (type)) {
241                         m.set (new ISOBinaryField (
242                             fieldNumber, 
243                                 ISOUtil.hex2byte (
244                                     value.getBytes(), 0, value.length()/2
245                                 )
246                             )
247                         );
248                     }
249                     else {
250                         m.set (new ISOField (fieldNumber, value));
251                     }
252                     
253                 }
254             } catch (ISOException e) {
255                 throw new SAXException 
256                     ("ISOException unpacking "+fieldNumber);
257             }
258         }
259     
260         public void endElement (String ns, String name, String qname) 
261             throws SAXException
262         {
263             if (name.equals (ISOMSG_TAG)) {
264                 ISOMsg m = (ISOMsg) stk.pop();
265                 if (stk.empty())
266                     stk.push (m); // push outter message
267             }
268         }
269     
270         public String getFieldDescription(ISOComponent m, int fldNumber) {
271             return "<notavailable/>";
272         }
273         public void setLogger (Logger logger, String realm) {
274             this.logger = logger;
275             this.realm  = realm;
276         }
277         public String getRealm () {
278             return realm;
279         }
280         public Logger getLogger() {
281             return logger;
282         }
283     }
