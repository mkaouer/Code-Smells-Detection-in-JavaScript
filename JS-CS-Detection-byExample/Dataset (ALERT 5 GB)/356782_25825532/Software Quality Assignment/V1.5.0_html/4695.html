38     package iso;
39     
40     import junit.framework.TestCase;
41     
42     import org.jpos.iso.IFB_LLNUM;
43     import org.jpos.iso.ISOField;
44     
45     /**
46      * @author joconnor
47      */
48     public class IFB_LLNUMTest extends TestCase
49     {
50         public void testPack() throws Exception
51         {
52             ISOField field = new ISOField(12, "1234");
53             IFB_LLNUM packager = new IFB_LLNUM(10, "Should be 041234", true);
54             TestUtils.assertEquals(new byte[]{0x04, 0x12, 0x34}, packager.pack(field));
55         }
56     
57         public void testUnpack() throws Exception
58         {
59             byte[] raw = new byte[]{0x04, 0x12, 0x34};
60             IFB_LLNUM packager = new IFB_LLNUM(10, "Should be 041234", true);
61             ISOField field = new ISOField(12);
62             packager.unpack(field, raw, 0);
63             assertEquals("1234", (String) field.getValue());
64         }
65     
66         public void testReversability() throws Exception
67         {
68             String origin = "1234567890";
69             ISOField f = new ISOField(12, origin);
70             IFB_LLNUM packager = new IFB_LLNUM(10, "Should be 1234567890", true);
71     
72             ISOField unpack = new ISOField(12);
73             packager.unpack(unpack, packager.pack(f), 0);
74             assertEquals(origin, (String) unpack.getValue());
75         }
76     }
