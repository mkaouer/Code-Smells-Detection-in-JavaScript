249         /**
250          * pad to the left
251          * @param s - original string
252          * @param len - desired len
253          * @param c - padding char
254          * @return padded string
255          */
256         public static String padleft(String s, int len, char c)
257             throws ISOException 
258         {
259             s = s.trim();
260             if (s.length() > len)
261                 throw new ISOException("invalid len " +s.length() + "/" +len);
262             StringBuffer d = new StringBuffer (len);
263             int fill = len - s.length();
264             while (fill-- > 0)
265                 d.append (c);
266             d.append(s);
267             return d.toString();
268         }

307         /**
308          * converts to BCD
309          * @param s - the number
310          * @param padLeft - flag indicating left/right padding
311          * @param d The byte array to copy into.
312          * @param offset Where to start copying into.
313          * @return BCD representation of the number
314          */
315         public static byte[] str2bcd(String s, boolean padLeft, byte[] d, int offset) {
316             int len = s.length();
317             int start = (((len & 1) == 1) && padLeft) ? 1 : 0;
318             for (int i=start; i < len+start; i++) 
319                 d [offset + (i >> 1)] |= (s.charAt(i-start)-'0') << ((i & 1) == 1 ? 0 : 4);
320             return d;
321         }

333         /**
334          * converts to BCD
335          * @param s - the number
336          * @param padLeft - flag indicating left/right padding
337          * @param fill - fill value
338          * @return BCD representation of the number
339          */
340         public static byte[] str2bcd(String s, boolean padLeft, byte fill) {
341             int len = s.length();
342             byte[] d = new byte[ (len+1) >> 1 ];
343             Arrays.fill (d, fill);
344             int start = (((len & 1) == 1) && padLeft) ? 1 : 0;
345             for (int i=start; i < len+start; i++) 
346                 d [i >> 1] |= (s.charAt(i-start)-'0') << ((i & 1) == 1 ? 0 : 4);
347             return d;
348         }
349         /**
350          * converts a BCD representation of a number to a String
351          * @param b - BCD representation
352          * @param offset - starting offset
353          * @param len - BCD field len
354          * @param padLeft - was padLeft packed?
355          * @return the String representation of the number
356          */
357         public static String bcd2str(byte[] b, int offset,
358                             int len, boolean padLeft)
359         {
360             StringBuffer d = new StringBuffer(len);
361             int start = (((len & 1) == 1) && padLeft) ? 1 : 0;
362             for (int i=start; i < len+start; i++) {
363                 int shift = ((i & 1) == 1 ? 0 : 4);
364                 char c = Character.forDigit (
365                     ((b[offset+(i>>1)] >> shift) & 0x0F), 16);
366                 if (c == 'd')
367                     c = '=';
368                 d.append (Character.toUpperCase (c));
369             }
370             return d.toString();
371         }

472         public static byte[] bitSet2byte (BitSet b)
473         {
474             int len = (b.length() > 65) ? 128 : 64;
475             byte[] d = new byte[len >> 3];
476             for (int i=0; i<len; i++) 
477                 if (b.get(i+1))

479             if (len>64)
480                 d[0] |= 0x80;
481             return d;

623          * @param s source String
624          * @param canonical true if we want to normalize \r and \n as well
625          * @return normalized string suitable for XML Output
626          */
627         public static String normalize (String s, boolean canonical) {
628             StringBuffer str = new StringBuffer();
629     
630             int len = (s != null) ? s.length() : 0;
631             for (int i = 0; i < len; i++) {
632                 char ch = s.charAt(i);
633                 switch (ch) {
634                     case '<': 
635                         str.append("&lt;");
636                         break;
637                     case '>': 
638                         str.append("&gt;");
639                         break;
640                     case '&': 
641                         str.append("&amp;");
642                         break;
643                     case '"': 
644                         str.append("&quot;");
645                         break;
646                     case '\r':
647                     case '\n': 
648                         if (canonical) {
649                             str.append("&#");
650                             str.append(Integer.toString(ch));
651                             str.append(';');
652                             break;
653                         }
654                         // else, default append char
655                     default: 
656                         if (ch >= 0x20 && ch <= 0x7F)
657                             str.append(ch);
658                         else {
659                             str.append("&#");
660                             str.append(Integer.toString(ch));
661                             str.append(';');
662                         }
663                 }
664             }
665             return (str.toString());
666         }

675         /**
676          * Protects PAN, Track2, CVC (suitable for logs).
677          *
678          * <pre>
679          * "40000101010001" is converted to "400001____0001"
680          * "40000101010001=020128375" is converted to "400001____0001=0201_____"
681          * "123" is converted to "___"
682          * </pre>
683          * @param s string to be protected 
684          * @return 'protected' String
685          */
686         public static String protect (String s) {
687             StringBuffer sb = new StringBuffer();
688             int len   = s.length();
689             int clear = len > 6 ? 6 : 0;
690             int lastFourIndex = -1;
691             if (clear > 0) {
692                 lastFourIndex = s.indexOf ('=') - 4;
693                 if (lastFourIndex < 0) {
694                     lastFourIndex = s.indexOf ('^') - 4;
695                     if (lastFourIndex < 0) 
696                         lastFourIndex = len - 4;
697                 }
698             }
699             for (int i=0; i<len; i++) {
700                 if (s.charAt(i) == '=')
701                     clear = 5;
702                 else if (s.charAt(i) == '^') {
703                     lastFourIndex = 0;
704                     clear = len - i;
705                 }
706                 else if (i == lastFourIndex)
707                     clear = 4;
708                 sb.append (clear-- > 0 ? s.charAt(i) : '_');
709             }
710             return sb.toString();
711         }

831         public static String unPadRight(String s, char c) {
832             if ( (s.trim().length() == 0) && (c == ' ') )

834             else if ( (s.trim().length() == 0) )
835                 return s;

837             int end = s.length();

1083         public static String hexdump (byte[] b, int offset, int len) {
1084             StringBuffer sb    = new StringBuffer ();
1085             StringBuffer hex   = new StringBuffer ();
1086             StringBuffer ascii = new StringBuffer ();
1087             String sep         = "  ";
1088             String lineSep     = System.getProperty ("line.separator");
1089     
1090             for (int i=offset; i<len; i++) {
1091                 char hi = Character.forDigit ((b[i] >> 4) & 0x0F, 16);
1092                 char lo = Character.forDigit (b[i] & 0x0F, 16);
1093                 hex.append (Character.toUpperCase(hi));
1094                 hex.append (Character.toUpperCase(lo));
1095                 hex.append (' ');
1096                 char c = (char) b[i];
1097                 ascii.append ((c >= 32 && c < 127) ? c : '.');
1098     
1099                 int j = i % 16;
1100                 switch (j) {
1101                     case 7 :
1102                         hex.append (' ');
1103                         break;
1104                     case 15 :
1105                         sb.append (hexOffset (i));
1106                         sb.append (sep);
1107                         sb.append (hex.toString());
1108                         sb.append (' ');
1109                         sb.append (ascii.toString());
1110                         sb.append (lineSep);
1111                         hex   = new StringBuffer ();
1112                         ascii = new StringBuffer ();
1113                         break;
