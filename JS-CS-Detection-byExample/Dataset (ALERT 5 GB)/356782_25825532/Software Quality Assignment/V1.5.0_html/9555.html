40     package org.jpos.iso;
41     
42     /**
43      * BcdPrefixer constructs a prefix storing the length in BCD.
44      * 
45      * @author joconnor
46      * @version $Revision$ $Date$
47      */
48     public class BcdPrefixer implements Prefixer
49     {
50         /**
51          * A length prefixer for upto 9 chars. The length is encoded with 1 ASCII
52          * char representing 1 decimal digit.
53          */
54         public static final BcdPrefixer L = new BcdPrefixer(1);
55         /**
56     	 * A length prefixer for upto 99 chars. The length is encoded with 2 ASCII
57     	 * chars representing 2 decimal digits.
58     	 */
59         public static final BcdPrefixer LL = new BcdPrefixer(2);
60         /**
61     	 * A length prefixer for upto 999 chars. The length is encoded with 3 ASCII
62     	 * chars representing 3 decimal digits.
63     	 */
64         public static final BcdPrefixer LLL = new BcdPrefixer(3);
65         /**
66     	 * A length prefixer for upto 9999 chars. The length is encoded with 4
67     	 * ASCII chars representing 4 decimal digits.
68     	 */
69         public static final BcdPrefixer LLLL = new BcdPrefixer(4);
70         /**
71          * A length prefixer for upto 99999 chars. The length is encoded with 5
72          * ASCII chars representing 5 decimal digits.
73          */
74         public static final BcdPrefixer LLLLL = new BcdPrefixer(5);
75     
76         /** The number of digits allowed to express the length */
77         private int nDigits;
78     
79         public BcdPrefixer(int nDigits)
80         {
81             this.nDigits = nDigits;
82         }
83     
84         /*
85     	 * (non-Javadoc)
86     	 * 
87     	 * @see org.jpos.iso.Prefixer#encodeLength(int, byte[])
88     	 */
89         public void encodeLength(int length, byte[] b)
90         {
91             for (int i = getPackedLength() - 1; i >= 0; i--) {
92                 int twoDigits = length % 100;
93                 length /= 100;
94                 b[i] = (byte)(((twoDigits / 10) << 4) + twoDigits % 10);
95             }
96         }
97     
98         /*
99     	 * (non-Javadoc)
100     	 * 
101     	 * @see org.jpos.iso.Prefixer#decodeLength(byte[], int)
102     	 */
103         public int decodeLength(byte[] b, int offset)
104         {
105             int len = 0;
106             for (int i = 0; i < (nDigits + 1) / 2; i++)
107             {
108                 len = 100 * len + ((b[offset + i] & 0xF0) >> 4) * 10 + ((b[offset + i] & 0x0F));
109             }
110             return len;
111         }
112     
113         /*
114     	 * (non-Javadoc)
115     	 * 
116     	 * @see org.jpos.iso.Prefixer#getLengthInBytes()
117     	 */
118         public int getPackedLength()
119         {
120             return (nDigits + 1) / 2;
121         }
122     }
