67         /** Should be called like this: put("ALL", "008", 2);
68          * Note: the second parameter is zero padded to three digits
69          */
70         private static void put(String alphaCode, String isoCode, int numDecimals) {
71             Currency ccy = new Currency(alphaCode, Integer.parseInt(isoCode), numDecimals);
72             currencies.put(alphaCode, ccy);
73             currencies.put(isoCode, ccy);
74         }

265         /** 
266          * Converts from an ISO Amount (12 digit string) to a double taking in 
267          * consideration the number of decimal digits according to currency
268          * 
269          * @param isoamount - The ISO amount to be converted (eg. ISOField 4)
270          * @param currency  - The ISO currency to be converted (eg. ISOField 49)
271          * @return result - A double representing the converted field
272          * @exception IllegalArgumentException
273          */
274         public static double convertFromIsoMsg(String isoamount, String currency) throws IllegalArgumentException
275         {
276             double d=0;                                                                                           
277             try
278             {
279                 Currency c=(Currency)currencies.get(currency.toUpperCase());
280                 int decimals=c.getDecimals();
281                 double m=1; if(decimals>0) for(int x=1;x<=decimals;x++) m*=10;
282                 d=new Double(isoamount).doubleValue();
283                 d/=m;
284             }
285             catch(Exception e)
286             {
287                 throw new IllegalArgumentException("Failed convertFromIsoMsg");
288             }
289             return d;
290         }
291     
292         /** 
293          * Converts an amount to an ISO Amount taking in consideration
294          * the number of decimal digits according to currency
295          * 
296          * @param amount - The amount to be converted 
297          * @param currency  - The ISO currency to be converted (eg. ISOField 49)
298          * @return result - An iso amount representing the converted field
299          * @exception IllegalArgumentException
300          */
301         public static String convertToIsoMsg(double amount,String currency) throws IllegalArgumentException
302         {
303             String z=null;
304             try
305             {
306                 Currency c=(Currency)currencies.get(currency.toUpperCase());
307                 if(c==null) throw new IllegalArgumentException("Bad currency parameter");
308                 int decimals=c.getDecimals();
309                 double m=1; if(decimals>0) for(int x=1;x<=decimals;x++) m*=10;
310                 amount*=m;
311                 z=ISOUtil.zeropad(Long.toString(Math.round (amount)),12);
312             }
313             catch(Exception e)
314             {
315                 throw new IllegalArgumentException("Failed convertToIsoMsg");
316             }
317             return z;
318         }

320         public static Object[] decomposeComposedCurrency(String incurr) throws IllegalArgumentException
321         {
322             Object[] outcurr=null;
323             try
324             {   
325                 StringTokenizer st=new StringTokenizer(incurr);
326                 String curr=st.nextToken();
327                 Double amount=new Double(st.nextToken());
328                 outcurr=new Object[2];
329                 outcurr[0]=curr;
330                 outcurr[1]=amount;
331             }
332             catch(Exception e)
333             {   
334                 throw new IllegalArgumentException("Failed decompose");
335             }
336             return outcurr;
337         }

339         public static String getIsoCodeFromAlphaCode(String alphacode) throws IllegalArgumentException
340         {
341             String isocode=null;
342             try
343             {
344                 Currency c=(Currency)currencies.get(alphacode.toUpperCase());
345                 isocode=ISOUtil.zeropad(Integer.toString(c.getIsoCode()),3);
346                 if(isocode==null) throw new IllegalArgumentException("AlphaCode not found, or incorrectly specified");
347             }
348             catch(Exception e)
349             {
350                 throw new IllegalArgumentException("Failed getIsoCodeFromAlphaCode");
351             }
352             return  isocode;
353         }

355         public static Currency getCurrency (int code) throws ISOException {
356             return (Currency) 
357                 currencies.get (ISOUtil.zeropad (Integer.toString (code), 3));
358         }
