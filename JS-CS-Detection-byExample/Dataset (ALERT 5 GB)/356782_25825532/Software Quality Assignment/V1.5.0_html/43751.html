50     package org.jpos.util;
51     
52     import java.util.Vector;
53     
54     import org.jpos.core.Configurable;
55     import org.jpos.core.Configuration;
56     import org.jpos.core.ConfigurationException;
57     import org.jpos.iso.ISOException;
58     import org.jpos.iso.ISOField;
59     import org.jpos.iso.ISOMsg;
60     import org.jpos.iso.ISOUtil;
61     
62     /**
63      * Protects selected fields from LogEvents.
64      *
65      * ProtectedLogListener acts like a filter for Event logs,
66      * it should be defined _before_ other standard LogListeners
67      * such as SimpleLogListener or RotateLogListeners.<br>
68      * i.e.
69      * <pre>
70      * <logger name="qsp">
71      *   <log-listener class="org.jpos.util.SimpleLogListener"/>
72      *   <log-listener class="org.jpos.util.ProtectedLogListener">
73      *     <property name="protect" value="2 35 45 55" />
74      *     <property name="wipe"    value="48" />
75      *   </log-listener>
76      *   <log-listener class="org.jpos.util.RotateLogListener">
77      *     <property name="file" value="/tmp/qsp.log" />
78      *     <property name="window" value="86400" />
79      *     <property name="copies" value="5" />
80      *     <property name="maxsize" value="1000000" />
81      *   </log-listener>
82      * </logger>
83      * </pre>
84      * 
85      * Order is important. In the previous example SimpleLogListener
86      * will dump unprotected LogEvents while RotateLogListener will
87      * dump protected ones (for selected fields)
88      *
89      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
90      * @version $Revision$ $Date$
91      * @see org.jpos.core.Configurable
92      * @since jPOS 1.3.3
93      */
94     public class ProtectedLogListener implements LogListener, Configurable
95     {
96         int[] protectFields = null;
97         int[] wipeFields    = null;
98         Configuration cfg   = null;
99     
100         public ProtectedLogListener () {
101             super();
102         }
103     
104        /**
105         * Configure this ProtectedLogListener<br>
106         * Properties:<br>
107         * <ul>
108         *  <li>[protect]   blank separated list of fields to be protected
109         *  <li>[wipe]      blank separated list of fields to be wiped
110         * </ul>
111         * @param cfg Configuration 
112         * @throws ConfigurationException
113         */
114         public void setConfiguration (Configuration cfg)
115             throws ConfigurationException
116         {
117             this.cfg = cfg;
118             protectFields = ISOUtil.toIntArray (cfg.get ("protect", ""));
119             wipeFields    = ISOUtil.toIntArray (cfg.get ("wipe", ""));
120         }
121         public synchronized LogEvent log (LogEvent ev) {
122             Vector payLoad = ev.getPayLoad();
123             int size = payLoad.size();
124             for (int i=0; i<size; i++) {
125                 Object obj = payLoad.elementAt (i);
126                 if (obj instanceof ISOMsg) {
127                     ISOMsg m = (ISOMsg) ((ISOMsg) obj).clone();
128                     try {
129                         checkProtected (m);
130                         checkHidden (m);
131                     } catch (ISOException e) { 
132                         ev.addMessage (e);
133                     }
134                     payLoad.setElementAt (m, i);
135                 }
136             }
137             return ev;
138         }
139         private void checkProtected (ISOMsg m) throws ISOException {
140             for (int i=0; i<protectFields.length; i++) {
141                 int f = protectFields[i];
142                 if (m.hasField (f))
143                     m.set (new ISOField (f, ISOUtil.protect (m.getString(f))));
144             }
145         }
146         private void checkHidden (ISOMsg m) throws ISOException {
147             for (int i=0; i<wipeFields.length; i++) {
148                 int f = wipeFields[i];
149                 if (m.hasField (f))
150                     m.set (new ISOField (f, "[WIPED]"));
151             }
152         }
153     }
