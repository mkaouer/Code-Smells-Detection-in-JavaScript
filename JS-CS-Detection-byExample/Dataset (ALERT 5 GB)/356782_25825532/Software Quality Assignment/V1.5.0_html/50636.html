130         public void setConfiguration(Configuration cfg) 
131             throws ConfigurationException 
132         {
133             setVetoUnmatched(cfg.getBoolean("veto-unmatched"));
134             setSpace(cfg.get("space",""));
135             setKeyPrefix(cfg.get("key-prefix", ""+hashCode()));
136             setTimeout(cfg.getLong("timeout", TIMEOUT));
137             setOverwriteOriginalFields(cfg.getBoolean("overwrite-original-fields", true));
138             int dir = cfg.get("direction","incoming").equals("incoming") ? 
139                 ISOMsg.INCOMING : ISOMsg.OUTGOING;
140             setMatchDirection(dir);
141             
142             setKey(ISOUtil.toIntArray(cfg.get("key", "11 41")));
143             setSavedFields(ISOUtil.toIntArray(cfg.get("saved-fields", "")));
144             setIgnoredFields(ISOUtil.toIntArray(cfg.get("ignored-fields", "")));
145         }

150         public ISOMsg filter(ISOChannel iSOChannel, ISOMsg m, LogEvent evt) 
151             throws ISOFilter.VetoException 
152         {
153             int[] key = getKey();
154             StringBuffer b = new StringBuffer(getKeyPrefix());
155             for(int i = 0; i < key.length; i++)
156                 b.append("|"+m.getString(key[i]));
157             String skey = b.toString();
158             if(m.getDirection() == getMatchDirection()){
159                 int[] savedFields = getSavedFields();
160                 ISOMsg saved = (ISOMsg)(
161                     (savedFields != null && savedFields.length != 0) ?
162                         m.clone(savedFields) : m.clone());
163                 int[] ignoredFields = getIgnoredFields();
164                 if (ignoredFields != null) saved.unset(ignoredFields);
165                 getSpace().out(skey, saved, getTimeout());
166                 return m;
167             } else {
168                 ISOMsg saved = (ISOMsg)getSpace().inp(skey);
169                 if (saved == null && isVetoUnmatched()) 
170                     throw new VetoException("unmatched iso message");
171                 else if(saved != null) {
172                     if (!isOverwriteOriginalFields()) m.merge(saved);
173                     else {
174                         saved.merge(m);
175                         m = saved;
176                     }
177                 }
178                 return m;
179             }
180         }
