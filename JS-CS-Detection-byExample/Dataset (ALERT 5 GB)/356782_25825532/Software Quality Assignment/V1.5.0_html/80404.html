7     package org.jpos.q2.spring;
8     
9     import org.jpos.q2.QBeanSupport;
10     import org.jpos.core.ConfigurationException;
11     import org.springframework.context.support.FileSystemXmlApplicationContext;
12     import org.springframework.context.ApplicationContext;
13     import org.jpos.util.NameRegistrar;
14     
15     /**
16      * @author Anthony Schexnaildre
17      * @version $Revision$ $Date$
18      * @jmx:mbean description="Spring QBean" extends="org.jpos.q2.QBeanSupportMBean"
19      */
20     public class Spring extends QBeanSupport implements SpringMBean {
21         ApplicationContext context;
22         String[] configFiles;
23     
24         public void initService () throws ConfigurationException {
25             configFiles = cfg.getAll( "config" ); 
26             if ( configFiles.length < 1 )
27                 throw new ConfigurationException ("config property not specified");
28     
29             context = new FileSystemXmlApplicationContext( configFiles );
30         }
31     
32         public void startService () {
33             NameRegistrar.register( getName(), this );
34         }
35     
36         public void stopService () {
37             NameRegistrar.unregister( getName() );
38         }
39     
40         /**
41          * Returns the Spring ApplicationContext
42          */
43         public ApplicationContext getContext () {
44     	    return context;
45         }
46     
47         /**
48          * @jmx:managed-attribute description="Configuration Files"
49          */
50         public String[] getConfig () {
51             return configFiles;
52         }
53     
54     }
