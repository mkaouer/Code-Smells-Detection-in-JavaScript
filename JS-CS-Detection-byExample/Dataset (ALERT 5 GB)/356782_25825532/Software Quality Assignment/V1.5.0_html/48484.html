50     package org.jpos.iso;
51     
52     import java.io.IOException;
53     import java.io.InputStream;
54     
55     /**
56      * Uses a 2 EBCDIC byte length field.
57      *
58      * right-justified with leading 0
59      * and packed data as BCD. 2 BCD digits 
60      * per byte and adding the value hex(0xF) 
61      * for padding if length is odd.
62      *
63      * @author julien.moebs@paybox.net
64      * @author doronf@xor-t.com
65      * @version $Id$
66      * @see ISOFieldPackager
67      * @see ISOComponent
68      */
69     public class IFEB_LLNUM extends ISOFieldPackager {
70         public IFEB_LLNUM () {
71             super();
72         }
73         /**
74          * @param len - field len
75          * @param description symbolic descrption
76          */
77         public IFEB_LLNUM (int len, String description) {
78             super(len, description);
79         }
80         /**
81          * @param c - a component
82          * @return packed component
83          * @exception ISOException
84          */
85         public byte[] pack(ISOComponent c) throws ISOException {
86             boolean odd = false;
87             int len;
88             String s = (String) c.getValue();
89      
90             if ((len=s.length()) > getLength() || len>99)   // paranoia settings
91                 throw new ISOException(
92                 "invalid len "+len +" packing LLEBCHAR field "+ c.getKey());
93             
94             // if odd length
95             if ( (len%2)==1 ) {
96                 odd = true;
97                 len = (len/2)+1;
98             } else {
99                 odd = false;
100                 len = len/2;
101             }
102     
103             String fieldLength = ISOUtil.zeropad(Integer.toString(len), 2);
104             
105             byte [] EBCDIClength = ISOUtil.asciiToEbcdic(fieldLength);
106     
107             // bcd stuff
108             byte[] bcd = ISOUtil.str2bcd(s, false);
109             
110             if(odd)
111                 bcd[len-1] = (byte) (bcd[len-1] | 0xf);
112     
113             byte[] b   = new byte[bcd.length + 2];
114             
115             b[0] = EBCDIClength[0];
116             b[1] = EBCDIClength[1];
117             System.arraycopy(bcd, 0, b, 2, bcd.length);
118     
119             return b;
120         }
121         /**
122          * @param c - the Component to unpack
123          * @param b - binary image
124          * @param offset - starting offset within the binary image
125          * @return consumed bytes
126          * @exception ISOException
127          */
128         public int unpack(ISOComponent c, byte[] b, int offset)
129         throws ISOException {
130             boolean pad = false;
131             int len = ((b[offset] & 0x0f) * 10)  + (b[offset+1] & 0x0f);
132             int tempLen = len*2;
133     
134             // odd handling
135             byte lastByte = b[ (offset+2+len-1) ];
136     
137             if((lastByte & 0x0f) == 0x0f)
138                 tempLen--;
139             
140             c.setValue(ISOUtil.bcd2str(b, offset+2, tempLen, pad));
141     
142             return len+2;
143         }
144     
145         /*
146          * code contributed by doronf@xor-t.com
147          */
148         public void unpack (ISOComponent c, InputStream in) 
149             throws IOException, ISOException
150         {
151             byte[] b = readBytes (in, 2);
152             int len = 
153                 (100 * ((((b[0] >> 4) & 0x0F) > 0x09 ? 0 : 
154                 ((b[0] >> 4) & 0x0F)) * 10 + (b[0] & 0x0F)) 
155                +(((b[1] >> 4) & 0x0F) > 0x09 ? 0 : 
156                   ((b[1] >> 4) & 0x0F)) * 10 + (b[1] & 0x0F)
157                 );
158             c.setValue (ISOUtil.bcd2str (readBytes (in, len), 0, 2*len, pad));
159         }
160         
161         public int getMaxPackedLength() {
162             return getLength()+2;
163         }
164     }
