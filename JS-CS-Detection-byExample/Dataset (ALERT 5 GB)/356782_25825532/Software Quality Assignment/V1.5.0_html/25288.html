73     public class VISA1Packager 
74         extends SimpleLogSource implements ISOPackager, VISA1ResponseFilter
75     {
76         public static final byte[] FS = { (byte)'\034' };
77         int[] sequence;
78         int respField;
79         String badResultCode;
80         String okPattern;
81         VISA1ResponseFilter filter;
82     
83         /**
84          * @param sequence array of fields that go to VISA1 request
85          * @param respField where to put response
86          * @param badResultCode (i.e. "05")
87          * @param okPattern (i.e. "AUT. ")
88          */
89         public VISA1Packager 
90             (int[] sequence, int respField, String badResultCode, String okPattern)
91         { 
92             super();
93             this.sequence      = sequence;
94             this.respField     = respField;
95             this.badResultCode = badResultCode;
96             this.okPattern     = okPattern;
97             setVISA1ResponseFilter (this);
98         }
99         public void setVISA1ResponseFilter (VISA1ResponseFilter filter) {
100             this.filter = filter;
101         }
102         protected int handleSpecialField35 (ISOMsg m, Vector v) 
103             throws ISOException
104         {
105             int len = 0;
106             byte[] entryMode = new byte[1];
107             if (m.hasField (35)) {
108                 entryMode[0] = (byte) '\001';
109                 byte[] value = ((String)m.getValue(35)).getBytes();
110                 v.addElement (entryMode);
111                 v.addElement (value);
112                 v.addElement (FS);
113                 len += value.length+2;
114             } else if (m.hasField (2) && m.hasField (14)) {
115                 entryMode[0] = (byte) '\000';
116                 String simulatedTrack2 = 
117                     (String) m.getValue(2) + "=" + (String) m.getValue(14);
118                 v.addElement (entryMode);
119                 v.addElement (simulatedTrack2.getBytes());
120                 v.addElement (FS);
121                 len += simulatedTrack2.length()+2;
122             }
123             return len;
124         }
125         public byte[] pack (ISOComponent c) throws ISOException
126         {
127             LogEvent evt = new LogEvent (this, "pack");
128             try {
129                 if (!(c instanceof ISOMsg))
130                     throw new ISOException
131                         ("Can't call VISA1 packager on non ISOMsg");
132             
133                 ISOMsg m = (ISOMsg) c;
134     
135                 int len  = 0;
136                 Vector v = new Vector();
137                 for (int i=0; i<sequence.length; i++) {
138                     int fld = sequence[i];
139                     if (fld == 35) 
140                         len += handleSpecialField35 (m, v);
141                     else if (m.hasField(fld)) {
142                         byte[] value;
143                         if (fld == 4) {
144                             long l = Long.parseLong (((String)m.getValue(4)));
145                             value = ISOUtil.formatAmount (l,12).trim().getBytes();
146                         }
147                         else
148                             value = ((String)m.getValue(fld)).getBytes();
149                         v.addElement (value);
150                         len += value.length;
151                         if (i < (sequence.length-1)) {
152                             v.addElement (FS);
153                             len++;
154                         }
155                     }
156                 }
157     
158                 int k = 0;
159                 byte[] d = new byte[len];
160                 for (int i=0; i<v.size(); i++) {
161                     byte[] b = (byte[]) v.elementAt(i);
162                     for (int j=0; j<b.length; j++)
163                         d[k++] = b[j];
164                 }
165                 if (logger != null)  // save a few CPU cycle if no logger available
166                     evt.addMessage (ISOUtil.dumpString (d));
167                 return d;
168             } catch (ISOException e) {
169                 evt.addMessage (e);
170                 throw e;
171             } finally {
172                 Logger.log(evt);
173             }
174         }
175     
176         public String guessAutNumber (String s) {
177             StringBuffer buf = new StringBuffer();
178             for (int i=0; i<s.length(); i++)
179                 if (Character.isDigit(s.charAt(i))) 
180                     buf.append (s.charAt(i));
181             if (buf.length() == 0)
182                 return null;
183     
184             while (buf.length() > 6)
185                 buf.deleteCharAt(0);
186             while (buf.length() < 6)
187                 buf.insert(0, "0");
188             
189             return buf.toString();
190         }
191     
192         public int unpack (ISOComponent m, byte[] b) throws ISOException
193         {
194             String response = new String (b);
195             m.set (new ISOField (respField, response));
196             m.set (new ISOField (39, badResultCode));
197             if (response.startsWith (okPattern)) {
198                 String autNumber = filter.guessAutNumber (response);
199                 if (autNumber != null) {
200                     m.set (new ISOField (39, "00"));
201                     m.set (new ISOField (38, autNumber));
202                 }
203             }
204             return b.length;
205         }
206         public void unpack (ISOComponent m, InputStream in) throws ISOException {
207             throw new ISOException ("not implemented");
208         }
209         public String getFieldDescription(ISOComponent m, int fldNumber)
210         {
211             return "VISA 1 fld "+fldNumber;
212         }
