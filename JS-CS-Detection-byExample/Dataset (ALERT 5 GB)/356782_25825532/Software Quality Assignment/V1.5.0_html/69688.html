22     /**
23      * DailyTask Adaptor
24      *
25      * @author Alejandro Revilla
26      * @version $Revision$ $Date$
27      */
28     public class DailyTaskAdaptor extends QBeanSupport implements Runnable {
29         Runnable task;
30         Thread thisThread = null;
31     
32         public DailyTaskAdaptor () {
33             super ();
34         }
35     
36         protected void initService () throws Exception {
37             QFactory factory = getServer().getFactory();
38             Element e = getPersist ();
39             task = (Runnable) factory.newInstance (e.getChildTextTrim ("class"));
40             factory.setLogger (task, e);
41         }
42         protected void startService () throws Exception {
43             if (task instanceof Configurable) {
44                 Element e = getPersist ();
45                 QFactory factory = getServer().getFactory();
46                 ((Configurable)task).setConfiguration (
47                     factory.getConfiguration (e)
48                 );
49             }
50             (thisThread = new Thread(this)).start();
51         }
52         protected void stopService () throws Exception {
53             if (thisThread != null)
54                 thisThread.interrupt();
55         }
56         public void run () {
57             while (running()) {
58                 waitUntilStartTime();
59                 if (running()) {
60                     Thread taskThread = new Thread(task);
61                     taskThread.setDaemon (true);
62                     taskThread.start();
63                     ISOUtil.sleep (1000);
64                 }
65             }
66         }
67         public Date getWhen() {
68             String s = cfg.get ("start")+":00:00";
69             int hh = Integer.parseInt(s.substring (0, 2));
70             int mm = Integer.parseInt(s.substring (3, 5));
71             int ss = Integer.parseInt(s.substring (6, 8));
72     
73             Date now = new Date();
74             Calendar cal = new GregorianCalendar();
75     
76             cal.setTime (now);
77             cal.set (Calendar.HOUR_OF_DAY, hh);
78             cal.set (Calendar.MINUTE, mm);
79             cal.set (Calendar.SECOND, ss);
80     
81             Date when = cal.getTime();
82             if (when.before(now)) 
83                 when = new Date(when.getTime() + 24*60*60*1000);
84     
85             return when;
86         }
87         protected void waitUntilStartTime() {
88             Date when = getWhen();
89             while (running()) {
90                 Date now = new Date();
91                 if (now.before (when)) {
92                     long sleepTime = when.getTime() - now.getTime();
93                     getLog().info ("sleeping",
94                         (sleepTime/1000) + " secs until " + when.toString()
95                     );
96                     try {
97                         Thread.sleep (sleepTime);
98                     } catch (InterruptedException e) { 
99                         when = getWhen();
100                     }
101                 } else
102                     break;
103             }
104         }
105     }
