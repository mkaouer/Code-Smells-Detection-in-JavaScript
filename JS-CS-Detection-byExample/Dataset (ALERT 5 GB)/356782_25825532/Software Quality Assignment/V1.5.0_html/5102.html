131         /**
132          * Construct a CardHolder based on content received on
133          * field 35 (track2) or field 2 (PAN) + field 14 (EXP)
134          * @param m an ISOMsg
135          * @throws InvalidCardException
136          */
137         public CardHolder (ISOMsg m)
138             throws InvalidCardException
139         {
140             super();
141             try {
142                 if (m.hasField(35))
143                     parseTrack2 ((String) m.getValue(35));
144                 else if (m.hasField(2) && m.hasField (14)) {
145                     setPAN ((String) m.getValue(2));
146                     setEXP ((String) m.getValue(14));
147                 } else {
148                     throw new InvalidCardException("required fields not present");
149                 }
150                 if (m.hasField(45)) {
151                     setTrack1((String) m.getValue(45));
152                 }
153                 if (m.hasField(55)) {
154                     setSecurityCode (m.getString(55));
155                 }
156             } catch (ISOException e) {
157                 throw new InvalidCardException();
158             }
159         }

161         /**
162          * extract pan/exp/trailler from track2
163          * @param s a valid track2
164          * @exception InvalidCardException
165          */
166         public void parseTrack2 (String s) 
167             throws InvalidCardException
168         {
169             if (s == null)
170                 throw new InvalidCardException ("null track2 data");
171             int separatorIndex = s.replace ('D','=').indexOf(TRACK2_SEPARATOR);
172             if ((separatorIndex > 0) && (s.length() > separatorIndex+4)) {
173                 pan = s.substring(0, separatorIndex);
174                 exp = s.substring(separatorIndex+1, separatorIndex+1+4);
175                 trailler = s.substring(separatorIndex+1+4);
176             } else 
177                 throw new InvalidCardException (s);
178         }

201         /**
202          * @return the Name written on the card (from track1)
203          */
204         public String getNameOnCard() {
205             String name = null;
206             if (track1!=null) {
207                 StringTokenizer st = 
208                         new StringTokenizer(track1, TRACK1_SEPARATOR);
209                 if (st.countTokens()<2)
210                     return null;
211                 st.nextToken(); // Skips the first token
212                 name = st.nextToken(); // This is the name
213             }
214             return name;
215         }

315         /**
316          * Y2K compliant expiration check
317          * @return true if card is expired (or invalid exp)
318          */
319         public boolean isExpired () {
320             if (exp == null || exp.length() != 4)
321                 return true;
322             String now = ISODate.formatDate(new java.util.Date(), "yyyyMM");
323             try {
324                 int mm = Integer.parseInt(exp.substring(2));
325                 int aa = Integer.parseInt(exp.substring(0,2));
326                 if ((aa < 100) && (mm > 0) && (mm <= 12)) {
327                     String expDate = ((aa < 70) ? "20" : "19") + exp;
328                     if (expDate.compareTo(now) >= 0)
329                         return false;
330                 }
331             } catch (NumberFormatException e) { }
332             return true;
333         }
334         public boolean isValidCRC () {
335             return isValidCRC(this.pan);
336         }
337         public static boolean isValidCRC (String p) {
338             int i, crc;
339     
340             int odd = p.length() % 2;
341             
342             for (i=crc=0; i<p.length(); i++) {
343                 char c = p.charAt(i);
344                 if (!Character.isDigit (c))
345                     return false;
346                 c = (char) (c - '0');
347                 if (i % 2 == odd)
348                     crc+=(c*2) >= 10 ? ((c*2)-9) : (c*2);        
349                 else
350                     crc+=c;
351             }
352             return crc % 10 == 0;
353         }

355         /**
356          * dumps CardHolder basic information<br>
357          * by default we do not dump neither track1/2 nor securityCode
358          * for security reasons.
359          * @param p a PrintStream usually suplied by Logger
360          * @param indent ditto
361          * @see org.jpos.util.Loggeable
362          */
363         public void dump (PrintStream p, String indent) {
364             p.print (indent + "<CardHolder");
365             if (hasTrack1())
366                 p.print (" trk1=\"true\"");
367     
368             if (hasTrack2())
369                 p.print (" trk2=\"true\"");
370     
371             if (hasSecurityCode())
372                 p.print (" sec=\"true\"");
373     
374             if (isExpired())
375                 p.print (" expired=\"true\"");
376     
377             p.println (">");
378             p.println (indent + "  " + "<pan>" +pan +"</pan>");
379             p.println (indent + "  " + "<exp>" +exp +"</exp>");
380             p.println (indent + "</CardHolder>");
381         }

386         public String getServiceCode () {

389                 "   ";

392             return trailler == null ? true : (trailler.trim().length() == 0);

395         /**
396          * compares two cardholder object<br>
397          * based on PAN and EXP
398          * @param obj a CardHolder instance
399          * @return true if pan and exp matches
400          */
401         public boolean equals(Object obj) {
402             if ((obj != null) && (obj instanceof CardHolder)) {
403                 CardHolder ch = (CardHolder) obj;
404                 if ( (pan != null) && (ch.pan != null) &&
405                      (exp != null) && (ch.exp != null) &&
406                      pan.equals (ch.pan) && exp.equals (ch.exp))
407                     return true;
408             }
409             return false;
410         }
