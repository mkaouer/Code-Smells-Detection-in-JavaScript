50     package  smadapter;
51     
52     import  org.jpos.util.Logger;
53     import  org.jpos.util.SimpleLogListener;
54     import  org.jpos.security.*;
55     import  org.jpos.security.jceadapter.JCESecurityModule;
56     import  java.util.Properties;
57     import  org.jpos.core.Configuration;
58     import  org.jpos.core.SimpleConfiguration;
59     import  org.jpos.iso.ISOUtil;
60     
61     /**
62      * A class that demonstrates the use of the SMAdapter interface, using
63      * JCE Security Module.
64      * @author Hani S. Kirollos
65      * @version $Revision$ $Date$
66      */
67     public class Test {
68     
69         public static void main (String args[]) {
70             try {
71                 Logger logger = new Logger();
72                 logger.addListener(new SimpleLogListener(System.out));
73                 Properties cfgProps = new Properties();
74                 //cfgProps.setProperty("jce", "org.bouncycastle.jce.provider.BouncyCastleProvider");
75                 cfgProps.setProperty("lmk", "src/ext-examples/smadapter/lmk");
76                 Configuration cfg = new SimpleConfiguration(cfgProps);
77                 SMAdapter sm = new JCESecurityModule(cfg, logger, "security-module");
78     
79                 /*
80                     An acquirer scenario
81                 */
82     
83                 /*
84                     The Zone Master Key (typically stored in a clear database)
85                     Used for exchanging keys with the issuer, in our scenario.
86                 */
87                 SecureDESKey zmk = new SecureDESKey(sm.LENGTH_DES3_2KEY, sm.TYPE_ZMK,
88                     "CB3EC0FF092D3786167566882685307B", "5B3368");
89     
90                 /*
91                     The exchanged Zone PIN Key (typically exchanged using ANSI X9.17,
92                     since we didn't generate the ZPK, then we are simulating the role
93                     of Party B in a point-to-point environment. See ANSI X9.17)
94                     Generated by the issuer and sent to us, in our scenario.
95                 */
96                 byte[] zpkUnderZmk = ISOUtil.hex2byte("28E054ABD112F1BC");
97     
98                 /*
99                     The Zone PIN Key needs to be imported (typically done by
100                     translating ZPK from encryption under ZMK to encryption under LMK)
101                     Used for sending encrypted PINs to the issuer, in our scenario.
102                 */
103                 SecureDESKey zpk = sm.importKey(sm.LENGTH_DES, sm.TYPE_ZPK, zpkUnderZmk, zmk, true);
104     
105     
106                 /*
107                     The Terminal Master Key (typically stored in a clear database)
108                     Used for exchanging keys with the terminal, in our scenario.
109                 */
110                 SecureDESKey tmk = new SecureDESKey(sm.LENGTH_DES3_2KEY, sm.TYPE_TMK,
111                     "A0DC2159505FE495FE59B4EAA63F89DF", "863605");
112     
113                 /*
114                     Generating a new Terminal PIN Key
115                 */
116                 // SecureDESKey tpk = sm.generateKey(sm.LENGTH_DES3_2KEY, sm.TYPE_TPK);
117                 /* just to make the test results, the same :-) */
118                     SecureDESKey tpk = new SecureDESKey(sm.LENGTH_DES3_2KEY, sm.TYPE_TPK,
119                     "99C2B0C250C7190DC5B11D1B28F5C007", "01D863");
120     
121                 /*
122                     Export the TPK to be sent to the terminal (See ANSI X9.24, Key
123                     Management method 2)
124                     At the terminal, the terminal would import TPK to its own security module
125                 */
126                 byte[] tpkUnderTmk = sm.exportKey(tpk, tmk);
127     
128     
129                 /*
130                     PIN under TPK received from terminal
131                 */
132                 EncryptedPIN pinUnderTpk = new EncryptedPIN("C77BC10670AC51FE", sm.FORMAT01,
133                     "400000123456");
134     
135                 /*
136                     Translating PIN to issuer.
137                     The issuer in this case accepts encrypted PINs using FORMAT01
138                 */
139                 EncryptedPIN pinUnderZpk = sm.translatePIN(pinUnderTpk, tpk, zpk, sm.FORMAT01);
140     
141             } catch (Exception e) {
142                 e.printStackTrace();
143             }
144         }
145     }
