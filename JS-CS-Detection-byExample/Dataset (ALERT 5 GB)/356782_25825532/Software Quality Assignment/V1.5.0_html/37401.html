100             super();

107          * parse message
108          *
109          * @param is input stream
110          *
111          * @throws IOException
112          * @throws JDOMException
113          * @throws MalformedURLException
114          */
115         public void unpack (InputStream is) 
116             throws IOException, JDOMException, MalformedURLException
117         {
118             try {
119                 unpack (is, getSchema (baseSchema));
120             } catch (EOFException e) {
121                 map.put ("EOF", "true");
122             }
123         }

140          * @return message string
141          */
142         public String pack () 
143             throws JDOMException, MalformedURLException, ISOException, IOException
144         {
145             StringBuffer sb = new StringBuffer ();
146             pack (getSchema (baseSchema), sb);
147             return sb.toString ();

150         protected String get (String id, String type, int length, String defValue) 
151             throws ISOException
152         {
153             String value = (String) map.get (id);
154             if (value == null)
155                 value = defValue == null ? "" : defValue;
156     
157             type   = type.toUpperCase ();
158     
159             switch (type.charAt (0)) {
160                 case 'N':
161                     value = ISOUtil.zeropad (value, length);
162                     break;
163                 case 'A':
164                     value = ISOUtil.strpad (value, length);
165                     break;
166                 case 'K':
167                     if (defValue != null)
168                         value = defValue;
169                     break;
170                 case 'B':
171                     try {
172                         value = new String (
173                             ISOUtil.hex2byte (value.substring (0, length << 1)),
174                             "ISO8859_1"
175                         );
176                     } catch (UnsupportedEncodingException e) {
177                         // ISO8859_1 is supported
178                     }
179                     break;
180             }
181             return (type.endsWith ("FS")) ? value.trim() : value;
182         }
183     
184         protected void pack (Element schema, StringBuffer sb) 
185             throws JDOMException, MalformedURLException, ISOException, IOException
186         {
187             String keyOff = "";
188             Iterator iter = schema.getChildren("field").iterator();
189             while (iter.hasNext()) {
190                 Element elem = (Element) iter.next ();
191                 String id    = elem.getAttributeValue ("id");
192                 int length   = Integer.parseInt (elem.getAttributeValue ("length"));
193                 String type  = elem.getAttributeValue ("type");
194                 boolean key  = "true".equals (elem.getAttributeValue ("key"));
195                 String defValue = elem.getText();
196                 String value = get (id, type, length, defValue);
197                 sb.append (value);
198                 if (type.endsWith ("FS"))
199                     sb.append (FS);
200                 if (key) 
201                     keyOff = keyOff + value;
202             }
203             if (keyOff.length() > 0) 
204                 pack (getSchema (getId (schema) + keyOff), sb);
205         }
206     
207         protected void unpack (InputStream is, Element schema) 
208             throws IOException, JDOMException, MalformedURLException 
209         
210         {
211             Iterator iter = schema.getChildren("field").iterator();
212             String keyOff = "";
213             while (iter.hasNext()) {
214                 Element elem = (Element) iter.next();
215     
216                 String id    = elem.getAttributeValue ("id");
217                 int length   = Integer.parseInt (elem.getAttributeValue ("length"));
218                 String type  = elem.getAttributeValue ("type").toUpperCase();
219                 boolean key  = "true".equals (elem.getAttributeValue ("key"));
220                 String value = readField (
221                     is, id, length, type.endsWith ("FS"), "B".equals (type)
222                 );
223     
224                 if (key)
225                     keyOff = keyOff + value;
226     
227                 if ("K".equalsIgnoreCase (type) && !value.equals (elem.getText()))
228                     throw new IllegalArgumentException (
229                         "Field "+id 
230                            + " value='"     +value
231                            + "' expected='" + elem.getText () + "'"
232                     );
233             }
234             if (keyOff.length() > 0) {
235                 unpack (is, 
236                     getSchema (getId (schema) + keyOff)
237                 );
238             }
239         }
240         private String getId (Element e) {
241             String s = e.getAttributeValue ("id");
242             return s == null ? "" : s;
243         }
244         protected String read (InputStream is, int len, boolean fs) 
245             throws IOException 
246         {
247             StringBuffer sb = new StringBuffer();
248             byte[] b = new byte[1];
249             for (int i=0; i<len; i++) {
250                 if (is.read (b) < 0)
251                     throw new EOFException ();
252                 if (b[0] == FS) {
253                     fs = false;
254                     break;
255                 }
256                 sb.append ((char) (b[0] & 0xff));
257             }
258             if (fs) {
259                 if (is.read (b) < 0)
260                     throw new EOFException ();
261             }
262             return sb.toString ();
263         }
264         protected String readField 
265             (InputStream is, String fieldName, int len, boolean fs, boolean binary) 
266             throws IOException
267         {
268             String fieldValue = read (is, len, fs);
269             if (binary)
270                 fieldValue = ISOUtil.hexString (fieldValue.getBytes ("ISO8859_1"));
271             map.put (fieldName, fieldValue);
272             // System.out.println (fieldName + ":" + fieldValue);
273             return fieldValue;
274         }
275         public void set (String name, String value) {
276             if (value != null)
277                 map.put (name, value);
278             else
279                 map.remove (name);
280         }

288             return header != null ? ISOUtil.hexString (header).substring (2) : "";
289         }

291             return (String) map.get (name);
292         }

294             map.put (name, msg.get (name));
295         }
296         public byte[] getHexBytes (String name) {
297             String s = get (name);
298             return s == null ? null : ISOUtil.hex2byte (s);
299         }
300         public int getInt (String name) {
301             int i = 0;
302             try {
303                 i = Integer.parseInt (get (name));
304             } catch (Exception e) { }
305             return i;
306         }
307         public Element toXML () {
308             Element e = new Element ("message");
309             if (header != null) {
310                 e.addContent (
311                     new Element ("header")
312                         .setText (getHexHeader ())
313                 );
314             }
315             Iterator iter = map.keySet().iterator();
316             while (iter.hasNext()) {
317                 String fieldName = (String) iter.next();
318                 Element inner = new Element (fieldName);
319                 inner.addContent (ISOUtil.normalize ((String) map.get (fieldName)));
320                 e.addContent (inner);
321             }
322             return e;
323         }
324         protected Element getSchema () 
325             throws JDOMException, MalformedURLException, IOException
326         {
327             return getSchema (baseSchema);
328         }
329         protected Element getSchema (String message) 
330             throws JDOMException, MalformedURLException, IOException
331         {
332             StringBuffer sb = new StringBuffer (basePath);
333             sb.append (message);
334             sb.append (".xml");
335             String uri = sb.toString ();
336     
337             Space sp = TransientSpace.getSpace();
338             Element schema = (Element) sp.rdp (uri);
339             if (schema == null) {
340                 synchronized (FSDMsg.class) {
341                     schema = (Element) sp.rdp (uri);
342                     if (schema == null) {
343                         SAXBuilder builder = new SAXBuilder ();
344                         schema = builder.build (new URL (uri)).getRootElement ();
345                     }
346                     sp.out (uri, schema);
347                 }
348             } 
349             return schema;
350         }
351         public void dump (PrintStream p, String indent) {
352             try {
353                 XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
354                 out.output (toXML(), p);
355             } catch (IOException e) {
356                 e.printStackTrace (p);
357             }
358         }

360          * @return message's Map
361          */
362         public Map getMap () {
363             return map;
364         }
