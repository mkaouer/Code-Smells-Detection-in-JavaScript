58     import org.jpos.core.Sequencer;

61     import org.jpos.iso.ISODate;
62     import org.jpos.iso.ISOException;

64     import org.jpos.iso.ISOFilter;

70     /**
71      * MacroFilter useful to set sequencers, date, unset iso fields, etc.
72      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
73      * @version $Revision$ $Date$
74      */
75     public class MacroFilter implements ISOFilter, ReConfigurable {
76         Sequencer seq;
77         Configuration cfg;
78         int[] unsetFields  = new int[0];
79         int[] validFields  = new int[0];
80         public MacroFilter () {
81             super();
82         }
83        /**
84         * @param cfg
85         * <ul>
86         *  <li>sequencer - a sequencer used to store counters
87         *  <li>unset - space delimited list of fields to be unset 
88         *  <li>valid - space delimited list of valid fields 
89         *  <li>comma delimited list of fields to be unset when applying filter
90         *  <li>xzy - property named "xyz"
91         * </ul>
92         */
93         public void setConfiguration (Configuration cfg) 
94             throws ConfigurationException
95         {
96             this.cfg = cfg;
97             try {
98                 String seqName  = cfg.get ("sequencer", null);
99                 unsetFields     = ISOUtil.toIntArray (cfg.get ("unset", ""));
100                 validFields     = ISOUtil.toIntArray (cfg.get ("valid", ""));
101                 if (seqName != null) {
102                     seq = (Sequencer) NameRegistrar.get (
103                         "sequencer."+cfg.get("sequencer")
104                     );
105                 } else if (seq == null) {
106                     seq = new VolatileSequencer();
107                 }
108             } catch (NameRegistrar.NotFoundException e) {
109                 throw new ConfigurationException (e);
110             }
111         }
112     
113         public ISOMsg filter (ISOChannel channel, ISOMsg m, LogEvent evt) 
114             throws VetoException
115         {
116             try {
117                 applyProps (m);
118                 if (validFields.length > 0) 
119                     m = (ISOMsg) m.clone (validFields);
120                 if (unsetFields.length > 0)
121                     m.unset (unsetFields);
122             } catch (ISOException e) {
123                 evt.addMessage (e);
124                 throw new VetoException (e);
125             }
126             return m;
127         }
128         private void applyProps (ISOMsg m) throws ISOException {
129             int maxField = m.getMaxField ();
130             for (int i=0; i<=maxField; i++) {
131                 Object o = null;
132                 if (m.hasField (i))
133                     o = m.getValue (i);
134                 if (o instanceof String) {
135                     String value = (String) o;
136                     if (value.length () == 0)
137                         continue;
138                     if (value.equalsIgnoreCase ("$date") )
139                         m.set (new ISOField (i, ISODate.getDateTime(new Date())));
140                     else if ((value.toLowerCase().startsWith ("$date") ) && (value.indexOf("GMT") > 0)) {
141                         String zoneID = value.substring(value.indexOf("GMT"));
142                         m.set (new ISOField (i, ISODate.getDateTime(new Date(), TimeZone.getTimeZone(zoneID))));
143                     }                    
144                     else if (value.charAt (0) == '#')
145                         m.set (new ISOField (i,
146                           ISOUtil.zeropad (
147                             Integer.toString(seq.get (value.substring(1))),6)
148                           )
149                         );
150                     else if (value.charAt (0) == '$') {
151                         String p = cfg.get (value.substring(1), null);
152                         if (p != null)
153                             m.set (new ISOField (i, p));
154                     }
155                 }
156             }
157         }
158     }
