98         private TrustManager[] getTrustManagers(KeyStore ks)
99             throws GeneralSecurityException {
100             if (serverAuthNeeded) {
101                 TrustManagerFactory tm = TrustManagerFactory.getInstance("SunX509" );
102                 tm.init( ks ); 
103                 return tm.getTrustManagers(); 
104             } else {
105                 // Create a trust manager that does not validate certificate chains
106                 return new TrustManager[]{
107                     new X509TrustManager() {
108                         public java.security.cert.X509Certificate[] getAcceptedIssuers() {
109                             return new java.security.cert.X509Certificate[] {};
110                         }
111                         public void checkClientTrusted(
112                             java.security.cert.X509Certificate[] certs, String authType) {
113                         }
114                         public void checkServerTrusted(
115                             java.security.cert.X509Certificate[] certs, String authType) {
116                         }
117                     }
118                 };
119             }
120         }

122         /**
123          * Create a SSLSocket Context
124          * @return the SSLContext
125          * @returns null if exception occurrs
126          */
127         private SSLContext getSSLContext() throws ISOException {
128             if(password==null)  password=getPassword();
129             if(keyPassword ==null)  keyPassword=getKeyPassword();
130             if(keyStore==null || keyStore.length()==0) {
131                 keyStore=System.getProperty("user.home")+File.separator+".keystore";
132             }
133     
134             try{
135                 KeyStore ks = KeyStore.getInstance( "JKS" );
136                 ks.load( new FileInputStream( new File( keyStore) ),password.toCharArray());
137                 KeyManagerFactory km = KeyManagerFactory.getInstance( "SunX509"); 
138                 km.init( ks, keyPassword.toCharArray() );
139                 KeyManager[] kma = km.getKeyManagers();
140                 TrustManager[] tma = getTrustManagers( ks );
141                 SSLContext sslc = SSLContext.getInstance( "SSL" ); 
142                 sslc.init( kma, tma, SecureRandom.getInstance( "SHA1PRNG" ) ); 
143            
144                 return sslc;
145             } catch(Exception e) {
146                 throw new ISOException (e);
147             } finally { 
148                 password=null;
149                 keyPassword=null;
150             }
151         }

199          * @param  host   the host name
200          * @param  port   the port number
201          * @return a socket connected to the specified host and port.
202          * @exception IOException if an I/O error occurs during socket creation
203          * @exception ISOException should any other error occurs

205         public Socket createSocket(String host, int port) 
206             throws IOException, ISOException
207         {
208             if(socketFactory==null) socketFactory=createSocketFactory();
209             SSLSocket s = (SSLSocket) socketFactory.createSocket(host,port);
210             verifyHostname(s);
211             return s;

214         /**
215          * Verify that serverName and CN equals.
216          *
217          * <pre>
218          * Origin:      jakarta-commons/httpclient
219          * File:        StrictSSLProtocolSocketFactory.java
220          * Revision:    1.5
221          * License:     Apache-2.0
222          * </pre>
223          *
224          * @param socket a SSLSocket value
225          * @exception SSLPeerUnverifiedException  If there are problems obtaining
226          * the server certificates from the SSL session, or the server host name 
227          * does not match with the "Common Name" in the server certificates 
228          * SubjectDN.
229          * @exception UnknownHostException  If we are not able to resolve
230          * the SSL sessions returned server host name. 
231          */
232         private void verifyHostname(SSLSocket socket)
233             throws SSLPeerUnverifiedException, UnknownHostException
234         {
235             if (!serverAuthNeeded) {
236                 return; 
237             }
238     
239             SSLSession session = socket.getSession();
240     
241             if (serverName==null || serverName.length()==0) {
242                 serverName = session.getPeerHost();
243                 try {
244                     InetAddress addr = InetAddress.getByName(serverName);
245                 } catch (UnknownHostException uhe) {
246                     throw new UnknownHostException("Could not resolve SSL " +
247                                                    "server name " + serverName);
248                 }
249             }
250     
251     
252             X509Certificate[] certs = session.getPeerCertificateChain();
253             if (certs==null || certs.length==0) 
254                 throw new SSLPeerUnverifiedException("No server certificates found");
255     
256             //get the servers DN in its string representation
257             String dn = certs[0].getSubjectDN().getName();
258     
259             //get the common name from the first cert
260             String cn = getCN(dn);
261             if (!serverName.equalsIgnoreCase(cn)) {
262                 throw new SSLPeerUnverifiedException("Invalid SSL server name. "+
263                                                      "Expected '" + serverName +
264                                                      "', got '" + cn + "'");
265             }
266         }

268         /**
269          * Parses a X.500 distinguished name for the value of the 
270          * "Common Name" field.
271          * This is done a bit sloppy right now and should probably be done a bit
272          * more according to RFC 2253.
273          *
274          * <pre>
275          * Origin:      jakarta-commons/httpclient
276          * File:        StrictSSLProtocolSocketFactory.java
277          * Revision:    1.5
278          * License:     Apache-2.0
279          * </pre>
280          *
281          * @param dn  a X.500 distinguished name.
282          * @return the value of the "Common Name" field.
283          */
284         private String getCN(String dn) {
285             int i = dn.indexOf("CN=");
286             if (i == -1) {
287                 return null;
288             }
289             //get the remaining DN without CN=
290             dn = dn.substring(i + 3);  
291             // System.out.println("dn=" + dn);
292             char[] dncs = dn.toCharArray();
293             for (i = 0; i < dncs.length; i++) {
294                 if (dncs[i] == ','  && i > 0 && dncs[i - 1] != '\\') {
295                     break;
296                 }
297             }
298             return dn.substring(0, i);
299         }
