50     package org.jpos.iso.filter;
51     
52     import java.io.ByteArrayInputStream;
53     import java.io.ByteArrayOutputStream;
54     
55     import javax.xml.transform.Transformer;
56     import javax.xml.transform.TransformerConfigurationException;
57     import javax.xml.transform.TransformerFactory;
58     import javax.xml.transform.stream.StreamResult;
59     import javax.xml.transform.stream.StreamSource;
60     
61     import org.jpos.core.Configurable;
62     import org.jpos.core.Configuration;
63     import org.jpos.core.ConfigurationException;
64     import org.jpos.iso.ISOChannel;
65     import org.jpos.iso.ISOException;
66     import org.jpos.iso.ISOFilter;
67     import org.jpos.iso.ISOMsg;
68     import org.jpos.iso.packager.XMLPackager;
69     import org.jpos.util.LogEvent;
70     
71     /**
72      * Implements ISOFilter by means of XSL-Transformations
73      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
74      * @version $Revision$ $Date$
75      */
76     public class XSLTFilter implements ISOFilter, Configurable {
77         boolean reread;
78         String xsltfile;
79         TransformerFactory tfactory;
80         Transformer transformer;
81         XMLPackager packager;
82     
83         /**
84          * Default noargs constructor
85          * @throws ISOException
86          */
87         public XSLTFilter () throws ISOException {
88             super();
89             packager    = new XMLPackager();
90             tfactory    = TransformerFactory.newInstance();
91             transformer = null;
92             reread      = true;
93         }
94     
95         /**
96          * @param xsltfile XSL Transformation file
97          * @param reread true if you want XSLT file re-read from disk
98          * @throws ISOException
99          */
100         public XSLTFilter (String xsltfile, boolean reread) 
101             throws ISOException
102         {
103             this();
104             try {
105                 this.xsltfile    = xsltfile;
106                 this.reread      = reread;
107                 this.transformer = 
108                     tfactory.newTransformer(new StreamSource(xsltfile));
109             } catch (TransformerConfigurationException e) {
110                 throw new ISOException (e);
111             }
112     
113         }
114     
115        /**
116         * configure filter.
117         *
118         * <ul>
119         *  <li>xsltfile - source XSL-T file
120         *  <li>reread   - something != "no" will re-read source file
121         * </ul>
122         *
123         * @param cfg new ConfigurationFile
124         */
125         public void setConfiguration (Configuration cfg) 
126             throws ConfigurationException
127         {
128             try {
129                 transformer = tfactory.newTransformer(
130                     new StreamSource(cfg.get("xsltfile"))
131                 );
132                 String s = cfg.get ("reread");
133                 reread   =  (s == null || s.equals ("no"));
134             } catch (Exception e) {
135                 throw new ConfigurationException (e);
136             }
137         }
138     
139         /**
140          * @param channel current ISOChannel instance
141          * @param m ISOMsg to filter
142          * @param evt LogEvent
143          * @return an ISOMsg (possibly parameter m)
144          * @throws VetoException
145          */
146         public ISOMsg filter (ISOChannel channel, ISOMsg m, LogEvent evt) 
147             throws VetoException
148         {
149             try {
150                 m.setPackager (packager);
151                 ByteArrayOutputStream os = new ByteArrayOutputStream();
152     
153                 if (reread || transformer == null)
154                     transformer = tfactory.newTransformer(
155                         new StreamSource(xsltfile)
156                     );
157                 transformer.transform (
158                     new StreamSource(new ByteArrayInputStream (m.pack())),
159                     new StreamResult(os)
160                 );
161                 m.unpack (os.toByteArray());
162             } catch (Exception e) {
163                 throw new VetoException(e);
164             }
165             return m;
166         }
167     }
