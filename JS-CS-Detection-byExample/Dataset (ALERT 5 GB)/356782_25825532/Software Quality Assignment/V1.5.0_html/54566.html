50     package  org.jpos.security;
51     
52     import java.io.File;
53     import java.io.FileInputStream;
54     import java.io.FileOutputStream;
55     import java.util.Properties;
56     
57     import org.jpos.core.Configuration;
58     import org.jpos.core.ConfigurationException;
59     import org.jpos.core.ReConfigurable;
60     import org.jpos.iso.ISOUtil;

62     import org.jpos.util.LogSource;

66     /**
67      * Implements SecureKeyStore using a properties file.
68      * @author Hani S. Kirollos
69      * @version $Revision$ $Date$
70      * @see java.util.Properties
71      */
72     public class SimpleKeyFile
73             implements SecureKeyStore, ReConfigurable, LogSource  {
74         Properties props = new Properties();
75         File file;
76         String header = "Key File";
77         protected Logger logger = null;
78         protected String realm = null;
79     
80         public SimpleKeyFile () {
81         }
82     
83         public SimpleKeyFile (String keyFileName) throws SecureKeyStoreException
84         {
85             init(keyFileName);
86         }
87     
88         public void init (String keyFileName) throws SecureKeyStoreException {
89             file = new File(keyFileName);
90             try {
91                 if (!file.exists())
92                     file.createNewFile();
93                 load();
94             } catch (Exception e) {
95                 throw  new SecureKeyStoreException(e);
96             }
97         }
98     
99         public void setLogger (Logger logger, String realm) {
100             this.logger = logger;
101             this.realm = realm;
102         }
103     
104         public Logger getLogger () {
105             return  logger;
106         }
107     
108         public String getRealm () {
109             return  realm;
110         }
111     
112     
113         /**
114          *
115          * @param cfg configuration object
116          * @throws ConfigurationException
117          */
118         public void setConfiguration (Configuration cfg) throws ConfigurationException {
119             try {
120                 init(cfg.get("key-file"));
121                 header = cfg.get("file-header", header);
122             } catch (Exception e) {
123                 throw  new ConfigurationException(e);
124             }
125         }
126     
127     
128         public synchronized SecureKey getKey (String alias) throws SecureKeyStoreException {
129             SecureKey secureKey = null;
130             LogEvent evt = new LogEvent(this, "get-key");
131             evt.addMessage("alias", alias);
132             try {
133                 load();
134                 String keyClassName = getProperty(alias, "class");
135                 Class c = Class.forName(keyClassName);
136                 secureKey = (SecureKey)c.newInstance();
137                 if (!(secureKey instanceof SecureDESKey))
138                     throw  new SecureKeyStoreException("Unsupported SecureKey class: " +
139                             secureKey.getClass().getName());
140                 byte[] keyBytes = ISOUtil.hex2byte(getProperty(alias, "key"));
141                 short keyLength = Short.parseShort(getProperty(alias, "length"));
142                 String keyType = getProperty(alias, "type");
143                 byte[] KeyCheckValue = ISOUtil.hex2byte(getProperty(alias, "checkvalue"));
144                 secureKey = new SecureDESKey(keyLength, keyType, keyBytes, KeyCheckValue);
145                 evt.addMessage(secureKey);
146             } catch (Exception e) {
147                 evt.addMessage(e);
148                 throw  e instanceof SecureKeyStoreException ? (SecureKeyStoreException) e : new SecureKeyStoreException(e);
149             } finally {
150                 Logger.log(evt);
151             }
152             return  secureKey;
153         }
154     
155         public synchronized void setKey (String alias, SecureKey secureKey) throws SecureKeyStoreException {
156             LogEvent evt = new LogEvent(this, "set-key");
157             evt.addMessage("alias", alias);
158             evt.addMessage(secureKey);
159             try {
160                 if (!(secureKey instanceof SecureDESKey))
161                     throw  new SecureKeyStoreException("Unsupported SecureKey class: " +
162                             secureKey.getClass().getName());
163                 load();                 // load new changes (possibly made manually on the file)
164                 setProperty(alias, "class", secureKey.getClass().getName());
165                 setProperty(alias, "key", ISOUtil.hexString(secureKey.getKeyBytes()));
166                 setProperty(alias, "length", new Short(secureKey.getKeyLength()).toString());
167                 setProperty(alias, "type", secureKey.getKeyType());
168                 String keyCheckValueHexString = ISOUtil.hexString(((SecureDESKey)secureKey).getKeyCheckValue());
169                 setProperty(alias, "checkvalue", keyCheckValueHexString);
170                 store();
171             } catch (Exception e) {
172                 evt.addMessage(e);
173                 throw  e instanceof SecureKeyStoreException ? (SecureKeyStoreException) e : new SecureKeyStoreException(e);
174             } finally {
175                 Logger.log(evt);
176             }
177         }
178     
179         void load () throws SecureKeyStoreException {
180             FileInputStream in;
181             try {
182                 if (!file.canRead())
183                     throw  new SecureKeyStoreException("Can't read from file: " + file.getCanonicalPath());
184                 in = new FileInputStream(file);
185                 props.load(in);
186                 in.close();
187             } catch (Exception e) {
188                 throw  new SecureKeyStoreException(e);
189             }
190         }
191     
192         void store () throws SecureKeyStoreException {
193             FileOutputStream out;
194             try {
195                 if (!file.canWrite())
196                     throw  new SecureKeyStoreException("Can't write to file: " + file.getCanonicalPath());
197                 out = new FileOutputStream(file);
198                 props.store(out, header);
199                 out.flush();
200                 out.close();
201             } catch (Exception e) {
202                 throw  new SecureKeyStoreException(e);
203             }
204         }
205     
206         public String getProperty (String alias, String subName) throws SecureKeyStoreException {
207             // key here has nothing to do with cryptographic keys
208             String key = alias + "." + subName;
209             String value = props.getProperty(key);
210             if (value == null)
211                 throw  new SecureKeyStoreException("Key can't be retrieved. Can't get property: " + key);
212             return  value;
213         }
214     
215         public void setProperty (String alias, String subName, String value) {
216             // key here has nothing to do with cryptographic keys
217             String key = alias + "." + subName;
218             props.setProperty(key, value);
219         }
220     }
