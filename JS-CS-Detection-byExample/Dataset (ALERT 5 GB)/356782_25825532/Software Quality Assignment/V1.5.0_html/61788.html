80         public String getLogger () 
81         {
82             return log.getLogger().getName();

85         public void init ()
86         {
87             if (state != -1)
88                 return;
89             try {
90                 setModified (false);
91                 processorName = new ObjectName("MX4J:name=mx4j.tools.adaptor.http.XSLTProcessor");
92                 mx4j.tools.adaptor.http.XSLTProcessor processorMBean = new mx4j.tools.adaptor.http.XSLTProcessor();
93                 getServer().getMBeanServer().registerMBean(processorMBean,processorName);
94                 setProcessorName(processorName);
95                 if (user != null && password != null) {
96                     addAuthorization(user, password);
97                     setAuthenticationMethod ("basic");
98                 }
99                 state = QBean.STOPPED;
100             } catch (Exception e) {
101                 if (log != null)
102                     log.warn ("init", e);
103             }
104     
105         }
106         public void start()
107         {
108             if (state != QBean.DESTROYED &&
109                 state != QBean.STOPPED   &&
110                 state != QBean.FAILED)
111                return;
112     
113             this.state = QBean.STARTING;
114     
115             try {
116                super.start();
117             } catch (Throwable t) {
118                state = QBean.FAILED;
119                t.printStackTrace();
120                return;
121             }
122             state = QBean.STARTED;
123         }
124         public void stop ()
125         {
126             if (state != QBean.STARTED)
127                return;
128             state = QBean.STOPPING;
129             try {
130                super.stop();
131             } catch (Throwable t) {
132                state = QBean.FAILED;
133                t.printStackTrace();
134                return;
135             }
136             state = QBean.STOPPED;
137         }
138         public void destroy ()
139         {
140             if (state == QBean.DESTROYED)
141                return;
142             if (state != QBean.STOPPED)
143                stop();
144     
145             try {
146                 getServer().getMBeanServer().unregisterMBean(processorName);
147             } catch (Exception e) {
148                 if (log != null)
149                     log.warn ("destroy", e);
150             }
151     
152             state = QBean.DESTROYED;
153         }
154         public int getState () {
155             return state;
156         }
157         public String getStateAsString () {
158             return QBeanSupport.stateString[state];
159         }
160         public void setState (int state) {
161             this.state = state;
162         }

164             this.persist = persist ;
165         }

167         private Element createAttr (String name, String value, String type) {
168             Element e = new Element ("attr");
169             e.setAttribute ("name", name);
170             if (type != null)
171                 e.setAttribute ("type", type);
172             e.setText (value);
173             return e;
174         }
175         private Element createAttr (String name, String value) {
176             return createAttr (name, value, null);
177         }
178         public synchronized Element getPersist () {
179             setModified (false);
180             Element e = new Element (persist.getName());
181             Element classPath = persist.getChild ("classpath");
182             if (classPath != null)
183                 e.addContent (classPath);
184             e.setAttribute ("class", getClass().getName());
185             if (!e.getName().equals (getName ()))
186                 e.setAttribute ("name", getName());
187             if (loggerName != null)
188                 e.setAttribute ("logger", loggerName);
189             e.addContent (
190                 createAttr ("host", getHost())
191             );
192             e.addContent (
193                 createAttr (
194                     "port", Integer.toString (getPort()), "java.lang.Integer")
195             );
196             if (user != null && password != null) {
197                 e.addContent (createAttr ("user", user));
198                 e.addContent (createAttr ("password", password));
199             }
200             return (persist = e);
201         }
202         public synchronized void setModified (boolean modified) {
203             this.modified = modified;
204         }
205         public synchronized boolean isModified () {
206             return modified;
207         }
208         protected boolean running () {

210         }

216             setModified (true);
217             super.setPort (port);
218         }
