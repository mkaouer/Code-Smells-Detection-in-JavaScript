40     package org.jpos.iso;
41     
42     /**
43      * This interpreter does no conversion and leaves the input the same as the output.
44      * @author jonathan.oconnor@xcom.de
45      */
46     public class LiteralBinaryInterpreter implements BinaryInterpreter
47     {
48         /**
49          * The only instance of this interpreter.
50          */
51         public static final LiteralBinaryInterpreter INSTANCE = new LiteralBinaryInterpreter();
52     
53         /**
54          * Private constructor so we don't allow multiple instances.
55          */
56         private LiteralBinaryInterpreter()
57         {
58         }
59     
60         /**
61          * Copies the input to the output.
62          * @see org.jpos.iso.BinaryInterpreter#interpret(byte[], byte[], int)
63          */
64         public void interpret(byte[] data, byte[] b, int offset)
65         {
66             System.arraycopy(data, 0, b, offset, data.length);
67         }
68     
69         /**
70          * Copies the data out of the byte array.
71          * @see org.jpos.iso.BinaryInterpreter#uninterpret(byte[], int, int)
72          */
73         public byte[] uninterpret(byte[] rawData, int offset, int length)
74         {
75             byte[] ret = new byte[length];
76             System.arraycopy(rawData, offset, ret, 0, length);
77             return ret;
78         }
79     
80         /**
81          * Returns nBytes because we are not doing any conversion.
82          * @see org.jpos.iso.BinaryInterpreter#getPackedLength(int)
83          */
84         public int getPackedLength(int nBytes)
85         {
86             // TODO Auto-generated method stub
87             return nBytes;
88         }
89     }
