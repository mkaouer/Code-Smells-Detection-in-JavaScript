50     package org.jpos.core;
51     
52     import java.util.Iterator;
53     import java.util.List;
54     import java.util.Vector;
55     
56     
57     /**
58      * @author apr@cs.com.uy
59      * @version $Id$
60      * @since jPOS 1.1
61      *
62      * helper class maps BIN (Bank Issuer Number) to CardNames<br>
63      * We use a custom <strong>not-so-singleton</strong> pattern
64      * here. Although <strong>getDefault()</strong> returns a unique
65      * instance of CardSelector we have the chance to use multiple
66      * instances of CardSelector within the same JVM if ever needed.
67      */
68     public class CardSelector {
69         List list;
70         static CardSelector defaultInstance;
71     
72         public class Entry {
73             CardBrand card;
74             int low;
75             int high;
76             public Entry (CardBrand card, int low, int high) {
77                 this.card = card;
78                 this.low  = low;
79                 this.high = high;
80             }
81         }
82         public CardSelector() {
83             list = new Vector();
84         }
85         public CardSelector (List list) {
86             this.list=list;
87         }
88         /**
89          * @param card CardBrand
90          * @param low BIN
91          * @param high BIN
92          */
93         public void add (CardBrand card, int low, int high) {
94             list.add (new Entry (card, low, high));
95         }
96         /**
97          * @param card CardBrand
98          * @param bin
99          */
100         public void add (CardBrand card, int bin) {
101             list.add (new Entry (card, bin, bin));
102         }
103         public static void setDefault (CardSelector def) {
104             defaultInstance = def;
105         }
106         public static CardSelector getDefault () {
107             if (defaultInstance == null)
108                 defaultInstance = new CardSelector();
109             return defaultInstance;
110         }
111         public CardBrand getCardBrand(String bin) 
112             throws UnknownCardException
113         {
114             try {
115                 int b = Integer.parseInt (bin);
116                 Iterator i = list.iterator();
117                 while (i.hasNext()) {
118                     Entry e = (Entry) i.next();
119                     if ((b >= e.low) && (b <= e.high)) 
120                         return e.card;
121                 }
122             } catch (NumberFormatException x) { }
123             throw new UnknownCardException (bin);
124     
125         }
126         public void remove (CardBrand card) {
127             Iterator i = list.iterator();
128             while (i.hasNext()) {
129                 Entry e = (Entry) i.next();
130                 if (e.card == card)
131                     i.remove ();
132             }
133         }
134     }
