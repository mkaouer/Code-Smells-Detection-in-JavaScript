78     public class ISOMeter extends JComponent implements Runnable {
79         /**
80          * @serial
81          */
82         Color color = new Color (255, 255, 255);
83         /**
84          * @serial
85          */
86         Image im;
87         /**
88          * @serial
89          */
90         Graphics img;
91         /**
92          * @serial
93          */
94         Font fontBig, fontSmall;
95         /**
96          * @serial
97          */
98         String positiveText;
99         /**
100          * @serial
101          */
102         String negativeText;
103         /**
104          * @serial
105          */
106         Timer ti;
107         /**
108          * handle ISOMeter's counters outside of this class in order
109          * to reduce 'int' to 'String' conversions.
110          * @serial
111          */
112         String positiveCounter;
113         /**
114          * @serial
115          */
116         String negativeCounter;
117         /**
118          * @serial
119          */
120         int lastPositive;
121         /**
122          * @serial
123          */
124         int lastNegative;
125         /**
126          * @serial
127          */
128         boolean connected;
129         /**
130          * @serial
131          */
132         ISOChannelPanel parent;

134         final static int width    = 200;
135         final static int height   = 60;
136         final static int mass     = height/2;
137     
138         /**
139          * @serial
140          */
141         int[] yPoints;
142         /**
143          * @serial
144          */
145         int[] xPoints;
146         /**
147          * counter to keep the scrolling active
148          */
149         int continueScroll;
150         /**
151          * used to determine if to scroll mark to end of graph
152          */
153         boolean scroll = true;
154         /**
155          * Refresh panel in millseconds
156          */
157         int refreshPanel = 50;

160         private Graphics imbCopy;

162         public ISOMeter(ISOChannelPanel parent) {
163             super();
164             this.parent = parent;
165     
166             fontBig   = new Font ("Helvetica", Font.ITALIC, mass*3/4);
167             fontSmall = new Font ("Helvetica", Font.PLAIN, 10);
168             yPoints = new int[width];
169             xPoints = new int[width];
170             for (int i=0; i<width; i++) {
171                 xPoints[i] = i;
172                 yPoints[i] = mass;
173             }
174             positiveText  = null;
175             negativeText  = null;
176             positiveCounter = negativeCounter = "";
177             connected     = false;
178     
179             MouseListener mouseListener = new MouseAdapter() {
180                 public void mouseClicked(MouseEvent e) {
181                     showLogList();
182                 }
183             };
184             addMouseListener(mouseListener);
185         }
186     
187         public void start() {
188             Thread t = new Thread (this);
189             t.setPriority (Thread.NORM_PRIORITY-1);
190             t.setName ("ISOMeter");
191             t.start();
192         }
193     
194         public void showLogList() {
195             JFrame f = new JFrame(parent.getSymbolicName());
196             f.getContentPane().add(createLogList());
197             f.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
198             f.validate();
199             f.pack();
200             f.setSize(width,width+50);
201             f.show();
202         }
203     
204         public JComponent createLogList() {
205             final JList logList = new JList(parent.getLog());
206             JPanel A = new JPanel();
207             A.setLayout(new BorderLayout());
208     
209             MouseListener mouseListener = new MouseAdapter() {
210                 public void mouseClicked(MouseEvent e) {
211                     ISOMsg m = (ISOMsg) logList.getSelectedValue();
212                     if (m != null) {
213                         JFrame f = new JFrame(
214                             parent.getSymbolicName()+":"+m.toString());
215                         ISOMsgPanel p = new ISOMsgPanel(m);
216                         f.getContentPane().add(p);
217                         f.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
218                         f.pack();
219                         f.show();
220                     }
221                 }
222             };
223             logList.addMouseListener(mouseListener);
224     
225             logList.setPrototypeCellValue("9999 99999999 999999");
226             JScrollPane scrollPane = new JScrollPane(logList);
227             A.add(scrollPane, BorderLayout.CENTER);
228             return A;
229         }
230     
231         public void setValue(int val) {
232             int y = mass - (val*height/2000);
233             yPoints[width-1] = y;
234             continueScroll = width;
235             scroll();
236         }
237     
238         public void setScroll (boolean scroll) {
239             this.scroll = scroll;
240         }
241         public void setRefresh (int refreshPanel) {
242             if (refreshPanel > 0)
243                 this.refreshPanel = refreshPanel;
244         }
245         public void setConnected(boolean connected) {
246             if (this.connected != connected)
247                 if (!scroll)
248                     if (connected)
249                         continueScroll = width;
250                     else
251                         continueScroll = 1;
252             this.connected = connected;
253         }
254         public void setPositiveCounter(String s) {
255             positiveCounter = s;
256         }
257         public void setNegativeCounter(String s){
258             negativeCounter = s;
259         }
260         public void setValue(int val, String textString) {
261             setValue(val);
262             if (val < 0) {
263                 negativeText = textString;
264                 lastNegative = 0;
265             }
266             else {
267                 positiveText = textString;
268                 lastPositive = 0;
269             }
270         }
271         public void paint (Graphics g) {
272             plot();
273             g.drawImage (im, 0, 0, null);
274         }
275         public Dimension getPreferredSize() {
276             return new Dimension(width, height);
277         }
278         private void scroll() {
279             for (int i=0; i<width-1; i++) 
280                 yPoints[i] = yPoints[i+1];
281             if (continueScroll > 0)
282                 continueScroll--;
283         }
284         public void plot() {
285            if (im == null) {
286                 im = createImage(width, height);
287                 img = im.getGraphics ();
288                 img.setColor (Color.black);
289                 img.fillRoundRect (0, 0, width, height, 10, 10);
290                 img.clipRect (0, 0, width, height);
291                 plotGrid();
292     
293                 /* save a copy of the image */
294                 imb = createImage(width, height);
295                 imbCopy = imb.getGraphics ();
296                 imbCopy.drawImage (im, 0, 0, this);
297             }
298             img.drawImage (imb, 0, 0, this);
299             if (continueScroll > 0)
300                 scroll();
301             plotText(positiveText, lastPositive++, 3, mass-3);
302             plotText(negativeText, lastNegative++, 3, height-3);
303             plotCounters(positiveCounter, negativeCounter);
304             img.setColor (connected ? Color.green : Color.red);
305             img.drawPolyline(xPoints, yPoints, width);
306         }
307         private void plotGrid() {
308             img.setColor(Color.blue);
309             for (int i=0; i<width; i++)
310                 if (i % 20 == 0) 
311                     img.drawLine(i,0,i,height);
312             for (int i=-1000; i<1000; i+= 200) {
313                 int y = mass + (i*height/2000);
314                 img.drawLine(0,y,width,y);
315             }
316         }
317         private void plotText(String t, int l, int x, int y) {
318             if (t != null && l < 20) {
319                 img.setColor(Color.lightGray);
320                 img.setFont(fontBig);
321                 img.drawString (t, x, y);
322             }
323         }
324         private void plotCounters(String p, String n) {
325             img.setColor(Color.lightGray);
326             img.setFont(fontSmall);
327             img.drawString (p, width-45, 13);
328             img.drawString (n, width-45, height-3);
329         }
330         public void run () {
331             for (;;) {
332                 if (continueScroll > 0)
333                     repaint();
334                 try { 
335                     Thread.sleep(refreshPanel);
336                 } catch (InterruptedException e) { }
337             }
338         }
339         public void update (Graphics g) {
340             paint (g);
341         }
342     }
