25             super(arg0);

28         protected void setUp() throws Exception {
29             super.setUp();        
30             dirPoll = new DirPoll();
31             dirPoll.setPath(BASE_PATH);
32             dirPoll.setProcessor(new DirPoll.FileProcessor() {
33                 public void process(File name) throws DirPollException { 
34                     System.out.println ("File processed " + name.toString());
35                     fileProcessed = true;
36                 }
37             });
38             dirPoll.createDirs();       
39             new Thread(dirPoll).start();
40         }
41         
42         protected void tearDown() throws Exception {
43             super.tearDown();
44             dirPoll.destroy();
45             File archiveDir = new File(ARCHIVE_DIR);
46             File[] files = archiveDir.listFiles();
47             for (int i = 0; i < files.length; i++) {
48                 files[i].delete();
49             } 
50         }

52         public void testArchiveFile() throws IOException, InterruptedException {
53             String dateFormatString = "yyyyMMddHHmmss";
54             String filename = "dodgyTestFile.test";
55             
56             dirPoll.setShouldArchive(true);
57             dirPoll.setArchiveDateFormat(dateFormatString);
58             dirPoll.setShouldTimestampArchive(true);
59             testIncomingFile = new File(BASE_PATH + "/request/" + filename);
60             FileOutputStream fileOutputStream = new FileOutputStream(testIncomingFile);
61             fileOutputStream.write(new String("test").getBytes());
62             fileOutputStream.flush();
63             fileOutputStream.close();
64             assertTrue(testIncomingFile.exists());
65             
66             while(!fileProcessed) {            
67                 Thread.yield();
68             }
69             System.out.println ("");
70             Thread.sleep(200);
71             File archiveDirectory = new File(ARCHIVE_DIR);        
72             assertEquals(1, archiveDirectory.listFiles().length);
73             assertEquals((dateFormatString.length() + filename.length() + 1), archiveDirectory.listFiles()[0].getName().length());
74         }
