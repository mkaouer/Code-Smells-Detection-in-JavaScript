52     import java.io.IOException;
53     import java.io.EOFException;
54     import java.net.ServerSocket;
55     import org.jpos.iso.ISOMsg;
56     import org.jpos.iso.ISOField;
57     import org.jpos.iso.ServerChannel;
58     import org.jpos.iso.ISOException;
59     import org.jpos.iso.ISOFactory;
60     import org.jpos.iso.ISOPackager;
61     import org.jpos.iso.ISOUtil;
62     import org.jpos.util.LogSource;
63     import org.jpos.util.LogEvent;
64     import org.jpos.util.SimpleLogListener;
65     import org.jpos.util.Logger;
66     import org.jpos.util.LogSource;
67     import org.jpos.core.Configuration;
68     import org.jpos.core.SimpleConfiguration;
69     
70     public class Test implements Runnable, LogSource {
71         Logger logger;
72         String realm;
73         public static final String CFG_PORT = "simpleserver.port";
74         boolean testMux;
75     
76         public Test (Logger logger, String realm, boolean testMux) {
77     	super();
78     	setLogger (logger, realm);
79     	this.testMux = testMux;
80         }
81         public void setLogger (Logger logger, String realm) {
82     	this.logger = logger;
83     	this.realm  = realm;
84         }
85         public String getRealm () {
86     	return realm;
87         }
88         public Logger getLogger() {
89     	return logger;
90         }
91         protected class Session implements Runnable, LogSource {
92             ServerChannel channel;
93             protected Session(ServerChannel channel) {
94                 this.channel = channel;
95             }
96             public void run() {
97     	    Logger.log (new LogEvent (this, "SessionStart"));
98                 try {
99                     for (;;) {
100                         ISOMsg m = channel.receive();
101     		    m.setResponseMTI();
102                         m.set (new ISOField(39, "00"));
103     		    if (testMux)
104     			Thread.sleep (10);  // simulate server delay
105                         channel.send(m);
106     		    if (testMux && m.getMTI().equals ("0810")) {
107     			// on 'testMux' mode we originate an unexpected
108     			// message that will be handled by ISOMUX's ISORequestListener
109     			m.setMTI ("0800");
110     			channel.send(m);
111     		    }
112                     }
113     	    } catch (EOFException e) {
114                 } catch (Exception e) { 
115     		Logger.log (new LogEvent (this, "SessionError", e));
116                 } 
117     	    Logger.log (new LogEvent (this, "SessionEnd"));
118             }
119     	public void setLogger (Logger logger, String realm) { }
120     	public String getRealm () {
121     	    return realm + ".session";
122     	}
123     	public Logger getLogger() {
124     	    return logger;
125     	}
126         }
127     
128         public void run() {
129             ServerChannel  channel;
130     	String cfgFile    = System.getProperties().getProperty("jpos.config");
131             try {
132     	    Configuration cfg = new SimpleConfiguration (cfgFile);
133     	    int port          = cfg.getInt (CFG_PORT);
134                 ServerSocket serverSocket = new ServerSocket(port);
135     	    Logger.log (
136     	       new LogEvent (this, "message", "listening on port "+port)
137     	    );
138                 for (;;) {
139     		channel = (ServerChannel) ISOFactory.newChannel 
140     		    (cfg, "simpleserver", logger, realm);
141                     channel.accept(serverSocket);
142                     Thread t = new Thread (new Session(channel));
143                     t.setDaemon(true);
144                     t.start();
145                 }
146             } catch (ISOException e) {
147     	    Logger.log (new LogEvent (this, "mainrun", e));
148             } catch (IOException e) {
149     	    Logger.log (new LogEvent (this, "mainrun", e));
150             }
151         }
152         public static void main(String args[]) {
153     	boolean testMux = args.length > 0 && args[0].equals ("-testmux");
154     	Logger logger = new Logger();
155     	logger.addListener (new SimpleLogListener (System.out));
156             new Thread(new Test(logger, "SimpleServer", testMux)).start();
157         }
158     }
