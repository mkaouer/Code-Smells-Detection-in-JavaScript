73     /**
74      * ISOChannel is an abstract class that provides functionality that
75      * allows the transmision and reception of ISO 8583 Messages
76      * over a TCP/IP session.
77      * <p>
78      * It is not thread-safe, ISOMUX takes care of the
79      * synchronization details

82      * such as ISOChannelPanel.
83      * <br>
84      * It now support the new Logger architecture so we will
85      * probably setup ISOChannelPanel to be a LogListener insteado
86      * of being an Observer in future releases.

88      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>

92      * @see ISOException

243         protected void connect (Socket socket) 
244     	throws IOException, SocketException
245         {
246             this.socket = socket;
247     	applyTimeout();
248     
249             serverIn = new DataInputStream (
250                 new BufferedInputStream (socket.getInputStream ())
251             );
252             serverOut = new DataOutputStream(
253                 new BufferedOutputStream(socket.getOutputStream())
254             );
255             usable = true;
256             cnt[CONNECT]++;
257             setChanged();
258             notifyObservers();
259         }

291         /**
292          * Connects client ISOChannel to server
293          * @exception IOException
294          */
295         public void connect () throws IOException {
296     	LogEvent evt = new LogEvent (this, "connect");
297     	try {
298                 if (serverSocket != null) {
299     		evt.addMessage ("local port "+serverSocket.getLocalPort()
300     		    +" remote host "+serverSocket.getInetAddress());
301     		accept(serverSocket);
302     	    }
303     	    else {
304     		evt.addMessage (host+":"+port);
305     		connect(newSocket (host, port));
306     	    }
307     	    applyTimeout();
308     	    Logger.log (evt);
309     	} catch (ConnectException e) {
310     	    Logger.log (new LogEvent (this, "connection-refused",
311     		getHost()+":"+getPort())
312     	    );
313     	} catch (IOException e) {
314     	    evt.addMessage (e);
315     	    Logger.log (evt);
316     	    throw e;
317     	}
318         }

365         /**
366          * sends an ISOMsg over the TCP/IP session
367          * @param m the Message to be sent
368          * @exception IOException
369          * @exception ISOException
370          * @exception ISOFilter.VetoException;
371          */
372         public void send (ISOMsg m) 
373     	throws IOException, ISOException, VetoException
374         {
375     	LogEvent evt = new LogEvent (this, "send");
376     	evt.addMessage (m);
377     	try {
378     	    if (!isConnected())
379     		throw new ISOException ("unconnected ISOChannel");
380     	    m.setDirection(ISOMsg.OUTGOING);
381     	    m = applyOutgoingFilters (m, evt);
382     	    m.setDirection(ISOMsg.OUTGOING); // filter may have drop this info
383     	    m.setPackager (getDynamicPackager(m));
384     	    byte[] b = m.pack();
385     	    synchronized (serverOut) {
386     		sendMessageLength(b.length + getHeaderLength());
387     		sendMessageHeader(m, b.length);
388     		serverOut.write(b, 0, b.length);
389     		sendMessageTrailler(m, b.length);
390     		serverOut.flush ();
391     	    }
392     	    cnt[TX]++;
393     	    setChanged();
394     	    notifyObservers(m);
395     	} catch (VetoException e) {
396     	    evt.addMessage (e);
397     	    throw e;
398     	} catch (ISOException e) {
399     	    evt.addMessage (e);
400     	    throw e;
401     	} catch (IOException e) {
402     	    evt.addMessage (e);
403     	    throw e;
404     	} finally {
405     	    Logger.log (evt);
406     	}
407         }

412         /**
413          * Waits and receive an ISOMsg over the TCP/IP session
414          * @return the Message received
415          * @exception IOException
416          * @exception ISOException
417          */
418         public ISOMsg receive() throws IOException, ISOException {
419             byte[] b, header=null;
420     	LogEvent evt = new LogEvent (this, "receive");
421     	ISOMsg m = new ISOMsg();
422     	try {
423     	    if (!isConnected())
424     		throw new ISOException ("unconnected ISOChannel");
425     
426     	    synchronized (serverIn) {
427     		int len  = getMessageLength();
428     		int hLen = getHeaderLength();
429     
430     		if (len == -1) {
431     		    header = new byte [hLen];
432     		    serverIn.readFully(header,0,hLen);
433     		    b = streamReceive();
434     		}
435     		else if (len > 10 && len <= 4096) {
436     		    int l;
437     		    if (hLen > 0) {
438     			// ignore message header (TPDU)
439     			header = new byte [hLen];
440     			serverIn.readFully(header,0,hLen);
441     			if (isRejected(header))
442     			    throw new ISOException 
443     				("Unhandled Rejected Message");
444     			len -= hLen;
445     		    }
446     		    b = new byte[len];
447     		    serverIn.readFully(b,0,len);
448     		}
449     		else
450     		    throw new ISOException(
451     			"receive length " +len + " seems extrange");
452     	    }
453     	    m.setPackager (getDynamicPackager(b));
454     	    if (b.length > 0)  // Ignore NULL messages
455     		m.unpack (b);
456     
457     	    if (header != null && header.length > 0)
458     		m.setHeader(header);
459     	    m.setDirection(ISOMsg.INCOMING);
460     	    m = applyIncomingFilters (m, evt);
461     	    m.setDirection(ISOMsg.INCOMING);
462     	    evt.addMessage (m);
463     	    cnt[RX]++;
464     	    setChanged();
465     	    notifyObservers(m);
466     	} catch (ISOException e) {
467     	    evt.addMessage (e);
468     	    throw e;
469     	} catch (EOFException e) {
470     	    evt.addMessage ("<peer-disconnect/>");
471     	    throw e;
472     	} catch (IOException e) { 
473     	    if (usable) 
474     		evt.addMessage (e);
475     	    throw e;
476     	} finally {
477     	    Logger.log (evt);
478     	}
479     	return m;
480         }

489         /**
490          * disconnects the TCP/IP session. The instance is ready for
491          * a reconnection. There is no need to create a new ISOChannel<br>
492          * @exception IOException
493          */
494         public void disconnect () throws IOException {
495     	LogEvent evt = new LogEvent (this, "disconnect");
496             if (serverSocket != null) 
497     	    evt.addMessage ("local port "+serverSocket.getLocalPort()
498     		+" remote host "+serverSocket.getInetAddress());
499     	else
500     	    evt.addMessage (host+":"+port);
501     	try {
502     	    usable = false;
503     	    setChanged();
504     	    notifyObservers();
505     	    serverIn  = null;
506     	    serverOut = null;
507     	    if (socket != null)
508     		socket.close ();
509     	    socket = null;
510     	    Logger.log (evt);
511     	} catch (IOException e) {
512     	    evt.addMessage (e);
513     	    Logger.log (evt);
514     	    throw e;
515     	}
516         }

623         protected ISOMsg applyOutgoingFilters (ISOMsg m, LogEvent evt) 
624     	throws VetoException
625         {
626     	Iterator iter  = outgoingFilters.iterator();
627     	while (iter.hasNext())
628     	    m = ((ISOFilter) iter.next()).filter (this, m, evt);
629     	return m;
630         }

635     	while (iter.hasNext())
636     	    m = ((ISOFilter) iter.next()).filter (this, m, evt);
637     	return m;

639        /**

663     	    } catch (SocketException e) {
664     		throw new ConfigurationException (e);
665     	    }
