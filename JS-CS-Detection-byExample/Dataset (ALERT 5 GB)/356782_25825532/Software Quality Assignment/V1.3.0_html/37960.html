63     /**
64      * <b><i>j</i>POS</b> applications usually constructs
65      * objects such as ISOChannels, ISOMUXs, etc. based on 
66      * configuration parameters.<br>
67      * We'll group here factory methods for those repeated tasks
68      *
69      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a> 
70      * @version $Revision$ $Date$
71      * @deprecated This whole class will be deprecated, use QSP application instead
72      *
73      * @deprecated Use QSP instead
74      */
75     public class ISOFactory {
76         /**
77          * Creates a client or server ISOChannel based on Configuration properties:
78          * <br>
79          * If [prefix].host is available newChannel will create an unconnected
80          * client ISOChannel.
81          * <br>
82          * If [prefix].host is not available (null) it will create an unconnected
83          * server ISOChannel.
84          *
85          * <ul>
86          *  <li>[prefix].channel
87          *  <li>[prefix].packager
88          *  <li>[prefix].header
89          *  <li>[prefix].port
90          *  <li>[prefix].host
91          * </ul>
92          *
93          * @see org.jpos.core.Configuration
94          * @see ISOChannel
95          *
96          * @param cfg Configuration Object
97          * @param prefix propertyPrefix
98          * @param logger optional logger (may be null)
99          * @param realm  optional realm  (may be null if logger is null)
100          * @exception ISOException catches all possible exceptions into one
101          */
102         public static ISOChannel newChannel
103     	(Configuration cfg, String prefix, Logger logger, String realm)
104     	throws ISOException
105         {
106     	String channelName  = cfg.get    (prefix + ".channel", null);
107     	String packagerName = cfg.get    (prefix + ".packager", null);
108     	String header       = cfg.get    (prefix + ".header", null);
109     	String host         = cfg.get    (prefix + ".host", null);
110     	int    port         = cfg.getInt (prefix + ".port");
111     	int    timeout      = cfg.getInt (prefix + ".timeout");
112             ISOChannel channel  = newChannel
113     	    (channelName, packagerName, logger, realm);
114     	if (timeout != 0 && channel instanceof BaseChannel) {
115     	    try {
116     		((BaseChannel)channel).setTimeout (timeout);
117     	    } catch (SocketException e) {
118     		throw new ISOException (e);
119     	    }
120     	}
121     	if (host != null && channel instanceof ClientChannel)
122     	    ((ClientChannel)channel).setHost (host, port);
123     	if (header != null) {
124     	    if (channel instanceof RawChannel) {
125     		((RawChannel)channel).setHeader (
126     		    ISOUtil.str2bcd(header, false)
127     		);
128     	    } else if (channel instanceof BASE24Channel) {
129     		((BASE24Channel)channel).setHeader (header.getBytes());
130     	    } else if (channel instanceof PADChannel) {
131     		((PADChannel)channel).setHeader (
132     		    ISOUtil.hex2byte
133     			(header.getBytes(), 0, header.getBytes().length)
134     		);
135     	    }
136     	}
137             return channel;
138         }

140         /**
141          * Creates an ISOChannel, possibly assigning packager and Logger
142          * @see ISOChannel
143          *
144          * @param channelName channel class name
145          * @param packagerName packager class name
146          * @param logger optional logger (may be null)
147          * @param realm  optional realm  (may be null if logger is null)
148          * @exception ISOException catches all possible exceptions into one
149          */
150         public static ISOChannel newChannel
151     	(String channelName, String packagerName, Logger logger, String realm)
152     	throws ISOException
153         {
154             ISOChannel channel  = null;
155     	ISOPackager packager= null;
156             try {
157                 Class c = Class.forName(channelName);
158                 if (c != null) {
159                     channel = (ISOChannel) c.newInstance();
160     		if (packagerName != null) {
161     		    Class p = Class.forName(packagerName);
162     		    packager = (ISOPackager) p.newInstance();
163     		    channel.setPackager(packager);
164     		}
165     		if (logger != null && (channel instanceof LogSource))
166     		    ((LogSource) channel) .
167     			setLogger (logger, realm + ".channel");
168                 }
169             } catch (ClassNotFoundException e) {
170     	    throw new ISOException ("newChannel:"+channelName, e);
171             } catch (InstantiationException e) {
172     	    throw new ISOException ("newChannel:"+channelName, e);
173             } catch (IllegalAccessException e) {
174     	    throw new ISOException ("newChannel:"+channelName, e);
175     	}
176             return channel;
177         }

179         /**
180          * Creates an ISOChannel based on Configuration property [prefix].packager
181          *
182          * @see org.jpos.core.Configuration
183          * @see ISOPackager
184          *
185          * @param cfg Configuration Object
186          * @param prefix propertyPrefix
187          * @param logger optional logger (may be null)
188          * @param realm  optional realm  (may be null if logger is null)
189          * @exception ISOException catches all possible exceptions into one
190          */
191         public static ISOPackager newPackager
192     	(Configuration cfg, String prefix, Logger logger, String realm)
193     	throws ISOException
194         {
195     	ISOPackager packager = null;
196     	String packagerName = cfg.get    (prefix + ".packager");
197             try {
198                 Class p = Class.forName(packagerName);
199                 if (p != null) {
200     		packager = (ISOPackager) p.newInstance();
201     		if (logger != null)
202     		    packager.setLogger (logger, realm + ".packager");
203                 }
204             } catch (ClassNotFoundException e) {
205     	    throw new ISOException ("newPackager", e);
206             } catch (InstantiationException e) {
207     	    throw new ISOException ("newPackager", e);
208             } catch (IllegalAccessException e) {
209     	    throw new ISOException ("newPackager", e);
210     	}
211             return packager;
212         }

214         /**
215          * Create an ISOChannel, builds an ISOMUX around it and start it in
216          * a newly created daemon Thread.
217          * @see ISOFactory#newChannel
218          * @see org.jpos.core.Configuration
219          * @see ISOChannel
220          *
221          * @param cfg Configuration Object
222          * @param prefix propertyPrefix
223          * @param logger optional logger (may be null)
224          * @param realm  optional realm  (may be null if logger is null)
225          * @exception ISOException catches all possible exceptions into one
226          */
227         public static ISOMUX newMUX
228     	(Configuration cfg, String prefix, Logger logger, String realm)
229     	throws ISOException
230         {
231     	ISOChannel channel = newChannel (cfg, prefix, logger, realm);
232     	ISOMUX mux = new ISOMUX (channel, logger, realm + ".mux");
233     	Thread t = new Thread (mux);
234     	t.setName (realm + ".mux");
235     	t.setDaemon (true);
236     	t.start();
237     	return mux;
238         }

242          * @throws NameRegistrar.NotFoundException;
243          * @see NameRegistrar
244          */
245         public static ISOChannel getChannel (String name)
246     	throws NameRegistrar.NotFoundException
247         {
248     	return (ISOChannel) NameRegistrar.get ("channel."+name);
