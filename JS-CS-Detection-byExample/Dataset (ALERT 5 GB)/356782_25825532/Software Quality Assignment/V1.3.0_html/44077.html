201         private void recalcMaxField() {
202             ISOComponent c;
203             maxField = 0;
204             for (int i=1; i<=192; i++)
205                 if ((c = (ISOComponent) fields.get (new Integer (i))) != null)
206                     maxField = i;
207             maxFieldDirty = false;
208         }

219             return packager;
220         }
221         /**
222          * Set a field within this message
223          * @param c - a component
224          * @exception ISOException
225          */
226         public void set (ISOComponent c) throws ISOException {
227             Integer i = (Integer) c.getKey();
228             fields.put (i, c);
229             if (i.intValue() > maxField)
230                 maxField = i.intValue();
231             dirty = true;
232         }
233         /**
234          * Unset a field if it exists, otherwise ignore.
235          * @param fldno - the field number
236          */
237         public void unset (int fldno) {
238             if (fields.remove (new Integer (fldno)) != null)
239     	    dirty = maxFieldDirty = true;
240         }

251         /**
252          * setup BitMap
253          * @exception ISOException
254          */
255         public void recalcBitMap () throws ISOException {
256             ISOComponent c;
257             if (!dirty)
258                 return;
259     
260     	if(maxField>128)
261     	{
262     	    BitSet bmap=new BitSet(64);
263     	    for (int i=1; i<=64; i++)
264     		if((c=(ISOComponent) fields.get(new Integer (i+128))) != null) 
265     		    bmap.set (i);
266     	    set (new ISOBitMap (65, bmap));
267     	}
268     		
269             BitSet bmap = new BitSet (getMaxField() > 64 ? 128 : 64);
270     		int tmpMaxField=maxField > 128 ? 128 : maxField;
271     
272             for (int i=1; i<=tmpMaxField; i++)
273                 if ((c = (ISOComponent) fields.get (new Integer (i))) != null) 
274                     bmap.set (i);
275             set (new ISOBitMap (-1, bmap));
276     		
277             dirty = false;
278         }

307         /**
308          * dump the message to a PrintStream. The output is sorta
309          * XML, intended to be easily parsed.
310          * <br>
311          * Each component is responsible for its own dump function,
312          * ISOMsg just calls dump on every valid field.
313          * @param p - print stream
314          * @param indent - optional indent string
315          */
316         public void dump (PrintStream p, String indent) {
317             ISOComponent c;
318     	p.print (indent + "<" + XMLPackager.ISOMSG_TAG);
319     	switch (direction) {
320     	    case INCOMING:
321     		p.print (" direction=\"incoming\"");
322     		break;
323     	    case OUTGOING:
324     		p.print (" direction=\"outgoing\"");
325     		break;
326     	}
327     	if (fieldNumber != -1)
328     	    p.print (" "+XMLPackager.ID_ATTR +"=\""+fieldNumber +"\"");
329     	p.println (">");
330     	String newIndent = indent + "  ";
331             for (int i=0; i<=maxField; i++) {
332                 if ((c = (ISOComponent) fields.get (new Integer (i))) != null)
333                     c.dump (p, newIndent);
334     	    //
335     	    // Uncomment to include bitmaps within logs
336     	    // 
337     	    // if (i == 0) {
338     	    //  if ((c = (ISOComponent) fields.get (new Integer (-1))) != null)
339     	    //    c.dump (p, newIndent);
340     	    // }
341     	    //
342     	}
343     
344             p.println (indent + "</" + XMLPackager.ISOMSG_TAG+">");
345         }

381             try {
382                 ISOMsg m = (ISOMsg) super.clone();

384                 return (Object) m;
385             } catch (CloneNotSupportedException e) {
386                 throw new InternalError();
387             }

390         /**
391          * Partially clone an ISOMsg
392          * @param fields int array of fields to go
393          * @return new ISOMsg instance
394          */
395         public Object clone(int[] fields) {
396             try {
397                 ISOMsg m = (ISOMsg) super.clone();
398                 m.fields = new Hashtable();
399                 for (int i=0; i<fields.length; i++) {
400                     if (hasField(fields[i])) {
401                         try {
402                             m.set (getComponent(fields[i]));
403                         } catch (ISOException e) { 
404                             // it should never happen
405                         }
406                     }
407                 }
408                 return (Object) m;
409             } catch (CloneNotSupportedException e) {
410                 throw new InternalError();
411             }
412         }

414         /**
415          * add all fields present on received parameter to this ISOMsg<br>
416          * please note that received fields take precedence over 
417          * existing ones (simplifying card agent message creation 
418          * and template handling)
419          * @param m ISOMsg to merge
420          */
421         public void merge (ISOMsg m) {
422     	for (int i=0; i<=m.getMaxField(); i++) 
423     	    try {
424     		if (m.hasField(i))
425     		    set (m.getComponent(i));
426     	    } catch (ISOException e) {
427     		// should never happen 
428     	    }
429         }

431         /**
432          * @return a string suitable for a log
433          */
434         public String toString() {
435             StringBuffer s = new StringBuffer();
436             if (isIncoming())
437                 s.append("<-- ");
438             else if (isOutgoing())
439                 s.append("--> ");
440             else
441                 s.append("    ");
442     
443             try {
444                 s.append((String) getValue(0));
445                 if (hasField(11)) {
446                     s.append(' ');
447                     s.append((String) getValue(11));
448                 }
449                 if (hasField(41)) {
450                     s.append(' ');
451                     s.append((String) getValue(41));
452                 }
453             } catch (ISOException e) { }
454             return s.toString();
455         }

471          * @param mti new MTI

474         public void setMTI (String mti) throws ISOException {
475     	if (isInner())
476     	    throw new ISOException ("can't setMTI on inner message");
477     	set (new ISOField (0, mti));

479         /**
480          * @return current MTI
481          * @exception ISOException on inner message or MTI not set
482          */
483         public String getMTI() throws ISOException {
484     	if (isInner())
485     	    throw new ISOException ("can't getMTI on inner message");
486     	else if (!hasField(0))
487     	    throw new ISOException ("MTI not available");
488     	return (String) getValue(0);
489         }
490         /**
491          * @return true if message "seems to be" a request
492          * @exception ISOException on MTI not set
493          */
494         public boolean isRequest() throws ISOException {
495     	return Character.getNumericValue(getMTI().charAt (2))%2 == 0;
496         }
497         /**
498          * @return true if message "seems not to be" a request
499          * @exception ISOException on MTI not set
500          */
501         public boolean isResponse() throws ISOException {
502     	return !isRequest();
503         }
504         /**
505          * @return true if message is Retransmission
506          * @exception ISOException on MTI not set
507          */
508         public boolean isRetransmission() throws ISOException {
509     	return getMTI().charAt(3) == '1';
510         }
511         /**
512          * sets an appropiate response MTI<br>
513          * i.e. 0110 becomes 0120<br>
514          * i.e. 0111 becomes 0120<br>
515          * i.e. 1201 becomes 1210<br>
516          * @exception ISOException on MTI not set or it is not a request
517          */
518         public void setResponseMTI() throws ISOException {
519     	if (!isRequest())
520     	    throw new ISOException ("not a request - can't set response MTI");
521     
522     	String mti = getMTI();
523     	set (new ISOField (0,
524     	    mti.substring(0,2)
525     		+(Character.getNumericValue(getMTI().charAt (2))+1) + "0"
526     	    )
527     	);
528         }

530         public void writeExternal (ObjectOutput out) throws IOException {
531             try {
532                 ISOPackager p = new ISO93BPackager();
533                 byte[] b;
534                 synchronized (this) {
535                     recalcBitMap();
536                     b = getInternalPackager().pack(this);
537                 }
538                 out.writeObject (b);
539             } catch (ISOException e) {
540                 throw new IOException (e.getMessage());
541             }
542         }
543         public void readExternal  (ObjectInput in) 
544             throws IOException, ClassNotFoundException
545         {
546             byte[] b = (byte[]) in.readObject();
547             try {
548                 synchronized (this) {
549                     getInternalPackager().unpack(this, b);
550                 }
551             } catch (ISOException e) {
552                 throw new IOException (e.getMessage());
553             }
554         }
555         protected ISOPackager getInternalPackager() {
556             synchronized (ISOMsg.class) {
557                 if (internalPackager == null)
558                     internalPackager = new ISO93BPackager();
559             }
560             return internalPackager;
561         }
