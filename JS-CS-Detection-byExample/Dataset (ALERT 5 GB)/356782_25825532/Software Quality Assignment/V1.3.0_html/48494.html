70     /**
71      * Accept ServerChannel sessions and forwards them to ISORequestListeners
72      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
73      * @version $Revision$ $Date$
74      */
75     public class ISOServer extends Observable 
76         implements LogSource, Runnable, Observer
77     {
78         int port;
79         ISOChannel clientSideChannel;
80         Class clientSideChannelClass;
81         ISOPackager clientPackager;
82         Collection clientOutgoingFilters, clientIncomingFilters, listeners;
83         ThreadPool pool;
84         public static final int DEFAULT_MAX_THREADS = 100;
85         String name;
86         protected Logger logger;
87         protected String realm;
88     
89        /**
90         * @param port port to listen
91         * @param clientSide client side ISOChannel (where we accept connections)
92         * @param pool ThreadPool (created if null)
93         */
94         public ISOServer(int port, ServerChannel clientSide, ThreadPool pool) {
95     	super();
96     	this.port = port;
97     	this.clientSideChannel = clientSide;
98     	this.clientSideChannelClass = clientSide.getClass();
99     	this.clientPackager = clientSide.getPackager();
100     	if (clientSide instanceof FilteredChannel) {
101     	    FilteredChannel fc = (FilteredChannel) clientSide;
102     	    this.clientOutgoingFilters = fc.getOutgoingFilters();
103     	    this.clientIncomingFilters = fc.getIncomingFilters();
104     	}
105     	this.pool = (pool == null) ?  
106     	    new ThreadPool (1, DEFAULT_MAX_THREADS) : pool;
107     	listeners = new Vector();
108     	name = "";
109         }
110     
111         protected class Session implements Runnable, LogSource {
112             ServerChannel channel;
113             protected Session(ServerChannel channel) {
114                 this.channel = channel;
115             }
116             public void run() {
117     	    Logger.log (new LogEvent (this, "SessionStart"));
118                 try {
119                     for (;;) {
120                         ISOMsg m = channel.receive();
121     		    Iterator iter = listeners.iterator();
122     		    while (iter.hasNext())
123     			if (((ISORequestListener)iter.next()).process
124     			    (channel, m)) 
125     			    break;
126                     }
127     	    } catch (EOFException e) {
128                 } catch (Exception e) { 
129     		Logger.log (new LogEvent (this, "SessionError", e));
130                 } 
131     	    Logger.log (new LogEvent (this, "SessionEnd"));
132             }
133     	public void setLogger (Logger logger, String realm) { 
134     	}
135     	public String getRealm () {
136     	    return ISOServer.this.getRealm() + ".session";
137     	}
138     	public Logger getLogger() {
139     	    return ISOServer.this.getLogger();
140     	}
141         }
142        /**
143         * add an ISORequestListener
144         * @param l request listener to be added
145         * @see ISORequestListener
146         */
147         public void addISORequestListener(ISORequestListener l) {
148     	listeners.add (l);
149         }
150        /**
151         * remove an ISORequestListener
152         * @param l a request listener to be removed
153         * @see ISORequestListener
154         */
155         public void removeISORequestListener(ISORequestListener l) {
156     	listeners.remove (l);
157         }
158     
159         public void run() {
160             ServerChannel  channel;
161     	for (;;) {
162     	    try {
163     		ServerSocket serverSocket = new ServerSocket(port);
164     		Logger.log (new LogEvent (this, "iso-server", 
165     		    "listening on port "+port));
166     		for (;;) {
167     		    try {
168     			channel = (ServerChannel) 
169     			    clientSideChannelClass.newInstance();
170     			channel.setPackager (clientPackager);
171     			if (clientSideChannel instanceof BaseChannel)
172     			    ((BaseChannel)channel).setHeader (
173     				((BaseChannel)clientSideChannel).getHeader());
174     			setFilters (channel);
175     			if (channel instanceof LogSource) {
176     			    ((LogSource)channel) .
177     				setLogger (getLogger(), getRealm()+".channel");
178     			}
179     			if (channel instanceof Observable)
180     			    ((Observable)channel).addObserver (this);
181     			channel.accept (serverSocket);
182     			pool.execute (new Session(channel));
183     		    } catch (IOException e) {
184     			Logger.log (new LogEvent (this, "iso-server", e));
185     			relax();
186     		    } catch (InstantiationException e) {
187     			Logger.log (new LogEvent (this, "iso-server", e));
188     			relax();
189     		    } catch (IllegalAccessException e) {
190     			Logger.log (new LogEvent (this, "iso-server", e));
191     			relax();
192     		    }
193     		}
194     	    } catch (Throwable e) {
195     		Logger.log (new LogEvent (this, "iso-server", e));
196     		relax();
197     	    }
198     	}
199         }
200     
201         private void setFilters (ISOChannel channel) {
202     	if (clientOutgoingFilters != null)
203     	    ((FilteredChannel)channel) .
204     		setOutgoingFilters (clientOutgoingFilters);
205     	if (clientIncomingFilters != null)
206     	    ((FilteredChannel)channel) .
207     		setIncomingFilters (clientOutgoingFilters);
208         }
209         private void relax() {
210     	try {
211     	    Thread.sleep (5000);
212     	} catch (InterruptedException e) { }
213         }

216          * associates this ISOServer with a name using NameRegistrar
217          * @param name name to register
218          * @see NameRegistrar
219          */
220         public void setName (String name) {
221     	this.name = name;
222     	NameRegistrar.register ("server."+name, this);
223         }
224         /**
225          * @return ISOMUX instance with given name.
226          * @throws NameRegistrar.NotFoundException;
227          * @see NameRegistrar
228          */
229         public static ISOServer getServer (String name)
230     	throws NameRegistrar.NotFoundException
231         {
232     	return (ISOServer) NameRegistrar.get ("server."+name);
233         }
234         /**
235          * @return this ISOServer's name ("" if no name was set)
236          */
237         public String getName() {
238     	return this.name;
239         }
240         public void setLogger (Logger logger, String realm) {
241     	this.logger = logger;
242     	this.realm  = realm;
243         }
244         public String getRealm () {
245     	return realm;
246         }
247         public Logger getLogger() {
248     	return logger;
249         }
250         public void update(Observable o, Object arg) {
251     	setChanged ();
252     	notifyObservers (arg);
253         }
254     }
