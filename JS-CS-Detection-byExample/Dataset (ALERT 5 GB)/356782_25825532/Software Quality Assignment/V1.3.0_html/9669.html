95     /**
96      * @author apr@cs.com.uy
97      * @version $Id$
98      * @serial
99      */
100     public class LogEvent extends EventObject {
101        /**
102         * @serial
103         */
104         LogSource source;
105        /**
106         * @serial
107         */
108         public String tag;
109        /**
110         * @serial
111         */
112         public Vector payLoad;
113         
114         public LogEvent (LogSource source, String tag) {
115     	super (source);
116     	this.source  = source;
117     	this.tag     = tag;
118     	this.payLoad = new Vector();
119         }
120         public LogEvent (LogSource source, String tag, Object msg) {
121     	super (source);
122     	this.source  = source;
123     	this.tag     = tag;
124     	this.payLoad = new Vector(1);
125     	addMessage(msg);
126         }
127         public void addMessage (Object msg) {
128     	this.payLoad.addElement(msg);
129         }
130         public void addMessage (String tagname, String message) {
131     	this.payLoad.addElement("<"+tagname+">"+message+"</"+tagname+">");
132         }
133         public void dump (PrintStream p, String indent) {
134     	if (payLoad.size() == 0) 
135     	    p.println (indent + "<" + tag + "/>");
136     	else {
137     	    p.println (indent + "<" + tag + ">");
138     	    String newIndent = indent + "  ";
139     	    Iterator i = payLoad.iterator();
140     	    while (i.hasNext()) {
141     		Object o = i.next();
142     		if (o instanceof Loggeable) 
143     		    ((Loggeable)o).dump (p, newIndent);
144     		else if (o instanceof SQLException) {
145     		    SQLException e = (SQLException) o;
146     		    p.println (newIndent + "<SQLException>"
147     				+e.getMessage() + "</SQLException>");
148     		    p.println (newIndent + "<SQLState>"
149     				+e.getSQLState() + "</SQLState>");
150     		    p.println (newIndent + "<VendorError>"
151     				+e.getErrorCode() + "</VendorError>");
152     		    ((Throwable)o).printStackTrace (p);
153     		}
154     		else if (o instanceof Throwable) {
155     		    p.println (newIndent + "<exception name=\""
156     			+ ((Throwable)o).getMessage() + "\">");
157     		    p.print (newIndent);
158     		    ((Throwable)o).printStackTrace (p);
159     		    p.println (newIndent + "</exception>");
160     		}
161     		else
162     		    p.println (newIndent + o.toString());
163     	    }
164     	    p.println (indent + "</" + tag + ">");
165     	}
166         }
167         public String getRealm() {
168     	return source.getRealm();
169         }
170     }
