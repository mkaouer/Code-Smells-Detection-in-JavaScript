73     /**
74      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>

78     public class QSP implements ErrorHandler, LogSource {
79         Document config;
80         Logger logger;
81         String realm;
82         File configFile;
83         long lastModified;
84         static ControlPanel controlPanel = null;
85         long monitorConfigInterval = 60 * 1000;
86         Collection reconfigurables;
87     
88         public static String[] SUPPORTED_TAGS = 
89     	{ "logger",
90     	  "qsp-config",
91     	  "log-listener",
92     	  "persistent-engine",
93     	  "sequencer",
94     	  "control-panel",
95     	  "channel",
96     	  "filter",
97     	  "mux",
98     	  "server",
99     	  "request-listener",
100     	  "card-agent",
101     	  "dir-poll",
102     	  "task",
103     	  "daily-task"
104     	};
105     
106         public QSP () {
107     	super();
108     	reconfigurables = new ArrayList();
109         }
110         public void setConfig (Document config) {
111     	this.config = config;
112         }
113         public void setConfigFile (File f) {
114     	this.configFile = f;
115     	this.lastModified = f.lastModified();
116         }
117         public void setMonitorConfigInterval (long l) {
118     	monitorConfigInterval = l;
119         }
120         public File getConfigFile () {
121     	return configFile;
122         }
123         public Collection getReConfigurables() {
124     	return reconfigurables;
125         }
126         public ControlPanel initControlPanel (int rows, int cols) {
127     	if (controlPanel == null) {
128     	    synchronized (QSP.class) {
129     		if (controlPanel == null) 
130     		    controlPanel = new ControlPanel (this, rows, cols);
131     	    }
132     	}
133     	return controlPanel;
134         }
135         public ControlPanel getControlPanel (){
136     	return controlPanel;
137         }
138         public void warning (SAXParseException e) throws SAXException {
139     	Logger.log (new LogEvent (this, "warning", e));
140     	throw e;
141         }
142         public void error (SAXParseException e) throws SAXException {
143     	Logger.log (new LogEvent (this, "error", e));
144     	throw e;
145         }
146     
147         public void fatalError (SAXParseException e) throws SAXException {
148     	Logger.log (new LogEvent (this, "fatalError", e));
149     	throw e;
150         }
151         public void setLogger (Logger logger, String realm) {
152     	this.logger = logger;
153     	this.realm  = realm;
154         }
155         public String getRealm () {
156     	return realm;
157         }
158         public Logger getLogger () {
159     	return logger;
160         }
161         public void configure (String tagname) throws ConfigurationException {
162     	QSPConfigurator configurator = QSPConfiguratorFactory.create (tagname);
163     	NodeList nodes = config.getElementsByTagName (tagname);
164     	if (configurator instanceof QSPReConfigurator && nodes.getLength()>0)
165     	    reconfigurables.add (tagname);
166     	for (int i=0; i<nodes.getLength(); i++) {
167     	    configurator.config (this, nodes.item(i));
168     	}
169         }
170         public void reconfigure (String tagname) throws ConfigurationException {
171     	QSPConfigurator configurator = QSPConfiguratorFactory.create (tagname);
172     	if (configurator instanceof QSPReConfigurator) {
173     	    NodeList nodes = config.getElementsByTagName (tagname);
174     	    for (int i=0; i<nodes.getLength(); i++) 
175     		((QSPReConfigurator)configurator).reconfig 
176     		    (this, nodes.item(i));
177     	}
178         }
179         private boolean monitorConfigFile () {
180     	long l;
181     	while (lastModified == (l=configFile.lastModified()))
182     	    try {
183     		Thread.sleep (monitorConfigInterval);
184     	    } catch (InterruptedException e) { }
185     	lastModified = l;
186     	return true;
187         }
188         public static void main (String args[]) {
189     	if (args.length != 1) {
190     	    System.out.println ("Usage: org.jpos.apps.qsp.QSP <configfile>");
191     	    System.exit (1);
192     	}
193     	DOMParser parser = new DOMParser();
194     	QSP qsp = new QSP();
195     	// qsp.getLogger().addListener (new SimpleLogListener(System.out));
196     	try {
197     	    qsp.setConfigFile (new File (args[0]));
198     	    parser.setFeature("http://xml.org/sax/features/validation", true);
199     	    parser.setErrorHandler (qsp);
200     	    parser.parse (qsp.getConfigFile().getPath());
201     	    qsp.setConfig (parser.getDocument());
202     	    for (int i=0; i<SUPPORTED_TAGS.length; i++)
203     		qsp.configure (SUPPORTED_TAGS[i]);
204     
205     	    if (controlPanel != null)
206     		controlPanel.showUp();
207     
208     	    if (qsp.getLogger() != null) 
209     		new SystemMonitor (3600000, qsp.getLogger(), "monitor");
210     		    
211     	    while (qsp.monitorConfigFile ()) {
212     		parser.parse (qsp.getConfigFile().getPath());
213     		qsp.setConfig (parser.getDocument());
214     		Iterator iter = qsp.getReConfigurables().iterator();
215     		while (iter.hasNext())
216     		    qsp.reconfigure ((String) iter.next());
217     	    }
218     	} catch (IOException e) {
219     	    Logger.log (new LogEvent (qsp, "error", e));
220     	    System.out.println (e);
221     	} catch (SAXException e) {
222     	    Logger.log (new LogEvent (qsp, "error", e));
223     	    System.out.println (e);
224     	} catch (ConfigurationException e) {
225     	    Logger.log (new LogEvent (qsp, "error", e));
226     	    System.out.println (e);
227     	}
228         }
