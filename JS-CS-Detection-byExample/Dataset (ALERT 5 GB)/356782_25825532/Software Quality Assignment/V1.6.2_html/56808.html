19     package org.jpos.q2.cli;
20     
21     import org.jpos.util.SystemMonitor;
22     import org.jpos.q2.CLI;
23     
24     public class MEM implements CLI.Command {
25         public void exec (CLI cli, String[] args) throws Exception {
26             if (args.length > 1 && "--gc".equals (args[1])) {
27                 System.gc();
28             }
29             Runtime r = Runtime.getRuntime();
30             StringBuffer sb = new StringBuffer();
31             sb.append ("total="); sb.append (r.totalMemory());
32             sb.append (" free="); sb.append (r.freeMemory());
33             sb.append (" in-use="); sb.append (r.totalMemory()-r.freeMemory());
34             cli.println (sb.toString());
35         }
36     }
