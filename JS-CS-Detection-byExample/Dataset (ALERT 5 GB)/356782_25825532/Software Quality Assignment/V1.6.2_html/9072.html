38         /** Should be called like this: put("ALL", "008", 2);
39          * Note: the second parameter is zero padded to three digits
40          */
41         private static void put(String alphaCode, String isoCode, int numDecimals) {
42             Currency ccy = new Currency(alphaCode, Integer.parseInt(isoCode), numDecimals);
43             currencies.put(alphaCode, ccy);
44             currencies.put(isoCode, ccy);
45         }
46     
47         static
48         {
49             currencies=new Hashtable();
50     
51             ResourceBundle r = ResourceBundle.getBundle(ISOCurrency.class.getName());
52             Enumeration en = r.getKeys();
53     
54             // property example:
55             //   DKK=208 2
56             while (en.hasMoreElements()) {
57                 String alphaCode  = (String)en.nextElement();
58                 String[] tmp = ((String)r.getString(alphaCode)).split(" ");
59                 String isoCode = tmp[0];
60                 int numDecimals = Integer.parseInt(tmp[1]);
61                 put(alphaCode, isoCode, numDecimals);
62             }
63         }

65         /** 
66          * Converts from an ISO Amount (12 digit string) to a double taking in 
67          * consideration the number of decimal digits according to currency
68          * 
69          * @param isoamount - The ISO amount to be converted (eg. ISOField 4)
70          * @param currency  - The ISO currency to be converted (eg. ISOField 49)
71          * @return result - A double representing the converted field
72          * @exception IllegalArgumentException
73          */
74         public static double convertFromIsoMsg(String isoamount, String currency) throws IllegalArgumentException
75         {
76             double d=0;                                                                                           
77             try
78             {
79                 Currency c=(Currency)currencies.get(currency.toUpperCase());
80                 int decimals=c.getDecimals();
81                 double m=1; if(decimals>0) for(int x=1;x<=decimals;x++) m*=10;
82                 d=new Double(isoamount).doubleValue();
83                 d/=m;
84             }
85             catch(Exception e)
86             {
87                 throw new IllegalArgumentException("Failed convertFromIsoMsg");
88             }
89             return d;
90         }
91     
92         /** 
93          * Converts an amount to an ISO Amount taking in consideration
94          * the number of decimal digits according to currency
95          * 
96          * @param amount - The amount to be converted 
97          * @param currency  - The ISO currency to be converted (eg. ISOField 49)
98          * @return result - An iso amount representing the converted field
99          * @exception IllegalArgumentException
100          */
101         public static String convertToIsoMsg(double amount,String currency) throws IllegalArgumentException
102         {
103             String z=null;
104             try
105             {
106                 Currency c=(Currency)currencies.get(currency.toUpperCase());
107                 if(c==null) throw new IllegalArgumentException("Bad currency parameter");
108                 int decimals=c.getDecimals();
109                 double m=1; if(decimals>0) for(int x=1;x<=decimals;x++) m*=10;
110                 amount*=m;
111                 z=ISOUtil.zeropad(Long.toString(Math.round (amount)),12);
112             }
113             catch(Exception e)
114             {
115                 throw new IllegalArgumentException("Failed convertToIsoMsg");
116             }
117             return z;
118         }
119         
120         public static Object[] decomposeComposedCurrency(String incurr) throws IllegalArgumentException
121         {
122             Object[] outcurr=null;
123             try
124             {   
125                 StringTokenizer st=new StringTokenizer(incurr);
126                 String curr=st.nextToken();
127                 Double amount=new Double(st.nextToken());
128                 outcurr=new Object[2];
129                 outcurr[0]=curr;
130                 outcurr[1]=amount;
131             }
132             catch(Exception e)
133             {   
134                 throw new IllegalArgumentException("Failed decompose");
135             }
136             return outcurr;
137         }
138         
139         public static String getIsoCodeFromAlphaCode(String alphacode) throws IllegalArgumentException
140         {
141             String isocode=null;
142             try
143             {
144                 Currency c=(Currency)currencies.get(alphacode.toUpperCase());
145                 isocode=ISOUtil.zeropad(Integer.toString(c.getIsoCode()),3);
146                 if(isocode==null) throw new IllegalArgumentException("AlphaCode not found, or incorrectly specified");
147             }
148             catch(Exception e)
149             {
150                 throw new IllegalArgumentException("Failed getIsoCodeFromAlphaCode");
151             }
152             return  isocode;
153         }
154     
155         public static Currency getCurrency (int code) throws ISOException {
156             return (Currency) 
157                 currencies.get (ISOUtil.zeropad (Integer.toString (code), 3));
158         }
159         public static Currency getCurrency (String code) throws ISOException {
160             return (Currency) currencies.get (ISOUtil.zeropad (code, 3));
161         }
