19     package org.jpos.transaction;
20     
21     import java.io.Serializable;
22     import org.jpos.iso.ISOUtil;
23     import org.jpos.core.Configurable;
24     import org.jpos.core.Configuration;
25     
26     public class TestPauseParticipant implements TransactionParticipant {
27         Configuration cfg;
28         TransactionManager txnmgr;
29     
30         public int prepare (long id, Serializable o) { 
31             final Context ctx = (Context) o;
32             new Thread() {
33                 public void run() {
34                     ISOUtil.sleep (1000);
35                     txnmgr.queue (ctx);   // re-inject paused transaction
36                 }
37             }.start();
38             return PREPARED | PAUSE;
39         }
40         public void commit (long id, Serializable o) { 
41             Context ctx = (Context) o;
42             ctx.log ("TestPausedParticipant commit has been called, id: " + id);
43         }
44         public void abort  (long id, Serializable o) { 
45             Context ctx = (Context) o;
46             ctx.log ("TestPausedParticipant abort has been called, id:" + id);
47         }
48         public void setTransactionManager (TransactionManager txnmgr) {
49             this.txnmgr = txnmgr;
50         }
51     }
