137         public static void runPack(String packagerName, int len, String data) throws Exception
138         {
139             Class packagerClass = Class.forName(packagerName);
140             ISOFieldPackager packager = (ISOFieldPackager)packagerClass.newInstance();
141             packager.setLength(len);
142             ISOField f = new ISOField(12, data);
143             long start = System.currentTimeMillis();
144             runPackTest(packager, f);
145             long end = System.currentTimeMillis();
146             boolean isOldImplementation = packagerClass.getSuperclass().getName().endsWith("ISOFieldPackager")
147                 || packagerClass.getSuperclass().getName().endsWith("IF_TBASE");
148             System.out.println("Pack: " + (isOldImplementation ? "old " : "new ") + packager.getClass().getName() + " = " + (end - start));
149             
150         }

166         private static void runPackTest(ISOFieldPackager p, ISOComponent c) throws ISOException
167         {
168             for (int i = 0; i < 1000000; i++)
169                 p.pack(c);
