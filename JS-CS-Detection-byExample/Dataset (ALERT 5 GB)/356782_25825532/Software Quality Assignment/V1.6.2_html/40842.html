34     /**
35      * <p>
36      * Provides some higher level methods that are needed by the JCE
37      * Security Module, yet they are generic and can be used elsewhere.
38      * </p>
39      * <p>
40      * It depends on the Java<font size=-1><sup>TM</sup></font> Cryptography Extension (JCE).
41      * </p>
42      * @author Hani S. Kirollos
43      * @version $Revision$ $Date$
44      */
45     public class JCEHandler {
46         static final String ALG_DES = "DES";
47         static final String ALG_TRIPLE_DES = "DESede";
48     
49         /**
50          * Registers the JCE provider whose name is providerName and sets it to be the
51          * only provider to be used in this instance of the JCEHandler class.
52          * @param jceProviderClassName Name of the JCE provider
53          * (e.g. "com.sun.crypto.provider.SunJCE" for Sun's implementation,
54          * or "org.bouncycastle.jce.provider.BouncyCastleProvider" for bouncycastle.org
55          * implementation)
56          * @throws JCEHandlerException
57          */
58         public JCEHandler (String jceProviderClassName) throws JCEHandlerException
59         {
60             try {
61                 provider = (Provider)Class.forName(jceProviderClassName).newInstance();
62                 Security.addProvider(provider);
63             } catch (Exception e) {
64                 throw  new JCEHandlerException(e);
65             }
66         }

68         /**
69          * Uses the JCE provider specified
70          * @param provider
71          */
72         public JCEHandler (Provider provider) {
73             this.provider = provider;
74         }

76         /**
77          * Generates a clear DES (DESede) key
78          * @param keyLength the bit length (key size) of the generated key (LENGTH_DES, LENGTH_DES3_2KEY or LENGTH_DES3_3KEY)
79          * @return generated clear DES (or DESede) key
80          * @exception JCEHandlerException
81          */
82         public Key generateDESKey (short keyLength) throws JCEHandlerException {
83             Key generatedClearKey = null;
84             try {
85                 KeyGenerator k1 = null;
86                 if (keyLength > SMAdapter.LENGTH_DES) {
87                     k1 = KeyGenerator.getInstance(ALG_TRIPLE_DES, provider.getName());
88                 }
89                 else {
90                     k1 = KeyGenerator.getInstance(ALG_DES, provider.getName());
91                 }
92                 generatedClearKey = k1.generateKey();
93                 /* These 3 steps not only enforce correct parity, but also enforces
94                  that when keyLength=128, the third key of the triple DES key is equal
95                  to the first key. This is needed because, JCE doesn't differenciate
96                  between Triple DES with 2 keys and Triple DES with 3 keys
97                  */
98                 byte[] clearKeyBytes = extractDESKeyMaterial(keyLength, generatedClearKey);
99                 Util.adjustDESParity(clearKeyBytes);
100                 generatedClearKey = formDESKey(keyLength, clearKeyBytes);
101             } catch (Exception e) {
102                 if (e instanceof JCEHandlerException)
103                     throw  (JCEHandlerException)e;
104                 else
105                     throw  new JCEHandlerException(e);
106             }
107             return  generatedClearKey;
108         }
109     
110         /**
111          * Encrypts (wraps) a clear DES Key, it also sets odd parity before encryption
112          * @param keyLength bit length (key size) of the clear DES key (LENGTH_DES, LENGTH_DES3_2KEY or LENGTH_DES3_3KEY)
113          * @param clearDESKey DES/Triple-DES key whose format is "RAW"
114          * (for a DESede with 2 Keys, keyLength = 128 bits, while DESede key with 3 keys keyLength = 192 bits)
115          * @param encryptingKey can be a key of any type (RSA, DES, DESede...)
116          * @return encrypted DES key
117          * @throws JCEHandlerException
118          */
119         public byte[] encryptDESKey (short keyLength, Key clearDESKey, Key encryptingKey) throws JCEHandlerException {
120             byte[] encryptedDESKey = null;
121             byte[] clearKeyBytes = extractDESKeyMaterial(keyLength, clearDESKey);
122             // enforce correct (odd) parity before encrypting the key
123             Util.adjustDESParity(clearKeyBytes);
124             encryptedDESKey = doCryptStuff(clearKeyBytes, encryptingKey, Cipher.ENCRYPT_MODE);
125             return  encryptedDESKey;
126         }
127     
128         /**
129          * Extracts the DES/DESede key material
130          * @param keyLength bit length (key size) of the DES key. (LENGTH_DES, LENGTH_DES3_2KEY or LENGTH_DES3_3KEY)
131          * @param clearDESKey DES/Triple-DES key whose format is "RAW"
132          * @return encoded key material
133          * @throws JCEHandlerException
134          */
135         protected byte[] extractDESKeyMaterial (short keyLength, Key clearDESKey) throws JCEHandlerException {
136             byte[] clearKeyBytes = null;
137             String keyAlg = clearDESKey.getAlgorithm();
138             String keyFormat = clearDESKey.getFormat();
139             if (keyFormat.compareTo("RAW") != 0) {
140                 throw  new JCEHandlerException("Unsupported DES key encoding format: "
141                         + keyFormat);
142             }
143             if (!keyAlg.startsWith(ALG_DES)) {
144                 throw  new JCEHandlerException("Unsupported key algorithm: " + keyAlg);
145             }
146             clearKeyBytes = clearDESKey.getEncoded();
147             clearKeyBytes = ISOUtil.trim(clearKeyBytes, getBytesLength(keyLength));
148             return  clearKeyBytes;
149         }
150     
151         /**
152          * Decrypts an encrypted DES/Triple-DES key
153          * @param keyLength bit length (key size) of the DES key to be decrypted. (LENGTH_DES, LENGTH_DES3_2KEY or LENGTH_DES3_3KEY)
154          * @param encryptedDESKey the byte[] representing the encrypted key
155          * @param encryptingKey can be of any algorithm (RSA, DES, DESede...)
156          * @param checkParity if true, the parity of the key is checked
157          * @return clear DES (DESede) Key
158          * @throws JCEHandlerException if checkParity==true and the key does not have correct parity
159          */
160         public Key decryptDESKey (short keyLength, byte[] encryptedDESKey, Key encryptingKey,
161                 boolean checkParity) throws JCEHandlerException {
162             Key key = null;
163             byte[] clearKeyBytes = doCryptStuff(encryptedDESKey, encryptingKey, Cipher.DECRYPT_MODE);
164             if (checkParity) {
165                 if (!Util.isDESParityAdjusted(clearKeyBytes)) {
166                     throw new JCEHandlerException("Parity not adjusted");
167                 }
168             }
169             key = formDESKey(keyLength, clearKeyBytes);
170             return  key;
171         }
172     
173         /**
174          * Forms the clear DES key given its "RAW" encoded bytes
175          * Does the inverse of extractDESKeyMaterial
176          * @param keyLength bit length (key size) of the DES key. (LENGTH_DES, LENGTH_DES3_2KEY or LENGTH_DES3_3KEY)
177          * @param clearKeyBytes the RAW DES/Triple-DES key
178          * @return clear key
179          * @throws JCEHandlerException
180          */
181         protected Key formDESKey (short keyLength, byte[] clearKeyBytes) throws JCEHandlerException {
182             Key key = null;
183             switch (keyLength) {
184                 case SMAdapter.LENGTH_DES:
185                     {
186                         key = new SecretKeySpec(clearKeyBytes, ALG_DES);
187                     }
188                     ;
189                     break;
190                 case SMAdapter.LENGTH_DES3_2KEY:
191                     {
192                         // make it 3 components to work with JCE
193                         clearKeyBytes = ISOUtil.concat(
194                             clearKeyBytes, 0, getBytesLength(SMAdapter.LENGTH_DES3_2KEY),
195                             clearKeyBytes, 0, getBytesLength(SMAdapter.LENGTH_DES)
196                             );
197                     }
198                 case SMAdapter.LENGTH_DES3_3KEY:
199                     {
200                         key = new SecretKeySpec(clearKeyBytes, ALG_TRIPLE_DES);
201                     }
202             }
203             if (key == null)
204                 throw  new JCEHandlerException("Unsupported DES key length: " + keyLength
205                         + " bits");
206             return  key;
207         }

219             return  encryptedData;

222         /**
223          * Decrypts data
224          * @param encryptedData
225          * @param key
226          * @return clear data
227          * @exception JCEHandlerException
228          */
229         public byte[] decryptData (byte[] encryptedData, Key key) throws JCEHandlerException {
230             byte[] clearData;
231             clearData = doCryptStuff(encryptedData, key, Cipher.DECRYPT_MODE);
232             return  clearData;
233         }
234     
235         /**
236          * performs cryptographic operations (encryption/decryption) using JCE Cipher
237          * @param data
238          * @param key
239          * @param CipherMode Cipher.ENCRYPT_MODE or Cipher.DECRYPT_MODE
240          * @return result of the cryptographic operations
241          * @throws JCEHandlerException
242          */
243         byte[] doCryptStuff (byte[] data, Key key, int CipherMode) throws JCEHandlerException {
244             byte[] result;
245             String transformation;
246             if (key.getAlgorithm().startsWith(ALG_DES)) {
247                 transformation = key.getAlgorithm() + "/" + desMode + "/" + desPadding;
248                 ;
249             }
250             else {
251                 transformation = key.getAlgorithm();
252             }
253             try {
254                 Cipher c1 = Cipher.getInstance(transformation, provider.getName());
255                 c1.init(CipherMode, key);
256                 result = c1.doFinal(data);
257             } catch (Exception e) {
258                 throw  new JCEHandlerException(e);
259             }
260             return  result;
261         }
262     
263         /**
264          * Calculates the length of key in bytes
265          * @param keyLength bit length (key size) of the DES key. (LENGTH_DES, LENGTH_DES3_2KEY or LENGTH_DES3_3KEY)
266          * @return keyLength/8
267          * @throws JCEHandlerException if unknown key length
268          */
269         int getBytesLength(short keyLength) throws JCEHandlerException{
270             int bytesLength = 0;
271             switch (keyLength) {
272                 case SMAdapter.LENGTH_DES: bytesLength = 8;break;
273                 case SMAdapter.LENGTH_DES3_2KEY: bytesLength = 16;break;
274                 case SMAdapter.LENGTH_DES3_3KEY: bytesLength = 24; break;
275                 default: throw new JCEHandlerException("Unsupported key length: " + keyLength + " bits");
276             }
277             return bytesLength;
278         }

280         /**
281          * The JCE provider
282          */
283         Provider provider = null;
284         String desMode = "ECB";
285         String desPadding = "NoPadding";
286     }
