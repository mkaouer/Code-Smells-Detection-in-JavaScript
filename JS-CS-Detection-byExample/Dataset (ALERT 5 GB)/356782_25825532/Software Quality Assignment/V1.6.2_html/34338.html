260         protected void connect (Socket socket) 
261             throws IOException, SocketException
262         {
263             this.socket = socket;
264             applyTimeout();
265             setLogger(getLogger(), getOriginalRealm() + 
266                 "/" + socket.getInetAddress().getHostAddress() + ":" 
267                 + socket.getPort()
268             );
269             serverIn = new DataInputStream (
270                 new BufferedInputStream (socket.getInputStream ())
271             );
272             serverOut = new DataOutputStream(
273                 new BufferedOutputStream(socket.getOutputStream(), 2048)
274             );
275             usable = true;
276             cnt[CONNECT]++;
277             setChanged();
278             notifyObservers();
279         }
280         /**
281          * factory method pattern (as suggested by Vincent.Greene@amo.com)
282          * @throws IOException
283          * Use Socket factory if exists. If it is missing create a normal socket
284          * @see ISOClientSocketFactory
285          */
286         protected Socket newSocket(String host, int port) throws IOException {
287             try {
288                 if (socketFactory != null)
289                     return socketFactory.createSocket (host, port);
290                 else {
291                     if (connectTimeout > 0) {
292                         Socket s = new Socket();
293                         s.connect (
294                             new InetSocketAddress (host, port),
295                             connectTimeout
296                         );
297                         return s;
298                     } else if (localIface == null && localPort == 0){
299                         return new Socket(host,port);
300                     } else {
301                         InetAddress addr = (localIface == null) ?
302                             InetAddress.getLocalHost() : 
303                             InetAddress.getByName(localIface);
304                         return new Socket(host, port, addr, localPort);
305                     }
306                 }
307             } catch (ISOException e) {
308                 throw new IOException (e.getMessage());
309             }
310         }
311         protected Socket newSocket (String[] hosts, int[] ports, LogEvent evt) 
312             throws IOException
313         {
314             Socket s = null;
315             for (int i=0; i<hosts.length; i++) {
316                 try {
317                     evt.addMessage (hosts[i]+":"+ports[i]);
318                     s = newSocket (hosts[i], ports[i]);
319                     break;
320                 } catch (IOException e) {
321                     evt.addMessage ("  " + e.getMessage());
322                 }
323             }
324             if (s == null)
325                 throw new IOException ("Unable to connect");
326             return s;
327         }

358         /**
359          * Connects client ISOChannel to server
360          * @exception IOException
361          */
362         public void connect () throws IOException {
363             LogEvent evt = new LogEvent (this, "connect");
364             try {
365                 if (serverSocket != null) {
366                     accept(serverSocket);
367                     evt.addMessage ("local port "+serverSocket.getLocalPort()
368                         +" remote host "+socket.getInetAddress());
369                 }
370                 else {
371                     connect(newSocket (hosts, ports, evt));
372                 }
373                 applyTimeout();
374                 if (socket != null)
375                     socket.setKeepAlive (keepAlive);
376                 Logger.log (evt);
377             } catch (ConnectException e) {
378                 Logger.log (new LogEvent (this, "connection-refused",
379                     getHost()+":"+getPort())
380                 );
381             } catch (IOException e) {
382                 evt.addMessage (e.getMessage ());
383                 Logger.log (evt);
384                 throw e;
385             }
386         }

485         /**
486          * sends an ISOMsg over the TCP/IP session
487          * @param m the Message to be sent
488          * @exception IOException
489          * @exception ISOException
490          * @exception ISOFilter.VetoException;
491          */
492         public void send (ISOMsg m) 
493             throws IOException, ISOException, VetoException
494         {
495             LogEvent evt = new LogEvent (this, "send");
496             try {
497                 if (!isConnected())
498                     throw new ISOException ("unconnected ISOChannel");
499                 m.setDirection(ISOMsg.OUTGOING);
500                 m = applyOutgoingFilters (m, evt);
501                 evt.addMessage (m);
502                 m.setDirection(ISOMsg.OUTGOING); // filter may have drop this info
503                 m.setPackager (getDynamicPackager(m));
504                 byte[] b = m.pack();
505                 synchronized (serverOut) {
506                     sendMessageLength(b.length + getHeaderLength(m));
507                     sendMessageHeader(m, b.length);
508                     sendMessage (b, 0, b.length);
509                     sendMessageTrailler(m, b);
510                     serverOut.flush ();
511                 }
512                 cnt[TX]++;
513                 setChanged();
514                 notifyObservers(m);
515             } catch (VetoException e) {
516                 //if a filter vets the message it was not added to the event
517                 evt.addMessage (m);
518                 evt.addMessage (e);
519                 throw e;
520             } catch (ISOException e) {
521                 evt.addMessage (e);
522                 throw e;
523             } catch (IOException e) {
524                 evt.addMessage (e);
525                 throw e;
526             } catch (Exception e) {
527                 evt.addMessage (e);
528                 throw new ISOException ("unexpected exception", e);
529             } finally {
530                 Logger.log (evt);
531             }
532         }

570         /**
571          * Waits and receive an ISOMsg over the TCP/IP session
572          * @return the Message received
573          * @exception IOException
574          * @exception ISOException
575          */
576         public ISOMsg receive() throws IOException, ISOException {
577             byte[] b=null;
578             byte[] header=null;
579             LogEvent evt = new LogEvent (this, "receive");
580             ISOMsg m = createMsg ();  // call createMsg instead of createISOMsg for 
581                                       // backward compatibility
582             m.setSource (this);
583             try {
584                 if (!isConnected())
585                     throw new ISOException ("unconnected ISOChannel");
586     
587                 synchronized (serverIn) {
588                     int len  = getMessageLength();
589                     int hLen = getHeaderLength();
590     
591                     if (len == -1) {
592                         if (hLen > 0) {
593                             header = readHeader(hLen);
594                         }
595                         b = streamReceive();
596                     }
597                     else if (len > 0 && len <= getMaxPacketLength()) {
598                         if (hLen > 0) {
599                             // ignore message header (TPDU)
600                             // Note header length is not necessarily equal to hLen (see VAPChannel)
601                             header = readHeader(hLen);
602                             len -= header.length;
603                         }
604                         b = new byte[len];
605                         getMessage (b, 0, len);
606                         getMessageTrailler();
607                     }
608                     else
609                         throw new ISOException(
610                             "receive length " +len + " seems strange - maxPacketLength = " + getMaxPacketLength());
611                 }
612                 m.setPackager (getDynamicPackager(b));
613                 m.setHeader (getDynamicHeader(header));
614                 if (b.length > 0 && !shouldIgnore (header))  // Ignore NULL messages
615                     unpack (m, b);
616                 m.setDirection(ISOMsg.INCOMING);
617                 m = applyIncomingFilters (m, header, b, evt);
618                 m.setDirection(ISOMsg.INCOMING);
619                 evt.addMessage (m);
620                 cnt[RX]++;
621                 setChanged();
622                 notifyObservers(m);
623             } catch (ISOException e) {
624                 evt.addMessage (e);
625                 if (header != null) {
626                     evt.addMessage ("--- header ---");
627                     evt.addMessage (ISOUtil.hexdump (header));
628                 }
629                 if (b != null) {
630                     evt.addMessage ("--- data ---");
631                     evt.addMessage (ISOUtil.hexdump (b));
632                 }
633                 throw e;
634             } catch (EOFException e) {
635                 closeSocket();
636                 evt.addMessage ("<peer-disconnect/>");
637                 throw e;
638             } catch (SocketException e) {
639                 closeSocket();
640                 if (usable)
641                     evt.addMessage ("<peer-disconnect>" + e.getMessage() + "</peer-disconnect>");
642                 throw e;
643             } catch (InterruptedIOException e) {
644                 closeSocket();
645                 evt.addMessage ("<io-timeout/>");
646                 throw e;
647             } catch (IOException e) { 
648                 closeSocket();
649                 if (usable) 
650                     evt.addMessage (e);
651                 throw e;
652             } catch (Exception e) { 
653                 evt.addMessage (m);
654                 evt.addMessage (e);
655                 throw new ISOException ("unexpected exception", e);
656             } finally {
657                 Logger.log (evt);
658             }
659             return m;
660         }

669         /**
670          * disconnects the TCP/IP session. The instance is ready for
671          * a reconnection. There is no need to create a new ISOChannel<br>
672          * @exception IOException
673          */
674         public void disconnect () throws IOException {
675             LogEvent evt = new LogEvent (this, "disconnect");
676             if (serverSocket != null) 
677                 evt.addMessage ("local port "+serverSocket.getLocalPort()
678                     +" remote host "+serverSocket.getInetAddress());
679             else
680                 evt.addMessage (host+":"+port);
681             try {
682                 usable = false;
683                 setChanged();
684                 notifyObservers();
685                 closeSocket();
686                 if (serverIn != null) {
687                     try {
688                         serverIn.close();
689                     } catch (IOException ex) { evt.addMessage (ex); }
690                     serverIn  = null;
691                 }
692                 if (serverOut != null) {
693                     try {
694                         serverOut.close();
695                     } catch (IOException ex) { evt.addMessage (ex); }
696                     serverOut = null;
697                 }
698             } catch (IOException e) {
699                 evt.addMessage (e);
700                 Logger.log (evt);
701                 throw e;
702             }
703             socket = null;
704         }

784          */
785         public void removeFilter (ISOFilter filter, int direction) {
786             switch (direction) {
787                 case ISOMsg.INCOMING :
788                     incomingFilters.remove (filter);
789                     break;
790                 case ISOMsg.OUTGOING :
791                     outgoingFilters.remove (filter);
792                     break;
793                 case 0 :
794                     incomingFilters.remove (filter);
795                     outgoingFilters.remove (filter);
796                     break;
797             }
798         }

830         protected ISOMsg applyIncomingFilters (ISOMsg m, byte[] header, byte[] image, LogEvent evt) 
831             throws VetoException
832         {
833             Iterator iter  = incomingFilters.iterator();
834             while (iter.hasNext()) {
835                 ISOFilter f = (ISOFilter) iter.next ();
836                 if (image != null && (f instanceof RawIncomingFilter))
837                     m = ((RawIncomingFilter)f).filter (this, m, header, image, evt);
838                 else
839                     m = f.filter (this, m, evt);
840             }
841             return m;
842         }

846        /**
847         * Implements Configurable<br>
848         * Properties:<br>
849         * <ul>
850         * <li>host - destination host (if ClientChannel)
851         * <li>port - port number      (if ClientChannel)
852         * <li>local-iface - local interfase to use (if ClientChannel)
853         * <li>local-port - local port to bind (if ClientChannel)
854         * </ul>
855         * (host not present indicates a ServerChannel)
856         *
857         * @param cfg Configuration
858         * @throws ConfigurationException
859         */
860         public void setConfiguration (Configuration cfg)
861             throws ConfigurationException 
862         {
863             this.cfg = cfg;
864             String h    = cfg.get    ("host");
865             int port    = cfg.getInt ("port");
866             maxPacketLength = cfg.getInt ("max-packet-length", 100000);
867             if (h != null && h.length() > 0) {
868                 if (port == 0)
869                     throw new ConfigurationException 
870                         ("invalid port for host '"+h+"'");
871                 setHost (h, port);
872                 setLocalAddress (cfg.get("local-iface", null),cfg.getInt("local-port"));
873                 String[] altHosts = cfg.getAll  ("alternate-host");
874                 int[] altPorts = cfg.getInts ("alternate-port");
875                 hosts = new String[altHosts.length + 1];
876                 ports = new int[altPorts.length + 1];
877                 if (hosts.length != ports.length) {
878                     throw new ConfigurationException (
879                         "alternate host/port misconfiguration"
880                     );
881                 }
882                 hosts[0] = host;
883                 ports[0] = port;
884                 System.arraycopy (altHosts, 0, hosts, 1, altHosts.length);
885                 System.arraycopy (altPorts, 0, ports, 1, altPorts.length);
886             }
887             setOverrideHeader(cfg.getBoolean ("override-header", false));
888             keepAlive = cfg.getBoolean ("keep-alive", false);
889             if (socketFactory != this && socketFactory instanceof Configurable)
890                 ((Configurable)socketFactory).setConfiguration (cfg);
891             try {
892                 setTimeout (cfg.getInt ("timeout"));
893                 connectTimeout = cfg.getInt ("connect-timeout", timeout);
894             } catch (SocketException e) {
895                 throw new ConfigurationException (e);
896             }
897         }
