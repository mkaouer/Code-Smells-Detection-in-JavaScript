35     /**
36      * DirPoll Adaptor
37      *
38      * @author Alejandro Revilla
39      * @version $Revision$ $Date$
40      * @jmx:mbean description="DirPoll adaptor QBean"
41      *                  extends="org.jpos.q2.QBeanSupportMBean"
42      */
43     public class DirPollAdaptor 
44         extends QBeanSupport
45         implements DirPollAdaptorMBean
46     {
47         String path, priorities, processorClass;
48         int poolSize;
49         long pollInterval;
50         DirPoll dirPoll;
51         public DirPollAdaptor () {
52             super ();
53             poolSize = 1;
54             pollInterval = 1000;
55         }
56     
57         protected void initService () throws Exception {
58             QFactory factory = getServer().getFactory();
59             dirPoll  = new DirPoll ();
60             dirPoll.setPath (getPath ());
61             dirPoll.setThreadPool (new ThreadPool (1, poolSize));
62             dirPoll.setPollInterval (pollInterval);
63             if (priorities != null) 
64                 dirPoll.setPriorities (priorities);
65             dirPoll.setLogger (getLog().getLogger(), getLog().getRealm ());
66             Configuration cfg = factory.getConfiguration (getPersist());
67             dirPoll.setConfiguration (cfg);
68             dirPoll.createDirs ();
69             Object dpp = factory.newInstance (getProcessor());
70             if (dpp instanceof LogSource) {
71                 ((LogSource) dpp).setLogger (
72                     getLog().getLogger(), getLog().getRealm ()
73                 );
74                 dirPoll.setProcessor (dpp);
75             }
76             if (dpp instanceof Configurable) {
77                 ((Configurable) dpp).setConfiguration (cfg);
78             }
79         }
80         protected void startService () throws Exception {
81             new Thread (dirPoll).start ();
82         }
83     
84         protected void stopService () throws Exception {
85             dirPoll.destroy ();
86         }
87

98          * @jmx:managed-attribute description="thread pool size"
99          */
100         public synchronized void setPoolSize (int size) {
101             this.poolSize = size;
102             setModified (true);
103         }
104         /**
105          * @jmx:managed-attribute description="thread pool size"
106          */
107         public int getPoolSize () {
108             return poolSize;
109         }
110     
111         /**
112          * @jmx:managed-attribute description="Base path"
113          */
114         public String getPath () {
115             return path == null ? "." : path;
116         }
117     
118         /**
119          * @jmx:managed-attribute description="poll time in millis"
120          */
121         public synchronized void setPollInterval (long pollInterval) {
122             this.pollInterval = pollInterval;
123             setModified (true);
124         }
125         /**
126          * @jmx:managed-attribute description="poll time in millis"
127          */
128         public long getPollInterval () {
129             return pollInterval;
130         }
131         /**
132          * @jmx:managed-attribute description="priorities"
133          */
134         public synchronized void setPriorities (String priorities) {
135             this.priorities = priorities;
136             setModified (true);
137         }
138         /**
139          * @jmx:managed-attribute description="priorities"
140          */
141         public String getPriorities () {
142             return priorities;
143         }
144     
145         /**
146          * @jmx:managed-attribute description="processor class"
147          */
148         public synchronized void setProcessor (String processor) {
149             this.processorClass = processor;
150             setModified (true);
151         }
152         /**
153          * @jmx:managed-attribute description="processor class"
154          */
155         public String getProcessor() {
156             return processorClass;
157         }
158     }
