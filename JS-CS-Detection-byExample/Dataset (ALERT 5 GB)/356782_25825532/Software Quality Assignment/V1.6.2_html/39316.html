19     package org.jpos.iso;
20     
21     /**
22      * AsciiPrefixer constructs a prefix for ASCII messages.
23      * 
24      * @author joconnor
25      * @version $Revision$ $Date$
26      */
27     public class AsciiPrefixer implements Prefixer
28     {
29         /**
30          * A length prefixer for upto 9 chars. The length is encoded with 1 ASCII
31          * char representing 1 decimal digit.
32          */
33         public static final AsciiPrefixer L = new AsciiPrefixer(1);
34         /**
35     	 * A length prefixer for upto 99 chars. The length is encoded with 2 ASCII
36     	 * chars representing 2 decimal digits.
37     	 */
38         public static final AsciiPrefixer LL = new AsciiPrefixer(2);
39         /**
40     	 * A length prefixer for upto 999 chars. The length is encoded with 3 ASCII
41     	 * chars representing 3 decimal digits.
42     	 */
43         public static final AsciiPrefixer LLL = new AsciiPrefixer(3);
44         /**
45     	 * A length prefixer for upto 9999 chars. The length is encoded with 4
46     	 * ASCII chars representing 4 decimal digits.
47     	 */
48         public static final AsciiPrefixer LLLL = new AsciiPrefixer(4);
49         /**
50          * A length prefixer for upto 99999 chars. The length is encoded with 5
51          * ASCII chars representing 5 decimal digits.
52          */
53         public static final AsciiPrefixer LLLLL = new AsciiPrefixer(5);
54     
55         /**
56          * A length prefixer for upto 999999 chars. The length is encoded with 6
57          * ASCII chars representing 6 decimal digits.
58          */
59         public static final AsciiPrefixer LLLLLL = new AsciiPrefixer(6);
60     
61         //private static final LeftPadder PADDER = LeftPadder.ZERO_PADDER;
62         //private static final AsciiInterpreter INTERPRETER = AsciiInterpreter.INSTANCE;
63     
64         /** The number of digits allowed to express the length */
65         private int nDigits;
66         
67         public AsciiPrefixer(int nDigits)
68         {
69             this.nDigits = nDigits;
70         }
71     
72         /*
73     	 * (non-Javadoc)
74     	 * 
75     	 * @see org.jpos.iso.Prefixer#encodeLength(int, byte[])
76     	 */
77         public void encodeLength(int length, byte[] b) throws ISOException
78         {
79             int n = length;
80             // Write the string backwards - I don't know why I didn't see this at first.
81             for (int i = nDigits - 1; i >= 0; i--)
82             {
83                 b[i] = (byte)(n % 10 + '0');
84                 n /= 10;
85             }
86             if (n != 0)
87             {
88                 throw new ISOException("invalid len "+ length + ". Prefixing digits = " + nDigits);
89             }
90         }
91     
92         /*
93     	 * (non-Javadoc)
94     	 * 
95     	 * @see org.jpos.iso.Prefixer#decodeLength(byte[], int)
96     	 */
97         public int decodeLength(byte[] b, int offset)
98         {
99             int len = 0;
100             for (int i = 0; i < nDigits; i++)
101             {
102                 len = len * 10 + b[offset + i] - (byte)'0';
103             }
104             return len;
105         }
106     
107         /*
108     	 * (non-Javadoc)
109     	 * 
110     	 * @see xcom.traxbahn.iso.Prefixer#getLengthInBytes()
111     	 */
112         public int getPackedLength()
113         {
114             return nDigits;
115         }
116     }
