19     package org.jpos.iso;
20     
21     /**
22      * Base validation-error class. Contains a reference to error details.
23      * Error description, Error reject code: optional code used in some
24      * financial systems to specifya field reject code. It refer to error.
25      * Error Id: A string of " " separated ids. The ids are the fields,
26      * subfields, ... ids for the component with error.
27      * For example: id="48 0 1" indicates the error was in field 48,
28      * subfield 0, subfield 1.
29      * <p>Title: jPOS</p>
30      * <p>Description: Java Framework for Financial Systems</p>
31      * <p>Copyright: Copyright (c) 2000 jPOS.org.  All rights reserved.</p>
32      * <p>Company: www.jPOS.org</p>
33      * @author Jose Eduardo Leon
34      * @version 1.0
35      */
36     public class ISOVError {
37     
38         public ISOVError( String Description ) {
39             this.description = Description;
40         }
41     
42         public ISOVError( String Description, String RejectCode ) {
43             this.description = Description;
44             this.rejectCode = RejectCode;
45         }
46     
47         public String getRejectCode(){
48             return rejectCode;
49         }
50     
51         public String getId(){
52             return id;
53         }
54     
55         public void setId ( String ID ){
56             id = ID;
57         }
58     
59         public String getDescription() {
60             return description;
61         }
62     
63         /** Used by error parsers to set field tree path **/
64         protected String id;
65         protected String description = "";
66         protected String rejectCode;
67         /** default error types **/
68         public static final int ERR_INVALID_LENGTH = 1;
69         public static final int ERR_INVALID_VALUE = 2;
70     }
