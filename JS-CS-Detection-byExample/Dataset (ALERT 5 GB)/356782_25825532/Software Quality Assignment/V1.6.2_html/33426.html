19     package org.jpos.core;
20     
21     import java.io.ByteArrayInputStream;
22     import java.io.ObjectInputStream;
23     import java.util.ArrayList;
24     import java.util.Iterator;
25     
26     
27     /**
28      * @author apr@cs.com.uy
29      * @version $Id$
30      * @since jPOS 1.1
31      *
32      * Singleton in charge of registering and further locating Agents
33      * @see CardAgent
34      * @see CardTransaction
35      */
36     public class CardAgentLookup {
37         private static CardAgentLookup instance = new CardAgentLookup();
38         private ArrayList agents;
39     
40         /**
41          * no external instantiation - thank you
42          */
43         private CardAgentLookup () {
44             agents = new ArrayList();
45         }
46         /**
47          * register an Agent (at the end of the list)
48          * @param agent Agent to add
49          */
50         synchronized public static void add (CardAgent agent) {
51             instance.agents.add (agent);
52         }
53         /**
54          * remove all ocurrences of agent
55          * @param agent Agent to remove
56          */
57         synchronized public static void remove (CardAgent agent) {
58             ArrayList a = instance.agents;
59             int i;
60             while ( (i=a.indexOf(agent)) >= 0)
61                 a.remove(i);
62         }
63         /**
64          * @return all available agents
65          */
66         synchronized public static CardAgent[] getAgents() {
67             ArrayList a = instance.agents;
68             return (CardAgent[]) a.toArray(new CardAgent[a.size()]);
69         }
70     
71         /**
72          * locate agent able to process a given CardTransaction
73          * @param t CardTransaction holding an Operation to be performed
74          * @return suitable array of agents
75          */
76         synchronized public static CardAgent[] getAgents (CardTransaction t) {
77             ArrayList l = new ArrayList();
78             Iterator i = instance.agents.iterator();
79             while (i.hasNext()) {
80                 CardAgent a = (CardAgent) i.next();
81                 if (a.canHandle (t)) 
82                     l.add (a);
83             }
84             return (CardAgent[]) l.toArray(new CardAgent[l.size()]);
85         }
86         /**
87          * locate an agent giving its class Name
88          * @param class name
89          * @return given agent
90          * @exception CardAgentNotFoundException
91          */
92         synchronized public static CardAgent getAgent (String name)
93             throws CardAgentNotFoundException
94         {
95             Iterator i = instance.agents.iterator();
96             while (i.hasNext()) {
97                 CardAgent a = (CardAgent) i.next();
98                 if ( (a.getClass().getName()).endsWith (name) ) 
99                     return a;
100             }
101             throw new CardAgentNotFoundException (name);
102         }
103         /**
104          * locate an agent giving its unique agent ID
105          * @param id
106          * @return given agent
107          * @exception CardAgentNotFoundException
108          */
109         synchronized public static CardAgent getAgent (int id)
110             throws CardAgentNotFoundException
111         {
112             Iterator i = instance.agents.iterator();
113             while (i.hasNext()) {
114                 CardAgent a = (CardAgent) i.next();
115                 if (a.getID() == id)
116                     return a;
117             }
118             throw new CardAgentNotFoundException (Integer.toString(id));
119         }
120     
121         /**
122          * locate an agent giving a transaction Image
123          * @param b a transaction image
124          * @return given agent
125          * @exception CardAgentNotFoundException
126          */
127         synchronized public static CardAgent getAgent (byte[] b)
128             throws CardAgentNotFoundException
129         {
130             try {
131                 ByteArrayInputStream i = new ByteArrayInputStream (b);
132                 ObjectInputStream o    = new ObjectInputStream (i);
133                 int id = o.readInt();
134                 return getAgent (id);
135             } catch (Exception e) { }
136             throw new CardAgentNotFoundException ();
137         }
138     
139         /**
140          * locate an agent of a given class
141          * @param class 
142          * @return given agent
143          * @exception CardAgentNotFoundException
144          */
145         synchronized public static CardAgent getAgent (Class t)
146             throws CardAgentNotFoundException
147         {
148             Iterator i = instance.agents.iterator();
149             while (i.hasNext()) {
150                 CardAgent a = (CardAgent) i.next();
151                 if (a.getClass() == t)
152                     return a;
153             }
154             throw new CardAgentNotFoundException (t.getName());
155         }
156         /**
157          * locate an agent able to process a given CardTransaction
158          * @param t CardTransaction holding an Operation to be performed
159          * @return suitable agent
160          * @exception CardAgentNotFoundException
161          */
162         synchronized public static CardAgent getAgent (CardTransaction t)
163             throws CardAgentNotFoundException
164         {
165             Iterator i = instance.agents.iterator();
166             while (i.hasNext()) {
167                 CardAgent a = (CardAgent) i.next();
168                 if (a.canHandle (t)) 
169                     return a;
170             }
171             throw new CardAgentNotFoundException ();
172         }
173     }
