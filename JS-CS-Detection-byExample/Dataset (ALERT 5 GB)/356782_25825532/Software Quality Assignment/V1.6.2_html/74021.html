29     /**
30      * VALIDATES FIELD INTERDEPENDENCY. IF FIELD 0 ENDS WITH 1 THEN
31      * FIELD 1 MUST END WITH 0.

40     public class MSGTEST02 extends ISOBaseValidator {
41     
42         public MSGTEST02() {
43             super();
44         }
45     
46         public MSGTEST02( boolean breakOnError ) {
47             super(breakOnError);
48         }
49     
50         private String makeStrFromArray( int[] validFields ){
51             if ( validFields == null ) return null;
52             StringBuffer result = new StringBuffer();
53             for (int i = 0; i < validFields.length; i++){
54                 result.append( validFields[i] );
55                 result.append( ", " );
56             }
57             result.delete( result.length()-2, result.length()-1 );
58             return result.toString(  );
59         }
60     
61         public ISOComponent validate(ISOComponent m) throws org.jpos.iso.ISOException {
62             LogEvent evt = new LogEvent( this, "validate" );
63             try {
64                 super.validate ( m );
65                 ISOMsg msg = (ISOMsg)m;
66                 int[] validFields = { 0,1 };
67                 if ( !msg.hasFields( validFields ) ){
68                     ISOVError e = new ISOVError( "Fields " + makeStrFromArray( validFields ) + " must appear in msg.", "002" );
69                     if ( msg instanceof ISOVMsg )
70                         ((ISOVMsg)msg).addISOVError( e );
71                     else
72                         msg = new ISOVMsg( msg, e );
73                     if ( breakOnError )
74                         throw new ISOVException ( "Error on msg. " , msg );
75                 }
76                 /** field interdependency **/
77                 if ( msg.getString( 0 ).endsWith( "1" )  && !msg.getString( 1 ).endsWith( "0" ) ){
78                     ISOVError e = new ISOVError( "If field 0 ends with 1 then field 1 must end with 0.", "003" );
79                     if ( msg instanceof ISOVMsg )
80                         ((ISOVMsg)msg).addISOVError( e );
81                     else
82                         msg = new ISOVMsg( msg, e );
83                     if ( breakOnError )
84                         throw new ISOVException ( "Error on msg. " , msg );
85                 }
86                 return msg;
87             } catch ( ISOVException ex ) {
88                 throw ex;
89             } finally {
90                 Logger.log( evt );
91             }
92         }
93     }
