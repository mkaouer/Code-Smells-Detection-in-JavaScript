19     package org.jpos.q2.ui;
20     
21     import org.jdom.Element;
22     import org.jdom.JDOMException;
23     import org.jpos.q2.QBean;
24     import org.jpos.q2.QBeanSupport;
25     import org.jpos.ui.UIObjectFactory;
26     
27     /**
28      * @author Alejandro Revilla
29      *
30      * org.jpos.ui.UI adapter
31      */
32     public class UI extends QBeanSupport implements UIObjectFactory {
33         org.jpos.ui.UI ui;
34         public UI () {
35             super();
36         }
37         public void startService () throws Exception {
38             Element config = getPersist ();
39             String provider = 
40                 config.getAttributeValue ("provider", "org.jpos.ui.UI");
41             ui = (org.jpos.ui.UI) getFactory().newInstance (provider);
42             ui.setConfig (config);
43             ui.setLog (getLog ());
44             ui.setObjectFactory (this);
45             ui.configure ();
46         }
47     
48         public void stopService () {
49             if (ui != null) 
50                 ui.dispose ();
51         }
52         public Object newInstance (String clazz) throws Exception {
53             return getFactory().newInstance (clazz);
54         }
55     }
