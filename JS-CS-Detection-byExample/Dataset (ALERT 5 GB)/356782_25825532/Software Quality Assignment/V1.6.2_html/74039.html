19     package org.jpos.q2.qbean;
20     
21     import org.jdom.Element;
22     import org.jpos.q2.QBeanSupport;
23     
24     import bsh.BshClassManager;
25     import bsh.Interpreter;
26     import bsh.UtilEvalError;
27     
28     public class BSH extends QBeanSupport implements Runnable {
29         protected Interpreter bsh;
30         public void initService() {
31             bsh = new Interpreter ();
32             BshClassManager bcm = bsh.getClassManager();
33             try {
34                 bcm.setClassPath(getServer().getLoader().getURLs());
35             } catch (UtilEvalError e) {
36                 e.printStackTrace();
37             }
38             bcm.setClassLoader(getServer().getLoader());
39         }
40         public void startService() {
41             new Thread (this, "BSH-" + getName()).start ();
42         }
43         public void run () {
44             Element config = getPersist();
45             try {
46                 bsh.set  ("qbean", this);
47                 bsh.set  ("log", getLog());
48                 bsh.eval (config.getText());
49                 String source = config.getAttributeValue ("source");
50                 if (source != null)
51                     bsh.source (source);
52             } catch (Throwable e) {
53                 getLog().warn (e);
54             }
55         }
56     }
