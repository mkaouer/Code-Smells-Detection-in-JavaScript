19     package org.jpos.iso.channel;
20     
21     import java.io.IOException;
22     import java.net.ServerSocket;
23     
24     import org.jpos.iso.BaseChannel;
25     import org.jpos.iso.ISOChannel;
26     import org.jpos.iso.ISOException;
27     import org.jpos.iso.ISOMsg;
28     import org.jpos.iso.ISOPackager;
29     import org.jpos.iso.ISOUtil;
30     
31     /**
32      * ISOChannel implementation - Postilion Channel
33      * Send packet len (2 bytes network byte order MSB/LSB) followed by
34      * raw data. 
35      *
36      * @author salaman@teknos.com
37      * @version Id: PostChannel.java,v 1.0 1999/05/14 19:00:00 may Exp 
38      * @see ISOMsg
39      * @see ISOException
40      * @see ISOChannel
41      */
42     public class PostChannel extends BaseChannel {
43         /**
44          * Public constructor (used by Class.forName("...").newInstance())
45          */
46         public PostChannel () {
47             super();
48         }
49         /**
50          * Construct client ISOChannel
51          * @param host  server TCP Address
52          * @param port  server port number
53          * @param p     an ISOPackager
54          * @see ISOPackager
55          */
56         public PostChannel (String host, int port, ISOPackager p) {
57             super(host, port, p);
58         }
59         /**
60          * Construct server ISOChannel
61          * @param p     an ISOPackager
62          * @exception IOException
63          * @see ISOPackager
64          */
65         public PostChannel (ISOPackager p) throws IOException {
66             super(p);
67         }
68         /**
69          * constructs a server ISOChannel associated with a Server Socket
70          * @param p     an ISOPackager
71          * @param serverSocket where to accept a connection
72          * @exception IOException
73          * @see ISOPackager
74          */
75         public PostChannel (ISOPackager p, ServerSocket serverSocket) 
76             throws IOException
77         {
78             super(p, serverSocket);
79         }
80         protected void sendMessageLength(int len) throws IOException {
81             serverOut.write (len >> 8);
82             serverOut.write (len);
83         }
84         protected int getMessageLength() throws IOException, ISOException {
85             byte[] b = new byte[2];
86             serverIn.readFully(b,0,2);
87             return (int) (
88                 ((((int)b[0])&0xFF) << 8) | 
89                 (((int)b[1])&0xFF));
90         }
91        /**
92         *      * @param header Hex representation of header
93         */
94         public void setHeader (String header) {
95             super.setHeader (
96                 ISOUtil.hex2byte (header.getBytes(), 0, header.length() / 2)
97             );
98         }
99     }
