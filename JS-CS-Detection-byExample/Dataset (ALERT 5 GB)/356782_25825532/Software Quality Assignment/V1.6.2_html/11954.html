19     package org.jpos.log4j;
20     
21     import java.io.ByteArrayOutputStream;
22     import java.io.PrintStream;
23     
24     import org.apache.log4j.or.ObjectRenderer;
25     import org.jpos.util.LogEvent;
26         
27     /** 
28      * 
29      * This class can be used by the Log4J framework to render LogEvents objects.
30      * 
31      * <pre>
32      * Add the following section into the Log4J configuration
33      * file to use this renderer
34      *
35      * &lt;renderer
36      *     renderedClass="org.jpos.util.LogEvent"
37      *     renderingClass="org.jpos.log4j.LogEventRenderer"&gt;
38      * &lt;/renderer&gt;
39      * </pre>
40      *
41      * @see ObjectRenderer
42      * @see LogEvent
43      * @author Alejandro Revilla (based on Eoin's work)
44      */
45     public class LogEventRenderer implements ObjectRenderer
46     {
47         /**
48          * Convert an ISOMsg to an XML representation
49          * that is suitable for logging.
50          */
51         public String doRender (Object o)
52         {
53             if (o instanceof LogEvent)
54             {
55                 return "\n" + ((LogEvent) o).toString();
56             }
57             else
58                 return "ERROR: LogEventRenderer can only render LogEvent objects";
59         }
60     }
