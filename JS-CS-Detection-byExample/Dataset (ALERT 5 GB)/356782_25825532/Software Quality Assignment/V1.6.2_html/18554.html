19     package org.jpos.iso.gui;
20     
21     /**
22      * allows for easy visualization of channel utilization. 
23      * It shows messages coming through in an
24      * 'Oscilloscope' style clickeable window.
25      * @see ISOMeter
26      * @see ISOMsgPanel
27      * @serial
28      */
29     import java.awt.BorderLayout;
30     import java.awt.FlowLayout;
31     import java.awt.Insets;
32     import java.util.Observable;
33     import java.util.Observer;
34     
35     import javax.swing.BorderFactory;
36     import javax.swing.DefaultListModel;
37     import javax.swing.JLabel;
38     import javax.swing.JPanel;
39     import javax.swing.ListModel;
40     
41     import org.jpos.iso.BaseChannel;
42     import org.jpos.iso.ISOChannel;
43     import org.jpos.iso.ISOException;
44     import org.jpos.iso.ISOMsg;
45     import org.jpos.iso.ISOUtil;
46     
47     public class ISOChannelPanel extends JPanel implements Observer {
48         private static final long serialVersionUID = -8069489863639386589L;
49         /**
50          * @serial
51          */
52         ISOMeter meter;
53         /**
54          * @serial
55          */
56         DefaultListModel log;
57         /**
58          * @serial
59          */
60         String symbolicName;
61     
62         private int[] protectFields = null;
63         private int[] wipeFields    = null;
64     
65         public static final int LOG_CAPACITY = 250;
66     
67         public ISOChannelPanel (ISOChannel channel, String symbolicName)
68         {
69             super();
70             this.symbolicName = symbolicName;
71             setLayout(new FlowLayout());
72             setBorder(BorderFactory.createRaisedBevelBorder());
73             log = new DefaultListModel();
74             add(createCountersPanel());
75             meter.setConnected(channel.isConnected());
76             if (channel instanceof Observable) 
77                 ((Observable)channel).addObserver(this);
78         }
79         /**
80          * Unconnected constructor allows for instantiation of
81          * ISOChannelPanel without associated ISOChannel
82          * (that can be attached later)
83          */
84         public ISOChannelPanel (String symbolicName) {
85             super();
86             this.symbolicName = symbolicName;
87             setLayout(new FlowLayout());
88             setBorder(BorderFactory.createRaisedBevelBorder());
89             log = new DefaultListModel();
90             add(createCountersPanel());
91             meter.setConnected(false);
92         }
93     
94         public final String getSymbolicName() {
95             return symbolicName;
96         }
97         public final ListModel getLog() {
98             return log;
99         }
100     
101         public void update(Observable o, Object arg) {
102             if (arg != null && arg instanceof ISOMsg) {
103                 ISOMsg m = (ISOMsg) arg;
104                 try {
105                     String mti = (String) m.getValue(0);
106                     int imti   = Integer.parseInt(mti);
107                     if (m.isIncoming())
108                         meter.setValue(-imti, mti);
109                     else 
110                         meter.setValue(imti, mti);
111     
112                     // log.insertElementAt(m,0);
113                     log.addElement(getProtectedClone (m));
114                     if (log.getSize() > LOG_CAPACITY) 
115                         log.remove(0);
116     
117                 } catch (ISOException e) { 
118                     meter.setValue(ISOMeter.mass, "ERROR");
119                 }
120                 meter.setValue(ISOMeter.mass);
121             }
122             if (o instanceof BaseChannel) {
123                 BaseChannel c = (BaseChannel) o;
124                 meter.setConnected(c.isConnected());
125                 int cnt[] = c.getCounters();
126                 try {
127                     meter.setPositiveCounter(
128                         ISOUtil.zeropad(Integer.toString(cnt[ISOChannel.TX]), 6)
129                     );
130                     meter.setNegativeCounter(
131                         ISOUtil.zeropad(Integer.toString(cnt[ISOChannel.RX]), 6)
132                     );
133                 } catch (ISOException e) { }
134             } else 
135                 meter.setConnected(true);
136         }
137         public ISOMeter getISOMeter() {
138             return meter;
139         }
140         public void setProtectFields (int[] fields) {
141             protectFields = fields;
142         }
143         public void setWipeFields (int[] fields) {
144             wipeFields    = fields;
145         }
146         private JPanel createCountersPanel() {
147             JPanel A = new JPanel() {
148     
149                 private static final long serialVersionUID = 1175437215105556679L;
150     
151                 public Insets getInsets() {
152                     return new Insets(10,10,10,10);
153                 }
154             };
155     
156             A.setLayout(new BorderLayout());
157     
158             meter = new ISOMeter(this);
159     
160             JLabel l = new JLabel(symbolicName);
161             A.add(l, BorderLayout.NORTH);
162             A.add(meter, BorderLayout.CENTER);
163             meter.start();
164             return A;
165         }
166         private ISOMsg getProtectedClone (ISOMsg m) throws ISOException {
167             ISOMsg pm = (ISOMsg) m.clone ();
168             if (protectFields != null)
169                 checkProtected(pm);
170             if (wipeFields != null)
171                 checkHidden (pm);
172             return pm;
173         }
174         private void checkProtected (ISOMsg m) throws ISOException {
175             for (int i=0; i<protectFields.length; i++) {
176                 int f = protectFields[i];
177                 if (m.hasField (f))
178                     m.set (f, ISOUtil.protect (m.getString(f)));
179             }
180         }
181         private void checkHidden (ISOMsg m) throws ISOException {
182             for (int i=0; i<wipeFields.length; i++) {
183                 int f = wipeFields[i];
184                 if (m.hasField (f))
185                     m.set (f, "[WIPED]");
186             }
187         }
188     }
