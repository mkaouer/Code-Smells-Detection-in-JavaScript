23     /*
24      * BASE1 Header
25      * <pre>
26      *   0 hlen;         Fld  1: Header Length        1B      (Byte     0)
27      *   1 hformat;      Fld  2: Header Format        8N,bit  (Byte     1)
28      *   2 format;       Fld  3: Text Format          1B      (Byte     2)
29      *   3 len[2];       Fld  4: Total Message Length 2B      (Byte  3- 4)
30      *   5 dstId[3];     Fld  5: Destination Id       6N,BCD  (Byte  5- 7)
31      *   8 srcId[3];     Fld  6: Source Id            6N,BCD  (Byte  8-10)
32      *  11 rtCtl;        Fld  7: Round-Trip Ctrl Info 8N,bit  (Byte    11)
33      *  12 flags[2];     Fld  8: BASE I Flags        16N,bit  (Byte 12-13)
34      *  14 status[3];    Fld  9: Message Status Flags 24bits  (Byte 14-16)
35      *  17 batchNbr;     Fld 10: Batch Number        1B       (Byte    17)
36      *  18 reserved[3];  Fld 11: Reserved            3B       (Byte 18-20)
37      *  21 userInfo;     Fld 12: User Info           1B       (Byte    21)
38      *  The following fields are only presend in a reject message
39      *  22 bitmap;       Fld 13: Bitmap              2B       (Byte 22-23)
40      *  24 rejectdata;   Fld 14: Reject Data Group   2B       (Byte 24-25)
41      * </pre>
42      *
43      */
44     public class BASE1Header extends BaseHeader {
45     
46         private static final long serialVersionUID = 6466427524726021374L;
47         public static final int LENGTH = 22;
48     
49         public BASE1Header() {
50             this("000000", "000000");
51         }
52         public BASE1Header(String source, String destination) {
53             super();
54             header = new byte[LENGTH];
55             header[0] = LENGTH; // hlen
56             setHFormat(1);
57             setFormat(2);
58             setSource(source);
59             setDestination(destination);
60         }
61         public BASE1Header(byte[] header) {
62             super(header);
63         }
64         public int unpack(byte[] header) {
65             this.header = header;
66             return header.length;
67         }
68         public int getHLen() {
69             return (int) (header[0] & 0xFF);
70         }
71         public void setHFormat(int hformat) {
72             header[1] = (byte) hformat;
73         }
74         public int getFormat() {
75             return (int) (header[2] & 0xFF);
76         }
77         public void setRtCtl(int i) {
78             header[11] = (byte) i;
79         }
80         public void setFlags(int i) {
81             header[12] = (byte) ((i >> 8) & 0xFF);
82             header[13] = (byte) (i & 0xFF);
83         }
84         public void setStatus(int i) {
85             header[14] = (byte) ((i >> 16) & 0xFF);
86             header[15] = (byte) ((i >> 8) & 0xFF);
87             header[16] = (byte) (i & 0xFF);
88         }
89         public void setBatchNumber(int i) {
90             header[17] = (byte) (i & 0xFF);
91         }
92         public void setFormat(int format) {
93             header[2] = (byte) format;
94         }
95         public void setLen(int len) {
96             len += header.length;
97             header[3]  = (byte) ((len >> 8) & 0xff);
98             header[4]  = (byte) (len        & 0xff);
99         }
100         public void setDestination(String dest) {
101             byte[] d = ISOUtil.str2bcd(dest, true);
102             System.arraycopy(d, 0, header, 5, 3);
103         }
104         public void setSource(String src) {
105             byte[] d = ISOUtil.str2bcd(src, true);
106             System.arraycopy(d, 0, header, 8, 3);
107         }
108         public String getSource() {
109             return ISOUtil.bcd2str (this.header, 8, 6, false);
110         }
111         public String getDestination() {
112             return ISOUtil.bcd2str (this.header, 5, 6, false);
113         }
114         public void swapDirection() {
115             if (header != null && header.length >= LENGTH) {
116                 byte[] source = new byte[3];
117                 System.arraycopy(header, 8, source, 0, 3);
118                 System.arraycopy(header, 5, header, 8, 3);
119                 System.arraycopy(source, 0, header, 5, 3);
120             }
121         }
122         public boolean isRejected() {
123             // Header length must be 26 or gerater
124             // And header field 13 bit 1 must be 1 (field 13 starts at byte 22)
125             return (getLength() >= 26) && ((header[22] & 0x80) == 0x80);
126         }
127     	
128         /**
129          * Gets the BASE 1 Reject Code.
130          * 
131          * @return If the message is a reject return the Reject Code Otherwise return "" 
132          */ 
133         public String getRejectCode() {
134             return isRejected() ? ISOUtil.bcd2str (this.header, 24, 4, false) : "";
135         }
