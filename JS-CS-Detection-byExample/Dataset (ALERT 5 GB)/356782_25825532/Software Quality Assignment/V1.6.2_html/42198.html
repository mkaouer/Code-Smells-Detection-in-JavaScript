38     /** A TransactionParticipant whichï¿½s prepare, commit and abort methods can be 
39      *  specified through beanshell scripts. <BR>
40      *
41      *  To indicate what code to execute for any of the methods just add an element 
42      *  named 'prepare', 'commit' or 'abort' contained in that of the participant. <BR>
43      *
44      *  See BSHMethod for details on the syntax of these elemets. The value to return 
45      *  in the prepare method should be stored in the script variable named "result".
46      *  None of these tags are mandatory. <BR>
47      *
48      *  You can subclass BSHTransactionParticipant and override the default... 
49      *  methods. That way you can provide default behaviour for a participant and 
50      *  override it at deploy time through scripts. 
51      * 
52      * @see BSHMethod
53      * @author  AMarques
54      */
55     public class BSHTransactionParticipant extends SimpleLogSource 
56         implements TransactionParticipant, AbortParticipant, XmlConfigurable 
57     {
58         
59         protected BSHMethod prepareMethod;
60         protected BSHMethod prepareForAbortMethod;
61         protected BSHMethod commitMethod;
62         protected BSHMethod abortMethod;
63     
64         boolean trace;
65         
66         /** Creates a new instance of BSHTransactionParticipant */
67         public BSHTransactionParticipant() {
68             super();
69         }
70         
71         public void abort(long id, java.io.Serializable context) {
72             LogEvent ev = new LogEvent(this, "abort");
73             if (abortMethod != null) {
74                 try {
75                     executeMethod(abortMethod, id, context, ev, "");
76                 } catch (Exception ex) {
77                     ev.addMessage(ex);
78                 }
79             } else {
80                 defaultAbort(id, context, ev);
81             }
82             if (trace)
83                 Logger.log(ev);
84         }
85         
86         protected void defaultAbort(long id, Serializable context, LogEvent ev) {}
87         
88         public void commit(long id, java.io.Serializable context) {
89             LogEvent ev = new LogEvent(this, "commit");
90             if (commitMethod != null) {
91                 try {
92                     executeMethod(commitMethod, id, context, ev, "");
93                 } catch (Exception ex) {
94                     ev.addMessage(ex);
95                 }
96             } else {
97                 defaultCommit(id, context, ev);
98             }
99             if (trace)
100                 Logger.log(ev);
101         }
102         
103         protected void defaultCommit(long id, Serializable context, LogEvent ev) {}
104         
105         public int prepare(long id, java.io.Serializable context) {
106             LogEvent ev = new LogEvent(this, "prepare");
107             int result = ABORTED | READONLY;
108             if (prepareMethod != null) {
109                 try {
110                     result = ((Integer) executeMethod(prepareMethod, id, context, ev, "result")).intValue();
111                 } catch (Exception ex) {
112                     ev.addMessage(ex);
113                 }
114             } else {
115                 result = defaultPrepare(id, context, ev);
116             }
117             ev.addMessage("result", Integer.toBinaryString(result));
118             if (trace)
119                 Logger.log(ev);
120             return result;
121         }
122     
123         public int prepareForAbort(long id, java.io.Serializable context) {
124             LogEvent ev = new LogEvent(this, "prepare-for-abort");
125             int result = ABORTED | READONLY;
126             if (prepareForAbortMethod != null) {
127                 try {
128                     result = ((Integer) executeMethod(prepareForAbortMethod, id, context, ev, "result")).intValue();
129                 } catch (Exception ex) {
130                     ev.addMessage(ex);
131                 }
132             } 
133             ev.addMessage("result", Integer.toBinaryString(result));
134             if (trace)
135                 Logger.log(ev);
136             return result;
137         }
138         
139         protected int defaultPrepare(long id, Serializable context, LogEvent ev) {
140             return PREPARED | READONLY;
141         }
142         
143         public void setConfiguration(Element e) throws ConfigurationException {
144     	try {
145                 prepareMethod = BSHMethod.createBshMethod(e.getChild("prepare"));
146                 prepareForAbortMethod = BSHMethod.createBshMethod(e.getChild("prepare-for-abort"));
147                 commitMethod = BSHMethod.createBshMethod(e.getChild("commit"));
148                 abortMethod = BSHMethod.createBshMethod(e.getChild("abort"));
149                 trace = "yes".equals (e.getAttributeValue ("trace"));
150             } catch (Exception ex) {
151                 throw new ConfigurationException(ex.getMessage(), ex);
152             }
153         }
154         
155         protected Object executeMethod(BSHMethod m, long id, Serializable context, LogEvent evt, String resultName) 
156         throws EvalError, FileNotFoundException, IOException {
157             Map params = new HashMap();
158             params.put("context", context);
159             params.put("id", new Long(id));
160             params.put("evt", evt);
161             params.put("self", this);
162             return m.execute(params, resultName);
163         }
164     }
