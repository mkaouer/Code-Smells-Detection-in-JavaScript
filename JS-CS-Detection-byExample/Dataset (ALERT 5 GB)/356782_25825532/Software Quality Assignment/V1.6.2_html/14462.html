19     package org.jpos.tpl;
20     
21     import java.sql.Connection;
22     import java.sql.DriverManager;
23     import java.sql.ResultSet;
24     import java.sql.SQLException;
25     import java.sql.Statement;
26     
27     import org.jpos.core.Configurable;
28     import org.jpos.core.Configuration;
29     import org.jpos.core.ConfigurationException;
30     import org.jpos.util.LogEvent;
31     import org.jpos.util.LogSource;
32     import org.jpos.util.Logger;

34     /**
35      * DataSource implementation used by PersistentPeer implementations
36      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
37      * @version $Id$
38      */
39     public class PersistentEngine implements LogSource, Configurable {
40         Configuration cfg;
41         Logger logger;
42         String realm;
43     
44         /**
45          * @param cfg Configuration
46          * @param logger logger
47          * @param realm logger's realm
48          * @throws ConfigurationException
49          */
50         public PersistentEngine (Configuration cfg, Logger logger, String realm) 
51             throws ConfigurationException
52         {
53             super();
54             this.cfg = cfg;
55             this.logger = logger;
56             this.realm  = realm;
57             initEngine ();
58         }
59         /**
60          * no args constructor
61          */
62         public PersistentEngine () {
63             super();
64         }
65         /**
66          * Implements Configurable
67          * @param cfg Configuration
68          * @throws ConfigurationException
69          */
70         public void setConfiguration (Configuration cfg)
71             throws ConfigurationException
72         {
73             this.cfg = cfg;
74             initEngine();
75         }
76         private void initEngine () throws ConfigurationException {
77             initJDBC();
78         }
79         private void initJDBC() throws ConfigurationException {
80             try {
81                 Class.forName(cfg.get("jdbc.driver")).newInstance();
82             } catch (Exception e) {
83                 throw new ConfigurationException (e);
84             }
85         }
86         public synchronized Connection getConnection() {
87             for (;;) {
88                 try {
89                     String url  = cfg.get ("jdbc.url");
90                     String user = cfg.get ("jdbc.user");
91                     String pass = cfg.get ("jdbc.password");
92                     return DriverManager.getConnection(url,user,pass);
93                 } catch (SQLException e) {
94                     Logger.log (new LogEvent(this, "sql-connection", e));
95                     try {
96                         Thread.sleep (2000);
97                     } catch (InterruptedException ex) { }
98                 }
99             }
100         }
101         public void releaseConnection (Connection conn) {
102             // Connection pooling hook
103             try {
104                 conn.close();
105             } catch (SQLException e) {
106                 Logger.log (new LogEvent(this, "sql-release-connection", e));
107             }
108         }
109         public void setLogger (Logger logger, String realm) {
110             this.logger = logger;
111             this.realm  = realm;
112         }
113         public String getRealm () {
114             return realm;
115         }
116         public Logger getLogger() {
117             return logger;
118         }
119         /**
120          * finds a peer for this object
121          * @param obj Main object
122          * @throws NoPeerException
123          */
124         public PersistentPeer getPeer (Object obj) throws NoPeerException {
125             if (obj instanceof PersistentPeer) 
126                 return (PersistentPeer) obj;
127             else {
128                 try {
129                     Class c = Class.forName (obj.getClass().getName()+"Peer");
130                     return (PersistentPeer) c.newInstance();
131                 } catch (Exception e) {
132                     throw new NoPeerException (e.toString());
133                 }
134             }
135         }
136         /**
137          * Execute SQL Update
138          * @param sql sql command
139          * @exception SQLException
140          */
141         public void executeUpdate (String sql) throws SQLException {
142             Connection conn = getConnection();
143             try {
144                 executeUpdate (sql, conn);
145             } finally {
146                 releaseConnection (conn);
147             }
148         }
149         /**
150          * Execute SQL Update
151          * @param sql  sql command
152          * @param conn sql connection
153          * @exception SQLException
154          */
155         public void executeUpdate (String sql, Connection conn) 
156             throws SQLException
157         {
158             Statement s = null;
159             try {
160                 s = conn.createStatement();
161                 if (logger != null && logger.hasListeners()) 
162                     Logger.log (new LogEvent (this, "sql-update", sql));
163                 s.executeUpdate (sql);
164             } finally {
165                 if (s != null)
166                     s.close();
167             }
168         }
169     
170         /**
171          * Execute SQL Query. 
172          * @param sql  sql command
173          * @param conn sql connection
174          * @return ResultSet (please close() it after using - thanks)
175          * @exception SQLException
176          */
177         public ResultSet executeQuery (String sql, Connection conn) 
178             throws SQLException
179         {
180             Statement s = null;
181             ResultSet rs;
182             s = conn.createStatement();
183             if (logger != null && logger.hasListeners()) 
184                 Logger.log (new LogEvent (this, "sql-query", sql));
185             return s.executeQuery (sql);
186         }
187     
188         /**
189          * creates a new persistent object
190          * @param o object to create
191          * @throws NoPeerException
192          * @throws SQLException
193          */
194         public void create (Object o) 
195             throws NoPeerException, SQLException
196         {
197             PersistentPeer peer = getPeer(o);
198             peer.setPersistentEngine (this);
199             peer.create (o);
200         }
201         /**
202          * load object from persistent storage
203          * @param o object to load
204          * @throws NoPeerException
205          * @throws SQLException
206          */
207         public void load (Object o) 
208             throws NoPeerException, SQLException, NotFoundException
209         {
210             PersistentPeer peer = getPeer(o);
211             peer.setPersistentEngine (this);
212             peer.load (o);
213         }
214         /**
215          * remove object from persistent storage
216          * @param o object to load
217          * @throws NoPeerException
218          * @throws SQLException
219          */
220         public void remove (Object o) 
221             throws NoPeerException, SQLException, NotFoundException
222         {
223             PersistentPeer peer = getPeer(o);
224             peer.setPersistentEngine (this);
225             peer.remove (o);
226         }
227         /**
228          * update object to persistent storage
229          * @param o object to load
230          * @throws NoPeerException
231          * @throws SQLException
232          */
233         public void update (Object o) 
234             throws NoPeerException, SQLException, NotFoundException
235         {
236             PersistentPeer peer = getPeer(o);
237             peer.setPersistentEngine (this);
238             peer.update(o);
239         }
240     
241         public long getOID (Connection conn) throws SQLException {
242             String sql = "SELECT last_insert_id()";
243             ResultSet rs = executeQuery (sql, conn);
244             if (rs.isBeforeFirst()) 
245                 rs.next();
246             long oid = rs.getLong (1);
247             rs.close();
248             return oid;
249         }
250     }
