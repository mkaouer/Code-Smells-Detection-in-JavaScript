19     package org.jpos.util;
20     
21     /**
22      * Represents a LogSource and adds several helpers
23      *
24      * @author apr@cs.com.uy
25      * @version $Revision$ $Date$
26      * @see LogSource
27      */
28     public class Log implements LogSource {
29         protected Logger logger;
30         protected String realm;
31     
32         public static final String TRACE   = "trace";
33         public static final String DEBUG   = "debug";
34         public static final String INFO    = "info";
35         public static final String WARN    = "warn";
36         public static final String ERROR   = "error";
37         public static final String FATAL   = "fatal";
38     
39         public Log () {
40             super();
41         }
42         public static Log getLog (String logName, String realm) {
43             return new Log (Logger.getLogger (logName), realm);
44         }
45         public Log (Logger logger, String realm) {
46             setLogger (logger, realm);
47         }
48         public void setLogger (Logger logger, String realm) {
49             this.logger = logger;
50             this.realm  = realm;
51         }
52         public String getRealm () {
53             return realm;
54         }
55         public Logger getLogger() {
56             return logger;
57         }
58         public void setLogger (Logger logger) {
59             this.logger = logger;
60         }
61         public void setRealm (String realm) {
62             this.realm = realm;
63         }
64         public void trace (Object detail) {
65             Logger.log (createTrace (detail));
66         }
67         public void trace (Object detail, Object obj) {
68             LogEvent evt = createTrace (detail);
69             evt.addMessage (obj);
70             Logger.log (evt);
71         }
72         public void debug (Object detail) {
73             Logger.log (createDebug (detail));
74         }
75         public void debug (Object detail, Object obj) {
76             LogEvent evt = createDebug (detail);
77             evt.addMessage (obj);
78             Logger.log (evt);
79         }
80         public void info (Object detail) {
81             Logger.log (createInfo (detail));
82         }
83         public void info (Object detail, Object obj) {
84             LogEvent evt = createInfo (detail);
85             evt.addMessage (obj);
86             Logger.log (evt);
87         }
88         public void warn (Object detail) {
89             Logger.log (createWarn (detail));
90         }
91         public void warn (Object detail, Object obj) {
92             LogEvent evt = createWarn (detail);
93             evt.addMessage (obj);
94             Logger.log (evt);
95         }
96         public void error (Object detail) {
97             Logger.log (createError (detail));
98         }
99         public void error (Object detail, Object obj) {
100             LogEvent evt = createError (detail);
101             evt.addMessage (obj);
102             Logger.log (evt);
103         }
104         public void fatal (Object detail) {
105             Logger.log (createFatal (detail));
106         }
107         public void fatal (Object detail, Object obj) {
108             LogEvent evt = createFatal (detail);
109             evt.addMessage (obj);
110             Logger.log (evt);
111         }
112         public LogEvent createLogEvent (String level) {
113             return new LogEvent (this, level);
114         }
115         public LogEvent createLogEvent (String level, Object detail) {
116             return new LogEvent (this, level, detail);
117         }
118         public LogEvent createTrace () {
119             return createLogEvent (TRACE);
120         }
121         public LogEvent createTrace (Object detail) {
122             return createLogEvent (TRACE, detail);
123         }
124         public LogEvent createDebug() {
125             return createLogEvent (TRACE);
126         }
127         public LogEvent createDebug(Object detail) {
128             return createLogEvent (TRACE, detail);
129         }
130         public LogEvent createInfo () {
131             return createLogEvent (INFO);
132         }
133         public LogEvent createInfo (Object detail) {
134             return createLogEvent (INFO, detail);
135         }
136         public LogEvent createWarn () {
137             return createLogEvent (WARN);
138         }
139         public LogEvent createWarn (Object detail) {
140             return createLogEvent (WARN, detail);
141         }
142         public LogEvent createError () {
143             return createLogEvent (ERROR);
144         }
145         public LogEvent createError (Object detail) {
146             return createLogEvent (ERROR, detail);
147         }
148         public LogEvent createFatal () {
149             return createLogEvent (FATAL);
150         }
151         public LogEvent createFatal (Object detail) {
152             return createLogEvent (FATAL, detail);
153         }
154     }
