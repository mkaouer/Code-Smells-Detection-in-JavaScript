19     package org.jpos.util;
20     
21     import junit.framework.*;
22     
23     
24     public class ThroughputControlTestCase extends TestCase {
25         public void testSingleThread () throws Exception {
26             ThroughputControl tc = new ThroughputControl (2, 1000);
27             long start = System.currentTimeMillis();
28             assertTrue ("Control should return 0L", tc.control() == 0L);
29             assertTrue (
30                 "Elapsed time should be less than one second", 
31                 System.currentTimeMillis() - start < 1000L
32             );
33             tc.control();
34             assertTrue (
35                 "Elapsed time should still be less than one second", 
36                 System.currentTimeMillis() - start < 1000L
37             );
38             tc.control();
39             assertTrue (
40                 "Elapsed time should be greater than one second", 
41                 System.currentTimeMillis() - start > 1000L
42             );
43             tc.control();
44             assertTrue (
45                 "second transaction should be less than two seconds", 
46                 System.currentTimeMillis() - start < 2000L
47             );
48         }
49         public void testFifty () throws Exception {
50             ThroughputControl tc = new ThroughputControl (10, 1000);
51             long start = System.currentTimeMillis();
52             for (int i=0; i<50; i++)
53                 tc.control();
54     
55             long elapsed =  System.currentTimeMillis() - start;
56             assertTrue (
57                 "50 transactions should take at least 4 seconds but took " + elapsed, 
58                 elapsed >= 4000L
59             );
60             assertTrue (
61                 "50 transactions shouldn't take more than aprox 4 seconds but took " + elapsed, 
62                 elapsed < 4300L
63             );
64         }
65         public void testDualPeriod () throws Exception {
66             ThroughputControl tc = new ThroughputControl (
67                 new int[] { 100, 150 }, 
68                 new int[] { 1000, 5000 }
69             );
70             long start = System.currentTimeMillis();
71             for (int i=0; i<100; i++)
72                 tc.control();
73     
74             long elapsed =  System.currentTimeMillis() - start;
75             assertTrue (
76                 "100 initial transactions should take more than about one second but took " + elapsed, 
77                 elapsed <= 1000L
78             );
79             for (int i=0; i<100; i++)
80                 tc.control();
81     
82             elapsed =  System.currentTimeMillis() - start;
83             assertTrue (
84                 "100 additional transactions should take more than five seconds but took " + elapsed, 
85                 elapsed > 5000L
86             );
87         }
88         public void testMultiThread() throws Exception {
89             final ThroughputControl tc = new ThroughputControl (2, 1000);
90             long start = System.currentTimeMillis();
91             Thread[] t = new Thread[10];
92             for (int i=0; i<10; i++) {
93                 t[i] = new Thread() {
94                     public void run() {
95                         tc.control();
96                     }
97                 };
98                 t[i].start();
99             }
100             for (int i=0; i<10; i++) {
101                 t[i].join();
102             }
103             long elapsed =  System.currentTimeMillis() - start;
104             assertTrue (
105                 "10 transactions should take about four seconds but took " + elapsed, 
106                 elapsed > 4000L && elapsed < 5000L
107             );
108         }
109     }
