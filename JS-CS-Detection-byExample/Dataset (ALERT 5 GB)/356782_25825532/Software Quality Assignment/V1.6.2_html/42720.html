39     /**
40      * MacroFilter useful to set sequencers, date, unset iso fields, etc.
41      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
42      * @version $Revision$ $Date$
43      */
44     public class MacroFilter implements ISOFilter, ReConfigurable {
45         Sequencer seq;
46         Configuration cfg;
47         int[] unsetFields  = new int[0];
48         int[] validFields  = new int[0];
49         public MacroFilter () {
50             super();
51         }
52        /**
53         * @param cfg
54         * <ul>
55         *  <li>sequencer - a sequencer used to store counters
56         *  <li>unset - space delimited list of fields to be unset 
57         *  <li>valid - space delimited list of valid fields 
58         *  <li>comma delimited list of fields to be unset when applying filter
59         *  <li>xzy - property named "xyz"
60         * </ul>
61         */
62         public void setConfiguration (Configuration cfg) 
63             throws ConfigurationException
64         {
65             this.cfg = cfg;
66             try {
67                 String seqName  = cfg.get ("sequencer", null);
68                 unsetFields     = ISOUtil.toIntArray (cfg.get ("unset", ""));
69                 validFields     = ISOUtil.toIntArray (cfg.get ("valid", ""));
70                 if (seqName != null) {
71                     seq = (Sequencer) NameRegistrar.get (
72                         "sequencer."+cfg.get("sequencer")
73                     );
74                 } else if (seq == null) {
75                     seq = new VolatileSequencer();
76                 }
77             } catch (NameRegistrar.NotFoundException e) {
78                 throw new ConfigurationException (e);
79             }
80         }
81     
82         public ISOMsg filter (ISOChannel channel, ISOMsg m, LogEvent evt) 
83             throws VetoException
84         {
85             try {
86                 applyProps (m);
87                 if (validFields.length > 0) 
88                     m = (ISOMsg) m.clone (validFields);
89                 if (unsetFields.length > 0)
90                     m.unset (unsetFields);
91             } catch (ISOException e) {
92                 evt.addMessage (e);
93                 throw new VetoException (e);
94             }
95             return m;
96         }
97         private void applyProps (ISOMsg m) throws ISOException {
98             int maxField = m.getMaxField ();
99             for (int i=0; i<=maxField; i++) {
100                 Object o = null;
101                 if (m.hasField (i))
102                     o = m.getValue (i);
103                 if (o instanceof String) {
104                     String value = (String) o;
105                     if (value.length () == 0)
106                         continue;
107                     if (value.equalsIgnoreCase ("$date") )
108                         m.set (new ISOField (i, ISODate.getDateTime(new Date())));
109                     else if ((value.toLowerCase().startsWith ("$date") ) && (value.indexOf("GMT") > 0)) {
110                         String zoneID = value.substring(value.indexOf("GMT"));
111                         m.set (new ISOField (i, ISODate.getDateTime(new Date(), TimeZone.getTimeZone(zoneID))));
112                     }                    
113                     else if (value.charAt (0) == '#')
114                         m.set (new ISOField (i,
115                           ISOUtil.zeropad (
116                             Integer.toString(seq.get (value.substring(1))),6)
117                           )
118                         );
119                     else if (value.charAt (0) == '$') {
120                         String p = cfg.get (value.substring(1), null);
121                         if (p != null)
122                             m.set (new ISOField (i, p));
123                     }
124                 }
125             }
126         }
127     }
