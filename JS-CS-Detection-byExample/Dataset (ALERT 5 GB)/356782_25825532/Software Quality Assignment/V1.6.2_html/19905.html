19     package org.jpos.transaction;
20     
21     import java.io.ObjectOutputStream;
22     import java.io.ObjectInputStream;
23     import java.io.ByteArrayOutputStream;
24     import java.io.ByteArrayInputStream;
25     import org.jpos.iso.ISOUtil;
26     import junit.framework.*;
27     
28     public class ContextTestCase extends TestCase {
29         public void testExternalizable () throws Exception {
30             Context ctx = new Context();
31             ctx.put ("volatile", "the quick brown fox");
32             ctx.put ("persistent", "jumped over the lazy dog", true);
33     
34             ByteArrayOutputStream baos = new ByteArrayOutputStream();
35             ObjectOutputStream os = new ObjectOutputStream (baos);
36             os.writeObject (ctx);
37     
38             ByteArrayInputStream bain = new ByteArrayInputStream (baos.toByteArray());
39             ObjectInputStream is = new ObjectInputStream (bain);
40             Context ctx1 = (Context) is.readObject();
41     
42             // ctx.dump (System.out, "ctx> ");
43             // ctx1.dump (System.out, "ctx1>");
44     
45             assertNull (ctx1.getString ("volatile"));
46             assertNotNull (ctx1.getString ("persistent"));
47             assertEquals ("jumped over the lazy dog", ctx1.getString ("persistent"));
48         }
49     }
