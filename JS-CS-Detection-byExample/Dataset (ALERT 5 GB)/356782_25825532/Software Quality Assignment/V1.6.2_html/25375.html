19     package org.jpos.transaction.participant;
20     
21     import java.io.Serializable;
22     import org.jpos.core.Configurable;
23     import org.jpos.core.Configuration;
24     import org.jpos.transaction.Context;
25     import org.jpos.transaction.TransactionParticipant;
26     
27     public class CheckPoint implements TransactionParticipant, Configurable  {
28         Configuration cfg;
29         public int prepare (long id, Serializable o) {
30             if (o instanceof Context) 
31                 ((Context)o).checkPoint (cfg.get ("message", "checkpoint"));
32     
33             return PREPARED | NO_JOIN | READONLY;
34         }
35         public void commit (long id, Serializable o) { }
36         public void abort  (long id, Serializable o) { }
37         public void setConfiguration (Configuration cfg) {
38             this.cfg = cfg;
39         }
40     }
