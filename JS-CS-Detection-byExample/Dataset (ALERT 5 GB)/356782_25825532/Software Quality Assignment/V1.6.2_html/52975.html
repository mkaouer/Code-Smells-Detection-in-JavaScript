19     package org.jpos.transaction.participant;
20     
21     import java.io.Serializable;
22     import java.util.List;
23     import java.util.ArrayList;
24     import java.util.Iterator;
25     import org.jpos.transaction.Context;
26     import org.jpos.transaction.TransactionConstants;
27     import org.jpos.transaction.TransactionParticipant;
28     import org.jpos.transaction.AbortParticipant;
29     import org.jpos.core.Configurable;
30     import org.jpos.core.Configuration;
31     import org.jpos.core.ConfigurationException;
32     import org.jpos.core.XmlConfigurable;
33     import org.jpos.transaction.TransactionManager;
34     import org.jdom.Element;
35     
36     public class Join 
37            implements TransactionConstants, AbortParticipant, 
38                       Configurable, XmlConfigurable 
39     {
40         private Configuration cfg;
41         private TransactionManager mgr;
42         private List participants = new ArrayList ();
43     
44         public int prepare (long id, Serializable o) {
45             return mergeActions(
46                 joinRunners(prepare (createRunners(id, o)))
47             );
48         }
49         public int prepareForAbort  (long id, Serializable o) { 
50             return mergeActions(
51                 joinRunners(prepareForAbort (createRunners(id, o)))
52             );
53         }
54         public void commit (long id, Serializable o) { 
55             joinRunners(commit (createRunners(id, o)));
56         }
57         public void abort  (long id, Serializable o) { 
58             joinRunners(abort (createRunners(id, o)));
59         }
60         public void setConfiguration (Configuration cfg) {
61             this.cfg = cfg;
62         }
63         public void setConfiguration (Element e) 
64             throws ConfigurationException
65         {
66             Iterator iter = e.getChildren ("participant").iterator();
67             while (iter.hasNext()) {
68                 participants.add (mgr.createParticipant ((Element) iter.next()));
69             }
70         }
71         public void setTransactionManager (TransactionManager mgr) {
72             this.mgr = mgr;
73         }
74         private Runner[] prepare (Runner[] runners) {
75             for (int i=0; i<runners.length; i++)
76                 runners[i].prepare();
77             return runners;
78         }
79         private Runner[] prepareForAbort (Runner[] runners) {
80             for (int i=0; i<runners.length; i++)
81                 runners[i].prepareForAbort();
82             return runners;
83         }
84         private Runner[] commit (Runner[] runners) {
85             for (int i=0; i<runners.length; i++)
86                 runners[i].commit();
87             return runners;
88         }
89         private Runner[] abort (Runner[] runners) {
90             for (int i=0; i<runners.length; i++)
91                 runners[i].abort();
92             return runners;
93         }
94         private Runner[] createRunners(long id, Serializable o) {
95             Runner[] runners = new Runner[participants.size()];
96             Iterator iter = participants.iterator();
97             for (int i=0; iter.hasNext(); i++) {
98                 runners[i] = new Runner (
99                     (TransactionParticipant) iter.next(), id, o
100                 );
101             }
102             return runners;
103         }
104         private Runner[] joinRunners (Runner[] runners) {
105             for (int i=0; i<runners.length; i++)
106                 runners[i].join();
107             return runners;
108         }
109         private int mergeActions (Runner[] runners) {
110             boolean prepared = true;
111             boolean readonly = true;;
112             boolean no_join = true;
113             boolean retry = false;
114             for (int i=0; i<runners.length; i++) {
115                 int action = runners[i].rc;
116                 retry  = (action & RETRY) == RETRY;
117                 if (retry)
118                     return RETRY; 
119                if ((action & PREPARED) == ABORTED)
120                    prepared = false;;
121                if ((action & READONLY) != READONLY)
122                    readonly = false;
123                if ((action & NO_JOIN) != NO_JOIN)
124                    no_join = false;
125             }
126             return (prepared ? PREPARED : ABORTED) |
127                    (no_join  ? NO_JOIN  : 0) |
128                    (readonly ? READONLY : 0);
129         }
130         public static class Runner implements Runnable {
131             TransactionParticipant p;
132             public int rc;
133             long id;
134             int mode;
135             Serializable ctx;
136             Thread t;
137             public static final int PREPARE = 0;
138             public static final int PREPARE_FOR_ABORT = 1;
139             public static final int COMMIT = 2;
140             public static final int ABORT = 3;
141             public static final String[] MODES = {
142                 "prepare", "prepareForAbort", "commit", "abort"
143             };
144     
145             public Runner (TransactionParticipant p, long id, Serializable ctx) {
146                 this.p = p;
147                 this.id = id;
148                 this.ctx = ctx;
149             }
150             public void prepare() {
151                 createThread (PREPARE);
152             }
153             public void prepareForAbort() {
154                 createThread (PREPARE_FOR_ABORT);
155             }
156             public void commit () {
157                 createThread (COMMIT);
158             }
159             public void abort () {
160                 createThread (ABORT);
161             }
162             public void run() {
163                 switch (mode) {
164                     case PREPARE:
165                         rc = p.prepare(id, ctx);
166                         break;
167                     case PREPARE_FOR_ABORT:
168                         if (p instanceof AbortParticipant)
169                             rc = ((AbortParticipant)p).prepareForAbort (id, ctx);
170                         break;
171                     case COMMIT:
172                         if ((rc & NO_JOIN) == 0)
173                             p.commit (id, ctx);
174                         break;
175                     case ABORT:
176                         if ((rc & NO_JOIN) == 0)
177                             p.abort (id, ctx);
178                         break;
179                 }
180             }
181             public void join () {
182                 try {
183                     t.join ();
184                 } catch (InterruptedException e) { }
185             }
186             private void createThread (int m) {
187                 this.t = new Thread(this);
188                 t.setName (
189                     MODES[m] +
190                     this.getClass().getName() + ":" + p.getClass().getName()
191                 );
192                 this.mode = m;
193                 t.start();
194             }
195         }
196     }
