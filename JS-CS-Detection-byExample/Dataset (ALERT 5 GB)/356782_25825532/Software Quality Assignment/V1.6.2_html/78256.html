19     package org.jpos.util;
20     
21     import java.util.LinkedList;
22     
23     /**
24      * implements a blocking queue 
25      * @see ThreadPool
26      * @since 1.1
27      */
28     public class BlockingQueue {
29         private LinkedList queue = new LinkedList();
30         private boolean closed = false;
31         private int consumers = 0;
32     
33         public static class Closed extends RuntimeException {
34     
35             private static final long serialVersionUID = 3404885702116373450L;
36     
37             public Closed() {
38                 super ("queue closed");
39             }
40         }
41     
42         public synchronized void enqueue (Object o) throws Closed {
43             if (closed)
44                 throw new Closed();
45             queue.addLast (o);
46             notify();
47         }
48         public synchronized void requeue (Object o) throws Closed {
49             if (closed)
50                 throw new Closed();
51             queue.addFirst (o);
52             notify();
53         }
54     
55         public synchronized Object dequeue()
56             throws InterruptedException, Closed
57         {
58             consumers++;
59             try {
60                 while (queue.size() == 0) {
61                     wait();
62                     if (closed)
63                         throw new Closed();
64                 }
65             } finally {
66                 consumers--;
67             }
68             return queue.removeFirst();
69         }
70     
71         public synchronized Object dequeue (long timeout)
72             throws InterruptedException, Closed
73         {
74             if (timeout == 0)
75                 return dequeue ();
76     
77             consumers++;
78             long maxTime = System.currentTimeMillis() + timeout;
79             try {
80                 while (queue.size() == 0 && System.currentTimeMillis() < maxTime) {
81                     wait (timeout);
82                     if (closed)
83                         throw new Closed();
84                 }
85             } finally {
86                 consumers--;
87             }
88             return queue.size() > 0 ? queue.removeFirst() : null;
89         }
90         public synchronized void close() {
91             closed = true;
92             notifyAll();
93         }
94         public synchronized int consumerCount() {
95             return consumers;
96         }
97         public synchronized boolean ready() {
98             return !closed;
99         }
100         public synchronized int pending() {
101             return queue.size();
102         }
103         public LinkedList getQueue () {
104             return queue;
105         }
106         public void setQueue (LinkedList queue) {
107             this.queue = queue;
108         }
109     }
