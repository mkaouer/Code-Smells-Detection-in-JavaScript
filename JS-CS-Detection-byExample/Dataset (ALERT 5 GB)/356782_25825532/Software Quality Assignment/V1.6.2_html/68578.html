19     package org.jpos.q2.qbean;
20     
21     import java.io.PrintStream;
22     
23     import org.jpos.q2.QBeanSupport;
24     import org.jpos.util.Loggeable;
25     import org.jpos.util.Logger;
26     import org.jpos.util.NameRegistrar;
27     
28     /**
29      * Periodically dumps Thread and memory usage
30      * 
31      * @author apr@cs.com.uy
32      * @version $Id$
33      * @jmx:mbean description="System Monitor"
34      *            extends="org.jpos.q2.QBeanSupportMBean"
35      * @see Logger
36      */
37     public class SystemMonitor extends QBeanSupport implements Runnable,
38             SystemMonitorMBean, Loggeable {
39     
40         private long sleepTime = 60 * 60 * 1000;
41     
42         private long delay = 0;
43     
44         private boolean detailRequired = false;
45     
46         private Thread me = null;
47     
48         public void startService() {
49             try {
50                 log.info("Starting SystemMonitor");
51                 me = new Thread(this,"SystemMonitor");
52                 me.start();
53             } catch (Exception e) {
54                 log.warn("error starting service", e);
55             }
56         }
57     
58         public void stopService() {
59             log.info("Stopping SystemMonitor");
60             if (me != null)
61                 me.interrupt();
62         }
63     
64         /**
65          * @jmx:managed-attribute description="Milliseconds between dump"
66          */
67         public synchronized void setSleepTime(long sleepTime) {
68             this.sleepTime = sleepTime;
69             setModified(true);
70             if (me != null)
71                 me.interrupt();
72         }
73     
74         /**
75          * @jmx:managed-attribute description="Milliseconds between dump"
76          */
77         public synchronized long getSleepTime() {
78             return sleepTime;
79         }
80     
81         /**
82          * @jmx:managed-attribute description="Detail required?"
83          */
84         public synchronized void setDetailRequired(boolean detail) {
85             this.detailRequired = detail;
86             setModified(true);
87             if (me != null)
88                 me.interrupt();
89         }
90     
91         /**
92          * @jmx:managed-attribute description="Detail required?"
93          */
94         public synchronized boolean getDetailRequired() {
95             return detailRequired;
96         }
97     
98         void dumpThreads(ThreadGroup g, PrintStream p, String indent) {
99             Thread[] list = new Thread[g.activeCount() + 5];
100             int nthreads = g.enumerate(list);
101             for (int i = 0; i < nthreads; i++)
102                 p.println(indent + list[i]);
103         }
104     
105         public void showThreadGroup(ThreadGroup g, PrintStream p, String indent) {
106             if (g.getParent() != null)
107                 showThreadGroup(g.getParent(), p, indent + "  ");
108             else
109                 dumpThreads(g, p, indent + "    ");
110         }
111     
112         public void run() {
113             while (running()) {
114                 log.info(this);
115                 try {
116                     long expected = System.currentTimeMillis() + sleepTime;
117                     Thread.sleep(sleepTime);
118                     delay = (System.currentTimeMillis() - expected);
119                 } catch (InterruptedException e) {
120                 }
121             }
122         }
123     
124         public void dump(PrintStream p, String indent) {
125             String newIndent = indent + "  ";
126             Runtime r = Runtime.getRuntime();
127             p.println (indent+"<release>"+getServer().getRelease()+"</release>");
128             p.println(indent + "<memory>");
129             p.println(newIndent + " freeMemory=" + r.freeMemory());
130             p.println(newIndent + "totalMemory=" + r.totalMemory());
131             p.println(newIndent + "inUseMemory="
132                     + (r.totalMemory() - r.freeMemory()));
133             p.println(indent + "</memory>");
134             p.println(indent + "<threads>");
135             p.println(newIndent + "      delay=" + delay + " ms");
136             p.println(newIndent + "    threads=" + Thread.activeCount());
137             showThreadGroup(Thread.currentThread().getThreadGroup(), p, newIndent);
138             p.println(indent + "</threads>");
139             NameRegistrar.getInstance().dump(p, indent, detailRequired);
140         }
141     
142     }
