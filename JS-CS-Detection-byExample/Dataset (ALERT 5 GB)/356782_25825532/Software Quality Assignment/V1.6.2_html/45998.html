25     /**
26      * EBCDIC [unpacked] Bitmap
27      *
28      * @author apr
29      * @see ISOComponent
30      * @see ISOBitMapPackager
31      */
32     public class IFE_BITMAP extends ISOBitMapPackager {
33         public IFE_BITMAP() {
34             super();
35         }
36         /**
37          * @param len - field len
38          * @param description symbolic descrption
39          */
40         public IFE_BITMAP(int len, String description) {
41             super(len, description);
42         }
43         /**
44          * @param c - a component
45          * @return packed component
46          * @exception ISOException
47          */
48         public byte[] pack (ISOComponent c) throws ISOException {
49         	BitSet bitMapValue = (BitSet) c.getValue();
50         	
51         	int maxBytesPossible = getLength();
52         	int maxBitsAllowedPhysically = maxBytesPossible<<3;
53         	
54         	int lastBitOn = bitMapValue.length()-1;    
55                                 
56             int actualLastBit=lastBitOn; // takes into consideration 2nd and 3rd bit map flags
57             if (lastBitOn > 128)
58             	if (bitMapValue.get(65))
59             		actualLastBit = 192;
60             	else actualLastBit = 128;
61             else if (lastBitOn > 64)
62             	if (bitMapValue.get(1))
63             		actualLastBit = 128;
64             	else actualLastBit = 64;
65             
66                      
67            	if (actualLastBit > maxBitsAllowedPhysically)
68         			{
69         				throw new ISOException ("Bitmap can only hold bits numbered up to " + maxBitsAllowedPhysically + " in the " + 
70         						getLength() + " bytes available.");
71         			}
72             
73            	int requiredLengthInBytes = (actualLastBit >> 3) + (actualLastBit % 8 > 0 ? 1 : 0);
74            	
75            	int requiredBitMapLengthInBytes;
76            	if (requiredLengthInBytes>4 && requiredLengthInBytes<=8) requiredBitMapLengthInBytes = 8;
77            	else if (requiredLengthInBytes>8 && requiredLengthInBytes<=16) requiredBitMapLengthInBytes = 16;
78            	else if (requiredLengthInBytes>16 && requiredLengthInBytes<=24) requiredBitMapLengthInBytes = 24;
79            	else requiredBitMapLengthInBytes=maxBytesPossible;
80            		     	
81             byte[] b = ISOUtil.bitSet2byte (bitMapValue, requiredBitMapLengthInBytes);
82             return ISOUtil.asciiToEbcdic(ISOUtil.hexString(b).getBytes());
83             
84         }
85         public int getMaxPackedLength() {
86             return getLength() >> 2;
87         }
88         /**
89          * @param c - the Component to unpack
90          * @param b - binary image
91          * @param offset - starting offset within the binary image
92          * @return consumed bytes
93          * @exception ISOException
94          */
95         public int unpack (ISOComponent c, byte[] b, int offset)
96             throws ISOException
97         {
98     //    	TODO: calculate bytes to read based on bits 1, 65 on/off in the actual data 
99         	int bytes;
100         	byte [] b1 = ISOUtil.ebcdicToAsciiBytes (b, offset, getLength()*2 );
101         	BitSet bmap = ISOUtil.hex2BitSet (b1, 0, getLength() << 3);
102             c.setValue(bmap);
103             bytes = b1.length;;
104             // check for 2nd bit map indicator
105             if ((bytes > 16) && bmap.get(1)==false) { 
106               bytes = 16; 
107             // check for 3rd bit map indicator
108             } else if ((bytes > 32) && bmap.get(65)==false) { 
109               bytes = 32; 
110             } 
111             return bytes;
112         }
113         public void unpack (ISOComponent c, InputStream in) 
114             throws IOException, ISOException
115         {
116         	byte [] b1 = ISOUtil.ebcdicToAsciiBytes (readBytes (in, 16), 0, 16);
117             BitSet bmap = ISOUtil.hex2BitSet (new BitSet (64), b1, 0);
118             if (getLength() > 8 && bmap.get (1)) {
119             	byte [] b2 = ISOUtil.ebcdicToAsciiBytes (readBytes (in, 16), 0, 16);        	
120                 ISOUtil.hex2BitSet (bmap, b2, 64);
121             }
122             c.setValue(bmap);
123         }
