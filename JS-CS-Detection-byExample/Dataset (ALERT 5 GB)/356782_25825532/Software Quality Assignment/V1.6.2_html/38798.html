45     /**
46      * packs/unpacks ISOMsgs from jPOS logs
47      *
48      * @author apr@cs.com.uy
49      * @version $Id$
50      * @see ISOPackager
51      */
52     public class LogPackager extends DefaultHandler
53                              implements ISOPackager, LogSource
54     {
55         protected Logger logger = null;
56         protected String realm = null;
57         private ByteArrayOutputStream out;
58         private PrintStream p;
59         private XMLReader reader = null;
60         private Stack stk;
61     
62         public static final String LOG_TAG       = "log";
63         public static final String ISOMSG_TAG    = "isomsg";
64         public static final String ISOFIELD_TAG  = "field";
65         public static final String ID_ATTR       = "id";
66         public static final String VALUE_ATTR    = "value";
67         public static final String TYPE_ATTR     = "type";
68         public static final String TYPE_BINARY   = "binary";
69         public static final String TYPE_BITMAP   = "bitmap";
70     
71         public LogPackager() throws ISOException {
72             super();
73             out = new ByteArrayOutputStream();
74             p   = new PrintStream(out);
75             stk = new Stack();
76             try {
77                 reader = XMLReaderFactory.createXMLReader(
78                     System.getProperty( "sax.parser",
79                                         "org.apache.crimson.parser.XMLReaderImpl")
80                 );
81                 reader.setFeature ("http://xml.org/sax/features/validation",false);
82                 reader.setContentHandler(this);
83                 reader.setErrorHandler(this);
84             } catch (Exception e) {
85                 throw new ISOException (e.toString());
86             }
87         }
88         public byte[] pack (ISOComponent c) throws ISOException {
89             LogEvent evt = new LogEvent (this, "pack");
90             try {
91                 if (!(c instanceof ISOMsg))
92                     throw new ISOException ("cannot pack "+c.getClass());
93                 ISOMsg m = (ISOMsg) c;
94                 byte[] b;
95                 synchronized (this) {
96                     p.println ("<log>");
97                     c.dump (p, " ");
98                     p.println ("</log>");
99                     b = out.toByteArray();
100                     out.reset();
101                 }
102                 if (logger != null)
103                     evt.addMessage (m);
104                 return b;
105             } catch (ISOException e) {
106                 evt.addMessage (e);
107                 throw e;
108             } finally {
109                 Logger.log(evt);
110             }
111         }
112     
113         public synchronized int unpack (ISOComponent c, byte[] b) 
114             throws ISOException
115         {
116             LogEvent evt = new LogEvent (this, "unpack");
117             try {
118                 if (!(c instanceof ISOMsg))
119                     throw new ISOException 
120                         ("Can't call packager on non Composite");
121     
122                 while (!stk.empty())    // purge from possible previous error
123                     stk.pop();
124     
125                 InputSource src = new InputSource (new ByteArrayInputStream(b));
126                 reader.parse (src);
127                 if (!stk.empty()) {
128                     ISOMsg m = (ISOMsg) c;
129                     m.merge ((ISOMsg) stk.pop());
130                     if (logger != null)     
131                         evt.addMessage (m);
132                 }
133             } catch (ISOException e) {
134                 evt.addMessage (e);
135                 // throw e;
136             } catch (IOException e) {
137                 evt.addMessage (e);
138                 // throw new ISOException (e.toString());
139             } catch (SAXException e) {
140                 evt.addMessage (e);
141                 // throw new ISOException (e.toString());
142             } finally {
143                 Logger.log (evt);
144             }
145             return b.length;
146         }
147     
148         public synchronized void unpack (ISOComponent c, InputStream in) 
149             throws ISOException, IOException
150         {
151             LogEvent evt = new LogEvent (this, "unpack");
152             try {
153                 if (!(c instanceof ISOMsg))
154                     throw new ISOException 
155                         ("Can't call packager on non Composite");
156     
157                 while (!stk.empty())    // purge from possible previous error
158                     stk.pop();
159     
160                 reader.parse (new InputSource (in));
161                 if (!stk.empty()) {
162                     ISOMsg m = (ISOMsg) c;
163                     m.merge ((ISOMsg) stk.pop());
164                     if (logger != null)     
165                         evt.addMessage (m);
166                 }
167             } catch (ISOException e) {
168                 evt.addMessage (e);
169                 // throw e;
170             } catch (IOException e) {
171                 evt.addMessage (e);
172                 // throw new ISOException (e.toString());
173             } catch (SAXException e) {
174                 evt.addMessage (e);
175                 // throw new ISOException (e.toString());
176             } finally {
177                 Logger.log (evt);
178             }
179         }
180     
181         public void startElement 
182             (String ns, String name, String qName, Attributes atts)
183             throws SAXException
184         {
185             int fieldNumber = -1;
186             try {
187                 String id       = atts.getValue(ID_ATTR);
188                 if (id != null) {
189                     try {
190                         fieldNumber = Integer.parseInt (id);
191                     } catch (NumberFormatException ex) { }
192                 }
193                 if (name.equals (ISOMSG_TAG)) {
194                     if (fieldNumber >= 0) {
195                         if (stk.empty())
196                             throw new SAXException ("inner without outter");
197     
198                         ISOMsg inner = new ISOMsg(fieldNumber);
199                         ((ISOMsg)stk.peek()).set (inner);
200                         stk.push (inner);
201                     } else {
202                         stk.push (new ISOMsg(0));
203                     }
204                 } else if (name.equals (ISOFIELD_TAG)) {
205                     ISOMsg m     = (ISOMsg) stk.peek();
206                     String value = atts.getValue(VALUE_ATTR);
207                     String type  = atts.getValue(TYPE_ATTR);
208                     if (id == null || value == null)
209                         throw new SAXException ("invalid field");   
210                     if (TYPE_BINARY.equals (type)) {
211                         m.set (new ISOBinaryField (
212                             fieldNumber, 
213                                 ISOUtil.hex2byte (
214                                     value.getBytes(), 0, value.length()/2
215                                 )
216                             )
217                         );
218                     }
219                     else {
220                         m.set (new ISOField (fieldNumber, value));
221                     }
222                 }
223             } catch (ISOException e) {
224                 throw new SAXException 
225                     ("ISOException unpacking "+fieldNumber);
226             }
227         }
228     
229         public void endElement (String ns, String name, String qname) 
230             throws SAXException
231         {
232             if (name.equals (ISOMSG_TAG)) {
233                 ISOMsg m = (ISOMsg) stk.pop();
234                 if (stk.empty())
235                     stk.push (m); // push outter message
236             }
237         }

239         public String getFieldDescription(ISOComponent m, int fldNumber) {
240             return "<notavailable/>";
241         }
242         public void setLogger (Logger logger, String realm) {
243             this.logger = logger;
244             this.realm  = realm;
245         }
246         public String getRealm () {
247             return realm;
248         }
249         public Logger getLogger() {
250             return logger;
251         }
252         public ISOMsg createISOMsg() {
253             return new ISOMsg();
254         }
255     }
