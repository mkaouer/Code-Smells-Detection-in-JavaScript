19     package org.jpos.util;
20     
21     import java.io.File;
22     import java.io.FileOutputStream;
23     import java.io.IOException;
24     import java.io.PrintStream;
25     import java.util.Date;
26     import java.util.Timer;
27     import java.util.TimerTask;
28     
29     import org.jpos.core.Configurable;
30     import org.jpos.core.Configuration;
31     import org.jpos.core.ConfigurationException;
32     
33     /**
34      * Rotates logs
35      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
36      * @version $Revision$ $Date$
37      * @see org.jpos.core.Configurable
38      * @since jPOS 1.2
39      */
40     public class RotateLogListener extends SimpleLogListener 
41         implements Configurable, Destroyable
42     {
43         FileOutputStream f;
44         String logName;
45         int maxCopies;
46         long sleepTime;
47         long maxSize;
48         int  msgCount;
49         Rotate rotate;
50         public static final int CHECK_INTERVAL = 100;
51         public static final long DEFAULT_MAXSIZE = 10000000;
52     
53         /**
54          * @param name base log filename
55          * @param sleepTime switch logs every t seconds
56          * @param maxCopies number of old logs
57          * @param maxSize in bytes 
58          */
59     
60         public RotateLogListener 
61             (String logName, int sleepTime, int maxCopies, long maxSize) 
62             throws IOException
63         {
64             super();
65             this.logName   = logName;
66             this.maxCopies = maxCopies;
67             this.sleepTime = sleepTime * 1000;
68             this.maxSize   = maxSize;
69             f = null;
70             openLogFile ();
71             Timer timer = DefaultTimer.getTimer();
72             if (sleepTime != 0) {
73                 timer.schedule (rotate = new Rotate(), 
74                         this.sleepTime, this.sleepTime);
75             }
76         }
77     
78         public RotateLogListener 
79             (String logName, int sleepTime, int maxCopies) 
80             throws IOException
81         {
82             this (logName, sleepTime, maxCopies, DEFAULT_MAXSIZE); 
83         }
84     
85         public RotateLogListener () {
86             super();
87         }
88     
89        /**
90         * Configure this RotateLogListener<br>
91         * Properties:<br>
92         * <ul>
93         *  <li>file      base log filename
94         *  <li>[window]  in seconds (default 0 - never rotate)
95         *  <li>[count]   number of copies (default 0 == single copy)
96         *  <li>[maxsize] max log size in bytes (aprox)
97         * </ul>
98         * @param cfg Configuration 
99         * @throws ConfigurationException
100         */
101         public void setConfiguration (Configuration cfg)
102             throws ConfigurationException
103         {
104             maxCopies = cfg.getInt  ("copies");
105             sleepTime = cfg.getInt  ("window") * 1000;
106             logName   = cfg.get     ("file");
107             maxSize   = cfg.getLong ("maxsize");
108             maxSize   = maxSize <= 0 ? DEFAULT_MAXSIZE : maxSize;
109     
110             try {
111                 openLogFile();
112             } catch (IOException e) {
113                 throw new ConfigurationException (e);
114             }
115             Timer timer = DefaultTimer.getTimer();
116             if (sleepTime != 0) 
117                 timer.schedule (rotate = new Rotate(), sleepTime, sleepTime);
118         }
119         public synchronized LogEvent log (LogEvent ev) {
120             if (msgCount++ > CHECK_INTERVAL) {
121                 checkSize();
122                 msgCount = 0;
123             }
124             
125             return super.log (ev);
126         }
127         protected synchronized void openLogFile() throws IOException {
128             if (f != null)
129                 f.close();
130             f = new FileOutputStream (logName, true);
131             setPrintStream (new PrintStream(f));
132         }
133         protected synchronized void closeLogFile() throws IOException {
134             if (f != null)
135                 f.close();
136             f = null;
137         }
138         public synchronized void logRotate ()
139             throws IOException
140         {
141             super.close ();
142             setPrintStream (null);
143             closeLogFile ();
144             for (int i=maxCopies; i>0; ) {
145                 File dest   = new File (logName + "." + i);
146                 File source = new File (logName + ((--i > 0) ? ("." + i) : ""));
147                 dest.delete();
148                 source.renameTo(dest);
149             }
150             openLogFile();
151         }
152         protected synchronized void logDebug (String msg) {
153             if (p != null) {
154                 p.println ("<log realm=\"rotate-log-listener\" at=\""+new Date().toString() +"\">");
155                 p.println ("   "+msg);
156                 p.println ("</log>");
157             }
158         }
159         protected void checkSize() {
160             File logFile = new File (logName);
161             if (logFile.length() > maxSize) {
162                 try {
163                     logDebug ("maxSize ("+maxSize+") threshold reached");
164                     logRotate();
165                 } catch (IOException e) {
166                     e.printStackTrace (System.err);
167                 }
168             }
169         }
170         public class Rotate extends TimerTask {
171             public void run() {
172                 try {
173                     logDebug ("time exceeded - log rotated");
174                     logRotate();
175                 } catch (IOException e) {
176                     e.printStackTrace (System.err);
177                 }
178             }
179         }
180         public void destroy () {
181             if (rotate != null)
182                 rotate.cancel ();
183             try {
184                 closeLogFile ();
185             } catch (IOException e) {
186                 // nothing we can do.
187             }
188         }
189     }
