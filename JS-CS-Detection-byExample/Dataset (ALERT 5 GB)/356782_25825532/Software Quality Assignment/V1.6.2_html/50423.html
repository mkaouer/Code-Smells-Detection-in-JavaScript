40     public class Context implements Externalizable, Loggeable, Pausable {
41         private transient Map map; // transient map
42         private Map pmap;          // persistent (serializable) map
43         private long timeout;
44         private int resuming=0;
45         private boolean resumeOnPause = false;
46     
47         public static String LOGEVT = "LOGEVT";
48         public static String PROFILER = "PROFILER";
49         public static String PAUSED_TRANSACTION = ":paused_transaction";
50     
51         public Context () {
52             super ();
53         }
54         /**
55          * puts an Object in the transient Map
56          */
57         public void put (Object key, Object value) {
58             getMap().put (key, value);
59         }
60         /**
61          * puts an Object in the transient Map
62          */
63         public void put (Object key, Object value, boolean persist) {
64             getMap().put (key, value);
65             if (persist && value instanceof Serializable)
66                 getPMap().put (key, value);
67         }
68         /**
69          * Get
70          */
71         public Object get (Object key) {
72             return getMap().get (key);
73         }
74         public Object get (Object key, Object defValue) {
75             Object obj = getMap().get (key);
76             return obj != null ? obj : defValue;
77         }
78         /**
79          * Transient remove
80          */
81         public synchronized Object remove (Object key) {
82             getPMap().remove (key);
83             return getMap().remove (key);
84         }
85         public String getString (Object key) {
86             return (String) getMap().get (key);
87         }
88         public String getString (Object key, Object defValue) {
89             return (String) get (key, defValue);
90         }
91         public void dump (PrintStream p, String indent) {
92             String inner = indent + "  ";
93             p.println (indent + "<context>");
94             dumpMap (p, inner);
95             p.println (indent + "</context>");
96         }
97         /**
98          * persistent get with timeout
99          * @param key the key
100          * @param timeout timeout
101          * @return object (null on timeout)
102          */
103         public synchronized Object get (Object key, long timeout) {
104             Object obj;
105             long now = System.currentTimeMillis();
106             long end = now + timeout;
107             while ((obj = map.get (key)) == null && 
108                     ((now = System.currentTimeMillis()) < end))
109             {
110                 try {
111                     this.wait (end - now);
112                 } catch (InterruptedException e) { }
113             }
114             return obj;
115         }
116         public void writeExternal (ObjectOutput out) throws IOException {
117             out.writeByte (0);  // reserved for future expansion (version id)
118             Set s = getPMap().entrySet();
119             out.writeInt (s.size());
120             Iterator iter = s.iterator();
121             while (iter.hasNext()) {
122                 Map.Entry entry = (Map.Entry) iter.next();
123                 out.writeObject(entry.getKey());
124                 out.writeObject(entry.getValue());
125             }
126         }
127         public void readExternal  (ObjectInput in) 
128             throws IOException, ClassNotFoundException
129         {
130             in.readByte();  // ignore version for now
131             getMap();       // force creation of map
132             getPMap();      // and pmap
133             int size = in.readInt();
134             for (int i=0; i<size; i++) {
135                 Object k = in.readObject();
136                 Object v = in.readObject();
137                 map.put (k, v);
138                 pmap.put (k, v);
139             }
140         }
141         /**
142          * @return persistent map
143          */
144         private synchronized Map getPMap() {
145             if (pmap == null)
146                 pmap = Collections.synchronizedMap (new LinkedHashMap ());
147             return pmap;
148         }
149         /**
150          * @return transient map
151          */
152         public synchronized Map getMap() {
153             if (map == null)
154                 map = Collections.synchronizedMap (new LinkedHashMap ());
155             return map;
156         }
157         private void dumpMap (PrintStream p, String indent) {
158             if (map == null)
159                 return;
160     
161             Iterator iter = map.entrySet().iterator();
162             while (iter.hasNext()) {
163                 Map.Entry entry = (Map.Entry) iter.next ();
164                 if (pmap != null && pmap.containsKey(entry.getKey())) 
165                     p.print (indent + "<entry key='" + entry.getKey().toString() + "' p='true'>"); 
166                 else
167                     p.print (indent + "<entry key='" + entry.getKey().toString() + "'>"); 
168                 Object value = entry.getValue();
169                 if (value instanceof Loggeable) {
170                     p.println ("");
171                     ((Loggeable) value).dump (p, indent + " ");
172                     p.print (indent);
173                 } else if (value instanceof Element) {
174                     p.println ("");
175                     p.println (indent+ "<![CDATA[");
176                     XMLOutputter out = new XMLOutputter (Format.getPrettyFormat ());
177                     out.getFormat().setLineSeparator ("\n");
178                     try {
179                         out.output ((Element) value, p);
180                     } catch (IOException ex) {
181                         ex.printStackTrace (p);
182                     }
183                     p.println ("");
184                     p.println (indent + "]]>");
185                 } else if (value instanceof byte[]) {
186                     byte[] b = (byte[]) value;
187                     p.println ("");
188                     p.println (ISOUtil.hexdump (b));
189                 } else if (value != null) {
190                     try {
191                         p.print (value.toString ());
192                     } catch (Exception e) {
193                         p.println (e.getMessage());
194                     }
195                 } else {
196                     p.print ("nil");
197                 }
198                 p.println ("</entry>");
199             }
200         }
201         /**
202          * return a LogEvent used to store trace information
203          * about this transaction.
204          * If there's no LogEvent there, it creates one.
205          * @return LogEvent
206          */
207         public LogEvent getLogEvent () {
208             LogEvent evt = (LogEvent) get (LOGEVT);
209             if (evt == null) {
210                 synchronized (this) {
211                     evt = (LogEvent) get (LOGEVT);
212                     if (evt == null) {
213                         evt = new LogEvent ("log");
214                         put (LOGEVT, evt);
215                     }
216                 }
217             }
218             return evt;
219         }
220         /**
221          * return (or creates) a Profiler object
222          * @return Profiler object
223          */
224         public Profiler getProfiler () {
225             Profiler prof = (Profiler) get (PROFILER);
226             if (prof == null) {
227                 synchronized (this) {
228                     prof = (Profiler) get (PROFILER);
229                     if (prof == null) {
230                         prof = new Profiler();
231                         put (PROFILER, prof);
232                     }
233                 }
234             }
235             return prof;
236         }
237         /**
238          * adds a trace message
239          * @msg trace information
240          */
241         public void log (Object msg) {
242             getLogEvent().addMessage (msg);
243         }
244         /**
245          * add a checkpoint to the profiler
246          */
247         public void checkPoint (String detail) {
248             getProfiler().checkPoint (detail);
249         }
250         public void setPausedTransaction (PausedTransaction p) {
251             put (PAUSED_TRANSACTION, p);
252             synchronized (this) {
253                 if (resumeOnPause) {
254                     resume();
255                 }
256             }
257         }
258         public PausedTransaction getPausedTransaction() {
259             return (PausedTransaction) get (PAUSED_TRANSACTION);
260     
261         }
262         public void setTimeout (long timeout) {
263             this.timeout = timeout;
264         }
265         public long getTimeout () {
266             return timeout;
267         }
268         public synchronized void resume() {
269             PausedTransaction pt = getPausedTransaction();
270             if (pt != null && !pt.isResumed()) {
271                 pt.setResumed (true);
272                 pt.getTransactionManager().push (this);
273             } else {
274                 resumeOnPause = true;
275             }
276         }
277         static final long serialVersionUID = 6056487212221438338L;
278     }
