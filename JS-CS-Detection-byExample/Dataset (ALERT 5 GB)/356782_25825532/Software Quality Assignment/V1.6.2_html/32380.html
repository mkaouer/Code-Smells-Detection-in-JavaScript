19     package org.jpos.iso;
20     
21     import junit.framework.TestCase;
22     
23     /**
24      * @author joconnor
25      */
26     public class IFA_LLLLLCHARTest extends TestCase
27     {
28         public void testPack() throws Exception
29         {
30             ISOField field = new ISOField(12, "ABCD");
31             IFA_LLLLLCHAR packager = new IFA_LLLLLCHAR(10, "Should be 00004ABCD");
32             TestUtils.assertEquals("00004ABCD".getBytes(), packager.pack(field));
33         }
34     
35         public void testUnpack() throws Exception
36         {
37             byte[] raw = "00004ABCD".getBytes();
38             IFA_LLLLLCHAR packager = new IFA_LLLLLCHAR(10, "Should be 00004ABCD");
39             ISOField field = new ISOField(12);
40             packager.unpack(field, raw, 0);
41             assertEquals("ABCD", (String) field.getValue());
42         }
43     
44         public void testReversability() throws Exception
45         {
46             String origin = "Abc123:.-";
47             ISOField f = new ISOField(12, origin);
48             IFA_LLLLLCHAR packager = new IFA_LLLLLCHAR(10, "Should be Abc123:.-");
49     
50             ISOField unpack = new ISOField(12);
51             packager.unpack(unpack, packager.pack(f), 0);
52             assertEquals(origin, (String) unpack.getValue());
53         }
54     }
