34     /**
35      * Base Packager class envolving validators. It implements
36      * ISOValidator interface and define an implementation for validate method.
37      * Validation is for composed components.
38      * <p>Title: jPOS</p>

45     public class ISOBaseValidatingPackager extends ISOBasePackager implements ISOValidator {
46     
47         public ISOBaseValidatingPackager() {
48             super();
49         }
50     
51         public ISOComponent validate(ISOComponent m) throws ISOException {
52             LogEvent evt = new LogEvent( this, "validate" );
53             try {
54                 ISOComponent c;
55                 Hashtable fields = ((ISOMsg)m).getChildren();
56                 /** Field  validations **/
57                 for (int i=0; i<fldVld.length; i++) {
58                     if ( fldVld[i] != null && (c=(ISOComponent) fields.get (new Integer ( ((ISOFieldValidator)fldVld[i]).getFieldId() ))) != null ){
59                         try {
60                             m.set( fldVld[i].validate( c ) );
61                         } catch ( ISOVException e ) {
62                             if ( !e.treated() ) {
63                                 m.set( e.getErrComponent() );
64                                 e.setTreated( true );
65                             }
66                             evt.addMessage( "Component Validation Error." );
67                             throw e;
68                         }
69                     }
70                 }
71                 /** msg validations **/
72                 try {
73                     if ( msgVld != null ){
74                         for (int i = 0; i < this.msgVld.length; i++) {
75                             if ( msgVld[i] != null )
76                                 m = msgVld[i].validate( m );
77                         }
78                     }
79                 }
80                 catch (ISOVException ex) {
81                     evt.addMessage( "Component Validation Error." );
82                     throw ex;
83                 }
84                 return m;
85             }
86             finally {
87                 Logger.log( evt );
88             }
89         }
