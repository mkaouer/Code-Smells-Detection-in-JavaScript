19     package org.jpos.iso;
20     
21     import java.io.IOException;
22     import java.io.InputStream;
23     
24     /**
25      * @author joconnor
26      * @version $Revision$ $Date$
27      */
28     public class ISOFilledStringFieldPackager extends ISOFieldPackager
29     {
30         private Interpreter interpreter;
31         private Padder padder;
32         private Prefixer prefixer;
33     
34         /**
35          * Constructs a default ISOFilledStringFieldPackager. There is no padding,
36          * no length prefix and a literal interpretation. The set methods must be called to
37          * make this useful.
38          */
39         public ISOFilledStringFieldPackager()
40         {
41             super();
42             this.padder = NullPadder.INSTANCE;
43             this.interpreter = LiteralInterpreter.INSTANCE;
44             this.prefixer = NullPrefixer.INSTANCE;
45         }
46     
47         /**
48          * Constructs an ISOFilledStringFieldPackager with a specific Padder, Interpreter and Prefixer.
49          * The length and description should be set with setLength() and setDescription methods.
50          * @param padder The type of padding used.
51          * @param interpreter The interpreter used to encode the field.
52          * @param prefixer The type of length prefixer used to encode this field.
53          */
54         public ISOFilledStringFieldPackager(Padder padder, Interpreter interpreter, Prefixer prefixer)
55         {
56             super();
57             this.padder = padder;
58             this.interpreter = interpreter;
59             this.prefixer = prefixer;
60         }
61     
62         /**
63          * Creates an ISOFilledStringFieldPackager.
64          * @param maxLength The maximum length of the field in characters or bytes depending on the datatype.
65          * @param description The description of the field. For human readable output.
66          * @param interpreter The interpreter used to encode the field.
67          * @param padder The type of padding used.
68          * @param prefixer The type of length prefixer used to encode this field.
69          */
70         public ISOFilledStringFieldPackager(int maxLength, String description, Padder padder,
71                                       Interpreter interpreter, Prefixer prefixer)
72         {
73             super(maxLength, description);
74             this.padder = padder;
75             this.interpreter = interpreter;
76             this.prefixer = prefixer;
77         }
78     
79         /**
80          * Sets the Padder.
81          * @param padder The padder to use during packing and unpacking.
82          */
83         public void setPadder(Padder padder)
84         {
85             this.padder = padder;
86         }
87     
88         /**
89          * Sets the Interpreter.
90          * @param interpreter The interpreter to use in packing and unpacking.
91          */
92         public void setInterpreter(Interpreter interpreter)
93         {
94             this.interpreter = interpreter;
95         }
96     
97         /**
98          * Sets the length prefixer.
99          * @param prefixer The length prefixer to use during packing and unpacking.
100          */
101         public void setPrefixer(Prefixer prefixer)
102         {
103             this.prefixer = prefixer;
104         }
105     
106         /**
107          * Returns the prefixer's packed length and the interpreter's packed length.
108     	 * @see org.jpos.iso.ISOFieldPackager#getMaxPackedLength()
109     	 */
110         public int getMaxPackedLength()
111         {
112             return prefixer.getPackedLength() + interpreter.getPackedLength(getLength());
113         }
114     
115         /** Create a nice readable message for errors */
116         private String makeExceptionMessage(ISOComponent c, String operation) {
117             Object fieldKey = "unknown";
118             if (c != null)
119             {
120                 try
121                 {
122                     fieldKey = c.getKey();
123                 } catch (Exception ignore)
124                 {
125                 }
126             }
127             return this.getClass().getName() + ": Problem " + operation + " field " + fieldKey;
128         }
129     
130         /**
131     	 * Convert the component into a byte[].
132     	 */
133         public byte[] pack(ISOComponent c) throws ISOException
134         {
135             try
136             {
137                 String data = (String)c.getValue();
138                 if (data.length() > getLength())
139                 {
140                     throw new ISOException("Field length " + data.length() + " too long. Max: " + getLength());
141                 }
142                 String paddedData = padder.pad(data, getLength());
143                 byte[] rawData = new byte[prefixer.getPackedLength()
144                         + interpreter.getPackedLength(paddedData.length())];
145                 prefixer.encodeLength(data.length(), rawData);
146                 interpreter.interpret(paddedData, rawData, prefixer.getPackedLength());
147                 return rawData;
148             } catch(Exception e)
149             {
150                 throw new ISOException(makeExceptionMessage(c, "packing"), e);
151             }
152         }
153     
154         /**
155          * Unpacks the byte array into the component.
156          * @param c The component to unpack into.
157          * @param b The byte array to unpack.
158          * @param offset The index in the byte array to start unpacking from.
159          * @return The number of bytes consumed unpacking the component.
160          */
161         public int unpack(ISOComponent c, byte[] b, int offset) throws ISOException
162         {
163             try
164             {
165                 int len = prefixer.decodeLength(b, offset);
166                 if (len == -1)
167                 {
168                     // The prefixer doesn't know how long the field is, so use
169         			// maxLength instead
170                     len = getLength();
171                 }
172                 int lenLen = prefixer.getPackedLength();
173                 String unpacked = interpreter.uninterpret(b, offset + lenLen, len);
174                 c.setValue(unpacked);
175                 return lenLen + interpreter.getPackedLength(getLength());
176             } catch(Exception e)
177             {
178                 throw new ISOException(makeExceptionMessage(c, "unpacking"), e);
179             }
180         }
181     
182         /**
183          * Unpack the input stream into the component.
184          * @param c  The Component to unpack into.
185          * @param in Input stream where the packed bytes come from.
186          * @exception IOException Thrown if there's a problem reading the input stream.
187          */
188         public void unpack (ISOComponent c, InputStream in) 
189             throws IOException, ISOException
190         {
191             try
192             {
193                 int lenLen = prefixer.getPackedLength ();
194                 int len;
195                 if (lenLen == 0)
196                 {
197                     len = getLength();
198                 } else
199                 {
200                     len = prefixer.decodeLength (readBytes (in, lenLen), 0);
201                 }
202                 int packedLen = interpreter.getPackedLength(len);
203                 String unpacked = interpreter.uninterpret(readBytes (in, packedLen), 0, len);
204                 c.setValue(unpacked);
205                 in.skip(interpreter.getPackedLength(getLength()) - packedLen);
206             } catch(ISOException e)
207             {
208                 throw new ISOException(makeExceptionMessage(c, "unpacking"), e);
209             }
210         }
211     
212         /**
213          * Checks the length of the data against the maximum, and throws an IllegalArgumentException.
214          * This is designed to be called from field Packager constructors and the setLength()
215          * method.
216          * @param len The length of the data for this field packager.
217          * @param maxLength The maximum length allowed for this type of field packager.
218          *          This depends on the prefixer that is used.
219          * @throws IllegalArgumentException If len > maxLength.
220          */
221         protected void checkLength(int len, int maxLength) throws IllegalArgumentException
222         {
223             if (len > maxLength)
224             {
225                 throw new IllegalArgumentException("Length " + len + " too long for " + getClass().getName());
226             }
227         }
228     }
