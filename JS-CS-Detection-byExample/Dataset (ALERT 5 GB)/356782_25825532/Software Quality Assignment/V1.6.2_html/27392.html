19     package org.jpos.transaction;
20     
21     import java.io.Serializable;
22     
23     /**
24      * 2 phase commit participant
25      * @author apr
26      * @since 1.4.7
27      */
28     public interface TransactionParticipant extends TransactionConstants {
29         /**
30          * Called by TransactionManager in preparation for a transaction
31          * @param id the Transaction identifier
32          * @param context transaction context
33          * @return PREPARED or ABORTED (| NO_JOIN | READONLY)
34          */
35         public int  prepare (long id, Serializable context);
36     
37         /**
38          * Called by TransactionManager upon transaction commit.
39          * Warning: implementation should be able to handle multiple calls
40          * with the same transaction id (rare crash recovery)
41          *
42          * @param id the Transaction identifier
43          * @param context transaction context
44          */
45         public void commit  (long id, Serializable context);
46     
47         /**
48          * Called by TransactionManager upon transaction commit.
49          * Warning: implementation should be able to handle multiple calls
50          * with the same transaction id (rare crash recovery)
51          *
52          * @param id the Transaction identifier
53          * @param context transaction context
54          */
55         public void abort   (long id, Serializable context);
56     }
