37     /**
38      * packs/unpacks ISOMsgs into XML representation
39      *
40      * @author apr@cs.com.uy
41      * @version $Id: XMLPackager.java 2594 2008-01-22 16:41:31Z apr $
42      * @see ISOPackager
43      */
44     public class XML2003Packager extends DefaultHandler
45                              implements ISOPackager, LogSource
46     {
47         protected Logger logger = null;
48         protected String realm = null;
49         private ByteArrayOutputStream out;
50         private PrintStream p;
51         private XMLReader reader = null;
52         private Stack stk;
53     
54         public static final String ISOMSG_TAG    = "isomsg";
55         public static final String ISOFIELD_TAG  = "field";
56         public static final String ID_ATTR       = "id";
57         public static final String VALUE_ATTR    = "value";
58         public static final String TYPE_ATTR     = "type";
59         public static final String TYPE_BINARY   = "binary";
60         public static final String TYPE_BITMAP   = "bitmap";
61         public static final String TYPE_AMOUNT   = "amount";
62         public static final String CURRENCY_ATTR = "currency";
63         public static final String HEADER_TAG    = "header";
64         public static final int[] BINARY_FIELDS  = new int[] { 72 };
65     
66         public XML2003Packager() throws ISOException {
67             super();
68             out = new ByteArrayOutputStream();
69             p   = new PrintStream(out);
70             stk = new Stack();
71             try {
72                 reader = createXMLReader();
73             } catch (Exception e) {
74                 throw new ISOException (e.toString());
75             }
76         }
77         public byte[] pack (ISOComponent c) throws ISOException {
78             LogEvent evt = new LogEvent (this, "pack");
79             try {
80                 if (!(c instanceof ISOMsg))
81                     throw new ISOException ("cannot pack "+c.getClass());
82                 ISOMsg m = (ISOMsg) c;
83                 byte[] b;
84                 synchronized (this) {
85                     m.setDirection(0);  // avoid "direction=xxxxxx" in XML msg
86                     m.dump (p, "");
87                     b = out.toByteArray();
88                     out.reset();
89                 }
90                 if (logger != null)
91                     evt.addMessage (m);
92                 return b;
93             } catch (ISOException e) {
94                 evt.addMessage (e);
95                 throw e;
96             } finally {
97                 Logger.log(evt);
98             }
99         }
100     
101         public synchronized int unpack (ISOComponent c, byte[] b) 
102             throws ISOException
103         {
104             LogEvent evt = new LogEvent (this, "unpack");
105             try {
106                 if (!(c instanceof ISOMsg))
107                     throw new ISOException 
108                         ("Can't call packager on non Composite");
109     
110                 while (!stk.empty())    // purge from possible previous error
111                     stk.pop();
112     
113                 InputSource src = new InputSource (new ByteArrayInputStream(b));
114                 reader.parse (src);
115                 if (stk.empty())
116                     throw new ISOException ("error parsing");
117     
118                 ISOMsg m = (ISOMsg) c;
119                 ISOMsg m1 = (ISOMsg) stk.pop();
120                 m.merge (m1);
121                 m.setHeader (m1.getHeader());
122                 fixup (m, BINARY_FIELDS);
123     
124                 if (logger != null) 
125                     evt.addMessage (m);
126                 return b.length;
127             } catch (ISOException e) {
128                 evt.addMessage (e);
129                 throw e;
130             } catch (IOException e) {
131                 evt.addMessage (e);
132                 throw new ISOException (e.toString());
133             } catch (SAXException e) {
134                 evt.addMessage (e);
135                 throw new ISOException (e.toString());
136             } finally {
137                 Logger.log (evt);
138             }
139         }
140     
141         public synchronized void unpack (ISOComponent c, InputStream in) 
142             throws ISOException, IOException
143         {
144             LogEvent evt = new LogEvent (this, "unpack");
145             try {
146                 if (!(c instanceof ISOMsg))
147                     throw new ISOException 
148                         ("Can't call packager on non Composite");
149     
150                 while (!stk.empty())    // purge from possible previous error
151                     stk.pop();
152     
153                 reader.parse (new InputSource (in));
154                 if (stk.empty())
155                     throw new ISOException ("error parsing");
156     
157                 ISOMsg m = (ISOMsg) c;
158                 m.merge ((ISOMsg) stk.pop());
159                 fixup (m, BINARY_FIELDS);
160     
161                 if (logger != null) 
162                     evt.addMessage (m);
163             } catch (ISOException e) {
164                 evt.addMessage (e);
165                 throw e;
166             } catch (SAXException e) {
167                 evt.addMessage (e);
168                 throw new ISOException (e.toString());
169             } finally {
170                 Logger.log (evt);
171             }
172         }

174         public void startElement 
175             (String ns, String name, String qName, Attributes atts)
176             throws SAXException
177         {
178             int fieldNumber = -1;
179             try {
180                 String id       = atts.getValue(ID_ATTR);
181                 if (id != null) {
182                     try {
183                         fieldNumber = Integer.parseInt (id);
184                     } catch (NumberFormatException ex) { }
185                 }
186                 if (name.equals (ISOMSG_TAG)) {
187                     if (fieldNumber >= 0) {
188                         if (stk.empty())
189                             throw new SAXException ("inner without outter");
190     
191                         ISOMsg inner = new ISOMsg(fieldNumber);
192                         ((ISOMsg)stk.peek()).set (inner);
193                         stk.push (inner);
194                     } else {
195                         stk.push (new ISOMsg(0));
196                     }
197                 } else if (name.equals (ISOFIELD_TAG)) {
198                     ISOMsg m     = (ISOMsg) stk.peek();
199                     String value = atts.getValue(VALUE_ATTR);
200                     String type  = atts.getValue(TYPE_ATTR);
201                     if (id == null || value == null)
202                         throw new SAXException ("invalid field");   
203                     if (TYPE_BINARY.equals (type)) {
204                         m.set (new ISOBinaryField (
205                             fieldNumber, 
206                                 ISOUtil.hex2byte (
207                                     value.getBytes(), 0, value.length()/2
208                                 )
209                             )
210                         );
211                     }
212                     else if (TYPE_AMOUNT.equals (type)) {
213                         ISOAmount amount = new ISOAmount(
214                             fieldNumber,
215                             Integer.parseInt (atts.getValue(CURRENCY_ATTR)),
216                             new BigDecimal (value)
217                         );
218                         m.set (amount);
219                     }
220                     else {
221                         m.set (new ISOField (fieldNumber, value));
222                     }
223                     
224                 } else if (HEADER_TAG.equals (name)) {
225                     stk.push (new BaseHeader());
226                 }
227             } catch (ISOException e) {
228                 throw new SAXException 
229                     ("ISOException unpacking "+fieldNumber);
230             }
231         }
232         public void characters (char ch[], int start, int length) {
233             if (stk.peek() instanceof BaseHeader) {
234                 ((BaseHeader)stk.peek()).unpack (
235                     ISOUtil.hex2byte (new String(ch,start,length))
236                 );
237             }
238         }
239         public void endElement (String ns, String name, String qname) 
240             throws SAXException
241         {
242             if (name.equals (ISOMSG_TAG)) {
243                 ISOMsg m = (ISOMsg) stk.pop();
244                 if (stk.empty())
245                     stk.push (m); // push outter message
246             } else if (HEADER_TAG.equals (name)) {
247                 BaseHeader h = (BaseHeader) stk.pop();
248                 ISOMsg m = (ISOMsg) stk.peek ();
249                 m.setHeader (h);
250             }
251         }

267             return new ISOMsg();
268         }
269         public String getDescription () {
270             return getClass().getName();
271         }    
272         private void fixup (ISOMsg m, int[] bfields) throws ISOException {
273             for (int i=0; i<bfields.length; i++) {
274                 int f = bfields[i];
275                 if (m.hasField(f)) {
276                     ISOComponent c = m.getComponent (f);
277                     if (c instanceof ISOField) 
278                         m.set (f, ((ISOField) c).getBytes());
279                 }
280             }
281         }
282         private XMLReader createXMLReader () throws SAXException {
283             XMLReader reader = null;
284             try {
285                 reader = XMLReaderFactory.createXMLReader();
286             } catch (SAXException e) {
287                 reader = XMLReaderFactory.createXMLReader (
288                     System.getProperty( 
289                         "org.xml.sax.driver", 
290                         "org.apache.crimson.parser.XMLReaderImpl"
291                     )
292                 );
293             }
294             reader.setFeature ("http://xml.org/sax/features/validation",false);
295             reader.setContentHandler(this);
296             reader.setErrorHandler(this);
297             return reader;
298         }
