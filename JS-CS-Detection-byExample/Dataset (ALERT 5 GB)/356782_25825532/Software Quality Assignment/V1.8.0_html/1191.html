19     package org.jpos.q2.iso;
20     
21     import org.jdom.Element;
22     import org.jpos.core.ConfigurationException;
23     import org.jpos.iso.*;
24     import org.jpos.q2.QBeanSupport;
25     import org.jpos.util.NameRegistrar;
26     
27     import java.util.StringTokenizer;
28     
29     /**
30      * @author apr
31      */
32     public class MUXPool extends QBeanSupport implements MUX {
33         int strategy = 0;
34         String[] muxName;
35         MUX[] mux;
36         int msgno = 0;
37         public static final int ROUND_ROBIN = 1;
38         public static final int PRIMARY_SECONDARY = 0;
39            
40         public void initService () throws ConfigurationException {
41             Element e = getPersist ();
42             muxName = toStringArray(e.getChildTextTrim ("muxes"));
43             String s = e.getChildTextTrim ("strategy");
44             strategy = "round-robin".equals (s) ? ROUND_ROBIN : PRIMARY_SECONDARY;
45     
46             mux = new MUX[muxName.length];
47             try {
48                 for (int i=0; i<mux.length; i++)
49                     mux[i] = QMUX.getMUX (muxName[i]);
50             } catch (NameRegistrar.NotFoundException ex) {
51                 throw new ConfigurationException (ex);
52             }
53             NameRegistrar.register ("mux."+getName (), this);
54         }
55         public void stopService () {
56             NameRegistrar.unregister ("mux."+getName ());
57         }
58         public ISOMsg request (ISOMsg m, long timeout) throws ISOException {
59             int mnumber = 0;
60             long maxWait = System.currentTimeMillis() + timeout;
61             synchronized (this) {
62                 mnumber = msgno++;
63             }
64             MUX mux = strategy == ROUND_ROBIN ? 
65                 nextAvailableMUX (mnumber, maxWait) :
66                 firstAvailableMUX (maxWait);
67     
68             if (mux != null) {
69                 timeout = maxWait - System.currentTimeMillis();
70                 if (timeout >= 0)
71                     return mux.request (m, timeout);
72             }
73             return null;
74         }
75         public boolean isConnected() {
76             for (int i=0; i<mux.length; i++)
77                 if (mux[i].isConnected())
78                     return true;
79             return false;
80         }
81         private MUX firstAvailableMUX (long maxWait) {
82             do {
83                 for (int i=0; i<mux.length; i++)
84                     if (mux[i].isConnected())
85                         return mux[i];
86                 ISOUtil.sleep (1000);
87             } while (System.currentTimeMillis() < maxWait);
88             return null;
89         }
90         private MUX nextAvailableMUX (int mnumber, long maxWait) {
91             do {
92                 for (int i=0; i<mux.length; i++) {
93                     int j = (mnumber+i) % mux.length;
94                     if (mux[j].isConnected())
95                         return mux[j];
96                 }
97                 ISOUtil.sleep (1000);
98             } while (System.currentTimeMillis() < maxWait);
99             return null;
100         }
101         private String[] toStringArray (String s) {
102             String[] ss = null;
103             if (s != null && s.length() > 0) {
104                 StringTokenizer st = new StringTokenizer (s);
105                 ss = new String[st.countTokens()];
106                 for (int i=0; st.hasMoreTokens(); i++)
107                     ss[i] = st.nextToken();
108             }
109             return ss;
110         }
111         public void request (ISOMsg m, long timeout, final ISOResponseListener r, final Object handBack) 
112             throws ISOException 
113         {
114             int mnumber = 0;
115             long maxWait = System.currentTimeMillis() + timeout;
116             synchronized (this) {
117                 mnumber = msgno++;
118             }
119             MUX mux = strategy == ROUND_ROBIN ?
120             nextAvailableMUX (mnumber, maxWait) :
121             firstAvailableMUX (maxWait);
122     
123             if (mux != null) {
124                 timeout = maxWait - System.currentTimeMillis();
125                 if (timeout >= 0)
126                     mux.request(m, timeout,r, handBack);
127                 else {
128                     new Thread() {
129                         public void run() {
130                             r.expired (handBack);
131                         }
132                     }.start();
133                 }
134             } else 
135                 throw new ISOException ("No MUX available");
136         }
137     }
