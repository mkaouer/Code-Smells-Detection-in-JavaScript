153         private void recalcMaxField() {
154             maxField = 0;
155             Iterator iter = fields.keySet().iterator();
156             while (iter.hasNext()) {
157                 Object obj = iter.next();
158                 if (obj instanceof Integer)
159                     maxField = Math.max (maxField, ((Integer)obj).intValue());
160             }
161             maxFieldDirty = false;
162         }

189        /**
190         * Creates an ISOField associated with fldno within this ISOMsg
191         * @param fldno field number
192         * @param value field value
193         */
194         public void set(int fldno, String value) throws ISOException {
195             if (value != null) {
196                 if (!(packager instanceof ISOBasePackager)) {
197                     // No packager is available, we can't tell what the field
198                     // might be, so treat as a String!
199                     set(new ISOField(fldno, value));
200                 }
201                 else {
202                     // This ISOMsg has a packager, so use it
203                     Object obj = ((ISOBasePackager) packager).getFieldPackager(fldno);
204                     if (obj instanceof ISOBinaryFieldPackager) {
205                         set(new ISOBinaryField(fldno, ISOUtil.hex2byte(value)));
206                     } else {
207                         set(new ISOField(fldno, value));
208                     }
209                 }
210             }
211             else
212                 unset(fldno);
213         }

215        /**
216         * Creates an ISOField associated with fldno within this ISOMsg
217         * @param fpath dot-separated field path (i.e. 63.2)
218         * @param value field value
219         */
220         public void set (String fpath, String value) throws ISOException {
221             StringTokenizer st = new StringTokenizer (fpath, ".");
222             ISOMsg m = this;
223             for (;;) {
224                 int fldno = Integer.parseInt(st.nextToken());
225                 if (st.hasMoreTokens()) {
226                     Object obj = m.getValue(fldno);
227                     if (obj instanceof ISOMsg)
228                         m = (ISOMsg) obj;
229                     else
230                         /* 
231                          * we need to go deeper, however, if the value == null then
232                          * there is nothing to do (unset) at the lower levels, so break now and save some processing.
233                          */
234                         if (value == null) {
235                             break;
236                         } else {
237                             // We have a value to set, so adding a level to hold it is sensible.
238                             m.set (m = new ISOMsg (fldno));
239                         }
240                 } else {
241                     m.set (fldno, value);
242                     break;
243                 }
244             }
245         }

331         /**
332          * Unset a field referenced by a fpath if it exists, otherwise ignore.
333          * @param fpath dot-separated field path (i.e. 63.2)
334          * @throws ISOException 
335         */
336         public void unset (String fpath) throws ISOException {
337             StringTokenizer st = new StringTokenizer (fpath, ".");
338             ISOMsg m = this;
339             ISOMsg lastm = m;
340             int fldno = -1 ;
341             int lastfldno ;
342             for (;;) {
343                 lastfldno = fldno;
344                 fldno = Integer.parseInt(st.nextToken());
345                 if (st.hasMoreTokens()) {
346                     Object obj = m.getValue(fldno);
347                     if (obj instanceof ISOMsg) {
348                         lastm = m;
349                         m = (ISOMsg) obj;
350                     }
351                     else {
352                         // No real way of unset further subfield, exit.  Perhaps should be ISOException?
353                         break;
354                     }
355                 } else {
356                     m.unset(fldno);
357                     if (m.hasFields() == false  && (lastfldno != -1)) {
358                         lastm.unset(lastfldno);
359                     }
360                     break;
361                 }
362             }
363         }

374         /**
375          * setup BitMap
376          * @exception ISOException
377          */
378         public void recalcBitMap () throws ISOException {
379             if (!dirty)
380                 return;
381     
382             int mf = Math.min (getMaxField(), 192);
383     
384             BitSet bmap = new BitSet (((mf+62)>>6)<<6);
385             for (int i=1; i<=mf; i++)
386                 if ((fields.get (i)) != null)
387                     bmap.set (i);
388             set (new ISOBitMap (-1, bmap));
389             dirty = false;
390         }

425          * dump the message to a PrintStream. The output is sorta
426          * XML, intended to be easily parsed.
427          * <br>
428          * Each component is responsible for its own dump function,
429          * ISOMsg just calls dump on every valid field.
430          * @param p - print stream
431          * @param indent - optional indent string
432          */
433         public void dump (PrintStream p, String indent) {
434             ISOComponent c;
435             p.print (indent + "<" + XMLPackager.ISOMSG_TAG);
436             switch (direction) {
437                 case INCOMING:
438                     p.print (" direction=\"incoming\"");
439                     break;
440                 case OUTGOING:
441                     p.print (" direction=\"outgoing\"");
442                     break;
443             }
444             if (fieldNumber != -1)
445                 p.print (" "+XMLPackager.ID_ATTR +"=\""+fieldNumber +"\"");
446             p.println (">");
447             String newIndent = indent + "  ";
448             if (getPackager() != null) {
449                p.println (
450                   newIndent
451                + "<!-- " + getPackager().getDescription() + " -->"
452                );
453             }
454             if (header instanceof Loggeable)
455                 ((Loggeable) header).dump (p, newIndent);
456     
457             for (int i=0; i<=maxField; i++) {
458                 if ((c = (ISOComponent) fields.get (i)) != null)
459                     c.dump (p, newIndent);
460                 //
461                 // Uncomment to include bitmaps within logs

464                 //  if ((c = (ISOComponent) fields.get (new Integer (-1))) != null)
465                 //    c.dump (p, newIndent);
466                 // }
467                 //
468             }
469     
470             p.println (indent + "</" + XMLPackager.ISOMSG_TAG+">");
471         }

667         public Object clone() {
668             try {
669                 ISOMsg m = (ISOMsg) super.clone();
670                 m.fields = (TreeMap) ((TreeMap) fields).clone();
671                 if (header != null)
672                     m.header = (ISOHeader) header.clone();
673     
674                 Iterator iter = fields.keySet().iterator();
675                 while (iter.hasNext()) {
676                     Integer k = (Integer) iter.next();
677                     ISOComponent c = (ISOComponent) m.fields.get (k);
678                     if (c instanceof ISOMsg) 
679                         m.fields.put (k, ((ISOMsg)c).clone());
680                 }
681                 return m;
682             } catch (CloneNotSupportedException e) {
683                 throw new InternalError();
684             }
685         }

691          */
692         public Object clone(int[] fields) {
693             try {
694                 ISOMsg m = (ISOMsg) super.clone();
695                 m.fields = new TreeMap();
696                 for (int i=0; i<fields.length; i++) {
697                     if (hasField(fields[i])) {
698                         try {
699                             m.set (getComponent(fields[i]));
700                         } catch (ISOException e) { 
701                             // it should never happen
702                         }
703                     }
704                 }
705                 return m;
706             } catch (CloneNotSupportedException e) {
707                 throw new InternalError();
708             }
709         }

713          * please note that received fields take precedence over 
714          * existing ones (simplifying card agent message creation

718         public void merge (ISOMsg m) {
719             for (int i=0; i<=m.getMaxField(); i++) 
720                 try {
721                     if (m.hasField(i))
722                         set (m.getComponent(i));

824         /**
825          * sets an appropiate response MTI.
826          *
827          * i.e. 0100 becomes 0110<br>
828          * i.e. 0201 becomes 0210<br>
829          * i.e. 1201 becomes 1210<br>
830          * @exception ISOException on MTI not set or it is not a request
831          */
832         public void setResponseMTI() throws ISOException {
833             if (!isRequest())
834                 throw new ISOException ("not a request - can't set response MTI");
835     
836             String mti = getMTI();
837             char c1 = mti.charAt(3);
838             char c2 = '0';
839             switch (c1)
840             {
841                 case '0' :
842                 case '1' : c2='0';break;
843                 case '2' :
844                 case '3' : c2='2';break;
845                 case '4' :
846                 case '5' : c2='4';break;
847     
848             }
849             set (new ISOField (0,
850                 mti.substring(0,2)
851                 +(Character.getNumericValue(getMTI().charAt (2))+1) + c2
852                 )
853             );
854         }

882             out.writeByte('P');

889         ClassNotFoundException {
890             byte[] b = new byte[in.readShort()];
891             in.readFully(b);
892             try {
893                 Class mypClass = Class.forName(new String(b));
894                 ISOPackager myp = (ISOPackager) mypClass.newInstance();
895                 setPackager(myp);
896             } catch (Exception e) {
897                 setPackager(null);
898             }

911         public void writeExternal (ObjectOutput out) throws IOException {
912             out.writeByte (0);  // reserved for future expansion (version id)
913             out.writeShort (fieldNumber);
914     
915             if (header != null) 
916                 writeHeader (out);
917             if (packager != null)
918                 writePackager(out);
919             if (direction > 0)
920                 writeDirection (out);
921     
922             // List keySet = new ArrayList (fields.keySet());
923             // Collections.sort (keySet);
924             Iterator iter = fields.values().iterator();
925             while (iter.hasNext()) {
926                 ISOComponent c = (ISOComponent) iter.next();
927                 if (c instanceof ISOMsg) {
928                     writeExternal (out, 'M', c);
929                 }
930                 else if (c instanceof ISOBinaryField) {
931                     writeExternal (out, 'B', c);
932                 }
933                 else if (c instanceof ISOAmount) {
934                     writeExternal (out, 'A', c);
935                 }
936                 else if (c instanceof ISOField) {
937                     writeExternal (out, 'F', c);
938                 }
939             }
940             out.writeByte ('E');
941         }

943         public void readExternal  (ObjectInput in)
944             throws IOException, ClassNotFoundException
945         {
946             in.readByte();  // ignore version for now
947             fieldNumber = in.readShort();
948             byte fieldType;
949             ISOComponent c;
950             try {
951                 while ((fieldType = in.readByte()) != 'E') {
952                     c = null;
953                     switch (fieldType) {
954                         case 'F':
955                             c = new ISOField ();
956                             break;
957                         case 'A':
958                             c = new ISOAmount ();
959                             break;
960                         case 'B':
961                             c = new ISOBinaryField ();
962                             break;
963                         case 'M':
964                             c = new ISOMsg ();
965                             break;
966                         case 'H':
967                             readHeader (in);
968                             break;
969                         case 'P':
970                             readPackager(in);
971                             break;
972                         case 'D':
973                             readDirection (in);
974                             break;
975                         default:
976                             throw new IOException ("malformed ISOMsg");
977                     }
978                     if (c != null) {
979                         ((Externalizable)c).readExternal (in);
980                         set (c);
981                     }
982                 }
983             }
984             catch (ISOException e) {
985                 throw new IOException (e.getMessage());
986             }
987         }
