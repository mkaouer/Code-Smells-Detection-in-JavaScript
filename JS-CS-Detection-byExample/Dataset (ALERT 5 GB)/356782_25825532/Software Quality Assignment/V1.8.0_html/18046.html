19     package org.jpos.core;
20     
21     
22     /**
23      * SubConfiguration objects lets childs objects access attributes
24      * in its parent object with a given prefix, for example "child.".
25      * Child objects can access properties with their prefix removed.
26      *
27      * @author <a href="mailto:alcarraz@iie.edu.uy">Andr?s Alcarraz</a>
28      * @version $Id$
29      */
30     public class SubConfiguration implements Configuration{
31         /** Parent Configuration */
32         protected Configuration cfg;
33         /** prefix identifying the child parameters */
34         protected String prefix;
35         /** Creates a new empty SubConfiguration object */
36         public SubConfiguration() {
37             super();
38         }
39         /**
40          * Creates a new SubConfiguration from its parent's Configuration 
41          * and the a given <code>prefix</code>.
42          * @param cfg Parent's Configuration 
43          * @param prefix prefix identifying child parameters.
44          */
45         public SubConfiguration(Configuration cfg, String prefix) {
46             super ();
47             this.cfg = cfg;
48             this.prefix = prefix;
49         }
50         /**
51          * Sets the container configuration.
52          * @param newCfg New container configuration.
53          */
54         public void setConfiguration(Configuration newCfg){
55             cfg=newCfg;
56         }
57         /**
58          * Sets the prefix that identifies the parameters of the child object
59          * inside the parent configuration.
60          * @param newPrefix New prefix
61          */
62         public void setPrefix(String newPrefix){
63             prefix = newPrefix;
64         }
65         public String get(String propertyName){
66             return cfg.get(prefix + propertyName);
67         }
68         public String[] getAll(String propertyName){
69             return cfg.getAll(prefix + propertyName);
70         }
71         public int[] getInts (String propertyName) {
72             return cfg.getInts (prefix + propertyName);
73         }
74         public long[] getLongs (String propertyName) {
75             return cfg.getLongs (prefix + propertyName);
76         }
77         public double[] getDoubles (String propertyName) {
78             return cfg.getDoubles (prefix + propertyName);
79         }
80         public boolean[] getBooleans (String propertyName) {
81             return cfg.getBooleans(prefix + propertyName);
82         }
83         public String get(String propertyName, String defaultValue){
84             return cfg.get(prefix + propertyName, defaultValue);
85         }
86         public boolean getBoolean(String propertyName){
87             return cfg.getBoolean(prefix + propertyName);
88         }
89         public boolean getBoolean(String propertyName, boolean defaultValue){
90             return cfg.getBoolean(prefix + propertyName, defaultValue);
91         }
92         public double getDouble(String propertyName){
93             return cfg.getDouble(prefix + propertyName);
94         }
95         public double getDouble(String propertyName, double defaultValue){
96             return cfg.getDouble(prefix + propertyName, defaultValue);
97         }
98         public long getLong(String propertyName){
99             return cfg.getLong(prefix + propertyName);
100         }
101         public long getLong(String propertyName, long defaultValue){
102             return cfg.getLong(prefix + propertyName, defaultValue);
103         }
104         public int getInt(String propertyName){
105             return cfg.getInt(prefix + propertyName);
106         }
107         public int getInt(String propertyName, int defaultValue){
108             return cfg.getInt(prefix + propertyName, defaultValue);
109         }
110         public void put (String name, Object value) {
111             cfg.put (prefix + name, value);
112         }
113         /**
114          * Creates a new object, it takes the class from the value of the property
115          * <code>propertyName</code>
116          * @param propertyName Property whose value is the class name of  
117          * the object being created.
118          */
119         public Object getObject (String propertyName) throws ConfigurationException{
120             try{
121                 Object ret = 
122                     Class.forName (get (propertyName)).newInstance();
123                 if(ret instanceof Configurable) 
124                     ((Configurable)ret).setConfiguration(this);
125                 return ret;
126             } catch (Exception e){
127                 throw new ConfigurationException ("Error trying to create an " 
128                                                  + "object from property " 
129                                                  + prefix + propertyName, e
130                                                  );
131             }
132         }
133     }
