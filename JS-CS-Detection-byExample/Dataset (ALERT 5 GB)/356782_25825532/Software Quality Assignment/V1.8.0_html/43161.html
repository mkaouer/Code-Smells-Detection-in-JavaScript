19     package org.jpos.iso;
20     
21     import org.jpos.util.FSDMsg;
22     import org.jdom.JDOMException;
23     
24     import java.io.*;
25     import java.util.Iterator;
26     import java.util.Map;
27     
28     public class FSDISOMsg extends ISOMsg implements Cloneable  {
29         FSDMsg fsd;
30         public  FSDISOMsg () {
31             super();
32         }
33         public FSDISOMsg (FSDMsg fsd) {
34             super();
35             this.fsd = fsd;
36         }
37         public String getMTI() {
38             return getString(0);
39         }
40         public byte[] pack() throws ISOException {
41             try {
42                 return fsd.packToBytes();
43             } catch (Exception e) {
44                 throw new ISOException (e);
45             }
46         }
47         public int unpack(byte[] b) throws ISOException {
48             try {
49                 fsd.unpack (b);
50                 return b.length;
51             } catch (Exception e) {
52                 throw new ISOException (e);
53             }
54         }
55         public void unpack (InputStream in) throws IOException, ISOException {
56             synchronized (this) {
57                 try {
58                     fsd.unpack(in);
59                 } catch (JDOMException e) {
60                     throw new ISOException (e);
61                 }
62             }
63         }
64     
65         public FSDMsg getFSDMsg() {
66             return fsd;
67         }
68         public String getString (int fldno) {
69             return fsd.get (Integer.toString(fldno));
70         }
71         public boolean hasField (int fldno) {
72             return getString(fldno) != null;
73         }
74         public void dump (PrintStream p, String indent) {
75             if (fsd != null)
76                 fsd.dump (p, indent);
77         }
78         public void writeExternal (ObjectOutput out) throws IOException {
79             out.writeByte (0);  // reserved for future expansion (version id)
80             out.writeUTF (fsd.getBasePath());
81             out.writeUTF (fsd.getBaseSchema());
82             out.writeObject (fsd.getMap());
83         }
84         public void readExternal  (ObjectInput in) 
85             throws IOException, ClassNotFoundException
86         {
87             in.readByte();  // ignore version for now
88             String basePath = in.readUTF();
89             String baseSchema = in.readUTF();
90             fsd = new FSDMsg (basePath, baseSchema);
91             Map map = (Map) in.readObject();
92             Iterator iter = map.entrySet().iterator();
93             while (iter.hasNext()) {
94                 Map.Entry entry = (Map.Entry) iter.next();
95                 fsd.set ((String) entry.getKey(), (String) entry.getValue());
96             }
97         }
98         public Object clone() {
99             FSDISOMsg m = (FSDISOMsg) super.clone();
100             m.fsd = (FSDMsg) fsd.clone();
101             return m;
102         }
103         public Object clone(int[] fields) {
104             FSDISOMsg m = (FSDISOMsg) super.clone();
105             m.fsd = new FSDMsg(fsd.getBasePath(), fsd.getBaseSchema());
106             for (int i=0; i<fields.length; i++) {
107                 String f = Integer.toString(fields[i]);
108                 m.fsd.set (f, fsd.get (f));
109             }
110             return m;
111         }
112         public void merge (ISOMsg m) {
113             if (m instanceof FSDISOMsg) {
114                 fsd.merge (((FSDISOMsg)m).getFSDMsg());
115             } else {
116                 for (int i=0; i<=m.getMaxField(); i++) {
117                     if (m.hasField(i))
118                         fsd.set (Integer.toString(i), m.getString(i));
119                 }
120             }
121         }
122         public void setResponseMTI() {
123             try {
124                 super.setResponseMTI();
125             } catch (ISOException ignored) { }               
126         }
127         public void set (String name, String value) {
128             if (value != null)
129                 this.fsd.set (name, value);
130         }   
131         private static final long serialVersionUID = 1L;
132     }
