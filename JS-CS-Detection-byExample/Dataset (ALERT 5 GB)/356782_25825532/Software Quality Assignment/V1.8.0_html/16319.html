25     /**
26      * Validator for ASCII numeric fields. By default radix is 10.
27      * <p>Title: jPOS</p>
28      * <p>Description: Java Framework for Financial Systems</p>
29      * <p>Copyright: Copyright (c) 2000 jPOS.org.  All rights reserved.</p>
30      * <p>Company: www.jPOS.org</p>
31      * @author Jose Eduardo Leon
32      * @version 1.0
33      */
34     public class IVA_NUM extends ISOFieldValidator {
35     
36         public IVA_NUM(  ) {
37             super(  );
38         }
39     
40         /**
41          * radix = 10.
42          * @param Description Brief description.
43          */
44         public IVA_NUM( String Description ) {
45             super( Description );
46         }
47     
48         public IVA_NUM( String Description, int radix ) {
49             super( Description );
50             this.radix = radix;
51         }
52     
53         public IVA_NUM( int maxLen, String Description ) {
54             super( maxLen, Description );
55         }
56     
57         public IVA_NUM( int maxLen, String Description, int radix ) {
58             super( maxLen, Description );
59             this.radix = radix;
60         }
61     
62         /**
63          * Create the validator. Radix is 10.
64          * @param minLen min length.
65          * @param maxLen max length
66          * @param Description Validator description
67          */
68         public IVA_NUM( int minLen, int maxLen, String Description ) {
69             super( minLen, maxLen, Description );
70         }
71     
72         /**
73          * Create the validator
74          * @param minLen min length.
75          * @param maxLen max length
76          * @param Description Validator description
77          * @param radix numeric radix for numeric validation
78          */
79         public IVA_NUM( int minLen, int maxLen, String Description, int radix ) {
80             super( minLen, maxLen, Description );
81             this.radix = radix;
82         }
83     
84         public IVA_NUM( boolean breakOnError, String Description ) {
85             this( Description );
86             this.breakOnError = breakOnError;
87         }
88     
89         public IVA_NUM( boolean breakOnError, String Description, int radix ) {
90             this( Description, radix );
91             this.breakOnError = breakOnError;
92         }
93     
94         public IVA_NUM( boolean breakOnError, int maxLen, String Description ) {
95             this( maxLen, Description );
96             this.breakOnError = breakOnError;
97         }
98     
99         public IVA_NUM( boolean breakOnError, int maxLen, String Description, int radix ) {
100             this( maxLen, Description, radix );
101             this.breakOnError = breakOnError;
102         }
103     
104         public IVA_NUM( boolean breakOnError, int minLen, int maxLen, String Description ) {
105             this( minLen, maxLen, Description );
106             this.breakOnError = breakOnError;
107         }
108     
109         public IVA_NUM( boolean breakOnError, int minLen, int maxLen, String Description, int radix ) {
110             this( minLen, maxLen, Description, radix );
111             this.breakOnError = breakOnError;
112         }
113     
114         /**
115          * Configure the validator. @see ISOFieldValidator class.
116          * Take config param "radix" wich specify the numeric radix.
117          * @param cfg configuration instance
118          * @throws ConfigurationException
119          */
120         public void setConfiguration(Configuration cfg) throws ConfigurationException {
121             super.setConfiguration( cfg );
122             this.radix = cfg.getInt( "radix", 10 );
123         }
124     
125         /**
126          * Validate numeric condition. @see ISOFieldValidator class.
127          * @param f ISOField to validate
128          * @return see validate method in ISOFieldValidator class.
129          * @throws ISOException if any validation error.
130          */
131         public ISOComponent validate( ISOComponent f ) throws ISOException {
132             ISOField c = (ISOField)f;
133             c = (ISOField)super.validate( c );
134             if ( !ISOUtil.isNumeric( (String)c.getValue(), this.radix ) ){
135                 ISOVError e = new ISOVError(
136                         "Invalid Value Error. " + (String)c.getValue() +
137                         " is not a numeric value in radix " +
138                         this.radix, getRejCode( ISOVError.ERR_INVALID_VALUE ) );
139                 if ( c instanceof ISOVField )
140                     ((ISOVField)c).addISOVError( e );
141                 else
142                     c = new ISOVField( c, e );
143                 if ( breakOnError )
144                     throw new ISOVException ( "Error on field " + ((Integer)c.getKey()).intValue(), c );
145             }
146             return c;
147         }
148     
149         /** by default is decimal **/
150         protected int radix = 10;
151     }
