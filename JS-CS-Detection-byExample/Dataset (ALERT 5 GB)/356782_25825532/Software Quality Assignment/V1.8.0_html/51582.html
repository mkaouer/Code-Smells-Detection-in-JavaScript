19     package org.jpos.iso.channel;
20     
21     import org.jpos.core.Configuration;
22     import org.jpos.core.ConfigurationException;
23     import org.jpos.iso.FSDISOMsg;
24     import org.jpos.iso.ISOException;
25     import org.jpos.iso.ISOMsg;
26     import org.jpos.util.FSDMsg;
27     import org.jpos.util.LogEvent;
28     import org.jpos.util.Logger;
29     
30     import java.io.IOException;
31     
32     public class FSDChannel extends NACChannel {
33         String schema;
34     
35         public ISOMsg createMsg() {
36             return new FSDISOMsg (new FSDMsg (schema));
37         }
38         public void setConfiguration (Configuration cfg)
39             throws ConfigurationException 
40         {
41             super.setConfiguration (cfg);
42             schema = cfg.get ("schema");
43         }
44         protected int getMessageLength() throws IOException, ISOException {
45             int len = super.getMessageLength();
46             LogEvent evt = new LogEvent (this, "fsd-channel-debug");
47             evt.addMessage ("received message length: " + len);
48             Logger.log (evt);
49             return len;
50         }
51     }
