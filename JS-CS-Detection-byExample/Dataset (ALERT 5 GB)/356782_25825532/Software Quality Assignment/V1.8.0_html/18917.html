19     package  org.jpos.security;
20     
21     import org.jpos.util.Loggeable;
22     
23     import java.io.PrintStream;
24     import java.io.Serializable;
25     
26     
27     /**
28      * Key Serial Number (also called Key Name in the ANSI X9.24).
29      * Needed for deriving the Transaction Key when  DUKPT (Derived Unique Key Per
30      * Transaction) method is used.<br>
31      * Refer to ANSI X9.24 for more information about DUKPT
32      * @author Hani S. Kirollos
33      * @version $Revision$ $Date$
34      * @see EncryptedPIN
35      */
36     public class KeySerialNumber
37             implements Serializable, Loggeable {
38     
39         private static final long serialVersionUID = -8388775376202253082L;
40         /**
41          * baseKeyID a HexString representing the BaseKeyID (also called KeySet ID)
42          */
43         String baseKeyID;
44         /**
45          * deviceID a HexString representing the Device ID (also called TRSM ID)
46          */
47         String deviceID;
48         /**
49          * transactionCounter a HexString representing the transaction counter
50          */
51         String transactionCounter;
52     
53         /**
54          * Constructs a key serial number object
55          */
56         public KeySerialNumber () {
57         }
58     
59         /**
60          * Constructs a key serial number object
61          * @param baseKeyID a HexString representing the BaseKeyID (also called KeySet ID)
62          * @param deviceID a HexString representing the Device ID (also called TRSM ID)
63          * @param transactionCounter a HexString representing the transaction counter
64          */
65         public KeySerialNumber (String baseKeyID, String deviceID, String transactionCounter) {
66             setBaseKeyID(baseKeyID);
67             setDeviceID(deviceID);
68             setTransactionCounter(transactionCounter);
69         }
70     
71         /**
72          *
73          * @param baseKeyID a HexString representing the BaseKeyID (also called KeySet ID)
74          */
75         public void setBaseKeyID (String baseKeyID) {
76             this.baseKeyID = baseKeyID;
77         }
78     
79         /**
80          *
81          * @return baseKeyID a HexString representing the BaseKeyID (also called KeySet ID)
82          */
83         public String getBaseKeyID () {
84             return  baseKeyID;
85         }
86     
87         /**
88          *
89          * @param deviceID a HexString representing the Device ID (also called TRSM ID)
90          */
91         public void setDeviceID (String deviceID) {
92             this.deviceID = deviceID;
93         }
94     
95         /**
96          *
97          * @return deviceID a HexString representing the Device ID (also called TRSM ID)
98          */
99         public String getDeviceID () {
100             return  deviceID;
101         }
102     
103         /**
104          *
105          * @param transactionCounter a HexString representing the transaction counter
106          */
107         public void setTransactionCounter (String transactionCounter) {
108             this.transactionCounter = transactionCounter;
109         }
110     
111         /**
112          *
113          * @return transactionCounter a HexString representing the transaction counter
114          */
115         public String getTransactionCounter () {
116             return  transactionCounter;
117         }
118     
119         /**
120          * dumps Key Serial Number
121          * @param p a PrintStream usually supplied by Logger
122          * @param indent indention string, usually suppiled by Logger
123          * @see org.jpos.util.Loggeable
124          */
125         public void dump (PrintStream p, String indent) {
126             String inner = indent + "  ";
127             p.println(indent + "<key-serial-number>");
128             p.println(inner + "<base-key-id>" + getBaseKeyID() + "</base-key-id>");
129             p.println(inner + "<device-id>" + getDeviceID() + "</device-id>");
130             p.println(inner + "<transaction-counter>" + getTransactionCounter() + "</transaction-counter");
131             p.println(indent + "</key-serial-number>");
132         }
133     }
