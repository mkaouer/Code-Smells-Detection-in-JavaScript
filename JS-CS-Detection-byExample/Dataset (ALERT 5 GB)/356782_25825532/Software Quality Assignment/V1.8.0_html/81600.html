21     /**
22      * It is possible to ask ISOMUX to forward all unmatched
23      * messages received through its associated ISOChannel
24      * to be processed by an ISORequestListener.
25      * 
26      * @author apr@cs.com.uy
27      * @version $Id$
28      * @see ISOMUX#setISORequestListener
29      */
30     public interface ISORequestListener {
31         /**
32          * @param  source source where you optionally can reply
33          * @param  m   the unmatched request
34          * @return true if request was handled by this listener
35          */
36         public boolean process (ISOSource source, ISOMsg m);
37     }
