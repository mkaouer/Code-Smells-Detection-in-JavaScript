19     package org.jpos.iso;
20     
21     import org.jpos.iso.packager.XMLPackager;
22     
23     import java.io.*;
24     import java.math.BigDecimal;
25     
26     public class ISOAmount 
27         extends ISOComponent 
28         implements Cloneable, Externalizable
29     {
30         static final long serialVersionUID = -6130248734056876225L;
31         private int fieldNumber;
32         private int currencyCode;
33         private String value;
34         private BigDecimal amount;
35     
36         public ISOAmount () {
37             super();
38             setFieldNumber (-1);
39         }
40         public ISOAmount (int fieldNumber) {
41             super ();
42             setFieldNumber (fieldNumber);
43         }
44         public ISOAmount (int fieldNumber, int currencyCode, BigDecimal amount) {
45             super ();
46             setFieldNumber (fieldNumber);
47             this.amount = amount;
48             this.currencyCode = currencyCode;
49         }
50         public Object getKey() {
51             return fieldNumber;
52         }
53         public Object getValue() throws ISOException {
54             if (value == null) {
55                 StringBuilder sb = new StringBuilder();
56                 sb.append (ISOUtil.zeropad (Integer.toString(currencyCode), 3));
57                 sb.append (Integer.toString (amount.scale()));
58                 sb.append (
59                     ISOUtil.zeropad (
60                         amount.movePointRight(amount.scale()).toString(),12
61                     )
62                 );
63                 value = sb.toString();
64             }
65             return value;
66     
67         }
68         public void setValue (Object obj) throws ISOException {
69             if (obj instanceof String) {
70                 String s = (String) obj;
71                 if (s.length() < 12) {
72                     throw new ISOException (
73                         "ISOAmount invalid length " + s.length()
74                     );
75                 }
76                 try {
77                     currencyCode = Integer.parseInt (s.substring(0,3));
78                     int dec = Integer.parseInt (s.substring(3,4));
79                     amount = new BigDecimal (s.substring(4)).movePointLeft (dec);
80                     value  = s;
81                 } catch (NumberFormatException e) {
82                     throw new ISOException (e.getMessage());
83                 }
84             }
85         }
86         public void setFieldNumber (int fieldNumber) {
87             this.fieldNumber = fieldNumber;
88         }
89         public BigDecimal getAmount () {
90             return amount;
91         }
92         public int getScale() {
93             return amount.scale() % 10;
94         }
95         public String getScaleAsString() {
96             return Integer.toString(getScale());
97         }
98         public int getCurrencyCode() {
99             return currencyCode;
100         }
101         public String getCurrencyCodeAsString() throws ISOException {
102             return ISOUtil.zeropad(Integer.toString(currencyCode),3);
103         }
104         public String getAmountAsLegacyString() throws ISOException {
105             return ISOUtil.zeropad (amount.unscaledValue().toString(), 12);
106         }
107         public String getAmountAsString() throws ISOException {
108             StringBuffer sb = new StringBuffer(16);
109             sb.append (ISOUtil.zeropad (Integer.toString (currencyCode),3));
110             sb.append (Integer.toString(amount.scale() % 10));
111             sb.append (ISOUtil.zeropad (amount.unscaledValue().toString(), 12));
112             return sb.toString();
113         }
114         public byte[] pack() throws ISOException {
115             throw new ISOException ("Not available");
116         }
117         public int unpack(byte[] b) throws ISOException {
118             throw new ISOException ("Not available");
119         }
120         public void unpack(InputStream in) throws ISOException {
121             throw new ISOException ("Not available");
122         }
123         public void dump (PrintStream p, String indent) {
124             p.println (indent +"<"+XMLPackager.ISOFIELD_TAG + " " 
125               +XMLPackager.ID_ATTR +"=\"" +fieldNumber +"\" "
126               +"currency=\"" +ISOUtil.zeropad (currencyCode, 3)+"\" "
127               +XMLPackager.TYPE_ATTR +"=\"amount\" "
128               +XMLPackager.VALUE_ATTR+"=\"" + amount.toString() +"\"/>"
129             );
130         }
131         public void writeExternal (ObjectOutput out) throws IOException {
132             out.writeShort (fieldNumber);
133             try {
134                 out.writeUTF ((String) getValue());
135             } catch (ISOException e) {
136                 throw new IOException (e);
137             }
138         }
139         public void readExternal  (ObjectInput in) 
140             throws IOException, ClassNotFoundException
141         {
142             fieldNumber = in.readShort ();
143             try {
144                 setValue(in.readUTF());
145             } catch (ISOException e) {
146                 throw new IOException (e.getMessage());
147             }
148         }       
149     }
