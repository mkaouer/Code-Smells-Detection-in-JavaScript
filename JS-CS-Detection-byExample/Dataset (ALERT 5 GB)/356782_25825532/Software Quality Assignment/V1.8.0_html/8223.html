19     package org.jpos.q2.qbean;
20     
21     import org.jdom.Element;
22     import org.jpos.core.Configurable;
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.q2.QBeanSupport;
25     import org.jpos.q2.QFactory;
26     import org.jpos.util.LogListener;
27     import org.jpos.util.Logger;
28     
29     import java.util.Iterator;
30     
31     public class LoggerAdaptor extends QBeanSupport {
32         Logger logger;
33     
34         protected void initService () {
35             logger = Logger.getLogger (getName());
36         }
37         protected void startService () throws ConfigurationException {
38             logger.removeAllListeners ();
39             Iterator iter = getPersist ().getChildren ("log-listener").iterator();
40             while (iter.hasNext()) 
41                 addListener ((Element) iter.next ());
42         }
43         protected void stopService() {
44             logger.removeAllListeners ();
45         }
46         protected void destroyService() {
47             // we don't destroy (that would unregister the logger from the
48             // NameRegistrar) because other components might have references
49             // to this logger.
50             //
51             // logger.destroy ();
52         }
53         private void addListener (Element e) 
54             throws ConfigurationException
55         {
56             QFactory factory = getServer().getFactory();
57             String clazz  = e.getAttributeValue ("class");
58             LogListener listener = (LogListener) factory.newInstance (clazz);
59             if (listener instanceof Configurable) {
60                 try {
61                     ((Configurable) listener).setConfiguration (
62                         factory.getConfiguration (e)
63                     );
64                 } catch (ConfigurationException ex) {
65                     throw new ConfigurationException (ex);
66                 }
67             }
68             
69             logger.addListener (listener);
70         }
71     }
