19     package org.jpos.iso;
20     
21     import org.jpos.iso.ISOFilter.VetoException;
22     
23     import java.io.IOException;
24     
25     /**
26      * Source for an ISORequest (where to send a reply)
27      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
28      * @version $Id$
29      * @see ISORequestListener
30      */
31     public interface ISOSource {
32         /**
33          * sends (or hands back) an ISOMsg
34          * @param m the Message to be sent
35          * @exception IOException
36          * @exception ISOException
37          * @exception ISOFilter.VetoException;
38          */
39         public void send (ISOMsg m) 
40             throws IOException, ISOException, VetoException;
41     
42         /**
43          * @return true if source is connected and usable
44          */
45         public boolean isConnected();
46     }
