32      * Test validatingPackager for subelements in field 48.

41     public class CTCSubElementPackager extends ISOBaseValidatingPackager {
42     
43         public CTCSubElementPackager() {
44             super();
45         }
46     
47         public byte[] pack ( ISOComponent c ) throws ISOException {
48             try     {
49                 Map tab = c.getChildren();
50                 StringBuffer sb = new StringBuffer();
51                 for ( int i = 0; i < fld.length; i++ ) {
52                     ISOMsg f = (ISOMsg) tab.get (new Integer(i));
53                     if ( f != null ) {
54                         sb.append ( ISOUtil.zeropad( ((Integer)f.getKey()).toString(), 2 ) + new String( fld[i].pack( f ) ) );
55                     }
56                 }
57                 return sb.toString().getBytes();
58             }
59             catch ( Exception ex ) {
60                 throw new ISOException ( this.getRealm() + ":" + ex.getMessage(), ex );
61             }
62         }
63     
64         public int unpack ( ISOComponent m, byte[] b ) throws ISOException {
65             LogEvent evt = new LogEvent ( this, "unpack" );
66             int consumed = 0;
67             for ( int i=0; consumed < b.length ; i++ ) {
68                 ISOComponent c = fld[i].createComponent( i );
69                 consumed += fld[i].unpack ( c, b, consumed );
70                 if ( logger != null )       {
71                     evt.addMessage ("<unpack fld=\"" + i
72                                     +"\" packager=\""
73                                     +fld[i].getClass().getName()+ "\">");
74                     if (c.getValue() instanceof ISOMsg)
75                         evt.addMessage (c.getValue());
76                     else
77                         evt.addMessage ("  <value>"
78                                         +c.getValue().toString()
79                                         + "</value>");
80                     evt.addMessage ("</unpack>");
81                 }
82                 m.set(c);
83             }
84             Logger.log (evt);
85             return consumed;
86         }
87     
88         /**
89          * Always return false
90          * <br><br>
91          **/
92         protected boolean emitBitMap() {
93             return false;
94         }
95     
96         public ISOComponent validate( ISOComponent c ) throws org.jpos.iso.ISOException {
97             LogEvent evt = new LogEvent( this, "validate" );
98             try {
99                 Map tab = c.getChildren();
100                 for ( int i = 0; i < fldVld.length; i++ ) {
101                     ISOMsg f = (ISOMsg) tab.get (new Integer(i));
102                     if ( f != null )
103                         c.set( (ISOMsg)fldVld[i].validate( f ) );
104                 }
105                 return c;
106             } catch ( ISOVException ex ) {
107                 if ( !ex.treated() ) {
108                     c.set( ex.getErrComponent() );
109                     ex.setTreated( true );
110                 }
111                 evt.addMessage( ex );
112                 throw ex;
113             } finally {
114                 Logger.log( evt );
115             }
116         }

118     }
