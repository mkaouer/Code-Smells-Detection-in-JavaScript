88         public void startService () throws Exception {
89             NameRegistrar.register (getName (), this);
90             recover ();
91             int sessions = cfg.getInt ("sessions", 1);
92             threads = new Thread[sessions];
93             if (tps != null)
94                 tps.stop();
95             tps = new TPS (cfg.getBoolean ("auto-update-tps", true));
96             for (int i=0; i<sessions; i++) {
97                 Thread t = new Thread (this);
98                 t.setName (getName() + "-" + i);
99                 t.setDaemon (false);
100                 t.start ();
101                 threads[i] = t;
102             }
103             if (psp.rdp (RETRY_QUEUE) != null)
104                 checkRetryTask();
105         }
106         public void stopService () throws Exception {
107             NameRegistrar.unregister (getName ());
108             long sessions = cfg.getLong ("sessions", 1);
109             for (int i=0; i<sessions; i++) {
110                 isp.out (queue, Boolean.FALSE, 60*1000);
111             }
112             for (int i=0; i<sessions; i++) {
113                 try {
114                     threads[i].join (60*1000);
115                 } catch (InterruptedException e) {
116                     getLog().warn ("Session " +i +" does not response - attempting to interrupt");
117                     threads[i].interrupt();
118                 }
119                 threads[i] = null;
120             }
121             tps.stop();
122         }

142         public void run () {
143             long id = 0;
144             int session;
145             List members = null;
146             Iterator iter = null;
147             PausedTransaction pt;
148             boolean abort = false;
149             LogEvent evt = null;
150             Profiler prof = null;
151             String threadName = Thread.currentThread().getName();
152             getLog().info (threadName + " start");
153             long startTime = 0L;
154             boolean paused;
155             synchronized (HEAD) { 
156                 session = activeSessions++;
157             }
158             while (running()) {
159                 Serializable context = null;
160                 paused = false;
161                 try {
162                     if (hasStatusListeners)
163                         notifyStatusListeners (session, TransactionStatusEvent.State.READY, id, "", null);
164     
165                     Object obj = isp.in (queue);
166                     if (obj == Boolean.FALSE)
167                         continue;   // stopService ``hack''
168                     if (!(obj instanceof Serializable)) {
169                         getLog().error (
170                             "non serializable '" + obj.getClass().getName()
171                            + "' on queue '" + queue + "'"
172                         );
173                         continue;
174                     }
175                     context = (Serializable) obj;
176                     if (obj instanceof Pausable) {
177                         Pausable pausable = (Pausable) obj;
178                         pt = pausable.getPausedTransaction();
179                         if (pt != null) {
180                             pt.cancelExpirationMonitor();
181                             id      = pt.id();
182                             members = pt.members();
183                             iter    = pt.iterator();
184                             abort   = pt.isAborting();
185                         }
186                     } else 
187                         pt = null;
188     
189                     if (pt == null) {
190                         int running = getRunningSessions();
191                         if (maxActiveSessions > 0 && running >= maxActiveSessions) {
192                             evt = getLog().createLogEvent ("warn",
193                                 Thread.currentThread().getName() 
194                                 + ": emergency retry, running-sessions=" + running 
195                                 + ", max-active-sessions=" + maxActiveSessions
196                             );
197                             evt.addMessage (obj);
198                             psp.out (RETRY_QUEUE, obj, retryTimeout);
199                             checkRetryTask();
200                             continue;
201                         }
202                         abort = false;
203                         id = nextId ();
204                         members = new ArrayList ();
205                         iter = getParticipants (DEFAULT_GROUP).iterator();
206                     }
207                     if (debug) {
208                         evt = getLog().createLogEvent ("debug",
209                             Thread.currentThread().getName() 
210                             + ":" + Long.toString(id) +
211                             (pt != null ? " [resuming]" : "")
212                         );
213                         prof = new Profiler();
214                         startTime = System.currentTimeMillis();
215                     }
216                     snapshot (id, context, PREPARING);
217                     int action = prepare (session, id, context, members, iter, abort, evt, prof);
218                     switch (action) {
219                         case PAUSE:
220                             paused = true;
221                             break;
222                         case PREPARED:
223                             setState (id, COMMITTING);
224                             commit (session, id, context, members, false, evt, prof);
225                             break;
226                         case ABORTED:
227                             abort (session, id, context, members, false, evt, prof);
228                             break;
229                         case RETRY:
230                             psp.out (RETRY_QUEUE, context);
231                             checkRetryTask();
232                             break;
233                         case NO_JOIN:
234                             break;
235                     }
236                     if ((action & PAUSE) == 0) {
237                         snapshot (id, null, DONE);
238                         if (id == tail) {
239                             checkTail ();
240                         }
241                         tps.tick();
242                     }
243                 } catch (Throwable t) {
244                     if (evt == null)
245                         getLog().fatal (t); // should never happen
246                     else
247                         evt.addMessage (t);
248                 } finally {
249                     if (hasStatusListeners) {
250                         notifyStatusListeners (
251                             session,
252                             paused ? TransactionStatusEvent.State.PAUSED : TransactionStatusEvent.State.DONE, 
253                             id, "", context);
254                     }
255     
256                     if (evt != null) {
257                         evt.addMessage (
258                             String.format ("head=%d, tail=%d, outstanding=%d, %s, elapsed=%dms",
259                                 head, tail, getOutstandingTransactions(),
260                                 tps.toString(),
261                                 (System.currentTimeMillis() - startTime)
262                             )
263                         );
264                          if (prof != null)
265                             evt.addMessage (prof);
266                         Logger.log (evt);
267                         evt = null;
268                     }
269                 }
270             }
271             getLog().info (threadName + " stop");
272             synchronized (HEAD) {
273                 activeSessions--;
274             }
275             getLog().info ("stop " + Thread.currentThread() + ", active sessions=" + activeSessions);
276         }

315         protected void commit 
316             (int session, long id, Serializable context, List members, boolean recover, LogEvent evt, Profiler prof)
317         {
318             Iterator iter = members.iterator();
319             while (iter.hasNext ()) {
320                 TransactionParticipant p = (TransactionParticipant) iter.next();
321                 if (recover && p instanceof ContextRecovery) {
322                     context = ((ContextRecovery) p).recover (id, context, true);
323                     if (evt != null)
324                         evt.addMessage (" commit-recover: " + p.getClass().getName());
325                 }
326                 if (hasStatusListeners)
327                     notifyStatusListeners (
328                         session, TransactionStatusEvent.State.COMMITING, id, p.getClass().getName(), context
329                     );
330                 commit (p, id, context);
331                 if (evt != null) {
332                     evt.addMessage ("         commit: " + p.getClass().getName());
333                     if (prof != null)
334                         prof.checkPoint (" commit: " + p.getClass().getName());
335                 }
336             }
337         }
338         protected void abort 
339             (int session, long id, Serializable context, List members, boolean recover, LogEvent evt, Profiler prof)
340         {
341             Iterator iter = members.iterator();
342             while (iter.hasNext ()) {
343                 TransactionParticipant p = (TransactionParticipant) iter.next();
344                 if (recover && p instanceof ContextRecovery) {
345                     context = ((ContextRecovery) p).recover (id, context, false);
346                     if (evt != null)
347                         evt.addMessage ("  abort-recover: " + p.getClass().getName());
348                 }
349                 if (hasStatusListeners)
350                     notifyStatusListeners (
351                         session, TransactionStatusEvent.State.ABORTING, id, p.getClass().getName(), context
352                     );
353     
354                 abort (p, id, context);
355                 if (evt != null) {
356                     evt.addMessage ("          abort: " + p.getClass().getName());
357                     if (prof != null)
358                         prof.checkPoint ("  abort: " + p.getClass().getName());
359                 }
360             }

401         protected int prepare
402             (int session, long id, Serializable context, List members, Iterator iter, boolean abort, LogEvent evt, Profiler prof)
403         {
404             boolean retry = false;
405             boolean pause = false;
406             for (int i=0; iter.hasNext (); i++) {
407                 int action;
408                 if (i > MAX_PARTICIPANTS) {
409                     getLog().warn (
410                         "loop detected - transaction " +id + " aborted."
411                     );
412                     return ABORTED;
413                 }
414                 TransactionParticipant p = (TransactionParticipant) iter.next();
415                 if (abort) {
416                     if (hasStatusListeners)
417                         notifyStatusListeners (
418                             session, TransactionStatusEvent.State.PREPARING_FOR_ABORT, id, p.getClass().getName(), context
419                         );
420                     action = prepareForAbort (p, id, context);
421                     if (evt != null && (p instanceof AbortParticipant))
422                         evt.addMessage ("prepareForAbort: " + p.getClass().getName());
423                 } else {
424                     if (hasStatusListeners)
425                         notifyStatusListeners (
426                             session, TransactionStatusEvent.State.PREPARING, id, p.getClass().getName(), context
427                         );
428                     action = prepare (p, id, context);
429                     abort  = (action & PREPARED) == ABORTED;
430                     retry  = (action & RETRY) == RETRY;
431                     pause  = (action & PAUSE) == PAUSE;
432                     if (evt != null) {
433                         evt.addMessage ("        prepare: "
434                                 + p.getClass().getName() 
435                                 + (abort ? " ABORTED" : "")
436                                 + (retry ? " RETRY" : "")
437                                 + (pause ? " PAUSE" : "")
438                                 + ((action & READONLY) == READONLY ? " READONLY" : "")
439                                 + ((action & NO_JOIN) == NO_JOIN ? " NO_JOIN" : ""));
440                         if (prof != null)
441                             prof.checkPoint ("prepare: " + p.getClass().getName());
442                     }
443                 }
444                 if ((action & READONLY) == 0) {
445                     snapshot (id, context);
446                 }
447                 if ((action & NO_JOIN) == 0) {
448                     members.add (p);
449                 }
450                 if (p instanceof GroupSelector) {
451                     String groupName = null;
452                     try {
453                         groupName = ((GroupSelector)p).select (id, context);
454                     } catch (Exception e) {
455                         if (evt != null) 
456                             evt.addMessage ("  groupSelector " + p + " - " + e.getMessage());
457                         else 
458                             getLog().error (" groupSelector: " + p + " - " + e.getMessage());
459                     }
460                     if (evt != null) 
461                         evt.addMessage ("  groupSelector: " + groupName);
462                     if (groupName != null) {
463                         StringTokenizer st = new StringTokenizer (groupName, " ,");
464                         List participants = new ArrayList();
465                         while (st.hasMoreTokens ()) {
466                             String grp = st.nextToken();
467                             addGroup (id, grp);
468                             participants.addAll (getParticipants (grp));
469                         }
470                         while (iter.hasNext())
471                             participants.add (iter.next());
472     
473                         iter = participants.iterator();
474                         continue;
475                     }
476                 }
477                 if (pause) {
478                     if (context instanceof Pausable) {
479                         Pausable pausable = (Pausable) context;
480                         long t = pausable.getTimeout();
481                         if (t == 0) 
482                             t = pauseTimeout;
483                         TimerTask expirationMonitor = null;
484                         if (t > 0)
485                             expirationMonitor = new PausedMonitor (pausable);
486                         PausedTransaction pt = new PausedTransaction (
487                             this, id, members, iter, abort, expirationMonitor
488                         );
489                         pausable.setPausedTransaction (pt);
490                         if (expirationMonitor != null) {
491                             synchronized (context) {
492                                 if (!pt.isResumed()) {
493                                     DefaultTimer.getTimer().schedule (
494                                         expirationMonitor, t
495                                     );
496                                 }
497                             }
498                         }
499                     } else {
500                         throw new RuntimeException ("Unable to PAUSE transaction - Context is not Pausable");
501                     }
502                     return PAUSE;
503                 }
504             }
505             return members.size() == 0 ? NO_JOIN : 
506                 (abort ? (retry ? RETRY : ABORTED) : PREPARED);
507         }

514         protected List getParticipants (long id) {
515         	// Use a local copy of participant to avoid adding the 
516             // GROUP participant to the DEFAULT_GROUP
517         	ArrayList participantsChain = new ArrayList();
518             List participants = getParticipants (DEFAULT_GROUP);
519             // Add DEFAULT_GROUP participants 
520             participantsChain.addAll(participants);
521             String key = getKey(GROUPS, id);
522             String grp;
523             // now add participants of Group 
524             while ( (grp = (String) psp.inp (key)) != null) {
525                 participantsChain.addAll (getParticipants (grp));
526             }
527             return participantsChain;
528         }

621         protected void initTailLock () {
622             tailLock = TAILLOCK + "." + Integer.toString (this.hashCode());
623             SpaceUtil.wipe (sp, tailLock);
624             sp.out (tailLock, TAILLOCK);
625         }
626         protected void checkTail () {
627             Object lock = sp.in (tailLock);
628             while (tailDone()) {
629                 // if (debug) {
630                 //    getLog().debug ("tailDone " + tail);
631                 // }
632                 tail++;
633             }
634             syncTail ();
635             sp.out (tailLock, lock);
636         }
637         protected boolean tailDone () {
638             String stateKey = getKey (STATE, tail);
639             if (DONE.equals (psp.rdp (stateKey))) {
640                 purge (tail);
641                 return true;
642             }
643             return false;
644         }
645         protected long nextId () {
646             long h;
647             synchronized (psp) {
648                 commitOff (psp);
649                 psp.in  (HEAD);
650                 h = head;
651                 psp.out (HEAD, ++head);
652                 commitOn (psp);
653             }
654             return h;
655         }

659         protected void snapshot (long id, Serializable context, Integer status) {
660             String contextKey = getKey (CONTEXT, id);
661             synchronized (psp) {
662                 commitOff (psp);
663                 while (psp.inp (contextKey) != null)
664                     ;
665                 if (context != null)
666                     psp.out (contextKey, context);
667     
668                 if (status != null) {
669                     String stateKey  = getKey (STATE, id);
670                     while (psp.inp (stateKey) != null)
671                         ;
672                     psp.out (stateKey, status);
673                 }
674                 commitOn (psp);
675             }
676         }
677         protected void setState (long id, Integer state) {
678             String stateKey  = getKey (STATE, id);
679             synchronized (psp) {
680                 commitOff (psp);
681                 while (psp.inp (stateKey) != null)
682                     ;
683                 if (state!= null)
684                     psp.out (stateKey, state);
685                 commitOn (psp);
686             }
687         }

692         protected void purge (long id) {
693             String stateKey   = getKey (STATE, id);
694             String contextKey = getKey (CONTEXT, id);
695             String groupsKey  = getKey (GROUPS, id);
696             synchronized (psp) {
697                 commitOff (psp);
698                 while (psp.inp (stateKey) != null)
699                     ;
700                 while (psp.inp (contextKey) != null)
701                     ;
702                 while (psp.inp (groupsKey) != null)
703                     ;
704                 commitOn (psp);
705             }
706         }
707         protected void recover () {
708             if (doRecover) {
709                 if (tail < head) {
710                     getLog().info ("recover - tail=" +tail+", head="+head);
711                 }
712                 while (tail < head) {
713                     recover (0, tail++);
714                 }
715             } else
716                 tail = head;
717             syncTail ();
718         }
719         protected void recover (int session, long id) {
720             LogEvent evt = getLog().createLogEvent ("recover");
721             Profiler prof = new Profiler();
722             evt.addMessage ("<id>" + id + "</id>");
723             try {
724                 String stateKey   = getKey (STATE, id);
725                 String contextKey = getKey (CONTEXT, id);
726                 Integer state = (Integer) psp.rdp (stateKey);
727                 if (state == null) {
728                     evt.addMessage ("unknown stateKey " + stateKey);
729                     SpaceUtil.wipe (psp, contextKey);   // just in case ...
730                     return;
731                 }
732                 Serializable context = (Serializable) psp.rdp (contextKey);
733                 if (context != null)
734                     evt.addMessage (context);
735     
736                 if (DONE.equals (state)) {
737                     evt.addMessage ("<done/>");
738                 } else if (COMMITTING.equals (state)) {
739                     commit (session, id, context, getParticipants (id), true, evt, prof);
740                 } else if (PREPARING.equals (state)) {
741                     abort (session, id, context, getParticipants (id), true, evt, prof);
742                 }
743                 purge (id);
744             } finally {
745                 evt.addMessage (prof);
746                 Logger.log (evt);
747             }
748         }

750             if (retryTask == null) {
751                 retryTask = new RetryTask();
752                 new Thread(retryTask).start();
753             }

767         public class RetryTask implements Runnable {
768             public void run() {
769                 Thread.currentThread().setName (getName()+"-retry-task");
770                 while (running()) {
771                     for (Object context; (context = psp.rdp (RETRY_QUEUE)) != null;) 
772                     {
773                         isp.out (queue, context, retryTimeout);
774                         psp.inp (RETRY_QUEUE);
775                     }
776                     try {
777                         Thread.sleep (retryInterval);
778                     } catch (InterruptedException ignored) { } 
779                 }
780             }
781         }
