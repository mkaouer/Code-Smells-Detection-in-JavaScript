31     package org.jpos.q2.iso;
32     
33     import java.text.ParseException;
34     import java.text.SimpleDateFormat;
35     import java.util.Date;
36     import java.util.Timer;
37     import java.util.TimerTask;
38     import org.jpos.core.ConfigurationException;
39     import org.jpos.util.NameRegistrar;
40     
41     /**
42      *
43      * @author amarqueslin
44      */
45     public class SchedulingTaskAdaptor extends TaskAdaptor {
46         
47         private Timer timer = new Timer();
48         private Long delay;
49         private Date firstTime;
50         private long period;
51         
52         
53         /** Creates a new instance of SchedulingTaskAdaptor */
54         public SchedulingTaskAdaptor() {
55         }
56         
57         public class RunnableWrapper extends TimerTask {
58             
59             private Runnable innerRunnable;
60             
61             public RunnableWrapper(Runnable innerRunnable) {
62                 this.innerRunnable = innerRunnable;
63             }
64             
65             public void run() {
66                 innerRunnable.run();
67             }
68         }
69     
70         public void setConfiguration(org.jpos.core.Configuration cfg) throws org.jpos.core.ConfigurationException {
71     
72             super.setConfiguration(cfg);
73             boolean daemon = cfg.getBoolean("daemon", true);
74             if (daemon) {
75                 timer = new Timer(true);
76             }
77             period = cfg.getLong("period", 24 * 60 * 60 * 1000); //By default execute task every 24hs
78             
79             String initTimeInfo = "";
80             
81             long delayLong = cfg.getLong("delay", -1);
82             if (delayLong != -1) {
83                 delay = new Long(delayLong);
84                 initTimeInfo = "delay = " + delay;
85             } else {
86                 String strFirstTime = cfg.get("firstTime", "00:00:00"); //By default start at 12 AM
87                 Date now = new Date();
88                 if (strFirstTime.equals("now")) {
89                     firstTime = now;
90                 } else {
91                     SimpleDateFormat dateTimeFormat = new SimpleDateFormat("dd/MM/yyyy HH:mm:ss");
92                     SimpleDateFormat dateFormat = new SimpleDateFormat("dd/MM/yyyy");
93                     try {
94                         String today = dateFormat.format(now);
95                         Date time = dateTimeFormat.parse(today + " " + strFirstTime);
96                         if (time.after(now)) {
97                             firstTime = time;
98                         } else {
99                             firstTime = new Date(time.getTime() + 1000 * 60 * 60 * 24);
100                         }
101                     } catch (ParseException ex) {
102                         throw new ConfigurationException("The format in which to specify the first execution time is HH:mm:ss");
103                     }
104                     initTimeInfo = "firstTime = " + dateTimeFormat.format(firstTime);
105                 }
106             }
107             
108             String strConfig = "daemon = " + daemon + "\n" +
109                                initTimeInfo  + "\n"
110                                 + "period = " + period;
111                                 
112             log.info("Configuration", strConfig);
113         }
114         
115         protected void startService() throws Exception {
116             getServer().getFactory().setConfiguration(task, getPersist());
117             NameRegistrar.register (getName (), task);
118             if (task instanceof TimerTask) {
119                 schedule((TimerTask) task);
120                 return;
121             }
122             if (task instanceof Runnable) {
123                 schedule(new RunnableWrapper((Runnable) task));
124             }
125         }
126         
127         protected void schedule(TimerTask task) {
128             if (delay != null) {
129                 timer.schedule(task, delay.longValue(), period);
130             } else {
131                 timer.schedule(task, firstTime, period);
132             }
133         }
134     
135         protected void stopService() throws Exception {
136     
137             super.stopService();
138             timer.cancel();
139         }
