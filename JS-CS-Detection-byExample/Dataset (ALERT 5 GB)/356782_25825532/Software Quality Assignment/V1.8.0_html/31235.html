19     package org.jpos.util;
20     
21     import org.jpos.core.Configurable;
22     import org.jpos.core.Configuration;
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.util.BlockingQueue.Closed;
25     
26     import java.io.PrintStream;

28     /**
29      * Implements a ThreadPool with the ability to run simple Runnable
30      * tasks as well as Jobs (supervised Runnable tasks)
31      * @since 1.1
32      * @author apr@cs.com.uy
33      */
34     public class ThreadPool extends ThreadGroup implements LogSource, Loggeable, Configurable, ThreadPoolMBean
35     {
36         private static int poolNumber=0;
37         private static int threadNumber=0;
38         private int maxPoolSize = 1;
39         private int available;
40         private int running = 0;
41         private int active  = 0;
42         private BlockingQueue pool = new BlockingQueue();
43         private Logger logger;
44         private String realm;
45         private int jobs = 0;
46         public static final int DEFAULT_MAX_THREADS = 100;
47         public interface Supervised {
48             public boolean expired ();
49         }
50     
51         private class PooledThread extends Thread {
52             Object currentJob = null;
53     
54             public PooledThread() {
55                 super ((ThreadGroup) ThreadPool.this, 
56                     "PooledThread-" + (threadNumber++));
57                 setDaemon(true);
58             }
59             public void run () {
60                 String name = getName();
61                 try {
62                     while (pool.ready()) {
63                         Object job = pool.dequeue();
64                         if (job instanceof Runnable) {
65                             setName (name + "-running");
66                             synchronized (this) {
67                                 currentJob = job;
68                             }
69                             try {
70                                 active++;
71                                 ((Runnable) job).run();
72                                 setName (name + "-idle");
73                             } catch (Throwable t) {
74                                 setName (name + "-idle-"+t.getMessage());
75                             }
76                             synchronized (this) {
77                                 currentJob = null;
78                                 available++;
79                                 active--;
80                             }
81                         }
82                     }
83                 } catch (InterruptedException e) {
84                 } catch (Closed e) {
85                 }
86             }
87             public synchronized void supervise () {
88                 if (currentJob != null && currentJob instanceof Supervised) 
89                     if ( ((Supervised)currentJob).expired() )
90                         this.interrupt();
91             }
92         }
93     
94         /**
95          * @param poolSize starting pool size
96          * @param maxPoolSize maximum number of threads on this pool
97          */
98         public ThreadPool (int poolSize, int maxPoolSize) {
99             super ("ThreadPool-" + poolNumber++);
100             this.maxPoolSize = maxPoolSize > 0 ? maxPoolSize : DEFAULT_MAX_THREADS ;
101             this.available = this.maxPoolSize;
102             init (poolSize);
103         }
104         /**
105          * @param name pool name
106          * @param poolSize starting pool size
107          * @param maxPoolSize maximum number of threads on this pool
108          */
109         public ThreadPool (int poolSize, int maxPoolSize, String name) {
110             super (name + "-" + poolNumber++);
111             this.maxPoolSize = maxPoolSize > 0 ? maxPoolSize : DEFAULT_MAX_THREADS ;
112             this.available = this.maxPoolSize;
113             init (poolSize);
114         }
115         
116         private void init(int poolSize){
117             while (running < Math.min (poolSize > 0 ? poolSize : 1, maxPoolSize)) {
118                 running++;
119                 new PooledThread().start();
120             }
121         }
122         /**
123          * Default constructor for ThreadPool
124          */
125         public ThreadPool () {
126             this(1, DEFAULT_MAX_THREADS);
127         }
128         public void close () {
129             pool.close();
130         }
131         public synchronized void execute (Runnable action) throws Closed
132         {
133             if (!pool.ready())
134                 throw new Closed();
135     
136             if (++jobs % this.maxPoolSize == 0 || pool.consumerCount() <= 0)
137                 supervise();
138     
139             synchronized (pool) {
140             	if (running < maxPoolSize && pool.consumerCount() <= 0) {
141                 	new PooledThread().start();
142     	            running++;
143         	    }
144         	}
145             available--;
146             pool.enqueue (action);
147         }
148         public void dump (PrintStream p, String indent) {
149             String inner = indent + "  ";
150             p.println (indent + "<thread-pool name=\""+getName()+"\">");
151             if (!pool.ready())
152                 p.println (inner  + "<closed/>");
153             p.println (inner  + "<jobs>" +jobs+"</jobs>");
154             p.println (inner  + "<size>" +available+"</size>");
155             p.println (inner  + "<max>"  +maxPoolSize+"</max>");
156             p.println (inner  + "<active>" + running +"</active>");
157             p.println (inner  + "<idle>"  + pool.consumerCount() +"</idle>");
158             p.println (inner  + "<pending>"  +pool.pending()+"</pending>");
159             p.println (indent + "</thread-pool>");
160         }

163          * @return number of jobs processed by this pool
164          */
165         public int getJobCount () {
166             return jobs;
167         }
168         /**
169          * @return number of active threads
170          */
171         public int getPoolSize () {
172             return available;
173         }
174         /**
175          * @return max number of active threads allowed
176          */
177         public int getMaxPoolSize () {
178             return maxPoolSize;
179         }
180         /**
181          * @return number of running threads
182          */
183         public int getActiveCount () {
184             return active;
185         }
186         /**
187          * @return number of idle threads
188          */
189         public int getIdleCount () {
190             return pool.consumerCount ();
191         }
192         /**
193          * @return number of available threads
194          */
195         synchronized public int getAvailableCount () {
196             return available;
197         }
198     
199         /**
200          * @return number of Pending jobs
201          */
202         public int getPendingCount () {
203             return pool.pending ();
204         }
205     
206         public void supervise () {
207             Thread[] t = new Thread[maxPoolSize];
208             int cnt = enumerate (t);
209             for (int i=0; i<cnt; i++) 
210                 if (t[i] instanceof PooledThread)
211                     ((PooledThread) t[i]).supervise();
212         }
213     
214         public void setLogger (Logger logger, String realm) {
215             this.logger = logger;
216             this.realm  = realm;
217         }
218         public String getRealm () {
219             return realm;
220         }
221         public Logger getLogger() {
222             return logger;
223         }
224         
225        /** 
226         * @param cfg Configuration object
227         * @throws ConfigurationException
228         */
229         public void setConfiguration(Configuration cfg) throws ConfigurationException {
230             maxPoolSize = cfg.getInt("max-size", DEFAULT_MAX_THREADS);
231             init (cfg.getInt("initial-size"));
232         }

234         /** 
235          * Retrieves a thread pool from NameRegistrar given its name, unique identifier.
236          *
237          * @param name Name of the thread pool to retrieve, must be the same as the name property of the thread-pool tag in the QSP config file
238          * @throws NotFoundException thrown when there is not a thread-pool registered under this name.
239          * @return returns the retrieved instance of thread pool
240          */    
241         public static ThreadPool getThreadPool(java.lang.String name) throws org.jpos.util.NameRegistrar.NotFoundException {
242             return (ThreadPool)NameRegistrar.get("thread.pool." + name);
243         }
