52     /**
53      * <pre>
54      * GenericPackager uses an XML config file to describe the layout of an ISOMessage
55      * The general format is as follows
56      * &lt;isopackager&gt;
57      *     &lt;isofield 
58      *         id="[field id]"
59      *         name="[field name]"
60      *         length="[max field length]"
61      *         class="[org.jpos.iso.IF_*]"
62      *         pad="true|false"&gt;  
63      *     &lt;/isofield&gt;
64      *     ...
65      * &lt;/isopackager&gt;
66      *
67      * Fields that contain subfields can be handled as follows
68      * &lt;isofieldpackager
69      *     id="[field id]"
70      *     name="[field name]"
71      *     length="[field length]"
72      *     class="[org.jpos.iso.IF_*]"
73      *     packager="[org.jpos.iso.packager.*]"&gt;
74      *      
75      *     &lt;isofield 
76      *         id="[subfield id]"
77      *         name="[subfield name]"
78      *         length="[max subfield length]"
79      *         class="[org.jpos.iso.IF_*]"
80      *         pad="true|false"&gt;
81      *     &lt;/isofield&gt;
82      *         ...
83      * &lt;/isofieldpackager&gt;
84      *
85      * The optional attributes maxValidField, bitmapField and emitBitmap
86      * are allowed on the isopackager node.
87      *
88      * </pre>
89      * @author Eoin Flood
90      * @version $Revision$ $Date$
91      * @see ISOPackager
92      * @see ISOBasePackager
93      */
94

156                     this.filename = cfg.get("packager-config")
157                 );
158             } 
159             catch (ISOException e) 
160             {
161                 throw new ConfigurationException(e);
162             }
163         }

183         /**
184          * Parse the field descriptions from an XML file.
185          *
186          * <pre>
187          * Uses the sax parser specified by the system property 'sax.parser'
188          * The default parser is org.apache.crimson.parser.XMLReaderImpl
189          * </pre>
190          * @param filename The XML field description file
191          */
192         public void readFile(String filename) throws ISOException
193         {
194             try {
195                 createXMLReader().parse(filename);  
196             } 
197             catch (Exception e) {
198                 throw new ISOException("Error reading " + filename, e);
199             }
200         }

220         @Override
221         public void setLogger (Logger logger, String realm) {
222             super.setLogger (logger, realm);
223             if (fld != null) {
224                 for (int i=0; i<fld.length; i++) {
225                     if (fld[i] instanceof ISOMsgFieldPackager) {
226                         Object o = ((ISOMsgFieldPackager)fld[i]).getISOMsgPackager();
227                         if (o instanceof LogSource) {
228                             ((LogSource)o).setLogger (logger, realm + "-fld-" + i);
229                         }
230                     } 
231                 }
232             }

234         private XMLReader createXMLReader () throws SAXException {
235             XMLReader reader;
236             try {
237                 reader = XMLReaderFactory.createXMLReader();
238             } catch (SAXException e) {
239                 reader = XMLReaderFactory.createXMLReader (
240                     System.getProperty( 
241                         "org.xml.sax.driver", 
242                         "org.apache.crimson.parser.XMLReaderImpl"
243                     )
244                 );
245             }
246             reader.setFeature ("http://xml.org/sax/features/validation", true);
247             GenericContentHandler handler = new GenericContentHandler();
248             reader.setContentHandler(handler);
249             reader.setErrorHandler(handler);
250             reader.setEntityResolver(new GenericEntityResolver());
251             return reader;
252         }

264         private void setGenericPackagerParams (Attributes atts)
265         {
266             String maxField  = atts.getValue("maxValidField");
267             String emitBmap  = atts.getValue("emitBitmap");
268             String bmapfield = atts.getValue("bitmapField");
269             firstField = atts.getValue("firstField");
270             String headerLenStr = atts.getValue("headerLength");
271     
272             if (maxField != null)
273                 maxValidField = Integer.parseInt(maxField); 
274     
275             if (emitBmap != null)
276                 emitBitmap = Boolean.valueOf(emitBmap);
277     
278             if (bmapfield != null)
279                 bitmapField = Integer.parseInt(bmapfield);
280     
281             if (firstField != null)
282                 Integer.parseInt (firstField);  // attempt to parse just to
283                                                 // force an exception if the
284                                                 // data is not correct.
285             
286             if (headerLenStr != null)
287             	setHeaderLength(Integer.parseInt(headerLenStr));
288         }

290         public class GenericEntityResolver implements EntityResolver
291         {
292             /**
293              * Allow the application to resolve external entities.
294              * <p/>
295              * The strategy we follow is:<p>
296              * We first check whether the DTD points to a well defined URI,
297              * and resolve to our internal DTDs.<p>
298              *
299              * If the systemId points to a file, then we attempt to read the 
300              * DTD from the filesystem, in case they've been modified by the user.
301              * Otherwise, we fallback to the built-in DTDs inside jPOS.<p>
302              *
303              * @param publicId The public identifier of the external entity
304              *                 being referenced, or null if none was supplied.
305              * @param systemId The system identifier of the external entity
306              *                 being referenced.
307              * @return An InputSource object describing the new input source,
308              *         or null to request that the parser open a regular
309              *         URI connection to the system identifier.
310              * @throws org.xml.sax.SAXException Any SAX exception, possibly
311              *                                  wrapping another exception.
312              * @throws java.io.IOException      A Java-specific IO exception,
313              *                                  possibly the result of creating a new InputStream
314              *                                  or Reader for the InputSource.
315              * @see org.xml.sax.InputSource
316              */
317             public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException
318             {
319                 if(systemId==null) return null;
320     
321                 ClassLoader cl =Thread.currentThread().getContextClassLoader();
322                 cl = cl ==null?ClassLoader.getSystemClassLoader() : cl;
323     
324                 if(systemId.equals("http://jpos.org/dtd/generic-packager-1.0.dtd"))
325                 {
326                     final URL resource = cl.getResource("org/jpos/iso/packager/genericpackager.dtd");
327                     return new InputSource(resource.toExternalForm());
328                 }
329                 else if(systemId.equals("http://jpos.org/dtd/generic-validating-packager-1.0.dtd"))
330                 {
331                     final URL resource = cl.getResource("org/jpos/iso/packager/generic-validating-packager.dtd");
332                     return new InputSource(resource.toExternalForm());
333                 }
334     
335                 URL url=new URL(systemId);
336                 if(url.getProtocol().equals("file"))
337                 {
338                     String file=url.getFile();
339                     if(file.endsWith(".dtd"))
340                     {
341                         File f=new File(file);
342                         InputStream res=null;
343                         if(f.exists())
344                         {
345                             res=new FileInputStream(f);
346                         }
347                         if(res==null)
348                         {
349                             String dtdResource="org/jpos/iso/packager/"+f.getName();
350                             res= cl.getResourceAsStream(dtdResource);
351                         }
352                         if(res!=null) return new InputSource(res);
353                     }
354                 }
355                 return null;
356             }
357         }

359         public class GenericContentHandler extends DefaultHandler
360         {
361             private Stack<Object> fieldStack;
362     
363             @Override
364             public void startDocument()
365             {
366                 fieldStack = new Stack<Object>();
367             }
368     
369             @Override
370             public void endDocument() throws SAXException
371             {
372                 if (!fieldStack.isEmpty())
373                 {
374                     throw new SAXException ("Format error in XML Field Description File");
375                 }
376             }
377     
378             @Override
379             public void startElement(String namespaceURI, String localName, String qName, Attributes atts)
380                     throws SAXException
381             {
382                 try
383                 {
384                     String id   = atts.getValue("id");
385                     String type = atts.getValue("class");
386                     String name = atts.getValue("name");
387                     String size = atts.getValue("length");
388                     String pad  = atts.getValue("pad");
389                     // Modified for using IF_TBASE
390                     String token = atts.getValue("token");
391     
392                     if (localName.equals("isopackager"))
393                     {
394                         // Stick a new Map on stack to collect the fields
395                         fieldStack.push(new TreeMap());
396     
397                         setGenericPackagerParams (atts);
398                     }
399     
400                     if (localName.equals("isofieldpackager"))
401                     {
402                         /*
403                         For a isofield packager node push the following fields
404                         onto the stack.
405                         1) an Integer indicating the field ID
406                         2) an instance of the specified ISOFieldPackager class
407                         3) an instance of the specified ISOBasePackager (msgPackager) class
408                         4) a Map to collect the subfields
409                         */
410                         String packager = atts.getValue("packager");
411     
412                         fieldStack.push(new Integer(id));
413     
414                         ISOFieldPackager f;
415                         f = (ISOFieldPackager) Class.forName(type).newInstance();   
416                         f.setDescription(name);
417                         f.setLength(Integer.parseInt(size));
418                         f.setPad(Boolean.parseBoolean(pad));
419                         // Modified for using IF_TBASE
420                         if( f instanceof IF_TBASE){
421                           ((IF_TBASE)f).setToken( token );
422                         }
423                         fieldStack.push(f);
424     
425                         ISOBasePackager p;
426                         p = (ISOBasePackager) Class.forName(packager).newInstance();
427                         if (p instanceof GenericPackager)
428                         {
429                             GenericPackager gp = (GenericPackager) p;
430                             gp.setGenericPackagerParams (atts);
431                         }
432                         fieldStack.push(p);
433     
434                         fieldStack.push(new TreeMap());
435                     }
436     
437                     if (localName.equals("isofield"))
438                     {
439                         Class c = Class.forName(type);
440                         ISOFieldPackager f;
441                         f = (ISOFieldPackager) c.newInstance();     
442                         f.setDescription(name);
443                         f.setLength(Integer.parseInt(size));
444                         f.setPad(Boolean.parseBoolean(pad));
445                         // Modified for using IF_TBASE
446                         if( f instanceof IF_TBASE){
447                           ((IF_TBASE)f).setToken( token );
448                         }
449                         // Insert this new isofield into the Map
450                         // on the top of the stack using the fieldID as the key
451                         Map m = (Map) fieldStack.peek();
452                         m.put(new Integer(id), f);
453                     }
454                 }
455                 catch (Exception ex)
456                 {
457                     throw new SAXException(ex);
458                 }
459             }

461             /**
462              * Convert the ISOFieldPackagers in the Map
463              * to an array of ISOFieldPackagers
464              */
465             private ISOFieldPackager[] makeFieldArray(Map<Integer,ISOFieldPackager> m)
466             {
467                 int maxField = 0;
468     
469                 // First find the largest field number in the Map
470                 for (Entry<Integer,ISOFieldPackager> ent :m.entrySet())
471                     if (ent.getKey() > maxField)
472                         maxField = ent.getKey();
473     
474                 // Create the array
475                 ISOFieldPackager fld[] = new ISOFieldPackager[maxField+1];
476     
477                 // Populate it
478                 for (Entry<Integer,ISOFieldPackager> ent :m.entrySet())
479                    fld[ent.getKey()] = ent.getValue();
480                 return fld;
481             }
482     
483             @Override
484             public void endElement(String namespaceURI, String localName, String qName)
485             {
486                 Map<Integer,ISOFieldPackager> m;
487                 if (localName.equals("isopackager"))
488                 {
489                     m  = (Map)fieldStack.pop();
490     
491                     setFieldPackager(makeFieldArray(m));
492                 }
493     
494                 if (localName.equals("isofieldpackager"))
495                 {
496                     // Pop the 4 entries off the stack in the correct order
497                     m = (Map)fieldStack.pop();
498     
499                     ISOBasePackager msgPackager = (ISOBasePackager) fieldStack.pop();
500                     msgPackager.setFieldPackager (makeFieldArray(m));
501     
502                     ISOFieldPackager fieldPackager = (ISOFieldPackager) fieldStack.pop();
503     
504                     Integer fno = (Integer) fieldStack.pop();
505     
506                     msgPackager.setLogger (getLogger(), getRealm() + "-fld-" + fno);
507     
508                     // Create the ISOMsgField packager with the retrieved msg and field Packagers
509                     ISOMsgFieldPackager mfp = 
510                         new ISOMsgFieldPackager(fieldPackager, msgPackager);
511     
512                     // Add the newly created ISOMsgField packager to the
513                     // lower level field stack
514     
515                     m=(Map)fieldStack.peek();
516                     m.put(fno, mfp);
517                 }
518             }
