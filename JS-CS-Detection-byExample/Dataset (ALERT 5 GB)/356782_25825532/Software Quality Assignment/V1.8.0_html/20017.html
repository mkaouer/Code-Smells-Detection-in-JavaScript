19     package org.jpos.util;
20     
21     import org.jpos.core.Configurable;
22     import org.jpos.core.Configuration;
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.iso.FSDISOMsg;
25     import org.jpos.iso.ISOException;
26     import org.jpos.iso.ISOUtil;
27     import java.util.List;
28     
29     /**
30      * Protects selected fields from LogEvents.
31      *
32      * FSDProtectedLogListener acts like a filter for Event logs,
33      * it should be defined _before_ other standard LogListeners
34      * such as SimpleLogListener or RotateLogListeners. It is
35      * based upon the ProtectedLogListener<br>
36      * i.e.
37      * <pre>
38      * <logger name="Q2">
39      *   <log-listener class="org.jpos.util.SimpleLogListener"/>
40      *   <log-listener class="org.jpos.util.FSDProtectedLogListener">
41      *     <property name="truncate" value="field1:100 field2:50" />
42      *     <property name="protect" value="account-number track2-data" />
43      *     <property name="wipe"    value="pinblock" />
44      *   </log-listener>
45      *   <log-listener class="org.jpos.util.RotateLogListener">
46      *     <property name="file" value="/log/q2.log" />
47      *     <property name="window" value="86400" />
48      *     <property name="copies" value="5" />
49      *     <property name="maxsize" value="1000000" />
50      *   </log-listener>
51      * </logger>
52      * </pre>
53      * 
54      * Order is important. In the previous example SimpleLogListener
55      * will dump unprotected LogEvents while RotateLogListener will
56      * dump protected ones (for selected fields)
57      *
58      * @author Alejandro P. Revilla
59      * @author Dave Bergert 
60      * @version $Revision$ 
61      * @see org.jpos.core.Configurable
62      * @since jPOS 1.6.5
63      */
64     
65     public class FSDProtectedLogListener implements LogListener, Configurable
66     {
67         String[] protectFields = null;
68         String[] wipeFields    = null;
69         String[] truncateFields    = null;
70         Configuration cfg   = null;
71         public static final String WIPED = "[WIPED]";
72         public static final byte[] BINARY_WIPED = ISOUtil.hex2byte ("AA55AA55");
73     
74         public FSDProtectedLogListener () {
75             super();
76         }
77     
78        /**
79         * Configure this FSDProtectedLogListener<br>
80         * Properties:<br>
81         * <ul>
82         *  <li>[protect]   blank separated list of fields to be protected
83         *  <li>[wipe]      blank separated list of fields to be wiped
84         *  <li>[truncate]  blank separated list of fields:length to be truncated
85         * </ul>
86         * @param cfg Configuration 
87         * @throws ConfigurationException
88         */
89         public void setConfiguration (Configuration cfg)
90             throws ConfigurationException
91         {
92             this.cfg = cfg;
93             truncateFields  = ISOUtil.toStringArray (cfg.get ("truncate", ""));
94             protectFields   = ISOUtil.toStringArray (cfg.get ("protect", ""));
95             wipeFields      = ISOUtil.toStringArray (cfg.get ("wipe", ""));
96         }
97         public synchronized LogEvent log (LogEvent ev) {
98             List payLoad = ev.getPayLoad();
99             int size = payLoad.size();
100             for (int i=0; i<size; i++) {
101                 Object obj = payLoad.get (i);
102                 if (obj instanceof FSDISOMsg) {
103                     FSDISOMsg m = (FSDISOMsg) ((FSDISOMsg) obj).clone();
104                     try {
105                         checkTruncated (m);
106                         checkProtected (m);
107                         checkHidden (m);
108                     } catch (ISOException e) {
109                         ev.addMessage (e);
110                     }
111                     payLoad.set (i, m);
112                 }
113             }
114             return ev;
115         }
116          private void checkTruncated (FSDISOMsg m) throws ISOException {
117             for (int i=0; i<truncateFields.length; i++) {
118         	String truncate[] = truncateFields[i].split(":");
119         	    if (truncate.length == 2) {
120                     String f = truncate[0];
121     		int len = Integer.parseInt(truncate[1]);
122                     Object v = null;
123                     try {
124                         v = m.getFSDMsg().get (f);
125                     } catch (Exception e) {
126                         // ignore error
127                     }
128                     if (v != null) {
129                         if (v instanceof String) {
130                            String x  = ((String) v);
131                            if (x.length()>len){
132                                m.getFSDMsg().set (f, x.substring(0,len));
133                                }
134                         }
135                     }
136            	    }
137     	 }
138         }
139         private void checkProtected (FSDISOMsg m) throws ISOException {
140             for (int i=0; i<protectFields.length; i++) {
141                 String f = protectFields[i];
142                 Object v = null;
143                 try {
144                     v = m.getFSDMsg().get(f);
145                 } catch (Exception e) {
146                     // ignore error
147                 }
148                 if (v != null) {
149                     if (v instanceof String)
150                         m.getFSDMsg().set (f, ISOUtil.protect ((String) v));
151                     else
152                         m.getFSDMsg().set (f, new String(BINARY_WIPED));
153                 }
154             }
155         }
156         private void checkHidden (FSDISOMsg m) throws ISOException {
157             for (int i=0; i<wipeFields.length; i++) {
158                 String f = wipeFields[i];
159                 Object v = null;
160                 try {
161                     v = m.getFSDMsg().get (f);
162                 } catch (Exception e) {
163                     // ignore error
164                 }
165                 if (v != null) {
166                     if (v instanceof String)
167                         m.getFSDMsg().set (f, WIPED);
168                     else
169                         m.getFSDMsg().set (f, new String(BINARY_WIPED));
170                 }
171             }
172         }

175     
176     }
