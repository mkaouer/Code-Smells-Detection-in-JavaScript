19     package org.jpos.iso;
20     
21     import java.io.Serializable;
22     
23     
24     /**
25      * ISO Currency Conversion package
26      *
27      * @author salaman@teknos.com
28      * @version $Id$
29      */
30     public class Currency implements Serializable
31     {
32         String alphacode;
33         int isocode;
34         int numdecimals;
35     
36         public Currency(String alphacode, int isocode, int numdecimals)
37         {
38             this.alphacode = alphacode;
39             this.isocode = isocode;
40             this.numdecimals = numdecimals;
41         }
42     
43         public int getDecimals()
44         {
45             return numdecimals;
46         }
47     
48         public int getIsoCode()
49         {
50             return isocode;
51         }
52     
53         public String getAlphaCode()
54         {
55             return alphacode;
56         }
57     
58         public String formatAmountForISOMsg(double amount)
59         {
60             try
61             {
62                 double m = Math.pow(10, getDecimals()) * amount;
63                 return ISOUtil.zeropad(String.valueOf(Math.round(m)), 12);
64             }
65             catch (ISOException e)
66             {
67                 throw new IllegalArgumentException("Failed to convert amount",e);
68             }
69         }
70     
71         public double parseAmountFromISOMsg(String isoamount)
72         {
73             return new Double(isoamount)/Math.pow(10, getDecimals());
74         }
75     
76         @Override
77         public String toString()
78         {
79             return "Currency{" +
80                    "alphacode='" + alphacode + '\'' +
81                    ", isocode=" + isocode +
82                    ", numdecimals=" + numdecimals +
83                    '}';
84         }
85     }
