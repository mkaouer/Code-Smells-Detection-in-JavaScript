19     package org.jpos.iso.channel;
20     
21     import org.jpos.iso.*;
22     
23     import java.io.IOException;
24     import java.net.ServerSocket;
25     import java.util.zip.GZIPInputStream;
26     import java.util.zip.GZIPOutputStream;
27     
28     /**
29      * ISOChannel implementation.
30      *
31      * @author apr@jpos.org
32      * @version $Id$
33      * @see ISOMsg
34      * @see ISOException
35      * @see ISOChannel
36      */
37     public class GZIPChannel extends BaseChannel {
38         public GZIPChannel () {
39             super();
40         }
41         /**
42          * Construct client ISOChannel
43          * @param host  server TCP Address
44          * @param port  server port number
45          * @param p     an ISOPackager
46          * @see ISOPackager
47          */
48         public GZIPChannel (String host, int port, ISOPackager p) {
49             super(host, port, p);
50         }
51         /**
52          * Construct server ISOChannel
53          * @param p     an ISOPackager
54          * @exception IOException
55          * @see ISOPackager
56          */
57         public GZIPChannel (ISOPackager p) throws IOException {
58             super(p);
59         }
60         /**
61          * constructs a server ISOChannel associated with a Server Socket
62          * @param p     an ISOPackager
63          * @param serverSocket where to accept a connection
64          * @exception IOException
65          * @see ISOPackager
66          */
67         public GZIPChannel (ISOPackager p, ServerSocket serverSocket) 
68             throws IOException
69         {
70             super(p, serverSocket);
71         }
72         /**
73          * @param len the packed Message len
74          * @exception IOException
75          */
76         protected void sendMessageLength(int len) throws IOException {
77             serverOut.write (len >> 8);
78             serverOut.write (len);
79         }
80         /**
81          * @return the Message len
82          * @exception IOException, ISOException
83          */
84         protected int getMessageLength() throws IOException, ISOException {
85             int l = 0;
86             byte[] b = new byte[4];
87             while (l == 0) {
88                 serverIn.readFully(b,0,2);
89                 l = ((((int)b[0])&0xFF) << 8) | (((int)b[1])&0xFF);
90                 if (l == 0) {
91                     serverOut.write(b);
92                     serverOut.flush();
93                 }
94             }
95             return l;
96         }
97         protected void sendMessage (byte[] b, int offset, int len) 
98             throws IOException
99         {
100             GZIPOutputStream gzip = new GZIPOutputStream(serverOut);
101             gzip.write(b, offset, len);
102             gzip.finish();
103             gzip.flush();
104         }
105         protected void getMessage (byte[] b, int offset, int len) throws IOException, ISOException { 
106         	int total = 0;
107             GZIPInputStream gzip = new GZIPInputStream(serverIn);
108             while (total < len) {
109             	int nread = gzip.read (b, offset, len - total);
110             	if (nread == -1) {
111             		throw new ISOException("End of compressed stream reached before all data was read"); 
112             	}
113             	total += nread;
114             	offset += nread;
115             }
116         }
117     }
