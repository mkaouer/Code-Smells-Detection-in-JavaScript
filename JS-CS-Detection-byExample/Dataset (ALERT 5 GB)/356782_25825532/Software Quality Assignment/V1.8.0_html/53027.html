19     package org.jpos.util;
20     
21     import org.apache.log4j.Level;
22     import org.apache.log4j.helpers.FileWatchdog;
23     import org.apache.log4j.xml.DOMConfigurator;
24     import org.jpos.core.Configuration;
25     import org.jpos.core.ConfigurationException;
26     import org.jpos.core.ReConfigurable;
27     
28     import java.io.ByteArrayOutputStream;
29     import java.io.PrintStream;
30     
31     /**
32      * @author Eoin P. FLood</a>
33      * @version $Revision$ $Date$
34      * @see org.jpos.util.LogListener
35      * @see org.jpos.util.LogEvent
36      * @since jPOS 1.3
37      *
38      * This class acts as a simple bridge between jPOS's logging system
39      * and log4j.
40      * The jPOS <code>realm</code> is used as the log4j <code>Logger</code>
41      * and messages are by default logged with the DEBUG level. This can
42      * be changed by calling <code>setLevel</code>
43      */
44     
45     public class Log4JListener implements LogListener, ReConfigurable
46     {
47         private Level _level;
48         /** 
49          * Create a new Log4JListener with DEBUG level.
50          */
51         public Log4JListener ()
52         {
53             setLevel (Level.DEBUG_INT);
54         }
55     
56         public void setLevel (int level) {
57             _level = Level.toLevel (level);
58         }
59     
60         public void setLevel (String level) {
61             _level = Level.toLevel (level);
62         }
63     
64         public void close() 
65         {
66         }
67     
68         /**
69          * Expects the following properties:
70          * <ul>
71          *  <li>config   - Configuration file path
72          *  <li>priority - Log4J priority (debug, info, warn, error)
73          *  <li>watch    - interval (in ms) to monitor XML config file for changes 
74          * </ul>
75          */
76         public void setConfiguration (Configuration cfg) 
77             throws ConfigurationException
78         {
79             String config = cfg.get ("config");
80             long watch =  cfg.getLong ("watch");
81     
82             if (watch == 0)
83                 watch = FileWatchdog.DEFAULT_DELAY;
84     
85             if ( (config!=null) && (!config.trim().equals("")) )
86                 DOMConfigurator.configureAndWatch (config, watch);
87     
88             setLevel (cfg.get ("priority"));
89         }
90     
91         public synchronized LogEvent log (LogEvent ev) {
92     
93             org.apache.log4j.Logger logger = org.apache.log4j.Logger.getLogger (
94                 ev.getRealm().replace('/', ':')
95             );
96             if (logger.isEnabledFor ( _level)) {
97                 ByteArrayOutputStream w = new ByteArrayOutputStream ();
98                 PrintStream p = new PrintStream (w);
99                 ev.dump (p, "");
100                 logger.log (_level, w.toString());
101             }
102             return ev;
103         }
104     }
