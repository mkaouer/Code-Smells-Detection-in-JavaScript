37             super(len, description, NullPadder.INSTANCE, AsciiInterpreter.INSTANCE, BcdPrefixer.LLL);
38             checkLength(len, 999);
39         }
40     
41         public void setLength(int len)
42         {
43             checkLength(len, 999);
44             super.setLength(len);
45         }
46     }
