13     package edu.cmu.sphinx.linguist.acoustic.tiedstate;
14     
15     import edu.cmu.sphinx.frontend.Data;
16     import edu.cmu.sphinx.frontend.DoubleData;
17     import edu.cmu.sphinx.frontend.FloatData;
18     import edu.cmu.sphinx.util.LogMath;
19     
20     /**
21      * Represents a concrete implementation of a simple {@link Senone senone}. A simple senone is a set of probability
22      * density functions implemented as a Gaussian mixture.
23      * <p/>
24      * All scores and weights are maintained in LogMath log base.
25      */
26     
27     public class GaussianMixture extends ScoreCachingSenone {
28     
29         // these data element in a senone may be shared with other senones
30         // and therefore should not be written to.
31         private float[] logMixtureWeights;
32         private MixtureComponent[] mixtureComponents;
33         private long id;
34     
35         private LogMath logMath;
36     
37     
38         /**
39          * Creates a new senone from the given components.
40          *
41          * @param logMath           the log math
42          * @param logMixtureWeights the mixture weights for this senone in LogMath log base
43          * @param mixtureComponents the mixture components for this senone
44          */
45         public GaussianMixture(LogMath logMath, float[] logMixtureWeights,
46                                MixtureComponent[] mixtureComponents, long id) {
47     
48             assert mixtureComponents.length == logMixtureWeights.length;
49     
50             this.logMath = logMath;
51             this.mixtureComponents = mixtureComponents;
52             this.logMixtureWeights = logMixtureWeights;
53             this.id = id;
54         }
55     
56     
57         /**
58          * Dumps this senone.
59          *
60          * @param msg annotation message
61          */
62         @Override
63         public void dump(String msg) {
64             System.out.println(msg + " GaussianMixture: ID " + getID());
65         }
66     
67         /**
68          * Determines if two objects are equal
69          *
70          * @param o the object to compare to this.
71          * @return true if the objects are equal
72          */
73         @Override
74         public boolean equals(Object o) {
75             if (!(o instanceof Senone)) {
76                 return false;
77             }
78             Senone other = (Senone) o;
79             return this.getID() == other.getID();
80         }
81     
82     
83         /**
84          * Returns the hashcode for this object
85          *
86          * @return the hashcode
87          */
88         @Override
89         public int hashCode() {
90             long id = getID();
91             int high = (int) ((id >> 32));
92             int low = (int) (id);
93             return high + low;
94         }
95     
96     
97         /**
98          * Gets the ID for this senone
99          *
100          * @return the senone id
101          */
102         @Override
103         public long getID() {
104             return id;
105         }
106     
107     
108         /**
109          * Retrieves a string form of this object
110          *
111          * @return the string representation of this object
112          */
113         @Override
114         public String toString() {
115             return "senone id: " + getID();
116         }
117     
118         @Override
119         public float calculateScore(Data feature) {
120             if (feature instanceof DoubleData)
121                 System.err.println("DoubleData conversion required on mixture level!");
122     
123             float[] featureVector = FloatData.toFloatData(feature).getValues();
124     
125             float logTotal = LogMath.getLogZero();
126             for (int i = 0; i < mixtureComponents.length; i++) {
127                 // In linear form, this would be:
128                 //
129                 // Total += Mixture[i].score * MixtureWeight[i]
130                 logTotal = logMath.addAsLinear(logTotal,
131                         mixtureComponents[i].getScore(featureVector) + logMixtureWeights[i]);
132             }
133     
134             return logTotal;
135         }
136     
137     
138         /**
139          * Calculates the scores for each component in the senone.
140          *
141          * @param feature the feature to score
142          * @return the LogMath log scores for the feature, one for each component
143          */
144         @Override
145         public float[] calculateComponentScore(Data feature) {
146             if (feature instanceof DoubleData)
147                 System.err.println("DoubleData conversion required on mixture level!");
148     
149             float[] featureVector = FloatData.toFloatData(feature).getValues();
150     
151             float[] logComponentScore = new float[mixtureComponents.length];
152             for (int i = 0; i < mixtureComponents.length; i++) {
153                 // In linear form, this would be:
154                 //
155                 // Total += Mixture[i].score * MixtureWeight[i]
156                 logComponentScore[i] = mixtureComponents[i].getScore(featureVector) + logMixtureWeights[i];
157             }
158     
159             return logComponentScore;
160         }
161     
162     
163         /**
164          * Returns the mixture components associated with this Gaussian
165          *
166          * @return the array of mixture components
167          */
168         public MixtureComponent[] getMixtureComponents() {
169             return mixtureComponents;
170         }
171     
172     
173         /** @return the dimension of the modeled feature space */
174         public int dimension() {
175             return mixtureComponents[0].getMean().length;
176         }
177     
178     
179         /** @return the number of component densities of this <code>GaussianMixture</code>. */
180         public int numComponents() {
181             return mixtureComponents.length;
182         }
183     
184     
185         /** @return the (linearly scaled) mixture weights of the component densities */
186         public float[] getComponentWeights() {
187             float[] mixWeights = new float[getMixtureComponents().length];
188             for (int i = 0; i < mixWeights.length; i++)
189                 mixWeights[i] = (float) logMath.logToLinear(logMixtureWeights[i]);
190     
191             return mixWeights;
192         }
193     
194     
195         public LogMath getLogMath() {
196             return logMath;
197         }
198     
199     
200     
201         /** @return the (log-scaled) mixture weight of the component density <code>index</code> */
202         public float getLogComponentWeight(int index) {
203             return logMixtureWeights[index];
204         }
205     
206     }
