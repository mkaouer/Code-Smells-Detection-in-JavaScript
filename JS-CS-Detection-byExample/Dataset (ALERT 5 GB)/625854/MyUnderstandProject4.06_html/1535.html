67         /**
68          * Displays a finalized result.
69          * 
70          * @param result
71          *                A result which was accepted by {@link #resultUpdate}.
72          * @returns the spoken text
73          */
74         private String displayResult(Result result) {
75             StringBuffer spoken = new StringBuffer();
76     
77             // Get the array of tokens that the recognizer thinks
78             // the user most likely spoke.
79             ResultToken[] best = result.getBestTokens();
80     
81             // Concatenate all of the result's tokens.
82             for (int i = 0; i < best.length; i++) {
83                 if (i > 0)
84                     spoken.append(' ');
85                 spoken.append(best[i].getSpokenText());
86             }
87     
88             // Append the spoken tokens to the user interface.
89             return spoken.toString();
90         }
