27     /**
28      * Classes that implement this interface create grammars. A grammar is represented internally as a graph of {@link
29      * GrammarNode GrammarNodes} linked together by {@link GrammarArc GrammarArcs}. Calling {@link #getInitialNode()
30      * getInitialNode} will return the first node of the grammar graph. To traverse the grammar graph, one should call
31      * GrammarNode.getSuccessors, which will return an array of GrammarArcs, from which you can reach the neighboring
32      * GrammarNodes.
33      * <p/>
34      * Note that all grammar probabilities are maintained in LogMath log domain.
35      */
36

141         /**
142          * Perform the standard set of grammar post processing. This can include inserting silence nodes and optimizing out
143          * empty nodes
144          */
145         protected void postProcessGrammar() {
146             if (!postProcessed) {
147                 if (addFillerWords) {
148                     addFillerWords();
149                 } else if (addSilenceWords) {
150                     addSilenceWords();
151                 }
152     
153                 if (optimizeGrammar) {
154                     optimizeGrammar();
155                 }
156                 dumpStatistics();
157                 if (showGrammar) {
158                     dumpGrammar("grammar.gdl");
159                     dumpRandomSentences("sentences.txt", 100);
160                     logger.info("Total number of nodes " + grammarNodes.size());
161                 }
162                 postProcessed = true;
163             }
164         }

167         /** Dumps statistics for this grammar */
168         public void dumpStatistics() {
169             if (logger.isLoggable(Level.INFO)) {
170                 int successorCount = 0;
171                 logger.info("Num nodes : " + getNumNodes());
172                 for (GrammarNode grammarNode : grammarNodes)
173                     successorCount += grammarNode.getSuccessors().length;
174     
175                 logger.info("Num arcs  : " + successorCount);
176                 logger.info("Avg arcs  : "
177                         + ((float) successorCount / getNumNodes()));
178             }
179         }

230         /**
231          * Returns a random sentence that fits this grammar
232          *
233          * @return a random sentence that fits this grammar
234          */
235         public String getRandomSentence() {
236             StringBuilder sb = new StringBuilder();
237             GrammarNode node = getInitialNode();
238             while (!node.isFinalNode()) {
239                 if (!node.isEmpty()) {
240                     Word word = node.getWord();
241                     if (!word.isFiller())
242                         sb.append(word.getSpelling()).append(' ');
243                 }
244                 node = selectRandomSuccessor(node);
245             }
246             return sb.toString().trim();
247         }

250         /**
251          * Given a node, select a random successor from the set of possible successor nodes
252          *
253          * @param node the node
254          * @return a random successor node.
255          */
256         private GrammarNode selectRandomSuccessor(GrammarNode node) {
257             GrammarArc[] arcs = node.getSuccessors();
258     
259             // select a transition arc with respect to the arc-probabilities (which are log and we don't have a logMath here
260             // which makes the implementation a little bit messy
261             if (arcs.length > 1) {
262                 double[] linWeights = new double[arcs.length];
263                 double linWeightsSum = 0;
264     
265                 final double EPS = 1E-10;
266     
267                 for (int i = 0; i < linWeights.length; i++) {
268                     linWeights[i] = (arcs[0].getProbability() + EPS) / (arcs[i].getProbability() + EPS);
269                     linWeightsSum += linWeights[i];
270                 }
271     
272                 for (int i = 0; i < linWeights.length; i++) {
273                     linWeights[i] /= linWeightsSum;
274                 }
275     
276     
277                 double selIndex = randomizer.nextDouble();
278                 int index = 0;
279                 for (int i = 0; selIndex > EPS; i++) {
280                     index = i;
281                     selIndex -= linWeights[i];
282                 }
283     
284                 return arcs[index].getGrammarNode();
285     
286             } else {
287                 return arcs[0].getGrammarNode();
288             }
289         }

361         /**
362          * Returns a new GrammarNode with the given set of alternatives.
363          *
364          * @param identity the id for this node
365          * @param alts     the set of alternative word lists for this GrammarNode
366          */
367         protected GrammarNode createGrammarNode(int identity, String[][] alts) {
368             GrammarNode node;
369             Word[][] alternatives = new Word[alts.length][];
370             for (int i = 0; i < alternatives.length; i++) {
371                 alternatives[i] = new Word[alts[i].length];
372                 for (int j = 0; j < alts[i].length; j++) {
373                     Word word = getDictionary().getWord(alts[i][j]);
374                     // Pronunciation[] pronunciation =
375                     // word.getPronunciations(null);
376                     if (word == null) {
377                         alternatives = EMPTY_ALTERNATIVE;
378                         break;
379                     } else {
380                         alternatives[i][j] = word;
381                     }
382                 }
383             }
384             node = new GrammarNode(identity, alternatives);
385             add(node);
386     
387             return node;
388         }

415         /**
416          * Returns a new GrammarNode with the given single word. If the word is not in the dictionary, an empty node is
417          * created
418          *
419          * @param identity the id for this node
420          * @param word     the word for this grammar node
421          */
422         protected GrammarNode createGrammarNode(int identity, String word) {
423             GrammarNode node;
424             Word[][] alternatives = EMPTY_ALTERNATIVE;
425             Word wordObject = getDictionary().getWord(word);
426             // Pronunciation[] pronunciation = wordObject.getPronunciations(null);
427             if (wordObject != null) {
428                 alternatives = new Word[1][];
429                 alternatives[0] = new Word[1];
430                 alternatives[0][0] = wordObject;
431                 node = new GrammarNode(identity, alternatives);
432                 add(node);
433             } else {
434                 node = createGrammarNode(identity, false);
435                 logger.warning("Can't find pronunciation for " + word);
436             }
437             return node;
438         }

456         /**
457          * Adds the given grammar node to the set of nodes for this grammar
458          *
459          * @param node the grammar node
460          * @throws Error
461          */
462         private void add(GrammarNode node) throws Error {
463             if (node.getID() > maxIdentity) {
464                 maxIdentity = node.getID();
465             }
466     
467             // check to see if there is already a node with the given ID.
468             if (idCheck) {
469                 for (GrammarNode grammarNode : grammarNodes) {
470                     if (grammarNode.getID() == node.getID()) {
471     
472                         throw new Error("DUP ID " + grammarNode + " and " + node);
473                     }
474                 }
475             }
476     
477             grammarNodes.add(node);
478     
479         }

493         /** Adds an optional silence word after every non-filler word in the grammar */
494         private void addSilenceWords() {
495             HashSet<GrammarNode> nodes = new HashSet<GrammarNode>(getGrammarNodes());
496             for (GrammarNode g : nodes) {
497                 if (!g.isEmpty() && !g.getWord().isFiller()) {
498                     GrammarNode silNode = createGrammarNode(maxIdentity + 1,
499                             dictionary.getSilenceWord().getSpelling());
500     
501                     GrammarNode branchNode = g.splitNode(maxIdentity + 1);
502                     add(branchNode);
503     
504                     g.add(silNode, 0.00f);
505                     silNode.add(branchNode, 0.0f);
506                     silNode.add(silNode, 0.0f);
507                 }
508             }
509         }

512         /** Adds an optional filler word loop after every non-filler word in the grammar */
513         private void addFillerWords() {
514             Set<GrammarNode> nodes = new HashSet<GrammarNode>(getGrammarNodes());
515     
516             Word[] fillers = getInterWordFillers();
517     
518             if (fillers.length == 0) {
519                 return;
520             }
521     
522             for (GrammarNode wordNode : nodes) {
523                 if (!wordNode.isEmpty() && !wordNode.getWord().isFiller()) {
524                     GrammarNode wordExitNode = wordNode.splitNode(maxIdentity + 1);
525                     add(wordExitNode);
526                     GrammarNode fillerStart = createGrammarNode(false);
527                     GrammarNode fillerEnd = createGrammarNode(false);
528                     fillerEnd.add(fillerStart, 0.0f);
529                     fillerEnd.add(wordExitNode, 0.0f);
530                     wordNode.add(fillerStart, 0.0f);
531     
532                     for (Word filler : fillers) {
533                         GrammarNode fnode = createGrammarNode(maxIdentity + 1, filler.getSpelling());
534                         fillerStart.add(fnode, 0.0f);
535                         fnode.add(fillerEnd, 0.0f);
536                     }
537                 }
538             }
539         }

542         /**
543          * Gets the set of fillers after filtering out fillers that don't go between words.
544          *
545          * @return the set of inter-word fillers
546          */
547         private Word[] getInterWordFillers() {
548             List<Word> fillerList = new ArrayList<Word>();
549             Word[] fillers = dictionary.getFillerWords();
550     
551             for (Word fillerWord : fillers) {
552                 if (fillerWord != dictionary.getSentenceStartWord()
553                         && fillerWord != dictionary.getSentenceEndWord()) {
554                     fillerList.add(fillerWord);
555                 }
556             }
557             return fillerList.toArray(new Word[fillerList.size()]);
558         }
