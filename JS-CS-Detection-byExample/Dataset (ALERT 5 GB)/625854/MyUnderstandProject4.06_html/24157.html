32     /**
33      * Loads a tied-state acoustic model generated by the Sphinx-3 trainer.
34      * <p>
35      * The acoustic model is stored as a directory specified by a URL.
36      * The dictionary and language model files are not required to be in the
37      * package. You can specify their locations separately. 
38      * </p>
39      * <p>Configuration file should set mandatory property of component:
40      * <ul>
41      * <li><b>location</b> - this specifies the directory where the actual model
42      * data files are. You can use <b>resource:</b> prefix to refer to files packed
43      * inside jar or any other URI scheme.
44      * </ul>
45      * The actual model data files are named "mdef", "means", "variances",
46      * "transition_matrices", "mixture_weights".
47      * </p>
48      * <p>
49      * If model has some layout that is different default generated by SphinxTrain, you may 
50      * specify additional properties like <b>dataLocation</b> to set the path to the binary 
51      * files, <b>mdef</b> to set the path to the model definition file.
52      * </p>
53      * <p>
54      * As an example, lets look at the Wall Street Journal acoustic model JAR file,
55      * which is located at the <code>sphinx4/lib</code> directory. If you run
56      * <code>"jar tf lib/WSJ_8gau_13dCep_16k_40mel_130Hz_6800Hz.jar"</code>, you
57      * will find that its internal structure looks roughly like:
58      * <p/>
59      * 
60      * <pre>
61      * WSJ_8gau_13dCep_16k_40mel_130Hz_6800Hz.jar
62      *   |
63      *   +- license.terms
64      *   +- cd_continuous_8gau
65      *   |   |
66      *   |   +- means
67      *   |   +- variances
68      *   |   +- mixture_weights
69      *   |   +- transition_matrices
70      *   |
71      *   +- dict
72      *   |   |
73      *   |   +- alpha.dict
74      *   |   +- cmudict.0.6d
75      *   |   +- digits.dict
76      *   |   +- fillerdict
77      *   |
78      *   +- etc
79      *       |
80      *       +- WSJ_clean_13dCep_16k_40mel_130Hz_6800Hz.4000.mdef
81      *       +- WSJ_clean_13dCep_16k_40mel_130Hz_6800Hz.ci.mdef
82      *       +- variables.def
83      * </pre>
84      * <p/>
85      * <p>
86      *  So the configuration to load this model should look like
87      *  <pre>
88      *  &lt;component name="wsjLoader" type="edu.cmu.sphinx.linguist.acoustic.tiedstate.Sphinx3Loader"&gt;
89      *     &lt;property name="logMath" value="logMath"/&gt;
90      *        &lt;property name="unitManager" value="unitManager"/&gt;
91      *        &lt;property name="location" value="resource:/WSJ_8gau_13dCep_16k_40mel_130Hz_6800Hz"/&gt;
92      *        &lt;property name="modelDefinition" value="etc/WSJ_clean_13dCep_16k_40mel_130Hz_6800Hz.4000.mdef"/&gt;
93      *        &lt;property name="dataLocation" value="cd_continuous_8gau/"/&gt;
94      *    &lt;/component&gt;
95      *  </pre>
96      * </p>
97      * <p>
98      * For more details on using SphinxTrain models in sphinx4, see <a href="../../../../../../../doc/UsingSphinxTrainModels.html">documentation</a>
99      * </p>

276                 transformMatrix = null;

329             logger.config("    modelName: " + this.model);

332             meansPool = loadDensityFile(dataLocation + "means", -Float.MAX_VALUE);

344             // load the HMM modelDef file

346             if (modelStream == null) {
347                 throw new IOException("can't find modelDef " + this.model);
348             }
349             loadHMMPool(useCDUnits, modelStream, this.model);

359         /**
360          * Creates the senone pool from the rest of the pools.
361          * 
362          * @param distFloor
363          *            the lowest allowed score
364          * @param varianceFloor
365          *            the lowest allowed variance
366          * @return the senone pool
367          */
368         protected Pool<Senone> createSenonePool(float distFloor, float varianceFloor) {
369             Pool<Senone> pool = new Pool<Senone>("senones");
370             int numMixtureWeights = mixtureWeightsPool.size();
371     
372             int numMeans = meansPool.size();
373             int numVariances = variancePool.size();
374             int numGaussiansPerSenone = mixtureWeightsPool.getFeature(
375                     NUM_GAUSSIANS_PER_STATE, 0);
376             int numSenones = mixtureWeightsPool.getFeature(NUM_SENONES, 0);
377             int numStreams = mixtureWeightsPool.getFeature(NUM_STREAMS, 0);
378             int whichGaussian = 0;
379     
380             logger.fine("Senones " + numSenones);
381             logger.fine("Gaussians Per Senone " + numGaussiansPerSenone);
382             logger.fine("MixtureWeights " + numMixtureWeights);
383             logger.fine("Means " + numMeans);
384             logger.fine("Variances " + numVariances);
385     
386             assert numGaussiansPerSenone > 0;
387             assert numMixtureWeights == numSenones;
388             assert numVariances == numSenones * numGaussiansPerSenone;
389             assert numMeans == numSenones * numGaussiansPerSenone;
390             
391             float [][] meansTransformationMatrix = meanTransformationMatrixPool == null ?
392                     null : meanTransformationMatrixPool.get(0);
393             float [] meansTransformationVector = meanTransformationVectorPool == null ?
394                     null : meanTransformationVectorPool.get(0);
395             float [][] varianceTransformationMatrix = varianceTransformationMatrixPool == null ?
396                     null : varianceTransformationMatrixPool.get(0);
397             float [] varianceTransformationVector = varianceTransformationVectorPool == null ?
398                     null : varianceTransformationVectorPool.get(0);
399             
400             for (int i = 0; i < numSenones; i++) {
401                 MixtureComponent[] mixtureComponents = new MixtureComponent[numGaussiansPerSenone * numStreams];
402                 for (int j = 0; j < numGaussiansPerSenone; j++) {
403                     mixtureComponents[j] = new MixtureComponent(logMath, 
404                             meansPool.get(whichGaussian), 
405                             meansTransformationMatrix, 
406                             meansTransformationVector,
407                             variancePool.get(whichGaussian),
408                             varianceTransformationMatrix,
409                             varianceTransformationVector, 
410                             distFloor,
411                             varianceFloor);
412     
413                     whichGaussian++;
414                 }
415     
416                 Senone senone = new GaussianMixture(logMath, mixtureWeightsPool
417                         .get(i), mixtureComponents, i);
418     
419                 pool.put(i, senone);
420             }
421             return pool;
422         }

424         /**
425          * Loads the sphinx3 density file, a set of density arrays are created and
426          * placed in the given pool.
427          * 
428          * @param path
429          *            the name of the data
430          * @param floor
431          *            the minimum density allowed
432          * @return a pool of loaded densities
433          * @throws FileNotFoundException
434          *             if a file cannot be found
435          * @throws IOException
436          *             if an error occurs while loading the data
437          */
438         protected Pool<float[]> loadDensityFile(String path, float floor)
439                 throws IOException, URISyntaxException {
440             Properties props = new Properties();
441             int blockSize = 0;
442     
443             DataInputStream dis = readS3BinaryHeader(path, props);
444     
445             String version = props.getProperty("version");
446     
447             if (version == null || !version.equals(DENSITY_FILE_VERSION)) {
448                 throw new IOException("Unsupported version in " + path);
449             }
450     
451             String checksum = props.getProperty("chksum0");
452             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
453             resetChecksum();
454     
455             int numStates = readInt(dis);
456             int numStreams = readInt(dis);
457             int numGaussiansPerState = readInt(dis);
458     
459             int[] vectorLength = new int[numStreams];
460             for (int i = 0; i < numStreams; i++) {
461                 vectorLength[i] = readInt(dis);
462             }
463     
464             int rawLength = readInt(dis);
465     
466             logger.fine("Number of states " + numStates);
467             logger.fine("Number of streams " + numStreams);
468             logger.fine("Number of gaussians per state " + numGaussiansPerState);
469             logger.fine("Vector length " + vectorLength.length);
470             logger.fine("Raw length " + rawLength);
471     
472             for (int i = 0; i < numStreams; i++) {
473                 blockSize += vectorLength[i];
474             }
475     
476             assert rawLength == numGaussiansPerState * blockSize * numStates;
477     
478             Pool<float[]> pool = new Pool<float[]>(path);
479             pool.setFeature(NUM_SENONES, numStates);
480             pool.setFeature(NUM_STREAMS, numStreams);
481             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
482     
483             for (int i = 0; i < numStates; i++) {
484                 for (int j = 0; j < numStreams; j++) {
485                     for (int k = 0; k < numGaussiansPerState; k++) {
486                         float[] density = readFloatArray(dis, vectorLength[j]);
487                         Utilities.floorData(density, floor);
488                         pool.put(i * numStreams * numGaussiansPerState + j * numGaussiansPerState + k, density);
489                     }
490                 }
491             }
492     
493             validateChecksum(dis, doCheckSum);
494             
495             dis.close();
496             return pool;
497         }

499         /**
500          * Reads the S3 binary header from the given location + path. Adds header
501          * information to the given set of properties.
502          * 
503          * @param path
504          *            the name of the file
505          * @param props
506          *            the properties
507          * @return the input stream positioned after the header
508          * @throws IOException
509          *             on error
510          */
511         protected DataInputStream readS3BinaryHeader(String path, Properties props)
512                 throws IOException, URISyntaxException {
513     
514             InputStream inputStream = getDataStream(path);
515     
516             if (inputStream == null) {
517                 throw new IOException("Can't open " + path);
518             }
519             DataInputStream dis = new DataInputStream(new BufferedInputStream(
520                     inputStream));
521             String id = readWord(dis);
522             if (!id.equals("s3")) {
523                 throw new IOException("Not proper s3 binary file " + path);
524             }
525             String name;
526             while ((name = readWord(dis)) != null) {
527                 if (!name.equals("endhdr")) {
528                     String value = readWord(dis);
529                     props.setProperty(name, value);
530                 } else {
531                     break;
532                 }
533             }
534             int byteOrderMagic = dis.readInt();
535             if (byteOrderMagic == BYTE_ORDER_MAGIC) {
536                 logger.fine("Not swapping " + path);
537                 swap = false;
538             } else if (Utilities.swapInteger(byteOrderMagic) == BYTE_ORDER_MAGIC) {
539                 logger.fine("Swapping  " + path);
540                 swap = true;
541             } else {
542                 throw new IOException("Corrupted S3 file " + path);
543             }
544             return dis;
545         }

547         /**
548          * Reads the next word (text separated by whitespace) from the given stream.
549          * 
550          * @param dis
551          *            the input stream
552          * @return the next word
553          * @throws IOException
554          *             on error
555          */
556         String readWord(DataInputStream dis) throws IOException {
557             StringBuilder sb = new StringBuilder();
558             char c;
559             // skip leading whitespace
560             do {
561                 c = readChar(dis);
562             } while (Character.isWhitespace(c));
563             // read the word
564             do {
565                 sb.append(c);
566                 c = readChar(dis);
567             } while (!Character.isWhitespace(c));
568             return sb.toString();
569         }

584         /* Stores checksum during loading */
585         private long calculatedCheckSum = 0;
586         
587         /**
588          * Resets the checksum before loading a new chunk of data
589          */
590         private void resetChecksum() {
591             calculatedCheckSum = 0;

594         /**
595          *  Validates checksum in the stream
596          *  
597          *  @param dis input stream
598          *  @param doCheckSum validates
599          *  @throws IOException on error
600          **/
601         private void validateChecksum(DataInputStream dis, boolean doCheckSum) throws IOException {
602             if (!doCheckSum)
603                 return;
604             int oldCheckSum = (int)calculatedCheckSum;
605             int checkSum = readInt(dis);
606             if (checkSum != oldCheckSum) {
607                 throw new IOException("Invalid checksum "
608                         + Long.toHexString(calculatedCheckSum) + " must be "
609                         + Integer.toHexString(checkSum));
610             }
611         }

613         /**
614          * Read an integer from the input stream, byte-swapping as necessary.
615          * 
616          * @param dis
617          *            the input stream
618          * @return an integer value
619          * @throws IOException
620          *             on error
621          */
622         protected int readInt(DataInputStream dis) throws IOException {
623             int val;
624             if (swap) {
625                 val = Utilities.readLittleEndianInt(dis);
626             } else {
627                 val = dis.readInt();
628             }
629             calculatedCheckSum = ((calculatedCheckSum << 20 | calculatedCheckSum >> 12) + val) & 0xFFFFFFFFL;
630             return val;
631         }

633         /**
634          * Read a float from the input stream, byte-swapping as necessary.
635          * 
636          * @param dis
637          *            the input stream
638          * @return a floating pint value
639          * @throws IOException
640          *             on error
641          */
642         protected float readFloat(DataInputStream dis) throws IOException {
643             int val;
644             if (swap) {
645                 val = Utilities.readLittleEndianInt(dis);
646             } else {
647                 val = dis.readInt();
648             }
649             calculatedCheckSum = ((calculatedCheckSum << 20 | calculatedCheckSum >> 12) + val) & 0xFFFFFFFFL;
650             return Float.intBitsToFloat(val);
651         }

683          * @param path
684          *            the path to a density file
685          * @throws FileNotFoundException
686          *             if a file cannot be found
687          * @throws IOException
688          *             if an error occurs while loading the data
689          */
690         protected void loadHMMPool(boolean useCDUnits, InputStream inputStream,
691                 String path) throws IOException {
692             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream,
693                     '#', false);
694     
695             logger.fine("Loading HMM file from: " + path);
696     
697             est.expectString(MODEL_VERSION);
698     
699             int numBase = est.getInt("numBase");
700             est.expectString("n_base");
701     
702             int numTri = est.getInt("numTri");
703             est.expectString("n_tri");
704     
705             int numStateMap = est.getInt("numStateMap");
706             est.expectString("n_state_map");
707     
708             int numTiedState = est.getInt("numTiedState");
709             est.expectString("n_tied_state");
710     
711             int numContextIndependentTiedState = est
712                     .getInt("numContextIndependentTiedState");
713             est.expectString("n_tied_ci_state");
714     
715             int numTiedTransitionMatrices = est.getInt("numTiedTransitionMatrices");
716             est.expectString("n_tied_tmat");
717     
718             int numStatePerHMM = numStateMap / (numTri + numBase);
719     
720             assert numTiedState == mixtureWeightsPool.getFeature(NUM_SENONES, 0);
721             assert numTiedTransitionMatrices == transitionsPool.size();
722     
723             // Load the base phones
724             for (int i = 0; i < numBase; i++) {
725                 String name = est.getString();
726                 String left = est.getString();
727                 String right = est.getString();
728                 String position = est.getString();
729                 String attribute = est.getString();
730                 int tmat = est.getInt("tmat");
731     
732                 int[] stid = new int[numStatePerHMM - 1];
733     
734                 for (int j = 0; j < numStatePerHMM - 1; j++) {
735                     stid[j] = est.getInt("j");
736                     assert stid[j] >= 0 && stid[j] < numContextIndependentTiedState;
737                 }
738                 est.expectString("N");
739     
740                 assert left.equals("-");
741                 assert right.equals("-");
742                 assert position.equals("-");
743                 assert tmat < numTiedTransitionMatrices;
744     
745                 Unit unit = unitManager.getUnit(name, attribute.equals(FILLER));
746                 contextIndependentUnits.put(unit.getName(), unit);
747     
748                 if (logger.isLoggable(Level.FINE)) {
749                     logger.fine("Loaded " + unit);
750                 }
751     
752                 // The first filler
753                 if (unit.isFiller() && unit.getName().equals(SILENCE_CIPHONE)) {
754                     unit = UnitManager.SILENCE;
755                 }
756     
757                 float[][] transitionMatrix = transitionsPool.get(tmat);
758                 SenoneSequence ss = getSenoneSequence(stid);
759     
760                 HMM hmm = new SenoneHMM(unit, ss, transitionMatrix, HMMPosition
761                         .lookup(position));
762                 hmmManager.put(hmm);
763             }
764     
765             if (hmmManager.get(HMMPosition.UNDEFINED, UnitManager.SILENCE) == null) {
766         	    throw new IOException("Could not find SIL unit in acoustic model");
767             }
768     
769             // Load the context dependent phones. If the useCDUnits
770             // property is false, the CD phones will not be created, but
771             // the values still need to be read in from the file.
772     
773             String lastUnitName = "";
774             Unit lastUnit = null;
775             int[] lastStid = null;
776             SenoneSequence lastSenoneSequence = null;
777     
778             for (int i = 0; i < numTri; i++) {
779                 String name = est.getString();
780                 String left = est.getString();
781                 String right = est.getString();
782                 String position = est.getString();
783                 String attribute = est.getString();
784                 int tmat = est.getInt("tmat");
785     
786                 int[] stid = new int[numStatePerHMM - 1];
787     
788                 for (int j = 0; j < numStatePerHMM - 1; j++) {
789                     stid[j] = est.getInt("j");
790                     assert stid[j] >= numContextIndependentTiedState
791                             && stid[j] < numTiedState;
792                 }
793                 est.expectString("N");
794     
795                 assert !left.equals("-");
796                 assert !right.equals("-");
797                 assert !position.equals("-");
798                 assert attribute.equals("n/a");
799                 assert tmat < numTiedTransitionMatrices;
800     
801                 if (useCDUnits) {
802                     Unit unit;
803                     String unitName = (name + ' ' + left + ' ' + right);
804     
805                     if (unitName.equals(lastUnitName)) {
806                         unit = lastUnit;
807                     } else {
808                         Unit[] leftContext = new Unit[1];
809                         leftContext[0] = contextIndependentUnits.get(left);
810     
811                         Unit[] rightContext = new Unit[1];
812                         rightContext[0] = contextIndependentUnits.get(right);
813     
814                         Context context = LeftRightContext.get(leftContext,
815                                 rightContext);
816                         unit = unitManager.getUnit(name, false, context);
817                     }
818                     lastUnitName = unitName;
819                     lastUnit = unit;
820     
821                     if (logger.isLoggable(Level.FINE)) {
822                     	logger.fine("Loaded " + unit);
823                     }
824     
825                     float[][] transitionMatrix = transitionsPool.get(tmat);
826     
827                     SenoneSequence ss = lastSenoneSequence;
828                     if (ss == null || !sameSenoneSequence(stid, lastStid)) {
829                         ss = getSenoneSequence(stid);
830                     }
831                     lastSenoneSequence = ss;
832                     lastStid = stid;
833     
834                     HMM hmm = new SenoneHMM(unit, ss, transitionMatrix, HMMPosition
835                             .lookup(position));
836                     hmmManager.put(hmm);
837                 }
838             }
839     
840             est.close();
841         }

843         /**
844          * Returns true if the given senone sequence IDs are the same.
845          * 
846          * @return true if the given senone sequence IDs are the same, false
847          *         otherwise
848          */
849         protected boolean sameSenoneSequence(int[] ssid1, int[] ssid2) {
850             if (ssid1.length == ssid2.length) {
851                 for (int i = 0; i < ssid1.length; i++) {
852                     if (ssid1[i] != ssid2[i]) {
853                         return false;
854                     }
855                 }
856                 return true;
857             } else {
858                 return false;
859             }
860         }

864          *

866          *            is the array of senone state ids
867          * @return the senone sequence associated with the states
868          */
869         protected SenoneSequence getSenoneSequence(int[] stateid) {

877         /**
878          * Loads the mixture weights (Binary).
879          * 
880          * @param path
881          *            the path to the mixture weight file
882          * @param floor
883          *            the minimum mixture weight allowed
884          * @return a pool of mixture weights
885          * @throws FileNotFoundException
886          *             if a file cannot be found
887          * @throws IOException
888          *             if an error occurs while loading the data
889          */
890         protected Pool<float[]> loadMixtureWeights(String path, float floor)
891                 throws IOException, URISyntaxException {
892             logger.fine("Loading mixture weights from: " + path);
893     
894             Properties props = new Properties();
895     
896             DataInputStream dis = readS3BinaryHeader(path, props);
897     
898             String version = props.getProperty("version");
899     
900             if (version == null || !version.equals(MIXW_FILE_VERSION)) {
901                 throw new IOException("Unsupported version in " + path);
902             }
903     
904             String checksum = props.getProperty("chksum0");
905             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
906             resetChecksum();
907     
908             Pool<float[]> pool = new Pool<float[]>(path);
909     
910             int numStates = readInt(dis);
911             int numStreams = readInt(dis);
912             int numGaussiansPerState = readInt(dis);
913             int numValues = readInt(dis);
914     
915             logger.fine("Number of states " + numStates);
916             logger.fine("Number of streams " + numStreams);
917             logger.fine("Number of gaussians per state " + numGaussiansPerState);
918     
919             assert numValues == numStates * numStreams * numGaussiansPerState;
920     
921             pool.setFeature(NUM_SENONES, numStates);
922             pool.setFeature(NUM_STREAMS, numStreams);
923             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
924     
925             if (numStreams != 1) {
926                 for (int i = 0; i < numStates; i++) {
927                     float[] logMixtureWeight = new float[numGaussiansPerState
928                             * numStreams];
929                     for (int j = 0; j < numStreams; j++) {
930                         float[] logStreamMixtureWeight = readFloatArray(dis,
931                                 numGaussiansPerState);
932                         Utilities.normalize(logStreamMixtureWeight);
933                         Utilities.floorData(logStreamMixtureWeight, floor);
934                         logMath.linearToLog(logStreamMixtureWeight);
935                         System.arraycopy(logStreamMixtureWeight, 0,
936                                 logMixtureWeight, numGaussiansPerState * j,
937                                 numGaussiansPerState);
938                     }
939                     pool.put(i, logMixtureWeight);
940                 }
941             } else {
942                 for (int i = 0; i < numStates; i++) {
943                     float[] logMixtureWeight = readFloatArray(dis,
944                             numGaussiansPerState);
945                     Utilities.normalize(logMixtureWeight);
946                     Utilities.floorData(logMixtureWeight, floor);
947                     logMath.linearToLog(logMixtureWeight);
948                     pool.put(i, logMixtureWeight);
949     
950                 }
951     
952             }
953     
954             validateChecksum(dis, doCheckSum);
955     
956             dis.close();
957             return pool;
958         }

960         /**
961          * Loads the transition matrices (Binary).
962          * 
963          * @param path
964          *            the path to the transitions matrices
965          * @return a pool of transition matrices
966          * @throws FileNotFoundException
967          *             if a file cannot be found
968          * @throws IOException
969          *             if an error occurs while loading the data
970          */
971         protected Pool<float[][]> loadTransitionMatrices(String path)
972                 throws IOException, URISyntaxException {
973             logger.fine("Loading transition matrices from: " + path);
974     
975             Properties props = new Properties();
976             DataInputStream dis = readS3BinaryHeader(path, props);
977     
978             String version = props.getProperty("version");
979     
980             if (version == null || !version.equals(TMAT_FILE_VERSION)) {
981                 throw new IOException("Unsupported version in " + path);
982             }
983     
984             String checksum = props.getProperty("chksum0");
985             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
986             resetChecksum();
987     
988             Pool<float[][]> pool = new Pool<float[][]>(path);
989     
990             int numMatrices = readInt(dis);
991             int numRows = readInt(dis);
992             int numStates = readInt(dis);
993             int numValues = readInt(dis);
994     
995             assert numValues == numStates * numRows * numMatrices;
996     
997             for (int i = 0; i < numMatrices; i++) {
998                 float[][] tmat = new float[numStates][];
999                 // last row should be zeros
1000                 tmat[numStates - 1] = new float[numStates];
1001                 logMath.linearToLog(tmat[numStates - 1]);
1002     
1003                 for (int j = 0; j < numRows; j++) {
1004                     tmat[j] = readFloatArray(dis, numStates);
1005                     Utilities.nonZeroFloor(tmat[j], 0f);
1006                     Utilities.normalize(tmat[j]);
1007                     logMath.linearToLog(tmat[j]);
1008                 }
1009                 pool.put(i, tmat);
1010             }
1011     
1012             validateChecksum(dis, doCheckSum);
1013     
1014             dis.close();
1015             return pool;
1016         }

1018         /**
1019          * Loads the transform matrices (Binary).
1020          * 
1021          * @param path
1022          *            the path to the transform matrix
1023          * @return a transform matrix
1024          * @throws java.io.FileNotFoundException
1025          *             if a file cannot be found
1026          * @throws java.io.IOException
1027          *             if an error occurs while loading the data
1028          */
1029         protected float[][] loadTransformMatrix(String path) throws IOException {
1030             logger.fine("Loading transform matrix from: " + path);
1031     
1032             Properties props = new Properties();
1033     
1034             DataInputStream dis;
1035             try {
1036                 dis = readS3BinaryHeader(path, props);
1037             } catch (URISyntaxException e) {
1038                 throw new RuntimeException(e);
1039             } catch (IOException e) {
1040                 return null;
1041             }
1042     
1043     
1044             String version = props.getProperty("version");
1045     
1046             if (version == null || !version.equals(TRANSFORM_FILE_VERSION)) {
1047                 throw new IOException("Unsupported version in " + path);
1048             }
1049     
1050             String checksum = props.getProperty("chksum0");
1051             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
1052             resetChecksum();
1053     
1054             readInt(dis);
1055             int numRows = readInt(dis);
1056             int numValues = readInt(dis);
1057             int num = readInt(dis);
1058     
1059             assert num == numRows * numValues;
1060     
1061             float[][] result = new float[numRows][];
1062             for (int i = 0; i < numRows; i++) {
1063                 result[i] = readFloatArray(dis, numValues);
1064             }
1065     
1066             validateChecksum(dis, doCheckSum);
1067     
1068             dis.close();
1069             return result;
1070         }
