12     package edu.cmu.sphinx.linguist.util;
13     
14     import edu.cmu.sphinx.linguist.*;
15     
16     import java.util.*;
17     
18     /** A linguist processor that dumps out statistics about the search space */
19     public class LinguistStats extends LinguistProcessor {
20     
21         private final Map<Class<? extends SearchState>, Integer> stateCountByType = new HashMap<Class<? extends SearchState>, Integer>();
22     
23         public LinguistStats( Linguist linguist ) {
24             super(linguist);
25         }
26     
27         public LinguistStats() {
28             
29         }
30     
31         /** Dumps the statistics of the linguist */
32         @Override
33         public void run() {
34             Linguist linguist = getLinguist();
35             List<SearchState> queue = new LinkedList<SearchState>();
36             Set<SearchState> visitedStates = new HashSet<SearchState>();
37             int stateCount = 0;
38             queue.add(linguist.getSearchGraph().getInitialState());
39             while (!queue.isEmpty()) {
40                 SearchState state = queue.remove(0);
41                 if (!visitedStates.contains(state)) {
42                     stateCount++;
43                     incrementStateTypeCount(state);
44                     visitedStates.add(state);
45                     SearchStateArc[] arcs = state.getSuccessors();
46                     for (int i = arcs.length - 1; i >= 0; i--) {
47                         SearchState nextState = arcs[i].getState();
48                         queue.add(nextState);
49                     }
50                     // Dump out the unique word states for debugging
51                     // if (state instanceof WordSearchState) {
52                     //    System.out.println("WS: " + state);
53                     // }
54                     // if (false && state instanceof HMMSearchState) {
55                     //    System.out.println("HS: " + state);
56                     // }
57                 }
58             }
59             System.out.println("# ----------- linguist stats ------------ ");
60             System.out.println("# Total states: " + stateCount);
61             dumpStateTypeCounts();
62         }
63     
64     
65         /**
66          * Keeps track of state counts by class
67          *
68          * @param state the state to track
69          */
70         private void incrementStateTypeCount(SearchState state) {
71             Integer count = stateCountByType.get(state.getClass());
72             if (count == null) {
73                 count = 0;
74             }
75             count = count + 1;
76             stateCountByType.put(state.getClass(), count);
77         }
78     
79     
80         /** Dumps all of the class counts */
81         private void dumpStateTypeCounts() {
82             for (Map.Entry<Class<? extends SearchState>, Integer> entry : stateCountByType.entrySet())
83                 System.out.println("# " + entry.getKey() + ": " + entry.getValue());
84         }
85     }
