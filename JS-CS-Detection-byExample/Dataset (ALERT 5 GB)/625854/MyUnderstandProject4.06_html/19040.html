7     /**
8      * Recognition thread to run the recognizer in parallel.
9      *
10      * @author Dirk Schnelle-Walka
11      * @version $Revision: 1110 $
12      */
13     final class RecognitionThread
14             extends Thread {
15         /** The wrapper for the sphinx4 recognizer. */
16         private SphinxRecognizer recognizer;
17     
18         /**
19          * Creates a new object.
20          * @param rec The wrapper for the sphinx4 recognizer.
21          */
22         public RecognitionThread(final SphinxRecognizer rec) {
23             recognizer = rec;
24             setDaemon(true);
25             setName("SphinxRecognitionThread");
26         }
27     
28         /**
29          * {@inheritDoc}
30          */
31         public void run() {
32             final Recognizer rec = recognizer.getRecognizer();
33             final Microphone microphone = getMicrophone();
34             final boolean started;
35     
36             if (microphone != null) {
37                 microphone.clear();
38                 started = microphone.startRecording();
39             } else {
40                 started = true;
41             }
42     
43             if (started) {
44                 recognize(rec, microphone);
45             }
46             if (microphone != null) {
47                 // Stop recording from the microphone.
48                 while (microphone.isRecording()) {
49                     microphone.stopRecording();
50                 }
51             }
52         }
53     
54         /**
55          * Recognition loop. Continue recognizing until this thread is
56          * requested to stop.
57          * @param rec The recognizer to use.
58          * @param mic The microphone to use.
59          */
60         private void recognize(final Recognizer rec, final Microphone mic) {
61             while (hasMoreData(mic) && !isInterrupted()) {
62                 try {
63                     rec.recognize();
64                 } catch (IllegalArgumentException e) {
65                     e.printStackTrace();
66                 }
67             }
68         }
69     
70         /**
71          * Checks, if the microphone has more data to deliver.
72          * @param mic The microphone or <code>null</code> if the data processor
73          * is not a microphone.
74          * @return <code>true</code> if there is more data.
75          */
76         private boolean hasMoreData(final Microphone mic) {
77             if (mic == null) {
78                 return true;
79             }
80     
81             return mic.hasMoreData();
82         }
83     
84         /**
85          * Stop this recognition thread.
86          */
87         public void stopRecognition() {
88             final Microphone microphone = getMicrophone();
89             if (microphone != null) {
90                 microphone.stopRecording();
91             }
92     
93             interrupt();
94         }
95     
96         /**
97          * Retrieves the microphone.
98          * @return The microphone, <code>null</code> if the data processor is
99          * not a microphone.
100          * @since 0.5.5
101          */
102         private Microphone getMicrophone() {
103             final DataProcessor dataProcessor = recognizer.getDataProcessor();
104             if (dataProcessor instanceof Microphone) {
105                 return (Microphone) dataProcessor;
106             }
107     
108             return null;
109         }
