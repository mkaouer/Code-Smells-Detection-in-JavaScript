15     import java.io.ByteArrayInputStream;
16     import java.io.IOException;
17     import java.io.InputStream;
18     import java.io.OutputStream;
19     import java.io.PipedInputStream;
20     import java.io.PipedOutputStream;
21     
22     import javax.sound.sampled.AudioFormat;
23     import javax.sound.sampled.AudioInputStream;
24     import javax.sound.sampled.AudioSystem;
25     
26     import org.jvoicexml.jsapi2.BaseAudioManager;
27     
28     /**
29      * Utility to convert audio from a source format to another format.
30      * @author Renato Cassaca
31      * @author Dirk Schnelle-Walka
32      *
33      */
34     public class AudioFormatConverter {
35     
36         /** The source audio format. */
37         private final AudioFormat sourceFormat;
38         /** The target audio format. */
39         private final AudioFormat targetFormat;
40     
41         public AudioFormatConverter(BaseAudioManager manager,
42                 AudioFormat sourceFormat, AudioFormat targetFormat)
43             throws IOException {
44             this.sourceFormat = sourceFormat;
45             this.targetFormat = targetFormat;
46         }
47     
48         /**
49          * Convert the given audio data from the source format to the target format.
50          * @param in bytes to convert.
51          * @return converted bytes stream
52          * @throws IOException
53          *         error converting
54          */
55         public InputStream getConvertedAudio(final byte[] in)
56             throws IOException {
57             final ByteArrayInputStream stream = new ByteArrayInputStream(in);
58             return getConvertedStream(stream, sourceFormat, targetFormat);
59         }
60     
61         /**
62          * @todo Insure that this is robust...
63          *
64          * @param is InputStream
65          * @param sourceFormat AudioFormat
66          * @param targetFormat AudioFormat
67          * @return InputStream
68          */
69         public InputStream getConvertedStream(final InputStream is,
70                 final AudioFormat sourceFormat, final AudioFormat targetFormat) {
71             /** @todo Compare more precisely AudioFormat (not using AudioFormat.matches()) */
72             if (sourceFormat.matches(targetFormat)) {
73                 return is;
74             }
75     
76             // Describe source stream as an AudioFormat
77             final AudioInputStream sourceStream = new AudioInputStream(is,
78                     sourceFormat, AudioSystem.NOT_SPECIFIED);
79             return AudioSystem.getAudioInputStream(targetFormat, sourceStream);
80         }
81     
82         /**
83          * @todo Insure that this is robust...
84          *
85          * @param os OutputStream
86          * @param engineAudioFormat AudioFormat
87          * @param audioFormat AudioFormat
88          * @return OutputStream
89          */
90         public OutputStream getConvertedStream(final OutputStream os,
91                                                 AudioFormat engineAudioFormat,
92                                                 AudioFormat targetFormat) {
93             /** @todo Compare more preciselly AudioFormat (not using AudioFormat.matches()) */
94             if (engineAudioFormat.matches(targetFormat)) {
95                 return os;
96             }
97     
98             try {
99                 //Basic Conversion support
100                 PipedInputStream pis = new PipedInputStream(16000000);
101                 PipedOutputStream pos = new PipedOutputStream(pis);
102     
103                 //Describe source audio
104                 getConvertedStream(pis, engineAudioFormat, targetFormat);
105     
106                 return pos;
107             } catch (IOException ex) {
108                 ex.printStackTrace();
109             }
110     
111             /** @todo Should never reach this point */
112             return os;
113         }
114     }
