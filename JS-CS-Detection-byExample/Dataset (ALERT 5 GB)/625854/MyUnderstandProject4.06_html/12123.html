20         /**
21          * Dumps the config as a set of HTML tables
22          *
23          * @param path where to output the HTML
24          * @throws java.io.IOException if an error occurs
25          */
26         public static void showConfigAsHTML(ConfigurationManager ConfigurationManager, String path) throws IOException {
27             PrintStream out = new PrintStream(new FileOutputStream(path));
28             dumpHeader(out);
29             for (String componentName : ConfigurationManager.getInstanceNames(Configurable.class)) {
30                 dumpComponentAsHTML(out, componentName, ConfigurationManager.getPropertySheet(componentName));
31             }
32             dumpFooter(out);
33             out.close();
34         }

38          * Dumps the footer for HTML output

40          * @param out the output stream
41          */
42         public static void dumpFooter(PrintStream out) {

44             out.println("</html>");
45         }

49          * Dumps the header for HTML output
50          *
51          * @param out the output stream

61         /**
62          * Dumps the given component as HTML to the given stream
63          *
64          * @param out  where to dump the HTML
65          * @param name the name of the component to dump
66          */
67         public static void dumpComponentAsHTML(PrintStream out, String name, PropertySheet properties) {
68             out.println("<table border=1>");
69             //        out.println("<table border=1 width=\"%80\">");
70             out.print("    <tr><th bgcolor=\"#CCCCFF\" colspan=2>");
71             //       out.print("<a href="")
72             out.print(name);
73             out.print("</a>");
74             out.println("</td></tr>");
75     
76             out.println("    <tr><th bgcolor=\"#CCCCFF\">Property</th><th bgcolor=\"#CCCCFF\"> Value</th></tr>");
77             Collection<String> propertyNames = properties.getRegisteredProperties();
78     
79             for (String propertyName : propertyNames) {
80                 out.print("    <tr><th align=\"leftt\">" + propertyName + "</th>");
81                 Object obj;
82                 obj = properties.getRaw(propertyName);
83                 if (obj instanceof String) {
84                     out.println("<td>" + obj + "</td></tr>");
85                 } else if (obj instanceof List<?>) {
86                     List<?> l = (List<?>) obj;
87                     out.println("    <td><ul>");
88                     for (Object listElement : l) {
89                         out.println("        <li>" + listElement + "</li>");
90                     }
91                     out.println("    </ul></td>");
92                 } else {
93                     out.println("<td>DEFAULT</td></tr>");
94                 }
95             }
96             out.println("</table><br>");
97         }
