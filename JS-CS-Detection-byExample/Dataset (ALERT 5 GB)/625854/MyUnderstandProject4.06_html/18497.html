13     package edu.cmu.sphinx.tools.audio;
14     
15     import javax.sound.sampled.AudioFormat;
16     import javax.sound.sampled.AudioInputStream;
17     import javax.swing.event.ChangeEvent;
18     import javax.swing.event.ChangeListener;
19     import java.io.IOException;
20     import java.util.ArrayList;
21     import java.util.List;
22     
23     /** Represents a 16bit, SIGNED_PCM, big endian audio clip with a sample rate specified by AudioFormat. */
24     public class AudioData {
25     
26         protected AudioFormat format;
27         protected short[] shorts;
28         protected final List<ChangeListener> listeners = new ArrayList<ChangeListener>();
29         protected int selectionStart = -1;
30         protected int selectionEnd = -1;
31     
32     
33         /** No-arg constructor.  Creates an empty clip at 16kHz sample rate. */
34         public AudioData() {
35             this.format = new AudioFormat(16000f,
36                     16,    // sample size in bits
37                     1,     // mono
38                     true,  // signed
39                     true); // big endian
40             shorts = new short[0];
41         }
42     
43     
44         /**
45          * Creates a new AudioData with the given data and sample rate. Expects the data to be 16bit, big endian,
46          * SIGNED_PCM.
47          *
48          * @param data       the audio samples; one sample per element in the array
49          * @param sampleRate the sample rate in Hz
50          */
51         public AudioData(short[] data, float sampleRate) {
52             this.shorts = data;
53             this.format = new AudioFormat(sampleRate,
54                     16,    // sample size in bits
55                     1,     // mono
56                     true,  // signed
57                     true); // big endian
58         }
59     
60     
61         /**
62          * Creates a new AudioData from the given AudioInputStream, converting the data to 16bit, big endian, SIGNED_PCM if
63          * needed.
64          *
65          * @param ais the AudioInputStream
66          * @throws IOException if problems happen when reading from ais
67          */
68         public AudioData(AudioInputStream ais) throws IOException {
69             this.shorts = Utils.toSignedPCM(ais);
70             this.format = new AudioFormat(ais.getFormat().getSampleRate(),
71                     16,    // sample size in bits
72                     1,     // mono
73                     true,  // signed
74                     true); // big endian
75         }
76     
77     
78         /**
79          * Gets the SIGNED_PCM 16 bit big endian audio data.  NOTE:  this the actual array held by this object, so only use
80          * it as a reference (i.e., don't modify the contents).
81          *
82          * @return the SIGNED_PCM 16 bit big endian samples
83          */
84         public short[] getAudioData() {
85             return shorts;
86         }
87     
88     
89         /**
90          * Sets the audio data and notifies all ChangeListeners.
91          *
92          * @param data the new SIGNED_PCM 16 bit big endian samples
93          */
94         public void setAudioData(short[] data) {
95             this.shorts = data;
96             fireStateChanged();
97         }
98     
99     
100         /**
101          * Gets the audio format.
102          *
103          * @return the AudioFormat for the data managed by this object
104          */
105         public AudioFormat getAudioFormat() {
106             return format;
107         }
108     
109     
110         /**
111          * Add a ChangeListener.
112          *
113          * @param listener the listener to add
114          */
115         public void addChangeListener(ChangeListener listener) {
116             listeners.add(listener);
117         }
118     
119     
120         /**
121          * Remove a ChangeListener.
122          *
123          * @param listener the listener to remove
124          */
125         public void removeChangeListener(ChangeListener listener) {
126             listeners.remove(listener);
127         }
128     
129     
130         /** Notify all ChangeListeners of a change. */
131         protected void fireStateChanged() {
132             ChangeEvent event = new ChangeEvent(this);
133             for (ChangeListener listener : listeners)
134                 listener.stateChanged(event);
135         }
136     }
