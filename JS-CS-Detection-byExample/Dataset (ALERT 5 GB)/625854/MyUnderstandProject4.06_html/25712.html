34     /**
35      * Supports the JSAPI 2.0 <code>AudioManager</code> interface. Actual JSAPI
36      * implementations might want to extend or modify this implementation.
37      */
38     public class BaseRecognizerAudioManager extends JseBaseAudioManager {
39         /** The input stream for the recognizer. */
40         private InputStream inputStream;
41     
42         /**
43          * Constructs a new object.
44          * @param format native engine audio format
45          */
46         public BaseRecognizerAudioManager(final AudioFormat format) {
47             super(format);
48         }
49     
50         /**
51          * {@inheritDoc}
52          */
53         public void handleAudioStart() throws AudioException {
54             // Just convert samples 
55             if (inputStream instanceof AudioInputStream) {
56                 AudioInputStream ais = (AudioInputStream)inputStream;
57                 targetAudioFormat = ais.getFormat();
58                 inputStream = AudioSystem.getAudioInputStream(engineAudioFormat, ais);
59                 return;
60             }
61     
62             final String locator = getMediaLocator();
63             // Open URL described in locator
64             final InputStream is;
65             if (locator == null) {
66                 targetAudioFormat = getEngineAudioFormat();
67                 final InputStream source = new LineInputStream(this);
68                 is = new BufferedInputStream(source);
69             } else {
70                 URL url;
71                 try {
72                     url = new URL(locator);
73                 } catch (MalformedURLException e) {
74                     throw new AudioException(e.getMessage());
75                 }
76                 try {
77                     final AudioFileFormat format =
78                         AudioSystem.getAudioFileFormat(url);
79                     targetAudioFormat = format.getFormat();
80                 } catch (UnsupportedAudioFileException e) {
81                 } catch (IOException e) {
82                 }
83                 if (targetAudioFormat == null) {
84                     try {
85                         targetAudioFormat = JavaSoundParser.parse(url);
86                     } catch (URISyntaxException e) {
87                         throw new AudioException(e.getMessage());
88                     }
89                 }
90                 final URLConnection urlConnection;
91                 try {
92                     urlConnection = openURLConnection();
93                 } catch (IOException e) {
94                     throw new AudioException(e.getMessage());
95                 }
96     
97                 try {
98                     final InputStream source  = urlConnection.getInputStream();
99                     is = new BufferedInputStream(source);
100                 } catch (IOException ex) {
101                     throw new AudioException("Cannot get InputStream from URL: "
102                             + ex.getMessage());
103                 }
104             }
105             System.err.println(targetAudioFormat);
106             final AudioInputStream ais = new AudioInputStream(is,
107                         targetAudioFormat, AudioSystem.NOT_SPECIFIED);
108             inputStream = AudioSystem.getAudioInputStream(engineAudioFormat, ais);
109         }
110     
111         /**
112          * {@inheritDoc}
113          */
114         public void handleAudioStop() throws AudioException {
115             // Release IO
116             if (inputStream != null) {
117                 try {
118                     inputStream.close();
119                 } catch (IOException ex) {
120                     throw new AudioException(ex.getMessage());
121                 }
122             }
123         }
124     
125         /**
126          * {@inheritDoc}
127          */
128         public void setMediaLocator(final String locator, final InputStream stream)
129                 throws AudioException {
130             super.setMediaLocator(locator);
131             inputStream = stream;
132         }
133     
134         /**
135          * {@inheritDoc}
136          *
137          * Throws an {@link IllegalArgumentException} since output streams are not
138          * supported.
139          */
140         public void setMediaLocator(String locator, OutputStream stream)
141                 throws AudioException, EngineStateException,
142                 IllegalArgumentException, SecurityException {
143             throw new IllegalArgumentException("output streams are not supported");
144         }
145     
146         /**
147          * {@inheritDoc}
148          */
149         @Override
150         public InputStream getInputStream() {
151             return inputStream;
152         }
153     
154         /**
155          * {@inheritDoc}
156          */
157         @Override
158         public OutputStream getOutputStream() {
159             throw new IllegalArgumentException("output streams are not supported");
160         }
161     }
