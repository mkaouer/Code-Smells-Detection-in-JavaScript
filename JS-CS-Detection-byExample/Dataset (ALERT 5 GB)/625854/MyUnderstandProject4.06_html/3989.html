3     import java.util.concurrent.ThreadFactory;
4     import java.util.concurrent.atomic.AtomicInteger;

6     /**
7      * A custom thread factory, able to create threads with custom name prefix, daemon-ness and priority.
8      * Based on java.util.concurrent.Executors.DefaultThreadFactory.
9      *
10      * @author Yaniv Kunda
11      * @see java.util.concurrent.Executors.DefaultThreadFactory
12      */
13     public class CustomThreadFactory implements ThreadFactory {
14         static final AtomicInteger poolNumber = new AtomicInteger(1);
15         final ThreadGroup group;
16         final AtomicInteger threadNumber = new AtomicInteger(1);
17         final String namePrefix;
18         final boolean daemon;
19         final int priority;
20     
21         public CustomThreadFactory(String namePrefix, boolean daemon, int priority) {
22             if (priority > Thread.MAX_PRIORITY || priority < Thread.MIN_PRIORITY)
23                 throw new IllegalArgumentException("illegal thread priority");
24             SecurityManager s = System.getSecurityManager();
25             this.group = s != null ? s.getThreadGroup() : Thread.currentThread().getThreadGroup();
26             this.namePrefix = namePrefix + "-" + poolNumber.getAndIncrement() + "-thread-";
27             this.daemon = daemon;
28             this.priority = priority;
29         }
30     
31         @Override
32         public Thread newThread(Runnable r) {
33             Thread t = new Thread(group, r, namePrefix + threadNumber.getAndIncrement(), 0);
34             if (t.isDaemon() != daemon)
35                 t.setDaemon(daemon);
36             if (t.getPriority() != priority)
37                 t.setPriority(priority);
38             return t;
39         }
