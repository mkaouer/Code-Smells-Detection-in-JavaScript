9      * <code>DataProcessor</code> unchanged.

26         }

29         * (non-Javadoc)

32         */
33         @Override
34         public void newProperties(PropertySheet ps) throws PropertyException {
35             super.newProperties(ps);

37         }

40         @Override
41         public Data getData() throws DataProcessingException {
42             Data data = getPredecessor().getData();
43     
44             if (data instanceof FloatData) {
45                 float[] values = ((FloatData) data).getValues();
46                 if (gainFactor != 1.0) {
47                     // apply the gain-factor
48                     for (int i = 0; i < values.length; i++) {
49                         values[i] *= gainFactor;
50     
51                     }
52                 }
53     
54             } else if (data instanceof DoubleData) {
55                 double[] values = ((DoubleData) data).getValues();
56                 if (gainFactor != 1.0) {
57                     // apply the gain-factor
58                     for (int i = 0; i < values.length; i++) {
59                         values[i] *= gainFactor;
60     
61                     }
62                 }
63             }
64     
65             return data;
66         }
