9     package edu.cmu.sphinx.demo.jsapi.tags;
10     
11     import java.util.ArrayList;
12     import java.util.List;
13     
14     public class Burger implements OrderItem {
15         public List<String> toppings = new ArrayList<String>();
16         public List<String> condiments = new ArrayList<String>();
17     
18         public Burger() {
19         }
20     
21         public void addTopping(String topping) {
22             toppings.add(topping);
23         }
24     
25         public void addCondiment(String condiment) {
26             condiments.add(condiment);
27         }
28         
29         public String toString() {
30             List<String> allToppings = new ArrayList<String>(toppings);
31             allToppings.addAll(condiments);
32             int numToppings = allToppings.size();
33             if (numToppings == 0) {
34                 return "plain burger.";
35             } else {
36                 StringBuilder sb = new StringBuilder("burger with ");
37                 for (int i = 0; i < numToppings; i++) {
38                     sb.append(allToppings.get(i));
39                     if (numToppings > 1) {
40                         if (i == (numToppings - 2)) {
41                             sb.append(" and ");
42                         } else if (i < (numToppings - 1)) {
43                             sb.append(", ");
44                         }
45                     }
46                 }
47                 sb.append('.');
48                 return sb.toString();
49             }
50         }
51     }
