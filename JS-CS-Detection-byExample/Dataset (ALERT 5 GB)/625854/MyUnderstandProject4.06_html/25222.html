80         /**
81          * Grow branches using Bushderby.
82          * <p/>
83          * Goes through the active list of tokens and expands each token, finding the set of successor tokens until all the
84          * successor tokens are emitting tokens.  With bushderby, non-emitting green nodes are collected into a listed
85          * called the delayedExpansionList.  This is a list of green nodes that are non-emitting. Non-emitting nodes are
86          * usually grown immediately until we find emitting nodes, but with bushderby, we have to delay this expansion until
87          * the node is completely scored. To do this, the non-emitting green nodes are placed in a 'delayedExpansionList'.
88          * After all normal tokens are grown, the tokens in this delayedExpansionList are given the second bushderby pass
89          * and then grown. This process repeats until the delayedExpansionList is empty.
90          */
91         protected void growBranches() {
92             getGrowTimer().start();
93     
94             setBestTokenMap(new HashMap<SearchState, Token>(getActiveList().size() * 10));
95     
96             int pass = 0;
97             boolean moreTokensToExpand = true;
98     
99             Iterator<Token> iterator = getActiveList().iterator();
100     
101             // TODO: fixme!!
102             // setResultList(new LinkedList());
103             // setActiveList(getActiveList().createNew());
104     
105             while (moreTokensToExpand) {
106                 pass++;
107                 List<Token> delayedExpansionList = new ArrayList<Token>();
108     
109                 while (iterator.hasNext()) {
110                     Token token = iterator.next();
111                     collectSuccessorTokens(token, delayedExpansionList);
112                 }
113     
114                 if (delayedExpansionList != null &&
115                     !delayedExpansionList.isEmpty()) {
116                     finalizeBushderby(delayedExpansionList.iterator());
117                     iterator = delayedExpansionList.iterator();
118                     logger.finest("Pass " + pass + " Processing " + delayedExpansionList.size() + " delayed tokens");
119                 } else {
120                     moreTokensToExpand = false;
121                 }
122             }
123     
124             finalizeBushderby(getActiveList().iterator());
125     
126             getGrowTimer().stop();
127         }

130         /** Chase through the list, find all Green nodes and convert the scores to the final bushderby score */
131         private void finalizeBushderby(Iterator<Token> iterator) {
132     
133             while (iterator.hasNext()) {
134                 Token token = iterator.next();
135                 SearchState state = token.getSearchState();
136                 if (isGreenState(state)) {
137                     float logNewScore = (float)
138                             (token.getWorkingScore() / bushderbyEta);
139                     logger.finest("OS: " + token.getScore() + " NS: " + logNewScore);
140                     token.setScore(logNewScore);
141                 }
142             }
143         }

157         /**
158          * Collects the next set of emitting tokens from a token and accumulates them in the active or result lists
159          *
160          * @param token                the token to collect successors from
161          * @param delayedExpansionList the place where tokens that cannot be immediately expaned are placed. Null if we
162          *                             should always expand all nodes.
163          */
164         private void collectSuccessorTokens(Token token,
165                                             List<Token> delayedExpansionList) {
166     
167             // System.out.println("Entering cst...");
168     
169             // If this is a final state, add it to the final list
170             if (token.isFinal()) {
171                 getResultList().add(token);
172                 return;
173             }
174     
175             SearchState state = token.getSearchState();
176             SearchStateArc[] arcs = state.getSuccessors();
177     
178             // For each successor
179             // calculate the entry score for the token based upon the
180             // predecessor token score and the transition probabilities
181             // if the score is better than the best score encountered for
182             // the SentenceHMMState and frame then create a new token, add
183             // it to the lattice and the SentenceHMMState.
184             // If the token is an emitting token add it to the list,
185             // otherwise recursively collect the new tokens successors.
186     
187             for (SearchStateArc arc : arcs) {
188                 SearchState nextState = arc.getState();
189     
190                 if (filterSuccessors && !isValidTransition(token, nextState)) {
191                     continue;
192                 }
193     
194                 float logLanguageProbability = getLanguageProbability(token, arc);            
195     
196                 // We're actually multiplying the variables, but since
197                 // these come in log(), multiply gets converted to add
198                 float logCurrentScore = token.getScore() +
199                     logLanguageProbability +
200                     arc.getInsertionProbability();
201     
202                 boolean firstToken = (getBestToken(nextState) == null);
203                 boolean greenToken = isGreenState(nextState);
204     
205                 float logWorkingScore = firstToken ? LogMath.getLogZero() :
206                     getBestToken(nextState).getWorkingScore();
207     
208                 Token predecessor = getResultListPredecessor(token);
209                 if (firstToken || getBestToken(nextState).getScore() <= logCurrentScore) {
210     
211                     // we may want to create  green tokens all the time
212                     if (greenToken) {
213     
214                         Token newToken = new Token(
215                             predecessor,
216                             nextState,
217                             logCurrentScore,
218                             arc.getInsertionProbability(),
219                             logLanguageProbability,
220                             getCurrentFrameNumber()
221                         );
222                         getTokensCreated().value++;
223     
224                         newToken = collapseToken(newToken);
225     
226                         Token oldBestToken = setBestToken(newToken, nextState);
227     
228                         if (!newToken.isEmitting()) {
229                             if (greenToken && delayedExpansionList != null) {
230                                 if (oldBestToken != null &&
231                                     oldBestToken.getScore() <= logCurrentScore) {
232                                     int oldTokenIdx =
233                                         delayedExpansionList.indexOf(oldBestToken);
234                                     if (oldTokenIdx >= 0)
235                                         delayedExpansionList.remove(oldTokenIdx);
236                                 }
237                                 delayedExpansionList.add(newToken);
238                             } else {
239                                 // System.out.println("Recursing into cst...");
240                                 collectSuccessorTokens(newToken,
241                                     delayedExpansionList);
242                             }
243                         } else if (firstToken) {
244                             getActiveList().add(newToken);
245                         } else {
246                             getActiveList().replace(oldBestToken, newToken);
247                         }
248                     }
249                 }
250     
251                 // with Bushderby nodes are either 'combine' or 'compete'
252                 // Green nodes are 'combine' nodes.  If bushderby is not
253                 // enabled, all nodes are considered to be 'compete'
254                 // nodes.
255                 //
256                 // If bushderby is enabled then we accumulate a working
257                 // score for this token which combines the working scores
258                 // for all paths into this token
259     
260     
261                 if (greenToken) {
262                     Token bestToken = getBestToken(nextState);
263                     if (bestToken != null) {
264                         logWorkingScore = logMath.addAsLinear(
265                             logWorkingScore,
266                             (float)(logCurrentScore * bushderbyEta));
267                         bestToken.setWorkingScore(logWorkingScore);
268                     }
269                     logger.finest("CS: " + logCurrentScore + " WS: " + logWorkingScore);
270                 }
271             }
272         }

275         /**
276          * Determines if the transition from the given token (given its perfect knowledge of context history) to the
277          * nextState is valid
278          *
279          * @param token     the current token
280          * @param nextState the next sentence hmm state
281          * @return true if the transition is valid
282          */
283         private boolean isValidTransition(Token token, SearchState nextState) {
284             Unit prevUnit;
285             Unit thisUnit;
286             Unit[] thisLC;
287             Unit[] thisRC;
288             Unit[] prevLC;
289             Unit[] prevRC;
290     
291             // if we are not transitioning to a unit state, then it is a
292             // valid transition
293     
294             if (!(nextState instanceof UnitSearchState)) {
295                 return true;
296             }
297     
298             // if we are transitioning to a unit state we have to check to
299             // make sure that the contexts (left and right) of the
300             // previous unit align properly with those of the next
301     
302     
303             thisUnit = ((UnitSearchState) nextState).getUnit();
304             thisLC = getLeftContext(thisUnit);
305             thisRC = getRightContext(thisUnit);
306     
307             prevUnit = getPreviousUnit(token);
308     
309             // if there is no previous unit, then the next unit's left
310             // context should be empty
311     
312     
313             if (prevUnit == null) {
314                 return thisLC.length == 0;
315             }
316     
317             prevLC = getLeftContext(prevUnit);
318             prevRC = getRightContext(prevUnit);
319     
320             // we have a transition between units,
321             // prev  RC had better not be empty or its not a valid transition
322             // this  LC had better not be empty or its not a valid transition
323     
324             if (prevRC != null && (prevRC.length == 0 || thisLC.length == 0)) {
325                 return false;
326             }
327     
328             // if the previous right context is longer than the new unit
329             // and its right context then its not a valid transition
330     
331             if (prevRC != null && thisRC != null
332                     && prevRC.length > (thisRC.length + 1)) {
333                 return false;
334             }
335     
336             // if this left context is longer than the previous left
337             // context and unit then its not a valid transition
338     
339             if (thisLC.length > (prevLC.length + 1)) {
340                 return false;
341             }
342     
343             // now check the the previous right context matches this unit
344             // and its context
345     
346             if (prevRC != null && !prevRC[0].getName().equals(thisUnit.getName())) {
347                 return false;
348             }
349     
350             for (int i = 1; prevRC != null && thisRC != null
351                     && i < prevRC.length; i++) {
352                 if (prevRC[i] != thisRC[i - 1]) {
353                     return false;
354                 }
355             }
356     
357             // now check that this left context matches the previous  unit
358             // and its context
359     
360             if (!thisLC[0].getName().equals(prevUnit.getName())) {
361                 return false;
362             }
363     
364             for (int i = 1; i < thisLC.length; i++) {
365                 if (thisLC[i] != prevLC[i - 1]) {
366                     return false;
367                 }
368             }
369     
370             // if we made it to here, its a valid unit-to-unit transition
371             // so we can return true
372     
373             return true;
374         }

399         /**
400          * Given a token return the previous unit
401          *
402          * @param token the token where the unit search is started
403          * @return the unit or null if no unit is found
404          */
405         private Unit getPreviousUnit(Token token) {
406             while (token != null) {
407                 if (token.getSearchState() instanceof UnitSearchState) {
408                     return ((UnitSearchState) token.getSearchState()).getUnit();
409                 }
410                 token = token.getPredecessor();
411             }
412             return null;
413         }

416         /**
417          * Given a linguist and an arc to the next token, determine a language probability for the next state
418          *
419          * @param token the next token
420          * @param arc   the arc to the next state
421          * @return the language probability for the transition to the next state (in LogMath log domain)
422          */
423         private float getLanguageProbability(Token token,
424                                              SearchStateArc arc) {
425             float logProbability = arc.getLanguageProbability();
426             if (languageModel != null && arc.getState() instanceof WordSearchState) {
427                 WordSearchState state = (WordSearchState) arc.getState();
428                 int depth = languageModel.getMaxDepth();
429                 Word word = state.getPronunciation().getWord();
430     
431                 if (isWord(word)) {
432                     List<Word> wordList = new ArrayList<Word>(depth);
433                     wordList.add(word);
434                     while (token != null && wordList.size() < depth) {
435                         if (token.getSearchState() instanceof WordSearchState) {
436                             WordSearchState prevWord = (WordSearchState) token.getSearchState();
437                             Word prevWordObject = prevWord.getPronunciation().getWord();
438                             if (isWord(prevWordObject)) {
439                                 wordList.add(prevWordObject);
440                             }
441                         }
442                         token = token.getPredecessor();
443                     }
444     
445                     if (token == null && wordList.size() < depth) {
446                         // bug: should be adding a word instead, but we
447                         // don't have the dictionary, so we can't get the
448                         // sentence start word
449                         wordList.add(null);
450                         // new Word(Dictionary.SENTENCE_START_SPELLING,
451                         // null));
452                     }
453                     Collections.reverse(wordList);
454                     logProbability = languageModel.getProbability(new WordSequence(wordList));
455                 }
456             }
457             return logProbability;
458         }
