23     /** Generic interface for a saver of acoustic models */

38         public final static String PROP_SPARSE_FORM = "sparseForm";

40         /**

42          *
43          * @return the pool
44          */
45         public Pool<float[]> getMeansPool();

47         /**
48          * Gets the pool of means transformation matrices for this loader.
49          *
50          * @return the pool
51          */
52         public Pool<float[][]> getMeansTransformationMatrixPool();

54         /**
55          * Gets the pool of means transformation vectors for this loader.
56          *
57          * @return the pool
58          */
59         public Pool<float[]> getMeansTransformationVectorPool();
60     
61         /**
62          * Gets the variance pool.
63          *
64          * @return the pool
65          */
66         public Pool<float[]> getVariancePool();
67     
68         /**
69          * Gets the variance transformation matrix pool.
70          *
71          * @return the pool
72          */
73         public Pool<float[][]> getVarianceTransformationMatrixPool();
74     
75         /**
76          * Gets the variance transformation vectorpool.
77          *
78          * @return the pool
79          */
80         public Pool<float[]> getVarianceTransformationVectorPool();

82         /**
83          * Gets the senone pool for this loader.
84          *
85          * @return the pool
86          */
87         public Pool<Senone> getSenonePool();
88     
89         /**
90          * Returns the HMM Manager for this loader.
91          *
92          * @return the HMM Manager
93          */
94         public HMMManager getHMMManager();

96         /**
97          * Returns the map of context indepent units. The map can be accessed by unit name.

99          * @return the map of context independent units

101         public Map<String, Unit> getContextIndependentUnits();

103         /** logs information about this loader */
104         public void logInfo();
105     
106         /**
107          * Returns the size of the left context for context dependent units.

109          * @return the left context size
110          */
111         public int getLeftContextSize();

113         /**
114          * Returns the size of the right context for context dependent units.
115          *
116          * @return the left context size
117          */
118         public int getRightContextSize();
119         
120         public void save(String name, boolean b) throws IOException;
