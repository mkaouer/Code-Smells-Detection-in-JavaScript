10     public class JSGFRuleToken extends JSGFRule {
11     	protected String text;
12     
13     	public JSGFRuleToken() {
14     		setText(null);
15     	}
16     
17     	public JSGFRuleToken(String text) {
18     		setText(text);
19     	}
20     
21     	private boolean containsWhiteSpace(String text) {
22     		for (int i = 0; i < text.length(); ++i) {
23     			if (Character.isWhitespace(text.charAt(i)))
24     				return true;
25     		}
26     		return false;
27     	}
28     
29     	public String getText() {
30     		return text;
31     	}
32     
33     	public void setText(String text) {
34     		this.text = text;
35     	}
36     
37     	public String toString() {
38     		if ((containsWhiteSpace(text)) || (text.indexOf('\\') >= 0)
39     				|| (text.indexOf('"') >= 0)) {
40     			StringBuilder stringBuilder = new StringBuilder(text);
41     
42     			for (int j = stringBuilder.length() - 1; j >= 0; --j) {
43     				int i;
44     				i = stringBuilder.charAt(j);
45     				if ((i == '"') || (i == '\\')) {
46     					stringBuilder.insert(j, '\\');
47     				}
48     			}
49     			stringBuilder.insert(0, '"');
50     			stringBuilder.append('"');
51     
52     			return stringBuilder.toString();
53     		}
54     		return text;
55     	}
