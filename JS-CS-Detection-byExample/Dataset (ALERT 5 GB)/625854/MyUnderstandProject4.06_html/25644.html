14

28     /**
29      * This program takes in an audio file, does frontend signal processing to it,
30      * and then dumps the resulting Feature into a separate file. Also it can
31      * process a list of files at once.
32      * <p/>
33      * Available options:
34      * <ul>
35      * <li>-config configFile - the XML configuration file</li>
36      * <li>-name frontendName - the name of the feature extractor inside the
37      * configuration file</li>
38      * <li>-i audioFile - the name of the audio file</li>
39      * <li>-ctl controlFile - the name of the input file for batch processing</li>
40      * <li>-o outputFile - the name of the output file or output folder</li>
41      * <li>-format binary/ascii - output file format</li>
42      * </ul>
43      */
44     public class FeatureFileDumper {
45     
46         private FrontEnd frontEnd;
47         private StreamDataSource audioSource;
48         private List<float[]> allFeatures;
49         private int featureLength = -1;
50     
51         /** The logger for this class */
52         private static final Logger logger = Logger
53                 .getLogger("edu.cmu.sphinx.tools.feature.FeatureFileDumper");
54     
55         /**
56          * Constructs a FeatureFileDumper.
57          * 
58          * @param cm
59          *            the configuration manager
60          * @param frontEndName
61          *            the name for the frontend
62          */
63         public FeatureFileDumper(ConfigurationManager cm, String frontEndName)
64                 throws IOException {
65             try {
66                 frontEnd = (FrontEnd) cm.lookup(frontEndName);
67                 audioSource = (StreamDataSource) cm.lookup("streamDataSource");
68             } catch (Exception e) {
69                 e.printStackTrace();
70             }
71         }
72     
73         /**
74          * Process the file and store the features
75          * 
76          * @param inputAudioFile
77          *            the input audio file
78          * @throws FileNotFoundException
79          */
80         public void processFile(String inputAudioFile) throws FileNotFoundException {
81             audioSource
82                     .setInputStream(new FileInputStream(inputAudioFile), "audio");
83             allFeatures = new LinkedList<float[]>();
84             getAllFeatures();
85             logger.info("Frames: " + allFeatures.size());
86         }
87     
88         /**
89          * Retrieve all Features from the frontend, and cache all those with actual
90          * feature data.
91          */
92         private void getAllFeatures() {
93             /*
94              * Run through all the data and produce feature.
95              */
96             try {
97                 assert (allFeatures != null);
98                 Data feature = frontEnd.getData();
99                 while (!(feature instanceof DataEndSignal)) {
100                     if (feature instanceof DoubleData) {
101                         double[] featureData = ((DoubleData) feature).getValues();
102                         if (featureLength < 0) {
103                             featureLength = featureData.length;
104                             logger.info("Feature length: " + featureLength);
105                         }
106                         float[] convertedData = new float[featureData.length];
107                         for (int i = 0; i < featureData.length; i++) {
108                             convertedData[i] = (float) featureData[i];
109                         }
110                         allFeatures.add(convertedData);
111                     } else if (feature instanceof FloatData) {
112                         float[] featureData = ((FloatData) feature).getValues();
113                         if (featureLength < 0) {
114                             featureLength = featureData.length;
115                             logger.info("Feature length: " + featureLength);
116                         }
117                         allFeatures.add(featureData);
118                     }
119                     feature = frontEnd.getData();
120                 }
121             } catch (Exception e) {
122                 e.printStackTrace();
123             }
124         }
125     
126         /**
127          * Returns the total number of data points that should be written to the
128          * output file.
129          * 
130          * @return the total number of data points that should be written
131          */
132         private int getNumberDataPoints() {
133             return (allFeatures.size() * featureLength);
134         }
135     
136         /**
137          * Dumps the feature to the given binary output.
138          * 
139          * @param outputFile
140          *            the binary output file
141          */
142         public void dumpBinary(String outputFile) throws IOException {
143             DataOutputStream outStream = new DataOutputStream(new FileOutputStream(
144                     outputFile));
145             outStream.writeInt(getNumberDataPoints());
146     
147             for (float[] feature : allFeatures) {
148                 for (float val : feature) {
149                     outStream.writeFloat(val);
150                 }
151             }
152     
153             outStream.close();
154         }
155     
156         /**
157          * Dumps the feature to the given ASCII output file.
158          * 
159          * @param outputFile
160          *            the ASCII output file
161          */
162         public void dumpAscii(String outputFile) throws IOException {
163             PrintStream ps = new PrintStream(new FileOutputStream(outputFile), true);
164             ps.print(getNumberDataPoints());
165             ps.print(' ');
166     
167             for (float[] feature : allFeatures) {
168                 for (float val : feature) {
169                     ps.print(val);
170                     ps.print(' ');
171                 }
172             }
173     
174             ps.close();
175         }
176     
177         /**
178          * Main program for this dumper.
179          */
180         public static void main(String[] argv) {
181     
182             String configFile = null;
183             String frontEndName = null;
184             String inputFile = null;
185             String inputCtl = null;
186             String outputFile = null;
187             String format = "binary";
188     
189             for (int i = 0; i < argv.length; i++) {
190                 if (argv[i].equals("-c")) {
191                     configFile = argv[++i];
192                 }
193                 if (argv[i].equals("-name")) {
194                     frontEndName = argv[++i];
195                 }
196                 if (argv[i].equals("-i")) {
197                     inputFile = argv[++i];
198                 }
199                 if (argv[i].equals("-ctl")) {
200                     inputCtl = argv[++i];
201                 }
202                 if (argv[i].equals("-o")) {
203                     outputFile = argv[++i];
204                 }
205                 if (argv[i].equals("-format")) {
206                     format = argv[++i];
207                 }
208             }
209     
210             if (frontEndName == null || (inputFile == null && inputCtl == null)
211                     || outputFile == null || format == null) {
212                 System.out
213                         .println("Usage: FeatureFileDumper "
214                                 + "[ -config configFile ] -name frontendName "
215                                 + "< -i input File -o outputFile | -ctl inputFile -i inputFolder -o outputFolder >");
216                 System.exit(1);
217             }
218     
219             logger.info("Input file: " + inputFile);
220             logger.info("Output file: " + outputFile);
221             logger.info("Format: " + format);
222     
223             try {
224                 URL url;
225                 if (configFile != null) {
226                     url = new File(configFile).toURI().toURL();
227                 } else {
228                     url = FeatureFileDumper.class
229                             .getResource("frontend.config.xml");
230                 }
231                 ConfigurationManager cm = new ConfigurationManager(url);
232                 FeatureFileDumper dumper = new FeatureFileDumper(cm, frontEndName);
233     
234                 if (inputCtl == null)
235                     dumper.processFile(inputFile, outputFile, format);
236                 else
237                     dumper.processCtl(inputCtl, inputFile, outputFile, format);
238             } catch (IOException ioe) {
239                 System.err.println("I/O Error " + ioe);
240             } catch (PropertyException p) {
241                 System.err.println("Bad configuration " + p);
242             }
243         }
244     
245         private void processFile(String inputFile, String outputFile, String format)
246                 throws MalformedURLException, IOException {
247             processFile(inputFile);
248             if (format.equals("binary")) {
249                 dumpBinary(outputFile);
250             } else if (format.equals("ascii")) {
251                 dumpAscii(outputFile);
252             } else {
253                 System.out.println("ERROR: unknown output format: " + format);
254             }
255         }
256     
257         private void processCtl(String inputCtl, String inputFolder,
258                 String outputFolder, String format) throws MalformedURLException,
259                 IOException {
260     
261             Scanner scanner = new Scanner(new File(inputCtl));
262             while (scanner.hasNext()) {
263                 String fileName = scanner.next();
264                 String inputFile = inputFolder + "/" + fileName + ".wav";
265                 String outputFile = outputFolder + "/" + fileName + ".mfc";
266     
267                 processFile(inputFile);
268                 if (format.equals("binary")) {
269                     dumpBinary(outputFile);
270                 } else if (format.equals("ascii")) {
271                     dumpAscii(outputFile);
272                 } else {
273                     System.out.println("ERROR: unknown output format: " + format);
274                 }
275             }
276         }
277     }
