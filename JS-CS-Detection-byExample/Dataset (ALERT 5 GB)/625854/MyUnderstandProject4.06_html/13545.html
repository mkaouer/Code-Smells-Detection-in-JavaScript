15     package edu.cmu.sphinx.tools.gui.writer;

17     import edu.cmu.sphinx.tools.gui.ConfigProperties;
18     import edu.cmu.sphinx.util.props.RawPropertyData;

20     import java.util.Map;
21     import java.util.List;

23     /**
24      * This is a helper class to convert from ConfigProperties to String format 
25      * that's ready to be written to .config.XML file
26      *
27      * @author Ariani
28      */
29     public class ConfigConverter {
30         
31         /* this class uses Singleton pattern for the constructor */
32         private ConfigConverter() {}
33         
34         private static class ConfigConverterHolder {
35           private static final ConfigConverter instance = new ConfigConverter();
36         }
37           
38         /**
39          * Get reference to the <code>ConfigConverter</code> 
40          * 
41          * @return <code>ConfigConverter</code>
42          */
43         public static ConfigConverter getInstance(){
44           return ConfigConverterHolder.instance;
45         }
46         
47         private final static String XML_HEADING = "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n";
48         private final static String XML_SUBHEAD = '\n' +
49             "<!--    Sphinx-4 Configuration file--> \n" +
50             "<!--  ******************************************************** -->\n\n";
51         private final static String XML_ROOT = "<config>\n";
52         private final static String XML_ROOT_CLOSE = "</config>\n";
53         private final static String XML_GLOBAL = '\n' +
54             "<!--  *******************************************-->\n"+ 
55             "<!--  frequently tuned properties                -->\n" +
56             "<!-- ********************************************-->\n"; 
57         private final static String XML_CONF = '\n' +
58          "<!--  ********************************************************-->\n"+
59          "<!--  component       configuration                           -->\n"+
60          "<!--  ********************************************************-->\n\n"; 
61     
62         private final static String XML_PROP = "\t <property name=\"";
63         private final static String XML_COMPONENT = "<component name=\"";
64         private final static String XML_COMPONENT_CLOSE = "</component>\n\n";
65         private final static String XML_TYPE = " type=\"";
66         private final static String XML_VALUE = " value=\"";
67         private final static String XML_PROPLIST = "\t <propertylist name=\"";
68         private final static String XML_PROPLIST_CLOSE = "\t </propertylist>\n";
69         private final static String XML_ITEM = "\t\t <item>";
70         private final static String XML_ITEM_CLOSE = "</item>\n";
71         
72         private final static String XML_CLOSE_ELEMENT = "\" /> \n";
73         private final static String XML_CLOSE_TAG = "\" > \n";
74         private final static String XML_QUOTE = "\"";
75                 
76         // a helper method to write the information from a single RawPropertyData
77         private void writeComponent(StringBuilder sb, RawPropertyData rpd) {
78             String name = rpd.getName();
79             String classname = rpd.getClassName();
80             Map<String, Object> properties = rpd.getProperties();
81             if (name == null || classname == null) {
82                 return;
83             }
84             sb.append(XML_COMPONENT).append(name).append(XML_QUOTE + XML_TYPE)
85                     .append(classname).append(XML_CLOSE_TAG);
86     
87             for (Map.Entry<String, Object> entry : properties.entrySet()) {
88                 String propName = entry.getKey();
89                 Object propVal = entry.getValue();
90                 if (propName == null && propVal == null) {
91                     continue;
92                 }
93                 if (propVal instanceof String) {
94                     sb.append(XML_PROP).append(propName).append(
95                             XML_QUOTE + XML_VALUE).append(propVal).append(
96                             XML_CLOSE_ELEMENT);
97                 } else if (propVal instanceof List<?>) {
98                     sb.append(XML_PROPLIST).append(propName).append(XML_CLOSE_TAG);
99                     // iterate the propertyList
100                     for (Object obj : (List<?>) propVal) {
101                         if (obj instanceof String) {
102                             String item = (String) obj;
103                             if (item != null && !item.trim().isEmpty()) {
104                                 sb.append(XML_ITEM).append(item).append(
105                                         XML_ITEM_CLOSE);
106                             }
107                         }
108                     }
109                     sb.append(XML_PROPLIST_CLOSE);
110                 }
111             }
112             sb.append(XML_COMPONENT_CLOSE);
113         }
114         
115         /**
116          * Write the property name and values into a StringBuilder
117          * 
118          * @param cp <code>ConfigProperties</code> that keeps the property name-values
119          * @return StringBuilder to write the output
120          */
121         public StringBuilder writeOutput(ConfigProperties cp) {
122     
123             StringBuilder sb = new StringBuilder();
124             Map<String, String> global = cp.getGlobal();
125             Map<String, RawPropertyData> property = cp.getProperty();
126     
127             sb.append(XML_HEADING + XML_SUBHEAD + XML_ROOT + '\n');
128             sb.append(XML_GLOBAL + '\n');
129             
130             // iterate the global properties
131             if (global != null) {
132                 for (Map.Entry<String, String> entry : global.entrySet()) {
133                     String name = entry.getKey();
134                     if (name != null) {
135                         String value = entry.getValue();
136                         sb.append(XML_PROP).append(name).append(XML_QUOTE + XML_VALUE).append(value).append(XML_CLOSE_ELEMENT);
137                     }
138     
139                 }
140             }
141             sb.append(XML_CONF);
142     
143             if (property != null) {
144                 for (Map.Entry<String, RawPropertyData> entry : property.entrySet()) {
145                     String name = entry.getKey();
146                     RawPropertyData rpd = entry.getValue();
147                     if (name != null && rpd != null) {
148                         writeComponent(sb, rpd);
149                     }
150                 }
151             }
152             sb.append(XML_ROOT_CLOSE + '\n');
153             return sb;
154         }
155     }
