27     package javax.speech;
28     
29     import java.util.Vector;
30     
31     //Comp. 2.0.6
32     
33     public class EngineEvent extends SpeechEvent {
34         // Events.
35         public static final int ENGINE_ALLOCATED = 0x1000001;
36     
37         public static final int ENGINE_DEALLOCATED = 0x1000002;
38     
39         public static final int ENGINE_ALLOCATING_RESOURCES = 0x1000004;
40     
41         public static final int ENGINE_DEALLOCATING_RESOURCES = 0x1000008;
42     
43         public static final int ENGINE_PAUSED = 0x1000010;
44     
45         public static final int ENGINE_RESUMED = 0x1000020;
46     
47         public static final int ENGINE_DEFOCUSED = 0x1000040;
48     
49         public static final int ENGINE_FOCUSED = 0x1000080;
50     
51         public static final int ENGINE_ERROR = 0x1000100;
52     
53         public static final int DEFAULT_MASK = ENGINE_ALLOCATED
54                 | ENGINE_DEALLOCATED | ENGINE_PAUSED | ENGINE_RESUMED
55                 | ENGINE_FOCUSED | ENGINE_DEFOCUSED | ENGINE_ERROR;
56     
57         private long oldEngineState;
58     
59         private long newEngineState;
60     
61         private Throwable problem;
62     
63         public EngineEvent(Engine source, int id, long oldEngineState,
64                 long newEngineState, Throwable problem)
65             throws IllegalArgumentException {
66             super(source, id);
67     
68             if ((problem != null) && (id != ENGINE_ERROR)) {
69                 throw new IllegalArgumentException(
70                         "A problem can only be provided for ENGINE_ERROR");
71             }
72             if ((problem == null) && (id == ENGINE_ERROR)) {
73                 throw new IllegalArgumentException(
74                         "A problem must be provided for ENGINE_ERROR");
75             }
76             this.oldEngineState = oldEngineState;
77             this.newEngineState = newEngineState;
78             this.problem = problem;
79         }
80     
81         public long getNewEngineState() {
82             return newEngineState;
83         }
84     
85         public long getOldEngineState() {
86             return oldEngineState;
87         }
88     
89         public Throwable getEngineError() {
90             final int id = getId();
91             if (id == ENGINE_ERROR) {
92                 return problem;
93             }
94     
95             return null;
96         }
97     
98         /**
99          * {@inheritDoc}
100          */
101         protected void id2String(StringBuffer str) {
102             maybeAddId(str, ENGINE_ALLOCATED, "ENGINE_ALLOCATED");
103             maybeAddId(str, ENGINE_DEALLOCATED, "ENGINE_DEALLOCATED");
104             maybeAddId(str, ENGINE_ALLOCATING_RESOURCES, 
105                     "ENGINE_ALLOCATING_RESOURCES");
106             maybeAddId(str, ENGINE_DEALLOCATING_RESOURCES, 
107                 "ENGINE_DEALLOCATING_RESOURCES");
108             maybeAddId(str, ENGINE_PAUSED, "ENGINE_PAUSED");
109             maybeAddId(str, ENGINE_RESUMED, "ENGINE_RESUMED");
110             maybeAddId(str, ENGINE_DEFOCUSED, "ENGINE_DEFOCUSED");
111             maybeAddId(str, ENGINE_FOCUSED, "ENGINE_FOCUSED");
112             maybeAddId(str, ENGINE_ERROR, "ENGINE_ERROR");
113             super.id2String(str);
114         }
115     
116         /**
117          * {@inheritDoc}
118          */
119         protected Vector getParameters() {
120             final Vector parameters = super.getParameters();
121     
122             final Long oldEngineStateObject = new Long(oldEngineState);
123             parameters.addElement(oldEngineStateObject);
124             final Long newEngineStateObject = new Long(newEngineState);
125             parameters.addElement(newEngineStateObject);
126             parameters.addElement(problem);
127     
128             return parameters;
129         }
130     }
