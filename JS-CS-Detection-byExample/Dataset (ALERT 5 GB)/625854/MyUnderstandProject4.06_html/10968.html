124         /**
125          * Parse a set of tokens against a particular rule from a particluar grammar
126          * returning a RuleParse data structure is successful and null otherwise.
127          * @param inputTokens String[]
128          * @param grammarManager GrammarManager
129          * @param grammarReference String
130          * @param ruleName String
131          * @return RuleParse[]
132          */
133         public static RuleParse[] mparse(final String[] inputTokens,
134                                          final GrammarManager grammarManager,
135                                          final String grammarReference,
136                                          final String ruleName) {
137             final RuleParser rp = new RuleParser(grammarManager, 0);
138             String[] rNames;
139             RuleComponent startRule = null;
140             final Grammar gram = grammarManager.getGrammar(grammarReference);
141             final RuleGrammar grammar;
142             if (gram instanceof RuleGrammar) {
143                 grammar = (RuleGrammar) gram;
144             } else {
145                 return null;
146             }
147             if (ruleName != null) {
148                 rNames = new String[1];
149                 rNames[0] = ruleName;
150             } else {
151                 rNames = grammar.listRuleNames();
152             }
153             Vector parsed = new Vector();
154             for (int j = 0; j < rNames.length; j++) {
155                 if ((ruleName == null) && !(grammar.isActivatable(rNames[j]))) {
156                     continue;
157                 }
158                 final Rule currentRule = grammar.getRule(rNames[j]);
159                 startRule = currentRule.getRuleComponent();
160                 if (startRule == null) {
161                     LOGGER.severe("BAD RULENAME " + rNames[j]);
162                     continue;
163                 }
164                 final GrammarGraph grammarGraph = rp.buildGrammarGraph(grammar,
165                         rNames[j]);
166                 final GrammarNode node = grammarGraph.getStartNode();
167                 if (rp.parse(node, inputTokens)) {
168                     final Object element = rp.grammarElements.pop();
169                     parsed.add(element);
170                 }
171             }
172             if (parsed.size() == 0) {
173                 //No parse is available
174                 return null;
175             }
176             final RuleParse[] rpa = new RuleParse[parsed.size()];
177             parsed.copyInto(rpa);
178             return rpa;
179         }

274         /**
275         * Creates an sub-graph that represents a rule reference.
276         * @param rg the rule grammar
277         * @param r the rule reference
278         * @return GrammarGraph
279         */
280        private GrammarGraph buildGrammarGraph(final RuleGrammar rg,
281                                               final RuleReference r) {
282            RuleGrammar currentRuleGrammar = rg;
283            GrammarNode startNode = new GrammarNode(false,
284                                                    GrammarNode.START_REFERENCE, r);
285            GrammarNode endNode = new GrammarNode(false, GrammarNode.END_REFERENCE);
286     
287            String simpleName = r.getRuleName();
288            RuleComponent ruleref;
289            if (currentRuleGrammar.getRule(simpleName) == null) {
290                ruleref = null;
291            } else {
292                ruleref = currentRuleGrammar.getRule(simpleName).getRuleComponent();
293            }
294            if (ruleref == null) {
295                String gname = r.getGrammarReference();
296                if ((gname != null) && (gname.length() > 0)) {
297                    RuleGrammar rg1 = null;
298                    try {
299                        rg1 = (RuleGrammar) grammarManager.getGrammar(gname);
300                    } catch (EngineStateException ex) {
301                        ex.printStackTrace();
302                    }
303                    if (rg1 != null) {
304                        ruleref = rg1.getRule(simpleName).getRuleComponent();
305                        currentRuleGrammar = rg1;
306                    } else {
307                        LOGGER.severe("ERROR: UNKNOWN GRAMMAR " + gname);
308                    }
309                }
310                if (ruleref == null) {
311                    LOGGER.severe("ERROR: UNKNOWN RULE NAME " + r.getRuleName()
312                                   + " " + r);
313                    return null;
314                }
315            }
316     
317            GrammarGraph rc = buildGrammarGraph(currentRuleGrammar, ruleref);
318            if (rc == null) {
319                return null;
320            }
321            startNode.addArc(rc.getStartNode());
322            rc.getEndNode().addArc(endNode);
323     
324            return new GrammarGraph(startNode, endNode);
325     
326        }

378         /**
379         * Creates an sub-graph that represents a rule sequence.
380         * @param rg the rule grammar
381         * @param r the rule sequence
382         * @return GrammarGraph
383         */
384         private GrammarGraph buildGrammarGraph(final RuleGrammar rg,
385                                                final RuleSequence r) {
386             GrammarNode startNode = new GrammarNode(false,
387                                                     GrammarNode.START_SEQUENCE, r);
388             GrammarNode endNode = new GrammarNode(false, GrammarNode.END_SEQUENCE);
389     
390             RuleComponent[] rules = r.getRuleComponents();
391     
392             GrammarNode lastGrammarNode = null;
393     
394             // expand and connect each rule in the sequence serially
395             for (int i = 0; i < rules.length; i++) {
396                 RuleComponent rule = rules[i];
397                 GrammarGraph newNodes = buildGrammarGraph(rg, rule);
398     
399                 // first node
400                 if (i == 0) {
401                     startNode.addArc(newNodes.getStartNode());
402                 }
403     
404                 // last node
405                 if (i == (rules.length - 1)) {
406                     newNodes.getEndNode().addArc(endNode);
407                 }
408     
409                 if (i > 0) {
410                     lastGrammarNode.addArc(newNodes.getStartNode());
411                 }
412                 lastGrammarNode = newNodes.getEndNode();
413             }
414     
415             return new GrammarGraph(startNode, endNode);
416         }

431         * Creates an sub-graph that represents a rule count.
432         * @param rg the rule grammar
433         * @param r the rule count
434         * @return GrammarGraph
435         */
436        private GrammarGraph buildGrammarGraph(final RuleGrammar rg,
437                                               final RuleCount r) {
438            GrammarNode startNode = new GrammarNode(false, GrammarNode.START_COUNT,
439                                                    r);
440            GrammarNode endNode = new GrammarNode(false, GrammarNode.END_COUNT);
441     
442            int minRepeat = r.getRepeatMin();
443            int maxRepeat = r.getRepeatMax();
444            GrammarGraph newNodes = buildGrammarGraph(rg, r.getRuleComponent());
445            int countNodes = 1;
446     
447            GrammarGraph lastNode = newNodes;
448     
449            if (minRepeat > 1) {
450                GrammarGraph tmpGraph;
451                while (countNodes < minRepeat) {
452                    countNodes++;
453                    /** @todo how can i copy a graph */
454                    tmpGraph = buildGrammarGraph(rg, r.getRuleComponent());
455                    lastNode = tmpGraph;
456                    newNodes.getEndNode().addArc(tmpGraph.getStartNode());
457                    /** @todo review this */
458                    newNodes.setEndNode(tmpGraph.getEndNode());
459                }
460            }
461     
462            if (maxRepeat != RuleCount.REPEAT_INDEFINITELY) {
463                GrammarGraph tmpGraph;
464                ArrayList<GrammarNode> v = new ArrayList();
465                lastNode = newNodes;
466                while (countNodes < maxRepeat) {
467                    ++countNodes;
468                    /** @todo how can i copy a graph */
469                    tmpGraph = buildGrammarGraph(rg, r.getRuleComponent());
470                    v.add(lastNode.getEndNode());
471                    newNodes.getEndNode().addArc(tmpGraph.getStartNode());
472                    /** @todo review this */
473                    newNodes.setEndNode(tmpGraph.getEndNode());
474                    lastNode = tmpGraph;
475                }
476     
477                //set this nodes optional
478                for (GrammarNode g : v) {
479                    g.addArc(endNode);
480                }
481            }
482     
483            startNode.addArc(newNodes.getStartNode());
484            newNodes.getEndNode().addArc(endNode);
485     
486            // if this is optional, add a bypass arc
487            if (minRepeat == 0) {
488                startNode.addArc(endNode);
489            }
490     
491            // if this can possibly occur indefinitely add a loopback
492            if (maxRepeat == RuleCount.REPEAT_INDEFINITELY) {
493                if (lastNode != null) {
494                    newNodes.getEndNode().addArc(lastNode.getStartNode());
495                }
496            }
497            return new GrammarGraph(startNode, endNode);
498        }

500        /** this method parses a token.
501          * @param currentNode the current node of this grammar
502          * @param input the set of tokens
503          * @return <code>true</code> if this nodes accepts the current (iPos) input
504          */
505         public final boolean parseToken(final GrammarNode currentNode,
506                                         final String[] input) {
507             if (iPos >= input.length) {
508                 return false;
509             }
510             // @todo what about case sensitivity ??????
511             String tText = ((RuleToken) currentNode.getRuleComponent()).getText().
512                            toLowerCase();
513             if (tText.equals(input[iPos]) || (input[iPos].equals("%"))
514                 || (input[iPos].equals("*"))) {
515                 iPos++;
516                 if (parse(currentNode.getArcList().get(0).getGrammarNode(),
517                           input)) {
518                     grammarElements.push(new RuleToken(tText));
519                     return true;
520                 } else {
521                     return false;
522                 }
523             } else {
524                 if (tText.indexOf(' ') < 0) {
525                     return false;
526                 }
527                 if (!tText.startsWith(input[iPos])) {
528                     return false;
529                 }
530                 String[] ta = tokenize(tText);
531                 int j = 0;
532                 StringBuffer strBuffer = new StringBuffer("");
533                 // this while is necessary because an token
534                 // can contain more than a single word
535                 while (true) {
536                     if (j >= ta.length) {
537                         break;
538                     }
539                     if (iPos >= input.length) {
540                         return false;
541                     }
542                     if (!ta[j].equals(input[iPos])) {
543                         return false;
544                     }
545                     if (j > 0) {
546                         strBuffer.append(" ");
547                     }
548                     strBuffer.append(ta[j]);
549                     iPos++;
550                     j++;
551                 }
552                 if (parse(currentNode.getArcList().get(0).getGrammarNode(),
553                           input)) {
554                     grammarElements.push(new RuleToken(strBuffer.toString()));
555                     return true;
556                 } else {
557                     return false;
558                 }
559             }
560         }

563          * This method constructs an RuleSequence for the rule parser.
564          */
565         private void posParseStartSequence() {
566             Vector<RuleComponent> arSeq = new Vector<RuleComponent>();
567             while (true) {
568                 if (grammarElements.empty()) {
569                     break;
570                 }
571                 Object topElement = grammarElements.pop();
572                 if (topElement instanceof GrammarNode
573                     && ((GrammarNode) topElement).getNodeType()
574                     == GrammarNode.END_SEQUENCE) {
575                     RuleComponent[] rc = new RuleComponent[arSeq.size()];
576                     arSeq.copyInto(rc);
577                     grammarElements.push(new RuleSequence(rc));
578                     break;
579                 } else if (topElement instanceof RuleComponent) {
580                     arSeq.add((RuleComponent) topElement);
581                 }
582             }
583         }

586          * This method constructs an RuleCount for the rule parser.
587          * @param currentNode the current node
588          */
589         private void posParseStartCount(final GrammarNode currentNode) {
590             Vector ruleComponents = new Vector();
591             int count = 0;
592             while (true) {
593                 if (grammarElements.empty()) {
594                     break;
595                 }
596                 Object topElement = grammarElements.pop();
597                 if (topElement instanceof GrammarNode
598                     && ((GrammarNode) topElement).getNodeType()
599                     == GrammarNode.END_COUNT) {
600                     RuleComponent[] copy = new RuleComponent[ruleComponents.size()];
601                     ruleComponents.copyInto(copy);
602     
603                     int repeatProb = ((RuleCount) currentNode.getRuleComponent()).
604                                      getRepeatProbability();
605                     if (repeatProb != RuleCount.MAX_PROBABILITY) {
606                         grammarElements.push(new RuleCount(new RuleSequence(copy),
607                                 count, count, repeatProb));
608                     } else {
609                         grammarElements.push(new RuleCount(new RuleSequence(copy),
610                                 count, count));
611                     }
612                     break;
613                 } else if (topElement instanceof RuleComponent) {
614                     ruleComponents.add(topElement);
615                     count++;
616                 }
617             }
618         }

670         /** this method parses recursively an grammar graph, to check if
671          * a set of tokens belongs at this grammar.
672          * @param currentNode the current node of this grammar
673          * @param input the set of tokens
674          * @return <code>true</code> if starting in the current node exists an
675          * way in the graph to ends in a final node.
676          */
677         private boolean parse(final GrammarNode currentNode,
678                                     final String[] input) {
679             final int currentIPos = iPos;
680     
681             if (currentNode.isFinalNode()) {
682                 if (iPos == input.length) {
683                     grammarElements.push(currentNode);
684                     return true;
685                 } else {
686                     return false;
687                 }
688             }
689     
690             if (currentNode.getNodeType() == GrammarNode.TOKEN) {
691                 return parseToken(currentNode, input);
692             } else {
693                 for (GrammarArc arc : currentNode.getArcList()) {
694                     if (parse(arc.getGrammarNode(), input)) {
695                         switch (currentNode.getNodeType()) {
696                         case GrammarNode.END_ALTERNATIVE:
697                         case GrammarNode.END_COUNT:
698                         case GrammarNode.END_REFERENCE:
699                         case GrammarNode.END_SEQUENCE:
700                             grammarElements.push(currentNode);
701                             break;
702                         case GrammarNode.START_SEQUENCE:
703                             posParseStartSequence();
704                             break;
705                         case GrammarNode.START_COUNT:
706                             posParseStartCount(currentNode);
707                             break;
708                         case GrammarNode.START_REFERENCE:
709                             posParseStartReference(currentNode);
710                             break;
711                         case GrammarNode.START_ALTERNATIVE:
712                             posParseStartAlternative();
713                             break;
714                         default:
715                             break;
716                         }
717                         return true;
718                     }
719                 }
720                 iPos = currentIPos;
721                 return false;
722             }
723         }

726         /**
727          * tokenize a string.
728          * @param text the text to tokenize
729          * @return an array of tokens
730          **/
731         static String[] tokenize(final String text) {
732             StringTokenizer st = new StringTokenizer(text);
733             int size = st.countTokens();
734             String[] res = new String[size];
735             int i = 0;
736             while (st.hasMoreTokens()) {
737                 res[i++] = st.nextToken().toLowerCase();
738             }
739             return res;
740         }
