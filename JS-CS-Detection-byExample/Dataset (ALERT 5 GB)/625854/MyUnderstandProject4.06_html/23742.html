73         /**
74          * Loads the sphinx3 density file, a set of density arrays are created and
75          * placed in the given pool.
76          * 
77          * @param path
78          *            the name of the data
79          * @param floor
80          *            the minimum density allowed
81          * @return a pool of loaded densities
82          * @throws FileNotFoundException
83          *             if a file cannot be found
84          * @throws IOException
85          *             if an error occurs while loading the data
86          */
87         protected Pool<float[]> loadDensityFile(String path, float floor) throws IOException, URISyntaxException {
88             logger.fine("Loading density file from: " + path);
89             InputStream inputStream = getDataStream(path);
90             if (inputStream == null) {
91                 throw new FileNotFoundException("Error trying to read file " + path);
92             }
93     
94             // 'false' argument refers to EOL is insignificant
95             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream, '#', false);
96             Pool<float[]> pool = new Pool<float[]>(path);
97             est.expectString("param");
98             int numStates = est.getInt("numStates");
99             int numStreams = est.getInt("numStreams");
100             int numGaussiansPerState = est.getInt("numGaussiansPerState");
101             pool.setFeature(NUM_SENONES, numStates);
102             pool.setFeature(NUM_STREAMS, numStreams);
103             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
104             int vectorLength = 39;
105     
106             for (int i = 0; i < numStates; i++) {
107                 est.expectString("mgau");
108                 est.expectInt("mgau index", i);
109                 est.expectString("feat");
110                 est.expectInt("feat index", 0);
111                 for (int j = 0; j < numGaussiansPerState; j++) {
112                     est.expectString("density");
113                     est.expectInt("densityValue", j);
114                     float[] density = new float[vectorLength];
115                     for (int k = 0; k < vectorLength; k++) {
116                         density[k] = est.getFloat("val");
117                         if (density[k] < floor) {
118                             density[k] = floor;
119                         }
120                     }
121                     int id = i * numGaussiansPerState + j;
122                     pool.put(id, density);
123                 }
124             }
125             est.close();
126             return pool;
127         }

129         /**
130          * Loads the mixture weights.
131          * 
132          * @param path
133          *            the path to the mixture weight file
134          * @param floor
135          *            the minimum mixture weight allowed
136          * @return a pool of mixture weights
137          * @throws FileNotFoundException
138          *             if a file cannot be found
139          * @throws IOException
140          *             if an error occurs while loading the data
141          */
142         protected Pool<float[]> loadMixtureWeights(String path, float floor) throws IOException, URISyntaxException {
143             logger.fine("Loading mixture weights from: " + path);
144             InputStream inputStream = getDataStream(path);
145             if (inputStream == null) {
146                 throw new FileNotFoundException("Error trying to read file " + path);
147             }
148     
149             Pool<float[]> pool = new Pool<float[]>(path);
150             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream, '#', false);
151             est.expectString("mixw");
152             int numStates = est.getInt("numStates");
153             int numStreams = est.getInt("numStreams");
154             int numGaussiansPerState = est.getInt("numGaussiansPerState");
155             pool.setFeature(NUM_SENONES, numStates);
156             pool.setFeature(NUM_STREAMS, numStreams);
157             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
158             for (int i = 0; i < numStates; i++) {
159                 est.expectString("mixw");
160                 est.expectString("[" + i);
161                 est.expectString("0]");
162                 // float total = est.getFloat("total");
163                 float[] logMixtureWeight = new float[numGaussiansPerState];
164                 for (int j = 0; j < numGaussiansPerState; j++) {
165                     float val = est.getFloat("mixwVal");
166                     if (val < floor) {
167                         val = floor;
168                     }
169                     logMixtureWeight[j] = val;
170                 }
171                 logMath.linearToLog(logMixtureWeight);
172                 pool.put(i, logMixtureWeight);
173             }
174             est.close();
175             return pool;
176         }

178         /**
179          * Loads the transition matrices.
180          * 
181          * @param path
182          *            the path to the transitions matrices
183          * @return a pool of transition matrices
184          * @throws FileNotFoundException
185          *             if a file cannot be found
186          * @throws IOException
187          *             if an error occurs while loading the data
188          */
189         protected Pool<float[][]> loadTransitionMatrices(String path) throws IOException, URISyntaxException {
190             logger.fine("Loading transition matrices from: " + path);
191     
192             InputStream inputStream = getDataStream(path);
193             if (inputStream == null) {
194                 throw new FileNotFoundException("Error trying to read file " + path);
195             }
196     
197             Pool<float[][]> pool = new Pool<float[][]>(path);
198             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream, '#', false);
199             est.expectString("tmat");
200             int numMatrices = est.getInt("numMatrices");
201             int numStates = est.getInt("numStates");
202             logger.fine("with " + numMatrices + " and " + numStates + " states");
203     
204             // read in the matrices
205             for (int i = 0; i < numMatrices; i++) {
206                 est.expectString("tmat");
207                 est.expectString("[" + i + ']');
208                 float[][] tmat = new float[numStates][numStates];
209                 for (int j = 0; j < numStates; j++) {
210                     for (int k = 0; k < numStates; k++) {
211                         // the last row is just zeros, so we just do
212                         // the first (numStates - 1) rows
213                         if (j < numStates - 1) {
214                             if (k == j || k == j + 1) {
215                                 tmat[j][k] = est.getFloat("tmat value");
216                             }
217                         }
218                         tmat[j][k] = logMath.linearToLog(tmat[j][k]);
219                         if (logger.isLoggable(Level.FINE)) {
220                             logger.fine("tmat j " + j + " k " + k + " tm " + tmat[j][k]);
221                         }
222                     }
223                 }
224                 pool.put(i, tmat);
225             }
226             est.close();
227             return pool;
228         }
