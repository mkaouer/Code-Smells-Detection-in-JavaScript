25     /**
26      * Implements a level tracking endpointer invented by Bent Schmidt Nielsen.
27      * <p/>
28      * <p>This endpointer is composed of three main steps. <ol> <li>classification of audio into speech and non-speech
29      * <li>inserting SPEECH_START and SPEECH_END signals around speech <li>removing non-speech regions </ol>
30      * <p/>
31      * <p>The first step, classification of audio into speech and non-speech, uses Bent Schmidt Nielsen's algorithm. Each
32      * time audio comes in, the average signal level and the background noise level are updated, using the signal level of
33      * the current audio. If the average signal level is greater than the background noise level by a certain threshold
34      * value (configurable), then the current audio is marked as speech. Otherwise, it is marked as non-speech.
35      * <p/>
36      * <p>The second and third step of this endpointer are documented in the classes {@link SpeechMarker SpeechMarker} and
37      * {@link NonSpeechDataFilter NonSpeechDataFilter}.
38      *
39      * @see SpeechMarker
40      */
41     public class SpeechClassifier extends AbstractVoiceActivityDetector {
42     
43         /** The property specifying the endpointing frame length in milliseconds. */
44         @S4Integer(defaultValue = 10)
45         public static final String PROP_FRAME_LENGTH_MS = "frameLengthInMs";
46     
47         /** The property specifying the minimum signal level used to update the background signal level. */
48         @S4Double(defaultValue = 0)
49         public static final String PROP_MIN_SIGNAL = "minSignal";
50     
51         /**
52          * The property specifying the threshold. If the current signal level is greater than the background level by
53          * this threshold, then the current signal is marked as speech. Therefore, a lower threshold will make the
54          * endpointer more sensitive, that is, mark more audio as speech. A higher threshold will make the endpointer less
55          * sensitive, that is, mark less audio as speech.
56          */
57         @S4Double(defaultValue = 10)
58         public static final String PROP_THRESHOLD = "threshold";
59     
60         /** The property specifying the adjustment. */
61         @S4Double(defaultValue = 0.003)
62         public static final String PROP_ADJUSTMENT = "adjustment";
63     
64         protected final double averageNumber = 1;
65         protected double adjustment;
66         protected double level;               // average signal level
67         protected double background;          // background signal level
68         protected double minSignal;           // minimum valid signal level
69         protected double threshold;
70         protected float frameLengthSec;
71         protected boolean isSpeech;
72     
73         /* Statistics */
74         protected long speechFrames;
75         protected long backgroundFrames;
76         protected double totalBackgroundLevel;
77         protected double totalSpeechLevel;
78         
79         public SpeechClassifier(int frameLengthMs, double adjustment, double threshold, double minSignal ) {
80             initLogger();
81             this.frameLengthSec = ((float) frameLengthMs) / 1000.f;
82     
83             this.adjustment = adjustment;
84             this.threshold = threshold;
85             this.minSignal = minSignal;
86     
87             initialize();
88         }

91         }

95             super.newProperties(ps);
96             int frameLengthMs = ps.getInt(PROP_FRAME_LENGTH_MS);
97             frameLengthSec = ((float) frameLengthMs) / 1000.f;

99             adjustment = ps.getDouble(PROP_ADJUSTMENT);
100             threshold = ps.getDouble(PROP_THRESHOLD);

104             //logger.setLevel(Level.FINEST);

107         }

110         /** Initializes this LevelTracker endpointer and DataProcessor predecessor. */
111         @Override
112         public void initialize() {
113             super.initialize();
114             reset();
115         }

118         /** Resets this LevelTracker to a starting state. */
119         protected void reset() {
120             level = 0;
121             background = 300;
122             resetStats();
123         }

126         /**
127          * Returns the logarithm base 10 of the root mean square of the given samples.
128          *
129          * @param samples the samples
130          * @return the calculated log root mean square in log 10
131          */
132         public static double logRootMeanSquare(double[] samples) {
133             assert samples.length > 0;
134             double sumOfSquares = 0.0f;
135             for (double sample : samples) {
136                 sumOfSquares += sample * sample;
137             }
138             double rootMeanSquare = Math.sqrt
139                     (sumOfSquares / samples.length);
140             rootMeanSquare = Math.max(rootMeanSquare, 1);
141             return (LogMath.log10((float) rootMeanSquare) * 20);
142         }
143     
144     
145         /**
146          * Classifies the given audio frame as speech or not, and updates the endpointing parameters.
147          *
148          * @param audio the audio frame
149          */
150         protected SpeechClassifiedData classify(DoubleData audio) {
151             double current = logRootMeanSquare(audio.getValues());
152             // System.out.println("current: " + current);
153             isSpeech = false;
154             if (current >= minSignal) {
155                 level = ((level * averageNumber) + current) / (averageNumber + 1);
156                 if (current < background) {
157                     background = current;
158                 } else {
159                     background += (current - background) * adjustment;
160                 }
161                 if (level < background) {
162                     level = background;
163                 }
164                 isSpeech = (level - background > threshold);
165             }
166     
167             SpeechClassifiedData labeledAudio = new SpeechClassifiedData(audio, isSpeech);
168     
169             if (logger.isLoggable(Level.FINEST)) {
170                 String speech = "";
171                 if (labeledAudio.isSpeech())
172                     speech = "*";
173     
174                 System.out.println("Bkg: " + background + ", level: " + level +
175                         ", current: " + current + ' ' + speech);
176             }
177     
178             collectStats (isSpeech);
179             
180             return labeledAudio;
181         }
182     
183         /**
184          * Reset statistics
185          */
186         private void resetStats () {
187             backgroundFrames = 1;
188             speechFrames = 1;
189             totalSpeechLevel = 0;
190             totalBackgroundLevel = 0;
191         }
192         
193         /**
194          * Collects the statistics to provide information about signal to noise ratio in channel
195          * 
196          * @param isSpeech if the current frame is classified as speech
197          */
198         private void collectStats(boolean isSpeech) {
199             if (isSpeech) {
200                 totalSpeechLevel = totalSpeechLevel + level;
201                 speechFrames = speechFrames + 1;
202             } else {
203                 totalBackgroundLevel = totalBackgroundLevel + background;
204                 backgroundFrames = backgroundFrames + 1;
205             }        
206         }
207     
208         /**
209          * Returns the next Data object.
210          *
211          * @return the next Data object, or null if none available
212          * @throws DataProcessingException if a data processing error occurs
213          */
214         @Override
215         public Data getData() throws DataProcessingException {
216             Data audio = getPredecessor().getData();
217     
218             if (audio instanceof DataStartSignal)
219                 reset();
220     
221             if (audio instanceof DoubleData) {
222                 DoubleData data = (DoubleData) audio;
223                 audio = classify(data);
224             }
225             return audio;
226         }
227         
228         /**
229          * Method that returns if current returned frame contains speech. 
230          * It could be used by noise filter for example to adjust noise 
231          * spectrum estimation.
232          * 
233          * @return if current frame is speech 
234          */
235         public boolean isSpeech() {
236         	return isSpeech;
237         }
238         
239         /** 
240          * Retrieves accumulated signal to noise ratio in dbScale 
241          * 
242          * @return signal to noise ratio
243          */
244         public double getSNR () {
245             double snr = (totalBackgroundLevel / backgroundFrames - totalSpeechLevel / speechFrames);
246             logger.fine ("Background " + totalBackgroundLevel / backgroundFrames);
247             logger.fine ("Speech " + totalSpeechLevel / speechFrames);
248             logger.fine ("SNR is " + snr);
249             return snr;
250         }
251      
252         /** 
253          * Return the estimation if input data was noisy enough to break
254          * recognition. The audio is counted noisy if signal to noise ratio
255          * is less then -20dB.
256          * 
257          * @return estimation of data being noisy
258          */
259         public boolean getNoisy () {
260             return (getSNR() > -20);
261         }
