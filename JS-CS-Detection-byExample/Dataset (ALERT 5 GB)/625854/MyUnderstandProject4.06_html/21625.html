30     import javax.sound.sampled.AudioSystem;
31     import javax.sound.sampled.UnsupportedAudioFileException;
32     
33     import org.junit.Test;
34     
35     public class MAPConfidenceTest {
36     
37     	@Test
38     	public void testMAPConfidence ()
39     			throws UnsupportedAudioFileException, IOException {
40     		URL audioFileURL = new File("src/test/edu/cmu/sphinx/result/test/green.wav")
41     				.toURI().toURL();
42     		URL configURL = new File("src/test/edu/cmu/sphinx/result/test/config.xml")
43     				.toURI().toURL();
44     
45     		ConfigurationManager cm = new ConfigurationManager(configURL);
46     		Recognizer recognizer = (Recognizer) cm.lookup("recognizer");
47     
48     		recognizer.allocate();
49     
50     		StreamDataSource reader = (StreamDataSource) cm
51     				.lookup("streamDataSource");
52     
53     		reader.setInputStream(AudioSystem.getAudioInputStream(audioFileURL),
54     				audioFileURL.getFile());
55     
56     		Result result = recognizer.recognize();
57     
58     		if (result != null) {
59     			ConfidenceScorer cs = (ConfidenceScorer) cm
60     					.lookup("confidenceScorer");
61     			ConfidenceResult cr = cs.score(result);
62     			Path best = cr.getBestHypothesis();
63     
64     			/* print out confidence of individual words in the best path */
65     			WordResult[] words = best.getWords();
66     			for (WordResult wr : words) {
67     				System.out.println(wr.getPronunciation().getWord()
68     						.getSpelling());
69     				System.out.println("   (confidence: "
70     						+ wr.getLogMath().logToLinear(
71     								(float) wr.getConfidence()) + ')');
72     			}
73     
74     			System.out.println();
75     
76     			/* confidence of the best path */
77     			System.out.println(best.getTranscription());
78     			System.out.println("   (confidence: "
79     					+ best.getLogMath().logToLinear(
80     							(float) best.getConfidence()) + ')');
81     		}
82     	}
83     }
