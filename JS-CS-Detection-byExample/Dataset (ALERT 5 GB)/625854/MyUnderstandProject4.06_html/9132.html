27     package javax.speech;
28

31     public final class SpeechLocale {
32         public static final SpeechLocale ENGLISH;
33     
34         public static final SpeechLocale US;
35         
36         public static final SpeechLocale FRENCH;
37     
38         public static final SpeechLocale GERMAN;
39     
40         
41         /**
42          * The default locale. Except for during bootstrapping, this should never be
43          * null. Note the logic in the main constructor, to detect when
44          * bootstrapping has completed.
45          */
46         private static SpeechLocale DEFAULT_LOCALE;
47     
48         static {
49             ENGLISH = new SpeechLocale("en");
50             US = new SpeechLocale("en", "US");
51             FRENCH = new SpeechLocale("fr");
52             GERMAN = new SpeechLocale("de");
53     
54             String defaultLanguage = System.getProperty("microedition.locale");
55             if (defaultLanguage == null) {
56                 defaultLanguage = "en";
57             }
58             DEFAULT_LOCALE = new SpeechLocale(defaultLanguage);
59         }
60     
61         private String language;
62     
63         private String country;
64     
65         private String variant;
66     
67         /**
68          * Convert new iso639 codes to the old ones.
69          * 
70          * @param language
71          *            the language to check
72          * @return the appropriate code
73          */
74         private String convertLanguage(String language) {
75             if (language.equals(""))
76                 return language;
77             language = language.toLowerCase();
78             int index = "he,id,yi".indexOf(language);
79             if (index != -1)
80                 return "iw,in,ji".substring(index, index + 2);
81             return language;
82         }
83     
84         public SpeechLocale(String language, String country, String variant) {
85             this.language = language;
86             this.country = country;
87             this.variant = variant;
88         }
89     
90         public SpeechLocale(String language, String country) {
91             this(language, country, "");
92         }
93     
94         public SpeechLocale(String language) {
95             this(language, "", "");
96         }
97     
98         public static SpeechLocale getDefault() {
99             return DEFAULT_LOCALE;
100         }
101     
102         public static SpeechLocale[] getAvailableLocales() {
103             return new SpeechLocale[] { ENGLISH, US, FRENCH, GERMAN };
104         }
105     
106         public String getLanguage() {
107             return language;
108         }
109     
110         public String getCountry() {
111             return country;
112         }
113     
114         public String getVariant() {
115             return variant;
116         }
117     
118         public int hashCode() {
119             final int prime = 31;
120             int result = 1;
121             result = prime * result + ((country == null) ? 0 : country.hashCode());
122             result = prime * result
123                     + ((language == null) ? 0 : language.hashCode());
124             result = prime * result + ((variant == null) ? 0 : variant.hashCode());
125             return result;
126         }
127     
128         public boolean equals(Object obj) {
129             if (this == obj) {
130                 return true;
131             }
132             if (obj == null) {
133                 return false;
134             }
135             if (getClass() != obj.getClass()) {
136                 return false;
137             }
138             SpeechLocale other = (SpeechLocale) obj;
139             if (country == null) {
140                 if (other.country != null) {
141                     return false;
142                 }
143             } else if (!country.equals(other.country)) {
144                 return false;
145             }
146             if (language == null) {
147                 if (other.language != null) {
148                     return false;
149                 }
150             } else if (!language.equals(other.language)) {
151                 return false;
152             }
153             if (variant == null) {
154                 if (other.variant != null) {
155                     return false;
156                 }
157             } else if (!variant.equals(other.variant)) {
158                 return false;
159             }
160             return true;
161         }
162     
163         public boolean match(SpeechLocale require) {
164             if (require == null) {
165                 return true;
166             }
167             if (require.country.length() > 0) {
168                 if (!require.country.equals(country)) {
169                     return false;
170                 }
171             }
172             if (require.language.length() > 0) {
173                 if (!require.language.equals(language)) {
174                     return false;
175                 }
176             }
177             if (require.variant.length() > 0) {
178                 if (!require.variant.equals(variant)) {
179                     return false;
180                 }
181             }
182             return true;
183         }
184         
185         public final String toString() {
186             if ((language.length() == 0) && (country.length() == 0)) {
187                 return "";
188             }
189     
190             StringBuffer str = new StringBuffer();
191             str.append(language);
192             if (country.length() != 0) {
193                 str.append('_').append(country);
194             }
195             if (variant.length() != 0) {
196                 str.append('_').append(variant);
197             }
198     
199             return str.toString();
200         }
