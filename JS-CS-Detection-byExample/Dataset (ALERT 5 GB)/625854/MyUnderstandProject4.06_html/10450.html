16     /**
17      * Computes the delta and double delta of input cepstrum (or plp or ...). The delta is the first order derivative and
18      * the double delta (a.k.a. delta delta) is the second order derivative of the original cepstrum. They help model the
19      * speech signal dynamics. The output data is a {@link FloatData} object with a float array of size three times the
20      * original cepstrum, formed by the concatenation of cepstra, delta cepstra, and double delta cepstra. The output is the
21      * feature vector used by the decoder. Figure 1 shows the arrangement of the output feature data array:
22      * <p/>
23      * <img src="doc-files/feature.jpg"> <br> <b>Figure 1: Layout of the returned features. </b>
24      * <p/>
25      * Suppose that the original cepstrum has a length of N, the first N elements of the feature are just the original
26      * cepstrum, the second N elements are the delta of the cepstrum, and the last N elements are the double delta of the
27      * cepstrum.
28      * <p/>
29      * Figure 2 below shows pictorially the computation of the delta and double delta of a cepstrum vector, using the last 3
30      * cepstra and the next 3 cepstra. <img src="doc-files/deltas.jpg"> <br> <b>Figure 2: Delta and double delta vector
31      * computation. </b>
32      * <p/>
33      * Referring to Figure 2, the delta is computed by subtracting the cepstrum that is two frames behind of the current
34      * cepstrum from the cepstrum that is two frames ahead of the current cepstrum. The computation of the double delta is
35      * similar. It is computed by subtracting the delta cepstrum one time frame behind from the delta cepstrum one time
36      * frame ahead. Replacing delta cepstra with cepstra, this works out to a formula involving the cepstra that are one and
37      * three behind and after the current cepstrum.
38      */
39     public class DeltasFeatureExtractor extends AbstractFeatureExtractor {
40     
41         /**
42          *
43          * @param window
44          */
45         public DeltasFeatureExtractor( int window ) {
46             super(window);
47         }
48     
49         public DeltasFeatureExtractor( ) {
50         }
51         
52         /**
53          * Computes the next feature. Advances the pointers as well.
54          *
55          * @return the feature Data computed
56          */
57         @Override
58         protected Data computeNextFeature() {
59     
60         	int jp1 = (currentPosition - 1 + cepstraBufferSize) % cepstraBufferSize;
61             int jp2 = (currentPosition - 2 + cepstraBufferSize) % cepstraBufferSize;
62             int jp3 = (currentPosition - 3 + cepstraBufferSize) % cepstraBufferSize;
63             int jf1 = (currentPosition + 1) % cepstraBufferSize;
64             int jf2 = (currentPosition + 2) % cepstraBufferSize;
65             int jf3 = (currentPosition + 3) % cepstraBufferSize;
66             
67         	DoubleData currentCepstrum = cepstraBuffer[currentPosition];
68             double[] mfc3f = cepstraBuffer[jf3].getValues();
69             double[] mfc2f = cepstraBuffer[jf2].getValues();
70             double[] mfc1f = cepstraBuffer[jf1].getValues();
71             double[] current = currentCepstrum.getValues();
72             double[] mfc1p = cepstraBuffer[jp1].getValues();
73             double[] mfc2p = cepstraBuffer[jp2].getValues();
74             double[] mfc3p = cepstraBuffer[jp3].getValues();
75             float[] feature = new float[current.length * 3];
76     
77             currentPosition = (currentPosition + 1) % cepstraBufferSize;
78     
79             // CEP; copy all the cepstrum data
80             int j = 0;
81             for (double val : current) {
82                 feature[j++] = (float)val;
83             }
84             // System.arraycopy(current, 0, feature, 0, j);
85             // DCEP: mfc[2] - mfc[-2]
86             for (int k = 0; k < mfc2f.length; k++) {
87                 feature[j++] = (float) (mfc2f[k] - mfc2p[k]);
88             }
89             // D2CEP: (mfc[3] - mfc[-1]) - (mfc[1] - mfc[-3])
90             for (int k = 0; k < mfc3f.length; k++) {
91                 feature[j++] = (float) ((mfc3f[k] - mfc1p[k]) - (mfc1f[k] - mfc3p[k]));
92             }
93             return (new FloatData(feature,
94                     currentCepstrum.getSampleRate(),
95                     currentCepstrum.getCollectTime(),
96                     currentCepstrum.getFirstSampleNumber()));
97         }
