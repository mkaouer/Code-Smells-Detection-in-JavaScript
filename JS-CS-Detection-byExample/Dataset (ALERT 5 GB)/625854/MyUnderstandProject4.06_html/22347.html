31     public class Word {
32         public static final long UNKNOWN = 0x00000;
33     
34         public static final long DONT_CARE = 0x00001;
35     
36         public static final long OTHER = 0x00002;
37     
38         public static final long NOUN = 0x00004;
39     
40         public static final long PROPER_NOUN = 0x00008;
41     
42         public static final long PRONOUN = 0x00010;
43     
44         public static final long VERB = 0x00020;
45     
46         public static final long ADVERB = 0x00040;
47     
48         public static final long ADJECTIVE = 0x00080;
49     
50         public static final long PROPER_ADJECTIVE = 0x00100;
51     
52         public static final long AUXILIARY = 0x00200;
53     
54         public static final long DETERMINER = 0x00400;
55     
56         public static final long CARDINAL = 0x00800;
57     
58         public static final long CONJUNCTION = 0x01000;
59     
60         public static final long PREPOSITION = 0x02000;
61     
62         public static final long CONTRACTION = 0x04000;
63     
64         public static final long ABBREVIATION = 0x08000;
65     
66         public static final long ACOUSTIC = 0x10000;
67     
68         private String text;
69     
70         private String[] pronunciations;
71     
72         private String spokenForm;
73     
74         private AudioSegment audioSegment;
75     
76         private long categories;
77     
78         private SpeechLocale locale;
79     
80         public Word(String text, String[] pronunciations, String spokenForm,
81                 AudioSegment audioSegment, long categories)
82             throws IllegalArgumentException {
83             this(text, pronunciations, spokenForm, audioSegment, categories, null);
84         }
85     
86         public Word(String text, String[] pronunciations, String spokenForm,
87                 AudioSegment audioSegment, long categories, SpeechLocale locale)
88             throws IllegalArgumentException {
89             if (text == null) {
90                 throw new IllegalArgumentException(
91                         "Written form text must be specified");
92             }
93             this.text = text;
94             this.pronunciations = pronunciations;
95             this.spokenForm = spokenForm;
96             this.audioSegment = audioSegment;
97             this.categories = categories;
98             this.locale = locale;
99         }
100     
101         public AudioSegment getAudioSegment() {
102             return audioSegment;
103         }
104     
105         public long getCategories() {
106             return categories;
107         }
108     
109         public String[] getPronunciations() {
110             return pronunciations;
111         }
112     
113         public String getSpokenForm() {
114             return spokenForm;
115         }
116     
117         public String getText() {
118             return text;
119         }
120     
121         public SpeechLocale getSpeechLocale() {
122             return locale;
123         }
124     
125         public String toString() {
126             final StringBuffer str = new StringBuffer();
127     
128             str.append(getClass());
129             str.append("[");
130     
131             str.append(text);
132     
133             if (pronunciations == null) {
134                 str.append(pronunciations);
135             } else {
136                 str.append("[");
137                 int max = pronunciations.length;
138                 for (int i = 0; i < max; i++) {
139                     str.append(pronunciations[i]);
140                     if (i != max - 1) {
141                         str.append(",");
142                     }
143                 }
144                 str.append("]");
145             }
146     
147             str.append(spokenForm);
148             str.append(audioSegment);
149             str.append(categories);
150     
151             str.append("]");
152     
153             return str.toString();
154         }
155     }
