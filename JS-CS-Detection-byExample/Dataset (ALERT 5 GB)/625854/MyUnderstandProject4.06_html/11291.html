12     package org.jvoicexml.jsapi2;
13     
14     import java.util.Enumeration;
15     import java.util.Vector;
16     
17     import javax.speech.Engine;
18     import javax.speech.EngineProperties;
19     import javax.speech.EnginePropertyEvent;
20     import javax.speech.EnginePropertyListener;
21     import javax.speech.SpeechEventExecutor;
22     
23     /**
24      * Supports the JSAPI 2.0 <code>EngineProperties</code> interface.
25      * 
26      * <p>
27      * Actual JSAPI 2 implementations may want to override this class to apply the
28      * settings to the engine.
29      * </p>
30      *
31      * <p>
32      * Engines are requested to implement the
33      * {@link EnginePropertyChangeRequestListener} interface and register
34      * to this object via the
35      * {@link #addEnginePropertyChangeRequestListener(EnginePropertyChangeRequestListener)}
36      * method to get notifications about change requests-
37      * Properties are not set directly. Instead the
38      * {@link EnginePropertyChangeRequestListener} is notified about a change
39      * request.
40      * Once a requested property
41      * is changed the engine must call the callback method
42      * {@link #commitPropertyChange(String, Object, Object)} to have all
43      * registered {@link EnginePropertyListener}s notified about the change.
44      * </p>
45      *
46      * @author Renato Cassaca
47      * @author Dirk Schnelle-Walka
48      * @version $Revision: 295 $
49      */
50     public abstract class BaseEngineProperties implements EngineProperties {
51         /** Name of the base property in events. */
52         public static final String BASE = "base";
53     
54         /** Name of the priority property in events. */
55         public static final String PRIORITY = "priority";
56     
57         /**
58          * List of <code>PropertyChangeListeners</code> registered for
59          * <code>PropertyChangeEvents</code> on this object.
60          */
61         private final Vector propertyChangeListeners;
62     
63         /**
64          * List of {@link EnginePropertyChangeRequestListener} registered for
65          * {@link EnginePropertyChangeRequestEvent} on this object.
66          */
67         private final Vector propertyChangeRequestListeners;
68     
69         /** The engine for which these properties apply. */
70         private final Engine engine;
71     
72         /**
73          * The base location to resolve relative URLs against for this Engine
74          * instance.
75          */
76         private String base;
77     
78         /** The priority for this engine instance. */
79         private int priority;
80     
81         /**
82          * Constructs a new object.
83          * 
84          * @param eng
85          *            the engine for which these properties apply.
86          */
87         protected BaseEngineProperties(final Engine eng) {
88             propertyChangeListeners = new Vector();
89             propertyChangeRequestListeners = new Vector();
90             engine = eng;
91             priority = EngineProperties.NORM_TRUSTED_PRIORITY;
92             base = "";
93         }
94     
95         /**
96          * Retrieves the engine.
97          * 
98          * @return the engine
99          */
100         protected final Engine getEngine() {
101             return engine;
102         }
103     
104         /**
105          * Returns all properties to reasonable defaults for the <code>Engine</code>
106          * . A <code>PropertyChangeEvent</code> is issued for each property that
107          * changes as the reset takes effect.
108          */
109         public void reset() {
110             setPriority(EngineProperties.NORM_TRUSTED_PRIORITY);
111             setBase("");
112         }
113     
114         /**
115          * {@inheritDoc}
116          */
117         public int getPriority() {
118             return priority;
119         }
120     
121         /**
122          * {@inheritDoc}
123          */
124         public void setPriority(final int prio) {
125             postPropertyChangeRequestEvent(PRIORITY, new Integer(this.priority),
126                     new Integer(prio));
127         }
128     
129         /**
130          * {@inheritDoc}
131          */
132         public void setBase(final String uri) {
133             postPropertyChangeRequestEvent(BASE, new Integer(this.priority),
134                     new Integer(priority));
135         }
136     
137         /**
138          * {@inheritDoc}
139          */
140         public String getBase() {
141             return base;
142         }
143     
144         /**
145          * {@inheritDoc}
146          */
147         public void addEnginePropertyListener(
148                 final EnginePropertyListener listener) {
149             if (!propertyChangeListeners.contains(listener)) {
150                 propertyChangeListeners.addElement(listener);
151             }
152         }
153     
154         /**
155          * {@inheritDoc}
156          */
157         public void removeEnginePropertyListener(
158                 final EnginePropertyListener listener) {
159             propertyChangeListeners.removeElement(listener);
160         }
161     
162         /**
163          * Adds the given {@link EnginePropertyChangeRequestListener} to the list of
164          * known listeners.
165          * 
166          * @param listener
167          *            the listener to add
168          */
169         public void addEnginePropertyChangeRequestListener(
170                 final EnginePropertyChangeRequestListener listener) {
171             if (!propertyChangeRequestListeners.contains(listener)) {
172                 propertyChangeRequestListeners.addElement(listener);
173             }
174         }
175     
176         /**
177          * {@inheritDoc}
178          */
179         public void removeEnginePropertyChangeRequestListener(
180                 final EnginePropertyChangeRequestListener listener) {
181             propertyChangeRequestListeners.removeElement(listener);
182         }
183     
184         /**
185          * Commit the property changes and sets the value.
186          * <p>
187          * This method is called after the new values are applied to the engine.
188          * </p>
189          * @param propName
190          * @param oldValue
191          * @param newValue
192          */
193         public void commitPropertyChange(final String propName,
194                 final Object oldValue, final Object newValue) {
195             if (propName == null) {
196                 throw new IllegalArgumentException(
197                         "Property name must not be null!");
198             }
199             if (propName.equals(PRIORITY)) {
200                 final Integer value = (Integer) newValue;
201                 priority = value.intValue();
202             } else if (propName.equals("base")) {
203                 base = (String) newValue;
204             } else {
205                 final boolean set = setProperty(propName, newValue);
206                 if (!set) {
207                     throw new IllegalArgumentException("Unknown property '"
208                             + propName + "'");
209                 }
210             }
211             postPropertyChangeEvent(propName, oldValue, newValue);
212         }
213     
214         /**
215          * Must be overwritten to set engine specific properties which are not
216          * covered by the standard after the values are applied to the engine.
217          * @param propName name of the property
218          * @param value value to set
219          * @return <code>true</code> if the value has been set.
220          */
221         protected abstract boolean setProperty(final String propName,
222                 final Object value);
223     
224         /**
225          * Synchronously notifies the registered listeners about the property change
226          * request.
227          * 
228          * @param propName
229          *            name of the property
230          * @param oldValue
231          *            old value of the property
232          * @param newValue
233          *            requested value of the property
234          */
235         protected void postPropertyChangeRequestEvent(final String propName,
236                 final Object oldValue, final Object newValue) {
237             final EnginePropertyChangeRequestEvent event =
238                 new EnginePropertyChangeRequestEvent(
239                     this, propName, oldValue, newValue);
240             final Enumeration e = propertyChangeRequestListeners.elements();
241             while (e.hasMoreElements()) {
242                 final EnginePropertyChangeRequestListener listener
243                     = (EnginePropertyChangeRequestListener) e.nextElement();
244                 listener.propertyChangeRequest(event);
245             }
246         }
247     
248         /**
249          * Generates a {@link PropertyChangeEvent} for an <code>Object</code> value
250          * and posts it to the event queue using the configured
251          * {@link SpeechEventExecutor}.
252          * 
253          * <p>
254          * Registered listeners are notified using the
255          * {@link #firePropertyChangeEvent(PropertyChangeEvent)} method.
256          * </p>
257          * 
258          * @param propName
259          *            the name of the property
260          * @param oldValue
261          *            the old value
262          * @param newValue
263          *            the new value
264          * 
265          * @see #firePropertyChangeEvent
266          * @see #dispatchSpeechEvent
267          */
268         protected void postPropertyChangeEvent(final String propName,
269                 final Object oldValue, final Object newValue) {
270     
271             if (propertyChangeListeners.size() < 1) {
272                 return;
273             }
274     
275             final EnginePropertyEvent event = new EnginePropertyEvent(this,
276                     propName, oldValue, newValue);
277     
278             final Runnable runnable = new Runnable() {
279                 public void run() {
280                     firePropertyChangeEvent(event);
281                 }
282             };
283     
284             final SpeechEventExecutor executor = engine.getSpeechEventExecutor();
285             executor.execute(runnable);
286         }
287     
288         /**
289          * Sends a {@link PropertyChangeEvent} to all
290          * <code>PropertyChangeListeners</code> registered with this object.
291          * 
292          * <p>
293          * This method runs within the configured {@link SpeechEventExecutor}.
294          * </p>
295          * 
296          * @param event
297          *            the <code>PropertyChangeEvent</code> to send
298          * 
299          * @see #firePropertyChangeEvent
300          * @see #dispatchSpeechEvent
301          */
302         public void firePropertyChangeEvent(final EnginePropertyEvent event) {
303             final Enumeration e = propertyChangeListeners.elements();
304             while (e.hasMoreElements()) {
305                 final EnginePropertyListener listener = (EnginePropertyListener) e
306                         .nextElement();
307                 listener.propertyUpdate(event);
308             }
309         }
310     }
