14     package edu.cmu.sphinx.linguist.acoustic.tiedstate;
15     
16     import java.io.IOException;
17     import java.util.ArrayList;
18     import java.util.Collections;
19     import java.util.List;
20     import java.util.Properties;
21     
22     /** Dumps out information about an acoustic model. */
23     
24     public class PropertiesDumper {
25     
26         private final Properties props;
27     
28         /** Dumps the properties file 'model.props' that is in the same directory as this class.
29          * @param argv*/
30         public static void main(String[] argv) {
31             try {
32                 PropertiesDumper dumper = new PropertiesDumper("model.props");
33                 System.out.println();
34                 System.out.println(dumper);
35             } catch (IOException ioe) {
36                 ioe.printStackTrace();
37             }
38         }
39     
40     
41         /**
42          * Constructs a PropertiesDumper of the given acoustic model properties file.
43          *
44          * @param propsFile the properties file to dump
45          * @throws java.io.IOException
46          */
47         public PropertiesDumper(String propsFile) throws IOException {
48             props = new Properties();
49             props.load(getClass().getResource(propsFile).openStream());
50         }
51     
52     
53         /**
54          * Constructs a PropertiesDumper of the given acoustic model properties.
55          *
56          * @param properties the Properties object to dump
57          * @throws java.io.IOException
58          */
59         public PropertiesDumper(Properties properties) throws IOException {
60             props = properties;
61         }
62     
63     
64         /**
65          * Returns a string of the properties.
66          *
67          * @return a string of the properties
68          */
69         @Override
70         public String toString() {
71             StringBuilder result = new StringBuilder().append(props.get("description")).append('\n');
72             List<?> propnames = Collections.list(props.propertyNames());
73             List<String> list = new ArrayList<String>();
74             for (Object obj : propnames) {
75                 if (obj instanceof String) {
76                     list.add((String)obj);
77                 }
78             }
79             Collections.sort(list);
80             for (String key : list) {
81                 String value = (String)props.get(key);
82                 result.append("\n\t").append(key).append(": ").append(value);
83             }
84             result.append('\n');
85             return result.toString();
86         }
87     }
