21     /**
22      * Creates a Lattice from a GDL (AISee) Lattice file. One can obtain such a GDL file from a lattice by calling the
23      * <code>Lattice.dumpAISee</code> method.
24      */
25     public class GDLLatticeFactory {
26     
27         private GDLLatticeFactory() {
28         }
29     
30     
31         /**
32          * Create a Lattice from a GDL (AISee) Lattice file.
33          *
34          * @param gdlFile    the lattice file
35          * @param dictionary the dictionary to use to look up words
36          */
37         public static Lattice getLattice(String gdlFile, Dictionary dictionary)
38                 throws IOException {
39             Lattice lattice = new Lattice();
40     
41             BufferedReader reader = new BufferedReader(new FileReader(gdlFile));
42             String line = null;
43     
44             while ((line = reader.readLine()) != null) {
45                 if (line.startsWith("node")) {
46                     createNode(line, lattice, dictionary);
47                 } else if (line.startsWith("edge")) {
48                     createEdge(line, lattice);
49                 }
50             }
51     
52             return lattice;
53         }
54     
55     
56         private static void createNode(String line, Lattice lattice,
57                                        Dictionary dictionary) {
58             String[] text = line.split("\\s");
59             String id = text[3].substring(1, text[3].length() - 1);
60             String contents = text[5].substring(1);
61             String posterior = text[6].substring(2, text[6].length() - 2);
62     
63             String word = contents.substring(0, contents.indexOf('['));
64             contents = contents.substring(contents.indexOf('[') + 1);
65     
66             String start = contents.substring(0, contents.indexOf(','));
67             String end = contents.substring(contents.indexOf(',') + 1);
68     
69             Node node = lattice.addNode(id, dictionary.getWord(word),
70                     Integer.parseInt(start),
71                     Integer.parseInt(end));
72             node.setPosterior(Double.parseDouble(posterior));
73     
74             if (word.equals("<s>")) {
75                 lattice.setInitialNode(node);
76             } else if (word.equals("</s>")) {
77                 lattice.setTerminalNode(node);
78             }
79         }
80     
81     
82         private static void createEdge(String line, Lattice lattice) {
83             String[] text = line.split("\\s");
84             String src = text[3].substring(1, text[3].length() - 1);
85             String dest = text[5].substring(1, text[5].length() - 1);
86             String contents = text[7].substring(1, text[7].length() - 1);
87             String[] scores = contents.split(",");
88     
89             lattice.addEdge(lattice.getNode(src), lattice.getNode(dest),
90                     Double.parseDouble(scores[0]),
91                     Double.parseDouble(scores[1]));
92         }
93     }
