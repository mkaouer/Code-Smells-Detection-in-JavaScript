27     package javax.speech.recognition;
28     
29     //Comp 2.0.6
30     
31     public class Rule {
32         public static int PUBLIC = 0x1;
33     
34         public static int PRIVATE = 0x2;
35     
36         private String ruleName;
37     
38         private RuleComponent ruleComponent;
39     
40         private int scope;
41     
42         public Rule(String ruleName, RuleComponent ruleComponent)
43             throws IllegalArgumentException {
44             this(ruleName, ruleComponent, PRIVATE);
45         }
46     
47         public Rule(String ruleName, RuleComponent ruleComponent, int scope) {
48             RuleComponent.checkValidGrammarText(ruleName);
49     
50             if ((scope != PRIVATE) && (scope != PUBLIC)) {
51                 throw new IllegalArgumentException(
52                         "Scope must be either PRIVATE or PUBLIC!");
53             }
54             if (ruleComponent == null) {
55                 throw new IllegalArgumentException(
56                         "Rule component must not be null!");
57             }
58             this.ruleName = ruleName;
59             this.ruleComponent = ruleComponent;
60             this.scope = scope;
61         }
62     
63         public RuleComponent getRuleComponent() {
64             return ruleComponent;
65         }
66     
67         public String getRuleName() {
68             return ruleName;
69         }
70     
71         public int getScope() {
72             return scope;
73         }
74     
75         public String toString() {
76             StringBuffer str = new StringBuffer();
77     
78             str.append("<rule id=\"");
79             str.append(ruleName);
80             str.append("\" scope=\"");
81             if (scope == PRIVATE) {
82                 str.append("private");
83             } else {
84                 str.append("public");
85             }
86             str.append("\">");
87             str.append(ruleComponent);
88             str.append("</rule>");
89             return str.toString();
90         }
91     }
