28     /**
29      * Defines a simple grammar based upon a language model. It generates one {@link GrammarNode grammar node}per word. This
30      * grammar can deal with unigram and bigram grammars of up to 1000 or so words. Note that all probabilities are in the
31      * log math domain.
32      */
33     public class LMGrammar extends Grammar {
34     
35         /** The property for the language model to be used by this grammar */
36         @S4Component(type = LanguageModel.class)
37         public final static String PROP_LANGUAGE_MODEL = "languageModel";
38         // ------------------------
39         // Configuration data
40         // ------------------------
41         private LanguageModel languageModel;
42     
43         public LMGrammar(LanguageModel languageModel, boolean showGrammar, boolean optimizeGrammar, boolean addSilenceWords, boolean addFillerWords, Dictionary dictionary) {
44             super(showGrammar,optimizeGrammar,addSilenceWords,addFillerWords,dictionary);
45             this.languageModel = languageModel;
46         }
47     
48         public LMGrammar() {
49     
50         }
51     
52         /*
53         * (non-Javadoc)
54         *
55         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
56         */
57         @Override
58         public void newProperties(PropertySheet ps) throws PropertyException {
59             super.newProperties(ps);
60             languageModel = (LanguageModel) ps.getComponent(PROP_LANGUAGE_MODEL);
61         }
62     
63     
64         /**
65          * Creates the grammar from the language model. This Grammar contains one word per grammar node. Each word (and
66          * grammar node) is connected to all other words with the given probability
67          *
68          * @return the initial grammar node
69          */
70         @Override
71         protected GrammarNode createGrammar() throws IOException {
72             languageModel.allocate();
73             TimerPool.getTimer(this,"LMGrammar.create").start();
74             GrammarNode firstNode = null;
75             if (languageModel.getMaxDepth() > 2) {
76                 System.out.println("Warning: LMGrammar  limited to bigrams");
77             }
78             List<GrammarNode> nodes = new ArrayList<GrammarNode>();
79             Set<String> words = languageModel.getVocabulary();
80             // create all of the word nodes
81             for (String word : words) {
82                 GrammarNode node = createGrammarNode(word);
83                 if (node != null && !node.isEmpty()) {
84                     if (node.getWord().equals(
85                             getDictionary().getSentenceStartWord())) {
86                         firstNode = node;
87                     } else if (node.getWord().equals(
88                             getDictionary().getSentenceEndWord())) {
89                         node.setFinalNode(true);
90                     }
91                     nodes.add(node);
92                 }
93             }
94             if (firstNode == null) {
95                 throw new Error("No sentence start found in language model");
96             }
97             for (GrammarNode prevNode : nodes) {
98                 // don't add any branches out of the final node
99                 if (prevNode.isFinalNode()) {
100                     continue;
101                 }
102                 for (GrammarNode nextNode : nodes) {
103                     String prevWord = prevNode.getWord().getSpelling();
104                     String nextWord = nextNode.getWord().getSpelling();
105                     Word[] wordArray = {getDictionary().getWord(prevWord),
106                             getDictionary().getWord(nextWord)};
107                     float logProbability = languageModel
108                             .getProbability((new WordSequence(wordArray)));
109                     prevNode.add(nextNode, logProbability);
110                 }
111             }
112             TimerPool.getTimer(this,"LMGrammar.create").stop();
113             languageModel.deallocate();
114             return firstNode;
115         }
116     }
