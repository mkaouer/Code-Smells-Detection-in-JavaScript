12     package org.jvoicexml.jsapi2.jse;
13     
14     import java.util.Vector;
15     
16     import org.jvoicexml.jsapi2.TerminatableSpeechEventExecutor;
17     
18     /**
19      * A speech event executor that is based on a thread.
20      *
21      * <p>
22      * There is only one single thread that is responsible to execute the
23      * commands asynchronously.
24      * </p>
25      * @author Renato Cassaca
26      * @version $Revision: 627 $
27      */
28     public final class ThreadSpeechEventExecutor
29         implements TerminatableSpeechEventExecutor, Runnable {
30         /** Number of msec to wait before inspecting the command queue. */
31         private static final int COMMAND_POLL_INTERVALL = 1000;
32     
33         /** The thread that executes the commands. */
34         private final Thread thread;
35     
36         /** Commands to execute. */
37         private final Vector commands;
38     
39         /** <code>false</code> if the executor is terminating. */
40         private boolean shouldRun;
41     
42         /**
43          * Constructs a new object.
44          */
45         public ThreadSpeechEventExecutor() {
46             commands = new Vector();
47             thread = new Thread(this);
48             shouldRun = true;
49             thread.start();
50         }
51     
52         /**
53          * {@inheritDoc}
54          *
55          * Terminates the execution thread.
56          */
57         protected void finalize() {
58             terminate();
59         }
60     
61         /**
62          * {@inheritDoc}
63          */
64         public void terminate() {
65             shouldRun = false;
66             synchronized (commands) {
67                 commands.notifyAll();
68             }
69         }
70     
71         /**
72          * Executes the given command.
73          *
74          * @param command the command to execute.
75          */
76         public void execute(final Runnable command) {
77             if (command == null) {
78                 throw new NullPointerException("Command must not be null!");
79             }
80             if (!shouldRun) {
81                 throw new IllegalStateException(
82                         "SpeechEventExecutor is terminated!");
83             }
84             commands.addElement(command);
85             synchronized (commands) {
86                 commands.notify();
87             }
88         }
89     
90         /**
91          * {@inheritDoc}
92          */
93         public void run() {
94             while (shouldRun) {
95                 while ((commands.isEmpty()) && (shouldRun)) {
96                     synchronized (commands) {
97                         try {
98                             commands.wait(COMMAND_POLL_INTERVALL);
99                         } catch (InterruptedException ex) {
100                             return;
101                         }
102                     }
103                 }
104                 if (!shouldRun) {
105                     return;
106                 }
107     
108                 //Use this thread to run the command
109                 final Runnable command = (Runnable) commands.firstElement();
110                 commands.removeElementAt(0);
111                 command.run();
112             }
113         }
114     }
