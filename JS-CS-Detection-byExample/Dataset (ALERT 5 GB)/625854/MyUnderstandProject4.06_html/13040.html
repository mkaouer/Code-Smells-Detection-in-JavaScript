21     import java.io.File;
22     import java.io.IOException;
23     import java.net.URL;
24     import java.util.Map;
25     import java.util.HashMap;

27     /**
28      * This class implements the process of reading from XML file .config.xml
29      *
30      * @author Ariani
31      * @see GUIReader
32      */
33     public class XMLConfigReader implements GUIReader {
34         
35         /* Singleton pattern, with this details:
36          public class Singleton {
37              // Private constructor suppresses generation of a (public) default constructor
38             private Singleton() {}
39      
40             private static class SingletonHolder {
41                 private static Singleton instance = new Singleton();
42             } 
43      
44             public static Singleton getInstance() {
45                 return SingletonHolder.instance;
46             }
47          }
48          */
49         
50           /**
51            * This class uses Singleton pattern for the constructor 
52            * to make sure that there is only one instance of reader created
53            * to get the reader instance, use <code>XMLConfigReader.getInstance()</code>
54            *       
55            */
56           private XMLConfigReader() {}
57     
58           private static class XMLReaderHolder {
59               private static final XMLConfigReader instance = new XMLConfigReader();
60           }
61           
62           /**
63            * Singleton constructor 
64            *
65            * @return    the xml reader
66            */
67           public static XMLConfigReader getInstance(){
68               return XMLReaderHolder.instance;
69           }
70         
71           /**
72            * The method called for reading from xml file
73            *
74            * @param     fFile the File to be read
75            * @return    <code>ConfigProperties</code> object that holds the property type-value 
76            * @throws    GUIReaderExceptoin  if there are any errors while finding
77            *            and reading from file.
78            */
79           @Override
80           public ConfigProperties read(File fFile) throws GUIReaderException {
81             if (fFile == null || fFile.getName().trim().isEmpty()) /* no filename for input */
82             {
83                 throw new GUIReaderException
84                         ("No input filename specified",GUIReaderException.EXCEPTION_NO_FILENAME);
85             }
86             else /* start reading from config file */
87             {
88                 try{          
89                    URL url = fFile.toURI().toURL();
90                    
91                    Map<String, String> global = new HashMap<String, String>();
92                    Map<String, RawPropertyData> pm = new SaxLoader(url, global).load();
93                    ConfigProperties cp = new ConfigProperties ();
94                    cp.setGlobal(global);
95                    cp.setProperty(pm);
96                    return cp;
97                    
98                 }catch(IOException e){
99                     throw new GUIReaderException("IO Exception during read",GUIReaderException.EXCEPTION_IO);
100                 }
101             }
102         }
103     }
