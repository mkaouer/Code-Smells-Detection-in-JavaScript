14     
15     import edu.cmu.sphinx.decoder.scorer.Scoreable;
16     import edu.cmu.sphinx.linguist.WordSearchState;
17     import edu.cmu.sphinx.linguist.dictionary.Word;

19     import edu.cmu.sphinx.util.props.PropertySheet;

23     import java.util.*;

25     /**
26      * A factory for WordActiveList. The word active list is active list designed to hold word tokens only. In addition to
27      * the usual active list properties such as absolute and relative beams, the word active list allows restricting the
28      * number of copies of any particular word in the word beam.  Also the word active list can restrict the number of
29      * fillers in the beam.
30      */
31     public class WordActiveListFactory extends ActiveListFactory {
32     
33         /** property that sets the max paths for a single word. (zero disables this feature) */
34         @S4Integer(defaultValue = 0)
35         public final static String PROP_MAX_PATHS_PER_WORD = "maxPathsPerWord";
36     
37         /** property that sets the max filler words allowed in the beam. (zero disables this feature) */
38         @S4Integer(defaultValue = 1)
39         public final static String PROP_MAX_FILLER_WORDS = "maxFillerWords";
40     
41         private int maxPathsPerWord;
42         private int maxFiller;
43     
44         /**
45          * 
46          * @param absoluteBeamWidth
47          * @param relativeBeamWidth
48          * @param logMath
49          * @param maxPathsPerWord
50          * @param maxFiller
51          */
52         public WordActiveListFactory(int absoluteBeamWidth, double relativeBeamWidth, LogMath logMath, int maxPathsPerWord, int maxFiller ) {
53             super(absoluteBeamWidth, relativeBeamWidth, logMath);
54             this.maxPathsPerWord = maxPathsPerWord;
55             this.maxFiller = maxFiller;
56         }
57     
58         public WordActiveListFactory() {
59         }
60     
61     
62         /*
63         * (non-Javadoc)
64         *
65         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
66         */
67         @Override
68         public void newProperties(PropertySheet ps) throws PropertyException {
69             super.newProperties(ps);
70     
71             maxPathsPerWord = ps.getInt(PROP_MAX_PATHS_PER_WORD);
72             maxFiller = ps.getInt(PROP_MAX_FILLER_WORDS);
73         }
74     
75     
76         /*
77         * (non-Javadoc)
78         *
79         * @see edu.cmu.sphinx.decoder.search.ActiveListFactory#newInstance()
80         */
81         @Override
82         public ActiveList newInstance() {
83             return new WordActiveList();
84         }
85     
86     
87         /**
88          * An active list that manages words. Guarantees only one version of a word.
89          * <p/>
90          * <p/>
91          * Note that all scores are maintained in the LogMath log domain
92          */
93         class WordActiveList implements ActiveList {
94     
95             private Token bestToken;
96             private List<Token> tokenList = new LinkedList<Token>();
97     
98     
99             /**
100              * Adds the given token to the list
101              *
102              * @param token the token to add
103              */
104             @Override
105             public void add(Token token) {
106                 tokenList.add(token);
107                 if (bestToken == null || token.getScore() > bestToken.getScore()) {
108                     bestToken = token;
109                 }
110             }
111     
112     
113             /**
114              * Replaces an old token with a new token
115              *
116              * @param oldToken the token to replace (or null in which case, replace works like add).
117              * @param newToken the new token to be placed in the list.
118              */
119             @Override
120             public void replace(Token oldToken, Token newToken) {
121                 add(newToken);
122                 if (oldToken != null) {
123                     tokenList.remove(oldToken);
124                 }
125             }
126     
127     
128             /**
129              * Purges excess members. Remove all nodes that fall below the relativeBeamWidth
130              *
131              * @return a (possible new) active list
132              */
133     
134             @Override
135             public ActiveList purge() {
136                 int fillerCount = 0;
137                 Map<Word, Integer> countMap = new HashMap<Word, Integer>();
138                 Collections.sort(tokenList, Scoreable.COMPARATOR);
139                 // remove word duplicates
140                 for (Iterator<Token> i = tokenList.iterator(); i.hasNext();) {
141                     Token token = i.next();
142                     WordSearchState wordState = (WordSearchState)token.getSearchState();
143     
144                     Word word = wordState.getPronunciation().getWord();
145     
146                     // only allow  maxFiller words
147                     if (maxFiller > 0) {
148                         if (word.isFiller()) {
149                             if (fillerCount < maxFiller) {
150                                 fillerCount++;
151                             } else {
152                                 i.remove();
153                                 continue;
154                             }
155                         }
156                     }
157     
158                     if (maxPathsPerWord > 0) {
159                         Integer count = countMap.get(word);
160                         int c = count == null ? 0 : count;
161     
162                         // Since the tokens are sorted by score we only
163                         // keep the n tokens for a particular word
164     
165                         if (c < maxPathsPerWord - 1) {
166                             countMap.put(word, c + 1);
167                         } else {
168                             i.remove();
169                         }
170                     }
171                 }
172     
173                 if (tokenList.size() > absoluteBeamWidth) {
174                     tokenList = tokenList.subList(0, absoluteBeamWidth);
175                 }
176     
177                 return this;
178             }
179     
180     
181             /**
182              * Retrieves the iterator for this tree.
183              *
184              * @return the iterator for this token list
185              */
186             @Override
187             public Iterator<Token> iterator() {
188                 return tokenList.iterator();
189             }
190     
191     
192             /**
193              * Gets the set of all tokens
194              *
195              * @return the set of tokens
196              */
197             @Override
198             public List<Token> getTokens() {
199                 return tokenList;
200             }
201     
202     
203             /**
204              * Returns the number of tokens on this active list
205              *
206              * @return the size of the active list
207              */
208             @Override
209             public final int size() {
210                 return tokenList.size();
211             }
212     
213     
214             /**
215              * gets the beam threshold best upon the best scoring token
216              *
217              * @return the beam threshold
218              */
219             @Override
220             public float getBeamThreshold() {
221                 return getBestScore() + logRelativeBeamWidth;
222             }
223     
224     
225             /**
226              * gets the best score in the list
227              *
228              * @return the best score
229              */
230             @Override
231             public float getBestScore() {
232                 float bestScore = -Float.MAX_VALUE;
233                 if (bestToken != null) {
234                     bestScore = bestToken.getScore();
235                 }
236                 return bestScore;
237             }
238     
239     
240             /**
241              * Sets the best scoring token for this active list
242              *
243              * @param token the best scoring token
244              */
245             @Override
246             public void setBestToken(Token token) {
247                 bestToken = token;
248             }
249     
250     
251             /**
252              * Gets the best scoring token for this active list
253              *
254              * @return the best scoring token
255              */
256             @Override
257             public Token getBestToken() {
258                 return bestToken;
259             }
260     
261     
262             /* (non-Javadoc)
263             * @see edu.cmu.sphinx.decoder.search.ActiveList#createNew()
264             */
265             @Override
266             public ActiveList newInstance() {
267                 return WordActiveListFactory.this.newInstance();
268             }
269         }
270     }
