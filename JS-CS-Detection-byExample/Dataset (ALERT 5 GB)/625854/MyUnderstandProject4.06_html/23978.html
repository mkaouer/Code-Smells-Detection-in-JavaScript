13     package edu.cmu.sphinx.demo.classbased;
14     
15     import edu.cmu.sphinx.frontend.util.Microphone;
16     import edu.cmu.sphinx.recognizer.Recognizer;
17     import edu.cmu.sphinx.result.Result;
18     import edu.cmu.sphinx.util.props.ConfigurationManager;
19     
20     import java.io.File;
21     import java.net.MalformedURLException;
22     import java.net.URL;
23     
24     
25     public class ClassBased {
26     
27     
28         public static void main(String[] args) throws MalformedURLException {
29             URL url;
30             if (args.length > 0) {
31                 url = new File(args[0]).toURI().toURL();
32             } else {
33                 url = ClassBased.class.getResource("classbased.config.xml");
34             }
35     
36             System.out.println("Loading...");
37     
38             ConfigurationManager cm = new ConfigurationManager(url);
39     
40             Recognizer recognizer = (Recognizer) cm.lookup("recognizer");
41             Microphone microphone = (Microphone) cm.lookup("microphone");
42     
43             /* allocate the resource necessary for the recognizer */
44             recognizer.allocate();
45     
46             printInstructions();
47     
48             /* the microphone will keep recording until the program exits */
49             if (microphone.startRecording()) {
50     
51                 while (true) {
52                     System.out.println
53                             ("Start speaking. Press Ctrl-C to quit.\n");
54     
55                     /*
56                     * This method will return when the end of speech
57                     * is reached. Note that the endpointer will determine
58                     * the end of speech.
59                     */
60                     Result result = recognizer.recognize();
61     
62                     if (result != null) {
63                         String resultText = result.getBestResultNoFiller();
64                         System.out.println("You said: " + resultText + "\n");
65                     } else {
66                         System.out.println("I can't hear what you said.\n");
67                     }
68                 }
69             } else {
70                 System.out.println("Cannot start microphone.");
71                 recognizer.deallocate();
72                 System.exit(1);
73             }
74     
75         }
76     
77         /**
78          * Prints out what to say for this demo.
79          */
80         private static void printInstructions() {
81             System.out.println
82                     ("Sample sentences:\n" +
83                             "\n" +
84                             "one ticket to boston\n" +
85                             "two tickets to washington for tuesday\n" +
86                             "\n");
87         }
88     }
