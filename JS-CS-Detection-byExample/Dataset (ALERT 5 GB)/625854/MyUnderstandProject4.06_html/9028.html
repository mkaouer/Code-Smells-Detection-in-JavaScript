27     package javax.speech.recognition;
28     
29     import java.util.Vector;
30     
31     import javax.speech.EngineEvent;
32     
33     //Comp. 2.0.6
34     
35     public class RecognizerEvent extends EngineEvent {
36         public static final int RECOGNIZER_PROCESSING = 0x2000800;
37     
38         public static final int RECOGNIZER_LISTENING = 0x2001000;
39     
40         public static final int CHANGES_COMMITTED = 0x2002000;
41     
42         public static final int CHANGES_REJECTED = 0x2040000;
43     
44         public static final int SPEECH_STARTED = 0x2004000;
45     
46         public static final int SPEECH_STOPPED = 0x2008000;
47     
48         public static final int RECOGNIZER_NOT_BUFFERING = 0x2010000;
49     
50         public static final int RECOGNIZER_BUFFERING = 0x2020000;
51     
52         public static final int RECOGNIZER_STOPPED = 0x2040000;
53     
54         public static final int UNKNOWN_AUDIO_POSITION = -1;
55     
56         public static final int DEFAULT_MASK = EngineEvent.DEFAULT_MASK
57                 | CHANGES_COMMITTED | CHANGES_REJECTED | RECOGNIZER_LISTENING
58                 | RECOGNIZER_PROCESSING | SPEECH_STARTED | SPEECH_STOPPED;
59     
60         private GrammarException grammarException;
61     
62         private long audioPosition;
63     
64         public RecognizerEvent(Recognizer source, int id, long oldEngineState,
65                 long newEngineState, Throwable problem,
66                 GrammarException grammarException, long audioPosition) 
67             throws IllegalArgumentException {
68             super(source, id, oldEngineState, newEngineState, problem);
69             if ((id != ENGINE_ERROR) && (problem != null)) {
70                 throw new IllegalArgumentException(
71                         "A problem can only be specified for ENGINE_ERROR");
72             }
73             if ((id == SPEECH_STARTED) || (id == SPEECH_STOPPED)
74                     || (id == RECOGNIZER_BUFFERING)
75                     || (id == RECOGNIZER_NOT_BUFFERING)) {
76                 if (audioPosition < 0) {
77                     throw new IllegalArgumentException(
78                             "Audio position must be a non-negative integer!");
79                 }
80             } else {
81                 if (audioPosition != UNKNOWN_AUDIO_POSITION) {
82                     StringBuffer str = new StringBuffer();
83                     id2String(str);
84                     throw new IllegalArgumentException(
85                             "Audio position must be UNKNOWN_AUDIO_POSITION for"
86                             + " the given event id (" + str.toString() + ")!");
87                 }
88             }
89             this.grammarException = grammarException;
90             this.audioPosition = audioPosition;
91         }
92     
93         public long getAudioPosition() {
94             return audioPosition;
95         }
96     
97         public GrammarException getGrammarException() {
98             final int id = getId();
99             if (id == CHANGES_REJECTED) {
100                 return grammarException;
101             }
102     
103             return null;
104         }
105     
106         /**
107          * {@inheritDoc}
108          */
109         protected Vector getParameters() {
110             final Vector parameters = super.getParameters();
111     
112             final Long audioPositionObject = new Long(audioPosition);
113             parameters.addElement(audioPositionObject);
114             parameters.addElement(grammarException);
115     
116             return parameters;
117         }
118     }
