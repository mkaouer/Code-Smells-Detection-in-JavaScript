132                 dataProcessor = (DataProcessor) configuration

136                 if (!(dataProcessor instanceof SphinxInputDataProcessor)) {
137                     throw new EngineException("Unsupported input type");
138                 }
139     
140             } catch (Exception ex) {
141                 LOGGER.warning("error creating engine properties "
142                         + ex.getMessage());
143                 ex.printStackTrace();
144             }
145     
146             // hard-coded audio format
147             ((JseBaseAudioManager) getAudioManager())
148                     .setEngineAudioFormat(new AudioFormat(16000, 16, 1, true, true));
149             resultListener = new Sphinx4ResultListener(this);

152         /**
153          * Called from the <code>allocate</code> method.
154          * 
155          * @throws EngineException
156          *             if problems are encountered
157          */
158         public void handleAllocate() throws AudioException, EngineException,
159                 EngineStateException, SecurityException {
160     
161             if (recognizer == null) {
162                 throw new EngineException(
163                         "cannot allocate: no recognizer configured!");
164             }
165     
166             if (LOGGER.isLoggable(Level.FINE)) {
167                 LOGGER.fine("allocating recognizer...");
168             }
169     
170             // Get and set input
171             InputStream inputStream = ((JseBaseAudioManager) getAudioManager())
172                     .getInputStream();
173     
174             ((SphinxInputDataProcessor) dataProcessor).setInputStream(inputStream);
175     
176             // allocate recognizer and wait for State.READY
177             recognizer.allocate();
178             waitForRecognizerState(State.READY);
179     
180             // Register state and result listener
181             recognizer.addResultListener(resultListener);
182             recognizer.addStateListener(this);
183     
184             if (LOGGER.isLoggable(Level.FINE)) {
185                 LOGGER.fine("...allocated");
186                 LOGGER.fine("state: " + recognizer.getState());
187             }
188     
189             setEngineState(CLEAR_ALL_STATE, ALLOCATED);
190         }

192         /**
193          * Called from the <code>resume</code> method.
194          */
195         public boolean handleResume() {
196             if (recognizer == null) {
197                 LOGGER.warning("no recognizer: cannot resume!");
198                 return false;
199             }
200     
201             if (recognitionThread != null) {
202                 LOGGER.warning("recognition thread already started.");
203                 return false;
204             }
205     
206             if (recognizer.getState() != State.READY) {
207                 LOGGER
208                         .warning("Cannot resume, recognizer not ready, but in state: "
209                                 + recognizer.getState());
210                 return false;
211             }
212     
213             // start data source for sphinx if it is not running
214             if (dataProcessor instanceof SphinxInputDataProcessor) {
215                 final SphinxInputDataProcessor sidp = (SphinxInputDataProcessor) dataProcessor;
216                 sidp.isRunning(true);
217             }
218     
219             // start the recognizer thread and wait for the recognizer to recognize
220             recognitionThread = new RecognitionThread(this);
221             recognitionThread.start();
222             waitForRecognizerState(State.RECOGNIZING);
223     
224             if (LOGGER.isLoggable(Level.FINE)) {
225                 LOGGER.fine("recognition started");
226             }
227     
228             return true;
229         }

231         /**
232          * Called from the <code>pause</code> method.
233          */
234         public void handlePause() {
235             if (recognitionThread == null) {
236                 throw new EngineStateException("Cannot pause, no decoder started");
237             }
238     
239             // prevent further calls to recognize()
240             recognitionThread.stopRecognition();
241     
242             // stop the sphinx4 frontend
243             if (dataProcessor instanceof Microphone) {
244                 final Microphone microphone = (Microphone) dataProcessor;
245                 microphone.stopRecording();
246             }
247             if (dataProcessor instanceof SphinxInputDataProcessor) {
248                 final SphinxInputDataProcessor sidp = (SphinxInputDataProcessor) dataProcessor;
249                 sidp.isRunning(false);
250             }
251     
252             // wait for the recognizer to transit from RECOGNIZING to READY
253             waitForRecognizerState(State.READY);
254     
255             // get rid of the recognizer thread
256             stopRecognitionThread();
257     
258         }

267         /**
268          * Called from the <code>deallocate</code> method.
269          * 
270          * According to JSAPI2 specs, pause is transitioned before dealloc.
271          * 
272          * @throws EngineException
273          *             if this <code>Engine</code> cannot be deallocated.
274          * @todo Implement this com.sun.speech.engine.BaseEngine method
275          */
276         public void handleDeallocate() {
277     
278             if (LOGGER.isLoggable(Level.FINE)) {
279                 LOGGER.fine("deallocating recognizer...");
280             }
281     
282             // pause is not called before dealloc obviously
283             if (recognizer.getState() != State.READY)
284                 handlePause();
285     
286             // Deallocate the recognizer and wait until it stops recognizing.
287             recognizer.deallocate();
288             waitForRecognizerState(State.DEALLOCATED);
289             recognizer.resetMonitors();
290     
291             // recognizer.removeResultListener(resultListener);
292     
293             if (LOGGER.isLoggable(Level.FINE)) {
294                 LOGGER.fine("...deallocated");
295             }
296     
297         }

317         /**
318          * Stop the recognition thread and wait until it is terminated.
319          */
320         private void stopRecognitionThread() {
321             if (recognitionThread == null) {
322                 LOGGER.fine("recognition thread already stopped");
323                 return;
324             }
325     
326             if (LOGGER.isLoggable(Level.FINE)) {
327                 LOGGER.fine("stopping recognition thread...");
328             }
329             recognitionThread.stopRecognition();
330     
331             final long maxSleepTime = 5000;
332             long sleepTime = 0;
333     
334             while (recognitionThread.isAlive() && (sleepTime < maxSleepTime)) {
335                 try {
336                     Thread.sleep(SLEEP_MSEC);
337                     sleepTime += SLEEP_MSEC;
338                 } catch (InterruptedException ie) {
339                     if (LOGGER.isLoggable(Level.FINE)) {
340                         LOGGER.fine("recognition thread interrupted");
341                     }
342                 }
343             }
344     
345             recognitionThread = null;
346     
347             if (LOGGER.isLoggable(Level.FINE)) {
348                 LOGGER.fine("recognition thread stopped");
349             }
350         }

352         /**
353          * Get the current rule grammar or the one that produced the list of tokens
354          * in case of the SRGS container.
355          * 
356          * @param token
357          * 
358          * @return Active grammar.
359          */
360         RuleGrammar getRuleGrammar(Token token) {
361             if (grammar instanceof SRGSGrammar) {
362                 return ((SRGSGrammar) grammar).getRuleGrammar();
363             }
364             if (grammar instanceof SRGSGrammarContainer) {
365                 return ((SRGSGrammarContainer) grammar).getRuleGrammar(token);
366             }
367             return null;
368         }

371          * @todo: in case of grammarDefinition.size > 1, make <one-of> of all the
372          *        grammars
373          * @param newGrammars
374          *            String[]
375          * @return boolean
376          */
377         protected boolean setGrammars(Vector grammarDefinition) {
378             if (grammar instanceof SRGSGrammar) {
379                 // old behavior with only a single active grammar
380                 if (grammarDefinition.size() == 1) {
381                     try {
382                         ((SRGSGrammar) grammar)
383                                 .loadSRGS(((GrammarDefinition) grammarDefinition
384                                         .get(0)).getGrammar());
385                     } catch (IOException ex) {
386                         return false;
387                     }
388                     return true;
389                 } else {
390                     return false;
391                 }
392             } else if (grammar instanceof SRGSGrammarContainer) {
393                 // the big one-of dispatcher
394                 try {
395                     ((SRGSGrammarContainer) grammar)
396                             .loadGrammars(grammarDefinition);
397                 } catch (IOException ex) {
398                     System.err.println(ex);
399                     return false;
400                 }
401                 return true;
402             }
403             return false;
404         }

414         public void fireResultEvent(final ResultEvent event) {
415             Enumeration listeners = resultListeners.elements();
416             while (listeners.hasMoreElements()) {
417                 ResultListener el = (ResultListener) listeners.nextElement();
418                 // only notify result listeners for the given grammar
419                 if (RuleGrammar.class.isAssignableFrom(el.getClass())
420                         && event.getSource().getClass().isAssignableFrom(
421                                 BaseResult.class)) {
422                     if (((RuleGrammar) el).getReference().equals(
423                             ((BaseResult) event.getSource()).getGrammar()
424                                     .getReference())) {
425                         ((ResultListener) el).resultUpdate((ResultEvent) event);
426                     }
427                 } else {
428                     ((ResultListener) el).resultUpdate((ResultEvent) event);
429     
430                 }
431             }
432         }

508         /**
509          * Wait for the recognizer to enter the given state.
510          * 
511          * @param status
512          *            The state of the recognizer to wait for.
513          */
514         private synchronized void waitForRecognizerState(State status) {
515             while (recognizer.getState() != status) {
516                 try {
517                     wait();
518                 } catch (InterruptedException e) {
519                     e.printStackTrace();
520                 }
521             }
522             LOGGER.info("Sphinx4Recognizer in state: " + status);
523         }
