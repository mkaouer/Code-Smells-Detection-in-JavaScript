27     package javax.speech.recognition;
28     
29     //Comp 2.0.6
30     
31     public class RuleCount extends RuleComponent {
32         public static final int MAX_PROBABILITY = 0x7fffffff;
33     
34         public static final int REPEAT_INDEFINITELY = 0x7fffffff;
35     
36         private RuleComponent ruleComponent;
37     
38         private int repeatMin;
39     
40         private int repeatMax;
41     
42         private int repeatProbability;
43     
44         public RuleCount(RuleComponent ruleComponent, int repeatMin)
45             throws IllegalArgumentException {
46             if (repeatMin < 0) {
47                 throw new IllegalArgumentException(
48                         "Repeat minimum must be greater or equal to 0!");
49             }
50             this.ruleComponent = ruleComponent;
51             this.repeatMax = REPEAT_INDEFINITELY;
52             this.repeatMin = repeatMin;
53             this.repeatProbability = -1;
54         }
55     
56         public RuleCount(RuleComponent ruleComponent, int repeatMin, int repeatMax)
57             throws IllegalArgumentException {
58             if (repeatMin < 0 || (repeatMin > repeatMax)) {
59                 throw new IllegalArgumentException(
60                         "Repeat minimum must be greater or equal to 0 and smaller "
61                                 + "than or equal to repeat maximum!");
62             }
63             this.ruleComponent = ruleComponent;
64             this.repeatMin = repeatMin;
65             this.repeatMax = repeatMax;
66             this.repeatProbability = -1;
67         }
68     
69         public RuleCount(RuleComponent ruleComponent, int repeatMin, int repeatMax,
70                 int repeatProbability) throws IllegalArgumentException {
71             if (repeatMin < 0 || (repeatMin > repeatMax)) {
72                 throw new IllegalArgumentException(
73                         "Repeat minimum must be greater or equal to 0 and smaller "
74                                 + "than or equal to repeat maximum!");
75             }
76     
77             if (repeatProbability < 0) {
78                 throw new IllegalArgumentException(
79                         "Repeat propability must be greater or equal to 0!");
80             }
81             this.ruleComponent = ruleComponent;
82             this.repeatMin = repeatMin;
83             this.repeatMax = repeatMax;
84             this.repeatProbability = repeatProbability;
85         }
86     
87         public int getRepeatMax() {
88             return repeatMax;
89         }
90     
91         public int getRepeatMin() {
92             return repeatMin;
93         }
94     
95         public int getRepeatProbability() {
96             if (repeatProbability < 0) {
97                 return REPEAT_INDEFINITELY;
98             }
99     
100             return repeatProbability;
101         }
102     
103         public RuleComponent getRuleComponent() {
104             return ruleComponent;
105         }
106     
107         public String toString() {
108             StringBuffer str = new StringBuffer();
109     
110             str.append("<item repeat=\"");
111             str.append(repeatMin);
112             if (repeatMin != repeatMax) {
113                 str.append("-");
114     
115                 if (repeatMax != REPEAT_INDEFINITELY) {
116                     str.append(repeatMax);
117                 }
118             }
119             str.append("\"");
120     
121             if (repeatProbability >= 0) {
122                 str.append(" repeat-prop=\"");
123                 str.append(repeatProbability);
124                 str.append("\"");
125             }
126     
127             str.append(">");
128     
129             // TODO: What to do with null rule components?
130             if (ruleComponent != null) {
131                 str.append(ruleComponent);
132             }
133     
134             str.append("</item>");
135     
136             return str.toString();
137         }
138     }
