25     /**
26      * Manages a particular point in a grammar. The GrammarPoint is used to manage the look-ahead for generating
27      * right-contexts. Since we haven't built the HMM tree yet, looking ahead can be difficult. The GrammarPoint class
28      * points to a particular unit within a pronunciation/word/grammar.  From a particular grammar point, it is possible to
29      * get the set of next grammar points.
30      */
31     
32     public class GrammarPoint {
33     
34         private GrammarNode node;        // the grammar node
35         private int alternativeIndex;    // which alternative in the grammar
36         private int wordIndex;        // which word in the alternative
37         private int pronunciationIndex;    // which pronunciation in the word
38         private int unitIndex;        // which unit in the pronunciation
39         private static boolean bounded;
40     
41     
42         /**
43          * Creates a grammar point that points to the given unit of the given pronunciation state.
44          *
45          * @param state the pronunciation of interest
46          */
47         public GrammarPoint(SentenceHMMState state) {
48             while (state != null) {
49                 if (state instanceof UnitState) {
50                     unitIndex = state.getWhich();
51                 } else if (state instanceof PronunciationState) {
52                     pronunciationIndex = state.getWhich();
53                 } else if (state instanceof WordState) {
54                     wordIndex = state.getWhich();
55                 } else if (state instanceof AlternativeState) {
56                     alternativeIndex = state.getWhich();
57                 } else if (state instanceof GrammarState) {
58                     node = ((GrammarState) state).getGrammarNode();
59                 }
60                 state = state.getParent();
61             }
62             assert node != null;
63         }

66         /**
67          * Creates a grammar node that points to the first unit of the first pronunciation of the first word of the given
68          * grammar node
69          *
70          * @param node the grammar node of interest
71          */
72         public GrammarPoint(GrammarNode node) {
73             this(node, -1, 0, 0, 0);
74         }

78          * Creates a GrammarPoint that corresponds to the given unit of the given pronunciation

80          * @param state the pronunciation state
81          * @param which the index of the unit

83         public GrammarPoint(PronunciationState state, int which) {
84             this(state);
85             unitIndex = which;

89         /**
90          * Creates a GrammarPoint that points to a fully specified unit
91          *
92          * @param node               the grammar node
93          * @param wordIndex          the index of the word in the node
94          * @param pronunciationIndex the index of the pronunciation in the word.
95          * @param unitIndex          the index of the unit in the pronunciation
96          */
97         public GrammarPoint(GrammarNode node, int alternativeIndex,
98                             int wordIndex, int pronunciationIndex, int unitIndex) {
99             assert node != null;
100             this.node = node;
101             this.alternativeIndex = alternativeIndex;
102             this.wordIndex = wordIndex;
103             this.pronunciationIndex = pronunciationIndex;
104             this.unitIndex = unitIndex;
105         }
106     
107     
108         /**
109          * Gets the unit associated with this point in the grammar
110          *
111          * @return the unit, or null if there is no unit associated with this point in the grammar
112          */
113         private Unit getUnit() {
114             Unit unit = null;
115             Word[][] alternatives = node.getAlternatives();
116             if (alternativeIndex != -1 && alternativeIndex < alternatives.length) {
117                 Word[] words = alternatives[alternativeIndex];
118                 if (wordIndex < words.length) {
119                     Pronunciation[] pronunciations =
120                             words[wordIndex].getPronunciations(null);
121                     if (pronunciationIndex < pronunciations.length) {
122                         Unit[] units =
123                                 pronunciations[pronunciationIndex].getUnits();
124                         if (unitIndex < units.length) {
125                             unit = units[unitIndex];
126                         }
127                     }
128                 }
129             }
130             return unit;
131         }
132     
133     
134         /**
135          * Gets the unit associated with this point in the grammar. If there is no unit, return filler
136          *
137          * @return the unit for this grammar node or a filler unit
138          */
139         private Unit getUnitOrFill() {
140             Unit unit = getUnit();
141             if (unit == null) {
142                 unit = UnitManager.SILENCE;
143             }
144             return unit;
145         }

148         /**
149          * Gets all of the right contexts for this grammar point. The contexts returned are guaranteed to be 'size' units
150          * in length, The number of contexts returned depends upon the perplexity of the grammar downstream from this
151          * GrammarPoint
152          *
153          * @param size             the size of each context returned
154          * @param startWithCurrent include the current state in the context
155          * @param maxContexts      the maxium number of right contexts to return
156          * @return a list of containing Unit[] contexts.
157          */
158         public List<Unit[]> getRightContexts(int size, boolean startWithCurrent,
159                                      int maxContexts) {
160             List<Unit[]> contexts = new ArrayList<Unit[]>();
161             List<GrammarPoint> nextPoints = getNextGrammarPoints(startWithCurrent);
162     
163             if (nextPoints.isEmpty()) {
164                 Unit[] units = Unit.getEmptyContext(size);
165                 addContext(contexts, units);
166             } else {
167                 for (GrammarPoint gp : nextPoints) {
168                     if (size == 1) {
169                         Unit[] units = new Unit[size];
170                         units[0] = gp.getUnitOrFill();
171                         addContext(contexts, units);
172                     } else {
173                         List<Unit[]> rc = gp.getRightContexts(size - 1, false,
174                             maxContexts - contexts.size());
175                         for (Unit[] rcUnits : rc) {
176                             Unit[] units = Unit.getEmptyContext(rcUnits.length + 1);
177                             units[0] = gp.getUnitOrFill();
178                             System.arraycopy(rcUnits, 0, units, 1, rcUnits.length);
179                             addContext(contexts, units);
180                         }
181                     }
182                     if (contexts.size() >= maxContexts) {
183                         break;
184                     }
185                 }
186             }
187             return contexts;
188         }

191         /**
192          * Add a context to a list of contexts after ensuring that no identical contexts exist on the list. When a right
193          * context is collected it may contain duplicates in certain cases (when this unit is the last unit in a grammar
194          * node, and there is a branch to multiple words in subsequent nodes, for instance)
195          *
196          * @param contexts the list of contexts to add the new units to
197          * @param units    the units to add to the context
198          */
199         private void addContext(List<Unit[]> contexts, Unit[] units) {
200             for (Unit[] onList : contexts) {
201                 if (Unit.isContextMatch(onList, units)) {
202                     return; // found on list so bailout
203                 }
204             }
205             contexts.add(units);
206         }

209         /**
210          * Returns a list of next GrammarPoints for this GrammarPoint. If there are no more downstream grammar points with
211          * words, an empty list is returned.
212          *
213          * @param startWithCurrent include the current state in the context
214          * @return the (possibly empty) list of next GrammarPoint objects
215          */
216         private List<GrammarPoint> getNextGrammarPoints(boolean startWithCurrent) {
217             List<GrammarPoint> nextPoints = new ArrayList<GrammarPoint>();
218             int unitsLength = 0;
219     
220             // if this GrammarPoint is associated with a grammar node
221             // and the grannar node has alternatives, add points for each
222             // alternative
223             if (alternativeIndex == -1 && node.getAlternatives().length > 0) {
224                 for (int i = 0; i < node.getAlternatives().length; i++) {
225                     GrammarPoint gp = new GrammarPoint(node, i, 0, 0, 0);
226                     nextPoints.add(gp);
227                 }
228             }
229     
230             // If we don't have any alternatives, (i.e. this grammar node
231             // has no words at all associated with it, then just go and
232             // find the set of next grammar nodes with words, collect
233             // them up, expand them and return that set.
234     
235             else if (node.getAlternatives().length == 0) {
236                 addNextGrammarPointsWithWords(node, nextPoints);
237             } else {
238     
239                 // At this point we are at a node with a set of alternatives
240     
241                 GrammarPoint next;
242     
243                 if (startWithCurrent) {
244                     next = this;
245                 } else {
246                     next = new GrammarPoint(node, alternativeIndex, wordIndex,
247                             pronunciationIndex, unitIndex + 1);
248                 }
249                 Pronunciation[] pronunciations = node.
250                         getAlternatives()[alternativeIndex][wordIndex].
251                         getPronunciations(null);
252     
253                 unitsLength = pronunciations[pronunciationIndex].getUnits().length;
254     
255                 if (next.unitIndex < unitsLength) {
256                     nextPoints.add(next);
257                 } else {
258                     next.unitIndex = 0;
259                     Word[] alternative =
260                             next.node.getAlternatives()[alternativeIndex];
261                     if (++next.wordIndex < alternative.length) {
262                         Word word = alternative[next.wordIndex];
263                         for (int i = 0; i < word.getPronunciations(null).length;
264                              i++) {
265                             GrammarPoint newGP = new GrammarPoint(next.node,
266                                     next.alternativeIndex, next.wordIndex, i, 0);
267                             nextPoints.add(newGP);
268                         }
269                     } else if (!bounded) {
270                         addNextGrammarPointsWithWords(next.node, nextPoints);
271                     }
272                 }
273             }
274             return nextPoints;
275         }

278         /**
279          * Given a GrammarNode return a list of successors GrammarNodes that contain words
280          *
281          * @param node successors are gathered from this node
282          * @return list the list of grammar nodes
283          */
284         private static List<GrammarNode> getNextGrammarNodesWithWords(GrammarNode node) {
285             List<GrammarNode> list = new ArrayList<GrammarNode>();
286     
287             for (GrammarArc arc : node.getSuccessors()) {
288                 GrammarNode gnode = arc.getGrammarNode();
289                 if (gnode.getAlternatives().length == 0) {
290                     if (gnode.isFinalNode()) {
291                         list.add(gnode);
292                     } else {
293                         list.addAll(getNextGrammarNodesWithWords(gnode));
294                     }
295                 } else {
296                     list.add(gnode);
297                 }
298             }
299             return list;
300         }
301     
302     
303         /**
304          * Adds the next set of grammar points that contain words to the given list
305          *
306          * @param node       the grammar node
307          * @param nextPoints where the grammar points should be added
308          */
309         private static void addNextGrammarPointsWithWords(GrammarNode
310                 node, List<GrammarPoint> nextPoints) {
311             for (GrammarNode nextNode : getNextGrammarNodesWithWords(node)) {
312                 for (int j = 0; j < nextNode.getAlternatives().length; j++) {
313                     GrammarPoint gp = new GrammarPoint(nextNode, j, 0, 0, 0);
314                     nextPoints.add(gp);
315                 }
316             }
317         }
