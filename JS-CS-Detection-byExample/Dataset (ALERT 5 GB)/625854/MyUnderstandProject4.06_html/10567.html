122          *
123          * @param result the new result
124          */
125         protected void fireResultListeners(Result result) {
126             if (fireNonFinalResults || result.isFinal()) {
127                 for (ResultListener resultListener : resultListeners) {
128                     resultListener.newResult(result);
129                 }
130             }else {
131                 logger.finer("skipping non-final result " + result);
