13     package edu.cmu.sphinx.tools.batch;
14     
15     import java.io.File;
16     import java.io.FileInputStream;
17     import java.io.FileWriter;
18     import java.io.IOException;
19     import java.io.PrintWriter;
20     import java.util.Scanner;
21     
22     import edu.cmu.sphinx.frontend.util.StreamDataSource;
23     import edu.cmu.sphinx.recognizer.Recognizer;
24     import edu.cmu.sphinx.result.Result;
25     import edu.cmu.sphinx.util.props.ConfigurationManager;
26     
27     /**
28      * Batch decoder recognizer which can be used inside Sphinxtrain
29      * to build lattices and generate decoding results.
30      */
31     public class BatchDecoderRecognizer {
32     
33         int ctlOffset = -1;
34         int ctlCount = 1000000;
35         String config;
36         String hmm;
37         String ctl;
38         String hyp;
39         String featDir;
40         
41         ConfigurationManager manager;
42         StreamDataSource source;
43         Recognizer recognizer;
44         PrintWriter writer;
45         
46         void parseArgs(String[] argv) {
47             for (int i = 0; i < argv.length; i++) {
48                 if (argv[i].equals("-ctl")) {
49                     ctl = argv[++i];
50                 }
51                 if (argv[i].equals("-config")) {
52                     config = argv[++i];
53                 }
54                 if (argv[i].equals("-hmm")) {
55                     hmm = argv[++i];
56                 }
57                 if (argv[i].equals("-ctloffset")) {
58                     ctlOffset = Integer.parseInt(argv[++i]);
59                 }
60                 if (argv[i].equals("-ctlcount")) {
61                     ctlCount = Integer.parseInt(argv[++i]);
62                 }
63                 if (argv[i].equals("-hyp")) {
64                     hyp = argv[++i];
65                 }
66                 if (argv[i].equals("-feat")) {
67                     featDir = argv[++i];
68                 }        
69             }        
70         }
71         
72         void recognize() throws IOException {
73             
74             init();
75             writer = new PrintWriter (new File(hyp), "UTF-8");
76             Scanner scanner = new Scanner(new File(ctl));
77             
78             for (int i = 0; i < ctlOffset; i++) {
79                 if (scanner.hasNext())
80                     scanner.next();
81             }
82             
83             for (int i = 0; i < ctlCount; i++) {
84                 if (scanner.hasNext()) {
85                     String utteranceId = scanner.next();
86                     String inputFile =  featDir + "/" + utteranceId + ".wav";
87                     processFile(utteranceId, inputFile);
88                 }
89             }
90             writer.close();
91             recognizer.deallocate();
92         }
93         
94         private void processFile(String utteranceId, String inputFile) throws IOException {
95             FileInputStream stream = new FileInputStream(inputFile);
96             source.setInputStream(stream, null);
97             Result result = recognizer.recognize();
98             writer.println (result.getBestFinalResultNoFiller() + " (" + utteranceId + ")");
99         }
100     
101         public static void main(String[] argv) throws IOException {
102             BatchDecoderRecognizer batchRecognizer = new BatchDecoderRecognizer();
103             batchRecognizer.parseArgs(argv);
104             batchRecognizer.recognize();
105         }
106     
107         private void init() throws IOException {
108             manager = new ConfigurationManager(config);
109             recognizer = (Recognizer)manager.lookup("recognizer");
110             source = (StreamDataSource)manager.lookup("streamDataSource");
111             
112             recognizer.allocate();        
113         }
114     }
