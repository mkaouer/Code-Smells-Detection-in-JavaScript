27     package javax.speech.recognition;
28     
29     import java.util.Vector;
30     
31     import javax.speech.SpeechEvent;
32     
33     // Comp 2.0.6
34     
35     public class ResultEvent extends SpeechEvent {
36     
37         public static final int RESULT_CREATED = 0x6000001;
38     
39         public static final int RESULT_UPDATED = 0x6000002;
40     
41         public static final int GRAMMAR_FINALIZED = 0x6000004;
42     
43         public static final int RESULT_ACCEPTED = 0x6000008;
44     
45         public static final int RESULT_REJECTED = 0x6000010;
46     
47         public static final int AUDIO_RELEASED = 0x6000020;
48     
49         public static final int TRAINING_INFO_RELEASED = 0x6000040;
50     
51         public static final int DEFAULT_MASK = RESULT_CREATED | RESULT_UPDATED;
52     
53         private boolean tokensFinalized;
54     
55         private boolean unfinalizedTokensChanged;
56     
57         public ResultEvent(Result source, int id) throws IllegalArgumentException {
58             super(source, id);
59         }
60     
61         public ResultEvent(Result source, int id, boolean tokensFinalized,
62                 boolean unfinalizedTokensChanged) throws IllegalArgumentException {
63             super(source, id);
64     
65             this.tokensFinalized = tokensFinalized;
66             this.unfinalizedTokensChanged = unfinalizedTokensChanged;
67         }
68     
69         public boolean isFinalizedChanged() {
70             final int id = getId();
71             if ((id == RESULT_CREATED) || (id == RESULT_UPDATED)
72                     || (id == RESULT_ACCEPTED) || (id == RESULT_REJECTED)) {
73                 return tokensFinalized;
74             }
75     
76             return false;
77         }
78     
79         public boolean isUnfinalizedChanged() {
80             final int id = getId();
81             if ((id == RESULT_CREATED) || (id == RESULT_UPDATED)
82                     || (id == RESULT_ACCEPTED) || (id == RESULT_REJECTED)) {
83                 return tokensFinalized;
84             }
85     
86             return false;
87         }
88     
89         /**
90          * {@inheritDoc}
91          */
92         protected Vector getParameters() {
93             final Vector parameters = super.getParameters();
94     
95             final Boolean tokensFinalizedObject = new Boolean(tokensFinalized);
96             parameters.addElement(tokensFinalizedObject);
97             final Boolean unfinalizedTokensChangedObject = new Boolean(
98                     unfinalizedTokensChanged);
99             parameters.addElement(unfinalizedTokensChangedObject);
100     
101             return parameters;
102         }
103     }
