14     import edu.cmu.sphinx.frontend.*;

16     /**
17      * Computes the delta and double delta of input cepstrum (or plp or ...). The delta is the first order derivative and
18      * the double delta (a.k.a. delta delta) is the second order derivative of the original cepstrum. They help model the
19      * speech signal dynamics. The output data is a {@link FloatData} object with a float array of size three times the
20      * original cepstrum.
21      * <p/>
22      * <p/>
23      * The format of the outputted feature is:
24      * <p/>
25      * 12 cepstra (c[1] through c[12]) <br>followed by delta cepstra (delta c[1] through delta c[12]) <br>followed by c[0],
26      * delta c[0] <br>followed by delta delta c[0] through delta delta c[12] </p>
27      */
28     public class S3FeatureExtractor extends AbstractFeatureExtractor {
29     
30         /**
31          * Computes the next feature. Advances the pointers as well.
32          *
33          * @return the feature Data computed
34          */
35         @Override
36         protected Data computeNextFeature() {
37     
38         	int jp1 = (currentPosition - 1 + cepstraBufferSize) % cepstraBufferSize;
39             int jp2 = (currentPosition - 2 + cepstraBufferSize) % cepstraBufferSize;
40             int jp3 = (currentPosition - 3 + cepstraBufferSize) % cepstraBufferSize;
41             int jf1 = (currentPosition + 1) % cepstraBufferSize;
42             int jf2 = (currentPosition + 2) % cepstraBufferSize;
43             int jf3 = (currentPosition + 3) % cepstraBufferSize;
44     
45         	DoubleData currentCepstrum = cepstraBuffer[currentPosition];
46             double[] mfc3f = cepstraBuffer[jf3].getValues();
47             double[] mfc2f = cepstraBuffer[jf2].getValues();
48             double[] mfc1f = cepstraBuffer[jf1].getValues();
49             double[] current = currentCepstrum.getValues();
50             double[] mfc1p = cepstraBuffer[jp1].getValues();
51             double[] mfc2p = cepstraBuffer[jp2].getValues();
52             double[] mfc3p = cepstraBuffer[jp3].getValues();
53             float[] feature = new float[current.length * 3];
54     
55             currentPosition = (currentPosition + 1)% cepstraBufferSize;
56     
57             // CEP; skip C[0]
58             int j = 0;
59             for (int k = 1; k < current.length; k++) {
60                 feature[j++] = (float) current[k];
61             }
62     
63             // DCEP: mfc[2] - mfc[-2], skip DC[0]
64             for (int k = 1; k < mfc2f.length; k++) {
65                 feature[j++] = (float) (mfc2f[k] - mfc2p[k]);
66             }
67     
68             // POW: C0, DC0
69             feature[j++] = (float) current[0];
70             feature[j++] = (float) (mfc2f[0] - mfc2p[0]);
71     
72             // D2CEP: (mfc[3] - mfc[-1]) - (mfc[1] - mfc[-3])
73             for (int k = 0; k < mfc3f.length; k++) {
74                 feature[j++] = (float)
75                         ((mfc3f[k] - mfc1p[k]) - (mfc1f[k] - mfc3p[k]));
76             }
77     
78             return (new FloatData(feature,
79                     currentCepstrum.getSampleRate(),
80                     currentCepstrum.getCollectTime(),
81                     currentCepstrum.getFirstSampleNumber()));
82         }
83     }
