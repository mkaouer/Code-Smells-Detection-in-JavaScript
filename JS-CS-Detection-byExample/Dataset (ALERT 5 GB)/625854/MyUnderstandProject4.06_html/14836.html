13     package edu.cmu.sphinx.util;
14     
15     import java.util.HashMap;
16     import java.util.Map;
17     
18     /**
19      * Represents a named value. A StatisticsVariable may be used to track data in a fashion that will allow the data to be
20      * viewed or dumped at any time.  Statistics are kept in a pool and are grouped in contexts. Statistics can be dumped
21      * as a whole or by context.
22      */
23     public class StatisticsVariable {
24     
25         private static final Map<String, StatisticsVariable> pool = new HashMap<String, StatisticsVariable>();
26     
27         /** the value of this StatisticsVariable. It can be manipulated directly by the application. */
28         public double value;
29     
30         private final String name;        // the name of this value
31         private boolean enabled;        // if true this var is enabled
32     
33     
34         /**
35          * Gets the StatisticsVariable with the given name from the given context. If the statistic does not currently
36          * exist, it is created. If the context does not currently exist, it is created.
37          *
38          * @param statName the name of the StatisticsVariable
39          * @return the StatisticsVariable with the given name and context
40          */
41         static public StatisticsVariable getStatisticsVariable(String statName) {
42     
43             StatisticsVariable stat = pool.get(statName);
44             if (stat == null) {
45                 stat = new StatisticsVariable(statName);
46                 pool.put(statName, stat);
47             }
48             return stat;
49         }
50     
51     
52         /**
53          * Gets the StatisticsVariable with the given name for the given instance and context. This is a convenience
54          * function.
55          *
56          * @param instanceName the instance name of creator
57          * @param statName     the name of the StatisticsVariable
58          */
59         static public StatisticsVariable getStatisticsVariable(
60                 String instanceName, String statName) {
61             return getStatisticsVariable(instanceName + '.' + statName);
62         }
63     
64     
65         /** Dump all of the StatisticsVariable in the given context */
66         static public void dumpAll() {
67             System.out.println(" ========= statistics  " + "=======");
68             for (StatisticsVariable stats : pool.values()) {
69                 stats.dump();
70             }
71         }
72     
73     
74         /** Resets all of the StatisticsVariables in the given context */
75         static public void resetAll() {
76             for (StatisticsVariable stats : pool.values()) {
77                 stats.reset();
78             }
79         }
80     
81     
82         /**
83          * Contructs a StatisticsVariable with the given name and context
84          *
85          * @param statName the name of this StatisticsVariable
86          */
87         private StatisticsVariable(String statName) {
88             this.name = statName;
89             this.value = 0.0;
90         }
91     
92     
93         /**
94          * Retrieves the name of this StatisticsVariable
95          *
96          * @return the name of this StatisticsVariable
97          */
98         public String getName() {
99             return name;
100         }
101     
102     
103         /**
104          * Retrieves the value for this StatisticsVariable
105          *
106          * @return the current value for this StatisticsVariable
107          */
108         public double getValue() {
109             return value;
110         }
111     
112     
113         /**
114          * Sets the value for this StatisticsVariable
115          *
116          * @param value the new value
117          */
118         public void setValue(double value) {
119             this.value = value;
120         }
121     
122     
123         /** Resets this StatisticsVariable. The value is set to zero. */
124         public void reset() {
125             setValue(0.0);
126         }
127     
128     
129         /** Dumps this StatisticsVariable. */
130         public void dump() {
131             if (isEnabled()) {
132                 System.out.println(name + ' ' + value);
133             }
134         }
135     
136     
137         /**
138          * Determines if this StatisticsVariable is enabled
139          *
140          * @return true if enabled
141          */
142         public boolean isEnabled() {
143             return enabled;
144         }
145     
146     
147         /**
148          * Sets the enabled state of this StatisticsVariable
149          *
150          * @param enabled the new enabled state
151          */
152         public void setEnabled(boolean enabled) {
153             this.enabled = enabled;
154         }
155     
156     
157         /** Some simple tests for the StatisticsVariable */
158         public static void main(String[] args) {
159             StatisticsVariable loops =
160                     StatisticsVariable.getStatisticsVariable("main", "loops");
161             StatisticsVariable sum =
162                     StatisticsVariable.getStatisticsVariable("main", "sum");
163     
164             StatisticsVariable foot =
165                     StatisticsVariable.getStatisticsVariable("body", "foot");
166             StatisticsVariable leg =
167                     StatisticsVariable.getStatisticsVariable("body", "leg");
168             StatisticsVariable finger =
169                     StatisticsVariable.getStatisticsVariable("body", "finger");
170     
171             foot.setValue(2);
172             leg.setValue(2);
173             finger.setValue(10);
174     
175             StatisticsVariable.dumpAll();
176             StatisticsVariable.dumpAll();
177     
178             for (int i = 0; i < 1000; i++) {
179                 loops.value++;
180                 sum.value += i;
181             }
182     
183             StatisticsVariable.dumpAll();
184     
185     
186             StatisticsVariable loopsAlias =
187                     StatisticsVariable.getStatisticsVariable("main", "loops");
188             StatisticsVariable sumAlias =
189                     StatisticsVariable.getStatisticsVariable("main", "sum");
190     
191             for (int i = 0; i < 1000; i++) {
192                 loopsAlias.value++;
193                 sumAlias.value += i;
194             }
195     
196             StatisticsVariable.dumpAll();
197             StatisticsVariable.resetAll();
198             StatisticsVariable.dumpAll();
199         }
200     }
