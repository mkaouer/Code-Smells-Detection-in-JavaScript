32     /**
33      * Grammar for GrXML W3C Standard
34      * @author shmyrev
35      *
36      */
37     public class GrXMLGrammar extends JSGFGrammar {
38     
39         Map<String, JSGFRule> rules;
40         /** The property that defines the location of the JSGF grammar file. */
41         @S4String
42         public final static String PROP_BASE_GRAMMAR_URL = "grammarLocation";
43     
44         protected void loadXML() throws IOException {
45             try {
46                 SAXParserFactory factory = SAXParserFactory.newInstance();
47                 XMLReader xr = factory.newSAXParser().getXMLReader();
48                 rules = new HashMap<String, JSGFRule>();
49                 GrXMLHandler handler = new GrXMLHandler(baseURL, rules, logger);
50                 xr.setContentHandler(handler);
51                 xr.setErrorHandler(handler);
52                 InputStream is = baseURL.openStream();
53                 xr.parse(new InputSource(is));
54                 is.close();
55             } catch (SAXParseException e) {
56                 String msg = "Error while parsing line " + e.getLineNumber() + " of " + baseURL + ": " + e.getMessage();
57                 throw new IOException(msg);
58             } catch (SAXException e) {
59                 throw new IOException("Problem with XML: " + e);
60             } catch (ParserConfigurationException e) {
61                 throw new IOException(e.getMessage());
62             }
63     
64             return;
65         }
66         
67         /**
68          * Commit changes to all loaded grammars and all changes of grammar since
69          * the last commitChange
70          * 
71          * @throws JSGFGrammarParseException
72          * @throws JSGFGrammarException
73          */
74         public void commitChanges() throws IOException, JSGFGrammarParseException,
75                 JSGFGrammarException {
76             try {
77                 if (loadGrammar) {
78                     if (manager == null)
79                         getGrammarManager();
80                     loadXML();
81                     loadGrammar = false;
82                 }
83     
84                 ruleStack = new RuleStack();
85                 newGrammar();
86     
87                 firstNode = createGrammarNode("<sil>");
88                 GrammarNode finalNode = createGrammarNode("<sil>");
89                 finalNode.setFinalNode(true);
90     
91                 // go through each rule and create a network of GrammarNodes
92                 // for each of them
93     
94                 for (Map.Entry<String, JSGFRule> entry : rules.entrySet()) {
95                     
96                         GrammarGraph publicRuleGraph = new GrammarGraph();
97                         ruleStack.push(entry.getKey(), publicRuleGraph);
98                         GrammarGraph graph = processRule(entry.getValue());
99                         ruleStack.pop();
100     
101                         firstNode.add(publicRuleGraph.getStartNode(), 0.0f);
102                         publicRuleGraph.getEndNode().add(finalNode, 0.0f);
103                         publicRuleGraph.getStartNode().add(graph.getStartNode(),
104                                 0.0f);
105                         graph.getEndNode().add(publicRuleGraph.getEndNode(), 0.0f);
106                 }
107                 postProcessGrammar();
108             } catch (MalformedURLException mue) {
109                 throw new IOException("bad base grammar URL " + baseURL + ' ' + mue);
110             }
111         }
