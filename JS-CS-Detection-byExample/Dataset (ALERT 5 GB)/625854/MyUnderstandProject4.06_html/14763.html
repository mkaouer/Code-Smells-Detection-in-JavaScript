34             super(absoluteBeamWidth, relativeBeamWidth, logMath);
35         }

37         public SortingActiveListFactory() {

39         }

42         * (non-Javadoc)

54         *

57         @Override
58         public ActiveList newInstance() {
59             return new SortingActiveList(absoluteBeamWidth, logRelativeBeamWidth);
60         }

63         /**
64          * An active list that tries to be simple and correct. This type of active list will be slow, but should exhibit
65          * correct behavior. Faster versions of the ActiveList exist (HeapActiveList, TreeActiveList).
66          * <p/>
67          * This class is not thread safe and should only be used by a single thread.
68          * <p/>
69          * Note that all scores are maintained in the LogMath log base.
70          */
71     
72         class SortingActiveList implements ActiveList {
73     
74             private final static int DEFAULT_SIZE = 1000;
75             private final int absoluteBeamWidth;
76             private final float logRelativeBeamWidth;
77             private Token bestToken;
78             // when the list is changed these things should be
79             // changed/updated as well
80             private List<Token> tokenList;
81     
82     
83             /** Creates an empty active list
84              * @param absoluteBeamWidth
85              * @param logRelativeBeamWidth*/
86             public SortingActiveList(int absoluteBeamWidth, float logRelativeBeamWidth) {
87                 this.absoluteBeamWidth = absoluteBeamWidth;
88                 this.logRelativeBeamWidth = logRelativeBeamWidth;
89     
90                 int initListSize = absoluteBeamWidth > 0 ? absoluteBeamWidth : DEFAULT_SIZE;
91                 this.tokenList = new ArrayList<Token>(initListSize);
92             }
93     
94     
95             /**
96              * Adds the given token to the list
97              *
98              * @param token the token to add
99              */
100             @Override
101             public void add(Token token) {
102                 token.setLocation(tokenList.size());
103                 tokenList.add(token);
104                 if (bestToken == null || token.getScore() > bestToken.getScore()) {
105                     bestToken = token;
106                 }
107             }
108     
109     
110             /**
111              * Replaces an old token with a new token
112              *
113              * @param oldToken the token to replace (or null in which case, replace works like add).
114              * @param newToken the new token to be placed in the list.
115              */
116             @Override
117             public void replace(Token oldToken, Token newToken) {
118                 if (oldToken != null) {
119                     int location = oldToken.getLocation();
120                     // just a sanity check:
121                     if (tokenList.get(location) != oldToken) {
122                         System.out.println("SortingActiveList: replace " + oldToken
123                                 + " not where it should have been.  New "
124                                 + newToken + " location is " + location + " found "
125                                 + tokenList.get(location));
126                     }
127                     tokenList.set(location, newToken);
128                     newToken.setLocation(location);
129                     if (bestToken == null
130                             || newToken.getScore() > bestToken.getScore()) {
131                         bestToken = newToken;
132                     }
133                 } else {
134                     add(newToken);
135                 }
136             }
137     
138     
139             /**
140              * Purges excess members. Reduce the size of the token list to the absoluteBeamWidth
141              *
142              * @return a (possible new) active list
143              */
144             @Override
145             public ActiveList purge() {
146                 // if the absolute beam is zero, this means there
147                 // should be no constraint on the abs beam size at all
148                 // so we will only be relative beam pruning, which means
149                 // that we don't have to sort the list
150                 if (absoluteBeamWidth > 0 && tokenList.size() > absoluteBeamWidth) {
151                     Collections.sort(tokenList, Scoreable.COMPARATOR);
152                     tokenList = tokenList.subList(0, absoluteBeamWidth);
153                 }
154                 return this;
155             }
156     
157     
158             /**
159              * gets the beam threshold best upon the best scoring token
160              *
161              * @return the beam threshold
162              */
163             @Override
164             public float getBeamThreshold() {
165                 return getBestScore() + logRelativeBeamWidth;
166             }
167     
168     
169             /**
170              * gets the best score in the list
171              *
172              * @return the best score
173              */
174             @Override
175             public float getBestScore() {
176                 float bestScore = -Float.MAX_VALUE;
177                 if (bestToken != null) {
178                     bestScore = bestToken.getScore();
179                 }
180                 return bestScore;
181             }
182     
183     
184             /**
185              * Sets the best scoring token for this active list
186              *
187              * @param token the best scoring token
188              */
189             @Override
190             public void setBestToken(Token token) {
191                 bestToken = token;
192             }
193     
194     
195             /**
196              * Gets the best scoring token for this active list
197              *
198              * @return the best scoring token
199              */
200             @Override
201             public Token getBestToken() {
202                 return bestToken;
203             }
204     
205     
206             /**
207              * Retrieves the iterator for this tree.
208              *
209              * @return the iterator for this token list
210              */
211             @Override
212             public Iterator<Token> iterator() {
213                 return tokenList.iterator();
214             }
215     
216     
217             /**
218              * Gets the list of all tokens
219              *
220              * @return the list of tokens
221              */
222             @Override
223             public List<Token> getTokens() {
224                 return tokenList;
225             }
226     
227             /**
228              * Returns the number of tokens on this active list
229              *
230              * @return the size of the active list
231              */
232             @Override
233             public final int size() {
234                 return tokenList.size();
235             }
236     
237     
238             /* (non-Javadoc)
239             * @see edu.cmu.sphinx.decoder.search.ActiveList#newInstance()
240             */
241             @Override
242             public ActiveList newInstance() {
243                 return SortingActiveListFactory.this.newInstance();
244             }
245         }
246     }
