14     package edu.cmu.sphinx.frontend.feature;
15     
16     import edu.cmu.sphinx.frontend.*;
17     import edu.cmu.sphinx.linguist.acoustic.tiedstate.*;
18     import edu.cmu.sphinx.util.props.*;
19     
20     /**
21      * Implements a linear feature transformation transformation. It might be a
22      * dimension reduction or just a decorrelation transform. This component
23      * requires a special model trained with LDA/MLLT transform.
24      */
25     public class FeatureTransform extends BaseDataProcessor {
26     
27         /** The name of the transform matrix file */
28         @S4Component(type = Loader.class)
29         public final static String PROP_LOADER = "loader";
30     
31         float[][] transformMatrix;
32         protected Loader loader;
33     
34         int rows;
35         int values;
36     
37         public FeatureTransform(Loader loader) {
38             initLogger();
39             init(loader);
40         }
41     
42         public FeatureTransform() {
43         }
44     
45         /*
46          * (non-Javadoc)
47          * 
48          * @see
49          * edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util
50          * .props.PropertySheet)
51          */
52         @Override
53         public void newProperties(PropertySheet ps) throws PropertyException {
54             super.newProperties(ps);
55     
56             init((Loader) ps.getComponent(PROP_LOADER));
57         }
58     
59         private void init(Loader loader) {
60             this.loader = loader;
61     
62             try {
63                 loader.load();
64             } catch (Exception e) {
65                 e.printStackTrace();
66             }
67             this.transformMatrix = loader.getTransformMatrix();
68     
69             if (transformMatrix == null)
70                 throw new RuntimeException("Model doesn't include transformation matrix");
71     
72             this.rows = transformMatrix.length;
73             this.values = transformMatrix[0].length;
74         }
75     
76         /**
77          * Returns the next Data object being processed by this LDA, or if it is a
78          * Signal, it is returned without modification.
79          * 
80          * @return the next available Data object, returns null if no Data object is
81          *         available
82          * @throws DataProcessingException
83          *             if there is a processing error
84          * @see Data
85          */
86         @Override
87         public Data getData() throws DataProcessingException {
88             Data input = getPredecessor().getData();
89             Data output;
90             getTimer().start();
91             if (input != null && input instanceof FloatData) {
92                 FloatData inputData = (FloatData) input;
93                 float[] in = inputData.getValues();
94                 float[] out = new float[rows];
95     
96                 assert in.length == values;
97     
98                 for (int i = 0; i < rows; i++) {
99                     out[i] = 0;
100                     float transform_matrix_i[] = transformMatrix[i];
101                     for (int j = 0; j < values; j++) {
102                         out[i] += in[j] * transform_matrix_i[j];
103                     }
104                 }
105                 output = new FloatData(out, inputData.getSampleRate(), inputData.getCollectTime(), inputData.getFirstSampleNumber());
106             } else {
107                 output = input;
108             }
109             getTimer().stop();
110             return output;
111         }
112     }
