16     /** A 16bit, linear SIGNED_PCM, big endian, sinusoid with a sample rate of 16kHz. */
17     public class Sinusoid extends AudioData {
18     
19         /**
20          * Creates a 16bit, linear SIGNED_PCM, big endian, sinusoid with a sample rate of 16kHz.
21          *
22          * @param frequency cycles per second
23          * @param phase     phase offset in radians
24          * @param amplitude amplitude
25          * @param duration  duration in seconds
26          */
27         public Sinusoid(double frequency,
28                         double phase,
29                         double amplitude,
30                         double duration) {
31             this.shorts = new short[(int) (16000 * duration)];
32             double radiansPerSample = (frequency * 2.0 * Math.PI) / 16000.0;
33             for (int i = 0; i < shorts.length; i++) {
34                 shorts[i] = (short) (
35                         amplitude * Math.cos((radiansPerSample * (double) i) + phase));
36             }
37         }
38     }
