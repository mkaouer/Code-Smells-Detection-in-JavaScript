25     /**
26      * An acoustic model saver that saves sphinx3 ascii data.
27      * <p/>
28      * Mixture weights and transition probabilities are saved in linear scale.
29      */
30     class Sphinx4Saver extends Sphinx3Saver {
31     
32         /** The logger for this class */
33         private static Logger logger = Logger.getLogger(TrainerAcousticModel.PROP_PREFIX + "AcousticModel");
34     
35         protected final static String TMAT_FILE_VERSION = "4.0";
36     
37         @Override
38         protected void saveTransitionMatricesAscii(Pool<float[][]> pool, String path, boolean append)
39                 throws IOException {
40     
41             String location = super.getLocation();
42             OutputStream outputStream = StreamFactory.getOutputStream(location, path, append);
43             if (outputStream == null) {
44                 throw new IOException("Error trying to write file " + location + path);
45             }
46             PrintWriter pw = new PrintWriter(outputStream, true);
47     
48             LogMath logMath = super.getLogMath();
49     
50             logger.info("Saving transition matrices to: ");
51             logger.info(path);
52             int numMatrices = pool.size();
53     
54             assert numMatrices > 0;
55             float[][] tmat = pool.get(0);
56     
57             pw.println("tmat " + numMatrices + " X");
58     
59             for (int i = 0; i < numMatrices; i++) {
60                 tmat = pool.get(i);
61                 int numStates = tmat[0].length;
62     
63                 pw.println("tmat [" + i + ']');
64                 pw.println("nstate " + (numStates - 1));
65                 for (int j = 0; j < numStates; j++) {
66                     for (int k = 0; k < numStates; k++) {
67     
68                         // the last row is just zeros, so we just do
69                         // the first (numStates - 1) rows
70     
71                         if (j < numStates - 1) {
72                             if (sparseForm) {
73                                 if (k < j) {
74                                     pw.print("\t");
75                                 }
76                                 if (k == j || k == j + 1) {
77                                     pw.print((float)logMath.logToLinear(tmat[j][k]));
78                                 }
79                             } else {
80                                 pw.print((float)logMath.logToLinear(tmat[j][k]));
81                             }
82                             if (numStates - 1 == k) {
83                                 pw.println();
84                             } else {
85                                 pw.print(" ");
86                             }
87     
88                         }
89     
90                         if (logger.isLoggable(Level.FINE)) {
91                             logger.fine("tmat j " + j + " k " + k + " tm " + tmat[j][k]);
92                         }
93                     }
94                 }
95             }
96             outputStream.close();
97         }

99         @Override
100         protected void saveTransitionMatricesBinary(Pool<float[][]> pool, String path, boolean append)
101                 throws IOException {
102     
103             logger.info("Saving transition matrices to: ");
104             logger.info(path);
105             Properties props = new Properties();
106     
107             String strCheckSum = super.getCheckSum();
108             boolean doCheckSum = super.getDoCheckSum();
109             props.setProperty("version", TMAT_FILE_VERSION);
110             if (doCheckSum) {
111                 props.setProperty("chksum0", strCheckSum);
112             }
113     
114             String location = super.getLocation();
115     
116             DataOutputStream dos = writeS3BinaryHeader(location, path, props, append);
117     
118             int numMatrices = pool.size();
119             assert numMatrices > 0;
120             writeInt(dos, numMatrices);
121     
122             // Now we count number of states. Since each model can have an
123             // arbitrary number of states, we have to visit all tmats, and
124             // count the total number of states.
125             int numStates = 0;
126             int numValues = 0;
127             for (int i = 0; i < numMatrices; i++) {
128                 float[][] tmat = pool.get(i);
129                 int nStates = tmat[0].length;
130                 numStates += nStates;
131                 // Number of elements in each transition matrix is the
132                 // number of row, i.e. number of emitting states, times
133                 // number of columns, i.e., number of emitting states plus
134                 // final non-emitting state.
135                 numValues += nStates * (nStates - 1);
136             }
137             writeInt(dos, numValues);
138     
139             for (int i = 0; i < numMatrices; i++) {
140                 float[][] tmat = pool.get(i);
141                 numStates = tmat[0].length;
142                 writeInt(dos, numStates);
143     
144                 // Last row should be all zeroes
145                 float[] logTmatRow = tmat[numStates - 1];
146                 float[] tmatRow = new float[logTmatRow.length];
147     
148                 for (int j = 0; j < numStates; j++) {
149                     assert tmatRow[j] == 0.0f;
150                 }
151     
152                 for (int j = 0; j < numStates - 1; j++) {
153                     logTmatRow = tmat[j];
154                     tmatRow = new float[logTmatRow.length];
155                     logMath.logToLinear(logTmatRow, tmatRow);
156                     writeFloatArray(dos, tmatRow);
157                 }
158             }
159             if (doCheckSum) {
160                 assert doCheckSum = false : "Checksum not supported";
161                 // writeInt(dos, checkSum);
162             }
163             dos.close();
164         }
