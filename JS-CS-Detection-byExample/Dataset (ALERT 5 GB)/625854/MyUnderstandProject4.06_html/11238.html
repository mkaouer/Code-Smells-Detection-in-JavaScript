36     public class SynthesizerMode extends EngineMode {
37         /**
38          * Returns a hash code value for the array
39          * @param array the array to create a hash code value for
40          * @return a hash code value for the array
41          */
42         private static int hashCode(Object[] array) {
43             int prime = 31;
44             if (array == null)
45                 return 0;
46             int result = 1;
47             for (int index = 0; index < array.length; index++) {
48                 result = prime * result
49                         + (array[index] == null ? 0 : array[index].hashCode());
50             }
51             return result;
52         }

54         public static final SynthesizerMode DEFAULT = new SynthesizerMode();
55     
56         private Voice[] voices;
57     
58         public SynthesizerMode() {
59             super();
60         }
61     
62         public SynthesizerMode(SpeechLocale locale) {
63             super();
64     
65             voices = new Voice[1];
66             voices[0] = new Voice(locale, null, Voice.GENDER_DONT_CARE,
67                     Voice.AGE_DONT_CARE, Voice.VARIANT_DONT_CARE);
68         }
69     
70         public SynthesizerMode(String engineName, String modeName, Boolean running,
71                 Boolean supportsLetterToSound, Boolean supportsMarkup,
72                 Voice[] voices) {
73             super(engineName, modeName, running, supportsLetterToSound,
74                     supportsMarkup);
75     
76             this.voices = voices;
77         }
78     
79     
80         /* (non-Javadoc)
81          * @see java.lang.Object#hashCode()
82          */
83         public int hashCode() {
84             final int prime = 31;
85             int result = super.hashCode();
86             result = prime * result + SynthesizerMode.hashCode(voices);
87             return result;
88         }
89     
90         /* (non-Javadoc)
91          * @see java.lang.Object#equals(java.lang.Object)
92          */
93         public boolean equals(Object obj) {
94             if (this == obj) {
95                 return true;
96             }
97             if (!super.equals(obj)) {
98                 return false;
99             }
100             if (getClass() != obj.getClass()) {
101                 return false;
102             }
103             SynthesizerMode other = (SynthesizerMode) obj;
104             if (voices == null) {
105                 if (other.voices != null) {
106                     return false;
107                 }
108             } else if (voices.length != other.voices.length) {
109                 return false;
110             } else {
111                 for (int i=0; i<voices.length; i++) {
112                     if (!voices[i].equals(other.voices[i])) {
113                         return false;
114                     }
115                 }
116             }
117             return true;
118         }

120         public Boolean getMarkupSupport() {
121             return super.getSupportsMarkup();
122         }
123     
124         public Voice[] getVoices() {
125             return voices;
126         }
127     
128         public boolean match(EngineMode require) {
129             if (!super.match(require)) {
130                 return false;
131             }
132     
133             if (require instanceof SynthesizerMode) {
134                 final SynthesizerMode mode = (SynthesizerMode) require;
135                 Voice[] otherVoices = mode.getVoices();
136                 if (otherVoices != null) {
137                     if (voices == null) {
138                         return false;
139                     }
140     
141                     for (int i = 0; i < otherVoices.length; i++) {
142                         Voice otherVoice = otherVoices[i];
143     
144                         boolean voiceMatch = false;
145                         for (int k = 0; (k < voices.length) && !voiceMatch; k++) {
146                             final Voice voice = voices[k];
147                             if (otherVoice.match(voice)) {
148                                 voiceMatch = true;
149                             }
150                         }
151     
152                         if (!voiceMatch) {
153                             return false;
154                         }
155                     }
156                 }
157             }
158     
159             return true;
160         }
161     
162         /**
163          * Creates a collection of all parameters.
164          * 
165          * @return collection of all parameters.
166          */
167         protected Vector getParameters() {
168             final Vector parameters = super.getParameters();
169     
170             if (voices == null) {
171                 parameters.addElement(null);
172             } else {
173                 final Vector col = new Vector();
174                 for (int i = 0; i < voices.length; i++) {
175                     col.addElement(voices[i]);
176                 }
177                 parameters.addElement(col);
178             }
179     
180             return parameters;
181         }
