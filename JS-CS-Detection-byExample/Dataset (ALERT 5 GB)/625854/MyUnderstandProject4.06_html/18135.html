71         public void calculateVULevels(byte[] data, int offset, int cnt) {
72             short[] samples = new short[cnt / 2];
73             for (int i = 0; i < cnt / 2; i++) {
74                 int o = offset + (2 * i);
75                 samples[i] = (short) ((data[o] << 8) | (0x000000FF & data[o + 1]));
76                 //System.out.print(data[2*i] + "+" +data[(2*i)+1] + "=" + samples[i] + " ");
77             }
78             calculateVULevels(samples);
79         }

82         private synchronized void calculateVULevels(double[] samples) {
83             double energy = 0.0;
84             average = 0.0;
85     
86             double y1 = 0.0f;
87             double y2 = 0.0f;
88     
89     
90             for (int i = 0; i < samples.length; i++) {
91     
92                 // remove the DC offset with a filter
93     
94                 //System.out.print(samples[i] + " ");
95     
96                 double i1 = samples[i];
97                 double j = 0;
98                 double k = 0;
99     
100                 if (i > 0) {
101                     j = samples[i - 1];
102                 }
103                 if (i > 1) {
104                     k = samples[i - 2];
105                 }
106     
107                 double y = b1 * i1 + b2 * j + b3 * k - a2 * y1 - a3 * y2;
108     
109                 y2 = y1;
110                 y1 = y;
111     
112                 double v2 = Math.abs(y);
113     
114                 long now = System.currentTimeMillis();
115     
116                 energy += v2 * v2;
117                 average += v2;
118     
119                 if (v2 > peak) {
120                     peak = v2;
121                 } else if ((now - then) > peakHoldTime) {
122                     peak = v2;
123                     then = now;
124                 }
125     
126             }
127     
128             rms = energy / samples.length;
129             rms = Math.sqrt(rms);
130             average /= samples.length;
131         }
