22     /**
23      * <p/>
24      * A node is part of Lattices, representing the theory that a word was spoken over a given period of time. A node also
25      * has a set of entering and leaving {@link edu.cmu.sphinx.result.Edge edges}, connecting it to other nodes. One can get
26      * and set the beginning and end frames of the word via the getBeginTime and getEndTime methods. When setting these
27      * times, the beginning time must be earlier or equal to the end time, otherwise an error will be thrown. </p>
28      * <p/>
29      * <p/>
30      * The posterior probability of any word in a word lattice is the probability that the node representing that word
31      * occurs on any path through the lattice. It is usually computed as the ratio of the total likelihood scores of all
32      * paths through the lattice that pass through the node, to the total likelihood score of all paths through the lattice.
33      * Path scores are usually computed using the acoustic likelihoods of the nodes, although language scores can also be
34      * incorporated. The posterior probabilities of an entire lattice is usually computed efficiently using the
35      * Forward-Backward Algorithm. Refer to the {@link edu.cmu.sphinx.result.Lattice#computeNodePosteriors
36      * computeNodePosteriors} method in the Lattice class for details. </p>
37      */

77          * Create a new Node with given ID. Used when creating a Lattice from a .LAT file
78          *
79          * @param id
80          * @param word
81          * @param beginTime
82          * @param endTime

84         protected Node(String id, Word word, int beginTime, int endTime) {
85             this.id = id;
86             this.word = word;
87             this.beginTime = beginTime;
88             this.endTime = endTime;
89             if (endTime != -1) {
90                 if (beginTime > endTime) {
91                     throw new Error("Begin time (" + beginTime +
92                             ") later than end time (" + endTime + ')');
93                 }
94             }
95             this.forwardScore = LogMath.getLogZero();
96             this.backwardScore = LogMath.getLogZero();
97             this.posterior = LogMath.getLogZero();
98         }

165         /**
166          * Test if a Node has all Edges from the same Nodes and another Node.
167          *
168          * @param n
169          * @return true if this Node has Edges from the same Nodes as n
170          */
171         protected boolean hasEquivalentEnteringEdges(Node n) {
172             if (enteringEdges.size() != n.getEnteringEdges().size()) {
173                 return false;
174             }
175             for (Edge e : enteringEdges) {
176                 Node fromNode = e.getFromNode();
177                 if (!n.hasEdgeFromNode(fromNode)) {
178                     return false;
179                 }
180             }
181             return true;
182         }

187          *
188          * @param n the node of interest
189          * @return true if this Node has all Edges to the sames Nodes as n
190          */
191         public boolean hasEquivalentLeavingEdges(Node n) {
192             if (leavingEdges.size() != n.getLeavingEdges().size()) {
193                 return false;
194             }
195             for (Edge e : leavingEdges) {
196                 Node toNode = e.getToNode();
197                 if (!n.hasEdgeToNode(toNode)) {
198                     return false;
199                 }
200             }
201             return true;
202         }

508         /**
509          * Calculates the begin time of this node, in the event that the begin time was not specified. The begin time is the
510          * latest of the end times of its predecessor nodes.
511          */
512         private void calculateBeginTime() {
513             beginTime = 0;
514             for (Edge edge : enteringEdges) {
515                 if (edge.getFromNode().getEndTime() > beginTime) {
516                     beginTime = edge.getFromNode().getEndTime();
517                 }
518             }
519         }

536         protected void cacheDescendants() {
537             descendants = new HashSet<Node>();
538             cacheDescendantsHelper(this);
539         }

542         protected void cacheDescendantsHelper(Node n) {
543             for (Node child : n.getChildNodes()) {
544                 if (descendants.contains(child)) {
545                     continue;
546                 }
547                 descendants.add(child);
548                 cacheDescendantsHelper(child);
549             }
550         }

552     
553         protected boolean isAncestorHelper(List<Node> children, Node node, Set<Node> seenNodes) {
554             for (Node n : children) {
555                 if (seenNodes.contains(n)) {
556                     continue;
557                 }
558                 seenNodes.add(n);
559                 if (n.equals(node)) {
560                     return true;
561                 }
562                 if (isAncestorHelper(n.getChildNodes(), node, seenNodes)) {
563                     return true;
564                 }
565             }
566             return false;
567         }

570         /**
571          * Check whether this node is an ancestor of another node.
572          *
573          * @param node the Node to check
574          * @return whether this node is an ancestor of the passed in node.
575          */
576         public boolean isAncestorOf(Node node) {
577             if (descendants != null) {
578                 return descendants.contains(node);
579             }
580             if (this.equals(node)) {
581                 return true; // node is its own ancestor
582             }
583             Set<Node> seenNodes = new HashSet<Node>();
584             seenNodes.add(this);
585             return isAncestorHelper(this.getChildNodes(), node, seenNodes);
586         }

601         /**
602          * Returns true if the given node is equivalent to this node. Two nodes are equivalent only if they have the same
603          * word, the same number of entering and leaving edges, and that their begin and end times are the same.
604          *
605          * @param other the Node we're comparing to
606          * @return true if the Node is equivalent; false otherwise
607          */
608         public boolean isEquivalent(Node other) {
609             return
610                     ((word.getSpelling().equals(other.getWord().getSpelling()) &&
611                             (getEnteringEdges().size() == other.getEnteringEdges().size() &&
612                                     getLeavingEdges().size() == other.getLeavingEdges().size())) &&
613                             (beginTime == other.getBeginTime() &&
614                                     endTime == other.getEndTime()));
615         }

623          * @return an equivalent edge, if any; or null if no equivalent edge

626             for (Edge e : leavingEdges) {
