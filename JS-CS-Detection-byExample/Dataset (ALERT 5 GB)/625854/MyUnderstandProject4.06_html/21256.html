15      * @author Holger Brandl
16      */
17     public class BackgroundModelNormalizer implements ScoreNormalizer {

19         /**
20          * The active list provider used to determined the best token for normalization. If this reference is not defined no
21          * normalization will be applied.
22          */
23         @S4Component(type = SimpleBreadthFirstSearchManager.class, mandatory = false)
24         public static final String ACTIVE_LIST_PROVIDER = "activeListProvider";

29         public BackgroundModelNormalizer() {       
30         }

32         @Override
33         public void newProperties(PropertySheet ps) throws PropertyException {
34             this.activeListProvider = (SimpleBreadthFirstSearchManager) ps.getComponent(ACTIVE_LIST_PROVIDER);
35             this.logger = ps.getLogger();
36     
37             logger.warning("no active list set.");
38         }
39     
40         /**
41          * @param activeListProvider The active list provider used to determined the best token for normalization. If this reference is not defined no
42          * normalization will be applied.
43          */
44         public BackgroundModelNormalizer(SimpleBreadthFirstSearchManager activeListProvider) {
45             this.activeListProvider = activeListProvider;
46             this.logger = Logger.getLogger(getClass().getName());
47     
48             logger.warning("no active list set.");
49         }
50         
51         @Override
52         public Scoreable normalize(List<? extends Scoreable> scoreableList, Scoreable bestToken) {
53             if (activeListProvider == null) {
54                 return bestToken;
55             }
56     
57             Token normToken = activeListProvider.getActiveList().getBestToken();
58     
59             assert bestToken.getFrameNumber() == normToken.getFrameNumber() - 1 : "frame numbers should be equal for a meaningful normalization";
60     
61             float normScore = normToken.getScore();
62     
63             for (Scoreable scoreable : scoreableList) {
64                 if (scoreable instanceof Token) {
65                     scoreable.normalizeScore(normScore);
66                 }
67             }
68     
69             return bestToken;
70         }
