22         /**
23          * Enables or disables the token tracker
24          *

32         /** Starts the per-utterance tracking */
33         void startUtterance() {
34             if (enabled) {
35                 frame = 0;

38                 utteranceSumStates = 0;

44         void stopUtterance() {

52         void startFrame() {
53             if (enabled) {

60          * Adds a new token to the tracker

62          * @param t the token to add.

81         /** Dumps summary info about the tokens */
82         public void dumpSummary() {
83             if (enabled) {
84                 float avgStates = 0f;
85                 if (utteranceStateCount > 0) {
86                     avgStates = ((float) utteranceSumStates) / utteranceStateCount;
87                 }

91                 if (utteranceStateCount > 0) {

102         /** Dumps detailed info about the tokens */
103         public void dumpDetails() {
104             if (enabled) {
105                 int maxStates = -Integer.MAX_VALUE;
106                 int hmmCount = 0;
107                 int sumStates = 0;
108     
109                 for (TokenStats stats : stateMap.values()) {
110                     if (stats.isHMM) {
111                         hmmCount++;
112                     }
113                     sumStates += stats.count;
114                     utteranceSumStates += stats.count;
115                     if (stats.count > maxStates) {
116                         maxStates = stats.count;
117                     }
118     
119                     if (stats.count > utteranceMaxStates) {
120                         utteranceMaxStates = stats.count;
121                     }
122                 }
123     
124                 utteranceStateCount += stateMap.size();
125     
126                 float avgStates = 0f;
127                 if (!stateMap.isEmpty()) {
128                     avgStates = ((float) sumStates) / stateMap.size();
129                 }
130                 System.out.print("# Frame " + frame);
131                 System.out.print(" States: " + stateMap.size());
132     
133                 if (!stateMap.isEmpty()) {
134                     System.out.print(" Paths: " + sumStates);
135                     System.out.print(" Max: " + maxStates);
136                     System.out.print(" Avg: " + avgStates);
137                     System.out.print(" HMM: " + hmmCount);
138                 }
139     
140                 System.out.println();
141             }
142         }

145         /**
146          * Gets the statistics for a particular token
147          *
148          * @param t the token of interest
149          * @return the token statistics associated with the given token
150          */
151         private TokenStats getStats(Token t) {
152             TokenStats stats = stateMap.get(t.getSearchState()
153                     .getLexState());
154             if (stats == null) {
155                 stats = new TokenStats();
156                 stateMap.put(t.getSearchState().getLexState(), stats);
157             }
158             return stats;
159         }
160     
161         /**
162          * A class for keeping track of statistics about tokens. Tracks the count,
163          * minimum and maximum score for a particular state.
164          */
165         class TokenStats {
166     
167             int count;
168             float maxScore;
169             float minScore;
170             boolean isHMM;
171     
172     
173             TokenStats() {
174                 count = 0;
175                 maxScore = -Float.MAX_VALUE;
176                 minScore = Float.MIN_VALUE;
177             }
178     
179     
180             /** Update this state with the given token
181              * @param t*/
182             public void update(Token t) {
183                 count++;
184                 if (t.getScore() > maxScore) {
185                     maxScore = t.getScore();
186                 }
187     
188                 if (t.getScore() < minScore) {
189                     minScore = t.getScore();
190                 }
191     
192                 isHMM = t.getSearchState() instanceof HMMSearchState;
193             }
194         }
