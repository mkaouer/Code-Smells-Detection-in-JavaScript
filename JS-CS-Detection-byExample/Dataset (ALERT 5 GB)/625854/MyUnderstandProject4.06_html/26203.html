30         /**
31          * Returns a string with the given number of spaces.
32          *
33          * @param padding the number of spaces in the string
34          * @return a string of length 'padding' containing only the SPACE char.
35          */
36         public static String pad(int padding) {
37             if (padding > 0) {
38                 StringBuilder sb = new StringBuilder(padding);
39                 for (int i = 0; i < padding; i++) {
40                     sb.append(' ');
41                 }
42                 return sb.toString();
43             } else {
44                 return "";
45             }
46         }

49         /**
50          * Pads with spaces or truncates the given string to guarantee that it is exactly the desired length.
51          *
52          * @param string    the string to be padded
53          * @param minLength the desired length of the string
54          * @return a string of length containing string padded with whitespace or truncated
55          */
56         public static String pad(String string, int minLength) {
57             String result = string;
58             int pad = minLength - string.length();
59             if (pad > 0) {
60                 result = string + pad(minLength - string.length());
61             } else if (pad < 0) {
62                 result = string.substring(0, minLength);
63             }
64             return result;
65         }

91         /**
92          * Converts class name into readable string
93          * @param name  input name
94          * @return readable class name
95          */
96         public static String getReadable(String name) {
97             StringBuilder builder = new StringBuilder();
98             
99             for (int i = 0; i < name.length(); i++) {
100                 char c = name.charAt(i);
101                 if (i == 0) {
102                     builder.append(Character.toUpperCase(c));
103                 } else if (Character.isUpperCase(c) && !Character.isUpperCase(name.charAt(i - 1))) {
104                     builder.append(' ');
105                     builder.append(c);
106                 } else {
107                     builder.append(c);
108                 }
109             }
110             return builder.toString();
111         }

145          * Dumps  out memory information
146          *
147          * @param msg addditional text for the dump
148          */
149     
150         public static void dumpMemoryInfo(String msg) {
151             Runtime rt = Runtime.getRuntime();
152             long free = rt.freeMemory();
153             rt.gc();
154             long reclaimedMemory = (rt.freeMemory() - free)
155                     / (1024 * 1024);
156             long freeMemory = rt.freeMemory() / (1024 * 1024);
157             long totalMemory = rt.totalMemory() / (1024 * 1024);
158             long usedMemory = rt.totalMemory() - rt.freeMemory();
159     
160             if (usedMemory > maxUsed) {
161                 maxUsed = usedMemory;
162             }
163     
164             System.out.println("Memory (mb) "
165                     + " total: " + totalMemory
166                     + " reclaimed: " + reclaimedMemory
167                     + " free: " + freeMemory
168                     + " Max Used: " + (maxUsed / (1024 * 1024))
169                     + " -- " + msg);
170         }

173         /**
174          * Returns the string representation of the given double value in normalized scientific notation. The
175          * <code>fractionDigits</code> argument gives the number of decimal digits in the fraction portion. For example, if
176          * <code>fractionDigits</code> is 4, then the 123450 will be "1.2345e+05". There will always be two digits in the
177          * exponent portion, and a plus or minus sign before the exponent.
178          *
179          * @param number         the double to convert
180          * @param fractionDigits the number of digits in the fraction part, e.g., 4 in "1.2345e+05".
181          * @return the string representation of the double in scientific notation
182          */
183         public static String doubleToScientificString(double number,
184                                                       int fractionDigits) {
185             DecimalFormat format = new DecimalFormat();
186     
187             StringBuilder formatter = new StringBuilder(5 + fractionDigits).append("0.");
188             for (int i = 0; i < fractionDigits; i++) {
189                 formatter.append('0');
190             }
191             formatter.append("E00");
192     
193             format.applyPattern(formatter.toString());
194             String formatted = format.format(number);
195     
196             int index = formatted.indexOf('E');
197             if (formatted.charAt(index + 1) != '-') {
198                 return formatted.substring(0, index + 1) + '+' +
199                         formatted.substring(index + 1);
200             } else {
201                 return formatted;
202             }
203         }

206         /**
207          * Returns true if the given binary cepstra file is in big-endian format. It assumes that the first 4 bytes of the
208          * file tells you how many 4-byte floating point cepstra values are in the file.
209          *
210          * @param filename the cepstra file name
211          * @return true if the given binary cepstra file is big-endian
212          */
213         public static boolean isCepstraFileBigEndian(String filename)
214                 throws IOException {
215             File cepstraFile = new File(filename);
216             int fileSize = (int) cepstraFile.length();
217             DataInputStream stream =
218                     new DataInputStream(new FileInputStream(filename));
219             int numberBytes = stream.readInt() * 4 + 4;
220             stream.close();
221             return (fileSize == numberBytes);
222         }

237         /**
238          * Reads the next little-endian integer from the given DataInputStream.
239          *
240          * @param dataStream the DataInputStream to read from
241          * @return an integer
242          */
243         public static int readLittleEndianInt(DataInputStream dataStream)
244                 throws IOException {
245             return dataStream.readUnsignedByte() | dataStream.readUnsignedByte() << 8 |
246                    dataStream.readUnsignedByte() << 16 | dataStream.readUnsignedByte() << 24;
247         }

250         /**
251          * Byte-swaps the given integer to the other endian. That is, if this integer is big-endian, it becomes
252          * little-endian, and vice-versa.
253          *
254          * @param integer the integer to swap
255          */
256         public static int swapInteger(int integer) {
257             return (((0x000000ff & integer) << 24) |
258                     ((0x0000ff00 & integer) << 8) |
259                     ((0x00ff0000 & integer) >> 8) |
260                     ((0xff000000 & integer) >> 24));
261         }
262     
263     
264         /**
265          * Byte-swaps the given float to the other endian. That is, if this float is big-endian, it becomes little-endian,
266          * and vice-versa.
267          *
268          * @param floatValue the float to swap
269          */
270         public static float swapFloat(float floatValue) {
271             return Float.intBitsToFloat
272                     (swapInteger(Float.floatToRawIntBits(floatValue)));
273         }

276         /**
277          * If a data point is below 'floor' make it equal to floor.
278          *
279          * @param data  the data to floor
280          * @param floor the floored value
281          */
282         public static void floorData(float[] data, float floor) {
283             for (int i = 0; i < data.length; i++) {
284                 if (data[i] < floor) {
285                     data[i] = floor;
286                 }
287             }
288         }
289         /**
290          * If a data point is non-zero and below 'floor' make it equal to floor
291          * (don't floor zero values though).
292          * 
293          * @param data the data to floor
294          * @param floor the floored value
295          */
296         public static void nonZeroFloor(float[] data, float floor) {
297             for (int i = 0; i < data.length; i++) {
298                 if (data[i] != 0.0 && data[i] < floor) {
299                     data[i] = floor;
300                 }
301             }
302         }

305         /**
306          * Normalize the given data.
307          * 
308          * @param data the data to normalize
309          */
310         public static void normalize(float[] data) {
311             float sum = 0;
312             for (float val : data) {
313                 sum += val;
314             }
315             if (sum != 0.0f) {
316                 for (int i = 0; i < data.length; i++) {
317                     data[i] = data[i] / sum;
318                 }
319             }
320         }
