13     package org.jvoicexml.jsapi2.jse.protocols.playback;
14     
15     import java.io.IOException;
16     import java.net.InetAddress;
17     import java.net.URL;
18     import java.net.URLConnection;
19     import java.net.URLStreamHandler;
20     import java.net.UnknownHostException;
21     
22     /**
23      * Protocol handler for the capture protocol.
24      *
25      * @author Renato Cassaca
26      * @author Dirk Schnelle-Walka
27      * @version 1.0
28      */
29     public class Handler extends URLStreamHandler {
30         /**
31          * Constructs a new object.
32          */
33         public Handler() {
34             super();
35         }
36     
37         /**
38          * {@inheritDoc}
39          */
40         protected URLConnection openConnection(final URL url) throws IOException {
41             return new PlaybackURLConnection(url);
42         }
43     
44         /**
45          * {@inheritDoc}
46          */
47         protected int getDefaultPort() {
48             return 0;
49         }
50     
51         /**
52          * {@inheritDoc}
53          */
54         protected synchronized InetAddress getHostAddress(final URL url) {
55             try {
56                 return InetAddress.getLocalHost();
57             } catch (UnknownHostException ex) {
58                 return null;
59             }
60         }
61     }
