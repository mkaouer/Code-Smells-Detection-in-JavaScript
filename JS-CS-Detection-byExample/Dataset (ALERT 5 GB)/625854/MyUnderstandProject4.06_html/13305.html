162          * {@inheritDoc}
163          */
164         public final void requestFocus() throws EngineStateException {
165             checkEngineState(DEALLOCATED | DEALLOCATING_RESOURCES);
166     
167             if (testEngineState(ALLOCATING_RESOURCES)) {
168                 try {
169                     waitEngineState(ALLOCATED);
170                 } catch (InterruptedException ex) {
171                     return;
172                 }
173             }
174     
175             // Do nothing if the state is already OK
176             if (testEngineState(FOCUSED)) {
177                 return;
178             }
179     
180             long[] states = setEngineState(DEFOCUSED, FOCUSED);
181             postEngineEvent(states[0], states[1], EngineEvent.ENGINE_FOCUSED);
182     
183             notifyGrammarActivation();
184             handleRequestFocus();
185         }

188          * {@inheritDoc}
189          */
190         public final void releaseFocus() throws EngineStateException {
191             checkEngineState(DEALLOCATED | DEALLOCATING_RESOURCES);
192     
193             if (testEngineState(ALLOCATING_RESOURCES)) {
194                 try {
195                     waitEngineState(ALLOCATED);
196                 } catch (InterruptedException ex) {
197                     return;
198                 }
199             }
200     
201             if (testEngineState(DEFOCUSED)) {
202                 return;
203             }
204     
205             long[] states = setEngineState(FOCUSED, DEFOCUSED);
206             postEngineEvent(states[0], states[1], EngineEvent.ENGINE_DEFOCUSED);
207     
208             notifyGrammarActivation();
209             handleReleaseFocus();
210         }

261          * {@inheritDoc}
262          */
263         public void fireEvent(final EngineEvent event) {
264             synchronized (engineListeners) {
265                 final RecognizerEvent recognizerEvent =
266                     (RecognizerEvent) event;
267                 Enumeration enumeration = engineListeners.elements();
268                 while (enumeration.hasMoreElements()) {
269                     final RecognizerListener listener =
270                         (RecognizerListener) enumeration.nextElement();
271                     listener.recognizerUpdate(recognizerEvent);
272                 }
273             }
274         }

291         public void postEngineEvent(long oldState, long newState, int eventType,

293             switch (eventType) {
294             case RecognizerEvent.SPEECH_STARTED:
295             case RecognizerEvent.SPEECH_STOPPED:
296             case RecognizerEvent.RECOGNIZER_BUFFERING:
297             case RecognizerEvent.RECOGNIZER_NOT_BUFFERING:
298                 break;
299             default:
300                 audioPosition = RecognizerEvent.UNKNOWN_AUDIO_POSITION;
301             }
302     
303             final RecognizerEvent event = new RecognizerEvent(this,
304                     eventType,
305                     oldState,
306                     newState,
307                     null,
308                     null,
309                     audioPosition);
310             postEngineEvent(event);
311         }

313         protected void postResultEvent(final ResultEvent event) {
314             final SpeechEventExecutor executor = getSpeechEventExecutor();
315             try {
316                 executor.execute(new Runnable() {
317                     public void run() {
318                         fireResultEvent(event);
319                     }
320                 });
321             } catch (RuntimeException e) {
322                 e.printStackTrace();
323             }
324             final Result result = (Result) event.getSource();
325             postResultEvent(result, event, executor);
326         }

338         public void fireResultEvent(final ResultEvent event) {
339             Enumeration listeners = resultListeners.elements();
340             while (listeners.hasMoreElements()) {
341                 ResultListener el = (ResultListener) listeners.nextElement();
342                 ((ResultListener) el).resultUpdate((ResultEvent) event);
343             }

426         /**
427          *
428          * @throws EngineStateException
429          */
430         public void processGrammars() throws EngineStateException {
431     
432             //Flag that indicates if grammars were changed
433             boolean existChanges = false;
434     
435             //Build a new grammar set, with all enabled grammars
436             Vector newGrammars = new Vector();
437     
438             //Commit all grammars pending changes
439             final Grammar[] grammars = grammarManager.listGrammars();
440             for (int i = 0; i < grammars.length; i++) {
441                 final Grammar grammar = grammars[i];
442                 boolean updated;
443                 try {
444                     updated = processGrammar(grammar);
445                 } catch (GrammarException e) {
446                     updated = false;
447                     e.printStackTrace();
448                 }
449                 // Update "modified-flag"
450                 existChanges = existChanges || updated;
451                 if (grammar.isActivatable()) {
452                     final GrammarDefinition definition =
453                         new GrammarDefinition(grammar.toString(),
454                                 grammar.getReference(), existChanges);
455                     newGrammars.addElement(definition);
456                 }
457             }
458     
459             // Set grammars
460             boolean setGrammarsResult = setGrammars(newGrammars);
461     
462             //Raise proper events
463             if (existChanges) {
464                 if (setGrammarsResult) {
465                     postEngineEvent(PAUSED, RESUMED,
466                             RecognizerEvent.CHANGES_COMMITTED);
467                     for (int i = 0; i < grammars.length; i++) {
468                         final BaseGrammar baseGrammar = (BaseGrammar) grammars[i];
469                         baseGrammar.postGrammarChangesCommitted();
470                     }
471                 } else {
472                     for (int i = 0; i < grammars.length; i++) {
473                         final BaseGrammar baseGrammar = (BaseGrammar) grammars[i];
474                         baseGrammar.postGrammarChangesRejected();
475                     }
476                 }
477             }
478         }

514          * Called from the <code>allocate</code> method.
515          *
516          * @see #allocate
517          *
518          * @throws AudioException
519          *          if any audio request fails 
520          * @throws EngineException
521          *          if an allocation error occurred or the Engine is not
522          *          operational.

524          *          if called for an Engine in the DEALLOCATING_RESOURCES state 
525          * @throws SecurityException

528         protected final void baseAllocate() throws EngineStateException,
529                 EngineException, AudioException, SecurityException {
530             final AudioManager audioManager = getAudioManager();
531             audioManager.audioStart();
532     
533             // Proceed to real engine allocation
534             handleAllocate();
535             long[] states = setEngineState(CLEAR_ALL_STATE,
536                     ALLOCATED | PAUSED | DEFOCUSED |
537                     NOT_BUFFERING);
538             postEngineEvent(states[0], states[1], EngineEvent.ENGINE_ALLOCATED);
539         }

600         /**
601          * Called from the <code>resume</code> method.  Override in subclasses.
602          *
603          * @todo Handle grammar updates
604          */
605         protected final boolean baseResume() throws EngineStateException {
606     
607             //Process grammars
608             processGrammars();
609     
610             boolean status = handleResume();
611             if (status) {
612                 setEngineState(0, LISTENING);
613                 postEngineEvent(0, LISTENING, RecognizerEvent.RECOGNIZER_LISTENING);
614                 setEngineState(NOT_BUFFERING, BUFFERING);
615             }
616     
617             return status;
618         }
