33     public class Voice {

36         public static final int AGE_TEENAGER = 0x80020000;
37     
38         public static final int AGE_YOUNGER_ADULT = 0x80040000;
39     
40         public static final int AGE_MIDDLE_ADULT = 0x80080000;
41     
42         public static final int AGE_OLDER_ADULT = 0x80100000;
43     
44         public static final int AGE_SPECIFIC = 0xC0000000;
45     
46         public static final int AGE_DONT_CARE = -1;
47     
48         public static final int GENDER_FEMALE = 0x1;
49     
50         public static final int GENDER_MALE = 0x2;
51     
52         public static final int GENDER_NEUTRAL = 0x4;
53     
54         public static final int GENDER_DONT_CARE = -1;
55     
56         public static final int VARIANT_DEFAULT = 1;

58         public static final int VARIANT_DONT_CARE = -1;

70         public Voice() {
71             gender = GENDER_DONT_CARE;
72             age = AGE_DONT_CARE;
73             variant = VARIANT_DONT_CARE;
74             locale = null;
75             name = null;
76         }

78         public Voice(SpeechLocale locale, String name, int gender, int age,
79                 int variant) throws IllegalArgumentException {
80             if ((age < 0) && (age != AGE_DONT_CARE)) {
81                 if ((age & 0xFFFF0000) != AGE_SPECIFIC) {
82                     if ((age & 0x7FE0FFFF) != 0) {
83                         throw new IllegalArgumentException(
84                                 "Age must be a positive integer or AGE_DONT_CARE!");
85                     }
86                 }
87             }
88             if ((variant < 0) && (variant != VARIANT_DONT_CARE)) {
89                 throw new IllegalArgumentException(
90                         "Variant must be a positive integer or VARIANT_DONT_CARE!");
91             }
92             if ((gender != GENDER_DONT_CARE) && (gender != GENDER_FEMALE)
93                     && (gender != GENDER_MALE) && (gender != GENDER_NEUTRAL)) {
94                 throw new IllegalArgumentException(gender
95                         + " is not a valid value for gender!");
96             }
97             this.locale = locale;
98             this.name = name;
99             this.gender = gender;
100             this.age = age;
101             this.variant = variant;
102         }

124         public int hashCode() {
125             final int prime = 31;
126             int result = 1;
127             result = prime * result + age;
128             result = prime * result + gender;
129             result = prime * result + ((locale == null) ? 0 : locale.hashCode());
130             result = prime * result + ((name == null) ? 0 : name.hashCode());
131             result = prime * result + variant;
132             return result;
133         }

143                 return false;

191         public boolean match(Voice require) {
192             if (require == null) {
193                 return true;
194             }
195     
196             final boolean namesMatch;
197             final String requiredName = require.getName();
198             if ((requiredName == null) || (requiredName.length() == 0)) {
199                 namesMatch = true;
200             } else {
201                 namesMatch = requiredName.equals(name);
202             }
203     
204             final boolean genderMatch;
205             final int requiredGender = require.getGender();
206             if (requiredGender == GENDER_DONT_CARE) {
207               genderMatch = true;
208             } else {
209               genderMatch = (gender == requiredGender);
210             }
211             final boolean localeMatch;
212             final SpeechLocale requiredLocale = require.getSpeechLocale();
213             if (requiredLocale == null) {
214                 localeMatch = true;
215             } else {
216                 localeMatch = requiredLocale.equals(locale);
217             }
218     
219             final boolean ageMatch;
220             final int requiredAge = require.getAge();
221             if (requiredAge == AGE_DONT_CARE) {
222                 ageMatch = true;
223             } else {
224                 final int closestAge = getClosestAge(requiredAge);
225                 ageMatch = (age == closestAge);
226             }
227     
228             final boolean variantMatch;
229             final int requiredVariant = require.getVariant();
230             if (requiredVariant == VARIANT_DONT_CARE) {
231                 variantMatch = true;
232             } else {
233                 variantMatch = (variant == requiredVariant);
234             }
235     
236             return namesMatch && genderMatch && localeMatch && ageMatch
237                     && variantMatch;
238         }

240         /**
241          * Determines the age closest to the given age.
242          *
243          * @param age
244          *            the given age.
245          * @return Age defined as a constant closest to the given age.
246          */
247         private int getClosestAge(int age) {
248             if (age <= AGE_CHILD + (AGE_TEENAGER - AGE_CHILD) / 2) {
249                 return AGE_CHILD;
250             }
251     
252             if (age <= AGE_TEENAGER + (AGE_YOUNGER_ADULT - AGE_TEENAGER) / 2) {
253                 return AGE_TEENAGER;
254             }
255     
256             if (age <= AGE_YOUNGER_ADULT + (AGE_MIDDLE_ADULT - AGE_YOUNGER_ADULT)
257                     / 2) {
258                 return AGE_YOUNGER_ADULT;
259             }
260     
261             if (age <= AGE_MIDDLE_ADULT + (AGE_OLDER_ADULT - AGE_MIDDLE_ADULT) / 2) {
262                 return AGE_MIDDLE_ADULT;
263             }
264     
265             return AGE_OLDER_ADULT;
266         }
