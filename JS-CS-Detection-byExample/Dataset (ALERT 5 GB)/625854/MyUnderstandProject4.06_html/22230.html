135         private Rule[] parseGrammar(Node grammarNode) {
136             ArrayList<Rule> rules = new ArrayList<Rule>();
137     
138     //        try{
139     //            String root = xpath.evaluate("@root", grammarNode);
140     //            attributes.remove("root");
141     //            attributes.put("root", root);
142     //            System.out.println("new root : "+ root);
143     //            
144     //        }catch (XPathExpressionException ex) {
145     //            ex.printStackTrace();
146     //        }
147             
148             
149             try {
150                 NodeList ruleNodes = (NodeList) xpath.evaluate("rule", grammarNode,
151                         XPathConstants.NODESET);
152                 for (int i = 0; i < ruleNodes.getLength(); i++) {
153                     Node ruleNode = ruleNodes.item(i);
154                     String ruleId = xpath.evaluate("@id", ruleNode);
155                     int scope = Rule.PRIVATE;
156                     String scopeStr = xpath.evaluate("@scope", ruleNode);
157                     if (scopeStr != "") {
158                         if (scopeStr.equalsIgnoreCase("public"))
159                             scope = Rule.PUBLIC;
160                     }
161     
162                     ArrayList<RuleComponent> rcs = evalChildNodes(ruleNode);
163                     if (rcs.size() == 1) {
164                         Rule rule = new Rule(ruleId, rcs.get(0), scope);
165                         rules.add(rule);
166                     } else if (rcs.size() > 1) {
167                         RuleSequence rs = new RuleSequence(rcs.toArray(new
168                                 RuleComponent[] {}));
169                         Rule rule = new Rule(ruleId, rs, scope);
170                         rules.add(rule);
171                     }
172                 }
173             } catch (XPathExpressionException ex) {
174                 ex.printStackTrace();
175             }
176             return rules.toArray(new Rule[] {});
177         }

180         private ArrayList<RuleComponent> evalNode(Node node) throws
181                 XPathExpressionException {
182             ArrayList<RuleComponent> ruleComponents = new ArrayList<RuleComponent>();
183             String nodeName = node.getNodeName();
184             if (nodeName.equalsIgnoreCase("#text")) {
185                 String text = node.getNodeValue().trim();
186                 if (text.length() > 0) {
187                     RuleToken ruleToken = new RuleToken(text);
188                     ruleComponents.add(ruleToken);
189                 }
190             } else if (nodeName.equalsIgnoreCase("one-of")) {
191                 ArrayList<RuleComponent> rcs = evalChildNodes(node);
192                 RuleAlternatives ra = new RuleAlternatives(rcs.toArray(new
193                         RuleComponent[] {}));
194                 ruleComponents.add(ra);
195             } else if (nodeName.equalsIgnoreCase("item")) {
196                 int repeatMin = -1;
197                 int repeatMax = -1;
198                 double repeatProb = -1;
199                 try {
200                     String repeatStr = xpath.evaluate("@repeat", node);
201                     String repeatProbStr = xpath.evaluate("@repeat-prob", node);
202     
203                     if (repeatStr != "") {
204                         String minStr = repeatStr.substring(0,
205                                 repeatStr.indexOf('-'));
206                         String maxStr = repeatStr.substring(repeatStr.indexOf('-') +
207                                 1);
208                         if (minStr.trim().length() > 0)
209                             repeatMin = Integer.parseInt(minStr);
210                         if (maxStr.trim().length() > 0)
211                             repeatMax = Integer.parseInt(maxStr);
212                     }
213     
214                     if (repeatProbStr != "") {
215                         repeatProb = Double.parseDouble(repeatProbStr);
216                     }
217     
218                 } catch (XPathExpressionException ex) {
219                     ex.printStackTrace();
220                 }
221     
222                 ArrayList<RuleComponent> rcs = evalChildNodes(node);
223                 RuleSequence rs = new RuleSequence(rcs.toArray(new RuleComponent[] {}));
224     
225                 if ((repeatMin != -1) && (repeatMax != -1) && (repeatProb != -1)) {
226                     RuleCount rc = new RuleCount(rs, repeatMin, repeatMax, (int)(repeatProb*RuleCount.MAX_PROBABILITY));
227                     ruleComponents.add(rc);
228                 }
229                 else if ((repeatMin != -1) && (repeatMax != -1)) {
230                     RuleCount rc = new RuleCount(rs, repeatMin, repeatMax);
231                     ruleComponents.add(rc);
232                 }
233                 else if (repeatMin != -1) {
234                     if (repeatProb != -1){
235                         RuleCount rc = new RuleCount(rs, repeatMin, RuleCount.REPEAT_INDEFINITELY, (int)(repeatProb*RuleCount.MAX_PROBABILITY));
236                         ruleComponents.add(rc);
237                     } else {
238                         RuleCount rc = new RuleCount(rs, repeatMin);
239                         ruleComponents.add(rc);
240                     }
241                 }
242                 else {
243                     ruleComponents.add(rs);
244                 }
245     
246             } else if (nodeName.equalsIgnoreCase("ruleref")) {
247                 String specialStr = (String) xpath.evaluate("@special", node);
248                 if (specialStr != "") {
249                     if (specialStr.equalsIgnoreCase("NULL")) {
250                         ruleComponents.add(RuleSpecial.NULL);
251                     } else if (specialStr.equalsIgnoreCase("VOID")) {
252                         ruleComponents.add(RuleSpecial.VOID);
253                     } else if (specialStr.equalsIgnoreCase("GARBAGE")) {
254                         ruleComponents.add(RuleSpecial.GARBAGE);
255                     }
256                 } else {
257                     String uriStr = (String) xpath.evaluate("@uri", node);
258                     String ruleName = uriStr.substring(uriStr.indexOf("#") + 1);
259                     String grammarName = uriStr.substring(0, uriStr.indexOf("#"));
260                     if ((uriStr.indexOf("#") == -1) || (grammarName.length() < 1)) {
261                         ruleComponents.add(new RuleReference(ruleName));
262                     } else {
263                         String typeStr = (String) xpath.evaluate("@type", node);
264                         if (typeStr == "") {
265                             ruleComponents.add(new RuleReference(grammarName,
266                                     ruleName));
267                         } else {
268                             ruleComponents.add(new RuleReference(grammarName,
269                                     ruleName, typeStr));
270                         }
271                     }
272                 }
273             } else if (nodeName.equalsIgnoreCase("token")) {
274                 String tokenText = (String) xpath.evaluate(".", node,
275                         XPathConstants.STRING);
276                 RuleToken ruleToken = new RuleToken(tokenText);
277                 ruleComponents.add(ruleToken);
278             } else if (nodeName.equalsIgnoreCase("tag")) {
279                 Object tagObject = xpath.evaluate(".", node, XPathConstants.STRING);
280                 RuleTag ruleTag = new RuleTag(tagObject);
281                 ruleComponents.add(ruleTag);
282             } else if (nodeName.equalsIgnoreCase("example")) {
283                 //Ignore
284             }
285     
286             return ruleComponents;
287         }

289         private ArrayList<RuleComponent> evalChildNodes(Node nodes) throws
290                 XPathExpressionException {
291             ArrayList<RuleComponent> ruleComponents = new ArrayList<RuleComponent>();
292             NodeList childs = (NodeList) xpath.evaluate("child::node()", nodes,
293                     XPathConstants.NODESET);
294             for (int i = 0; i < childs.getLength(); i++) {
295                 ruleComponents.addAll(evalNode(childs.item(i)));
296             }
297     
298             return ruleComponents;
299         }
