100         @Override
101         public Data getData() throws DataProcessingException {
102             Data data = getPredecessor().getData();
103     
104             if (data instanceof DataStartSignal)
105                 sampleRate = ((DataStartSignal) data).getSampleRate();
106     
107             if (data instanceof DataStartSignal || (data instanceof SpeechStartSignal && captureUtts)) {
108                 baos = new ByteArrayOutputStream();
109                 dos = new DataOutputStream(baos);
110             }
111     
112     
113             if ((data instanceof DataEndSignal && !captureUtts) || (data instanceof SpeechEndSignal && captureUtts)) {
114             	
115                 String wavName;
116                 if (isCompletePath)
117                     wavName = outFileNamePattern;
118                 else
119                     wavName = getNextFreeIndex(outFileNamePattern);
120     
121                 writeFile(wavName);
122     
123                 isInSpeech = false;
124             }
125     
126             if (data instanceof SpeechStartSignal)
127                 isInSpeech = true;
128     
129             if ((data instanceof DoubleData || data instanceof FloatData) && (isInSpeech || !captureUtts)) {
130                 DoubleData dd = data instanceof DoubleData ? (DoubleData) data : DataUtil.FloatData2DoubleData((FloatData) data);
131                 double[] values = dd.getValues();
132     
133                 for (double value : values) {
134                     try {
135                         dos.writeShort(new Short((short) value));
136                     } catch (IOException e) {
137                         e.printStackTrace();
138                     }
139                 }
140             }
141     
142             return data;
143         }

146         private static String getNextFreeIndex(String outPattern) {
147     
148             int fileIndex = 0;
149             String fileName;
150     
151             while (true) {
152                 String indexString = Integer.toString(fileIndex);
153     
154                 fileName = outPattern.substring(0,
155                         Math.max(0, outPattern.length() - indexString.length()))
156                         + indexString + ".wav";
157     
158                 if (!new File(fileName).isFile())
159                     break;
160     
161                 fileIndex++;
162             }
163     
164             return fileName;
165         }

186         private static AudioFileFormat.Type getTargetType(String extension) {
187             AudioFileFormat.Type[] typesSupported = AudioSystem.getAudioFileTypes();
188     
189             for (AudioFileFormat.Type aTypesSupported : typesSupported) {
190                 if (aTypesSupported.getExtension().equals(extension)) {
191                     return aTypesSupported;
192                 }
193             }
194     
195             return null;
196         }

199         /**
200          * Converts a big-endian byte array into an array of doubles. Each consecutive bytes in the byte array are converted
201          * into a double, and becomes the next element in the double array. The size of the returned array is
202          * (length/bytesPerValue). Currently, only 1 byte (8-bit) or 2 bytes (16-bit) samples are supported.
203          *
204          * @param values
205          * @param bytesPerValue the number of bytes per value
206          * @param signedData    whether the data is signed
207          * @return a double array, or <code>null</code> if byteArray is of zero length
208          * @throws ArrayIndexOutOfBoundsException
209          */
210         public static byte[] valuesToBytes(double[] values, int bytesPerValue, boolean signedData)
211                 throws ArrayIndexOutOfBoundsException {
212     
213             byte[] byteArray = new byte[bytesPerValue * values.length];
214     
215             int byteArInd = 0;
216     
217             for (double value : values) {
218                 int val = (int) value;
219     
220     
221                 for (int j = bytesPerValue - 1; j >= 0; j++) {
222                     byteArray[byteArInd + j] = (byte) (val & 0xff);
223                     val = val >> 8;
224                 }
225     
226                 byteArInd += bytesPerValue;
227             }
228     
229             return byteArray;
230         }

232         /**
233         * Writes the current stream to disc; override this method if you want to take 
234         * additional action on file writes
235         *
236         * @param wavName name of the file to be written
237         */
238         protected void writeFile(String wavName) {
239             AudioFormat wavFormat = new AudioFormat(sampleRate, bitsPerSample, 1, isSigned, true);
240             AudioFileFormat.Type outputType = getTargetType("wav");
241     
242             byte[] abAudioData = baos.toByteArray();
243             ByteArrayInputStream bais = new ByteArrayInputStream(abAudioData);
244             AudioInputStream ais = new AudioInputStream(bais, wavFormat, abAudioData.length / wavFormat.getFrameSize());
245     
246             File outWavFile = new File(wavName);
247     
248             if (AudioSystem.isFileTypeSupported(outputType, ais)) {
249                 try {
250                     AudioSystem.write(ais, outputType, outWavFile);
251                 } catch (IOException e) {
252                     e.printStackTrace();
253                 }
254             }   	
255         }
