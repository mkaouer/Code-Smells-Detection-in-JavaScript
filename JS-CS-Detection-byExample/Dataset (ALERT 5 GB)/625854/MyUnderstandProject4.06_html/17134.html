15     package edu.cmu.sphinx.result;
16     
17     import edu.cmu.sphinx.linguist.dictionary.Word;
18     import edu.cmu.sphinx.util.LogMath;
19     
20     import java.util.Iterator;
21     import java.util.List;
22     import java.util.ArrayList;
23     
24     /**
25      * An implementation of a result Path that computes scores and confidences on the fly.
26      *
27      * @author P. Gorniak
28      */
29     public class WordResultPath implements Path {
30     
31         private final List<WordResult> path = new ArrayList<WordResult>();
32     
33     
34         /**
35          * Constructs a WordResultPath with the given list of WordResults and LogMath.
36          *
37          * @param wordResults the list of WordResults
38          */
39         WordResultPath(List<WordResult> wordResults) {
40             path.addAll(wordResults);
41         }
42     
43     
44         /** Constructs an empty WordResultPath. */
45         WordResultPath() {
46         }
47     
48     
49         /** @see edu.cmu.sphinx.result.Path#getScore() */
50         @Override
51         public double getScore() {
52             double score = LogMath.getLogOne();
53             for (WordResult wr : path) {
54                 score += wr.getScore();
55             }
56             return score;
57         }
58     
59     
60         /** @see edu.cmu.sphinx.result.Path#getConfidence() */
61         @Override
62         public double getConfidence() {
63             double confidence = LogMath.getLogOne();
64             for (WordResult wr : path) {
65                 confidence += wr.getConfidence();
66             }
67             return confidence;
68         }
69     
70     
71         /**
72          * Returns the LogMath of the scores.
73          *
74          * @return the LogMath of the scores
75          */
76         @Override
77         public LogMath getLogMath() {
78             Iterator<WordResult> i = path.iterator();
79             if (i.hasNext()) {
80                 WordResult wr = i.next();
81                 return wr.getLogMath();
82             } else {
83                 return null;
84             }
85         }
86     
87     
88         /** @see edu.cmu.sphinx.result.Path#getWords() */
89         @Override
90         public WordResult[] getWords() {
91             return path.toArray(new WordResult[path.size()]);
92         }
93     
94     
95         /** @see edu.cmu.sphinx.result.Path#getTranscription() */
96         @Override
97         public String getTranscription() {
98             StringBuilder sb = new StringBuilder();
99             for (WordResult wr : path)
100                 sb.append(wr).append(' ');
101             return sb.toString().trim();
102         }
103     
104         /** @see edu.cmu.sphinx.result.Path#getTranscriptionNoFiller() */
105         public String getTranscriptionNoFiller() {
106             StringBuilder sb = new StringBuilder();
107             for (WordResult wordResult : path) {
108                 Word word = wordResult.getPronunciation().getWord();
109                 if (!word.isFiller() && !word.getSpelling().equals("<unk>")) {
110                     sb.append(word.getSpelling()).append(' ');
111                 }
112             }
113             return sb.toString().trim();
114         }
115     
116         public void add(WordResult wr) {
117             path.add(wr);
118         }
119     
120     }
