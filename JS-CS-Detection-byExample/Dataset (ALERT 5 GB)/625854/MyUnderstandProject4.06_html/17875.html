28     /**
29      * Some tests which test the functionality of <code>MixtureComponentt</code>s.
30      * <p/>
31      * Currently testing is restricted to univariate gaussians. It should be extended to test highdimensional gaussians as
32      * well.
33      */
34     public class MixtureComponentTest {
35     
36         private LogMath lm;
37     
38     
39         @Before
40         public void setup() {
41             Map<String, Object> props = new HashMap<String, Object>();
42             props.put(LogMath.PROP_USE_ADD_TABLE, false);
43     
44             lm = ConfigurationManager.getInstance(LogMath.class, props);
45         }
46     
47     
48         /**
49          * Compute the density values of a sampled interval with an univariate <code>MixtureComponent</code> and compare
50          * values with the precomputed-computed ones.
51          */
52         @Test
53         public void testUnivariateDensity() {
54     
55             float minX = 10;
56             float maxX = 30;
57             float resolution = 0.1f;
58     
59             float mean = 20;
60             float var = 3;
61     
62             MixtureComponent gaussian = new MixtureComponent(lm, new float[]{mean}, new float[]{var});
63     
64             for (float curX = minX; curX <= maxX; curX += resolution) {
65                 double gauLogScore = gaussian.getScore(new FloatData(new float[]{curX}, 0, 0, 0));
66     
67                 double manualScore = (1 / sqrt(var * 2 * PI)) * exp((-0.5 / var) * (curX - mean) * (curX - mean));
68                 double gauScore = lm.logToLinear((float) gauLogScore);
69     
70                 assertEquals(manualScore, gauScore, 1E-5);
71             }
72         }
73     
74     
75         /** Tests whether working with different types transformations works properly. */
76         @Test
77         public void testUnivariateMeanTransformation() {
78             float mean = 20;
79             float var = 0.001f;
80     
81             MixtureComponent gaussian = new MixtureComponent(lm, new float[]{mean}, new float[][]{{2}}, new float[]{5}, new float[]{var}, null, null);
82             assertTrue(lm.logToLinear(gaussian.getScore(new float[]{2 * mean + 5})) > 10);
83         }
84     
85     
86         /** Tests whether a <code>MixtureComponent</code>s can be cloned (using deep copying). */
87         @Test
88         public void testClone() throws CloneNotSupportedException {
89             MixtureComponent gaussian = new MixtureComponent(lm, new float[]{2}, new float[][]{{3}}, new float[]{4}, new float[]{5}, new float[][]{{6}}, new float[]{7});
90     
91             MixtureComponent clonedGaussian = gaussian.clone();
92     
93             assertTrue(!clonedGaussian.equals(gaussian));
94     
95             assertTrue(gaussian.getMean() != clonedGaussian.getMean());
96             assertTrue(gaussian.getVariance() != clonedGaussian.getVariance());
97             assertTrue(gaussian.getScore(new float[]{2}) == clonedGaussian.getScore(new float[]{2}));
98         }
