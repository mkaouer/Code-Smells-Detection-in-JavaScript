82         /** Dumps the information about a line. */
83         private static void dumpLineInfo(String indent,
84                                          Line.Info[] lineInfo) {
85             int numDumped = 0;
86     
87             if (lineInfo != null) {
88                 for (Line.Info info : lineInfo) {
89                     if (info instanceof DataLine.Info) {
90                         AudioFormat[] formats =
91                             ((DataLine.Info)info).getFormats();
92                         for (AudioFormat format : formats) {
93                             System.out.println(indent + format);
94                         }
95                         numDumped++;
96                     } else if (info instanceof Port.Info) {
97                         System.out.println(indent + info);
98                         numDumped++;
99                     }
100                 }
101             }
102     
103             if (numDumped == 0) {
104                 System.out.println(indent + "none");
105             }
106         }

131             int returnVal;
132     
133             fileChooser.setDialogTitle(title);
134             fileChooser.setCurrentDirectory(file);
135             fileChooser.setDialogType(type);

137             if (type == JFileChooser.OPEN_DIALOG) {
138                 returnVal = fileChooser.showOpenDialog(jframe);

150         /**
151          */
152         static public void populateAudio(String filename) {
153             try {
154                 AudioData newAudio = Utils.readAudioFile(filename);
155                 if (newAudio == null) {
156                     newAudio = Utils.readRawFile(filename);
157                 }
158                 zoomReset();
159                 audio.setAudioData(newAudio.getAudioData());
160                 /*
161                 * Play only if user requests. Auto play is annoying if
162                 * the audio is too long
163                 *
164                 * player.play(audioPanel.getSelectionStart(),
165                 * audioPanel.getSelectionEnd());
166                */
167             } catch (IOException e) {
168                 /* just ignore bad files. */
169             }
170         }

173         /**
174          */
175         static public void getAudioFromFile(String filename) throws IOException {
176             /* Supports alignment data.  The format of the alignment file
177              * is as follows:
178              *
179              * input filename                String
180              * number of (time tag) lines    int
181              * time tag                      float String
182              * time tag                      float String
183              * time tag                      float String
184              * ...
185              *
186              * Times are in seconds.
187              */
188             if (filename.endsWith(".align")) {
189                 BufferedReader reader = new BufferedReader(
190                         new InputStreamReader(new FileInputStream(filename)));
191     
192                 populateAudio(reader.readLine());
193     
194                 int numPoints = Integer.parseInt(reader.readLine());
195                 float[] times = new float[numPoints];
196                 String[] labels = new String[numPoints];
197                 for (int i = 0; i < numPoints; i++) {
198                     StringTokenizer tokenizer = new StringTokenizer(
199                             reader.readLine());
200                     while (tokenizer.hasMoreTokens()) {
201                         times[i] = Float.parseFloat(tokenizer.nextToken());
202                         labels[i] = tokenizer.nextToken();
203                     }
204                 }
205                 audioPanel.setLabels(times, labels);
206     
207                 reader.close();
208             } else {
209                 populateAudio(filename);
210             }
211         }

214         /** Gets the audio that's in the recorder.  This should only be called after recorder.stopRecording is called. */
215         static private short[] getRecordedAudio(Microphone recorder) {
216             short[] shorts = new short[0];
217             int sampleRate = 16000;
218     
219             /* [[[WDW - TODO: this is not the most efficient way
220              * to do this, but it at least works for now.]]]
221              */
222             while (recorder.hasMoreData()) {
223                 try {
224                     Data data = recorder.getData();
225                     if (data instanceof DoubleData) {
226                         sampleRate =
227                                 ((DoubleData) data).getSampleRate();
228                         double[] values =
229                                 ((DoubleData) data).getValues();
230                         short[] newShorts = Arrays.copyOf(shorts, shorts.length + values.length);
231                         for (int i = 0; i < values.length; i++) {
232                             newShorts[shorts.length + i] = (short)values[i];
233                         }
234                         shorts = newShorts;
235                     }
236                 } catch (Exception e) {
237                     e.printStackTrace();
238                 }
239             }
240     
241             if (sampleRate > 16000) {
242                 System.out.println("Downsampling from " +
243                         sampleRate + " to 16000.");
244                 shorts = Downsampler.downsample(
245                         shorts,
246                         sampleRate / 1000,
247                         16);
248             }
249     
250             return shorts;
251         }

530         /**
531          * Main method.
532          *
533          * @param args argv[0] : The name of an audio file argv[1] : SphinxProperties file
534          */
535         static public void main(String[] args) {
536             FrontEnd frontEnd;
537             StreamDataSource dataSource;
538     
539             prefs = Preferences.userRoot().node(PREFS_CONTEXT);
540             filename = prefs.get(FILENAME_PREFERENCE, "untitled.raw");
541             file = new File(filename);
542     
543             if ((args.length == 1) && (args[0].equals("-dumpMixers"))) {
544                 dumpMixers();
545                 System.exit(0);
546             }
547     
548             try {
549                 URL url;
550                 if (args.length > 0) {
551                     filename = args[0];
552                 }
553                 if (args.length == 2) {
554                     url = new File(args[1]).toURI().toURL();
555                 } else {
556                     url = AudioTool.class.getResource("spectrogram.config.xml");
557                 }
558                 ConfigurationManager cm = new ConfigurationManager(url);
559     
560                 recorder = (Microphone) cm.lookup(MICROPHONE);
561                 recorder.initialize();
562                 audio = new AudioData();
563     
564                 frontEnd = (FrontEnd) cm.lookup(FRONT_END);
565                 dataSource = (StreamDataSource) cm.lookup(DATA_SOURCE);
566     
567                 PropertySheet ps = cm.getPropertySheet(WINDOWER);
568                 float windowShiftInMs = ps.getFloat(RaisedCosineWindower.PROP_WINDOW_SHIFT_MS);
569     
570                 final JFrame jframe = new JFrame("AudioTool");
571                 fileChooser = new JFileChooser();
572                 createMenuBar(jframe);
573     
574                 /* Scale the width according to the size of the
575                 * spectrogram.
576                 */
577                 float windowShiftInSamples = windowShiftInMs
578                         * audio.getAudioFormat().getSampleRate() / 1000.0f;
579                 audioPanel = new AudioPanel(audio,
580                         1.0f / windowShiftInSamples,
581                         0.004f);
582                 spectrogramPanel = new SpectrogramPanel(frontEnd, dataSource, audio);
583     
584                 JPanel panel = new JPanel(new BorderLayout());
585                 panel.add(audioPanel, BorderLayout.CENTER);
586                 panel.add(spectrogramPanel, BorderLayout.SOUTH);
587     
588                 JScrollPane scroller = new JScrollPane(panel);
589     
590                 JPanel outerPanel = new JPanel(new BorderLayout());
591                 outerPanel.add(createButtonPanel(), BorderLayout.NORTH);
592                 outerPanel.add(scroller);
593     
594                 player = new AudioPlayer(audio);
595                 player.start();
596     
597                 getAudioFromFile(filename);
598     
599                 jframe.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
600                 jframe.setContentPane(outerPanel);
601                 jframe.pack();
602                 jframe.setSize(640, 540);
603                 jframe.setVisible(true);
604             } catch (Exception e) {
605                 e.printStackTrace();
606             }
607         }
