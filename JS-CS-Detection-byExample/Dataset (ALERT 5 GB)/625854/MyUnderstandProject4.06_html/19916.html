24     /**
25      * The HMMPool provides the ability to manage units via small integer IDs.  Context Independent units and context
26      * dependent units can be converted to an ID. IDs can be used to quickly retrieve a unit or an hmm associated with the
27      * unit.  This class operates under the constraint that context sizes are exactly one, which is generally only valid for
28      * large vocabulary tasks.
29      */
30     
31     public class HMMPool {
32     
33         private AcousticModel model;
34         private Unit[] unitTable;
35         private Map<HMMPosition, HMM[]> hmmTable;
36         private int numCIUnits;
37         private Logger logger;
38         private UnitManager unitManager;
39     
40     
41         protected HMMPool(){
42         }

44         /**
45          * Constructs a HMMPool object.
46          *
47          * @param model the model to use for the pool
48          */
49         public HMMPool(AcousticModel model, Logger logger, UnitManager unitManager) {
50             this.logger = logger;
51             int maxCIUnits = 0;
52             this.model = model;
53             this.unitManager = unitManager;
54             TimerPool.getTimer(this,"Build HMM Pool").start();
55     
56             if (model.getLeftContextSize() != 1) {
57                 throw new Error("LexTreeLinguist: Unsupported left context size");
58             }
59     
60             if (model.getRightContextSize() != 1) {
61                 throw new Error("LexTreeLinguist: Unsupported right context size");
62             }
63     
64             // count CI units:
65     
66             for (Iterator<Unit> i = model.getContextIndependentUnitIterator(); i.hasNext();) {
67                 Unit unit = i.next();
68                 logger.fine("CI unit " + unit);
69                 if (unit.getBaseID() > maxCIUnits) {
70                     maxCIUnits = unit.getBaseID();
71                 }
72             }
73     
74             numCIUnits = maxCIUnits + 1;
75     
76             unitTable = new Unit[numCIUnits * numCIUnits * numCIUnits];
77     
78             for (Iterator<HMM> i = model.getHMMIterator(); i.hasNext();) {
79                 HMM hmm = i.next();
80                 Unit unit = hmm.getUnit();
81                 int id = getID(unit);
82                 unitTable[id] = unit;
83                 if (logger.isLoggable(Level.FINER)) {
84                     logger.finer("Unit " + unit + " id " + id);
85                 }
86             }
87     
88             // build up the hmm table to allow quick access to the hmms
89             hmmTable = new EnumMap<HMMPosition, HMM[]>(HMMPosition.class);
90             for (HMMPosition position : HMMPosition.values()) {
91                 HMM[] hmms = new HMM[unitTable.length];
92                 hmmTable.put(position, hmms);
93                 for (int j = 1; j < unitTable.length; j++) {
94                     Unit unit = unitTable[j];
95                     if (unit == null) {
96                         unit = synthesizeUnit(j);
97                     }
98                     if (unit != null) {
99                         hmms[j] = model.lookupNearestHMM(unit, position, false);
100                         assert hmms[j] != null;
101                     }
102                 }
103             }
104             TimerPool.getTimer(this,"Build HMM Pool").stop();
105         }

113         /**
114          * Given a unit ID, generate a full context dependent unit that will allow us to look for a suitable hmm
115          *
116          * @param id the unit id
117          * @return a context dependent unit for the ID
118          */
119         private Unit synthesizeUnit(int id) {
120             int centralID = getCentralUnitID(id);
121             int leftID = getLeftUnitID(id);
122             int rightID = getRightUnitID(id);
123     
124             if (centralID == 0 || leftID == 0 || rightID == 0) {
125                 return null;
126             }
127     
128             Unit centralUnit = unitTable[centralID];
129             Unit leftUnit = unitTable[leftID];
130             Unit rightUnit = unitTable[rightID];
131     
132             assert centralUnit != null;
133             assert leftUnit != null;
134             assert rightUnit != null;
135     
136             Unit[] lc = new Unit[1];
137             Unit[] rc = new Unit[1];
138             lc[0] = leftUnit;
139             rc[0] = rightUnit;
140             LeftRightContext context = LeftRightContext.get(lc, rc);
141     
142             Unit unit = unitManager.getUnit(
143                     centralUnit.getName(), centralUnit.isFiller(),
144                     context);
145     
146             if (logger.isLoggable(Level.FINER)) {
147                 logger.finer("Missing " + getUnitNameFromID(id)
148                         + " returning " + unit);
149             }
150             return unit;
151         }

187         /**
188          * given a unit return its ID
189          *
190          * @param unit the unit
191          * @return an ID
192          */
193         public int getID(Unit unit) {
194             if (unit.isContextDependent()) {
195                 LeftRightContext context = (LeftRightContext) unit.getContext();
196                 assert context.getLeftContext().length == 1;
197                 assert context.getRightContext().length == 1;
198                 return buildID(
199                         getSimpleUnitID(unit),
200                         getSimpleUnitID(context.getLeftContext()[0]),
201                         getSimpleUnitID(context.getRightContext()[0]));
202             } else {
203                 return getSimpleUnitID(unit);
204             }
205         }

208         /**
209          * Returns a context independent ID
210          *
211          * @param unit the unit of interest
212          * @return the ID of the central unit (ignoring any context)
213          */
214         private int getSimpleUnitID(Unit unit) {
215             return unit.getBaseID();
216         }

219         public boolean isValidID(int unitID) {
220             return unitID >= 0 && unitID < unitTable.length && unitTable[unitID] != null;
221         }
222     
223     
224         /**
225          * Builds an id from the given unit and its left and right unit ids
226          *
227          * @param unitID  the id of the central unit
228          * @param leftID  the id of the left context unit
229          * @param rightID the id of the right context unit
230          * @return the id for the context dependent unit
231          */
232         public int buildID(int unitID, int leftID, int rightID) {
233             // special case ... if the unitID is associated with
234             // filler than we have no context ... so use the CI
235             // form
236     
237             if (unitTable[unitID] == null) {
238                 return -1;
239             }
240     
241             int id;
242             if (unitTable[unitID].isFiller()) {
243                 id = unitID;
244             } else {
245                 id = unitID * (numCIUnits * numCIUnits)
246                         + (leftID * numCIUnits)
247                         + rightID;
248             }
249     
250             assert id < unitTable.length;
251     
252             return id;
253         }

256         /**
257          * Given a unit id extract the left context unit id
258          *
259          * @param id the unit id
260          * @return the unit id of the left context (0 means no left context)
261          */
262         private int getLeftUnitID(int id) {
263             return (id / numCIUnits) % numCIUnits;
264         }
265     
266     
267         /**
268          * Given a unit id extract the right context unit id
269          *
270          * @param id the unit id
271          * @return the unit id of the right context (0 means no right context)
272          */
273         private int getRightUnitID(int id) {
274             return id % numCIUnits;
275         }

278         /**
279          * Given a unit id extract the central unit id
280          *
281          * @param id the unit id
282          * @return the central unit id
283          */
284         private int getCentralUnitID(int id) {
285             return id / (numCIUnits * numCIUnits);

289         /**
290          * Given an ID, build up a name for display
291          *
292          * @return the name baed on the ID
293          */
294         private String getUnitNameFromID(int id) {
295             int centralID = getCentralUnitID(id);
296             int leftID = getLeftUnitID(id);
297             int rightID = getRightUnitID(id);
298     
299             String cs = unitTable[centralID] == null ? "(" + centralID + ')' :
300                     unitTable[centralID].toString();
301             String ls = unitTable[leftID] == null ? ("(" + leftID + ')') :
302                     unitTable[leftID].toString();
303             String rs = unitTable[rightID] == null ? "(" + rightID + ')' :
304                     unitTable[rightID].toString();
305     
306             return cs + '[' + ls + ',' + rs + ']';
307         }

309         /**
310          * Retrieves an HMM for a unit in context. If there is no direct match, the
311          * nearest match will be used. Note that we are currently only dealing with,
312          * at most, single unit left and right contexts.
313          * 
314          * @param base
315          *            the base CI unit
316          * @param lc
317          *            the left context
318          * @param rc
319          *            the right context
320          * @param pos
321          *            the position of the base unit within the word
322          * @return the HMM. (This should never return null)
323          */
324         public HMM getHMM(Unit base, Unit lc, Unit rc, HMMPosition pos) {
325             int id = -1;
326             int bid = getID(base);
327             int lid = getID(lc);
328             int rid = getID(rc);
329     
330             if (!isValidID(bid)) {
331                 logger.severe("Bad HMM Unit: " + base.getName());
332                 return null;
333             }
334             if (!isValidID(lid)) {
335                 logger.severe("Bad HMM Unit: " + lc.getName());
336                 return null;
337             }
338             if (!isValidID(rid)) {
339                 logger.severe("Bad HMM Unit: " + rc.getName());
340                 return null;
341             }
342             id = buildID(bid, lid, rid);
343             if (id < 0) {
344                 logger.severe("Unable to build HMM Unit ID for " + base.getName()
345                         + " lc=" + lc.getName() + " rc=" + rc.getName());
346                 return null;
347             }
348             HMM hmm = getHMM(id, pos);
349             if (hmm == null) {
350                 logger.severe("Missing HMM Unit for " + base.getName() + " lc="
351                         + lc.getName() + " rc=" + rc.getName());
352             }
353     
354             return hmm;
355         }

372         /**
373          * A quick and dirty benchmark to get an idea how long the HMM lookups will take.  This experiment shows that on a
374          * 1GHZ sparc system, the lookup takes a little less than 1uSec.  This is probably fast enough.
375          */
376     
377         static final HMMPosition[] pos = {
378                 HMMPosition.BEGIN, HMMPosition.END, HMMPosition.SINGLE,
379                 HMMPosition.INTERNAL};    
380         
381         static final int[] ids = {9206, 9320, 9620, 9865, 14831, 15836};
382     
383         void benchmark() {
384             int nullCount = 0;
385             System.out.println("benchmarking ...");
386             TimerPool.getTimer(this,"hmmPoolBenchmark").start();
387     
388             for (int i = 0; i < 1000000; i++) {
389                 int id = ids[i % ids.length];
390                 HMMPosition position = pos[i % pos.length];
391                 HMM hmm = getHMM(id, position);
392                 if (hmm == null) {
393                     nullCount++;
394                 }
395             }
396             TimerPool.getTimer(this,"hmmPoolBenchmark").stop();
397             System.out.println("null count " + nullCount);
398         }
