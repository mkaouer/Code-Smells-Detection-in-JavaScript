13     package edu.cmu.sphinx.decoder.search;
14     
15     import java.util.Arrays;
16     import java.util.HashMap;
17     import java.util.Map;
18     
19     import edu.cmu.sphinx.decoder.scorer.Scoreable;
20     import edu.cmu.sphinx.linguist.SearchState;
21     /**
22      * The token heap search manager that maintains the heap of best tokens for each
23      * search state instead of single one best token
24      * 
25      */
26     public class TokenHeapSearchManager extends WordPruningBreadthFirstSearchManager {
27     
28         protected final int maxTokenHeapSize = 3;
29     
30         Map<Object, TokenHeap> bestTokenMap;
31     
32         @Override
33         protected void createBestTokenMap() {
34             int mapSize = activeList.size() << 2;
35             if (mapSize == 0) {
36                 mapSize = 1;
37             }
38             bestTokenMap = new HashMap<Object, TokenHeap>(mapSize, 0.3F);
39         }
40     
41         @Override
42         protected void setBestToken(Token token, SearchState state) {
43             Object key = getStateKey(state);
44     
45             TokenHeap th = (TokenHeap) bestTokenMap.get(key);
46             if (th == null) {
47                 th = new TokenHeap(maxTokenHeapSize);
48                 bestTokenMap.put(key, th);
49             }
50             th.add(token);
51         }
52     
53         @Override
54         protected Token getBestToken(SearchState state) {
55             Object key = getStateKey(state);
56     
57             // new way... if the heap for this state isn't full return
58             // null, otherwise return the worst scoring token
59             TokenHeap th = (TokenHeap) bestTokenMap.get(key);
60             Token t;
61     
62             if (th == null) {
63                 return null;
64             } else if ((t = th.get(state)) != null) {
65                 return t;
66             } else if (!th.isFull()) {
67                 return null;
68             } else {
69                 return th.getSmallest();
70             }
71         }
72     
73         /**
74          * A quick and dirty token heap that allows us to perform token stack
75          * experiments. It is not very efficient. We will likely replace this with
76          * something better once we figure out how we want to prune things.
77          */
78     
79         class TokenHeap {
80     
81             final Token[] tokens;
82             int curSize;
83     
84             /**
85              * Creates a token heap with the maximum size
86              * 
87              * @param maxSize
88              *            the maximum size of the heap
89              */
90             TokenHeap(int maxSize) {
91                 tokens = new Token[maxSize];
92             }
93     
94             /**
95              * Adds a token to the heap
96              * 
97              * @param token
98              *            the token to add
99              */
100             void add(Token token) {
101                 // first, if an identical state exists, replace
102                 // it.
103     
104                 if (!tryReplace(token)) {
105                     if (curSize < tokens.length) {
106                         tokens[curSize++] = token;
107                     } else if (token.getScore() > tokens[curSize - 1].getScore()) {
108                         tokens[curSize - 1] = token;
109                     }
110                 }
111                 fixupInsert();
112             }
113     
114             /**
115              * Returns the smallest scoring token on the heap
116              * 
117              * @return the smallest scoring token
118              */
119             Token getSmallest() {
120                 if (curSize == 0) {
121                     return null;
122                 } else {
123                     return tokens[curSize - 1];
124                 }
125             }
126     
127             /**
128              * Determines if the heap is full
129              * 
130              * @return <code>true</code> if the heap is full
131              */
132             boolean isFull() {
133                 return curSize == tokens.length;
134             }
135     
136             /**
137              * Checks to see if there is already a token t on the heap that has the
138              * same search state. If so, this token replaces that one
139              * 
140              * @param t
141              *            the token to try to add to the heap
142              * @return <code>true</code> if the token was added
143              */
144             private boolean tryReplace(Token t) {
145                 for (int i = 0; i < curSize; i++) {
146                     if (t.getSearchState().equals(tokens[i].getSearchState())) {
147                         assert t.getScore() > tokens[i].getScore();
148                         tokens[i] = t;
149                         return true;
150                     }
151                 }
152                 return false;
153             }
154     
155             /** Orders the heap after an insert */
156             private void fixupInsert() {
157                 Arrays.sort(tokens, 0, curSize - 1, Scoreable.COMPARATOR);
158             }
159     
160             /**
161              * returns a token on the heap that matches the given search state
162              * 
163              * @param s
164              *            the search state
165              * @return the token that matches, or null
166              */
167             Token get(SearchState s) {
168                 for (int i = 0; i < curSize; i++) {
169                     if (tokens[i].getSearchState().equals(s)) {
170                         return tokens[i];
171                     }
172                 }
173                 return null;
174             }
175         }
176     }
