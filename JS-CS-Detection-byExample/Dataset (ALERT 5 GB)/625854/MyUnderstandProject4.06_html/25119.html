13     package edu.cmu.sphinx.instrumentation;
14     
15     import edu.cmu.sphinx.recognizer.Recognizer;
16     import edu.cmu.sphinx.recognizer.Recognizer.State;
17     import edu.cmu.sphinx.recognizer.StateListener;
18     import edu.cmu.sphinx.result.Result;
19     import edu.cmu.sphinx.decoder.ResultListener;
20     import edu.cmu.sphinx.util.NISTAlign;
21     import edu.cmu.sphinx.util.props.*;
22     
23     /** Tracks and reports recognition accuracy */
24     abstract public class AccuracyTracker
25             implements
26             ResultListener,
27             Resetable,
28             StateListener,
29             Monitor {
30     
31         /** The property that defines which recognizer to monitor */
32         @S4Component(type = Recognizer.class)
33         public final static String PROP_RECOGNIZER = "recognizer";
34     
35         /** The property that defines whether summary accuracy information is displayed */
36         @S4Boolean(defaultValue = true)
37         public final static String PROP_SHOW_SUMMARY = "showSummary";
38     
39         /** The property that defines whether detailed accuracy information is displayed */
40         @S4Boolean(defaultValue = true)
41         public final static String PROP_SHOW_DETAILS = "showDetails";
42     
43         /** The property that defines whether recognition results should be displayed. */
44         @S4Boolean(defaultValue = true)
45         public final static String PROP_SHOW_RESULTS = "showResults";
46     
47     
48         /** The property that defines whether recognition results should be displayed. */
49         @S4Boolean(defaultValue = true)
50         public final static String PROP_SHOW_ALIGNED_RESULTS = "showAlignedResults";
51     
52         /** The property that defines whether recognition results should be displayed. */
53         @S4Boolean(defaultValue = true)
54         public final static String PROP_SHOW_RAW_RESULTS = "showRawResults";
55     
56         // ------------------------------
57         // Configuration data
58         // ------------------------------
59         private String name;
60         private Recognizer recognizer;
61         private boolean showSummary;
62         private boolean showDetails;
63         private boolean showResults;
64         private boolean showAlignedResults;
65         private boolean showRaw;
66     
67         private final NISTAlign aligner = new NISTAlign(false, false);
68     
69         public AccuracyTracker(Recognizer recognizer, boolean showSummary, boolean showDetails, boolean showResults, boolean showAlignedResults, boolean showRawResults) {
70     
71             initRecognizer(recognizer);
72     
73             this.showSummary = showSummary;
74             this.showDetails = showDetails;
75             this.showResults = showResults;
76             this.showAlignedResults = showAlignedResults;
77     
78             this.showRaw = showRawResults;
79     
80             aligner.setShowResults(showResults);
81             aligner.setShowAlignedResults(showAlignedResults);
82         }
83     
84         public AccuracyTracker() {
85         }
86     
87         /*
88         * (non-Javadoc)
89         *
90         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
91         */
92         @Override
93         public void newProperties(PropertySheet ps) throws PropertyException {
94     
95             initRecognizer((Recognizer) ps.getComponent(PROP_RECOGNIZER));
96     
97             showSummary = ps.getBoolean(PROP_SHOW_SUMMARY);
98             showDetails = ps.getBoolean(PROP_SHOW_DETAILS);
99             showResults = ps.getBoolean(PROP_SHOW_RESULTS);
100             showAlignedResults = ps.getBoolean(PROP_SHOW_ALIGNED_RESULTS);
101     
102             showRaw = ps.getBoolean(PROP_SHOW_RAW_RESULTS);
103     
104             aligner.setShowResults(showResults);
105             aligner.setShowAlignedResults(showAlignedResults);
106         }
107     
108         private void initRecognizer(Recognizer newRecognizer) {
109             if (recognizer == null) {
110                 recognizer = newRecognizer;
111                 recognizer.addResultListener(this);
112                 recognizer.addStateListener(this);
113             } else if (recognizer != newRecognizer) {
114                 recognizer.removeResultListener(this);
115                 recognizer.removeStateListener(this);
116                 recognizer = newRecognizer;
117                 recognizer.addResultListener(this);
118                 recognizer.addStateListener(this);
119             }
120         }
121     
122     
123         /*
124          * (non-Javadoc)
125          * 
126          * @see edu.cmu.sphinx.instrumentation.Resetable
127          */
128         @Override
129         public void reset() {
130             aligner.resetTotals();
131         }
132     
133     
134         /*
135         * (non-Javadoc)
136         *
137         * @see edu.cmu.sphinx.util.props.Configurable#getName()
138         */
139         public String getName() {
140             return name;
141         }
142     
143     
144         /**
145          * Retrieves the aligner used to track the accuracy stats
146          *
147          * @return the aligner
148          */
149         public NISTAlign getAligner() {
150             return aligner;
151         }
152     
153     
154         /**
155          * Shows the complete details.
156          *
157          * @param rawText the RAW result
158          */
159         protected void showDetails(String rawText) {
160             if (showDetails) {
161                 System.out.println();
162                 aligner.printSentenceSummary();
163                 if (showRaw) {
164                     System.out.println("RAW     " + rawText);
165                 }
166                 System.out.println();
167                 aligner.printTotalSummary();
168             }
169         }
170     
171     
172         /*
173         * (non-Javadoc)
174         *
175         * @see edu.cmu.sphinx.decoder.ResultListener#newResult(edu.cmu.sphinx.result.Result)
176         */
177         @Override
178         abstract public void newResult(Result result);
179     
180         @Override
181         public void statusChanged(Recognizer.State status) {
182             if (status == State.DEALLOCATED) {
183                 if (showSummary) {
184                     System.out.println("\n# --------------- Summary statistics ---------");
185                     aligner.printTotalSummary();
186                 }
187             }
188         }
189     }
