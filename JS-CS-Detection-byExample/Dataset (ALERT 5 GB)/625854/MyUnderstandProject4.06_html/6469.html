45         @Override
46         public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
47             if (qName.equals("config")) {
48                 // test if this configuration extends another one
49                 String extendedConfigName = attributes.getValue("extends");
50                 if (extendedConfigName != null) {
51                     mergeConfigs(extendedConfigName, true);
52                     replaceDuplicates = true;
53                 }
54             } else if (qName.equals("include")) {
55                 String includeFileName = attributes.getValue("file");
56                 mergeConfigs(includeFileName, false);
57             } else if (qName.equals("extendwith")) {
58                 String includeFileName = attributes.getValue("file");
59                 mergeConfigs(includeFileName, true);
60             } else if (qName.equals("component")) {
61                 String curComponent = attributes.getValue("name");
62                 String curType = attributes.getValue("type");
63                 if (rpdMap.get(curComponent) != null && !replaceDuplicates) {
64                     throw new SAXParseException("duplicate definition for " + curComponent, locator);
65                 }
66                 rpd = new RawPropertyData(curComponent, curType);
67             } else if (qName.equals("property")) {
68                 String name = attributes.getValue("name");
69                 String value = attributes.getValue("value");
70                 if (attributes.getLength() != 2 || name == null || value == null) {
71                     throw new SAXParseException("property element must only have 'name' and 'value' attributes", locator);
72                 }
73                 if (rpd == null) {
74                     // we are not in a component so add this to the global
75                     // set of symbols
76     //                    String symbolName = "${" + name + "}"; // why should we warp the global props here
77                     globalProperties.put(name, value);
78                 } else if (rpd.contains(name) && !replaceDuplicates) {
79                     throw new SAXParseException("Duplicate property: " + name, locator);
80                 } else {
81                     rpd.add(name, value);
82                 }
83             } else if (qName.equals("propertylist")) {
84                 itemListName = attributes.getValue("name");
85                 if (attributes.getLength() != 1 || itemListName == null) {
86                     throw new SAXParseException("list element must only have the 'name'  attribute", locator);
87                 }
88                 itemList = new ArrayList<String>();
89             } else if (qName.equals("item")) {
90                 if (attributes.getLength() != 0) {
91                     throw new SAXParseException("unknown 'item' attribute", locator);
92                 }
93                 curItem = new StringBuilder();
94             } else {
95                 throw new SAXParseException("Unknown element '" + qName + '\'', locator);
96             }
97         }

99         @Override
100         public void characters(char ch[], int start, int length) throws SAXParseException {
101             if (curItem != null) {
102                 curItem.append(ch, start, length);
103             }
104         }

106         @Override
107         public void endElement(String uri, String localName, String qName) throws SAXParseException {
108             if (qName.equals("component")) {
109                 rpdMap.put(rpd.getName(), rpd);
110                 rpd = null;
111             } else if (qName.equals("property")) {
112                 // nothing to do
113             } else if (qName.equals("propertylist")) {
114                 if (rpd.contains(itemListName)) {
115                     throw new SAXParseException("Duplicate property: " + itemListName, locator);
116                 } else {
117                     rpd.add(itemListName, itemList);
118                     itemList = null;
119                 }
120             } else if (qName.equals("item")) {
121                 itemList.add(curItem.toString().trim());
122                 curItem = null;
123             }
124         }

131         private void mergeConfigs(String configFileName, boolean replaceDuplicates) {
132             try {
133                 File parent = new File(baseURL.toURI().getPath()).getParentFile();
134                 URL fileURL = new File(parent.getPath() + File.separatorChar +  configFileName).toURI().toURL();
135     
136                 Logger logger = Logger.getLogger(ConfigHandler.class.getSimpleName());
137                 logger.fine((replaceDuplicates ? "extending" : "including") + " config:" + fileURL.toURI());
138     
139                 SaxLoader saxLoader = new SaxLoader(fileURL, globalProperties, rpdMap, replaceDuplicates);
140                 saxLoader.load();
141             } catch (IOException e) {
142                 throw new RuntimeException("Error while processing <include file=\"" + configFileName + "\">: " + e, e);
143             } catch (URISyntaxException e) {
144                 e.printStackTrace();
145             }
146         }
