13     package edu.cmu.sphinx.linguist.acoustic;
14     
15     import java.util.HashMap;
16     import java.util.Map;
17     
18     /** Represents  the context for a unit */
19     public class LeftRightContext extends Context {
20     
21         String stringRepresentation;
22         final Unit[] leftContext;
23         final Unit[] rightContext;
24         private static final boolean CACHING_CONTEXTS = true;
25         private static final Map<String, LeftRightContext> cache;
26     
27         static {
28             if (CACHING_CONTEXTS) {
29                 cache = new HashMap<String, LeftRightContext>();
30             }
31         }
32     
33         /**
34          * Creates a LeftRightContext
35          *
36          * @param leftContext  the left context or null if no left context
37          * @param rightContext the right context or null if no right context
38          */
39         private LeftRightContext(Unit[] leftContext, Unit[] rightContext) {
40             this.leftContext = leftContext;
41             this.rightContext = rightContext;
42         }
43     
44         /** Provides a string representation of a context */
45         @Override
46         public String toString() {
47             if (stringRepresentation == null) {
48                 stringRepresentation = getStringRepresentation(leftContext, rightContext);
49             }
50             return stringRepresentation;
51         }
52     
53         /**
54          * Factory method for creating a left/right context
55          *
56          * @param leftContext  the left context or null if no left context
57          * @param rightContext the right context or null if no right context
58          * @return a left right context
59          */
60         public static LeftRightContext get(Unit[] leftContext, Unit[] rightContext) {
61             LeftRightContext context;
62             if (CACHING_CONTEXTS) {
63                 String name = getStringRepresentation(leftContext, rightContext);
64     
65                 context = cache.get(name);
66                 if (context == null) {
67                     context = new LeftRightContext(leftContext, rightContext);
68                     cache.put(name, context);
69                 }
70             } else {
71                 context = new LeftRightContext(leftContext, rightContext);
72             }
73             return context;
74         }
75     
76         private static String getStringRepresentation(Unit[] leftContext, Unit[] rightContext) {
77             return getContextName(leftContext) + ',' + getContextName(rightContext);
78         }
79     
80         /**
81          * Retrieves the left context for this unit
82          *
83          * @return the left context
84          */
85         public Unit[] getLeftContext() {
86             return leftContext;
87         }
88     
89         /**
90          * Retrieves the right context for this unit
91          *
92          * @return the right context
93          */
94         public Unit[] getRightContext() {
95             return rightContext;
96         }
97     
98         /**
99          * Gets the context name for a particular array of units
100          *
101          * @param context the context
102          * @return the context name
103          */
104         public static String getContextName(Unit[] context) {
105             if (context == null)
106                 return "*";
107             if (context.length == 0)
108                 return "(empty)";
109             StringBuilder sb = new StringBuilder();
110             for (Unit unit : context) {
111                 sb.append(unit == null ? null : unit.getName()).append('.');
112             }
113             sb.setLength(sb.length() - 1); // remove last period
114             return sb.toString();
115         }
116     
117         /**
118          * Checks to see if there is a partial match with the given context. If both contexts are LeftRightContexts then  a
119          * left or right context that is null is considered a wild card and matches anything, othewise the contexts must
120          * match exactly. Anything matches the Context.EMPTY_CONTEXT
121          *
122          * @param context the context to check
123          * @return true if there is a partial match
124          */
125         @Override
126         public boolean isPartialMatch(Context context) {
127             if (context instanceof LeftRightContext) {
128                 LeftRightContext lrContext = (LeftRightContext)context;
129                 Unit[] lc = lrContext.getLeftContext();
130                 Unit[] rc = lrContext.getRightContext();
131     
132                 return (lc == null || leftContext == null || Unit.isContextMatch(lc, leftContext))
133                     && (rc == null || rightContext == null || Unit.isContextMatch(rc, rightContext));
134             }
135             return context == Context.EMPTY_CONTEXT && leftContext == null && rightContext == null;
136         }
