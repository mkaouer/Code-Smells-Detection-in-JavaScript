9     /**
10      * Use a largeNGramModel that also can be adjusted depending on context.
11      * A model that optimizes the search by giving a preference to 
12      * the list of keywords.
13      * <p>
14      * Example of use in calling program using config file:
15      * <br>
16      * <code>
17      *         KeywordOptimizerLargeNGramModel model = 
18      *      	(KeywordOptimizerLargeNGramModel) cm.lookup("trigramModel");
19      *         model.keywordProbs = this.hashProbs;
20      *          </code> <p>
21      * Create hashProbs by loading keywords and changes to probabilities,
22      *    string, float pairs. Keywords may be all lower case.<br>
23      *    <code>
24      *    hashProbs.put("keyword", 0.5f);  </code>
25      * @author daktari3
26      * @version 2010-12-16
27      * @see LargeNGramModel
28      * @see edu.cmu.sphinx.linguist.language.ngram.KeywordOptimizerModel
29      *
30      */
31     
32     public class KeywordOptimizerLargeNGramModel extends LargeNGramModel {
33     	
34     	   /**
35     	    * hash map of probability adjustments settable by
36     	    * user program.
37     	    * 
38     	    */
39     	   public HashMap<String, Float> keywordProbs;
40     
41         /**
42          * Gets the ngram probability of the word sequence represented by the word list
43          *
44          * @param wordSequence the word sequence
45          * @return the probability of the word sequence. Probability is in logMath log base
46          */
47         @Override
48         public float getProbability(WordSequence wordSequence) {
49         	float prob = super.getProbability(wordSequence);
50         	
51         	if (keywordProbs == null)
52         			return prob;
53         	
54         	for (Word word : wordSequence.getWords()) {
55         		String ws = word.toString();
56         		if (keywordProbs.containsKey(ws)) {
57         			prob *= keywordProbs.get(ws);
58         		}
59         	}
60         	
61         	return prob;
62         }
