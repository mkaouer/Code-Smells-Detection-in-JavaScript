195             TimerPool.getTimer(this, "Load LM").start();
196             
197             // create the log file if specified
198             if (ngramLogFile != null)
199                 logFile = new PrintWriter(new FileOutputStream(ngramLogFile));
200             
201             if (location.getProtocol() == null || location.getProtocol().equals("file")) {
202                 loader = new BinaryLoader(new File(location.getFile()), format, applyLanguageWeightAndWip,
203                                           logMath, languageWeight, wip, unigramWeight);
204             } else {
205                 loader = new BinaryStreamLoader(location, format, applyLanguageWeightAndWip,
206                         logMath, languageWeight, wip, unigramWeight);            
207             }
208                     
209             unigramIDMap = new HashMap<Word, UnigramProbability>();
210             unigrams = loader.getUnigrams();
211             loadedNGramBuffers = new Map[loader.getMaxDepth()];
212             ngramProbTable = new float[loader.getMaxDepth()][];
213             ngramBackoffTable = new float[loader.getMaxDepth()][];
214             ngramSegmentTable = new int[loader.getMaxDepth()][];
215     
216     		for (int i = 1; i <= loader.getMaxDepth(); i++) {
217     			loadedNGramBuffers[i - 1] = new HashMap<WordSequence, NGramBuffer>();
218     
219     			if (i >= 2)
220     				ngramProbTable[i - 1] = loader.getNGramProbabilities(i);
221     
222     			if (i > 2) {
223     				ngramBackoffTable[i - 1] = loader.getNGramBackoffWeights(i);
224     				ngramSegmentTable[i - 1] = loader.getNGramSegments(i);
225     			}
226     		}
227             
228             ngramDepthCache = new LRUCache<WordSequence, ProbDepth>(ngramCacheSize);
229             buildUnigramIDMap(dictionary);
230             loadedBigramBuffers = new NGramBuffer[unigrams.length];
231     
232             if (maxDepth <= 0 || maxDepth > loader.getMaxDepth())
233                 maxDepth = loader.getMaxDepth();
234     
235             for (int i = 1; i <= loader.getMaxDepth(); i++)
236             	logger.info(Integer.toString(i) + "-grams: " + loader.getNumberNGrams(i));
237     
238             if (fullSmear) {
239                 System.out.println("Full Smear");
240                 try {
241                     System.out.println("... Reading ...");
242                     readSmearInfo("smear.dat");
243                     System.out.println("... Done ");
244                 } catch (IOException e) {
245                     System.out.println("... " + e);
246                     System.out.println("... Calculating");
247                     buildSmearInfo();
248                     System.out.println("... Writing");
249                     // writeSmearInfo("smear.dat");
250                     System.out.println("... Done");
251                 }
252             }
253             
254             TimerPool.getTimer(this,"Load LM").stop();
255         }

268         /** Builds the map from unigram to unigramID. Also finds the startWordID and endWordID.
269          * @param dictionary
270          * */
271         private void buildUnigramIDMap(Dictionary dictionary) {
272             int missingWords = 0;
273             String[] words = loader.getWords();
274             for (int i = 0; i < words.length; i++) {
275                 Word word = dictionary.getWord(words[i]);
276     
277                 if (word == null) {
278                     logger.info("Missing word: " + words[i]);
279                     missingWords++;
280                 }
281                 
282                 unigramIDMap.put(word, unigrams[i]);
283                 
284                 if (logger.isLoggable(Level.FINE))
285                     logger.fine("Word: " + word);
286             }
287     
288             if (missingWords > 0)
289                 logger.warning("Dictionary is missing " + missingWords + " words that are contained in the language model.");
290         }

313         /** Clears the various N-gram caches. */
314         private void clearCache() {
315             for (int i = 0; i < loadedBigramBuffers.length; i++) {
316                 NGramBuffer buffer = loadedBigramBuffers[i];
317                 
318                 if (buffer != null) {
319                     if (!buffer.getUsed())
320                         loadedBigramBuffers[i] = null; // free the BigramBuffer
321                     else
322                         buffer.setUsed(false);
323                 }
324             }
325     
326             
327             loadedBigramBuffers = new NGramBuffer[unigrams.length];        
328             for (int i = 2; i <= loader.getMaxDepth(); i++) {
329                 loadedNGramBuffers[i - 1] = new HashMap<WordSequence, NGramBuffer>();
330             }
331             logger.info("LM Cache Size: " + ngramDepthCache.size() + " Hits: " + ngramHits + " Misses: " + ngramMisses);
332         	if (clearCacheAfterUtterance) {
333         		ngramDepthCache = new LRUCache<WordSequence, ProbDepth>(ngramCacheSize);
334         	}
335         }

337         /**
338          * Returns predicted probability and depth. Uses caching for 
339          * high order ngrams.
340          * 
341          * @param wordSequence sequence to get the probability
342          */
343         public ProbDepth getProbDepth(WordSequence wordSequence) {
344             int numberWords = wordSequence.size();
345             ProbDepth probDepth = null;
346     
347             if (numberWords > maxDepth) {
348                 throw new Error("Unsupported NGram: " + wordSequence.size());
349             }
350     
351             if (numberWords == maxDepth) {
352                 probDepth = ngramDepthCache.get(wordSequence);
353     
354                 if (probDepth != null) {
355                     ngramHits++;
356                     return probDepth;
357                 }
358                 ngramMisses++;
359             }
360     
361             probDepth = getNGramProbDepth(wordSequence);
362     
363             if (numberWords == maxDepth)
364                 ngramDepthCache.put(wordSequence, probDepth);
365     
366             if (logFile != null && probDepth != null)
367                 logFile.println(wordSequence.toString().replace("][", " ") + " : " + Float.toString(probDepth.probability) + " : "
368                         + probDepth.depth);
369     
370             return probDepth;
371         }

374         private ProbDepth getNGramProbDepth(WordSequence wordSequence) {
375             int numberWords = wordSequence.size();
376             Word firstWord = wordSequence.getWord(0);
377     
378             if (loader.getNumberNGrams(numberWords) == 0 || !hasUnigram(firstWord))
379                 return getNGramProbDepth(wordSequence.getNewest());
380     
381             if (numberWords < 2) {
382                 return getUnigramProbDepth(wordSequence);
383             }
384     
385             NGramProbability nGProbability = findNGram(wordSequence);
386     
387             if (nGProbability != null) {
388                 float probability = ngramProbTable[numberWords - 1][nGProbability.getProbabilityID()];
389                 return new ProbDepth(probability, numberWords);
390             }
391     
392             if (numberWords == 2) {
393                 UnigramProbability unigramProb = getUnigram(firstWord);
394                 UnigramProbability unigramProb1 = getUnigram(wordSequence.getWord(1));
395                 float probability = unigramProb.getLogBackoff() + unigramProb1.getLogProbability();
396                 return new ProbDepth(probability, 1);
397             }
398     
399             NGramProbability nMinus1Gram = findNGram(wordSequence.getOldest());
400     
401             if (nMinus1Gram != null) {
402                 ProbDepth result1 = getProbDepth(wordSequence.getNewest());
403                 float probability = ngramBackoffTable[numberWords - 1][nMinus1Gram.getBackoffID()] + result1.probability;
404                 return new ProbDepth(probability, result1.depth);
405             }
406     
407             return getProbDepth(wordSequence.getNewest());
408         }

426         /**
427          * Finds or loads the NGram probability of the given NGram.
428          *
429          * @param wordSequence the NGram to load
430          * @return a NGramProbability of the given NGram
431          */
432         private NGramProbability findNGram(WordSequence wordSequence) {
433         	int numberWords = wordSequence.size();
434             NGramProbability nGram = null; 
435             
436             WordSequence oldest = wordSequence.getOldest();
437             NGramBuffer nGramBuffer = loadedNGramBuffers[numberWords - 1].get(oldest);
438             if (nGramBuffer == null) {
439             	nGramBuffer = getNGramBuffer(oldest);
440                 if (nGramBuffer != null)
441                     loadedNGramBuffers[numberWords - 1].put(oldest, nGramBuffer);
442             }
443     
444             if (nGramBuffer != null) {
445                 int nthWordID = getWordID(wordSequence.getWord(numberWords - 1));
446                 nGram = nGramBuffer.findNGram(nthWordID);
447             }
448     
449             return nGram;
450         }

463         /**
464          * Loads into a buffer all the NGram followers of the given N-1Gram.
465          * 
466          * @param ws the N-1Gram to find followers
467          * 
468          * @return a NGramBuffer of all the NGram followers of the given sequence
469          */
470         private NGramBuffer loadNGramBuffer(WordSequence ws) { 
471         	int firstWordID = getWordID(ws.getWord(0));
472         	int firstCurrentNGramEntry = 0;
473         	int numberNGrams = 0;
474         	int size = 0;
475     		long position = 0;
476     		int orderBuffer = ws.size() + 1;
477     		NGramBuffer currentBuffer = null;
478     		NGramBuffer nMinus1Buffer = null;
479     
480     		firstCurrentNGramEntry = unigrams[firstWordID].getFirstBigramEntry(); 
481     		numberNGrams = getNumberBigramFollowers(firstWordID) + 1;
482     		
483     		if (numberNGrams == 1) // 1 means that there is no bigram starting with firstWordID
484     			return null;
485             
486     		if (orderBuffer == 2)  {
487     			size = numberNGrams * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField();
488     			position = (long) (loader.getNGramOffset(orderBuffer) + (firstCurrentNGramEntry * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField()));
489     		} else { // only for ws.size() >= 2
490     			int lastWordId = getWordID(ws.getWord(ws.size() - 1));
491     			nMinus1Buffer = getNGramBuffer(ws.getOldest());
492     			int index = nMinus1Buffer.findNGramIndex(lastWordId);
493     
494     			if (index == -1)
495     				return null;
496     
497     			int firstNMinus1GramEntry = nMinus1Buffer.getFirstNGramEntry();
498     			firstCurrentNGramEntry = getFirstNGramEntry(nMinus1Buffer.getNGramProbability(index), firstNMinus1GramEntry, orderBuffer);
499     			int firstNextNGramEntry = getFirstNGramEntry(nMinus1Buffer.getNGramProbability(index + 1), firstNMinus1GramEntry, orderBuffer);
500     			numberNGrams = firstNextNGramEntry - firstCurrentNGramEntry;
501     			
502     			if (numberNGrams == 0)
503     				return null;
504     			
505     			if (loader.getMaxDepth() != orderBuffer) 
506     				numberNGrams++;
507     
508     			size = numberNGrams * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField();
509     			position = (long) loader.getNGramOffset(orderBuffer) + (long) firstCurrentNGramEntry * (long) ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * (long) loader.getBytesPerField();
510     		}
511     
512             try {
513                 byte[] buffer = loader.loadBuffer(position, size);
514     
515                 if (loader.getMaxDepth() == orderBuffer) {
516                     currentBuffer = new NMaxGramBuffer(buffer, numberNGrams, loader.getBigEndian(), is32bits(), orderBuffer,
517                             firstCurrentNGramEntry);
518                 } else {
519                     currentBuffer = new NGramBuffer(buffer, numberNGrams, loader.getBigEndian(), is32bits(), orderBuffer,
520                             firstCurrentNGramEntry);
521                 }
522             }
523     		catch (IOException ioe) {
524     			ioe.printStackTrace();
525     			throw new Error("Error loading " + orderBuffer + "-Grams.");
526     		}
527     
528     		return currentBuffer;
529         }

532         /**
533          * Returns the NGrams of the given word sequence
534          *
535          * @param wordSequence the word sequence 
536          * 					   from which to get the buffer
537          * @return the NGramBuffer of the word sequence
538          */
539         private NGramBuffer getNGramBuffer(WordSequence wordSequence) {
540         	NGramBuffer nGramBuffer = null;
541         	int order = wordSequence.size();
542         	
543         	if (order > 1)
544         		nGramBuffer = loadedNGramBuffers[order - 1].get(wordSequence); // better when using containsKey
545     
546         	if (nGramBuffer == null) {
547         		nGramBuffer = loadNGramBuffer(wordSequence);
548     
549         		if (nGramBuffer != null) 
550         			loadedNGramBuffers[order - 1].put(wordSequence, nGramBuffer); // optimizable by adding an 'empty' nGramBuffer
551         	}
552     
553         	return nGramBuffer;
554         }

557         /**
558          * Returns the index of the first NGram entry of the given N-1Gram
559          *
560          * @param nMinus1Gram            the N-1Gram which first NGram entry we're looking for
561          * @param firstNMinus1GramEntry  the index of the first N-1Gram entry of the N-1Gram in question
562          * @param n						 the order of the NGram
563          * @return the index of the first NGram entry of the given N-1Gram
564          */
565         private int getFirstNGramEntry(NGramProbability nMinus1Gram, int firstNMinus1GramEntry, int n) {
566             int firstNGramEntry = ngramSegmentTable[n - 1][(firstNMinus1GramEntry + nMinus1Gram.getWhichFollower()) 
567                                                       >> loader.getLogNGramSegmentSize()] 
568                                                       + nMinus1Gram.getFirstNPlus1GramEntry();
569             
570             return firstNGramEntry;
571         }

639                 int length = wordSequence.size();
640                 
641                 if (length > 0) {
642                     int wordID = getWordID(wordSequence.getWord(length - 1));
643                     smearTerm = unigramSmearTerm[wordID];
644                 }
645             }
646             
647             if (fullSmear && logger.isLoggable(Level.FINE))
648                 logger.fine("SmearTerm: " + smearTerm);
649     
650             return smearTerm;
651         }

656         @Override
657         public float getSmear(WordSequence wordSequence) {
658             float smearTerm = 0.0f;
659             
660             if (fullSmear) {
661                 smearCount++;
662                 int length = wordSequence.size();
663                 
664                 if (length == 1) {
665                     int wordID = getWordID(wordSequence.getWord(0));
666                     smearTerm = unigramSmearTerm[wordID];
667                 } else if (length >= 2) {
668                     int size = wordSequence.size();
669                     int wordID1 = getWordID(wordSequence.getWord(size - 2));
670                     int wordID2 = getWordID(wordSequence.getWord(size - 1));
671                     Float st = getSmearTerm(wordID1, wordID2);
672                     
673                     if (st == null)
674                         smearTerm = unigramSmearTerm[wordID2];
675                     else {
676                         smearTerm = st;
677                         smearBigramHit++;
678                     }
679                 }
680     
681                 if (smearCount % 100000 == 0)
682                     System.out.println("Smear hit: " + smearBigramHit + " tot: " + smearCount);
683             }
684             
685             if (fullSmear && logger.isLoggable(Level.FINE))
686                 logger.fine("SmearTerm: " + smearTerm);
687     
688             return smearTerm;
689         }

788         private void buildSmearInfo() throws IOException {
789             double S0 = 0;
790             double R0 = 0;
791     
792             bigramSmearMap = new HashMap<Long, Float>();
793     
794             double[] ugNumerator = new double[unigrams.length];
795             double[] ugDenominator = new double[unigrams.length];
796             double[] ugAvgLogProb = new double[unigrams.length];
797     
798             unigramSmearTerm = new float[unigrams.length];
799     
800             for (UnigramProbability unigram : unigrams) {
801                 float logp = unigram.getLogProbability();
802                 double p = logMath.logToLinear(logp);
803                 S0 += p * logp;
804                 R0 += p * logp * logp;
805             }
806     
807             System.out.println("R0 S0 " + R0 + ' ' + S0);
808     
809             for (int i = 0; i < loadedBigramBuffers.length; i++) {
810                 NGramBuffer bigram = getBigramBuffer(i);
811                 
812                 if (bigram == null) {
813                     unigramSmearTerm[i] = LogMath.getLogOne();
814                     continue;
815                 }
816     
817                 ugNumerator[i] = 0.0;
818                 ugDenominator[i] = 0.0;
819                 ugAvgLogProb[i] = 0.0;
820     
821                 float logugbackoff = unigrams[i].getLogBackoff();
822                 double ugbackoff = logMath.logToLinear(logugbackoff);
823     
824                 for (int j = 0; j < bigram.getNumberNGrams(); j++) {
825                     int wordID = bigram.getWordID(j);
826                     NGramProbability bgProb = bigram.getNGramProbability(j);
827     
828                     float logugprob = unigrams[wordID].getLogProbability();
829                     float logbgprob = ngramProbTable[1][bgProb.getProbabilityID()];
830     
831                     double ugprob = logMath.logToLinear(logugprob);
832                     double bgprob = logMath.logToLinear(logbgprob);
833     
834                     double backoffbgprob = ugbackoff * ugprob;
835                     double logbackoffbgprob = logMath.linearToLog(backoffbgprob);
836     
837                     ugNumerator[i] += (bgprob * logbgprob
838                             - backoffbgprob * logbackoffbgprob) * logugprob;
839     
840                     ugDenominator[i] += (bgprob - backoffbgprob) * logugprob;
841                     // dumpProbs(ugNumerator, ugDenominator, i, j, logugprob,
842     				//		logbgprob, ugprob, bgprob, backoffbgprob,
843     				//		logbackoffbgprob);
844                 }
845                 
846                 ugNumerator[i] += ugbackoff * (logugbackoff * S0 + R0);
847                 ugAvgLogProb[i] = ugDenominator[i] + ugbackoff * S0;
848                 ugDenominator[i] += ugbackoff * R0;
849     
850                 // System.out.println("n/d " + ugNumerator[i] + " " +
851                 //                     ugDenominator[i]);
852     
853                 unigramSmearTerm[i] = (float) (ugNumerator[i] / ugDenominator[i]);
854                 /// unigramSmearTerm[i] = 
855                 //   logMath.linearToLog(ugNumerator[i] / ugDenominator[i]);
856                 //  System.out.println("ugs " + unigramSmearTerm[i]);
857             }
858     
859             for (int i = 0; i < loadedBigramBuffers.length; i++) {
860                 System.out.println("Processed " + i
861                         + " of " + loadedBigramBuffers.length);
862                 NGramBuffer bigram = getBigramBuffer(i);
863                 
864                 if (bigram == null)
865                     continue;
866     
867                 for (int j = 0; j < bigram.getNumberNGrams(); j++) {
868                     float smearTerm;
869                     NGramProbability bgProb = bigram.getNGramProbability(j);
870                     float logbgbackoff = ngramBackoffTable[2][bgProb.getBackoffID()];
871                     double bgbackoff = logMath.logToLinear(logbgbackoff);
872                     int k = bigram.getWordID(j);
873                     NGramBuffer trigram = loadTrigramBuffer(i, k);
874     
875                     if (trigram == null)
876                         smearTerm = unigramSmearTerm[k];
877                     else {
878                         double bg_numerator = 0;
879                         double bg_denominator = 0;
880                         for (int l = 0; l < trigram.getNumberNGrams(); l++) {
881                             int m = trigram.getWordID(l);
882                             float logtgprob
883                                     = ngramProbTable[2][trigram.getProbabilityID(l)];
884                             double tgprob = logMath.logToLinear(logtgprob);
885                             float logbgprob = getBigramProb(k, m);
886                             double bgprob = logMath.logToLinear(logbgprob);
887                             float logugprob = unigrams[m].getLogProbability();
888                             double backofftgprob = bgbackoff * bgprob;
889                             double logbackofftgprob
890                                     = logMath.linearToLog(backofftgprob);
891     
892                             bg_numerator += (tgprob * logtgprob
893                                     - backofftgprob * logbackofftgprob) * logugprob;
894     
895                             bg_denominator += (tgprob - backofftgprob)
896                                     * logugprob * logugprob;
897                         }
898                         
899                         bg_numerator += bgbackoff * (logbgbackoff * ugAvgLogProb[k] - ugNumerator[k]);
900                         bg_denominator += bgbackoff * ugDenominator[k];
901                         // bigram.ugsmear = bg_numerator / bg_denominator;
902                         smearTerm = (float) (bg_numerator / bg_denominator);
903                         smearTermCount++;
904                     }
905                     
906                     putSmearTerm(i, k, smearTerm);
907                 }
908             }
909             
910             System.out.println("Smear count is " + smearTermCount);
911         }

975          * Reads the smear info from the given file
976          *
977          * @param filename where to read the smear info from
978          * @throws IOException if an inconsistent file is found or on any general I/O error
979          */

981             DataInputStream in = new DataInputStream(new FileInputStream(filename));
982     
983     
984             if (in.readInt() != SMEAR_MAGIC)
985                 throw new IOException("Bad smear format for " + filename);
986     
987             if (in.readInt() != unigrams.length)
988                 throw new IOException("Bad unigram length in " + filename);
989     
990             bigramSmearMap = new HashMap<Long, Float>();
991             unigramSmearTerm = new float[unigrams.length];
992             System.out.println("Reading " + unigrams.length);
993             
994             for (int i = 0; i < unigrams.length; i++)
995                 unigramSmearTerm[i] = in.readFloat();
996     
997             for (int i = 0; i < unigrams.length; i++) {
998                 System.out.println("Processed " + i
999                         + " of " + loadedBigramBuffers.length);
1000                 int numBigrams = in.readInt();
1001                 NGramBuffer bigram = getBigramBuffer(i);
1002                 
1003                 if (bigram.getNumberNGrams() != numBigrams)
1004                     throw new IOException("Bad ngrams for unigram " + i + " Found " + numBigrams + " expected " + bigram.getNumberNGrams());
1005     
1006                 for (int j = 0; j < numBigrams; j++) {
1007                     int k = bigram.getWordID(j);
1008                     putSmearTerm(i, k, in.readFloat());
1009                 }
1010             }
1011             
1012             in.close();
1013         }
