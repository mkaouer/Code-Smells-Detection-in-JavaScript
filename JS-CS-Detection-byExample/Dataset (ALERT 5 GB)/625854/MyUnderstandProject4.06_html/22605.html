13     package edu.cmu.sphinx.result;
14     
15     import edu.cmu.sphinx.decoder.search.AlternateHypothesisManager;
16     import edu.cmu.sphinx.decoder.search.Token;
17     import edu.cmu.sphinx.linguist.HMMSearchState;
18     import edu.cmu.sphinx.linguist.SearchState;
19     import edu.cmu.sphinx.linguist.UnitSearchState;
20     import edu.cmu.sphinx.linguist.WordSearchState;
21     
22     import java.io.FileWriter;
23     import java.io.IOException;
24     import java.util.*;
25     
26     /**
27      * Dumps out the GDL graph of all the result token chains in a Result, as well as all the alternate hypotheses along
28      * those chains.
29      */
30     public class TokenGraphDumper {
31     
32         private final AlternateHypothesisManager loserManager;
33         private final Result result;
34         private final Map<Token, Integer> tokenIDMap;
35         private final Set<Token> dumpedTokens;
36         private int ID;
37     
38     
39         /**
40          * Constructs a TokenGraphDumper from the given result.
41          *
42          * @param result The result which search space we want to dump.
43          */
44         public TokenGraphDumper(Result result) {
45             this.result = result;
46             this.loserManager = result.getAlternateHypothesisManager();
47             tokenIDMap = new HashMap<Token, Integer>();
48             dumpedTokens = new HashSet<Token>();
49         }
50     
51     
52         /**
53          * Dumps the GDL output of the search space to the given file.
54          *
55          * @param title    the title of the GDL graph
56          * @param fileName
57          */
58         public void dumpGDL(String title, String fileName) {
59             try {
60                 System.err.println("Dumping " + title + " to " + fileName);
61                 FileWriter f = new FileWriter(fileName);
62                 f.write(dumpGDL(title));
63                 f.close();
64             } catch (IOException ioe) {
65                 ioe.printStackTrace();
66             }
67         }
68     
69     
70         /**
71          * Dumps the GDL output.
72          *
73          * @param title the title of the GDL graph
74          * @return the GDL output string
75          */
76         public String dumpGDL(String title) {
77             StringBuilder gdl = new StringBuilder("graph: {\n");
78             gdl.append("title: \"").append(title).append("\"\n");
79             gdl.append("display_edge_labels: yes\n");
80     
81             for (Token token : result.getResultTokens()) {
82                 gdl.append(dumpTokenGDL(token));
83             }
84     
85             gdl.append("}\n");
86             return gdl.toString();
87         }
88     
89     
90         /**
91          * Dumps the GDL output for a token, and any of its predecessors or alternate hypotheses.
92          *
93          * @param token the token to dump
94          * @return the GDL output string
95          */
96         private String dumpTokenGDL(Token token) {
97     
98             if (dumpedTokens.contains(token)) {
99                 return "";
100             } else {
101                 String label = ("[" + token.getAcousticScore() + token.getInsertionScore() + ',' +
102                         token.getLanguageScore() + ']');
103                 if (token.isWord()) {
104                     label = token.getWord().getSpelling() + label;
105                 }
106     
107                 String color = null;
108     
109                 if (token.getSearchState() != null) {
110                     color = getColor(token.getSearchState());
111                 }
112     
113                 StringBuilder gdl = new StringBuilder().append("node: { title: \"").append(getTokenID(token))
114                     .append("\" label: \"").append(label).append("\" color: ");
115                 if (color != null) {
116                     gdl.append(color).append(" }");
117                 } else {
118                     gdl.append(" }");
119                 }
120                 gdl.append('\n');
121     
122                 dumpedTokens.add(token);
123     
124                 if (token.getPredecessor() != null) {
125                     gdl.append("edge: { sourcename: \"").append(getTokenID(token))
126                         .append("\" targetname: \"").append(getTokenID(token.getPredecessor()))
127                         .append("\" }").append('\n').append(dumpTokenGDL(token.getPredecessor()));
128                 }
129     
130                 if (loserManager != null) {
131                     List<Token> list = loserManager.getAlternatePredecessors(token);
132                     if (list != null) {
133                         for (Token loser : list) {
134                             gdl.append("edge: { sourcename: \"").append(getTokenID(token))
135                                 .append("\" targetname: \"").append(getTokenID(loser))
136                                 .append("\" }").append('\n').append(dumpTokenGDL(loser));
137                         }
138                     }
139                 }
140                 return gdl.toString();
141             }
142         }
143     
144     
145         /**
146          * Gets the color for a particular state
147          *
148          * @param state the state
149          * @return its color
150          */
151         private String getColor(SearchState state) {
152             String color = "lightred";
153             if (state.isFinal()) {
154                 color = "magenta";
155             } else if (state instanceof UnitSearchState) {
156                 color = "green";
157             } else if (state instanceof WordSearchState) {
158                 color = "lightblue";
159             } else if (state instanceof HMMSearchState) {
160                 color = "orange";
161             }
162             return color;
163         }
164     
165     
166         /**
167          * Returns the next available token ID.
168          *
169          * @param token the token for which we want an ID
170          * @return the next available token ID
171          */
172         private Integer getTokenID(Token token) {
173             Integer id = tokenIDMap.get(token);
174             if (id == null) {
175                 id = ID++;
176                 tokenIDMap.put(token, id);
177             }
178             return id;
179         }
180     }
