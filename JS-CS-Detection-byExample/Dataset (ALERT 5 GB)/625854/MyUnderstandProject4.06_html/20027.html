32         /** Copy a grammar from gramFrom to gramTo. */
33         public static void copyGrammar(RuleGrammar gramFrom, RuleGrammar gramTo)
34                 throws GrammarException {
35             
36         	// Copy imports
37             for (RuleName ruleName : gramFrom.listImports()) {
38                 gramTo.addImport(ruleName);
39             }
40             // Copy each rule
41             for (String ruleName : gramFrom.listRuleNames()) {
42                 boolean isPublic = false;
43                 try {
44                     isPublic = gramFrom.isRulePublic(ruleName);
45                 } catch (IllegalArgumentException nse) {
46                     throw new GrammarException(nse.toString(), null);
47                 }
48     
49                 Rule rule = gramFrom.getRule(ruleName);
50                 try {
51                     gramTo.setRule(ruleName, rule, isPublic);
52                     if (gramFrom.isEnabled(ruleName)) {
53                         gramTo.setEnabled(ruleName, true);
54                     }
55                 } catch (IllegalArgumentException E) {
56                     throw new GrammarException(E.toString(), null);
57                 }
58             }
59         }

62         /** Copy all RuleGrammars from recFrom to recTo. */
63         public static void copyGrammars(Recognizer recFrom, Recognizer recTo)
64                 throws GrammarException {
65             RuleGrammar gramFrom[] = recFrom.listRuleGrammars();
66     
67             for (RuleGrammar grammar : gramFrom) {
68                 String name = grammar.getName();
69                 RuleGrammar gramTo = recTo.getRuleGrammar(name);
70     
71                 // Create a new grammar - if necessary
72                 if (gramTo == null) {
73                     try {
74                         gramTo = recTo.newRuleGrammar(name);
75                     } catch (IllegalArgumentException gse) {
76                         throw new GrammarException("copyGrammars: " + gse, null);
77                     }
78                 }
79                 copyGrammar(grammar, gramTo);
80             }
81         }

83         /*
84           * transform grammar to eliminate <NULL> and <VOID>
85           *
86           * NULL and VOID can be removed by transforming the grammar
87           * as follows:
88           *
89           * RuleSequence:
90           *   (x  y  <NULL>) --> (x y)
91           *   (x  y  <VOID>) --> <VOID>
92           *   (<NULL>) --> <NULL>
93           *   (<VOID>) --> <VOID>
94           *
95           * RuleAlternatives:
96           *   (x | y | <NULL>) --> [(x | y)]
97           *   (x | y | <VOID>) --> (x | y)
98           *   (<NULL>) --> <NULL>
99           *   (<VOID>) --> <VOID>
100           *   (<VOID> | <NULL>) --> <NULL>
101         *
102         * RuleCount:
103         *   [<VOID>]  --> <NULL>
104         *    <VOID>*  --> <NULL>
105         *    <VOID>+  --> <VOID>
106         *   [<NULL>]  --> <NULL>
107         *    <NULL>*  --> <NULL>
108         *    <NULL>+  --> <NULL>
109         *
110         * RuleTag:
111         *   <NULL>{tag} --> <NULL>
112         *   <VOID>{tag} --> <VOID>
113         *
114         * RuleName:
115         *    If a rule has been marked as being void or null then apply the
116         *    following:
117         *    <RuleName> --> <VOID>
118         *
119         * The steps of transformation are:
120         *
121         *     just return the original grammar
122         *
123         *  2. Make a copy of the grammar and Mark all rules as non-void & non-null
124         *
125         *  3. Go through each rule in series and apply the transformations.
126         *
127         *  4. Repeat step 3 until one pass has been made through all the
128         *     rules in which no transformations were performed
129         *
130         *  5. Remove NULL and VOID rules from copies grammars
131         *
132         *  6. Remove any empty grammars
133         *
134         *  7. Return the transformed array of grammars
135         *
136         * Because all transforms are some form of a reduction, the process is
137         * guaranteed to complete.
138         *
139         */

232         static Rule transform(Rule r, RuleGrammar RGL[], boolean en, boolean ev) {
233             //System.out.println("xf " + r.getClass() + r);
234     
235             /*
236             * RULESEQUENCE
237             */
238             if (r instanceof RuleSequence) {
239                 RuleSequence rs = (RuleSequence) r;
240                 Rule rarry[] = rs.getRules();
241                 if ((rarry == null) || (rarry.length == 0)) {
242                     return null;
243                 }
244                 int cnull = 0;
245                 for (int i = 0; i < rarry.length; i++) {
246                     Rule r1 = transform(rarry[i], RGL, en, ev);
247                     RuleState s1 = new RuleState(r1, RGL);
248                     if (ev && s1.isVoid) return RuleName.VOID;
249                     if ((en && s1.isNull) || (r1 != rarry[i])) {
250                         if (rs == r) {
251                             rs = (RuleSequence) rs.copy();
252                             rarry = rs.getRules();
253                         }
254                         if (en && s1.isNull) {
255                             cnull++;
256                             r1 = null;
257                         }
258                         rarry[i] = r1;
259                         rs.setRules(rarry);
260                     }
261                 }
262                 if (cnull == 0) return rs;
263                 if (cnull == rarry.length) return RuleName.NULL;
264                 Rule barry[] = new Rule[rarry.length - cnull];
265                 int j = 0;
266                 for (Rule rule : rarry) {
267                     if (rule != null) {
268                         barry[j++] = rule;
269                     }
270                 }
271                 rs.setRules(barry);
272                 return rs;
273             }
274     
275             /*
276             * RULEALTERNATIVES
277             */
278             if (r instanceof RuleAlternatives) {
279                 RuleAlternatives ra = (RuleAlternatives) r;
280                 Rule rarry[] = ra.getRules();
281                 if ((rarry == null) || (rarry.length == 0)) {
282                     return null;
283                 }
284                 int cnull = 0;
285                 int cvoid = 0;
286                 for (int i = 0; i < rarry.length; i++) {
287                     Rule r1 = transform(rarry[i], RGL, en, ev);
288                     //System.out.println("rs: " + r1);
289                     RuleState s1 = new RuleState(r1, RGL);
290                     if ((en && s1.isNull) || (ev && s1.isVoid) || (r1 != rarry[i])) {
291                         if (ra == r) {
292                             ra = (RuleAlternatives) ra.copy();
293                             rarry = ra.getRules();
294                         }
295                         if (en && s1.isNull) {
296                             cnull++;
297                             r1 = null;
298                         }
299                         if (ev && s1.isVoid) {
300                             cvoid++;
301                             r1 = null;
302                         }
303                         rarry[i] = r1;
304                         ra.setRules(rarry);
305                     }
306                 }
307                 if ((cnull == 0) && (cvoid == 0)) return ra;
308                 if (cvoid == rarry.length) return RuleName.VOID;
309                 if ((cvoid + cnull) == rarry.length) return RuleName.NULL;
310                 Rule barry[] = new Rule[rarry.length - (cnull + cvoid)];
311                 int j = 0;
312                 for (Rule rule : rarry) {
313                     if (rule != null) {
314                         barry[j++] = rule;
315                     }
316                 }
317                 ra.setRules(barry);
318                 if (cnull > 0) return new RuleCount(ra, RuleCount.OPTIONAL);
319                 else return ra;
320             }
321     
322             /*
323             * RULECOUNT (e.g. [], *, or + )
324             */
325             if (r instanceof RuleCount) {
326                 RuleCount rc = (RuleCount) r;
327                 int rcount = rc.getCount();
328                 Rule r1 = rc.getRule();
329                 Rule r2 = transform(r1, RGL, en, ev);
330                 RuleState s1 = new RuleState(r2, RGL);
331                 if (ev && s1.isVoid) {
332                     if (rcount == RuleCount.ONCE_OR_MORE) return RuleName.VOID;
333                     return RuleName.NULL;
334                 }
335                 if (en && s1.isNull) return RuleName.NULL;
336                 if (r1 == r2) return rc;
337                 return new RuleCount(r2, rcount);
338             }
339     
340             /*
341             * TAGS
342             */
343             if (r instanceof RuleTag) {
344                 RuleTag rtag = (RuleTag) r;
345                 Rule rtr = rtag.getRule();
346                 Rule r1 = transform(rtr, RGL, en, ev);
347                 RuleState s1 = new RuleState(r1, RGL);
348                 if (en && s1.isNull) return RuleName.NULL;
349                 if (ev && s1.isVoid) return RuleName.VOID;
350                 if (r1 == rtr) return rtag;
351                 return new RuleTag(r1, rtag.getTag());
352             }
353     
354             return r;
355         }

358         static RuleGrammar[] transform(RuleGrammar RG[], boolean eliminateNULL, boolean eliminateVOID)
359                 throws GrammarException {
360             //if (!eliminateNULL && !eliminateVOID) return RG;
361             RuleGrammar NRG[] = RG.clone();
362             boolean somethingchanged = true;
363             boolean nochange = true;
364             while (somethingchanged) {
365                 somethingchanged = false;
366                 for (int i = 0; i < NRG.length; i++) {
367                     for (String rname : NRG[i].listRuleNames()) {
368                         Rule r = NRG[i].getRuleInternal(rname);
369                         Rule nr = transform(r, NRG, eliminateNULL, eliminateVOID);
370                         if (nr != r) {
371                             nochange = false;
372                             somethingchanged = true;
373                             if (NRG[i] == RG[i]) {
374                                 NRG[i] = new BaseRuleGrammar(null, RG[i].getName());
375                                 RecognizerUtilities.copyGrammar(RG[i], NRG[i]);
376                             }
377                             boolean isPublic;
378                             try {
379                                 isPublic = NRG[i].isRulePublic(rname);
380                             } catch (IllegalArgumentException nse) {
381                                 throw new GrammarException(nse.toString(), null);
382                             }
383                             NRG[i].setRule(rname, nr, isPublic);
384                         }
385                     }
386                 }
387             }
388     
389             if (nochange) return NRG;
390     
391             /*
392             * Now remove unused rules
393             */
394             List<RuleGrammar> nr = new ArrayList<RuleGrammar>();
395             for (RuleGrammar grammar : NRG) {
396                 int k = 0;
397                 for (String rname : grammar.listRuleNames()) {
398                     Rule r1 = grammar.getRuleInternal(rname);
399                     RuleState s = new RuleState(r1, NRG);
400                     //System.out.println(rname + " " + r1.getClass() + " " + s.isvoid + " " + s.isnull);
401                     if ((eliminateVOID && s.isVoid) || (eliminateNULL && s.isNull)) {
402                         grammar.deleteRule(rname);
403                     } else k++;
404                 }
405                 if (k > 0) nr.add(grammar);
406             }
407     
408             /*
409             * Now return all non-null grammars
410             */
411             return nr.toArray(new RuleGrammar[nr.size()]);
412         }

415     class RuleState {
416     
417         boolean isNull;
418     
419         boolean isVoid;
420     
421         RuleState(Rule r, RuleGrammar RGL[]) {
422             if (r instanceof RuleName) {
423                 RuleName rn = (RuleName) r;
424     
425                 /* simple check for void/null */
426                 String simpleName = rn.getSimpleRuleName();
427                 if (simpleName.equals("VOID")) {
428                     isVoid = true;
429                     return;
430                 }
431                 if (simpleName.equals("NULL")) {
432                     isNull = true;
433                     return;
434                 }
435     
436                 /* now lookup the rule and see if its RHS is null/void */
437                 Rule r1 = getRule(rn, RGL);
438                 if (r1 == null) System.out.println("ERROR: Could not find rule " + rn);
439                 if (r1 instanceof RuleName) {
440                     RuleName rn1 = (RuleName) r1;
441     
442                     /* simple check for void/null */
443                     String simpleName1 = rn1.getSimpleRuleName();
444                     if (simpleName1.equals("VOID")) {
445                         isVoid = true;
446                         return;
447                     }
448                     if (simpleName1.equals("NULL")) {
449                         isNull = true;
450                         return;
451                     }
452                 }
453                 return;
454             }
455         }
456         Rule getRule(RuleName rn, RuleGrammar RGL[]) {
457             String gname = rn.getSimpleGrammarName();
458             for (RuleGrammar grammar : RGL) {
459                 if (grammar.getName().equals(gname)) {
460                     return grammar.getRuleInternal(rn.getSimpleRuleName());
461                 }
462             }
463             return null;
464         }
