8     package com.sun.speech.engine;

10     import javax.speech.EngineProperties;
11     import javax.speech.SpeechError;
12     import javax.speech.SpeechEvent;
13     import java.beans.PropertyChangeEvent;
14     import java.beans.PropertyChangeListener;
15     import java.util.*;
16     
17     /** Supports the JSAPI 1.0 <code>EngineProperties</code> interface. */
18     public abstract class BaseEngineProperties
19             implements EngineProperties, SpeechEventDispatcher {
20     
21         /** List of <code>PropertyChangeListeners</code> registered for <code>PropertyChangeEvents</code> on this object. */
22         protected final List<PropertyChangeListener> propertyChangeListeners = new ArrayList<PropertyChangeListener>();
23     
24         /**
25          * Obtains the AWT <code>Component</code> that provides the default user interface for setting the properties of the
26          * <code>Engine</code> associated with this object.
27          *
28          * @return an AWT <code>Component</code> to manipulate this object
29          */
30         public java.awt.Component getControlComponent() {
31             return null;
32         }
33     
34     
35         /**
36          * Returns all properties to reasonable defaults for the <code>Engine</code>.  A <code>PropertyChangeEvent</code> is
37          * issued for each property that changes as the reset takes effect.
38          */
39         public abstract void reset();
40     
41     
42         /**
43          * Adds a <code>PropertyChangeListener</code> to the listener list.
44          *
45          * @param listener the <code>PropertyChangeListener</code> to add
46          */
47         public void addPropertyChangeListener(PropertyChangeListener listener) {
48             if (!propertyChangeListeners.contains(listener)) {
49                 propertyChangeListeners.add(listener);
50             }
51         }
52     
53     
54         /**
55          * Removes a <code>PropertyChangeListener</code> from the listener list.
56          *
57          * @param listener the <code>PropertyChangeListener</code> to remove
58          */
59         public void removePropertyChangeListener(PropertyChangeListener listener) {
60             propertyChangeListeners.remove(listener);
61         }
62     
63     
64         /**
65          * Generates a <code>PropertyChangeEvent</code> for an <code>Object</code> value and posts it to the event queue.
66          * Eventually <code>firePropertyChangeEvent</code> will be called by <code>dispatchSpeechEvent</code> as a result of
67          * this action.
68          *
69          * @param propName the name of the property
70          * @param oldValue the old value
71          * @param newValue the new value
72          * @see #firePropertyChangeEvent
73          * @see #dispatchSpeechEvent
74          */
75         protected void postPropertyChangeEvent(String propName,
76                                                Object oldValue,
77                                                Object newValue) {
78             EventObject e = new PropertyChangeEvent(this,
79                     propName,
80                     oldValue,
81                     newValue);
82             SpeechEvent se = new SpeechEventWrapper(e);
83             SpeechEventUtilities.postSpeechEvent(this, se);
84         }
85     
86     
87         /**
88          * Generates a <code>PropertyChangeEvent</code> for a <code>float</code> value and posts it to the event queue.
89          * Eventually <code>firePropertyChangeEvent</code> will be called by <code>dispatchSpeechEvent</code> as a result of
90          * this action.
91          *
92          * @param propName the name of the property
93          * @param oldValue the old value
94          * @param newValue the new value
95          * @see #firePropertyChangeEvent
96          * @see #dispatchSpeechEvent
97          */
98         protected void postPropertyChangeEvent(String propName,
99                                                float oldValue,
100                                                float newValue) {
101             EventObject e = new PropertyChangeEvent(this, propName, oldValue, newValue);
102             SpeechEvent se = new SpeechEventWrapper(e);
103             SpeechEventUtilities.postSpeechEvent(this, se);
104         }
105

108          * Generates a <code>PropertyChangeEvent</code> for a <code>int</code> value and posts it to the event queue.
109          * Eventually <code>firePropertyChangeEvent</code> will be called by <code>dispatchSpeechEvent</code> as a result of
110          * this action.
111          *
112          * @param propName the name of the property
113          * @param oldValue the old value
114          * @param newValue the new value
115          * @see #firePropertyChangeEvent
116          * @see #dispatchSpeechEvent
117          */
118         protected void postPropertyChangeEvent(String propName,
119                                                int oldValue,
120                                                int newValue) {

127         /**
128          * Generates a <code>PropertyChangeEvent</code> for a <code>boolean</code> value and posts it to the event queue.
129          * Eventually <code>firePropertyChangeEvent</code> will be called by <code>dispatchSpeechEvent</code> as a result of
130          * this action.
131          *
132          * @param propName the name of the property
133          * @param oldValue the old value
134          * @param newValue the new value
135          * @see #firePropertyChangeEvent
136          * @see #dispatchSpeechEvent
137          */
138         protected void postPropertyChangeEvent(String propName,
139                                                boolean oldValue,
140                                                boolean newValue) {
141             EventObject e = new PropertyChangeEvent(this, propName, oldValue, newValue);
142             SpeechEvent se = new SpeechEventWrapper(e);
143             SpeechEventUtilities.postSpeechEvent(this, se);

147         /**
148          * Sends a <code>PropertyChangeEvent</code> to all <code>PropertyChangeListeners</code> registered with this object.
149          * Called by <code>dispatchSpeechEvent</code>.
150          *
151          * @param event the <code>PropertyChangeEvent</code> to send
152          * @see #firePropertyChangeEvent
153          * @see #dispatchSpeechEvent
154          */
155         public void firePropertyChangeEvent(PropertyChangeEvent event) {
156             for (PropertyChangeListener pl : propertyChangeListeners)
157                 pl.propertyChange(event);
158         }
159     
160     
161         /**
162          * Dispatches a <code>PropertyChangeEvent</code>. The dispatcher should notify all
163          * <code>PropertyChangeListeners</code> from this method.  The <code>SpeechEvent</code> was added via the various
164          * post methods of this class.
165          *
166          * @param event the <code>SpeechEvent</code> containing a <code>PropertyChangeEvent</code>
167          * @see #postPropertyChangeEvent
168          */
169         public void dispatchSpeechEvent(SpeechEvent event) {
170             if (event instanceof SpeechEventWrapper) {
171                 SpeechEventWrapper se = (SpeechEventWrapper) event;
172                 PropertyChangeEvent pe =
173                         (PropertyChangeEvent) (se.getEventObject());
174                 firePropertyChangeEvent(pe);
175             } else {
176                 throw new SpeechError(
177                         "BaseEngineProperties: speech event type error");
178             }
179         }
180     }
