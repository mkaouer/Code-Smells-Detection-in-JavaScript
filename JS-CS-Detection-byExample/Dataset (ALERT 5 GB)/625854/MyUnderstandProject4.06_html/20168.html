45     /**
46      * Result listener for results from the sphinx recognizer.
47      * 
48      * @author Dirk Schnelle
49      * @author Stefan Radomski
50      * @version $Revision: 608 $
51      * 
52      *          <p>
53      *          Copyright &copy; 2005-2008 JVoiceXML group - <a
54      *          href="http://jvoicexml.sourceforge.net">
55      *          http://jvoicexml.sourceforge.net/</a>
56      *          </p>
57      */
58     class Sphinx4ResultListener implements ResultListener {
59         /** Logger for this class. */
60         private static final Logger LOGGER = Logger
61                 .getLogger(Sphinx4ResultListener.class.getName());
62     
63         /** The recognizer which is notified when a result is obtained. */
64         private final Sphinx4Recognizer recognizer;
65     
66         /** The current Result. */
67         private BaseResult currentResult;
68     
69         /**
70          * Construct a new result listener.
71          * 
72          * @param rec
73          *            The recognizer.
74          */
75         public Sphinx4ResultListener(final Sphinx4Recognizer rec) {
76             recognizer = rec;
77         }
78     
79         /**
80          * Creates a vector of ResultToken (jsapi) from a sphinx result.
81          * 
82          * @param result
83          *            The Sphinx4 result
84          * @param current
85          *            The current BaseResult (jsapi)
86          * @return ResultToken[]
87          */
88         private ResultToken[] sphinx4ResultToResultToken(final Result result,
89                 final BaseResult current) {
90             String strRes = result.getBestFinalResultNoFiller();
91             StringTokenizer st = new StringTokenizer(strRes);
92             int nTokens = st.countTokens();
93     
94             final ResultToken[] res = new ResultToken[nTokens];
95     
96             for (int i = 0; i < nTokens; ++i) {
97                 String text = st.nextToken();
98     
99                 BaseResultToken brt = new BaseResultToken(current, text);
100                 if (current.getResultState() == BaseResult.ACCEPTED) {
101                     // @todo set confidenceLevel, startTime and end time,
102                     // of each token
103                 }
104     
105                 res[i] = brt;
106             }
107             return res;
108         }
109     
110         /**
111          * Method called when a result is generated.
112          * 
113          * @param result
114          *            The new result.
115          */
116         public void newResult(final Result result) {
117             LOGGER.info("received result: " + result);
118             LOGGER.info("isFinal: " + result.isFinal());
119     
120             if (!result.isFinal()) {
121                 LOGGER.warning("result is not final. forget about it.");
122                 return;
123             }
124     
125             /**
126              * For the current implementation with the SRGSGrammarContainer in
127              * Sphinx4, all active grammars are actually a single one. Call the
128              * recognizer with the best token and let him figure out, which grammar
129              * actually produced the result.
130              */
131             final RuleGrammar grammar = recognizer.getRuleGrammar(result
132                     .getBestFinalToken());
133             try {
134                 currentResult = new BaseResult(grammar);
135             } catch (GrammarException e) {
136                 LOGGER.warning(e.getMessage());
137                 return;
138             }
139     
140             final ResultEvent created = new ResultEvent(currentResult,
141                     ResultEvent.RESULT_CREATED, false, false);
142             recognizer.postResultEvent(created);
143     
144             ResultToken[] rt = sphinx4ResultToResultToken(result, currentResult);
145             int numTokens = rt.length;
146     
147             final ResultEvent grammarFinalized = new ResultEvent(currentResult,
148                     ResultEvent.GRAMMAR_FINALIZED);
149             recognizer.postResultEvent(grammarFinalized);
150     
151             if (numTokens == 0) {
152                 currentResult.setResultState(BaseResult.REJECTED);
153                 final ResultEvent rejected = new ResultEvent(currentResult,
154                         ResultEvent.RESULT_REJECTED, false, false);
155                 recognizer.postResultEvent(rejected);
156             } else {
157                 currentResult.setResultState(BaseResult.ACCEPTED);
158                 currentResult.setNumTokens(numTokens);
159                 currentResult.setTokens(rt);
160                 final ResultEvent accepted = new ResultEvent(currentResult,
161                         ResultEvent.RESULT_ACCEPTED, false, false);
162                 recognizer.postResultEvent(accepted);
163             }
164         }

168                 throws PropertyException {
169         }
170     }
