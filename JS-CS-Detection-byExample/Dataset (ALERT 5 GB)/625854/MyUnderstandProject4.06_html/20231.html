21     /**
22      * Computes confidences for the highest scoring path in a Result. The highest scoring path refers to the path with the
23      * maximum a posteriori (MAP) probability, which is why this class is so named. Note that this MAPConfidenceScorer
24      * creates a {@link edu.cmu.sphinx.result.Lattice} from the result first, which means that you should only use this
25      * confidence scorer if the result is created from the {@link edu.cmu.sphinx.linguist.lextree.LexTreeLinguist} and the
26      * {@link edu.cmu.sphinx.decoder.search.WordPruningBreadthFirstSearchManager}.
27      */
28     public class MAPConfidenceScorer implements ConfidenceScorer, Configurable {
29     
30         /** The property that defines the weight multiplier that will be applied to language score already scaled by language weight. */
31         @S4Double(defaultValue = 1.0)
32         public final static String PROP_LANGUAGE_WEIGHT_ADJUSTMENT = "languageWeightAdjustment";
33     
34     
35         /** The property that specifies whether to dump the lattice. */
36         @S4Boolean(defaultValue = false)
37         public final static String PROP_DUMP_LATTICE = "dumpLattice";
38     
39     
40         /** The property that specifies whether to dump the sausage. */
41         @S4Boolean(defaultValue = false)
42         public final static String PROP_DUMP_SAUSAGE = "dumpSausage";
43     
44     
45         private float languageWeightAdjustment;
46         private boolean dumpLattice;
47         private boolean dumpSausage;
48     
49         public MAPConfidenceScorer(float languageWeightAdjustment, boolean dumpLattice, boolean dumpSausage) {
50             this.languageWeightAdjustment = languageWeightAdjustment;
51             this.dumpLattice = dumpLattice;
52             this.dumpSausage = dumpSausage;
53         }
54     
55         public MAPConfidenceScorer() {
56         }
57     
58         /*
59          * (non-Javadoc)
60          *
61          * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
62          */
63         @Override
64         public void newProperties(PropertySheet ps) throws PropertyException {
65             languageWeightAdjustment = ps.getFloat(PROP_LANGUAGE_WEIGHT_ADJUSTMENT);
66             dumpLattice = ps.getBoolean(PROP_DUMP_LATTICE);
67             dumpSausage = ps.getBoolean(PROP_DUMP_SAUSAGE);
68         }
69     
70     
71         /**
72          * Computes confidences for a Result and returns a ConfidenceResult, a compact representation of all the hypothesis
73          * contained in the result together with their per-word and per-path confidences.
74          *
75          * @param result the result to compute confidences for
76          * @return a confidence result
77          */
78         @Override
79         public ConfidenceResult score(Result result) {
80             Lattice lattice = new Lattice(result);
81             LatticeOptimizer lop = new LatticeOptimizer(lattice);
82             lop.optimize();
83             lattice.computeNodePosteriors(languageWeightAdjustment);
84             SausageMaker sm = new SausageMaker(lattice);
85             Sausage s = sm.makeSausage();
86     
87             if (dumpLattice) {
88                 lattice.dumpAISee("mapLattice.gdl", "MAP Lattice");
89             }
90             if (dumpSausage) {
91                 s.dumpAISee("mapSausage.gdl", "MAP Sausage");
92             }
93     
94             WordResultPath mapPath = new WordResultPath();
95             List<Token> wordTokens = getWordTokens(result.getBestToken());
96     
97             /* start with the first slot */
98             int slot = 0;
99     
100             for (Token wordToken : wordTokens) {
101                 String word = wordToken.getWord().getSpelling();
102                 WordResult wr = null;
103                 ConfusionSet cs = null;
104     
105                 /* track through all the slots to find the word */
106                 while (slot < s.size() && wr == null) {
107                     cs = s.getConfusionSet(slot);
108                     wr = cs.getWordResult(word);
109                     if (wr == null) {
110                         slot++;
111                     }
112                 }
113                 if (wr != null) {
114                     mapPath.add(wr);
115                 } else {
116                     cs.dump("Slot " + slot);
117                     throw new Error
118                             ("Can't find WordResult in ConfidenceResult slot " +
119                                     slot + " for word " + word);
120                 }
121                 slot++;
122             }
123     
124             return (new MAPConfidenceResult(s, mapPath));
125         }
126     
127     
128         /**
129          * Returns all the word tokens ending at the given token as a List.
130          *
131          * @param lastToken the last token in the token chain
132          * @return a list of word tokens in order of appearance
133          */
134         private List<Token> getWordTokens(Token lastToken) {
135             List<Token> wordTokens = new LinkedList<Token>();
136             Token token = lastToken;
137             while (token != null) {
138                 if (token.isWord()) {
139                     wordTokens.add(0, token);
140                 }
141                 token = token.getPredecessor();
142             }
143             return wordTokens;
144         }

147         /** The confidence result for the highest scoring path. */
148         class MAPConfidenceResult implements ConfidenceResult {
149     
150             private final ConfidenceResult sausage;
151             private final Path mapPath;
152     
153     
154             /**
155              * Constructs a MAPConfidenceResult.
156              *
157              * @param sausage the sausge that this MAPConfidenceResult is based on
158              * @param mapPath the maximum posterior probability path
159              */
160             public MAPConfidenceResult(ConfidenceResult sausage, Path mapPath) {
161                 this.sausage = sausage;
162                 this.mapPath = mapPath;
163             }
164     
165     
166             /**
167              * Returns the path with the maximum posterior probability path. This path should be the same as that returned
168              * by Result.getBestToken().
169              */
170             @Override
171             public Path getBestHypothesis() {
172                 return mapPath;
173             }
174     
175     
176             /**
177              * Get the number of word slots contained in this result
178              *
179              * @return length of the result
180              */
181             @Override
182             public int size() {
183                 return sausage.size();
184             }
185     
186     
187             /**
188              * Iterator through the confusion sets in this result.
189              *
190              * @return confusion set iterator
191              */
192             @Override
193             public Iterator<ConfusionSet> iterator() {
194                 return sausage.iterator();
195             }
196     
197     
198             /**
199              * Get the nth confusion set in this result
200              *
201              * @param i the index of the confusion set to get
202              * @return the requested confusion set
203              */
204             @Override
205             public ConfusionSet getConfusionSet(int i) {
206                 return sausage.getConfusionSet(i);
207             }
208         }
