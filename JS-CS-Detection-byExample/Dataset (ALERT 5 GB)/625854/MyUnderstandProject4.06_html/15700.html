48     import javax.sound.sampled.AudioFormat;
49     import javax.speech.AudioManager;
50     import javax.speech.SpeechEventExecutor;
51     import javax.speech.recognition.Grammar;
52     import javax.speech.recognition.GrammarException;
53     import javax.speech.recognition.GrammarManager;
54     import javax.speech.recognition.RecognizerMode;
55     import javax.speech.recognition.Result;
56     import javax.speech.recognition.ResultEvent;
57     
58     import org.jvoicexml.jsapi2.jse.ThreadSpeechEventExecutor;
59     import org.jvoicexml.jsapi2.recognition.BaseRecognizer;
60     
61     
62     /**
63      * Skeletal Implementation of the JSAPI Recognizer interface.
64      *
65      * This class is useful by itself for debugging, e.g. you
66      * can load grammars and simulate a recognizer recognizing
67      * some text, etc.
68      *
69      * <P>
70      *
71      * Actual JSAPI recognizer implementations might want to extend or
72      * modify this implementation.
73      *
74      */
75     public abstract class JseBaseRecognizer extends BaseRecognizer {
76         /**
77          * Create a new Recognizer in the DEALLOCATED state.
78          */
79         public JseBaseRecognizer() {
80         }
81     
82         /**
83          * Create a new Recognizer in the DEALLOCATED state.
84          * @param mode the recognizer mode
85          */
86         public JseBaseRecognizer(final RecognizerMode mode) {
87             super(mode);
88         }
89     
90         /**
91          * {@inheritDoc}
92          */
93         @Override
94         protected GrammarManager createGrammarManager() {
95             return new BaseGrammarManager(this);
96         }
97     
98         /**
99          * {@inheritDoc}
100          */
101         @Override
102         protected AudioManager createAudioManager() {
103             final AudioFormat format = new AudioFormat(8000f, 16, 1, true, true);
104             return new BaseRecognizerAudioManager(format);
105         }
106     
107         /**
108          * {@inheritDoc}
109          */
110         @Override
111         protected boolean processGrammar(final Grammar grammar)
112             throws GrammarException {
113             if (grammar instanceof BaseRuleGrammar) {
114                 BaseRuleGrammar baseRuleGrammar = (BaseRuleGrammar) grammar;
115                 return baseRuleGrammar.commitChanges();
116             }
117             return false;
118         }
119     
120         /**
121          * {@inheritDoc}
122          */
123         @Override
124         protected void postResultEvent(final Result result, final ResultEvent event,
125                 final SpeechEventExecutor executor) {
126             final BaseResult base = (BaseResult) result;
127             base.postResultEvent(executor, event);
128         }
129     
130     
131         /**
132          * {@inheritDoc}
133          */
134         @Override
135         protected SpeechEventExecutor createSpeechEventExecutor() {
136             return new ThreadSpeechEventExecutor();
137         }
138     }
