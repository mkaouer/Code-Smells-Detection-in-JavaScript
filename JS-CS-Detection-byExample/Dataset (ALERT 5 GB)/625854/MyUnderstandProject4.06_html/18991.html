21     /**
22      * Defines a grammar based upon a list of words in a file. The format of the file is just one word per line. For
23      * example, for an isolated digits grammar the file will simply look like:
24      * <pre>
25      * zero
26      * one
27      * two
28      * three
29      * four
30      * five
31      * six
32      * seven
33      * eight
34      * nine
35      * </pre>
36      * The path to the file is defined by the {@link #PROP_PATH PROP_PATH} property. If the {@link #PROP_LOOP PROP_LOOP}
37      * property is true, the grammar created will be a looping grammar. Using the above digits grammar example, setting
38      * PROP_LOOP to true will make it a connected-digits grammar.
39      * <p/>
40      * All probabilities are maintained in LogMath log base.
41      */
42     public class SimpleWordListGrammar extends Grammar implements Configurable {
43     
44         /** The property that defines the location of the word list grammar */
45         @S4String(defaultValue = "spelling.gram")
46         public final static String PROP_PATH = "path";
47     
48         /** The property that if true, indicates that this is a looping grammar */
49         @S4Boolean(defaultValue = true)
50         public final static String PROP_LOOP = "isLooping";
51     
52         /** The property that defines the logMath component. */
53         @S4Component(type = LogMath.class)
54         public final static String PROP_LOG_MATH = "logMath";
55     
56     
57         // ---------------------
58         // Configurable data
59         // ---------------------
60         private String path;
61         private boolean isLooping;
62         private LogMath logMath;
63     
64         public SimpleWordListGrammar(String path, boolean isLooping, LogMath logMath, boolean showGrammar, boolean optimizeGrammar, boolean addSilenceWords, boolean addFillerWords, edu.cmu.sphinx.linguist.dictionary.Dictionary dictionary) {
65             super(showGrammar,optimizeGrammar,addSilenceWords,addFillerWords,dictionary);
66             this.path = path;
67             this.isLooping = isLooping;
68             this.logMath = logMath;
69         }
70     
71         public SimpleWordListGrammar() {
72     
73         }
74     
75         /*
76         * (non-Javadoc)
77         *
78         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
79         */
80         @Override
81         public void newProperties(PropertySheet ps) throws PropertyException {
82             super.newProperties(ps);
83             
84             path = ps.getString(PROP_PATH);
85             isLooping = ps.getBoolean(PROP_LOOP);
86             logMath = (LogMath) ps.getComponent(PROP_LOG_MATH);
87         }

91          * Create class from reference text (not implemented).
92          *
93          * @param bogusText dummy variable
94          */
95         @Override
96         protected GrammarNode createGrammar(String bogusText)
97                 throws NoSuchMethodException {
98             throw new NoSuchMethodException("Does not create "
99                     + "grammar with reference text");
100         }
101     
102     
103         /** Creates the grammar. */
104         @Override
105         protected GrammarNode createGrammar() throws IOException {
106             ExtendedStreamTokenizer tok = new ExtendedStreamTokenizer(path, true);
107             GrammarNode initialNode = createGrammarNode("<sil>");
108             GrammarNode branchNode = createGrammarNode(false);
109             GrammarNode finalNode = createGrammarNode("<sil>");
110             finalNode.setFinalNode(true);
111             List<GrammarNode> wordGrammarNodes = new LinkedList<GrammarNode>();
112             while (!tok.isEOF()) {
113                 String word;
114                 while ((word = tok.getString()) != null) {
115                     word = word.toLowerCase();
116                     GrammarNode wordNode = createGrammarNode(word);
117                     wordGrammarNodes.add(wordNode);
118                 }
119             }
120             // now connect all the GrammarNodes together
121             initialNode.add(branchNode, LogMath.getLogOne());
122             float branchScore = logMath.linearToLog(
123                     1.0 / wordGrammarNodes.size());
124             for (GrammarNode wordNode : wordGrammarNodes) {
125                 branchNode.add(wordNode, branchScore);
126                 wordNode.add(finalNode, LogMath.getLogOne());
127                 if (isLooping) {
128                     wordNode.add(branchNode, LogMath.getLogOne());
129                 }
130             }
131     
132             return initialNode;
133         }
