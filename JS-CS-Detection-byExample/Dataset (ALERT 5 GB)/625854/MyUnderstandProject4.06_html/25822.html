33     /**
34      * Sphinx4 SRGS grammar container.
35      * 
36      * <p>
37      * This is a grammar for a linguist in sphinx4. There can not be more than one
38      * grammar per linguist so we cheat by having a single grammar which contains
39      * all the active grammars from the GrammarManager.
40      * </p>
41      * 
42      * @author Stefan Radomski
43      * @version $Revision: 1 $
44      */
45     
46     public class SRGSGrammarContainer extends Grammar {
47     
48         @S4Component(type = LogMath.class)
49         public final static String PROP_LOG_MATH = "logMath";
50     
51         private static final Logger LOGGER = Logger
52                 .getLogger(Sphinx4Recognizer.class.getName());
53     
54         /**
55          * The GrammarDefinitions as set through loadGrammars from the
56          * GrammarManager
57          */
58         private Map<String, GrammarDefinition> grammarDefs = new HashMap<String, GrammarDefinition>();
59     
60         /** All active SRGSGrammars */
61         private Map<String, SRGSGrammar> grammars = new HashMap<String, SRGSGrammar>();
62     
63         /** The logMath used */
64         private LogMath logMath;
65     
66         /** The initial node for the searchGraph of the linguist */
67         private GrammarNode firstNode = null;
68     
69         /** The rule grammar as the union of all contained grammars */
70         private BaseRuleGrammar ruleGrammar = null;
71     
72         /** All GrammarNodes of contained grammars plus the firstNode */
73         private Set<GrammarNode> grammarNodes = new LinkedHashSet<GrammarNode>();

75         /*
76          * (non-Javadoc)
77          *

87         /**
88          * Load all the given grammars into this one.
89          * 
90          * The argument contains all active grammars with their names, their XML
91          * representation and an indication whether they have changed.
92          * 
93          * @param grammarDefinition
94          *            The set of all grammars from the GrammarManager
95          * @throws IOException
96          */
97         public synchronized void loadGrammars(Vector grammarDefinitions)
98                 throws IOException {
99             grammarDefs.clear();
100             for (Object obj : grammarDefinitions) {
101                 GrammarDefinition gd = (GrammarDefinition) obj;
102                 grammarDefs.put(gd.getName(), gd);
103             }
104             commitChanges();
105         }

107         /**
108          * Create the grammar.
109          * 
110          * @return the initial node of the Grammar
111          */
112         @Override
113         protected synchronized GrammarNode createGrammar() throws IOException {
114             commitChanges();
115             return firstNode;
116         }

118         /**
119          * Returns the initial node for the grammar.
120          * 
121          * The sphinx4 linguist will rebuild its searchGraph iff this is a different

123          * RecognizerThread started in handleResume().
124          * 
125          * @return the initial grammar node
126          */
127         @Override
128         public synchronized GrammarNode getInitialNode() {
129             return firstNode;
130         }

142         /**
143          * The Spinx4ResultListener asked us for the grammar that produced this list
144          * of tokens.
145          * 
146          * @param token
147          * @return
148          */
149         public synchronized RuleGrammar getRuleGrammar(Token token) {
150             for (SRGSGrammar grammar : grammars.values()) {
151                 if (grammar.getGrammarNodes().contains(
152                         ((GrammarState) token.getSearchState()).getGrammarNode())) {
153                     return grammar.getRuleGrammar();
154                 }
155             }
156             return ruleGrammar;
157         }

159         /**
160          * Commit all pending changes.
161          * 
162          * All active grammars are in grammarDefs, if they were changed, they have
163          * their hasChanges flag set. See if there is a new grammar, if an existing
164          * one was deactivated or there have been changes to an active one. Build
165          * new grammars in grammars hash and adapt firstNaode and ruleGrammar.
166          * 
167          * @throws IOException
168          */
169         public synchronized void commitChanges() throws IOException {
170     
171             // we need a ruleGrammar in any case
172             if (ruleGrammar == null) {
173                 ruleGrammar = new BaseRuleGrammar(null, "srgs_container");
174             }
175     
176             boolean existsChanges = false;
177     
178             // name of all active grammars
179             HashSet<String> activeGrammarNames = new HashSet<String>();
180             activeGrammarNames.addAll(grammarDefs.keySet());
181     
182             // is an active grammar to be removed?
183             for (Object obj : grammars.keySet()) {
184                 String name = (String) obj;
185                 if (!activeGrammarNames.contains(name)) {
186                     grammars.remove(name);
187                     existsChanges = true;
188                 }
189             }
190     
191             // is there a new or changed grammar?
192             for (String grammarName : activeGrammarNames) {
193                 GrammarDefinition grammarDef = grammarDefs.get(grammarName);
194                 if (!grammars.containsKey(grammarName) || grammarDef.hasChanged()) {
195                     // no grammar with that name yet or changes to the grammar
196                     existsChanges = true;
197     
198                     // reload the grammar (TODO: reuse existing object?)
199                     SRGSGrammar grammar = new SRGSGrammar(false, false, false,
200                             false, dictionary);
201                     grammar.setGrammarName(grammarName);
202                     String grammarString = grammarDef.getGrammar();
203                     grammar.loadSRGS(grammarString);
204     
205                     // insert or replace new grammar
206                     grammars.put(grammarName, grammar);
207                 }
208             }
209     
210             // rebuild ruleGrammar, firstNode and the set of all grammar nodes
211             if (existsChanges) {
212                 // new first node, so the linguist will rebuild its searchgraph
213                 firstNode = createGrammarNode("<sil>");
214                 firstNode.setFinalNode(false);
215     
216                 grammarNodes.clear();
217                 grammarNodes.add(firstNode);
218     
219                 for (SRGSGrammar grammar : grammars.values()) {
220                     GrammarNode srgsStart = grammar.getInitialNode();
221     
222                     /**
223                      * Every SRGS Grammar starts with <sil>, drop it and add
224                      * transitions from our firstNode.
225                      */
226                     for (GrammarArc transition : srgsStart.getSuccessors()) {
227                         firstNode.add(transition.getGrammarNode(), LogMath
228                                 .getLogOne());
229                     }
230     
231                     /**
232                      * Add all the rules to the rule grammar for JSAPI2.
233                      */
234                     for (String name : grammar.getRuleGrammar().listRuleNames()) {
235                         ruleGrammar.addRule(grammar.getRuleGrammar().getRule(name));
236                     }
237     
238                     // Gather all grammar nodes
239                     grammarNodes.addAll(grammar.getGrammarNodes());
240                 }
241             }
242     
243             // If we did not yet create a firstnode, create an empty one
244             if (firstNode == null) {
245                 newGrammar();
246                 firstNode = createGrammarNode("<sil>");
247                 firstNode.setFinalNode(false);
248                 grammarNodes.clear();
249                 grammarNodes.add(firstNode);
250             }
251     
252             if (LOGGER.isLoggable(Level.INFO)) {
253                 StringBuilder sb = new StringBuilder();
254                 for (String activeGrammar : grammars.keySet()) {
255                     sb.append(activeGrammar + " ");
256                 }
257                 LOGGER.info("Activate grammars: " + sb.toString());
258             }
259         }
