13     package edu.cmu.sphinx.instrumentation;
14     
15     import edu.cmu.sphinx.decoder.search.Token;
16     import edu.cmu.sphinx.result.Lattice;
17     import edu.cmu.sphinx.result.LatticeOptimizer;
18     import edu.cmu.sphinx.result.Result;
19     import edu.cmu.sphinx.result.Sausage;
20     import edu.cmu.sphinx.result.SausageMaker;
21     import edu.cmu.sphinx.util.props.*;
22     import edu.cmu.sphinx.recognizer.Recognizer;
23     
24     /** Tracks and reports recognition accuracy based upon the highest scoring path in a Result. */
25     public class SausageAccuracyTracker extends AccuracyTracker {
26     
27         /** The property that defines whether the full token path is displayed */
28         @S4Boolean(defaultValue = false)
29         public final static String PROP_SHOW_FULL_PATH = "showFullPath";
30     
31         /** The property with language model weight for posterior probability computation */
32         @S4Double(defaultValue = 10.5f)
33         public final static String PROP_LANGUAGE_WEIGHT = "languageWeight";
34     
35         private boolean showFullPath;
36         private float languageModelWeight;
37         
38         public SausageAccuracyTracker(Recognizer recognizer, boolean showSummary, boolean showDetails, boolean showResults, boolean showAlignedResults, boolean showRawResults, boolean showFullPath, float languageWeight) {
39             super(recognizer, showSummary, showDetails, showResults, showAlignedResults, showRawResults);
40             this.showFullPath = showFullPath;
41             this.languageModelWeight = languageWeight;
42         }
43     
44         public SausageAccuracyTracker() {
45     
46         }
47     
48     
49         /*
50         * (non-Javadoc)
51         *
52         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
53         */
54         @Override
55         public void newProperties(PropertySheet ps) throws PropertyException {
56             super.newProperties(ps);
57             showFullPath = ps.getBoolean(PROP_SHOW_FULL_PATH);
58             languageModelWeight = ps.getFloat(PROP_LANGUAGE_WEIGHT);
59         }
60     
61     
62         /**
63          * Dumps the best path
64          *
65          * @param result the result to dump
66          */
67         private void showFullPath(Result result) {
68             if (showFullPath) {
69                 System.out.println();
70                 Token bestToken = result.getBestToken();
71                 if (bestToken != null) {
72                     bestToken.dumpTokenPath();
73                 } else {
74                     System.out.println("Null result");
75                 }
76                 System.out.println();
77             }
78         }
79     
80     
81         /*
82         * (non-Javadoc)
83         *
84         * @see edu.cmu.sphinx.decoder.ResultListener#newResult(edu.cmu.sphinx.result.Result)
85         */
86         @Override
87         public void newResult(Result result) {
88             String ref = result.getReferenceText();
89             if (result.isFinal() && ref != null) {
90                 Lattice lattice = new Lattice(result);
91                 LatticeOptimizer optimizer = new LatticeOptimizer(lattice);
92                 optimizer.optimize();           
93                 lattice.computeNodePosteriors(languageModelWeight);
94                 SausageMaker sausageMaker = new SausageMaker(lattice);
95                 Sausage sausage = sausageMaker.makeSausage();
96                 sausage.removeFillers();
97                     
98                 getAligner().alignSausage(ref, sausage);
99                 showFullPath(result);
100                 showDetails(result.toString());
101             }
102         }
103     }
