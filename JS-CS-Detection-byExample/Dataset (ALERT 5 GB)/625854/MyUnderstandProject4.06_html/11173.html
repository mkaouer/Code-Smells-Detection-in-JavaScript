37         private static int hashCode(Object[] array) {
38             int prime = 31;
39             if (array == null)
40                 return 0;
41             int result = 1;
42             for (int index = 0; index < array.length; index++) {
43                 result = prime * result
44                         + (array[index] == null ? 0 : array[index].hashCode());
45             }
46             return result;
47         }

74         public RecognizerMode(String engineName, String modeName, Boolean running,
75                 Boolean supportsLetterToSound, Boolean supportsMarkup,
76                 Integer vocabSupport, SpeechLocale[] locales,
77                 SpeakerProfile[] profiles) {
78             super(engineName, modeName, running, supportsLetterToSound,
79                     supportsMarkup);
80             this.vocabSupport = vocabSupport;
81             this.locales = locales;
82             this.profiles = profiles;
83         }

86         public int hashCode() {
87             final int prime = 31;
88             int result = super.hashCode();
89             result = prime * result + RecognizerMode.hashCode(locales);
90             result = prime * result + RecognizerMode.hashCode(profiles);
91             result = prime * result
92                     + ((vocabSupport == null) ? 0 : vocabSupport.hashCode());
93             return result;
94         }

96         public boolean equals(Object obj) {
97             if (this == obj) {
98                 return true;
99             }
100             if (!super.equals(obj)) {
101                 return false;
102             }
103             if (getClass() != obj.getClass()) {
104                 return false;
105             }
106             RecognizerMode other = (RecognizerMode) obj;
107             if (locales == null) {
108                 if (other.locales != null) {
109                     return false;
110                 }
111             } else if (locales.length != other.locales.length) {
112                 return false;
113             } else {
114                 for (int i=0; i<locales.length; i++) {
115                     if (!locales[i].equals(other.locales[i])) {
116                         return false;
117                     }
118                 }
119             }
120             if (profiles == null) {
121                 if (other.profiles != null) {
122                     return false;
123                 }
124             } else if (profiles.length != other.profiles.length) {
125                 return false;
126             } else {
127                 for (int i=0; i<profiles.length; i++) {
128                     if (!profiles[i].equals(other.profiles[i])) {
129                         return false;
130                     }
131                 }
132             }
133             if (vocabSupport == null) {
134                 if (other.vocabSupport != null) {
135                     return false;
136                 }
137             } else if (!vocabSupport.equals(other.vocabSupport)) {
138                 return false;
139             }
140             return true;
141         }

155         public boolean match(EngineMode require) {
156             if (!super.match(require)) {
157                 return false;
158             }
159     
160             if (require instanceof RecognizerMode) {
161                 final RecognizerMode mode = (RecognizerMode) require;
162                 final SpeechLocale[] otherLocales = mode.getSpeechLocales();
163                 if (otherLocales != null) {
164                     if (locales == null) {
165                         return false;
166                     }
167     
168                     boolean match = false;
169                     for (int i = 0; (i < otherLocales.length) && !match; i++) {
170                         final SpeechLocale otherLocale = otherLocales[i];
171     
172                         for (int k = 0; k < locales.length; k++) {
173                             final SpeechLocale locale = locales[k];
174                             if (locale.equals(otherLocale)) {
175                                 match = true;
176                             }
177                         }
178     
179                         if (!match) {
180                             return false;
181                         }
182                     }
183                 }
184     
185                 final SpeakerProfile[] otherProfiles = mode.getSpeakerProfiles();
186                 if (otherProfiles != null) {
187                     if (profiles == null) {
188                         return false;
189                     }
190     
191                     boolean match = false;
192                     for (int i = 0; (i < otherProfiles.length) && !match; i++) {
193                         final SpeakerProfile otherProfile = otherProfiles[i];
194     
195                         for (int k = 0; k < profiles.length; k++) {
196                             final SpeakerProfile profile = profiles[k];
197                             if (profile.equals(otherProfile)) {
198                                 match = true;
199                             }
200                         }
201     
202                         if (!match) {
203                             return false;
204                         }
205                     }
206                 }
207             }
208     
209             return true;
210         }

217         protected Vector getParameters() {
218             final Vector parameters = super.getParameters();
219     
220             parameters.addElement(vocabSupport);
221             if (locales == null) {
222                 parameters.addElement(null);
223             } else {
224                 final Vector vec = new Vector();
225                 for (int i = 0; i < locales.length; i++) {
226                     vec.addElement(locales[i]);
227                 }
228                 parameters.addElement(vec);
229             }
230     
231             if (profiles == null) {
232                 parameters.addElement(null);
233             } else {
234                 final Vector vec = new Vector();
235                 for (int i = 0; i < profiles.length; i++) {
236                     vec.addElement(profiles[i]);
237                 }
238                 parameters.addElement(vec);
239             }
240     
241             return parameters;
242         }
