13     package org.jvoicexml.jsapi2.jse.protocols.playback;
14     
15     import java.io.Closeable;
16     import java.io.IOException;
17     import java.io.OutputStream;
18     
19     import javax.sound.sampled.SourceDataLine;
20     
21     /**
22      * An {@link OutputStream} that writes to a {@link SourceDataLine}.
23      * @author Dirk Schnelle-Walka
24      *
25      */
26     public class LineOutputStream extends OutputStream
27         implements Closeable {
28         /** The source data line. */
29         private SourceDataLine line;
30     
31         /**
32          * Constructs a new object.
33          * @param source the line to write to.
34          */
35         public LineOutputStream(final SourceDataLine source) {
36             line = source;
37         }
38     
39         /**
40          * {@inheritDoc}
41          */
42         @Override
43         public void write(final int b) throws IOException {
44             byte[] bytes = new byte[1];
45             write(bytes, 0, bytes.length);
46         }
47     
48         /**
49          * {@inheritDoc}
50          */
51         @Override
52         public void write(final byte[] b, final int off, final int len)
53             throws IOException {
54             line.write(b, off, len);
55         }
56     
57         /**
58          * {@inheritDoc}
59          */
60         @Override
61         public void write(final byte[] b) throws IOException {
62             write(b, 0, b.length);
63         }
64     
65         /**
66          * {@inheritDoc}
67          */
68         @Override
69         public void close() throws IOException {
70             line.drain();
71             line.stop();
72             line.close();
73             super.close();
74         }
75     
76         /**
77          * {@inheritDoc}
78          */
79         @Override
80         public void flush() throws IOException {
81             line.drain();
82             super.flush();
83         }
84     
85     }
