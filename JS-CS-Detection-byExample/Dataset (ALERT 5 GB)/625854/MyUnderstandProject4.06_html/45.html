13     package org.jvoicexml.jsapi2.jse.protocols;
14     
15     import java.net.URI;
16     import java.net.URISyntaxException;
17     import java.net.URL;
18     import java.util.HashMap;
19     import java.util.Map;
20     
21     import javax.sound.sampled.AudioFormat;
22     
23     /**
24      * A parser for the javasound URL schema.
25      *
26      * @author Dirk Schnelle-Walka
27      *
28      */
29     public final class JavaSoundParser {
30         /** The default sample rate. */
31         public static final int DEFAULT_SAMPLE_RATE = 8000;
32     
33         /** Number of bits per byte. */
34         private static final int BITS_PER_BYTE = 8;
35     
36         /**
37          * Prevent construction from outside.
38          */
39         private JavaSoundParser() {
40         }
41     
42         /**
43          * Parses the given URI into an audio format.
44          *
45          * @param url
46          *            the URL to parse.
47          * @return audio format.
48          * @throws URISyntaxException
49          *         error parsing the URL
50          */
51         public static AudioFormat parse(final URL url) throws URISyntaxException {
52             URI uri = url.toURI();
53             final Map<String, String> parameters = new HashMap<String, String>();
54             if (uri.getQuery() != null) {
55                 String[] parametersString = uri.getQuery().split("\\&");
56                 for (String part : parametersString) {
57                     String[] queryElement = part.split("\\=");
58                     parameters.put(queryElement[0], queryElement[1]);
59                 }
60             }
61     
62             // Default values for AudioFormat parameters
63             AudioFormat.Encoding encoding = AudioFormat.Encoding.ULAW;
64             float sampleRate = DEFAULT_SAMPLE_RATE;
65             int bits = BITS_PER_BYTE;
66             int channels = 1;
67             boolean endian = true;
68             boolean signed = true;
69     
70             // Change default values as specified
71             final String signedStr = parameters.get("signed");
72             if (signedStr != null) {
73                 signed = Boolean.valueOf(signedStr);
74             }
75     
76             final String encodingStr = parameters.get("encoding");
77             if (encodingStr != null) {
78                 if (encodingStr.equals("pcm")) {
79                     if (signed) {
80                         encoding = AudioFormat.Encoding.PCM_SIGNED;
81                     } else {
82                         encoding = AudioFormat.Encoding.PCM_UNSIGNED;
83                     }
84                 } else if (encodingStr.equals("alaw")) {
85                     encoding = AudioFormat.Encoding.ALAW;
86                 } else if (encodingStr.equals("ulaw")) {
87                     encoding = AudioFormat.Encoding.ULAW;
88                 } else if (encodingStr.equals("gsm")) {
89                     throw new URISyntaxException(url.toString(),
90                             "gsm is currently not supported!");
91                 }
92             }
93     
94             final String rateStr = parameters.get("rate");
95             if (rateStr != null) {
96                 sampleRate = Float.valueOf(rateStr);
97             }
98     
99             final String bitsStr = parameters.get("bits");
100             if (bitsStr != null) {
101                 bits = Integer.valueOf(bitsStr);
102             }
103     
104             final String channelsStr = parameters.get("channels");
105             if (channelsStr != null) {
106                 channels = Integer.valueOf(channelsStr);
107             }
108     
109             final String endianStr = parameters.get("endian");
110             if (endianStr != null) {
111                 if (endianStr.equals("little")) {
112                     endian = false;
113                 } else if (endianStr.equals("big")) {
114                     endian = true;
115                 }
116             }
117     
118             // Construct the AudioFormat
119             return new AudioFormat(encoding, sampleRate,
120                     bits, channels, bits / BITS_PER_BYTE, sampleRate, endian);
121         }
122     }
