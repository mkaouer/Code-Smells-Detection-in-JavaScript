16     import edu.cmu.sphinx.frontend.BaseDataProcessor;
17     import edu.cmu.sphinx.frontend.Data;

22     /**
23      * Computes the PLP cepstrum from a given PLP Spectrum. The power spectrum has the amplitude compressed by computing the
24      * cubed root of the PLP spectrum.  This operation is an approximation to the power law of hearing and simulates the
25      * non-linear relationship between sound intensity and perceived loudness.  Computationally, this operation is used to
26      * reduce the spectral amplitude of the critical band to enable all-pole modeling with relatively low order AR filters.
27      * The inverse discrete cosine transform (IDCT) is then applied to the autocorrelation coefficients. A linear prediction
28      * filter is then estimated from the autocorrelation values, and the linear prediction cepstrum (LPC cepstrum) is
29      * finally computed from the LP filter.
30      *
31      * @author <a href="mailto:rsingh@cs.cmu.edu">rsingh</a>
32      * @version 1.0
33      * @see LinearPredictor
34      */
35     public class PLPCepstrumProducer extends BaseDataProcessor {
36     
37         /** The property for the number of filters in the filter bank. */
38         @S4Integer(defaultValue = 32)
39         public static final String PROP_NUMBER_FILTERS = "numberFilters";
40     
41         /** The property specifying the length of the cepstrum data. */
42         @S4Integer(defaultValue = 13)
43         public static final String PROP_CEPSTRUM_LENGTH
44                 = "cepstrumLength";
45     
46         /** The property specifying the LPC order. */
47         @S4Integer(defaultValue = 14)
48         public static final String PROP_LPC_ORDER = "lpcOrder";
49     
50         private int cepstrumSize;       // size of a Cepstrum
51         private int LPCOrder;           // LPC Order to compute cepstrum
52         private int numberPLPFilters;   // number of PLP filters
53         private double[][] cosine;
54     
55         public PLPCepstrumProducer(int numberPLPFilters,int cepstrumSize,int LPCOrder) {
56             initLogger();
57             this.numberPLPFilters = numberPLPFilters;
58             this.cepstrumSize = cepstrumSize;
59             this.LPCOrder = LPCOrder;
60         }
61     
62         public PLPCepstrumProducer() {
63         }
64     
65         /*
66         * (non-Javadoc)
67         *
68         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
69         */
70         @Override
71         public void newProperties(PropertySheet ps) throws PropertyException {
72             super.newProperties(ps);
73             numberPLPFilters = ps.getInt(PROP_NUMBER_FILTERS);
74             cepstrumSize = ps.getInt(PROP_CEPSTRUM_LENGTH);
75             LPCOrder = ps.getInt(PROP_LPC_ORDER);
76         }
77     
78     
79         /** Constructs a PLPCepstrumProducer */
80         @Override
81         public void initialize() {
82             super.initialize();
83             computeCosine();
84         }
85     
86     
87         /** Compute the Cosine values for IDCT. */
88         private void computeCosine() {
89             cosine = new double[LPCOrder + 1][numberPLPFilters];
90     
91             double period = (double) 2 * numberPLPFilters;
92     
93             for (int i = 0; i <= LPCOrder; i++) {
94                 double frequency = 2 * Math.PI * (double) i / period;
95     
96                 for (int j = 0; j < numberPLPFilters; j++) {
97                     cosine[i][j] = Math.cos(frequency * (j + 0.5));
98                 }
99             }
100         }
101     
102     
103         /**
104          * Applies the intensity loudness power law. This operation is an approximation to the power law of hearing and
105          * simulates the non-linear relationship between sound intensity and percieved loudness. Computationally, this
106          * operation is used to reduce the spectral amplitude of the critical band to enable all-pole modeling with
107          * relatively low order AR filters.
108          * @param inspectrum
109          */
110         private double[] powerLawCompress(double[] inspectrum) {
111             double[] compressedspectrum = new double[inspectrum.length];
112     
113             for (int i = 0; i < inspectrum.length; i++) {
114                 compressedspectrum[i] = Math.pow(inspectrum[i], 1.0 / 3.0);
115             }
116             return compressedspectrum;
117         }
118     
119     
120         /**
121          * Returns the next Data object, which is the PLP cepstrum of the input frame. However, it can also be other Data
122          * objects like a EndPointSignal.
123          *
124          * @return the next available Data object, returns null if no Data object is available
125          * @throws DataProcessingException if there is an error reading the Data objects
126          */
127         @Override
128         public Data getData() throws DataProcessingException {
129     
130             Data input = getPredecessor().getData();
131             Data output = input;
132     
133             getTimer().start();
134     
135             if (input != null) {
136                 if (input instanceof DoubleData) {
137                     output = process((DoubleData) input);
138                 }
139             }
140     
141             getTimer().stop();
142     
143             return output;
144         }
145     
146     
147         /**
148          * Process data, creating the PLP cepstrum from an input audio frame.
149          *
150          * @param input a PLP Spectrum frame
151          * @return a PLP Data frame
152          * @throws IllegalArgumentException
153          */
154         private Data process(DoubleData input) throws IllegalArgumentException {
155     
156             double[] plpspectrum = input.getValues();
157     
158             if (plpspectrum.length != numberPLPFilters) {
159                 throw new IllegalArgumentException
160                         ("PLPSpectrum size is incorrect: plpspectrum.length == " +
161                                 plpspectrum.length + ", numberPLPFilters == " +
162                                 numberPLPFilters);
163             }
164     
165             // power law compress spectrum
166             double[] compressedspectrum = powerLawCompress(plpspectrum);
167     
168             // compute autocorrelation values
169             double[] autocor = applyCosine(compressedspectrum);
170     
171             LinearPredictor LPC = new LinearPredictor(LPCOrder);
172             // Compute LPC Parameters
173             LPC.getARFilter(autocor);
174             // Compute LPC Cepstra
175             double[] cepstrumDouble = LPC.getData(cepstrumSize);
176     
177             DoubleData cepstrum = new DoubleData
178                     (cepstrumDouble, input.getSampleRate(), input.getCollectTime(),
179                             input.getFirstSampleNumber());
180     
181             return cepstrum;
182         }
183     
184     
185         /**
186          * Compute the discrete Cosine transform for the given power spectrum
187          *
188          * @param plpspectrum the PLPSpectrum data
189          * @return autocorrelation computed from PLP spectral values
190          */
191         private double[] applyCosine(double[] plpspectrum) {
192     
193             double[] autocor = new double[LPCOrder + 1];
194             double period = (double) numberPLPFilters;
195             double beta = 0.5f;
196     
197             // apply the idct
198             for (int i = 0; i <= LPCOrder; i++) {
199     
200                 if (numberPLPFilters > 0) {
201                     double[] cosine_i = cosine[i];
202                     int j = 0;
203                     autocor[i] += (beta * plpspectrum[j] * cosine_i[j]);
204     
205                     for (j = 1; j < numberPLPFilters; j++) {
206                         autocor[i] += (plpspectrum[j] * cosine_i[j]);
207                     }
208                     autocor[i] /= period;
209                 }
210             }
211     
212             return autocor;
213         }
