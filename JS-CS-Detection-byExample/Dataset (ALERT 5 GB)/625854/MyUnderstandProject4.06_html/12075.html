19     /**
20      * Implements a buffer that contains NGrams of model's MAX order. 
21      * It assumes that the first two bytes of each n-gram entry is the 
22      * ID of the n-gram.
23      */
24     
25     class NMaxGramBuffer extends NGramBuffer {
26     
27         /**
28          * Constructs a NMaxGramBuffer object with the given byte[].
29          *
30          * @param buffer       the byte[] with NGrams
31          * @param numberNGrams the number of N-gram
32          * @param bigEndian	   the buffer's endianness
33          * @param is32bits     whether the buffer is 16 or 32 bits
34          * @param n	           the buffer's order
35          * @param firstCurrentNGramEntry the first Current NGram Entry
36         */
37         public NMaxGramBuffer(byte[] buffer, int numberNGrams, boolean bigEndian, boolean is32bits, int n, int firstCurrentNGramEntry) {
38             super(buffer, numberNGrams, bigEndian, is32bits, n, firstCurrentNGramEntry);
39         }
40     
41     
42         /**
43          * Returns the NGramProbability of the nth follower.
44          *
45          * @param nthFollower which follower
46          * @return the NGramProbability of the nth follower
47          */
48         public int getProbabilityID(int nthFollower) {
49         	int nthPosition = 0;
50         	
51         	nthPosition = nthFollower * LargeNGramModel.BYTES_PER_NMAXGRAM * ((is32bits()) ? 4 : 2);
52         	setPosition(nthPosition + ((is32bits()) ? 4 : 2)); // to skip the word ID
53         	
54             return readBytesAsInt();
55         }
56         
57         
58         /**
59          * Finds the NGram probabilities for the given nth word in a NGram.
60          *
61          * @param nthWordID the ID of the nth word
62          * @return the NGramProbability of the given nth word
63          */
64         public NGramProbability findNGram(int nthWordID) {
65     
66             int mid, start = 0, end = getNumberNGrams();
67             NGramProbability ngram = null;
68     
69             while ((end - start) > 0) {
70                 mid = (start + end) / 2;
71                 int midWordID = getWordID(mid);
72                 if (midWordID < nthWordID) {
73                     start = mid + 1;
74                 } else if (midWordID > nthWordID) {
75                     end = mid;
76                 } else {
77                     ngram = getNGramProbability(mid);
78                     break;
79                 }
80             }
81     
82             return ngram;
83         }
84         
85     
86         /**
87          * Returns the NGramProbability of the nth follower.
88          *
89          * @param nthFollower which follower
90          * @return the NGramProbability of the nth follower
91          */
92         public NGramProbability getNGramProbability(int nthFollower) {
93         	int nthPosition = 0, wordID = 0, probID = 0, backoffID = 0, firstNGram = 0;
94     
95         	nthPosition = nthFollower * LargeNGramModel.BYTES_PER_NMAXGRAM * ((is32bits()) ? 4 : 2);
96     
97             setPosition(nthPosition);
98             
99             wordID = readBytesAsInt();
100             probID = readBytesAsInt();
101                 
102             return (new NGramProbability(nthFollower, wordID, probID, backoffID, firstNGram));
103         }
