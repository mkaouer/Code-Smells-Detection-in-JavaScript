27     package javax.speech.synthesis;
28     
29     import java.util.Vector;
30     
31     import javax.speech.EngineEvent;
32     
33     //Comp. 2.0.6
34     
35     public class SynthesizerEvent extends EngineEvent {
36         public static final int QUEUE_EMPTIED = 0x3000800;
37     
38         public static final int QUEUE_UPDATED = 0x3001000;
39     
40         public static final int SYNTHESIZER_BUFFER_UNFILLED = 0x3002000;
41     
42         public static final int SYNTHESIZER_BUFFER_READY = 0x50348032;
43     
44         public static final int DEFAULT_MASK = EngineEvent.DEFAULT_MASK | QUEUE_EMPTIED
45                 | SYNTHESIZER_BUFFER_UNFILLED | SYNTHESIZER_BUFFER_READY;
46     
47         private boolean topOfQueueChanged;
48     
49         public SynthesizerEvent(Synthesizer source, int id, long oldEngineState,
50                 long newEngineState, Throwable problem, boolean topOfQueueChanged)
51             throws IllegalArgumentException {
52             super(source, id, oldEngineState, newEngineState, problem);
53             if ((id != ENGINE_ERROR) && (problem != null)) {
54                 throw new IllegalArgumentException(
55                         "A problem can only be specified for ENGINE_ERROR");
56             }
57             if (topOfQueueChanged && (id != QUEUE_UPDATED)
58                     && (id != QUEUE_EMPTIED)) {
59                 throw new IllegalArgumentException(
60                         "topOfQueueChanged can only be set for the events" +
61                         " QUEUE_UPDATED and QUEUE_EMPTIED");
62             }
63             this.topOfQueueChanged = topOfQueueChanged;
64         }
65     
66         public boolean isTopOfQueueChanged() {
67             final int id = getId();
68             if (id == QUEUE_UPDATED) {
69                 return topOfQueueChanged;
70             }
71     
72             return false;
73         }
74     
75         /**
76          * {@inheritDoc}
77          */
78         protected void id2String(StringBuffer str) {
79             maybeAddId(str, QUEUE_EMPTIED, "QUEUE_EMPTIED");
80             maybeAddId(str, QUEUE_UPDATED, "QUEUE_UPDATED");
81             maybeAddId(str, SYNTHESIZER_BUFFER_READY, 
82                     "SYNTHESIZER_BUFFER_READY");
83             maybeAddId(str, SYNTHESIZER_BUFFER_UNFILLED, 
84                 "SYNTHESIZER_BUFFER_UNFILLED");
85             super.id2String(str);
86         }
87     
88         /**
89          * {@inheritDoc}
90          */
91         protected Vector getParameters() {
92             final Vector parameters = super.getParameters();
93     
94             final Boolean topOfQueueChangedObject = new Boolean(topOfQueueChanged);
95             parameters.addElement(topOfQueueChangedObject);
96     
97             return parameters;
98         }
99     }
