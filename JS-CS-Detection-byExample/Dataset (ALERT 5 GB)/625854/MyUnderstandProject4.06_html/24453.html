26     /**
27      * Test cases for {@link BaseSynthesizerProperties}.
28      * @author Dirk Schnelle-Walka
29      *
30      */
31     public final class BaseSynthesizerPropertiesTest extends TestCase
32         implements EnginePropertyChangeRequestListener, EnginePropertyListener {
33         /** The test object. */
34         private BaseSynthesizerProperties props;
35     
36         private String name;
37         private Object oldValue;
38         private Object newValue;
39         private EnginePropertyEvent propertyEvent;
40     
41         /**
42          * Setup the test environment.
43          * @throws java.lang.Exception
44          *         setup failed
45          */
46         public void setUp() throws Exception {
47             final Synthesizer synthesizer = new DummySynthesizer();
48             props = new BaseSynthesizerProperties(synthesizer);
49             props.addEnginePropertyChangeRequestListener(this);
50             props.addEnginePropertyListener(this);
51         }
52     
53         /**
54          * Test method for {@link org.jvoicexml.jsapi2.synthesis.BaseSynthesizerProperties#setPitch(int)}.
55          * @exception Exception
56          *            test failed
57          */
58         public void testSetPitch() throws Exception {
59             final int pitch = props.getPitch();
60             final int val1 = 240;
61             props.setPitch(val1);
62             Assert.assertEquals(pitch, props.getPitch());
63             props.commitPropertyChange(name, oldValue, newValue);
64             synchronized (props) {
65                 props.wait(300);
66             }
67             Assert.assertEquals(val1, props.getPitch());
68             Assert.assertNotNull(propertyEvent);
69             Assert.assertEquals(BaseSynthesizerProperties.PITCH,
70                     propertyEvent.getPropertyName());
71             Assert.assertEquals(new Integer(pitch), propertyEvent.getOldValue());
72             Assert.assertEquals(new Integer(val1), propertyEvent.getNewValue());
73         }
74     
75         public void propertyChangeRequest(
76                 final EnginePropertyChangeRequestEvent event) {
77             name = event.getPropertyName();
78             oldValue = event.getOldValue();
79             newValue = event.getNewValue();
80         }
81     
82         public void propertyUpdate(EnginePropertyEvent event) {
83             propertyEvent = event;
84             synchronized (props) {
85                 props.notifyAll();
86             }
87         }
