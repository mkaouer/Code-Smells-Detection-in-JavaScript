35         /**
36          * Constructor for class BuildTranscriptHMM. When called, this method creates graphs for the transcript at several
37          * levels of detail, subsequently mapping from a word graph to a phone graph, to a state graph.
38          *
39          * @param context       this object's context
40          * @param transcript    the transcript to be converted to HMM
41          * @param acousticModel the acoustic model to be used
42          */
43         public BuildTranscriptHMM(String context, Transcript transcript,
44                                   AcousticModel acousticModel) {
45     
46             this.acousticModel = acousticModel;
47             wordGraph = buildWordGraph(transcript);
48             assert wordGraph.validate() : "Word graph not validated";
49             phonemeGraph = buildPhonemeGraph(wordGraph);
50             assert phonemeGraph.validate() : "Phone graph not validated";
51             contextDependentPhoneGraph =
52                     buildContextDependentPhonemeGraph(phonemeGraph);
53             assert contextDependentPhoneGraph.validate() :
54                     "Context dependent graph not validated";
55             hmmGraph = buildHMMGraph(contextDependentPhoneGraph);
56             assert hmmGraph.validate() : "HMM graph not validated";
57             // hmmGraph.printGraph();
58         }

71         /*
72         * Build a word graph from this transcript
73         */
74         private Graph buildWordGraph(Transcript transcript) {
75             Graph graph;
76             Dictionary transcriptDict = transcript.getDictionary();
77             // Make sure the dictionary is a TrainerDictionary before we cast
78             assert
79                     transcriptDict.getClass().getName().endsWith("TrainerDictionary");
80             dictionary = (TrainerDictionary) transcriptDict;
81     
82             transcript.startWordIterator();
83             /* Shouldn't node and edge be part of the graph class? */
84     
85             /* The wordgraph must always begin with the <s> */
86             graph = new Graph();
87             Node initialNode = new Node(NodeType.UTTERANCE_BEGIN);
88             graph.addNode(initialNode);
89             graph.setInitialNode(initialNode);
90     
91             if (transcript.isExact()) {
92                 Node prevNode = initialNode;
93                 for (transcript.startWordIterator();
94                      transcript.hasMoreWords();) {
95                     /* create a new node for the next word */
96                     Node wordNode = new Node(NodeType.WORD,
97                             transcript.nextWord());
98                     /* Link the new node into the graph */
99                     graph.linkNodes(prevNode, wordNode);
100     
101                     prevNode = wordNode;
102                 }
103                 /* All words are done. Just add the </s> */
104                 Node wordNode = new Node(NodeType.UTTERANCE_END);
105                 graph.linkNodes(prevNode, wordNode);
106                 graph.setFinalNode(wordNode);
107             } else {
108                 /* Begin the utterance with a loopy silence */
109                 Node silLoopBack =
110                         new Node(NodeType.SILENCE_WITH_LOOPBACK);
111                 graph.linkNodes(initialNode, silLoopBack);
112     
113                 // Create links with words from the transcript
114                 for (transcript.startWordIterator();
115                      transcript.hasMoreWords();) {
116                     String word = transcript.nextWord();
117                     Pronunciation[] pronunciations =
118                             dictionary.getWord(word).getPronunciations(null);
119                     int numberOfPronunciations = pronunciations.length;
120     
121                     Node[] pronNode = new Node[numberOfPronunciations];
122     
123                     // Create node at the beginning of the word
124                     Node dummyWordBeginNode = new Node(NodeType.DUMMY);
125                     // Allow the silence to be skipped
126                     // TODO: don't link this, for debugging.
127                     // graph.linkNodes(prevNode, dummyWordBeginNode);
128                     // Link the latest silence to the dummy too
129                     graph.linkNodes(silLoopBack, dummyWordBeginNode);
130                     // Add word ending dummy node
131                     Node dummyWordEndNode = new Node(NodeType.DUMMY);
132                     for (int i = 0; i < numberOfPronunciations; i++) {
133                         String wordAlternate
134                                 = pronunciations[i].getWord().getSpelling();
135                         if (i > 0) {
136                             wordAlternate += "(" + i + ')';
137                         }
138                         pronNode[i] = new Node(NodeType.WORD, wordAlternate);
139                         graph.linkNodes(dummyWordBeginNode, pronNode[i]);
140                         graph.linkNodes(pronNode[i], dummyWordEndNode);
141                     }
142     
143                     /* Add silence */
144                     silLoopBack = new
145                             Node(NodeType.SILENCE_WITH_LOOPBACK);
146                     graph.linkNodes(dummyWordEndNode, silLoopBack);
147     
148                 }
149                 Node wordNode = new Node(NodeType.UTTERANCE_END);
150                 // Link previous node, a dummy word end node
151                 // TODO: disable this link for now.
152                 // graph.linkNodes(prevNode, wordNode);
153                 // Link also the previous silence node
154                 graph.linkNodes(silLoopBack, wordNode);
155                 graph.setFinalNode(wordNode);
156             }
157             return graph;
158         }

161         /** Convert word graph to phoneme graph */
162         private Graph buildPhonemeGraph(Graph wordGraph) {
163             Graph phonemeGraph = new Graph();
164             phonemeGraph.copyGraph(wordGraph);
165     
166             for (Node node : phonemeGraph.nodeToArray()) {
167                 if (node.getType().equals(NodeType.WORD)) {
168                     String word = node.getID();
169                     // "false" means graph won't have additional dummy
170                     // nodes surrounding the word
171                     Graph pronunciationGraph = dictionary.getWordGraph(word, false);
172                     phonemeGraph.insertGraph(pronunciationGraph, node);
173                 }
174             }
175             return phonemeGraph;
176         }

194         /**
195          * Convert the phoneme graph to an HMM.
196          *
197          * @param cdGraph a context dependent phoneme graph
198          * @return an HMM graph for a context dependent phoneme graph
199          */
200         public Graph buildHMMGraph(Graph cdGraph) {
201             Graph hmmGraph = new Graph();
202     
203             hmmGraph.copyGraph(cdGraph);
204     
205             for (Node node : hmmGraph.nodeToArray()) {
206                 Unit unit = null;
207                 if (node.getType().equals(NodeType.PHONE)) {
208                     // unit = acousticModel.lookupUnit(node.getID());
209     //		unit = Unit.getUnit(node.getID());
210                 } else if (node.getType().equals(NodeType.SILENCE_WITH_LOOPBACK)) {
211     //		unit = Unit.SILENCE;
212                 } else {
213                     // if it's not a phone, and it's not silence, it's a
214                     // dummy node, and we don't care.
215                     continue;
216                 }
217                 HMM hmm =
218                     acousticModel.lookupNearestHMM(unit, HMMPosition.UNDEFINED, false);
219                 Graph modelGraph = buildModelGraph((SenoneHMM)hmm);
220                 modelGraph.validate();
221                 hmmGraph.insertGraph(modelGraph, node);
222             }
223             return hmmGraph;
224         }

227         /**
228          * Build a graph given an HMM. The graph will not be surrounded by dummy nodes. The number of nodes in the graph is
229          * the number of emitting states in the hmm plus one, to account for a final, non-emitting state.
230          *
231          * @param hmm the HMM
232          * @return the graph
233          */
234         private Graph buildModelGraph(SenoneHMM hmm) {
235             Graph graph = new Graph();
236             Node prevNode;
237             Node stateNode = null;
238             float[][] tmat = hmm.getTransitionMatrix();
239     
240             prevNode = new Node(NodeType.DUMMY);
241             graph.addNode(prevNode);
242             graph.setInitialNode(prevNode);
243     
244             // 'hmm.getOrder() + 1' to account for final, non-emitting state.
245             for (int i = 0; i < hmm.getOrder() + 1; i++) {
246                 /* create a new node for the next hmmState */
247                 stateNode = new Node(NodeType.STATE, hmm.getUnit().getName());
248                 stateNode.setObject(hmm.getState(i));
249                 graph.addNode(stateNode);
250                 /* Link the new node into the graph */
251                 if (i == 0) {
252                     graph.linkNodes(prevNode, stateNode);
253                 }
254                 for (int j = 0; j <= i; j++) {
255                     // System.out.println("TMAT: " + j + " " + i + " " +
256                     // tmat[j][i]);
257                     if (tmat[j][i] != LogMath.getLogZero()) {
258                         // 'j + 1' to account for the initial dummy node
259                         graph.linkNodes(graph.getNode(j + 1), stateNode);
260                     }
261                 }
262                 prevNode = stateNode;
263             }
264             /* All words are done. Just add the final dummy */
265             // stateNode = new Node(NodeType.DUMMY);
266             // graph.linkNodes(prevNode, stateNode);
267             graph.setFinalNode(stateNode);
268     
269             return graph;
270         }
