27     package javax.speech.recognition;
28     
29     //Comp. 2.0.6
30     
31     public class RuleSequence extends RuleComponent {
32         private RuleComponent[] ruleComponents;
33     
34         public RuleSequence(RuleComponent[] ruleComponents)
35             throws IllegalArgumentException {
36             if (ruleComponents == null) {
37                 throw new IllegalArgumentException(
38                         "Rule components must not be null!");
39             }
40             this.ruleComponents = ruleComponents;
41         }
42     
43         public RuleSequence(String[] tokens) throws IllegalArgumentException {
44             if (tokens == null) {
45                 throw new IllegalArgumentException(
46                         "Tokens must not be null!");
47             }
48             ruleComponents = new RuleComponent[tokens.length];
49     
50             for (int i = 0; i < tokens.length; i++) {
51                 final String token = tokens[i];
52                 ruleComponents[i] = new RuleToken(token);
53             }
54         }
55     
56         public RuleComponent[] getRuleComponents() {
57             return ruleComponents;
58         }
59     
60         public String toString() {
61             if (ruleComponents == null) {
62                 return "";
63             }
64     
65             final StringBuffer str = new StringBuffer();
66             for (int i = 0; i < ruleComponents.length; i++) {
67                 final RuleComponent component = ruleComponents[i];
68                 if (component == null) {
69                     str.append(RuleSpecial.NULL.toString());
70                 } else {
71                     str.append(component.toString());
72                 }
73             }
74     
75             return str.toString();
76         }
77     
78     }
