13     package edu.cmu.sphinx.tools.audio;
14     
15     import java.util.Arrays;
16     
17     /** Downsamples an audio clip. */
18     public class Downsampler {
19     
20         /**
21          * Downsamples the given audio clip from the given input sample rate to the given output sample rate.
22          *
23          * @param inSamples the clip to down sample - one sample per element
24          * @param srIn      the sample rate of the given clip
25          * @param srOut     the sample to downsample to
26          * @return an array of downsampled samples
27          */
28         public static short[] downsample(short[] inSamples,
29                                          int srIn,
30                                          int srOut) {
31     
32             /* [[[WDW - this was very back of the napkin for me.  The main
33              * idea was to break a series of input samples into chunks
34              * and have each sample in the chunk contribute to the average
35              * value.  It's brute force, but I didn't have time to think
36              * of anything less or more grand.]]]
37              */
38             short[] temp = new short[inSamples.length];
39             int inSampleIndex = -1;
40             int outSampleIndex = 0;
41             int k = srOut;
42             boolean done = false;
43             while (!done) {
44                 int sum = 0;
45                 for (int i = 0; i < srIn; i++) {
46                     if (k == srOut) {
47                         inSampleIndex++;
48                         if (inSampleIndex >= inSamples.length) {
49                             done = true;
50                             break;
51                         }
52                         k = 0;
53                     }
54                     sum += inSamples[inSampleIndex];
55                     k++;
56                 }
57                 temp[outSampleIndex++] = (short) (sum / srIn);
58             }
59             return Arrays.copyOf(temp, outSampleIndex);
60         }
61     }
