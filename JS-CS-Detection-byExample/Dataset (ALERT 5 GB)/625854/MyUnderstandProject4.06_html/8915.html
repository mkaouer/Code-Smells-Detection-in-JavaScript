129         /**
130          * Returns the word associated with the particular unit
131          *
132          * @return the word associated with this state, or null if there is no word associated with this state.
133          */
134         public Word getAssociatedWord() {
135             Word word = null;
136             SentenceHMMState state = this;
137     
138             while (state != null && !(state instanceof WordState)) {
139                 state = state.getParent();
140             }
141     
142             if (state != null) {
143                 WordState wordState = (WordState) state;
144                 word = wordState.getWord();
145             }
146             return word;
147         }

295          * Connects the arc to this sentence hmm, but don't affect the predecessor relation ship
296          *
297          * @param arc the arc to the next state
298          */
299         private void rawConnect(SentenceHMMStateArc arc) {
300             SentenceHMMState state = (SentenceHMMState) arc.getState();

414         public String toString() {
415             if (cachedName == null) {
416                 StringBuilder sb = new StringBuilder();
417                 if (isEmitting()) {
418                     sb.append('*');
419                 }
420                 sb.append(getName());
421     
422                 // String base = (isEmitting() ? "*" : "") + getName()
423                 //       + getWhich() + (isFinal() ? "!" : "");
424     
425                 if (parent != null) {
426                     sb.append('_');
427                     sb.append(parent);
428                 }
429     
430                 if (isFinal()) {
431                     sb.append('!');
432                 }
433                 cachedName = sb.toString();
434             }
435             return cachedName;

543         /**
544          * Visit all of the states starting at start with the given vistor
545          *
546          * @param visitor the state visitor
547          * @param start   the place to start the search
548          * @param sorted  if true, states are sorted before visited
549          * @return true if the visiting was terminated before all nodes were visited
550          */
551         public static boolean visitStates(SentenceHMMStateVisitor visitor,
552                                           SentenceHMMState start, boolean sorted) {
553             Set<SentenceHMMState> states = collectStates(start);
554     
555             if (sorted) {
556                 // sort the states by stateNumber
557     
558                 TreeSet<SentenceHMMState> sortedStates = new TreeSet<SentenceHMMState>(new Comparator<SentenceHMMState>() {
559     
560                     @Override
561                     public int compare(SentenceHMMState o1, SentenceHMMState o2) {
562                         SentenceHMMState so1 = o1;
563                         SentenceHMMState so2 = o2;
564                         return so1.stateNumber - so2.stateNumber;
565                     }
566                 });
567     
568                 sortedStates.addAll(states);
569                 states = sortedStates;
570             }
571     
572             for (SentenceHMMState state : states) {
573                 if (visitor.visit(state)) {
574                     return true;
575                 }
576             }
577             return false;
578         }

619         /**
620          * Collect all states starting from the given start state
621          *
622          * @param start the state to start the search from
623          * @return the set of collected state
624          */
625         public static Set<SentenceHMMState> collectStates(SentenceHMMState start) {
626             Set<SentenceHMMState> visitedStates = new HashSet<SentenceHMMState>();
627             List<SentenceHMMState> queue = new LinkedList<SentenceHMMState>();
628     
629             queue.add(start);
630     
631             while (!queue.isEmpty()) {
632                 SentenceHMMState state = queue.remove(0);
633     
634                 visitedStates.add(state);
635                 SearchStateArc[] successors = state.getSuccessors();
636                 for (SearchStateArc arc : successors) {
637                     SentenceHMMState nextState = (SentenceHMMState)arc.getState();
638                     if (!visitedStates.contains(nextState) && !queue.contains(nextState)) {
639                         queue.add(nextState);
640                     }
641                 }
642             }
643             return visitedStates;
644         }
