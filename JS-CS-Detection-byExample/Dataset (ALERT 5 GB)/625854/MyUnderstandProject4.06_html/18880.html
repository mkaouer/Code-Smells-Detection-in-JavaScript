227         /**
228          * Resolve a simple or qualified rule name as a full rule name.
229          * 
230          * @param ruleName
231          *            the name of the rule.
232          */
233         public JSGFRuleName resolve(JSGFRuleName ruleName) throws JSGFGrammarException {
234             // System.out.println ("Resolving " + ruleName);
235             JSGFRuleName rn = new JSGFRuleName(ruleName.getRuleName());
236     
237             String simpleName = rn.getSimpleRuleName();
238             String grammarName = rn.getSimpleGrammarName();
239             String packageName = rn.getPackageName();
240             String fullGrammarName = rn.getFullGrammarName();
241     
242             // Check for badly formed RuleName
243             if (packageName != null && grammarName == null) {
244                 throw new JSGFGrammarException("Error: badly formed rulename " + rn);
245             }
246     
247             if (ruleName.getSimpleRuleName().equals("NULL")) {
248                 return JSGFRuleName.NULL;
249             }
250     
251             if (ruleName.getSimpleRuleName().equals("VOID")) {
252                 return JSGFRuleName.VOID;
253             }
254     
255             // Check simple case: a local rule reference
256             if (fullGrammarName == null && this.getRule(simpleName) != null) {
257                 return new JSGFRuleName(name + '.' + simpleName);
258             }
259     
260             // Check for fully-qualified reference
261             if (fullGrammarName != null) {
262                 JSGFRuleGrammar g = manager.retrieveGrammar(fullGrammarName);
263                 if (g != null) {
264                     if (g.getRule(simpleName) != null) {
265                         // we have a successful resolution
266                         return new JSGFRuleName(fullGrammarName + '.' + simpleName);
267                     }
268                 }
269             }
270     
271             // Collect all matching imports into a list. After trying to
272             // match rn to each import statement the vec will have
273             // size()=0 if rn is unresolvable
274             // size()=1 if rn is properly resolvable
275             // size()>1 if rn is an ambiguous reference
276             List<JSGFRuleName> matches = new ArrayList<JSGFRuleName>();
277     
278             // Get list of imports
279             // Add local grammar to simply the case of checking for
280             // a qualified or fully-qualified local reference.
281             List<JSGFRuleName> imports = new ArrayList<JSGFRuleName>(this.imports);
282             imports.add(new JSGFRuleName(name + ".*"));
283     
284             // Check each import statement for a possible match
285             for (JSGFRuleName importName : imports) {
286                 // TO-DO: update for JSAPI 1.0
287                 String importSimpleName = importName.getSimpleRuleName();
288                 String importGrammarName = importName.getSimpleGrammarName();
289                 String importFullGrammarName = importName.getFullGrammarName();
290     
291                 // Check for badly formed import name
292                 if (importFullGrammarName == null) {
293                     throw new JSGFGrammarException("Error: badly formed import " + ruleName);
294                 }
295     
296                 // Get the imported grammar
297                 JSGFRuleGrammar gref = manager.retrieveGrammar(importFullGrammarName);
298                 if (gref == null) {
299                     System.out.println("Warning: import of unknown grammar " + ruleName + " in " + name);
300                     continue;
301                 }
302     
303                 // If import includes simpleName, test that it really exists
304                 if (!importSimpleName.equals("*") && gref.getRule(importSimpleName) == null) {
305                     System.out.println("Warning: import of undefined rule " + ruleName + " in " + name);
306                     continue;
307                 }
308     
309                 // Check for fully-qualified or qualified reference
310                 if (importFullGrammarName.equals(fullGrammarName) || importGrammarName.equals(fullGrammarName)) {
311                     // Know that either
312                     // import <ipkg.igram.???> matches <pkg.gram.???>
313                     // OR
314                     // import <ipkg.igram.???> matches <gram.???>
315                     // (ipkg may be null)
316     
317                     if (importSimpleName.equals("*")) {
318                         if (gref.getRule(simpleName) != null) {
319                             // import <pkg.gram.*> matches <pkg.gram.rulename>
320                             matches.add(new JSGFRuleName(importFullGrammarName + '.' + simpleName));
321                         }
322                         continue;
323                     } else {
324                         // Now testing
325                         // import <ipkg.igram.iRuleName> against <??.gram.ruleName>
326                         //
327                         if (importSimpleName.equals(simpleName)) {
328                             // import <pkg.gram.rulename> exact match for
329                             // <???.gram.rulename>
330                             matches.add(new JSGFRuleName(importFullGrammarName + '.' + simpleName));
331                         }
332                         continue;
333                     }
334                 }
335     
336                 // If we get here and rulename is qualified or fully-qualified
337                 // then the match failed - try the next import statement
338                 if (fullGrammarName != null) {
339                     continue;
340                 }
341     
342                 // Now test
343                 // import <ipkg.igram.*> against <simpleName>
344     
345                 if (importSimpleName.equals("*")) {
346                     if (gref.getRule(simpleName) != null) {
347                         // import <pkg.gram.*> matches <simpleName>
348                         matches.add(new JSGFRuleName(importFullGrammarName + '.' + simpleName));
349                     }
350                     continue;
351                 }
352     
353                 // Finally test
354                 // import <ipkg.igram.iSimpleName> against <simpleName>
355     
356                 if (importSimpleName.equals(simpleName)) {
357                     matches.add(new JSGFRuleName(importFullGrammarName + '.' + simpleName));
358                     continue;
359                 }
360             }
361     
362             // The return behavior depends upon number of matches
363             switch (matches.size()) {
364             case 0: // Return null if rulename is unresolvable
365                 return null;
366             case 1: // Return successfully
367                 return matches.get(0);
368             default: // Throw exception if ambiguous reference
369                 StringBuilder b = new StringBuilder();
370                 b.append("Warning: ambiguous reference ").append(rn).append(" in ").append(name).append(" to ");
371                 for (JSGFRuleName tmp : matches) {
372                     b.append(tmp).append(" and ");
373                 }
374                 b.setLength(b.length() - 5);
375                 throw new JSGFGrammarException(b.toString());
376             }
377         }

379         /** Resolve and link up all rule references contained in all rules. */
380         public void resolveAllRules() throws JSGFGrammarException {
381             StringBuilder b = new StringBuilder();
382     
383             // First make sure that all imports are resolvable
384             for (JSGFRuleName ruleName : imports) {
385                 String grammarName = ruleName.getFullGrammarName();
386                 JSGFRuleGrammar GI = manager.retrieveGrammar(grammarName);
387                 if (GI == null) {
388                     b.append("Undefined grammar ").append(grammarName).append(" imported in ").append(name).append('\n');
389                 }
390             }
391             if (b.length() > 0) {
392                 throw new JSGFGrammarException(b.toString());
393             }
394     
395             for (JSGFRuleState state : rules.values()) {
396                 resolveRule(state.rule);
397             }
398         }

400         /** Resolve the given rule. */
401         protected void resolveRule(JSGFRule r) throws JSGFGrammarException {
402     
403             if (r instanceof JSGFRuleToken) {
404                 return;
405             }
406     
407             if (r instanceof JSGFRuleAlternatives) {
408                 for (JSGFRule rule : ((JSGFRuleAlternatives) r).getRules()) {
409                     resolveRule(rule);
410                 }
411                 return;
412             }
413     
414             if (r instanceof JSGFRuleSequence) {
415                 for (JSGFRule rule : ((JSGFRuleSequence) r).getRules()) {
416                     resolveRule(rule);
417                 }
418                 return;
419             }
420     
421             if (r instanceof JSGFRuleCount) {
422                 resolveRule(((JSGFRuleCount) r).getRule());
423                 return;
424             }
425     
426             if (r instanceof JSGFRuleTag) {
427                 JSGFRuleTag rt = (JSGFRuleTag) r;
428     
429                 JSGFRule rule = rt.getRule();
430                 String ruleStr = rule.toString();
431     
432                 // add the tag the tag-table
433                 Collection<String> tags = ruleTags.get(ruleStr);
434                 if (tags == null) {
435                     tags = new HashSet<String>();
436                     ruleTags.put(ruleStr, tags);
437                 }
438                 tags.add(rt.getTag());
439     
440                 resolveRule(rule);
441                 return;
442             }
443     
444             if (r instanceof JSGFRuleName) {
445                 JSGFRuleName rn = (JSGFRuleName) r;
446                 JSGFRuleName resolved = resolve(rn);
447     
448                 if (resolved == null) {
449                     throw new JSGFGrammarException("Unresolvable rulename in grammar " + name + ": " + rn);
450                 } else {
451                     // TODO: This forces all rule names to be fully resolved.
452                     // This should be changed.
453                     rn.resolvedRuleName = resolved.getRuleName();
454                     rn.setRuleName(resolved.getRuleName());
455                     return;
456                 }
457             }
458     
459             throw new JSGFGrammarException("Unknown rule type");
460         }

513         /**
514          * Returns a string containing the specification for this grammar.
515          * 
516          * @return specification for this grammar.
517          */
518         public String toString() {
519             StringBuilder sb = new StringBuilder();
520             sb.append("#JSGF V1.0;").append(LINE_SEPARATOR);
521             sb.append(LINE_SEPARATOR);
522             sb.append(formatComment(grammarDocComment));
523             sb.append(LINE_SEPARATOR);
524             sb.append("grammar ").append(name).append(';').append(LINE_SEPARATOR);
525             sb.append(LINE_SEPARATOR);
526             // Set of comment keys (The import such comment belongs to).
527             Set<Object> docComments = importDocComments.keySet();
528             for (int i = 0; i < imports.size(); i++) {
529                 String curImport = '<' + imports.get(i).getRuleName() + '>';
530                 if (docComments.contains(curImport)) {
531                     sb.append(formatComment((String) importDocComments.get(curImport)));
532                     sb.append(LINE_SEPARATOR);
533                     sb.append("import ").append(curImport + ';').append(LINE_SEPARATOR);
534                     sb.append(LINE_SEPARATOR);
535                 }
536             }
537             docComments = ruleDocComments.keySet();
538             for (Map.Entry<String, JSGFRuleState> entry : rules.entrySet()) {
539                 Object rule = entry.getKey();
540                 if ((docComments.size() > 0) && docComments.contains(rule)) {
541                     sb.append(formatComment((String) ruleDocComments.get(rule))).append(LINE_SEPARATOR);
542                 }
543                 JSGFRuleState state = entry.getValue();
544                 if (state.isPublic) {
545                     sb.append("public ");
546                 }
547                 sb.append('<').append(rule).append("> = ").append(state.rule).append(';').append(LINE_SEPARATOR);
548                 sb.append(LINE_SEPARATOR);
549             }
550             return sb.toString();
551         }

553         /**
554          * Expands the given String comment into: A. a multi-line comment if the
555          * provided String contains any newline characters. B. a single-line comment
556          * if comment does not contain any newline characters.
557          * 
558          * @param comment
559          *            The String to expand into a multi or single line comment.
560          * @return If the provided string is not null, the multi or single line
561          *         representation of the provided comment, otherwise an empty string
562          *         ("").
563          */
564         private String formatComment(String comment) {
565             StringBuilder sb = new StringBuilder("");
566             if (comment == null) {
567                 return sb.toString();
568             } else if (java.util.regex.Pattern.compile("[\\n\\r\\f]+").matcher(comment).find()) {
569                 String tokens[] = comment.split('[' + LINE_SEPARATOR + "]+");
570                 sb.append("/**").append(LINE_SEPARATOR);
571                 sb.append("  *").append(tokens[0]).append(LINE_SEPARATOR);
572                 for (int i = 1; i < tokens.length; i++) {
573                     sb.append("  *").append(tokens[i]).append(LINE_SEPARATOR);
574                 }
575                 sb.append("  */");
576                 return sb.toString();
577             } else {
578                 return "//" + comment;
579             }
580         }
