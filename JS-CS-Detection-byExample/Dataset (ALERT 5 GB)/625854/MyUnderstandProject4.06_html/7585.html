13     package edu.cmu.sphinx.decoder.search;
14     
15     import edu.cmu.sphinx.util.LogMath;
16     import edu.cmu.sphinx.util.props.*;
17     
18     /** Creates new active lists. */
19     public abstract class ActiveListFactory implements Configurable {
20     
21     
22         /** The property that defines the name of the logmath to be used by this search manager. */
23         @S4Component(type = LogMath.class)
24         public final static String PROP_LOG_MATH = "logMath";
25     
26         /**
27          * property that sets the desired (or target) size for this active list.  This is sometimes referred to as the beam
28          * size
29          */
30         @S4Integer(defaultValue = -1)
31         public final static String PROP_ABSOLUTE_BEAM_WIDTH = "absoluteBeamWidth";
32     
33         /**
34          * Property that sets the minimum score relative to the maximum score in the list for pruning.  Tokens with a score
35          * less than relativeBeamWidth * maximumScore will be pruned from the list
36          */
37         @S4Double(defaultValue = 1E-80)
38         public final static String PROP_RELATIVE_BEAM_WIDTH = "relativeBeamWidth";
39     
40         /**
41          * Property that indicates whether or not the active list will implement 'strict pruning'.  When strict pruning is
42          * enabled, the active list will not remove tokens from the active list until they have been completely scored.  If
43          * strict pruning is not enabled, tokens can be removed from the active list based upon their entry scores. The
44          * default setting is false (disabled).
45          */
46         @S4Boolean(defaultValue = true)
47         public final static String PROP_STRICT_PRUNING = "strictPruning";
48     
49         protected int absoluteBeamWidth;
50         protected float logRelativeBeamWidth;
51     
52         /**
53          * 
54          * @param absoluteBeamWidth
55          * @param relativeBeamWidth
56          * @param logMath
57          */
58         public ActiveListFactory( int absoluteBeamWidth,double relativeBeamWidth, LogMath logMath ){
59             this.absoluteBeamWidth = absoluteBeamWidth;
60             this.logRelativeBeamWidth = logMath.linearToLog(relativeBeamWidth);      
61         }
62     
63         public ActiveListFactory() {
64         }
65     
66     
67         @Override
68         public void newProperties(PropertySheet ps) throws PropertyException {
69             absoluteBeamWidth = ps.getInt(PROP_ABSOLUTE_BEAM_WIDTH);
70             double relativeBeamWidth = ps.getDouble(PROP_RELATIVE_BEAM_WIDTH);
71     
72             LogMath logMath = (LogMath) ps.getComponent(PROP_LOG_MATH);
73             logRelativeBeamWidth = logMath.linearToLog(relativeBeamWidth);
74         }
75     
76     
77         /**
78          * Creates a new active list of a particular type
79          *
80          * @return the active list
81          */
82         public abstract ActiveList newInstance();
83     }
