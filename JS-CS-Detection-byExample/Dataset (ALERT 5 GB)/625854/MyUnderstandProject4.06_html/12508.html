48         public static List<DoubleData> createFeatVectors(double lengthSec, int sampleRate, long startSample, int featDim, double shiftMs) {
49             int numFrames = (int) Math.ceil((lengthSec * 1000) / shiftMs);
50             List<DoubleData> datas = new ArrayList<DoubleData>(numFrames);
51     
52             long curStartSample = startSample;
53             long shiftSamples = ms2samples((int) shiftMs, sampleRate);
54             for (int i = 0; i < numFrames; i++) {
55                 double[] values = createRandFeatureVector(featDim, null, null);
56                 datas.add(new DoubleData(values, sampleRate, System.currentTimeMillis(), curStartSample));
57     
58                 curStartSample += shiftSamples;
59             }
60     
61             return datas;
62         }

65         public static double[] createRandFeatureVector(int featDim, double[] mean, double[] variance) {
66             if (mean == null) {
67                 mean = new double[featDim];
68             }
69     
70             if (variance == null) {
71                 variance = new double[featDim];
72                 for (int i = 0; i < variance.length; i++) {
73                     variance[i] = 1;
74                 }
75             }
76     
77             assert featDim == mean.length && featDim == variance.length;
78     
79             double[] updBlock = new double[featDim];
80     
81             for (int i = 0; i < updBlock.length; i++) {

83             }
