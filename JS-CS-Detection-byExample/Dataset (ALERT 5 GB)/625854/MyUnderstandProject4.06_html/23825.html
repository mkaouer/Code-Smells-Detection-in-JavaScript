163          * @return Grammar[]

165          */
166         public Grammar[] listGrammars() throws EngineStateException {
167     
168             // Validate current state
169             ensureValidEngineState();
170     
171             // List of all grammars
172             ArrayList<Grammar> allGrammars = new ArrayList<Grammar> ();
173     
174             // Get engine built-in grammars
175             if (recognizer != null) {
176                 Vector builtInGrammars = recognizer.getBuiltInGrammars();
177                 if (builtInGrammars != null) {
178                     allGrammars.addAll(builtInGrammars);
179                 }
180             }
181     
182             // Add local managed grammars
183             allGrammars.addAll(grammars.values());
184     
185             // Return an array with all know grammars
186             return (Grammar[]) allGrammars.toArray(new Grammar[allGrammars.size()]);

426         /**
427          * Checks if the recognizer is in a valid state to perform grammar
428          * operations. If the recognizer is currently allocating resources, this
429          * methods waits until the the resoucres are allocated.
430          * @throws EngineStateException
431          *         invalid engine state
432          */
433         private void ensureValidEngineState() throws EngineStateException {
434             if (recognizer == null) {
435                 return;
436             }
437             //Validate current state
438             if (recognizer.testEngineState(
439                     Recognizer.DEALLOCATED | Recognizer.DEALLOCATING_RESOURCES)) {
440                 throw new EngineStateException(
441                   "Cannot execute GrammarManager operation: invalid engine state: "
442                   + recognizer.stateToString(recognizer.getEngineState()));
443             }
444     
445             //Wait until end of allocating (if it's currently allocating)
446             while (recognizer.testEngineState(Recognizer.ALLOCATING_RESOURCES)) {
447                 try {
448                     recognizer.waitEngineState(Recognizer.ALLOCATED);
449                 } catch (InterruptedException ex) {
450                     throw new EngineStateException(ex.getMessage());
451                 }
452             }
453         }
