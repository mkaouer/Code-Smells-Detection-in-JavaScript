39     public class EngineManager {
40         private static final Vector ENGINE_LIST_FACTORIES;
41     
42         private static SpeechEventExecutor executor;
43     
44         static {
45             ENGINE_LIST_FACTORIES = new Vector();
46     
47             final InputStream input;
48             try {
49                 final Class clazz = Class.forName("javax.speech.EngineManager"); 
50                 input = clazz.getResourceAsStream( "/speech.properties");
51             } catch (ClassNotFoundException e) {
52                 throw new IllegalArgumentException(e.getMessage());
53             }
54             if (input != null) {
55                 final Properties props = new Properties();
56                 try {
57                     props.load(input);
58     
59                     //Close input
60                     input.close();
61                 } catch (IOException e) {
62                     // Ignore.
63                 }
64                 
65                 final Enumeration keys = props.keys();
66                 while (keys.hasMoreElements()) {
67                     final String key = (String) keys.nextElement();
68                     final String className = (String) props.get(key);
69                     try {
70                         registerEngineListFactory(className);
71                     } catch (IllegalArgumentException e) {
72                         // Ignore.
73                     } catch (EngineException e) {
74                         // Ignore.
75                     } catch (SecurityException e) {
76                         // Ignore.
77                     }
78                 }
79             }
80         }

82         public static EngineList availableEngines(EngineMode require) {
83             final Vector modes = new Vector();
84     
85             final Enumeration enumeration = ENGINE_LIST_FACTORIES.elements();
86             while (enumeration.hasMoreElements()) {
87                 final EngineListFactory factory = (EngineListFactory) enumeration
88                         .nextElement();
89                 EngineList list = factory.createEngineList(require);
90                 if (list != null) {
91                     final Enumeration currentModes = list.elements();
92                     while (currentModes.hasMoreElements()) {
93                         final EngineMode mode = (EngineMode) currentModes
94                                 .nextElement();
95                         modes.addElement(mode);
96                     }
97                 }
98             }
99     
100             EngineMode[] foundModes = new EngineMode[modes.size()];
101             modes.copyInto(foundModes);
102     
103             return new EngineList(foundModes);
104         }

106         public static Engine createEngine(EngineMode require)
107                 throws IllegalArgumentException, EngineException {
108             if (require == null) {
109                 throw new IllegalArgumentException(
110                         "An engine mode must be specified to create an engine!");
111             }
112     
113             final SpeechLocale defaultLocale = SpeechLocale.getDefault();
114             // TODO Evaluate the default Locale
115             final EngineList list = availableEngines(require);
116     
117             final Enumeration enumeration = list.elements();
118             EngineFactory preferredFactory = null;
119             Boolean preferredFactoryRunning = null;
120             while (enumeration.hasMoreElements()) {
121                 final EngineMode mode = (EngineMode) enumeration.nextElement();
122                 if (mode instanceof EngineFactory) {
123                     final EngineFactory factory = (EngineFactory) mode;
124                     if (preferredFactory == null) {
125                         preferredFactory = factory;
126                         preferredFactoryRunning = mode.getRunning();
127                     }
128     
129                     final Boolean currentFactoryRunning = mode.getRunning();
130                     final Boolean trueVal = new Boolean(true);
131                     if (trueVal.equals(currentFactoryRunning)) {
132                         if (!trueVal.equals(preferredFactoryRunning)) {
133                             preferredFactory = factory;
134                         }
135                     }
136                 }
137             }
138     
139             if (preferredFactory == null) {
140                 return null;
141             }
142     
143             return preferredFactory.createEngine();
144         }

159         public static void registerEngineListFactory(String className)
160                 throws IllegalArgumentException, EngineException,
161                     SecurityException {
162             final Class clazz;
163             try {
164                 clazz = Class.forName(className);
165             } catch (ClassNotFoundException e) {
166                 throw new IllegalArgumentException("'" + className
167                         + "' cannot be loaded!");
168             }
169     
170             final EngineListFactory engineListFactory;
171             try {
172                 engineListFactory = (EngineListFactory) clazz.newInstance();
173             } catch (InstantiationException e) {
174                 throw new IllegalArgumentException("'" + className
175                         + "' cannot be created!");
176             } catch (IllegalAccessException e) {
177                 throw new SecurityException("'" + className
178                         + "' cannot be created!");
179             } catch (ClassCastException e) {
180                 throw new IllegalArgumentException("'" + className
181                         + "' does not implement EngineListFactory!");
182             }
183     
184             if (!(engineListFactory instanceof EngineListFactory)) {
185                 throw new IllegalArgumentException("'" + className
186                         + "' does not implement EngineListFactory");
187             }
188     
189             final Enumeration enumeration = ENGINE_LIST_FACTORIES.elements();
190             while (enumeration.hasMoreElements()) {
191                 final Object current = enumeration.nextElement();
192                 final Class currentClass = current.getClass();
193                 final String currentName = currentClass.getName();
194                 if (className.equals(currentName)) {
195                     return;
196                 }
197             }
198     
199             ENGINE_LIST_FACTORIES.addElement(engineListFactory);
200         }
