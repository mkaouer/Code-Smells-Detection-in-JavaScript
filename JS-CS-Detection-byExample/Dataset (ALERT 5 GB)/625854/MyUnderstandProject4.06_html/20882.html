27     public class BatchForcedAlignerGrammar extends ForcedAlignerGrammar implements GrammarInterface {

29         /** Property that defines the reference file containing the transcripts used to create the froced align grammar */

31         public final static String PROP_REF_FILE = "refFile";

33         protected String refFile;
34         protected final Map<String, GrammarNode> grammars = new HashMap<String, GrammarNode>();
35         protected String currentUttName = "";
36     
37         public BatchForcedAlignerGrammar(String refFile, boolean showGrammar, boolean optimizeGrammar, boolean addSilenceWords,
38                 boolean addFillerWords, Dictionary dictionary) {
39             super(showGrammar, optimizeGrammar, addSilenceWords, addFillerWords, dictionary);
40             this.refFile = refFile;
41         }
42         
43         public BatchForcedAlignerGrammar () {
44         }

46         /*
47         * (non-Javadoc)
48         *
49         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
50         */
51         @Override
52         public void newProperties(PropertySheet ps) throws PropertyException {
53             super.newProperties(ps);
54     
55             refFile = ps.getString(PROP_REF_FILE);
56         }

59         @Override
60         protected GrammarNode createGrammar() {
61             // TODO: FlatLinguist requires the initial grammar node
62             // to contain a single silence. We'll do that for now,
63             // but once the FlatLinguist is fixed, this should be
64             // returned to its former method of creating an empty
65             // initial grammar node
66             //   initialNode = createGrammarNode(initialID, false);
67     
68             initialNode = null;
69             finalNode = createGrammarNode(true);
70             try {
71                 LineNumberReader in = new LineNumberReader(new FileReader(refFile));
72                 String line;
73                 while (true) {
74                     line = in.readLine();
75     
76                     if (line == null || line.isEmpty())
77                         break;
78     
79                     int uttNameStart = line.indexOf('(') + 1;
80                     int uttNameEnd = line.indexOf(')');
81     
82                     if (uttNameStart < 0 || uttNameStart > uttNameEnd)
83                         continue;
84     
85                     String uttName = line.substring(uttNameStart, uttNameEnd);
86                     String transcript = line.substring(0, uttNameStart - 1).trim();
87     
88                     if (transcript.isEmpty())
89                         continue;
90     
91                     initialNode = createGrammarNode(Dictionary.SILENCE_SPELLING);
92                     createForcedAlignerGrammar(initialNode, finalNode, transcript);
93                     grammars.put(uttName, initialNode);
94                     currentUttName = uttName;
95                 }
96             } catch (FileNotFoundException e) {
97                 throw new Error(e);
98             } catch (IOException e) {
99                 throw new Error(e);
100             } catch (NoSuchMethodException e) {
101                 throw new Error(e);
102             }
103             return initialNode;
104         }
105     
106     
107         @Override
108         public GrammarNode getInitialNode() {
109             return initialNode;
110         }
111     
112     
113         public void setUtterance(String utteranceName) {
114             initialNode = grammars.get(utteranceName);
115             assert initialNode != null;
116         }
