65          * Start or stop this data processor
66          * 
67          * We need to send a DataStartSignal before commencing to send samples. When
68          * we were stopped, send a DataEndSignal and return null for subsequent
69          * calls to getData().
70          * 
71          * This way, we can stop this processor and have calls to recognize() return

73          */
74         public synchronized void isRunning(boolean running) {
75             if (this.running != running) {
76                 if (running)
77                     sentStarted = false;
78                 if (!running)
79                     sentEnded = false;
80                 this.running = running;
81             }
82         }

87          * @return an Data object that has been processed by this DataProcessor
88          * @throws DataProcessingException
89          *             if a data processor error occurs
90          */
91         public Data getData() throws DataProcessingException {
92     
93             /** @todo AudioFormat is hardcoded */
94             int channels = 1;
95             int sampleSizeInBytes = 2;
96             boolean signed = true;
97             int sampleRate = 16000;
98             int frameSizeInBytes = sampleRate * sampleSizeInBytes * channels * 10
99                     / 1000;
100     
101             // we are not running anymore, but did not sent DataEnd yet
102             if (!running && !sentEnded) {
103                 sentEnded = true;
104                 long duration = (long) (((double) totalSamplesRead
105                         / (double) sampleRate * 1000.0));
106                 if (LOGGER.isLoggable(Level.FINE)) {
107                     LOGGER.fine("Sending end signal");
108                 }
109                 totalSamplesRead = 0;
110                 return new DataEndSignal(duration);
111             }
112     
113             // we are running, but have not sent DataStart yet
114             if (running && !sentStarted) {
115                 sentStarted = true;
116                 if (LOGGER.isLoggable(Level.FINE)) {
117                     LOGGER.fine("Sending start signal");
118                 }
119                 return new DataStartSignal(sampleRate);
120             }
121     
122             // we are not running at all
123             if (!running) {
124                 if (LOGGER.isLoggable(Level.FINE)) {
125                     LOGGER.fine("Sending null");
126                 }
127                 return null;
128             }
129     
130             long collectTime = System.currentTimeMillis();
131             long firstSampleNumber = totalSamplesRead / channels;
132     
133             // Read data
134             byte[] data = new byte[frameSizeInBytes];
135             int numBytesRead = 0;
136             while (numBytesRead == 0) {
137                 try {
138                     numBytesRead = inputStream.read(data);
139                 } catch (IOException e) {
140                     throw new DataProcessingException(e.getMessage());
141                 }
142                 if (numBytesRead == 0) {
143                     try {
144                         Thread.sleep(100);
145                     } catch (InterruptedException e) {
146                         throw new DataProcessingException(e.getMessage());
147                     }
148                 }
149             }
150             if (numBytesRead == -1) {
151                 long duration = (long) (((double) totalSamplesRead
152                         / (double) sampleRate * 1000.0));
153     
154                 return new DataEndSignal(duration);
155             }
156     
157             if (numBytesRead != frameSizeInBytes) {
158                 if (numBytesRead % sampleSizeInBytes != 0) {
159                     // Is it an error?

161                 }
162     
163                 byte[] shrinked = new byte[numBytesRead];
164                 System.arraycopy(data, 0, shrinked, 0, numBytesRead);
165                 data = shrinked;
166             }
167     
168             totalSamplesRead += (numBytesRead / sampleSizeInBytes);
169     
170             // Convert it to double
171             double[] samples = DataUtil.bytesToValues(data, 0, data.length,
172                     sampleSizeInBytes, signed);
173     
174             return new DoubleData(samples, (int) sampleRate, collectTime,
175                     firstSampleNumber);
