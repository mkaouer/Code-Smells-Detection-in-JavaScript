13     package edu.cmu.sphinx.util;
14     
15     import java.io.*;
16     import java.net.InetAddress;
17     import java.nio.channels.FileLock;
18     import java.util.List;
19     
20     /** A simple implementation of the batch manager suitable for single threaded batch processing */
21     public class PooledBatchManager implements BatchManager {
22     
23         private final String batchFile;
24         private final int skip;
25         private File processingFile;
26     
27         private final static File topDir = new File("tests");
28         private final static File inputDir = new File(topDir, "ToRun");
29         private final static File inProcessDir = new File(topDir, "InProcess");
30         private final static File completedDir = new File(topDir, "Completed");
31         private final static File resultsDir = new File(topDir, "Results");
32         private final static File lockFile = new File(".lock");
33         private FileLock lock;
34         private PrintStream oldOut;
35         private final FileFilter testFileFilter = new TestFileFilter();
36     
37     
38         /**
39          * Creates a pooled batch manager
40          *
41          * @param filename the name of the batch file
42          * @param skip     items to skip between runs
43          */
44     
45         public PooledBatchManager(String filename, int skip) {
46             this.batchFile = filename;
47             this.skip = skip;
48         }
49     
50     
51         /** Starts processing the batch */
52         @Override
53         public void start() throws IOException {
54     
55             // redirect standard out to a file
56             lock();
57             try {
58                 createDirectories();
59                 redirectStdout();
60             } finally {
61                 unlock();
62             }
63         }
64     
65     
66         /**
67          * Gets the next available batch item or null if no more are available
68          *
69          * @return the next available batch item
70          * @throws IOException if an I/O error occurs while reading the next item from the batch file.
71          */
72         @Override
73         public BatchItem getNextItem() throws IOException {
74             lock();
75             try {
76                 // move the last 'in process' file to the 'completed'
77                 // tests section.
78     
79                 if (processingFile != null) {
80                     File completedFile = getCompletedFile(processingFile);
81                     processingFile.renameTo(completedFile);
82                     processingFile = null;
83                 }
84                 File testFile = getNextFile();
85                 if (testFile != null) {
86                     processingFile = getProcessingFile(testFile);
87                     testFile.renameTo(processingFile);
88                     System.out.println("Processing: " + processingFile);
89                     return getBatchItem(processingFile);
90                 } else {
91                     return null;
92                 }
93             } finally {
94                 unlock();
95             }
96         }
97     
98     
99         /** Stops processing the batch */
100         @Override
101         public void stop() throws IOException {
102             lock();
103             try {
104                 closeStdout();
105             } finally {
106                 unlock();
107             }
108         }
109     
110     
111         /**
112          * Returns the name of the file
113          *
114          * @return the filename
115          */
116         @Override
117         public String getFilename() {
118             return batchFile;
119         }
120     
121     
122         /** Creates the test directories as necessary */
123         private void createDirectories() throws IOException {
124             if (!topDir.isDirectory()) {
125                 topDir.mkdir();
126                 inProcessDir.mkdir();
127                 completedDir.mkdir();
128                 resultsDir.mkdir();
129                 createInputDirectory();
130             }
131         }
132     
133     
134         /** Creates the input directory */
135         private void createInputDirectory() throws IOException {
136             inputDir.mkdir();
137             // read in the batch file
138             List<String> list = BatchFile.getLines(batchFile, skip);
139     
140             for (int i = 0; i < list.size(); i++) {
141                 String name = Integer.toString(i + 1);
142                 String line = list.get(i);
143                 createInputFile(inputDir, name, line);
144             }
145         }
146     
147     
148         /**
149          * Creates the individual batch files
150          *
151          * @param dir  the directory to place the input file in
152          * @param name the name of the file
153          * @param line the contents of the file
154          */
155         private void createInputFile(File dir, String name, String line)
156                 throws IOException {
157     
158             File path = new File(dir, name);
159             FileOutputStream fos = new FileOutputStream(path);
160             PrintStream ps = new PrintStream(fos);
161             ps.println(line);
162             ps.close();
163         }
164     
165     
166         /** Redirects standard out to a file in the results directory with a name 'Results_xxx.out' */
167         private void redirectStdout() throws IOException {
168             String myName = getMyName();
169             File resultFile = File.createTempFile(myName, ".out", resultsDir);
170     
171             FileOutputStream fos = new FileOutputStream(resultFile);
172             PrintStream ps = new PrintStream(fos);
173             oldOut = System.out;
174             System.setOut(ps);
175     
176             System.out.println("# These results collected on " + getMyName());
177         }
178     
179     
180         /**
181          * Gets my network name
182          *
183          * @return my network name
184          */
185         private String getMyName() throws IOException {
186             return InetAddress.getLocalHost().getHostName();
187         }
188     
189     
190         /** Close the redirected stdout and restore it to what it was before we redirected it. */
191         private void closeStdout() throws IOException {
192             System.out.close();
193             System.setOut(oldOut);
194         }
195     
196     
197         /** Lock the test suite so we can manipulate the set of tests */
198         private void lock() throws IOException {
199             RandomAccessFile raf = new RandomAccessFile(lockFile, "rw");
200             lock = raf.getChannel().lock();
201         }
202     
203     
204         /** unlock the test suite so we can manipulate the set of tests */
205         private void unlock() throws IOException {
206             lock.release();
207             lock = null;
208     
209         }
210     
211     
212         /**
213          * Given an 'in process' file, generate the corresponding completed file.
214          *
215          * @param file the in process file
216          * @return the completed file
217          */
218         private File getCompletedFile(File file) {
219             return new File(completedDir, file.getName());
220         }
221     
222     
223         /**
224          * Given an 'input' file, generate the corresponding inProcess file.
225          *
226          * @param file the in process file
227          * @return the completed file
228          */
229         private File getProcessingFile(File file) {
230             return new File(inProcessDir, file.getName());
231         }
232     
233     
234         /** returns the next batch item file in the input directory */
235         private File getNextFile() throws IOException {
236             File[] match = inputDir.listFiles(testFileFilter);
237     
238             if (match.length > 0) {
239                 return match[0];
240             }
241             return null;
242         }
243     
244     
245         /**
246          * Given a file parse the contents of the file into a BatchItem
247          *
248          * @param file the file to parse
249          * @return the contents in the form of a batch item
250          */
251         private BatchItem getBatchItem(File file) throws IOException {
252             List<String> list = BatchFile.getLines(file.getPath());
253             if (list.size() != 1) {
254                 throw new IOException("Bad batch file size");
255             }
256             String line = list.get(0);
257             return new BatchItem(BatchFile.getFilename(line),
258                     BatchFile.getReference(line));
259         }
260     }
261     
262     /** Filter that only yields filenames that are integer numbers */
263     class TestFileFilter implements FileFilter {
264     
265         @Override
266         public boolean accept(File pathname) {
267             String name = pathname.getName();
268             try {
269                 Integer.parseInt(name);
270                 return true;
271             } catch (NumberFormatException nfe) {
272                 return false;
273             }
274         }
275     }
