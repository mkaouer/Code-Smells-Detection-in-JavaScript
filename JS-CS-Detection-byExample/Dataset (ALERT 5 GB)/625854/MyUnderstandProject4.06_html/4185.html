50         /**
51          * Creates the grammar from the language model. This Grammar contains one word per grammar node. Each word (and
52          * grammar node) is connected to all other words with the given probability
53          *
54          * @return the initial grammar node
55          */
56         @Override
57         protected GrammarNode createGrammar() throws IOException {
58             if (lattice == null) {
59                 return createGrammarNode("<s>");
60             }
61             
62             lattice.removeFillers();
63             
64             GrammarNode firstNode = null;
65             HashMap<Node, GrammarNode> nodeMap = new HashMap<Node, GrammarNode>();
66             for (Node n : lattice.getNodes()) { 
67                 String word = n.getWord().toString();
68                 GrammarNode node = createGrammarNode(word);
69                 if (n.equals(lattice.getInitialNode()))
70                     firstNode = node;
71                 if (n.equals(lattice.getTerminalNode()))
72                     node.setFinalNode(true);
73                 nodeMap.put(n, node);
74             }
75             if (firstNode == null) {
76                 throw new Error("No lattice start found");
77             }
78     
79             for (Edge e : lattice.getEdges()) {
80                 float logProbability = (float)e.getLMScore();           
81                 GrammarNode prevNode = nodeMap.get(e.getFromNode());
82                 GrammarNode toNode = nodeMap.get(e.getToNode());            
83                 prevNode.add(toNode, logProbability);
84             }
85     
86             return firstNode;
87             
88         }
