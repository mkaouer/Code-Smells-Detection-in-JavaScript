13     package edu.cmu.sphinx.decoder.search;
14     
15     import java.util.Arrays;
16     import edu.cmu.sphinx.decoder.scorer.Scoreable;
17     
18     /**
19      * Partitions a list of tokens according to the token score, used
20      * in {@link PartitionActiveListFactory}. This method is supposed 
21      * to provide O(n) performance so it's more preferable than 
22      */
23     public class Partitioner {
24     
25         /** Max recursion depth **/
26         final private int MAX_DEPTH = 50;
27     
28     
29         /**
30          * Partitions sub-array of tokens around the end token. 
31          * Put all elements less or equal then pivot to the start of the array,
32          * shifting new pivot position
33          *
34          * @param tokens the token array to partition
35          * @param start      the starting index of the subarray
36          * @param end      the pivot and the ending index of the subarray, inclusive
37          * @return the index (after partitioning) of the element around which the array is partitioned
38          */
39         private int endPointPartition(Token[] tokens, int start, int end) {
40             Token pivot = tokens[end];
41             float pivotScore = pivot.getScore();
42                    
43             int i = start;
44             int j = end - 1;
45             
46             while (true) {
47     
48                 while (i < end && tokens[i].getScore() >= pivotScore)
49                     i++;                
50                 while (j > i && tokens[j].getScore() < pivotScore)
51                     j--;
52                 
53                 if (j <= i)
54                     break;
55                 
56                 Token current = tokens[j];
57                 setToken(tokens, j, tokens[i]);
58                 setToken(tokens, i, current);            
59             }
60     
61             setToken(tokens, end, tokens[i]);
62             setToken(tokens, i, pivot);
63             return i;
64         }
65     
66     
67         /**
68          * Partitions sub-array of tokens around the x-th token by selecting the midpoint of the token array as the pivot.
69          * Partially solves issues with slow performance on already sorted arrays.
70          *
71          * @param tokens the token array to partition
72          * @param start      the starting index of the subarray
73          * @param end      the ending index of the subarray, inclusive
74          * @return the index of the element around which the array is partitioned
75          */
76         private int midPointPartition(Token[] tokens, int start, int end) {
77             int middle = (start + end) >>> 1;
78             Token temp = tokens[end];
79             setToken(tokens, end, tokens[middle]);
80             setToken(tokens, middle, temp);
81             return endPointPartition(tokens, start, end);
82         }
83     
84     
85         /**
86          * Partitions the given array of tokens in place, so that the highest scoring n token will be at the beginning of
87          * the array, not in any order.
88          *
89          * @param tokens the array of tokens to partition
90          * @param size   the number of tokens to partition
91          * @param n      the number of tokens in the final partition
92          * @return the index of the last element in the partition
93          */
94         public int partition(Token[] tokens, int size, int n) {
95             if (tokens.length > n) {
96                 return midPointSelect(tokens, 0, size - 1, n, 0);
97             } else {
98                 return findBest(tokens, size);
99             }
100         }
101     
102         /**
103          * Simply find the best token and put it in the last slot
104          * 
105          * @param tokens array of tokens
106          * @param size the number of tokens to partition
107          * @return index of the best token
108          */
109         private int findBest(Token[] tokens, int size) {
110             int r = -1;
111             float lowestScore = Float.MAX_VALUE;
112             for (int i = 0; i < tokens.length; i++) {
113                 float currentScore = tokens[i].getScore();
114                 if (currentScore <= lowestScore) {
115                     lowestScore = currentScore;
116                     r = i; // "r" is the returned index
117                 }
118             }
119     
120             // exchange tokens[r] <=> last token,
121             // where tokens[r] has the lowest score
122             int last = size - 1;
123             if (last >= 0) {
124                 Token lastToken = tokens[last];
125                 setToken(tokens, last, tokens[r]);
126                 setToken(tokens, r, lastToken);
127             }
128     
129             // return the last index
130             return last;
131         }
132     
133     
134         private void setToken(Token[] list, int index, Token token) {
135             list[index] = token;
136             token.setLocation(index);
137         }
138     
139         /**
140          * Selects the token with the ith largest token score.
141          *
142          * @param tokens       the token array to partition
143          * @param start        the starting index of the subarray
144          * @param end          the ending index of the subarray, inclusive
145          * @param targetSize   target size of the partition
146          * @param depth        recursion depth to avoid stack overflow and fall back to simple partition.
147          * @return the index of the token with the ith largest score
148          */
149         private int midPointSelect(Token[] tokens, int start, int end, int targetSize, int depth) {
150             if (depth > MAX_DEPTH) {
151                 return simplePointSelect (tokens, start, end, targetSize);
152             }
153             if (start == end) {
154                 return start;
155             }
156             int partitionToken = midPointPartition(tokens, start, end);
157             int newSize = partitionToken - start + 1;
158             if (targetSize == newSize) {
159                 return partitionToken;
160             } else if (targetSize < newSize) {
161                 return midPointSelect(tokens, start, partitionToken - 1, targetSize, depth + 1);
162             } else {
163                 return midPointSelect(tokens, partitionToken + 1, end, targetSize - newSize, depth + 1);
164             }
165         }
166         
167         /**
168          * Fallback method to get the partition
169          *
170          * @param tokens       the token array to partition
171          * @param start        the starting index of the subarray
172          * @param end          the ending index of the subarray, inclusive
173          * @param targetSize   target size of the partition
174          * @return the index of the token with the ith largest score
175          */
176         private int simplePointSelect(Token[] tokens, int start, int end, int targetSize) {
177             System.out.println ("Falling back to non-recursive partition");
178             Arrays.sort(tokens, start, end + 1, Scoreable.COMPARATOR);
179             return start + targetSize - 1;
180         }
181         
182     }
