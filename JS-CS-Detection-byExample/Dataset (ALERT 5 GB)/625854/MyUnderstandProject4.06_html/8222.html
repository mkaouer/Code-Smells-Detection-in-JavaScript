9     package edu.cmu.sphinx.demo.jsapi.tags;
10     
11     import com.sun.speech.engine.recognition.BaseRecognizer;
12     import edu.cmu.sphinx.tools.tags.ObjectTagsParser;
13     
14     import javax.speech.recognition.RuleGrammar;
15     import javax.speech.recognition.RuleParse;
16     
17     /**
18      * A simple application that uses the embedded object method to
19      * determine the actions to perform when a user says something.
20      * This uses the LiveConnectDemo.gram JSGF grammar.
21      */
22     public class LiveConnectDemo {
23         static final String[] utterances = {
24             "I want a pizza with mushrooms and onions",
25             "Mushroom pizza",
26             "Sausage and pepperoni pizza",
27             "I would like a pizza",
28             "I want a cheese and mushroom pizza with onions",
29             "I would like a burger",
30             "I would like a burger with pickles onions lettuce and cheese",
31             "I would like a burger with special sauce lettuce and cheese",
32             "I want a pizza with pepperoni and cheese",
33             "Cheeseburger with onions",
34         };
35         
36         /**
37          * The ActionTags parser.  This one is a subclass of the ActionTagsParser
38          * and allows you to embed object instances within the tags in the
39          * Grammar.
40          */
41         ObjectTagsParser parser;
42     
43         /**
44          * Method to get an instance of an ObjectTagsParser.
45          */
46         private ObjectTagsParser getTagsParser() {
47             if (parser == null) {
48                 parser = new ObjectTagsParser();
49                 parser.put("appObj", this);
50             }
51             return parser;
52         }
53         
54         /**
55          * Create a new LiveConnectDemo.
56          */
57         public LiveConnectDemo() {
58         }
59     
60         public void doTest() throws Exception {
61             BaseRecognizer recognizer = new BaseRecognizer();
62             recognizer.allocate();
63             
64             RuleGrammar grammar = recognizer.loadJSGF(
65                 null, "edu.cmu.sphinx.demo.jsapi.tags.LiveConnectDemo");
66             grammar.setEnabled(true);
67             recognizer.commitChanges();
68     
69             for (String utterance : utterances) {
70                 RuleParse p = grammar.parse(utterance, null);
71     
72                 if (p == null) {
73                     System.out.println("ILLEGAL UTTERANCE: " + utterance);
74                     continue;
75                 }
76     
77                 System.out.println();
78                 System.out.println("Utterance: " + utterance);
79     
80                 getTagsParser().parseTags(p);
81             }
82         }
83         
84         /**
85          * Submit the order.
86          */
87         public static void submitOrder(OrderItem item) {
88             System.out.println("    Order: " + item);
89         }
90     
91         /**
92          * Standalone operation.
93          */
94         static public void main(String[] args) {
95             try {
96                 LiveConnectDemo demo = new LiveConnectDemo();
97                 demo.doTest();
98             } catch (Exception e) {
99                 e.printStackTrace();
100             }
101         }
102     }
