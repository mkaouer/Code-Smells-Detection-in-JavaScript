13     package edu.cmu.sphinx.research.parallel;
14     
15     import edu.cmu.sphinx.decoder.search.ActiveList;
16     import edu.cmu.sphinx.frontend.FrontEnd;
17     import edu.cmu.sphinx.linguist.acoustic.AcousticModel;
18     import edu.cmu.sphinx.util.props.*;
19     
20     /**
21      * Implements a feature stream used in parallel decoding. A feature stream describes the type of features that are
22      * generated, so it is represented by the {@link #PROP_FRONT_END front end} that generates the features, and the {@link
23      * #PROP_ACOUSTIC_MODEL acoustic model} that should be used to decode those features.
24      * <p/>
25      * Since the parallel decoder uses information from multiple feature streams for decoding, the {@link #PROP_ETA eta
26      * value} of the feature stream represents the relative weight that is applied for the scores from each feature stream.
27      * <p/>
28      * To allow stream-specific pruning, each feature stream maintains its own {@link edu.cmu.sphinx.decoder.search.ActiveList
29      * active list}, which is where tokens of that particular stream should be stored.
30      */
31     public class FeatureStream implements Configurable {
32     
33         /** Property for the acoustic model of this feature stream. */
34         @S4Component(type = AcousticModel.class)
35         public static final String PROP_ACOUSTIC_MODEL = "acousticModel";
36     
37         /** Property for the front end of this feature stream. */
38         @S4Component(type = FrontEnd.class)
39         public static final String PROP_FRONT_END = "frontEnd";
40     
41         /** Property for the eta value of this feature stream. */
42         @S4Double(defaultValue = 1.0f)
43         public static final String PROP_ETA = "eta";
44     
45         private String name;
46         private AcousticModel model;
47         private FrontEnd frontEnd;
48         private float eta;
49         private ActiveList activeList;
50     
51     
52         /*
53          * (non-Javadoc)
54          *
55          * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
56          */
57         public void newProperties(PropertySheet ps) throws PropertyException {
58             model = (AcousticModel) ps.getComponent(PROP_ACOUSTIC_MODEL);
59             frontEnd = (FrontEnd) ps.getComponent(PROP_FRONT_END);
60             eta = ps.getFloat(PROP_ETA);
61     
62             name = model.getName();
63         }
64     
65     
66         /**
67          * Returns the name of this FeatureStream.
68          *
69          * @return the name of this FeatureStream
70          */
71         public String getName() {
72             return name;
73         }
74     
75     
76         /**
77          * Returns the acoustic model of this feature stream.
78          *
79          * @return the acoustic model of this feature stream
80          */
81         public AcousticModel getAcousticModel() {
82             return model;
83         }
84     
85     
86         /**
87          * Returns the front end of this feature stream.
88          *
89          * @return the front end of this feature stream
90          */
91         public FrontEnd getFrontEnd() {
92             return frontEnd;
93         }
94     
95     
96         /**
97          * Returns the eta value of this FeatureStream.
98          *
99          * @return the eta value of this FeatureStream
100          */
101         public float getEta() {
102             return eta;
103         }
104     
105     
106         /**
107          * Returns the active list
108          *
109          * @return the active list
110          */
111         public ActiveList getActiveList() {
112             return activeList;
113         }
114     
115     
116         /** Frees the acoustic model. */
117         public void freeAcousticModel() {
118             model = null;
119         }
120     
121     
122         /**
123          * Sets the eta value of this FeatureStream
124          *
125          * @param eta the eta value
126          */
127         public void setEta(float eta) {
128             this.eta = eta;
129         }
130     
131     
132         /**
133          * Sets the active list
134          *
135          * @param list the active list
136          */
137         public void setActiveList(ActiveList list) {
138             this.activeList = list;
139         }
140     }
