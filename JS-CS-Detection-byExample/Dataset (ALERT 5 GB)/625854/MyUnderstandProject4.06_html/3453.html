13     package edu.cmu.sphinx.linguist.acoustic.tiedstate.trainer;

15     import edu.cmu.sphinx.util.LogMath;
16     
17     /** Used to accumulate data for updating of models. */
18     class Buffer {
19     
20         private double[] numerator;
21         private double denominator;
22         private boolean wasUsed;
23         // Maybe isLog should be used otherwise: one single, say,
24         // accumulate(), which would be directed according to isLog. But
25         // then again having accumulate() and logAccumulate() makes it
26         // clearer if we're dealing with log scale or not...
27         private boolean isLog;
28         private int id;
29     
30     
31         /**
32          * Creates a new buffer. If the values will be in log, the buffer is initialized to all
33          * <code>LogMath.getLogZero()</code>.
34          *
35          * @param size  the number of elements in this buffer
36          * @param isLog if true, the values in the buffer will be in log
37          */
38         Buffer(int size, boolean isLog, int id) {
39             this.id = id;
40             this.isLog = isLog;
41             wasUsed = false;
42             numerator = new double[size];
43             if (isLog) {
44                 denominator = LogMath.getLogZero();
45                 for (int i = 0; i < size; i++) {
46                     numerator[i] = LogMath.getLogZero();
47                 }
48             }
49         }
50     
51     
52         /**
53          * Accumulates data to this buffer. Data are accumulated to a given numerator buffer and to the denominator buffer.
54          *
55          * @param data  the data to be added
56          * @param entry the numerator entry to be accumulated to
57          */
58         void accumulate(float data, int entry) {
59             // Not needed anymore??
60             assert false;
61             assert numerator != null;
62             assert !isLog;
63             numerator[entry] += data;
64             denominator += data;
65             wasUsed = true;
66         }
67     
68     
69         /**
70          * Accumulates data to this buffer. Data are accumulated to a given numerator buffer and to the denominator buffer.
71          *
72          * @param data    the data to be added
73          * @param entry   the numerator entry to be accumulated to
74          * @param logMath the logMath to use
75          */
76         void logAccumulate(float data, int entry, LogMath logMath) {
77             assert numerator != null;
78             assert isLog;
79             numerator[entry] = logMath.addAsLinear((float) numerator[entry], data);
80             denominator = logMath.addAsLinear((float) denominator, data);
81             wasUsed = true;
82         }
83     
84     
85         /**
86          * Accumulates data to this buffer. Data are accumulated to a given numerator buffer and to the denominator buffer.
87          *
88          * @param numeratorData   the data to be added to the numerator
89          * @param denominatorData the data to be added to the denominator
90          */
91         void accumulate(double[] numeratorData, double denominatorData) {
92             assert numerator != null;
93             assert numeratorData != null;
94             assert numerator.length == numeratorData.length;
95             assert !isLog;
96             for (int i = 0; i < numerator.length; i++) {
97                 numerator[i] += numeratorData[i];
98             }
99             denominator += denominatorData;
100             wasUsed = true;
101         }
102     
103     
104         /**
105          * Accumulates data to this buffer. Data are accumulated to a given numerator buffer and to the denominator buffer.
106          *
107          * @param logNumeratorData   the data to be added to the numerator
108          * @param logDenominatorData the data to be added to the denominator
109          * @param logMath            the LogMath instance to be used
110          */
111         void logAccumulate(float[] logNumeratorData, float logDenominatorData,
112                            LogMath logMath) {
113             assert numerator != null;
114             assert logNumeratorData != null;
115             assert numerator.length == logNumeratorData.length;
116             assert isLog;
117             for (int i = 0; i < numerator.length; i++) {
118                 numerator[i] =
119                         logMath.addAsLinear((float) numerator[i], logNumeratorData[i]);
120             }
121             denominator = logMath.addAsLinear((float) denominator, logDenominatorData);
122             wasUsed = true;
123         }
124     
125     
126         /**
127          * Normalize the buffer. This method divides the numerator by the denominator, storing the result in the numerator,
128          * and setting denominator to 1.
129          */
130         void normalize() {
131             assert !isLog;
132             if (denominator == 0) {
133                 System.out.println("Empty denominator: " + id);
134                 // dump();
135                 // assert false;
136                 wasUsed = false;
137                 return;
138             }
139     
140             double invDenominator = (1.0 / denominator);
141             for (int i = 0; i < numerator.length; i++) {
142                 numerator[i] *= invDenominator;
143             }
144             denominator = 1.0;
145         }
146     
147     
148         /**
149          * Normalize the buffer in log scale. This method divides the numerator by the denominator, storing the result in
150          * the numerator, and setting denominator to log(1) = 0.
151          */
152         void logNormalize() {
153             assert isLog;
154             for (int i = 0; i < numerator.length; i++) {
155                 numerator[i] -= denominator;
156             }
157             denominator = 0.0f;
158         }
159     
160     
161         /**
162          * Normalize the non-zero elements in a buffer in log scale. This method divides the numerator by the denominator,
163          * storing the result in the numerator, and setting denominator to log(1) = 0. A mask is used to tell whether a
164          * component should be normalized (non-zero) or not (zero).
165          *
166          * @param mask a vector containing zero/non-zero values.
167          */
168         void logNormalizeNonZero(float[] mask) {
169             assert isLog;
170             assert mask.length == numerator.length;
171             for (int i = 0; i < numerator.length; i++) {
172                 if (mask[i] != LogMath.getLogZero()) {
173                     numerator[i] -= denominator;
174                 }
175             }
176             denominator = 0.0;
177         }
178     
179     
180         /** Normalize the buffer. The normalization is done so that the summation of elements in the buffer is 1. */
181         void normalizeToSum() {
182             assert !isLog;
183             float den = 0.0f;
184             for (double val : numerator) {
185                 den += val;
186             }
187             float invDenominator = (float) (1.0 / den);
188             for (int i = 0; i < numerator.length; i++) {
189                 numerator[i] *= invDenominator;
190             }
191             denominator = 1.0;
192         }
193     
194     
195         /**
196          * Normalize the buffer in log scale. The normalization is done so that the summation of elements in the buffer is
197          * log(1) = 0. In this, we assume that if an element has a value of zero, it won't be updated.
198          *
199          * @param logMath the logMath to use
200          */
201         void logNormalizeToSum(LogMath logMath) {
202             assert isLog;
203             float logZero = LogMath.getLogZero();
204             float den = logZero;
205             for (double val : numerator) {
206                 if (val != logZero) {
207                     den = logMath.addAsLinear(den, (float)val);
208                 }
209             }
210             for (int i = 0; i < numerator.length; i++) {
211                 if (numerator[i] != logZero) {
212                     numerator[i] -= den;
213                 }
214             }
215             denominator = 0.0;
216         }

219         /**
220          * Floor the buffer.
221          *
222          * @param floor the floor for this buffer
223          * @return if true, the buffer was modified
224          */
225         protected boolean floor(float floor) {
226             assert !isLog;
227             boolean wasModified = false;
228             for (int i = 0; i < numerator.length; i++) {
229                 if (numerator[i] < floor) {
230                     wasModified = true;
231                     numerator[i] = floor;
232                 }
233             }
234             return wasModified;
235         }
236     
237     
238         /**
239          * Floor the buffer in log scale.
240          *
241          * @param logFloor the floor for this buffer, in log scale
242          * @return if true, the buffer was modified
243          */
244         protected boolean logFloor(float logFloor) {
245             assert isLog;
246             boolean wasModified = false;
247             for (int i = 0; i < numerator.length; i++) {
248                 if (numerator[i] < logFloor) {
249                     wasModified = true;
250                     numerator[i] = logFloor;
251                 }
252             }
253             return wasModified;
254         }

257         /**
258          * Retrieves a value from this buffer. Make sure you normalize the buffer first.
259          *
260          * @param entry the index into the buffer
261          * @return the value
262          */
263         protected float getValue(int entry) {
264             return (float) numerator[entry];
265         }
266     
267     
268         /**
269          * Set the entry in this buffer to a value.
270          *
271          * @param entry the index into the buffer
272          * @param value the value
273          */
274         protected void setValue(int entry, float value) {
275             numerator[entry] = value;
276         }
277     
278     
279         /**
280          * Retrieves a vector from this buffer. Make sure you normalize the buffer first.
281          *
282          * @return the value
283          */
284         protected float[] getValues() {
285             float[] returnVector = new float[numerator.length];
286             for (int i = 0; i < numerator.length; i++) {
287                 returnVector[i] = (float) numerator[i];
288             }
289             return returnVector;
290         }
291     
292     
293         /**
294          * Returns whether the buffer was used.
295          *
296          * @return if true, the buffer was used
297          */
298         protected boolean wasUsed() {
299             return wasUsed;
300         }

303         /** Dump info about this buffer. */
304         public void dump() {
305             System.out.println("Denominator= " + denominator);
306             System.out.println("Numerators= ");
307             for (int i = 0; i < numerator.length; i++) {
308                 System.out.println("[" + i + "]= " + numerator[i]);
309             }
310         }
