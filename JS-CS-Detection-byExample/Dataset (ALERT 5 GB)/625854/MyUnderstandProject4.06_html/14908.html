13     package edu.cmu.sphinx.result.test;
14     
15     import static org.junit.Assert.*;
16     
17     import org.junit.Test;
18     
19     import edu.cmu.sphinx.result.Node;
20     import edu.cmu.sphinx.result.Lattice;
21     import edu.cmu.sphinx.util.LogMath;
22     
23     /**
24      * Tests the posterior score computation code. Sets up a simple lattice, and
25      * dumps out the posterior probabilities of each node.
26      */
27     public class PosteriorTest {
28     
29     	@Test
30     	public void testPosterior() {
31     
32     		LogMath logMath = new LogMath(1.0001f, true);
33     
34     		Lattice lattice = new Lattice(logMath);
35     
36     		Node a = lattice.addNode("A", "A", 0, 0);
37     		Node b = lattice.addNode("B", "B", 0, 0);
38     		Node c = lattice.addNode("C", "C", 0, 0);
39     		Node d = lattice.addNode("D", "D", 0, 0);
40     
41     		double acousticAB = 4;
42     		double acousticAC = 6;
43     		double acousticCB = 1;
44     		double acousticBD = 5;
45     		double acousticCD = 2;
46     
47     		lattice.setInitialNode(a);
48     		lattice.setTerminalNode(d);
49     
50     		lattice.addEdge(a, b, logMath.linearToLog(acousticAB), 0);
51     		lattice.addEdge(a, c, logMath.linearToLog(acousticAC), 0);
52     		lattice.addEdge(c, b, logMath.linearToLog(acousticCB), 0);
53     		lattice.addEdge(b, d, logMath.linearToLog(acousticBD), 0);
54     		lattice.addEdge(c, d, logMath.linearToLog(acousticCD), 0);
55     
56     		lattice.computeNodePosteriors(1.0f);
57     		double pathABD = acousticAB * acousticBD;
58     		double pathACBD = acousticAC * acousticCB * acousticBD;
59     		double pathACD = acousticAC * acousticCD;
60     		double allPaths = pathABD + pathACBD + pathACD;
61     
62     		double bPosterior = (pathABD + pathACBD) / allPaths;
63     		double cPosterior = (pathACBD + pathACD) / allPaths;
64     
65     		double delta = 1e-4;
66     		assertEquals (logMath.logToLinear((float) a.getPosterior()), 1.0, delta);
67     		assertEquals (logMath.logToLinear((float) b.getPosterior()), bPosterior, delta);
68     		assertEquals (logMath.logToLinear((float) c.getPosterior()), cPosterior, delta);
69     		assertEquals (logMath.logToLinear((float) d.getPosterior()), 1.0, delta);
70     	}
71     }
