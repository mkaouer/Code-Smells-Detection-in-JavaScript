64          * Registers a new property which type and default value are defined by the given sphinx property.
65          *
66          * @param propName The name of the property to be registered.
67          * @param property The property annotation masked by a proxy.
68          */
69         private void registerProperty(String propName, S4PropWrapper property) {
70             if (property == null || propName == null)
71                 throw new InternalConfigurationException(getInstanceName(), propName, "property or its value is null");
72     
73             if (!registeredProperties.containsKey(propName))
74                 registeredProperties.put(propName, property);
75     
76             if (!propValues.containsKey(propName)) {
77                 propValues.put(propName, null);
78                 rawProps.put(propName, null);
79             }
80         }

83         /** Returns the property names <code>name</code> which is still wrapped into the annotation instance. */
84         public S4PropWrapper getProperty(String name, Class<?> propertyClass) throws PropertyException {
85             if (!propValues.containsKey(name))
86                 throw new InternalConfigurationException(getInstanceName(), name,
87                         "Unknown property '" + name + "' ! Make sure that you've annotated it.");
88     
89             S4PropWrapper s4PropWrapper = registeredProperties.get(name);
90     
91             try {
92                 propertyClass.cast(s4PropWrapper.getAnnotation());
93             } catch (ClassCastException e) {
94                 throw new InternalConfigurationException(e, getInstanceName(), name, name + " is not an annotated sphinx property of '" + getConfigurableClass().getName() + "' !");
95             }
96     
97             return s4PropWrapper;
98         }

101         /**
102          * Gets the value associated with this name
103          *
104          * @param name the name
105          * @return the value
106          */
107         public String getString(String name) throws PropertyException {
108             S4PropWrapper s4PropWrapper = getProperty(name, S4String.class);
109             S4String s4String = ((S4String) s4PropWrapper.getAnnotation());
110     
111             if (propValues.get(name) == null) {
112                 boolean isDefDefined = !s4String.defaultValue().equals(S4String.NOT_DEFINED);
113     
114                 if (s4String.mandatory()) {
115                     if (!isDefDefined)
116                         throw new InternalConfigurationException(getInstanceName(), name, "mandatory property is not set!");
117                 }
118                 propValues.put(name, isDefDefined ? s4String.defaultValue() : null);
119             }
120     
121             String propValue = flattenProp(name);
122     
123             // Check range
124             List<String> range = Arrays.asList(s4String.range());
125             if (!range.isEmpty() && !range.contains(propValue))
126                 throw new InternalConfigurationException(getInstanceName(), name, " is not in range (" + range + ')');
127     
128             return propValue;
129         }

149

156                 } else if (!isDefDefined)

260         /**
261          * Gets a component associated with the given parameter name. First search
262          * the component in property table, then try to get component by name from
263          * the manager, then creates component with default properties.
264          * 
265          * @param name
266          *            the parameter name
267          * @return the component associated with the name
268          * @throws edu.cmu.sphinx.util.props.PropertyException
269          *             if the component does not exist or is of the wrong type.
270          */
271         public Configurable getComponent(String name) throws PropertyException {
272             S4PropWrapper s4PropWrapper = getProperty(name, S4Component.class);
273             Configurable configurable = null;
274     
275             S4Component s4Component = (S4Component) s4PropWrapper.getAnnotation();
276             Class<? extends Configurable> expectedType = s4Component.type();
277     
278             Object propVal = propValues.get(name);
279     
280             if (propVal != null && propVal instanceof Configurable) {
281                 return (Configurable) propVal;
282             }
283     
284             if (propVal != null && propVal instanceof String) {
285                 PropertySheet ps = cm.getPropertySheet(flattenProp(name));
286                 if (ps != null)
287                     configurable = ps.getOwner();
288                 else
289                     throw new InternalConfigurationException(getInstanceName(), name, "component '" + flattenProp(name)
290                             + "' is missing");
291             }
292     
293             if (configurable != null && !expectedType.isInstance(configurable))
294                 throw new InternalConfigurationException(getInstanceName(), name, "mismatch between annotation and component type");
295     
296             if (configurable != null) {
297                 propValues.put(name, configurable);
298                 return configurable;
299             }
300     
301             configurable = getComponentFromAnnotation(name, s4Component);
302     
303             propValues.put(name, configurable);
304             return configurable;
305         }

308         private Configurable getComponentFromAnnotation(String name, S4Component s4Component) {
309             Configurable configurable;
310             Class<? extends Configurable> defClass = s4Component.defaultClass();
311     
312             if (defClass.equals(Configurable.class) && s4Component.mandatory()) {
313                 throw new InternalConfigurationException(getInstanceName(), name, "mandatory property is not set!");
314             }
315     
316             if (Modifier.isAbstract(defClass.getModifiers()) && s4Component.mandatory())
317                 throw new InternalConfigurationException(getInstanceName(), name, defClass.getName() + " is abstract!");
318     
319             // because we're forced to use the default type, make sure that it
320             // is set
321             if (defClass.equals(Configurable.class)) {
322                 if (s4Component.mandatory()) {
323                     throw new InternalConfigurationException(getInstanceName(), name, instanceName
324                             + ": no default class defined for " + name);
325                 } else {
326                     return null;
327                 }
328             }
329     
330             configurable = ConfigurationManager.getInstance(defClass);
331             if (configurable == null) {
332                 throw new InternalConfigurationException(getInstanceName(), name, "instantiation of referenenced configurable failed");
333             }
334             
335             return configurable;
336         }

338         /** Returns the class of of a registered component property without instantiating it. */
339         public Class<? extends Configurable> getComponentClass(String propName) {
340             Class<? extends Configurable> defClass = null;
341     
342             if (propValues.get(propName) != null)
343                 try {
344                     Class<?> objClass = Class.forName((String) propValues.get(propName));
345                     defClass = objClass.asSubclass(Configurable.class);
346                 } catch (ClassNotFoundException e) {
347                     PropertySheet ps = cm.getPropertySheet(flattenProp(propName));
348                     defClass = ps.ownerClass;
349                 }
350             else {
351                 S4Component comAnno = (S4Component) registeredProperties.get(propName).getAnnotation();
352                 defClass = comAnno.defaultClass();
353                 if (comAnno.mandatory())
354                     defClass = null;
355             }
356     
357             return defClass;
358         }

373         /**
374          * Gets a list of components associated with the given parameter name
375          * 
376          * @param name
377          *            the parameter name
378          * @param tclass
379          *            the class of the list elements
380          * @return the component associated with the name
381          * @throws edu.cmu.sphinx.util.props.PropertyException
382          *             if the component does not exist or is of the wrong type.
383          */
384         public <T> List<T> getComponentList(String name, Class<T> tclass)
385                 throws InternalConfigurationException {
386             getProperty(name, S4ComponentList.class);
387     
388             List<?> components = (List<?>) propValues.get(name);
389     
390             assert registeredProperties.get(name).getAnnotation() instanceof S4ComponentList;
391             S4ComponentList annotation = (S4ComponentList) registeredProperties
392                     .get(name).getAnnotation();
393     
394             // no components names are available and no component list was yet
395             // loaded therefore load the default list of components from the 
396             // annotation
397             if (components == null) {
398                 List<Class<? extends Configurable>> defClasses = Arrays
399                         .asList(annotation.defaultList());
400     
401                 // if (annotation.mandatory() && defClasses.isEmpty())
402                 // throw new InternalConfigurationException(getInstanceName(), name,
403                 // "mandatory property is not set!");
404     
405                 List<Configurable> defaultComponents = new ArrayList<Configurable>();
406     
407                 for (Class<? extends Configurable> defClass : defClasses) {
408                     defaultComponents.add(ConfigurationManager.getInstance(defClass));
409                 }
410     
411                 propValues.put(name, defaultComponents);
412             
413             } else if (!components.isEmpty()
414                     && !(components.get(0) instanceof Configurable)) {
415     
416                 List<Configurable> resolvedComponents = new ArrayList<Configurable>();
417     
418                 for (Object componentName : components) {
419                     Configurable configurable = cm.lookup((String) componentName);
420     
421                     if (configurable != null) {
422                         resolvedComponents.add(configurable);
423                     } else if (!annotation.beTolerant()) {
424                         throw new InternalConfigurationException(name,
425                                 (String) componentName, "lookup of list-element '"
426                                         + componentName + "' failed!");
427                     }
428                 }
429     
430                 propValues.put(name, resolvedComponents);
431             }
432     
433             List<?> values = (List<?>) propValues.get(name);
434             ArrayList<T> result = new ArrayList<T>();
435             for (Object obj : values) {
436                 if (tclass.isInstance(obj)) {
437                     result.add(tclass.cast(obj));
438                 } else {
439                     throw new InternalConfigurationException(getInstanceName(),
440                             name, "Not all elements have required type " + tclass + " Found one of type " + obj.getClass());
441                 }
442             }
443             return result;
444         }

490          * Returns the owner of this property sheet. In most cases this will be the configurable instance which was
491          * instrumented by this property sheet.
492          */
493         public synchronized Configurable getOwner() {
494             try {
495     
496                 if (!isInstanciated()) {
497                     // ensure that all mandatory properties are set before instantiating the component
498                     Collection<String> undefProps = getUndefinedMandatoryProps();
499                     if (!undefProps.isEmpty()) {
500                         throw new InternalConfigurationException(getInstanceName(),
501                                 undefProps.toString(), "not all mandatory properties are defined");
502                     }
503     
504                     owner = ownerClass.newInstance();
505                     owner.newProperties(this);
506                 }
507             } catch (IllegalAccessException e) {
508                 throw new InternalConfigurationException(e, getInstanceName(), null, "Can't access class " + ownerClass);
509             } catch (InstantiationException e) {
510                 throw new InternalConfigurationException(e, getInstanceName(), null, "Can't instantiate class " + ownerClass);
511             }
512     
513             return owner;
514         }

517         /**
518          * Returns the set of all component properties which were tagged as mandatory but which are not set (or no default
519          * value is given).
520          */
521         public Collection<String> getUndefinedMandatoryProps() {
522             Collection<String> undefProps = new ArrayList<String>();
523             for (String propName : getRegisteredProperties()) {
524                 Proxy anno = registeredProperties.get(propName).getAnnotation();
525     
526                 boolean isMandatory = false;
527                 if (anno instanceof S4Component) {
528                     isMandatory = ((S4Component) anno).mandatory() && ((S4Component) anno).defaultClass() == null;
529                 } else if (anno instanceof S4String) {
530                     isMandatory = ((S4String) anno).mandatory() && ((S4String) anno).defaultValue().equals(S4String.NOT_DEFINED);
531                 } else if (anno instanceof S4Integer) {
532                     isMandatory = ((S4Integer) anno).mandatory() && ((S4Integer) anno).defaultValue() == S4Integer.NOT_DEFINED;
533                 } else if (anno instanceof S4Double) {
534                     isMandatory = ((S4Double) anno).mandatory() && ((S4Double) anno).defaultValue() == S4Double.NOT_DEFINED;
535                 }
536     
537                 if (isMandatory && !((rawProps.get(propName) != null) || (propValues.get(propName) != null)))
538                     undefProps.add(propName);
539             }
540             return undefProps;
541         }

550         /**
551          * Sets the configurable class of this object.
552          *
553          * @throws RuntimeException if the the <code>Configurable</code> is already instantiated.
554          */
555         void setConfigurableClass(Class<? extends Configurable> confClass) {
556             ownerClass = confClass;
557     
558             // Don't allow changes of the class if the configurable has already been instantiated
559             if (isInstanciated())
560                 throw new RuntimeException("class is already instantiated");
561     
562             // clean up the properties if necessary
563     	// registeredProperties.clear();
564     
565             final Collection<String> classProperties = new HashSet<String>();
566             final Map<Field, Annotation> classProps = parseClass(ownerClass);
567             for (Map.Entry<Field, Annotation> entry : classProps.entrySet()) {
568                 try {
569                     String propertyName = (String)entry.getKey().get(null);
570     
571                     // make sure that there is not already another property with this name
572                     assert !classProperties.contains(propertyName) :
573                             "duplicate property-name for different properties: " + propertyName;
574     
575                     registerProperty(propertyName, new S4PropWrapper((Proxy)entry.getValue()));
576                     classProperties.add(propertyName);
577                 } catch (IllegalAccessException e) {
578                     e.printStackTrace();
579                 }
580             }
581         }

691         public void setComponentList(String name, List<String> valueNames, List<Configurable> value) throws PropertyException {
692             if (!registeredProperties.containsKey(name))
693                 throw new InternalConfigurationException(getInstanceName(), name, '\'' + name +
694                         "' is not a registered component-list");
695     
696             Proxy annotation = registeredProperties.get(name).getAnnotation();
697             if (!(annotation instanceof S4ComponentList))
698                 throw new InternalConfigurationException(getInstanceName(), name, '\'' + name + "' is of type component-list");
699     
700             rawProps.put(name, valueNames);
701             propValues.put(name, value);
702     
703             applyConfigurationChange(name, valueNames, value);
704         }

708             rawProps.put(propName, cmName);
709             propValues.put(propName, value != null ? value : cmName);
710     
711             if (getInstanceName() != null)
712                 cm.fireConfChanged(getInstanceName(), propName);
713     
714             if (owner != null)
715                 owner.newProperties(this);
716         }

753         /** Returns the type of the given property. */
754         public PropertyType getType(String propName) {
755             S4PropWrapper wrapper = registeredProperties.get(propName);
756             if (wrapper == null) {
757                 throw new InternalConfigurationException(getInstanceName(), propName, " is not a valid property of" + getConfigurableClass());
758             }
759     
760             Proxy annotation = wrapper.getAnnotation();
761             if (annotation instanceof S4Component)
762                 return PropertyType.COMPONENT;
763             else if (annotation instanceof S4ComponentList)
764                 return PropertyType.COMPONENT_LIST;
765             else if (annotation instanceof S4Integer)
766                 return PropertyType.INT;
767             else if (annotation instanceof S4Double)
768                 return PropertyType.DOUBLE;
769             else if (annotation instanceof S4Boolean)
770                 return PropertyType.BOOLEAN;
771             else if (annotation instanceof S4String)
772                 return PropertyType.STRING;
773             else
774                 throw new RuntimeException("Unknown property type");
775         }

788         /**
789          * Returns a logger to use for this configurable component. The logger can be configured with the property:
790          * 'logLevel' - The default logLevel value is defined (within the xml configuration file by the global property
791          * 'defaultLogLevel' (which defaults to WARNING).
792          * <p/>
793          * implementation note: the logger became configured within the constructor of the parenting configuration manager.
794          *
795          * @return the logger for this component
796          * @throws edu.cmu.sphinx.util.props.PropertyException
797          *          if an error occurs
798          */
799         public Logger getLogger() {
800             Logger logger;
801     
802             String baseName = ConfigurationManagerUtils.getLogPrefix(cm) + ownerClass.getName();
803             if (instanceName != null) {
804                 logger = Logger.getLogger(baseName + '.' + instanceName);
805             } else
806                 logger = Logger.getLogger(baseName);
807     
808             // if there's a logLevel set for component apply to the logger
809             Object rawLogLevel = rawProps.get(COMP_LOG_LEVEL);
810             if (rawLogLevel != null)
811                 logger.setLevel(rawLogLevel instanceof String ? Level.parse((String) rawLogLevel) : (Level) rawLogLevel);
812     
813             return logger;
814         }

856         @Override
857         protected PropertySheet clone() throws CloneNotSupportedException {
858             PropertySheet ps = (PropertySheet)super.clone();
859     
860             ps.registeredProperties = new HashMap<String, S4PropWrapper>(this.registeredProperties);
861             ps.propValues = new HashMap<String, Object>(this.propValues);
862     
863             ps.rawProps = new HashMap<String, Object>(this.rawProps);
864     
865             // make deep copy of raw-lists
866             for (String regProp : ps.getRegisteredProperties()) {
867                 if (getType(regProp) == PropertyType.COMPONENT_LIST) {
868                     ps.rawProps.put(regProp, ConfigurationManagerUtils.toStringList(rawProps.get(regProp)));
869                     ps.propValues.put(regProp, null);
870                 }
871             }
872     
873             ps.cm = cm;
874             ps.owner = null;
875             ps.instanceName = this.instanceName;
876     
877             return ps;
878         }

882         public boolean validate() {
883             for (String propName : rawProps.keySet()) {
884                 if (propName.equals(ConfigurationManagerUtils.GLOBAL_COMMON_LOGLEVEL))
885                     continue;

891             return true;
892         }

895         /**
896          * use annotation based class parsing to detect the configurable properties of a <code>Configurable</code>-class
897          *
898          * @param configurable of type Class
899          */
900         private static Map<Field, Annotation> parseClass(Class<? extends Configurable> configurable) {
901             Field[] classFields = configurable.getFields();
902     
903             Map<Field, Annotation> s4props = new HashMap<Field, Annotation>();
904             for (Field field : classFields) {
905                 Annotation[] annotations = field.getAnnotations();
906     
907                 for (Annotation annotation : annotations) {
908                     Annotation[] superAnnotations = annotation.annotationType().getAnnotations();
909     
910                     for (Annotation superAnnotation : superAnnotations) {
911                         if (superAnnotation instanceof S4Property) {
912                             int fieldModifiers = field.getModifiers();
913                             assert Modifier.isStatic(fieldModifiers) : "property fields are assumed to be static";
914                             assert Modifier.isPublic(fieldModifiers) : "property fields are assumed to be public";
915                             assert Modifier.isFinal(fieldModifiers) : "property fields are assumed to be final";
916                             assert field.getType().equals(String.class) : "properties fields are assumed to be instances of java.lang.String";
917     
918                             s4props.put(field, annotation);
919                         }
920                     }
921                 }
922             }
923     
924             return s4props;
925         }
