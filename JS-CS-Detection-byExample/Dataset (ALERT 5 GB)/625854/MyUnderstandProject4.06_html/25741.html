27     package javax.speech.recognition;
28     
29     //Comp 2.0.6
30     
31     public class RuleComponent {
32         /**
33          * Checks if the given text is a valid grammar text.
34          * 
35          * @param text
36          *            the text to check.
37          */
38         protected static void checkValidGrammarText(String text) {
39             if ((text == null) || (text.length() == 0)) {
40                 throw new IllegalArgumentException(" No text is not a valid for a grammar ");
41             }
42     
43             final char[] chars = text.toCharArray();
44             // The first symbol must be a character.
45             final char first = chars[0];
46             if (!isLetter(first) && (first != '_')) {
47                 throw new IllegalArgumentException("'" + text
48                         + "' is not a valid grammar text : '"+
49                             first +"' Element is not a valid first symbol");
50             }
51             // Following symbols must be a character or a digit.
52             for (int i = 1; i < chars.length; i++) {
53                 final char ch = chars[i];
54                 if (!isLetter(ch) && !Character.isDigit(ch)
55                         && (ch != '_')) {
56                     throw new IllegalArgumentException("'" + text
57                             + "' is not a valid grammar text : '"+
58                             ch +"' Element is not a valid symbol");
59                 }
60             }
61         }
62     
63         static boolean isLetter(char ch) {
64             return isUpperCase(ch) || isLowerCase(ch);
65         }
66         
67         static boolean isUpperCase(char ch) {
68             return (ch >= 'A') && (ch <= 'Z');
69         }
70     
71         static boolean isLowerCase(char ch) {
72             return (ch >= 'a') && (ch <= 'z');
73         }
74     
75         static boolean isWhitespace(char ch) {
76             switch (ch) {
77             case ' ':
78             case '\t':
79                 return true;
80             default:
81                 return false;
82             }
83         }
84         
85         public String toString() {
86             return null;
87         }
88     }
