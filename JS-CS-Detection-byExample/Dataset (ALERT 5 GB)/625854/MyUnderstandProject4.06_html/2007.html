9     /**
10      * A simple converter which converts <code>DoubleData</code> to <code>FloatData</code> and vv (depending on its
11      * configuration). All remaining <code>Data</code>s will pass this processor unchanged.
12      *
13      * @author Holger Brandl
14      */
15     
16     public class DataConverter extends BaseDataProcessor {
17     
18         public static final String CONVERT_D2F = "d2f";
19         public static final String CONVERT_F2D = "f2d";
20     
21         @S4String(defaultValue = "d2f", range = {CONVERT_D2F, CONVERT_F2D})
22         public static final String PROP_CONVERSION_MODE = "conversionMode";
23         private String convMode;
24     
25         public DataConverter(String convMode) throws PropertyException {
26             initLogger();
27             this.convMode = convMode;
28         }
29     
30         public DataConverter() {
31     
32         }
33     
34         /*
35         * (non-Javadoc)
36         *
37         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
38         */
39         @Override
40         public void newProperties(PropertySheet ps) throws PropertyException {
41             super.newProperties(ps);
42     
43             convMode = ps.getString(PROP_CONVERSION_MODE);
44         }
45     
46     
47         @Override
48         public Data getData() throws DataProcessingException {
49             Data d = getPredecessor().getData();
50     
51             if (d instanceof DoubleData && convMode.equals(CONVERT_D2F)) {
52                 DoubleData dd = (DoubleData) d;
53                 d = new FloatData(MatrixUtils.double2float(dd.getValues()), dd.getSampleRate(),
54                         dd.getCollectTime(), dd.getFirstSampleNumber());
55             } else if (d instanceof FloatData && convMode.equals(CONVERT_F2D)) {
56                 FloatData fd = (FloatData) d;
57                 d = new DoubleData(MatrixUtils.float2double(fd.getValues()), fd.getSampleRate(),
58                         fd.getCollectTime(), fd.getFirstSampleNumber());
59             }
60     
61             return d;
62         }
63     }
