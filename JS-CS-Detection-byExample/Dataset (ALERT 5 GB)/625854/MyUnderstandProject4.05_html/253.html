25     
26     import java.io.File;
27     import java.io.IOException;
28     import java.net.URL;
29     
30     import javax.sound.sampled.AudioInputStream;
31     import javax.sound.sampled.AudioSystem;
32     import javax.sound.sampled.UnsupportedAudioFileException;
33     
34     import org.junit.Test;
35     
36     /**
37      * Compares the lattices generated when the LexTreeLinguist flag 'keepAllTokens'
38      * is turned on/off.
39      */
40     public class LatticeCompTest {
41     
42     	/**
43     	 * Main method for running the LatticeCompTest demo.
44     	 * @throws IOException 
45     	 * @throws UnsupportedAudioFileException 
46     	 */
47     	@Test
48     	public void testLatticeComp() throws UnsupportedAudioFileException, IOException {
49     
50     		URL audioFileURL = new File("src/test/edu/cmu/sphinx/result/test/green.wav").toURI().toURL();
51     		URL configURL = new File("src/test/edu/cmu/sphinx/result/test/config.xml").toURI().toURL();
52     
53     		ConfigurationManager cm = new ConfigurationManager(configURL);
54     
55     		Recognizer recognizer = (Recognizer) cm.lookup("recognizer");
56     
57     		StreamDataSource reader = (StreamDataSource) cm
58     				.lookup("streamDataSource");
59     
60     		AudioInputStream ais = AudioSystem.getAudioInputStream(audioFileURL);
61     
62     		/* set the stream data source to read from the audio file */
63     		reader.setInputStream(ais, audioFileURL.getFile());
64     		
65     		/* allocate the resource necessary for the recognizer */
66     		recognizer.allocate();
67     
68     		/* decode the audio file */
69     		Result result = recognizer.recognize();
70     
71     		/* print out the results */
72     		Lattice lattice = new Lattice(result);
73     		lattice.dumpAISee("logs/lattice.gdl", "lattice");
74     
75     		recognizer.deallocate();
76     		
77     		cm = new ConfigurationManager(configURL);
78     		
79     		ConfigurationManagerUtils.setProperty(cm, "keepAllTokens", "true");
80     		
81     		recognizer = (Recognizer) cm.lookup("recognizer");
82     		recognizer.allocate();
83     		
84     		reader = (StreamDataSource) cm.lookup("streamDataSource");
85     		reader.setInputStream(AudioSystem.getAudioInputStream(audioFileURL),
86     				audioFileURL.getFile());
87     
88     		Result allResult = recognizer.recognize();
89     
90     		Lattice allLattice = new Lattice(allResult);
91     		allLattice.dumpAISee("logs/allLattice.gdl", "All Lattice");
92     
93     		assertTrue(lattice.isEquivalent(allLattice));
94     	}
95     }
