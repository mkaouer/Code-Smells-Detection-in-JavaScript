32     /** A client-side BatchDecoder. It talks to a ServerDeocder to decode a list of files in a batch file. */
33     public class BatchClient implements Configurable {
34     
35         @S4String
36         private static final String BATCH_FILE = "batchFile";
37         private String batchFile;
38     
39         @S4Integer(defaultValue = 0)
40         public static final String SKIP = "skip";
41         private int skip;
42     
43     
44         @S4Component(type = ClientFrontEndImpl.class)
45         private final static String CLIENT = "client";
46         private ClientFrontEndImpl clientFrontEnd;
47     
48     
49         private static DecimalFormat timeFormat = new DecimalFormat("0.00");
50     
51         private Timer decodeTimer;
52         private long cumulativeProcessingTime;
53     
54         private NISTAlign aligner;
55     
56     
57         public void newProperties(PropertySheet ps) throws PropertyException {
58             this.batchFile = ps.getString(BATCH_FILE);
59     
60             skip = ps.getInt(SKIP);
61             clientFrontEnd = (ClientFrontEndImpl) ps.getComponent(CLIENT);
62     
63             decodeTimer = TimerPool.getTimer(this, "BatchClientDecode");
64             aligner = new NISTAlign(true, true);
65         }
66     
67     
68         /**
69          * Decodes the batch of audio files
70          *
71          * @throws DataProcessingException if a data processing error occurs
72          * @throws IOException             if an I/O error occurs
73          */
74         public void decode() throws DataProcessingException, IOException {
75             clientFrontEnd.connect();
76     
77             int curCount = skip;
78             System.out.println("\nBatchDecoder: decoding files in " + batchFile);
79             System.out.println("----------");
80     
81             for (String line : BatchFile.getLines(batchFile)) {
82                 String file = BatchFile.getFilename(line);
83                 String reference = BatchFile.getReference(line);
84     
85                 if (++curCount >= skip) {
86                     curCount = 0;
87                     decodeFile(file, reference);
88                 }
89             }
90     
91             System.out.println("\nBatchDecoder: All files decoded\n");
92             TimerPool.dumpAll();
93     
94             clientFrontEnd.close();
95         }
96     
97     
98         /**
99          * Decodes the given file.
100          *
101          * @param file      the file to decode
102          * @param reference the reference string (or null if not available)
103          * @throws DataProcessingException if a data processing error occurs
104          * @throws IOException             if an I/O error occurs
105          */
106         public void decodeFile(String file, String reference)
107                 throws DataProcessingException, IOException {
108     
109             InputStream is = new FileInputStream(file);
110     
111             decodeTimer.start();
112             String result = clientFrontEnd.decode(is, file);
113             decodeTimer.stop();
114     
115             aligner.align(reference, result);
116             aligner.printSentenceSummary();
117             aligner.printTotalSummary();
118     
119             long processingTime = decodeTimer.getCurTime();
120             cumulativeProcessingTime += processingTime;
121     
122             System.out.println("   This Time Proc: " +
123                     timeFormat.format(processingTime / 1000.f));
124             System.out.println("   Total Time Proc: " +
125                     timeFormat.format(cumulativeProcessingTime / 1000.f));
126             System.out.println("----------");
127         }
128     
129     
130         /**
131          * Main method of this BatchDecoder.
132          *
133          * @param argv argv[0] : SphinxProperties file argv[1] : a file listing all the audio files to decode
134          */
135         public static void main(String[] argv) {
136     
137             if (argv.length < 2) {
138                 System.out.println
139                         ("Usage: BatchClient propertiesFile batchFile");
140                 System.exit(1);
141             }
142     
143             String batchFile = argv[1];
144     
145             try {
146                 Map<String, Object> props = new HashMap<String, Object>();
147                 props.put(BATCH_FILE, batchFile);
148                 BatchClient client = ConfigurationManager.getInstance(BatchClient.class, props);
149     
150                 client.decode();
151     
152             } catch (Exception e) {
153                 e.printStackTrace();
154             }
155         }
156     
157     }
