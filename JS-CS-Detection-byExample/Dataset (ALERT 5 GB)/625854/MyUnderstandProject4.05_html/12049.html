12     /**
13      * Computes the linear predictive model using the Levinson-Durbin algorithm. Linear prediction assumes that a signal can
14      * be model as a linear combination of previous samples, that is, the current sample x[i] can be modeled as:
15      * <p/>
16      * <pre> x[i] = a[0] + a[1] * x[i - 1] + a[2] * x[i - 2] + ... </pre>
17      * <p/>
18      * The summation on the right hand side of the equation involves a finite number of terms. The number of previous
19      * samples used is the order of the linear prediction.
20      * <p/>
21      * This class also provides a method to compute LPC cepstra, that is, the cepstra computed from LPC coefficients, as
22      * well as a method to compute the bilinear transformation of the LPC
23      */
24     public class LinearPredictor {
25     
26         private int order;
27         private int cepstrumOrder;
28         private double[] reflectionCoeffs;
29         private double[] ARParameters;
30         private double alpha;
31         private double[] cepstra;
32         private final double[] bilinearCepstra;

36          * Constructs a LinearPredictor with the given order.
37          *
38          * @param order the order of the LinearPredictor
39          */
40         public LinearPredictor(int order) {
41             this.order = order;
42     
43             // Set the rest to null values
44             reflectionCoeffs = null;
45             ARParameters = null;
46             alpha = 0;
47             cepstra = null;
48             bilinearCepstra = null;
49         }

52         /**
53          * Method to compute Linear Prediction Coefficients for a frame of speech. Assumes the following 
54          * sign convention:<br> prediction(x[t]) = Sum_i {Ar[i] * x[t-i]}
55          *
56          * @param autocor
57          * @return the energy of the frame (alpha in the Levinson recursion)
58          */
59         public double[] getARFilter(double[] autocor) {
60             /* No signal */
61             if (autocor[0] == 0) {
62                 return null;
63             }
64             reflectionCoeffs = new double[order + 1];
65             ARParameters = new double[order + 1];
66             double[] backwardPredictor = new double[order + 1];
67     
68             alpha = autocor[0];
69             reflectionCoeffs[1] = -autocor[1] / autocor[0];
70             ARParameters[0] = 1.0;
71             ARParameters[1] = reflectionCoeffs[1];
72             alpha *= (1 - reflectionCoeffs[1] * reflectionCoeffs[1]);
73     
74             for (int i = 2; i <= order; i++) {
75                 for (int j = 1; j < i; j++) {
76                     backwardPredictor[j] = ARParameters[i - j];
77                 }
78                 reflectionCoeffs[i] = 0;
79                 for (int j = 0; j < i; j++) {
80                     reflectionCoeffs[i] -= ARParameters[j] * autocor[i - j];
81                 }
82                 reflectionCoeffs[i] /= alpha;
83     
84                 for (int j = 1; j < i; j++) {
85                     ARParameters[j] += reflectionCoeffs[i] * backwardPredictor[j];
86                 }
87                 ARParameters[i] = reflectionCoeffs[i];
88                 alpha *= (1 - reflectionCoeffs[i] * reflectionCoeffs[i]);
89                 if (alpha <= 0.0) {
90                     return null;
91                 }
92             }
93             return ARParameters;
94         }

97         /**
98          * Computes AR parameters from a given set of reflection coefficients.
99          *
100          * @param RC       double array of reflection coefficients. The RC array must begin at 1 (RC[0] is a dummy value)
101          * @param lpcorder AR order desired
102          * @return AR parameters
103          */
104         public double[] reflectionCoeffsToARParameters(double[] RC, int lpcorder) {
105             double[][] tmp = new double[lpcorder + 1][lpcorder + 1];
106     
107             order = lpcorder;
108             reflectionCoeffs = RC.clone();
109     
110             for (int i = 1; i <= lpcorder; i++) {
111                 for (int m = 1; m < i; m++) {
112                     tmp[i][m] = tmp[i - 1][m] - RC[i] * tmp[i - 1][i - m];
113                 }
114                 tmp[i][i] = RC[i];
115             }
116             ARParameters[0] = 1;
117             for (int m = 1; m <= lpcorder; m++) {
118                 ARParameters[m] = tmp[m][m];
119             }
120             return ARParameters;
121         }

124         /**
125          * Computes LPC Cepstra from the AR predictor parameters and alpha using a recursion invented by Oppenheim et al.
126          * The literature shows the optimal value of cepstral order to be:
127          * <p/>
128          * <pre>0.75 * LPCorder <= ceporder <= 1.25 * LPCorder</pre>
129          *
130          * @param ceporder is the order of the LPC cepstral vector to be computed.
131          * @return LPC cepstra
132          */
133         public double[] getData(int ceporder) {
134             int i;
135             double sum;
136     
137             if (ceporder <= 0) {
138                 return null;
139             }
140     
141             cepstrumOrder = ceporder;
142             cepstra = new double[cepstrumOrder];
143     
144             cepstra[0] = Math.log(alpha);
145             if (cepstrumOrder == 1) {
146                 return cepstra;
147             }
148     
149             cepstra[1] = -ARParameters[1];
150     
151             for (i = 2; i < Math.min(cepstrumOrder, order + 1); i++) {
152                 sum = (double) i * ARParameters[i];
153                 for (int j = 1; j < i; j++) {
154                     sum += ARParameters[j] * cepstra[i - j] * (double) (i - j);
155                 }
156                 cepstra[i] = -sum / (double) i;
157             }
158             for (; i < cepstrumOrder; i++) { // Only if cepstrumOrder > order+1
159                 sum = 0;
160                 for (int j = 1; j <= order; j++) {
161                     sum += ARParameters[j] * cepstra[i - j] * (double) (i - j);
162                 }
163                 cepstra[i] = -sum / (double) i;
164             }
165             return cepstra;
166         }

169         /**
170          * Computes a bi-linear frequency warped version of the LPC cepstrum from the LPC cepstrum. The recursive algorithm
171          * used is defined in Oppenheim's paper in Proceedings of IEEE, June 1972 The program has been written using g[x,y]
172          * = g_o[x,-y] where g_o is the array used by Oppenheim. To handle the reversed array index the recursion has been
173          * done DOWN the array index.
174          *
175          * @param warp          is the warping coefficient. For 16KHz speech 0.6 is good valued.
176          * @param nbilincepstra is the number of bilinear cepstral values to be computed from the linear frequency
177          *                      cepstrum.
178          * @return a bi-linear frequency warped version of the LPC cepstrum
179          */
180         public double[] getBilinearCepstra(double warp, int nbilincepstra) {
181             double[][] g = new double[nbilincepstra][cepstrumOrder];
182     
183             // Make a local copy as this gets destroyed
184             double[] lincep = Arrays.copyOf(cepstra, cepstrumOrder);
185     
186             bilinearCepstra[0] = lincep[0];
187             lincep[0] = 0;
188             g[0][cepstrumOrder - 1] = lincep[cepstrumOrder - 1];
189             for (int i = 1; i < nbilincepstra; i++) {
190                 g[i][cepstrumOrder - 1] = 0;
191             }
192     
193             for (int i = cepstrumOrder - 2; i >= 0; i--) {
194                 g[0][i] = warp * g[0][i + 1] + lincep[i];
195                 g[1][i] = (1 - warp * warp) * g[0][i + 1] + warp * g[1][i + 1];
196                 for (int j = 2; j < nbilincepstra; j++) {
197                     g[j][i] = warp * (g[j][i + 1] - g[j - 1][i]) + g[j - 1][i + 1];
198                 }
199             }
200     
201             for (int i = 1; i <= nbilincepstra; i++) {
202                 bilinearCepstra[i] = g[i][0];
203             }
204     
205             return bilinearCepstra;
206         }
