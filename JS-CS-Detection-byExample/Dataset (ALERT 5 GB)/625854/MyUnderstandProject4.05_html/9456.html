58         /**
59          * Loads the sphinx4 density file, a set of density arrays are created and placed in the given pool.
60          *
61          * {@inheritDoc}
62          */
63         @Override
64         protected void loadHMMPool(boolean useCDUnits, InputStream inputStream, String path)
65                 throws IOException {
66             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream, '#', false);
67     
68             logger.info("Loading HMM file from: ");
69             logger.info(path);
70     
71             est.expectString(MODEL_VERSION);
72     
73             int numBase = est.getInt("numBase");
74             est.expectString("n_base");
75     
76             int numTri = est.getInt("numTri");
77             est.expectString("n_tri");
78     
79             int numStateMap = est.getInt("numStateMap");
80             est.expectString("n_state_map");
81     
82             int numTiedState = est.getInt("numTiedState");
83             est.expectString("n_tied_state");
84     
85             int numContextIndependentTiedState = est.getInt("numContextIndependentTiedState");
86             est.expectString("n_tied_ci_state");
87     
88             int numTiedTransitionMatrices = est.getInt("numTiedTransitionMatrices");
89             est.expectString("n_tied_tmat");
90     
91             int[] maxStid = new int[maxModelSize];
92     
93             HMMManager hmmManager = super.getHmmManager();
94             Pool<float[][]> matrixPool = super.getMatrixPool();
95             Pool<float[]> mixtureWeightsPool = super.getMixtureWeightsPool();
96             Map<String, Unit> contextIndependentUnits = super.getContextIndependentUnits();
97     
98             assert numTiedState == mixtureWeightsPool.getFeature(NUM_SENONES, 0);
99             assert numTiedTransitionMatrices == matrixPool.size();
100     
101             // Load the base phones
102             for (int i = 0; i < numBase; i++) {
103                 String name = est.getString();
104                 String left = est.getString();
105                 String right = est.getString();
106                 String position = est.getString();
107                 String attribute = est.getString();
108                 int tmat = est.getInt("tmat");
109                 int numStates = 0;
110     
111                 // Read all state ID in the line...
112                 for (int j = 0; ; j++) {
113                     String str = est.getString();
114     
115                     // ... until we reach a "N"
116     
117                     if (!str.equals("N")) {
118                         int id = Integer.parseInt(str);
119                         try {
120                             maxStid[j] = id;
121                         } catch (ArrayIndexOutOfBoundsException aie) {
122                             throw new Error("Use a larger value for " +
123                                     "maxStatePerModel");
124                         }
125                         assert maxStid[j] >= 0 &&
126                                 maxStid[j] < numContextIndependentTiedState;
127                     } else {
128                         numStates = j;
129                         break;
130                     }
131                 }
132                 int[] stid = Arrays.copyOf(maxStid, numStates);
133     
134                 assert left.equals("-");
135                 assert right.equals("-");
136                 assert position.equals("-");
137                 assert tmat < numTiedTransitionMatrices;
138     
139     //            Unit unit = Unit.getUnit(name, attribute.equals(FILLER));
140                 Unit unit = null;
141                 contextIndependentUnits.put(unit.getName(), unit);
142     
143                 if (logger.isLoggable(Level.FINE)) {
144                     logger.fine("Loaded " + unit);
145                 }
146     
147                 // The first filler
148                 if (unit.isFiller() && unit.getName().equals(SILENCE_CIPHONE)) {
149     //                unit = Unit.SILENCE;
150                     unit = null;
151                 }
152     
153                 float[][] transitionMatrix = matrixPool.get(tmat);
154                 SenoneSequence ss = getSenoneSequence(stid);
155     
156                 HMM hmm = new SenoneHMM(unit, ss, transitionMatrix, HMMPosition.lookup(position));
157                 hmmManager.put(hmm);
158             }
159     
160             // Load the context dependent phones.  If the useCDUnits
161             // property is false, the CD phones will not be created, but
162             // the values still need to be read in from the file.
163     
164             String lastUnitName = "";
165             Unit lastUnit = null;
166             int[] lastStid = null;
167             SenoneSequence lastSenoneSequence = null;
168     
169             for (int i = 0; i < numTri; i++) {
170                 String name = est.getString();
171                 String left = est.getString();
172                 String right = est.getString();
173                 String position = est.getString();
174                 String attribute = est.getString();
175                 int tmat = est.getInt("tmat");
176     
177                 int numStates = 0;
178     
179                 // Read all state ID in the line...
180                 for (int j = 0; ; j++) {
181                     String str = est.getString();
182     
183                     // ... until we reach a "N"
184     
185                     if (!str.equals("N")) {
186                         int id = Integer.parseInt(str);
187                         maxStid[j] = id;
188                         assert maxStid[j] >= numContextIndependentTiedState &&
189                                 maxStid[j] < numTiedState;
190                     } else {
191                         numStates = j;
192                         break;
193                     }
194                 }
195                 int[] stid = Arrays.copyOf(maxStid, numStates);
196     
197                 assert !left.equals("-");
198                 assert !right.equals("-");
199                 assert !position.equals("-");
200                 assert attribute.equals("n/a");
201                 assert tmat < numTiedTransitionMatrices;
202     
203                 if (useCDUnits) {
204                     Unit unit = null;
205                     String unitName = (name + ' ' + left + ' ' + right);
206     
207                     if (unitName.equals(lastUnitName)) {
208                         unit = lastUnit;
209                     } else {
210                         Unit[] leftContext = new Unit[1];
211                         leftContext[0] = contextIndependentUnits.get(left);
212     
213                         Unit[] rightContext = new Unit[1];
214                         rightContext[0] = contextIndependentUnits.get(right);
215     
216                         Context context = LeftRightContext.get(leftContext, rightContext);
217     //                    unit = Unit.getUnit(name, false, context);
218                         unit = null;
219                     }
220                     lastUnitName = unitName;
221                     lastUnit = unit;
222     
223                     if (logger.isLoggable(Level.FINE)) {
224                         logger.fine("Loaded " + unit);
225                     }
226     
227                     float[][] transitionMatrix = matrixPool.get(tmat);
228     
229                     SenoneSequence ss = lastSenoneSequence;
230                     if (ss == null || !sameSenoneSequence(stid, lastStid)) {
231                         ss = getSenoneSequence(stid);
232                     }
233                     lastSenoneSequence = ss;
234                     lastStid = stid;
235     
236                     HMM hmm = new SenoneHMM(unit, ss, transitionMatrix, HMMPosition.lookup(position));
237                     hmmManager.put(hmm);
238                 }
239             }
240     
241             est.close();
242         }

244         @Override
245         protected Pool<float[][]> loadTransitionMatricesAscii(String path)
246                 throws FileNotFoundException, IOException {
247     
248             String location = "";
249             InputStream inputStream = StreamFactory.getInputStream(location, path);
250     
251             LogMath logMath = ConfigurationManager.getInstance(LogMath.class);
252             logger.info("Loading transition matrices from: ");
253             logger.info(path);
254     
255             Pool<float[][]> pool = new Pool<float[][]>(path);
256             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream, '#', false);
257     
258             est.expectString("tmat");
259             int numMatrices = est.getInt("numMatrices");
260             est.expectString("X");
261             // numStates = est.getInt("numStates");
262     
263             for (int i = 0; i < numMatrices; i++) {
264                 est.expectString("tmat");
265                 est.expectString("[" + i + ']');
266                 est.expectString("nstate");
267                 // Number of emitting states + 1, final non-emitting state
268                 int numStates = est.getInt("numStates") + 1;
269                 float[][] tmat = new float[numStates][numStates];
270     
271                 for (int j = 0; j < numStates; j++) {
272                     for (int k = 0; k < numStates; k++) {
273     
274                         // the last row is just zeros, so we just do
275                         // the first (numStates - 1) rows
276     
277                         if (j < numStates - 1) {
278                             if (sparseForm) {
279                                 if (k == j || k == j + 1) {
280                                     tmat[j][k] = est.getFloat("tmat value");
281                                 }
282                             } else {
283                                 tmat[j][k] = est.getFloat("tmat value");
284                             }
285                         }
286     
287                         tmat[j][k] = logMath.linearToLog(tmat[j][k]);
288     
289                         if (logger.isLoggable(Level.FINE)) {
290                             logger.fine("tmat j " + j + " k "
291                                     + k + " tm " + tmat[j][k]);
292                         }
293                     }
294                 }
295                 pool.put(i, tmat);
296             }
297             est.close();
298             return pool;
299         }

301         @Override
302         protected Pool<float[][]> loadTransitionMatricesBinary(String path) throws IOException, URISyntaxException {
303     
304             logger.info("Loading transition matrices from: ");
305             logger.info(path);
306     
307             Properties props = new Properties();
308             DataInputStream dis = readS3BinaryHeader(path, props);
309     
310             String version = props.getProperty("version");
311     
312             if (version == null || !version.equals(TMAT_FILE_VERSION)) {
313                 throw new IOException("Unsupported version in " + path);
314             }
315     
316             String checksum = props.getProperty("chksum0");
317             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
318     
319             Pool<float[][]> pool = new Pool<float[][]>(path);
320     
321             int numMatrices = readInt(dis);
322             // numRows = readInt(dis);
323             // numStates = readInt(dis);
324             int numValues = readInt(dis);
325     
326             // assert numValues == numStates * numRows * numMatrices;
327     
328             int count = 0;
329             for (int i = 0; i < numMatrices; i++) {
330                 int numStates = readInt(dis);
331                 float[][] tmat = new float[numStates][];
332                 // last row should be zeros
333                 tmat[numStates - 1] = new float[numStates];
334                 logMath.linearToLog(tmat[numStates - 1]);
335     
336                 for (int j = 0; j < numStates - 1; j++) {
337                     tmat[j] = readFloatArray(dis, numStates);
338                     count += numStates;
339                     Utilities.nonZeroFloor(tmat[j], 0f);
340                     Utilities.normalize(tmat[j]);
341                     logMath.linearToLog(tmat[j]);
342                 }
343                 pool.put(i, tmat);
344             }
345             dis.close();
346             assert numValues == count;
347             return pool;
348         }
