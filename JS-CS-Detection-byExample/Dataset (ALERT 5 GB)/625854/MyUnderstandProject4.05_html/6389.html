13     package edu.cmu.sphinx.linguist.flat;
14     
15     import edu.cmu.sphinx.linguist.WordSearchState;
16     import edu.cmu.sphinx.linguist.dictionary.Pronunciation;
17     
18     
19     /** Represents a pronunciation in an SentenceHMMS */
20     public class PronunciationState extends SentenceHMMState implements        WordSearchState {
21     
22         private final Pronunciation pronunciation;
23     
24         /**
25          * Creates a PronunciationState
26          *
27          * @param parent the parent word of the current pronunciation
28          * @param which  the pronunciation of interest
29          */
30         public PronunciationState(WordState parent, int which) {
31             super("P", parent, which);
32     
33             pronunciation = parent.getWord().getPronunciations(null)[which];
34         }
35     
36     
37         /**
38          * Creates a PronunciationState
39          *
40          * @param name  the name of the pronunciation associated with this state
41          * @param p     the pronunciation
42          * @param which the index for the pronunciation
43          */
44         public PronunciationState(String name, Pronunciation p, int which) {
45             super(name, null, which);
46             pronunciation = p;
47         }
48     
49     
50         /**
51          * Gets the pronunciation associated with this state
52          *
53          * @return the pronunciation
54          */
55         @Override
56         public Pronunciation getPronunciation() {
57             return pronunciation;
58         }
59     
60     
61         /**
62          * Retrieves a short label describing the type of this state. Typically, subclasses of SentenceHMMState will
63          * implement this method and return a short (5 chars or less) label
64          *
65          * @return the short label.
66          */
67         @Override
68         public String getTypeLabel() {
69             return "Pron";
70         }
71     
72     
73         /**
74          * Returns the state order for this state type
75          *
76          * @return the state order
77          */
78         @Override
79         public int getOrder() {
80             return 4;
81         }
82     
83     
84         /**
85          * Returns true if this PronunciationState indicates the start of a word. Returns false if this PronunciationState
86          * indicates the end of a word.
87          *
88          * @return true if this PronunciationState indicates the start of a word, false if this PronunciationState indicates
89          *         the end of a word
90          */
91         @Override
92         public boolean isWordStart() {
93             return true;
94         }
95     }
