13     package edu.cmu.sphinx.util;
14     
15     import java.text.DecimalFormat;
16     
17     /**
18      * Keeps track of execution times. This class provides methods that can be used for timing processes. The process to be
19      * timed should be bracketed by calls to timer.start() and timer.stop().  Repeated operations can be timed more than
20      * once. The timer will report the minimum, maximum, average and last time executed for all start/stop pairs when the
21      * timer.dump is called.
22      * <p/>
23      * Timer instances can be obtained from a global cache implemented in {@code TimerPool}.
24      *
25      * @see TimerPool
26      */
27     public class Timer {
28     
29         private final static DecimalFormat timeFormatter = new DecimalFormat("###0.0000");
30     
31         private final String name;
32     
33         private double sum;
34         private long count;
35         private long startTime;
36         private long curTime;
37         private long minTime = Long.MAX_VALUE;
38         private long maxTime;
39         private boolean notReliable; // if true, timing is not reliable
40     
41     
42         /**
43          * Creates a timer.
44          *
45          * @param name the name of the timer
46          */
47         Timer(String name) {
48             assert name != null : "timers must have a name!";
49             this.name = name;
50             reset();
51         }
52     
53     
54         /**
55          * Retrieves the name of the timer
56          *
57          * @return the name of the timer
58          */
59         public String getName() {
60             return name;
61         }
62     
63     
64         /** Resets the timer as if it has never run before. */
65         public void reset() {
66             startTime = 0L;
67             count = 0L;
68             sum = 0L;
69             minTime = Long.MAX_VALUE;
70             maxTime = 0L;
71             notReliable = false;
72         }
73     
74     
75         /**
76          * Returns true if the timer has started.
77          *
78          * @return true if the timer has started; false otherwise
79          */
80         public boolean isStarted() {
81             return (startTime > 0L);
82         }
83     
84     
85         /** Starts the timer running. */
86         public void start() {
87             if (startTime != 0L) {
88                 notReliable = true; // start called while timer already running
89                 System.out.println
90                         (getName() + " timer.start() called without a stop()");
91             }
92             startTime = System.currentTimeMillis();
93         }
94     
95     
96         /**
97          * Starts the timer at the given time.
98          *
99          * @param time the starting time
100          */
101         public void start(long time) {
102             if (startTime != 0L) {
103                 notReliable = true; // start called while timer already running
104                 System.out.println
105                         (getName() + " timer.start() called without a stop()");
106             }
107             if (time > System.currentTimeMillis()) {
108                 throw new IllegalStateException
109                         ("Start time is later than current time");
110             }
111             startTime = time;
112         }
113     
114     
115         /**
116          * Stops the timer.
117          *
118          * @param verbose if <code>true</code>, print out details from this run; otherwise, don't print the details
119          * @return the duration since start in milliseconds
120          */
121         public long stop(boolean verbose) {
122             if (startTime == 0L) {
123                 notReliable = true;        // stop called, but start never called
124                 System.out.println
125                         (getName() + " timer.stop() called without a start()");
126             }
127             curTime = System.currentTimeMillis() - startTime;
128             startTime = 0L;
129             if (curTime > maxTime) {
130                 maxTime = curTime;
131             }
132             if (curTime < minTime) {
133                 minTime = curTime;
134             }
135             count++;
136             sum += curTime;
137             if (verbose) {
138                 dump();
139             }
140             return curTime;
141         }
142     
143     
144         /** Stops the timer. */
145         public void stop() {
146             stop(false);
147         }
148     
149     
150         /** Dump the timer. Shows the timer details. */
151         public void dump() {
152             showTimesShort();
153         }
154     
155     
156         /**
157          * Gets the count of starts for this timer
158          *
159          * @return the count
160          */
161         public long getCount() {
162             return count;
163         }
164     
165     
166         /**
167          * Returns the latest time gathered
168          *
169          * @return the time in milliseconds
170          */
171         public long getCurTime() {
172             return curTime;
173         }
174     
175     
176         /**
177          * Gets the average time for this timer in milliseconds
178          *
179          * @return the average time
180          */
181         public double getAverageTime() {
182             if (count == 0) {
183                 return 0.0;
184             }
185             return sum / count;
186         }
187     
188     
189         /**
190          * Gets the min time for this timer in milliseconds
191          *
192          * @return the min time
193          */
194         public long getMinTime() {
195             return minTime;
196         }
197     
198     
199         /**
200          * Gets the max time for this timer in milliseconds
201          *
202          * @return the max time in milliseconds
203          */
204         public long getMaxTime() {
205             return maxTime;
206         }
207     
208     
209         /**
210          * Formats times into a standard format.
211          *
212          * @param time the time (in milliseconds) to be formatted
213          * @return a string representation of the time.
214          */
215         private String fmtTime(long time) {
216             return fmtTime(time / 1000.0);
217         }
218     
219     
220         /**
221          * Formats times into a standard format.
222          *
223          * @param time the time (in seconds) to be formatted
224          * @return a string representation of the time.
225          */
226         private String fmtTime(double time) {
227             return Utilities.pad(timeFormatter.format(time) + 's', 10);
228         }
229     
230     
231         /** Shows brief timing statistics . */
232         private void showTimesShort() {
233             double avgTime = 0.0;
234     
235             if (count == 0) {
236                 return;
237             }
238     
239             if (count > 0) {
240                 avgTime = sum / count / 1000.0;
241             }
242     
243             if (notReliable) {
244                 System.out.print(Utilities.pad(name, 20) + ' ');
245                 System.out.println("Not reliable.");
246             } else {
247                 System.out.print(Utilities.pad(name, 20) + ' ');
248                 System.out.print(Utilities.pad(String.valueOf(count), 8));
249                 System.out.print(fmtTime(curTime));
250                 System.out.print(fmtTime(minTime));
251                 System.out.print(fmtTime(maxTime));
252                 System.out.print(fmtTime(avgTime));
253                 System.out.print(fmtTime(sum / 1000.0));
254                 System.out.println();
255             }
256         }
257     }
