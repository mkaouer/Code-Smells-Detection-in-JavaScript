13     package edu.cmu.sphinx.tools.audio;
14     
15     import javax.sound.sampled.AudioFormat;
16     import java.io.IOException;
17     import java.io.OutputStream;
18     
19     
20     /** Writes raw audio to a file, handling sample size, endian format, and signed/unsigned data. */
21     public class RawWriter {
22     
23         private final OutputStream outputStream;
24         private final int bytesPerSample;
25         private boolean signedData;
26     
27     
28         /**
29          */
30         public RawWriter(OutputStream outputStream,
31                          AudioFormat audioFormat) {
32             AudioFormat.Encoding encoding = audioFormat.getEncoding();
33             this.outputStream = outputStream;
34             this.bytesPerSample = audioFormat.getSampleSizeInBits() / 8;
35             if (encoding == AudioFormat.Encoding.PCM_SIGNED) {
36                 this.signedData = true;
37             } else if (encoding == AudioFormat.Encoding.PCM_UNSIGNED) {
38                 this.signedData = false;
39             } else {
40                 System.err.println("Unsupported audio encoding: " + encoding);
41                 System.exit(-1);
42             }
43         }
44     
45     
46         /**
47          * Writes the sample to the output stream.
48          *
49          * @throws java.io.IOException
50          */
51         public void writeSample(int sample) throws IOException {
52             /* First byte contains the big endian byte that carries the sign.
53              */
54             if (signedData) {
55                 outputStream.write(sample >> ((bytesPerSample - 1) * 8));
56             } else {
57                 outputStream.write((sample >> ((bytesPerSample - 1) * 8)) & 0xff);
58             }
59     
60             /* Now just output the rest of the data in big endian form.
61              */
62             for (int i = bytesPerSample - 2; i >= 0; i--) {
63                 outputStream.write((sample >> (i * 8)) & 0xff);
64             }
65         }
66     }
