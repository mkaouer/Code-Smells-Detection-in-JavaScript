13     package edu.cmu.sphinx.research.parallel;
14     
15     import edu.cmu.sphinx.decoder.search.Token;
16     
17     
18     /** An array implementation of TokenStack. */
19     public class ArrayTokenStack implements TokenStack {
20     
21         private final Token[] stack;
22     
23     
24         /**
25          * Constructs a ArrayTokenStack object with the given capacity.
26          *
27          * @param capacity the height of the stack
28          */
29         public ArrayTokenStack(int capacity) {
30             this.stack = new Token[capacity];
31             clear();
32         }
33     
34     
35         /**
36          * Adds the given Token to this TokenStack.
37          *
38          * @param newToken the new Token to add
39          * @return the replaced Token, or null if no Token was replaced
40          */
41         @Override
42         public Token add(Token newToken) {
43             Token lowestToken = null;
44             int lowestTokenIndex = -1;
45     
46             // first look for the lowest scoring token
47             for (int i = 0; i < stack.length; i++) {
48                 Token token = stack[i];
49                 if (token == null) {
50                     stack[i] = newToken;
51                     return null;
52                 } else {
53                     if (token.getFrameNumber() != newToken.getFrameNumber()) {
54                         stack[i] = newToken;
55                         return token;
56                     }
57                     if (lowestToken == null) {
58                         lowestToken = token;
59                         lowestTokenIndex = i;
60                     } else if (token.getScore() <= lowestToken.getScore()) {
61                         lowestToken = token;
62                         lowestTokenIndex = i;
63                     }
64                 }
65             }
66     
67             // if found the lowest scoring token
68             if (lowestTokenIndex != -1) {
69                 stack[lowestTokenIndex] = newToken;
70                 return lowestToken;
71             } else {
72                 // if no tokens at all
73                 stack[0] = newToken;
74                 return null;
75             }
76         }
77     
78     
79         /**
80          * Returns true if the given score is higher than the lowest scoring token in this TokenStack.
81          *
82          * @param score the score the check
83          * @return true if the given score is higher than the lowest scoring token, false otherwise
84          */
85         @Override
86         public boolean isInsertable(float score, int frameNumber) {
87             for (Token token : stack) {
88                 if (token == null ||
89                         token.getFrameNumber() != frameNumber ||
90                         token.getScore() <= score) {
91                     return true;
92                 }
93             }
94             return false;
95         }
96     
97     
98         /**
99          * Returns the capacity of this TokenStack, that is, the maximum number of elements allowed in this TokenStack.
100          *
101          * @return the capacity of this TokenStack
102          */
103         @Override
104         public int getCapacity() {
105             return stack.length;
106         }
107     
108     
109         /** Removes all of the Tokens from this TokenStack */
110         @Override
111         public void clear() {
112             for (int i = 0; i < stack.length; i++) {
113                 stack[i] = null;
114             }
115         }
116     }
