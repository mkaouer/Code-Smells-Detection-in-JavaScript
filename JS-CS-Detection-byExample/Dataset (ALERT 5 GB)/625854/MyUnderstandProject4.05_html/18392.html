31     /** 
32      * Triphone selection algorithm:
33      * 
34      * 1) Look for a triphone that matches exactly in terms of unit, unit 
35      * context and word position. Word position can be one of 'beginning', 
36      * 'end', 'internal', 'single' and 'unknown'. 
37      * 2) If not found, try to find a match at any word position. 
38      * 3) If still not found, if any of the context units are non-silence  
39      *    filler units, replace them with silence and try again. 
40      * 4) Finally, if still not found, the context-independent version is used 
41      *  
42      * The code is in TiedStateAcousticModel.lookupNearestHMM
43      */
44     
45     /**
46      * 
47      * Remark1: S4 does not use HMM tying: the HTK "tiedlist" is not loaded nor user
48      * for now
49      * 
50      * Remark2: HTK does nearly never backoff to monophones, whereas S4 might do it.
51      * Hence, all 1ph must be present, which is not always the case in HTK models.
52      * In this file, we have simply tied all 1phs to the first 3ph found with the
53      * same base phone. This is clearly not the best option, this may impact the
54      * performances. A better alternative should be proposed.
55      * 
56      * Remark3: HTK is case-sensitive, while S4 requires upper-case names. A good
57      * option would be to modify S4 to make it case-sensitive, but in this version,
58      * I have rather built a class "NamesConversion" to transform the models,
59      * lexicons and grammar.
60      * 
61      * Remark4: HTK does not have separate filler file. It should be quite easy to
62      * define the fillers in the configuration of HTKLoader, but I have not yet done
63      * it. For now, a separate filler file must be built.
64      * 
65      * Remark5: The HTK/Julius lexicon has different fields for the word name and
66      * the printed output. We simply delete and don't use this field ("[...]") in
67      * this version. This modify the way scoring must be done. WARNING !
68      * 
69      * Remark6: Diphones are not allowed in S4 (?), and when they occur in the HTK
70      * MMF, the 2ph is transformed into a 3ph with "SIL" context. When both the
71      * transformed 2ph and the 3ph coexists in the MMF file, only the first one is
72      * kept (!)
73      * 
74      * Remark7: HTKloader does not support yet all the HTK format. However it should
75      * work for standard 3ph models (it supports state/transition tying).
76      * 
77      */

295         /**
296          * Creates the senone pool from the rest of the pools. assumes the means and
297          * variances are in the same order
298          * 
299          * @param distFloor the lowest allowed score
300          * @param varianceFloor the lowest allowed variance
301          * @return the senone pool
302          */
303         private Pool<Senone> createSenonePool(float distFloor, float varianceFloor) {
304             Pool<Senone> pool = new Pool<Senone>("senones");
305             int numMixtureWeights = mixtureWeightsPool.size();
306     
307             int numMeans = meansPool.size();
308             int numVariances = variancePool.size();
309             // TODO: remove this variable to allow different number of states per
310             // HMM
311             int numGaussiansPerSenone = mixtureWeightsPool.getFeature(
312                     NUM_GAUSSIANS_PER_STATE, 0);
313             int numSenones = mixtureWeightsPool.getFeature(NUM_SENONES, 0);
314             int whichGaussian = 0;
315     
316             logger.fine("NG " + numGaussiansPerSenone);
317             logger.fine("NS " + numSenones);
318             logger.fine("NMIX " + numMixtureWeights);
319             logger.fine("NMNS " + numMeans);
320             logger.fine("NMNS " + numVariances);
321     
322             assert numGaussiansPerSenone > 0;
323             assert numMixtureWeights == numSenones;
324             assert numVariances == numSenones * numGaussiansPerSenone;
325             assert numMeans == numSenones * numGaussiansPerSenone;
326             
327             float [][] meansTransformationMatrix = meanTransformationMatrixPool == null ?
328                     null : meanTransformationMatrixPool.get(0);
329             float [] meansTransformationVector = meanTransformationVectorPool == null ?
330                     null : meanTransformationVectorPool.get(0);
331             float [][] varianceTransformationMatrix = varianceTransformationMatrixPool == null ?
332                     null : varianceTransformationMatrixPool.get(0);
333             float [] varianceTransformationVector = varianceTransformationVectorPool == null ?
334                     null : varianceTransformationVectorPool.get(0);
335     
336             for (int i = 0; i < numSenones; i++) {
337                 MixtureComponent[] mixtureComponents = new MixtureComponent[numGaussiansPerSenone];
338                 for (int j = 0; j < numGaussiansPerSenone; j++) {
339                     mixtureComponents[j] = new MixtureComponent(logMath,
340                             meansPool.get(whichGaussian), 
341                             meansTransformationMatrix, 
342                             meansTransformationVector,
343                             variancePool.get(whichGaussian),
344                             varianceTransformationMatrix,
345                             varianceTransformationVector, 
346                             distFloor,
347                             varianceFloor);
348     
349                     whichGaussian++;
350                 }
351     
352                 Senone senone = new GaussianMixture(logMath, mixtureWeightsPool
353                         .get(i), mixtureComponents, i);
354     
355                 pool.put(i, senone);
356             }
357             return pool;
358         }

360         /**
361          * Reads the next word (text separated by whitespace) from the given stream.
362          * 
363          * the input stream
364          * 
365          * @return the next word
366          * @throws IOException
367          *             on error
368          */
369         String readWord(DataInputStream dis) throws IOException {
370             StringBuilder sb = new StringBuilder();
371             char c;
372             // skip leading whitespace
373             do {
374                 c = readChar(dis);
375             } while (Character.isWhitespace(c));
376             // read the word
377             do {
378                 sb.append(c);
379                 c = readChar(dis);
380             } while (!Character.isWhitespace(c));
381             return sb.toString();
382         }

453         /**
454          * Loads the model file. A set of density arrays are created and placed in
455          * the given pool.
456          * 
457          * 
458          * @param useCDUnits if true, loads also the context dependent units
459          * @param htkModels the set of HTK models
460          * @param path the path to a density file
461          * 
462          * @throws IOException
463          *             if an error occurs while loading the data
464          */
465         protected void loadHMMPool(boolean useCDUnits, HTKStruct htkModels,
466                 String path) throws IOException {
467             int numStatePerHMM;
468     
469             // assert numTiedState == mixtureWeightsPool.getFeature(NUM_SENONES, 0);
470             // assert numTiedTransitionMatrices == matrixPool.size();
471     
472             // Load the base phones
473             /**
474              * This version loads the phones from the .MMF But contrary to S4, HTK
475              * do not produce reliable monophones, and some monophones may be
476              * missing ! An option might be to load 1ph trained independently from
477              * the 3ph but this file may not be always available. An easy fallback
478              * solution for now is to tie all the monophone to the first 3ph with
479              * the same base name found...
480              */
481             if (!tie1ph) {
482                 for (Iterator<SingleHMM> monoPhones = htkModels.hmmsHTK.get1phIt(); monoPhones
483                         .hasNext();) {
484                     SingleHMM hmm = monoPhones.next();
485                     if (hmm == null)
486                         break;
487                     String name = hmm.getName();
488                     String attribute;
489                     if (name.equals("sil") || name.equals("sp")
490                             || name.equals("bb") || name.equals("xx")
491                             || name.equals("hh"))
492                         attribute = FILLER;
493                     else
494                         attribute = "nofiller";
495                     // get the transition index
496                     int tmat = hmm.trIdx;
497     
498                     // This includes non-emitting states !
499                     numStatePerHMM = hmm.getNstates();
500                     int[] stid = new int[hmm.getNbEmittingStates()];
501     
502                     // stid contains the number of senone as it in senone pool
503                     int j = 0;
504                     for (int ii = 0; ii < numStatePerHMM; ii++) {
505                         if (hmm.isEmitting(ii)) {
506                             HMMState s = hmm.getState(ii);
507                             // we use the index of HMM in MMF file
508                             stid[j] = htkModels.hmmsHTK.getStateIdx(s);
509                             j++;
510                             // assert stid[j] >= 0 && stid[j] <
511                             // numContextIndependentTiedState;
512                         }
513                     }
514                     // assert tmat < numTiedTransitionMatrices;
515     
516                     Unit unit = unitManager.getUnit(name, attribute.equals(FILLER));
517                     contextIndependentUnits.put(unit.getName(), unit);
518     
519                     if (logger.isLoggable(Level.FINE)) {
520                         logger.fine("Loaded " + unit);
521                     }
522     
523                     // The first filler
524                     if (unit.isFiller() && unit.getName().equals(SILENCE_CIPHONE)) {
525                         unit = UnitManager.SILENCE;
526                     }
527     
528                     float[][] transitionMatrix = matrixPool.get(tmat);
529                     SenoneSequence ss = getSenoneSequence(stid);
530     
531                     HMM hmm2 = new SenoneHMM(unit, ss, transitionMatrix,
532                             HMMPosition.lookup("-"));
533                     hmmManager.put(hmm2);
534                 }
535             } else {
536                 // build the 1ph by tying to the first 3ph
537                 for (int i = 0; i < htkModels.hmmsHTK.getNhmms(); i++) {
538                     SingleHMM hmm = htkModels.hmmsHTK.getHMM(i);
539                     if (hmm == null)
540                         break;
541                     String name = hmm.getBaseName();
542                     if (!contextIndependentUnits.containsKey(name)) {
543                         String attribute;
544                         
545                         if (name.equals("SIL") || name.equals("SP")
546                                 || name.equals("BB") || name.equals("XX")
547                                 || name.equals("HH"))
548                             attribute = FILLER;
549                         else
550                             attribute = "nofiller";
551                         // get the transition index
552                         int tmat = hmm.trIdx;
553     
554                         // warning ! this includes non-emitting states !
555                         numStatePerHMM = hmm.getNstates();
556                         int[] stid = new int[hmm.getNbEmittingStates()];
557     
558                         // stid contains the identifier of senone used during
559                         // creation of senone pool
560                         int j = 0;
561                         for (int ii = 0; ii < numStatePerHMM; ii++) {
562                             if (hmm.isEmitting(ii)) {
563                                 HMMState s = hmm.getState(ii);
564                                 // We get an index of HMM in MMF file
565                                 stid[j] = htkModels.hmmsHTK.getStateIdx(s);
566                                 j++;
567                                 // assert stid[j] >= 0 && stid[j] <
568                                 // numContextIndependentTiedState;
569                             }
570                         }
571                         // assert tmat < numTiedTransitionMatrices;
572     
573                         Unit unit = unitManager.getUnit(name, attribute
574                                 .equals(FILLER));
575                         contextIndependentUnits.put(unit.getName(), unit);
576     
577                         if (logger.isLoggable(Level.FINE)) {
578                             logger.fine("Loaded " + unit);
579                         }
580     
581                         // The first filler
582                         if (unit.isFiller()
583                                 && unit.getName().equals(SILENCE_CIPHONE)) {
584                             unit = UnitManager.SILENCE;
585                         }
586     
587                         float[][] transitionMatrix = matrixPool.get(tmat);
588                         SenoneSequence ss = getSenoneSequence(stid);
589     
590                         HMM hmm2 = new SenoneHMM(unit, ss, transitionMatrix,
591                                 HMMPosition.lookup("-"));
592                         hmmManager.put(hmm2);
593                     }
594                 }
595             }
596             // Load the context dependent phones. If the useCDUnits
597             // property is false, the CD phones will not be created, but
598             // the values still need to be read in from the file.
599     
600             String lastUnitName = "";
601             Unit lastUnit = null;
602             int[] lastStid = null;
603             SenoneSequence lastSenoneSequence = null;
604     
605             List<String> HMMdejavu = new ArrayList<String>();
606             for (Iterator<SingleHMM> triPhones = htkModels.hmmsHTK.get3phIt(); triPhones
607                     .hasNext();) {
608                 SingleHMM hmm = triPhones.next();
609                 if (hmm == null)
610                     break;
611                 String name = hmm.getBaseName();
612                 String left = hmm.getLeft();
613                 String right = hmm.getRight();
614     
615                 {
616                     // diphones are transformed into triphones with SIL context, as
617                     // it looks
618                     // like S4 do not support diphones
619                     if (left.equals("-"))
620                         left = "SIL";
621                     if (right.equals("-"))
622                         right = "SIL";
623                     String s = left + ' ' + name + ' ' + right;
624                     if (HMMdejavu.contains(s)) {
625                         // this may happen when a diphone is transformed into a
626                         // triphone with SIL
627                         continue;
628                     }
629                     HMMdejavu.add(s);
630                 }
631     
632                 // TODO: For this moment we don't know if the HMMS are in the end or
633                 // not
634                 String position = "i";
635                 int tmat = hmm.trIdx;
636     
637                 numStatePerHMM = hmm.getNstates();
638                 int[] stid = new int[hmm.getNbEmittingStates()];
639     
640                 int j = 0;
641                 for (int ii = 0; ii < numStatePerHMM; ii++) {
642                     if (hmm.isEmitting(ii)) {
643                         HMMState s = hmm.getState(ii);
644                         stid[j] = htkModels.hmmsHTK.getStateIdx(s);
645                         j++;
646                         // assert stid[j] >= 0 && stid[j] <
647                         // numContextIndependentTiedState;
648                     }
649                 }
650     
651                 if (useCDUnits) {
652                     Unit unit;
653                     String unitName = (name + ' ' + left + ' ' + right);
654     
655                     if (unitName.equals(lastUnitName)) {
656                         unit = lastUnit;
657                     } else {
658                         Unit[] leftContext = new Unit[1];
659                         leftContext[0] = contextIndependentUnits.get(left);
660     
661                         Unit[] rightContext = new Unit[1];
662                         rightContext[0] = contextIndependentUnits.get(right);
663     
664                         Context context = LeftRightContext.get(leftContext,
665                                 rightContext);
666                         unit = unitManager.getUnit(name, false, context);
667                     }
668                     lastUnitName = unitName;
669                     lastUnit = unit;
670     
671                     if (logger.isLoggable(Level.FINE)) {
672                         logger.fine("Loaded " + unit);
673                     }
674     
675                     float[][] transitionMatrix = matrixPool.get(tmat);
676     
677                     SenoneSequence ss = lastSenoneSequence;
678                     if (ss == null || !sameSenoneSequence(stid, lastStid)) {
679                         ss = getSenoneSequence(stid);
680                     }
681                     lastSenoneSequence = ss;
682                     lastStid = stid;
683     
684                     HMM hmm2 = new SenoneHMM(unit, ss, transitionMatrix,
685                             HMMPosition.lookup(position));
686                     hmmManager.put(hmm2);
687                 }
688             }
689         }

691         /**
692          * Returns true if the given senone sequence IDs are the same.
693          * 
694          * @return true if the given senone sequence IDs are the same, false
695          *         otherwise
696          */
697         protected boolean sameSenoneSequence(int[] ssid1, int[] ssid2) {
698             if (ssid1.length == ssid2.length) {
699                 for (int i = 0; i < ssid1.length; i++) {
700                     if (ssid1[i] != ssid2[i]) {
701                         return false;
702                     }
703                 }
704                 return true;
705             } else {
706                 return false;
707             }
708         }

716         protected SenoneSequence getSenoneSequence(int[] stateId) {

719                 senones[i] = senonePool.get(stateId[i]);

721             return new SenoneSequence(senones);

844             public Pool<float[]> htkMeans(String path) {
845                 Pool<float[]> pool = new Pool<float[]>(path);
846                 // Suppose this is the number of shared states
847                 int numStates = getNumStates();
848                 int numStreams = 1;
849                 int numGaussiansPerState = getGMMSize();
850                 pool.setFeature(NUM_SENONES, numStates);
851                 pool.setFeature(NUM_STREAMS, numStreams);
852                 pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
853                 int ncoefs = getNcoefs();
854                 for (int i = 0; i < numStates; i++) {
855                     GMMDiag gmm = hmmsHTK.gmms.get(i);
856                     for (int j = 0; j < numGaussiansPerState; j++) {
857                         float[] density = new float[ncoefs];
858                         for (int k = 0; k < ncoefs; k++) {
859                             density[k] = gmm.getMean(j, k);
860                         }
861                         int id = i * numGaussiansPerState + j;
862                         // the order of the means is the order in the HMMSet.gmms
863                         // vector which is the order of appearance in the MMF file
864                         pool.put(id, density);
865                     }
866                 }
867                 return pool;
868             }

920             public Pool<float[][]> htkTrans(String path) {
921                 Pool<float[][]> pool = new Pool<float[][]>(path);
922                 int numMatrices = getNumHMMs();
923                 int i = 0;
924     
925                 /*
926                  * the order of the transitions is first, the order of the trans
927                  * macro, and then the order of the trans in the HMMs... in other
928                  * words, this is simply the order of the trans in the MMF !
929                  */
930     
931                 // tied-trans
932                 if (hmmsHTK.transitions != null)
933                     for (; i < hmmsHTK.transitions.size(); i++) {
934                         float[][] tr = hmmsHTK.transitions.get(i);
935                         float[][] tmat = new float[tr.length][tr[0].length];
936                         for (int j = 0; j < tmat.length; j++)
937                             for (int k = 0; k < tmat[j].length; k++) {
938                                 tmat[j][k] = logMath.linearToLog(tr[j][k]);
939                             }
940                         pool.put(i, tmat);
941                     }
942     
943                 // untied-trans
944                 for (int l = 0; l < numMatrices; l++) {
945                     SingleHMM hmm = hmmsHTK.getHMM(l);
946                     if (hmm.trans != null) {
947                         float[][] tr = hmm.trans;
948                         float[][] tmat = new float[tr.length][tr[0].length];
949                         for (int j = 0; j < tmat.length; j++)
950                             for (int k = 0; k < tmat[j].length; k++) {
951                                 tmat[j][k] = logMath.linearToLog(tr[j][k]);
952                             }
953                         hmm.trIdx = i;
954                         pool.put(i++, tmat);
955                     } else {
956                         // The index in the pool is the same as the index in the
957                         // macros
958                         hmm.trIdx = hmm.getTransIdx();
959                     }
960                 }
961                 return pool;
962             }
