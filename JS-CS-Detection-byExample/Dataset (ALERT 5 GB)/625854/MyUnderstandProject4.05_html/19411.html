115     	public void load(String name) {
116     		try {
117     			BufferedReader fin = new BufferedReader(new FileReader(name));
118     			String s = fin.readLine();
119     			String[] ss = s.split(" ");
120     			ngauss = Integer.parseInt(ss[0]);
121     			ncoefs = Integer.parseInt(ss[1]);
122     			allocate();
123     			for (int i = 0; i < ngauss; i++) {
124     				s = fin.readLine();
125     				ss = s.split(" ");
126     				if (!ss[0].equals("gauss") || Integer.parseInt(ss[1]) != i) {
127     					System.err.println("Error loading GMM " + s + ' ' + i);
128     					System.exit(1);
129     				}
130     				setWeight(i, Float.parseFloat(ss[2]));
131     				// means
132     				s = fin.readLine();
133     				ss = s.split(" ");
134     				for (int j = 0; j < ncoefs; j++) {
135     					setMean(i, j, Float.parseFloat(ss[j]));
136     				}
137     				// covariances
138     				s = fin.readLine();
139     				ss = s.split(" ");
140     				for (int j = 0; j < ncoefs; j++) {
141     					setVar(i, j, Float.parseFloat(ss[j]));
142     				}
143     			}
144     			s = fin.readLine();
145     			if (s != null) {
146     				// can be added to store the amount of data on which the GMM has been
147     				// learned
148     				nT = Integer.parseInt(s);
149     			}
150     			fin.close();
151     			precomputeDistance();
152     		} catch (IOException e) {
153     			e.printStackTrace();
154     		}
155     	}

256     	public void loadHTK(String nom) {
257     		try {
258     			BufferedReader fin = new BufferedReader(new FileReader(nom));
259     			String s, s2;
260     			StringTokenizer st;
261     			ngauss = 0;
262     			ncoefs = 0;
263     			for (;;) {
264     				s = fin.readLine();
265     				if (s == null)
266     					break;
267     				if (s.contains("<MEAN>")) {
268     					ngauss++;
269     					if (ncoefs == 0) {
270     						st = new StringTokenizer(s);
271     						st.nextToken();
272     						ncoefs = Integer.parseInt(st.nextToken());
273     					}
274     				}
275     			}
276     			fin.close();
277     			allocate();
278     			fin = new BufferedReader(new FileReader(nom));
279     			for (int g = 0;;) {
280     				s = fin.readLine();
281     				if (s == null)
282     					break;
283     				if (s.contains("<MEAN>")) {
284     					s = fin.readLine();
285     					st = new StringTokenizer(s);
286     					for (int c = 0; st.hasMoreTokens(); c++) {
287     						s2 = st.nextToken();
288     						setMean(g, c, Float.parseFloat(s2));
289     					}
290     					s = fin.readLine();
291     					if (!s.contains("<VARIANCE>")) {
292     						throw new IOException();
293     					}
294     					s = fin.readLine();
295     					st = new StringTokenizer(s);
296     					for (int c = 0; st.hasMoreTokens(); c++) {
297     						s2 = st.nextToken();
298     						setVar(g, c, Float.parseFloat(s2));
299     					}
300     					g++;
301     				}
302     			}
303     			fin.close();
304     			precomputeDistance();
305     		} catch (IOException e) {
306     			e.printStackTrace();
307     		}
308     	}

355     	private void allocateWeights() {
356     		logMath = new LogMath();
357     		weights = new float[ngauss];
358     		for (int i = 0; i < ngauss; i++) {
359     			setWeight(i, 1f / (float) ngauss);
360     		}
361     	}

363     	public void precomputeDistance() {
364     		for (int gidx = 0; gidx < ngauss; gidx++) {
365     			float fact = 0.0f;
366     			for (int i = 0; i < ncoefs; i++) {
367     				fact += logMath.linearToLog(getVar(gidx, i));
368     			}
369     			fact += logMath.linearToLog(2.0 * Math.PI) * ncoefs;
370     			logPreComputedGaussianFactor[gidx] = fact * 0.5f;
371     		}
372     	}
373     
374     	private void allocate() {
375     		if (weights == null)
376     			allocateWeights();
377     		if (means == null) {
378     			loglikes = new float[ngauss];
379     			means = new float[ngauss][ncoefs];
380     			covar = new float[ngauss][ncoefs];
381     			logPreComputedGaussianFactor = new float[ngauss];
382     		}
383     	}

386     	 * Log likelihood calculation
387     	 */

389     	private static final float distFloor = -Float.MAX_VALUE;
390     
391     	public void computeLogLikes(float[] data) {
392     		float logDval1gauss = 0f;
393     		for (int gidx = 0; gidx < ngauss; gidx++) {
394     			logDval1gauss = 0f;
395     			for (int i = 0; i < data.length; i++) {
396     				float logDiff = data[i] - means[gidx][i];
397     				logDval1gauss += logDiff * logDiff * covar[gidx][i];
398     			}
399     			logDval1gauss -= logPreComputedGaussianFactor[gidx];
400     			if (Float.isNaN(logDval1gauss)) {
401     				System.err.println("gs2 is Nan, converting to 0 debug " + gidx
402     						+ ' ' + logPreComputedGaussianFactor[gidx] + ' '
403     						+ means[gidx][0] + ' ' + covar[gidx][0]);
404     				logDval1gauss = LogMath.getLogZero();
405     			}
406     			if (logDval1gauss < distFloor) {
407     				logDval1gauss = distFloor;
408     			}
409     			// Including apriori probability for each gaussian
410     			loglikes[gidx] = weights[gidx] + logDval1gauss;
411     		}
412     	}

415     	 * Calculate log probability of the observation
416     	 * must be called AFTER next() !
417     	 * 
418     	 * @return log likelihood

420     	public float getLogLike() {
421     		float sc = loglikes[0];
422     		for (int i = 1; i < ngauss; i++) {
423     			sc = logMath.addAsLinear(sc, loglikes[i]);
424     		}
425     		return sc;
426     	}

450     	public GMMDiag getMarginal(boolean[] mask) {
451     		int nc = 0;
452     		for (boolean flag : mask)
453     			if (flag)
454     				nc++;
455     		GMMDiag g = new GMMDiag(getNgauss(), nc);
456     		int curc = 0;
457     		for (int j = 0; j < ncoefs; j++) {
458     			if (mask[j]) {
459     				for (int i = 0; i < ngauss; i++) {
460     					g.setMean(i, curc, getMean(i, j));
461     					g.setVar(i, curc, getVar(i, j));
462     				}
463     				curc++;
464     			}
465     		}
466     		for (int i = 0; i < ngauss; i++) {
467     			g.setWeight(i, getWeight(i));
468     		}
469     		g.precomputeDistance();
470     		return g;
471     	}

474     	 * 
475     	 * @param g  second GMM for the merge
476     	 * @param w1 weight of the first GMM for the merge
477     	 * @return gaussian
478     	 */
479     	public GMMDiag merge(GMMDiag g, float w1) {
480     		GMMDiag res = new GMMDiag(getNgauss() + g.getNgauss(), getNcoefs());
481     		for (int i = 0; i < getNgauss(); i++) {
482     			System.arraycopy(means[i], 0, res.means[i], 0, getNcoefs());
483     			System.arraycopy(covar[i], 0, res.covar[i], 0, getNcoefs());
484     			res.setWeight(i, getWeight(i) * w1);
485     		}
486     		for (int i = 0; i < g.getNgauss(); i++) {
487     			System.arraycopy(g.means[i], 0, res.means[ngauss + i], 0,
488     					getNcoefs());
489     			System.arraycopy(g.covar[i], 0, res.covar[ngauss + i], 0,
490     					getNcoefs());
491     			res.setWeight(ngauss + i, g.getWeight(i) * (1f - w1));
492     		}
493     		res.precomputeDistance();
494     		return res;
495     	}

516     	/**
517     	 * 2 GMMs are considered to be equal when all of their parameters do not
518     	 * differ from more than 1%
519     	 * 
520     
521     	 * @return if GMMs are equal
522     	 */
523     	public boolean isEqual(GMMDiag g) {
524     		if (getNgauss() != g.getNgauss())
525     			return false;
526     		if (getNgauss() != g.getNcoefs())
527     			return false;
528     		for (int i = 0; i < getNgauss(); i++) {
529     			if (isDiff(getWeight(i), g.getWeight(i)))
530     				return false;
531     			for (int j = 0; j < getNcoefs(); j++) {
532     				if (isDiff(getMean(i, j), g.getMean(i, j)))
533     					return false;
534     				if (isDiff(getVar(i, j), g.getVar(i, j)))
535     					return false;
536     			}
537     		}
538     		return true;
539     	}
