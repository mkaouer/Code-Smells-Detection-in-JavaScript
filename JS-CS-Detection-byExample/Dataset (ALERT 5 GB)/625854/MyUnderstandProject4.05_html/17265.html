13     package edu.cmu.sphinx.research.parallel;
14     
15     import edu.cmu.sphinx.decoder.search.Token;
16     import edu.cmu.sphinx.linguist.flat.SentenceHMMState;
17     
18     
19     /**
20      * A Token for the within parallel stream sentence HMM states. The <code>getScore()</code> method returns the parallel
21      * stream score.
22      */
23     public class ParallelToken extends Token {
24     
25         private FeatureStream featureStream;
26         private float combinedScore;    // the combined score
27         private boolean pruned;         // is this token pruned?
28         private int lastCombineTime;    // the last combination time
29         private String lastCombineStamp;
30     
31     
32         /**
33          * Constructs a ParallelToken
34          *
35          * @param predecessor   the predecessor for this token
36          * @param state         the SentenceHMMState associated with this token
37          * @param featureScore  the score for this feature stream
38          * @param combinedScore the combinedScore
39          * @param frameNumber   the frame number associated with this token
40          */
41         public ParallelToken(ParallelToken predecessor,
42                              SentenceHMMState state,
43                              float featureScore,
44                              float combinedScore,
45                              int frameNumber) {
46             super(predecessor, state, featureScore, 0.f, 0.0f, frameNumber);
47             if (predecessor != null) {
48                 this.featureStream = predecessor.getFeatureStream();
49             }
50             this.combinedScore = combinedScore;
51             this.pruned = false;
52         }
53     
54     
55         /**
56          * Constructs a ParallelToken
57          *
58          * @param predecessor      the predecessor for this token
59          * @param state            the SentenceHMMState associated with this token
60          * @param featureScore     the score for this feature stream
61          * @param combinedScore    the combinedScore
62          * @param frameNumber      the frame number associated with this token
63          * @param lastCombineFrame the frame number at which score combination last occurred
64          */
65         public ParallelToken(ParallelToken predecessor,
66                              SentenceHMMState state,
67                              float featureScore,
68                              float combinedScore,
69                              int frameNumber,
70                              int lastCombineFrame) {
71             this(predecessor, state, featureScore, combinedScore, frameNumber);
72             this.lastCombineTime = lastCombineFrame;
73         }
74     
75     
76         /**
77          * Constructs a ParallelToken
78          *
79          * @param predecessor      the predecessor for this token
80          * @param state            the SentenceHMMState associated with this token
81          * @param featureScore     the score for this feature stream
82          * @param combinedScore    the combinedScore
83          * @param frameNumber      the frame number associated with this token
84          * @param lastCombineStamp the stamp used to identify the last score combination event
85          */
86         public ParallelToken(ParallelToken predecessor,
87                              SentenceHMMState state,
88                              float featureScore,
89                              float combinedScore,
90                              int frameNumber,
91                              String lastCombineStamp) {
92             this(predecessor, state, featureScore, combinedScore, -1);
93             this.lastCombineStamp = lastCombineStamp;
94         }
95     
96     
97         /**
98          * Constructs a ParallelToken with no predecessors, i.e., this is the ParallelToken at the first state of the
99          * SentenceHMM.
100          *
101          * @param state         the SentenceHMMState associated with this token
102          * @param featureStream the feature stream of this ParallelToken
103          * @param frameNumber   the frame number of this token
104          */
105         public ParallelToken(SentenceHMMState state, FeatureStream featureStream,
106                              int frameNumber) {
107             this(null, state, 0.0f, 0.0f, frameNumber);
108             this.featureStream = featureStream;
109         }
110     
111     
112         /**
113          * Returns the FeatureStream of this ParallelToken.
114          *
115          * @return the FeatureStream
116          */
117         public FeatureStream getFeatureStream() {
118             return featureStream;
119         }
120     
121     
122         /**
123          * Returns the eta factor for the feature stream of this ParallelToken.
124          *
125          * @return the eta factor
126          */
127         public float getEta() {
128             return featureStream.getEta();
129         }
130     
131     
132         /**
133          * Returns the combined score of this ParallelToken.
134          *
135          * @return the combined score of this ParallelToken
136          */
137         public float getCombinedScore() {
138             return combinedScore;
139         }
140     
141     
142         /**
143          * Returns the feature score of this ParallelToken.
144          *
145          * @return the feature score of this ParallelToken
146          */
147         public float getFeatureScore() {
148             return getScore();
149         }
150     
151     
152         /**
153          * Returns the name of the acoustic model.
154          *
155          * @return the name of the acoustic model
156          */
157         public String getModelName() {
158             return featureStream.getName();
159         }
160     
161     
162         /**
163          * Returns the last frame score combination takes place.
164          *
165          * @return the last frame score combination takes place
166          */
167         public int getLastCombineTime() {
168             return lastCombineTime;
169         }
170     
171     
172         /**
173          * Returns the stamp used to identify the last score combination event.
174          *
175          * @return the stamp used to identify the last score combination event
176          */
177         public String getLastCombineStamp() {
178             return lastCombineStamp;
179         }
180     
181     
182         /**
183          * Sets the combined score of this ParallelToken.
184          *
185          * @param combinedScore the new combined score
186          */
187         public void setCombinedScore(float combinedScore) {
188             this.combinedScore = combinedScore;
189         }
190     
191     
192         /**
193          * Sets the feature score of this ParallelToken.
194          *
195          * @param featureScore the new feature score
196          */
197         public void setFeatureScore(float featureScore) {
198             setScore(featureScore);
199         }
200     
201     
202         /**
203          * Sets the frame number at which score combination takes place.
204          *
205          * @param frameNumber the frame at which score combination last took place.
206          */
207         public void setLastCombineTime(int frameNumber) {
208             this.lastCombineTime = frameNumber;
209         }
210     
211     
212         /**
213          * Sets the stamp used to identify the last score combination event.
214          *
215          * @param stamp the stamp for identification
216          */
217         public void setLastCombineStamp(String stamp) {
218             this.lastCombineStamp = stamp;
219         }
220     
221     
222         /**
223          * Returns true if this ParallelToken has been pruned.
224          *
225          * @return true if this ParallelToken has been pruned.
226          */
227         public boolean isPruned() {
228             return pruned;
229         }
230     
231     
232         /**
233          * Sets this ParallelToken to be pruned.
234          *
235          * @param pruned sets this ParallelToken to be pruned or not
236          */
237         public void setPruned(boolean pruned) {
238             this.pruned = pruned;
239         }
240     }
