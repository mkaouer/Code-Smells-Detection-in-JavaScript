4     import javax.speech.recognition.ResultEvent;
5     import javax.speech.recognition.RuleGrammar;
6     
7     import edu.cmu.sphinx.decoder.ResultListener;
8     import edu.cmu.sphinx.result.Result;
9     import edu.cmu.sphinx.util.props.PropertyException;

12     /**
13      * Result listener for results from the sphinx recognizer.
14      *
15      * @author Dirk Schnelle-Walka
16      * @version $Revision: 1864 $
17      */
18     class Sphinx4ResultListener
19             implements ResultListener {
20         /** The recognizer which is notified when a result is obtained. */
21         private final SphinxRecognizer recognizer;
22     
23         /**
24          * Construct a new result listener.
25          * @param rec The recognizer.
26          */
27         public Sphinx4ResultListener(final SphinxRecognizer rec) {
28             recognizer = rec;
29         }
30     
31         /**
32          * Method called when a result is generated.
33          * @param result The new result.
34          */
35         public void newResult(final Result result) {
36             if (!result.isFinal()) {
37                 return;
38             }
39     
40             final RuleGrammar grammar = recognizer.getRuleGrammar();
41             final Sphinx4Result res = new Sphinx4Result(grammar, result);
42     
43             if (res.numTokens() == 0) {
44                 res.setResultState(javax.speech.recognition.Result.REJECTED);
45     
46                 final ResultEvent event =
47                         new ResultEvent(res, ResultEvent.RESULT_REJECTED);
48                 recognizer.fireResultRejected(event);
49             } else {
50                 res.setResultState(javax.speech.recognition.Result.ACCEPTED);
51     
52                 final ResultEvent event =
53                         new ResultEvent(res, ResultEvent.RESULT_ACCEPTED);
54                 recognizer.fireResultAccepted(event);
55             }
56     
57         }
58     
59         /**
60          * {@inheritDoc}
61          */
62         @Override
63         public void newProperties(final PropertySheet sheet)
64             throws PropertyException {
65         }
66     }
