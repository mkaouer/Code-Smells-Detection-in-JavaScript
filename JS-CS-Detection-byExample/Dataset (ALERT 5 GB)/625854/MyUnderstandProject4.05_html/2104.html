130         protected void initFrontEnd() {
131     
132             this.dataBlocker = new DataBlocker(
133                     10 // blockSizeMs
134             );
135             this.speechClassifier = new SpeechClassifier(
136                     10,     // frameLengthMs,
137                     0.003, // adjustment,
138                     10,     // threshold,
139                     0       // minSignal
140             );
141     
142             this.speechMarker = new SpeechMarker(
143                     200, // startSpeechTime,
144                     500, // endSilenceTime,
145                     100, // speechLeader,
146                     50,  // speechLeaderFrames
147                     100, // speechTrailer
148                     15.0 // endSilenceDecay
149             );
150     
151             this.nonSpeechDataFilter = new NonSpeechDataFilter();
152     
153             this.premphasizer = new Preemphasizer(
154                     0.97 // preemphasisFactor
155             );
156             this.windower = new RaisedCosineWindower(
157                     0.46, // double alpha
158                     25.625f, // windowSizeInMs
159                     10.0f // windowShiftInMs
160             );
161             this.fft = new DiscreteFourierTransform(
162                     -1, // numberFftPoints
163                     false // invert
164             );
165             this.melFilterBank = new MelFrequencyFilterBank(
166                     130.0, // minFreq,
167                     6800.0, // maxFreq,
168                     40 // numberFilters
169             );
170             this.dct = new DiscreteCosineTransform(
171                     40, // numberMelFilters,
172                     13  // cepstrumSize
173             );
174             this.cmn = new LiveCMN(
175                     12.0, // initialMean,
176                     100,  // cmnWindow,
177                     160   // cmnShiftWindow
178             );
179             this.featureExtraction = new DeltasFeatureExtractor(
180                     3 // window
181             );
182     
183             ArrayList<DataProcessor> pipeline = new ArrayList<DataProcessor>();
184             pipeline.add(audioDataSource);
185             pipeline.add(dataBlocker);
186             pipeline.add(speechClassifier);
187             pipeline.add(speechMarker);
188             pipeline.add(nonSpeechDataFilter);
189             pipeline.add(premphasizer);
190             pipeline.add(windower);
191             pipeline.add(fft);
192             pipeline.add(melFilterBank);
193             pipeline.add(dct);
194             pipeline.add(cmn);
195             pipeline.add(featureExtraction);
196     
197             this.frontend = new FrontEnd(pipeline);
198     
199         }

201         protected void initRecognizer() {
202     
203             this.scorer = new ThreadedAcousticScorer(frontend, null, 10, true, 0, Thread.NORM_PRIORITY);
204     
205             this.pruner = new SimplePruner();
206     
207             this.activeListFactory = new PartitionActiveListFactory(absoluteBeamWidth, relativeBeamWidth, logMath);
208     
209             this.searchManager = new SimpleBreadthFirstSearchManager(
210                     logMath, linguist, pruner,
211                     scorer, activeListFactory,
212                     false, 0.0, 0, false);
213     
214             this.decoder = new Decoder(searchManager,
215                     false, false,
216                     new ArrayList<ResultListener>(),
217                     100000);
218     
219             this.recognizer = new Recognizer(decoder, monitors);
220     
221             this.monitors = new ArrayList<Monitor>();
222             this.monitors.add(new BestPathAccuracyTracker(recognizer, false, false, false, false, false, false));
223             this.monitors.add(new MemoryTracker(recognizer, false, false));
224             this.monitors.add(new SpeedTracker(recognizer, frontend, true, false, false, false));
225         }
