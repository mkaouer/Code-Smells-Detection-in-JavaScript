152         /**
153          * Gets the ngram probability of the word sequence represented by the word list
154          *
155          * @param wordSequence the word sequence
156          * @return the probability of the word sequence. Probability is in logMath log base
157          */
158         @Override
159         public float getProbability(WordSequence wordSequence) {
160             float logProbability = 0.0f;
161             Probability prob = getProb(wordSequence);
162             if (prob == null) {
163                 if (wordSequence.size() > 1) {
164                     logProbability = getBackoff(wordSequence.getOldest())
165                             + getProbability(wordSequence.getNewest());
166                 } else { // if the single word is not in the model at all
167                     // then its zero likelihood that we'll use it
168                     logProbability = LogMath.getLogZero();
169                 }
170             } else {
171                 logProbability = prob.logProbability;
172             }
173              //    System.out.println("Search: " + wordSequence + " : "
174              //           + logProbability + " "
175              //           + logMath.logToLinear(logProbability));
176              return logProbability;
177         }

288         /**
289          * Loads the language model from the given location.
290          *
291          * @param location      the URL location of the model
292          * @param unigramWeight the unigram weight
293          * @throws IOException if an error occurs while loading
294          */
295         private void load(URL location, float unigramWeight,
296                           Dictionary dictionary) throws IOException {
297             String line;
298             float logUnigramWeight = logMath.linearToLog(unigramWeight);
299             float inverseLogUnigramWeight = logMath
300                     .linearToLog(1.0 - unigramWeight);
301     
302             open(location);
303             // look for beginning of data
304             readUntil("\\data\\");
305             // look for ngram statements
306             List<Integer> ngramList = new ArrayList<Integer>();
307             while ((line = readLine()) != null) {
308                 if (line.startsWith("ngram")) {
309                     StringTokenizer st = new StringTokenizer(line, " \t\n\r\f=");
310                     if (st.countTokens() != 3) {
311                         corrupt("corrupt ngram field " + line + ' '
312                                 + st.countTokens());
313                     }
314                     st.nextToken();
315                     int index = Integer.parseInt(st.nextToken());
316                     int count = Integer.parseInt(st.nextToken());
317                     ngramList.add(index - 1, count);
318                     if (index > maxNGram) {
319                         maxNGram = index;
320                     }
321                 } else if (line.equals("\\1-grams:")) {
322                     break;
323                 }
324             }
325             int numUnigrams = ngramList.get(0) - 1;
326             // -log(x) = log(1/x)
327             float logUniformProbability = -logMath.linearToLog(numUnigrams);
328             for (int index = 0; index < ngramList.size(); index++) {
329                 int ngram = index + 1;
330                 int ngramCount = ngramList.get(index);
331                 for (int i = 0; i < ngramCount; i++) {
332                     StringTokenizer tok = new StringTokenizer(readLine());
333                     int tokenCount = tok.countTokens();
334                     if (tokenCount != ngram + 1 && tokenCount != ngram + 2) {
335                         corrupt("Bad format");
336                     }
337                     float log10Prob = Float.parseFloat(tok.nextToken());
338                     float log10Backoff = 0.0f;
339                     // construct the WordSequence for this N-Gram
340                     List<Word> wordList = new ArrayList<Word>(maxNGram);
341                     for (int j = 0; j < ngram; j++) {
342                         String word = tok.nextToken().toLowerCase();
343                         vocabulary.add(word);
344                         Word wordObject = dictionary.getWord(word);
345                         if (wordObject == null) {
346                             wordObject = Word.UNKNOWN;
347                         }
348                         wordList.add(wordObject);
349                     }
350                     WordSequence wordSequence = new WordSequence(wordList);
351                     if (tok.hasMoreTokens()) {
352                         log10Backoff = Float.parseFloat(tok.nextToken());
353                     }
354                     float logProb = logMath.log10ToLog(log10Prob);
355                     float logBackoff = logMath.log10ToLog(log10Backoff);
356                     // Apply unigram weights if this is a unigram probability
357                     if (ngram == 1) {
358                         float p1 = logProb + logUnigramWeight;
359                         float p2 = logUniformProbability + inverseLogUnigramWeight;
360                         logProb = logMath.addAsLinear(p1, p2);
361     //                    System.out
362     //                    .println("p1 " + p1 + " p2 " + p2 + " luw "
363     //                    		+ logUnigramWeight + " iluw "
364     //                    		+ inverseLogUnigramWeight + " lup "
365     //                    		+ logUniformProbability + " logprog "
366     //                    		+ logProb);
367                     }
368                     put(wordSequence, logProb, logBackoff);
369                 }
370                 if (index < ngramList.size() - 1) {
371                     String next = "\\" + (ngram + 1) + "-grams:";
372                     readUntil(next);
373                 }
374             }
375             readUntil("\\end\\");
376             close();
377         }

383          * @param wordSequence the tag for the prob.
384          * @param logProb      the probability in log math base
385          * @param logBackoff   the backoff probability in log math base

387         private void put(WordSequence wordSequence, float logProb, float logBackoff) {
388     //        System.out.println("Putting " + wordSequence + " p " + logProb

390             map.put(wordSequence, new Probability(logProb, logBackoff));

394         /**
395          * Reads the next line from the LM file. Keeps track of line number.
396          *
397          * @throws IOException if an error occurs while reading the input or an EOF is encountered.
398          */
399         private String readLine() throws IOException {
400             String line;
401             lineNumber++;
402             line = reader.readLine();
403             if (line == null) {
404                 corrupt("Premature EOF");
405             }
406             return line;
407         }

410         /**
411          * Opens the language model at the given location
412          *
413          * @param location the path to the language model
414          * @throws IOException if an error occurs while opening the file
415          */
416         private void open(URL location) throws
417                 IOException {
418             lineNumber = 0;
419             fileName = location.toString();
420             reader = new BufferedReader
421                     (new InputStreamReader(location.openStream()));
422         }

425         /**
426          * Reads from the input stream until the input matches the given string
427          *
428          * @param match the string to match on
429          * @throws IOException if an error occurs while reading the input or an EOF is encountered before finding the match
430          */
431         private void readUntil(String match) throws IOException {
432             try {
433                 while (!readLine().equals(match)) {
434                 }
435             } catch (IOException ioe) {
436                 corrupt("Premature EOF while waiting for " + match);
437             }
438         }

442          * Closes the language model file
443          *
444          * @throws IOException if an error occurs

446         private void close() throws IOException {
447             reader.close();
448             reader = null;
449         }
