95          * @param mean                         the mean vector for this PDF
96          * @param meanTransformationMatrix     transformation matrix for this pdf

98          * @param variance                     the variance for this PDF
99          * @param varianceTransformationMatrix var. transform matrix for this PDF

101          * @param distFloor                    the lowest score value (in linear domain)
102          * @param varianceFloor                the lowest value for the variance
103          */
104         public MixtureComponent(
105                 LogMath logMath,
106                 float[] mean,
107                 float[][] meanTransformationMatrix,
108                 float[] meanTransformationVector,
109                 float[] variance,
110                 float[][] varianceTransformationMatrix,
111                 float[] varianceTransformationVector,
112                 float distFloor,
113                 float varianceFloor) {
114     
115             assert variance.length == mean.length;
116     
117             this.logMath = logMath;
118             this.mean = mean;
119             this.meanTransformationMatrix = meanTransformationMatrix;
120             this.meanTransformationVector = meanTransformationVector;
121             this.variance = variance;
122             this.varianceTransformationMatrix = varianceTransformationMatrix;
123             this.varianceTransformationVector = varianceTransformationVector;
124     
125             assert distFloor >= 0.0 : "distFloot seems to be already in log-domain";
126             this.distFloor = logMath.linearToLog(distFloor);
127             this.varianceFloor = varianceFloor;
128     
129             transformStats();
130     
131             logPreComputedGaussianFactor = precomputeDistance();
132         }

170         /**
171          * Calculate the score for this mixture against the given feature. We model the output
172          * distributions using a mixture of Gaussians, therefore the current implementation is simply
173          * the computation of a multi-dimensional Gaussian. <p/> <p><b>Normal(x) = exp{-0.5 * (x-m)' *
174          * inv(Var) * (x-m)} / {sqrt((2 * PI) ^ N) * det(Var))}</b></p>
175          * <p/>
176          * where <b>x</b> and <b>m</b> are the incoming cepstra and mean vector respectivally,
177          * <b>Var</b> is the Covariance matrix, <b>det()</b> is the determinant of a matrix,
178          * <b>inv()</b> is its inverse, <b>exp</b> is the exponential operator, <b>x'</b> is the
179          * transposed vector of <b>x</b> and <b>N</b> is the dimension of the vectors <b>x</b> and
180          * <b>m</b>.
181          *
182          * @param feature the feature to score
183          * @return the score, in log, for the given feature
184          */
185         public float getScore(float[] feature) {
186             // float logVal = 0.0f;
187             float logDval = 0.0f;
188     
189             // First, compute the argument of the exponential function in
190             // the definition of the Gaussian, then convert it to the
191             // appropriate base. If the log base is <code>Math.E</code>,
192             // then no operation is necessary.
193     
194             for (int i = 0; i < feature.length; i++) {
195                 float logDiff = feature[i] - meanTransformed[i];
196                 logDval += logDiff * logDiff * precisionTransformed[i];
197             }
198             // logDval = -logVal / 2;
199     
200             // At this point, we have the ln() of what we need, that is,
201             // the argument of the exponential in the javadoc comment.
202     
203             // Convert to the appropriate base.
204             logDval = logMath.lnToLog(logDval);
205     
206             // Add the precomputed factor, with the appropriate sign.
207             logDval -= logPreComputedGaussianFactor;
208     
209             // System.out.println("MC: getscore " + logDval);
210     
211             // TODO: Need to use mean and variance transforms here
212     
213             if (Float.isNaN(logDval)) {
214                 System.out.println("gs is Nan, converting to 0");
215                 logDval = LogMath.getLogZero();
216             }
217     
218             if (logDval < distFloor) {
219                 logDval = distFloor;
220             }
221     
222             return logDval;
223         }

226         /**
227          * Pre-compute factors for the Mahalanobis distance. Some of the Mahalanobis distance
228          * computation can be carried out in advance. Specifically, the factor containing only variance
229          * in the Gaussian can be computed in advance, keeping in mind that the the determinant of the
230          * covariance matrix, for the degenerate case of a mixture with independent components - only
231          * the diagonal elements are non-zero - is simply the product of the diagonal elements. <p/>
232          * We're computing the expression: <p/> <p><b>{sqrt((2 * PI) ^ N) * det(Var))}</b></p>
233          *
234          * @return the precomputed distance
235          */
236         public float precomputeDistance() {
237             float logPreComputedGaussianFactor = 0.0f; // = log(1.0)
238             // Compute the product of the elements in the Covariance
239             // matrix's main diagonal. Covariance matrix is assumed
240             // diagonal - independent dimensions. In log, the product
241             // becomes a summation.
242             for (int i = 0; i < variance.length; i++) {
243                 logPreComputedGaussianFactor +=
244                         logMath.linearToLog(precisionTransformed[i] * -2);
245                 //	     variance[i] = 1.0f / (variance[i] * 2.0f);
246             }
247     
248             // We need the minus sign since we computed
249             // logPreComputedGaussianFactor based on precision, which is
250             // the inverse of the variance. Therefore, in the log domain,
251             // the two quantities have opposite signs.
252     
253             // The covariance matrix's dimension becomes a multiplicative
254             // factor in log scale.
255             logPreComputedGaussianFactor =
256                     logMath.linearToLog(2.0 * Math.PI) * variance.length
257                             - logPreComputedGaussianFactor;
258     
259             // The sqrt above is a 0.5 multiplicative factor in log scale.
260             return logPreComputedGaussianFactor * 0.5f;
261         }

264         /** Applies transformations to means and variances. */
265         public void transformStats() {
266             int featDim = mean.length;
267             /*
268             * The transformed mean vector is given by:
269             *
270             * <p><b>M = A * m + B</b></p>
271             *
272             * where <b>M</b> and <b>m</b> are the mean vector after and
273             * before transformation, respectively, and <b>A</b> and
274             * <b>B</b> are the transformation matrix and vector,
275             * respectively.
276             *
277             * if A or B are <code>null</code> the according substeps are skipped
278             */
279             if (meanTransformationMatrix != null) {
280                 meanTransformed = new float[featDim];
281                 for (int i = 0; i < featDim; i++)
282                     for (int j = 0; j < featDim; j++)
283                         meanTransformed[i] += mean[j] * meanTransformationMatrix[i][j];
284             } else {
285                 meanTransformed = mean;
286             }
287     
288             if (meanTransformationVector != null)
289                 for (int k = 0; k < featDim; k++)
290                     meanTransformed[k] += meanTransformationVector[k];
291     
292             /**
293              * We do analogously with the variance. In this case, we also
294              * invert the variance, and work with precision instead of
295              * variance.
296              */
297             if (varianceTransformationMatrix != null) {
298                 precisionTransformed = new float[variance.length];
299                 for (int i = 0; i < featDim; i++)
300                     for (int j = 0; j < featDim; j++)
301                         precisionTransformed[i] += variance[j] * varianceTransformationMatrix[i][j];
302             } else
303                 precisionTransformed = variance.clone();
304     
305             if (varianceTransformationVector != null)
306                 for (int k = 0; k < featDim; k++)
307                     precisionTransformed[k] += varianceTransformationVector[k];
308     
309             for (int k = 0; k < featDim; k++) {
310                 float flooredPrecision = (precisionTransformed[k] < varianceFloor ? varianceFloor : precisionTransformed[k]);
311                 precisionTransformed[k] = 1.0f / (-2.0f * flooredPrecision);
312             }
313         }

315         @Override
316         public MixtureComponent clone() throws CloneNotSupportedException {
317             MixtureComponent mixComp = (MixtureComponent)super.clone();
318     
319             mixComp.distFloor = distFloor;
320             mixComp.varianceFloor = varianceFloor;
321             mixComp.logPreComputedGaussianFactor = logPreComputedGaussianFactor;
322     
323             mixComp.mean = this.mean != null ? this.mean.clone() : null;
324             if (meanTransformationMatrix != null) {
325                 mixComp.meanTransformationMatrix = this.meanTransformationMatrix.clone();
326                 for (int i = 0; i < meanTransformationMatrix.length; i++)
327                     mixComp.meanTransformationMatrix[i] = meanTransformationMatrix[i].clone();
328             }
329             mixComp.meanTransformationVector = this.meanTransformationVector != null ?
330                     this.meanTransformationVector.clone() : null;
331             mixComp.meanTransformed = this.meanTransformed != null ? this.meanTransformed.clone() : null;
332     
333             mixComp.variance = this.variance != null ? this.variance.clone() : null;
334             if (varianceTransformationMatrix != null) {
335                 mixComp.varianceTransformationMatrix = this.varianceTransformationMatrix.clone();
336                 for (int i = 0; i < varianceTransformationMatrix.length; i++)
337                     mixComp.varianceTransformationMatrix[i] = varianceTransformationMatrix[i].clone();
338             }
339             mixComp.varianceTransformationVector = this.varianceTransformationVector != null ?
340                     this.varianceTransformationVector.clone() : null;
341             mixComp.precisionTransformed = this.precisionTransformed != null ?
342                     this.precisionTransformed.clone() : null;
343     
344             return mixComp;
345         }
