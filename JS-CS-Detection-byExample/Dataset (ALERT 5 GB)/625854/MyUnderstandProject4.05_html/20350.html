30     /**
31      * Creates a grammar from a text. This grammar could be used to create a precise
32      * recognizer that will force align the text with the audio.
33      */
34     public class TextAlignerGrammar extends Grammar implements ResultListener {
35     
36         /** The property that defines the logMath component. */
37         @S4Component(type = LogMath.class)
38         public final static String PROP_LOG_MATH = "logMath";
39         
40         protected GrammarNode finalNode;
41         private LogMath logMath;
42     
43         private final List<String> tokens = new ArrayList<String>();
44     
45         private int start;
46         private final int step = 40;
47     
48         public TextAlignerGrammar(final String text, final LogMath logMath, final boolean showGrammar, final boolean optimizeGrammar,
49                 final boolean addSilenceWords, final boolean addFillerWords, final Dictionary dictionary) {
50             super(showGrammar, optimizeGrammar, addSilenceWords, addFillerWords, dictionary);
51             this.logMath = logMath;
52             setText(text);
53         }
54     
55         public TextAlignerGrammar() {
56         }
57     
58         /*
59          * (non-Javadoc)
60          *
61          * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
62          */
63          @Override
64          public void newProperties(PropertySheet ps) throws PropertyException {
65              super.newProperties(ps);
66              logMath = (LogMath) ps.getComponent(PROP_LOG_MATH);
67          }
68         
69         public void setText(String text) {
70             String word;
71             try {
72                 final ExtendedStreamTokenizer tok = new ExtendedStreamTokenizer(new StringReader(text), true);
73     
74                 tokens.clear();
75                 while (!tok.isEOF()) {
76                     while ((word = tok.getString()) != null) {
77                         word = word.toLowerCase();
78                         tokens.add(word);
79                     }
80                 }
81             } catch (Exception e) {
82                 e.printStackTrace();
83             }
84         }
85     
86         /**
87          * Create a branch of the grammar that corresponds to a transcript. For each
88          * word create a node, and link the nodes with arcs. The branch is connected
89          * to the initial node and to the final node.
90          * 
91          * @return the first node of this branch
92          */
93         protected GrammarNode createGrammar() {
94     
95             initialNode = createGrammarNode(Dictionary.SILENCE_SPELLING);
96             finalNode = createGrammarNode(Dictionary.SILENCE_SPELLING);
97             finalNode.setFinalNode(true);
98             final GrammarNode branchNode = createGrammarNode(false);
99     
100             final List<GrammarNode> wordGrammarNodes = new ArrayList<GrammarNode>();
101             final int end = start + step > tokens.size() ? tokens.size() : start + step;
102             for (final String word : tokens.subList(start, end)) {
103                 // System.out.println ("Creating grammar from " + word);
104                 final GrammarNode wordNode = createGrammarNode(word);
105                 wordGrammarNodes.add(wordNode);
106             }
107     
108             // now connect all the GrammarNodes together
109             initialNode.add(branchNode, LogMath.getLogOne());
110             final float branchScore = logMath.linearToLog(1.0 / wordGrammarNodes.size());
111     
112             for (int i = 0; i < wordGrammarNodes.size(); i++) {
113                 final GrammarNode wordNode = wordGrammarNodes.get(i);
114     
115                 branchNode.add(wordNode, branchScore);
116                 wordNode.add(finalNode, LogMath.getLogOne());
117     
118                 // add connections to close words
119                 for (int j = i + 1; j < i + 2; j++) {
120                     if (0 <= j && j < wordGrammarNodes.size()) {
121                         final GrammarNode neighbour = wordGrammarNodes.get(j);
122                         wordNode.add(neighbour, LogMath.getLogOne());
123                     }
124                 }
125             }
126             return initialNode;
127         }
128     
129         @Override
130         public void newResult(final Result result) {
131             if (result == null)
132                 return;
133     
134             int max = -1;
135             for (final String word : result.toString().split(" ")) {
136                 for (int i = start; i < start + step && i < tokens.size(); i++) {
137                     if (tokens.get(i).equals(word)) {
138                         max = i;
139                         break;
140                     }
141                 }
142             }
143             if (max > 0) {
144                 start = max;
145                 newGrammar();
146                 createGrammar();
147                 postProcessGrammar();
148             }
149         }
150     }
