10     /**
11      * A tool for tracking the types tokens created and placed in the beam
12      * <p/>
13      * TODO: Develop a mechanism  for adding trackers such as these in a more general fashion.
14      */
15     public class TokenTypeTracker {
16         // keep track of the various types of states
17     
18         private int numWords;
19         private int numUnits;
20         private int numHMMs;
21         private int numOthers;
22         private int numHMMBegin;
23         private int numHMMEnd;
24         private int numHMMSingle;
25         private int numHMMInternal;
26         private int numTokens;
27     
28     
29         /**
30          * Adds a token to this tracker. Records statistics about the type of token.
31          *
32          * @param t the token to track
33          */
34         public void add(Token t) {
35             numTokens++;
36             SearchState s = t.getSearchState();
37     
38             if (s instanceof WordSearchState) {
39                 numWords++;
40             } else if (s instanceof UnitSearchState) {
41                 numUnits++;
42             } else if (s instanceof HMMSearchState) {
43                 numHMMs++;
44                 HMM hmm = ((HMMSearchState) s).getHMMState().getHMM();
45                 switch (hmm.getPosition()) {
46                     case BEGIN: numHMMBegin++; break;
47                     case END: numHMMEnd++; break;
48                     case SINGLE: numHMMSingle++; break;
49                     case INTERNAL: numHMMInternal++; break;
50                 }
51             } else {
52                 numOthers++;
53             }
54         }
55     
56     
57         /** Shows the accumulated statistics */
58         public void dump() {
59             System.out.println("TotalTokens: " + numTokens);
60             System.out.println("      Words: " + numWords + pc(numWords));
61             System.out.println("      Units: " + numUnits + pc(numUnits));
62             System.out.println("      HMM-b: " + numHMMBegin + pc(numHMMBegin));
63             System.out.println("      HMM-e: " + numHMMEnd + pc(numHMMEnd));
64             System.out.println("      HMM-s: " + numHMMSingle + pc(numHMMSingle));
65             System.out.println("      HMM-i: " + numHMMInternal +
66                     pc(numHMMInternal));
67             System.out.println("     Others: " + numOthers + pc(numOthers));
68         }
69     
70     
71         /**
72          * Utility method for generating integer percents
73          *
74          * @param num the value to be converted into percent
75          * @return a string representation as a percent
76          */
77         private String pc(int num) {
78             int percent = ((100 * num) / numTokens);
79             return " (" + percent + "%)";
80         }
