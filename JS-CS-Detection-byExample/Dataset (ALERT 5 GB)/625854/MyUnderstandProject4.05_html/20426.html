12     package edu.cmu.sphinx.demo.raw;
13     
14     import edu.cmu.sphinx.frontend.util.AudioFileDataSource;
15     import edu.cmu.sphinx.recognizer.Recognizer;
16     import edu.cmu.sphinx.result.Result;
17     import edu.cmu.sphinx.demo.raw.TranscriberConfiguration;
18     import edu.cmu.sphinx.util.props.ConfigurationManagerUtils;
19     
20     import javax.sound.sampled.UnsupportedAudioFileException;
21     import java.io.IOException;
22     import java.net.URISyntaxException;
23     import java.net.URL;
24     import java.net.MalformedURLException;
25     
26     /** 
27      * @author Peter Wolf
28      */
29     public class RawTranscriber {
30     
31         public static void main(String[] args) throws IOException, UnsupportedAudioFileException, URISyntaxException, ClassNotFoundException {
32             new RawTranscriber().run(args);
33         }
34     
35         public void run(String[] args) throws MalformedURLException, URISyntaxException, ClassNotFoundException {
36             System.out.println("\nRunning " + getClass().getSimpleName() + "...");
37     
38             URL audioURL;
39             if (args.length > 2) {
40                 throw new Error("USAGE: " + getClass().getSimpleName() + " [<WAV file>]");
41             }
42             else if(args.length == 2) {
43                 audioURL = new URL(args[1]);
44             }
45             else {
46                 audioURL = ConfigurationManagerUtils.resourceToURL("resource:/edu/cmu/sphinx/demo/transcriber/10001-90210-01803.wav");
47             }
48     
49             // get a recognizer configured for the Transcriber task
50             TranscriberConfiguration config = new TranscriberConfiguration();
51             Recognizer recognizer = config.getRecognizer();
52     
53             // allocate the resource necessary for the recognizer
54             recognizer.allocate();
55     
56             // configure the audio input for the recognizer
57             AudioFileDataSource audioSource = config.getAudioFileDataSource();
58             audioSource.setAudioFile(audioURL, null);
59     
60             // Loop until last utterance in the audio file has been decoded, in
61             // which case the recognizer will return null.
62             Result result;
63             while ((result = recognizer.recognize()) != null) {
64                 String resultText = result.getBestResultNoFiller();
65                 System.out.println(resultText);
66             }
67         }

69     }
