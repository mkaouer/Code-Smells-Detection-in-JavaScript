8     package com.sun.speech.engine;
9     
10     import java.util.ArrayList;
11     import java.util.List;
12     
13     import javax.speech.AudioListener;
14     import javax.speech.AudioManager;
15     
16     /**
17      * Supports the JSAPI 1.0 <code>AudioManager</code> interface. 
18      * Actual JSAPI implementations might want to extend or
19      * modify this implementation.
20      */
21     public class BaseAudioManager implements AudioManager {
22     
23         /** List of <code>AudioListeners</code> registered for <code>AudioEvents</code> on this object. */
24         protected final List<AudioListener> listeners  = new ArrayList<AudioListener>();
25     
26         /**
27          * Requests notification of <code>AudioEvents</code> from the <code>AudioManager</code>.
28          *
29          * @param listener the listener to add
30          */
31         public void addAudioListener(AudioListener listener) {
32             if (!listeners.contains(listener)) {
33                 listeners.add(listener);
34             }
35         }
36     
37         /**
38          * Removes an <code>AudioListener</code> from the list of <code>AudioListeners</code>.
39          *
40          * @param listener the listener to remove
41          */
42         public void removeAudioListener(AudioListener listener) {
43             listeners.remove(listener);
44         }
45     }
