13     package edu.cmu.sphinx.trainer;
14     
15     
16     /** Interface to the UtteranceGraph, a graph of an utterance. */
17     public interface UtteranceGraph {
18     
19         /**
20          * Add a transcript graph to the current utterance graph.
21          *
22          * @param transcriptGraph the transcript graph to add
23          */
24         public void add(Graph transcriptGraph);
25     
26     
27         /** Returns the size of a graph. This is the number of nodes in a graph. */
28         public int size();
29     
30     
31         /**
32          * Get node at the specified position in the list. The order is the same in which the nodes were entered.
33          *
34          * @param index index of item to retun
35          * @return the node
36          */
37         public Node getNode(int index);
38     
39     
40         /**
41          * Gets the initial node in this graph
42          *
43          * @return the initial node
44          */
45         public Node getInitialNode();
46     
47     
48         /**
49          * Gets the final node in this graph
50          *
51          * @return the final node
52          */
53         public Node getFinalNode();
54     
55     
56         /**
57          * Returns whether the given node is the initial node in this graph.
58          *
59          * @param node the node we want to compare
60          * @return if true, the node is the initial node
61          */
62         public boolean isInitialNode(Node node);
63     
64     
65         /**
66          * Returns whether the given node is the final node in this graph.
67          *
68          * @param node the node we want to compare
69          * @return if true, the node is the final node
70          */
71         public boolean isFinalNode(Node node);
72     
73     
74         /**
75          * Gets the index of a particular node in the graph.
76          *
77          * @param node the node
78          * @return the index
79          */
80         public int indexOf(Node node);
81     
82     
83         /**
84          * Validate the graph. It checks out basics about the graph, such as whether all nodes have at least one incoming
85          * and outgoing edge, except for the initial and final.
86          *
87          * @return if true, graph validation passed
88          */
89         public boolean validate();
90     }
