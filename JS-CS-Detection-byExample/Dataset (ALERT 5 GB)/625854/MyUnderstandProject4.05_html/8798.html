10      * A <code>DataProcessor</code> which wraps incoming <code>DoubleData</code>-objects into equally size blocks of defined
11      * length.
12      */
13     public class DataBlocker extends BaseDataProcessor {
14     
15         /** The property for the block size of generated data-blocks in milliseconds. */
16         @S4Double(defaultValue = 10)
17         public static final String PROP_BLOCK_SIZE_MS = "blockSizeMs";

20         private int blockSizeSamples = Integer.MAX_VALUE;

22         private int curFirstSamplePos;
23         private int sampleRate = -1;
24     
25         private final LinkedList<DoubleData> inBuffer = new LinkedList<DoubleData>();
26

29

31         }

34          * 
35          * @param blockSizeMs
36          */
37         public DataBlocker(double blockSizeMs) {
38             initLogger();
39             this.blockSizeMs = blockSizeMs;
40         }

42         @Override
43         public void newProperties(PropertySheet propertySheet) throws PropertyException {
44             super.newProperties(propertySheet);
45             blockSizeMs = propertySheet.getDouble(PROP_BLOCK_SIZE_MS);
46         }
47     
48     
49         public double getBlockSizeMs() {
50             return blockSizeMs;
51         }

54         @Override
55         public Data getData() throws DataProcessingException {
56             while (curInBufferSize < blockSizeSamples || curInBufferSize == 0) {
57                 Data data = getPredecessor().getData();
58     
59                 if (data instanceof DataStartSignal) {
60                     sampleRate = ((DataStartSignal) data).getSampleRate();
61                     blockSizeSamples = (int) Math.round(sampleRate * blockSizeMs / 1000);
62     
63                     curInBufferSize = 0;
64                     curFirstSamplePos = 0;
65                     
66                     inBuffer.clear();
67                 }
68     
69                 if (!(data instanceof DoubleData)) {
70                     return data;
71                 }
72     
73                 DoubleData dd = (DoubleData) data;
74     
75                 inBuffer.add(dd);
76                 curInBufferSize += dd.getValues().length;
77             }
78     
79             // now we are ready to merge all data blocks into one
80             double[] newSampleBlock = new double[blockSizeSamples];
81     
82             int copiedSamples = 0;
83     
84             long firstSample = inBuffer.get(0).getFirstSampleNumber() + curFirstSamplePos;
85             long collectTime = inBuffer.get(0).getCollectTime();
86     
87             while (!inBuffer.isEmpty()) {
88                 DoubleData dd = inBuffer.remove(0);
89                 double[] values = dd.getValues();
90                 int copyLength = Math.min(blockSizeSamples - copiedSamples, values.length - curFirstSamplePos);
91     
92                 System.arraycopy(values, curFirstSamplePos, newSampleBlock, copiedSamples, copyLength);
93     
94                 // does the current data-object contains more samples than necessary? -> keep the rest for the next block
95                 if (copyLength < (values.length - curFirstSamplePos)) {
96                     assert inBuffer.isEmpty();
97     
98                     curFirstSamplePos += copyLength;
99                     inBuffer.add(0, dd);
100                     break;
101                 } else {
102                     copiedSamples += copyLength;
103                     curFirstSamplePos = 0;
104                 }
105             }
106     
107             curInBufferSize = inBuffer.isEmpty() ? 0 : inBuffer.get(0).getValues().length - curFirstSamplePos;
108     
109     //        for (int i = 0; i < newSampleBlock.length; i++) {
110     //            newSampleBlock[i] *= 10;
111     //        }
112             return new DoubleData(newSampleBlock, sampleRate, collectTime, firstSample);
113         }
