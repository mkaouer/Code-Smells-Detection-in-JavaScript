8     /**
9      * A VU meter to be plugged into a front-end. Preferably this component should be plugged directly behind the
10      * <code>DataBlocker</code> in order to ensure that only equally sized blockes of meaningful length are used for RMS
11      * computation.
12      * <p/>
13      * Because vu-monitoring makes sense only for online speech processing the vu-meter will be visible only if data source
14      * which preceeds it is a <code>Microphone</code>.
15      *
16      * @author Holger Brandl
17      */
18     
19     public class VUMeterMonitor extends BaseDataProcessor {
20     
21         final VUMeter vumeter;
22         final VUMeterPanel vuMeterPanel;
23         final JDialog vuMeterDialog;
24     
25     
26         public VUMeterMonitor() {
27             vumeter = new VUMeter();
28     
29             vuMeterPanel = new VUMeterPanel();
30             vuMeterPanel.setVu(vumeter);
31             vuMeterPanel.start();
32     
33             vuMeterDialog = new JDialog();
34             vuMeterDialog.setBounds(100, 100, 100, 400);
35     
36             vuMeterDialog.getContentPane().setLayout(new BorderLayout());
37             vuMeterDialog.getContentPane().add(vuMeterPanel);
38     
39             vuMeterDialog.setVisible(true);
40         }
41     
42     
43         @Override
44         public Data getData() throws DataProcessingException {
45             Data d = getPredecessor().getData();
46     
47             // show the panel only if  a microphone is used as data source
48             if (d instanceof DataStartSignal)
49                 vuMeterPanel.setVisible(FrontEndUtils.getFrontEndProcessor(this, Microphone.class) != null);
50     
51             if (d instanceof DoubleData)
52                 vumeter.calculateVULevels(d);
53     
54             return d;
55         }
56     
57     
58         public JDialog getVuMeterDialog() {
59             return vuMeterDialog;
60         }
