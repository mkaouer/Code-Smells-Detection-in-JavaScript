20     public class VUMeterPanel extends JPanel {

22         public void setVu(VUMeter vu) {
23             this.vu = vu;

27         VUMeter vu;
28         boolean quit;
29         Thread thread;

31     
32         public void start() {
33             quit = false;
34             thread = new VUMeterPanelThread();
35             thread.start();
36         }
37     
38     
39         public void stop() {
40             quit = true;
41             boolean hasQuit = false;
42             while (!hasQuit) {
43                 try {
44                     thread.join();
45                     hasQuit = true;
46                 } catch (InterruptedException e) {
47                 }
48             }
49         }

51     
52         class VUMeterPanelThread extends Thread {
53     
54             @Override
55             public void run() {
56                 while (!quit) {
57                     repaint();  // probably this one should be replaced by a more appropriate method call in order to get rid of the annoying flickering
58                     try {
59                         Thread.sleep(10);
60                     } catch (InterruptedException e) {
61     
62                     }
63                 }
64             }
65         }
66     
67         /**
68          * Paint the component.  This will be called by AWT/Swing.
69          *
70          * @param g The <code>Graphics</code> to draw on.
71          */
72         @Override
73         public void paintComponent(Graphics g) {
74             super.paintComponent(g);
75     
76             if (vu != null) {
77                 paintVUMeter(g);
78             }
79         }
80     
81     
82         final int numberOfLights = 50;
83         final int greenLevel = (int) (numberOfLights * 0.3);
84         final int yellowLevel = (int) (numberOfLights * 0.7);
85         final int redLevel = (int) (numberOfLights * 0.9);
86     
87     
88         public VUMeter getVu() {
89             return vu;
90         }
91     
92     
93         private void paintVUMeter(Graphics g) {
94             int level = (int) ((vu.getRmsDB() / vu.getMaxDB()) * numberOfLights);
95             int peak = (int) ((vu.getPeakDB() / vu.getMaxDB()) * numberOfLights);
96     
97             assert level >= 0;
98             assert level < numberOfLights;
99     
100             Dimension sz = getSize();
101             int w = sz.width;
102             int h = (sz.height / numberOfLights);
103     
104             g.setColor(Color.BLACK);
105             g.fillRect(0, 0, sz.width - 1, sz.height - 1);
106     
107             for (int i = 0; i < level; i++) {
108                 setLevelColor(i, g);
109                 g.fillRect(1, sz.height - (i * h) + 1, w - 2, h - 2);
110             }
111     
112             setLevelColor(peak, g);
113             g.fillRect(1, sz.height - (peak * h) + 1, w - 2, h - 2);
114     
115         }
116     
117     
118         private void setLevelColor(int i, Graphics g) {
119             if (i < greenLevel)
120                 g.setColor(Color.BLUE);
121             else if (i < yellowLevel)
122                 g.setColor(Color.GREEN);
123             else if (i < redLevel)
124                 g.setColor(Color.YELLOW);
125             else
126                 g.setColor(Color.RED);
127         }
128     }
