68         /**
69          * Speaks JSML text provided as a <code>Speakable</code> object.
70          *
71          * @param jsmlText the JSML text to speak
72          * @param listener the listener to be notified as the
73          *   <code>jsmlText</code> is processed
74          *
75          * @throws JSMLException if the JSML text contains errors
76          * @throws EngineStateError 
77          *   if this <code>Synthesizer</code> in the <code>DEALLOCATED</code> or 
78          *   <code>DEALLOCATING_RESOURCES</code> states
79          */
80         public void speak(Speakable jsmlText, SpeakableListener listener)
81             throws JSMLException, EngineStateError {
82             checkEngineState(DEALLOCATED | DEALLOCATING_RESOURCES);
83             BaseSynthesizerQueueItem item = createQueueItem();
84             item.setData(this, jsmlText, listener);
85             appendQueue(item);
86         }

134          * Speaks a plain text <code>String</code>.  No JSML parsing is
135          * performed.
136          *
137          * @param text a <code>String</code> containing plain text.
138          * @param listener the listener to be notified as the
139          *   text is processed

141          * @throws EngineStateError

143          *   <code>DEALLOCATING_RESOURCES</code> states
144          */
145         public void speakPlainText(String text, SpeakableListener listener)
146             throws EngineStateError {
147             checkEngineState(DEALLOCATED | DEALLOCATING_RESOURCES);
148             try {
149                 BaseSynthesizerQueueItem item = createQueueItem();
150                 item.setData(this, text, true, listener);
151                 appendQueue(item);
152             } catch (JSMLException e) {
153                 throw new RuntimeException("JSMLException should never occur");
154             }

157         /**
158          * Returns a String of the names of all the states implied
159          * in the given bit pattern.
160          *
161          * @param state the bit pattern of states
162          *
163          * @return a String of the names of all the states implied
164          *   in the given bit pattern.
165          */
166         protected String stateToString(long state) {
167     	    StringBuilder buf = new StringBuilder();
168             if ((state & Synthesizer.QUEUE_EMPTY) != 0)
169                 buf.append(" QUEUE_EMPTY ");
170             if ((state & Synthesizer.QUEUE_NOT_EMPTY) != 0)
171                 buf.append(" QUEUE_NOT_EMPTY ");
172     	    return super.stateToString(state) + buf.toString();
173         }

293         /**
294          * Factory constructor for <code>EngineProperties</code> object.
295          * Gets the default speaking voice from the
296          * <code>SynthesizerModeDesc</code>.
297          * Takes the default prosody values (pitch, range, volume, rate)
298          * from the default voice.  Override to set engine-specific defaults.
299          *
300          * @return a <code>BaseEngineProperties</code> object specific to
301          *   a subclass.
302          */
303         protected BaseEngineProperties createEngineProperties() {
304             SynthesizerModeDesc desc = (SynthesizerModeDesc)engineModeDesc;
305             BaseVoice defaultVoice = (BaseVoice)(desc.getVoices()[0]);
306             
307             float defaultPitch = defaultVoice.defaultPitch;
308             float defaultPitchRange = defaultVoice.defaultPitchRange;
309             float defaultSpeakingRate = defaultVoice.defaultSpeakingRate;
310             float defaultVolume = defaultVoice.defaultVolume;
311             
312             return new BaseSynthesizerProperties(defaultVoice,
313                                                  defaultPitch,
314                                                  defaultPitchRange,
315                                                  defaultSpeakingRate,
316                                                  defaultVolume);
317         }

337         /**
338          * Utility function that generates <code>QUEUE_UPDATED</code>
339          * event and posts it to the event queue.  Eventually
340          * <code>fireQueueUpdated</code> will be called
341          * by <code>dispatchSpeechEvent</code> as a result of this action.
342          *
343          * @param topOfQueueChanged <code>true</code> if the top of the
344          *   queue has changed
345          * @param oldState the old state of this <code>Synthesizer</code>
346          * @param newState the new state of this <code>Synthesizer</code>
347          *
348          * @see #fireQueueUpdated
349          * @see #dispatchSpeechEvent
350          * 
351          */
352         public void postQueueUpdated(boolean topOfQueueChanged,
353                                      long oldState, long newState) {
354             SpeechEventUtilities.postSpeechEvent(
355                 this,
356                 new SynthesizerEvent(this,
357                                      SynthesizerEvent.QUEUE_UPDATED,
358                                      topOfQueueChanged,
359                                      oldState, newState));
360         }
361         
362         /**
363          * Utility function that sends a <code>QUEUE_UPDATED</code>
364          * event to all <code>SynthesizerListeners</code>.  
365          *
366          * @param event the <code>QUEUE_UPDATED</code> event
367          *
368          * @see #postQueueUpdated
369          * @see #dispatchSpeechEvent
370          */
371         public void fireQueueUpdated(SynthesizerEvent event) {
372             if (engineListeners == null) {
373                 return;
374             }
375             for (EngineListener el : engineListeners) {
376                 if (el instanceof SynthesizerListener) {
377                     SynthesizerListener sl = (SynthesizerListener)el;
378                     sl.queueUpdated(event);
379                 }
380             }
381         }

383         /**
384          * Utility function that generates <code>QUEUE_EMPTIED</code>
385          * event and posts it to the event queue.  Eventually
386          * <code>fireQueueEmptied</code> will be called
387          * by <code>dispatchSpeechEvent</code> as a result of this action.
388          *
389          * @param oldState the old state of this <code>Synthesizer</code>
390          * @param newState the new state of this <code>Synthesizer</code>
391          *
392          * @see #fireQueueEmptied
393          * @see #dispatchSpeechEvent
394          */
395         public void postQueueEmptied(long oldState, long newState) {
396             SpeechEventUtilities.postSpeechEvent(
397                 this,
398                 new SynthesizerEvent(this,
399                                      SynthesizerEvent.QUEUE_EMPTIED,
400                                      false,
401                                      oldState, newState));
402         }

404         /**
405          * Utility function that sends a <code>QUEUE_EMPTIED</code>
406          * event to all <code>SynthesizerListeners</code>.  
407          *
408          * @param event the <code>QUEUE_EMPTIED</code> event
409          *
410          * @see #postQueueEmptied
411          * @see #dispatchSpeechEvent
412          */
413         public void fireQueueEmptied(SynthesizerEvent event) {
414             if (engineListeners == null) {
415                 return;
416             }
417             for (EngineListener el : engineListeners) {
418                 if (el instanceof SynthesizerListener) {
419                     SynthesizerListener sl = (SynthesizerListener)el;
420                     sl.queueEmptied(event);
421                 }
422             }
423         }

425         /**
426          * Dispatches a <code>SpeechEvent</code>.
427          * The dispatcher should notify all <code>SynthesizerListeners</code>
428          * from this method.  The <code>SpeechEvent</code> was added
429          * via the various post methods of this class.
430          *
431          * @param event the <code>SpeechEvent</code> to dispatch
432          *
433          * @see #postQueueUpdated
434          * @see #postQueueEmptied
435          */
436         public void dispatchSpeechEvent(SpeechEvent event) {
437             switch (event.getId()) {
438                 case SynthesizerEvent.QUEUE_EMPTIED:
439                     fireQueueEmptied((SynthesizerEvent) event);
440                     break;
441                 case SynthesizerEvent.QUEUE_UPDATED:
442                     fireQueueUpdated((SynthesizerEvent) event);
443                     break;
444     
445                 // Defer to BaseEngine to handle the rest.
446                 //
447                 default:
448                     super.dispatchSpeechEvent(event);
449                     break;
450             }
451         }
