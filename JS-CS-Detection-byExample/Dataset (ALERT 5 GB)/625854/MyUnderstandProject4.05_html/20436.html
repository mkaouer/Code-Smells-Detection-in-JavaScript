16     package edu.cmu.sphinx.linguist.language.ngram.large;
17     
18     /** Represents a word ID (Nth word of a N-gram), and a N-gram probability ID. */
19     class NGramProbability {
20     
21         private final int which;
22         private final int wordID;
23         private final int probabilityID;
24         private final int backoffID;
25         private final int firstNPlus1GramEntry;
26     
27         /**
28          * Constructs a NGramProbability
29          * 
30          * @param which
31          *            which follower of the first word is this NGram
32          * @param wordID
33          *            the ID of the Nth word in a NGram
34          * @param probabilityID
35          *            the index into the probability array
36          * @param backoffID
37          *            the index into the backoff probability array
38          * @param firstNPlus1GramEntry
39          *            the first N+1Gram entry
40          */
41         public NGramProbability(int which, int wordID, int probabilityID,
42                 int backoffID, int firstNPlus1GramEntry) {
43             this.which = which;
44             this.wordID = wordID;
45             this.probabilityID = probabilityID;
46             this.backoffID = backoffID;
47             this.firstNPlus1GramEntry = firstNPlus1GramEntry;
48         }
49     
50         /**
51          * Returns which follower of the first word is this NGram
52          * 
53          * @return which follower of the first word is this NGram
54          */
55         public int getWhichFollower() {
56             return which;
57         }
58     
59         /**
60          * Returns the Nth word ID of this NGram
61          * 
62          * @return the Nth word ID
63          */
64         public int getWordID() {
65             return wordID;
66         }
67     
68         /**
69          * Returns the NGram probability ID.
70          * 
71          * @return the NGram probability ID
72          */
73         public int getProbabilityID() {
74             return probabilityID;
75         }
76     
77         /**
78          * Returns the backoff weight ID.
79          * 
80          * @return the backoff weight ID
81          */
82         public int getBackoffID() {
83             return backoffID;
84         }
85     
86         /**
87          * Returns the index of the first N+1Gram entry.
88          * 
89          * @return the index of the first N+1Gram entry
90          */
91         public int getFirstNPlus1GramEntry() {
92             return firstNPlus1GramEntry;
93         }
94     }
