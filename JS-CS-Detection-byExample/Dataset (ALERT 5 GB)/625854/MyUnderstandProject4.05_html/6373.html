13     package edu.cmu.sphinx.linguist.flat;
14     
15     import edu.cmu.sphinx.linguist.acoustic.Unit;
16     
17     /**
18      * A unit state that modifies how the unit state is cached.  Caching keys are generated from the full name for the
19      * sentence hmm. The default behavior for the unit (and all sentence hmms) is to generate the full name by combining the
20      * name for this unit with the name of the parent.  For the simple linguist, this is undesirable, because there are many
21      * different names for the parent pronunciations (differing contexts).  We want to be able to combine units that have
22      * identical names and context and are in the same position in the same pronunciation.  By defining getFullName to
23      * combine the name and the pronunciation index we allow units with identical contexts in the same position in a
24      * pronunciation to be combined.
25      */
26     public class ExtendedUnitState extends UnitState {
27     
28         /**
29          * Creates a UnitState. Gets the left and right contexts from the unit itself.
30          *
31          * @param parent the parent state
32          * @param which  the index of the given state
33          * @param unit   the unit associated with this state
34          */
35         public ExtendedUnitState(PronunciationState parent, int which, Unit unit) {
36             super(parent, which, unit);
37         }
38     
39     
40         /**
41          * Gets the fullName for this state
42          *
43          * @return the full name for this state
44          */
45         @Override
46         public String getFullName() {
47             return getName() + " in P" + getParent().getWhich();
48         }
49     }
