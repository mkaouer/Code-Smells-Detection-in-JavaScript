96                 System.out.println("No component: " + name);

101             Collection<String> propertyNames = ps.getRegisteredProperties();

105                 try {

119                         System.out.print("  " + propertyName + " [" + svalue + "]: ");

148          * Configure the logger
149          */
150         public static void configureLogger(ConfigurationManager cm) {
151     
152             // Allow others to override the logging settings.
153             if (System.getProperty("java.util.logging.config.class") != null
154                     || System.getProperty("java.util.logging.config.file") != null) {
155                 return;
156             }
157             // apply the log level (if defined) for the root logger (because we're using package based logging now)
158     
159             String cmPrefix = getLogPrefix(cm);
160             Logger cmRootLogger = Logger.getLogger(cmPrefix.substring(0, cmPrefix.length() - 1));
161     
162             // we need to determine the root-level here, because the logManager will reset it
163             Level rootLevel = Logger.getLogger("").getLevel();
164     
165             configureLogger(cmRootLogger);
166     
167             String level = cm.getGlobalProperty(GLOBAL_COMMON_LOGLEVEL);
168             if (level == null)
169                 level = Level.WARNING.getName();
170     
171             cmRootLogger.setLevel(Level.parse(level));
172     
173             // restore the old root logger level
174             Logger.getLogger("").setLevel(rootLevel);
175         }

205                 handler.setFormatter(new SphinxLogFormatter());

246         public static String toXML(ConfigurationManager cm) {
247             StringBuilder sb = new StringBuilder();
248             sb.append("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n");
249             sb.append("\n<!--    Sphinx-4 Configuration file--> \n\n");
250     
251             sb.append("<config>");
252     
253             Pattern pattern = Pattern.compile("\\$\\{(\\w+)\\}");
254     
255             Map<String, String> globalProps = cm.getGlobalProperties();
256             for (Map.Entry<String, String> entry : globalProps.entrySet()) {
257                 String propName = entry.getKey();
258     
259                 Matcher matcher = pattern.matcher(propName);
260                 propName = matcher.matches() ? matcher.group(1) : propName;
261     
262                 sb.append("\n\t<property name=\"").append(propName).append("\" value=\"").append(entry.getValue()).append("\"/>");
263             }
264     
265             for (String instanceName : cm.getComponentNames())
266                 sb.append("\n\n").append(propSheet2XML(instanceName, cm.getPropertySheet(instanceName)));
267     
268             sb.append("\n</config>");
269             return sb.toString();
270         }

273         private static String propSheet2XML(String instanceName, PropertySheet ps) {
274             StringBuilder sb = new StringBuilder();
275             sb.append("\t<component name=\"").append(instanceName).append("\" type=\"").append(ps.getConfigurableClass().getName()).append("\">");
276     
277             for (String propName : ps.getRegisteredProperties()) {
278                 String predec = "\n\t\t<property name=\"" + propName + "\" ";
279                 if (ps.getRawNoReplacement(propName) == null)
280                     continue;  // if the property was net defined within the xml-file
281     
282                 switch (ps.getType(propName)) {
283     
284                     case COMPONENT_LIST:
285                         sb.append("\n\t\t<propertylist name=\"").append(propName).append("\">");
286                         List<String> compNames = toStringList(ps.getRawNoReplacement(propName));
287                         for (String compName : compNames)
288                             sb.append("\n\t\t\t<item>").append(compName).append("</item>");
289                         sb.append("\n\t\t</propertylist>");
290                         break;
291                     default:
292                         sb.append(predec).append("value=\"").append(ps.getRawNoReplacement(propName)).append("\"/>");
293                 }
294             }
295     
296             sb.append("\n\t</component>\n\n");
297             return sb.toString();
298         }

367         /**
368          * Applies the system properties to the raw property map. System properties should be of the form
369          * compName[paramName]=paramValue
370          * <p/>
371          * List types cannot currently be set from system properties.
372          *
373          * @param rawMap the map of raw property values
374          * @param global global properties
375          * @throws PropertyException if an attempt is made to set a parameter for an unknown component.
376          */
377         static void applySystemProperties(Map<String, RawPropertyData> rawMap, Map<String, String> global)
378                 throws PropertyException {
379             Properties props = System.getProperties();
380             for (Enumeration<?> e = props.keys(); e.hasMoreElements();) {
381                 String param = (String) e.nextElement();
382                 String value = props.getProperty(param);
383     
384                 // search for parameters of the form component[parameter]=value
385                 // these go in the property sheet for the component
386                 int lb = param.indexOf('[');
387                 int rb = param.indexOf(']');
388     
389                 if (lb > 0 && rb > lb) {
390                     String compName = param.substring(0, lb);
391                     String paramName = param.substring(lb + 1, rb);
392                     RawPropertyData rpd = rawMap.get(compName);
393                     if (rpd != null) {
394                         rpd.add(paramName, value);
395                     } else {
396                         throw new InternalConfigurationException(compName, param,
397                                 "System property attempting to set parameter "
398                                         + " for unknown component " + compName
399                                         + " (" + param + ')');
400                     }
401                 }
402     
403                 // look for parameters of the form foo=bar
404                 // these go in the global map
405     
406                 else if (param.indexOf('.') == -1) {
407                     global.put(param, value);
408                 }
409             }
410         }

413         /**
414          * Renames a given <code>Configurable</code>. The configurable component named <code>oldName</code> is assumed to be
415          * registered to the CM. Renaming does not only affect the configurable itself but possibly global property values
416          * and properties of other components.
417          */
418         static void renameComponent(ConfigurationManager cm, String oldName, String newName) {
419             assert cm != null;
420             assert oldName != null && newName != null;
421             if (cm.getPropertySheet(oldName) == null) {
422                 throw new RuntimeException("no configurable (to be renamed) named " + oldName + " is contained in the CM");
423             }
424     
425             // this iteration is a little hacky. It would be much better to maintain the links to a configurable in a special table
426             for (String instanceName : cm.getComponentNames()) {
427                 PropertySheet propSheet = cm.getPropertySheet(instanceName);
428     
429                 for (String propName : propSheet.getRegisteredProperties()) {
430                     if (propSheet.getRawNoReplacement(propName) == null)
431                         continue;  // if the property was net defined within the xml-file
432     
433                     switch (propSheet.getType(propName)) {
434     
435                         case COMPONENT_LIST:
436                             List<String> compNames = toStringList(propSheet.getRawNoReplacement(propName));
437                             for (int i = 0; i < compNames.size(); i++) {
438                                 String compName = compNames.get(i);
439                                 if (compName.equals(oldName)) {
440                                     compNames.set(i, newName);
441                                 }
442                             }
443     
444                             break;
445                         case COMPONENT:
446                             if (propSheet.getRawNoReplacement(propName).equals(oldName)) {
447                                 propSheet.setRaw(propName, newName);
448                             }
449                     }
450                 }
451             }
452     
453             PropertySheet ps = cm.getPropertySheet(oldName);
454             ps.setInstanceName(newName);
455     
456             // it might be possible that the component is the value of a global property
457             for (Map.Entry<String, String> entry : cm.getGlobalProperties().entrySet()) {
458                 if (entry.getValue().equals(oldName))
459                     cm.setGlobalProperty(entry.getKey(), newName);
460             }
461         }

472         public static URL getResource(String name, PropertySheet ps) throws PropertyException {
473     
474             String location = ps.getString(name);
475             if (location == null) {
476                 throw new InternalConfigurationException(ps.getInstanceName(), name, "Required resource property '" + name + "' not set");

485                 return url;
486             } catch (MalformedURLException e) {
487                 throw new InternalConfigurationException(e, ps.getInstanceName(), name, "Bad URL " + location + e.getMessage());
488             }

494         public static URL resourceToURL(String location) throws MalformedURLException {
495             Matcher jarMatcher = jarPattern.matcher(location);
496             if (jarMatcher.matches()) {
497                 String resourceName = jarMatcher.group(1);
498                 return ConfigurationManagerUtils.class.getResource(resourceName);
499             } else {
500                 if (location.indexOf(':') == -1) {

502                 }
503                 return new URL(location);
504             }

519         public static boolean isImplementingInterface(Class<?> aClass, Class<?> interfaceClass) {
520             assert interfaceClass.isInterface();
521     
522             Class<?> superClass = aClass.getSuperclass();
523             if (superClass != null && isImplementingInterface(superClass, interfaceClass))
524                 return true;
525     
526             for (Class<?> curInterface : aClass.getInterfaces()) {
527                 if (curInterface.equals(interfaceClass) || isImplementingInterface(curInterface, interfaceClass))
528                     return true;
529             }
530     
531             return false;
532         }

536             while (aClass != null && !aClass.equals(Object.class)) {
537                 aClass = aClass.getSuperclass();
538     
539                 if (aClass != null && aClass.equals(possibleSuperclass))

541             }
542     
543             return false;

605         /**
606          * Attempts to set the value of an arbitrary component-property. If the property-name is ambiguous  with respect to
607          * the given <code>ConfiguratioManager</code> an extended syntax (componentName->propName) can be used to access the
608          * property.
609          * <p/>
610          * Beside component properties it is also possible to modify the class of a configurable, but this is only allowed if
611          * the configurable under question has not been instantiated yet. Furthermore the user has to ensure to set all
612          * mandatory component properties.
613          */
614         public static void setProperty(ConfigurationManager cm, String propName, String propValue) {
615             assert propValue != null;
616     
617             Map<String, List<PropertySheet>> allProps = listAllsPropNames(cm);
618             Set<String> configurableNames = cm.getComponentNames();
619     
620             if (!allProps.containsKey(propName) && !propName.contains("->") && !configurableNames.contains(propName))
621                 throw new RuntimeException("No property or configurable '" + propName + "' in configuration '" + cm.getConfigURL() + "'!");
622     
623             // if a configurable-class should be modified
624             if (configurableNames.contains(propName)) {
625                 try {
626                     final Class<? extends Configurable> confClass = Class.forName(propValue).asSubclass(Configurable.class);
627                     ConfigurationManagerUtils.setClass(cm.getPropertySheet(propName), confClass);
628                 } catch (ClassNotFoundException e) {
629                     throw new RuntimeException(e);
630                 }
631     
632                 return;
633             }
634     
635             if (!propName.contains("->") && allProps.get(propName).size() > 1) {
636                 throw new RuntimeException("Property-name '" + propName + "' is ambiguous with respect to configuration '"
637                         + cm.getConfigURL() + "'. Use 'componentName->propName' to disambiguate your request.");
638             }
639     
640             String componentName;
641     
642             // if disambiguation syntax is used find the correct PS first
643             if (propName.contains("->")) {
644                 String[] splitProp = propName.split("->");
645                 componentName = splitProp[0];
646                 propName = splitProp[1];
647             } else {
648                 componentName = allProps.get(propName).get(0).getInstanceName();
649             }
650     
651             setProperty(cm, componentName, propName, propValue);
652         }

654         public static void setProperty(ConfigurationManager cm, String componentName, String propName, String propValue) {
655     
656             // now set the property
657             PropertySheet ps = cm.getPropertySheet(componentName);
658             if (ps == null)
659                 throw new RuntimeException("Component '" + propName + "' is not registered to this system configuration '");
660     
661             // set the value to null if the string content is 'null
662             if (propValue.equals("null"))
663                 propValue = null;
664     
665             switch (ps.getType(propName)) {
666                 case BOOLEAN:
667                     ps.setBoolean(propName, Boolean.valueOf(propValue));
668                     break;
669                 case DOUBLE:
670                     ps.setDouble(propName, new Double(propValue));
671                     break;
672                 case INT:
673                     ps.setInt(propName, new Integer(propValue));
674                     break;
675                 case STRING:
676                     ps.setString(propName, propValue);
677                     break;
678                 case COMPONENT:
679                     ps.setComponent(propName, propValue, null);
680                     break;
681                 case COMPONENT_LIST:
682                     List<String> compNames = new ArrayList<String>();
683                     for (String component : propValue.split(";")) {
684                         compNames.add(component.trim());
685                     }
686     
687                     ps.setComponentList(propName, compNames, null);
688                     break;
689                 default:
690                     throw new RuntimeException("unknown property-type");

705         /**
706          * Returns the not yet instantiated components registered to this configuration manager.
707          */
708         public static Collection<String> getNonInstaniatedComps(ConfigurationManager cm) {
709             Collection<String> nonInstComponents = new ArrayList<String>();
710     
711             for (String compName : cm.getComponentNames()) {
712                 if (!cm.getPropertySheet(compName).isInstanciated())
713                     nonInstComponents.add(compName);
714             }
715             return nonInstComponents;
716         }

727         public static List<String> toStringList(Object obj) {
728             List<String> result = new ArrayList<String>();
729             if (!(obj instanceof List<?>))
730                 return null;
731             for (Object o : (List<?>) obj) {
732                 if (o instanceof String) {
733                     result.add((String) o);
734                 }
735             }
736             return result;
