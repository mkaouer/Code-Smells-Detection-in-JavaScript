231             for (int i = 1; i <= loader.getMaxDepth(); i++)
232             	logger.info(Integer.toString(i) + "-grams: " + loader.getNumberNGrams(i));

265         /** Builds the map from unigram to unigramID. Also finds the startWordID and endWordID.
266          * @param dictionary
267          * */
268         private void buildUnigramIDMap(Dictionary dictionary) {
269             int missingWords = 0;
270             String[] words = loader.getWords();
271             for (int i = 0; i < words.length; i++) {
272                 Word word = dictionary.getWord(words[i]);
273     
274                 if (word == null) {
275                     logger.info("Missing word: " + words[i]);
276                     missingWords++;
277                 }
278                 
279                 unigramIDMap.put(word, unigrams[i]);
280                 
281                 if (logger.isLoggable(Level.FINE))
282                     logger.fine("Word: " + word);
283             }
284     
285             if (missingWords > 0)
286                 logger.warning("Dictionary is missing " + missingWords + " words that are contained in the language model.");
287         }

310         /** Clears the various N-gram caches. */
311         private void clearCache() {
312             for (int i = 0; i < loadedBigramBuffers.length; i++) {
313                 NGramBuffer buffer = loadedBigramBuffers[i];
314                 
315                 if (buffer != null) {
316                     if (!buffer.getUsed())
317                         loadedBigramBuffers[i] = null; // free the BigramBuffer
318                     else
319                         buffer.setUsed(false);
320                 }
321             }
322     
323             
324             loadedBigramBuffers = new NGramBuffer[unigrams.length];        
325             for (int i = 2; i <= loader.getMaxDepth(); i++) {
326                 loadedNGramBuffers[i - 1] = new HashMap<WordSequence, NGramBuffer>();
327             }
328             logger.info("LM Cache Size: " + ngramDepthCache.size() + " Hits: " + ngramHits + " Misses: " + ngramMisses);
329         	if (clearCacheAfterUtterance) {
330         		ngramDepthCache = new LRUCache<WordSequence, ProbDepth>(ngramCacheSize);
331         	}
332         }

334         /**
335          * Returns predicted probability and depth. Uses caching for 
336          * high order ngrams.
337          * 
338          * @param wordSequence sequence to get the probability
339          */
340         public ProbDepth getProbDepth(WordSequence wordSequence) {
341             int numberWords = wordSequence.size();
342             ProbDepth probDepth = null;
343     
344             if (numberWords > maxDepth) {
345                 throw new Error("Unsupported NGram: " + wordSequence.size());
346             }
347     
348             if (numberWords == maxDepth) {
349                 probDepth = ngramDepthCache.get(wordSequence);
350     
351                 if (probDepth != null) {
352                     ngramHits++;
353                     return probDepth;
354                 }
355                 ngramMisses++;
356             }
357     
358             probDepth = getNGramProbDepth(wordSequence);
359     
360             if (numberWords == maxDepth)
361                 ngramDepthCache.put(wordSequence, probDepth);
362     
363             if (logFile != null && probDepth != null)
364                 logFile.println(wordSequence.toString().replace("][", " ") + " : " + Float.toString(probDepth.probability) + " : "
365                         + probDepth.depth);
366     
367             return probDepth;
368         }

371         private ProbDepth getNGramProbDepth(WordSequence wordSequence) {
372             int numberWords = wordSequence.size();
373             Word firstWord = wordSequence.getWord(0);
374     
375             if (loader.getNumberNGrams(numberWords) == 0 || !hasUnigram(firstWord))
376                 return getNGramProbDepth(wordSequence.getNewest());
377     
378             if (numberWords < 2) {
379                 return getUnigramProbDepth(wordSequence);
380             }
381     
382             NGramProbability nGProbability = findNGram(wordSequence);
383     
384             if (nGProbability != null) {
385                 float probability = ngramProbTable[numberWords - 1][nGProbability.getProbabilityID()];
386                 return new ProbDepth(probability, numberWords);
387             }
388     
389             if (numberWords == 2) {
390                 UnigramProbability unigramProb = getUnigram(firstWord);
391                 UnigramProbability unigramProb1 = getUnigram(wordSequence.getWord(1));
392                 float probability = unigramProb.getLogBackoff() + unigramProb1.getLogProbability();
393                 return new ProbDepth(probability, 1);
394             }
395     
396             NGramProbability nMinus1Gram = findNGram(wordSequence.getOldest());
397     
398             if (nMinus1Gram != null) {
399                 ProbDepth result1 = getProbDepth(wordSequence.getNewest());
400                 float probability = ngramBackoffTable[numberWords - 1][nMinus1Gram.getBackoffID()] + result1.probability;
401                 return new ProbDepth(probability, result1.depth);
402             }
403     
404             return getProbDepth(wordSequence.getNewest());
405         }

423         /**
424          * Finds or loads the NGram probability of the given NGram.
425          *
426          * @param wordSequence the NGram to load
427          * @return a NGramProbability of the given NGram
428          */
429         private NGramProbability findNGram(WordSequence wordSequence) {
430         	int numberWords = wordSequence.size();
431             NGramProbability nGram = null; 
432             
433             WordSequence oldest = wordSequence.getOldest();
434             NGramBuffer nGramBuffer = loadedNGramBuffers[numberWords - 1].get(oldest);
435             if (nGramBuffer == null) {
436             	nGramBuffer = getNGramBuffer(oldest);
437                 if (nGramBuffer != null)
438                     loadedNGramBuffers[numberWords - 1].put(oldest, nGramBuffer);
439             }
440     
441             if (nGramBuffer != null) {
442                 int nthWordID = getWordID(wordSequence.getWord(numberWords - 1));
443                 nGram = nGramBuffer.findNGram(nthWordID);
444             }
445     
446             return nGram;
447         }

460         /**
461          * Loads into a buffer all the NGram followers of the given N-1Gram.
462          * 
463          * @param ws the N-1Gram to find followers
464          * 
465          * @return a NGramBuffer of all the NGram followers of the given sequence
466          */
467         private NGramBuffer loadNGramBuffer(WordSequence ws) { 
468         	int firstWordID = getWordID(ws.getWord(0));
469         	int firstCurrentNGramEntry = 0;
470         	int numberNGrams = 0;
471         	int size = 0;
472     		long position = 0;
473     		int orderBuffer = ws.size() + 1;
474     		NGramBuffer currentBuffer = null;
475     		NGramBuffer nMinus1Buffer = null;
476     
477     		firstCurrentNGramEntry = unigrams[firstWordID].getFirstBigramEntry(); 
478     		numberNGrams = getNumberBigramFollowers(firstWordID) + 1;
479     		
480     		if (numberNGrams == 1) // 1 means that there is no bigram starting with firstWordID
481     			return null;
482             
483     		if (orderBuffer == 2)  {
484     			size = numberNGrams * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField();
485     			position = (long) (loader.getNGramOffset(orderBuffer) + (firstCurrentNGramEntry * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField()));
486     		} else { // only for ws.size() >= 2
487     			int lastWordId = getWordID(ws.getWord(ws.size() - 1));
488     			nMinus1Buffer = getNGramBuffer(ws.getOldest());
489     			int index = nMinus1Buffer.findNGramIndex(lastWordId);
490     
491     			if (index == -1)
492     				return null;
493     
494     			int firstNMinus1GramEntry = nMinus1Buffer.getFirstNGramEntry();
495     			firstCurrentNGramEntry = getFirstNGramEntry(nMinus1Buffer.getNGramProbability(index), firstNMinus1GramEntry, orderBuffer);
496     			int firstNextNGramEntry = getFirstNGramEntry(nMinus1Buffer.getNGramProbability(index + 1), firstNMinus1GramEntry, orderBuffer);
497     			numberNGrams = firstNextNGramEntry - firstCurrentNGramEntry;
498     			
499     			if (numberNGrams == 0)
500     				return null;
501     			
502     			if (loader.getMaxDepth() != orderBuffer) 
503     				numberNGrams++;
504     
505     			size = numberNGrams * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField();
506     			position = (long) loader.getNGramOffset(orderBuffer) + (long) firstCurrentNGramEntry * (long) ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * (long) loader.getBytesPerField();
507     		}
508     
509             try {
510                 byte[] buffer = loader.loadBuffer(position, size);
511     
512                 if (loader.getMaxDepth() == orderBuffer) {
513                     currentBuffer = new NMaxGramBuffer(buffer, numberNGrams, loader.getBigEndian(), is32bits(), orderBuffer,
514                             firstCurrentNGramEntry);
515                 } else {
516                     currentBuffer = new NGramBuffer(buffer, numberNGrams, loader.getBigEndian(), is32bits(), orderBuffer,
517                             firstCurrentNGramEntry);
518                 }
519             }
520     		catch (IOException ioe) {
521     			ioe.printStackTrace();
522     			throw new Error("Error loading " + orderBuffer + "-Grams.");
523     		}
524     
525     		return currentBuffer;
526         }

529         /**
530          * Returns the NGrams of the given word sequence
531          *
532          * @param wordSequence the word sequence 
533          * 					   from which to get the buffer
534          * @return the NGramBuffer of the word sequence
535          */
536         private NGramBuffer getNGramBuffer(WordSequence wordSequence) {
537         	NGramBuffer nGramBuffer = null;
538         	int order = wordSequence.size();
539         	
540         	if (order > 1)
541         		nGramBuffer = loadedNGramBuffers[order - 1].get(wordSequence); // better when using containsKey
542     
543         	if (nGramBuffer == null) {
544         		nGramBuffer = loadNGramBuffer(wordSequence);
545     
546         		if (nGramBuffer != null) 
547         			loadedNGramBuffers[order - 1].put(wordSequence, nGramBuffer); // optimizable by adding an 'empty' nGramBuffer
548         	}
549     
550         	return nGramBuffer;
551         }

554         /**
555          * Returns the index of the first NGram entry of the given N-1Gram
556          *
557          * @param nMinus1Gram            the N-1Gram which first NGram entry we're looking for
558          * @param firstNMinus1GramEntry  the index of the first N-1Gram entry of the N-1Gram in question
559          * @param n						 the order of the NGram
560          * @return the index of the first NGram entry of the given N-1Gram
561          */
562         private int getFirstNGramEntry(NGramProbability nMinus1Gram, int firstNMinus1GramEntry, int n) {
563             int firstNGramEntry = ngramSegmentTable[n - 1][(firstNMinus1GramEntry + nMinus1Gram.getWhichFollower()) 
564                                                       >> loader.getLogNGramSegmentSize()] 
565                                                       + nMinus1Gram.getFirstNPlus1GramEntry();
566             
567             return firstNGramEntry;
568         }

640                     smearTerm = unigramSmearTerm[wordID];
641                 }
642             }
643             
644             if (fullSmear && logger.isLoggable(Level.FINE))
645                 logger.fine("SmearTerm: " + smearTerm);
646     
647             return smearTerm;
648         }

653         @Override
654         public float getSmear(WordSequence wordSequence) {
655             float smearTerm = 0.0f;
656             
657             if (fullSmear) {
658                 smearCount++;
659                 int length = wordSequence.size();
660                 
661                 if (length == 1) {
662                     int wordID = getWordID(wordSequence.getWord(0));
663                     smearTerm = unigramSmearTerm[wordID];
664                 } else if (length >= 2) {
665                     int size = wordSequence.size();
666                     int wordID1 = getWordID(wordSequence.getWord(size - 2));
667                     int wordID2 = getWordID(wordSequence.getWord(size - 1));
668                     Float st = getSmearTerm(wordID1, wordID2);
669                     
670                     if (st == null)
671                         smearTerm = unigramSmearTerm[wordID2];
672                     else {
673                         smearTerm = st;
674                         smearBigramHit++;
675                     }
676                 }
677     
678                 if (smearCount % 100000 == 0)
679                     System.out.println("Smear hit: " + smearBigramHit + " tot: " + smearCount);
680             }
681             
682             if (fullSmear && logger.isLoggable(Level.FINE))
683                 logger.fine("SmearTerm: " + smearTerm);
684     
685             return smearTerm;
686         }

785         private void buildSmearInfo() throws IOException {
786             double S0 = 0;
787             double R0 = 0;
788     
789             bigramSmearMap = new HashMap<Long, Float>();
790     
791             double[] ugNumerator = new double[unigrams.length];
792             double[] ugDenominator = new double[unigrams.length];
793             double[] ugAvgLogProb = new double[unigrams.length];
794     
795             unigramSmearTerm = new float[unigrams.length];
796     
797             for (UnigramProbability unigram : unigrams) {
798                 float logp = unigram.getLogProbability();
799                 double p = logMath.logToLinear(logp);
800                 S0 += p * logp;
801                 R0 += p * logp * logp;
802             }
803     
804             System.out.println("R0 S0 " + R0 + ' ' + S0);
805     
806             for (int i = 0; i < loadedBigramBuffers.length; i++) {
807                 NGramBuffer bigram = getBigramBuffer(i);
808                 
809                 if (bigram == null) {
810                     unigramSmearTerm[i] = LogMath.getLogOne();
811                     continue;
812                 }
813     
814                 ugNumerator[i] = 0.0;
815                 ugDenominator[i] = 0.0;
816                 ugAvgLogProb[i] = 0.0;
817     
818                 float logugbackoff = unigrams[i].getLogBackoff();
819                 double ugbackoff = logMath.logToLinear(logugbackoff);
820     
821                 for (int j = 0; j < bigram.getNumberNGrams(); j++) {
822                     int wordID = bigram.getWordID(j);
823                     NGramProbability bgProb = bigram.getNGramProbability(j);
824     
825                     float logugprob = unigrams[wordID].getLogProbability();
826                     float logbgprob = ngramProbTable[1][bgProb.getProbabilityID()];
827     
828                     double ugprob = logMath.logToLinear(logugprob);
829                     double bgprob = logMath.logToLinear(logbgprob);
830     
831                     double backoffbgprob = ugbackoff * ugprob;
832                     double logbackoffbgprob = logMath.linearToLog(backoffbgprob);
833     
834                     ugNumerator[i] += (bgprob * logbgprob
835                             - backoffbgprob * logbackoffbgprob) * logugprob;
836     
837                     ugDenominator[i] += (bgprob - backoffbgprob) * logugprob;
838                     // dumpProbs(ugNumerator, ugDenominator, i, j, logugprob,
839     				//		logbgprob, ugprob, bgprob, backoffbgprob,
840     				//		logbackoffbgprob);
841                 }
842                 
843                 ugNumerator[i] += ugbackoff * (logugbackoff * S0 + R0);
844                 ugAvgLogProb[i] = ugDenominator[i] + ugbackoff * S0;
845                 ugDenominator[i] += ugbackoff * R0;
846     
847                 // System.out.println("n/d " + ugNumerator[i] + " " +
848                 //                     ugDenominator[i]);
849     
850                 unigramSmearTerm[i] = (float) (ugNumerator[i] / ugDenominator[i]);
851                 /// unigramSmearTerm[i] = 
852                 //   logMath.linearToLog(ugNumerator[i] / ugDenominator[i]);
853                 //  System.out.println("ugs " + unigramSmearTerm[i]);
854             }
855     
856             for (int i = 0; i < loadedBigramBuffers.length; i++) {
857                 System.out.println("Processed " + i
858                         + " of " + loadedBigramBuffers.length);
859                 NGramBuffer bigram = getBigramBuffer(i);
860                 
861                 if (bigram == null)
862                     continue;
863     
864                 for (int j = 0; j < bigram.getNumberNGrams(); j++) {
865                     float smearTerm;
866                     NGramProbability bgProb = bigram.getNGramProbability(j);
867                     float logbgbackoff = ngramBackoffTable[2][bgProb.getBackoffID()];
868                     double bgbackoff = logMath.logToLinear(logbgbackoff);
869                     int k = bigram.getWordID(j);
870                     NGramBuffer trigram = loadTrigramBuffer(i, k);
871     
872                     if (trigram == null)
873                         smearTerm = unigramSmearTerm[k];
874                     else {
875                         double bg_numerator = 0;
876                         double bg_denominator = 0;
877                         for (int l = 0; l < trigram.getNumberNGrams(); l++) {
878                             int m = trigram.getWordID(l);
879                             float logtgprob
880                                     = ngramProbTable[2][trigram.getProbabilityID(l)];
881                             double tgprob = logMath.logToLinear(logtgprob);
882                             float logbgprob = getBigramProb(k, m);
883                             double bgprob = logMath.logToLinear(logbgprob);
884                             float logugprob = unigrams[m].getLogProbability();
885                             double backofftgprob = bgbackoff * bgprob;
886                             double logbackofftgprob
887                                     = logMath.linearToLog(backofftgprob);
888     
889                             bg_numerator += (tgprob * logtgprob
890                                     - backofftgprob * logbackofftgprob) * logugprob;
891     
892                             bg_denominator += (tgprob - backofftgprob)
893                                     * logugprob * logugprob;
894                         }
895                         
896                         bg_numerator += bgbackoff * (logbgbackoff * ugAvgLogProb[k] - ugNumerator[k]);
897                         bg_denominator += bgbackoff * ugDenominator[k];
898                         // bigram.ugsmear = bg_numerator / bg_denominator;
899                         smearTerm = (float) (bg_numerator / bg_denominator);
900                         smearTermCount++;
901                     }
902                     
903                     putSmearTerm(i, k, smearTerm);
904                 }
905             }
906             
907             System.out.println("Smear count is " + smearTermCount);
908         }

973          *
974          * @param filename where to read the smear info from
975          * @throws IOException if an inconsistent file is found or on any general I/O error
976          */

978             DataInputStream in = new DataInputStream(new FileInputStream(filename));
979     
980     
981             if (in.readInt() != SMEAR_MAGIC)
982                 throw new IOException("Bad smear format for " + filename);
983     
984             if (in.readInt() != unigrams.length)
985                 throw new IOException("Bad unigram length in " + filename);
986     
987             bigramSmearMap = new HashMap<Long, Float>();
988             unigramSmearTerm = new float[unigrams.length];
989             System.out.println("Reading " + unigrams.length);
990             
991             for (int i = 0; i < unigrams.length; i++)
992                 unigramSmearTerm[i] = in.readFloat();
993     
994             for (int i = 0; i < unigrams.length; i++) {
995                 System.out.println("Processed " + i
996                         + " of " + loadedBigramBuffers.length);
997                 int numBigrams = in.readInt();
998                 NGramBuffer bigram = getBigramBuffer(i);
999                 
1000                 if (bigram.getNumberNGrams() != numBigrams)
1001                     throw new IOException("Bad ngrams for unigram " + i + " Found " + numBigrams + " expected " + bigram.getNumberNGrams());
1002     
1003                 for (int j = 0; j < numBigrams; j++) {
1004                     int k = bigram.getWordID(j);
1005                     putSmearTerm(i, k, in.readFloat());
1006                 }
1007             }
1008             
1009             in.close();
1010         }
