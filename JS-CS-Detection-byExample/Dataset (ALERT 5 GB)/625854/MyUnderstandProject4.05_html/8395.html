8     package com.sun.speech.engine.recognition;

10     import com.sun.speech.engine.BaseAudioManager;
11     import com.sun.speech.engine.SpeechEventDispatcher;
12     import com.sun.speech.engine.SpeechEventUtilities;
13     
14     import javax.speech.AudioListener;
15     import javax.speech.SpeechEvent;
16     import javax.speech.recognition.Recognizer;
17     import javax.speech.recognition.RecognizerAudioEvent;
18     import javax.speech.recognition.RecognizerAudioListener;
19     
20     /**
21      * Skeletal Implementation of the JSAPI AudioManager interface for Recognizers.  Merely provides convenience function
22      * for calling the listeners.
23      * <p/>
24      * <P>Actual JSAPI implementations might want to extend or modify this implementation.
25      * <p/>
26      *
27      * @author Willie Walker
28      * @version 1.2 01/27/99 13:43:57
29      */
30     public class BaseRecognizerAudioManager extends BaseAudioManager
31             implements SpeechEventDispatcher {
32     
33     //////////////////////
34     // Begin utility methods for calling RecognizerAudioListeners
35     
36         //////////////////////
37     
38     
39         /**
40          * Utility function to generate AUDIO_LEVEL event and post it to the event queue.  Eventually fireAudioLevel will be
41          * called by dispatchSpeechEvent as a result of this action.
42          *
43          * @param source     the Recognizer causing the event
44          * @param audioLevel the audio level between 0.0 and 1.0
45          */
46         public void postAudioLevel(Recognizer source, float audioLevel) {
47             SpeechEventUtilities.postSpeechEvent(
48                     this,
49                     new RecognizerAudioEvent(source,
50                             RecognizerAudioEvent.AUDIO_LEVEL,
51                             audioLevel));
52         }
53     
54     
55         /** Utility function to send a AUDIO_LEVEL event to all listeners. */
56         public void fireAudioLevel(RecognizerAudioEvent event) {
57             for (AudioListener al : listeners)
58                 if (al instanceof RecognizerAudioListener)
59                     ((RecognizerAudioListener) al).audioLevel(event);
60         }
61     
62     
63         /**
64          * Utility function to generate SPEECH_STARTED event and post it to the event queue.  Eventually fireSpeechStarted
65          * will be called by dispatchSpeechEvent as a result of this action.
66          *
67          * @param source the Recognizer causing the event
68          */
69         public void postSpeechStarted(Recognizer source) {
70             SpeechEventUtilities.postSpeechEvent(
71                     this,
72                     new RecognizerAudioEvent(source,
73                             RecognizerAudioEvent.SPEECH_STARTED));
74         }
75     
76     
77         /** Utility function to send a SPEECH_STARTED event to all listeners. */
78         public void fireSpeechStarted(RecognizerAudioEvent event) {
79             for (AudioListener al : listeners)
80                 if (al instanceof RecognizerAudioListener)
81                     ((RecognizerAudioListener) al).speechStarted(event);
82         }
83     
84     
85         /**
86          * Utility function to generate SPEECH_STOPPED event and post it to the event queue.  Eventually fireSpeechStopped
87          * will be called by dispatchSpeechEvent as a result of this action.
88          *
89          * @param source the Recognizer causing the event
90          */
91         public void postSpeechStopped(Recognizer source) {
92             SpeechEventUtilities.postSpeechEvent(
93                     this,
94                     new RecognizerAudioEvent(source,
95                             RecognizerAudioEvent.SPEECH_STOPPED));
96         }
97     
98     
99         /** Utility function to send a SPEECH_STOPPED event to all listeners. */
100         public void fireSpeechStopped(RecognizerAudioEvent event) {
101             for (AudioListener al : listeners)
102                 if (al instanceof RecognizerAudioListener)
103                     ((RecognizerAudioListener) al).speechStopped(event);
104         }
105     //////////////////////
106     // End utility methods for calling RecognizerAudioListeners
107     //////////////////////
108     
109     
110         /**
111          * Dispatch a SpeechEvent.  This is a method from SpeechEventDispatcher. The dispatcher should notify all listeners
112          * of the speech event from this method.
113          */
114         public void dispatchSpeechEvent(SpeechEvent event) {
115             switch (event.getId()) {
116                 case RecognizerAudioEvent.AUDIO_LEVEL:
117                     fireAudioLevel((RecognizerAudioEvent) event);
118                     break;
119                 case RecognizerAudioEvent.SPEECH_STARTED:
120                     fireSpeechStarted((RecognizerAudioEvent) event);
121                     break;
122                 case RecognizerAudioEvent.SPEECH_STOPPED:
123                     fireSpeechStopped((RecognizerAudioEvent) event);
124                     break;
125             }
126         }
127     }
