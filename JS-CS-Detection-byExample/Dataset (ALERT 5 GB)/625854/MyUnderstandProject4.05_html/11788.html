35     /**
36      * A test program for JSGF grammars. This program will generate a number of
37      * random sentences from a JSGF grammar and attempt to validate them via a rule
38      * parse. A count of valid and invalid sentences is reported after the run.
39      * 
40      * This test will detect and reports only certain grammar failures. It will
41      * detect when a JSGFGrammar generates invalid sentences. It will not detect the
42      * case where a valid sentence can never be generated by the JSGFGrammar.
43      * 
44      * Note that there is a bug in the Rule Grammar parse where rules with nested
45      * recursion cause a stack overflow error.
46      * 
47      */
48     public class JSGFTest {
49     
50         JSGFGrammar jsgfGrammar;
51         
52         @Before
53         public void init() throws IOException {
54             ConfigurationManager cm = new ConfigurationManager(
55             "src/test/edu/cmu/sphinx/jsgf/test/jsgftest.config.xml");
56     
57             jsgfGrammar = (JSGFGrammar) cm.lookup("jsgfGrammar");
58             jsgfGrammar.allocate();       
59         }
60         
61         @Test
62         public void testParser() throws IOException, GrammarException, EngineException {
63             JSGFRuleGrammar jsgfRuleGrammar = jsgfGrammar.getRuleGrammar();
64             BaseRecognizer recognizer = new BaseRecognizer(jsgfGrammar.getGrammarManager());  
65             recognizer.allocate();
66             RuleGrammar grammar = new BaseRuleGrammar(recognizer, jsgfRuleGrammar);
67             Scanner scanner = new Scanner (new File ("src/test/edu/cmu/sphinx/jsgf/test/input.txt"));
68             scanner.useDelimiter("\n");
69             while (scanner.hasNext()) {
70                 String sentence = scanner.next();
71                 RuleParse rp = grammar.parse(sentence, null);
72                 Assert.assertNotNull(rp);
73             }
74         }
75         
76         @Test
77         public void testSave () {
78             JSGFRuleGrammar grammar = jsgfGrammar.getRuleGrammar();
79             System.out.println (grammar.toString());
80         }
