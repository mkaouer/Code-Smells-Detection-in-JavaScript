13     package edu.cmu.sphinx.linguist.dictionary;
14     
15     import edu.cmu.sphinx.linguist.acoustic.Unit;
16     
17     
18     /** Provides pronunciation information for a word. */
19     public class Pronunciation {
20     
21         public static final Pronunciation UNKNOWN = new Pronunciation(Unit.EMPTY_ARRAY, null, null, 1.0f);
22     
23         private Word word;
24         private final WordClassification wordClassification;
25         private final Unit[] units;
26         private final String tag;
27         private final float probability;
28     
29     
30         /**
31          * Creates a pronunciation
32          *
33          * @param units              represents the pronunciation
34          * @param tag                a grammar specific tag
35          * @param wordClassification the classification for this word
36          * @param probability        the probability of this pronunciation occurring
37          */
38         protected Pronunciation(Unit[] units,
39                       String tag,
40                       WordClassification wordClassification,
41                       float probability) {
42             this.wordClassification = wordClassification;
43             this.units = units;
44             this.tag = tag;
45             this.probability = probability;
46         }
47     
48     
49         /**
50          * Sets the word this pronunciation represents.
51          *
52          * @param word the Word this Pronunciation represents
53          */
54         protected void setWord(Word word) {
55             if (this.word == null) {
56                 this.word = word;
57             } else {
58                 throw new Error("Word of Pronunciation cannot be set twice.");
59             }
60         }
61     
62     
63         /**
64          * Retrieves the word that this Pronunciation object represents.
65          *
66          * @return the word
67          */
68         public Word getWord() {
69             return word;
70         }
71     
72     
73         /**
74          * Retrieves the word classification for this pronunciation
75          *
76          * @return the word classification for this pronunciation.
77          */
78         public WordClassification getWordClassification() {
79             return wordClassification;
80         }
81     
82     
83         /**
84          * Retrieves the units for this pronunciation
85          *
86          * @return the units for this pronunciation
87          */
88         public Unit[] getUnits() {
89             return units;
90         }
91     
92     
93         /**
94          * Retrieves the tag associated with the pronunciation or null if there is no tag associated with this
95          * pronunciation. Pronunciations can optionally be tagged to allow applications to distinguish between different
96          * pronunciations.
97          *
98          * @return the tag or null if no tag is available.
99          */
100         public String getTag() {
101             return tag;
102         }
103     
104     
105         /**
106          * Retrieves the probability for the pronunciation. A word may have multiple pronunciations that are not all equally
107          * probable. All probabilities for particular word sum to 1.0.
108          *
109          * @return the probability of this pronunciation as a value between 0 and 1.0.
110          *         <p/>
111          *         TODO: FIX Note that probabilities are currently maintained in the linear domain (unlike just about
112          *         everything else)
113          */
114         public float getProbability() {
115             return probability;
116         }
117     
118     
119         /** Dumps a pronunciation */
120         public void dump() {
121             System.out.println(toString());
122         }
123     
124     
125         /**
126          * Returns a string representation of this Pronunication.
127          *
128          * @return a string of this Pronunciation
129          */
130         @Override
131         public String toString() {
132             StringBuilder result = new StringBuilder().append(word).append('(');
133             for (Unit unit : units) {
134                 result.append(unit).append(' ');
135             }
136             result.append(')');
137             return result.toString();
138         }
139     
140     
141         /**
142          * Returns a detailed string representation of this Pronunication.
143          *
144          * @return a string of this Pronunciation
145          */
146         public String toDetailedString() {
147             StringBuilder result = new StringBuilder().append(word).append(' ');
148             for (Unit unit : units) {
149                 result.append(unit).append(' ');
150             }
151             result.append("\n   class: ").append(wordClassification)
152                 .append(" tag: ").append(tag).append(" prob: ").append(probability);
153     
154             return result.toString();
155         }
156     }
