23     /**
24      * This class holds information about one configurable class in the Sphinx model
25      * Which includes the classname, property list, class description,group name,  
26      * and the configuration sets that are defined for this class
27      * <p>
28      * This class, combined with <code>ConfigurableProperty</code> construct
29      * the Sphinx model. 
30      * <p>
31      * RawPropertyData is used to hold the configuration values set for this component
32      *
33      * @author Ariani
34      * @see ConfigurableProperty
35      * @see RawPropertyData
36      */
37     public class ConfigurableComponent {
38         
39         private final String _sectionName;
40         private final Map<String, ConfigurableProperty> _propList; // String propertyName, ConfigurableProperty
41         private final Class<?> _component;
42         private final String _componentDesc;
43         private final String _componentClassName;
44         private final Map<String, RawPropertyData> _confProp; // List of String name, Rpd for this class
45         
46         /** 
47          * Creates a new instance of <code>ConfigurableComponent</code>
48          * 
49          * @param section the group that this class belongs to
50          * @param component reference to the class that this object represents
51          * @param name class name
52          * @param desc brief description of this configurable component
53          */
54         public ConfigurableComponent(String section, Class<?> component, String name, String desc) {
55             _sectionName = section;
56             _component = component;
57             _componentClassName = name;
58             _componentDesc = desc;
59            _propList = new HashMap<String, ConfigurableProperty>();
60            _confProp = new HashMap<String, RawPropertyData>();
61         }
62             
63         /**
64          * used to list the information contained in this component
65          * mainly for debugging and testing
66          *
67          * @return string description of model
68          */
69         @Override
70         public String toString(){
71             StringBuilder output = new StringBuilder("Section :").append(_sectionName).append('\n');
72             output.append(" Class Name : ").append(_componentClassName).append('\n');
73             if (!_componentDesc.isEmpty()) {
74                 output.append(" Desc : ").append(_componentDesc).append('\n');
75             }
76             if( !_propList.isEmpty() ){
77                 output.append(" Property list : \n");
78                 for (ConfigurableProperty cp : _propList.values()) {
79                     output.append(cp);
80                 }
81             }        
82             if ( !_confProp.isEmpty()){
83                 output.append(" Configuration values : ***** \n");
84                 for (RawPropertyData rpd : _confProp.values()) {
85                     output.append(rpd);
86                 }
87             }
88             return output.toString();
89         }
90            
91         /**
92          * @return Returns group / section name that this component belongs to
93          */
94         public String getSectionName(){
95             return _sectionName;
96         }
97         
98         /**
99          * @return Returns the Map that holds the component properties 
100          *         (pair of <code>String, ConfigurableProperty</code>)
101          */
102         public Map<String, ConfigurableProperty> getPropertyMap(){
103             return _propList;
104         }
105         
106         /**
107          * Add a new property for this component
108          *
109          * @param cp <code>ConfigurableProperty</code>
110          */
111         public void addProperty(ConfigurableProperty cp){
112             _propList.put(cp.getName(),cp);
113         }
114         
115         /**
116          * Check if component has the property
117          *
118          * @param name Name of property 
119          * @return Boolean <code>true</code> if property exists
120          */
121         public boolean containsProperty(String name){
122             return _propList.containsKey(name);
123         }
124         
125         /** 
126          * get the property with specified name
127          *
128          * @param name Name of property
129          * @return <code>ConfigurableProperty</code> with the specified name;
130          *         returns null if the property does not exist
131          */
132         public ConfigurableProperty getProperty(String name){
133             return _propList.get(name);
134         }
135         
136         /**
137          * @return the <code>Class</code> that this component refers to
138          */
139         public Class<?> getComponentClass(){
140             return _component;
141         }
142         
143         /**
144          * @return class name of component
145          */
146         public String getName(){
147             return _componentClassName;
148         }
149            
150         /**
151          * @return class description
152          */
153         public String getDesc(){
154             return _componentDesc;
155         }
156         
157         /**
158          * This function is used to attach a configuration set for this component           
159          * 
160          * @param rpd <code>RawPropertyData</code> that holds the property values
161          */
162         public void addConfigurationProp(RawPropertyData rpd){
163             _confProp.put(rpd.getName(),rpd);        
164         }
165             
166         /**
167          * Delete all configuration sets from this component
168          */
169         public void deleteAllConfigurationProp(){
170             _confProp.clear();
171         }
172         
173         /**
174          * Delete the configuration set with specified name
175          *
176          * @param rpdname name of <code>RawPropertyData</code> to be deleted
177          */
178         public void deleteConfigurationProp(String rpdname){
179             if (_confProp.containsKey(rpdname))
180                 _confProp.remove(rpdname);
181         }

184         
185         /** 
186          * Remove a specific property from one of the configuration
187          * sets
188          *
189          * @param rpdname Name of <code>RawPropertyData</code> that stores the
190          *               property values
191          * @param propname Property name to be modified    
192          */
193         public void deleteOneConfigurationPropFromSet(String rpdname,String propname){
194             RawPropertyData rpd = _confProp.get(rpdname);

198         /**   
199          * Change the value of a specific property inside one of the configuration
200          * sets
201          *
202          * @param rpdname Name of <code>RawPropertyData</code> that stores the
203          *               property values
204          * @param propname Property name to be modified
205          * @param newvalue New value of the property
206          */
207         public void changeConfigurationPropValue(String rpdname,String propname,
208                 String newvalue){
209             RawPropertyData rpd = _confProp.get(rpdname);
210             rpd.add(propname,newvalue);
211         }
212         
213         /** 
214          * Change the value of a specific property inside one of the configuration
215          * sets
216          *
217          * @param rpdname Name of <code>RawPropertyData</code> that stores the
218          *               property values
219          * @param propname Property name to be modified
220          * @param newvalue List of new values of the property
221          */
222         public void changeConfigurationPropValue(String rpdname,String propname,
223                 List<String> newvalue){
224             RawPropertyData rpd = _confProp.get(rpdname);
225             rpd.add(propname,newvalue);
226         }
227             
228         /**
229          * Get current value of the configuration
230          *
231          * @param rpdName Name of <code>RawPropertyData</code> that holds the 
232          *              configuration values
233          * @param propname Property name
234          * @return Either a <code>String</code> or <code>List</code> that 
235          *          contains value of the property
236          */
237         public Object getConfigurationPropValue(String rpdName,String propname){        
238             if (_confProp.containsKey(rpdName)) {
239                 RawPropertyData rpd = _confProp.get(rpdName);
240                 if ( rpd.getProperties().containsKey(propname)) {
241                     return rpd.getProperties().get(propname);
242                 } else
243                     return null;
244             } else
245                 return null;
246         }
247         
248         /**
249          * @return <code>Map</code> of <code>RawPropertyData</code> that has all the
250          *          configuration sets of this component. Each entry consists of 
251          *          String setname, RawPropertyData setproperties
252          */
253         public Map<String, RawPropertyData> getConfigurationPropMap(){
254             return _confProp;
255         }
256         
257         /**
258          * check if this component has a <code>RawPropertyData</code>
259          * configuration set with specified name
260          *
261          * @return <code>true</code> if the configuration with specified name exists
262          */
263         public boolean containsConfigurationSet(String name){
264             return _confProp.containsKey(name);
265         }

268         /**
269          * add all the properties that are in the Configurable Component model,
270          * by adding them as default
271          */
272         private void addDefaultProps(RawPropertyData rpd){
273             Map<String, ConfigurableProperty> completePropMap = this.getPropertyMap();
274             for (Map.Entry<String, ConfigurableProperty> propentry : completePropMap.entrySet()) {
275                 String propname = propentry.getKey();
276                 // System.out.println("***** prop"+propname);           
277     
278                 // if it doesn't exist yet
279                 if (!rpd.contains(propname)) {
280                     ConfigurableProperty prop = propentry.getValue();
281                     String defaultVal = prop.getDefault();
282                     if (defaultVal != null && !defaultVal.trim().isEmpty()) {
283                         rpd.add(propname, defaultVal);
284                         // System.out.println("***** add prop "+propname);
285                     } else {
286                         rpd.add(propname, (String)null);
287                         System.out.println("***** add null prop " + propname);
288                     }
289                 }
290             }
291         }
292     
293         /** 
294          * This function is used to create a new configuration set for this type of class
295          * The property values are set as their default values
296          * 
297          * @param setname Name of new configuration set
298          */
299         public void createNewSet(String setname){
300             // System.out.println("*****" + this);
301             // the set name given should have been checked by GUI
302             RawPropertyData newrpd = new RawPropertyData(setname,_componentClassName);        
303             addDefaultProps(newrpd); // fill them up with default properties
304             this.addConfigurationProp(newrpd); //add new rpd to this ConfigurableComponent's set        
305         }
