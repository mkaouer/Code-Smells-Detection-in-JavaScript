21     /**
22      * Applies automatic gain control (CMN)
23      */
24     public class BatchAGC extends BaseDataProcessor {
25     
26         private List<Data> cepstraList;
27         private double agc;
28     
29         public BatchAGC() {
30             initLogger();
31         }
32     
33         /* (non-Javadoc)
34          * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
35          */
36         @Override
37         public void newProperties(PropertySheet ps) throws PropertyException {
38             super.newProperties(ps);
39         }
40     
41         /** Initializes this BatchCMN. */
42         @Override
43         public void initialize() {
44             super.initialize();
45             cepstraList = new LinkedList<Data>();
46         }
47     
48         /**
49          * Returns the next Data object, which is a normalized cepstrum. Signal objects are returned unmodified.
50          *
51          * @return the next available Data object, returns null if no Data object is available
52          * @throws DataProcessingException if there is an error processing data
53          */
54         @Override
55         public Data getData() throws DataProcessingException {
56     
57             Data output = null;
58     
59             if (!cepstraList.isEmpty()) {
60                 output = cepstraList.remove(0);
61             } else {
62     	    agc = 0.0;
63         	    cepstraList.clear();
64                 // read the cepstra of the entire utterance, calculate and substract gain
65                 if (readUtterance() > 0) {
66                     normalizeList();
67                     output = cepstraList.remove(0);
68                 }
69             }
70     
71             return output;
72         }
73     
74     
75         /**
76          * Reads the cepstra of the entire Utterance into the cepstraList.
77          *
78          * @return the number cepstra (with Data) read
79          * @throws DataProcessingException if an error occurred reading the Data
80          */
81         private int readUtterance() throws DataProcessingException {
82     
83             Data input = null;
84             int numFrames = 0;
85     
86             while (true) {
87                 input = getPredecessor().getData();
88                 if (input == null) {
89     		break;
90     	    } else if (input instanceof DataEndSignal || input instanceof SpeechEndSignal) {
91                     cepstraList.add(input);
92                     break;
93     	    } else if (input instanceof DoubleData) {
94     	        cepstraList.add(input);
95     		double c0 = ((DoubleData)input).getValues()[0];
96     		if (agc < c0)
97     		    agc = c0;
98                 } else { // DataStartSignal or other Signal
99                     cepstraList.add(input);
100                 }
101                 numFrames++;
102             }
103     
104             return numFrames;
105         }
106     
107         /** Normalizes the list of Data. */
108         private void normalizeList() {
109             for (Data data : cepstraList) {
110                 if (data instanceof DoubleData) {
111                     ((DoubleData)data).getValues()[0] -= agc;
112                     System.out.println(((DoubleData)data).getValues()[0]);
113                 }
114             }
115         }
116     }
