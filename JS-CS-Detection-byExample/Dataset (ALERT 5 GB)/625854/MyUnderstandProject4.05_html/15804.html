64     	public static boolean isLegalRuleName(String name) {
65     		if (name == null) {
66     			return false;
67     		}
68     		name = stripRuleName(name);
69     
70     		if (name.endsWith(".*")) {
71     			name = name.substring(0, name.length() - 2);
72     		}
73     
74     		if (name.length() == 0) {
75     			return false;
76     		}
77     
78     		if ((name.startsWith(".")) || (name.endsWith("."))
79     				|| (name.indexOf("..") >= 0)) {
80     			return false;
81     		}
82     
83     		StringTokenizer localStringTokenizer = new StringTokenizer(name, ".");
84     
85     		while (localStringTokenizer.hasMoreTokens()) {
86     			String str = localStringTokenizer.nextToken();
87     
88     			int i = str.length();
89     
90     			if (i == 0)
91     				return false;
92     
93     			for (int j = 0; j < i; ++j) {
94     				if (!(isRuleNamePart(str.charAt(j))))
95     					return false;
96     			}
97     		}
98     		return true;
99     	}

101     	public static boolean isRuleNamePart(char c) {
102     		if (Character.isJavaIdentifierPart(c)) {
103     			return true;
104     		}
105     		return ((c == '!') || (c == '#') || (c == '%')
106     				|| (c == '&') || (c == '(')
107     				|| (c == ')') || (c == '+')
108     				|| (c == ',') || (c == '-')
109     				|| (c == '/') || (c == ':')
110     				|| (c == ';') || (c == '=')
111     				|| (c == '@') || (c == '[')
112     				|| (c == '\\') || (c == ']')
113     				|| (c == '^') || (c == '|') || (c == '~'));
114     	}

116     	public void setRuleName(String ruleName) {
117     		String strippedName = stripRuleName(ruleName);
118     		fullRuleName = strippedName;
119     
120     		int j = strippedName.lastIndexOf('.');
121     
122     		if (j < 0) {
123     			packageName = null;
124     			simpleGrammarName = null;
125     			simpleRuleName = strippedName;
126     		} else {
127     			int i = strippedName.lastIndexOf('.', j - 1);
128     
129     			if (i < 0) {
130     				packageName = null;
131     				simpleGrammarName = strippedName.substring(0, j);
132     				simpleRuleName = strippedName.substring(j + 1);
133     			} else {
134     				packageName = strippedName.substring(0, i);
135     				simpleGrammarName = strippedName.substring(i + 1, j);
136     				simpleRuleName = strippedName.substring(j + 1);
137     			}
138     		}
139     	}

141     	public static String stripRuleName(String name) {
142     		if ((name.startsWith("<")) && (name.endsWith(">"))) {
143     			return name.substring(1, name.length() - 1);
144     		}
145     		return name;
146     	}
