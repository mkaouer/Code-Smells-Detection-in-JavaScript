15     package edu.cmu.sphinx.tools.gui.util;
16     
17     /**
18      * This is a helper class to retrieve the javadoc information of 
19      * Sphinx classes
20      *
21      * @author Ariani
22      */
23     public class JavadocExtractor {
24         
25         private static String _last_class;
26         private static String _last_comment;
27         
28         /** execute the javadoc command to command line and get the result String */
29         private static String extractJavadocComment(
30                 String classname, String class_path, String source_path )
31         {
32             
33             if ( _last_class == null || ! _last_class.equals(classname)){
34                 try {                
35                     if( ! source_path.trim().endsWith("/")){ // add '/' at the end of path
36                         source_path = source_path.concat("/");
37                     }
38                     String source_file = classname.replace('.', '/') + ".java"; // convert class name to file path
39                     String commandLine = "javadoc -docletpath \"" + class_path +
40                         "\" -doclet edu.cmu.sphinx.tools.gui.util.DocletHelper \"" +
41                         source_path.trim() + source_file.trim() + '\"';
42     
43                     SysCommandExecutor cmdExecutor = SysCommandExecutor.getInstance(); 		   		
44                     cmdExecutor.runCommand(commandLine);
45     
46                     // String cmdError = cmdExecutor.getCommandError();
47                     // for debugging: System.out.println(cmdOutput);
48                     // for debugging: System.err.println(cmdError);
49                     
50                     _last_comment = cmdExecutor.getCommandOutput(); 
51                     _last_class = classname;
52                 }catch(Exception e){
53                        System.err.println("Exception "+e.getMessage());
54                 }
55             }        
56             return _last_comment;
57         }
58         
59         /** 
60          * do string manipulation to get the property comment
61          * @param classname Fully qualified class name that holds the property
62          * @param class_path Absolute path to the root of .class files;
63          *              must end in '\' or '/'
64          * @param source_path Absolute path to the source code root directory; 
65          *              must end in '\' or '/'
66          * @param prop_name Name of property to be searched
67          * @return Javadoc comment for the property that belongs to class name
68          */
69         public static String getJavadocComment(String classname, String class_path,
70                 String source_path, String prop_name)
71         {
72             String allComment;
73             
74             if (source_path == null || source_path.trim().isEmpty()) { // no path to source code
75                 return null;
76             }
77             
78             if ( _last_class == null || ! _last_class.trim().equals(classname.trim())){
79                 allComment = extractJavadocComment(classname,class_path,source_path);
80             }
81             else allComment = _last_comment;
82             //  System.out.println("comment : " + allComment);        
83             
84             // do processing for the comment
85             // one property comment block is defined by a '==='
86             String[] comments = allComment.split("={3}"); 
87             
88              // index 0 is javadoc's result comment
89             for (int i=1; i<comments.length;i++){
90                 //  System.out.println("comment" + i + " : " + comments[i]);
91                 String[] oneComment = comments[i].trim().split("={2}");
92                 if(oneComment[0].trim().equals(prop_name) && oneComment.length > 1){
93                     // property found, remove newlines in the comment string         
94                     oneComment[1] = oneComment[1].replaceAll("(\n|\r)+"," ");
95                     return oneComment[1]; 
96                 }
97             }
98             
99             return null; // does not find the comment for the class property
100         }
101     }
