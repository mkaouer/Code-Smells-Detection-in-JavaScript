23     	 * parse a text string against a particular rule from a particular grammar

29     		return parse(inputTokens, R, G, ruleName);

32     	public static RuleParse parse(String inputTokens[], Recognizer R,
33     			RuleGrammar G, String ruleName) {
34     		List<RuleParse> list = mparse(inputTokens, R, G, ruleName);
35     		if (list != null)
36     			for (RuleParse rp : list)
37     				if (G.isRulePublic(rp.getRuleName().getRuleName()))
38     					return rp;
39     		return null;
40     	}

48     	public static List<RuleParse> mparse(String inputTokens[], Recognizer R,
49     			RuleGrammar G, String ruleName) {
50     		RuleParser rp = new RuleParser();
51     		rp.recognizer = R;
52     		List<RuleParse> res = new ArrayList<RuleParse>();
53     		String[] rNames = ruleName == null ? G.listRuleNames()
54     				: new String[] { ruleName };
55     		for (String rName : rNames) {
56     			if (ruleName == null && !G.isEnabled(rName)) {
57     				continue;
58     			}
59     			Rule startRule = G.getRule(rName);
60     			if (startRule == null) {
61     				System.out.println("BAD RULENAME " + rName);
62     				continue;
63     			}
64     			List<TokenPos> p = rp.parse(G, startRule, inputTokens, 0);
65     			if (p != null && !p.isEmpty()) {
66     				for (TokenPos tp : p) {
67     					if (tp.getPos() == inputTokens.length) {
68     						res.add(new RuleParse(new RuleName(rName), (Rule) tp));
69     					}
70     				}
71     			}
72     		}
73     		return res.isEmpty() ? null : res;
74     	}

77     	 * Parse routine called recursively while traversing the Rule structure in a
78     	 * depth first manner. Returns a list of valid parses.
79     	 */
80     	private List<TokenPos> parse(RuleGrammar grammar, Rule r, String[] input,
81     			int pos) {
82     
83     		// System.out.println("PARSE " + r.getClass().getName() + ' ' + pos +
84     		// ' ' + r);
85     
86     		if (r instanceof RuleName)
87     			return parse(grammar, (RuleName) r, input, pos);
88     		if (r instanceof RuleToken)
89     			return parse(grammar, (RuleToken) r, input, pos);
90     		if (r instanceof RuleCount)
91     			return parse(grammar, (RuleCount) r, input, pos);
92     		if (r instanceof RuleTag)
93     			return parse(grammar, (RuleTag) r, input, pos);
94     		if (r instanceof RuleSequence)
95     			return parse(grammar, (RuleSequence) r, input, pos);
96     		if (r instanceof RuleAlternatives)
97     			return parse(grammar, (RuleAlternatives) r, input, pos);
98     
99     		System.out.println("ERROR UNKNOWN OBJECT " + r);
100     		return null;
101     	}

103     	/*
104     	 * RULE REFERENCES
105     	 */
106     	private List<TokenPos> parse(RuleGrammar grammar, RuleName rn,
107     			String[] input, int pos) {
108     		if (rn.getFullGrammarName() == null) {
109     			rn.setRuleName(grammar.getName() + '.' + rn.getSimpleRuleName());
110     		}
111     		String simpleName = rn.getSimpleRuleName();
112     		if (simpleName.equals("VOID"))
113     			return null;
114     		if (simpleName.equals("NULL")) {
115     			List<TokenPos> res = new ArrayList<TokenPos>();
116     			res.add(new ParsedRuleParse(rn, RuleName.NULL, pos));
117     			return res;
118     		}
119     		Rule ruleref = grammar.getRule(simpleName);
120     		if (rn.getFullGrammarName() != grammar.getName()) {
121     			ruleref = null;
122     		}
123     		if (ruleref == null) {
124     			String gname = rn.getFullGrammarName();
125     			// System.out.println("gname=" + gname);
126     			if (gname != null && !gname.isEmpty()) {
127     				RuleGrammar RG1 = recognizer.getRuleGrammar(gname);
128     				if (RG1 != null) {
129     					// System.out.println("simpleName=" + simpleName);
130     					ruleref = RG1.getRule(simpleName);
131     					// System.out.println("ruleRef=" + ruleref);
132     					grammar = RG1;
133     				} else {
134     					System.out.println("ERROR: UNKNOWN GRAMMAR " + gname);
135     					// Thread.dumpStack();
136     				}
137     			}
138     			if (ruleref == null) {
139     				System.out.println("ERROR: UNKNOWN RULE NAME "
140     						+ rn.getRuleName() + ' ' + rn);
141     				// Thread.dumpStack();
142     				return null;
143     			}
144     		}
145     		List<TokenPos> p = parse(grammar, ruleref, input, pos);
146     		if (p == null) {
147     			return null;
148     		}
149     		List<TokenPos> res = new ArrayList<TokenPos>();
150     		for (TokenPos tp : p) {
151     			if (tp instanceof ParsedEmptyToken) {
152     				res.add(tp);
153     				continue;
154     			}
155     			try {
156     				res.add(new ParsedRuleParse(rn, (Rule) tp, tp.getPos()));
157     			} catch (IllegalArgumentException e) {
158     				System.out.println("ERROR " + e);
159     			}
160     		}
161     		return res;
162     	}

164     	/*
165     	 * LITERAL TOKENS
166     	 */
167     	private List<TokenPos> parse(RuleGrammar grammar, RuleToken rt,
168     			String[] input, int pos) {
169     		if (pos >= input.length) {
170     			return null;
171     		}
172     		// System.out.println(rt.getText() + " ?= " + input[pos]);
173     		// TODO: what about case sensitivity ??????
174     		String tText = rt.getText().toLowerCase();
175     		if (tText.equals(input[pos]) || (input[pos].equals("%"))
176     				|| (input[pos].equals("*"))) {
177     			List<TokenPos> res = new ArrayList<TokenPos>();
178     			res.add(new ParsedRuleToken(rt.getText(), pos + 1));
179     			if (input[pos].equals("*")) {
180     				res.add(new ParsedRuleToken(rt.getText(), pos));
181     			}
182     			return res;
183     		} else {
184     			if (tText.indexOf(' ') < 0) {
185     				return null;
186     			}
187     			if (!tText.startsWith(input[pos])) {
188     				return null;
189     			}
190     			String ta[] = tokenize(tText);
191     			int j = 0;
192     			while (true) {
193     				if (j >= ta.length) {
194     					break;
195     				}
196     				if (pos >= input.length) {
197     					return null;
198     				}
199     				if (!ta[j].equals(input[pos])) {
200     					return null;
201     				}
202     				pos++;
203     				j++;
204     			}
205     			List<TokenPos> res = new ArrayList<TokenPos>();
206     			res.add(new ParsedRuleToken(rt.getText(), pos));
207     			return res;
208     		}
209     	}

213     	 */
214     	private List<TokenPos> parse(RuleGrammar grammar, RuleAlternatives ra,
215     			String[] input, int pos) {
216     		List<TokenPos> res = new ArrayList<TokenPos>();
217     		for (Rule rule : ra.getRules()) {
218     			List<TokenPos> p = parse(grammar, rule, input, pos);
219     			if (p != null)
220     				res.addAll(p);
221     		}
222     		return res;
223     	}

225     	/*
226     	 * RULESEQUENCE
227     	 */
228     	private List<TokenPos> parse(RuleGrammar grammar, RuleSequence rs,
229     			String[] input, int pos) {
230     		Rule[] rarry = rs.getRules();
231     		if (rarry == null || rarry.length == 0) {
232     			return null;
233     		}
234     		List<TokenPos> p = parse(grammar, rarry[0], input, pos);
235     		if (p == null) {
236     			return null;
237     		}
238     		List<TokenPos> res = new ArrayList<TokenPos>();
239     		// System.out.println("seq sz" + p.size());
240     		for (TokenPos tp : p) {
241     			// System.out.println("seq  " + p.get(j));
242     			int nPos = tp.getPos();
243     			if (rarry.length == 1) {
244     				if (tp instanceof ParsedEmptyToken) {
245     					res.add(tp);
246     					continue;
247     				}
248     				try {
249     					res.add(new ParsedRuleSequence(new Rule[] { (Rule) tp }, tp
250     							.getPos()));
251     				} catch (IllegalArgumentException e) {
252     					System.out.println(e);
253     				}
254     				continue;
255     			}
256     			Rule[] nra = Arrays.copyOfRange(rarry, 1, rarry.length);
257     			RuleSequence nrs = new RuleSequence(nra);
258     			// System.out.println("2parse " + nPos + nrs);
259     			List<TokenPos> q = parse(grammar, nrs, input, nPos);
260     			if (q == null) {
261     				continue;
262     			}
263     			// System.out.println("2 seq sz " + p.size());
264     			for (TokenPos tp1 : q) {
265     				// System.out.println("2 seq  " + q.get(k));
266     				// System.out.println("tp " + tp);
267     				// System.out.println("tp1 " + tp1);
268     				if (tp1 instanceof ParsedEmptyToken) {
269     					res.add(tp);
270     					continue;
271     				}
272     				if (tp instanceof ParsedEmptyToken) {
273     					res.add(tp1);
274     					continue;
275     				}
276     				Rule[] ra;
277     				if (tp1 instanceof RuleSequence) {
278     					RuleSequence r2 = (RuleSequence) tp1;
279     					Rule[] r2r = r2.getRules();
280     					ra = new Rule[r2r.length + 1];
281     					ra[0] = (Rule) tp;
282     					System.arraycopy(r2r, 0, ra, 1, r2r.length);
283     				} else {
284     					ra = new Rule[] { (Rule) tp, (Rule) tp1 };
285     				}
286     				try {
287     					res.add(new ParsedRuleSequence(ra, tp1.getPos()));
288     				} catch (IllegalArgumentException e) {
289     					System.out.println(e);
290     				}
291     			}
292     		}
293     		return res;
294     	}

297     	 * TAGS
298     	 */
299     	private List<TokenPos> parse(RuleGrammar grammar, RuleTag rtag,
300     			String[] input, int pos) {
301     		String theTag = rtag.getTag();
302     		// System.out.println("tag="+theTag);
303     		List<TokenPos> p = parse(grammar, rtag.getRule(), input, pos);
304     		if (p == null) {
305     			return null;
306     		}
307     		List<TokenPos> res = new ArrayList<TokenPos>();
308     		for (TokenPos tp : p) {
309     			if (tp instanceof ParsedEmptyToken) {
310     				res.add(tp);
311     				continue;
312     			}
313     			res.add(new ParsedRuleTag((Rule) tp, theTag, tp.getPos()));
314     		}
315     		return res;
316     	}

318     	/*
319     	 * RULECOUNT (e.g. [], *, or + )
320     	 */
321     	private List<TokenPos> parse(RuleGrammar grammar, RuleCount rc,
322     			String[] input, int pos) {
323     		int rcount = rc.getCount();
324     		ParsedEmptyToken empty = new ParsedEmptyToken(pos);
325     		List<TokenPos> p = parse(grammar, rc.getRule(), input, pos);
326     		if (p == null) {
327     			if (rcount == RuleCount.ONCE_OR_MORE)
328     				return null;
329     			List<TokenPos> res = new ArrayList<TokenPos>();
330     			res.add(empty);
331     			return res;
332     		}
333     		if (rcount != RuleCount.ONCE_OR_MORE) {
334     			p.add(empty);
335     		}
336     		if (rcount == RuleCount.OPTIONAL) {
337     			return p;
338     		}
339     		for (int m = 2; m <= input.length - pos; m++) {
340     			Rule[] ar = new Rule[m];
341     			Arrays.fill(ar, rc.getRule());
342     			List<TokenPos> q = parse(grammar, new RuleSequence(ar), input, pos);
343     			if (q == null) {
344     				return p;
345     			}
346     			p.addAll(q);
347     		}
348     		return p;
349     	}

351     	/*
352     	 * tokenize a string
353     	 */
354     	static String[] tokenize(String text) {
355     		StringTokenizer st = new StringTokenizer(text);
356     		int size = st.countTokens();
357     		String res[] = new String[size];
358     		int i = 0;
359     		while (st.hasMoreTokens()) {
360     			res[i++] = st.nextToken().toLowerCase();
361     		}
362     		return res;
363     	}
