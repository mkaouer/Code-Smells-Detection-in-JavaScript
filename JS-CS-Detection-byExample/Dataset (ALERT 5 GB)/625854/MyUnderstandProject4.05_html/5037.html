34     /**
35      * A simple form of the linguist.
36      * <p/>
37      * The flat linguist takes a Grammar graph (as returned by the underlying, configurable grammar), and generates a search
38      * graph for this grammar.
39      * <p/>
40      * It makes the following simplifying assumptions:
41      * <p/>
42      * <ul> <li>Zero or one word per grammar node <li> No fan-in allowed ever <li> No composites (yet) <li> Only Unit,
43      * HMMState, and pronunciation states (and the initial/final grammar state are in the graph (no word, alternative or
44      * grammar states attached). <li> Only valid transitions (matching contexts) are allowed <li> No tree organization of
45      * units <li> Branching grammar states are  allowed </ul>
46      * <p/>
47      * <p/>
48      * Note that all probabilities are maintained in the log math domain
49      */
50     public class FlatLinguist implements Linguist, Configurable {

374         /**
375          * Compiles the grammar into a sentence HMM. A GrammarJob is created for
376          * the initial grammar node and added to the GrammarJob queue. While there
377          * are jobs left on the grammar job queue, a job is removed from the queue
378          * and the associated grammar node is expanded and attached to the tails.
379          * GrammarJobs for the successors are added to the grammar job queue.
380          */
381         /**
382          * Compiles the grammar into a sentence HMM. A GrammarJob is created for the
383          * initial grammar node and added to the GrammarJob queue. While there are
384          * jobs left on the grammar job queue, a job is removed from the queue and
385          * the associated grammar node is expanded and attached to the tails.
386          * GrammarJobs for the successors are added to the grammar job queue.
387          */
388         protected Collection<SentenceHMMState> compileGrammar() {
389             initialGrammarState = grammar.getInitialNode();
390     
391             nodeStateMap = new HashMap<GrammarNode, GState>();
392             arcPool = new Cache<SentenceHMMStateArc>();
393     
394             List<GState> gstateList = new ArrayList<GState>();
395             TimerPool.getTimer(this, "Compile").start();
396     
397             // get the nodes from the grammar and create states
398             // for them. Add the non-empty gstates to the gstate list.
399             TimerPool.getTimer(this, "Create States").start();
400             for (GrammarNode grammarNode : grammar.getGrammarNodes()) {
401                 GState gstate = createGState(grammarNode);
402                 gstateList.add(gstate);
403             }
404             TimerPool.getTimer(this, "Create States").stop();
405             addStartingPath();
406     
407             // ensures an initial path to the start state
408             // Prep all the gstates, by gathering all of the contexts up
409             // this allows each gstate to know about its surrounding
410             // contexts
411             TimerPool.getTimer(this, "Collect Contexts").start();
412             for (GState gstate : gstateList)
413                 gstate.collectContexts();
414             TimerPool.getTimer(this, "Collect Contexts").stop();
415     
416             // now all gstates know all about their contexts, we can
417             // expand them fully
418             TimerPool.getTimer(this, "Expand States").start();
419             for (GState gstate : gstateList)
420                 gstate.expand();
421             TimerPool.getTimer(this, "Expand States").stop();
422     
423             // now that all states are expanded fully, we can connect all
424             // the states up
425             TimerPool.getTimer(this, "Connect Nodes").start();
426             for (GState gstate : gstateList)
427                 gstate.connect();
428             TimerPool.getTimer(this, "Connect Nodes").stop();
429     
430             SentenceHMMState initialState = findStartingState();
431     
432             // add an out-of-grammar branch if configured to do so
433             if (addOutOfGrammarBranch) {
434                 CIPhoneLoop phoneLoop = new CIPhoneLoop(phoneLoopAcousticModel, logPhoneInsertionProbability);
435                 SentenceHMMState firstBranchState = (SentenceHMMState)
436                         phoneLoop.getSearchGraph().getInitialState();
437                 initialState.connect(getArc(firstBranchState, logOne, logOutOfGrammarBranchProbability));
438             }
439     
440             searchGraph = new FlatSearchGraph(initialState);
441             TimerPool.getTimer(this, "Compile").stop();
442             // Now that we are all done, dump out some interesting
443             // information about the process
444             if (dumpGStates) {
445                 for (GrammarNode grammarNode : grammar.getGrammarNodes()) {
446                     GState gstate = getGState(grammarNode);
447                     gstate.dumpInfo();
448                 }
449             }
450             nodeStateMap = null;
451             arcPool = null;
452             return SentenceHMMState.collectStates(initialState);
453         }

511         /**
512          * Gets a SentenceHMMStateArc. The arc is drawn from a pool of arcs.
513          *
514          * @param nextState               the next state
515          * @param logLanguageProbability  the log language probability
516          * @param logInsertionProbability the log insertion probability
517          */
518         protected SentenceHMMStateArc getArc(SentenceHMMState nextState,
519                                              float logLanguageProbability,
520                                              float logInsertionProbability) {
521             SentenceHMMStateArc arc = new SentenceHMMStateArc(nextState,
522                     logLanguageProbability * languageWeight,
523                     logInsertionProbability);
524             SentenceHMMStateArc pooledArc = arcPool.cache(arc);
525             actualArcs.value = arcPool.getMisses();
526             totalArcs.value = arcPool.getHits() + arcPool.getMisses();
527             return pooledArc == null ? arc : pooledArc;
528         }

585         /**
586          * This is a nested class that is used to manage the construction of the states in a grammar node. There is one
587          * GState created for each grammar node. The GState is used to collect the entry and exit points for the grammar
588          * node and for connecting up the grammar nodes to each other.
589          */
590         protected class GState {
591     
592             private final Map<ContextPair, List<SearchState>> entryPoints = new HashMap<ContextPair, List<SearchState>>();
593             private final Map<ContextPair, List<SearchState>> exitPoints = new HashMap<ContextPair, List<SearchState>>();
594             private final Map<String, SentenceHMMState> existingStates = new HashMap<String, SentenceHMMState>();
595     
596             private final GrammarNode node;
597     
598             private final Set<UnitContext> rightContexts = new HashSet<UnitContext>();
599             private final Set<UnitContext> leftContexts = new HashSet<UnitContext>();
600             private Set<UnitContext> startingContexts;
601     
602             private int exitConnections;
603     //        private GrammarArc[] successors = null;
604     
605     
606             /**
607              * Creates a GState for a grammar node
608              *
609              * @param node the grammar node
610              */
611             protected GState(GrammarNode node) {
612                 this.node = node;
613                 nodeStateMap.put(node, this);
614             }
615     
616     
617             /**
618              * Retrieves the set of starting contexts for this node. The starting contexts are the set of Unit[] with a size
619              * equal to the maximum right context size.
620              *
621              * @return the set of starting contexts across nodes.
622              */
623             private Set<UnitContext> getStartingContexts() {
624                 if (startingContexts == null) {
625                     startingContexts = new HashSet<UnitContext>();
626                     // if this is an empty node, the starting context is
627                     // the set of starting contexts for all successor
628                     // nodes, otherwise, it is built up from each
629                     // pronunciation of this word
630                     if (node.isEmpty()) {
631                         GrammarArc[] arcs = getSuccessors();
632                         for (GrammarArc arc : arcs) {
633                             GState gstate = getGState(arc.getGrammarNode());
634                             startingContexts.addAll(gstate.getStartingContexts());
635                         }
636                     } else {
637     //                    int maxSize = getRightContextSize();
638                         Word word = node.getWord();
639                         Pronunciation[] prons = word.getPronunciations(null);
640                         for (Pronunciation pron : prons) {
641                             UnitContext startingContext = getStartingContext(pron);
642                             startingContexts.add(startingContext);
643                         }
644                     }
645                 }
646                 return startingContexts;
647             }
648     
649     
650             /**
651              * Retrieves the starting UnitContext for the given pronunciation
652              *
653              * @param pronunciation the pronunciation
654              * @return a UnitContext representing the starting context of the pronunciation
655              */
656             private UnitContext getStartingContext(Pronunciation pronunciation) {
657                 int maxSize = getRightContextSize();
658                 Unit[] units = pronunciation.getUnits();
659                 Unit[] context = units.length > maxSize ? Arrays.copyOf(units, maxSize) : units;
660                 return UnitContext.get(context);
661             }
662     
663     
664             /**
665              * Retrieves the set of trailing contexts for this node. the trailing contexts are the set of Unit[] with a size
666              * equal to the maximum left context size that align with the end of the node
667     
668              */
669             Collection<UnitContext> getEndingContexts() {
670                 Collection<UnitContext> endingContexts = new ArrayList<UnitContext>();
671                 if (!node.isEmpty()) {
672                     int maxSize = getLeftContextSize();
673                     Word word = node.getWord();
674                     Pronunciation[] prons = word.getPronunciations(null);
675                     for (Pronunciation pron : prons) {
676                         Unit[] units = pron.getUnits();
677                         int size = units.length;
678                         Unit[] context = size > maxSize ? Arrays.copyOfRange(units, size - maxSize, size) : units;
679                         endingContexts.add(UnitContext.get(context));
680                     }
681                     // add a silence to the ending context since we want to
682                     // include an optional transition to a silence unit at
683                     // the end of all words
684                     endingContexts.add(UnitContext.SILENCE);
685                 }
686                 return endingContexts;
687             }

805             /**
806              * Expands each GState into the sentence HMM States
807              */
808             public void expand() {
809                 // for each left context/starting context pair create a list
810                 // of starting states.
811                 for (UnitContext leftContext : leftContexts) {
812                     for (UnitContext startingContext : getStartingContexts()) {
813                         ContextPair contextPair = ContextPair.get(leftContext, startingContext);
814                         entryPoints.put(contextPair, new ArrayList<SearchState>());
815                     }
816                 }
817                 // if this is a final node don't expand it, just create a
818                 // state and add it to all entry points
819                 if (node.isFinalNode()) {
820                     GrammarState gs = new GrammarState(node);
821                     for (List<SearchState> epList : entryPoints.values()) {
822                         epList.add(gs);
823                     }
824                 } else if (!node.isEmpty()) {
825                     // its a full fledged node with a word
826                     // so expand it. Nodes without words don't need
827                     // to be expanded.
828                     for (UnitContext leftContext : leftContexts) {
829                         expandWord(leftContext);
830                     }
831                 } else {
832                     //if the node is empty, populate the set of entry and exit
833                     //points with a branch state. The branch state
834                     // branches to the successor entry points for this
835                     // state
836                     // the exit point should consist of the set of
837                     // incoming left contexts and outgoing right contexts
838                     // the 'entryPoint' table already consists of such
839                     // pairs so we can use that
840                     for (Map.Entry<ContextPair, List<SearchState>> entry : entryPoints.entrySet()) {
841                         ContextPair cp = entry.getKey();
842                         List<SearchState> epList = entry.getValue();
843                         SentenceHMMState bs = new BranchState(cp.getLeftContext().toString(),
844                                 cp.getRightContext().toString(), node.getID());
845                         epList.add(bs);
846                         addExitPoint(cp, bs);
847                     }
848                 }
849                 addEmptyEntryPoints();
850             }

853             /**
854              * Adds the set of empty entry points. The list of entry points are tagged with a context pair. The context pair
855              * represent the left context for the state and the starting context for the state, this allows states to be
856              * hooked up properly. However, we may be transitioning from states that have no right hand context (CI units
857              * such as SIL fall into this category). In this case we'd normally have no place to transition to since we add
858              * entry points for each starting context. To make sure that there are entry points for empty contexts if
859              * necessary, we go through the list of entry points and find all left contexts that have a right hand context
860              * size of zero. These entry points will need an entry point with an empty starting context. These entries are
861              * synthesized and added to the the list of entry points.
862              */
863             private void addEmptyEntryPoints() {
864                 Map<ContextPair, List<SearchState>> emptyEntryPoints = new HashMap<ContextPair, List<SearchState>>();
865                 for (Map.Entry<ContextPair, List<SearchState>> entry : entryPoints.entrySet()) {
866                     ContextPair cp = entry.getKey();
867                     if (needsEmptyVersion(cp)) {
868                         ContextPair emptyContextPair = ContextPair.get(cp.getLeftContext(), UnitContext.EMPTY);
869                         List<SearchState> epList = emptyEntryPoints.get(emptyContextPair);
870                         if (epList == null) {
871                             epList = new ArrayList<SearchState>();
872                             emptyEntryPoints.put(emptyContextPair, epList);
873                         }
874                         epList.addAll(entry.getValue());
875                     }
876                 }
877                 entryPoints.putAll(emptyEntryPoints);
878             }

907              * Expand the the word given the left context
908              *
909              * @param leftContext the left context
910              */
911             private void expandWord(UnitContext leftContext) {
912                 Word word = node.getWord();
913                 T("  Expanding word " + word + " for lc " + leftContext);
914                 Pronunciation[] pronunciations = word.getPronunciations(null);
915                 for (int i = 0; i < pronunciations.length; i++) {

917                 }

921             /**
922              * Expand the pronunciation given the left context
923              *
924              * @param leftContext   the left context
925              * @param pronunciation the pronunciation to expand
926              * @param which         unique ID for this pronunciation
927              */
928             // Each GState maintains a list of entry points. This list of
929             // entry points is used when connecting up the end states of
930             // one GState to the beginning states in another GState. The
931             // entry points are tagged by a ContextPair which represents
932             // the left context upon entering the state (the left context
933             // of the initial units of the state), and the right context
934             // of the previous states (corresponding to the starting
935             // contexts for this state).
936             //
937             // When expanding a proununciation, the following steps are
938             // taken:
939             //      1) Get the starting context for the pronunciation.
940             //      This is the set of units that correspond to the start
941             //      of the pronunciation.
942             //
943             //      2) Create a new PronunciationState for the
944             //      pronunciation.
945             //
946             //      3) Add the PronunciationState to the entry point table
947             //      (a hash table keyed by the ContextPair(LeftContext,
948             //      StartingContext).
949             //
950             //      4) Generate the set of context dependent units, using
951             //      the left and right context of the GState as necessary.
952             //      Note that there will be fan out at the end of the
953             //      pronunciation to allow for units with all of the
954             //      various right contexts. The point where the fan-out
955             //      occurs is the (length of the pronunciation - the max
956             //      right context size).
957             //
958             //      5) Attach each cd unit to the tree
959             //
960             //      6) Expand each cd unit into the set of HMM states
961             //
962             //      7) Attach the optional and looping back silence cd
963             //      unit
964             //
965             //      8) Collect the leaf states of the tree and add them to
966             //      the exitStates list.
967             private void expandPronunciation(UnitContext leftContext,
968                                              Pronunciation pronunciation, int which) {
969                 UnitContext startingContext = getStartingContext(pronunciation);
970                 // Add the pronunciation state to the entry point list
971                 // (based upon its left and right context)
972                 String pname = "P(" + pronunciation.getWord() + '[' + leftContext
973                         + ',' + startingContext + "])-G" + getNode().getID();
974                 PronunciationState ps = new PronunciationState(pname, pronunciation, which);
975                 T("     Expanding " + ps.getPronunciation() + " for lc " + leftContext);
976                 ContextPair cp = ContextPair.get(leftContext, startingContext);
977                 List<SearchState> epList = entryPoints.get(cp);
978                 if (epList == null) {
979                     throw new Error("No EP list for context pair " + cp);
980                 } else {
981                     epList.add(ps);
982                 }
983                 Unit[] units = pronunciation.getUnits();
984                 int fanOutPoint = units.length - getRightContextSize();
985                 if (fanOutPoint < 0) {
986                     fanOutPoint = 0;
987                 }
988                 SentenceHMMState tail = ps;
989                 for (int i = 0; tail != null && i < fanOutPoint; i++) {
990                     tail = attachUnit(ps, tail, units, i, leftContext, UnitContext.EMPTY);
991                 }
992                 SentenceHMMState branchTail = tail;
993                 for (UnitContext finalRightContext : rightContexts) {
994                     tail = branchTail;
995                     for (int i = fanOutPoint; tail != null && i < units.length; i++) {
996                         tail = attachUnit(ps, tail, units, i, leftContext, finalRightContext);
997                     }
998                 }
999             }

1002             /**
1003              * Attaches the given unit to the given tail, expanding the unit if necessary. If an identical unit is already
1004              * attached, then this path is folded into the existing path.
1005              *
1006              * @param parent       the parent state
1007              * @param tail         the place to attach the unit to
1008              * @param units        the set of units
1009              * @param which        the index into the set of units
1010              * @param leftContext  the left context for the unit
1011              * @param rightContext the right context for the unit
1012              * @return the tail of the added unit (or null if the path was folded onto an already expanded path.
1013              */
1014             private SentenceHMMState attachUnit(PronunciationState parent,
1015                                                 SentenceHMMState tail, Unit[] units, int which,
1016                                                 UnitContext leftContext, UnitContext rightContext) {
1017                 Unit[] lc = getLC(leftContext, units, which);
1018                 Unit[] rc = getRC(units, which, rightContext);
1019                 UnitContext actualRightContext = UnitContext.get(rc);
1020                 LeftRightContext context = LeftRightContext.get(lc, rc);
1021                 Unit cdUnit = unitManager.getUnit(units[which].getName(), units[which] .isFiller(), context);
1022                 UnitState unitState = new ExtendedUnitState(parent, which, cdUnit);
1023                 float logInsertionProbability;
1024                 if (unitState.getUnit().isSilence()) {
1025                     logInsertionProbability = logSilenceInsertionProbability;
1026                 } else if (unitState.getUnit().isFiller()) {
1027                     logInsertionProbability = logFillerInsertionProbability;
1028                 } else if (unitState.getWhich() == 0) {
1029                     logInsertionProbability = logWordInsertionProbability;
1030                 } else {
1031                     logInsertionProbability = logUnitInsertionProbability;
1032                 }
1033                 // check to see if this state already exists, if so
1034                 // branch to it and we are done, otherwise, branch to
1035                 // the new state and expand it.
1036                 SentenceHMMState existingState = getExistingState(unitState);
1037                 if (existingState != null) {
1038                     attachState(tail, existingState, logOne, logInsertionProbability);
1039                     // T(" Folding " + existingState);
1040                     return null;
1041                 } else {
1042                     attachState(tail, unitState, logOne, logInsertionProbability);
1043                     addStateToCache(unitState);
1044                     // T(" Attaching " + unitState);
1045                     tail = expandUnit(unitState);
1046                     // if we are attaching the last state of a word, then
1047                     // we add it to the exitPoints table. the exit points
1048                     // table is indexed by a ContextPair, consisting of
1049                     // the exiting left context and the right context.
1050                     if (unitState.isLast()) {
1051                         UnitContext nextLeftContext = generateNextLeftContext(leftContext, units[which]);
1052                         ContextPair cp = ContextPair.get(nextLeftContext, actualRightContext);
1053                         // T(" Adding to exitPoints " + cp);
1054                         addExitPoint(cp, tail);
1055                         // if we have encountered a last unit with a right
1056                         // context of silence, then we add a silence unit
1057                         // to this unit. the silence unit has a self
1058                         // loopback.
1059                         if (actualRightContext == UnitContext.SILENCE) {
1060                             SentenceHMMState silTail;
1061                             UnitState silUnit = new ExtendedUnitState(parent, which + 1, UnitManager.SILENCE);
1062                             SentenceHMMState silExistingState = getExistingState(silUnit);
1063                             if (silExistingState != null) {
1064                                 attachState(tail, silExistingState, logOne, logSilenceInsertionProbability);
1065                             } else {
1066                                 attachState(tail, silUnit, logOne, logSilenceInsertionProbability);
1067                                 addStateToCache(silUnit);
1068                                 silTail = expandUnit(silUnit);
1069                                 ContextPair silCP = ContextPair.get(UnitContext.SILENCE, UnitContext.EMPTY);
1070                                 addExitPoint(silCP, silTail);
1071                             }
1072                         }
1073                     }
1074                     return tail;
1075                 }
1076             }

1094             /**
1095              * Get the left context for a unit based upon the left context size, the entry left context and the current
1096              * unit.
1097              *
1098              * @param left  the entry left context
1099              * @param units the set of units
1100              * @param index the index of the current unit
1101              */
1102             private Unit[] getLC(UnitContext left, Unit[] units, int index) {
1103                 Unit[] leftUnits = left.getUnits();
1104                 int curSize = leftUnits.length + index;
1105                 int actSize = Math.min(curSize, getLeftContextSize(units[index]));
1106                 int leftIndex = curSize - actSize;
1107                 return combineUnits(leftUnits, units, actSize, leftIndex);
1108             }

1110             /**
1111              * Get the right context for a unit based upon the right context size, the exit right context and the current
1112              * unit.
1113              *
1114              * @param units the set of units
1115              * @param index the index of the current unit
1116              * @param right the exiting right context
1117              */
1118             private Unit[] getRC(Unit[] units, int index, UnitContext right) {
1119                 Unit[] rightUnits = right.getUnits();
1120                 int leftIndex = index + 1;
1121                 int curSize = units.length - leftIndex + rightUnits.length;
1122                 int actSize = Math.min(curSize, getRightContextSize(units[index]));
1123                 return combineUnits(units, rightUnits, actSize, leftIndex);
1124             }
1125     
1126             private Unit[] combineUnits(Unit[] left, Unit[] right, int size, int leftIndex) {
1127                 if (size == left.length && leftIndex == 0)
1128                     return left;
1129                 if (leftIndex >= left.length)
1130                     return size == right.length ? right : Arrays.copyOf(right, size);
1131                 Unit[] units = Arrays.copyOfRange(left, leftIndex, leftIndex + size);
1132                 int fromLeft = left.length - leftIndex;
1133                 if (fromLeft < size)
1134                     System.arraycopy(right, 0, units, fromLeft, size - fromLeft);
1135                 return units;
1136             }

1217              * Given a unit state, return the set of sentence hmm states associated with the unit
1218              *
1219              * @param unitState the unit state of intereset
1220              * @return the hmm tree for the unit
1221              */
1222             private HMMStateState getHMMStates(UnitState unitState) {
1223                 HMMStateState hmmTree;
1224                 HMMStateState finalState;
1225                 Unit unit = unitState.getUnit();
1226                 HMMPosition position = unitState.getPosition();
1227                 HMM hmm = acousticModel.lookupNearestHMM(unit, position, false);
1228                 HMMState initialState = hmm.getInitialState();
1229                 hmmTree = new HMMStateState(unitState, initialState);
1230                 attachState(unitState, hmmTree, logOne, logOne);
1231                 addStateToCache(hmmTree);
1232                 finalState = expandHMMTree(unitState, hmmTree);
1233                 return finalState;

1238              * Expands the given hmm state tree
1239              *
1240              * @param parent the parent of the tree
1241              * @param tree   the tree to expand
1242              * @return the final state in the tree
1243              */
1244             private HMMStateState expandHMMTree(UnitState parent, HMMStateState tree) {
1245                 HMMStateState retState = tree;
1246                 for (HMMStateArc arc : tree.getHMMState().getSuccessors()) {
1247                     HMMStateState newState;
1248                     if (arc.getHMMState().isEmitting()) {
1249                         newState = new HMMStateState(parent, arc.getHMMState());
1250                     } else {
1251                         newState = new NonEmittingHMMState(parent, arc.getHMMState());
1252                     }
1253                     SentenceHMMState existingState = getExistingState(newState);
1254                     float logProb = arc.getLogProbability();
1255                     if (existingState != null) {
1256                         attachState(tree, existingState, logOne, logProb);
1257                     } else {
1258                         attachState(tree, newState, logOne, logProb);
1259                         addStateToCache(newState);
1260                         retState = expandHMMTree(parent, newState);
1261                     }
1262                 }
1263                 return retState;
1264             }

1267             /**
1268              * Connect up all of the GStates. Each state now has a table of exit points. These exit points represent tail
1269              * states for the node. Each of these tail states is tagged with a ContextPair, that indicates what the left
1270              * context is (the exiting context) and the right context (the entering context) for the transition. To connect
1271              * up a state, the connect does the following: 1) Iterate through all of the grammar successors for this state
1272              * 2) Get the 'entry points' for the successor that match the exit points. 3) Hook them up.
1273              * <p/>
1274              * Note that for a task with 1000 words this will involve checking on the order of 35,000,000 connections and
1275              * making about 2,000,000 connections
1276              */
1277             public void connect() {
1278                 // T("Connecting " + node.getWord());
1279                 for (GrammarArc arc : getSuccessors()) {
1280                     GState gstate = getGState(arc.getGrammarNode());
1281                     if (!gstate.getNode().isEmpty()
1282                             && gstate.getNode().getWord().getSpelling().equals(
1283                             Dictionary.SENTENCE_START_SPELLING)) {
1284                         continue;
1285                     }
1286                     float probability = arc.getProbability();
1287                     // adjust the language probability by the number of
1288                     // pronunciations. If there are 3 ways to say the
1289                     // word, then each pronunciation gets 1/3 of the total
1290                     // probability.
1291                     if (spreadWordProbabilitiesAcrossPronunciations && !gstate.getNode().isEmpty()) {
1292                         int numPronunciations = gstate.getNode().getWord().getPronunciations(null).length;
1293                         probability -= logMath.linearToLog(numPronunciations);
1294                     }
1295                     float fprob = probability;
1296                     for (Map.Entry<ContextPair, List<SearchState>> entry : exitPoints.entrySet()) {
1297                         List<SearchState> destEntryPoints = gstate.getEntryPoints(entry.getKey());
1298                         if (destEntryPoints != null) {
1299                             List<SearchState> srcExitPoints = entry.getValue();
1300                             connect(srcExitPoints, destEntryPoints, fprob);
1301                         }
1302                     }
1303                 }
1304             }

1307             /**
1308              * connect all the states in the source list to the states in the destination list
1309              *
1310              * @param sourceList the set of source states
1311              * @param destList   the set of destination states.
1312     
1313              */
1314             private void connect(List<SearchState> sourceList, List<SearchState> destList, float logLangProb) {
1315                 for (SearchState source : sourceList) {
1316                     SentenceHMMState sourceState = (SentenceHMMState) source;
1317                     for (SearchState dest : destList) {
1318                         SentenceHMMState destState = (SentenceHMMState) dest;
1319                         sourceState.connect(getArc(destState, logLangProb, logOne));
1320                         exitConnections++;
1321                     }
1322                 }
1323             }

1327              * Attaches one SentenceHMMState as a child to another, the transition has the given probability
1328              *
1329              * @param prevState              the parent state
1330              * @param nextState              the child state
1331              * @param logLanguageProbablity  the language probability of transition in the LogMath log domain
1332              * @param logInsertionProbablity insertion probability of transition in the LogMath log domain
1333              */
1334             protected void attachState(SentenceHMMState prevState,
1335                                        SentenceHMMState nextState,
1336                                        float logLanguageProbablity, 
1337                                        float logInsertionProbablity) {
1338                 prevState.connect(getArc(nextState,
1339                         logLanguageProbablity, logInsertionProbablity));
1340                 if (showCompilationProgress && totalStateCounter++ % 1000 == 0) {
1341                     System.out.print(".");
1342                 }
1343             }
