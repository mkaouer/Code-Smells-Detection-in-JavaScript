17     /**
18      * Combines the various unit feature stream scores in a CombineToken. This SameTimeScoreCombiner assumes that there can
19      * be more than one token from each feature stream. It will take the highest scoring token from each stream <b>at each
20      * last combine time</b>, combine their scores, and choose the one with the highest combined score.
21      * <p/>
22      * For example, if tokens T1 and T2 from feature F1 had time stamps (i.e., the last combine time) t1 and t2, and tokens
23      * Ta and Tb from feature F2 had time stamps t1 and t2, we must compare combinedScore(T1,Ta)  and combineScore(T2,Tb),
24      * and retain the one for which the combined score is higher.
25      * <p/>
26      * All scores are maintained internally in the LogMath logbase
27      */
28     public class SameTimeScoreCombiner implements ScoreCombiner {
29     
30         private SameTimeTokensReader reader;
31     
32         private static Comparator<ParallelToken> combineTimeComparator;
33     
34     
35         private static Comparator<ParallelToken> getCombineTimeComparator() {
36             if (combineTimeComparator == null) {
37                 combineTimeComparator = new Comparator<ParallelToken>() {
38                     public int compare(ParallelToken t1, ParallelToken t2) {
39     
40                         if (t1.getLastCombineTime() < t2.getLastCombineTime()) {
41                             return -1;
42                         } else if (t1.getLastCombineTime() ==
43                                 t2.getLastCombineTime()) {
44                             return 0;
45                         } else {
46                             return 1;
47                         }
48                     }
49                 };
50             }
51             return combineTimeComparator;
52         }
53     
54     
55         /**
56          * Constructs a SameTimeScoreCombiner.
57          *
58          * @param timeDifference the maximum difference in time between tokens for them to be considered approximately the
59          *                       'same time'
60          */
61         public SameTimeScoreCombiner(int timeDifference) {
62             reader = new SameTimeTokensReader(timeDifference);
63         }
64     
65     
66         /**
67          * Combines the scores from all the feature stream in the given CombineToken. The feature stream(s) are represented
68          * in the form of ParallelTokens in the given CombineToken.
69          *
70          * @param combineToken the CombineToken on which to combine the feature stream scores
71          */
72         public void combineScore(CombineToken combineToken) {
73     
74             // sort all the parallel tokens in the CombineToken
75             // according to their last combine time
76     
77             List<ParallelToken> tokenList = new ArrayList<ParallelToken>(combineToken.getParallelTokens());
78             Collections.sort(tokenList, getCombineTimeComparator());
79     
80             // System.out.println("TokenList size: " + tokenList.size());
81     
82             double logHighestCombinedScore = Double.NEGATIVE_INFINITY;
83             List<ParallelToken> highestList = null;
84     
85             reader.reset(tokenList);
86     
87             while (reader.hasMoreTokens()) {
88                 List<ParallelToken> sameTimeList = reader.readNextSameTimeTokens();
89                 // checkSameTime(sameTimeList);
90                 double logCombinedScore = getCombinedScore(sameTimeList);
91                 // System.out.println("logCombinedScore = " + logCombinedScore);
92                 if (logCombinedScore > logHighestCombinedScore ||
93                         logHighestCombinedScore == Double.NEGATIVE_INFINITY) {
94                     logHighestCombinedScore = logCombinedScore;
95                     highestList = sameTimeList;
96                 }
97             }
98     
99             assert highestList != null && !highestList.isEmpty();
100             // System.out.println("Highest TokenList size: " + highestList.size());
101     
102             for (ParallelToken token : highestList) {
103                 token.setCombinedScore((float) logHighestCombinedScore);
104             }
105     
106             combineToken.clear();
107             combineToken.addAll(highestList);
108             combineToken.setCombinedScore((float) logHighestCombinedScore);
109         }
110     
111     
112         /**
113          * Calculates the combined score of the given list of tokens. This method assumes that all tokens have the same last
114          * combine time. It will retain only the highest scoring token of each stream and remove all the other lower scoring
115          * tokens from the list.
116          *
117          * @param sameTimeTokenList the token list
118          * @return the combined log score
119          */
120         private double getCombinedScore(List<ParallelToken> sameTimeTokenList) {
121             Map<String, ParallelToken> uniqueMap = new HashMap<String, ParallelToken>();
122     
123     //      int combineTime = -1;
124     
125             // first retain the highest scoring token from each stream 
126             for (ParallelToken token : sameTimeTokenList) {
127     //	    	if (combineTime == -1) {
128     //					combineTime = token.getLastCombineTime();
129     //	    	} else {
130     //					assert token.getLastCombineTime() == combineTime;
131     //	    	}
132                 String modelName = token.getModelName();
133                 ParallelToken tokenInMap = uniqueMap.get(modelName);
134                 if (tokenInMap == null) {
135                     uniqueMap.put(modelName, token);
136                 } else {
137                     if (token.getFeatureScore() > tokenInMap.getFeatureScore()) {
138                         uniqueMap.put(modelName, token);
139                     }
140                 }
141             }
142     
143             // clear the list
144             sameTimeTokenList.clear();
145             assert sameTimeTokenList.isEmpty();
146     
147             // now calculate the combinedScore
148             double logTotalScore = 0;
149     
150             for (ParallelToken pToken : uniqueMap.values()) {
151                 // System.out.println("Highest: " + tokenToString(pToken));
152                 sameTimeTokenList.add(pToken);
153     
154                 // in linear domain, the following expression is:
155                 // score = pToken.getFeatureScore()^pToken.getEta()
156     
157                 double logScore = pToken.getFeatureScore() * pToken.getEta();
158     
159                 // in linear domain, the following expression is:
160                 // totalScore *= score
161     
162                 logTotalScore += logScore;
163             }
164     
165             return logTotalScore;
166         }

171             System.out.print("SameTimeList: ");
172             for (ParallelToken token : tokenList) {

179         private String tokenToString(ParallelToken token) {
180             return (" (" + token.getLastCombineTime() + ',' +
181                     token.getModelName() + ',' +
182                     token.getFeatureScore() + ") ");
183         }
184     }

187     class SameTimeTokensReader {
188     
189         private ListIterator<ParallelToken> listIterator;
190         private int timeDifference;
191     
192     
193         SameTimeTokensReader(int timeDiff) {
194             timeDifference = timeDiff;
195             System.out.println("MaxTimeDiff: " + timeDifference);
196         }
197     
198     
199         public void reset(List<ParallelToken> tokenList) {
200             this.listIterator = tokenList.listIterator();
201         }
202     
203     
204         public boolean hasMoreTokens() {
205             return listIterator.hasNext();
206         }
207     
208     
209         public boolean isSameTime(int time1, int time2) {
210             return (Math.abs(time1 - time2) <= timeDifference);
211         }
212     
213     
214         public List<ParallelToken> readNextSameTimeTokens() {
215             List<ParallelToken> subList = new LinkedList<ParallelToken>();
216     
217             if (listIterator.hasNext()) {
218     
219                 // first token
220     
221                 ParallelToken token = listIterator.next();
222                 subList.add(token);
223                 int lastCombineTime = token.getLastCombineTime();
224     
225                 // then the next token(s) with the same combine time
226     
227                 while (listIterator.hasNext()) {
228                     ParallelToken nextToken = listIterator.next();
229                     if (isSameTime(nextToken.getLastCombineTime(),
230                             lastCombineTime)) {
231                         subList.add(nextToken);
232                     } else {
233                         // return to the previous element
234                         listIterator.previous();
235                         break;
236                     }
237                 }
238             }
239     
240             return subList;
241         }
242     }
