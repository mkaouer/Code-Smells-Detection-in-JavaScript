5     /**
6      * Implements a Senone that contains a cache of the last scored data.
7      * <p>
8      * Subclasses should implement the abstract {@link #calculateScore} method,
9      * which is called by the {@link #getScore} method to calculate the score
10      * for each cache miss.
11      * <p>
12      * Note: this implementation is thread-safe and can be safely used
13      * across different threads without external synchronization.
14      *
15      * @author Yaniv Kunda
16      */
17     public abstract class ScoreCachingSenone implements Senone {
18     
19         private class ScoreCache {
20             private final Data feature;
21             private final float score;
22     
23             public ScoreCache(Data feature, float score) {
24                 this.feature = feature;
25                 this.score = score;
26             }
27         }
28     
29         private volatile ScoreCache scoreCache = new ScoreCache(null, 0.0f);
30     
31         /**
32          * Gets the cached score for this senone based upon the given feature.
33          * If the score was not cached, it is calculated using {@link #calculateScore},
34          * cached, and then returned.  
35          */
36         @Override
37         public float getScore(Data feature) {
38             ScoreCache cached = scoreCache;
39             if (feature != cached.feature) {
40                 cached = new ScoreCache(feature, calculateScore(feature));
41                 scoreCache = cached;
42             }
43             return cached.score;
44         }
45     
46         /**
47          * Calculates the score for this senone based upon the given feature.
48          *
49          * @param feature the feature vector to score this senone against
50          * @return the score for this senone in LogMath log base
51          */
52         protected abstract float calculateScore(Data feature);
