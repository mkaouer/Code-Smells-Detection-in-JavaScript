17          */

31         @Override

33             buildWordLattice = ps.getBoolean(PROP_BUILD_WORD_LATTICE);
34             keepAllTokens = ps.getBoolean(PROP_KEEP_ALL_TOKENS);
35         }

37         /**
38          * Find the token to use as a predecessor in resultList given a candidate
39          * predecessor. There are three cases here:
40          * 
41          * <ul>
42          * <li>We want to store everything in resultList. In that case
43          * {@link #keepAllTokens} is set to true and we just store everything that
44          * was built before.
45          * <li>We are only interested in sequence of words. In this case we just
46          * keep word tokens and ignore everything else. In this case timing and
47          * scoring information is lost since we keep scores in emitting tokens.
48          * <li>We want to keep words but we want to keep scores to build a lattice
49          * from the result list later and {@link #buildWordLattice} is set to true.
50          * In this case we want to insert intermediate token to store the score and
51          * this token will be used during lattice path collapse to get score on
52          * edge. See {@link edu.cmu.sphinx.result.Lattice} for details of resultList
53          * compression.
54          * </ul>
55          * 
56          * @param token
57          *            the token of interest
58          * @return the immediate successor word token
59          */
60         protected Token getResultListPredecessor(Token token) {
61     
62             if (keepAllTokens) {
63                 return token;
64             }
65     
66             if(!buildWordLattice) {
67                 if (token.isWord())
68                     return token;
69                 else
70                     return token.getPredecessor();
71             }
72     
73             float logAcousticScore = 0.0f;
74             float logLanguageScore = 0.0f;
75             float logInsertionScore = 0.0f;
76     
77             while (token != null && !token.isWord()) {
78                 logAcousticScore += token.getAcousticScore();
79                 logLanguageScore += token.getLanguageScore();
80                 logInsertionScore += token.getInsertionScore();
81                 token = token.getPredecessor();
82             }
83     
84             return new Token(token, token.getScore(), logInsertionScore, logAcousticScore, logLanguageScore);
85         }
