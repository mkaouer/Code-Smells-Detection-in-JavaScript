34     /**
35      * Creates a dictionary by quickly reading in an ASCII-based Sphinx-3 format dictionary. It is called the FastDictionary
36      * because the loading is fast. When loaded the dictionary just loads each line of the dictionary into the hash table,
37      * assuming that most words are not going to be used. Only when a word is actually used is its pronunciations massaged
38      * into an array of pronunciations.
39      * <p/>
40      * The format of the ASCII dictionary that it explains is the same as the {@link FullDictionary FullDictionary}, i.e.,
41      * the word, followed by spaces or tab, followed by the pronunciation(s). For example, a digits dictionary will look
42      * like:
43      * <p/>
44      * <pre>
45      *  ONE HH W AH N
46      *  ONE(2) W AH N
47      *  TWO T UW
48      *  THREE TH R IY
49      *  FOUR F AO R
50      *  FIVE F AY V
51      *  SIX S IH K S
52      *  SEVEN S EH V AH N
53      *  EIGHT EY T
54      *  NINE N AY N
55      *  ZERO Z IH R OW
56      *  ZERO(2) Z IY R OW
57      *  OH OW
58      * </pre>
59      * <p/>
60      * <p/>
61      * In the above example, the words "one" and "zero" have two pronunciations each.
62      */
63

67         /**
68          * The property for the custom dictionary file paths. This addenda property points to a possibly
69          * empty list of URLs to dictionary addenda.  Each addendum should contain word pronunciations in the same Sphinx-3
70          * dictionary format as the main dictionary.  Words in the addendum are added after the words in the main dictionary
71          * and will override previously specified pronunciations.  If you wish to extend the set of pronunciations for a
72          * particular word, add a new pronunciation by number.  For example, in the following addendum, given that the
73          * aforementioned main dictionary is specified, the pronunciation for 'EIGHT' will be overridden by the addenda,
74          * while the pronunciation for 'SIX' and 'ZERO' will be augmented and a new pronunciation for 'ELEVEN' will be
75          * added.
76          * <pre>
77          *          EIGHT   OW T
78          *          SIX(2)  Z IH K S
79          *          ZERO(3)  Z IY Rl AH
80          *          ELEVEN   EH L EH V AH N
81          * </pre>
82          */

207         @Override
208         public void allocate() throws IOException {
209             if (!allocated) {
210                 dictionary = new HashMap<String, String>();
211                 wordDictionary = new HashMap<String, Word>();
212     
213                 Timer loadTimer = TimerPool.getTimer(this, "Load Dictionary");
214                 fillerWords = new HashSet<String>();
215     
216                 loadTimer.start();
217     
218                 logger.info("Loading dictionary from: " + wordDictionaryFile);
219     
220                 loadDictionary(wordDictionaryFile.openStream(), false);
221     
222                 loadCustomDictionaries(addendaUrlList);
223     
224                 logger.info("Loading filler dictionary from: " +
225                         fillerDictionaryFile);
226     
227                 loadDictionary(fillerDictionaryFile.openStream(), true);
228     
229                 loadTimer.stop();
230             }
231     
232         }

250         /**
251          * Loads the given sphinx3 style simple dictionary from the given InputStream. The InputStream is assumed to contain
252          * ASCII data.
253          *
254          * @param inputStream  the InputStream of the dictionary
255          * @param isFillerDict true if this is a filler dictionary, false otherwise
256          * @throws java.io.IOException if there is an error reading the dictionary
257          */
258         protected void loadDictionary(InputStream inputStream, boolean isFillerDict)
259                 throws IOException {
260             InputStreamReader isr = new InputStreamReader(inputStream);
261             BufferedReader br = new BufferedReader(isr);
262             String line;
263     
264             while ((line = br.readLine()) != null) {
265                 if (!line.isEmpty()) {
266                     int spaceIndex = line.indexOf(' ');
267                     int spaceIndexTab = line.indexOf('\t');
268                     if (spaceIndex == -1) {
269                         // Case where there's no blank character
270                         spaceIndex = spaceIndexTab;
271                     } else if ((spaceIndexTab >= 0) && (spaceIndexTab < spaceIndex)) {
272                         // Case where there's a blank and a tab, but the tab
273                         // precedes the blank
274                         spaceIndex = spaceIndexTab;
275                     }
276                     // TODO: throw an exception if spaceIndex == -1 ?
277                     if (spaceIndex == -1) {
278                         throw new Error("Error loading word: " + line);
279                     }
280                     String word = line.substring(0, spaceIndex);
281                     word = word.toLowerCase();
282                     if (isFillerDict) {
283                         dictionary.put(word, (FILLER_TAG + line));
284                         fillerWords.add(word);
285                     } else {
286                         dictionary.put(word, line);
287                     }
288                 }
289             }
290     
291             br.close();
292             isr.close();
293             inputStream.close();
294         }

342         /**
343          * Returns a Word object based on the spelling and its classification. The behavior of this method is also affected
344          * by the properties wordReplacement, allowMissingWords, and createMissingWords.
345          *
346          * @param text the spelling of the word of interest.
347          * @return a Word object
348          * @see edu.cmu.sphinx.linguist.dictionary.Word
349          */
350         @Override
351         public Word getWord(String text) {
352             text = text.toLowerCase();
353             Word wordObject = wordDictionary.get(text);
354     
355             if (wordObject != null) {
356                 return wordObject;
357             }
358     
359             String word = dictionary.get(text);
360             if (word == null) { // deal with 'not found' case
361                 logger.warning("Missing word: " + text);
362                 if (wordReplacement != null) {
363                     wordObject = getWord(wordReplacement);
364                 } else if (allowMissingWords) {
365                     if (createMissingWords) {
366                         wordObject = createWord(text, null, false);
367                     }
368                 }
369             } else { // first lookup for this string
370                 wordObject = processEntry(text);
371             }
372     
373             return wordObject;
374         }

378          * Create a Word object with the given spelling and pronunciations, and insert it into the dictionary.
379          *
380          * @param text          the spelling of the word
381          * @param pronunciation the pronunciation of the word
382          * @param isFiller      if <code>true</code> this is a filler word
383          * @return the word
384          */
385         private Word createWord(String text, Pronunciation[] pronunciation,
386                                 boolean isFiller) {
387             Word word = new Word(text, pronunciation, isFiller);
388             dictionary.put(text, word.toString());
389             return word;
390         }

393         /**
394          * Processes a dictionary entry. When loaded the dictionary just loads each line of the dictionary into the hash
395          * table, assuming that most words are not going to be used. Only when a word is actually used is its pronunciations
396          * massaged into an array of pronunciations.
397          */
398         private Word processEntry(String word) {
399             List<Pronunciation> pList = new LinkedList<Pronunciation>();
400             String line;
401             int count = 0;
402             boolean isFiller = false;
403     
404             do {
405                 count++;
406                 String lookupWord = word;
407                 if (count > 1) {
408                     lookupWord = lookupWord + '(' + count + ')';
409                 }
410                 line = (String) dictionary.get(lookupWord);
411                 if (line != null) {
412                     StringTokenizer st = new StringTokenizer(line);
413     
414                     String tag = st.nextToken();
415                     isFiller = tag.startsWith(FILLER_TAG);
416                     int unitCount = st.countTokens();
417     
418                     Unit[] units = new Unit[unitCount];
419                     for (int i = 0; i < units.length; i++) {
420                         String unitName = st.nextToken();
421                         units[i] = getCIUnit(unitName, isFiller);
422                     }
423     
424                     if (!isFiller && addSilEndingPronunciation) {
425                         units = Arrays.copyOf(units, unitCount + 1);
426                         units[unitCount] = UnitManager.SILENCE;
427                     }
428                     pList.add(new Pronunciation(units, null, null, 1.f));
429                 }
430             } while (line != null);
431     
432             Pronunciation[] pronunciations = new Pronunciation[pList.size()];
433             pList.toArray(pronunciations);
434             Word wordObject = createWord(word, pronunciations, isFiller);
435     
436             for (Pronunciation pronunciation : pronunciations) {
437                 pronunciation.setWord(wordObject);
438             }
439             wordDictionary.put(word, wordObject);
440     
441             return wordObject;
442         }

456         /**
457          * Returns a string representation of this FastDictionary in alphabetical order.
458          *
459          * @return a string representation of this FastDictionary
460          */
461         @Override
462         public String toString() {
463     
464             SortedMap<String, String> sorted = new TreeMap<String, String>(dictionary);
465             StringBuilder result = new StringBuilder();
466     
467             for (Map.Entry<String, String> entry : sorted.entrySet()) {
468                 result.append(entry.getKey());
469                 result.append("   ").append(entry.getValue()).append('\n');
470             }
471     
472             return result.toString();
473         }

500         /**
501          * Creates a list of resource URLs given the name of the property. This method should eventually be replaced by a
502          * corresponding version in PropertySheet
503          *
504          * @param propertyListName the name of the propertyList
505          * @param ps               the FastDictionary property sheet
506          * @return the list of resource URLs - empty if no propertyList exists
507          * @throws PropertyException if a URL (from the list) isn't a valid file path
508          */
509         private List<URL> getResourceList(String propertyListName, PropertySheet ps)
510                 throws PropertyException {
511             List<URL> resourceList = new ArrayList<URL>();
512             String pathListString = ps.getString(propertyListName);
513     
514             if (pathListString != null) {
515                 for (String addendumPath : pathListString.split(";")) {
516                     try {
517                         URL addendaUrl = new URL(addendumPath);
518                         resourceList.add(addendaUrl);
519                     } catch (MalformedURLException mue) {
520                         throw new IllegalArgumentException(
521                                 "Addendum path: " + addendumPath + " is not a valid URL.");
522                     }
523                 }
524             }
525             return resourceList;
526         }
