52     		boolean keep = true;
53     		SearchState ss = t.getSearchState();
54     		if (ss instanceof HMMSearchState) {
55     			if (!t.isEmitting()) {
56     				// HMMSearchState hss = (HMMSearchState) ss;
57     				Token lastToken = skewMap.get(ss);
58     				if (lastToken != null) {
59     					int lastFrame = lastToken.getFrameNumber();
60     					if (t.getFrameNumber() - lastFrame > skew
61     							|| t.getScore() > lastToken.getScore()) {
62     						keep = true;
63     					} else {
64     						logger.fine("Dropped " + t + " in favor of " + lastToken);
65     						keep = false;
66     					}
67     
68     				} else {
69     					keep = true;
70     				}
71     
72     				if (keep) {
73     					skewMap.put(ss, t);
74     				}
75     			}
76     		}
77     		return keep;
78     	}
