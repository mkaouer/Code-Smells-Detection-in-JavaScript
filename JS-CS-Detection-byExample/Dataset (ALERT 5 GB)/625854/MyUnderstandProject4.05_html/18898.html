13     package edu.cmu.sphinx.research.parallel;
14     
15     import java.util.*;
16     
17     /**
18      * Combines the various unit feature stream scores in a CombineToken. This SameStampScoreCombiner assumes that there can
19      * be more than one token from each feature stream. It will take the highest scoring token from each stream <b>with the
20      * same identifying stamp</b>, combine their scores, and choose the one with the highest combined score.
21      * <p/>
22      * For example, if tokens T1 and T2 from feature F1 had stamps s1 and s2, and tokens Ta and Tb from feature F2 had
23      * stamps s1 and s2, we must compare combinedScore(s1,Ta)  and combineScore(s2,Tb), and retain the one for which the
24      * combined score is higher.
25      * <p/>
26      * All scores are maintained internally in the LogMath logbase
27      */
28     public class SameStampScoreCombiner implements ScoreCombiner {
29     
30         /**
31          * Remove all tokens in the given token list that has the given stamp.
32          *
33          * @param tokenList the token list
34          * @param stamp     the stamp of tokens to remove
35          * @return a list of the removed tokens
36          */
37         public List<ParallelToken> removeTokensByStamp(List<ParallelToken> tokenList, String stamp) {
38             List<ParallelToken> returnList = new LinkedList<ParallelToken>();
39             for (Iterator<ParallelToken> i = tokenList.iterator(); i.hasNext();) {
40                 ParallelToken token = i.next();
41                 if (token.getLastCombineStamp().equals(stamp)) {
42                     returnList.add(token);
43                     i.remove();
44                 }
45             }
46             return returnList;
47         }
48     
49     
50         /**
51          * Combines the scores from all the feature stream in the given CombineToken. The feature stream(s) are represented
52          * in the form of ParallelTokens in the given CombineToken.
53          *
54          * @param combineToken the CombineToken on which to combine the feature stream scores
55          */
56         public void combineScore(CombineToken combineToken) {
57     
58             // sort all the parallel tokens in the CombineToken
59             // according to their last combine time
60     
61             List<ParallelToken> tokenList = new ArrayList<ParallelToken>(combineToken.getParallelTokens());
62     
63             // System.out.println("TokenList size: " + tokenList.size());
64     
65             double logHighestCombinedScore = Double.NEGATIVE_INFINITY;
66             List<ParallelToken> highestList = null;
67     
68             while (!tokenList.isEmpty()) {
69                 ParallelToken firstToken = tokenList.get(0);
70                 List<ParallelToken> sameStampList = removeTokensByStamp
71                         (tokenList, firstToken.getLastCombineStamp());
72     
73                 double logCombinedScore = getCombinedScore(sameStampList);
74     
75                 // System.out.println("logCombinedScore = " + logCombinedScore);
76                 if (logCombinedScore > logHighestCombinedScore ||
77                         logHighestCombinedScore == Double.NEGATIVE_INFINITY) {
78                     logHighestCombinedScore = logCombinedScore;
79                     highestList = sameStampList;
80                 }
81             }
82     
83             assert highestList != null && highestList.size() > 1;
84             // System.out.println("Highest TokenList size: " + highestList.size());
85     
86             for (ParallelToken token : highestList) {
87                 token.setCombinedScore((float)logHighestCombinedScore);
88             }
89     
90             combineToken.clear();
91             combineToken.addAll(highestList);
92             combineToken.setCombinedScore((float) logHighestCombinedScore);
93         }
94     
95     
96         /**
97          * Calculates the combined score of the given list of tokens. This method assumes that all tokens have the same last
98          * combine time. It will retain only the highest scoring token of each stream and remove all the other lower scoring
99          * tokens from the list.
100          *
101          * @param sameStampTokenList the token list
102          * @return the combined log score
103          */
104         private double getCombinedScore(List<ParallelToken> sameStampTokenList) {
105             Map<String, ParallelToken> uniqueMap = new HashMap<String, ParallelToken>();
106     
107             // first retain the highest scoring token from each stream 
108             for (ParallelToken token : sameStampTokenList) {
109                 String modelName = token.getModelName();
110                 ParallelToken tokenInMap = uniqueMap.get(modelName);
111                 if (tokenInMap == null) {
112                     uniqueMap.put(modelName, token);
113                 } else {
114                     if (token.getFeatureScore() > tokenInMap.getFeatureScore()) {
115                         uniqueMap.put(modelName, token);
116                     }
117                 }
118             }
119     
120             // clear the list
121             sameStampTokenList.clear();
122             assert sameStampTokenList.isEmpty();
123     
124             // now calculate the combinedScore
125             double logTotalScore = 0;
126     
127             for (ParallelToken pToken : uniqueMap.values()) {
128                 // System.out.println("Highest: " + tokenToString(pToken));
129                 sameStampTokenList.add(pToken);
130     
131                 // in linear domain, the following expression is:
132                 // score = pToken.getFeatureScore()^pToken.getEta()
133     
134                 double logScore = pToken.getFeatureScore() * pToken.getEta();
135     
136                 // in linear domain, the following expression is:
137                 // totalScore *= score
138     
139                 logTotalScore += logScore;
140             }
141     
142             return logTotalScore;
143         }
144     }
