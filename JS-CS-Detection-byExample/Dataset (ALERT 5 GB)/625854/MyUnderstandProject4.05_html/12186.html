8     package com.sun.speech.engine.synthesis;

10     import java.awt.Component;
11     import java.util.ArrayList;
12     import java.util.Enumeration;
13     import java.util.List;
14     
15     import javax.speech.Central;
16     import javax.speech.EngineList;
17     import javax.speech.synthesis.SynthesizerModeDesc;
18     import javax.swing.JOptionPane;
19     
20     /**
21      * GUI that displays a list of <code>Synthesizer</code> names in a dialog box.
22      * When the user makes a selection, the selected
23      * <code>SynthesizerModeDesc</code> is returned to the caller.
24      */
25     public class SynthesizerSelector {
26     	/**
27     	 * Asks <code>Central</code> for a list of available synthesizers. If there
28     	 * are none, returns <code>null</code>. If there is only one, returns it.
29     	 * Otherwise, pops up an input dialog that gives the user a choice.
30     	 * 
31     	 * @param component
32     	 *            the component for JOptionPane.showInputDialog
33     	 * @param appName
34     	 *            the title for the input dialog
35     	 * 
36     	 * @return a <code>SynthesizerModeDesc</code> representing the synthesizer
37     	 *         to use.
38     	 */
39     	static public SynthesizerModeDesc getSynthesizerModeDesc(
40     			Component component, String appName) {
41     		List<SynthesizerModeDesc> synths = new ArrayList<SynthesizerModeDesc>();
42     		List<String> synthNames = new ArrayList<String>();
43     		EngineList list = Central.availableSynthesizers(null);
44     		Enumeration<?> e = list.elements();
45     		while (e.hasMoreElements()) {
46     			SynthesizerModeDesc desc = (SynthesizerModeDesc)e.nextElement();
47     			synths.add(desc);
48     			synthNames.add(desc.getEngineName());
49     		}
50     		Object[] synthNamesArray = synthNames.toArray();
51     
52     		if (synths.size() == 0) {
53     			return null;
54     		} else if (synths.size() == 1) {
55     			return (SynthesizerModeDesc) synths.get(0);
56     		}
57     
58     		String synthName = (String) JOptionPane.showInputDialog(component,
59     				"Select the Synthesizer to use:", appName
60     						+ ": Select Synthesizer", JOptionPane.QUESTION_MESSAGE,
61     				null, synthNamesArray, synthNamesArray[0]);
62     
63     		int index = synthNames.indexOf(synthName);
64     		if (index == -1) {
65     			return null;
66     		}
67     		return (SynthesizerModeDesc) synths.get(index);
68     	}
69     }
