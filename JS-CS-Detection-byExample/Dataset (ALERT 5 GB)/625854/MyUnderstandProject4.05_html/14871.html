13     package edu.cmu.sphinx.frontend.util;
14     
15     import javax.sound.sampled.AudioFileFormat;
16     import javax.sound.sampled.AudioFormat;
17     import javax.sound.sampled.AudioInputStream;
18     import javax.sound.sampled.AudioSystem;
19     import java.io.ByteArrayInputStream;
20     import java.io.ByteArrayOutputStream;
21     import java.io.File;
22     import java.io.IOException;
23     
24     
25     /** Represents the complete audio data of an utterance. */
26     public class Utterance {
27     
28         private final String name;
29         private final ByteArrayOutputStream audioBuffer;
30         private final AudioFormat audioFormat;
31     
32     
33         /**
34          * Constructs a default Utterance object.
35          *
36          * @param name   the name of this Utterance, e.g., it can be the name of the audio file where the audio came from
37          * @param format the audio format of this Utterance
38          */
39         public Utterance(String name, AudioFormat format) {
40             this.name = name;
41             this.audioFormat = format;
42             this.audioBuffer = new ByteArrayOutputStream();
43         }
44     
45     
46         /**
47          * Returns the name of this Utterance.
48          *
49          * @return the name of this Utterance
50          */
51         public String getName() {
52             return name;
53         }
54     
55     
56         /**
57          * Returns the audio format of this Utterance.
58          *
59          * @return the audio format
60          */
61         public AudioFormat getAudioFormat() {
62             return audioFormat;
63         }
64     
65     
66         /**
67          * Adds an audio frame into this Utterance.
68          *
69          * @param audio the audio frame to add
70          */
71         public void add(byte[] audio) {
72             synchronized (audioBuffer) {
73                 audioBuffer.write(audio, 0, audio.length);
74             }
75         }
76     
77     
78         /**
79          * Returns the complete audio stream of this utterance.
80          *
81          * @return the complete audio stream
82          */
83         public byte[] getAudio() {
84             return audioBuffer.toByteArray();
85         }
86     
87     
88         /**
89          * Returns the amount of time (in seconds) this Utterance lasts.
90          *
91          * @return how long is this utterance
92          */
93         public float getAudioTime() {
94             return ((float) audioBuffer.size()) /
95                     (audioFormat.getSampleRate() *
96                             audioFormat.getSampleSizeInBits() / 8);
97         }
98     
99     
100         /**
101          * Save the audio as a WAV file in the given file.
102          *
103          * @param fileName   the name of the audio file
104          * @param fileFormat the audio file format
105          * @throws java.io.IOException
106          */
107         public void save(String fileName, AudioFileFormat.Type fileFormat)
108                 throws IOException {
109             File file = new File(fileName);
110             byte[] audio = getAudio();
111             AudioInputStream ais = new AudioInputStream
112                     ((new ByteArrayInputStream(audio)), getAudioFormat(), audio.length);
113             AudioSystem.write(ais, fileFormat, file);
114         }
115     }
