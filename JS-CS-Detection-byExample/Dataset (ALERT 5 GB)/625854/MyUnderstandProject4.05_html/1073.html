30     /** Prunes an ActiveList of ParallelTokens based on their scores. */
31     public abstract class TokenScorePruner implements Pruner {
32     
33         @S4Integer(defaultValue = 2000)
34         public static final String PROP_ABSOLUTE_BEAM_WIDTH = "absoluteBeamWidth";
35     
36         @S4Integer(defaultValue = 0)
37         public static final String PROP_RELATIVE_BEAM_WIDTH = "relativeBeamWidth";
38     
39         /** The property that defines the name of the logmath to be used by this search manager. */
40         @S4Component(type = LogMath.class)
41         public final static String PROP_LOG_MATH = "logMath";
42     
43         /** The property that defines the name of the active list factory to be used by this search manager. */
44         public final static String PROP_ACTIVE_LIST_FACTORY = "activeListFactory";
45     
46         private String name;
47         private int absoluteBeamWidth;
48         private float relativeBeamWidth;
49         private boolean doRelativePruning;
50         private LogMath logMath;
51         private ActiveListFactory activeListFactory;
52     
53         private static Comparator<Token> tokenComparator;
54     
55     
56         /* (non-Javadoc)
57          * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
58          */
59         public void newProperties(PropertySheet ps) throws PropertyException {
60             logMath = (LogMath) ps.getComponent(PROP_LOG_MATH);
61             setAbsoluteBeamWidth(ps.getInt(PROP_ABSOLUTE_BEAM_WIDTH
62             ));
63             double linearRelativeBeamWidth = ps.getDouble(PROP_RELATIVE_BEAM_WIDTH);
64             setRelativeBeamWidth(logMath.linearToLog(linearRelativeBeamWidth));
65             activeListFactory = (ActiveListFactory)ps.getComponent(PROP_ACTIVE_LIST_FACTORY);
66         }
67     
68     
69         /**
70          * Returns the score that we use to compare this Token with other Tokens.
71          *
72          * @param token the Token to compare
73          */
74         protected abstract float getTokenScore(Token token);
75     
76     
77         /**
78          * Returns a token comparator that is used to order ParallelTokens in descending order of their combined score.
79          *
80          * @return a ParallelTokenComparator
81          */
82         protected Comparator<Token> getTokenComparator() {
83             if (tokenComparator == null) {
84                 tokenComparator = new Comparator<Token>() {
85                     public int compare(Token t1, Token t2) {
86                         if (getTokenScore(t1) > getTokenScore(t2)) {
87                             return -1;
88                         } else if (getTokenScore(t1) == getTokenScore(t2)) {
89                             return 0;
90                         } else {
91                             return 1;
92                         }
93                     }
94                 };
95             }
96             return tokenComparator;
97         }
98     
99     
100         /**
101          * Prunes the set of tokens in the given ActiveList. It is assumed that all token in the ActiveList are
102          * ParallelTokens. Pruning is performed according to the absolute and relative beam widths of the given ActiveList.
103          *
104          * @param activeList a activeList of tokens
105          */
106         public ActiveList prune(ActiveList activeList) {
107     
108             ActiveList newList = activeListFactory.newInstance();
109     
110             List<Token> tokenList = activeList.getTokens();
111             Collections.sort(tokenList, getTokenComparator());
112     
113             if (!tokenList.isEmpty()) {
114                 Token bestToken = tokenList.get(0);
115                 float highestScore = getTokenScore(bestToken);
116                 float pruneScore = highestScore + relativeBeamWidth;
117                 float thisScore = highestScore;
118     
119                 Iterator<Token> i = tokenList.iterator();
120     
121                 // do the pruning
122                 while (i.hasNext() && newList.size() < absoluteBeamWidth) {
123     
124                     Token token = i.next();
125                     thisScore = getTokenScore(token);
126     
127                     if (doRelativePruning) {
128                         if (thisScore > pruneScore) {
129                             newList.add(token);
130                         } else {
131                             break;
132                         }
133                     } else {
134                         newList.add(token);
135                     }
136                 }
137             }
138     
139             return newList;
140         }

143         /**
144          * Returns the absolute beam width of this TokenScorePruner.
145          *
146          * @return the absolute beam width (in the linear domain)
147          */
148         public int getAbsoluteBeamWidth() {
149             return this.absoluteBeamWidth;
150         }
151     
152     
153         /**
154          * Sets the absolute beam width of this TokenScorePruner
155          *
156          * @param absoluteBeamWidth the absolute beam width
157          */
158         public void setAbsoluteBeamWidth(int absoluteBeamWidth) {
159             this.absoluteBeamWidth = absoluteBeamWidth;
160         }
161     
162     
163         /**
164          * Returns the relative beam width (in the linear domain) of this TokenScorePruner
165          *
166          * @return the relative beam width (in the linear domain)
167          */
168         public float getRelativeBeamWidth() {
169             return relativeBeamWidth;
170         }
171     
172     
173         /**
174          * Sets the relative beam width (log domain) of this TokenScorePruner.
175          *
176          * @param relativeBeamWidth the relative beam width in log domain
177          */
178         public void setRelativeBeamWidth(float relativeBeamWidth) {
179             this.relativeBeamWidth = relativeBeamWidth;
180         }

184         public void startRecognition() {
185         }

187

189         public void stopRecognition() {
190         }

194         public void allocate() {
195         }
196     
197     
198         /** Deallocates resources necessary for this pruner */
199         public void deallocate() {
200         }

202     
203         /*
204         * (non-Javadoc)
205         *
206         * @see edu.cmu.sphinx.util.props.Configurable#getName()
207         */
208         public String getName() {
209             return name;
210         }
211     }
