14     package edu.cmu.sphinx.frontend.frequencywarp;
15     
16     
17     /**
18      * Defines a filter used by the {@link PLPFrequencyFilterBank} class. The filter is defined by a function in the {@link
19      * #PLPFilter Constructor}. A set of equally spaced frequencies in a linear scale is passed to the constructor, which
20      * returns the weights for each of the frequency bins, such that the filter has the shape defined by this piecewise
21      * function in the bark scale.
22      *
23      * @author <a href="mailto:rsingh@cs.cmu.edu">rsingh</a>
24      * @version 1.0
25      * @see PLPFrequencyFilterBank
26      */
27     public class PLPFilter {
28     
29         private double[] filterCoefficients;
30         private final int numDFTPoints;
31     
32         /** The center frequency of the filter in Hertz. */
33         public final double centerFreqInHz;
34     
35         /** The center frequency of the filter in Bark. */
36         public final double centerFreqInBark;
37     
38     
39         /**
40          * Constructs a PLP filter around a given center frequency.
41          * <p/>
42          * Defines a filter according to the following equation, defined piecewise (all frequencies in the equation are Bark
43          * frequencies):
44          * <p/>
45          * <pre>
46          * Filter(f) = 0 if f < -2.5 <br>
47          *           = 10^(-(f+0.5)) if -2.5 <= f <= -0.5 <br>
48          *           = 1  if -0.5 <= f <= 0.5 <br>
49          *           = 10^(2.5(f-0.5)) if 0.5 <= f <= 1.3 <br>
50          *           = 0 if f > 1.3 <br>
51          * </pre>
52          * The current implementation assumes that the calling routine passes in an array of frequencies, one for each of
53          * the DFT points in the spectrum of the frame of speech to be filtered. This is used in conjunction with a
54          * specified center frequency to determine the filter.
55          *
56          * @param DFTFrequenciesInHz is a double array containing the frequencies in Hertz corresponding to each of the DFT
57          *                           points in the spectrum of the signal to be filtered.
58          * @param centerFreqInHz     is the filter's center frequency
59          * @throws IllegalArgumentException
60          */
61     
62     
63         public PLPFilter(double[] DFTFrequenciesInHz,
64                          double centerFreqInHz) throws IllegalArgumentException {
65     
66             FrequencyWarper bark = new FrequencyWarper();
67     
68             numDFTPoints = DFTFrequenciesInHz.length;
69             this.centerFreqInHz = centerFreqInHz;
70             centerFreqInBark = bark.hertzToBark(centerFreqInHz);
71     
72             if (centerFreqInHz < DFTFrequenciesInHz[0] ||
73                     centerFreqInHz > DFTFrequenciesInHz[numDFTPoints - 1]) {
74                 throw new IllegalArgumentException
75                         ("Center frequency for PLP filter out of range");
76             }
77     
78             filterCoefficients = new double[numDFTPoints];
79     
80             for (int i = 0; i < numDFTPoints; i++) {
81                 double barkf;
82     
83                 barkf = bark.hertzToBark(DFTFrequenciesInHz[i]) - centerFreqInBark;
84                 if (barkf < -2.5)
85                     filterCoefficients[i] = 0.0;
86                 else if (barkf <= -0.5)
87                     filterCoefficients[i] = Math.pow(10.0, barkf + 0.5);
88                 else if (barkf <= 0.5)
89                     filterCoefficients[i] = 1.0;
90                 else if (barkf <= 1.3)
91                     filterCoefficients[i] = Math.pow(10.0, -2.5 * (barkf - 0.5));
92                 else
93                     filterCoefficients[i] = 0.0;
94             }
95         }
96     
97     
98         /**
99          * Compute the PLP spectrum at the center frequency of this filter for a given power spectrum.
100          *
101          * @param spectrum the input power spectrum to be filtered
102          * @return the PLP spectrum value
103          * @throws IllegalArgumentException if the input spectrum is of a different length than the array of filter
104          *                                  coefficients
105          */
106         public double filterOutput(double[] spectrum)
107                 throws IllegalArgumentException {
108     
109             if (spectrum.length != numDFTPoints) {
110                 throw new IllegalArgumentException
111                         ("Mismatch in no. of DFT points " + spectrum.length +
112                                 " in spectrum and in filter " + numDFTPoints);
113             }
114     
115             double output = 0.0;
116             for (int i = 0; i < numDFTPoints; i++) {
117                 output += spectrum[i] * filterCoefficients[i];
118             }
119             return output;
120         }
121     }
