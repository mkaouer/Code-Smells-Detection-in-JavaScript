15     package edu.cmu.sphinx.tools.gui;
16     
17     import edu.cmu.sphinx.tools.gui.reader.XMLConfigReader;
18     import edu.cmu.sphinx.tools.gui.reader.GUIReaderException;
19     import edu.cmu.sphinx.tools.gui.writer.GUIWriterException;
20     import edu.cmu.sphinx.tools.gui.writer.XMLConfigWriter;
21     import edu.cmu.sphinx.tools.gui.util.ModelBuilder;
22     import edu.cmu.sphinx.tools.gui.util.ConfigurableUtilException;
23     
24     import java.util.List;
25     import java.util.ArrayList;
26     import java.io.File;
27     
28     
29     /**
30      * This is one of the important classes that coordinates between GUI and model
31      * Its main operations are : 
32      * 1. to start and initialize the GUI and Sphinx model
33      * 2. drive the input - output operation, and 
34      * 3. to retrieve the most updated data from GUI or to update GUI of the new loaded data
35      *
36      * @author Ariani
37      */
38     public class GUIMediator {
39         
40         public static final String OPEN = "open";
41         public static final String EXIT = "exit";
42         public static final String NEW = "new";
43         public static final String SAVE = "save";
44         public static final String REFRESH = "refresh_model";
45         public static final String SHOW_CONFIG = "show_config";
46             
47         private MainJFrame _mainJF;
48         private final List<GUIFileActionListener> _panelList;
49         private final XMLConfigReader _xmlReader;
50         private final XMLConfigWriter _xmlWriter;
51         private ModelBuilder _mb;
52     
53         
54         /** 
55          * Creates a new instance of GUIMediator 
56          */
57         public GUIMediator() throws ConfigurableUtilException{
58             _panelList = new ArrayList<GUIFileActionListener>(); //for the GUI panels
59             
60             // load reader and writer
61             _xmlReader = XMLConfigReader.getInstance();
62             _xmlWriter = XMLConfigWriter.getInstance();
63                     
64             // NOTE: the model should be created before the GUI components,
65             // so that GUI TabPenel can be created based on model        
66             loadModel();
67             
68             // create main GUI frame        
69             _mainJF = new MainJFrame(this,_mb.getGroups());
70             
71         }
72         
73         // part of the constuctor operation - initialize and load Sphinx model
74         private void loadModel() throws ConfigurableUtilException{
75             _mb = ModelBuilder.getInstance();            
76             
77             // NOTE: the model should be created before the GUI components,
78             // so that GUI TabPenel can be created based on model
79             _mb.refresh();       
80             // _mb.printModel(); // for debugging
81         }
82         
83         /**
84          * A public function to obtain reference to the ModelBuilder, 
85          * that holds the complete set of classes and groups in the model
86          * 
87          * @return ModelBuilder
88          **/
89         public ModelBuilder getModelBuilder(){
90             return _mb;
91         }
92         
93         /**
94          * Used by GUI panel to register as one of the notified clients
95          *
96          * @param c the registering class must implement <code>GUIFileActionListener</code>
97          */
98         public void registerPanel(GUIFileActionListener c){
99             _panelList.add(c);
100             
101         }
102      
103         /**
104          * Start the GUI -  after load operations is completed successfully
105          */
106         public void execute(){
107             _mainJF.setSize(900,700);
108             _mainJF.setLocationRelativeTo(null);
109             _mainJF.setVisible(true);
110             
111             
112         }
113         
114         // read the file and update all GUI panels
115         private void updateList (File fFile) throws GUIReaderException, GUIWriterException {
116             ConfigProperties cp = _xmlReader.read(fFile);
117             _mb.update(cp);
118             if (_panelList != null) {
119                 for (GUIFileActionListener listener : _panelList) {
120                     listener.update(cp);
121                 }
122              }
123         }
124             
125         
126         private void saveToFile(File fFile) throws GUIReaderException, GUIWriterException {
127             ConfigProperties cp = new ConfigProperties(); 
128             if (readList(cp))
129                 _xmlWriter.writeOutput(cp,fFile);
130         }
131         
132         /** read the GUI entries and save changes to ConfigProperties */
133         private boolean readList ( ConfigProperties cp) throws GUIReaderException, GUIWriterException {
134             try {
135                 _mb.saveData(cp);
136                 
137                 if (_panelList != null) {
138                     for (GUIFileActionListener listener : _panelList) {
139                         listener.saveData(cp);
140                     }                
141                    return true;
142                 }
143                 return true;
144             } catch (GUIOperationException oe) {
145                 return false;
146             }
147         }
148         
149         /** inform GUI panels to clear all configuration data
150          */
151         private void clearAll() {
152             _mb.clearAll();
153             if (_panelList != null) {
154                 for (GUIFileActionListener listener : _panelList) {
155                     listener.clearAll();
156                 }
157             }        
158         }
159         
160         /**refresh the Sphinx model in Model Builder and GUI Panels
161          */
162         private void refreshModel()throws ConfigurableUtilException{
163             _mb.modelRefresh();
164             _mb.printModel();
165             _mainJF.addTextPanels(_mb.getGroups());
166         }
167         
168         /**
169          * all action will call this method, with its specific command
170          *
171          * @param command Action to be performed
172          * @param fFile   File to open/save to
173          * @throws GUIReaderException, GUIWriterException
174          */
175         public void action(String command,File fFile) throws GUIReaderException, GUIWriterException {  
176             if (command.equalsIgnoreCase(OPEN))
177                 updateList(fFile);
178             else if (command.equalsIgnoreCase(SAVE))
179                 saveToFile(fFile);
180         }
181         
182         /**
183          * action that needs String return, with its specific command
184          *
185          * @param command Action to be performed
186          * @param outputJTextArea text area to display the output
187          * @throws GUIReaderException, GUIWriterException
188          */
189         public void action(String command, javax.swing.JTextArea outputJTextArea) 
190                 throws GUIWriterException, GUIReaderException {
191             ConfigProperties cp = new ConfigProperties();
192             if (readList(cp)) {
193                 String output = _xmlWriter.getOutput(cp);
194                 outputJTextArea.setText(output);
195             }
196         }
197         
198         /** 
199          * action command that does not involve <code>File</code> operation
200          *
201          * @throws ConfigurableUtilException when there's error while reloading model
202          */
203         public void action(String command) throws ConfigurableUtilException {
204             if (command.equalsIgnoreCase(NEW))
205                 clearAll();        
206             else if (command.equalsIgnoreCase(REFRESH))
207                 refreshModel();
208         }
209     }
