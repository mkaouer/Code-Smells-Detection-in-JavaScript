60         /** Dumps the search space hmm in GDL format */
61         @Override
62         public void run() {
63             try {
64                 FileOutputStream fos = new FileOutputStream(filename);
65                 PrintStream out = new PrintStream(fos);
66                 SearchState firstState =
67                         getLinguist().getSearchGraph().getInitialState();
68                 dumpSearchGraph(out, firstState);
69                 out.close();
70             } catch (FileNotFoundException fnfe) {
71                 System.out.println("Can't dump to file " + filename + ' ' + fnfe);
72             }
73         }

138         /**
139          * Dumps the search graph
140          *
141          * @param out           place to dump the output
142          * @param startingState the initial state of the search space
143          */
144         private void dumpSearchGraph(PrintStream out, SearchState startingState) {
145             List<StateLevel> queue = new LinkedList<StateLevel>();
146             Set<String> visitedStates = new HashSet<String>();
147             startDump(out);
148             queue.add(new StateLevel(startingState, 0));
149             while (!queue.isEmpty()) {
150                 StateLevel stateLevel = queue.remove(0);
151                 int level = stateLevel.getLevel();
152                 SearchState state = stateLevel.getState();
153                 // equalCheck(state);
154                 if (!visitedStates.contains(state.getSignature())) {
155                     visitedStates.add(state.getSignature());
156                     startDumpNode(out, state, level);
157                     SearchStateArc[] arcs = state.getSuccessors();
158                     for (int i = arcs.length - 1; i >= 0; i--) {
159                         SearchState nextState = arcs[i].getState();
160                         dumpArc(out, state, arcs[i], level);
161                         if (depthFirst) {
162                             // if depth first, its a stack
163                             queue.add(0, new StateLevel(nextState, level + 1));
164                         } else {
165                             queue.add(new StateLevel(nextState, level + 1));
166                         }
167                     }
168                     endDumpNode(out, state, level);
169                 }
170             }
171             endDump(out);
172         }

175         final Cache<SearchState> eqStates = new Cache<SearchState>();
176         final Map<String, SearchState> eqSigs = new HashMap<String, SearchState>();
177     
178     
179         /**
180          * This is a bit of test/debugging code that ensures that the states that have equal signatures are also considered
181          * to be object.equals and vice versa.. This method will dump out any states where this contract is not true
182          *
183          * @param state the state to check
184          */
185         @SuppressWarnings("unused")
186         private void equalCheck(SearchState state) {
187             SearchState eqState = eqStates.cache(state);
188             SearchState eqSig = eqSigs.get(state.getSignature());
189             if (eqState == null ^ eqSig == null) {
190                 System.out.println("Missing one: ");
191                 System.out.println("  state val: " + state);
192                 System.out.println("  state sig: " + state.getSignature());
193                 System.out.println("  eqState val: " + eqState);
194                 System.out.println("  eqSig val: " + eqSig);
195                 if (eqState != null) {
196                     System.out.println("   eqState sig: " + eqState.getSignature());
197                 }
198                 if (eqSig != null) {
199                     System.out.println("   eqSig sig: " + eqSig.getSignature());
200                 }
201             }
202             if (eqState == null) {
203                 eqState = state;
204             }
205             if (eqSig == null) {
206                 eqSigs.put(state.getSignature(), state);
207                 eqSig = state;
208             }
209             if (!eqState.getSignature().equals(state.getSignature())) {
210                 System.out.println("Sigs mismatch for: ");
211                 System.out.println("  state sig: " + state.getSignature());
212                 System.out.println("  eqSig sig: " + eqSig.getSignature());
213                 System.out.println("  state val: " + state);
214                 System.out.println("  eqSig val: " + eqSig);
215             }
216             if (!eqState.equals(state)) {
217                 System.out.println("obj mismatch for: ");
218                 System.out.println("  state sig: " + state.getSignature());
219                 System.out.println("  eqSig sig: " + eqSig.getSignature());
220                 System.out.println("  state val: " + state);
221                 System.out.println("  eqSig val: " + eqSig);
222             }
223         }

226     /** A class for bundling together a SearchState and its level. */
227     
228     class StateLevel {
229     
230         private final int level;
231         private final SearchState state;
232     
233     
234         /**
235          * Constructs a StateLevel from its primitive components.
236          *
237          * @param state the state to be bundled in the StateLevel
238          * @param level the level of the state
239          */
240         StateLevel(SearchState state, int level) {
241             this.state = state;
242             this.level = level;
243         }
244     
245     
246         /**
247          * Returns the state
248          *
249          * @return the state
250          */
251         SearchState getState() {
252             return state;
253         }
254     
255     
256         /**
257          * Returns the level
258          *
259          * @return the level.
260          */
261         int getLevel() {
262             return level;
263         }
264     
265     
266         /**
267          * Returns a string representation of the object
268          *
269          * @return a string representation
270          */
271         @Override
272         public String toString() {
273             return String.valueOf(level) + ' ' + state.getSignature() + " 1 "
274                     + getTypeLabel(state);
275         }
276     
277     
278         /**
279          * Retrieves a type label for a state
280          *
281          * @param state the state of interest
282          * @return a label for the type of state (one of Unit, Word, HMM or other
283          */
284         public String getTypeLabel(SearchState state) {
285             if (state instanceof UnitSearchState) {
286                 return "Unit";
287             }
288             if (state instanceof WordSearchState) {
289                 return "Word";
290             }
291             if (state instanceof HMMSearchState) {
292                 return "HMM";
293             }
294             return "other";
295         }
296     }
