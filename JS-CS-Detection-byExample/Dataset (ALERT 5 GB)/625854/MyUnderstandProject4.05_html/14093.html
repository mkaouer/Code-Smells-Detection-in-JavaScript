13     package edu.cmu.sphinx.linguist.acoustic;
14     
15     /**
16      * Defines possible positions of HMMs. Note that even though the positions are defined to be within words, some
17      * recognizers may classify positions in terms of other elements besides words.
18      */
19     public enum HMMPosition {
20     
21         BEGIN     ('b'), // HMM is at the beginning position of the word
22         END       ('e'), // HMM is at the end position of the word
23         SINGLE    ('s'), // HMM is at the beginning and the end of the word
24         INTERNAL  ('i'), // HMM is completely internal to the word
25         UNDEFINED ('-'); // HMM is at an undefined position in the word
26     
27         private static final HMMPosition[] posByRep;
28         static {
29             int maxChar = 0;
30             for (HMMPosition pos : values()) // determine max char to use as index
31                 if (pos.rep.charAt(0) > maxChar)
32                     maxChar = pos.rep.charAt(0);
33             posByRep = new HMMPosition[maxChar + 1];
34             for (HMMPosition pos : values()) // cache HMMPositions according to rep
35                 posByRep[pos.rep.charAt(0)] = pos;
36         }
37     
38         private final String rep;
39     
40         /**
41          * Looks up an HMMPosition based upon its representation
42          *
43          * @param rep the string representation
44          * @return the HMMPosition represented by rep or null if not found
45          */
46         private HMMPosition(char rep) {
47             this.rep = String.valueOf(rep);
48         }
49     
50         /**
51          * Looks up an HMMPosition based upon its representation
52          *
53          * @param rep the string representation
54          * @return the HMMPosition represented by rep or null if not found
55          */
56         public static HMMPosition lookup(String rep) {
57             return rep == null || rep.isEmpty() ? null : posByRep[rep.charAt(0)];
58         }
59     
60         /**
61          * Determines if this position is an end word position
62          *
63          * @return true if this is an end of word position
64          */
65         public boolean isWordEnd() {
66             return this == SINGLE || this == END;
67         }
68     
69         /**
70          * Determines if this position is word beginning position
71          *
72          * @return true if this is a word beginning position
73          */
74         public boolean isWordBeginning() {
75             return this == SINGLE || this == BEGIN;
76         }
77     
78         /**
79          * Returns a string representation of this object
80          *
81          * @return the string representation
82          */
83         @Override
84         public String toString() {
85             return rep;
86         }
87     }
