20     /**
21      * Represents a composite senone. A composite senone consists of a set of all possible {@link Senone senones} for a
22      * given state. CompositeSenones are used when the exact context of a senone is not known. The CompositeSenone
23      * represents all the possible senones.
24      * <p/>
25      * This class currently only needs to be public for testing purposes.
26      * <p/>
27      * Note that all scores are maintained in LogMath log base
28      */
29     
30     public class CompositeSenone extends ScoreCachingSenone {
31     
32         private final static int MAX_SENONES = 20000;
33         private final static boolean wantMaxScore = true;
34         private final Senone[] senones;
35         private final float weight;
36     
37         /**
38          * a factory method that creates a CompositeSenone from a list of senones.
39          *
40          * @param senoneCollection the Collection of senones
41          * @return a composite senone
42          */
43         public static CompositeSenone create(Collection<Senone> senoneCollection,
44                                              float weight) {
45             return new CompositeSenone(senoneCollection.toArray(new Senone[senoneCollection.size()]), weight);
46         }
47     
48     
49         /**
50          * Constructs a CompositeSenone given the set of constiuent senones
51          *
52          * @param senones the set of constiuent senones
53          */
54         public CompositeSenone(Senone[] senones, float weight) {
55             this.senones = senones;
56             this.weight = weight;
57             System.out.print(" " + senones.length);
58         }
59     
60     
61         /**
62          * Dumps this senone
63          *
64          * @param msg annotatin for the dump
65          */
66         @Override
67         public void dump(String msg) {
68             System.out.println("   CompositeSenone " + msg + ": ");
69             for (Senone senone : senones) {
70                 senone.dump("   ");
71             }
72         }
73     
74     
75         /**
76          * Calculates the composite senone score. Typically this is the best score for all of the constituent senones
77          */
78         @Override
79         public float calculateScore(Data feature) {
80             float logScore;
81             if (wantMaxScore) {
82                 logScore = -Float.MAX_VALUE;
83                 for (Senone senone : senones) {
84                     logScore = Math.max(logScore, senone.getScore(feature));
85                 }
86             } else { // average score
87                 logScore = 0.0f;
88                 for (Senone senone : senones) {
89                     logScore += senone.getScore(feature);
90                 }
91                 logScore = logScore / senones.length;
92             }
93             return logScore + weight;
94         }

116             return senones;
117         }

120         /**
121          * Determines if two objects are equal
122          *
123          * @param o the object to compare to this.
124          * @return true if the objects are equal
125          */
126         @Override
127         public boolean equals(Object o) {
128             if (!(o instanceof Senone)) {
129                 return false;
130             }
131             Senone other = (Senone) o;
132             return this.getID() == other.getID();
133         }

135     
136         /**
137          * Returns the hashcode for this object
138          *
139          * @return the hashcode
140          */
141         @Override
142         public int hashCode() {
143             long id = getID();
144             int high = (int) ((id >> 32));
145             int low = (int) (id);
146             return high + low;
147         }
148     
149     
150         /**
151          * Gets the ID for this senone
152          *
153          * @return the senone id
154          */
155         @Override
156         public long getID() {
157             long factor = 1L;
158             long id = 0L;
159             for (Senone senone : senones) {
160                 id += senone.getID() * factor;
161                 factor = factor * MAX_SENONES;
162             }
163             return id;
164         }
165

168          * Retrieves a string form of this object
169          *
170          * @return the string representation of this object
171          */
172         @Override
173         public String toString() {
174             return "senone id: " + getID();
175         }
