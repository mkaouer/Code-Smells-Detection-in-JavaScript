12     package edu.cmu.sphinx.frontend.util;
13     
14     import edu.cmu.sphinx.frontend.*;
15     import edu.cmu.sphinx.frontend.endpoint.SpeechClassifiedData;
16     import edu.cmu.sphinx.util.props.*;
17     
18     import java.text.DecimalFormat;
19     import java.text.DecimalFormatSymbols;
20     import java.util.Locale;
21     
22     
23     /** Dumps the data */
24     public class DataDumper extends BaseDataProcessor {
25     
26         /** The property that specifies whether data dumping is enabled */
27         @S4Boolean(defaultValue = true)
28         public final static String PROP_ENABLE = "enable";
29     
30         /** The property that specifies the format of the output. */
31         @S4String(defaultValue = "0.00000E00;-0.00000E00")
32         public final static String PROP_OUTPUT_FORMAT = "outputFormat";
33     
34         /** The property that enables the output of signals. */
35         @S4Boolean(defaultValue = true)
36         public final static String PROP_OUTPUT_SIGNALS = "outputSignals";
37     
38         // --------------------------
39         // Configuration data
40         // --------------------------
41         private int frameCount;
42         private boolean enable;
43         private boolean outputSignals;
44         private DecimalFormat formatter;
45     
46         public DataDumper( boolean enable, String format, boolean outputSignals ) {
47             initLogger();
48             this.formatter = new DecimalFormat(format, new DecimalFormatSymbols(Locale.US));
49             this.outputSignals = outputSignals;
50             this.enable = enable;
51         }
52     
53         public DataDumper() {
54     
55         }
56         
57         /*
58         * (non-Javadoc)
59         *
60         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
61         */
62         @Override
63         public void newProperties(PropertySheet ps) throws PropertyException {
64             super.newProperties(ps);
65             
66             logger = ps.getLogger();
67     
68             enable = ps.getBoolean(PROP_ENABLE);
69             String format = ps.getString(PROP_OUTPUT_FORMAT);
70             formatter = new DecimalFormat(format, new DecimalFormatSymbols(Locale.US));
71             outputSignals = ps.getBoolean(PROP_OUTPUT_SIGNALS);
72         }
73     
74     
75         /** Constructs a DataDumper */
76         @Override
77         public void initialize() {
78             super.initialize();
79         }
80     
81     
82         /**
83          * Reads and returns the next Data object from this DataProcessor, return null if there is no more audio data.
84          *
85          * @return the next Data or <code>null</code> if none is available
86          * @throws DataProcessingException if there is a data processing error
87          */
88         @Override
89         public Data getData() throws DataProcessingException {
90             Data input = getPredecessor().getData();
91             if (enable) {
92                 dumpData(input);
93             }
94             return input;
95         }
96     
97     
98         /**
99          * Dumps the given input data
100          *
101          * @param input the data to dump
102          */
103         private void dumpData(Data input) {
104             logger.finer("dumping data...");
105             
106             if (input instanceof Signal) {
107                 if (outputSignals) {
108                     System.out.println("Signal: " + input);
109                     if (input instanceof DataStartSignal) {
110                         frameCount = 0;
111                     }
112                 }
113             } else if (input instanceof DoubleData) {
114                 DoubleData dd = (DoubleData) input;
115                 double[] values = dd.getValues();
116                 System.out.print("Frame " + values.length);
117                 for (double val : values) {
118                     System.out.print(' ' + formatter.format(val));
119                 }
120                 System.out.println();
121             } else if (input instanceof SpeechClassifiedData) {
122                 SpeechClassifiedData dd = (SpeechClassifiedData) input;
123                 double[] values = dd.getValues();
124                 System.out.print("Frame ");
125                 if (dd.isSpeech())
126                 	System.out.print('*');
127                 else
128                 	System.out.print(' ');
129                 System.out.print(" " + values.length);
130                 for (double val : values) {
131                     System.out.print(' ' + formatter.format(val));
132                 }
133                 System.out.println();
134             } else if (input instanceof FloatData) {
135                 FloatData fd = (FloatData) input;
136                 float[] values = fd.getValues();
137                 System.out.print("Frame " + values.length);
138                 for (float val : values) {
139                     System.out.print(' ' + formatter.format(val));
140                 }
141                 System.out.println();
142             }
143             frameCount++;
144         }
145     }
