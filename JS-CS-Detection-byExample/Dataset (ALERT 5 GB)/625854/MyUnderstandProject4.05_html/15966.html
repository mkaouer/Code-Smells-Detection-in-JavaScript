13     package edu.cmu.sphinx.tools.audio;
14     
15     import javax.sound.sampled.*;
16     import java.io.ByteArrayInputStream;
17     import java.io.ByteArrayOutputStream;
18     import java.io.IOException;
19     
20     /** Grabs audio from the microphone and returns an array of samples. */
21     public class RawRecorder {
22     
23         final Object lock = new Object();
24         RecordThread recorder;
25         AudioFormat inFormat;
26         final AudioFormat outFormat;
27         TargetDataLine microphone;
28         boolean downsample;
29     
30     
31         /**
32          * Create a new RawRecorder.
33          *
34          * @param audioFormat the desired output
35          * @throws LineUnavailableException if the audioFormat is not supported
36          */
37         public RawRecorder(AudioFormat audioFormat)
38                 throws LineUnavailableException {
39     
40             inFormat = audioFormat;
41             outFormat = audioFormat;
42     
43             /* Some machines, such as my Mac OS X PowerBook, don't support
44              * a wide range of input formats.  So...we may need to read
45              * data in using a different format and then resample to the
46              * desired format.  Here, I'm just going to go for 44.1kHz
47              * 16-bit signed big endian data if the given audio format
48              * is not supported.
49              */
50             DataLine.Info info = new DataLine.Info(TargetDataLine.class,
51                     inFormat);
52     
53             if (!AudioSystem.isLineSupported(info)) {
54                 downsample = true;
55                 inFormat = new AudioFormat(44100.0f, // sample rate
56                         16,       // sample size
57                         1,        // channels (1 == mono)
58                         true,     // signed
59                         true);    // big endian
60                 info = new DataLine.Info(TargetDataLine.class,
61                         inFormat);
62                 if (!AudioSystem.isLineSupported(info)) {
63                     throw new LineUnavailableException(
64                             "Unsupported format: " + audioFormat);
65                 }
66             }
67     
68             microphone = (TargetDataLine) AudioSystem.getLine(info);
69             microphone.open(audioFormat, microphone.getBufferSize());
70         }
71     
72     
73         /**
74          * Start recording.  The stop method will give us the clip.
75          *
76          * @see #stop
77          */
78         public void start() {
79             synchronized (lock) {
80                 if (recorder != null) {
81                     recorder.stopRecording();
82                 }
83                 recorder = new RecordThread();
84                 recorder.start();
85             }
86         }
87     
88     
89         /**
90          * Stop recording and give us the clip.
91          *
92          * @return the clip that was recorded since the last time start was called
93          * @see #start
94          */
95         public short[] stop() {
96             synchronized (lock) {
97                 if (recorder == null) {
98                     return new short[0];
99                 }
100                 ByteArrayOutputStream out = recorder.stopRecording();
101                 microphone.close();
102                 recorder = null;
103                 byte audioBytes[] = out.toByteArray();
104                 ByteArrayInputStream in = new ByteArrayInputStream(audioBytes);
105                 try {
106                     short[] samples = RawReader.readAudioData(in, inFormat);
107                     if (downsample) {
108                         samples = Downsampler.downsample(
109                                 samples,
110                                 (int) (inFormat.getSampleRate() / 1000.0f),
111                                 (int) (outFormat.getSampleRate() / 1000.0f));
112                     }
113                     return samples;
114                 } catch (IOException e) {
115                     e.printStackTrace();
116                     return new short[0];
117                 }
118             }
119         }
120     
121     
122         class RecordThread extends Thread {
123     
124             boolean done;
125             final Object lock = new Object();
126             ByteArrayOutputStream out;
127     
128     
129             public ByteArrayOutputStream stopRecording() {
130                 try {
131                     synchronized (lock) {
132                         done = true;
133                         lock.wait();
134                     }
135                 } catch (InterruptedException e) {
136                 }
137                 return out;
138             }
139     
140     
141             @Override
142             public void run() {
143                 byte[] data = new byte[microphone.getBufferSize()];
144                 out = new ByteArrayOutputStream();
145     
146                 try {
147                     microphone.flush();
148                     microphone.start();
149                     while (!done) {
150                         int numBytesRead = microphone.read(data, 0, data.length);
151                         if (numBytesRead != -1) {
152                             out.write(data, 0, numBytesRead);
153                         } else {
154                             break;
155                         }
156                     }
157                     microphone.stop();
158                     out.flush();
159                 } catch (IOException e) {
160                     e.printStackTrace();
161                 }
162                 synchronized (lock) {
163                     lock.notify();
164                 }
165             }
166         }
167     }
