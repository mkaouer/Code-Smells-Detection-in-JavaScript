20     import java.util.ArrayList;
21     import java.util.Collections;
22     import java.util.List;
23     import java.util.Random;
24     
25     /** Gets successors from a linguist and times them */
26     public class LinguistTimer {
27     
28         private final Linguist linguist;
29         private final boolean details;
30         int totalStates;
31         int totalEmittingStates;
32         int totalNonEmittingStates;
33         int totalFinalStates;
34         int maxSuccessors;
35     
36     
37         /**
38          * Creats a LinguistTimer
39          *
40          * @param linguist the linguist to time
41          * @param details  if true print out details
42          */
43         public LinguistTimer(Linguist linguist, boolean details) {
44             this.linguist = linguist;
45             this.details = details;
46         }
47         /**
48          * tests the linguist
49          */
50         /**
51          * Times the lingust
52          *
53          * @param numRuns   the number of simulated runs
54          * @param numFrames the number of simulated frames
55          * @param maxBeam   the size of the beam
56          */
57         public void timeLinguist(int numRuns, int numFrames, int maxBeam) {
58             // this test invokes the linguist using access patterns that
59             // are similar to a real search. It allows for timing and
60             // profiling of the linguist, independent of the search
61             // or scoring
62             Random random = new Random(1000);
63             Timer frameTimer = TimerPool.getTimer(this, "frameTimer");
64             Timer totalTimer = TimerPool.getTimer(this, "totalTimer");
65             // Note: this comparator imposes orderings that are
66             // inconsistent with equals.
67             System.out.println("TestLinguist: runs " + numRuns + " frames "
68                     + numFrames + " beam " + maxBeam);
69             totalTimer.start();
70             for (int runs = 0; runs < numRuns; runs++) {
71                 int level = 0;
72                 List<SearchState> activeList = new ArrayList<SearchState>();
73                 activeList.add(linguist.getSearchGraph().getInitialState());
74                 linguist.startRecognition();
75                 for (int i = 0; i < numFrames; i++) {
76                     List<SearchState> oldList = activeList;
77                     activeList = new ArrayList<SearchState>(maxBeam * 10);
78                     frameTimer.start();
79                     for (SearchState nextStates : oldList) {
80                         expandState(level, activeList, nextStates);
81                     }
82                     frameTimer.stop();
83                     Collections.shuffle(activeList, random);
84                     if (activeList.size() > maxBeam) {
85                         activeList = activeList.subList(0, maxBeam);
86                     }
87                 }
88                 linguist.stopRecognition();
89                 frameTimer.dump();
90             }
91             totalTimer.stop();
92             System.out.println(" MaxSuccessors : " + maxSuccessors);
93             System.out.println(" TotalStates   : " + totalStates);
94             System.out.println(" TotalEmitting : " + totalEmittingStates);
95             System.out.println("   NonEmitting : " + totalNonEmittingStates);
96             System.out.println("  Final States : " + totalFinalStates);
97             TimerPool.dumpAll();
98         }
99     
100     
101         /**
102          * expand the give search state
103          *
104          * @param level      the nesting level
105          * @param activeList where next states are placed
106          * @param state      the search state to expand
107          */
108         private void expandState(int level, List<SearchState> activeList, SearchState state) {
109             SearchStateArc[] newStates = state.getSuccessors();
110             totalStates++;
111             // System.out.println(Utilities.pad(level * 2) + state);
112             if (newStates.length > maxSuccessors) {
113                 maxSuccessors = newStates.length;
114             }
115             for (SearchStateArc newState : newStates) {
116                 SearchState ns = newState.getState();
117                 if (ns.isEmitting()) {
118                     totalEmittingStates++;
119                     activeList.add(ns);
120                 } else if (!ns.isFinal()) {
121                     totalNonEmittingStates++;
122                     activeList.add(ns);
123                     if (details && ns.isFinal()) {
124                         System.out.println("result " + ns.toPrettyString());
125                     }
126                     expandState(level + 1, activeList, ns);
127                 } else {
128                     totalFinalStates++;
129                 }
130                 totalStates++;
131             }
132         }
133     }
