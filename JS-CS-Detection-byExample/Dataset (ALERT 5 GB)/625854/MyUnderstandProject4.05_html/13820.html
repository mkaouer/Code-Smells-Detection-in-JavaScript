13     package edu.cmu.sphinx.util.props;
14     
15     /** Indicates that a problem occurred while setting one or more properties for this component */
16     public class PropertyException extends RuntimeException {
17     
18         private String instanceName;
19         private String propertyName;
20         private String msg;
21     
22     
23         /**
24          * Creates a new property exception.
25          *
26          * @param instanceName The component this exception is related to.  (or <code>null</code> if unknown)
27          * @param propertyName The name of the component-property which the problem is related. (or <code>null</code> if
28          *                     unknown)
29          * @param msg          a description of the problem.
30          */
31         public PropertyException(String instanceName, String propertyName, String msg) {
32             this(null, instanceName, propertyName, msg);
33         }
34     
35     
36         /**
37          * Creates a new property exception.
38          *
39          * @param cause        The cause of exception. (or <code>null</code> if unknown)
40          * @param instanceName The component this exception is related to.  (or <code>null</code> if unknown)
41          * @param propertyName The name of the component-property which the problem is related. (or <code>null</code> if
42          *                     unknown)
43          * @param msg          a description of the problem.
44          */
45         public PropertyException(Throwable cause, String instanceName, String propertyName, String msg) {
46             super(cause);
47     
48             this.instanceName = instanceName;
49             this.propertyName = propertyName;
50             this.msg = msg;
51         }
52     
53     
54         public PropertyException(Exception e) {
55             super(e);
56         }
57     
58     
59         /** @return Returns the msg. */
60         public String getMsg() {
61             return msg;
62         }
63     
64     
65         /**
66          * Retrieves the name of the offending property
67          *
68          * @return the name of the offending property
69          */
70         public String getProperty() {
71             return propertyName;
72         }
73     
74     
75         /**
76          * Returns a string representation of this object
77          *
78          * @return the string representation of the object.
79          */
80         @Override
81         public String toString() {
82             return "Property exception component:'" + instanceName + "' property:'" + propertyName + "' - " + msg + '\n'
83                     + super.toString();
84         }
85     }
