15     /**
16      * Defines the Bark frequency warping function. This class provides methods to convert frequencies from a linear scale
17      * to the bark scale. The bark scale is originated from measurements of the critical bandwidth. Please find more details
18      * in books about psychoacoustics or speech analysis/recognition.
19      *
20      * @author <a href="mailto:rsingh@cs.cmu.edu">rsingh</a>
21      * @version 1.0
22      */
23     public class FrequencyWarper {
24     
25     
26         /**
27          * Default null constructor. No fields to be initialized.
28          * <p/>
29          * Compute Bark frequency from linear frequency in Hertz. The function is: bark = 6.0*log(hertz/600 +
30          * sqrt((hertz/600)^2 + 1))
31          *
32          * @param hertz the input frequency in Hertz
33          * @return the frequency in a Bark scale
34          */
35         /*
36         public FrequencyWarper (){}
37         */
38     
39         /**
40          * Compute Bark frequency from linear frequency in Hertz.
41          * The function is:
42          * bark = 6.0*log(hertz/600 + sqrt((hertz/600)^2 + 1))
43          *
44          * @param hertz the input frequency in Hertz
45          *
46          * @return the frequency in a Bark scale
47          *
48          */
49         public double hertzToBark(double hertz) {
50             double x = hertz / 600;
51             return (6.0 * Math.log(x + Math.sqrt(x * x + 1)));
52         }
53     
54     
55         /**
56          * Compute linear frequency in Hertz from Bark frequency. The function is: hertz = 300*(exp(bark/6.0) -
57          * exp(-bark/6.0))
58          *
59          * @param bark the input frequency in Barks
60          * @return the frequency in Hertz
61          */
62         public double barkToHertz(double bark) {
63             double x = bark / 6.0;
64             return (300.0 * (Math.exp(x) - Math.exp(-x)));
65         }
66     
67     }// FrequencyWarper
