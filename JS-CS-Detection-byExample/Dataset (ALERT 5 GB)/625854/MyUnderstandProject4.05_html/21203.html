13     package edu.cmu.sphinx.decoder.search;
14     
15     import edu.cmu.sphinx.result.Result;
16     import edu.cmu.sphinx.util.props.Configurable;
17     
18     /**
19      * Defines the interface for the SearchManager. The SearchManager's primary role is to execute the search for a given
20      * number of frames. The SearchManager will return interim results as the recognition proceeds and when recognition
21      * completes a final result will be returned.
22      */
23     public interface SearchManager extends Configurable {
24     
25         /**
26          * Allocates the resources necessary for this search. This should be called once before an recognitions are
27          * performed
28          */
29         public void allocate();
30     
31     
32         /**
33          * Deallocates resources necessary for this search. This should be called once after all recognitions are completed
34          * at the search manager is no longer needed.
35          */
36         public void deallocate();
37     
38     
39         /**
40          * Prepares the SearchManager for recognition.  This method must be called before <code> recognize </code> is
41          * called. Typically, <code> start </code>  and <code> stop </code>  are called bracketing an utterance.
42          */
43         public void startRecognition();
44     
45     
46         /** Performs post-recognition cleanup. This method should be called after recognize returns a final result. */
47         public void stopRecognition();
48     
49     
50         /**
51          * Performs recognition. Processes no more than the given number of frames before returning. This method returns a
52          * partial result after nFrames have been processed, or a final result if recognition completes while processing
53          * frames.  If a final result is returned, the actual number of frames processed can be retrieved from the result.
54          * This method may block while waiting for frames to arrive.
55          *
56          * @param nFrames the maximum number of frames to process. A final result may be returned before all nFrames are
57          *                processed.
58          * @return the recognition result, the result may be a partial or a final result; or return null if no frames are
59          *         arrived
60          */
61         public Result recognize(int nFrames);
62     }
