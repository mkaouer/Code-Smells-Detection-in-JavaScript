13     package edu.cmu.sphinx.research.parallel;
14     
15     import edu.cmu.sphinx.decoder.search.Token;
16     import edu.cmu.sphinx.linguist.flat.SentenceHMMState;
17     
18     import java.util.*;
19     
20     
21     /**
22      * A token for the sentence HMM states that are not feature stream-specific. The <code>getScore()</code> method returns
23      * the combined score. A combined token carries a parallel token for each feature stream, so that scores pertaining to
24      * each stream can be propagated.
25      */
26     public class CombineToken extends Token implements Iterable<ParallelToken> {
27     
28         private Map<Object, ParallelToken> tokens;     // the list of parallel tokens
29     
30     
31         /**
32          * Constructs a CombineToken
33          *
34          * @param predecessor the predecessor for this token
35          * @param state       the SentenceHMMState associated with this token
36          * @param frameNumber the frame number associated with this token
37          */
38         public CombineToken(Token predecessor,
39                             SentenceHMMState state,
40                             int frameNumber) {
41             super(predecessor, state, 0.0f, 0.0f, 0.0f, frameNumber);
42             this.tokens = new HashMap<Object, ParallelToken>();
43         }
44     
45     
46         /**
47          * Returns the combined score of this CombineToken.
48          *
49          * @return the combined score
50          */
51         public float getCombinedScore() {
52             return getScore();
53         }
54     
55     
56         /**
57          * Sets the combined score.
58          *
59          * @param combinedScore the combined score
60          */
61         public void setCombinedScore(float combinedScore) {
62             setScore(combinedScore);
63         }
64     
65     
66         /**
67          * Adds the parallel score of the given parallel stream.
68          *
69          * @param key           the stream name
70          * @param parallelToken the the parallel token
71          */
72         public ParallelToken addParallelToken(Object key,
73                                               ParallelToken parallelToken) {
74             return tokens.put(key, parallelToken);
75         }
76     
77     
78         /**
79          * Adds all the ParallelTokens in the given list into this CombineToken. The added tokens are keyed by their model
80          * name.
81          *
82          * @param tokenList the list of ParallelTokens
83          */
84         public void addAll(List<ParallelToken> tokenList) {
85             for (ParallelToken token : tokenList)
86                 tokens.put(token.getFeatureStream(), token);
87         }
88     
89     
90         /** Removes all the ParallelTokens from this CombineToken. */
91         public void clear() {
92             tokens.clear();
93         }
94     
95     
96         /**
97          * Returns an Iterator for the ParallelToken(s).
98          *
99          * @return an Iterator for the ParallelToken(s)
100          */
101         @Override
102         public Iterator<ParallelToken> iterator() {
103             return tokens.values().iterator();
104         }
105     
106     
107         /**
108          * Returns a Collection of all the parallel tokens.
109          *
110          * @return a Collection of all the parallel tokens
111          */
112         public Collection<ParallelToken> getParallelTokens() {
113             return tokens.values();
114         }
115     
116     
117         /**
118          * Returns the parallel token of the given parallel stream.
119          *
120          * @return the parallel token of the given parallel stream
121          */
122         public ParallelToken getParallelToken(FeatureStream stream) {
123             return tokens.get(stream);
124         }
125     
126     
127         /**
128          * Sets the last combine time of all the ParallelTokens in this CombineToken.
129          *
130          * @param frameNumber the last combine time
131          */
132         public void setLastCombineTime(int frameNumber) {
133             for (ParallelToken pToken : this)
134                 pToken.setLastCombineTime(frameNumber);
135         }
136     
137     
138         /**
139          * Returns the string representation of this object.
140          *
141          * @return the string representation of this object
142          */
143         public String toString() {
144             StringBuilder parallelTokenScores = new StringBuilder("CombinedToken: ")
145                 .append("Frame: ").append(getNumberFormat().format(getFrameNumber()))
146                 .append(", score: ").append(getScoreFormat().format(getScore())).append('\n');
147             int t = 0;
148             for (ParallelToken token : this) {
149                 parallelTokenScores.append("   ParallelToken ").append(t).append(", ").append(token.getModelName())
150                     .append(", feature: ").append(getScoreFormat().format(token.getFeatureScore()))
151                     .append(", combined: ").append(getScoreFormat().format(token.getCombinedScore())).append('\n');
152             }
153             return parallelTokenScores.toString();
154         }
155     }
