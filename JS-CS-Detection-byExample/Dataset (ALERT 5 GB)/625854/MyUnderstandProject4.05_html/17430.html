25     /**
26      * Loads a tied-state acoustic model generated by the Sphinx-3 trainer.
27      * <p/>
28      * <p/>
29      * It is not the goal of this documentation to provide an explanation about the concept of HMMs. The explanation below
30      * is superficial, and provided only in a way that the files in the acoustic model package make sense.
31      * <p/>
32      * An HMM models a process using a sequence of states. Associated with each state, there is a probability density
33      * function. A popular choice for this function is a Gaussian mixture, that is, a summation of Gaussians. As you may
34      * recall, a single Gaussian is defined by a mean and a variance, or, in the case of a multidimensional Gaussian, by a
35      * mean vector and a covariance matrix, or, under some simplifying assumptions, a variance vector. The "means" and
36      * "variances" files in the "continuous" directory contain exactly this: a table in which each line contains a mean
37      * vector or a variance vector respectively. The dimension of these vectors is the same as the incoming data, the
38      * encoded speech signal. The Gaussian mixture is a summation of Gaussians, with different weights for different
39      * Gaussians. The "mixture_weights" file contains this: each line contains the weights for a combination of Gaussians.
40      * <p/>
41      * The HMM is a model with a set of states. The transitions between states have an associated probability. These
42      * probabilities make up the transition matrices stored in the "transition_matrices" file.
43      * <p/>
44      * The files in the "continuous" directory are, therefore, tables, or pools, of means, variances, mixture weights, and
45      * transition probabilities.
46      * <p/>
47      * The dictionary is a file that maps words to their phonetic transcriptions, that is, it maps words to sequences of
48      * phonemes.
49      * <p/>
50      * The language model contains information about probabilities of words in a language. These probabilities could be for
51      * individual words or for sequences of two or three words.
52      * <p/>
53      * The model definition file in a way ties everything together. If the recognition system models phonemes, there is an
54      * HMM for each phoneme. The model definition file has one line for each phoneme. The phoneme could be in a context
55      * dependent or independent. Each line, therefore, identifies a unique HMM. This line has the phoneme identification,
56      * the non-required left or right context, the index of a transition matrix, and, for each state, the index of a mean
57      * vector, a variance vector, and a set of mixture weights.
58      */

142         /**
143          * Gets a composite HMM for the given unit and context
144          *
145          * @param unit     the unit for the hmm
146          * @param position the position of the unit within the word
147          * @return a composite HMM
148          */
149         private HMM getCompositeHMM(Unit unit, HMMPosition position) {
150     
151             Unit ciUnit = unitManager.getUnit(unit.getName(), unit.isFiller(),
152                     Context.EMPTY_CONTEXT);
153     
154             SenoneSequence compositeSequence = getCompositeSenoneSequence(unit,
155                     position);
156     
157             SenoneHMM contextIndependentHMM = (SenoneHMM) lookupNearestHMM(ciUnit,
158                     HMMPosition.UNDEFINED, true);
159             float[][] tmat = contextIndependentHMM.getTransitionMatrix();
160             return new SenoneHMM(unit, compositeSequence, tmat, position);
161         }

164         /**
165          * Given a unit, returns the HMM that best matches the given unit. If exactMatch is false and an exact match is not
166          * found, then different word positions are used. If any of the contexts are non-silence filler units. a silence
167          * filler unit is tried instead.
168          *
169          * @param unit       the unit of interest
170          * @param position   the position of the unit of interest
171          * @param exactMatch if true, only an exact match is acceptable.
172          * @return the HMM that best matches, or null if no match could be found.
173          */
174         @Override
175         public HMM lookupNearestHMM(Unit unit, HMMPosition position,
176                                     boolean exactMatch) {
177     
178             if (exactMatch) {
179                 return lookupHMM(unit, position);
180             } else {
181                 HMMManager mgr = loader.getHMMManager();
182                 HMM hmm = mgr.get(position, unit);
183     
184                 if (hmm != null) {
185                     return hmm;
186                 }
187                 // no match, try a composite
188     
189                 if (useComposites && hmm == null) {
190                     if (isComposite(unit)) {
191     
192                         hmm = getCompositeHMM(unit, position);
193                         if (hmm != null) {
194                             mgr.put(hmm);
195                         }
196                     }
197                 }
198                 // no match, try at other positions
199                 if (hmm == null) {
200                     hmm = getHMMAtAnyPosition(unit);
201                 }
202                 // still no match, try different filler
203                 if (hmm == null) {
204                     hmm = getHMMInSilenceContext(unit, position);
205                 }
206     
207                 // still no match, backoff to base phone
208                 if (hmm == null) {
209                     Unit ciUnit = lookupUnit(unit.getName());
210     
211                     assert unit.isContextDependent();
212                     if (ciUnit == null) {
213                         logger.severe("Can't find HMM for " + unit.getName());
214                     }
215                     assert ciUnit != null;
216                     assert !ciUnit.isContextDependent();
217     
218                     hmm = mgr.get(HMMPosition.UNDEFINED, ciUnit);
219                 }
220     
221                 assert hmm != null;
222     
223                 // System.out.println("PROX match for "
224                 // 	+ unit + " at " + position + ":" + hmm);
225     
226                 return hmm;
227             }
228         }

231         /**
232          * Determines if a unit is a composite unit
233          *
234          * @param unit the unit to test
235          * @return true if the unit is missing a right context
236          */
237         private boolean isComposite(Unit unit) {
238     
239             if (unit.isFiller()) {
240                 return false;
241             }
242     
243             Context context = unit.getContext();
244             if (context instanceof LeftRightContext) {
245                 LeftRightContext lrContext = (LeftRightContext) context;
246                 if (lrContext.getRightContext() == null) {
247                     return true;
248                 }
249                 if (lrContext.getLeftContext() == null) {
250                     return true;
251                 }
252             }
253             return false;
254         }

292         /**
293          * Get a composite senone sequence given the unit The unit should have a LeftRightContext, where one or two of
294          * 'left' or 'right' may be null to indicate that the match should succeed on any context.
295          *
296          * @param unit the unit
297          */
298         public SenoneSequence getCompositeSenoneSequence(Unit unit,
299                                                          HMMPosition position) {
300             Context context = unit.getContext();
301             SenoneSequence compositeSenoneSequence = null;
302             compositeSenoneSequence = compositeSenoneSequenceCache.get(unit.toString());
303     
304             if (logger.isLoggable(Level.FINE)) {
305                 logger.fine("getCompositeSenoneSequence: " + unit + (compositeSenoneSequence == null ? "" : "Cached"));
306             }
307             if (compositeSenoneSequence != null) {
308                 return compositeSenoneSequence;
309             }
310     
311             // Iterate through all HMMs looking for
312             // a) An hmm with a unit that has the proper base
313             // b) matches the non-null context
314     
315             List<SenoneSequence> senoneSequenceList = new ArrayList<SenoneSequence>();
316     
317             // collect all senone sequences that match the pattern
318             for (Iterator<HMM> i = getHMMIterator(); i.hasNext();) {
319                 SenoneHMM hmm = (SenoneHMM) i.next();
320                 if (hmm.getPosition() == position) {
321                     Unit hmmUnit = hmm.getUnit();
322                     if (hmmUnit.isPartialMatch(unit.getName(), context)) {
323                         if (logger.isLoggable(Level.FINE)) {
324                             logger.fine("collected: " + hmm.getUnit());
325                         }
326                         senoneSequenceList.add(hmm.getSenoneSequence());
327                     }
328                 }
329             }
330     
331             // couldn't find any matches, so at least include the CI unit
332             if (senoneSequenceList.isEmpty()) {
333                 Unit ciUnit = unitManager.getUnit(unit.getName(), unit.isFiller());
334                 SenoneHMM baseHMM = lookupHMM(ciUnit, HMMPosition.UNDEFINED);
335                 senoneSequenceList.add(baseHMM.getSenoneSequence());
336             }
337     
338             // Add this point we have all of the senone sequences that
339             // match the base/context pattern collected into the list.
340             // Next we build a CompositeSenone consisting of all of the
341             // senones in each position of the list.
342     
343             // First find the longest senone sequence
344     
345             int longestSequence = 0;
346             for (SenoneSequence ss : senoneSequenceList) {
347                 if (ss.getSenones().length > longestSequence) {
348                     longestSequence = ss.getSenones().length;
349                 }
350             }
351     
352             // now collect all of the senones at each position into
353             // arrays so we can create CompositeSenones from them
354             // QUESTION: is is possible to have different size senone
355             // sequences. For now lets assume the worst case.
356     
357             List<CompositeSenone> compositeSenones = new ArrayList<CompositeSenone>();
358             float logWeight = 0.0f;
359             for (int i = 0; i < longestSequence; i++) {
360                 Set<Senone> compositeSenoneSet = new HashSet<Senone>();
361                 for (SenoneSequence senoneSequence : senoneSequenceList) {
362                     if (i < senoneSequence.getSenones().length) {
363                         Senone senone = senoneSequence.getSenones()[i];
364                         compositeSenoneSet.add(senone);
365                     }
366                 }
367                 compositeSenones.add(CompositeSenone.create(
368                         compositeSenoneSet, logWeight));
369             }
370     
371             compositeSenoneSequence = SenoneSequence.create(compositeSenones);
372             compositeSenoneSequenceCache.put(unit.toString(),
373                     compositeSenoneSequence);
374     
375             if (logger.isLoggable(Level.FINE)) {
376                 logger.fine(unit + " consists of " + compositeSenones.size() + " composite senones");
377                 if (logger.isLoggable(Level.FINEST)) {
378                     compositeSenoneSequence.dump("am");
379                 }
380             }
381             return compositeSenoneSequence;
382         }

450         /**
451          * Given a unit, search for the HMM associated with this unit by replacing all non-silence filler contexts with the
452          * silence filler context
453          *
454          * @param unit the unit of interest
455          * @return the associated hmm or null
456          */
457         private SenoneHMM getHMMInSilenceContext(Unit unit, HMMPosition position) {
458             SenoneHMM hmm = null;
459             HMMManager mgr = loader.getHMMManager();
460             Context context = unit.getContext();
461     
462             if (context instanceof LeftRightContext) {
463                 LeftRightContext lrContext = (LeftRightContext) context;
464     
465                 Unit[] lc = lrContext.getLeftContext();
466                 Unit[] rc = lrContext.getRightContext();
467     
468                 Unit[] nlc;
469                 Unit[] nrc;
470     
471                 if (hasNonSilenceFiller(lc)) {
472                     nlc = replaceNonSilenceFillerWithSilence(lc);
473                 } else {
474                     nlc = lc;
475                 }
476     
477                 if (hasNonSilenceFiller(rc)) {
478                     nrc = replaceNonSilenceFillerWithSilence(rc);
479                 } else {
480                     nrc = rc;
481                 }
482     
483                 if (nlc != lc || nrc != rc) {
484                     Context newContext = LeftRightContext.get(nlc, nrc);
485                     Unit newUnit = unitManager.getUnit(unit.getName(),
486                             unit.isFiller(), newContext);
487                     hmm = (SenoneHMM) mgr.get(position, newUnit);
488                     if (hmm == null) {
489                         hmm = getHMMAtAnyPosition(newUnit);
490                     }
491                 }
492             }
493             return hmm;
494         }

497         /**
498          * Returns true if the array of units contains a non-silence filler
499          *
500          * @param units the units to check
501          * @return true if the array contains a filler that is not the silence filler
502          */
503         private boolean hasNonSilenceFiller(Unit[] units) {
504             if (units == null) {
505                 return false;
506             }
507     
508             for (Unit unit : units) {
509                 if (unit.isFiller() &&
510                     !unit.equals(UnitManager.SILENCE)) {
511                     return true;
512                 }
513             }
514             return false;
515         }

518         /**
519          * Returns a unit array with all non-silence filler units replaced with the silence filler a non-silence filler
520          *
521          * @param context the context to check
522          * @return true if the array contains a filler that is not the silence filler
523          */
524         private Unit[] replaceNonSilenceFillerWithSilence(Unit[] context) {
525             Unit[] replacementContext = new Unit[context.length];
526             for (int i = 0; i < context.length; i++) {
527                 if (context[i].isFiller() &&
528                         !context[i].equals(UnitManager.SILENCE)) {
529                     replacementContext[i] = UnitManager.SILENCE;
530                 } else {
531                     replacementContext[i] = context[i];
532                 }
533             }
534             return replacementContext;
535         }
