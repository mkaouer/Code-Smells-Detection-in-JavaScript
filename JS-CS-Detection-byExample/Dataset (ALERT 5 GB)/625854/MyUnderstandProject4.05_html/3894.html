13     package edu.cmu.sphinx.linguist.acoustic;
14     
15     import edu.cmu.sphinx.util.props.Configurable;
16     
17     import java.io.IOException;
18     import java.util.Iterator;
19     import java.util.Properties;
20     import edu.cmu.sphinx.util.props.S4String;
21     
22     /** Represents the generic interface to the Acoustic Model for sphinx4 */
23     public interface AcousticModel extends Configurable {
24     
25         /** The directory where the acoustic model data can be found. */
26         @S4String(defaultValue = ".")
27         public final static String PROP_LOCATION = "location";
28     
29         /**
30          * Gets this acoustic model ready to use, allocating all necessary resources.
31          *
32          * @throws IOException if the model could not be loaded
33          */
34         public void allocate() throws IOException;
35     
36     
37         /** Deallocates previously allocated resources */
38         public void deallocate();
39     
40     
41         /**
42          * Returns the name of this AcousticModel, or null if it has no name.
43          *
44          * @return the name of this AcousticModel, or null if it has no name
45          */
46         public String getName();
47     
48     
49         /**
50          * Given a unit, returns the HMM that best matches the given unit. If exactMatch is false and an exact match is not
51          * found, then different word positions are used. If any of the contexts are non-silence filler units. a silence
52          * filler unit is tried instead.
53          *
54          * @param unit       the unit of interest
55          * @param position   the position of the unit of interest
56          * @param exactMatch if true, only an exact match is acceptable.
57          * @return the HMM that best matches, or null if no match could be found.
58          */
59         public HMM lookupNearestHMM(Unit unit, HMMPosition position,
60                                     boolean exactMatch);
61     
62     
63         /**
64          * Returns an iterator that can be used to iterate through all the HMMs of the acoustic model
65          *
66          * @return an iterator that can be used to iterate through all HMMs in the model. The iterator returns objects of
67          *         type <code>HMM</code>.
68          */
69         public Iterator<HMM> getHMMIterator();
70     
71     
72         /**
73          * Returns an iterator that can be used to iterate through all the CI units in the acoustic model
74          *
75          * @return an iterator that can be used to iterate through all CI units. The iterator returns objects of type
76          *         <code>Unit</code>
77          */
78         public Iterator<Unit> getContextIndependentUnitIterator();
79     
80     
81         /**
82          * Returns the size of the left context for context dependent units
83          *
84          * @return the left context size
85          */
86         public int getLeftContextSize();
87     
88     
89         /**
90          * Returns the size of the right context for context dependent units
91          *
92          * @return the left context size
93          */
94         public int getRightContextSize();
95     
96     
97         /**
98          * Returns the properties of this acoustic model.
99          *
100          * @return the properties of this acoustic model
101          */
102         public Properties getProperties();
103     }
