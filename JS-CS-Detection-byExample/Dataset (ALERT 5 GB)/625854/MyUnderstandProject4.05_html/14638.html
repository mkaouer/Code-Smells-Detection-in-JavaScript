81          * Recursively go through the rules and create the same structures from
82          * JSAPI
83          */
84         public JSGFRule convert(Rule rule) {
85     
86             if (rule == null)
87                 return null;
88     
89             if (rule instanceof RuleName) {
90                 if (rule == RuleName.NULL)
91                     return JSGFRuleName.NULL;
92                 if (rule == RuleName.VOID)
93                     return JSGFRuleName.NULL;
94                 return new JSGFRuleName(((RuleName) rule).getRuleName());
95             }
96             if (rule instanceof RuleAlternatives) {
97                 RuleAlternatives ruleAlternatives = (RuleAlternatives) rule;
98                 List<JSGFRule> subrules = new ArrayList<JSGFRule>();
99                 for (Rule r : ruleAlternatives.getRules()) {
100                     subrules.add(convert(r));
101                 }
102                 List<Float> weights = null;
103                 if (ruleAlternatives.getWeights() != null) {
104                     weights = new ArrayList<Float>();
105                     for (float f : ruleAlternatives.getWeights()) {
106                         weights.add(f);
107                     }
108                 }
109                 return new JSGFRuleAlternatives(subrules, weights);
110             }
111     
112             if (rule instanceof RuleSequence) {
113                 RuleSequence ruleSequence = (RuleSequence) rule;
114                 List<JSGFRule> subrules = new ArrayList<JSGFRule>();
115                 for (Rule r : ruleSequence.getRules()) {
116                     subrules.add(convert(r));
117                 }
118                 return new JSGFRuleSequence(subrules);
119             }
120     
121             if (rule instanceof RuleCount) {
122                 RuleCount ruleCount = (RuleCount) rule;
123                 return new JSGFRuleCount(convert(ruleCount.getRule()), ruleCount
124                         .getCount());
125             }
126     
127             if (rule instanceof RuleTag) {
128                 RuleTag ruleTag = (RuleTag) rule;
129                 return new JSGFRuleTag(convert(ruleTag.getRule()), ruleTag.getTag());
130             }
131     
132             if (rule instanceof RuleToken) {
133                 RuleToken ruleToken = (RuleToken) rule;
134                 return new JSGFRuleToken(ruleToken.getText());
135             }
136             System.out.println("Can't convert rule " + rule.getClass());
137             return null;
138         }

140         /*
141          * Recursively go through the rules and create the same structures for JSAPI
142          */
143         private Rule convert(JSGFRule rule) {
144     
145             if (rule == null)
146                 return null;
147     
148             if (rule instanceof JSGFRuleName) {
149                 if (rule == JSGFRuleName.NULL)
150                     return RuleName.NULL;
151                 if (rule == JSGFRuleName.VOID)
152                     return RuleName.NULL;
153     
154                 JSGFRuleName ruleName = (JSGFRuleName) rule;
155                 return new RuleName(ruleName.getRuleName());
156             }
157     
158             if (rule instanceof JSGFRuleAlternatives) {
159                 JSGFRuleAlternatives ruleAlternatives = (JSGFRuleAlternatives) rule;
160                 Rule[] subrules = new Rule[ruleAlternatives.getRules().size()];
161                 int i = 0;
162                 for (JSGFRule subrule : ruleAlternatives.getRules()) {
163                     subrules[i] = convert(subrule);
164                     i++;
165                 }
166                 float[] weights = null;
167                 if (ruleAlternatives.getWeights() != null) {
168                     weights = new float[ruleAlternatives.getWeights().size()];
169                     i = 0;
170                     for (Float f : ruleAlternatives.getWeights()) {
171                         weights[i] = f;
172                         i++;
173                     }
174                 }
175                 return new RuleAlternatives(subrules, weights);
176             }
177     
178             if (rule instanceof JSGFRuleSequence) {
179                 JSGFRuleSequence ruleSequence = (JSGFRuleSequence) rule;
180                 Rule[] subrules = new Rule[ruleSequence.getRules().size()];
181                 int i = 0;
182                 for (JSGFRule subrule : ruleSequence.getRules()) {
183                     subrules[i] = convert(subrule);
184                     i++;
185                 }
186                 return new RuleSequence(subrules);
187             }
188     
189             if (rule instanceof JSGFRuleCount) {
190                 JSGFRuleCount ruleCount = (JSGFRuleCount) rule;
191                 return new RuleCount(convert(ruleCount.getRule()), ruleCount
192                         .getCount());
193             }
194     
195             if (rule instanceof JSGFRuleTag) {
196                 JSGFRuleTag ruleTag = (JSGFRuleTag) rule;
197                 return new RuleTag(convert(ruleTag.getRule()), ruleTag.getTag());
198             }
199     
200             if (rule instanceof JSGFRuleToken) {
201                 JSGFRuleToken ruleToken = (JSGFRuleToken) rule;
202                 return new RuleToken(ruleToken.getText());
203             }
204     
205             System.out.println("Unknown rule type " + rule.getClass());
206             return null;
207         }

404         /**
405          * Parse the nth best result of a FinalRuleResult against the specified
406          * rule. Uses the RuleParser class. From
407          * javax.speech.recognition.RuleGrammar.
408          * 
409          * @param r
410          *            the FinalRuleResult.
411          * @param nBest
412          *            the nth best result to use.
413          * @param ruleName
414          *            the name of rule to use for parsing.
415          */
416         public RuleParse parse(FinalRuleResult r, int nBest, String ruleName)
417                 throws GrammarException {
418             // Some JSAPI implementations we run into are not JSAPI complaint,
419             // so try a few alternatives
420             ResultToken rt[] = r.getAlternativeTokens(nBest);
421             if (rt != null || (rt = r.getBestTokens()) != null) {
422                 String tokens[] = new String[rt.length];
423                 for (int i = 0; i < rt.length; i++) {
424                     tokens[i] = rt[i].getSpokenText();
425                 }
426                 return parse(tokens, ruleName);
427             } else {
428                 return parse(r.toString(), ruleName);
429             }
430         }
