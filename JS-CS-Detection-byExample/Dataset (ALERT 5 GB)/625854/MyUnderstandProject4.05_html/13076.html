13     package edu.cmu.sphinx.instrumentation;
14     
15     import edu.cmu.sphinx.decoder.search.Token;
16     import edu.cmu.sphinx.result.Result;
17     import edu.cmu.sphinx.util.props.*;
18     import edu.cmu.sphinx.recognizer.Recognizer;
19     
20     /** Tracks and reports recognition accuracy based upon the highest scoring path in a Result. */
21     public class BestPathAccuracyTracker extends AccuracyTracker {
22     
23         /** The property that define whether the full token path is displayed */
24         @S4Boolean(defaultValue = false)
25         public final static String PROP_SHOW_FULL_PATH = "showFullPath";
26     
27         private boolean showFullPath;
28         
29         public BestPathAccuracyTracker(Recognizer recognizer, boolean showSummary, boolean showDetails, boolean showResults, boolean showAlignedResults, boolean showRawResults, boolean showFullPath) {
30             super(recognizer, showSummary, showDetails, showResults, showAlignedResults, showRawResults);
31             this.showFullPath = showFullPath;
32         }
33     
34         public BestPathAccuracyTracker() {
35     
36         }
37     
38     
39         /*
40         * (non-Javadoc)
41         *
42         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
43         */
44         @Override
45         public void newProperties(PropertySheet ps) throws PropertyException {
46             super.newProperties(ps);
47             showFullPath = ps.getBoolean(PROP_SHOW_FULL_PATH);
48         }
49     
50     
51         /**
52          * Dumps the best path
53          *
54          * @param result the result to dump
55          */
56         private void showFullPath(Result result) {
57             if (showFullPath) {
58                 System.out.println();
59                 Token bestToken = result.getBestToken();
60                 if (bestToken != null) {
61                     bestToken.dumpTokenPath();
62                 } else {
63                     System.out.println("Null result");
64                 }
65                 System.out.println();
66             }
67         }
68     
69     
70         /*
71         * (non-Javadoc)
72         *
73         * @see edu.cmu.sphinx.decoder.ResultListener#newResult(edu.cmu.sphinx.result.Result)
74         */
75         @Override
76         public void newResult(Result result) {
77             String ref = result.getReferenceText();
78             if (result.isFinal() && ref != null) {
79                 String hyp = result.getBestResultNoFiller();
80                 getAligner().align(ref, hyp);
81                 showFullPath(result);
82                 showDetails(result.toString());
83             }
84         }
85     }
