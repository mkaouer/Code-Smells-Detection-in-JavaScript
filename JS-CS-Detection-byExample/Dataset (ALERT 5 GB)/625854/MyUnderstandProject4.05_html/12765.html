270         /**
271          * Performs the recognition for the given number of frames.
272          *
273          * @param nFrames the number of frames to recognize
274          * @return the current result
275          */
276         @Override
277         public Result recognize(int nFrames) {
278             boolean done = false;
279             Result result = null;
280             streamEnd = false;
281     
282             for (int i = 0; i < nFrames && !done; i++) {
283                 done = recognize();
284             }
285             
286             if (!streamEnd) {
287             	result = new Result(loserManager, activeList, resultList,
288             					    currentFrameNumber, done, logMath);
289             }
290     
291             // tokenTypeTracker.show();
292             if (showTokenCount) {
293                 showTokenCount();
294             }
295             return result;
296         }

298         private boolean recognize() {
299     
300             activeList = activeListManager.getEmittingList();
301             boolean more = scoreTokens();
302     
303             if (more) {
304                 pruneBranches();
305                 currentFrameNumber++;
306                 if (growSkipInterval == 0 || (currentFrameNumber % growSkipInterval) != 0) {            	
307                     clearCollectors();
308                     growEmittingBranches();
309                     growNonEmittingBranches();
310                 }
311             }
312             return !more;
313         }

378         /**
379          * Goes through the active list of tokens and expands each token, finding the set of successor tokens until all the
380          * successor tokens are emitting tokens.
381          */
382         protected void growBranches() {
383             growTimer.start();
384             float relativeBeamThreshold = activeList.getBeamThreshold();
385             if (logger.isLoggable(Level.FINE)) {
386                 logger.fine("Frame: " + currentFrameNumber
387                         + " thresh : " + relativeBeamThreshold + " bs "
388                         + activeList.getBestScore() + " tok "
389                         + activeList.getBestToken());
390             }
391             for (Token token : activeList) {
392                 if (token.getScore() >= relativeBeamThreshold && allowExpansion(token)) {
393                     collectSuccessorTokens(token);
394                 }
395             }
396             growTimer.stop();
397         }
398     
399     
400         /**
401          * Grows the emitting branches. This version applies a simple acoustic lookahead based upon the rate of change in
402          * the current acoustic score.
403          */
404         protected void growEmittingBranches() {
405             if (acousticLookaheadFrames > 0F) {
406                 growTimer.start();
407                 float bestScore = -Float.MAX_VALUE;
408                 for (Token t : activeList) {
409                     float score = t.getScore() + t.getAcousticScore()
410                             * acousticLookaheadFrames;
411                     if (score > bestScore) {
412                         bestScore = score;
413                     }
414                     t.setWorkingScore(score);
415                 }
416                 float relativeBeamThreshold = bestScore + relativeBeamWidth;
417     
418                 for (Token t : activeList) {
419                     if (t.getWorkingScore() >= relativeBeamThreshold) {
420                         collectSuccessorTokens(t);
421                     }
422                 }
423                 growTimer.stop();
424             } else {
425                 growBranches();
426             }
427         }

430         /** Grow the non-emitting branches, until the tokens reach an emitting state. */
431         private void growNonEmittingBranches() {
432             for (Iterator<ActiveList> i = activeListManager.getNonEmittingListIterator(); i.hasNext();) {
433                 activeList = i.next();
434                 if (activeList != null) {
435                     i.remove();
436                     pruneBranches();
437                     growBranches();
438                 }
439             }
440         }

444          * Calculate the acoustic scores for the active list. The active list should contain only emitting tokens.
445          *
446          * @return <code>true</code> if there are more frames to score, otherwise, false
447          */
448         protected boolean scoreTokens() {
449             boolean moreTokens;
450             scoreTimer.start();
451             Data data = scorer.calculateScores(activeList.getTokens());
452             scoreTimer.stop();
453     
454             Token bestToken = null;
455             if (data instanceof Token) {
456                 bestToken = (Token)data;
457             } else if (data == null) {
458                 streamEnd = true;
459             }
460      
461             moreTokens = (bestToken != null);
462             activeList.setBestToken(bestToken);
463     
464             //monitorWords(activeList);
465             monitorStates(activeList);
466     
467             // System.out.println("BEST " + bestToken);
468     
469             curTokensScored.value += activeList.size();
470             totalTokensScored.value += activeList.size();
471     
472             return moreTokens;
473         }

588         /** Checks that the given two states are in legitimate order.
589          * @param fromState
590          * @param toState*/
591         private void checkStateOrder(SearchState fromState, SearchState toState) {
592             if (fromState.getOrder() == numStateOrder - 1) {
593                 return;
594             }
595     
596             if (fromState.getOrder() > toState.getOrder()) {
597                 throw new Error("IllegalState order: from "
598                         + fromState.getClass().getName() + ' '
599                         + fromState.toPrettyString()
600                         + " order: " + fromState.getOrder()
601                         + " to "
602                         + toState.getClass().getName() + ' '
603                         + toState.toPrettyString()
604                         + " order: " + toState.getOrder());
605             }
606         }

609         /**
610          * Collects the next set of emitting tokens from a token and accumulates them in the active or result lists
611          *
612          * @param token the token to collect successors from be immediately expanded are placed. Null if we should always
613          *              expand all nodes.
614          */
615         protected void collectSuccessorTokens(Token token) {
616     
617             // tokenTracker.add(token);
618             // tokenTypeTracker.add(token);
619     
620             // If this is a final state, add it to the final list
621     
622             if (token.isFinal()) {
623                 resultList.add(getResultListPredecessor(token));
624                 return;
625             }
626     
627             // if this is a non-emitting token and we've already 
628             // visited the same state during this frame, then we
629             // are in a grammar loop, so we don't continue to expand.
630             // This check only works properly if we have kept all of the
631             // tokens (instead of skipping the non-word tokens).
632             // Note that certain linguists will never generate grammar loops
633             // (lextree linguist for example). For these cases, it is perfectly
634             // fine to disable this check by setting keepAllTokens to false
635     
636             if (!token.isEmitting() && (keepAllTokens && isVisited(token))) {
637                 return;
638             }
639     
640             SearchState state = token.getSearchState();
641             SearchStateArc[] arcs = state.getSuccessors();
642             Token predecessor = getResultListPredecessor(token);
643     
644             // For each successor
645             // calculate the entry score for the token based upon the
646             // predecessor token score and the transition probabilities
647             // if the score is better than the best score encountered for
648             // the SearchState and frame then create a new token, add
649             // it to the lattice and the SearchState.
650             // If the token is an emitting token add it to the list,
651             // otherwise recursively collect the new tokens successors.
652     
653             for (SearchStateArc arc : arcs) {
654                 SearchState nextState = arc.getState();
655     
656                 if (checkStateOrder) {
657                     checkStateOrder(state, nextState);
658                 }
659     
660                 // We're actually multiplying the variables, but since
661                 // these come in log(), multiply gets converted to add
662                 float logEntryScore = token.getScore() + arc.getProbability();
663     
664                 Token bestToken = getBestToken(nextState);
665                 boolean firstToken = bestToken == null;
666     
667                 if (firstToken || bestToken.getScore() < logEntryScore) {
668                     Token newBestToken = new Token(predecessor, nextState,
669                             logEntryScore, 
670                             arc.getInsertionProbability(),
671                             arc.getLanguageProbability(), 
672                             currentFrameNumber);
673                     tokensCreated.value++;
674     
675                     setBestToken(newBestToken, nextState);
676                     if (firstToken) {
677                         activeListAdd(newBestToken);
678                     } else {
679     //                    System.out.println("Replacing " + bestToken + " with " + newBestToken);
680                         activeListReplace(bestToken, newBestToken);
681                         if (buildWordLattice && newBestToken.isWord()) {
682     
683                             // Move predecessors of bestToken to precede
684                             // newBestToken, bestToken will be garbage collected.
685                             loserManager.changeSuccessor(newBestToken, bestToken);
686                             loserManager.addAlternatePredecessor(newBestToken,
687                                     bestToken.getPredecessor());
688                         }
689                     }
690                 } else {
691                     if (buildWordLattice && nextState instanceof WordSearchState) {
692                         if (predecessor != null) {
693                             loserManager.addAlternatePredecessor(bestToken,
694                                     predecessor);
695                         }
696                     }
697                 }
698             }
699         }

702         /**
703          * Determines whether or not we've visited the state associated with this token since the previous frame.
704          *
705          * @param t
706          * @return true if we've visited the search state since the last frame
707          */
708         private boolean isVisited(Token t) {
709             SearchState curState = t.getSearchState();
710     
711             t = t.getPredecessor();
712     
713             while (t != null && !t.isEmitting()) {
714                 if (curState.equals(t.getSearchState())) {
715                     System.out.println("CS " + curState + " match " + t.getSearchState());
716                     return true;
