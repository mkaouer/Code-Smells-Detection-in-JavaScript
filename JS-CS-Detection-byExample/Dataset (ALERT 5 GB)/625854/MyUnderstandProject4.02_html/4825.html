474         protected String execute(String[] args) {
475             String response = "";
476     
477             CommandInterface ci;
478     
479             if (args.length > 0) {
480     
481                 ci = commandList.get(args[0]);
482                 if (ci != null) {
483                     response = ci.execute(this, args);
484                 } else {
485                     response = "ERR  CMD_NOT_FOUND";
486                 }
487     
488                 totalCommands++;

507         /**
508          * Parses the given message into an array of strings.
509          *
510          * @param message the string to be parsed.
511          * @return the parsed message as an array of strings
512          */
513         protected String[] parseMessage(String message) {
514             int tokenType;
515             List<String> words = new ArrayList<String>(20);
516             StreamTokenizer st = new StreamTokenizer(new StringReader(message));
517     
518             st.resetSyntax();
519             st.whitespaceChars(0, ' ');
520             st.wordChars('!', 255);
521             st.quoteChar('"');
522             st.quoteChar('\"');
523             st.commentChar('#');
524     
525             while (true) {
526                 try {
527                     tokenType = st.nextToken();
528                     if (tokenType == StreamTokenizer.TT_WORD) {
529                         words.add(st.sval);
530                     } else if (tokenType == '\'' || tokenType == '"') {
531                         words.add(st.sval);
532                     } else if (tokenType == StreamTokenizer.TT_NUMBER) {
533                         System.out.println("Unexpected numeric token!");
534                     } else {
535                         break;
536                     }
537                 } catch (IOException e) {
538                     break;
539                 }
540             }
541             return (String[]) words.toArray(new String[words.size()]);
542         }

547         public void run() {
548             while (!done) {
549                 try {
550                     printPrompt();
551                     String message = getInputLine();
552                     if (message == null) {
553                         break;
554                     } else {
555                         if (trace) {
556                             System.out.println("\n----------");
557                             System.out.println("In : " + message);
558                         }
559                         message = message.trim();
560                         if (message.length() > 0) {
561                             putResponse(execute(message));
562                         }
563                     }
564                 }
565                 catch (IOException e) {
566                     System.out.println("Exception: CommandInterpreter.run()");
567                     break;
568                 }
569             }
570             onExit();
571         }

573         // some history patterns used by getInputLine()
574     
575         private static Pattern historyPush = Pattern.compile("(.+):p");
576         private static Pattern editPattern =
577                 Pattern.compile("\\^(.+?)\\^(.*?)\\^?");
578         private static Pattern bbPattern = Pattern.compile("(!!)");
579     
580     
581         /**
582          * Gets the input line. Deals with history. Currently we support simple csh-like history. !! - execute last command,
583          * !-3 execute 3 from last command, !2 execute second command in history list, !foo - find last command that started
584          * with foo and execute it. Also allows editing of the last command wich ^old^new^ type replacesments
585          *
586          * @return the next history line or null if done
587          */
588         private String getInputLine() throws IOException {
589             String message = in.readLine();
590             if (message == null)
591                 return null;
592     
593             boolean justPush = false;
594             boolean echo = false;
595             boolean error = false;
596     
597             Matcher m = historyPush.matcher(message);
598             if (m.matches()) {
599                 justPush = true;
600                 echo = true;
601                 message = m.group(1);
602             }
603             if (message.startsWith("^")) { // line editing ^foo^fum^
604                 m = editPattern.matcher(message);
605                 if (m.matches()) {
606                     String orig = m.group(1);
607                     String sub = m.group(2);
608                     try {
609                         Pattern pat = Pattern.compile(orig);
610                         Matcher subMatcher = pat.matcher(history.getLast(0));
611                         if (subMatcher.find()) {
612                             message = subMatcher.replaceFirst(sub);
613                             echo = true;
614                         } else {
615                             error = true;
616                             putResponse(message + ": substitution failed");
617                         }
618                     } catch (PatternSyntaxException pse) {
619                         error = true;
620                         putResponse("Bad regexp: " + pse.getDescription());
621                     }
622                 } else {
623                     error = true;
624                     putResponse("bad substitution sytax, use ^old^new^");
625                 }
626             } else if ((m = bbPattern.matcher(message)).find()) {
627                 message = m.replaceAll(history.getLast(0));
628                 echo = true;
629             } else if (message.startsWith("!")) {
630                 if (message.matches("!\\d+")) {
631                     int which = Integer.parseInt(message.substring(1));
632                     message = history.get(which);
633                 } else if (message.matches("!-\\d+")) {
634                     int which = Integer.parseInt(message.substring(2));
635                     message = history.getLast(which - 1);
636                 } else {
637                     message = history.findLast(message.substring(1));
638                 }
639                 echo = true;
640             }
641     
642             if (error) {
643                 return "";
644             }
645     
646             if (message.length() > 0) {
647                 history.add(message);
648             }
649     
650             if (echo) {
651                 putResponse(message);
652             }
653             return justPush ? "" : message;
654         }

672         public boolean load(String filename) {
673             try {
674                 FileReader fr = new FileReader(filename);
675                 BufferedReader br = new BufferedReader(fr);
676                 String inputLine;
677     
678                 while ((inputLine = br.readLine()) != null) {
679                     String response = CommandInterpreter.this.execute(inputLine);
680                     if (!response.equals("OK")) {
681                         putResponse(response);
682                     }
683                 }
684                 fr.close();
685                 return true;
686             } catch (IOException ioe) {
687                 return false;
688             }
689         }

740         class CommandHistory {
741     
742             private List<String> history = new ArrayList<String>(100);
743     
744     
745             /**
746              * Adds a command to the history
747              *
748              * @param command the command to add
749              */
750             public void add(String command) {
751                 history.add(command);
752             }
753     
754     
755             /**
756              * Gets the most recent element in the history
757              *
758              * @param offset the offset from the most recent command
759              * @return the last command executed
760              */
761             public String getLast(int offset) {
762                 if (history.size() > offset) {
763                     return history.get((history.size() - 1) - offset);
764                 } else {
765                     putResponse("command not found");
766                     return "";
767                 }
768             }
769     
770     
771             /**
772              * Gets the most recent element in the history
773              *
774              * @param which the offset from the most recent command
775              * @return the last command executed
776              */
777             public String get(int which) {
778                 if (history.size() > which) {
779                     return history.get(which);
780                 } else {
781                     putResponse("command not found");
782                     return "";
783                 }
784             }
785     
786     
787             /**
788              * Finds the most recent message that starts with the given string
789              *
790              * @param match the string to match
791              * @return the last command executed that matches match
792              */
793             public String findLast(String match) {
794                 for (int i = history.size() - 1; i >= 0; i--) {
795                     String cmd = get(i);
796                     if (cmd.startsWith(match)) {
797                         return cmd;
798                     }
799                 }
800                 putResponse("command not found");
801                 return "";
802             }
803     
804     
805             /** Dumps the current history */
806             public void dump() {
807                 for (int i = 0; i < history.size(); i++) {
808                     String cmd = get(i);
809                     putResponse(i + " " + cmd);
810                 }
811             }
812         }
