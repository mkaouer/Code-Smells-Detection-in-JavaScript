13     package edu.cmu.sphinx.trainer;
14     
15     import edu.cmu.sphinx.util.props.Configurable;
16     import edu.cmu.sphinx.util.props.S4Double;
17     import edu.cmu.sphinx.util.props.S4Integer;
18     
19     import java.io.IOException;
20     
21     /** Manages inputs and outputs to the other trainer classes. */
22     interface TrainManager extends Configurable {
23     
24         /** Prefix for SphinxProperties in this file. */
25         public final static String PROP_PREFIX =
26                 "edu.cmu.sphinx.trainer.Trainer.";
27     
28         /** The minimum relative improvement of the log likelihood associated with the training data. */
29         @S4Double(defaultValue = 0.2f)
30         public final static String
31                 PROP_MINIMUM_IMPROVEMENT = PROP_PREFIX + "minimumImprovement";
32     
33         /** Default value for minimumImprovement. */
34         public final static float PROP_MINIMUM_IMPROVEMENT_DEFAULT = 0.2f;
35     
36         /** The maximum number of iterations. */
37         @S4Integer(defaultValue = 15)
38         public final static String PROP_MAXIMUM_ITERATION = PROP_PREFIX + "maximumIteration";
39     
40         /** The default value for maximumIteration. */
41         public final static int PROP_MAXIMUM_ITERATION_DEFAULT = 15;
42     
43     
44         /** Starts the TrainManager. */
45         public void start();
46     
47     
48         /** Stops the TrainManager. */
49         public void stop();
50     
51     
52         /** Do the train. */
53         public void train();
54     
55     
56         /** Initialize the TrainManager. */
57         public void initialize();
58     
59     
60         /**
61          * Saves the acoustic models.
62          *
63          * @param context the context of this TrainManager
64          * @throws IOException if an error occurs while loading the data
65          */
66         void saveModels(String context) throws IOException;
67     
68     
69         /**
70          * Copy the model.
71          * <p/>
72          * This method copies to model set, possibly to a new location and new format. This is useful if one wants to
73          * convert from binary to ascii and vice versa, or from a directory structure to a JAR file. If only one model is
74          * used, then name can be null.
75          *
76          * @param context this TrainManager's context
77          * @throws IOException if an error occurs while loading the data
78          */
79         void copyModels(String context) throws IOException;
80     
81     
82         /**
83          * Initializes the acoustic models.
84          *
85          * @param context the context of this TrainManager
86          */
87         void initializeModels(String context) throws IOException;
88     
89     
90         /**
91          * Trains context independent models. If the initialization stage was skipped, it loads models from files,
92          * automatically.
93          *
94          * @param context the context of this train manager.
95          * @throws IOException
96          */
97         void trainContextIndependentModels(String context) throws IOException;
98     }
