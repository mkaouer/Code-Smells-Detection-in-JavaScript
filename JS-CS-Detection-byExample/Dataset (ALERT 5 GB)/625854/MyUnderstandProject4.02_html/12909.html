15     import edu.cmu.sphinx.util.LogMath;

17     
18     import java.io.IOException;
19     import java.util.*;

21     /**
22      * Defines a grammar based upon a list of words in a file. The format of the file is just one word per line. For
23      * example, for an isolated digits grammar the file will simply look like:
24      * <pre>
25      * zero
26      * one
27      * two
28      * three
29      * four
30      * five
31      * six
32      * seven
33      * eight
34      * nine
35      * </pre>
36      * The path to the file is defined by the {@link #PROP_PATH PROP_PATH} property. If the {@link #PROP_LOOP PROP_LOOP}
37      * property is true, the grammar created will be a looping grammar. Using the above digits grammar example, setting
38      * PROP_LOOP to true will make it a connected-digits grammar.
39      * <p/>
40      * All probabilities are maintained in LogMath log base.
41      */
42     public class SimpleWordListGrammar extends Grammar implements Configurable {
43     
44         /** Sphinx property that defines the location of the word list grammar */
45         @S4String(defaultValue = "spelling.gram")
46         public final static String PROP_PATH = "path";
47     
48         /** Sphinx property that if true, indicates that this is a looping grammar */
49         @S4Boolean(defaultValue = true)
50         public final static String PROP_LOOP = "isLooping";
51     
52         /** Sphinx property that defines the logMath component. */
53         @S4Component(type = LogMath.class)
54         public final static String PROP_LOG_MATH = "logMath";
55     
56     
57         // ---------------------
58         // Configurable data
59         // ---------------------
60         private String path;
61         private boolean isLooping;
62         private LogMath logMath;
63     
64     
65         /*
66         * (non-Javadoc)
67         *
68         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
69         */
70         public void newProperties(PropertySheet ps) throws PropertyException {
71             super.newProperties(ps);
72             
73             path = ps.getString(PROP_PATH);
74             isLooping = ps.getBoolean(PROP_LOOP);
75             logMath = (LogMath) ps.getComponent(PROP_LOG_MATH);
76         }
77     
78     
79         /**
80          * Create class from reference text (not implemented).
81          *
82          * @param bogusText dummy variable
83          */
84         protected GrammarNode createGrammar(String bogusText)
85                 throws NoSuchMethodException {
86             throw new NoSuchMethodException("Does not create "
87                     + "grammar with reference text");
88         }
89     
90     
91         /** Creates the grammar. */
92         protected GrammarNode createGrammar() throws IOException {
93             ExtendedStreamTokenizer tok = new ExtendedStreamTokenizer(path, true);
94             GrammarNode initialNode = createGrammarNode("<sil>");
95             GrammarNode branchNode = createGrammarNode(false);
96             GrammarNode finalNode = createGrammarNode("<sil>");
97             finalNode.setFinalNode(true);
98             List wordGrammarNodes = new LinkedList();
99             while (!tok.isEOF()) {
100                 String word;
101                 while ((word = tok.getString()) != null) {
102                     word = word.toLowerCase();
103                     GrammarNode wordNode = createGrammarNode(word);
104                     wordGrammarNodes.add(wordNode);
105                 }
106             }
107             // now connect all the GrammarNodes together
108             initialNode.add(branchNode, LogMath.getLogOne());
109             float branchScore = logMath.linearToLog(
110                     1.0 / wordGrammarNodes.size());
111             for (Object wordGrammarNode : wordGrammarNodes) {
112                 GrammarNode wordNode = (GrammarNode) wordGrammarNode;
113                 branchNode.add(wordNode, branchScore);
114                 wordNode.add(finalNode, LogMath.getLogOne());
115                 if (isLooping) {
116                     wordNode.add(branchNode, LogMath.getLogOne());
117                 }
118             }
119     
120             return initialNode;
121         }
122     }
