13     package edu.cmu.sphinx.demo.helloworld;
14     
15     import edu.cmu.sphinx.frontend.util.Microphone;
16     import edu.cmu.sphinx.recognizer.Recognizer;
17     import edu.cmu.sphinx.result.Result;
18     import edu.cmu.sphinx.util.props.ConfigurationManager;
19     
20     
21     /**
22      * A simple HelloWorld demo showing a simple speech application built using Sphinx-4. This application uses the Sphinx-4
23      * endpointer, which automatically segments incoming audio into utterances and silences.
24      */
25     public class HelloWorld {
26     
27         public static void main(String[] args) {
28             ConfigurationManager cm;
29     
30             if (args.length > 0) {
31                 cm = new ConfigurationManager(args[0]);
32             } else {
33                 cm = new ConfigurationManager(HelloWorld.class.getResource("helloworld.config.xml"));
34             }
35     
36             Recognizer recognizer = (Recognizer) cm.lookup("recognizer");
37             recognizer.allocate();
38     
39             // start the microphone or exit if the programm if this is not possible
40             Microphone microphone = (Microphone) cm.lookup("microphone");
41             if (!microphone.startRecording()) {
42                 System.out.println("Cannot start microphone.");
43                 recognizer.deallocate();
44                 System.exit(1);
45             }
46     
47             System.out.println("Say: (Good morning | Hello) ( Bhiksha | Evandro | Paul | Philip | Rita | Will )");
48     
49             // loop the recognition until the programm exits.
50             while (true) {
51                 System.out.println("Start speaking. Press Ctrl-C to quit.\n");
52     
53                 Result result = recognizer.recognize();
54     
55                 if (result != null) {
56                     String resultText = result.getBestFinalResultNoFiller();
57                     System.out.println("You said: " + resultText + "\n");
58                 } else {
59                     System.out.println("I can't hear what you said.\n");
60                 }
61             }
62         }
63     }
