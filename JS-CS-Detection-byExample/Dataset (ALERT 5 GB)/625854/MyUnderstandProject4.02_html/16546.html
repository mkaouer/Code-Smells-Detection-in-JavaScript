26     /**
27      * This class is used to execute command line at from Java runtime
28      *
29      * Usage of following class ...
30      * 
31      * <P><PRE><CODE>
32      * 		SysCommandExecutor cmdExecutor = SysCommandExecutor.getInstance(); 		
33      * 		int exitStatus = cmdExecutor.runCommand(commandLine);
34      * 
35      * 		String cmdError = cmdExecutor.getCommandError();
36      * 		String cmdOutput = cmdExecutor.getCommandOutput(); 
37      * </CODE></PRE></P> 
38      */
39     public class SysCommandExecutor
40     {	
41     
42     	private String fWorkingDirectory = null;
43     	private List fEnvironmentVarList = null;
44     	
45     	private StringBuffer fCmdOutput = null;
46     	private StringBuffer fCmdError = null;
47     	private AsyncStreamReader fCmdOutputThread = null;
48     	private AsyncStreamReader fCmdErrorThread = null;	
49     	
50             /* this class uses Singleton pattern for the constructor */
51             private SysCommandExecutor() {}
52         
53             private static class SysExecutorHolder {
54                 private static SysCommandExecutor instance = new SysCommandExecutor();
55             }
56           
57             /**
58              * Get reference to the <code>SysCommandExecutor</code> 
59              * 
60              * @return <code>SysCommandExecutor</code>
61              */
62             public static SysCommandExecutor getInstance(){
63                 return SysExecutorHolder.instance;
64             }
65             
66     	public void setWorkingDirectory(String workingDirectory) {
67     		fWorkingDirectory = workingDirectory;
68     	}
69     	
70     	public void setEnvironmentVar(String name, String value)
71     	{
72     		if( fEnvironmentVarList == null )
73     			fEnvironmentVarList = new ArrayList();
74     		
75     		fEnvironmentVarList.add(new EnvironmentVar(name, value));
76     	}
77     	
78             /** get the output of command */
79     	public String getCommandOutput() {		
80     		return fCmdOutput.toString();
81     	}
82     	
83             /** get the error message after command execution */
84     	public String getCommandError() {
85     		return fCmdError.toString();
86     	}
87     	
88             /** 
89              * execute the command
90              *
91              * @param commandLine command
92              * @return Exit status of command
93              */
94     	public int runCommand(String commandLine) throws Exception
95     	{
96     		/* run command */
97     		Process process = runCommandHelper(commandLine);
98     		
99     		/* start output and error read threads */
100     		startOutputAndErrorReadThreads(process.getInputStream(), process.getErrorStream());
101     	    
102     		/* wait for command execution to terminate */
103     		int exitStatus = -1;
104     		try {
105     			exitStatus = process.waitFor();
106     					
107     		} catch (Throwable ex) {
108     			throw new Exception("SysCommandExecutor :" +ex.getMessage());
109     			
110     		} finally {
111     			/* notify output and error read threads to stop reading */
112     			notifyOutputAndErrorReadThreadsToStopReading();
113     		}
114     		
115     		return exitStatus;
116     	}	
117     	
118     	private Process runCommandHelper(String commandLine) throws IOException
119     	{
120     		Process process = null;		
121     		if( fWorkingDirectory == null )
122     			process = Runtime.getRuntime().exec(commandLine, getEnvTokens());
123     		else
124     			process = Runtime.getRuntime().exec(commandLine, getEnvTokens(), new File(fWorkingDirectory));
125     		
126     		return process;
127     	}
128     	
129     	private void startOutputAndErrorReadThreads(InputStream processOut, InputStream processErr)
130     	{
131     		fCmdOutput = new StringBuffer();
132     		fCmdOutputThread = new AsyncStreamReader(processOut, fCmdOutput, "OUTPUT");		
133     		fCmdOutputThread.start();
134     		
135     		fCmdError = new StringBuffer();
136     		fCmdErrorThread = new AsyncStreamReader(processErr, fCmdError, "ERROR");
137     		fCmdErrorThread.start();
138     	}
139     	
140     	private void notifyOutputAndErrorReadThreadsToStopReading()
141     	{
142     		fCmdOutputThread.stopReading();
143     		fCmdErrorThread.stopReading();
144     	}
145     	
146     	private String[] getEnvTokens()
147     	{
148     		if( fEnvironmentVarList == null )
149     			return null;
150     		
151     		String[] envTokenArray = new String[fEnvironmentVarList.size()];
152     		Iterator envVarIter = fEnvironmentVarList.iterator();
153     		int nEnvVarIndex = 0; 
154     		while (envVarIter.hasNext() == true)
155     		{
156     			EnvironmentVar envVar = (EnvironmentVar)(envVarIter.next());
157     			String envVarToken = envVar.fName + "=" + envVar.fValue;
158     			envTokenArray[nEnvVarIndex++] = envVarToken;
159     		}
160     		
161     		return envTokenArray;
162     	}	
163     }

165     class AsyncStreamReader extends Thread
166     {
167     	private StringBuffer fBuffer = null;
168     	private InputStream fInputStream = null;
169     	private String fThreadId = null;
170     	private boolean fStop = false;
171     	
172     	private String fNewLine = null;
173     	
174     	public AsyncStreamReader(InputStream inputStream, StringBuffer buffer, String threadId)
175     	{
176     		fInputStream = inputStream;
177     		fBuffer = buffer;
178     		fThreadId = threadId;
179     
180     		
181     		fNewLine = System.getProperty("line.separator");
182     	}	
183     	
184     	public String getBuffer() {		
185     		return fBuffer.toString();
186     	}
187     	
188     	public void run()
189     	{
190     		try {
191     			readCommandOutput();
192     		} catch (Exception ex) {
193     			//ex.printStackTrace(); //DEBUG
194     		}
195     	}
196     	
197     	private void readCommandOutput() throws IOException
198     	{		
199     		BufferedReader bufOut = new BufferedReader(new InputStreamReader(fInputStream));		
200     		String line = null;
201     		while ( (fStop == false) && ((line = bufOut.readLine()) != null) )
202     		{
203     			fBuffer.append(line + fNewLine);
204     		}		
205     		bufOut.close();
206     		//printToConsole("END OF: " + fThreadId); //DEBUG
207     	}
208     	
209     	public void stopReading() {
210     		fStop = true;
211     	}
212     
213     	
214     	private synchronized void printToConsole(String line) {
215     		System.out.println(line);
216     	}
217     }
