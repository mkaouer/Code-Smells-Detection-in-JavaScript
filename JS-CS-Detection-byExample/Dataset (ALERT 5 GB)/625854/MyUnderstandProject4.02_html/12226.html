32         /**
33          * A Cluster is a set of Nodes together with their earliest start time and latest end time. A SausageMaker builds up
34          * a sequence of such clusters that then gets turned into a Sausage.
35          *
36          * @see Node
37          * @see Sausage
38          * @see SausageMaker
39          */
40         class Cluster {
41     
42             public int startTime;
43             public int endTime;
44             private List elements = new LinkedList();
45     
46     
47             public Cluster(Node n) {
48                 startTime = n.getBeginTime();
49                 endTime = n.getEndTime();
50                 elements.add(n);
51             }
52     
53     
54             public Cluster(int start, int end) {
55                 startTime = start;
56                 endTime = end;
57             }
58     
59     
60             public void add(Node n) {
61                 if (n.getBeginTime() < startTime) {
62                     startTime = n.getBeginTime();
63                 }
64                 if (n.getEndTime() > endTime) {
65                     endTime = n.getEndTime();
66                 }
67                 elements.add(n);
68             }
69     
70     
71             public void add(Cluster c) {
72                 if (c.startTime < startTime) {
73                     startTime = c.startTime;
74                 }
75                 if (c.endTime > endTime) {
76                     endTime = c.endTime;
77                 }
78                 elements.addAll(c.getElements());
79             }
80     
81     
82             public Iterator iterator() {
83                 return elements.iterator();
84             }
85     
86     
87             public String toString() {
88                 StringBuffer sb = new StringBuffer();
89                 sb.append("s: ").append(startTime).append(" e: ").append(endTime).append("[");
90                 Iterator i = elements.iterator();
91                 while (i.hasNext()) {
92                     sb.append(i.next());
93                     if (i.hasNext()) {
94                         sb.append(",");
95                     }
96                 }
97                 sb.append("]");
98                 return sb.toString();
99             }
100     
101     
102             /** @return Returns the elements. */
103             public List getElements() {
104                 return elements;
105             }
106     
107     
108             /** @param elements The elements to set. */
109             public void setElements(List elements) {
110                 this.elements = elements;
111             }
112         }

114         class ClusterComparator implements Comparator {
115     
116             /**
117              * Compares to clusters according to their topological relationship. Relies on strong assumptions about the
118              * possible constituents of clusters which will only be valid during the sausage creation process.
119              *
120              * @param o1 the first cluster (must be a List)
121              * @param o2 the second cluster (must be a List)
122              */
123             public int compare(Object o1, Object o2) {
124                 Cluster cluster1 = (Cluster) o1;
125                 Cluster cluster2 = (Cluster) o2;
126                 Iterator i = cluster1.iterator();
127                 while (i.hasNext()) {
128                     Node n1 = (Node) i.next();
129                     Iterator i2 = cluster2.iterator();
130                     while (i2.hasNext()) {
131                         Node n2 = (Node) i2.next();
132                         if (n1.isAncestorOf(n2)) {
133                             return -1;
134                         } else if (n2.isAncestorOf(n1)) {
135                             return 1;
136                         }
137                     }
138                 }
139                 return 0;
140             }
141         }

209         /**
210          * Calculate the sum of posteriors in this cluster.
211          *
212          * @param cluster the cluster to sum over
213          * @return the probability sum
214          */
215         protected double clusterProbability(List cluster) {
216             float p = LogMath.getLogZero();
217             Iterator i = cluster.iterator();
218             while (i.hasNext()) {
219                 p = lattice.getLogMath().addAsLinear(p, (float) ((Node) i.next()).getPosterior());
220             }
221             return p;
222         }

236         /**
237          * Form a subcluster by extracting all nodes corresponding to a given word.
238          *
239          * @param cluster the parent cluster
240          * @param word    the word to cluster by
241          * @return the subcluster.
242          */
243         protected List makeWordSubCluster(List cluster, String word) {
244             Vector sub = new Vector();
245             Iterator i = cluster.iterator();
246             while (i.hasNext()) {
247                 Node n = (Node) i.next();
248                 if (n.getWord().getSpelling().equals(word)) {
249                     sub.add(n);
250                 }
251             }
252             return sub;
253         }

263         protected Cluster makeWordSubCluster(Cluster cluster, String word) {
264             List l = makeWordSubCluster(cluster.elements, word);
265             Cluster c = new Cluster(cluster.startTime, cluster.endTime);

286         /**
287          * Turn a list of lattice node clusters into a Sausage object.
288          *
289          * @param clusters the list of node clusters in topologically correct order
290          * @return the Sausage corresponding to the cluster list
291          */
292         protected Sausage sausageFromClusters(List clusters) {
293             Sausage sausage = new Sausage(clusters.size());
294             ListIterator c1 = clusters.listIterator();
295             while (c1.hasNext()) {
296                 HashSet seenWords = new HashSet();
297                 int index = c1.nextIndex();
298                 Cluster cluster = ((Cluster) c1.next());
299                 Iterator c2 = cluster.iterator();
300                 while (c2.hasNext()) {
301                     Node node = (Node) c2.next();
302                     Word word = node.getWord();
303                     if (seenWords.contains(word.getSpelling())) {
304                         continue;
305                     }
306                     seenWords.add(word.getSpelling());
307                     SimpleWordResult swr = new SimpleWordResult
308                             (node,
309                                     wordSubClusterProbability(cluster, word.getSpelling()),
310                                     lattice.getLogMath());
311                     sausage.addWordHypothesis(index, swr);
312                 }
313             }
314             sausage.fillInBlanks(lattice.getLogMath());
315             return sausage;
316         }
