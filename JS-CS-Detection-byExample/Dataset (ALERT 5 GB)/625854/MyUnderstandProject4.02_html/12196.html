16     import edu.cmu.sphinx.frontend.*;
17     import edu.cmu.sphinx.frontend.util.StreamDataSource;
18     import edu.cmu.sphinx.util.props.*;
19     
20     import java.io.*;
21     import java.net.Socket;
22     
23     /**
24      * A FrontEnd that runs on client applications. The main interface between the application and this ClientFrontEnd is
25      * the <b>decode(InputStream inputStream, String streamName)</b> method, which returns the decoded result as a String.
26      * <p/>
27      * <p>The method <b>connect()</b> should be called before <b>decode()</b>, and the method <b>close()</b> should be
28      * called after all decoding is done. Therefore, the correct sequence of calls is: <code> connect();
29      * decode(inputstream1, name1); ... decode(inputstreamN, nameN); close(); </code>
30      */
31     public class ClientFrontEndImpl implements Configurable{
32     
33     
34         private BufferedReader reader;
35         private DataOutputStream dataWriter;
36     
37         @S4String(defaultValue = "localhost")
38         public static final String SERVER_ADRESS = "serverAdress";
39         private String serverAddress;
40     
41     
42         @S4Integer(defaultValue = 52703)
43         public static final String SERVER_PORT = "serverPort";
44         private int serverPort;
45     
46         private Socket socket;
47     
48         private StreamDataSource streamAudioSource;
49         private FrontEnd frontend;
50     
51         private boolean inUtterance = false;
52     
53     
54     
55         public void newProperties(PropertySheet ps) throws PropertyException {
56               streamAudioSource = new StreamDataSource();
57             streamAudioSource.initialize();
58     //        streamAudioSource.initialize("StreamDataSource", null, props, null);
59     
60     //        frontend = FrontEndFactory.getFrontEnd(props, "client");
61             frontend.setDataSource(streamAudioSource);
62     
63             serverAddress = ps.getString(SERVER_ADRESS);
64             serverPort = ps.getInt(SERVER_PORT);
65         }
66     
67     
68         /** Connects this ClientFrontEndImpl to the back-end server. */
69         public void connect() {
70             try {
71                 socket = connectToBackEnd();
72             } catch (IOException ioe) {
73                 ioe.printStackTrace();
74             }
75         }
76     
77     
78         /** Closes the connection to the back-end server. */
79         public void close() {
80             try {
81                 closeConnection(socket);
82             } catch (IOException ioe) {
83                 ioe.printStackTrace();
84             }
85         }
86     
87     
88         /**
89          * Decodes the data in the given InputStream.
90          *
91          * @param is         the InputStream to decode
92          * @param streamName the name of the InputStream
93          * @return the result string
94          */
95         public String decode(InputStream is, String streamName)
96                 throws DataProcessingException, IOException {
97             sendRecognition();
98             streamAudioSource.setInputStream(is, streamName);
99             sendData();
100             String result = readResult();
101             return result;
102         }
103     
104     
105         /** Connects this client to the server. */
106         private Socket connectToBackEnd() throws IOException {
107             Socket socket = new Socket(serverAddress, serverPort);
108             reader = new BufferedReader
109                     (new InputStreamReader(socket.getInputStream()));
110             dataWriter = new DataOutputStream
111                     (new BufferedOutputStream(socket.getOutputStream()));
112             return socket;
113         }
114     
115     
116         /** Sends a single int "1" to the back-end server to signal the cepstra will be sent for recognition. */
117         private void sendRecognition() throws IOException {
118             dataWriter.writeInt(1);
119             dataWriter.flush();
120         }
121     
122     
123         /** Closes the connection to the back-end server. */
124         private void closeConnection(Socket socket) throws IOException {
125             reader.close();
126             dataWriter.close();
127             socket.close();
128         }
129     
130     
131         /** Sends all the available cepstra to the back-end server. */
132         private void sendData() throws DataProcessingException, IOException {
133             Data cepstrum = null;
134             do {
135                 cepstrum = frontend.getData();
136                 if (cepstrum != null) {
137                     if (!inUtterance) {
138                         if (cepstrum instanceof DataStartSignal) {
139                             inUtterance = true;
140                             dataWriter.writeDouble(Double.MAX_VALUE);
141                             dataWriter.flush();
142                         } else {
143                             throw new IllegalStateException
144                                     ("No DataStartSignal read");
145                         }
146                     } else {
147                         if (cepstrum instanceof DoubleData) {
148                             // send the cepstrum data
149                             double[] data = ((DoubleData) cepstrum).getValues();
150                             for (int i = 0; i < data.length; i++) {
151                                 dataWriter.writeDouble(data[i]);
152                             }
153                         } else if (cepstrum instanceof DataEndSignal) {
154                             // send a DataEndSignal
155                             dataWriter.writeDouble(Double.MIN_VALUE);
156                             inUtterance = false;
157                         } else if (cepstrum instanceof DataStartSignal) {
158                             throw new IllegalStateException
159                                     ("Too many DataStartSignals.");
160                         }
161                         dataWriter.flush();
162                     }
163                 }
164             } while (cepstrum != null);
165         }
166     
167     
168         /**
169          * Reads the result string from the back-end server.
170          *
171          * @return the result string
172          */
173         private String readResult() throws IOException {
174             return reader.readLine();
175         }
176     }
