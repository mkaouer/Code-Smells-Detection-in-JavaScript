70         /** 
71          * do string manipulation to get the property comment
72          * @param classname Fully qualified class name that holds the property
73          * @param class_path Absolute path to the root of .class files;
74          *              must end in '\' or '/'
75          * @param source_path Absolute path to the source code root directory; 
76          *              must end in '\' or '/'
77          * @param prop_name Name of property to be searched
78          * @return Javadoc comment for the property that belongs to classname
79          */
80         public static String getJavadocComment(String classname, String class_path,
81                 String source_path, String prop_name)
82         {
83             String allcomment;
84             
85             if(source_path == null || source_path.trim().equals("")){// no path to source code
86                 return null;
87             }
88             
89             if ( _last_class == null || ! _last_class.trim().equals(classname.trim())){
90                 allcomment = extractJavadocComment(classname,class_path,source_path);
91             }
92             else allcomment = _last_comment;
93             //  System.out.println("comment : " + allcomment);        
94             
95             // do processing for the comment
96             // one property comment block is defined by a '==='
97             String[] comments = allcomment.split("={3}"); 
98             
99              // index 0 is javadoc's result comment
100             for (int i=1; i<comments.length;i++){
101                 //  System.out.println("comment" + i + " : " + comments[i]);
102                 String[] oneComment = comments[i].trim().split("={2}");
103                 if(oneComment[0].trim().equals(prop_name) && oneComment.length > 1){
104                     // property found, remove newlines in the comment string         
105                     oneComment[1] = oneComment[1].replaceAll("(\n|\r)+"," ");
106                     return oneComment[1]; 
107                 }
108             }
109             
110             return null; // does not find the comment for the class property
111         }
