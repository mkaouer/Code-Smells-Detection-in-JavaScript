12     package edu.cmu.sphinx.result;
13     
14     import java.io.FileWriter;
15     import java.io.IOException;
16     import java.io.PrintWriter;
17     import java.util.StringTokenizer;
18     
19     /** Edges are part of Lattices.  They connect Nodes, and contain the score associated with that sequence. */
20     public class Edge {
21     
22         protected double acousticScore;
23         protected double lmScore;
24         protected Node fromNode;
25         protected Node toNode;
26     
27     
28         /**
29          * Create an Edge from fromNode to toNode with acoustic and Language Model scores.
30          *
31          * @param fromNode
32          * @param toNode
33          * @param acousticScore
34          * @param lmScore
35          */
36         protected Edge(Node fromNode, Node toNode,
37                        double acousticScore, double lmScore) {
38             this.acousticScore = acousticScore;
39             this.lmScore = lmScore;
40             this.fromNode = fromNode;
41             this.toNode = toNode;
42         }
43     
44     
45         public String toString() {
46             return "Edge(" + fromNode + "-->" + toNode + "[" + acousticScore
47                     + "," + lmScore + "])";
48         }
49     
50     
51         /**
52          * Internal routine used when creating a Lattice from a .LAT file
53          *
54          * @param lattice
55          * @param tokens
56          */
57         static void load(Lattice lattice, StringTokenizer tokens) {
58     
59             String from = tokens.nextToken();
60             String to = tokens.nextToken();
61             int score = Integer.parseInt(tokens.nextToken());
62     
63             Node fromNode = lattice.getNode(from);
64             if (fromNode == null) {
65                 throw new Error("Edge fromNode \"" + from + "\" does not exist");
66             }
67     
68             Node toNode = lattice.getNode(to);
69             if (fromNode == null) {
70                 throw new Error("Edge toNode \"" + to + "\" does not exist");
71             }
72     
73             lattice.addEdge(fromNode, toNode, score, 0.0);
74         }
75     
76     
77         /**
78          * Internal routine used when dumping a Lattice as a .LAT file
79          *
80          * @param f
81          * @throws IOException
82          */
83         void dump(PrintWriter f) throws IOException {
84             f.println("edge: " + fromNode.getId() + " " + toNode.getId() + " "
85                     + acousticScore + " " + lmScore);
86         }
87     
88     
89         /**
90          * Internal routine used when dumping a Lattice as an AiSee file
91          *
92          * @param f
93          * @throws IOException
94          */
95         void dumpAISee(FileWriter f) throws IOException {
96             f.write("edge: { sourcename: \"" + fromNode.getId()
97                     + "\" targetname: \"" + toNode.getId()
98                     + "\" label: \"" + acousticScore + "," + lmScore + "\" }\n");
99         }
100     
101     
102         /**
103          * Get the acoustic score associated with an Edge. This is the acoustic score of the word that this edge is
104          * transitioning to, that is, the word represented by the node returned by the getToNode() method.
105          *
106          * @return the acoustic score of the word this edge is transitioning to
107          */
108         public double getAcousticScore() {
109             return acousticScore;
110         }
111     
112     
113         /**
114          * Get the language model score associated with an Edge
115          *
116          * @return the score
117          */
118         public double getLMScore() {
119             return lmScore;
120         }
121     
122     
123         /**
124          * Get the "from" Node associated with an Edge
125          *
126          * @return the Node
127          */
128         public Node getFromNode() {
129             return fromNode;
130         }
131     
132     
133         /**
134          * Get the "to" Node associated with an Edge
135          *
136          * @return the Node
137          */
138         public Node getToNode() {
139             return toNode;
140         }
141     
142     
143         /**
144          * Sets the acoustic score
145          *
146          * @param v the acoustic score.
147          */
148         public void setAcousticScore(double v) {
149             acousticScore = v;
150         }
151     
152     
153         /**
154          * Sets the language model score
155          *
156          * @param v the lm score.
157          */
158         public void setLMScore(double v) {
159             lmScore = v;
160         }
161     
162     
163         /**
164          * Returns true if the given edge is equivalent to this edge. Two edges are equivalent only if they have their
165          * 'fromNode' and 'toNode' are equivalent, and that their acoustic and language scores are the same.
166          *
167          * @param other the Edge to compare this Edge against
168          * @return true if the Edges are equivalent; false otherwise
169          */
170         public boolean isEquivalent(Edge other) {
171             /*
172              * TODO: Figure out why there would be minute differences
173              * in the acousticScore. Therefore, the equality of the acoustic
174              * score is judge based on whether the difference is bigger than 1.
175              */
176             double diff = Math.abs(acousticScore) * 0.00001;
177             return ((Math.abs(acousticScore - other.getAcousticScore()) <= diff &&
178                     lmScore == other.getLMScore()) &&
179                     (fromNode.isEquivalent(other.getFromNode()) &&
180                             toNode.isEquivalent(other.getToNode())));
181         }
182     }
