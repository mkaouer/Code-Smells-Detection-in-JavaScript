14     package edu.cmu.sphinx.tools.gui;
15     
16     import java.util.HashMap;
17     import java.util.List;
18     import java.util.Map;
19     import java.util.Iterator;
20     
21     /**
22      * Holds the raw property data just as it has come in from the properties
23      * file.
24      */
25     public class RawPropertyData {
26         private String name;
27         private String className;
28         private Map properties;
29         /**
30          * Creates a raw property data item
31          * 
32          * @param name
33          *            the name of the item
34          * @param className
35          *            the class name of the item
36          */
37        public RawPropertyData(String name, String className) {
38             this.name = name;
39             this.className = className;
40             properties = new HashMap();
41         }
42         /**
43          * Adds a new property
44          * 
45          * @param propName
46          *            the name of the property
47          * @param propValue
48          *            the value of the property
49          */
50         public void add(String propName, String propValue) {
51             properties.put(propName, propValue);
52         }
53         /**
54          * Adds a new property
55          * 
56          * @param propName
57          *            the name of the property
58          * @param propValue
59          *            the value of the property
60          */
61         public void add(String propName, List propValue) {
62             properties.put(propName, propValue);
63         }
64         /**
65          * Change property value
66          * 
67          * @param propName
68          *            the name of the property
69          * @param propValue
70          *            the new value of the property
71          */
72         public void change(String propName, String propValue) {
73             properties.remove(propName);
74             properties.put(propName, propValue);
75         }
76         
77         /**
78          * Remove property 
79          * 
80          * @param propName
81          *            the name of the property
82          */
83         public void remove(String propName) {
84             properties.remove(propName);       
85         }
86         
87         /**
88          * Change property value
89          * 
90          * @param propName
91          *            the name of the property
92          * @param propValue
93          *            the new value of the property
94          */
95         public void change(String propName, List propValue) {
96             properties.remove(propName);
97             properties.put(propName, propValue);
98         }
99         /**
100          * @return Returns the className.
101          */
102         public String getClassName() {
103             return className;
104         }
105         /**
106          * @return Returns the configuration name.
107          */
108         public String getName() {
109             return name;
110         }
111         /**
112          * @return Returns the properties.
113          */
114        public Map getProperties() {
115             return properties;
116         }
117         
118        
119        /**
120         * Provide information stored inside this Object,
121         * used mainly for debugging / testing
122         *
123         * @return Description of object
124         */
125        public String toString(){
126            String output = new String("name : " + name);
127            for(Iterator it=properties.entrySet().iterator();it.hasNext();){            
128                 Map.Entry entry = (Map.Entry) it.next();            
129                 String groupname = (String)entry.getKey();            
130                 if (entry.getValue() != null ){
131                     if (entry.getValue() instanceof String) {
132                         output = output.concat("value string : "+ (String)entry.getValue());
133                     }
134                     else    { // is a list
135                         output = output.concat(((List)entry.getValue()).toString());
136                     }
137                 }
138            }
139             return output;
140        }
141        
142         /**
143          * Determines if the map already contains an entry
144          * for this property
145          *
146          * @param propName the property of interest
147          * @return true if the map already contains this property
148          */
149          
150       public boolean contains(String propName) {
151             return properties.get(propName) != null;
152         }
153     }
