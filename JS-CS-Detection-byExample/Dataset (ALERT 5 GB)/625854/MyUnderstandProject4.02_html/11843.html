26     /**
27      * Slices up a Data object into a number of overlapping windows (usually refered to as "frames" in the speech world). In
28      * order to minimize the signal discontinuities at the boundaries of each frame, we multiply each frame with a raised
29      * cosine windowing function. Moreover, the system uses overlapping windows to capture information that may occur at the
30      * window boundaries. These events would not be well represented if the windows were juxtaposed. <p> The number of
31      * resulting windows depends on the {@link #PROP_WINDOW_SIZE_MS window size} and the {@link #PROP_WINDOW_SHIFT_MS window
32      * shift} (commonly known as frame shift in speech world). Figure 1 shows the relationship between the original data
33      * stream, the window size, the window shift, and the windows returned. <p> <img src="doc-files/framing.jpg">
34      * <br><b>Figure 1: Relationship between original data, window size, window shift, and the windows returned.</b> <p> The
35      * raised cosine windowing function will be applied to each such window. Since the {@link #getData()} method returns a
36      * window, and multiple windows are created for each Data object, this is a 1-to-many processor. Also note that the
37      * returned windows should have the same number of data points as the windowing function. <p> The applied windowing
38      * function, <i>W(n)</i>, of length <i>N</i> (the window size), is given by the following formula:
39      * <pre>
40      * W(n) = (1-a) - (a * cos((2 * Math.PI * n)/(N - 1)))
41      * </pre>
42      * where <b>a</b> is commonly known as the "alpha" value. This variable can be set by the user using the property
43      * defined by {@link #PROP_ALPHA}. Please follow the links to the see the constant field values. Some values of alpha
44      * receive special names, since they are used so often. A value of 0.46 for the alpha results in a window named Hamming
45      * window. A value of 0.5 results in the Hanning window. And a value of 0 results in the Rectangular window. The default
46      * for this system is the Hamming window, with alpha 0.46 !). Figure 2 below shows the Hamming window function (a =
47      * 0.46), using our default window size of 25.625 ms and assuming a sample rate of 16kHz, thus yielding 410 samples per
48      * window. <p> <img src="doc-files/hamming-window.gif"> <br><b>Figure 2: The Hamming window function.</b>
49      *
50      * @see Data
51      */
52     public class RaisedCosineWindower extends BaseDataProcessor {

103         /**
104          * Creates the Hamming Window.
105          *
106          * @param sampleRate
107          */
108         private void createWindow(int sampleRate) {
109             if (cosineWindow != null && sampleRate == this.sampleRate) {
110                 return;
111             }
112     
113             this.sampleRate = sampleRate;
114     
115             int windowSize = DataUtil.getSamplesPerWindow(sampleRate, windowSizeInMs);
116             cosineWindow = new double[windowSize];
117     
118             windowShift = DataUtil.getSamplesPerShift(sampleRate, windowShiftInMs);
119     
120             if (cosineWindow.length > 1) {
121                 double oneMinusAlpha = (1 - alpha);
122                 for (int i = 0; i < cosineWindow.length; i++) {
123                     cosineWindow[i] = oneMinusAlpha -
124                             alpha * Math.cos(2 * Math.PI * i / ((double) cosineWindow.length - 1.0));
125                 }
126             }
127     
128             overflowBuffer = new DoubleBuffer(windowSize);
129         }

132         /**
133          * Returns the next Data object, which is usually a window of the input Data, with the windowing function applied to
134          * it.
135          *
136          * @return the next available Data object, returns null if no Data object is available
137          * @throws DataProcessingException if a data processing error occurred
138          * @see Data
139          */
140         public Data getData() throws DataProcessingException {
141     
142             if (outputQueue.size() == 0) {
143                 Data input = getPredecessor().getData();
144     
145                 getTimer().start();
146     
147                 if (input != null) {
148                     if (input instanceof DoubleData) {
149                         DoubleData data = (DoubleData) input;
150                         if (currentFirstSampleNumber == -1) {
151                             currentFirstSampleNumber = data.getFirstSampleNumber();
152                         }
153     
154                         // should not be necesssary if all DataProcessor would forward Signals. Unfortunately this
155                         // is currently not the case.
156                         createWindow(data.getSampleRate());
157     
158                         // process the Data, and output the windows
159                         process(data);
160                     } else {
161                         if (input instanceof DataStartSignal) {
162                             DataStartSignal startSignal = (DataStartSignal) input;
163                             createWindow(startSignal.getSampleRate());
164     
165                             // attach the frame-length and the shift-length to the start-signal to allow
166                             Map<String, Object> props = startSignal.getProps();
167                             props.put(WINDOW_SHIFT_SAMPLES, windowShift);
168                             props.put(WINDOW_SIZE_SAMPLES, cosineWindow.length);
169     
170                             // reset the current first sample number
171                             currentFirstSampleNumber = -1;
172                         } else if (input instanceof DataEndSignal) {
173                             // end of utterance handling
174                             processUtteranceEnd();
175                         }
176                         outputQueue.add(input);
177                     }
178                 }
179     
180                 getTimer().stop();
181             }
182     
183             if (outputQueue.size() > 0) {
184                 Data output = outputQueue.remove(0);
185                 if (output instanceof DoubleData) {
186                     assert ((DoubleData) output).getValues().length ==
187                             cosineWindow.length;
188                 }
189                 return output;
190             } else {
191                 return null;
192             }
193         }

196         /**
197          * Applies the Windowing to the given Data. The resulting windows are cached in the outputQueue.
198          *
199          * @param input the input Data object
200          * @throws DataProcessingException if a data processing error occurs
201          */
202         private void process(DoubleData input) throws DataProcessingException {
203     
204             currentCollectTime = input.getCollectTime();
205     
206             double[] in = input.getValues();
207             int length = overflowBuffer.getOccupancy() + in.length;
208     
209             List<DoubleData> dataList = new LinkedList<DoubleData>();
210             dataList.add(input);
211     
212             Data utteranceEnd = null;
213     
214             // read in more Data if we have under one window's length of data
215             while (length < cosineWindow.length) {
216                 Data next = getPredecessor().getData();
217                 if (next instanceof DoubleData) {
218                     dataList.add((DoubleData) next);
219                     length += ((DoubleData) next).getValues().length;
220                 } else {
221                     if (next instanceof DataEndSignal) {
222                         utteranceEnd = next;
223                         break;
224                     }
225     
226                     outputQueue.add(next);
227                 }
228             }
229     
230             double[] allSamples = in;
231     
232             // prepend overflow samples
233             if (length != in.length) {
234     
235                 allSamples = new double[length];
236                 int start = 0;
237     
238                 // copy overflow samples to allSamples buffer
239                 System.arraycopy(overflowBuffer.getBuffer(), 0,
240                         allSamples, start, overflowBuffer.getOccupancy());
241                 start = overflowBuffer.getOccupancy();
242     
243                 // copy input samples to allSamples buffer
244                 for (DoubleData aDataList : dataList) {
245                     double[] samples = aDataList.getValues();
246                     System.arraycopy(samples, 0, allSamples, start, samples.length);
247                     start += samples.length;
248                 }
249             }
250     
251             // apply Hamming window
252             int residual = applyRaisedCosineWindow(allSamples, length);
253     
254             // save elements that also belong to the next window
255             overflowBuffer.reset();
256             if (length - residual > 0) {
257                 overflowBuffer.append(allSamples, residual, length - residual);
258             }
259             if (utteranceEnd != null) {
260                 // end of utterance handling
261                 processUtteranceEnd();
262                 outputQueue.add(utteranceEnd);
263             }
264         }

267         /**
268          * What happens when an DataEndSignal is received. Basically pads up to a window of the overflow buffer with zeros,
269          * and then apply the Hamming window to it.
270          */
271         private void processUtteranceEnd() {
272             overflowBuffer.padWindow(cosineWindow.length);
273             applyRaisedCosineWindow
274                     (overflowBuffer.getBuffer(), cosineWindow.length);
275             overflowBuffer.reset();
276         }

279         /**
280          * Applies the Hamming window to the given double array. The windows are added to the output queue. Returns the
281          * index of the first array element of next window that is not produced because of insufficient data.
282          *
283          * @param in     the audio data to apply window and the Hamming window
284          * @param length the number of elements in the array to apply the RaisedCosineWindow
285          * @return the index of the first array element of the next window
286          */
287         private int applyRaisedCosineWindow(double[] in, int length) {
288     
289             int windowCount;
290     
291             // if no windows can be created but there is some data,
292             // pad it with zeros
293             if (length < cosineWindow.length) {
294                 double[] padded = new double[cosineWindow.length];
295                 Arrays.fill(padded, 0);
296                 System.arraycopy(in, 0, padded, 0, length);
297                 in = padded;
298                 windowCount = 1;
299             } else {
300                 windowCount = getWindowCount(length, cosineWindow.length, windowShift);
301             }
302     
303             // create all the windows at once, not individually, saves time
304             double[][] windows = new double[windowCount][cosineWindow.length];
305     
306             int windowStart = 0;
307     
308             for (int i = 0; i < windowCount; windowStart += windowShift, i++) {
309     
310                 double[] myWindow = windows[i];
311     
312                 // apply the Hamming Window function to the window of data
313                 for (int w = 0, s = windowStart; w < myWindow.length; s++, w++) {
314                     myWindow[w] = in[s] * cosineWindow[w];
315                 }
316     
317                 // add the frame to the output queue
318                 outputQueue.add(new DoubleData
319                         (myWindow, sampleRate,
320                                 currentCollectTime, currentFirstSampleNumber));
321                 currentFirstSampleNumber += windowShift;
322             }
323     
324             return windowStart;
325         }

328         /**
329          * Returns the number of windows in the given array, given the windowSize and windowShift.
330          *
331          * @param arraySize   the size of the array
332          * @param windowSize  the window size
333          * @param windowShift the window shift
334          * @return the number of windows
335          */
336         private static int getWindowCount(int arraySize, int windowSize,
337                                           int windowShift) {
338             if (arraySize < windowSize) {
339                 return 0;
340             } else {
341                 int windowCount = 1;
342                 for (int windowEnd = windowSize;
343                      windowEnd + windowShift <= arraySize;
344                      windowEnd += windowShift) {
345                     windowCount++;
346                 }
347                 return windowCount;
348             }
349         }

369         /**
370          * Rounds a given sample-number to the number of samples will be processed by this instance including the padding
371          * samples at the end..
372          */
373         public long roundToFrames(long samples) {
374             int windowSize = DataUtil.getSamplesPerWindow(sampleRate, windowSizeInMs);
375             int windowShift = DataUtil.getSamplesPerShift(sampleRate, windowShiftInMs);
376     
377             long mxNumShifts = samples / windowShift;
378     
379             for (int i = (int) mxNumShifts; ; i--) {
380                 long remainingSamples = samples - windowShift * i;
381     
382                 if (remainingSamples > windowSize)
383                     return windowShift * (i + 1) + windowSize;
384             }
385         }

389     class DoubleBuffer {
390     
391         private double[] buffer;
392         private int occupancy;
393     
394     
395         /** Constructs a DoubleBuffer of the given size. */
396         DoubleBuffer(int size) {
397             buffer = new double[size];
398             occupancy = 0;
399         }
400     
401     
402         /**
403          * Returns the number of elements in this DoubleBuffer.
404          *
405          * @return the number of elements in this DoubleBuffer.
406          */
407         public int getOccupancy() {
408             return occupancy;
409         }
410     
411     
412         /**
413          * Returns the underlying double array used to store the data.
414          *
415          * @return the underlying double array
416          */
417         public double[] getBuffer() {
418             return buffer;
419         }
420     
421     
422         /**
423          * Appends all the elements in the given array to this DoubleBuffer.
424          *
425          * @param src the array to copy from
426          * @return the resulting number of elements in this DoubleBuffer.
427          */
428         public int appendAll(double[] src) {
429             return append(src, 0, src.length);
430         }
431     
432     
433         /**
434          * Appends the specified elements in the given array to this DoubleBuffer.
435          *
436          * @param src    the array to copy from
437          * @param srcPos where in the source array to start from
438          * @param length the number of elements to copy
439          * @return the resulting number of elements in this DoubleBuffer
440          */
441         public int append(double[] src, int srcPos, int length) {
442             if (occupancy + length > buffer.length) {
443                 length = buffer.length - occupancy;
444                 throw new Error("RaisedCosineWindower: " +
445                         "overflow-buffer: attempting to fill " +
446                         "buffer beyond its capacity.");
447             }
448             System.arraycopy(src, srcPos, buffer, occupancy, length);
449             occupancy += length;
450             return occupancy;
451         }
452     
453     
454         /**
455          * If there are less than windowSize elements in this DoubleBuffer, pad the up to windowSize elements with zero.
456          *
457          * @param windowSize the window size
458          */
459         public void padWindow(int windowSize) {
460             if (occupancy < windowSize) {
461                 Arrays.fill(buffer, occupancy, windowSize, 0);
462             }
463         }
464     
465     
466         /** Sets the number of elements in this DoubleBuffer to zero, without actually remove the elements. */
467         public void reset() {
468             occupancy = 0;
469         }
470     }
