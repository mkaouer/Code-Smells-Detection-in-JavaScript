27      * <p/>
28      * Runs a NIST corpus as used by the GALE project.  The inputs are a CTL file, and a REF file.  The output is a CTM
29      * file.
30      * <p/>
31      * A CTL file contains a list of utterances to decode. The format is
32      * <p/>
33      * <utterance file> <start offset> <end offset> <utterance name>
34      * <p/>
35      * The <utterance file> is a base to which the property "dataDirectory" is prepended, and ".raw" is appended.  The
36      * utterance file should be raw PCM that agrees with the "bitsPerSample", "channelCount", "samplesPerSecond", and
37      * "framesPerSecond" properties.
38      * <p/>
39      * The <start offset> and <end offset> are specified in frames, where
40      * <p/>
41      * bytesPerFrame = (bitsPerSample/8)*channelCount*samplesPerSecond/framesPerSecond
42      * <p/>
43      * The <utterance name> should be a unique string.  For example "<utterance file>_<start offset>_<end offset>".
44      * <p/>
45      * A REF file contains the correct transcripts of the utterances specified in the CTL file.  Each line should be of the
46      * form
47      * <p/>
48      * <ASCII transcript> (<utterance name>)
49      * <p/>
50      * The output is a "processed" CTM file.  It is used by the NIST tools to compute the performance on the copus.  The
51      * format is not documented because it is currently a hack to get the Dry Run going.  We need to think more about it. If
52      * you want to use this tool talk to Peter Wolf, or Arthur Chan.
53      */

150             CTLUtterance(String ctl, String ref) throws CTLException {
151                 /*
152                     example line:
153                     20040422_150000_NTDTV.80Hz-6400Hz 64155 65103 20040422_150000_NTDTV_64155-65103_spk8
154                 */
155                 this.ref = ref;
156                 String[] fields = ctl.split(" ");
157                 if (fields.length != 4) throw new CTLException("CTL Syntax Error: " + ctl);
158                 startOffset = Integer.parseInt(fields[1]);
159                 endOffset = Integer.parseInt(fields[2]);
160                 name = fields[3];
161                 data = new byte[(endOffset - startOffset) * bytesPerFrame];
162                 int i = fields[0].indexOf('.');
163                 file = fields[0];
164                 if (i >= 0) {
165                     file = file.substring(0, i);
166                 }
167                 file = dataDir + "/" + file + ".raw";
168                 try {
169                     InputStream dataStream = new FileInputStream(file);
170                     dataStream.skip(startOffset * bytesPerFrame);
171                     if (dataStream.read(data) != data.length) {
172                         throw new CTLException("Unable to read " + data.length + " bytes of utterance " + name);
173                     }
174                 }
175                 catch (IOException e) {
176                     throw new CTLException("Unable to read utterance " + name + ": " + e.getMessage());
177                 }
178             }

252         protected void setInputStream(CTLUtterance utt) throws IOException {
253             for (DataProcessor dataSource : inputDataProcessors) {
254                 if (dataSource instanceof StreamDataSource) {
255                     ((StreamDataSource) dataSource).setInputStream(utt.getInputStream(), utt.getName());
256                 } else if (dataSource instanceof StreamCepstrumSource) {
257                     boolean isBigEndian = Utilities
258                             .isCepstraFileBigEndian(utt.getName());
259                     StreamCepstrumSource cepstrumSource =
260                             (StreamCepstrumSource) dataSource;
261                     cepstrumSource.setInputStream(utt.getInputStream(), isBigEndian);
262                 }
263             }

267         public void decode() {
268     
269             try {
270                 utteranceId = 0;
271                 DataOutputStream ctm = new DataOutputStream(new FileOutputStream(ctmFile));
272                 recognizer.allocate();
273     
274                 for (Iterator i = new CTLIterator(); i.hasNext();) {
275                     CTLUtterance utt = (CTLUtterance) i.next();
276                     setInputStream(utt);
277                     Result result = recognizer.recognize();
278                     System.out.println("Utterance " + utteranceId + ": " + utt.getName());
279                     System.out.println("Reference: " + utt.getRef());
280                     System.out.println("Result   : " + result);
281                     logger.info("Utterance " + utteranceId + ": " + utt.getName());
282                     logger.info("Result   : " + result);
283                     handleResult(ctm, utt, result);
284                     utteranceId++;
285                 }
286     
287                 recognizer.deallocate();
288             } catch (IOException io) {
289                 logger.severe("I/O error during decoding: " + io.getMessage());
290             }
291             logger.info("BatchCTLDecoder: " + utteranceId + " utterances decoded");
292         }

300         private int dumpBestPath(DataOutputStream out, CTLUtterance utt, Token token) throws IOException {
301     
302             if (token == null) return 0;
303     
304             Token pred = token.getPredecessor();
305             int startFrame = dumpBestPath(out, utt, pred);
306             if (token.isWord()) {
307     
308                 int endFrame = token.getFrameNumber();
309     
310                 WordSearchState wordState = (WordSearchState) token.getSearchState();
311                 Word word = wordState.getPronunciation().getWord();
312                 String spelling = word.getSpelling();
313                 if (!spelling.startsWith("<")) {
314                     String[] names = utt.name.split("_");
315                     String id = names[0] + "_" + names[1] + "_" + names[2];
316                     out.write((id + " 1 " + (utt.startOffset + startFrame) / 100.0 + " " + (endFrame - startFrame) / 100.0 + " ").getBytes());
317                     out.write(hex2Binary(spelling));
318                     out.write(" 0.700000\n".getBytes());
319                 }
320                 return endFrame;
321             }
322             return startFrame;
323         }

326         static public byte[] hex2Binary(String spelling) {
327             byte[] bin = new byte[spelling.length() / 2];
328             for (int i = 0; i < spelling.length(); i += 2) {
329                 int i0 = hexToByte(spelling.charAt(i));
330                 int i1 = hexToByte(spelling.charAt(i + 1));
331                 bin[i / 2] = (byte) (i1 + (16 * i0));
332             }
333             return bin;
334         }

337         static private int hexToByte(char c) {
338             switch (c) {
339                 case '0':
340                     return 0;
341                 case '1':
342                     return 1;
343                 case '2':
344                     return 2;
345                 case '3':
346                     return 3;
347                 case '4':
348                     return 4;
349                 case '5':
350                     return 5;
351                 case '6':
352                     return 6;
353                 case '7':
354                     return 7;
355                 case '8':
356                     return 8;
357                 case '9':
358                     return 9;
359                 case 'a':
360                     return 10;
361                 case 'b':
362                     return 11;
363                 case 'c':
364                     return 12;
365                 case 'd':
366                     return 13;
367                 case 'e':
368                     return 14;
369                 case 'f':
370                     return 15;
371                 default:
372                     throw new Error("Bad hex char " + c);
373             }
