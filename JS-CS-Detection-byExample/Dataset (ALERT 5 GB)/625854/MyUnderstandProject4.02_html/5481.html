13     package edu.cmu.sphinx.tools.audio;
14     
15     import javax.sound.sampled.AudioFormat;
16     import java.io.IOException;
17     import java.io.InputStream;
18     import java.util.ArrayList;
19     
20     /**
21      * Reads a raw audio file (i.e., a file that is just plain raw samples and nothing else) and converts it to signed
22      * data.
23      */
24     public class RawReader {
25     
26         /**
27          * Reads raw bytes from the given audioStream and returns big endian signed samples based on the audioFormat.  Only
28          * PCM_SIGNED or PCM_UNSIGNED encodings are allowed.
29          *
30          * @param audioStream the stream containing the raw bytes
31          * @param audioFormat a hint of what to expect from the stream
32          * @return big endian signed samples, one sample per array element
33          */
34         public static short[] readAudioData(InputStream audioStream,
35                                             AudioFormat audioFormat)
36                 throws IOException {
37     
38             /* Initialize ourselves based on the intput data format.
39              */
40             int bytesPerSample;
41             boolean signedData = true;
42             boolean bigEndian;
43             AudioFormat.Encoding encoding = audioFormat.getEncoding();
44             bytesPerSample = audioFormat.getSampleSizeInBits() / 8;
45             if (encoding == AudioFormat.Encoding.PCM_SIGNED) {
46                 signedData = true;
47             } else if (encoding == AudioFormat.Encoding.PCM_UNSIGNED) {
48                 signedData = false;
49             } else {
50                 System.err.println("Unsupported audio encoding: " + encoding);
51                 System.exit(-1);
52             }
53             bigEndian = audioFormat.isBigEndian();
54     
55             /* Now read in the data, saving the samples in an array list.
56              * Along the way, convert each sample to big endian signed
57              * data.
58              */
59             byte[] buffer = new byte[bytesPerSample];
60             ArrayList<Short> samples = new ArrayList<Short>();
61             int read = 0;
62             int totalRead = 0;
63             boolean done = false;
64             while (!done) {
65                 totalRead = read = audioStream.read(buffer, 0, bytesPerSample);
66                 while (totalRead < bytesPerSample) {
67                     if (read == -1) {
68                         done = true;
69                         break;
70                     } else {
71                         read = audioStream.read(buffer,
72                                 totalRead,
73                                 bytesPerSample - totalRead);
74                         totalRead += read;
75                     }
76                 }
77                 if (!done) {
78                     int val = 0;
79                     if (bigEndian) {
80                         val = (int) buffer[0];
81                         if (!signedData) {
82                             val &= 0xff;
83                         }
84                         for (int i = 1; i < bytesPerSample; i++) {
85                             int temp = (int) buffer[i] & 0xff;
86                             val = (val << 8) + temp;
87                         }
88                     } else {
89                         val = (int) buffer[bytesPerSample - 1];
90                         if (!signedData) {
91                             val &= 0xff;
92                         }
93                         for (int i = bytesPerSample - 2; i >= 0; i--) {
94                             int temp = (int) buffer[i] & 0xff;
95                             val = (val << 8) + temp;
96                         }
97                     }
98     
99                     /* We'll always give signed data.  So, if the input
100                      * is unsigned, convert it to signed.
101                      */
102                     if (!signedData) {
103                         val = val - (1 << ((bytesPerSample * 8) - 1));
104                     }
105                     samples.add((short) val);
106                 }
107             }
108     
109             /* Convert the array list to an array of shorts and return.
110             */
111             short[] audioData = new short[samples.size()];
112             for (int i = 0; i < audioData.length; i++) {
113                 audioData[i] = (samples.get(i)).shortValue();
114             }
115             return audioData;
116         }
117     }
