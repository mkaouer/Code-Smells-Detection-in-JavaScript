13     package edu.cmu.sphinx.linguist.acoustic;
14     
15     import java.io.Serializable;
16     
17     /** Represents  the context for a unit */
18     public class Context implements Serializable {
19     
20         /** Represents an empty context */
21         public final static Context EMPTY_CONTEXT = new Context();
22     
23     
24         /** No instantiations allowed */
25         protected Context() {
26         }
27     
28     
29         /**
30          * Checks to see if there is a partial match with the given context. For a simple context such as this we always
31          * match.
32          *
33          * @param context the context to check
34          * @return true if there is a partial match
35          */
36         public boolean isPartialMatch(Context context) {
37             return true;
38         }
39     
40     
41         /** Provides a string representation of a context */
42         public String toString() {
43             return "";
44         }
45     
46     
47         /**
48          * Determines if an object is equal to this context
49          *
50          * @param o the object to check
51          * @return true if the objects are equal
52          */
53         public boolean equals(Object o) {
54             if (this == o) {
55                 return true;
56             } else if (o instanceof Context) {
57                 Context otherContext = (Context) o;
58                 return toString().equals(otherContext.toString());
59             } else {
60                 return false;
61             }
62         }
63     
64     
65         /**
66          * calculates a hashCode for this context. Since we defined an equals for context, we must define a hashCode as
67          * well
68          *
69          * @return the hashcode for this object
70          */
71         public int hashCode() {
72             return toString().hashCode();
73         }
74     }
