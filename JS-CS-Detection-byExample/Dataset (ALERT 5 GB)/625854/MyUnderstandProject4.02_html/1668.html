245         /**
246          * Adds a set of density arrays to a given pool.
247          *
248          * @param pool                 the pool to add densities to
249          * @param stateID              a vector with the senone id of the states in a model
250          * @param numStreams           the number of streams
251          * @param numGaussiansPerState the number of Gaussians per state
252          * @throws IOException if an error occurs while loading the data
253          */
254         private void addModelToDensityPool(Pool pool, int[] stateID,
255                                            int numStreams, int numGaussiansPerState)
256                 throws IOException {
257             int numStates;
258             int numInPool;
259     
260             assert pool != null;
261             assert stateID != null;
262     
263             numStates = stateID.length;
264     
265             numInPool = pool.getFeature(NUM_SENONES, 0);
266             pool.setFeature(NUM_SENONES, numStates + numInPool);
267             numInPool = pool.getFeature(NUM_STREAMS, -1);
268             if (numInPool == -1) {
269                 pool.setFeature(NUM_STREAMS, numStreams);
270             } else {
271                 assert numInPool == numStreams;
272             }
273             numInPool = pool.getFeature(NUM_GAUSSIANS_PER_STATE, -1);
274             if (numInPool == -1) {
275                 pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
276             } else {
277                 assert numInPool == numGaussiansPerState;
278             }
279     
280             // TODO: numStreams should be any number > 0, but for now....
281             assert numStreams == 1;
282             for (int i = 0; i < numStates; i++) {
283                 int state = stateID[i];
284                 for (int j = 0; j < numGaussiansPerState; j++) {
285                     // We're creating densities here, so it's ok if values
286                     // are all zero.
287                     float[] density = new float[vectorLength];
288                     int id = state * numGaussiansPerState + j;
289                     pool.put(id, density);
290                 }
291             }
292         }

295         /**
296          * If a data point is below 'floor' make it equal to floor.
297          *
298          * @param data  the data to floor
299          * @param floor the floored value
300          */
301         private void floorData(float[] data, float floor) {
302             for (int i = 0; i < data.length; i++) {
303                 if (data[i] < floor) {
304                     data[i] = floor;
305                 }
306             }

310         /**
311          * Normalize the given data
312          *
313          * @param data the data to normalize
314          */
315         private void normalize(float[] data) {
316             float sum = 0;
317             for (int i = 0; i < data.length; i++) {
318                 sum += data[i];
319             }
320     
321             if (sum != 0.0f) {
322                 // Invert, so we multiply instead of dividing inside the loop
323                 sum = 1.0f / sum;
324                 for (int i = 0; i < data.length; i++) {
325                     data[i] = data[i] * sum;
326                 }
327             }
328         }

331         /**
332          * Convert to log math
333          *
334          * @param data the data to normalize
335          */
336         // linearToLog returns a float, so zero values in linear scale
337         // should return -Float.MAX_VALUE.
338         private void convertToLogMath(float[] data) {
339             for (int i = 0; i < data.length; i++) {
340                 data[i] = logMath.linearToLog(data[i]);
341             }

345         /**
346          * Loads the phone list, which possibly contains the sizes (number of states) of models.
347          *
348          * @param ps
349          * @param useCDUnits  if true, uses context dependent units
350          * @param inputStream the open input stream to use
351          * @param path        the path to a density file @throws FileNotFoundException if a file cannot be found
352          * @throws IOException if an error occurs while loading the data
353          */
354         private void loadPhoneList(PropertySheet ps, boolean useCDUnits, InputStream inputStream, String path)
355                 throws IOException {
356             int numState = 0;
357             // TODO: this should be flexible, but we're hardwiring for now
358             int numStreams = 1;
359             // Since we're initializing, we start simple.
360             int numGaussiansPerState = 1;
361             String version;
362             boolean sameSizedModels;
363             boolean tmatSkip;
364     
365             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer
366                     (inputStream, '#', false);
367             //    	Pool pool = new Pool(path);
368     
369             // Initialize the pools we'll need.
370             meansPool = new Pool("means");
371             variancePool = new Pool("variances");
372             mixtureWeightsPool = new Pool("mixtureweights");
373             matrixPool = new Pool("transitionmatrices");
374             senonePool = new Pool("senones");
375     
376             float distFloor = ps.getFloat(PROP_MC_FLOOR);
377             float mixtureWeightFloor = ps.getFloat(PROP_MW_FLOOR);
378             float transitionProbabilityFloor = 0;
379     //                ps.getFloat(Sphinx3Loader.PROP_TP_FLOOR, Sphinx3Loader.PROP_TP_FLOOR_DEFAULT);
380             float varianceFloor = ps.getFloat(PROP_VARIANCE_FLOOR);
381     
382     
383             logger.info("Loading phone list file from: ");
384             logger.info(path);
385     
386             // At this point, we only accept version 0.1
387             version = "0.1";
388             est.expectString("version");
389             est.expectString(version);
390     
391             est.expectString("same_sized_models");
392             sameSizedModels = est.getString().equals("yes");
393     
394             if (sameSizedModels) {
395                 est.expectString("n_state");
396                 numState = est.getInt("numBase");
397             }
398     
399             // for this phone list version, let's assume left-to-right
400             // models, with optional state skip.
401             est.expectString("tmat_skip");
402             tmatSkip = est.getString().equals("yes");
403     
404             // Load the phones with sizes
405     
406             // stateIndex contains the absolute state index, that is, a
407             // unique index in the senone pool.
408             int stateIndex;
409             int unitCount;
410             String attribute;
411             for (stateIndex = 0, unitCount = 0; ;) {
412                 String phone = est.getString();
413                 if (est.isEOF()) {
414                     break;
415                 }
416                 int size = numState;
417                 if (!sameSizedModels) {
418                     size = est.getInt("ModelSize");
419                 }
420                 phoneList.put(phone, new Integer(size));
421                 logger.fine("Phone: " + phone + " size: " + size);
422                 int[] stid = new int[size];
423                 String position = "-";
424     
425                 for (int j = 0; j < size; j++, stateIndex++) {
426                     stid[j] = stateIndex;
427                 }
428     
429                 // The first filler
430                 if (phone.equals(SILENCE_CIPHONE)) {
431                     attribute = FILLER;
432                 } else {
433                     attribute = "-";
434                 }
435     
436     //            Unit unit = Unit.getUnit(phone, attribute.equals(FILLER));
437                 Unit unit = null;
438                 contextIndependentUnits.put(unit.getName(), unit);
439     
440                 if (logger.isLoggable(Level.FINE)) {
441                     logger.fine("Loaded " + unit + " with " + size + " states");
442                 }
443     
444                 // The first filler
445                 if (unit.isFiller() && unit.getName().equals(SILENCE_CIPHONE)) {
446     //                unit = Unit.SILENCE;
447                     unit = null;
448                 }
449     
450                 // Means
451                 addModelToDensityPool(meansPool, stid, numStreams,
452                         numGaussiansPerState);
453     
454                 // Variances
455                 addModelToDensityPool(variancePool, stid, numStreams,
456                         numGaussiansPerState);
457     
458                 // Mixture weights
459                 addModelToMixtureWeightPool(mixtureWeightsPool, stid, numStreams,
460                         numGaussiansPerState, mixtureWeightFloor);
461     
462                 // Transition matrix
463                 addModelToTransitionMatricesPool(matrixPool,
464                         unitCount,
465                         stid.length,
466                         transitionProbabilityFloor,
467                         tmatSkip);
468     
469                 // After creating all pools, we create the senone pool.
470                 addModelToSenonePool(senonePool, stid, distFloor, varianceFloor);
471     
472                 // With the senone pool in place, we go through all units, and
473                 // create the HMMs.
474     
475                 // Create tmat
476                 float[][] transitionMatrix = (float[][])
477                         matrixPool.get(unitCount);
478                 SenoneSequence ss = getSenoneSequence(stid);
479     
480                 HMM hmm = new SenoneHMM(unit, ss,
481                         transitionMatrix, HMMPosition.lookup(position));
482                 hmmManager.put(hmm);
483                 unitCount++;
484             }
485     
486             // If we want to use this code to load sizes/create models for
487             // CD units, we need to find another way of establishing the
488             // number of CI models, instead of just reading until the end
489             // of file.
490     
491             est.close();
492         }

514         /**
515          * Adds model to the mixture weights
516          *
517          * @param pool                 the pool to add models to
518          * @param stateID              vector containing state ids for hmm
519          * @param numStreams           the number of streams
520          * @param numGaussiansPerState the number of Gaussians per state
521          * @param floor                the minimum mixture weight allowed
522          * @throws IOException if an error occurs while loading the data
523          */
524         private void addModelToMixtureWeightPool(Pool pool, int[] stateID,
525                                                  int numStreams, int numGaussiansPerState, float floor)
526                 throws IOException {
527     
528             int numStates = stateID.length;
529             int numInPool;
530     
531             assert pool != null;
532     
533             numInPool = pool.getFeature(NUM_SENONES, 0);
534             pool.setFeature(NUM_SENONES, numStates + numInPool);
535             numInPool = pool.getFeature(NUM_STREAMS, -1);
536             if (numInPool == -1) {
537                 pool.setFeature(NUM_STREAMS, numStreams);
538             } else {
539                 assert numInPool == numStreams;
540             }
541             numInPool = pool.getFeature(NUM_GAUSSIANS_PER_STATE, -1);
542             if (numInPool == -1) {
543                 pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
544             } else {
545                 assert numInPool == numGaussiansPerState;
546             }
547     
548             // TODO: allow any number for numStreams
549             assert numStreams == 1;
550             for (int i = 0; i < numStates; i++) {
551                 int state = stateID[i];
552                 float[] logMixtureWeight = new float[numGaussiansPerState];
553                 // Initialize the weights with the same value, e.g. floor
554                 floorData(logMixtureWeight, floor);
555                 // Normalize, so the numbers are not all too low
556                 normalize(logMixtureWeight);
557                 convertToLogMath(logMixtureWeight);
558                 pool.put(state, logMixtureWeight);
559             }
560         }

563         /**
564          * Adds transition matrix to the transition matrices pool
565          *
566          * @param pool              the pool to add matrix to
567          * @param hmmId             current HMM's id
568          * @param numEmittingStates number of states in current HMM
569          * @param floor             the transition probability floor
570          * @param skip              if true, states can be skipped
571          * @throws IOException if an error occurs while loading the data
572          */
573         private void addModelToTransitionMatricesPool(Pool pool, int hmmId,
574                                                       int numEmittingStates, float floor,
575                                                       boolean skip)
576                 throws IOException {
577     
578             assert pool != null;
579     
580             // Add one to account for the last, non-emitting, state
581             int numStates = numEmittingStates + 1;
582     
583             float[][] tmat = new float[numStates][numStates];
584     
585             for (int j = 0; j < numStates; j++) {
586                 for (int k = 0; k < numStates; k++) {
587                     // Just to be sure...
588                     tmat[j][k] = 0.0f;
589     
590                     // the last row is just zeros, so we just do
591                     // the first (numStates - 1) rows
592     
593                     // The value assigned could be anything, provided
594                     // we normalize it.
595                     if (j < numStates - 1) {
596                         // Usual case: state can transition to itself
597                         // or the next state.
598                         if (k == j || k == j + 1) {
599                             tmat[j][k] = floor;
600                         }
601                         // If we can skip, we can also transition to
602                         // the next state
603                         if (skip) {
604                             if (k == j + 2) {
605                                 tmat[j][k] = floor;
606                             }
607                         }
608                     }
609                 }
610                 normalize(tmat[j]);
611                 convertToLogMath(tmat[j]);
612             }
613             pool.put(hmmId, tmat);
614         }

635                 }
636             }
637     
638             pool.put(0, matrix);
