13     package edu.cmu.sphinx.decoder.search;
14     
15     import java.util.Iterator;
16     import java.util.List;
17     
18     /**
19      * An active list is maintained as a sorted list
20      * <p/>
21      * Note that all scores are represented in LogMath logbase
22      */
23     public interface ActiveList {
24     
25         /**
26          * property that sets the desired (or target) size for this active list.  This is sometimes referred to as the beam
27          * size
28          */
29         public final static String PROP_ABSOLUTE_BEAM_WIDTH = "absoluteBeamWidth";
30     
31         /** The default value for the PROP_ABSOLUTE_BEAM_WIDTH property */
32         public final static int PROP_ABSOLUTE_BEAM_WIDTH_DEFAULT = 2000;
33     
34         /**
35          * Property that sets the minimum score relative to the maximum score in the list for pruning.  Tokens with a score
36          * less than relativeBeamWidth * maximumScore will be pruned from the list
37          */
38     
39         public final static String PROP_RELATIVE_BEAM_WIDTH = "relativeBeamWidth";
40     
41         /** The default value for the PROP_RELATIVE_BEAM_WIDTH property */
42         public final static double PROP_RELATIVE_BEAM_WIDTH_DEFAULT = 0.0;
43     
44         /**
45          * Property that indicates whether or not the active list will implement 'strict pruning'.  When strict pruning is
46          * enabled, the active list will not remove tokens from the active list until they have been completely scored.  If
47          * strict pruning is not enabled, tokens can be removed from the active list based upon their entry scores. The
48          * default setting is false (disabled).
49          */
50     
51         public final static String PROP_STRICT_PRUNING = "strictPruning";
52     
53         /** The default for the PROP_STRICT_PRUNING property */
54         public final static boolean PROP_STRICT_PRUNING_DEFAULT = true;
55     
56     
57         /**
58          * Adds the given token to the list, keeping track of the lowest scoring token
59          *
60          * @param token the token to add
61          */
62         public void add(Token token);
63     
64     
65         /**
66          * Replaces an old token with a new token
67          *
68          * @param oldToken the token to replace (or null in which case, replace works like add).
69          * @param newToken the new token to be placed in the list.
70          */
71         public void replace(Token oldToken, Token newToken);
72     
73     
74         /**
75          * Purges the active list of excess members returning a (potentially new) active list
76          *
77          * @return a purged active list
78          */
79         public ActiveList purge();
80     
81     
82         /**
83          * Returns an iterator over the elements in this active list
84          *
85          * @return an iterator
86          */
87         public Iterator<Token> iterator();
88     
89     
90         /**
91          * Returns the size of this list
92          *
93          * @return the size
94          */
95         public int size();
96     
97     
98         /**
99          * Gets the list of all tokens
100          *
101          * @return the set of tokens
102          */
103         public List<Token> getTokens();
104     
105     
106         /**
107          * gets the beam threshold best upon the best scoring token
108          *
109          * @return the beam threshold
110          */
111         public float getBeamThreshold();
112     
113     
114         /**
115          * gets the best score in the list
116          *
117          * @return the best score
118          */
119         public float getBestScore();
120     
121     
122         /**
123          * Sets the best scoring token for this active list
124          *
125          * @param token the best scoring token
126          */
127         public void setBestToken(Token token);
128     
129     
130         /**
131          * Gets the best scoring token for this active list
132          *
133          * @return the best scoring token
134          */
135         public Token getBestToken();
136     
137     
138         /**
139          * Creates a new empty version of this active list with the same general properties.
140          *
141          * @return a new active list.
142          */
143         public ActiveList newInstance();
144     }
