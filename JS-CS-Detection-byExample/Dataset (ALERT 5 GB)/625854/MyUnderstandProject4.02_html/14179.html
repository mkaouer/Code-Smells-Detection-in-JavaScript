37         /**
38          * Returns an appropriate InputStream of the given file in the given URL location. The location can be a plain
39          * directory or a ZIP file (these are the only two supported at this point). The <code>resolve</code> method is
40          * called to resolve whether "location" refers to a ZIP file or a directory.
41          * <p/>
42          * Suppose you want the InputStream to the file "dict/dictionary.txt" in the ZIP file
43          * "file:/lab/speech/sphinx4/data/wsj.zip". You will do: <code> StreamFactory.getInputStream(
44          * "file:/lab/speech/sphinx4/data/wsj.zip", "dict/dictionary.txt"); </code>
45          * <p/>
46          * Suppose you want the InputStream to the file "dict/dictionary.txt" in the directory
47          * "file:/lab/speech/sphinx4/data/wsj", you will do: <code> StreamFactory.getInputStream(
48          * "file:/lab/speech/sphinx4/data/wsj", "dict/dictionary.txt"); </code>
49          * <p/>
50          * The <code>StreamFactory.resolve()</code> method is called to resolve whether "location" refers to a ZIP file or a
51          * directory.
52          *
53          * @param location the URL location of the input data, it can now be a directory or a ZIP file
54          * @param file     the file in the given location to obtain the InputStream
55          * @return an InputStream of the given file in the given location
56          */
57         public static InputStream getInputStream(String location,
58                                                  String file) throws
59                 FileNotFoundException, IOException, ZipException {
60             if (location != null) {
61                 return StreamFactory.getInputStream
62                         (StreamFactory.resolve(location), location, file);
63             } else {
64                 return StreamFactory.getInputStream(StreamFactory.DIRECTORY,
65                         location, file);
66             }
67         }

70         /**
71          * According to the given data format, returns an appropriate InputStream of the given file in the given URL
72          * location. The location can be a plain directory or a JAR or ZIP file (these are the only ones supported at this
73          * point).
74          * <p/>
75          * Suppose you want the InputStream to the file "dict/dictionary.txt" in the ZIP file
76          * "file:/lab/speech/sphinx4/data/wsj.zip". You will do: <code> StreamFactory.getInputStream(StreamFactory.ZIP_FILE,
77          * "file:/lab/speech/sphinx4/data/wsj.zip", "dict/dictionary.txt"); </code>
78          * <p/>
79          * Suppose you want the InputStream to the file "dict/dictionary.txt" in the directory
80          * "file:/lab/speech/sphinx4/data/wsj", you will do: <code> StreamFactory.getInputStream(StreamFactory.DIRECTORY,
81          * "file:/lab/speech/sphinx4/data/wsj", "dict/dictionary.txt"); </code>
82          *
83          * @param format   the format of the input data, the currently supported formats are: <br>StreamFactory.ZIP_FILE
84          *                 <br>StreamFactory.DIRECTORY
85          * @param location the URL location of the input data, it can now be a directory or a JAR or ZIP file, or null if no
86          *                 location is given, which means that the <code>argument</code> also specifies the exact location
87          * @param file     the file in the given location to obtain the InputStream
88          * @return an InputStream of the given file in the given location
89          */
90         public static InputStream getInputStream(String format,
91                                                  String location,
92                                                  String file) throws
93                 FileNotFoundException, IOException, ZipException {
94             InputStream stream = null;
95             String absoluteLocation;
96     
97             if (location == null) {
98                 absoluteLocation = null;
99             } else {
100                 // Create a url from the location, possibly relative path
101                 URI uri = URI.create(location);
102                 // Get the scheme and the path
103                 String scheme = uri.getScheme();
104                 String path = uri.getSchemeSpecificPart();
105                 // Create a file with the path (aka scheme-specific part)
106                 File relativeFile = new File(path);
107                 // Make the path absolute and reconstruct the location, with
108                 // the correct scheme
109                 URI absoluteURI = relativeFile.getAbsoluteFile().toURI();
110                 if (scheme == null) {
111                     absoluteLocation = absoluteURI.getSchemeSpecificPart();
112                 } else {
113                     absoluteLocation = scheme + ":" +
114                             absoluteURI.getSchemeSpecificPart();
115                 }
116             }
117     
118             if (format.equals(ZIP_FILE)) {
119                 try {
120                     URI newURI = new URI(absoluteLocation);
121                     ZipFile zipFile =
122                             new ZipFile(new File(new URI(absoluteLocation)));
123                     ZipEntry entry = zipFile.getEntry(file);
124                     if (entry != null) {
125                         stream = zipFile.getInputStream(entry);
126                     }
127                 } catch (URISyntaxException use) {
128                     use.printStackTrace();
129                     throw new ZipException("URISyntaxException: " +
130                             use.getMessage());
131                 }
132             } else if (format.equals(DIRECTORY)) {
133                 if (absoluteLocation != null) {
134                     stream = new FileInputStream(absoluteLocation +
135                             File.separator + file);
136                 } else {
137                     stream = new FileInputStream(file);
138                 }
139             }
140             return stream;
141         }

146          * directory or a ZIP file (these are the only two supported at this point).  The <code>resolve</code> method is
147          * called to resolve whether "location" refers to a ZIP file or a directory. If saving to a zip or jar, the file can
148          * be appended or overwritten. If saving to a directory, files are always overwritten.
149          * <p/>
150          * Suppose you want the OutputStream to the file "dict/dictionary.txt" in the ZIP file
151          * "file:/lab/speech/sphinx4/data/wsj.zip". You will do: <code> StreamFactory.getOutputStream(

153          * <p/>
154          * Suppose you want the OutputStream to the file "dict/dictionary.txt" in the directory
155          * "file:/lab/speech/sphinx4/data/wsj", you will do: <code> StreamFactory.getOutputStream(
156          * "file:/lab/speech/sphinx4/data/wsj", "dict/dictionary.txt", false); </code>
157          * <p/>
158          * The <code>StreamFactory.resolve()</code> method is called to resolve whether "location" refers to a ZIP file or a
159          * directory.
160          *
161          * @param location the URL location of the output data, it can now be a directory or a ZIP file
162          * @param file     the file in the given location to obtain the OutputStream
163          * @param append   if true and saving to a zip file, then file is appended rather than overwritten.
164          * @return an OutputStream of the given file in the given location
165          */
166         public static OutputStream getOutputStream(String location,
167                                                    String file,
168                                                    boolean append) throws
169                 FileNotFoundException, IOException, ZipException {
170             if (location != null) {
171                 return StreamFactory.getOutputStream
172                         (StreamFactory.resolve(location), location, file, append);
173             } else {
174                 return StreamFactory.getOutputStream(StreamFactory.DIRECTORY,
175                         location, file);
176             }
177         }

180         /**
181          * According to the given data format, returns an appropriate OutputStream of the given file in the given URL
182          * location.  The location can be a plain directory or a JAR or ZIP file (these are the only ones supported at this
183          * point). If saving to a zip or jar, the file can be appended or overwritten. If saving to a directory, files are
184          * always overwritten.
185          * <p/>
186          * Suppose you want the OutputStream to the file "dict/dictionary.txt" in the ZIP file
187          * "file:/lab/speech/sphinx4/data/wsj.zip". You will do: <code> StreamFactory.getOutputStream(StreamFactory.ZIP_FILE,
188          * "file:/lab/speech/sphinx4/data/wsj.zip", "dict/dictionary.txt", true); </code>
189          * <p/>
190          * Suppose you want the OutputStream to the file "dict/dictionary.txt" in the directory
191          * "file:/lab/speech/sphinx4/data/wsj", you will do: <code> StreamFactory.getOutputStream(StreamFactory.DIRECTORY,
192          * "file:/lab/speech/sphinx4/data/wsj", "dict/dictionary.txt", false); </code>
193          *
194          * @param format   the format of the output data, the currently supported formats are: <br>StreamFactory.ZIP_FILE
195          *                 <br>StreamFactory.DIRECTORY
196          * @param location the URL location of the output data, it can now be a directory or a JAR or ZIP file, or null if
197          *                 no location is given, which means that the <code>argument</code> also specifies the exact
198          *                 location
199          * @param file     the file in the given location to obtain the OutputStream
200          * @param append   if true and saving to a zip file, then file is appended rather than overwritten.
201          * @return an OutputStream of the given file in the given location
202          */
203         public static OutputStream getOutputStream(String format,
204                                                    String location,
205                                                    String file,
206                                                    boolean append) throws
207                 FileNotFoundException, IOException, ZipException {
208             OutputStream stream = null;
209             if (format.equals(ZIP_FILE)) {
210                 try {
211                     System.out.println("WARNING: ZIP not yet fully supported.!");
212                     File path = new File(location);
213                     File parent = new File(path.getParent());
214                     if (!parent.exists()) {
215                         parent.mkdirs();
216                     }
217                     FileOutputStream fos =
218                             new FileOutputStream(new File(new URI(location)), append);
219                     stream = new ZipOutputStream(new BufferedOutputStream(fos));
220     
221                     ZipEntry entry = new ZipEntry(file);
222                     ((ZipOutputStream) stream).putNextEntry(entry);
223                 } catch (URISyntaxException use) {
224                     use.printStackTrace();
225                     throw new ZipException("URISyntaxException: " +
226                             use.getMessage());
227                 }
228             } else if (format.equals(DIRECTORY)) {
229                 if (location != null) {
230                     File path = new File(location + File.separator + file);
231                     File parent = new File(path.getParent());
232                     if (!parent.exists()) {
233                         parent.mkdirs();
234                     }
235                     stream = new FileOutputStream(location + File.separator + file);
236                 } else {
237                     File path = new File(file);
238                     File parent = new File(path.getParent());
239                     if (!parent.exists()) {
240                         parent.mkdirs();
241                     }
242                     stream = new FileOutputStream(file);
243                 }
244             } else {
245                 throw new IOException("Format not supported for writing");
246             }
247             return stream;
248         }

323         /**
324          * Returns the type of the given data source. The current supported types are: <code> StreamFactory.ZIP_FILE
325          * StreamFactory.DIRECTORY </code>
326          */
327         public static String resolve(String sourceName) {
328             if ((sourceName.endsWith(".jar")) || (sourceName.endsWith(".zip"))) {
329                 return StreamFactory.ZIP_FILE;
330             } else {
331                 return StreamFactory.DIRECTORY;
332             }
333         }
