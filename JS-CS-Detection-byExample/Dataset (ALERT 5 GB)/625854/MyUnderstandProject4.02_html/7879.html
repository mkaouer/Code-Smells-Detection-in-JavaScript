16     /** Implements a buffer for bigrams read from disk. */
17     class BigramBuffer extends NGramBuffer {
18     
19         /**
20          * Constructs a BigramBuffer object with the given byte[].
21          *
22          * @param bigramsOnDisk the byte[] with bigrams
23          * @param numberNGrams  the number of bigram follows in the byte[]
24          */
25         public BigramBuffer(byte[] bigramsOnDisk,
26                             int numberNGrams, boolean bigEndian) {
27             super(bigramsOnDisk, numberNGrams, bigEndian);
28         }
29     
30     
31         /**
32          * Finds the bigram probabilities for the given second word in a bigram.
33          *
34          * @param secondWordID the ID of the second word
35          * @return the BigramProbability of the given second word
36          */
37         public BigramProbability findBigram(int secondWordID) {
38     
39             int mid, start = 0, end = getNumberNGrams() - 1;
40             BigramProbability bigram = null;
41     
42             while ((end - start) > 0) {
43                 mid = (start + end) / 2;
44                 int midWordID = getWordID(mid);
45                 if (midWordID < secondWordID) {
46                     start = mid + 1;
47                 } else if (midWordID > secondWordID) {
48                     end = mid;
49                 } else {
50                     bigram = getBigramProbability(mid);
51                     break;
52                 }
53             }
54             return bigram;
55         }
56     
57     
58         /**
59          * Returns the BigramProbability of the nth follower.
60          *
61          * @param nthFollower which follower
62          * @return the BigramProbability of the nth follower
63          */
64         public final BigramProbability getBigramProbability(int nthFollower) {
65             int nthPosition = nthFollower * LargeTrigramModel.BYTES_PER_BIGRAM;
66             setPosition(nthPosition);
67     
68             int wordID = readTwoBytesAsInt();
69             int probID = readTwoBytesAsInt();
70             int backoffID = readTwoBytesAsInt();
71             int firstTrigram = readTwoBytesAsInt();
72     
73             return (new BigramProbability
74                     (nthFollower, wordID, probID, backoffID, firstTrigram));
75         }
76     }
