11     package edu.cmu.sphinx.instrumentation;
12     
13     import edu.cmu.sphinx.linguist.dictionary.Word;
14     import edu.cmu.sphinx.result.*;
15     import edu.cmu.sphinx.util.NISTAlign;
16     import edu.cmu.sphinx.util.props.*;
17     
18     /**
19      * Tracks and reports recognition accuracy using the "confidenceScorer" component specified in the ConfigurationManager.
20      * The "confidenceScorer" component is typically configured to be edu.cmu.sphinx.result.SausageMaker.
21      *
22      * @see edu.cmu.sphinx.result.SausageMaker
23      */
24     public class BestConfidenceAccuracyTracker extends AccuracyTracker {
25     
26         /** Defines the class to use for confidence scoring. */
27         @S4Component(type = ConfidenceScorer.class)
28         public final static String PROP_CONFIDENCE_SCORER = "confidenceScorer";
29     
30         /** The confidence scorer */
31         protected ConfidenceScorer confidenceScorer;
32     
33     
34         /*
35         * (non-Javadoc)
36         *
37         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
38         */
39         public void newProperties(PropertySheet ps) throws PropertyException {
40             super.newProperties(ps);
41             confidenceScorer = (ConfidenceScorer) ps.getComponent(
42                     PROP_CONFIDENCE_SCORER
43             );
44         }
45     
46     
47         /** Gets the transcription with no fillers and no "<unk>". */
48         protected String getTranscriptionNoFiller(Path path) {
49             StringBuffer buf = new StringBuffer();
50             WordResult[] words = path.getWords();
51             for (int i = 0; i < words.length; i++) {
52                 Word word = words[i].getPronunciation().getWord();
53                 if (!word.isFiller() && !word.getSpelling().equals("<unk>")) {
54                     buf.append(word.getSpelling()).append(" ");
55                 }
56             }
57             return (buf.toString().trim());
58         }
59     
60     
61         /** Gets the raw transcription */
62         protected String getTranscriptionRaw(Path path) {
63             StringBuffer buf = new StringBuffer();
64             WordResult[] words = path.getWords();
65             for (int i = 0; i < words.length; i++) {
66                 Word word = words[i].getPronunciation().getWord();
67                 buf.append(word.getSpelling()).append(" ");
68             }
69             return (buf.toString().trim());
70         }
71     
72     
73         /*
74         * (non-Javadoc)
75         *
76         * @see edu.cmu.sphinx.decoder.ResultListener#newResult(edu.cmu.sphinx.result.Result)
77         */
78         public void newResult(Result result) {
79             NISTAlign aligner = getAligner();
80             String ref = result.getReferenceText();
81             if (result.isFinal() && (ref != null)) {
82                 try {
83                     Path bestPath = null;
84                     String hyp = "";
85                     if (result.getBestFinalToken() != null) {
86                         ConfidenceResult confidenceResult =
87                                 confidenceScorer.score(result);
88                         bestPath = confidenceResult.getBestHypothesis();
89                         hyp = getTranscriptionNoFiller(bestPath);
90                     }
91                     aligner.align(ref, hyp);
92                     if (bestPath != null) {
93                         showDetails(getTranscriptionRaw(bestPath));
94                     } else {
95                         showDetails("");
96                     }
97                 } catch (Exception e) {
98                     e.printStackTrace();
99                 }
100             }
101         }
102     }
