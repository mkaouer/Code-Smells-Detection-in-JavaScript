13     package edu.cmu.sphinx.linguist.dictionary;
14     
15     
16     /** Represents a word, its spelling and its pronunciation. */
17     public class Word {
18     
19         /** The Word representing the unknown word. */
20         public static final Word UNKNOWN;
21     
22     
23         static {
24             Pronunciation[] pros = {Pronunciation.UNKNOWN};
25             UNKNOWN = new Word("<unk>", pros, false);
26             Pronunciation.UNKNOWN.setWord(UNKNOWN);
27         }
28     
29     
30         private String spelling;               // the spelling of the word
31         private Pronunciation[] pronunciations; // pronunciations of this word
32         private boolean isFiller = false;
33     
34     
35         /**
36          * Creates a Word
37          *
38          * @param spelling       the spelling of this word
39          * @param pronunciations the pronunciations of this word
40          * @param isFiller       true if the word is a filler word
41          */
42         public Word(String spelling, Pronunciation[] pronunciations,
43                     boolean isFiller) {
44             this.spelling = spelling;
45             this.pronunciations = pronunciations;
46             this.isFiller = isFiller;
47         }
48     
49     
50         /**
51          * Returns the spelling of the word.
52          *
53          * @return the spelling of the word
54          */
55         public String getSpelling() {
56             return spelling;
57         }
58     
59     
60         /**
61          * Determines if this is a filler word
62          *
63          * @return <code>true</code> if this word is a filler word, otherwise it returns <code>false</code>
64          */
65         public boolean isFiller() {
66             return isFiller;
67         }
68     
69     
70         /**
71          * Returns true if this word is an end of sentence word
72          *
73          * @return true if the word matches Dictionary.SENTENCE_END_SPELLING
74          */
75         public boolean isSentenceEndWord() {
76             return Dictionary.SENTENCE_END_SPELLING.equals(this.spelling);
77         }
78     
79     
80         /**
81          * Returns true if this word is a start of sentence word
82          *
83          * @return true if the word matches Dictionary.SENTENCE_START_SPELLING
84          */
85         public boolean isSentenceStartWord() {
86             return Dictionary.SENTENCE_START_SPELLING.equals(this.spelling);
87         }
88     
89     
90         /**
91          * Retrieves the pronunciations of this word
92          *
93          * @param wordClassification the classification of the word (typically part of speech classification) or null if all
94          *                           word classifications are acceptable. The word classification must be one of the set
95          *                           returned by <code>Dictionary.getPossibleWordClassifications</code>
96          * @return the pronunciations of this word
97          */
98         public Pronunciation[] getPronunciations
99                 (WordClassification wordClassification) {
100             return pronunciations;
101         }
102     
103     
104         /**
105          * Retrieves the pronunciations of this word
106          *
107          * @return the pronunciations of this word
108          */
109         public Pronunciation[] getPronunciations() {
110             return pronunciations;
111         }
112     
113     
114         /**
115          * Get the highest probability pronunciation for a word
116          *
117          * @return the highest probability pronunciation
118          */
119         public Pronunciation getMostLikelyPronunciation() {
120             float bestScore = Float.NEGATIVE_INFINITY;
121             Pronunciation best = null;
122             for (int i = 0; i < pronunciations.length; i++) {
123                 if (pronunciations[i].getProbability() > bestScore) {
124                     bestScore = pronunciations[i].getProbability();
125                     best = pronunciations[i];
126                 }
127             }
128             return best;
129         }
130     
131     
132         @Override
133         public int hashCode() {
134             return spelling.hashCode();
135         }
136     
137     
138         @Override
139         public boolean equals(Object obj) {
140             if (obj instanceof Word) {
141                 return spelling.equals(((Word)obj).spelling);
142             }
143     
144             return false;
145         }
146     
147     
148         /**
149          * Returns a string representation of this word, which is the spelling
150          *
151          * @return the spelling of this word
152          */
153         public String toString() {
154             return spelling;
155         }
156     }
