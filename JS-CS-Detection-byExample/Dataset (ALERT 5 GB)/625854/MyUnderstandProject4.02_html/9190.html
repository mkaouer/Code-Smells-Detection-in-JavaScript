25     /**
26      * Given a sequence of Data, filters out the non-speech regions. The sequence of Data should have the speech and
27      * non-speech regions marked out by the SpeechStartSignal and SpeechEndSignal, using the {@link SpeechMarker
28      * SpeechMarker}. Such a sequence of Data for an utterance should look like one of the following two:
29      * <p/>
30      * <b>Case 1: Only one speech region</b> <p>In the first case, the data stream has only one speech region: <p><img
31      * src="doc-files/one-region.gif"> <br><i>Figure 1: A data stream with only one speech region</i>. <p>After filtering,
32      * the non-speech regions are removed, and becomes: <p><img src="doc-files/one-region-filtered.gif"> <br><i>Figure 2: A
33      * data stream with only on speech region after filtering.</i>
34      * <p/>
35      * <br><b>Case 2: Multiple speech regions</b>
36      * <p/>
37      * We will use the example of a data stream with two speech regions to illustrate the case of a data stream with
38      * multiple speech regions: <p><img src="doc-files/two-regions.gif"> <br><i>Figure 3: A data stream with two speech
39      * regions.</i>
40      * <p/>
41      * This case is more complicated than one speech region. The property <b>mergeSpeechSegments</b> is very important in
42      * controlling the behavior of this filter. This property determines whether individual speech regions (and the
43      * non-speech regions between them) in an utterance should be merged into one big region, or whether the individual
44      * speech regions should be converted into individual utterances. If <b>mergeSpeechSegments</b> is set to true, all the
45      * Data from the first SpeechStartSignal to the last SpeechEndSignal will be considered as one Utterance, and enclosed
46      * by a pair of SpeechEndSignal and SpeechEndSignal (which itself becomes enclosed by a pair of DataStartSignal and
47      * DataEndSignal). All non-speech regions are removed from the stream. This gives:
48      * <p/>
49      * <img src="doc-files/two-regions-merge.gif"> <br><i>Figure 4: A data stream with two speech regions after filtering,
50      * when <b>mergeSpeechSegments</b> is set to <b>true</b>.</i>
51      * <p/>
52      * On the other hand, if <b>mergeSpeechSegments</b> is set to false (the default), then each speech region will become
53      * its own data stream. Pictorially, our data stream with two speech regions becomes: <p><img
54      * src="doc-files/two-regions-nonmerge.gif"> <br><i>Figure 5: A data stream with two speech regions after filtering,
55      * when <b>mergeSpeechSegments</b> is set to <b>false</b>.</i>
56      * <p/>
57      */

65         @S4Boolean(defaultValue = false)

114         /**
115          * Returns the next Data or Signal.
116          *
117          * @return the next Data, or null if no Data is available
118          * @throws DataProcessingException if a data processing error occurs
119          */
120         public Data getData() throws DataProcessingException {
121     
122             if (outputQueue.size() == 0) {
123                 Data audio = readData();
124     
125                 getTimer().start();
126     
127                 if (audio != null) {
128                     if (!mergeSpeechSegments) {
129                         audio = handleNonMergingData(audio);
130                     } else {
131                         audio = handleMergingData(audio);
132                     }
133                 }
134                 outputQueue.add(audio);
135     
136                 getTimer().stop();
137             }
138     
139             if (outputQueue.size() > 0) {
140                 return outputQueue.remove(0);
141             } else {
142                 return null;
143             }
144         }

147         /**
148          * Handles the given Data in the case when mergeSpeechSegment is true.
149          *
150          * @param audio the Data object to handle
151          * @throws DataProcessingException if a data processor error occurs
152          */
153         private Data handleMergingData(Data audio) throws DataProcessingException {
154             Data next = audio;
155     
156             if (audio instanceof DataStartSignal) {
157                 // Read (and discard) all the Data from DataStartSignal until
158                 // we hit a SpeechStartSignal. The SpeechStartSignal is discarded.
159                 List<Data> audioList = readUntilSpeechStartOrDataEnd();
160                 Data last = audioList.get(audioList.size() - 1);
161                 if (last != null) {
162                     assert !(last instanceof DataEndSignal) : "data end while in speech";
163                     if (last instanceof SpeechStartSignal) {
164                         outputQueue.add(audio);
165                         next = last;
166                     }
167                 }
168             } else if (audio instanceof SpeechEndSignal) {
169                 // read (and discard) all the Data from SpeechEndSignal
170                 // until we hit a DataEndSignal
171                 List<Data> audioList = readUntilSpeechStartOrDataEnd();
172                 Data last = audioList.get(audioList.size() - 1);
173     
174                 assert last != null;
175                 if (last instanceof SpeechStartSignal) {
176                     // first remove the SpeechStartSignal, then add
177                     // all the Data to the inputBuffer
178     
179                     audioList.remove(last);
180                     inputBuffer.addAll(audioList);
181                     next = readData();
182     
183                 } else if (last instanceof DataEndSignal) {
184                     // System.out.println("Last is DataEndSignal");
185                     outputQueue.add(audio);
186                     next = last;
187                 }
188             }
189     
190             return next;
191         }

268         /**
269          * Returns the duration of the current speech segment.
270          *
271          * @return the duration of the current speech segment
272          */
273         private long getDuration() {
274             return (long) (((double) numberSpeechSamples / (double) sampleRate) * 1000.0);
275         }

278         /**
279          * Returns the next Data, either from the inputBuffer or the predecessor.
280          *
281          * @return the next available Data
282          * @throws DataProcessingException if a data processor error occurs
283          */
284         private Data readData() throws DataProcessingException {
285             Data audio;
286             if (inputBuffer.size() > 0) {
287                 audio = inputBuffer.remove(0);
288             } else {
289                 audio = getPredecessor().getData();
290             }
291             return audio;
292         }

295         /**
296          * Read until we hit a SpeechStartSignal or DataEndSignal.
297          *
298          * @return a list of all the Data read, including the SpeechStartSignal or DataEndSignal
299          */
300         private List<Data> readUntilSpeechStartOrDataEnd() throws DataProcessingException {
301             List<Data> audioList = new LinkedList<Data>();
302             Data audio;
303             do {
304                 audio = readData();
305                 if (audio != null) {
306                     audioList.add(audio);
307                 }
308     
309                 assert !(audio instanceof SpeechEndSignal);
310             } while (!(audio instanceof SpeechStartSignal) && !(audio instanceof DataEndSignal));
311             return audioList;
312         }
