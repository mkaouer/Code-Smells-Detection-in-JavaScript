36     public class XMLConfigReader implements GUIReader {

53           /**
54            * This class uses Singleton pattern for the constructor 
55            * to make sure that there is only one instance of reader created
56            * to get the reader instance, use <code>XMLConfigReader.getInstance()</code>
57            *       
58            */
59           private XMLConfigReader() {}
60     
61           private static class XMLReaderHolder {
62               private static XMLConfigReader instance = new XMLConfigReader();
63           }
64           
65           /**
66            * Singleton constructor 
67            *
68            * @return    the xml reader
69            */
70           public static XMLConfigReader getInstance(){
71               return XMLReaderHolder.instance;
72           }
73         
74           /**
75            * The method called for reading from xml file
76            *
77            * @param     fFile the File to be read
78            * @return    <code>ConfigProperties</code> object that holds the property type-value 
79            * @throws    GUIReaderExceptoin  if there are any errors while finding
80            *            and reading from file.
81            */
82           public ConfigProperties read(File fFile) throws GUIReaderException {        
83             if ( fFile == null || fFile.getName().trim().equalsIgnoreCase("") )  /*no filename for input */
84             {
85                 throw new GUIReaderException
86                         ("No input filename specified",GUIReaderException.EXCEPTION_NO_FILENAME);
87             }
88             else /* start reading from config file */
89             {
90                 try{          
91                    URL url = fFile.toURI().toURL();
92                    
93                    Map global = new HashMap(); 
94                    Map pm = loader(url, global);
95                    ConfigProperties cp = new ConfigProperties ();
96                    cp.setGlobal(global);
97                    cp.setProperty(pm);
98                    return cp;
99                    
100                 }catch(IOException e){
101                     throw new GUIReaderException("IO Exception during read",GUIReaderException.EXCEPTION_IO);
102                 }
103             }
104         }
105     
106           /**
107          * Loads the configuration data from the given url and adds the info to the
108          * symbol table. Throws an <code>IOexception</code> if there is a problem loading the
109          * table. Note that this only performs a partial populating of the symbol
110          * table entry
111          * 
112          * @param url
113          *                the url to load from
114          * @throws IOException
115          *                 if an error occurs while loading the symbol table
116          */
117         private Map loader(URL url, Map global) throws IOException {
118             SaxLoader saxLoader = new SaxLoader(url, global);
119             return saxLoader.load();
120         }
