22         /**
23          * Dumps the given component as GDL to the given stream
24          *
25          * @param out  where to dump the GDL
26          * @param name the name of the component to dump
27          */
28         public static void dumpComponentAsGDL(ConfigurationManager cm, PrintStream out, String name) {
29     
30             out.println("node: {title: \"" + name + "\" color: " + getColor(cm, name)
31                     + "}");
32     
33             PropertySheet ps = cm.getPropertySheet(name);
34             Collection<String> propertyNames = ps.getRegisteredProperties();
35     
36             for (String propertyName : propertyNames) {
37                 PropertySheet.PropertyType propType = ps.getType(propertyName);
38                 Object val = ps.getRaw(propertyName);
39     
40                 if (val != null) {
41                     if (propType == PropertySheet.PropertyType.COMP) {
42                         out.println("edge: {source: \"" + name
43                                 + "\" target: \"" + val + "\"}");
44                     } else if (propType == PropertySheet.PropertyType.COMPLIST) {
45                         List list = (List) val;
46                         for (Object listElement : list) {
47                             out.println("edge: {source: \"" + name
48                                     + "\" target: \"" + listElement + "\"}");
49                         }
50                     }
51                 }
52             }
53         }

61          */
62         public static void showConfigAsGDL(ConfigurationManager ConfigurationManager, String path) throws IOException {
63             PrintStream out = new PrintStream(new FileOutputStream(path));
64             dumpGDLHeader(out);
65             for (String componentName : ConfigurationManager.getInstanceNames(Configurable.class)) {
66                 dumpComponentAsGDL(ConfigurationManager, out, componentName);
67             }
68             dumpGDLFooter(out);
69             out.close();

94         /**
95          * Gets the color for the given component
96          *
97          * @param ConfigurationManager
98          * @param componentName        the name of the component @return the color name for the component
99          */
100         public static String getColor(ConfigurationManager ConfigurationManager, String componentName) {
101             try {
102                 Configurable c = ConfigurationManager.lookup(componentName);
103                 Class cls = c.getClass();
104                 if (cls.getName().indexOf(".recognizer") > 1) {
105                     return "cyan";
106                 } else if (cls.getName().indexOf(".tools") > 1) {
107                     return "darkcyan";
108                 } else if (cls.getName().indexOf(".decoder") > 1) {
109                     return "green";
110                 } else if (cls.getName().indexOf(".frontend") > 1) {
111                     return "orange";
112                 } else if (cls.getName().indexOf(".acoustic") > 1) {
113                     return "turquoise";
114                 } else if (cls.getName().indexOf(".linguist") > 1) {
115                     return "lightblue";
116                 } else if (cls.getName().indexOf(".instrumentation") > 1) {
117                     return "lightgrey";
118                 } else if (cls.getName().indexOf(".util") > 1) {
119                     return "lightgrey";
120                 }
121             } catch (PropertyException e) {
122                 return "black";
123             }
124             return "darkgrey";
125         }
