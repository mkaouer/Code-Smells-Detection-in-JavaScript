13     package edu.cmu.sphinx.demo.lattice;
14     
15     import edu.cmu.sphinx.frontend.util.AudioFileDataSource;
16     import edu.cmu.sphinx.recognizer.Recognizer;
17     import edu.cmu.sphinx.result.Lattice;
18     import edu.cmu.sphinx.result.LatticeOptimizer;
19     import edu.cmu.sphinx.result.Result;
20     import edu.cmu.sphinx.util.props.ConfigurationManager;
21     
22     import javax.sound.sampled.UnsupportedAudioFileException;
23     import java.io.File;
24     import java.io.IOException;
25     import java.net.URL;
26     
27     /** A simple Lattice demo showing a simple speech application that generates a Lattice from a recognition result. */
28     public class LatticeDemo {
29     
30     
31         /** Main method for running the Lattice demo. */
32         public static void main(String[] args) throws IOException, UnsupportedAudioFileException {
33             URL audioURL, url;
34             if (args.length > 0) {
35                 audioURL = new File(args[0]).toURI().toURL();
36             } else {
37                 audioURL = LatticeDemo.class.getResource("10001-90210-01803.wav");
38             }
39     
40             if (args.length > 1) {
41                 url = new File(args[1]).toURI().toURL();
42             } else {
43                 url = LatticeDemo.class.getResource("config.xml");
44             }
45     
46             System.out.println("Loading...");
47             ConfigurationManager cm = new ConfigurationManager(url);
48     
49             Recognizer recognizer = (Recognizer) cm.lookup("recognizer");
50             recognizer.allocate();
51     
52             // configure the audio input for the recognizer
53             AudioFileDataSource dataSource = (AudioFileDataSource) cm.lookup("audioFileDataSource");
54             dataSource.setAudioFile(audioURL, null);
55     
56             boolean done = false;
57             while (!done) {
58                 /* This method will return when the end of speech
59                 * is reached. Note that the endpointer will determine
60                 * the end of speech.
61                 */
62                 Result result = recognizer.recognize();
63     
64                 if (result != null) {
65                     Lattice lattice = new Lattice(result);
66                     LatticeOptimizer optimizer = new LatticeOptimizer(lattice);
67                     optimizer.optimize();
68                     lattice.dumpAllPaths();
69                     String resultText = result.getBestResultNoFiller();
70                     System.out.println("I heard: " + resultText + "\n");
71                 } else {
72                     done = true;
73                 }
74             }
75         }
76     }
