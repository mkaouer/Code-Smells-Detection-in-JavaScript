12     /**
13      * Simple extension of RuleName to store hidden resolved rulename.
14      *
15      * @version 1.3 11/25/98 16:51:14
16      */
17     public class BaseRuleName extends RuleName {
18     
19         public BaseRuleName() {
20             super();
21         }
22     
23     
24         public BaseRuleName(String name) {
25             super(name);
26         }
27     
28     
29         /** Fully-qualified rulename generated by linkGrammars method. */
30         public String resolvedRuleName;
31     
32     
33         /** Override clone to copy resolvedRuleName as well. */
34         public Rule copy() {
35             BaseRuleName rn = new BaseRuleName(getRuleName());
36             rn.resolvedRuleName = resolvedRuleName;
37             return rn;
38         }
39     
40     
41         /**
42          * Copy the input rule object but change all instances of a RuleName to a BaseRuleName.  Used by setRule method in
43          * BaseRuleGrammar so there is a place to put the fully resolved rule name when linking a Grammar.
44          */
45         public static Rule copyForBaseRuleName(Rule r) {
46             if (r == null) {
47                 return null;
48             }
49     
50             if (r instanceof RuleToken) {
51                 RuleToken rt = (RuleToken) r;
52                 return new RuleToken(rt.getText());
53             }
54     
55             if (r instanceof RuleAlternatives) {
56                 RuleAlternatives ra = (RuleAlternatives) r;
57                 Rule array[] = ra.getRules();
58                 float[] weights = ra.getWeights();
59                 for (int i = 0; i < array.length; i++) {
60                     array[i] = copyForBaseRuleName(array[i]);
61                 }
62                 return new RuleAlternatives(array, weights);
63             }
64     
65             if (r instanceof RuleSequence) {
66                 RuleSequence rs = (RuleSequence) r;
67                 Rule array[] = rs.getRules();
68                 for (int i = 0; i < array.length; i++) {
69                     array[i] = copyForBaseRuleName(array[i]);
70                 }
71                 return new RuleSequence(array);
72             }
73     
74             if (r instanceof RuleCount) {
75                 RuleCount rc = (RuleCount) r;
76                 return new RuleCount(copyForBaseRuleName(rc.getRule()),
77                         rc.getCount());
78             }
79     
80             if (r instanceof RuleTag) {
81                 RuleTag rt = (RuleTag) r;
82                 return new RuleTag(copyForBaseRuleName(rt.getRule()), rt.getTag());
83             }
84     
85             if (r instanceof RuleName) {
86                 RuleName rn = (RuleName) r;
87                 return new BaseRuleName(rn.getRuleName());
88             }
89     
90             throw new RuntimeException("Unknown rule type");
91         }
92     }
