12     package edu.cmu.sphinx.linguist.util;
13     
14     import edu.cmu.sphinx.linguist.*;
15     
16     import java.util.*;
17     
18     /** A linguist processor that dumps out stats about the search space */
19     public class LinguistStats extends LinguistProcessor {
20     
21         private Map<Class, Integer> stateCountByType = new HashMap<Class, Integer>();
22     
23         /** Dumps the stats of the linguist */
24         public void run() {
25             Linguist linguist = getLinguist();
26             List<SearchState> queue = new LinkedList<SearchState>();
27             Set<SearchState> visitedStates = new HashSet<SearchState>();
28             int stateCount = 0;
29             queue.add(linguist.getSearchGraph().getInitialState());
30             while (queue.size() > 0) {
31                 SearchState state = queue.remove(0);
32                 if (!visitedStates.contains(state)) {
33                     stateCount++;
34                     incrementStateTypeCount(state);
35                     visitedStates.add(state);
36                     SearchStateArc[] arcs = state.getSuccessors();
37                     for (int i = arcs.length - 1; i >= 0; i--) {
38                         SearchState nextState = arcs[i].getState();
39                         queue.add(nextState);
40                     }
41                     // DEBUG: dump out the unique word states
42                     if (false && state instanceof WordSearchState) {
43                         System.out.println("WS: " + state);
44                     }
45                     if (false && state instanceof HMMSearchState) {
46                         System.out.println("HS: " + state);
47                     }
48                 }
49             }
50             System.out.println("# ----------- linguist stats ------------ ");
51             System.out.println("# Total states: " + stateCount);
52             dumpStateTypeCounts();
53         }
54     
55     
56         /**
57          * Keeps track of state counts by class
58          *
59          * @param state the state to track
60          */
61         private void incrementStateTypeCount(SearchState state) {
62             Integer count = stateCountByType.get(state.getClass());
63             if (count == null) {
64                 count = 0;
65             }
66             count = count + 1;
67             stateCountByType.put(state.getClass(), count);
68         }
69     
70     
71         /** Dumps all of the class counts */
72         private void dumpStateTypeCounts() {
73             for (Class clazz  : stateCountByType.keySet()) {
74                 System.out.println("# " + clazz + ": "
75                         + stateCountByType.get(clazz));
76             }
77         }
78     }
