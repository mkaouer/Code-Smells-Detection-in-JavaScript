83         public List<Data> collectOutput(double blocSizeMs) throws DataProcessingException {
84             DataBlocker dataBlocker = new DataBlocker(blocSizeMs);
85             dataBlocker.setPredecessor(this);
86     
87             List<Data> output = new ArrayList<Data>();
88     
89             while (true) {
90                 Data d = dataBlocker.getData();
91                 if (d instanceof DoubleData)
92                     output.add(d);
93                 if (d instanceof DataEndSignal)
94                     return output;
95             }
96         }

99         public static List<DoubleData> createDataInput(int numSamples, int blockSize, int sampleRate, int offSet) {
100             List<DoubleData> datas = new ArrayList<DoubleData>();
101     
102             double counter = 1;
103             for (int i = 0; i < numSamples / blockSize; i++) {
104                 double[] values = new double[blockSize];
105                 datas.add(new DoubleData(values, sampleRate, 0, (long) counter + offSet));
106     
107                 for (int j = 0; j < values.length; j++)
108                     values[j] = counter++ + offSet;
109             }
110     
111             return datas;
112         }

115         /**
116          * Tests wether the samples of all <code>Data</code>s in the list are ordered in increasing order with +1
117          * increments.
118          */
119         public static boolean hasIncreasingOrder(List<Data> output, int lastValue) {
120             double dataCounter = 0;
121     
122             for (Data data : output) {
123                 if (data instanceof DoubleData) {
124                     DoubleData dd = (DoubleData) data;
125     
126                     for (int i = 0; i < dd.getValues().length; i++) {
127                         if ((dataCounter + 1) == dd.getValues()[i])
128                             dataCounter++;
129                         else
130                             return false;
131                     }
132                 }
133             }
134     
135             return dataCounter == lastValue;
136         }
