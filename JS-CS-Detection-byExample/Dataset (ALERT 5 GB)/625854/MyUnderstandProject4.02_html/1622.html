27     /** Provides mechanisms for accessing a next utterance's file name and transcription. */
28     public class SimpleControlFile implements ControlFile {
29     
30         private String audioFile;           // the audio file
31         private String transcriptFile;      // the transcript file
32         private Dictionary dictionary;          // the dictionary
33         private String wordSeparator;       // the word separator
34         private int currentPartition;       // the current partition
35         private int numberOfPartitions;     // total number of partitions
36         private Iterator audioFileIterator; // iterator for the control file
37         private Iterator transcriptFileIterator; // iterator for the transcriptions
38         private List audioFileList;         // list containing the audio files
39         private List transcriptFileList;    // list containing the transcriptions
40     
41     
42         /*
43         * The logger for this class
44         */
45         private Logger logger;
46     
47     
48         public void newProperties(PropertySheet ps) throws PropertyException {
49             logger = ps.getLogger();
50     
51             this.audioFile = ps.getString(PROP_AUDIO_FILE);
52             this.transcriptFile = ps.getString(PROP_TRANSCRIPT_FILE);
53             this.currentPartition = ps.getInt(PROP_WHICH_BATCH);
54             this.numberOfPartitions = ps.getInt(PROP_TOTAL_BATCHES);
55     
56             logger.info("Audio control file: " + this.audioFile);
57             logger.info("Transcript file: " + this.transcriptFile);
58             this.dictionary = new TrainerDictionary();
59             this.wordSeparator = " \t\n\r\f"; // the white spaces
60             logger.info("Processing part " + this.currentPartition +
61                     " of " + this.numberOfPartitions);
62             try {
63                 this.audioFileList = getLines(audioFile);
64             } catch (IOException ioe) {
65                 throw new Error("IOE: Can't open file " + audioFile, ioe);
66             }
67             try {
68                 this.transcriptFileList = getLines(transcriptFile);
69             } catch (IOException ioe) {
70                 throw new Error("IOE: Can't open file " + transcriptFile, ioe);
71             }
72         }
73     
74     
75         /** Gets an iterator for utterances. */
76         public void startUtteranceIterator() {
77             audioFileIterator = audioFileList.iterator();
78             transcriptFileIterator = transcriptFileList.iterator();
79         }
80     
81     
82         /**
83          * Returns whether there is another utterance.
84          *
85          * @return true if there is another utterance.
86          */
87         public boolean hasMoreUtterances() {
88             // Should throw exception or break if one has next and the
89             // other doesn't.
90             return (audioFileIterator.hasNext()
91                     && transcriptFileIterator.hasNext());
92         }
93     
94     
95         /**
96          * Gets the next utterance.
97          *
98          * @return the next utterance.
99          */
100         public Utterance nextUtterance() {
101             logger.fine("processing ext utterance");
102             
103             String utteranceLine = (String) audioFileIterator.next();
104             Utterance utterance = new SimpleUtterance(utteranceLine);
105             String utteranceFilename =
106                     utteranceLine.replaceFirst("^.*/", "").replaceFirst("\\..*$", "");
107             String transcriptLine = (String) transcriptFileIterator.next();
108             // Finds out if the audio file name is part of the transcript line
109             assert transcriptLine.matches(".*[ \t]\\(" + utteranceFilename + "\\)$") :
110                     "File name in transcript \"" + transcriptLine +
111                             "\" and control file \"" + utteranceFilename +
112                             "\" have to match.";
113             // Removes the filename from the transcript line.
114             // The transcript line is of the form:
115             //    She washed her dark suit (st002)
116             String transcript = transcriptLine.replaceFirst("[ \t]\\(.*\\)$", "");
117             utterance.add(transcript, dictionary, false, wordSeparator);
118             return utterance;
119         }

124         /**
125          * Gets the set of lines from the file.
126          *
127          * @param file the name of the file
128          * @throws IOException if error occurs while reading file
129          */
130         private List getLines(String file) throws IOException {
131             List list = new ArrayList();
132             BufferedReader reader
133                     = new BufferedReader(new FileReader(file));
134     
135             String line = null;
136     
137             while ((line = reader.readLine()) != null) {
138                 list.add(line);
139             }
140             reader.close();
141     
142             if (numberOfPartitions > 1) {
143                 int linesPerBatch = list.size() / numberOfPartitions;
144                 if (linesPerBatch < 1) {
145                     linesPerBatch = 1;
146                 }
147                 if (currentPartition >= numberOfPartitions) {
148                     currentPartition = numberOfPartitions - 1;
149                 }
150                 int startLine = currentPartition * linesPerBatch;
151                 // last batch needs to get all remaining lines
152                 if (currentPartition == (numberOfPartitions - 1)) {
153                     list = list.subList(startLine, list.size());
154                 } else {
155                     list = list.subList(startLine, startLine +
156                             linesPerBatch);
157                 }
158             }
159             return list;
160         }
161     }
