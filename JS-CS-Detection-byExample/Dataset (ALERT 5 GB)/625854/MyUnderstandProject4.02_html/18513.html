30     /**
31      * An AudioFileDataSource generates a stream of audio data from a given audion file. All required information concerning
32      * the audio format are read directly from the file . One would need to call {@link #setAudioFile(java.io.File,String)}
33      * to set the input file, and call {@link #getData} to obtain the Data frames.
34      * <p/>
35      * Using JavaSound as backend this class is able to handle all sound files supported by JavaSound. Beside the built-in
36      * support for .wav, .au and .aiff. Using plugins (cf.  http://www.jsresources.org/ ) it can be extended to support
37      * .ogg, .mp3, .speex and others.
38      *
39      * @author Holger Brandl
40      */
41     
42     public class AudioFileDataSource extends BaseDataProcessor {

48         public static final int PROP_BYTES_PER_READ_DEFAULT = 3200;

105          * Sets the audio file from which the data-stream will be generated of.

120         /**
121          * Sets the audio file from which the data-stream will be generated of.
122          *
123          * @param audioFileURL The location of the audio file to use
124          * @param streamName   The name of the InputStream. if <code>null</code> the complete path of the audio file will be
125          *                     uses as stream name.
126          */
127         public void setAudioFile(URL audioFileURL, String streamName) {
128             // first close the last stream if there's such a one
129             if (dataStream != null) {
130                 try {
131                     dataStream.close();
132                 } catch (IOException e) {
133                     e.printStackTrace();
134                 }
135     
136                 dataStream = null;
137             }
138     
139             assert audioFileURL != null;
140             if (streamName != null)
141                 streamName = audioFileURL.getPath();
142     
143             AudioInputStream audioStream = null;
144             try {
145                 audioStream = AudioSystem.getAudioInputStream(audioFileURL);
146             } catch (UnsupportedAudioFileException e) {
147                 System.err.println("Audio file format not supported: " + e);
148                 e.printStackTrace();
149             } catch (IOException e) {
150                 e.printStackTrace();
151             }
152     
153             curAudioFile = new File(audioFileURL.getFile());
154             for (AudioFileProcessListener fileListener : fileListeners)
155                 fileListener.audioFileProcStarted(curAudioFile);
156     
157             setInputStream(audioStream, streamName);
158         }

161         /**
162          * Sets the InputStream from which this StreamDataSource reads.
163          *
164          * @param inputStream the InputStream from which audio data comes
165          * @param streamName  the name of the InputStream
166          */
167         public void setInputStream(AudioInputStream inputStream, String streamName) {
168             dataStream = inputStream;
169             streamEndReached = false;
170             utteranceEndSent = false;
171             utteranceStarted = false;
172     
173             AudioFormat format = inputStream.getFormat();
174             sampleRate = (int) format.getSampleRate();
175             bigEndian = format.isBigEndian();
176     
177             String s = format.toString();
178             logger.finer("input format is " + s);
179     
180             if (format.getSampleSizeInBits() % 8 != 0)
181                 throw new Error("StreamDataSource: bits per sample must be a multiple of 8.");
182             bytesPerValue = format.getSampleSizeInBits() / 8;
183     
184             // test wether all files in the stream have the same format
185     
186             AudioFormat.Encoding encoding = format.getEncoding();
187             if (encoding.equals(AudioFormat.Encoding.PCM_SIGNED))
188                 signedData = true;
189             else if (encoding.equals(AudioFormat.Encoding.PCM_UNSIGNED))
190                 signedData = false;
191             else
192                 throw new RuntimeException("used file encoding is not supported");
193     
194             totalValuesRead = 0;
195         }

198         /**
199          * Reads and returns the next Data from the InputStream of StreamDataSource, return null if no data is read and end
200          * of file is reached.
201          *
202          * @return the next Data or <code>null</code> if none is available
203          * @throws edu.cmu.sphinx.frontend.DataProcessingException
204          *          if there is a data processing error
205          */
206         public Data getData() throws DataProcessingException {
207             getTimer().start();
208             Data output = null;
209             if (streamEndReached) {
210                 if (!utteranceEndSent) {
211                     // since 'firstSampleNumber' starts at 0, the last
212                     // sample number should be 'totalValuesRead - 1'
213                     output = createDataEndSignal();
214                     utteranceEndSent = true;
215                 }
216             } else {
217                 if (!utteranceStarted) {
218                     utteranceStarted = true;
219                     output = new DataStartSignal(sampleRate);
220                 } else {
221                     if (dataStream != null) {
222                         output = readNextFrame();
223                         if (output == null) {
224                             if (!utteranceEndSent) {
225                                 output = createDataEndSignal();
226                                 utteranceEndSent = true;
227                             }
228                         }
229                     }
230                 }
231             }
232             getTimer().stop();
233             return output;
234         }

237         private DataEndSignal createDataEndSignal() {
238             if (!(this instanceof ConcatAudioFileDataSource))
239                 for (AudioFileProcessListener fileListener : fileListeners)
240                     fileListener.audioFileProcFinished(curAudioFile);
241     
242             return new DataEndSignal(getDuration());
243         }

246         /**
247          * Returns the next Data from the input stream, or null if there is none available
248          *
249          * @return a Data or null
250          * @throws java.io.IOException
251          */
252         private Data readNextFrame() throws DataProcessingException {
253             // read one frame's worth of bytes
254             int read;
255             int totalRead = 0;
256             final int bytesToRead = bytesPerRead;
257             byte[] samplesBuffer = new byte[bytesPerRead];
258             long collectTime = System.currentTimeMillis();
259             long firstSample = totalValuesRead;
260             try {
261                 do {
262                     read = dataStream.read(samplesBuffer, totalRead, bytesToRead
263                             - totalRead);
264                     if (read > 0) {
265                         totalRead += read;
266                     }
267                 } while (read != -1 && totalRead < bytesToRead);
268                 if (totalRead <= 0) {
269                     closeDataStream();
270                     return null;
271                 }
272                 // shrink incomplete frames
273                 totalValuesRead += (totalRead / bytesPerValue);
274                 if (totalRead < bytesToRead) {
275                     totalRead = (totalRead % 2 == 0)
276                             ? totalRead + 2
277                             : totalRead + 3;
278                     byte[] shrinkedBuffer = new byte[totalRead];
279                     System
280                             .arraycopy(samplesBuffer, 0, shrinkedBuffer, 0,
281                                     totalRead);
282                     samplesBuffer = shrinkedBuffer;
283                     closeDataStream();
284                 }
285             } catch (IOException ioe) {
286                 ioe.printStackTrace();
287                 throw new DataProcessingException("Error reading data");
288             }
289             // turn it into an Data object
290             double[] doubleData;
291             if (bigEndian) {
292                 doubleData = DataUtil.bytesToValues(samplesBuffer, 0, totalRead, bytesPerValue, signedData);
293             } else {
294                 doubleData = DataUtil.littleEndianBytesToValues(samplesBuffer, 0, totalRead, bytesPerValue, signedData);
295             }
296     
297             return new DoubleData(doubleData, sampleRate, collectTime, firstSample);
298         }

301         private void closeDataStream() throws IOException {
302             streamEndReached = true;
303             if (dataStream != null) {
304                 dataStream.close();
305             }
306         }

309         /**
310          * Returns the duration of the current data stream in milliseconds.
311          *
312          * @return the duration of the current data stream in milliseconds
313          */
314         private long getDuration() {
315             return (long) (((double) totalValuesRead / (double) sampleRate) * 1000.0);
316         }
