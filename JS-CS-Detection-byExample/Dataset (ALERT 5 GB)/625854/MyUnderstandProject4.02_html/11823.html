26     /**
27      * Plots positive energy values of a cepstrum to stdout. The energy value is assumed to be the first element of the
28      * double array returned by <code>Data.getValues()</code>. For negative energy value, a "-" will be printed out. The
29      * plots look like the following, one line per cepstrum. The energy value for that particular cepstrum is printed at the
30      * end of the line.
31      * <p/>
32      * <code> <br>+......7 <br>+......7 <br>Cepstrum: SPEECH_START <br>+......7 <br>+.......8 <br>+......7 <br>+.......8
33      * <br>+.......8 <br>+........9 <br>+............14 <br>+...........13 <br>+...........13 <br>+...........13
34      * <br>+.............15 <br>+.............15 <br>+..............16 <br>+..............16 <br>+..............16
35      * <br>+.............15 <br>+............14 <br>+............14 <br>+............14 <br>+............14
36      * <br>+.............15 <br>+..............16 <br>+...............17 <br>+...............17 <br>+...............17
37      * <br>+...............17 <br>+...............17 <br>+...............17 <br>+..............16 <br>+.............15
38      * <br>+............14 <br>+............14 <br>+............14 <br>+...........13 <br>+........9 <br>+.......8
39      * <br>+......7 <br>+......7 <br>+......7 <br>Cepstrum: SPEECH_END <br>+......7 </code>
40      */
41     public class EnergyPlotter implements Configurable {
42     
43         /** The maximum level of energy for which a plot string will be preconstructed. */
44         @S4Integer(defaultValue = 20)
45         public static final String PROP_MAX_ENERGY = "maxEnergy";
46     
47         private int maxEnergy;
48         private String[] plots;
49     
50     
51         public void newProperties(PropertySheet ps) throws PropertyException {
52             maxEnergy = ps.getInt(PROP_MAX_ENERGY);
53             buildPlots(maxEnergy);
54         }
55     
56         /**
57          * Builds the strings for the plots.
58          *
59          * @param maxEnergy the maximum energy value
60          */
61         private void buildPlots(int maxEnergy) {
62             plots = new String[maxEnergy + 1];
63             for (int i = 0; i < maxEnergy + 1; i++) {
64                 plots[i] = getPlotString(i);
65             }
66         }
67     
68     
69         /**
70          * Returns the plot string for the given energy.
71          *
72          * @param energy the energy level
73          */
74         private String getPlotString(int energy) {
75             char[] plot = new char[energy];
76             Arrays.fill(plot, '.');
77             if (energy > 0) {
78                 if (energy < 10) {
79                     plot[plot.length - 1] = (char) ('0' + energy);
80                 } else {
81                     plot[plot.length - 2] = '1';
82                     plot[plot.length - 1] = (char) ('0' + (energy - 10));
83                 }
84             }
85             return ("+" + new String(plot));
86         }
87     
88     
89         /**
90          * Plots the energy values of the given Data to System.out. If the Data contains a signal, it prints the signal.
91          *
92          * @param cepstrum the Data to plot
93          */
94         public void plot(Data cepstrum) {
95             if (cepstrum != null) {
96                 if (cepstrum instanceof DoubleData) {
97                     int energy = (int) ((DoubleData) cepstrum).getValues()[0];
98                     System.out.println(getPlot(energy));
99                 } else {
100                     System.out.println(cepstrum.toString());
101                 }
102             }
103         }
104     
105     
106         /**
107          * Returns the corresponding plot String for the given energy value. The energy value should be positive or zero. If
108          * its negative, It will output the string "-".
109          *
110          * @return energy the energy value
111          */
112         private String getPlot(int energy) {
113             if (energy < 0) {
114                 return "-";
115             } else if (energy <= maxEnergy) {
116                 return plots[energy];
117             } else {
118                 return getPlotString(energy);
119             }
120         }
121     }
