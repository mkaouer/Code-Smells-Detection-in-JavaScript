14     package edu.cmu.sphinx.frontend.frequencywarp;
15     
16     
17     /**
18      * Defines a triangular mel-filter. The {@link edu.cmu.sphinx.frontend.frequencywarp.MelFrequencyFilterBank} creates
19      * mel-filters and filters spectrum data using the method {@link #filterOutput(double[]) filterOutput}.
20      * <p/>
21      * A mel-filter is a triangular shaped bandpass filter.  When a mel-filter is constructed, the parameters
22      * <code>leftEdge</code>, <code>rightEdge</code>, <code>centerFreq</code>, <code>initialFreq</code>, and
23      * <code>deltaFreq</code> are given to the {@link MelFilter Constructor}. The first three arguments to the constructor,
24      * i.e. <code>leftEdge</code>, <code>rightEdge</code>, and <code>centerFreq</code>, specify the filter's slopes. The
25      * total area under the filter is 1. The filter is shaped as a triangle. Knowing the distance between the center
26      * frequency and each of the edges, it is easy to compute the slopes of the two sides in the triangle - the third side
27      * being the frequency axis. The last two arguments, <code>initialFreq</code> and <code>deltaFreq</code>, identify the
28      * first frequency bin that falls inside this filter and the spacing between successive frequency bins. All frequencies
29      * here are considered in a linear scale.
30      * <p/>
31      * Figure 1 below shows pictorially what the other parameters mean.
32      * <p/>
33      * <img src="doc-files/melfilter.jpg"> <br><center><b>Figure 1: A triangular mel-filter.</b></center>
34      *
35      * @see MelFrequencyFilterBank
36      */
37     public class MelFilter {
38     
39         private double[] weight;
40     
41         private int initialFreqIndex;
42     
43     
44         /**
45          * Constructs a filter from the parameters.
46          * <p/>
47          * In the current implementation, the filter is a bandpass filter with a triangular shape.  We're given the left and
48          * right edges and the center frequency, so we can determine the right and left slopes, which could be not only
49          * assymmetric but completely different. We're also given the initial frequency, which may or may not coincide with
50          * the left edge, and the frequency step.
51          *
52          * @param leftEdge    the filter's lowest passing frequency
53          * @param centerFreq  the filter's center frequency
54          * @param rightEdge   the filter's highest passing frequency
55          * @param initialFreq the first frequency bin in the pass band
56          * @param deltaFreq   the step in the frequency axis between frequency bins
57          * @throws IllegalArgumentException
58          */
59         public MelFilter(double leftEdge,
60                          double centerFreq,
61                          double rightEdge,
62                          double initialFreq,
63                          double deltaFreq) throws IllegalArgumentException {
64     
65             double filterHeight;
66             double leftSlope;
67             double rightSlope;
68             double currentFreq;
69             int indexFilterWeight;
70             int numberElementsWeightField;
71     
72             if (deltaFreq == 0) {
73                 throw new IllegalArgumentException("deltaFreq has zero value");
74             }
75             /**
76              * Check if the left and right boundaries of the filter are
77              * too close.
78              */
79             if ((Math.round(rightEdge - leftEdge) == 0)
80                     || (Math.round(centerFreq - leftEdge) == 0)
81                     || (Math.round(rightEdge - centerFreq) == 0)) {
82                 throw new IllegalArgumentException("Filter boundaries too close");
83             }
84             /**
85              * Let's compute the number of elements we need in the
86              * <code>weight</code> field by computing how many frequency
87              * bins we can fit in the current frequency range.
88              */
89             numberElementsWeightField =
90                     (int) Math.round((rightEdge - leftEdge) / deltaFreq + 1);
91             /**
92              * Initialize the <code>weight</code> field.
93              */
94             if (numberElementsWeightField == 0) {
95                 throw new IllegalArgumentException("Number of elements in mel"
96                         + " is zero.");
97             }
98             weight = new double[numberElementsWeightField];
99     
100             /**
101              * Let's make the filter area equal to 1.
102              */
103             filterHeight = 2.0f / (rightEdge - leftEdge);
104     
105             /**
106              * Now let's compute the slopes based on the height.
107              */
108             leftSlope = filterHeight / (centerFreq - leftEdge);
109             rightSlope = filterHeight / (centerFreq - rightEdge);
110     
111             /**
112              * Now let's compute the weight for each frequency bin.  We
113              * initialize and update two variables in the <code>for</code>
114              * line.
115              */
116             for (currentFreq = initialFreq, indexFilterWeight = 0;
117                  currentFreq <= rightEdge;
118                  currentFreq += deltaFreq, indexFilterWeight++) {
119                 /**
120                  * A straight line that contains point <b>(x0, y0)</b> and
121                  * has slope <b>m</b> is defined by:
122                  *
123                  * <b>y = y0 + m * (x - x0)</b>
124                  *
125                  * This is used for both "sides" of the triangular filter
126                  * below.
127                  */
128                 if (currentFreq < centerFreq) {
129                     weight[indexFilterWeight] = leftSlope
130                             * (currentFreq - leftEdge);
131                 } else {
132                     weight[indexFilterWeight] = filterHeight + rightSlope
133                             * (currentFreq - centerFreq);
134                 }
135             }
136             /**
137              * Initializing frequency related fields.
138              */
139             this.initialFreqIndex = (int) Math.round
140                     (initialFreq / deltaFreq);
141         }
142     
143     
144         /**
145          * Compute the output of a filter. We're given a power spectrum, to which we apply the appropriate weights.
146          *
147          * @param spectrum the input power spectrum to be filtered
148          * @return the filtered value, in fact a weighted average of power in the frequency range of the filter pass band
149          */
150         public double filterOutput(double[] spectrum) {
151             double output = 0.0f;
152             int indexSpectrum;
153     
154             for (int i = 0; i < this.weight.length; i++) {
155                 indexSpectrum = this.initialFreqIndex + i;
156                 if (indexSpectrum < spectrum.length) {
157                     output += spectrum[indexSpectrum] * this.weight[i];
158                 }
159             }
160             return output;
161         }
162     }
