182         /* the delete button is pressed, should delete the row that's selected from table */
183         private void ButtonDeleteActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_ButtonDeleteActionPerformed
184             ListSelectionModel mySelectionModel = TableParameter.getSelectionModel();
185             DefaultTableModel myModel = (DefaultTableModel)TableParameter.getModel();
186             
187             int myIndex = mySelectionModel.getMinSelectionIndex();
188             if (myIndex == -1 || myIndex == myModel.getRowCount() ){ // nothing is selected in the table
189                 displayError("The are nothing to be deleted");
190             }
191             else if (myModel.getRowCount() == 0){
192                 displayError("There are no data in the table");
193             }
194             else {
195                 if(myModel != null)
196                     removeTableData(myIndex,myModel);
197             }
198             
199         }//GEN-LAST:event_ButtonDeleteActionPerformed

225         /**
226          * add a row to the table, consisting property value and property name 
227          */
228         private void addTableData(String propName, String propVal){
229             DefaultTableModel model = (DefaultTableModel)TableParameter.getModel();
230             Vector myData = new Vector(2);
231             myData.add(propName);
232             myData.add(propVal);
233             model.insertRow(model.getRowCount(),myData);
234             TableParameter.revalidate();
235            
236         }

238         /**
239          * delete all rows from the table
240          */
241         public void clearAll() {
242             
243             DefaultTableModel model =(DefaultTableModel)TableParameter.getModel();
244             int numRows = model.getRowCount();
245             
246             for (int i=numRows-1;i>=0;i--) {            
247                removeTableData(i,model);
248             }      
249         }

251         /**
252          * update the table according to the configProperties data
253          */
254         public void update(ConfigProperties cp) {
255             Map myglobal=cp.getGlobal();
256             this.clearAll(); 
257             // iterate each item in the global properties Map, and display it
258             for ( Iterator it = myglobal.entrySet().iterator(); it.hasNext();){
259                 Map.Entry entry = (Map.Entry) it.next();            
260                 String name = (String)entry.getKey();
261                 String value = (String)entry.getValue();
262                 addTableData(name,value);
263             }        
264         }

268         public void saveData(ConfigProperties cp) throws GUIOperationException {
269             Map globalMap;
270             if (cp.getGlobal() == null)
271                 globalMap = new HashMap();
272             else
273                 globalMap = cp.getGlobal();
274                 
275             iterateTable(globalMap);
276             cp.setGlobal(globalMap);
277     
278         }

284         /**
285          * iterate through the Table and verify the values to make sure that they are valid
286          * 
287          * @throws GUIOperationException
288          */
289         private void iterateTable(Map globalMap)throws GUIOperationException{
290             DefaultTableModel model =(DefaultTableModel)TableParameter.getModel();
291             String propName;
292             String propVal;
293             int retval;
294             
295             int numRows = model.getRowCount();
296             
297             for (int i=0;i<numRows;i++) {            
298                 propName = ((String)TableParameter.getValueAt(i,0)).trim();
299                 propVal = ((String)TableParameter.getValueAt(i,1)).trim();
300                 
301                 System.out.print("validating " + propName + " and " + propVal);
302                 retval = validateProp(propName,propVal);
303                 switch(retval){
304                     case MODE_OK:
305                         globalMap.put(propName,propVal);
306                         break;
307                     case MODE_ERROR:
308                         displayError("File Save error : Global property row " + (i+1) + " is invalid ");
309                         throw new GUIOperationException(GUIOperationException.EXCEPTION_GLOBAL,
310                                 "Data from Global property table is invalid");                  
311                     case MODE_SKIP:
312                         break;
313                 }
314             }
315     
316         }
317         
318         /** check if there's a valid name and value for each global property in the table
319          */
320         private int validateProp(String propName, String propVal)
321         {
322              if( propName == null || propName.trim().equalsIgnoreCase("") )
323              {
324                  // if the name is empty, then value must be empty too
325                  if(propVal == null || propVal.trim().equalsIgnoreCase("") )
326                      return MODE_SKIP;
327                  else
328                      return MODE_ERROR;
329              }
330              else // name is not empty, check the prop value 
331              {
332                  // name is not empty, must check if it's a one word name
333                  if( propName.trim().indexOf(" ") == -1 ){
334                     //no space in propName, check propValue
335                     if( propVal!= null && propVal.trim().indexOf(" ") == -1 )
336                         return MODE_OK;
337                     else
338                         return MODE_ERROR;
339                  }                 
340                  else // there are space(s) in the property name
341                      return MODE_ERROR;
342              }             
343                         
344         }
