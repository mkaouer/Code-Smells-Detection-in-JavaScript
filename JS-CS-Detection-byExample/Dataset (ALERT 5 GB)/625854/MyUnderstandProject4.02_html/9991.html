14     package edu.cmu.sphinx.demo.jsapi.cardmatch;
15     
16     import java.util.Collections;
17     import java.util.Iterator;
18     import java.util.LinkedList;
19     import java.util.List;
20     
21     /**
22      * A CardMatch Game.
23      */
24     public class Game {
25         private List cards;
26     
27     
28         /**
29          * Constructs a Card with the following image file.
30          *
31          * @param numberOfCards the number of cards to use in the game
32          * @param imageFile the array of image file for the cards
33          */
34         public Game(int numberOfCards, List imageFiles) {
35             cards = createCards(numberOfCards, imageFiles);
36             startOver();
37         }
38     
39     
40         /**
41          * Return the list of Cards.
42          *
43          * @return the list of Cards
44          */
45         public List getCards() {
46             return cards;
47         }
48     
49     
50         /**
51          * Creates a list of cards.
52          *
53          * @param numberOfCards the number of cards in the created list
54          * @param imageFiles the image files of people to use
55          *
56          * @return a list of cards
57          */
58         private List createCards(int numberOfCards, List imageFiles) {
59             List list = new LinkedList();
60             for (int i = 0; i < numberOfCards; i++) {
61                 int whichImage = i % imageFiles.size();
62                 Card card = new Card((String) imageFiles.get(whichImage));
63                 list.add(card);
64             }
65             return list;
66         }
67     
68     
69         /**
70          * Starts a new game.
71          */
72         public void startOver() {
73             Collections.shuffle(cards);
74             int id = 1;
75             for (Iterator i = cards.iterator(); i.hasNext(); id++) {
76                 Card card = (Card) i.next();
77                 card.setID(String.valueOf(id));
78                 card.setDefaultImageFile(id + ".gif");
79                 card.setMatched(false);
80                 card.setSelected(false);
81             }
82         }
83     
84     
85         /**
86          * Returns the Card of the given ID, which can be "1", "2", ... "6".
87          *
88          * @param id the card id
89          * @return the Card of the given ID
90          */
91         public Card getCard(String id) {
92             for (Iterator i = cards.iterator(); i.hasNext(); ) {
93                 Card card = (Card) i.next();
94                 if (card.getID().equals(id)) {
95                     return card;
96                 }
97             }
98             return null;
99         }
100     
101     
102         /**
103          * Mark all cards matching the given ID as unmatched
104          *
105          * @param id the card id
106          */
107         public void unsetMatch(Card card) {
108             for (Iterator i = cards.iterator(); i.hasNext(); ) {
109                 Card otherCard = (Card) i.next();
110                 if (card.isMatch(otherCard)) {
111                     otherCard.setMatched(false);
112                 }
113             }
114         }
115     
116     
117         /**
118          * Turns over the card with the given ID, which can be "1", "2" ... "6".
119          *
120          * @param cardID the card to turn over
121          */
122         public void turnCard(String cardID) {
123             Card card = getCard(cardID);
124             if (card != null && !card.isMatched()) {
125                 card.setSelected(true);
126             }
127         }
128     
129     
130         /**
131          * Determines if the given card is already selected
132          * 
133          * @param cardID the card
134          *
135          * @return <code>true</code>  if the card is already selected
136          */
137         public boolean isSelected(String cardID) {
138             return getCard(cardID).isSelected();
139         }
140     
141         /**
142          * Checks if there are any matches.
143          *
144          * @return <code>true</code>  if the last two selected 
145          * cards match, false otherwise
146          */
147         public boolean processMatches() {
148             Card card0 = findUnmatchedCard(0);
149             Card card1 = findUnmatchedCard(1);
150     
151             if (card0 != null && card1 != null) {
152                 if (card0.isMatch(card1)) {
153                     card0.setMatched(true);
154                     card1.setMatched(true);
155                     return true;
156                 }
157             }
158             return false;
159         }
160     
161     
162         /**
163          * finds the first unmatched card
164          *
165          * @param which which card to find
166          *
167          * @return the first unmatched card
168          */
169         private Card findUnmatchedCard(int which) {
170             int count = 0;
171             for (Iterator i = cards.iterator(); i.hasNext(); ) {
172                 Card card = (Card) i.next();
173                 if (card.isSelected() && !card.isMatched()) {
174                     if (count == which) {
175                         return card;
176                     }
177                     count++;
178                 }
179             }
180             return null;
181         }
182             
183     
184     
185         /**
186          * Turn back the last two cards that were guessed.
187          */
188         public void turnGuessedCards() {
189             for (Iterator i = cards.iterator(); i.hasNext(); ) {
190                 Card card = (Card) i.next();
191                 if (!card.isMatched()) {
192                     card.setSelected(false);
193                 }
194             }
195         }
196     
197     
198         /**
199          * Returns the number of selected, but unmatched cares
200          *
201          * @return the number of unselected cards
202          */
203         public int getNumSelected() {
204             int numSelected = 0;
205             for (Iterator i = cards.iterator(); i.hasNext(); ) {
206                 Card card = (Card) i.next();
207                 if (card.isSelected() && !card.isMatched()) {
208                     numSelected ++;
209                 }
210             }
211             return numSelected;
212         }
213     
214     
215         /**
216          * Returns true if this game has been won.
217          *
218          * @return true if this game has been won
219          */
220         public boolean hasWon() {
221             for (Iterator i = cards.iterator(); i.hasNext(); ) {
222                 Card card = (Card) i.next();
223                 if (!card.isMatched()) {
224                     return false;
225                 }
226             }
227             return true;
228         }
229     }
