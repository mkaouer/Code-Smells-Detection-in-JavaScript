21     /**
22      * Computes confidences for the highest scoring path in a Result. The highest scoring path refers to the path with the
23      * maximum a posteriori (MAP) probability, which is why this class is so named. Note that this MAPConfidenceScorer
24      * creates a {@link edu.cmu.sphinx.result.Lattice} from the result first, which means that you should only use this
25      * confidence scorer if the result is created from the {@link edu.cmu.sphinx.linguist.lextree.LexTreeLinguist} and the
26      * {@link edu.cmu.sphinx.decoder.search.WordPruningBreadthFirstSearchManager}.
27      */
28     public class MAPConfidenceScorer implements ConfidenceScorer, Configurable {
29     
30         /** Sphinx property that defines the language model weight. */
31         @S4Double(defaultValue = 1.0)
32         public final static String PROP_LANGUAGE_WEIGHT = "languageWeight";
33     
34     
35         /** Sphinx property that specifies whether to dump the lattice. */
36         @S4Boolean(defaultValue = false)
37         public final static String PROP_DUMP_LATTICE = "dumpLattice";
38     
39     
40         /** Sphinx property that specifies whether to dump the sausage. */
41         @S4Boolean(defaultValue = false)
42         public final static String PROP_DUMP_SAUSAGE = "dumpSausage";
43     
44     
45         private float languageWeight;
46         private boolean dumpLattice;
47         private boolean dumpSausage;
48     
49     
50         /*
51          * (non-Javadoc)
52          *
53          * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
54          */
55         public void newProperties(PropertySheet ps) throws PropertyException {
56             languageWeight = ps.getFloat(PROP_LANGUAGE_WEIGHT);
57             dumpLattice = ps.getBoolean(PROP_DUMP_LATTICE);
58             dumpSausage = ps.getBoolean(PROP_DUMP_SAUSAGE);
59         }
60     
61     
62         /**
63          * Computes confidences for a Result and returns a ConfidenceResult, a compact representation of all the hypothesis
64          * contained in the result together with their per-word and per-path confidences.
65          *
66          * @param result the result to compute confidences for
67          * @return a confidence result
68          */
69         public ConfidenceResult score(Result result) {
70             Lattice lattice = new Lattice(result);
71             LatticeOptimizer lop = new LatticeOptimizer(lattice);
72             lop.optimize();
73             lattice.computeNodePosteriors(languageWeight);
74             SausageMaker sm = new SausageMaker(lattice);
75             Sausage s = sm.makeSausage();
76     
77             if (dumpLattice) {
78                 lattice.dumpAISee("mapLattice.gdl", "MAP Lattice");
79             }
80             if (dumpSausage) {
81                 s.dumpAISee("mapSausage.gdl", "MAP Sausage");
82             }
83     
84             WordResultPath mapPath = new WordResultPath();
85             List<Token> wordTokens = getWordTokens(result.getBestToken());
86     
87             /* start with the first slot */
88             int slot = 0;
89     
90             for (Token wordToken : wordTokens) {
91                 String word = wordToken.getWord().getSpelling();
92                 WordResult wr = null;
93                 ConfusionSet cs = null;
94     
95                 /* track through all the slots to find the word */
96                 while (slot < s.size() && wr == null) {
97                     cs = s.getConfusionSet(slot);
98                     wr = cs.getWordResult(word);
99                     if (wr == null) {
100                         slot++;
101                     }
102                 }
103                 if (wr != null) {
104                     mapPath.add(wr);
105                 } else {
106                     cs.dump("Slot " + slot);
107                     throw new Error
108                             ("Can't find WordResult in ConfidenceResult slot " +
109                                     slot + " for word " + word);
110                 }
111                 slot++;
112             }
113     
114             return (new MAPConfidenceResult(s, mapPath));
115         }
116     
117     
118         /**
119          * Returns all the word tokens ending at the given token as a List.
120          *
121          * @param lastToken the last token in the token chain
122          * @return a list of word tokens in order of appearance
123          */
124         private List<Token> getWordTokens(Token lastToken) {
125             List<Token> wordTokens = new LinkedList<Token>();
126             Token token = lastToken;
127             while (token != null) {
128                 if (token.isWord()) {
129                     wordTokens.add(0, token);
130                 }
131                 token = token.getPredecessor();
132             }
133             return wordTokens;
134         }

137         /** The confidence result for the highest scoring path. */
138         class MAPConfidenceResult implements ConfidenceResult {
139     
140             private ConfidenceResult sausage;
141             private Path mapPath;
142     
143     
144             /**
145              * Constructs a MAPConfidenceResult.
146              *
147              * @param sausage the sausge that this MAPConfidenceResult is based on
148              * @param mapPath the maximum posterior probability path
149              */
150             public MAPConfidenceResult(ConfidenceResult sausage, Path mapPath) {
151                 this.sausage = sausage;
152                 this.mapPath = mapPath;
153             }
154     
155     
156             /**
157              * Returns the path with the maximum posterior probability path. This path should be the same as that returned
158              * by Result.getBestToken().
159              */
160             public Path getBestHypothesis() {
161                 return mapPath;
162             }
163     
164     
165             /**
166              * Get the number of word slots contained in this result
167              *
168              * @return length of the result
169              */
170             public int size() {
171                 return sausage.size();
172             }
173     
174     
175             /**
176              * Iterator through the confusion sets in this result.
177              *
178              * @return confusion set iterator
179              */
180             public Iterator confusionSetIterator() {
181                 return sausage.confusionSetIterator();
182             }
183     
184     
185             /**
186              * Get the nth confusion set in this result
187              *
188              * @param i the index of the confusion set to get
189              * @return the requested confusion set
190              */
191             public ConfusionSet getConfusionSet(int i) {
192                 return sausage.getConfusionSet(i);
193             }
194         }
