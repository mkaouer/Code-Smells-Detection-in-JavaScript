52          * @param url The location of the configuration file.

55             configURL = url;
56             SaxLoader saxLoader = new SaxLoader(url, globalProperties, null, false);
57     
58             try {
59                 rawPropertyMap = saxLoader.load();
60             } catch (IOException e) {
61                 throw new RuntimeException(e);
62             }
63     
64             ConfigurationManagerUtils.applySystemProperties(rawPropertyMap, globalProperties);
65             ConfigurationManagerUtils.configureLogger(this);

67             // we can't config the configuration manager with itself so we
68             // do some of these config items manually.

76          * Returns the property sheet for the given object instance
77          *
78          * @param instanceName the instance name of the object
79          * @return the property sheet for the object.
80          */
81         public PropertySheet getPropertySheet(String instanceName) {
82             if (!symbolTable.containsKey(instanceName)) {
83                 // if it is not in the symbol table, so construct
84                 // it based upon our raw property data
85                 RawPropertyData rpd = rawPropertyMap.get(instanceName);
86                 if (rpd != null) {
87                     String className = rpd.getClassName();
88                     try {
89                         Class cls = Class.forName(className);
90     
91                         // now load the property-sheet by using the class annotation
92     //                    PropertySheet propertySheet = new PropertySheet(cls, this, rpd.flatten(globalProperties));
93                         PropertySheet propertySheet = new PropertySheet(cls, instanceName, this, rpd);
94     
95                         symbolTable.put(instanceName, propertySheet);
96     
97                     } catch (ClassNotFoundException e) {
98                         System.err.println("class not found !" + e.toString());
99                     } catch (ClassCastException e) {
100                         System.err.println("can not cast class !" + e.toString());
101                     } catch (ExceptionInInitializerError e) {
102                         System.err.println("couldn't load class !" + e.toString());
103                     }
104                 }
105             }
106     
107             return symbolTable.get(instanceName);
108         }

113          *
114          * @param type the desired type of instance
115          * @return the set of all instances
116          */
117         public Collection<String> getInstanceNames(Class<? extends Configurable> type) {
118             Collection<String> instanceNames = new ArrayList<String>();
119     
120             for (PropertySheet ps : symbolTable.values()) {
121                 if (!ps.isInstanciated())
122                     continue;
123     
124                 if (ConfigurationManagerUtils.isDerivedClass(ps.getConfigurableClass(), type))
125                     instanceNames.add(ps.getInstanceName());
126             }
127     
128             return instanceNames;
129         }

198          * Registers a new configurable to this configuration manager.
199          *
200          * @param confClass The class of the configurable to be instantiated and to be added to this configuration manager
201          *                  instance.
202          * @param name      The desired  lookup-name of the configurable
203          * @param props     The properties to be used for component configuration
204          * @throws IllegalArgumentException if the there's already a component with the same <code>name</code> registered to
205          *                                  this configuration manager instance.
206          */
207         public void addConfigurable(Class<? extends Configurable> confClass, String name, Map<String, Object> props) {
208             if (name == null) // use the class name as default if no name is given
209                 name = confClass.getName();
210     
211             if (symbolTable.containsKey(name))
212                 throw new IllegalArgumentException("tried to override existing component name");
213     
214             PropertySheet ps = getPropSheetInstanceFromClass(confClass, props, name, this);
215             symbolTable.put(name, ps);
216             rawPropertyMap.put(name, new RawPropertyData(name, confClass.getName()));
217     
218             for (ConfigurationChangeListener changeListener : changeListeners)
219                 changeListener.componentAdded(this, ps);
220         }

244             PropertySheet ps = getPropertySheet(oldName);

280         /**
281          * Adds a subconfiguration to this instance by registering all subCM-components and all its global properties.
282          *
283          * @param subCM                The subconfiguration that shouwld be  to this instance
284          * @param doOverrideComponents If <code>true</code> non-instantiated components will be overriden by elements of
285          *                             subCM even if already being registered to this CM-instance. The same holds for global
286          *                             properties.
287          * @throws RuntimeException if an already instantiated component in this instance is redefined in subCM.
288          */
289         public void addSubConfiguration(ConfigurationManager subCM, boolean doOverrideComponents) {
290             Collection<String> compNames = getComponentNames();
291     
292             for (String componentName : subCM.getComponentNames()) {
293                 if (compNames.contains(componentName)) {
294                     if (doOverrideComponents && !getPropertySheet(componentName).isInstanciated()) {
295                         PropertySheet ps = subCM.getPropertySheet(componentName);
296                         symbolTable.put(componentName, ps);
297                         rawPropertyMap.put(componentName, new RawPropertyData(componentName, ps.getConfigurableClass().getSimpleName()));
298     
299                     } else {
300                         throw new RuntimeException(componentName + " is already registered to system configuration");
301                     }
302                 }
303             }
304     
305             for (String globProp : subCM.globalProperties.keySet()) {
306                 // the second test is necessary because system-props will be global-props in both CMs
307                 if (globalProperties.keySet().contains(globProp) && !System.getProperties().keySet().contains(globProp)) {
308                     if (!doOverrideComponents)
309                         throw new RuntimeException(globProp + " is already registered as global property");
310                 }
311             }
312     
313             globalProperties.putAll(subCM.globalProperties);
314     
315             // correct the reference to the cm
316             for (PropertySheet ps : subCM.symbolTable.values()) {
317                 ps.setCM(this);
318             }
319     
320             symbolTable.putAll(subCM.symbolTable);
321             rawPropertyMap.putAll(subCM.rawPropertyMap);
322         }

358          * Sets a global property.
359          *
360          * @param propertyName The name of the global property.
361          * @param value        The new value of the global property. If the value is <code>null</code> the property becomes
362          *                     removed.
363          */
364         public void setGlobalProperty(String propertyName, String value) {
365             if (value == null)
366                 globalProperties.remove(propertyName);
367             else
368                 globalProperties.setValue(propertyName, value);
369     
370             // update all component configurations because they might be affected by the change
371             for (String instanceName : getInstanceNames(Configurable.class)) {
372                 PropertySheet ps = getPropertySheet(instanceName);
373                 if (ps.isInstanciated())
374                     try {
375                         ps.getOwner().newProperties(ps);
376                     } catch (PropertyException e) {
377                         e.printStackTrace();
378                     }
379             }
380         }

436         /**
437          * Test wether the given configuration manager instance equals this instance in terms of same configuration. This
438          * This equals implemenation does not care about instantiation of components.
439          */
440         public boolean equals(Object obj) {
441             if (!(obj instanceof ConfigurationManager))
442                 return false;
443     
444             ConfigurationManager cm = (ConfigurationManager) obj;
445     
446             Collection<String> setA = new HashSet<String>(getComponentNames());
447             Collection<String> setB = new HashSet<String>(cm.getComponentNames());
448             if (!setA.equals(setB))
449                 return false;
450     
451             // make sure that all components are the same
452             for (String instanceName : getComponentNames()) {
453                 PropertySheet myPropSheet = getPropertySheet(instanceName);
454                 PropertySheet otherPropSheet = cm.getPropertySheet(instanceName);
455     
456                 if (!otherPropSheet.equals(myPropSheet))
457                     return false;
458             }
459     
460             // make sure that both configuration managers have the same set of global properties
461             return cm.getGlobalProperties().equals(getGlobalProperties());
462         }

517         /**
518          * Instantiates the given <code>targetClass</code> and instruments it using default properties or the properties
519          * given by the <code>defaultProps</code>.
520          */
521         private static PropertySheet getPropSheetInstanceFromClass(Class<? extends Configurable> targetClass, Map<String, Object> defaultProps, String componentName, ConfigurationManager cm) {
522             RawPropertyData rpd = new RawPropertyData(componentName, targetClass.getName());
523     
524             for (String confName : defaultProps.keySet()) {
525                 Object property = defaultProps.get(confName);
526     
527                 if (property instanceof Class)
528                     property = ((Class) property).getName();
529     
530                 rpd.getProperties().put(confName, property);
531             }
532     
533             return new PropertySheet(targetClass, componentName, cm, rpd);
534         }
