24     /**
25      * Provides a zip to city, state lookup service
26      */
27     public class ZipDatabase {
28         private Map zipDB = new HashMap();
29     
30         /**
31          * Creates the zip database
32          *
33          * @throws IOException if there is a problem loading the zip
34          * database
35          */
36         public ZipDatabase() throws IOException {
37             int line = 0;
38             InputStream is = this.getClass().getResourceAsStream("zip.txt");
39             BufferedReader br = new BufferedReader(new InputStreamReader(is));
40             String input;
41             // parses entries of the form:
42             // 03064,NH,NASHUA,lat,long
43             try {
44                 while  ((input = br.readLine()) != null) {
45                     line++;
46                     StringTokenizer st = new StringTokenizer(input, ",");
47                     if (st.countTokens() == 5) {
48                         String zip = st.nextToken();
49                         String state = st.nextToken();
50                         String city = st.nextToken();
51                         String slong = st.nextToken();
52                         String slat = st.nextToken();
53                         city = fixupCase(city);
54     
55                         try {
56                             float latitude = Float.parseFloat(slat);
57                             float longitude = Float.parseFloat(slong);
58                             zipDB.put(zip, new ZipInfo(zip, city, state,
59                                         latitude, longitude));
60                         } catch (NumberFormatException nfe) {
61                             throw new IOException("Bad zip format, line " + line
62                                     + "(" + input + ")");
63                         }
64     
65                     } else {
66                         throw new IOException("Bad zip format, line " + line
67                                     + "(" + input + ")");
68                     }
69                 }
70             } finally {
71                 br.close();
72             }
73         }
74     
75         /**
76          * Looks up the city associated with the given zip code
77          *
78          * @param zipcode the zip code
79          *
80          * @return city info or null if the zip was not found
81          */
82         public ZipInfo lookup(String zipcode) {
83             return (ZipInfo) zipDB.get(zipcode);
84         }
85     
86         /**
87          * Returns the iterator for all entries
88          *
89          * @return an iterator for all entries
90          */
91         public Iterator iterator() {
92             return zipDB.values().iterator();
93         }
94     
95         /**
96          * Fixes up the case of a string.  In the zip data, some of the
97          * cities are all upper case, while some are mixed case. fixupCase
98          * converts the input string to a mixed case string
99          * 
100          * @param s the string to fixup
101          * @return a mixed-case form of string
102          */
103         private String fixupCase(String s) {
104             boolean nextIsUpper = true;
105             StringBuffer sb = new StringBuffer();
106     
107             for (int i = 0; i < s.length(); i++) {
108                 char c = s.charAt(i);
109                 if (nextIsUpper) {
110                     c = Character.toUpperCase(c);
111                 }  else {
112                     c = Character.toLowerCase(c);
113                 }
114                 sb.append(c);
115                 nextIsUpper = Character.isWhitespace(c);
116             }
117             return sb.toString();
118         }

120         /**
121          * A test program for zip info
122          */
123         public static void main(String[] args) throws Exception {
124             ZipDatabase zipDB = new ZipDatabase();
125     
126             BufferedReader br = 
127                 new BufferedReader(new InputStreamReader(System.in));
128             while (true) {
129                 System.out.print("Enter zip: ");
130                 String zip = br.readLine();
131                 System.out.println("Input: " + zip);
132     
133                 ZipInfo info = zipDB.lookup(zip);
134                 System.out.println("Output: " + info);
135             }
136         }

141     /**
142      * Holds the zip information, including the zip, city and state
143      */
144     class ZipInfo {
145         private String zip;
146         private String city;
147         private String state;
148         private float latitude;
149         private float longitude;
150     
151         /**
152          * Creates a ZipInfo 
153          *
154           * @param zip the zip code
155           * @param city the city
156           * @param state the state
157           * @param lat the latitude
158           * @param lat the longitude
159           */
160         ZipInfo(String zip, String city, String state, float lat, float
161                 longitude) {
162             this.zip = zip;
163             this.city = city;
164             this.state = state;
165             this.latitude = lat;
166             this.longitude = longitude;
167         }
168     
169         /**
170          * Gets the zip code
171          *
172          * @return the zip code
173          */
174         public String getZip() {
175             return zip;
176         }
177     
178         /**
179          * Gets the city
180          *
181          * @return the city
182          */
183         public String getCity() {
184             return city;
185         }
186     
187         /**
188          * Gets the state
189          *
190          * @return the state
191          */
192         public String getState() {
193             return state;
194         }
195     
196         /**
197          * Gets the latitude
198          *
199          * @return the latitude
200          */
201         public float getLatitude() {
202             return latitude;
203         }
204     
205         /**
206          * Gets the longitude
207          *
208          * @return the longitude
209          */
210         public float getLongitude() {
211             return longitude;
212         }
213     
214         /**
215          * Returns a string representation of this object
216          *
217          * @return a string representation of this object
218          */
219         public String toString() {
220             return "Zip: " + getZip() +
221                   " City: " + getCity() +
222                   " State: " + getState() +
223                   "  " + getLatitude() + "," + getLongitude();
224         }
225     }
