14     package edu.cmu.sphinx.frontend;
15     
16     import java.util.HashMap;
17     import java.util.Map;
18     
19     /**
20      * Indicates events like beginning or end of data, data dropped, quality changed, etc.. It implements the Data
21      * interface, and it will pass between DataProcessors to inform them about the Data that is passed between
22      * DataProcessors.
23      *
24      * @see Data
25      * @see DataProcessor
26      */
27     public class Signal implements Data {
28     
29         /** the time this Signal was issued. */
30         private long time;
31     
32         /**
33          * A (lazily initialized) collection of names properties of this signal. This collection might contain infos about
34          * the file being processed, shift-size of frame-length of the windowing process, etc.
35          */
36         private Map<String, Object> props;
37     
38     
39         /**
40          * Constructs a Signal with the given name.
41          *
42          * @param time the time this Signal is created
43          */
44         protected Signal(long time) {
45             this.time = time;
46         }
47     
48     
49         /**
50          * Returns the time this Signal was created.
51          *
52          * @return the time this Signal was created
53          */
54         public long getTime() {
55             return time;
56         }
57     
58     
59         /** Returns the properties associated to this signal. */
60         public synchronized Map<String, Object> getProps() {
61             if (props == null)
62                 props = new HashMap<String, Object>();
63     
64             return props;
65         }
66     }
