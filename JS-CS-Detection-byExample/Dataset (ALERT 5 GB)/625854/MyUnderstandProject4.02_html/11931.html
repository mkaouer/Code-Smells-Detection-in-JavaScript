24     /**
25      * This class stores all the configuration information to be written to output
26      * and the properties loaded from input file
27      * It does not create any HashMap instances, so these responsibilites are on the creator
28      * and be careful when modifying the globalProperties and propertyMap because
29      * the original data would also be modified
30      *
31      * @author Ariani
32      */
33     public class ConfigProperties {
34        
35         /* the properties to be written */
36         private Map _propertyMap=null; // a HashMap for RawPropertyData map with component name as key
37         private Map _globalProperties=null; // a HashMap for global properties (often fine tuned properties)
38         private Map _otherPropMap=null; 
39         // a two-tier HashMap for RawPropertyData with String classname as 1st key and rpd name as 2nd key
40         
41         /** 
42          * Creates a new instance of ConfigProperties 
43          */
44         public ConfigProperties() { 
45         }
46         
47         /**
48          * @return get the Map of Global properties
49          */
50         public Map getGlobal(){
51             return _globalProperties;
52         }
53         
54         /**
55          * @return get the Map of component properties / configuration set
56          *          with classname as the Map key
57          */
58         public Map getOtherProp(){
59             return _otherPropMap;
60         }
61         
62         /**
63          * @return the configuration set/component properties
64          *          with configuration name as Map key
65          */
66         public Map getProperty(){
67             return _propertyMap;
68         }
69         
70         /** 
71          * @param global Set the global Map to this Map
72          */
73         public void setGlobal (Map global){
74             _globalProperties = global;
75         }
76     
77         /**
78          * add a whole set of Raw Property Data properties ( NOT Global properties )
79          * @param p <code>Map</code> of properties with key-value of 
80          *          String name,RawPropertyData
81          */
82         public void addRPDProperties(Map p)
83         {   
84             
85             if (_propertyMap == null)
86                 _propertyMap = new HashMap();
87             if ( p != null && !p.isEmpty())
88                 _propertyMap.putAll(p);
89     
90             copyPropertiesToOtherMap(p);
91             
92         }
93             
94         /** copy all the data in the input Map into the Other Map
95          */
96         private void copyPropertiesToOtherMap(Map from)
97         {
98             // add new properties to the other map
99             if ( from != null && !from.isEmpty()){
100                 if ( _otherPropMap == null){
101                     _otherPropMap = new HashMap();
102                 }
103                 
104                 Map classmap;
105                 for (Iterator it = from.values().iterator(); it.hasNext();){
106                     RawPropertyData rpd = (RawPropertyData)it.next();     
107                     String classname = rpd.getClassName();
108                     if ( _otherPropMap.containsKey(classname) ){
109                        classmap = (Map)_otherPropMap.get(classname);     
110                        if ( !classmap.containsKey(rpd.getName()) )
111                            classmap.put(rpd.getName(),rpd);       
112                     }else{
113                        classmap = new HashMap();
114                        classmap.put(rpd.getName(),rpd);       
115                        _otherPropMap.put(classname,classmap); 
116                     }
117                 }
118             }
119         }
120             
121         /**
122          * @param c Set the component properties to this one
123          */
124         public void setProperty(Map c){
125             _propertyMap = c;
126             
127             //create a duplicate map that has the classname 
128             // as key instead of configuration name key
129             if ( _otherPropMap == null)
130                 _otherPropMap = new HashMap();
131             else
132                 _otherPropMap.clear();
133             copyPropertiesToOtherMap(_propertyMap);
134             
135         }
136         
137     }
