13     package edu.cmu.sphinx.tools.gui.reader;

15     import java.io.IOException;
16     import java.io.InputStream;
17     import java.net.URL;
18     import java.util.ArrayList;
19     import java.util.HashMap;
20     import java.util.List;
21     import java.util.Map;
22     import javax.xml.parsers.ParserConfigurationException;
23     import javax.xml.parsers.SAXParserFactory;
24     import org.xml.sax.Attributes;
25     import org.xml.sax.InputSource;
26     import org.xml.sax.Locator;
27     import org.xml.sax.SAXException;
28     import org.xml.sax.SAXParseException;
29     import org.xml.sax.XMLReader;
30     import org.xml.sax.helpers.DefaultHandler;
31     
32     import edu.cmu.sphinx.tools.gui.RawPropertyData;
33     
34     /**
35      * Loads configuration from an XML file
36      */
37     public class SaxLoader {
38         private URL url;
39         private Map rpdMap;
40         private Map globalProperties;
41         /**
42          * Creates a loader that will load from the given location
43          * 
44          * @param url
45          *            the location to load
46          * @param globalProperties
47          *             the map of global properties
48          */
49         public SaxLoader(URL url, Map globalProperties) {
50             this.url = url;
51             this.globalProperties = globalProperties;
52         }
53         /**
54          * Loads a set of configuration data from the location
55          * 
56          * @return a map keyed by component name containing RawPropertyData objects
57          * @throws IOException
58          *             if an I/O or parse error occurs
59          */
60         public Map load() throws IOException {
61             rpdMap = new HashMap();
62             try {
63                 SAXParserFactory factory = SAXParserFactory.newInstance();
64                 XMLReader xr = factory.newSAXParser().getXMLReader();
65                 ConfigHandler handler = new ConfigHandler();
66                 xr.setContentHandler(handler);
67                 xr.setErrorHandler(handler);
68                 InputStream is = url.openStream();
69                 xr.parse(new InputSource(is));
70                 is.close();
71             } catch (SAXParseException e) {
72                 String msg = "Error while parsing line " + e.getLineNumber()
73                         + " of " + url + ": " + e.getMessage();
74                 throw new IOException(msg);
75             } catch (SAXException e) {
76                 throw new IOException("Problem with XML: " + e);
77             } catch (ParserConfigurationException e) {
78                 throw new IOException(e.getMessage());
79             }
80             return rpdMap;
81         }
82         
83         /**
84          * A SAX XML Handler implementation that builds up the map of raw property
85          * data objects
86          *  
87          */
88         class ConfigHandler extends DefaultHandler {
89             RawPropertyData rpd = null;
90             Locator locator;
91             List itemList = null;
92             String itemListName = null;
93             StringBuffer curItem;
94             
95             
96             /* (non-Javadoc)
97              * @see org.xml.sax.ContentHandler#startElement(java.lang.String, java.lang.String, java.lang.String, org.xml.sax.Attributes)
98              */
99             public void startElement(String uri, String localName, String qName,
100                     Attributes attributes) throws SAXException {
101                 if (qName.equals("config")) {
102                     // nothing to do
103                 } else if (qName.equals("component")) {
104                     String curComponent = attributes.getValue("name");
105                     String curType = attributes.getValue("type");
106                     if (rpdMap.get(curComponent) != null) {
107                         throw new SAXParseException(
108                             "duplicate definition for " + curComponent, locator);
109                     }
110                     rpd = new RawPropertyData(curComponent, curType);
111                 } else if (qName.equals("property")) {
112                     String name = attributes.getValue("name");
113                     String value = attributes.getValue("value");
114                     if (attributes.getLength() != 2 || name == null
115                             || value == null) {
116                         throw new SAXParseException(
117                                 "property element must only have "
118                                         + "'name' and 'value' attributes", locator);
119                     }
120                     if (rpd == null) {
121                         // we are not in a component so add this to the global
122                         // set of symbols                    
123                         globalProperties.put(name, value);
124                     } else if (rpd.contains(name)) {
125                         throw new SAXParseException("Duplicate property: " + name,
126                                 locator);
127                     } else {
128                         rpd.add(name, value);
129                     }
130                 } else if (qName.equals("propertylist")) {
131                     itemListName = attributes.getValue("name");
132                     if (attributes.getLength() != 1 || itemListName == null) {
133                         throw new SAXParseException("list element must only have "
134                                 + "the 'name'  attribute", locator);
135                     }
136                     itemList = new ArrayList();
137                 } else if (qName.equals("item")) {
138                     if (attributes.getLength() != 0) {
139                         throw new SAXParseException("unknown 'item' attribute",
140                                 locator);
141                     }
142                     curItem = new StringBuffer();
143                 } else {
144                     throw new SAXParseException("Unknown element '" + qName + "'",
145                             locator);
146                 }
147             }
148             /* (non-Javadoc)
149              * @see org.xml.sax.ContentHandler#characters(char[], int, int)
150              */
151             public void characters(char ch[], int start, int length)
152                     throws SAXParseException {
153                 if (curItem != null) {
154                     curItem.append(ch, start, length);
155                 } 
156             }
157             
158             /* (non-Javadoc)
159              * @see org.xml.sax.ContentHandler#endElement(java.lang.String, java.lang.String, java.lang.String)
160              */
161             public void endElement(String uri, String localName, String qName)
162                     throws SAXParseException {
163                 if (qName.equals("component")) {
164                     rpdMap.put(rpd.getName(), rpd);
165                     rpd = null;
166                 } else if (qName.equals("property")) {
167                     // nothing to do
168                 } else if (qName.equals("propertylist")) {
169                     if (rpd.contains(itemListName)) {
170                         throw new SAXParseException("Duplicate property: "
171                                 + itemListName, locator);
172                     } else {
173                         rpd.add(itemListName, itemList);
174                         itemList = null;
175                     }
176                 } else if (qName.equals("item")) {
177                     itemList.add(curItem.toString().trim());
178                     curItem = null;
179                 }
180             }
181             /* (non-Javadoc)
182              * @see org.xml.sax.ContentHandler#setDocumentLocator(org.xml.sax.Locator)
183              */
184             public void setDocumentLocator(Locator locator) {
185                 this.locator = locator;
186             }
187         }
188     }
