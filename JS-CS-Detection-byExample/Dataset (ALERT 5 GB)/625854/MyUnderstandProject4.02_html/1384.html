14     /**
15      * An interface implemented by command functions typically added to a command interpreter
16      *
17      * @see CommandInterpreter
18      */
19     
20     public interface CommandInterface {
21     
22         /**
23          * Execute the given command.
24          *
25          * @param ci   the command interpretere that invoked this command.
26          * @param args command line arguments (just like main).
27          * @return a command result
28          */
29         public String execute(CommandInterpreter ci, String[] args);
30     
31     
32         /**
33          * Returns a one line description of the command
34          *
35          * @return a one-liner help message
36          */
37         public String getHelp();
38     }
