14     package edu.cmu.sphinx.frontend.feature;
15     
16     import edu.cmu.sphinx.frontend.*;
17     import edu.cmu.sphinx.util.props.*;
18     
19     /**
20      * Subtracts the mean of all the input so far from the Data objects. Unlike the {@link BatchCMN}, it does not read in
21      * the entire stream of Data objects before it calculates the mean. It estimates the mean from already seen data and
22      * subtracts the mean from the Data objects on the fly. Therefore, there is no delay introduced by LiveCMN.
23      * <p/>
24      * The Sphinx properties that affect this processor are defined by the fields {@link #PROP_INITIAL_MEAN}, {@link
25      * #PROP_CMN_WINDOW}, and {@link #PROP_CMN_SHIFT_WINDOW}. Please follow the link "Constant Field Values" below to see
26      * the actual name of the Sphinx properties.
27      * <p/>
28      * <p>The mean of all the input cepstrum so far is not reestimated for each cepstrum. This mean is recalculated after
29      * every {@link #PROP_CMN_SHIFT_WINDOW} cepstra. This mean is estimated by dividing the sum of all input cepstrum so
30      * far. After obtaining the mean, the sum is exponentially decayed by multiplying it by the ratio:
31      * <pre>
32      * cmnWindow/(cmnWindow + number of frames since the last recalculation)
33      * </pre>
34      * <p/>
35      * <p>This is a 1-to-1 processor.
36      *
37      * @see BatchCMN
38      */
39     public class LiveCMN extends BaseDataProcessor {
40     
41     
42         /** The name of the SphinxProperty for the initial cepstral mean. This is a front-end dependent magic number. */
43         @S4Double(defaultValue = 12.0)
44         public static final String PROP_INITIAL_MEAN = "initialMean";
45     
46     
47         /** The default value for PROP_INITIAL_MEAN. */
48         public static final float PROP_INITIAL_MEAN_DEFAULT = 12.0f;
49     
50     
51         /** The name of the SphinxProperty for the live CMN window size. */
52         @S4Integer(defaultValue = 100)
53         public static final String PROP_CMN_WINDOW = "cmnWindow";
54     
55     
56         /** The default value for PROP_CMN_WINDOW. */
57         public static final int PROP_CMN_WINDOW_DEFAULT = 100;
58     
59     
60         /**
61          * The name of the SphinxProperty for the CMN shifting window. The shifting window specifies how many cepstrum after
62          * which we re-calculate the cepstral mean.
63          */
64         @S4Integer(defaultValue = 160)
65         public static final String PROP_CMN_SHIFT_WINDOW = "shiftWindow";
66     
67     
68         /** The default value of PROP_CMN_SHIFT_WINDOW. */
69         public static final int PROP_CMN_SHIFT_WINDOW_DEFAULT = 160;
70     
71     
72         private double[] currentMean;   // array of current means
73         private double[] sum;           // array of current sums
74         private double initialMean;     // initial mean, magic number
75         private int numberFrame;        // total number of input Cepstrum
76         private int cmnShiftWindow;     // # of Cepstrum to recalculate mean
77         private int cmnWindow;
78     
79     
80         /*
81         * (non-Javadoc)
82         *
83         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
84         */
85         public void newProperties(PropertySheet ps) throws PropertyException {
86             super.newProperties(ps);
87             initialMean = ps.getDouble(PROP_INITIAL_MEAN);
88             cmnWindow = ps.getInt(PROP_CMN_WINDOW);
89             cmnShiftWindow = ps.getInt(PROP_CMN_SHIFT_WINDOW);
90         }
91     
92     
93         /** Initializes this LiveCMN. */
94         public void initialize() {
95             super.initialize();
96         }
97     
98     
99         /**
100          * Initializes the currentMean and sum arrays with the given cepstrum length.
101          *
102          * @param cepstrumLength the length of the cepstrum
103          */
104         private void initMeansSums(int cepstrumLength) {
105             currentMean = new double[cepstrumLength];
106             currentMean[0] = initialMean;
107             sum = new double[cepstrumLength];
108         }
109     
110     
111         /**
112          * Returns the next Data object, which is a normalized Data produced by this class. Signals are returned
113          * unmodified.
114          *
115          * @return the next available Data object, returns null if no Data object is available
116          * @throws DataProcessingException if there is a data processing error
117          */
118         public Data getData() throws DataProcessingException {
119     
120             Data input = getPredecessor().getData();
121     
122             if (input instanceof DataStartSignal)
123                 sum = null;
124     
125             getTimer().start();
126     
127             if (input != null) {
128                 if (input instanceof DoubleData) {
129                     DoubleData data = (DoubleData) input;
130                     if (sum == null) {
131                         initMeansSums(data.getValues().length);
132                     }
133                     normalize(data);
134                 } else if (input instanceof DataEndSignal) {
135                     updateMeanSumBuffers();
136                 }
137             }
138     
139             getTimer().stop();
140     
141             return input;
142         }
143     
144     
145         /**
146          * Normalizes the given Data with using the currentMean array. Updates the sum array with the given Data.
147          *
148          * @param cepstrumObject the Data object to normalize
149          */
150         private void normalize(DoubleData cepstrumObject) {
151     
152             double[] cepstrum = cepstrumObject.getValues();
153     
154             if (cepstrum.length != sum.length) {
155                 throw new Error("Data length (" + cepstrum.length +
156                         ") not equal sum array length (" +
157                         sum.length + ")");
158             }
159     
160             for (int j = 0; j < cepstrum.length; j++) {
161                 sum[j] += cepstrum[j];
162                 cepstrum[j] -= currentMean[j];
163             }
164     
165             numberFrame++;
166     
167             if (numberFrame > cmnShiftWindow) {
168                 updateMeanSumBuffers();
169             }
170         }
171     
172     
173         /**
174          * Updates the currentMean buffer with the values in the sum buffer. Then decay the sum buffer exponentially, i.e.,
175          * divide the sum with numberFrames.
176          */
177         private void updateMeanSumBuffers() {
178     
179             if (numberFrame > 0) {
180     
181                 // update the currentMean buffer with the sum buffer
182                 double sf = 1.0 / numberFrame;
183     
184                 System.arraycopy(sum, 0, currentMean, 0, sum.length);
185     
186                 multiplyArray(currentMean, sf);
187     
188                 // decay the sum buffer exponentially
189                 if (numberFrame >= cmnShiftWindow) {
190                     multiplyArray(sum, (sf * cmnWindow));
191                     numberFrame = cmnWindow;
192                 }
193             }
194         }
195     
196     
197         /**
198          * Multiplies each element of the given array by the multiplier.
199          *
200          * @param array      the array to multiply
201          * @param multiplier the amount to multiply by
202          */
203         private static void multiplyArray(double[] array,
204                                           double multiplier) {
205             for (int i = 0; i < array.length; i++) {
206                 array[i] *= multiplier;
207             }
208         }
209     }
