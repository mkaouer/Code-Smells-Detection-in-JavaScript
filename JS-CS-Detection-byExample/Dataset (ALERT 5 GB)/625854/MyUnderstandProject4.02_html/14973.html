25         /** Copy all RuleGrammars from recFrom to recTo. */
26         public static void copyGrammars(Recognizer recFrom, Recognizer recTo)
27                 throws GrammarException {
28             RuleGrammar gramFrom[] = recFrom.listRuleGrammars();
29     
30             for (int i = 0; i < gramFrom.length; i++) {
31                 String name = gramFrom[i].getName();
32                 RuleGrammar gramTo = recTo.getRuleGrammar(name);
33     
34                 // Create a new grammar - if necessary
35                 if (gramTo == null) {
36                     try {
37                         gramTo = recTo.newRuleGrammar(name);
38                     } catch (IllegalArgumentException gse) {
39                         throw new GrammarException("copyGrammars: " + gse.toString(), null);
40                     }
41                 }
42                 copyGrammar(gramFrom[i], gramTo);
43             }
44         }

47         /** Copy a grammar from gramFrom to gramTo. */
48         public static void copyGrammar(RuleGrammar gramFrom, RuleGrammar gramTo)
49                 throws GrammarException {
50             // Copy imports
51             RuleName imports[] = gramFrom.listImports();
52             for (int j = 0; j < imports.length; j++) {
53                 gramTo.addImport(imports[j]);
54             }
55     
56             // Copy each rule
57             String rnames[] = gramFrom.listRuleNames();
58             for (int j = 0; j < rnames.length; j++) {
59                 String ruleName = rnames[j];
60                 boolean isPublic = false;
61                 try {
62                     isPublic = gramFrom.isRulePublic(ruleName);
63                 } catch (IllegalArgumentException nse) {
64                     throw new GrammarException(nse.toString(), null);
65                 }
66     
67                 Rule rule = gramFrom.getRule(ruleName);
68                 try {
69                     gramTo.setRule(ruleName, rule, isPublic);
70                     if (gramFrom.isEnabled(ruleName)) {
71                         gramTo.setEnabled(ruleName, true);
72                     }
73                 } catch (IllegalArgumentException E) {
74                     throw new GrammarException(E.toString(), null);
75                 }
76     
77                 // copy sample sentences if appropriate
78                 if ((gramTo instanceof BaseRuleGrammar) && (gramFrom instanceof BaseRuleGrammar)) {
79                     Vector a = ((BaseRuleGrammar) gramFrom).getSampleSentences(ruleName);
80                     ((BaseRuleGrammar) gramTo).setSampleSentences(ruleName, a);
81                 }
82             }
83         }

89         /**
90          * For a specified recognizer, build a table that for each rule R lists all rules in all grammars that reference R.
91          *
92          * @see #getRuleNameRefs
93          * @see #getRefsToRuleName
94          */
95         static public void buildXrefTable(Recognizer rec) {
96             xrefs = new Hashtable();
97             RuleGrammar[] grams = rec.listRuleGrammars();
98     
99             if (grams == null) {
100                 return;
101             }
102     
103             for (int i = 0; i < grams.length; i++) {
104                 String[] names = grams[i].listRuleNames();
105                 if (names == null) {
106                     continue;
107                 }
108                 for (int j = 0; j < names.length; j++) {
109                     // Get the definition of rule name[j] in gram[i]
110                     Rule r = grams[i].getRule(names[j]);
111     
112                     // Build a fully-qualified RuleName for rule name[j] in gram[i]
113                     RuleName rn = new RuleName(grams[i].getName() + "." + names[j]);
114     
115                     // Identify all rules referenced in r
116                     Vector refs = new Vector();
117                     getRuleNameRefs(r, refs);
118     
119                     for (int k = 0; k < refs.size(); k++) {
120                         RuleName ref = (RuleName) (refs.elementAt(k));
121     
122                         // Get a fully-qualified reference
123                         RuleName fullref;
124                         try {
125                             fullref = ((BaseRuleGrammar) (grams[i])).resolve(ref);
126                         } catch (GrammarException e) {
127                             fullref = null;
128                         }
129     
130                         if (fullref != null) {
131                             Hashtable h = (Hashtable) (xrefs.get(fullref.toString().intern()));
132     
133                             if (h == null) {
134                                 h = new Hashtable();
135                             }
136     
137                             h.put(rn.toString().intern(), "dummy");
138                             xrefs.put(fullref.toString().intern(), h);
139                         } else {
140                             debugMessageOut("Warning: unresolved rule " + ref.toString() +
141                                     " in grammar " + grams[i].getName());
142                         }
143                     }
144                 }
145             }
146         }

149         /** Return an array of references to rule r in grammar g. */
150         static public RuleName[] getRefsToRuleName(RuleGrammar g, RuleName r) {
151             // Ensure we have a fully qualified rulename
152             // (that's how the xref table works)
153             r = new RuleName(g.getName() + "." + r.getSimpleRuleName());
154     
155             if (xrefs == null) {
156                 return null;
157             }
158     
159             Hashtable h = (Hashtable) (xrefs.get(r.toString().intern()));
160     
161             if (h == null) {
162                 return null;
163             }
164     
165             RuleName[] rulenames = new RuleName[h.size()];
166     
167             int i = 0;
168             for (Enumeration e = h.keys(); e.hasMoreElements();) {
169                 String name = (String) (e.nextElement());
170                 rulenames[i++] = new RuleName(name);
171             }
172     
173             return rulenames;
174         }

177         static protected void getRuleNameRefs(Rule r, Vector refs) {
178             if (r instanceof RuleAlternatives || r instanceof RuleSequence) {
179                 Rule[] array;
180                 if (r instanceof RuleAlternatives) {
181                     array = ((RuleAlternatives) r).getRules();
182                 } else {
183                     array = ((RuleSequence) r).getRules();
184                 }
185     
186                 if (array != null) {
187                     for (int i = 0; i < array.length; i++) {
188                         getRuleNameRefs(array[i], refs);
189                     }
190                 }
191             } else if (r instanceof RuleTag) {
192                 getRuleNameRefs(((RuleTag) r).getRule(), refs);
193             } else if (r instanceof RuleCount) {
194                 getRuleNameRefs(((RuleCount) r).getRule(), refs);
195             } else if (r instanceof RuleName) {
196                 // Put a copy in the Xref list (avoid side-effects of linking)
197                 RuleName tmp = (RuleName) r;
198                 refs.addElement(new RuleName(tmp.getRuleName()));
199             }
200         }

216         /*
217           * transform grammar to eliminate <NULL> and <VOID>
218           *
219           * NULL and VOID can be removed by transforming the grammar
220           * as follows:
221           *
222           * RuleSequence:
223           *   (x  y  <NULL>) --> (x y)
224           *   (x  y  <VOID>) --> <VOID>
225           *   (<NULL>) --> <NULL>
226           *   (<VOID>) --> <VOID>
227           *
228           * RuleAlternatives:
229           *   (x | y | <NULL>) --> [(x | y)]
230           *   (x | y | <VOID>) --> (x | y)
231           *   (<NULL>) --> <NULL>
232           *   (<VOID>) --> <VOID>
233           *   (<VOID> | <NULL>) --> <NULL>
234         *
235         * RuleCount:
236         *   [<VOID>]  --> <NULL>
237         *    <VOID>*  --> <NULL>
238         *    <VOID>+  --> <VOID>
239         *   [<NULL>]  --> <NULL>
240         *    <NULL>*  --> <NULL>
241         *    <NULL>+  --> <NULL>
242         *
243         * RuleTag:
244         *   <NULL>{tag} --> <NULL>
245         *   <VOID>{tag} --> <VOID>
246         *
247         * RuleName:
248         *    If a rule has been marked as being void or null then apply the
249         *    following:
250         *    <RuleName> --> <VOID>
251         *
252         * The steps of transformation are:
253         *
254         *     just return the original grammar
255         *
256         *  2. Make a copy of the grammar and Mark all rules as non-void & non-null
257         *
258         *  3. Go through each rule in series and apply the transformations.
259         *
260         *  4. Repeat step 3 until one pass has been made through all the
261         *     rules in which no transformations were performed
262         *
263         *  5. Remove NULL and VOID rules from copies grammars
264         *
265         *  6. Remove any empty grammars
266         *
267         *  7. Return the transformed array of grammars
268         *
269         * Because all transforms are some form of a reduction, the process is
270         * guaranteed to complete.
271         *
272         */
273

275         static RuleGrammar[] transform(RuleGrammar RG[], boolean eliminateNULL, boolean eliminateVOID)
276                 throws GrammarException {
277             //if (!eliminateNULL && !eliminateVOID) return RG;
278             Hashtable ruleinfo = new Hashtable();
279             RuleGrammar NRG[] = (RuleGrammar[]) RG.clone();
280             boolean somethingchanged = true;
281             boolean nochange = true;
282             while (somethingchanged) {
283                 somethingchanged = false;
284                 for (int i = 0; i < NRG.length; i++) {
285                     RuleGrammar NG = null;
286                     String rnames[] = NRG[i].listRuleNames();
287                     for (int j = 0; j < rnames.length; j++) {
288                         Rule r = NRG[i].getRuleInternal(rnames[j]);
289                         Rule nr = transform(r, NRG, eliminateNULL, eliminateVOID);
290                         //System.out.println("do: " + rnames[j]);
291                         //System.out.println("  ("+r+") ("+nr+")");
292                         if (nr != r) {
293                             nochange = false;
294                             somethingchanged = true;
295                             if (NRG[i] == RG[i]) {
296                                 NRG[i] = new BaseRuleGrammar(null, RG[i].getName());
297                                 RecognizerUtilities.copyGrammar(RG[i], NRG[i]);
298                             }
299                             boolean isPublic;
300                             try {
301                                 isPublic = NRG[i].isRulePublic(rnames[j]);
302                             } catch (IllegalArgumentException nse) {
303                                 throw new GrammarException(nse.toString(), null);
304                             }
305                             NRG[i].setRule(rnames[j], nr, isPublic);
306                         }
307                     }
308                 }
309             }
310     
311             if (nochange) return NRG;
312     
313             /*
314             * Now remove unused rules
315             */
316             Vector nr = new Vector();
317             for (int i = 0; i < NRG.length; i++) {
318                 RuleGrammar NG = null;
319                 String rnames[] = NRG[i].listRuleNames();
320                 int k = 0;
321                 for (int j = 0; j < rnames.length; j++) {
322                     Rule r1 = NRG[i].getRuleInternal(rnames[j]);
323                     RuleState s = new RuleState(r1, NRG);
324                     //System.out.println(rnames[j] + " " + r1.getClass() + " " + s.isvoid + " " + s.isnull);
325                     if ((eliminateVOID && s.isvoid) || (eliminateNULL && s.isnull)) {
326                         NRG[i].deleteRule(rnames[j]);
327                     } else k++;
328                 }
329                 if (k > 0) nr.addElement(NRG[i]);
330             }
331     
332             /*
333             * Now return all non-null grammars
334             */
335             RuleGrammar FRG[] = new RuleGrammar[nr.size()];
336             nr.copyInto(FRG);
337             return FRG;
338         }

341         static Rule transform(Rule r, RuleGrammar RGL[], boolean en, boolean ev) {
342             //System.out.println("xf " + r.getClass() + r);
343     
344             /*
345             * RULESEQUENCE
346             */
347             if (r instanceof RuleSequence) {
348                 RuleSequence rs = (RuleSequence) r;
349                 Rule rarry[] = rs.getRules();
350                 if ((rarry == null) || (rarry.length == 0)) {
351                     return null;
352                 }
353                 int cnull = 0;
354                 for (int i = 0; i < rarry.length; i++) {
355                     Rule r1 = transform(rarry[i], RGL, en, ev);
356                     RuleState s1 = new RuleState(r1, RGL);
357                     if (ev && s1.isvoid) return RuleName.VOID;
358                     if ((en && s1.isnull) || (r1 != rarry[i])) {
359                         if (rs == (RuleSequence) r) {
360                             rs = (RuleSequence) rs.copy();
361                             rarry = rs.getRules();
362                         }
363                         if (en && s1.isnull) {
364                             cnull++;
365                             r1 = null;
366                         }
367                         rarry[i] = r1;
368                         rs.setRules(rarry);
369                     }
370                 }
371                 if (cnull == 0) return rs;
372                 if (cnull == rarry.length) return RuleName.NULL;
373                 Rule barry[] = new Rule[rarry.length - cnull];
374                 int j = 0;
375                 for (int i = 0; i < rarry.length; i++) {
376                     if (rarry[i] != null) {
377                         barry[j++] = rarry[i];
378                     }
379                 }
380                 rs.setRules(barry);
381                 return rs;
382             }
383     
384             /*
385             * RULEALTERNATIVES
386             */
387             if (r instanceof RuleAlternatives) {
388                 RuleAlternatives ra = (RuleAlternatives) r;
389                 Rule rarry[] = ra.getRules();
390                 if ((rarry == null) || (rarry.length == 0)) {
391                     return null;
392                 }
393                 int cnull = 0;
394                 int cvoid = 0;
395                 for (int i = 0; i < rarry.length; i++) {
396                     Rule r1 = transform(rarry[i], RGL, en, ev);
397                     //System.out.println("rs: " + r1);
398                     RuleState s1 = new RuleState(r1, RGL);
399                     if ((en && s1.isnull) || (ev && s1.isvoid) || (r1 != rarry[i])) {
400                         if (ra == (RuleAlternatives) r) {
401                             ra = (RuleAlternatives) ra.copy();
402                             rarry = ra.getRules();
403                         }
404                         if (en && s1.isnull) {
405                             cnull++;
406                             r1 = null;
407                         }
408                         if (ev && s1.isvoid) {
409                             cvoid++;
410                             r1 = null;
411                         }
412                         rarry[i] = r1;
413                         ra.setRules(rarry);
414                     }
415                 }
416                 if ((cnull == 0) && (cvoid == 0)) return ra;
417                 if (cvoid == rarry.length) return RuleName.VOID;
418                 if ((cvoid + cnull) == rarry.length) return RuleName.NULL;
419                 Rule barry[] = new Rule[rarry.length - (cnull + cvoid)];
420                 int j = 0;
421                 for (int i = 0; i < rarry.length; i++) {
422                     if (rarry[i] != null) {
423                         barry[j++] = rarry[i];
424                     }
425                 }
426                 ra.setRules(barry);
427                 if (cnull > 0) return new RuleCount(ra, RuleCount.OPTIONAL);
428                 else return ra;
429             }
430     
431             /*
432             * RULECOUNT (e.g. [], *, or + )
433             */
434             if (r instanceof RuleCount) {
435                 RuleCount rc = (RuleCount) r;
436                 int rcount = rc.getCount();
437                 Rule r1 = rc.getRule();
438                 Rule r2 = transform(r1, RGL, en, ev);
439                 RuleState s1 = new RuleState(r2, RGL);
440                 if (ev && s1.isvoid) {
441                     if (rcount == RuleCount.ONCE_OR_MORE) return RuleName.VOID;
442                     return RuleName.NULL;
443                 }
444                 if (en && s1.isnull) return RuleName.NULL;
445                 if (r1 == r2) return rc;
446                 return new RuleCount(r2, rcount);
447             }
448     
449             /*
450             * TAGS
451             */
452             if (r instanceof RuleTag) {
453                 RuleTag rtag = (RuleTag) r;
454                 Rule rtr = rtag.getRule();
455                 Rule r1 = transform(rtr, RGL, en, ev);
456                 RuleState s1 = new RuleState(r1, RGL);
457                 if (en && s1.isnull) return RuleName.NULL;
458                 if (ev && s1.isvoid) return RuleName.VOID;
459                 if (r1 == rtr) return rtag;
460                 return new RuleTag(r1, rtag.getTag());
461             }
462     
463             return r;
464         }

561         Rule getRule(RuleName rn, RuleGrammar RGL[]) {
562             String gname = rn.getSimpleGrammarName();
563             for (int i = 0; i < RGL.length; i++) {
564                 if (RGL[i].getName().equals(gname)) {
565                     return RGL[i].getRuleInternal(rn.getSimpleRuleName());
566                 }
567             }
568             return null;
569         }

572         RuleState(Rule r, RuleGrammar RGL[]) {
573             if (r instanceof RuleName) {
574                 RuleName rn = (RuleName) r;
575     
576                 /* simple check for void/null */
577                 String simpleName = rn.getSimpleRuleName();
578                 if (simpleName.equals("VOID")) {
579                     isvoid = true;
580                     return;
581                 }
582                 if (simpleName.equals("NULL")) {
583                     isnull = true;
584                     return;
585                 }
586     
587                 /* now lookup the rule and see if its RHS is null/void */
588                 Rule r1 = getRule(rn, RGL);
589                 if (r1 == null) System.out.println("ERROR: Could not find rule " + rn);
590                 if (r1 instanceof RuleName) {
591                     RuleName rn1 = (RuleName) r1;
592     
593                     /* simple check for void/null */
594                     String simpleName1 = rn1.getSimpleRuleName();
595                     if (simpleName1.equals("VOID")) {
596                         isvoid = true;
597                         return;
598                     }
599                     if (simpleName1.equals("NULL")) {
600                         isnull = true;
601                         return;
602                     }
603                 }
604                 return;
605             }
606         }
