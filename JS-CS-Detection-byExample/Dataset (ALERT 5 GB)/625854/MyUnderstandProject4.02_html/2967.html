94         /** Initializes this log math */
95         private void init() {
96             if (logger != null) {
97                 logger.config("Log base is " + logBase);
98                 if (useAddTable) {
99                     logger.config("Using AddTable when adding logs");
100                 } else {
101                     logger.config("Performing actual computation when adding logs");
102                 }
103             }
104             naturalLogBase = (float) Math.log(logBase);
105             inverseNaturalLogBase = 1.0f / naturalLogBase;
106             // When converting a number from/to linear, we need to make
107             // sure it's within certain limits to prevent it from
108             // underflowing/overflowing.
109             // We compute the max value by computing the log of the max
110             // value that a float can contain.
111             maxLogValue = linearToLog(Double.MAX_VALUE);
112             // We compute the min value by computing the log of the min
113             // (absolute) value that a float can hold.
114             minLogValue = linearToLog(Double.MIN_VALUE);
115             if (useAddTable) {
116                 // Now create the addTable table.
117                 // summation needed in the loop
118                 float innerSummation;
119                 // First decide number of elements.
120                 int entriesInTheAddTable;
121                 final int veryLargeNumberOfEntries = 150000;
122                 final int verySmallNumberOfEntries = 0;
123                 // To decide size of table, take into account that a base
124                 // of 1.0001 or 1.0003 converts probabilities, which are
125                 // numbers less than 1, into integers. Therefore, a good
126                 // approximation for the smallest number in the table,
127                 // therefore the value with the highest index, is an
128                 // index that maps into 0.5: indices higher than that, if
129                 // they were present, would map to less values less than
130                 // 0.5, therefore they would be mapped to 0 as
131                 // integers. Since the table implements the expression:
132                 //
133                 // log(1.0 + base^(-index)))
134                 //
135                 // then the highest index would be:
136                 //
137                 // topIndex = - log(logBase^(0.5) - 1)
138                 //
139                 // where log is the log in the appropriate base.
140                 //
141                 // Added -Math.rint(...) to round to nearest
142                 // integer. Added the negation to match the preceeding
143                 // documentation
144                 entriesInTheAddTable = (int) -Math
145                         .rint(linearToLog(logToLinear(0.5f) - 1));
146                 // We reach this max if the log base is 1.00007. The
147                 // closer you get to 1, the higher the number of entries
148                 // in the table.
149                 if (entriesInTheAddTable > veryLargeNumberOfEntries) {
150                     entriesInTheAddTable = veryLargeNumberOfEntries;
151                 }
152                 if (entriesInTheAddTable <= verySmallNumberOfEntries) {
153                     throw new IllegalArgumentException("The log base " + logBase
154                             + " yields a very small addTable. "
155                             + "Either choose not to use the addTable, "
156                             + "or choose a logBase closer to 1.0");
157                 }
158                 // PBL added this just to see how many entries really are
159                 // in the table
160                 if (logger != null)
161                     logger.config("LogAdd table has " + entriesInTheAddTable + " entries.");
162     
163                 theAddTable = new float[entriesInTheAddTable];
164                 for (int index = 0; index < entriesInTheAddTable; index++) {
165                     // This loop implements the expression:
166                     //
167                     // log( 1.0 + power(base, index))
168                     //
169                     // needed to add two numbers in the log domain.
170                     innerSummation = (float) logToLinear(-index);
171                     innerSummation += 1.0f;
172                     theAddTable[index] = linearToLog(innerSummation);
173                 }
174             }
175         }

178         /**
179          * Returns the summation of two numbers when the arguments and the result are in log. <p/> <p/> That is, it returns
180          * log(a + b) given log(a) and log(b) </p> <p/> <p/> This method makes use of the equality: </p> <p/> <p/> <b>log(a
181          * + b) = log(a) + log (1 + exp(log(b) - log(a))) </b> </p> <p/> <p/> which is derived from: </p> <p/> <p/> <b>a + b
182          * = a * (1 + (b / a)) </b> </p> <p/> <p/> which in turns makes use of: </p> <p/> <p/> <b>b / a = exp (log(b) -
183          * log(a)) </b> </p> <p/> <p/> Important to notice that <code>subtractAsLinear(a, b)</code> is *not* the same as
184          * <code>addAsLinear(a, -b)</code>, since we're in the log domain, and -b is in fact the inverse. </p> <p/> <p/> No
185          * underflow/overflow check is performed. </p>
186          *
187          * @param logVal1 value in log domain (i.e. log(val1)) to add
188          * @param logVal2 value in log domain (i.e. log(val2)) to add
189          * @return sum of val1 and val2 in the log domain
190          */
191         public final float addAsLinear(float logVal1, float logVal2) {
192             float logHighestValue = logVal1;
193             float logDifference = logVal1 - logVal2;
194             /*
195              * [ EBG: maybe we should also have a function to add many numbers, *
196              * say, return the summation of all terms in a given vector, if *
197              * efficiency becomes an issue.
198              */
199             // difference is always a positive number
200             if (logDifference < 0) {
201                 logHighestValue = logVal2;
202                 logDifference = -logDifference;
203             }
204             return logHighestValue + addTable(logDifference);
205         }

208         /**
209          * Method used by add() internally. It returns the difference between the highest number and the total summation of
210          * two numbers. <p/> Considering the expression (in which we assume natural log) <p/> <p/> <b>log(a + b) = log(a) +
211          * log(1 + exp(log(b) - log(a))) </b> </p>
212          * <p/>
213          * the current function returns the second term of the right hand side of the equality above, generalized for the
214          * case of any log base. This function can be contructed as a table, if table lookup is faster than actual
215          * computation.
216          *
217          * @param index the index into the addTable
218          * @return the value pointed to by index
219          */
220         private float addTableActualComputation(float index) {
221             double logInnerSummation;
222             // Negate index, since the derivation of this formula implies
223             // the smallest number as a numerator, therefore the log of the
224             // ratio is negative
225             logInnerSummation = logToLinear(-index);
226             logInnerSummation += 1.0;
227             return linearToLog(logInnerSummation);
228         }

231         /**
232          * Method used by add() internally. It returns the difference between the highest number and the total summation of
233          * two numbers. <p/> Considering the expression (in which we assume natural log) <p/> <p/> <b>log(a + b) = log(a) +
234          * log(1 + exp(log(b) - log(a))) </b> </p>
235          * <p/>
236          * the current function returns the second term of the right hand side of the equality above, generalized for the
237          * case of any log base. This function is contructed as a table lookup.
238          *
239          * @param index the index into the addTable
240          * @return the value pointed to by index
241          * @throws IllegalArgumentException
242          */
243         private float addTable(float index) throws IllegalArgumentException {
244             if (useAddTable) {
245                 // int intIndex = (int) Math.rint(index);
246                 int intIndex = (int) (index + 0.5);
247                 // When adding two numbers, the highest one should be
248                 // preserved, and therefore the difference should always
249                 // be positive.
250                 if (0 <= intIndex) {
251                     if (intIndex < theAddTable.length) {
252                         return theAddTable[intIndex];
253                     } else {
254                         return 0.0f;
255                     }
256                 } else {
257                     throw new IllegalArgumentException("addTable index has "
258                             + "to be negative");
259                 }
260             } else {
261                 return addTableActualComputation(index);
262             }
263         }

266         /**
267          * Returns the difference between two numbers when the arguments and the result are in log. <p/> <p/> That is, it
268          * returns log(a - b) given log(a) and log(b) </p> <p/> <p/> Implementation is less efficient than add(), since
269          * we're less likely to use this function, provided for completeness. Notice however that the result only makes
270          * sense if the minuend is higher than the subtrahend. Otherwise, we should return the log of a negative number.
271          * </p> <p/> <p/> It implements the subtraction as: </p> <p/> <p/> <b>log(a - b) = log(a) + log(1 - exp(log(b) -
272          * log(a))) </b> </p> <p/> <p/> No need to check for underflow/overflow. </p>
273          *
274          * @param logMinuend    value in log domain (i.e. log(minuend)) to be subtracted from
275          * @param logSubtrahend value in log domain (i.e. log(subtrahend)) that is being subtracted
276          * @return difference between minuend and the subtrahend in the log domain
277          * @throws IllegalArgumentException <p/> This is a very slow way to do this, but this method should rarely be used.
278          *                                  </p>
279          */
280         public final float subtractAsLinear(float logMinuend, float logSubtrahend)
281                 throws IllegalArgumentException {
282             double logInnerSummation;
283             if (logMinuend < logSubtrahend) {
284                 throw new IllegalArgumentException("Subtraction results in log "
285                         + "of a negative number: " + logMinuend + " - "
286                         + logSubtrahend);
287             }
288             logInnerSummation = 1.0;
289             logInnerSummation -= logToLinear(logSubtrahend - logMinuend);
290             return logMinuend + linearToLog(logInnerSummation);
291         }

294         /**
295          * Converts the source, which is assumed to be a log value whose base is sourceBase, to a log value whose base is
296          * resultBase. Possible values for both the source and result bases include Math.E, 10.0, LogMath.getLogBase(). If a
297          * source or result base is not supported, an IllegalArgumentException will be thrown. <p/> <p/> It takes advantage
298          * of the relation: </p> <p/> <p/> <b>log_a(b) = log_c(b) / lob_c(a) </b> </p> <p/> <p/> or: </p> <p/> <p/>
299          * <b>log_a(b) = log_c(b) * lob_a(c) </b> </p> <p/> <p/> where <b>log_a(b) </b> is logarithm of <b>b </b> base <b>a
300          * </b> etc. </p>
301          *
302          * @param logSource  log value whose base is sourceBase
303          * @param sourceBase the base of the log the source
304          * @param resultBase the base to convert the source log to
305          * @throws IllegalArgumentException
306          */
307         //  [[[ TODO: This is slow, but it probably doesn't need
308         //  to be too fast ]]]
309         // [ EBG: it can be made more efficient if one of the bases is
310         // Math.E. So maybe we should consider two functions logToLn and
311         // lnToLog instead of a generic function like this??
312         //
313         public static float logToLog(float logSource, float sourceBase,
314                                      float resultBase) throws IllegalArgumentException {
315             if ((sourceBase <= 0) || (resultBase <= 0)) {
316                 throw new IllegalArgumentException("Trying to take log of "
317                         + " non-positive number: " + sourceBase + " or "
318                         + resultBase);
319             }
320             if (logSource == logZero) {
321                 return logZero;
322             }
323             float lnSourceBase = (float) Math.log(sourceBase);
324             float lnResultBase = (float) Math.log(resultBase);
325             return (logSource * lnSourceBase / lnResultBase);
326         }

368         /**
369          * Converts the value from linear scale to log scale. The log scale numbers are limited by the range of the type
370          * float. The linear scale numbers can be any double value.
371          *
372          * @param linearValue the value to be converted to log scale
373          * @return the value in log scale
374          * @throws IllegalArgumentException
375          */
376         public final float linearToLog(double linearValue)
377                 throws IllegalArgumentException {
378             double returnValue;
379             if (linearValue < 0.0) {
380                 throw new IllegalArgumentException(
381                         "linearToLog: param must be >= 0: " + linearValue);
382             } else if (linearValue == 0.0) {
383                 // [EBG] Shouldn't the comparison above be something like
384                 // linearValue < "epsilon"? Is it ever going to be 0.0?
385                 return getLogZero();
386             } else {
387                 returnValue = Math.log(linearValue) * inverseNaturalLogBase;
388                 if (returnValue > Float.MAX_VALUE) {
389                     return Float.MAX_VALUE;
390                 } else {
391                     if (returnValue < -Float.MAX_VALUE) {
392                         return -Float.MAX_VALUE;
393                     } else {
394                         return (float) returnValue;
395                     }
396                 }
397             }
398         }

401         /**
402          * Converts the value from log scale to linear scale.
403          *
404          * @param logValue the value to be converted to the linear scale
405          * @return the value in the linear scale
406          */
407         public final double logToLinear(float logValue) {
408             // return Math.pow(logBase, logValue);
409             double returnValue;
410             if (logValue < minLogValue) {
411                 returnValue = 0.0;
412             } else if (logValue > maxLogValue) {
413                 returnValue = Double.MAX_VALUE;
414             } else {
415                 returnValue = Math.exp(logToLn(logValue));
416             }
417             return returnValue;
418         }

467         /** Converts a vector from linear domain to logdomain using a given <code>LogMath</code>-instance for conversion. */
468         public static float[] linearToLog(float[] vector, LogMath logMath) {
469             float[] logMixtureWeights = new float[vector.length];
470             int nbGaussians = vector.length;
471             for (int i = 0; i < nbGaussians; i++) {
472                 logMixtureWeights[i] = logMath.linearToLog(vector[i]);
473             }
474             return logMixtureWeights;
475         }
