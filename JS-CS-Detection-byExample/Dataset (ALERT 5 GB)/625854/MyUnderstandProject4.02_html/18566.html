13     package edu.cmu.sphinx.research.parallel;

15     import java.util.*;
16     
17     /**
18      * Combines the various unit feature stream scores in a CombineToken. This SameStampScoreCombiner assumes that there can
19      * be more than one token from each feature stream. It will take the highest scoring token from each stream <b>with the
20      * same identifying stamp</b>, combine their scores, and choose the one with the highest combined score.
21      * <p/>
22      * For example, if tokens T1 and T2 from feature F1 had stamps s1 and s2, and tokens Ta and Tb from feature F2 had
23      * stamps s1 and s2, we must compare combinedScore(s1,Ta)  and combineScore(s2,Tb), and retain the one for which the
24      * combined score is higher.
25      * <p/>
26      * All scores are maintained internally in the LogMath logbase
27      */
28     public class SameStampScoreCombiner implements ScoreCombiner {
29     
30         /**
31          * Remove all tokens in the given token list that has the given stamp.
32          *
33          * @param tokenList the token list
34          * @param stamp     the stamp of tokens to remove
35          * @return a list of the removed tokens
36          */
37         public List removeTokensByStamp(List tokenList, String stamp) {
38             List returnList = new LinkedList();
39             for (ListIterator i = tokenList.listIterator(); i.hasNext();) {
40                 ParallelToken token = (ParallelToken) i.next();
41                 if (token.getLastCombineStamp().equals(stamp)) {
42                     returnList.add(token);
43                     i.remove();
44                 }
45             }
46             return returnList;
47         }
48     
49     
50         /**
51          * Combines the scores from all the feature stream in the given CombineToken. The feature stream(s) are represented
52          * in the form of ParallelTokens in the given CombineToken.
53          *
54          * @param combineToken the CombineToken on which to combine the feature stream scores
55          */
56         public void combineScore(CombineToken combineToken) {
57     
58             // sort all the parallel tokens in the CombineToken
59             // according to their last combine time
60     
61             List tokenList = Arrays.asList
62                     (combineToken.getParallelTokens().toArray());
63     
64             // System.out.println("TokenList size: " + tokenList.size());
65     
66             double logHighestCombinedScore = Double.NEGATIVE_INFINITY;
67             List highestList = null;
68     
69             while (tokenList.size() > 0) {
70                 ParallelToken firstToken = (ParallelToken) tokenList.get(0);
71                 List sameStampList = removeTokensByStamp
72                         (tokenList, firstToken.getLastCombineStamp());
73     
74                 double logCombinedScore = getCombinedScore(sameStampList);
75     
76                 // System.out.println("logCombinedScore = " + logCombinedScore);
77                 if (logCombinedScore > logHighestCombinedScore ||
78                         logHighestCombinedScore == Double.NEGATIVE_INFINITY) {
79                     logHighestCombinedScore = logCombinedScore;
80                     highestList = sameStampList;
81                 }
82             }
83     
84             assert highestList != null & highestList.size() > 1;
85             // System.out.println("Highest TokenList size: " + highestList.size());
86     
87             for (Iterator i = highestList.iterator(); i.hasNext();) {
88                 ParallelToken token = (ParallelToken) i.next();
89                 token.setCombinedScore((float) logHighestCombinedScore);
90             }
91     
92             combineToken.clear();
93             combineToken.addAll(highestList);
94             combineToken.setCombinedScore((float) logHighestCombinedScore);
95         }
96     
97     
98         /**
99          * Calculates the combined score of the given list of tokens. This method assumes that all tokens have the same last
100          * combine time. It will retain only the highest scoring token of each stream and remove all the other lower scoring
101          * tokens from the list.
102          *
103          * @param sameStampTokenList the token list
104          * @return the combined log score
105          */
106         private double getCombinedScore(List sameStampTokenList) {
107             Map uniqueMap = new HashMap();
108             int combineTime = -1;
109     
110             // first retain the highest scoring token from each stream 
111             for (Iterator i = sameStampTokenList.iterator(); i.hasNext();) {
112                 ParallelToken token = (ParallelToken) i.next();
113                 String modelName = token.getModelName();
114                 ParallelToken tokenInMap =
115                         (ParallelToken) uniqueMap.get(modelName);
116                 if (tokenInMap == null) {
117                     uniqueMap.put(modelName, token);
118                 } else {
119                     if (token.getFeatureScore() > tokenInMap.getFeatureScore()) {
120                         uniqueMap.put(modelName, token);
121                     }
122                 }
123             }
124     
125             // clear the list
126             sameStampTokenList.clear();
127             assert sameStampTokenList.size() == 0;
128     
129             // now calculate the combinedScore
130             double logTotalScore = 0;
131     
132             for (Iterator i = uniqueMap.values().iterator(); i.hasNext();) {
133                 ParallelToken pToken = (ParallelToken) i.next();
134                 // System.out.println("Highest: " + tokenToString(pToken));
135                 sameStampTokenList.add(pToken);
136     
137                 // in linear domain, the following expression is:
138                 // score = pToken.getFeatureScore()^pToken.getEta()
139     
140                 double logScore = pToken.getFeatureScore() * pToken.getEta();
141     
142                 // in linear domain, the following expression is:
143                 // totalScore *= score
144     
145                 logTotalScore += logScore;
146             }
147     
148             return logTotalScore;
149         }

162         private String tokenToString(ParallelToken token) {
163             return (" (" + token.getLastCombineTime() + "," +
164                     token.getModelName() + "," +
165                     token.getFeatureScore() + ") ");
166         }
