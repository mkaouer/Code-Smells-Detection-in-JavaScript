12     package edu.cmu.sphinx.linguist.dictionary;
13     
14     import edu.cmu.sphinx.linguist.acoustic.UnitManager;
15     import edu.cmu.sphinx.util.props.Configurable;
16     import edu.cmu.sphinx.util.props.S4Boolean;
17     import edu.cmu.sphinx.util.props.S4Component;
18     import edu.cmu.sphinx.util.props.S4String;
19     
20     import java.io.IOException;
21     
22     /**
23      * Provides a generic interface to a dictionary. The dictionary is responsibile for determining how a word is
24      * pronounced.
25      */
26     public interface Dictionary extends Configurable {
27     
28         /** Spelling of the sentence start word. */
29         public static final String SENTENCE_START_SPELLING = "<s>";
30         /** Spelling of the sentence end word. */
31         public static final String SENTENCE_END_SPELLING = "</s>";
32         /** Spelling of the 'word' that marks a silence */
33         public static final String SILENCE_SPELLING = "<sil>";
34     
35         /** The name of the SphinxProperty for the dictionary file path. */
36         @S4String
37         public static final String PROP_DICTIONARY = "dictionaryPath";
38     
39         /** The name of the SphinxProperty for the filler dictionary file path. */
40         @S4String
41         public static final String PROP_FILLER_DICTIONARY = "fillerPath";
42     
43         /** The name of the SphinxProperty that specifies whether to add a duplicate SIL-ending pronunication. */
44         @S4Boolean(defaultValue = false)
45         public static final String PROP_ADD_SIL_ENDING_PRONUNCIATION = "addSilEndingPronunciation";
46     
47         /**
48          * The name of the SphinxProperty that specifies the word to substitute when a lookup fails to find the word in the
49          * dictionary. If this is not set, no substitute is performed.
50          */
51         @S4String(mandatory = false)
52         public static final String PROP_WORD_REPLACEMENT = "wordReplacement";
53     
54         /**
55          * The name of the SphinxProperty that specifies whether the dictionary should return null if a word is not found in
56          * the dictionary, or whether it should throw an error. If true, a null is returned for words that are not found in
57          * the dictionary (and the 'PROP_WORD_REPLACEMENT' property is not set).
58          */
59         @S4Boolean(defaultValue = false)
60         public static final String PROP_ALLOW_MISSING_WORDS = "allowMissingWords";
61     
62         /**
63          * The SphinxProperty that specifies whether the Dictionary.getWord() method should return a Word object even if the
64          * word does not exist in the dictionary. If this property is true, and property allowMissingWords is also true, the
65          * method will return a Word, but the Word will have null Pronunciations. Otherwise, the method will return null.
66          * This property is usually only used for testing purposes.
67          */
68         @S4Boolean(defaultValue = false)
69         public static final String PROP_CREATE_MISSING_WORDS = "createMissingWords";
70     
71         /** The sphinx property that defines the name of the unit manager that is used to convert strings to Unit objects */
72         @S4Component(type = UnitManager.class, defaultClass = UnitManager.class)
73         public static final String PROP_UNIT_MANAGER = "unitManager";
74     
75     
76         /**
77          * Returns a Word object based on the spelling and its classification. The behavior of this method is also affected
78          * by the properties wordReplacement, allowMissingWords, and createMissingWords.
79          *
80          * @param text the spelling of the word of interest.
81          * @return a Word object
82          * @see edu.cmu.sphinx.linguist.dictionary.Pronunciation
83          */
84         public Word getWord(String text);
85     
86     
87         /**
88          * Returns the sentence start word.
89          *
90          * @return the sentence start word
91          */
92         public Word getSentenceStartWord();
93     
94     
95         /**
96          * Returns the sentence end word.
97          *
98          * @return the sentence end word
99          */
100         public Word getSentenceEndWord();
101     
102     
103         /**
104          * Returns the silence word.
105          *
106          * @return the silence word
107          */
108         public Word getSilenceWord();
109     
110     
111         /**
112          * Returns the set of all possible word classifications for this dictionary.
113          *
114          * @return the set of all possible word classifications
115          */
116         public WordClassification[] getPossibleWordClassifications();
117     
118     
119         /**
120          * Gets the set of all filler words in the dictionary
121          *
122          * @return an array (possibly empty) of all filler words
123          */
124         public Word[] getFillerWords();
125     
126     
127         /**
128          * Allocates the dictionary
129          *
130          * @throws IOException if there is trouble loading the dictionary
131          */
132         public void allocate() throws IOException;
133     
134     
135         /** Deallocates the dictionary */
136         public void deallocate();
137     }
