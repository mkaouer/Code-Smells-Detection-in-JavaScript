103         /** Allocate the resources for the Engine and put it in the ALLOCATED, RESUMED, QUEUE_EMPTY state. */
104         public void allocate()
105                 throws EngineException, EngineStateError {
106             // We don't need the following steps to be atomic
107             // so there's no need to synchronize on engineStateLock
108             if (testEngineState(ALLOCATED)) {
109                 return;
110             }
111     
112             // Temporarily go in to the ALLOCATING_RESOURCES state.
113             long[] states = setEngineState(CLEAR_ALL_STATE, ALLOCATING_RESOURCES);
114             postEngineAllocatingResources(states[0], states[1]);
115     
116             // Go in to the ALLOCATED, RESUMED, LISTENING, and FOCUS_ON states.
117             // Subclasses with shared engines should check all states before
118             // changing them here.
119             synchronized (engineStateLock) {
120                 long newState = ALLOCATED | RESUMED | LISTENING | FOCUS_ON;
121                 states = setEngineState(CLEAR_ALL_STATE, newState);
122             }
123             postEngineAllocated(states[0], states[1]);
124             handleAllocate();
125         }

127     
128         /** Deallocate the resources for the Engine and put it in the DEALLOCATED state. */
129     
130         public void deallocate()
131                 throws EngineException, EngineStateError {
132             // We don't need the following steps to be atomic
133             // so there's no need to synchronize on engineStateLock
134     
135             if (testEngineState(DEALLOCATED))
136                 return;
137     
138             // Clean up the focus state
139             releaseFocus();
140     
141             // Temporarily go in to the DEALLOCATING_RESOURCES state.
142             // Make sure we kill the PAUSE/RESUME, LISTENING, FOCUS states etc.
143             long[] states = setEngineState(CLEAR_ALL_STATE, DEALLOCATING_RESOURCES);
144             postEngineDeallocatingResources(states[0], states[1]);
145     
146             // Go in to the DEALLOCATED state.
147             states = setEngineState(CLEAR_ALL_STATE, DEALLOCATED);
148             postEngineDeallocated(states[0], states[1]);
149             handleDeallocate();
150         }

223         private static URL gnameToURL(URL baseURL, String grammarName) throws MalformedURLException {
224     
225             // Convert each period in the grammar name to a slash "/"
226             // Append a slash and the converted grammar name to the base URL
227             // Append the ".gram" suffix
228             grammarName = grammarName.replace('.', '/');
229             String urlstr = "";
230             if (baseURL != null) urlstr = baseURL.toString();
231             urlstr = urlstr.concat(((urlstr.lastIndexOf((int) '/')
232                     == (urlstr.length() - 1))
233                     ? ""
234                     : "/") + grammarName + ".gram");
235     
236             URL grammarURL = null;
237             try {
238                 grammarURL = new URL(urlstr);
239             }
240             catch (MalformedURLException me) {
241                 grammarURL = ClassLoader.getSystemResource(urlstr);
242                 if (grammarURL == null)
243                     throw new MalformedURLException(urlstr);
244             }
245     
246             return grammarURL;
247         }

250         /** From javax.speech.recognition.Recognizer. */
251         public RuleGrammar loadJSGF(URL context, String grammarName,
252                                     boolean loadImports, boolean reloadGrammars,
253                                     Vector loadedGrammars)
254                 throws GrammarException, MalformedURLException,
255                 IOException, EngineStateError {
256             checkEngineState(DEALLOCATED | DEALLOCATING_RESOURCES);
257     
258             URL grammarURL = gnameToURL(context, grammarName);
259     
260             RuleGrammar G = JSGFParser.newGrammarFromJSGF(grammarURL, this);
261     
262             if (G == null) {
263                 throw new IOException();    // Should never happen
264             }
265     
266             if (loadImports) loadImports(this, G, context, true, reloadGrammars, loadedGrammars);
267     
268             if (G.getName() != null) {
269                 storeGrammar(G);
270             }
271             return G;
272         }

286         /** Get a list of loaded or defined RuleGrammars known to the Recognizer. From javax.speech.recognition.Recognizer. */
287         public RuleGrammar[] listRuleGrammars() throws EngineStateError {
288             checkEngineState(DEALLOCATED | DEALLOCATING_RESOURCES);
289             if (grammarList == null) {
290                 return new RuleGrammar[0];
291             }
292             RuleGrammar rl[] = new RuleGrammar[grammarList.size()];
293             int i = 0;
294             Enumeration e = grammarList.elements();
295             while (e.hasMoreElements()) {
296                 rl[i++] = (RuleGrammar) e.nextElement();
297             }
298             return rl;

326         /** Commit grammar changes. From javax.speech.recognition.Recognizer. */
327         public void commitChanges()
328                 throws GrammarException, EngineStateError {
329             checkEngineState(DEALLOCATED | DEALLOCATING_RESOURCES);
330     
331             // All cases:
332             //   If there is a pending suspend call, it can be cleared.
333             //   THIS IS NOT IMPLEMENTED YET
334             // Case 1:
335             //   If the recognizer is in the LISTENING state we should
336             //   first transition to the SUSPENDED state, then commit,
337             //   then return the LISTENING state.
338             // Case 2:
339             //   If the recognizer is in the SUSPENDED state, we remain
340             //   in that state, commit, then return to LISTENING state.
341             //   EXCEPTION: if the recognizer is still issuing events
342             //   for a Result that brought it to the SUSPENDED state,
343             //   the commit should be deferred until the event processing
344             //   is complete -- THIS IS NOT IMPLEMENTED YET.
345             // Case 3:
346             //   If the recognizer is in the PROCESSING state, the
347             //   commit should be deferred until following result is
348             //   finalized and appropriate events have been issued.
349     
350             // The following hack avoids inappropriately going from
351             // PROCESSING to LISTENING states.  It does not correctly
352             // handle clearing of explicit suspend requests.
353             boolean goBackToListening = false;
354     
355             if (testEngineState(LISTENING)) {
356                 goBackToListening = true;
357                 long[] states = setEngineState(LISTENING | PROCESSING, SUSPENDED);
358                 postRecognizerSuspended(states[0], states[1]);
359             }
360     
361             // If we're in the PROCESSING state we shouldn't modify grammars.
362     
363             // Prepare the grammars to be committed
364             linkGrammars();
365     
366             //GRAMMAR_CHANGES_COMMITTED events are sent in commitChangesInternal
367     
368             // Activate or Deactivate any grammars
369             checkForModalGrammars();
370             notifyGrammarActivation();
371     
372             if (goBackToListening) {
373                 long[] states = setEngineState(SUSPENDED | PROCESSING, LISTENING);
374                 postChangesCommitted(states[0], states[1], null);
375             }
376         }

938         /**
939          * Cycle through each rule in each grammar calling the method changeRule for each method that had changed or is new.
940          * Also make a list of all enabled rules.
941          * <p/>
942          * Nominally calling changeRule method will propogate the new rule to the underlying recognizer
943          */
944         protected void commitChangeInternal() {
945             boolean haveChanges = false;
946             Vector enabled = new Vector();
947             RuleGrammar G[] = listRuleGrammars();
948             if (!supportsNULL || !supportsVOID) {
949                 try {
950                     G = RecognizerUtilities.transform(G, !supportsNULL, !supportsVOID);
951                 } catch (GrammarException e) {
952                     e.printStackTrace();
953                     return;
954                 }
955             }
956             for (int i = 0; i < G.length; i++) {
957                 BaseRuleGrammar JG = (BaseRuleGrammar) G[i];
958                 if (JG.grammarChanged) {
959                     haveChanges = true;
960                     break;
961                 }
962             }
963             // if (reloadAll) haveChanges=true;
964             if (haveChanges) {
965                 startGrammarChanges();
966             }
967             // find and output rules that have changes
968             for (int i = 0; i < G.length; i++) {
969                 String gname = G[i].getName();
970                 BaseRuleGrammar JG = (BaseRuleGrammar) G[i];
971                 String rnames[] = G[i].listRuleNames();
972                 for (int j = 0; j < rnames.length; j++) {
973                     String ruleName = rnames[j];
974                     if (G[i].isEnabled(ruleName)) {
975                         enabled.addElement(gname + "_" + ruleName);
976                     }
977                     if (!haveChanges
978                             || (!JG.isRuleChanged(ruleName) && !reloadAll)) {
979                         continue;
980                     }
981                     JG.setRuleChanged(ruleName, false);
982                     boolean isPublic = false;
983                     try {
984                         isPublic = G[i].isRulePublic(ruleName);
985                     } catch (IllegalArgumentException nse) {
986                     }
987                     Rule rule = G[i].getRule(ruleName);
988                     currentGrammar = G[i];
989                     changeRule(gname, ruleName, rule, isPublic);
990                 }
991                 JG.grammarChanged = false;
992                 JG.postGrammarChangesCommitted(); // send events
993             }
994             if (haveChanges) {
995                 endGrammarChanges();
996             }
997             changeEnabled(enabled);
998         }

1034         /** Load grammars imported by the specified RuleGrammar if they are not already loaded. */
1035         static private void loadImports(javax.speech.recognition.Recognizer R,
1036                                         RuleGrammar G,
1037                                         URL context,
1038                                         boolean recurse,
1039                                         boolean relo,
1040                                         Vector grams)
1041                 throws GrammarException, IOException {
1042             RuleGrammar G2 = null;
1043             RuleName imports[] = G.listImports();
1044     
1045             if (imports != null) {
1046                 for (int i = 0; i < imports.length; i++) {
1047                     //RecognizerUtilities.debugMessageOut("Checking import " +
1048                     //                                   imports[i].getRuleName());
1049                     String gname = imports[i].getFullGrammarName();
1050                     RuleGrammar GI = R.getRuleGrammar(gname);
1051     
1052                     if (GI == null) {
1053                         URL grammarURL = gnameToURL(
1054                                 context, imports[i].getFullGrammarName());
1055                         G2 = JSGFParser.newGrammarFromJSGF(grammarURL, R);
1056                         if (G2 != null) {
1057                             if (grams != null) {
1058                                 grams.addElement(G2);
1059                             }
1060                             if (recurse) {
1061                                 loadImports(R, G2, context, recurse, relo, grams);
1062                             }
1063                         }
1064                     }
1065                 }
1066                 loadFullQualifiedRules(R, G, context, recurse, relo, grams);
1067             }
1068         }

1071         /**
1072          * Load grammars imported by a fullqualified Rule Token if they are not already loaded.
1073          *
1074          * @param r
1075          * @param g
1076          * @param context
1077          * @param recurse
1078          * @param relo
1079          * @param grams
1080          * @throws IOException
1081          * @throws GrammarException
1082          */
1083         private static void loadFullQualifiedRules(
1084                 javax.speech.recognition.Recognizer r,
1085                 RuleGrammar g,
1086                 URL context,
1087                 boolean recurse,
1088                 boolean relo,
1089                 Vector grams) throws GrammarException, IOException {
1090     
1091             String[] ruleNames = g.listRuleNames();
1092             //go through every rule
1093             for (int i = 0; i < ruleNames.length; i++) {
1094                 String rule = g.getRuleInternal(ruleNames[i]).toString();
1095                 //check for rule-Tokens
1096                 int index = 0;
1097                 while (index < rule.length()) {
1098                     index = rule.indexOf("<", index);
1099                     if (index < 0) {
1100                         index = rule.length();
1101                     } else {
1102                         //extract rulename
1103                         RuleName rn = new RuleName(rule.substring(
1104                                 index + 1, rule.indexOf(">", index + 1)).trim());
1105                         index = rule.indexOf(">", index) + 1;
1106                         //check for fullqualified rulename
1107                         if (rn.getFullGrammarName() != null) {
1108                             String gname = rn.getFullGrammarName();
1109                             RuleGrammar GI = r.getRuleGrammar(gname);
1110                             if (GI == null) {
1111                                 URL grammarURL = gnameToURL(context, gname);
1112                                 RuleGrammar G2 = JSGFParser.newGrammarFromJSGF(
1113                                         grammarURL, r);
1114     
1115                                 if (G2 != null) {
1116                                     if (grams != null) {
1117                                         grams.addElement(G2);
1118                                     }
1119                                     if (recurse) {
1120                                         loadImports(r,
1121                                                 G2,
1122                                                 context,
1123                                                 recurse,
1124                                                 relo,
1125                                                 grams);
1126                                     }
1127                                 }
1128                             }
1129                         }
1130                     }
1131                 }
1132             }
1133         }

1162         /**
1163          * Determine if the given Grammar is active.  This is a combination of the enabled state and activation modes of the
1164          * Grammar as well as the current focus state of the recognizer.  NOT JSAPI.
1165          */
1166         protected boolean isActive(Grammar grammar) {
1167             //[[[WDW - check engineState?]]]
1168             if (!grammar.isEnabled()) {
1169                 return false;
1170             } else if (grammar.getActivationMode() == Grammar.GLOBAL) {
1171                 return true;
1172             } else if (testEngineState(FOCUS_ON)) {
1173                 if (grammar.getActivationMode() == Grammar.RECOGNIZER_MODAL) {
1174                     return true;
1175                 } else if (!hasModalGrammars) {
1176                     return true;
1177                 }
1178             }
1179             return false;
1180         }

1183         /** Notify any grammars if their activation state has been changed. */
1184         protected void notifyGrammarActivation() {
1185             if (grammarList == null) {
1186                 return;
1187             }
1188             Enumeration e = grammarList.elements();
1189             while (e.hasMoreElements()) {
1190                 BaseRuleGrammar rg = (BaseRuleGrammar) e.nextElement();
1191                 boolean active = isActive(rg);
1192                 if (active != rg.grammarActive) {
1193                     rg.grammarActive = active;
1194                     if (active) {
1195                         rg.postGrammarActivated();
1196                     } else {
1197                         rg.postGrammarDeactivated();
1198                     }
1199                 }
1200             }
1201         }

1243         /**
1244          * Dispatch a SpeechEvent.  This is a method from SpeechEventDispatcher. The dispatcher should notify all listeners
1245          * of the speech event from this method.
1246          */
1247         public void dispatchSpeechEvent(SpeechEvent event) {
1248             switch (event.getId()) {
1249                 case RecognizerEvent.CHANGES_COMMITTED:
1250                     fireChangesCommitted((RecognizerEvent) event);
1251                     break;
1252                 case RecognizerEvent.FOCUS_GAINED:
1253                     fireFocusGained((RecognizerEvent) event);
1254                     break;
1255                 case RecognizerEvent.FOCUS_LOST:
1256                     fireFocusLost((RecognizerEvent) event);
1257                     break;
1258                 case RecognizerEvent.RECOGNIZER_PROCESSING:
1259                     fireRecognizerProcessing((RecognizerEvent) event);
1260                     break;
1261                 case RecognizerEvent.RECOGNIZER_SUSPENDED:
1262                     fireRecognizerSuspended((RecognizerEvent) event);
1263                     break;
1264     
1265                 case ResultEvent.AUDIO_RELEASED:
1266                     fireAudioReleased((ResultEvent) event);
1267                     break;
1268                 case ResultEvent.GRAMMAR_FINALIZED:
1269                     fireGrammarFinalized((ResultEvent) event);
1270                     break;
1271                 case ResultEvent.RESULT_ACCEPTED:
1272                     fireResultAccepted((ResultEvent) event);
1273                     break;
1274                 case ResultEvent.RESULT_CREATED:
1275                     fireResultCreated((ResultEvent) event);
1276                     break;
1277                 case ResultEvent.RESULT_REJECTED:
1278                     fireResultRejected((ResultEvent) event);
1279                     break;
1280                 case ResultEvent.RESULT_UPDATED:
1281                     fireResultUpdated((ResultEvent) event);
1282                     break;
1283                 case ResultEvent.TRAINING_INFO_RELEASED:
1284                     fireTrainingInfoReleased((ResultEvent) event);
1285                     break;
1286     
1287                     // Defer to BaseEngine to handle the rest.
1288                     //
1289                 default:
1290                     super.dispatchSpeechEvent(event);
1291                     break;
1292             }
1293         }
