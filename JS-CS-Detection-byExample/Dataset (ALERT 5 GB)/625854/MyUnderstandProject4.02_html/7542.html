25     /** Represents a single state in an HMM */
26     public class SenoneHMMState implements HMMState {
27     
28         private SenoneHMM hmm;
29         private int state;
30         HMMStateArc[] arcs;
31         private boolean isEmitting;
32         private Senone senone;
33     
34         private static int objectCount;
35     
36     
37         /**
38          * Constructs a SenoneHMMState
39          *
40          * @param hmm   the hmm for this state
41          * @param which the index for this particular state
42          */
43         SenoneHMMState(SenoneHMM hmm, int which) {
44             this.hmm = hmm;
45             this.state = which;
46             this.isEmitting = ((hmm.getTransitionMatrix().length - 1) != state);
47             if (isEmitting) {
48                 SenoneSequence ss = hmm.getSenoneSequence();
49                 senone = ss.getSenones()[state];
50             }
51             Utilities.objectTracker("HMMState", objectCount++);
52         }
53     
54     
55         /**
56          * Gets the HMM associated with this state
57          *
58          * @return the HMM
59          */
60         public HMM getHMM() {
61             return hmm;
62         }
63     
64     
65         /**
66          * Gets the state
67          *
68          * @return the state
69          */
70         public int getState() {
71             return state;
72         }
73     
74     
75         /**
76          * Gets the score for this HMM state
77          *
78          * @param feature the feature to be scored
79          * @return the acoustic score for this state.
80          */
81         public float getScore(Data feature) {
82             return senone.getScore(feature);
83         }
84     
85     
86         /**
87          * Gets the scores for each mixture component in this HMM state
88          *
89          * @param feature the feature to be scored
90          * @return the acoustic scores for the components of this state.
91          */
92         public float[] calculateComponentScore(Data feature) {
93             SenoneSequence ss = hmm.getSenoneSequence();
94             return senone.calculateComponentScore(feature);
95         }
96     
97     
98         /**
99          * Gets the senone for this HMM state
100          *
101          * @return the senone for this state.
102          */
103         public Senone getSenone() {
104             return senone;
105         }
106     
107     
108         /**
109          * Determines if two HMMStates are equal
110          *
111          * @param other the state to compare this one to
112          * @return true if the states are equal
113          */
114         public boolean equals(Object other) {
115             if (this == other) {
116                 return true;
117             } else if (!(other instanceof SenoneHMMState)) {
118                 return false;
119             } else {
120                 SenoneHMMState otherState = (SenoneHMMState) other;
121                 return this.hmm == otherState.hmm &&
122                         this.state == otherState.state;
123             }
124         }
125     
126     
127         /**
128          * Returns the hashcode for this state
129          *
130          * @return the hashcode
131          */
132         public int hashCode() {
133             return hmm.hashCode() + state;
134         }
135     
136     
137         /**
138          * Determines if this HMMState is an emittting state
139          *
140          * @return true if the state is an emitting state
141          */
142         // TODO: We may have non-emitting entry states as well.
143         public final boolean isEmitting() {
144             return isEmitting;
145         }
146     
147     
148         /**
149          * Retrieves the state of successor states for this state
150          *
151          * @return the set of successor state arcs
152          */
153         public HMMStateArc[] getSuccessors() {
154             if (arcs == null) {
155                 List list = new ArrayList();
156                 float[][] transitionMatrix = hmm.getTransitionMatrix();
157     
158                 // dumpMatrix("arc", transitionMatrix);
159                 for (int i = 0; i < transitionMatrix.length; i++) {
160                     if (transitionMatrix[state][i] > LogMath.getLogZero()) {
161                         HMMStateArc arc = new HMMStateArc(hmm.getState(i),
162                                 transitionMatrix[state][i]);
163                         list.add(arc);
164                     }
165                 }
166                 arcs = (HMMStateArc[]) list.toArray(new HMMStateArc[list.size()]);
167             }
168             return arcs;
169         }
170     
171     
172         /**
173          * Dumps out a matrix
174          *
175          * @param title  the title for the dump
176          * @param matrix the matrix to dump
177          */
178         private void dumpMatrix(String title, float[][] matrix) {
179             System.out.println(" -- " + title + " --- ");
180             for (int i = 0; i < matrix.length; i++) {
181                 for (int j = 0; j < matrix[i].length; j++) {
182                     System.out.print(" " + matrix[i][j]);
183                 }
184                 System.out.println();
185             }
186         }
187     
188     
189         /**
190          * Determines if this state is an exit state of the HMM
191          *
192          * @return true if the state is an exit state
193          */
194         public boolean isExitState() {
195             // return (hmm.getTransitionMatrix().length - 1) == state;
196             return !isEmitting;
197         }
198     
199     
200         /**
201          * returns a string represntation of this object
202          *
203          * @return a string representation
204          */
205         public String toString() {
206             return "HMMS " + hmm + " state " + state;
207         }
208     }
