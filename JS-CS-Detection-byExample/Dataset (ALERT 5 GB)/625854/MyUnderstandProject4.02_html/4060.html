19     /**
20      * Computes the delta and double delta of input cepstrum (or plp or ...). The delta is the first order derivative and
21      * the double delta (a.k.a. delta delta) is the second order derivative of the original cepstrum. They help model the
22      * speech signal dynamics. The output data is a {@link FloatData} object with a float array of size three times the
23      * original cepstrum, formed by the concatenation of cepstra, delta cepstra, and double delta cepstra. The output is the
24      * feature vector used by the decoder. Figure 1 shows the arrangement of the output feature data array:
25      * <p/>
26      * <img src="doc-files/feature.jpg"> <br> <b>Figure 1: Layout of the returned features. </b>
27      * <p/>
28      * Suppose that the original cepstrum has a length of N, the first N elements of the feature are just the original
29      * cepstrum, the second N elements are the delta of the cepstrum, and the last N elements are the double delta of the
30      * cepstrum.
31      * <p/>
32      * Figure 2 below shows pictorially the computation of the delta and double delta of a cepstrum vector, using the last 3
33      * cepstra and the next 3 cepstra. <img src="doc-files/deltas.jpg"> <br> <b>Figure 2: Delta and double delta vector
34      * computation. </b>
35      * <p/>
36      * Refering to Figure 2, the delta is computed by subtracting the cepstrum that is two frames behind of the current
37      * cepstrum from the cepstrum that is two frames ahead of the current cepstrum. The computation of the double delta is
38      * similar. It is computed by subtracting the delta cepstrum one time frame behind from the delta cepstrum one time
39      * frame ahead. Replacing delta cepstra with cepstra, this works out to a formula involving the cepstra that are one and
40      * three behind and after the current cepstrum.
41      */
42     @SuppressWarnings({"UnnecessaryLocalVariable"})
43     public class DeltasFeatureExtractor extends BaseDataProcessor {
44     
45         /** The name of the SphinxProperty for the window of the DeltasFeatureExtractor. */
46         @S4Integer(defaultValue = 3)
47         public static final String PROP_FEATURE_WINDOW = "windowSize";
48         /** The default value of PROP_FEATURE_WINDOW. */
49         public static final int PROP_FEATURE_WINDOW_DEFAULT = 3;
50         private int cepstraBufferSize;
51         private int cepstraBufferEdge;
52         private int bufferPosition;
53         private int currentPosition;
54         private int window;
55         private int jp1, jp2, jp3, jf1, jf2, jf3;
56         private DoubleData[] cepstraBuffer;
57         private DataEndSignal dataEndSignal;
58         private List<Data> outputQueue;
59     
60     
61         /*
62         * (non-Javadoc)
63         *
64         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
65         */
66         public void newProperties(PropertySheet ps) throws PropertyException {
67             super.newProperties(ps);
68             window = ps.getInt(PROP_FEATURE_WINDOW);
69         }

71     
72         /*
73         * (non-Javadoc)
74         *
75         * @see edu.cmu.sphinx.frontend.DataProcessor#initialize(edu.cmu.sphinx.frontend.CommonConfig)
76         */
77         public void initialize() {
78             super.initialize();
79             cepstraBufferSize = 256;
80             cepstraBuffer = new DoubleData[cepstraBufferSize];
81             cepstraBufferEdge = cepstraBufferSize - (window * 2 + 2);
82             outputQueue = new Vector<Data>();
83             reset();
84         }

86     
87         /** Resets the DeltasFeatureExtractor to be ready to read the next segment of data. */
88         private void reset() {
89             bufferPosition = 0;
90             currentPosition = 0;
91         }
92     
93     
94         /**
95          * Returns the next Data object produced by this DeltasFeatureExtractor.
96          *
97          * @return the next available Data object, returns null if no Data is available
98          * @throws DataProcessingException if there is a data processing error
99          */
100         public Data getData() throws DataProcessingException {
101             if (outputQueue.size() == 0) {
102                 Data input = getNextData();
103                 if (input != null) {
104                     if (input instanceof DoubleData) {
105                         addCepstrum((DoubleData) input);
106                         computeFeatures(1);
107                     } else if (input instanceof DataStartSignal) {
108                         dataEndSignal = null;
109                         outputQueue.add(input);
110                         Data start = getNextData();
111                         int n = processFirstCepstrum(start);
112                         computeFeatures(n);
113                         if (dataEndSignal != null) {
114                             outputQueue.add(dataEndSignal);
115                         }
116                     } else if (input instanceof DataEndSignal) {
117                         // when the DataEndSignal is right at the boundary
118                         int n = replicateLastCepstrum();
119                         computeFeatures(n);
120                         outputQueue.add(input);
121                     }
122                 }
123             }
124             if (outputQueue.size() > 0) {
125                 Data feature = outputQueue.remove(0);
126                 return feature;
127             } else {
128                 return null;
129             }
130         }
131     
132     
133         private Data getNextData() throws DataProcessingException {
134             Data d = getPredecessor().getData();
135             while (d != null && !(d instanceof DoubleData || d instanceof DataEndSignal || d instanceof DataStartSignal)) {
136                 outputQueue.add(d);
137                 d = getPredecessor().getData();
138             }
139     
140             return d;
141         }
142     
143     
144         /**
145          * Replicate the given cepstrum Data object into the first window+1 number of frames in the cepstraBuffer. This is
146          * the first cepstrum in the segment.
147          *
148          * @param cepstrum the Data to replicate
149          * @return the number of Features that can be computed
150          */
151         private int processFirstCepstrum(Data cepstrum)
152                 throws DataProcessingException {
153             if (cepstrum instanceof DataEndSignal) {
154                 outputQueue.add(cepstrum);
155                 return 0;
156             } else if (cepstrum instanceof DataStartSignal) {
157                 throw new Error("Too many UTTERANCE_START");
158             } else {
159                 // At the start of an utterance, we replicate the first frame
160                 // into window+1 frames, and then read the next "window" number
161                 // of frames. This will allow us to compute the delta-
162                 // double-delta of the first frame.
163                 Arrays.fill(cepstraBuffer, 0, window + 1, cepstrum);
164                 bufferPosition = window + 1;
165                 bufferPosition %= cepstraBufferSize;
166                 currentPosition = window;
167                 currentPosition %= cepstraBufferSize;
168                 int numberFeatures = 1;
169                 dataEndSignal = null;
170                 for (int i = 0; i < window; i++) {
171                     Data next = getNextData();
172                     if (next != null) {
173                         if (next instanceof DoubleData) {
174                             // just a cepstra
175                             addCepstrum((DoubleData) next);
176                         } else if (next instanceof DataEndSignal) {
177                             // end of segment cepstrum
178                             dataEndSignal = (DataEndSignal) next;
179                             replicateLastCepstrum();
180                             numberFeatures += i;
181                             break;
182                         } else if (next instanceof DataStartSignal) {
183                             throw new Error("Too many UTTERANCE_START");
184                         }
185                     }
186                 }
187                 jp1 = currentPosition - 1;
188                 jp2 = currentPosition - 2;
189                 jp3 = currentPosition - 3;
190                 jf1 = currentPosition + 1;
191                 jf2 = currentPosition + 2;
192                 jf3 = currentPosition + 3;
193                 if (jp3 > cepstraBufferEdge) {
194                     jf3 %= cepstraBufferSize;
195                     jf2 %= cepstraBufferSize;
196                     jf1 %= cepstraBufferSize;
197                     jp1 %= cepstraBufferSize;
198                     jp2 %= cepstraBufferSize;
199                     jp3 %= cepstraBufferSize;
200                 }
201                 return numberFeatures;
202             }
203         }
204     
205     
206         /**
207          * Adds the given DoubleData object to the cepstraBuffer.
208          *
209          * @param cepstrum the DoubleData object to add
210          */
211         private void addCepstrum(DoubleData cepstrum) {
212             cepstraBuffer[bufferPosition++] = cepstrum;
213             bufferPosition %= cepstraBufferSize;
214         }
215     
216     
217         /**
218          * Replicate the last frame into the last window number of frames in the cepstraBuffer.
219          *
220          * @return the number of replicated Cepstrum
221          */
222         private int replicateLastCepstrum() {
223             DoubleData last;
224             if (bufferPosition > 0) {
225                 last = this.cepstraBuffer[bufferPosition - 1];
226             } else if (bufferPosition == 0) {
227                 last = cepstraBuffer[cepstraBuffer.length - 1];
228             } else {
229                 throw new Error("BufferPosition < 0");
230             }
231             for (int i = 0; i < window; i++) {
232                 addCepstrum(last);
233             }
234             return window;
235         }

238         /**
239          * Converts the Cepstrum data in the cepstraBuffer into a FeatureFrame.
240          *
241          * @param totalFeatures the number of Features that will be produced
242          */
243         private void computeFeatures(int totalFeatures) {
244             getTimer().start();
245             if (totalFeatures == 1) {
246                 computeFeature();
247             } else {
248                 // create the Features
249                 for (int i = 0; i < totalFeatures; i++) {
250                     computeFeature();
251                 }
252             }
253             getTimer().stop();
254         }
255     
256     
257         /** Computes the next Feature. */
258         private void computeFeature() {
259             Data feature = computeNextFeature();
260             outputQueue.add(feature);
261         }
262     
263     
264         /**
265          * Computes the next feature. Advances the pointers as well.
266          *
267          * @return the feature Data computed
268          */
269         private Data computeNextFeature() {
270             DoubleData currentCepstrum = cepstraBuffer[currentPosition++];
271             double[] mfc3f = cepstraBuffer[jf3++].getValues();
272             double[] mfc2f = cepstraBuffer[jf2++].getValues();
273             double[] mfc1f = cepstraBuffer[jf1++].getValues();
274             double[] current = currentCepstrum.getValues();
275             double[] mfc1p = cepstraBuffer[jp1++].getValues();
276             double[] mfc2p = cepstraBuffer[jp2++].getValues();
277             double[] mfc3p = cepstraBuffer[jp3++].getValues();
278             float[] feature = new float[current.length * 3];
279             // CEP; copy all the cepstrum data
280             int j = 0;
281             for (int k = 0; k < current.length; k++) {
282                 feature[j++] = (float) current[k];
283             }
284             // System.arraycopy(current, 0, feature, 0, j);
285             // DCEP: mfc[2] - mfc[-2]
286             for (int k = 0; k < mfc2f.length; k++) {
287                 feature[j++] = (float) (mfc2f[k] - mfc2p[k]);
288             }
289             // D2CEP: (mfc[3] - mfc[-1]) - (mfc[1] - mfc[-3])
290             for (int k = 0; k < mfc3f.length; k++) {
291                 feature[j++] = (float) ((mfc3f[k] - mfc1p[k]) - (mfc1f[k] - mfc3p[k]));
292             }
293             if (jp3 > cepstraBufferEdge) {
294                 jf3 %= cepstraBufferSize;
295                 jf2 %= cepstraBufferSize;
296                 jf1 %= cepstraBufferSize;
297                 currentPosition %= cepstraBufferSize;
298                 jp1 %= cepstraBufferSize;
299                 jp2 %= cepstraBufferSize;
300                 jp3 %= cepstraBufferSize;
301             }
302     
303             return (new FloatData(feature,
304                     currentCepstrum.getSampleRate(),
305                     currentCepstrum.getCollectTime(),
306                     currentCepstrum.getFirstSampleNumber()));
307         }
