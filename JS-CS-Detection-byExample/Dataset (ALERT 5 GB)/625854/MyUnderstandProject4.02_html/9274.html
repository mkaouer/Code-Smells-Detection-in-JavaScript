13     package edu.cmu.sphinx.linguist.flat;
14     
15     import edu.cmu.sphinx.linguist.UnitSearchState;
16     import edu.cmu.sphinx.linguist.acoustic.HMMPosition;
17     import edu.cmu.sphinx.linguist.acoustic.Unit;
18     import edu.cmu.sphinx.linguist.acoustic.UnitManager;
19     
20     /** Represents a unit in an SentenceHMMS */
21     public class UnitState extends SentenceHMMState implements UnitSearchState {
22     
23         private Unit unit;
24         private transient StatePath tail;
25         private HMMPosition position = HMMPosition.INTERNAL;
26     
27     
28         /**
29          * Creates a UnitState. Gets the left and right contexts from the unit itself.
30          *
31          * @param parent the parent state
32          * @param which  the index of the given state
33          * @param unit   the unit associated with this state
34          */
35         public UnitState(PronunciationState parent, int which, Unit unit) {
36             super("U", parent, which);
37             this.unit = unit;
38             Unit[] units = parent.getPronunciation().getUnits();
39             int length = units.length;
40     
41             // If the last phone is SIL, then we should be using
42             // a word-ending phone for the last phone. Decrementing
43             // length will make the phone before SIL the last phone.
44     
45             if (units[length - 1] == UnitManager.SILENCE && length > 1) {
46                 length--;
47             }
48     
49             if (length == 1) {
50                 position = HMMPosition.SINGLE;
51             } else if (which == 0) {
52                 position = HMMPosition.BEGIN;
53             } else if (which == length - 1) {
54                 position = HMMPosition.END;
55             }
56         }
57     
58     
59         /**
60          * Creates a UnitState with the given unit and HMM position.
61          *
62          * @param unit     the unit associated with this state
63          * @param position the HMM position of this unit
64          */
65         public UnitState(Unit unit, HMMPosition position) {
66             this.unit = unit;
67             this.position = position;
68         }
69     
70     
71         /**
72          * Gets the unit associated with this state
73          *
74          * @return the unit
75          */
76         public Unit getUnit() {
77             return unit;
78         }
79     
80     
81         /**
82          * Returns true if this unit is the last unit of the pronunciation
83          *
84          * @return <code>true</code> if the unit is the last unit
85          */
86         public boolean isLast() {
87             return position == HMMPosition.SINGLE || position == HMMPosition.END;
88         }
89     
90     
91         /**
92          * Gets the name for this state
93          *
94          * @return the name for this state
95          */
96         public String getName() {
97             return super.getName() + "<" + unit + ">";
98         }
99     
100     
101         /**
102          * Returns the value signature of this unit
103          *
104          * @return the value signature
105          */
106         public String getValueSignature() {
107             return unit.toString();
108         }
109     
110     
111         /**
112          * Gets the pretty name for this unit sate
113          *
114          * @return the pretty name
115          */
116         public String getPrettyName() {
117             return unit.toString();
118         }
119     
120     
121         /**
122          * Retrieves a short label describing the type of this state. Typically, subclasses of SentenceHMMState will
123          * implement this method and return a short (5 chars or less) label
124          *
125          * @return the short label.
126          */
127         public String getTypeLabel() {
128             return "Unit";
129         }
130     
131     
132         /**
133          * Gets the position for this unit
134          *
135          * @return the position for this unit
136          */
137         public HMMPosition getPosition() {
138             return position;
139         }
140     
141     
142         /**
143          * Returns the tail for this unit state.  This is used during the SentenceHMM contruction process. A UnitState may
144          * be expanded into a unit state followed by a tree of HMMStates.  Keeping the tail of the tree here helps us to
145          * quickly find the end of this collection of states when we are reusing states.
146          *
147          * @return the tail state
148          */
149         public StatePath getTail() {
150             return tail;
151         }
152     
153     
154         /**
155          * Sets the tail for this state
156          *
157          * @param tail the tail for the state
158          */
159         public void setTail(StatePath tail) {
160             this.tail = tail;
161         }
162     
163     
164         public boolean isUnit() {
165             return true;
166         }
167     
168     
169         /**
170          * Returns the state order for this state type
171          *
172          * @return the state order
173          */
174         public int getOrder() {
175             return 5;
176         }
177     }
