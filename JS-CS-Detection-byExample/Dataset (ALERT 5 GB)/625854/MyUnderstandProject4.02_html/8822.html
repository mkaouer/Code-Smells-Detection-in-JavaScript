10     package edu.cmu.sphinx.frontend.filter;
11     
12     import edu.cmu.sphinx.frontend.*;
13     import edu.cmu.sphinx.util.props.*;
14     
15     import java.util.*;
16     
17     /**
18      * Implements a dither for the incoming packet. The small random noise is added
19      * to the signal to avoid floating point errors and prevent the energy from
20      * being zero. <p/> Other {@link Data}objects are passed along unchanged through
21      * this Dither processor. <p/>
22      */
23     public class Dither extends BaseDataProcessor {
24     
25     	Random random;
26     
27     	/*
28     	 * (non-Javadoc)
29     	 * 
30     	 * @see
31     	 * edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util
32     	 * .props.PropertySheet)
33     	 */
34     	public void newProperties(PropertySheet ps) throws PropertyException {
35     		super.newProperties(ps);
36     		random = new Random(0);
37     	}
38     
39     	/**
40     	 * Returns the next Data object being processed by this Dither, or if it is
41     	 * a Signal, it is returned without modification.
42     	 * 
43     	 * @return the next available Data object, returns null if no Data object is
44     	 *         available
45     	 * @throws DataProcessingException
46     	 *             if there is a processing error
47     	 * @see Data
48     	 */
49     	public Data getData() throws DataProcessingException {
50     		Data input = getPredecessor().getData();
51     		getTimer().start();
52     		if (input != null && input instanceof DoubleData) {
53     			applyDither(((DoubleData) input).getValues());
54     		}
55     		getTimer().stop();
56     		return input;
57     	}
58     
59     	/**
60     	 * Applies dither to the given Audio. The preemphasis is applied in place.
61     	 * 
62     	 * @param in
63     	 *            audio data
64     	 */
65     	private void applyDither(double[] in) {
66     
67     		for (int i = 0; i < in.length; i++) {
68     			if (random.nextBoolean()) {
69     				in[i] = in[i] + 1;
70     			}
71     		}
72     	}
73     }
