62         /** Converts a byte array to a signed short value. */
63         static public short toShort(byte[] bytes, boolean bigEndian) {
64             if (bytes.length == 1) {
65                 return (short) bytes[0];
66             } else if (bigEndian) {
67                 return (short) ((bytes[0] << 8) | (0xff & bytes[1]));
68             } else {
69                 return (short) ((bytes[1] << 8) | (0xff & bytes[0]));
70             }
71         }

74         /** Converts a byte array into an unsigned short. */
75         static public int toUnsignedShort(byte[] bytes, boolean bigEndian) {
76             if (bytes.length == 1) {
77                 return 0xff & bytes[0];
78             } else if (bigEndian) {
79                 return ((bytes[0] & 0xff) << 8) | (0xff & bytes[1]);
80             } else {
81                 return ((bytes[1] & 0xff) << 8) | (0xff & bytes[0]);
82             }
83         }

86         /** Converts a short into a byte array. */
87         public static void toBytes(short sVal, byte[] bytes, boolean bigEndian) {
88             if (bigEndian) {
89                 bytes[0] = (byte) (sVal >> 8);
90                 bytes[1] = (byte) (sVal & 0xff);
91             } else {
92                 bytes[0] = (byte) (sVal & 0xff);
93                 bytes[1] = (byte) (sVal >> 8);
94             }
95         }

98         /**
99          * Convert the bytes starting at the given offset to a signed short based upon the AudioFormat.  If the frame size
100          * is 1, then the value is doubled to make it match a frame size of 2.
101          *
102          * @param format    the audio format
103          * @param byteArray the byte array
104          * @return a short
105          * @throws java.lang.ArrayIndexOutOfBoundsException
106          *
107          */
108         public static short bytesToShort(AudioFormat format,
109                                          byte[] byteArray) {
110             short result = 0;
111             Encoding encoding = format.getEncoding();
112             int frameSize = format.getFrameSize();
113     
114             if (encoding == Encoding.PCM_SIGNED) {
115                 result = toShort(byteArray, format.isBigEndian());
116                 if (frameSize == 1) {
117                     result = (short) (result << 8);
118                 }
119             } else if (encoding == Encoding.PCM_UNSIGNED) {
120                 int tmp = toUnsignedShort(byteArray, format.isBigEndian());
121                 if (frameSize == 1) {
122                     tmp = tmp << 8;
123                 }
124                 result = (short) (tmp - (2 << 14));
125             } else if (encoding == Encoding.ULAW) {
126                 result = ulawTable[byteArray[0] + 128];
127             } else {
128                 System.out.println("Unknown encoding: " + encoding);
129             }
130             return result;
131         }

134         /**
135          * Turns the AudioInputStream into a 16bit, SIGNED_PCM, big endian audio stream that preserves the original sample
136          * rate of the AudioInputStream.  NOTE:  this assumes the frame size can be only 1 or 2 bytes.  The AudioInputStream
137          * is left in a state of having all of its data being read.
138          */
139         static public short[] toSignedPCM(AudioInputStream ais)
140                 throws IOException {
141             AudioFormat aisFormat = ais.getFormat();
142     
143             short[] shorts = new short[ais.available() / aisFormat.getFrameSize()];
144             byte[] frame = new byte[aisFormat.getFrameSize()];
145     
146             int pos = 0;
147             while (ais.read(frame) != -1) {
148                 shorts[pos++] = bytesToShort(aisFormat, frame);
149             }
150     
151             return shorts;
152         }

155         /**
156          * Attempts to read an audio file using the Java Sound APIs.  If this file isn't a typical audio file, then this
157          * returns a null.  Otherwise, it converts the data into a 16kHz 16-bit signed PCM big endian clip.
158          *
159          * @param filename the file containing audio data
160          * @return the audio data or null if the audio cannot be parsed
161          */
162         static public AudioData readAudioFile(String filename) throws IOException {
163             try {
164                 BufferedInputStream stream = new BufferedInputStream(
165                         new FileInputStream(filename));
166                 AudioInputStream ais = AudioSystem.getAudioInputStream(stream);
167                 AudioFormat format = ais.getFormat();
168                 AudioData audioData = new AudioData(ais);
169                 stream.close();
170                 return audioData;
171             } catch (UnsupportedAudioFileException e) {
172                 return null;
173             }
174         }

177         /**
178          * Reads the given stream in as 16kHz 16-bit signed PCM big endian audio data and returns an audio clip.
179          *
180          * @param filename the file containing audio data
181          * @return the audio data or null if the audio cannot be parsed
182          */
183         static public AudioData readRawFile(String filename)
184                 throws IOException {
185             FileInputStream stream = new FileInputStream(filename);
186             AudioFormat format = new AudioFormat(16000.0f, // sample rate
187                     16,       // sample size
188                     1,        // channels (1 == mono)
189                     true,     // signed
190                     true);    // big endian
191             short[] audioData = RawReader.readAudioData(stream, format);
192             stream.close();
193             return new AudioData(audioData, 16000.0f);
194         }

197         /** Writes the given 16kHz 16-bit signed PCM audio clip to the given file as raw big endian data. */
198         static public void writeRawFile(AudioData audio, String filename)
199                 throws IOException {
200     
201             FileOutputStream outputStream = new FileOutputStream(filename);
202             AudioFormat format = new AudioFormat(
203                     16000.0f, // sample rate
204                     16,       // sample size
205                     1,        // channels (1 == mono)
206                     true,     // signed
207                     true);    // big endian
208             RawWriter writer = new RawWriter(outputStream, format);
209             short[] samples = audio.getAudioData();
210             for (int i = 0; i < samples.length; i++) {
211                 writer.writeSample(samples[i]);
212             }
213             outputStream.flush();
214             outputStream.close();
215         }
