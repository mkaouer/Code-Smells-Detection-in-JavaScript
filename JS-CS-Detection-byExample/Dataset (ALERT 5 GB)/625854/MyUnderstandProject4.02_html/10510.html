27     /**
28      * Produces Mel-cepstrum data from an InputStream. To set the inputstream with cepstral data, use the {@link
29      * #setInputStream(InputStream,boolean) setInputStream} method, and then call {@link #getData} to obtain the Data
30      * objects that have cepstra data in it.
31      */
32     public class StreamCepstrumSource extends BaseDataProcessor {
33     
34         /** The SphinxProperty specifying whether the input is in binary. */
35         @S4Boolean(defaultValue = true)
36         public final static String PROP_BINARY = "binary";
37         /** The default value for PROP_BINARY. */
38         public final static boolean PROP_BINARY_DEFAULT = true;
39         /** The sphinx property  name for frame size in milliseconds. */
40         @S4Double(defaultValue = 25.625)
41         public static final String PROP_FRAME_SIZE_MS = "frameSizeInMs";
42         /** The default value for PROP_FRAME_SIZE_MS. */
43         public static final float PROP_FRAME_SIZE_MS_DEFAULT = 25.625f;
44         /** The sphinx property  name for frame shift in milliseconds, which has a default value of 10F. */
45         @S4Double(defaultValue = 10.0)
46         public static final String PROP_FRAME_SHIFT_MS = "frameShiftInMs";
47         /** The default value for PROP_FRAME_SHIFT_MS. */
48         public static final float PROP_FRAME_SHIFT_MS_DEFAULT = 10.0f;
49         /** The sphinx property  specifying the length of the cepstrum data. */
50         @S4Integer(defaultValue = 13)
51         public static final String PROP_CEPSTRUM_LENGTH = "cepstrumLength";
52         /** The default value of PROP_CEPSTRUM_LENGTH. */
53         public static final int PROP_CEPSTRUM_LENGTH_DEFAULT = 13;
54         /** The sphinx property that defines the sample rate */
55         @S4Integer(defaultValue = 16000)
56         public static final String PROP_SAMPLE_RATE = "sampleRate";
57         /** The default value for PROP_SAMPLE_RATE */
58         public static final int PROP_SAMPLE_RATE_DEFAULT = 16000;
59     
60         private boolean binary;
61         private ExtendedStreamTokenizer est; // for ASCII files
62         private DataInputStream binaryStream; // for binary files
63         private int numPoints;
64         private int curPoint;
65         private int cepstrumLength;
66         private int frameShift;
67         private int frameSize;
68         private int sampleRate;
69         private long firstSampleNumber;
70         private boolean bigEndian = true;
71     
72     
73         /*
74         * (non-Javadoc)
75         *
76         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
77         */
78         public void newProperties(PropertySheet ps) throws PropertyException {
79             super.newProperties(ps);
80             cepstrumLength = ps.getInt(PROP_CEPSTRUM_LENGTH
81             );
82             binary = ps.getBoolean(PROP_BINARY);
83             float frameShiftMs = ps.getFloat(PROP_FRAME_SHIFT_MS
84             );
85             float frameSizeMs = ps.getFloat(PROP_FRAME_SIZE_MS
86             );
87             sampleRate = ps.getInt(PROP_SAMPLE_RATE);
88             frameShift = DataUtil.getSamplesPerWindow(sampleRate, frameShiftMs);
89             frameSize = DataUtil.getSamplesPerShift(sampleRate, frameSizeMs);
90         }

93         /** Constructs a StreamCepstrumSource that reads MelCepstrum data from the given path. */
94         public void initialize() {
95             super.initialize();
96             curPoint = -1;
97             firstSampleNumber = 0;
98             bigEndian = true;
99         }

102         /**
103          * Sets the InputStream to read cepstral data from.
104          *
105          * @param is        the InputStream to read cepstral data from
106          * @param bigEndian true if the InputStream data is in big-endian, false otherwise
107          * @throws IOException if an I/O error occurs
108          */
109         public void setInputStream(InputStream is, boolean bigEndian)
110                 throws IOException {
111             this.bigEndian = bigEndian;
112             if (binary) {
113                 binaryStream = new DataInputStream(new BufferedInputStream(is));
114                 if (bigEndian) {
115                     numPoints = binaryStream.readInt();
116                     System.out.println("BigEndian");
117                 } else {
118                     numPoints = Utilities.readLittleEndianInt(binaryStream);
119                     System.out.println("LittleEndian");
120                 }
121                 System.out.println("Frames: " + numPoints / cepstrumLength);
122             } else {
123                 est = new ExtendedStreamTokenizer(is, false);
124                 numPoints = est.getInt("num_frames");
125                 est.expectString("frames");
126             }
127             curPoint = -1;
128             firstSampleNumber = 0;
129         }
130     
131     
132         /**
133          * Returns the next Data object, which is the mel cepstrum of the input frame. However, it can also be other Data
134          * objects like DataStartSignal.
135          *
136          * @return the next available Data object, returns null if no Data object is available
137          * @throws DataProcessingException if a data processing error occurs
138          */
139         public Data getData() throws DataProcessingException {
140     
141             Data data;
142     
143             if (curPoint == -1) {
144                 data = new DataStartSignal(sampleRate);
145                 curPoint++;
146             } else if (curPoint == numPoints) {
147                 if (numPoints > 0) {
148                     firstSampleNumber =
149                             (firstSampleNumber - frameShift + frameSize - 1);
150                 }
151                 // send a DataEndSignal
152                 int numberFrames = curPoint / frameSize;
153                 int totalSamples = (numberFrames - 1) * frameShift + frameSize;
154                 long duration = (long)
155                         (((double) totalSamples / (double) sampleRate) * 1000.0);
156     
157                 data = new DataEndSignal(duration);
158     
159                 try {
160                     if (binary) {
161                         binaryStream.close();
162                     } else {
163                         est.close();
164                     }
165                     curPoint++;
166                 } catch (IOException ioe) {
167                     throw new DataProcessingException
168                             ("IOException closing cepstrum stream.");
169                 }
170             } else if (curPoint > numPoints) {
171                 data = null;
172             } else {
173                 double[] vectorData = new double[cepstrumLength];
174                 long collectTime = System.currentTimeMillis();
175     
176                 for (int i = 0; i < cepstrumLength; i++) {
177                     try {
178                         if (binary) {
179                             if (bigEndian) {
180                                 vectorData[i] = (double) binaryStream.readFloat();
181                             } else {
182                                 vectorData[i] = (double)
183                                         Utilities.readLittleEndianFloat(binaryStream);
184                             }
185                         } else {
186                             vectorData[i] = (double) est.getFloat("cepstrum data");
187                         }
188                         curPoint++;
189                     } catch (IOException ioe) {
190                         throw new DataProcessingException
191                                 ("IOException reading from cepstrum stream.");
192                     }
193                 }
194     
195                 // System.out.println("Read: " + curPoint);
196                 data = new DoubleData
197                         (vectorData, sampleRate, collectTime, firstSampleNumber);
198                 firstSampleNumber += frameShift;
199                 // System.out.println(data);
200             }
201             return data;
202         }
