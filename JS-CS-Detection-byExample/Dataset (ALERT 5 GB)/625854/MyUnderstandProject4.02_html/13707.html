10      * A <code>DataProcessor</code> which wraps incoming <code>DoubleData</code>-objects into equally size blocks of defined
11      * length.
12      */
13     public class DataBlocker extends BaseDataProcessor {
14     
15         /** The SphinxProperty name for the block size of generated data-blocks in milliseconds. */
16         @S4Double(defaultValue = 10)
17         public static final String PROP_BLOCK_SIZE_MS = "blockSizeMs";

20         public static final float PROP_BLOCK_SIZE_MS_DEFAULT = 10;

23         private int blockSizeSamples = Integer.MAX_VALUE;
24

26         private int sampleRate = -1;
27

29

42         public Data getData() throws DataProcessingException {
43             while (curInBufferSize < blockSizeSamples) {
44                 Data data = getPredecessor().getData();
45     
46                 if (data instanceof DataStartSignal) {
47                     sampleRate = ((DataStartSignal) data).getSampleRate();
48                     blockSizeSamples = (int) Math.round(sampleRate * blockSizeMs / 1000);
49     
50                     curInBufferSize = 0;
51                     inBuffer.clear();
52                 }
53     
54                 if (!(data instanceof DoubleData)) {
55                     return data;
56                 }
57     
58                 // collect enough data to be able to create a new data block
59                 if (!(data instanceof DoubleData))
60                     continue;
61     
62                 DoubleData dd = (DoubleData) data;
63     
64                 inBuffer.add(dd);
65                 curInBufferSize += dd.getValues().length;
66             }
67     
68             // now we are ready to merge all data blocks into one
69             double[] newSampleBlock = new double[blockSizeSamples];
70     
71             int copiedSamples = 0;
72     
73             long firstSample = inBuffer.get(0).getFirstSampleNumber() + curFirstSamplePos;
74             long collectTime = inBuffer.get(0).getCollectTime();
75     
76             while (!inBuffer.isEmpty()) {
77                 DoubleData dd = inBuffer.remove(0);
78                 double[] values = dd.getValues();
79                 int copyLength = Math.min(blockSizeSamples - copiedSamples, values.length - curFirstSamplePos);
80     
81                 System.arraycopy(values, curFirstSamplePos, newSampleBlock, copiedSamples, copyLength);
82     
83                 // does the current data-object contains more samples than necessary? -> keep the rest for the next block
84                 if (copyLength < (values.length - curFirstSamplePos)) {
85                     assert inBuffer.isEmpty();
86     
87                     curFirstSamplePos += copyLength;
88                     inBuffer.add(0, dd);
89                     break;
90                 } else {
91                     copiedSamples += copyLength;
92                     curFirstSamplePos = 0;
93                 }
94             }
95     
96             curInBufferSize = inBuffer.isEmpty() ? 0 : inBuffer.get(0).getValues().length - curFirstSamplePos;
97     
98     //        for (int i = 0; i < newSampleBlock.length; i++) {
99     //            newSampleBlock[i] *= 10;
100     //        }
101             return new DoubleData(newSampleBlock, sampleRate, collectTime, firstSample);
102         }
