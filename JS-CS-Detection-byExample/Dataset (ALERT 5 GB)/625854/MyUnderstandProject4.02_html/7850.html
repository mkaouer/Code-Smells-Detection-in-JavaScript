12     /**
13      * A VU meter to be plugged into a front-end. Preferably this component should be plugged directly behind the
14      * <code>DataBlocker</code> in order to ensure that only equally sized blockes of meaningful length are used for RMS
15      * computation.
16      * <p/>
17      * Because vu-monitoring makes sense only for online speech processing the vu-meter will be visible only if data source
18      * which preceeds it is a <code>Microphone</code>.
19      *
20      * @author Holger Brandl
21      */
22     
23     public class VUMeterMonitor extends BaseDataProcessor {
24     
25         VUMeter vumeter;
26         VUMeterPanel vuMeterPanel;
27         JDialog vuMeterDialog;
28     
29     
30         public VUMeterMonitor() {
31             vumeter = new VUMeter();
32     
33             vuMeterPanel = new VUMeterPanel();
34             vuMeterPanel.setVu(vumeter);
35             vuMeterPanel.start();
36     
37             vuMeterDialog = new JDialog();
38             vuMeterDialog.setBounds(100, 100, 100, 400);
39     
40             vuMeterDialog.getContentPane().setLayout(new BorderLayout());
41             vuMeterDialog.getContentPane().add(vuMeterPanel);
42     
43             vuMeterDialog.setVisible(true);
44         }
45     
46     
47         public Data getData() throws DataProcessingException {
48             Data d = getPredecessor().getData();
49     
50             // show the panel only if  a microphone is used as data source
51             if (d instanceof DataStartSignal)
52                 vuMeterPanel.setVisible(FrontEndUtils.getFrontEndProcessor(this, Microphone.class) != null);
53     
54             if (d instanceof DoubleData)
55                 vumeter.calculateVULevels(d);
56     
57             return d;
58         }
59     
60     
61         public JDialog getVuMeterDialog() {
62             return vuMeterDialog;
63         }
64

66         /** A little test-function which plugs a microphone directly into the vu-meter. */
67         public static void main(String[] args) throws DataProcessingException {
68             Microphone mic = new Microphone();
69     
70             PropertySheet propSheet = new PropertySheet(mic, null, new RawPropertyData("tt", mic.getClass().getName()), new ConfigurationManager());
71             try {
72                 propSheet.setString(Microphone.PROP_STEREO_TO_MONO, "selectChannel");
73                 propSheet.setInt(Microphone.PROP_SELECT_CHANNEL, 2);
74                 propSheet.setBoolean(Microphone.PROP_BIG_ENDIAN, true);
75     //            propSheet.setLogger(getLogger());
76     
77                 mic.newProperties(propSheet);
78                 mic.initialize();
79             } catch (PropertyException e) {
80                 e.printStackTrace();
81             }
82     
83             mic.startRecording();
84     
85             VUMeterMonitor monitor = new VUMeterMonitor();
86             monitor.getVuMeterDialog().setModal(true);
87             monitor.setPredecessor(mic);
88     
89             while (true) {
90                 monitor.getData();
91             }
92         }
