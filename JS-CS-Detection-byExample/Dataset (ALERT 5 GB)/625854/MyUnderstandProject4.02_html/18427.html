12     package edu.cmu.sphinx.util;
13     
14     
15     import java.io.BufferedReader;
16     import java.io.IOException;
17     import java.io.InputStreamReader;
18     import java.io.PrintWriter;
19     import java.net.ServerSocket;
20     import java.net.Socket;
21     import java.util.Date;
22     import java.util.HashMap;
23     import java.util.Map;
24     
25     /**
26      * This class is a command interpreter. It reads strings from an input stream, parses them into commands and executes
27      * them, results are sent back on the output stream.
28      *
29      * @see CommandInterpreter
30      */
31     
32     public class SocketCommandInterpreter extends Thread {
33     
34         private int port;
35         private Map<String, CommandInterface> commandList;
36         private boolean trace = false;
37         private boolean acceptConnections = true;
38     
39     
40         /**
41          * Creates a command interpreter at the given port.
42          *
43          * @param port the port to create the command interpreter
44          */
45         public SocketCommandInterpreter(int port) {
46             this.port = port;
47             commandList = new HashMap<String, CommandInterface>();
48         }
49     
50     
51         /**
52          * Adds the given command to the command list.
53          *
54          * @param name    the name of the command.
55          * @param command the command to be executed.
56          */
57     
58         public void add(String name, CommandInterface command) {
59             commandList.put(name, command);
60         }
61     
62         // Implements the run() method of Runnable
63     
64     
65         public final void run() {
66             ServerSocket ss;
67     
68             try {
69                 ss = new ServerSocket(port);
70                 System.out.println("Waiting on " + ss);
71             } catch (IOException ioe) {
72                 System.out.println("Can't open socket on port " + port);
73                 ioe.printStackTrace();
74                 return;
75             }
76     
77             while (acceptConnections) {
78                 try {
79                     Socket s = ss.accept();
80                     spawnCommandInterpreter(s);
81                 } catch (IOException ioe) {
82                     System.err.println("Could not accept socket " + ioe);
83                     ioe.printStackTrace();
84                     break;
85                 }
86             }
87     
88             try {
89                 ss.close();
90             } catch (IOException ioe) {
91                 System.err.println("Could not close server socket " + ioe);
92                 ioe.printStackTrace();
93             }
94         }
95     
96     
97         /** Stops this SocketCommandInterpreter from accepting connections. Effectively stops this thread. */
98         public void setStopAcceptConnections() {
99             acceptConnections = false;
100         }
101     
102     
103         /**
104          * Sets the trace mode of the command interpreter.
105          *
106          * @param trace true if tracing.
107          */
108     
109         public void setTrace(boolean trace) {
110             this.trace = trace;
111         }
112     
113     
114         /**
115          * spawns a command interpreter
116          *
117          * @param    s the socket where the interpeter reads/writes to/from
118          */
119         private void spawnCommandInterpreter(Socket s) {
120             try {
121                 BufferedReader inReader = new BufferedReader(
122                         new InputStreamReader(s.getInputStream()));
123                 PrintWriter outWriter = new PrintWriter(
124                         s.getOutputStream(), true);
125                 CommandInterpreter ci = new CommandInterpreter(inReader, outWriter);
126                 ci.setSocket(s);
127                 ci.add(commandList);
128                 ci.setTrace(trace);
129                 ci.start();
130             } catch (IOException ioe) {
131                 System.err.println("Could not attach CI to socket " + ioe);
132             }
133         }
134     
135     
136         /** manual tester for the command interpreter. */
137     
138         public static void main(String[] args) {
139             SocketCommandInterpreter sci = new SocketCommandInterpreter(7890);
140             sci.add("testCommand", new CommandInterface() {
141                 public String execute(CommandInterpreter ci, String[] args) {
142                     return "this is a test";
143                 }
144     
145     
146                 public String getHelp() {
147                     return "a test command";
148                 }
149             });
150             sci.add("time", new CommandInterface() {
151                 public String execute(CommandInterpreter ci, String[] args) {
152                     return "Time is " + new Date();
153                 }
154     
155     
156                 public String getHelp() {
157                     return "shows the current time";
158                 }
159             });
160     
161             System.out.println("Welcome to SocketCommand interpreter test program");
162             sci.setTrace(true);
163             sci.start();
164         }
165     }
