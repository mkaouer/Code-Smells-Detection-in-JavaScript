13     package edu.cmu.sphinx.linguist.flat;
14     
15     import edu.cmu.sphinx.linguist.dictionary.Word;
16     
17     /** Represents a word in an SentenceHMMS */
18     public class WordState extends SentenceHMMState {
19     
20         /** Creates a WordState */
21         public WordState(AlternativeState parent, int which) {
22             super("W", parent, which);
23         }
24     
25     
26         /**
27          * Gets the word associated with this state
28          *
29          * @return the word
30          */
31         public Word getWord() {
32             return ((AlternativeState) getParent()).getAlternative()[getWhich()];
33         }
34     
35     
36         /**
37          * Returns a pretty name for this state
38          *
39          * @return a pretty name for this state
40          */
41         public String getPrettyName() {
42             return getName() + "(" + getWord().getSpelling() + ")";
43         }
44     
45     
46         /**
47          * Retrieves a short label describing the type of this state. Typically, subclasses of SentenceHMMState will
48          * implement this method and return a short (5 chars or less) label
49          *
50          * @return the short label.
51          */
52         public String getTypeLabel() {
53             return "Word";
54         }
55     
56     
57         /**
58          * Returns the state order for this state type
59          *
60          * @return the state order
61          */
62         public int getOrder() {
63             return 1;
64         }
65     
66     }
