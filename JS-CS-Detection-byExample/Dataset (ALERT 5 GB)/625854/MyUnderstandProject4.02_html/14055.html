12     package edu.cmu.sphinx.linguist.language.ngram;
13     
14     import edu.cmu.sphinx.linguist.WordSequence;
15     import edu.cmu.sphinx.linguist.dictionary.Dictionary;
16     import edu.cmu.sphinx.util.props.*;
17     
18     import java.io.IOException;
19     import java.util.Set;
20     import java.util.logging.Logger;
21     
22     /**
23      * Represents the generic interface to an N-Gram language model.
24      * <p/>
25      * Note that all probabilities are in LogMath log base, except as otherwise noted.
26      */
27     public interface LanguageModel extends Configurable {
28     
29         /** The SphinxProperty specifying the format of the language model. */
30         @S4String(defaultValue = "arpa")
31         public final static String PROP_FORMAT = "format";
32     
33         /** The default value of PROP_FORMAT. */
34         public final static String PROP_FORMAT_DEFAULT = "arpa";
35     
36         /** The Sphinx Property specifying the location of the language model. */
37         @S4String(defaultValue = ".")
38         public final static String PROP_LOCATION = "location";
39         /** The default value of PROP_LOCATION. */
40         public final static String PROP_LOCATION_DEFAULT = ".";
41     
42         /** The Sphinx Property specifying the unigram weight */
43         @S4Double(defaultValue = 1.0)
44         public final static String PROP_UNIGRAM_WEIGHT = "unigramWeight";
45         /** The default value for PROP_UNIGRAM_WEIGHT */
46         public final static float PROP_UNIGRAM_WEIGHT_DEFAULT = 1.0f;
47         /**
48          * The Sphinx Property specifying the maximum depth reported by the language model (from a getMaxDepth()) call. If
49          * this property is set to (-1) (the default) the language model reports the implicit depth of the model. This
50          * property allows a deeper language model to be used. For instance, a trigram language model could be used as a
51          * bigram model by setting this property to 2. Note if this property is set to a value greater than the implicit
52          * depth, the implicit depth is used. Legal values for this property are 1..N and -1.
53          */
54         @S4Integer(defaultValue = -1)
55         public final static String PROP_MAX_DEPTH = "maxDepth";
56         /** The default value for PROP_MAX_DEPTH. */
57         public final static int PROP_MAX_DEPTH_DEFAULT = -1;
58     
59     
60         /** The Sphinx Property specifying the dictionary to use */
61         @S4Component(type = Dictionary.class)
62         public final static String PROP_DICTIONARY = "dictionary";
63     
64     
65         /** Create the language model */
66         public void allocate() throws IOException;
67     
68     
69         /** Deallocate resources allocated to this language model */
70         public void deallocate();
71     
72     
73         /** Called before a recognition */
74         public void start();
75     
76     
77         /** Called after a recognition */
78         public void stop();
79     
80     
81         /**
82          * Gets the ngram probability of the word sequence represented by the word list
83          *
84          * @param wordSequence the wordSequence
85          * @return the probability of the word sequence in LogMath log base
86          */
87         public float getProbability(WordSequence wordSequence);
88     
89     
90         /**
91          * Gets the smear term for the given wordSequence
92          *
93          * @param wordSequence the word sequence
94          * @return the smear term associated with this word sequence
95          */
96         public float getSmear(WordSequence wordSequence);
97     
98     
99         /**
100          * Returns the set of words in the lanaguage model. The set is unmodifiable.
101          *
102          * @return the unmodifiable set of words
103          */
104         public Set getVocabulary();
105     
106     
107         /**
108          * Returns the maximum depth of the language model
109          *
110          * @return the maximum depth of the language mdoel
111          */
112         public int getMaxDepth();
113     
114     
115         /**
116          * Used for reporting errors and warnings during loading
117          *
118          * @return the logger used by the LanguageModel
119          */
120         public Logger getLogger();
121     
122     }
