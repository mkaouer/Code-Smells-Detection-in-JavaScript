8     package com.sun.speech.engine;
9     
10     import javax.swing.*;
11     import java.awt.*;
12     import java.awt.event.ActionEvent;
13     import java.awt.event.ActionListener;
14     
15     /** Simple GUI for monitoring events of an <code>Engine</code>.  Used for debugging and testing purposes. */
16     public class EngineEventPanel extends JPanel {
17     
18         /** The area where engine events are posted. */
19         protected JTextArea textArea;
20     
21         /**
22          * The scroll pane containing the <code>textArea</code>.
23          *
24          * @see #textArea
25          */
26         protected JScrollPane scroller;
27     
28         /**
29          * The button for clearing the <code>textArea</code>.
30          *
31          * @see #textArea
32          */
33         protected JButton clearButton;
34     
35     
36         /** Class constructor. */
37         public EngineEventPanel() {
38             setLayout(new BorderLayout());
39             setBorder(BorderFactory.createTitledBorder("Events:"));
40     
41             clearButton = new JButton("Clear");
42             clearButton.setMnemonic('C');
43             clearButton.addActionListener(new ActionListener() {
44                 public void actionPerformed(ActionEvent evt) {
45                     clearText();
46                 }
47             });
48     
49             textArea = new JTextArea();
50             scroller = new JScrollPane(textArea);
51     
52             add(scroller, BorderLayout.CENTER);
53             add(clearButton, BorderLayout.SOUTH);
54         }
55     
56     
57         /** Clears the text in the text area. */
58         public void clearText() {
59             textArea.setText("");
60         }
61     
62     
63         /**
64          * Sets the text in the text area.
65          *
66          * @param s the new text
67          */
68         public void setText(String s) {
69             textArea.setText(s);
70         }
71     
72     
73         /**
74          * Appends text to the text area and scrolls the text area so the new text is visible.
75          *
76          * @param s the text to append
77          */
78         public void addText(String s) {
79             textArea.append(s);
80             Point pt = new Point(0, textArea.getHeight() - 1);
81             scroller.getViewport().setViewPosition(pt);
82         }
83     }
