13     package edu.cmu.sphinx.linguist.acoustic.tiedstate;
14     
15     import edu.cmu.sphinx.linguist.acoustic.HMM;
16     import edu.cmu.sphinx.linguist.acoustic.HMMPosition;
17     import edu.cmu.sphinx.linguist.acoustic.HMMState;
18     import edu.cmu.sphinx.linguist.acoustic.Unit;
19     import edu.cmu.sphinx.util.Utilities;

21     /**
22      * Represents a hidden-markov-model. An HMM consists of a unit (context dependent or independent), a transition matrix
23      * from state to state, and a sequence of senones associated with each state. This representation of an HMM is a
24      * specialized left-to-right markov model. No backward transitions are allowed.
25      */
26     public class SenoneHMM implements HMM {
27     
28         private Unit unit;
29         private Unit baseUnit;
30         private SenoneSequence senoneSequence;
31         private float[][] transitionMatrix;
32         private HMMPosition position;
33         private static int objectCount;
34         private HMMState[] hmmStates;
35     
36     
37         /**
38          * Constructs an HMM
39          *
40          * @param unit             the unit for this HMM
41          * @param senoneSequence   the sequence of senones for this HMM
42          * @param transitionMatrix the state transition matrix
43          * @param position         the position associated with this HMM
44          */
45         public SenoneHMM(Unit unit, SenoneSequence senoneSequence,
46                          float[][] transitionMatrix, HMMPosition position) {
47             this.unit = unit;
48             this.senoneSequence = senoneSequence;
49             this.transitionMatrix = transitionMatrix;
50             this.position = position;
51             Utilities.objectTracker("HMM", objectCount++);
52     
53             hmmStates = new HMMState[transitionMatrix.length];
54             for (int i = 0; i < hmmStates.length; i++) {
55                 hmmStates[i] = new SenoneHMMState(this, i);
56             }
57             // baseUnit = Unit.getUnit(unit.getName());
58             baseUnit = unit.getBaseUnit();
59         }
60     
61     
62         /**
63          * Gets the  unit associated with this HMM
64          *
65          * @return the unit associated with this HMM
66          */
67         public Unit getUnit() {
68             return unit;
69         }
70     
71     
72         /**
73          * Gets the  base unit associated with this HMM
74          *
75          * @return the unit associated with this HMM
76          */
77         public Unit getBaseUnit() {
78             return baseUnit;
79         }
80     
81     
82         /**
83          * Retrieves the hmm state
84          *
85          * @param which the state of interest
86          */
87         public HMMState getState(int which) {
88             return hmmStates[which];
89         }
90     
91     
92         /**
93          * Returns the order of the HMM
94          *
95          * @return the order of the HMM
96          */
97         // [[[NOTE: this method is probably not explicitly needed since
98         // getSenoneSequence.getSenones().length will provide the same
99         // value, but this is certainly more convenient and easier to
100         // understand
101         public int getOrder() {
102             return getSenoneSequence().getSenones().length;
103         }
104     
105     
106         /**
107          * Returns the SenoneSequence associated with this HMM
108          *
109          * @return the sequence of senones associated with this HMM. The length of the sequence is N, where N is the order
110          *         of the HMM. Note that senone sequences may be shared among HMMs.
111          */
112         // [[ NOTE: the senone sequence may in fact be a sequence of
113         // composite senones
114         public SenoneSequence getSenoneSequence() {
115             return senoneSequence;
116         }
117     
118     
119         /**
120          * Determines if this HMM is a composite HMM
121          *
122          * @return true if this is a composite hmm
123          */
124         public boolean isComposite() {
125             Senone[] senones = getSenoneSequence().getSenones();
126             for (int i = 0; i < senones.length; i++) {
127                 if (senones[i] instanceof CompositeSenone) {
128                     return true;
129                 }
130             }
131             return false;
132         }
133     
134     
135         /**
136          * Returns the transition matrix that determines the state transition probabilities for the matrix. Each entry in
137          * the transition matrix defines the probability of transitioning from one state to the next. For example, the
138          * probablility of transitioning from state 1 to state 2 can be determined byh accessesing transition matrix
139          * element[1][2].
140          *
141          * @return the transition matrix (in log domain) of size NxN where N is the order of the HMM
142          */
143         public float[][] getTransitionMatrix() {
144             return transitionMatrix;
145         }
146     
147     
148         /**
149          * Returns the transition probability between two states.
150          *
151          * @param stateFrom the index of the state this transition goes from
152          * @param stateTo   the index of the state this transition goes to
153          * @return the transition probability (in log domain)
154          */
155         public float getTransitionProbability(int stateFrom, int stateTo) {
156             return transitionMatrix[stateFrom][stateTo];
157         }
158     
159     
160         /**
161          * Retrieves the position of this HMM. Possible
162          *
163          * @return the position for this HMM
164          */
165         public HMMPosition getPosition() {
166             return position;
167         }
168     
169     
170         /**
171          * Determines if this HMM represents a filler unit. A filler unit is speech that is not meaningful such as a cough,
172          * 'um' , 'er', or silence.
173          *
174          * @return true if the HMM  represents a filler unit
175          */
176         public boolean isFiller() {
177             return unit.isFiller();
178         }
179     
180     
181         /**
182          * Determines if this HMM corresponds to a context dependent unit
183          *
184          * @return true if the HMM is context dependent
185          */
186         public boolean isContextDependent() {
187             return unit.isContextDependent();
188         }
189     
190     
191         /**
192          * Gets the initial states (with probabilities) for this HMM
193          *
194          * @return the set of arcs that transition to the initial states for this HMM
195          */
196         public HMMState getInitialState() {
197             return getState(0);
198         }
199     
200     
201         /**
202          * Returns the string representation of this object
203          *
204          * @return the string representation
205          */
206         public String toString() {
207             String name = isComposite() ? "HMM@" : "HMM";
208             return name + "(" + unit + "):" + position;
209         }
210     
211     
212         public int hashCode() {
213             return getTransitionMatrix().hashCode() * 3 +
214                     getSenoneSequence().hashCode();
215         }
216     
217     
218         public boolean equals(Object o) {
219             if (this == o) {
220                 return true;
221             } else if (o instanceof SenoneHMM) {
222                 SenoneHMM other = (SenoneHMM) o;
223                 return getTransitionMatrix() ==
224                         other.getTransitionMatrix()
225                         && getSenoneSequence().equals(
226                         other.getSenoneSequence());
227             }
228             return false;
229         }
230     }
