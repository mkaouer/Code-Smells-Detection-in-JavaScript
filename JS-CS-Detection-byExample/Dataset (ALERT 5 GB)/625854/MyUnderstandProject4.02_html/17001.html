201         /*
202          * This function is used to check the name of the new set
203          * it should only contains alphanumeric '_' or '-'
204          */
205         private boolean checkSetName(String s){
206     
207             final char[] chars = s.toCharArray();
208             for (int x = 0; x < chars.length; x++) {
209                 final char c = chars[x];
210                 if ((c >= 'a') && (c <= 'z')) continue; // lowercase
211                 if ((c >= 'A') && (c <= 'Z')) continue; // uppercase
212                 // numeric allowed for 2nd char onwards
213                 if ((x >= 1) && (c >= '0') && (c <= '9')) continue;
214                 if ((x >= 1) && (c == '_')) continue; // underscore after 2nd char 
215                 if ((x >= 1) && (c == '-')) continue; // dash for 2nd char onwards
216                 return false;
217             }
218             return true;
219     
220         }

580         private void jButtonRemoveActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonRemoveActionPerformed
581             String  classname = (String)jListOuter.getSelectedValue();
582             String prop = (String)jListInner.getSelectedValue();
583             String setname = (String)jComboName.getSelectedItem();
584             String delval = (String)jListPropVal.getSelectedValue();
585     
586             if ( delval != null && (classname != null) &&
587                     ( setname != null) && (prop != null) )
588             {
589                 List newlist = new ArrayList(
590                         Arrays.asList(((DefaultListModel)jListPropVal.getModel()).toArray()));
591                 if( newlist.remove(delval) ){
592                     try {
593                         if(newlist.isEmpty()){ // list is now empty
594                             removePropFromSet(); // delete this property from the set
595                         }else{ // list is not empty
596                             if(_pm.allowChangePropertyValue(classname,setname,prop,newlist)){
597                                 _pm.changePropertyValue(classname,setname,prop,newlist);
598                                 ((DefaultListModel)jListPropVal.getModel()).removeElement(delval);
599                             }
600                         }
601                     }catch(PanelMediatorException e){
602                         displayError("Internal Error : "+e.getMessage());
603                     }
604                 }
605             }
606         }//GEN-LAST:event_jButtonRemoveActionPerformed

731                             mytype.equals(PropertyType.COMPONENT_LIST) )

739                         jTextClassType.setText("");
740                     }
741

743                     jTextDesc.setText(cp.getDesc());

745                         initPropValue(_pm.getPropertyValue(classname,
746                             (String)jComboName.getSelectedItem(),prop));

754                     // component? list? other type? or not specified?
