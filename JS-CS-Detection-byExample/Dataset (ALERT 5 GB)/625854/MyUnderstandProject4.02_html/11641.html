64         /**
65          * Copy a result. If the result to be copied is a BaseResult then clone it otherwise create a BaseResult and copy
66          * the tokens onto it.
67          */
68         static BaseResult copyResult(Result R) {
69             BaseResult copy = null;
70             if (R instanceof BaseResult) {
71                 try {
72                     copy = (BaseResult) ((BaseResult) R).clone();
73                 } catch (CloneNotSupportedException e) {
74                     System.out.println("ERROR: " + e);
75                 }
76                 return copy;
77             } else {
78                 copy = new BaseResult(R.getGrammar());
79                 copy.nTokens = R.numTokens();
80                 copy.theText = new String[copy.nTokens];
81                 for (int i = 0; i < R.numTokens(); i++) {
82                     copy.theText[i] = R.getBestToken(i).getSpokenText();
83                 }
84                 return copy;
85             }
86         }

501         /** Concatenate the best tokens in the Result. */
502         public String toString() {
503             StringBuffer sb = new StringBuffer(getBestToken(0).getWrittenText());
504             for (int i = 1; i < numTokens(); i++)
505                 sb.append(" " + getBestToken(i).getWrittenText());
506             return sb.toString();
507         }

576         /**
577          * Dispatch a SpeechEvent.  This is a method from SpeechEventDispatcher. The dispatcher should notify all listeners
578          * of the speech event from this method.
579          */
580         public void dispatchSpeechEvent(SpeechEvent event) {
581             switch (event.getId()) {
582                 case ResultEvent.AUDIO_RELEASED:
583                     fireAudioReleased((ResultEvent) event);
584                     break;
585                 case ResultEvent.GRAMMAR_FINALIZED:
586                     fireGrammarFinalized((ResultEvent) event);
587                     break;
588                 case ResultEvent.RESULT_ACCEPTED:
589                     fireResultAccepted((ResultEvent) event);
590                     break;
591                 case ResultEvent.RESULT_CREATED:
592                     fireResultCreated((ResultEvent) event);
593                     break;
594                 case ResultEvent.RESULT_REJECTED:
595                     fireResultRejected((ResultEvent) event);
596                     break;
597                 case ResultEvent.RESULT_UPDATED:
598                     fireResultUpdated((ResultEvent) event);
599                     break;
600                 case ResultEvent.TRAINING_INFO_RELEASED:
601                     fireTrainingInfoReleased((ResultEvent) event);
602                     break;
603             }
604         }

613         /** Try to set the Grammar and tokens of this result.  NOT JSAPI. */
614         public boolean tryTokens(Grammar G, String S) {
615             if ((S == null) || (G == null)) {
616                 return false;
617             }
618     
619             if (G instanceof RuleGrammar) {
620                 try {
621                     RuleParse rp = ((RuleGrammar) G).parse(S, null);
622                     if (rp != null) {
623                         grammar = G;
624                         tags = rp.getTags();
625                         ruleName = rp.getRuleName().getSimpleRuleName();
626                         StringTokenizer st = new StringTokenizer(S);
627                         nTokens = st.countTokens();
628                         int i = 0;
629                         theText = new String[nTokens];
630                         while (st.hasMoreTokens()) {
631                             theText[i++] = st.nextToken();
632                         }
633                         return true;
634                     }
635                 } catch (GrammarException e) {
636                 }
637             }
638             return false;
639         }
