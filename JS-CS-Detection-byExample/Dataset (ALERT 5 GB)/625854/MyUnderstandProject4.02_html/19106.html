229         /**
230          * Resolve a simple or qualified rulename as a full rulename. From javax.speech.recognition.RuleGrammar.
231          *
232          * @param ruleName the name of the rule.
233          */
234         public RuleName resolve(RuleName name) throws GrammarException {
235             RuleName rn = new RuleName(name.getRuleName());
236     
237             String simpleName = rn.getSimpleRuleName();
238             String grammarName = rn.getSimpleGrammarName();
239             String packageName = rn.getPackageName();
240             String fullGrammarName = rn.getFullGrammarName();
241     
242             // Check for badly formed RuleName
243             if (packageName != null && grammarName == null) {
244                 throw new GrammarException("Error: badly formed rulename " + rn,
245                         null);
246             }
247     
248             if (name.getSimpleRuleName().equals("NULL")) {
249                 return RuleName.NULL;
250             }
251     
252             if (name.getSimpleRuleName().equals("VOID")) {
253                 return RuleName.VOID;
254             }
255     
256             // Check simple case: a local rule reference
257             if (fullGrammarName == null && this.getRule(simpleName) != null) {
258                 return new RuleName(myName + "." + simpleName);
259             }
260     
261             // Check for fully-qualified reference
262             if (fullGrammarName != null) {
263                 RuleGrammar g = myRec.getRuleGrammar(fullGrammarName);
264                 if (g != null) {
265                     if (g.getRule(simpleName) != null) {
266                         // we have a successful resolution
267                         return new RuleName(fullGrammarName + "." + simpleName);
268                     }
269                 }
270             }
271     
272             // Collect all matching imports into a vector.  After trying to
273             // match rn to each import statement the vec will have
274             // size()=0 if rn is unresolvable
275             // size()=1 if rn is properly resolvable
276             // size()>1 if rn is an ambiguous reference
277             Vector matches = new Vector();
278     
279             // Get list of imports
280             // Add local grammar to simply the case of checking for
281             // a qualified or fully-qualified local reference.
282             RuleName imports[] = listImports();
283     
284             if (imports == null) {
285                 imports = new RuleName[1];
286                 imports[0] = new RuleName(getName() + ".*");
287             } else {
288                 RuleName[] tmp = new RuleName[imports.length + 1];
289                 System.arraycopy(imports, 0, tmp, 0, imports.length);
290                 tmp[imports.length] = new RuleName(getName() + ".*");
291                 imports = tmp;
292             }
293     
294             // Check each import statement for a possible match
295             //
296             for (int i = 0; i < imports.length; i++) {
297                 // TO-DO: update for JSAPI 1.0
298                 String iSimpleName = imports[i].getSimpleRuleName();
299                 String iGrammarName = imports[i].getSimpleGrammarName();
300                 String iPackageName = imports[i].getPackageName();
301                 String iFullGrammarName = imports[i].getFullGrammarName();
302     
303                 // Check for badly formed import name
304                 if (iFullGrammarName == null)
305                     throw new GrammarException("Error: badly formed import " +
306                             imports[i], null);
307     
308                 // Get the imported grammar
309                 RuleGrammar gref = myRec.getRuleGrammar(iFullGrammarName);
310                 if (gref == null) {
311                     RecognizerUtilities.debugMessageOut(
312                             "Warning: import of unknown grammar " +
313                                     imports[i] + " in " + getName());
314                     continue;
315                 }
316     
317                 // If import includes simpleName, test that it really exists
318                 if (!iSimpleName.equals("*")
319                         && gref.getRule(iSimpleName) == null) {
320                     RecognizerUtilities.debugMessageOut(
321                             "Warning: import of undefined rule " +
322                                     imports[i] + " in " + getName());
323                     continue;
324                 }
325     
326                 // Check for fully-qualified or qualified reference
327                 if (iFullGrammarName.equals(fullGrammarName)
328                         || iGrammarName.equals(fullGrammarName)) {
329                     // Know that either
330                     //    import <ipkg.igram.???> matches <pkg.gram.???>
331                     // OR
332                     //    import <ipkg.igram.???> matches <gram.???>
333                     // (ipkg may be null)
334     
335                     if (iSimpleName.equals("*")) {
336                         if (gref.getRule(simpleName) != null) {
337                             // import <pkg.gram.*> matches <pkg.gram.rulename>
338                             matches.addElement(
339                                     new RuleName(iFullGrammarName + "." + simpleName));
340                         }
341                         continue;
342                     } else {
343                         // Now testing
344                         //    import <ipkg.igram.iRuleName> against <??.gram.ruleName>
345                         //
346                         if (iSimpleName.equals(simpleName)) {
347                             // import <pkg.gram.rulename> exact match for <???.gram.rulename>
348                             matches.addElement(new RuleName(iFullGrammarName +
349                                     "." + simpleName));
350                         }
351                         continue;
352                     }
353                 }
354     
355                 // If we get here and rn is qualified or fully-qualified 
356                 // then the match failed - try the next import statement
357                 if (fullGrammarName != null) {
358                     continue;
359                 }
360     
361                 // Now test
362                 //    import <ipkg.igram.*> against <simpleName>
363     
364                 if (iSimpleName.equals("*")) {
365                     if (gref.getRule(simpleName) != null) {
366                         // import <pkg.gram.*> matches <simpleName>
367                         matches.addElement(new RuleName(iFullGrammarName +
368                                 "." + simpleName));
369                     }
370                     continue;
371                 }
372     
373                 // Finally test
374                 //    import <ipkg.igram.iSimpleName> against <simpleName>
375     
376                 if (iSimpleName.equals(simpleName)) {
377                     matches.addElement(new RuleName(iFullGrammarName +
378                             "." + simpleName));
379                     continue;
380                 }
381             }
382     
383             //
384             // The return behavior depends upon number of matches
385             //
386             if (matches.size() == 0) {
387                 // Return null if rulename is unresolvable
388                 return null;
389             } else if (matches.size() > 1) {
390                 // Throw exception if ambiguous reference
391                 StringBuffer b = new StringBuffer();
392                 b.append("Warning: ambiguous reference " + rn + " in " +
393                         getName() + " to ");
394                 for (int i = 0; i < matches.size(); i++) {
395                     RuleName tmp = (RuleName) (matches.elementAt(i));
396                     if (i > 0) {
397                         b.append(" and ");
398                     }
399                     b.append(tmp);
400                 }
401                 throw new GrammarException(b.toString(), null);
402             } else {
403                 // Return successfully
404                 return (RuleName) (matches.elementAt(0));
405             }
406         }

482         /**
483          * Parse the nth best result of a FinalRuleResult against the specified rule. Uses the RuleParser class. From
484          * javax.speech.recognition.RuleGrammar.
485          *
486          * @param r        the FinalRuleResult.
487          * @param nBest    the nth best result to use.
488          * @param ruleName the name of rule to use for parsing.
489          */
490         public RuleParse parse(FinalRuleResult r, int nBest, String ruleName) throws GrammarException {
491             // Some JSAPI implementations we run into are not JSAPI compliant,
492             // so try a few alternatives
493             ResultToken rt[] = r.getAlternativeTokens(nBest);
494             if (rt != null || (rt = r.getBestTokens()) != null) {
495                 String tokens[] = new String[rt.length];
496                 for (int i = 0; i < rt.length; i++) {
497                     tokens[i] = rt[i].getSpokenText();
498                 }
499                 return parse(tokens, ruleName);
500             } else {
501                 return parse(r.toString(), ruleName);
502             }
503         }

549         protected void resolveAllRules()
550                 throws GrammarException {
551             StringBuffer b = new StringBuffer();
552     
553             // First make sure that all imports are resolvable
554             RuleName imports[] = listImports();
555     
556             if (imports != null) {
557                 for (int i = 0; i < imports.length; i++) {
558                     String gname = imports[i].getFullGrammarName();
559                     RuleGrammar GI = myRec.getRuleGrammar(gname);
560                     if (GI == null) {
561                         b.append("Undefined grammar " + gname +
562                                 " imported in " + getName() + "\n");
563                     }
564                 }
565             }
566             if (b.length() > 0) {
567                 throw new GrammarException(b.toString(), null);
568             }
569     
570             String rn[] = listRuleNames();
571             for (int i = 0; i < rn.length; i++) {
572                 resolveRule(getRuleInternal(rn[i]));
573             }
574         }

577         /** Resolve the given rule. */
578         protected void resolveRule(Rule r)
579                 throws GrammarException {
580     
581             if (r instanceof RuleToken) {
582                 return;
583             }
584     
585             if (r instanceof RuleAlternatives) {
586                 RuleAlternatives ra = (RuleAlternatives) r;
587                 Rule array[] = ra.getRules();
588                 for (int i = 0; i < array.length; i++) {
589                     resolveRule(array[i]);
590                 }
591                 return;
592             }
593     
594             if (r instanceof RuleSequence) {
595                 RuleSequence rs = (RuleSequence) r;
596                 Rule array[] = rs.getRules();
597                 for (int i = 0; i < array.length; i++) {
598                     resolveRule(array[i]);
599                 }
600                 return;
601             }
602     
603             if (r instanceof RuleCount) {
604                 RuleCount rc = (RuleCount) r;
605                 resolveRule(rc.getRule());
606                 return;
607             }
608     
609             if (r instanceof RuleTag) {
610                 RuleTag rt = (RuleTag) r;
611     
612                 Rule rule = rt.getRule();
613     
614                 // add the tag the tag-table
615                 if (!ruleTags.containsKey(rule.toString()))
616                     ruleTags.put(rule.toString(), new HashSet<String>());
617     
618                 ruleTags.get(rule.toString()).add(rt.getTag());
619     
620                 resolveRule(rule);
621                 return;
622             }
623     
624             if (r instanceof BaseRuleName) {
625                 BaseRuleName rn = (BaseRuleName) r;
626                 RuleName resolved = resolve(rn);
627     
628                 if (resolved == null) {
629                     throw new GrammarException(
630                             "Unresolvable rulename in grammar " +
631                                     getName() + ": " + rn.toString(), null);
632                 } else {
633                     //[[[WDW - This forces all rule names to be fully resolved.
634                     //This should be changed.]]]
635                     rn.resolvedRuleName = resolved.getRuleName();
636                     rn.setRuleName(resolved.getRuleName());
637                     return;
638                 }
639             }
640     
641             throw new GrammarException("Unknown rule type", null);
642         }

747     
748         /** add a sample sentence to the list of sample sentences that go with the specified rule */
749         void setSampleSentences(String rname, Vector samps) {
750             GRule r = rules.get(stripRuleName(rname));
751             /* TODO : exception */
752             if (r == null) {
753                 return;
754             }
755             r.samples = samps;
756         }
757     
758     
759         /** get the list of sample sentences that go with the specified rule */
760         public Vector getSampleSentences(String rname) {
761             GRule r = rules.get(stripRuleName(rname));
762             if (r == null) {
763                 return null;
764             }
765             return r.samples;
766         }
767     
768     
769         /** Non-JSAPI method returns a unique 32-bit id for the specified rule */
770         public int getRuleID(String rname) {
771             GRule r = rules.get(stripRuleName(rname));
772             if (r == null) {
773                 return -1;
774             }
775             return r.hashCode();
776         }

778     
779         /** Remove a pair of leading/trailing angle brackets . */
780         protected String stripRuleName(String n) {
781             if (n.startsWith("<") && n.endsWith(">")) {
782                 return n.substring(1, n.length() - 1);
783             }
784             return n;
785         }
