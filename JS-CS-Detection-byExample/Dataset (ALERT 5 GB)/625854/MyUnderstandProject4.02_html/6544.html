29         /**
30          * Creates a new AudioDataInputStream for the given AudioData.
31          *
32          * @param audio the AudioData to turn into an AudioDataInputStream
33          */
34         public AudioDataInputStream(AudioData audio) {
35             shorts = audio.getAudioData();
36             bytes = new byte[2 * shorts.length];
37     
38             byte[] sample = new byte[2];
39             for (int i = 0; i < shorts.length; i++) {
40                 Utils.toBytes(shorts[i], sample, true);
41                 bytes[i * 2] = sample[0];
42                 bytes[(i * 2) + 1] = sample[1];
43             }
44         }
