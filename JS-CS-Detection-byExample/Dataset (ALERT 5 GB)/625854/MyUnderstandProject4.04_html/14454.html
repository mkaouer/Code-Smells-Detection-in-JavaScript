29     /**
30      * A test program for JSGF grammars. This program will generate a number of
31      * random sentences from a JSGF grammar and attempt to validate them via a rule
32      * parse. A count of valid and invalid sentences is reported after the run.
33      * 
34      * This test will detect and reports only certain grammar failures. It will
35      * detect when a JSGFGrammar generates invalid sentences. It will not detect the
36      * case where a valid sentence can never be generated by the JSGFGrammar.
37      * 
38      * Note that there is a bug in the Rule Grammar parse where rules with nested
39      * recursion cause a stack overflow error.
40      * 
41      */
42     public class JSGFTest {
43     
44         @Test
45         public void testParser() throws IOException, GrammarException {
46             ConfigurationManager cm = new ConfigurationManager(
47                     "src/test/edu/cmu/sphinx/jsgf/test/jsgftest.config.xml");
48     
49             JSGFGrammar jsgfGrammar = (JSGFGrammar) cm.lookup("jsgfGrammar");
50     
51             jsgfGrammar.allocate();
52     
53             RuleGrammar grammar = jsgfGrammar.getRuleGrammar();
54             Scanner scanner = new Scanner (new File ("src/test/edu/cmu/sphinx/jsgf/test/input.txt"));
55             scanner.useDelimiter("\n");
56             while (scanner.hasNext()) {
57                 String sentence = scanner.next();
58                 RuleParse rp = grammar.parse(sentence, null);
59                 Assert.assertNotNull(rp);
60             }
61         }
62     }
