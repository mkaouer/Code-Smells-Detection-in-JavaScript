13     package edu.cmu.sphinx.instrumentation;
14     
15     import edu.cmu.sphinx.recognizer.Recognizer;
16     import edu.cmu.sphinx.recognizer.Recognizer.State;
17     import edu.cmu.sphinx.recognizer.StateListener;
18     import edu.cmu.sphinx.result.Result;
19     import edu.cmu.sphinx.decoder.ResultListener;
20     import edu.cmu.sphinx.util.props.*;
21     
22     import java.text.DecimalFormat;
23     
24     /** Monitors a recognizer for memory usage */
25     public class MemoryTracker
26             implements
27             ResultListener,
28             StateListener,
29             Monitor {
30     
31         /** The property that defines which recognizer to monitor */
32         @S4Component(type = Recognizer.class)
33         public final static String PROP_RECOGNIZER = "recognizer";
34     
35         /** The property that defines whether summary accuracy information is displayed */
36         @S4Boolean(defaultValue = true)
37         public final static String PROP_SHOW_SUMMARY = "showSummary";
38     
39         /** The property that defines whether detailed accuracy information is displayed */
40         @S4Boolean(defaultValue = true)
41         public final static String PROP_SHOW_DETAILS = "showDetails";
42     
43         private static final DecimalFormat memFormat = new DecimalFormat("0.00 Mb");
44         // ------------------------------
45         // Configuration data
46         // ------------------------------
47         private String name;
48         private Recognizer recognizer;
49         private boolean showSummary;
50         private boolean showDetails;
51         private float maxMemoryUsed;
52         private int numMemoryStats;
53         private float avgMemoryUsed;
54     
55         public MemoryTracker(Recognizer recognizer, boolean showSummary, boolean showDetails) {
56             initRecognizer(recognizer);
57             this.showSummary = showSummary;
58             this.showDetails = showDetails;
59         }
60     
61         public MemoryTracker() {
62             
63         }
64     
65         /*
66         * (non-Javadoc)
67         *
68         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
69         */
70         @Override
71         public void newProperties(PropertySheet ps) throws PropertyException {
72             Recognizer newRecognizer = (Recognizer) ps.getComponent(
73                     PROP_RECOGNIZER);
74             initRecognizer(newRecognizer);
75             showSummary = ps.getBoolean(PROP_SHOW_SUMMARY);
76             showDetails = ps.getBoolean(PROP_SHOW_DETAILS);
77         }
78     
79         private void initRecognizer(Recognizer newRecognizer) {
80             if (recognizer == null) {
81                 recognizer = newRecognizer;
82                 recognizer.addResultListener(this);
83                 recognizer.addStateListener(this);
84             } else if (recognizer != newRecognizer) {
85                 recognizer.removeResultListener(this);
86                 recognizer.removeStateListener(this);
87                 recognizer = newRecognizer;
88                 recognizer.addResultListener(this);
89                 recognizer.addStateListener(this);
90             }
91         }
92     
93     
94         /*
95         * (non-Javadoc)
96         *
97         * @see edu.cmu.sphinx.util.props.Configurable#getName()
98         */
99         public String getName() {
100             return name;
101         }
102     
103     
104         /** Shows memory usage
105          * @param show*/
106         private void calculateMemoryUsage(boolean show) {
107             float totalMem = Runtime.getRuntime().totalMemory()
108                     / (1024.0f * 1024.0f);
109             float freeMem = Runtime.getRuntime().freeMemory() / (1024.0f * 1024.0f);
110             float usedMem = totalMem - freeMem;
111             if (usedMem > maxMemoryUsed) {
112                 maxMemoryUsed = usedMem;
113             }
114     
115             numMemoryStats++;
116             avgMemoryUsed = ((avgMemoryUsed * (numMemoryStats - 1)) + usedMem)
117                     / numMemoryStats;
118     
119             if (show) {
120                 System.out.println("   Mem  Total: " + memFormat.format(totalMem)
121                         + "  " + "Free: " + memFormat.format(freeMem));
122                 System.out.println("   Used: This: " + memFormat.format(usedMem) + "  "
123                         + "Avg: " + memFormat.format(avgMemoryUsed) + "  " + "Max: "
124                         + memFormat.format(maxMemoryUsed));
125             }
126         }
127     
128     
129         /*
130         * (non-Javadoc)
131         *
132         * @see edu.cmu.sphinx.decoder.ResultListener#newResult(edu.cmu.sphinx.result.Result)
133         */
134         @Override
135         public void newResult(Result result) {
136             if (result.isFinal()) {
137                 calculateMemoryUsage(showDetails);
138             }
139         }
140     
141         @Override
142         public void statusChanged(Recognizer.State status) {
143             if (status == State.DEALLOCATED) {
144                 calculateMemoryUsage(showSummary);
145             }
146         }
147     }
