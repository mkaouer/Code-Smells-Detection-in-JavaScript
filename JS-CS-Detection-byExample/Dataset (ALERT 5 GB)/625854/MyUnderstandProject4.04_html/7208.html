21     /**
22      * Given a sequence of Data, filters out the non-speech regions. The sequence of Data should have the speech and
23      * non-speech regions marked out by the SpeechStartSignal and SpeechEndSignal, using the {@link SpeechMarker
24      * SpeechMarker}.
25      * <p/>
26      * Only one speech region</b> <p>In the first case, the data stream has only one speech region: <p><img
27      * src="doc-files/one-region.gif"> <br><i>Figure 1: A data stream with only one speech region</i>. <p>After filtering,
28      * the non-speech regions are removed, and becomes: <p><img src="doc-files/one-region-filtered.gif"> <br><i>Figure 2: A
29      * data stream with only on speech region after filtering.</i>
30      */
31     public class NonSpeechDataFilter extends BaseDataProcessor {
32     
33         private boolean inSpeech;
34     
35         public NonSpeechDataFilter() {
36             initLogger();
37         }
38     
39         @Override
40         public void newProperties(PropertySheet ps) throws PropertyException {
41             super.newProperties(ps);
42         }
43     
44     
45         /** Initializes this data processor */
46         @Override
47         public void initialize() {
48             super.initialize();
49     
50             this.inSpeech = false;
51         }
52     
53     
54         @Override
55         public Data getData() throws DataProcessingException {
56             Data data = readData();
57     
58             while (data != null
59                     && !(data instanceof DataEndSignal) && !(data instanceof DataStartSignal)
60                     && !(data instanceof SpeechEndSignal) && !inSpeech) {
61                 data = readData();
62             }
63     
64             return data;
65         }
66     
67     
68         private Data readData() {
69             Data data = getPredecessor().getData();
70     
71             if (data instanceof SpeechStartSignal)
72                 inSpeech = true;
73     
74             if (data instanceof SpeechEndSignal)
75                 inSpeech = false;
76     
77             return data;
78         }
79     }
