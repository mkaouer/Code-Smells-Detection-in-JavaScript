13     package edu.cmu.sphinx.research.parallel;
14     
15     /**
16      * Combines the various unit feature stream scores in a CombineToken. Each feature stream is represented as a {@link
17      * ParallelToken ParallelToken} in a CombineToken. Suppose that the CombineToken has two ParallelTokens, P1 and P2, then
18      * the combined score is given by:
19      * <pre>
20      * P1.getFeatureScore() * P1.getEta() + P2.getFeatureScore() * P2.getEta()
21      * </pre>
22      * where the feature scores are in LogMath logbase, and the method <code>getEta()</code> returns the eta value of that
23      * particular feature stream.
24      */
25     public class FeatureScoreCombiner implements ScoreCombiner {
26     
27         /**
28          * Combines the scores from all the feature stream in the given CombineToken. The feature stream(s) are represented
29          * in the form of ParallelTokens in the given CombineToken.
30          *
31          * @param token the CombineToken on which to combine the feature stream scores
32          */
33         public void combineScore(CombineToken token) {
34             double logTotalScore = 0.0f;
35     
36             for (ParallelToken pToken : token) {
37                 // in linear domain, the following expression is:
38                 // score = pToken.getFeatureScore()^pToken.getEta()
39     
40                 double logScore = pToken.getFeatureScore() * pToken.getEta();
41     
42                 // in linear domain, the following expression is:
43                 // totalScore *= score
44     
45                 logTotalScore += logScore;
46             }
47     
48             token.setCombinedScore((float) logTotalScore);
49     
50             // set the combined score of all ParallelTokens in this CombineToken
51             for (ParallelToken parallelToken : token) {
52                 parallelToken.setCombinedScore((float) logTotalScore);
53             }
54         }
55     }
