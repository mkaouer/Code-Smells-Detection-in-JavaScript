21     /**
22      * Manages TTS output
23      */
24     public class CardMatchVoice {
25     
26         private Synthesizer synthesizer;
27         
28         public CardMatchVoice() throws Exception {
29     	try {
30                 String domain = getProperty("domain", "general");
31     
32     	    // Create a new SynthesizerModeDesc that will match the FreeTTS
33     	    // Synthesizer.
34     	    SynthesizerModeDesc desc = new SynthesizerModeDesc
35     		(null,
36     		 domain,
37     		 Locale.US,
38     		 Boolean.FALSE,         // running?
39     		 null);                 // voice
40     
41     	    synthesizer = Central.createSynthesizer(desc);
42     
43     	    if (synthesizer == null) {
44     		String message = "Can't find synthesizer.\n" +
45     		    "Make sure that there is a \"speech.properties\" file " +
46     		    "at either of these locations: \n";
47     		message += "user.home    : " + 
48     		    System.getProperty("user.home") + '\n';
49     		message += "java.home/lib: " + System.getProperty("java.home")
50     		    + File.separator + "lib\n";
51     		
52     		System.err.println(message);
53                     throw new Exception("JSAPI Synthesizer cannot be constructed");
54     	    }
55     
56                 // create the voice
57                 String voiceName = System.getProperty("voiceName", "kevin16");
58     	    Voice voice = new Voice
59                     (voiceName, Voice.GENDER_DONT_CARE, Voice.AGE_DONT_CARE, null);
60     
61     	    // get it ready to speak
62     	    synthesizer.allocate();
63     	    synthesizer.resume();
64                 synthesizer.getSynthesizerProperties().setVoice(voice);
65     
66             } catch (Exception e) {
67                 e.printStackTrace();
68                 throw new Exception("Unable to initialize CardMatchVoice");
69             }
70         }

72         /**
73          * Gets a property by name and returns its value. If the property
74          * cannot be found, the default is returned
75          *
76          * @param name the name of the property
77          *
78          * @param defaultValue the default value to use if the property
79          * cannot be found.
80          *
81          * @return the string value for the property, or the defaultValue if 
82          *  the property cannot be found
83          */
84         public static String getProperty(String name, String defaultValue) {
85             String value;
86             try {
87                 value = System.getProperty(name, defaultValue);
88             } catch (SecurityException se) {
89                 value = defaultValue;
90             }
91             return value;
92         }
93     
94         /**
95          * Speaks the given string of text.
96          *
97          * @param text the text to speak
98          */
99         public void speak(String text) {
100             if (synthesizer != null) {
101                 try {
102                     synthesizer.speakPlainText(text, null);
103                 } catch (EngineStateError ese) {
104                     ese.printStackTrace();
105                 }
106             } else {
107                 System.out.println("CardMatchVoice unavailable");
108             }
109         }
110     
111         /**
112          * Shuts down this CardMatchVoice.
113          */
114         public void close() {
115             if (synthesizer != null) {
116                 try {
117                     synthesizer.deallocate();
118                 } catch (Exception e) {
119                     e.printStackTrace();
120                 }
121             }
122         }
123     }
