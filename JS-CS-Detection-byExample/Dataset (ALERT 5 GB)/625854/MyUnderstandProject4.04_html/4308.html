13     package edu.cmu.sphinx.util.props;
14     
15     import java.util.HashMap;
16     import java.util.List;
17     import java.util.Map;
18     
19     
20     /** Holds the raw property data just as it has come in from the properties file. */
21     public class RawPropertyData {
22     
23         private String name;
24         private String className;
25         private Map<String, Object> properties;
26         
27         /**
28          * Creates a raw property data item.
29          *
30          * @param name the name of the item
31          * @param className the class name of the item
32          */
33         public RawPropertyData(String name, String className) {
34             this(name, className, new HashMap<String, Object>());
35         }
36     
37         /**
38          * Creates a raw property data item, using a given property map.
39          *
40          * @param name the name of the item
41          * @param className the class name of the item
42          * @param properties existing property map to use
43          */
44         public RawPropertyData(String name, String className, Map<String, Object> properties) {
45             this.name = name;
46             this.className = className;
47             this.properties = properties;
48         }
49     
50         /**
51          * Adds a new property with a {@code String} value.
52          *
53          * @param propName the name of the property
54          * @param propValue the value of the property
55          */
56         public void add(String propName, String propValue) {
57             properties.put(propName, propValue);
58         }
59     
60         /**
61          * Adds a new property with a {@code List<String>} value.
62          *
63          * @param propName the name of the property
64          * @param propValue the value of the property
65          */
66         public void add(String propName, List<String> propValue) {
67             properties.put(propName, propValue);
68         }
69     
70         /**
71          * Removes an existing property.
72          *
73          * @param propName the name of the property
74          */
75         public void remove(String propName) {
76             properties.remove(propName);
77         }
78     
79         /** Returns the className. */
80         public String getClassName() {
81             return className;
82         }
83     
84         /** @return Returns the name. */
85         public String getName() {
86             return name;
87         }
88     
89         /** @return Returns the properties. */
90         public Map<String, Object> getProperties() {
91             return properties;
92         }
93     
94         /**
95          * Determines if the map already contains an entry for a property.
96          *
97          * @param propName the property of interest
98          * @return true if the map already contains this property
99          */
100         public boolean contains(String propName) {
101             return properties.get(propName) != null;
102         }
103     
104         /** Returns a copy of this property data instance with all ${}-fields resolved. */
105         public RawPropertyData flatten(ConfigurationManager cm) {
106             RawPropertyData copyRPD = new RawPropertyData(name, className);
107     
108             for (Map.Entry<String, Object> entry : properties.entrySet()) {
109                 Object propVal = entry.getValue();
110                 if (propVal instanceof String) {
111                     if (((String) propVal).startsWith("${"))
112                         propVal = cm.getGloPropReference(ConfigurationManagerUtils.stripGlobalSymbol((String) propVal));
113                 }
114     
115                 copyRPD.properties.put(entry.getKey(), propVal);
116             }
117     
118             return copyRPD;
119         }
120     
121         /**
122          * Lookup a global symbol with a given name (and resolves
123          *
124          * @param key              the name of the property
125          * @param globalProperties
126          * @return the property value or null if it doesn't exist.
127          */
128         public String getGlobalProperty(String key, Map<String, String> globalProperties) {
129             if (!key.startsWith("${")) // is symbol already flat
130                 return key;
131     
132             while (true) {
133                 key = globalProperties.get(key);
134                 if (key == null || !(key.startsWith("${") && key.endsWith("}")))
135                     return key;
136             }
137         }
138     
139         /**
140          * Provide information stored inside this Object, used mainly for debugging/testing.
141          *
142          * @return Description of object
143          */
144         @Override
145         public String toString(){
146             StringBuilder output = new StringBuilder().append("name : ").append(name);
147             for (Object value : properties.values()) {
148                 if (value != null) {
149                     if (value instanceof String) {
150                         output.append("value string : ");
151                     }
152                     output.append(value);
153                 }
154             }
155              return output.toString();
156         }
157     }
