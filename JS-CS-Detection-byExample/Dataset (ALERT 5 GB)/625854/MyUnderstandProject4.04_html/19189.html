12     package edu.cmu.sphinx.result;
13     
14     import java.util.LinkedList;
15     import java.util.List;
16     
17     import edu.cmu.sphinx.linguist.WordSequence;
18     import edu.cmu.sphinx.linguist.dictionary.Word;
19     import edu.cmu.sphinx.linguist.language.ngram.LanguageModel;
20     import edu.cmu.sphinx.util.LogMath;
21     
22     /**
23      * Class to rescore the lattice with the new Language model.
24      */
25     
26     public class LatticeRescorer {
27     
28         protected final Lattice lattice;
29         protected final LanguageModel model;
30         private int depth;
31         private float languageWeigth = 8.0f;
32     
33         /**
34          * Create a new Lattice optimizer
35          * 
36          * @param lattice
37          */
38         public LatticeRescorer(Lattice lattice, LanguageModel model) {
39             this.lattice = lattice;
40             this.model = model;
41             depth = model.getMaxDepth();
42         }
43     
44         private boolean isFillerNode(Node node) {
45             return node.getWord().getSpelling().equals("<sil>");
46         }
47     
48         public void removeFillers() {
49             for (Node node : lattice.sortNodes()) {
50                 if (isFillerNode(node)) {
51                     lattice.removeNodeAndCrossConnectEdges(node);
52                     assert lattice.checkConsistency();
53                 }
54             }
55         }
56     
57         private void rescoreEdges() {
58             for (Edge edge : lattice.edges) {
59     
60                 if (isFillerNode(edge.getToNode()))
61                     continue;
62     
63                 List<String> paths = allPathsTo("", edge, depth);
64     
65                 float minProb = LogMath.getLogZero();
66                 for (String path : paths) {
67                     List<Word> wordList = new LinkedList<Word>();
68                     for (String pathWord : path.split(" ")) {
69                         wordList.add(new Word(pathWord, null, false));
70                     }
71                     wordList.add(edge.getToNode().getWord());
72     
73                     WordSequence seq = new WordSequence(wordList);
74                     float prob = model.getProbability(seq) * languageWeigth;
75                     if (minProb < prob)
76                         minProb = prob;
77                 }
78                 edge.setLMScore(minProb);
79             }
80         }
81     
82         protected List<String> allPathsTo(String path, Edge edge, int currentDepth) {
83             List<String> l = new LinkedList<String>();
84             String p = edge.getFromNode().getWord().toString() + ' ' + path;
85     
86             if (currentDepth == 2
87                     || edge.getFromNode().equals(lattice.getInitialNode())) {
88                 l.add(p);
89             } else {
90                 for (Edge e : edge.getFromNode().getEnteringEdges()) {
91                     l.addAll(allPathsTo(p, e, currentDepth - 1));
92                 }
93             }
94             return l;
95         }
96     
97         public void rescore() {
98     
99             removeFillers();
100     
101             rescoreEdges();
102         }
103     }
