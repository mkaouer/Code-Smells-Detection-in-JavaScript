35     /**
36      * Decodes a batch file containing a list of files to decode. The files can be
37      * either audio files or cepstral files, but defaults to audio files. The audio
38      * data should be 16-bit, 16kHz, PCM-linear data. Since this classes makes use
39      * of Java Sound, it supports all the audio file formats that are supported by
40      * Java Sound. If the audio file does not correspond to a format supported by
41      * Java Sound, it is treated as a raw audio file (i.e., one without a header).
42      * Audio file formats differ in the endian order of the audio data. Therefore,
43      * it is important to specify it correctly in the configuration of the <a
44      * href="../../frontend/util/StreamDataSource.html">StreamDataSource</a>. Note
45      * that in the ideal situation, the audio format of the data should be passed
46      * into the StreamDataSource, so that no extra configuration is needed. This
47      * will be fixed in future releases.
48      * <p/>
49      * To run this BatchModeRecognizer:
50      * 
51      * <pre>
52      * java BatchModeRecognizer &lt;xmlConfigFile&gt; &lt;batchFile&gt;
53      * </pre>
54      * 
55      * where <code>xmlConfigFile</code> is an XML-based configuration file and
56      * <code>batchFile</code> is a file listing all the files to decode and
57      * transcript of those files. For information about the configuration file,
58      * refer to the document <a
59      * href="../../util/props/doc-files/ConfigurationManagement.html"> Sphinx-4
60      * Configuration Management</a>. For information about the batch file, refer to
61      * the <a href="../../../../../../index.html#batch_files"> batch file
62      * description</a>.
63      * <p/>
64      * This class will send recognition results to the logger if the log level is
65      * set to INFO.
66      */

199         /**
200          * Decodes the batch of audio files
201          */
202         public void decode(String batchFile) throws IOException {
203             BatchItem batchItem;
204             int count = 0;
205             try {
206                 recognizer.allocate();
207                 setBatchFile(batchFile);
208     
209                 batchManager.start();
210                 logger.info("BatchDecoder: decoding files in "
211                         + batchManager.getFilename());
212     
213                 while (count < utteranceId &&
214                         (batchItem = batchManager.getNextItem()) != null) {
215                     setInputStream(batchItem.getFilename());
216                     Result result = recognizer.recognize(batchItem.getTranscript());
217                     logger.info("File  : " + batchItem.getFilename());
218                     logger.info("Result: " + result);
219                     count++;
220                 }
221                 batchManager.stop();
222                 recognizer.deallocate();
223             } catch (IOException io) {
224                 logger.severe("I/O error during decoding: " + io.getMessage());
225                 throw io;
226             }
227             logger.info("BatchDecoder: " + count + " files decoded");
228         }

231         /**
232          * Sets the input stream to the given filename
233          *
234          * @param filename the filename to set the input stream to
235          * @throws IOException if an error occurs
236          */
237         void setInputStream(String filename) throws IOException {
238             for (DataProcessor dataSource : inputDataProcessors) {
239                 InputStream is;
240                 try {
241                     File file = new File(filename);
242                     logger.info
243                             (AudioSystem.getAudioFileFormat(file).toString());
244                     is = AudioSystem.getAudioInputStream(file);
245                 } catch (UnsupportedAudioFileException uafe) {
246                     logger.info
247                             ("Reading " + filename + " as raw audio file.");
248                     is = new FileInputStream(filename);
249                 }
250                 if (dataSource instanceof StreamDataSource) {
251                     ((StreamDataSource) dataSource).setInputStream(is, filename);
252                 } else if (dataSource instanceof StreamCepstrumSource) {
253                     boolean isBigEndian = Utilities
254                             .isCepstraFileBigEndian(filename);
255                     StreamCepstrumSource cepstrumSource =
256                             (StreamCepstrumSource) dataSource;
257                     cepstrumSource.setInputStream(is, isBigEndian);
258                     // TODO: christophe: should use an interface there !!
259                 } else if (dataSource instanceof StreamHTKCepstrum) {
260                     StreamHTKCepstrum cepstrumSource =
261                             (StreamHTKCepstrum) dataSource;
262                     cepstrumSource.setInputStream(is);
263                 }
264             }
265         }

587         public void shell(String batchfile) throws IOException {
588             try {
589                 CommandInterpreter ci = new CommandInterpreter();
590                 ci.setPrompt("s4> ");
591                 addCommands(ci);
592                 setBatchFile(batchfile);
593                 recognizer.allocate();
594                 ci.run();
595                 batchManager.stop();
596                 if (recognizer.getState() == State.READY) {
597                     recognizer.deallocate();
598                 }
599             } catch (IOException io) {
600                 logger.severe("I/O error during decoding: " + io.getMessage());
601                 throw io;
602             }
603         }

608          *
609          * @param argv argv[0] : config.xml argv[1] : a file listing all the audio files to decode
610          */
611         public static void main(String[] argv) {
612             if (argv.length < 2) {
613                 System.out.println(
614                         "Usage: BatchDecoder propertiesFile batchFile [-shell]");
615                 System.exit(1);
616             }
617             String cmFile = argv[0];
618             String batchFile = argv[1];
619             ConfigurationManager cm;
620             BatchModeRecognizer bmr;
621     
622             try {
623                 URL url = new File(cmFile).toURI().toURL();
624                 cm = new ConfigurationManager(url);
625                 bmr = (BatchModeRecognizer) cm.lookup("batch");
626                 if (bmr == null) {
627                     System.err.println("Can't find batchModeRecognizer in " + cmFile);
628                     return;
629                 }
630                 if (argv.length >= 3 && argv[2].equals("-shell")) {
631                     bmr.shell(batchFile);
632                 } else {
633                     bmr.decode(batchFile);
634                 }

672         public Result recognize() throws IOException {
673             Result result = null;
674             BatchItem batchItem;
675             if (count < utteranceId &&
676                     (batchItem = batchManager.getNextItem()) != null) {
677                 setInputStream(batchItem.getFilename());
678                 result = recognizer.recognize(batchItem.getTranscript());
679                 logger.info("File  : " + batchItem.getFilename());
680                 logger.info("Result: " + result);
681                 count++;
682             }
683             logger.info("BatchDecoder: " + count + " files decoded");
684             return result;
