13     package edu.cmu.sphinx.result;

15     import java.util.LinkedList;
16     import java.util.List;
17     import java.util.ListIterator;
18     
19     
20     /**
21      * This is an implementation of an alternative sausage making algorithm as described in the following paper. Runs in
22      * less time than the SausageMaker, but will also build different sausages. A general algorithm for word graph matrix
23      * decomposition, Hakkani-Tur, D. Riccardi, G., AT&T Labs.-Res., USA;, This paper appears in: 2003 IEEE International
24      * Conference on Acoustics, Speech, and Signal Processing, 2003. Proceedings. (ICASSP '03).
25      *
26      * @author P. Gorniak
27      */
28     public class PivotSausageMaker extends AbstractSausageMaker {
29     
30         protected List<Node> sortedNodes;
31         protected List<Cluster> clusters;
32     
33         /**
34          * Turn the lattice contained in this sausage maker into a sausage object.
35          *
36          * @return the sausage producing by collapsing the lattice.
37          */
38         public Sausage makeSausage() {
39             for (Node node : sortedNodes) {
40                 int slot = findMostOverlappingSlot(node);
41                 Cluster targetCluster = getCluster(slot);
42                 if (!containsAncestor(targetCluster, node)) {
43                     targetCluster.add(node);
44                 } else {
45                     Cluster newCluster = new Cluster(node);
46                     int newTime = (targetCluster.startTime + targetCluster.endTime) / 2;
47                     targetCluster.endTime = newTime;
48                     newCluster.startTime = newTime;
49                     clusters.add(slot + 1, newCluster);
50                 }
51             }
52             return sausageFromClusters(clusters);
53         }
54     
55     
56         /**
57          * Get a cluster by index
58          *
59          * @param i the index into the list of clusters
60          * @return the requested cluster
61          */
62         protected Cluster getCluster(int i) {
63             return clusters.get(i);
64         }
65     
66     
67         /**
68          * Check whether this given cluster contains and ancestor of the given node
69          *
70          * @param c the cluster to check
71          * @param n the node to check for ancestors
72          * @return whether and ancestor was found
73          */
74         protected boolean containsAncestor(Cluster c, Node n) {
75             for (Node cNode : c.getElements()) {
76                 if (cNode.isAncestorOf(n)) {
77                     return true;
78                 }
79             }
80             return false;
81         }
82     
83     
84         /**
85          * Find the most overlapping time slot in the list of clusters
86          *
87          * @param n the node whose times to check for
88          * @return the index of the most overlapping slot
89          */
90         protected int findMostOverlappingSlot(Node n) {
91             int maxOverlap = Integer.MIN_VALUE;
92             int bestCluster = -1;
93             ListIterator<Cluster> i = clusters.listIterator();
94             i.next(); //never cluster anything with the <s> node
95             while (i.hasNext()) {
96                 int index = i.nextIndex();
97                 if (!i.hasNext()) {
98                     //never cluster anything with the </s> node
99                     return bestCluster;
100                 }
101                 Cluster c = i.next();
102                 int overlap = getOverlap(n, c.startTime, c.endTime);
103                 if (overlap > maxOverlap) {
104                     maxOverlap = overlap;
105                     bestCluster = index;
106                 }
107             }
108             return bestCluster;
109         }
110     
111     
112         /** @see edu.cmu.sphinx.result.ConfidenceScorer#score(edu.cmu.sphinx.result.Result) */
113         @Override
114         public ConfidenceResult score(Result result) {
115             lattice = new Lattice(result);
116             LatticeOptimizer lop = new LatticeOptimizer(lattice);
117             lop.optimize();
118             lattice.computeNodePosteriors(languageWeight);
119             List<Node> seedPath = lattice.getViterbiPath();
120             sortedNodes = lattice.sortNodes();
121             sortedNodes.removeAll(seedPath);
122             clusters = new LinkedList<Cluster>();
123             for (Node n : seedPath) {
124                 clusters.add(new Cluster(n));
125             }
126             return makeSausage();
127         }
