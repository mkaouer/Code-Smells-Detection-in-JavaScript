294         /**
295          * Gets the feature frames associated with this result
296          *
297          * @return the set of feature frames associated with this result, or null if the frames are not available.
298          */
299         public List<Data> getDataFrames() {
300             // find the best token, and then trace back for all the features
301             Token token = getBestToken();
302     
303             if (token == null)
304                 return null;
305     
306             List<Data> featureList = new LinkedList<Data>();
307     
308             do {
309                 Data feature = token.getData();
310                 if (feature != null)
311                     featureList.add(0, feature);
312     
313                 token = token.getPredecessor();
314             } while (token != null);
315     
316             return featureList;
317         }

371          * Returns the string of words (with timestamp) for this token.
372          *
373          * @param wantFiller     true if we want filler words included, false otherwise
374          * @param wordTokenFirst true if the word tokens come before other types of tokens
375          * @return the string of words
376          */
377         public String getTimedBestResult(boolean wantFiller,
378                                          boolean wordTokenFirst) {
379             Token token = getBestToken();
380             if (token == null) {
381                 return "";
382             } else {
383                 if (wordTokenFirst) {
384                     return getTimedWordPath(token, wantFiller);
385                 } else {
386                     return getTimedWordTokenLastPath(token, wantFiller);
387                 }
388             }
389         }

392         /**
393          * Returns the string of words (with timestamp) for this token. This method assumes that the word tokens come before
394          * other types of token.
395          *
396          * @param wantFiller true if we want filler words, false otherwise
397          * @return the string of words
398          */
399         private String getTimedWordPath(Token token, boolean wantFiller) {
400             StringBuilder sb = new StringBuilder();
401     
402             // get to the first emitting token
403             while (token != null && !token.isEmitting()) {
404                 token = token.getPredecessor();
405             }
406     
407             if (token != null) {
408                 Data lastWordFirstFeature = token.getData();
409                 Data lastFeature = lastWordFirstFeature;
410                 token = token.getPredecessor();
411     
412                 while (token != null) {
413                     if (token.isWord()) {
414                         Word word = token.getWord();
415                         if (wantFiller || !word.isFiller()) {
416                             addWord(sb, word,
417                                     (FloatData) lastFeature,
418                                     (FloatData) lastWordFirstFeature);
419                         }
420                         lastWordFirstFeature = lastFeature;
421                     }
422                     Data feature = token.getData();
423                     if (feature != null) {
424                         lastFeature = feature;
425                     }
426                     token = token.getPredecessor();
427                 }
428             }
429             return sb.toString();
430         }

433         /**
434          * Returns the string of words for this token, each with the starting sample number as the timestamp. This method
435          * assumes that the word tokens come after the unit and hmm tokens.
436          *
437          * @return the string of words, each with the starting sample number
438          */
439         private String getTimedWordTokenLastPath(Token token, boolean wantFiller) {
440             StringBuilder sb = new StringBuilder();
441             Word word = null;
442             Data lastFeature = null;
443             Data lastWordFirstFeature = null;
444     
445             while (token != null) {
446                 if (token.isWord()) {
447                     if (word != null) {
448                         if (wantFiller || !word.isFiller()) {
449                             addWord(sb, word,
450                                     (FloatData) lastFeature,
451                                     (FloatData) lastWordFirstFeature);
452                         }
453                         word = token.getWord();
454                         lastWordFirstFeature = lastFeature;
455                     }
456                     word = token.getWord();
457                 }
458                 Data feature = token.getData();
459                 if (feature != null) {
460                     lastFeature = feature;
461                     if (lastWordFirstFeature == null) {
462                         lastWordFirstFeature = lastFeature;
463                     }
464                 }
465                 token = token.getPredecessor();
466             }
467     
468             return sb.toString();
469         }

472         /**
473          * Adds the given word into the given string builder with the start and end times from the given features.
474          *
475          * @param sb           the StringBuilder into which the word is added
476          * @param word         the word to add
477          * @param startFeature the starting feature
478          * @param endFeature   tne ending feature
479          */
480         private void addWord(StringBuilder sb, Word word,
481                              FloatData startFeature, FloatData endFeature) {
482             float startTime = startFeature == null ? -1 : ((float) startFeature.getFirstSampleNumber() /
483                     startFeature.getSampleRate());
484             float endTime = endFeature == null ? -1 : ((float) endFeature.getFirstSampleNumber() /
485                     endFeature.getSampleRate());
486             if (sb.length() > 0) {
487                 sb.insert(0, ' ');
488             }
489             sb.insert(0, (word.getSpelling() + '(' + startTime + ',' +
490                     endTime + ')'));
491         }

516         /**
517          * Determines if the Result is valid. This is used for testing and debugging
518          *
519          * @return true if the result is properly formed.
520          */
521         public boolean validate() {
522             boolean valid = true;
523             for (Token token : activeList) {
524                 if (!token.validate()) {
525                     valid = false;
526                     token.dumpTokenPath();
527                 }
528             }
529             return valid;
530         }
