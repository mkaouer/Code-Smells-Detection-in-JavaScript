11      * @author Holger Brandl

13     public class TimerPool {

16

24         /**
25          * Retrieves (or creates) a timer with the given name
26          *
27          * @param owner
28          * @param timerName the name of the particular timer to retrieve. If the timer does not already exist, it will be
29          *                  created  @return the timer.
30          */
31         public static synchronized Timer getTimer(Object owner, String timerName) {
32             if (!weakRefTimerPool.containsKey(owner))
33                 weakRefTimerPool.put(owner, new ArrayList<Timer>());
34     
35             List<Timer> ownerTimers = weakRefTimerPool.get(owner);
36     
37             for (Timer timer : ownerTimers) {
38                 if (timer.getName().equals(timerName))
39                     return timer;
40             }
41     
42             // there is no timer named 'timerName' yet, so create it
43             Timer requestedTimer = new Timer(timerName);
44             ownerTimers.add(requestedTimer);
45     
46             return requestedTimer;
47         }

50         /** Returns the number of currently caches {@code Timer} instances. */
51         public static int getNumCachedTimers() {
52             int counter = 0;
53             for (List<Timer> timers : weakRefTimerPool.values()) {
54                 counter += timers.size();
55             }
56     
57             return counter;
58         }
