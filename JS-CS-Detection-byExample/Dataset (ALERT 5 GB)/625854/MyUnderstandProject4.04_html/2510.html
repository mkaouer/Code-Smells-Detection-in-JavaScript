10     package edu.cmu.sphinx.frontend.feature;
11     
12     import edu.cmu.sphinx.frontend.*;
13     import edu.cmu.sphinx.linguist.acoustic.tiedstate.*;
14     import edu.cmu.sphinx.util.props.*;
15     
16     /**
17      * Implements an LDA transformation. The dimension of the feature stream is
18      * reduced with a matrix transform thus reducing speed and increasing accuracy.
19      * This component requires special model trained with LDA/MLLT transform.
20      */
21     public class LDA extends BaseDataProcessor {
22     
23     	/** The name of the transform matrix file */
24     	@S4Component(type = Loader.class)
25     	public final static String PROP_LOADER = "loader";
26     
27     	float[][] transformMatrix;
28     	protected Loader loader;
29     	
30     	int rows;
31     	int values;
32     
33         public LDA( Loader loader ) {
34             initLogger();
35             init(loader);
36         }
37     
38         public LDA() {        
39         }
40     
41     	/*
42     	 * (non-Javadoc)
43     	 * 
44     	 * @see
45     	 * edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util
46     	 * .props.PropertySheet)
47     	 */
48     	@Override
49         public void newProperties(PropertySheet ps) throws PropertyException {
50     		super.newProperties(ps);
51     		
52     		init( (Loader)ps.getComponent(PROP_LOADER) );
53     	}
54     
55         private void init(Loader loader) {
56             this.loader = loader;
57     
58     		try {
59     			loader.load();
60     		} catch (Exception e) {
61     			e.printStackTrace();
62     		}
63     		this.transformMatrix = loader.getTransformMatrix();
64     
65     		if (transformMatrix == null)
66     			throw new RuntimeException("Model doesn't include transformation matrix");
67     
68     		this.rows = transformMatrix.length;
69     		this.values = transformMatrix[0].length;
70         }
71     
72     	/**
73     	 * Returns the next Data object being processed by this LDA, or if it is a
74     	 * Signal, it is returned without modification.
75     	 * 
76     	 * @return the next available Data object, returns null if no Data object is
77     	 *         available
78     	 * @throws DataProcessingException
79     	 *             if there is a processing error
80     	 * @see Data
81     	 */
82     	@Override
83         public Data getData() throws DataProcessingException {
84     		Data input = getPredecessor().getData();
85     		Data output;
86     		getTimer().start();
87     		if (input != null && input instanceof FloatData) {
88     			FloatData inputData = (FloatData) input;
89     			float[] in = inputData.getValues();
90     			float[] out = new float[rows];
91     			
92     			assert in.length == values; 
93     
94     			for (int i = 0; i < rows; i++) {
95     				out [i] = 0;
96     				float transform_matrix_i[] = transformMatrix[i];
97     				for (int j = 0; j < values; j++) {
98     					out[i] += in[j] * transform_matrix_i[j];
99     				}
100     			}
101     			output = new FloatData(out, inputData.getSampleRate(), inputData
102     					.getCollectTime(), inputData.getFirstSampleNumber());
103     		} else {
104     			output = input;
105     		}
106     		getTimer().stop();
107     		return output;
108     	}
109     }
