102         /**
103          * Given a unit, returns the HMM that best matches the given unit. If exactMatch is false and an exact match is not
104          * found, then different word positions are used. If any of the contexts are non-silence filler units. a silence
105          * filler unit is tried instead.
106          *
107          * @param unit       the unit of interest
108          * @param position   the position of the unit of interest
109          * @param exactMatch if true, only an exact match is acceptable.
110          * @return the HMM that best matches, or null if no match could be found.
111          */
112         @Override
113         public HMM lookupNearestHMM(Unit unit, HMMPosition position,
114                                     boolean exactMatch) {
115             HMM hmm = null;
116             if (!exactMatch || position == HMMPosition.UNDEFINED) {
117                 unit = unit.getBaseUnit();
118                 hmm = hmmMap.get(unit);
119             }
120             return hmm;
121         }

202     /** Trivial HMM */
203     class TrivialHMM implements HMM {
204     
205         private final static int NUM_STATES = 4;
206         final Unit unit;
207         final HMMPosition position;
208         final HMMState[] hmmStates;
209         private final Unit baseUnit;
210     
211     
212         /**
213          * Creates a trivial hmm
214          *
215          * @param unit     the unit for the hmm
216          * @param position the position of the hmm
217          */
218         TrivialHMM(Unit unit, HMMPosition position) {
219             this.unit = unit;
220             this.position = position;
221             hmmStates = new HMMState[NUM_STATES];
222             // baseUnit = Unit.getUnit(unit.getName());
223             baseUnit = unit.getBaseUnit();
224     
225             for (int i = 0; i < hmmStates.length; i++) {
226                 boolean finalState = i == hmmStates.length - 1;
227                 hmmStates[i] = new TrivialHMMState(this, i, finalState);
228             }
229         }
230     
231     
232         /**
233          * Gets the  unit associated with this HMM
234          *
235          * @return the unit associated with this HMM
236          */
237         @Override
238         public Unit getUnit() {
239             return unit;
240         }
241     
242     
243         /**
244          * Gets the  base unit associated with this HMM
245          *
246          * @return the unit associated with this HMM
247          */
248         @Override
249         public Unit getBaseUnit() {
250             return baseUnit;
251         }
252     
253     
254         /**
255          * Retrieves the hmm state
256          *
257          * @param which the state of interest
258          */
259         @Override
260         public HMMState getState(int which) {
261             return hmmStates[which];
262         }
263     
264     
265         /**
266          * Returns the order of the HMM
267          *
268          * @return the order of the HMM
269          */
270         @Override
271         public int getOrder() {
272             return hmmStates.length;
273         }
274     
275     
276         /**
277          * Retrieves the position of this HMM.
278          *
279          * @return the position for this HMM
280          */
281         @Override
282         public HMMPosition getPosition() {
283             return position;
284         }

286     
287         /**
288          * Gets the initial states (with probabilities) for this HMM

299     /** A trivial implementation of an HMMState */
300     class TrivialHMMState implements HMMState {
301     
302         private static final HMMStateArc[] EMPTY_ARC = new HMMStateArc[0];
303         private final HMM hmm;
304         private final int which;
305         private final boolean isFinal;
306     
307     
308         TrivialHMMState(HMM hmm, int which, boolean isFinal) {
309             this.hmm = hmm;
310             this.which = which;
311             this.isFinal = isFinal;
312         }
313     
314     
315         /**
316          * Gets the HMM associated with this state
317          *
318          * @return the HMM
319          */
320         @Override
321         public HMM getHMM() {
322             return hmm;
323         }
324     
325     
326         /**
327          * Gets the state
328          *
329          * @return the state
330          */
331         @Override
332         public int getState() {
333             return which;
334         }
335     
336     
337         /**
338          * Gets the score for this HMM state
339          *
340          * @param feature the feature to be scored
341          * @return the acoustic score for this state.
342          */
343         @Override
344         public float getScore(Data feature) {
345             return 0.0f;
346         }
347     
348     
349         /**
350          * Determines if this HMMState is an emittting state
351          *
352          * @return true if the state is an emitting state
353          */
354         @Override
355         public boolean isEmitting() {
356             return !isFinal;
357         }
358     
359     
360         /**
361          * Retrieves the state of successor states for this state
362          *
363          * @return the set of successor state arcs
364          */
365         @Override
366         public HMMStateArc[] getSuccessors() {
367             if (isFinal) {
368                 return EMPTY_ARC;
369             } else {
370                 HMMStateArc[] arcs = new HMMStateArc[2];
371                 arcs[0] = new HMMStateArc(this, 0f);
372                 arcs[1] = new HMMStateArc(hmm.getState(which + 1), 0f);
373                 return arcs;
374             }
375         }
376     
377     
378         /**
379          * Determines if this state is an exit state of the HMM
380          *
381          * @return true if the state is an exit state
382          */
383         @Override
384         public boolean isExitState() {
385             return isFinal;
386         }
387     }
