66         /**
67          * Make the Lattice deterministic, so that no node has multiple outgoing edges to equivalent nodes.
68          * <p/>
69          * Given two edges from the same node to two equivalent nodes, replace with one edge to one node with outgoing edges
70          * that are a union of the outgoing edges of the old two nodes.
71          * <p/>
72          * A --> B --> C \--> B' --> Y
73          * <p/>
74          * where B and B' are equivalent.
75          * <p/>
76          * is replaced with
77          * <p/>
78          * A --> B" --> C \--> Y
79          * <p/>
80          * where B" is the merge of B and B'
81          * <p/>
82          * Note that equivalent nodes must have the same incomming edges. For example
83          * <p/>
84          * A --> B \ \ X --> B'
85          * <p/>
86          * B and B' would not be equivalent because the incomming edges are different
87          */
88         protected void optimizeForward() {
89             //System.err.println("*** Optimizing forward ***");
90     
91             boolean moreChanges = true;
92             while (moreChanges) {
93                 moreChanges = false;
94                 // search for a node that can be optimized
95                 // note that we use getCopyOfNodes to avoid concurrent changes to nodes
96                 for (Node n : lattice.getCopyOfNodes()) {
97                     // we are iterating down a list of node before optimization
98                     // previous iterations may have removed nodes from the list
99                     // therefore we have to check that the node stiff exists
100                     if (lattice.hasNode(n)) {
101                         moreChanges |= optimizeNodeForward(n);
102                     }
103                 }
104             }
105         }

108         /**
109          * Look for 2 "to" edges to equivalent nodes.  Replace the edges with one edge to one node that is a merge of the
110          * equivalent nodes
111          * <p/>
112          * nodes are equivalent if they have equivalent from edges, and the same label
113          * <p/>
114          * merged nodes have a union of "from" and "to" edges
115          *
116          * @param n
117          * @return true if Node n required an optimize forward
118          */
119         protected boolean optimizeNodeForward(Node n) {
120             assert lattice.hasNode(n);
121     
122             List<Edge> leavingEdges = new ArrayList<Edge>(n.getLeavingEdges());
123             for (int j = 0; j < leavingEdges.size(); j++) {
124                 Edge e = leavingEdges.get(j);
125                 for (int k = j + 1; k < leavingEdges.size(); k++) {
126                     Edge e2 = leavingEdges.get(k);
127     
128                     /*
129                      * If these are not the same edge, and they point to
130                      * equivalent nodes, we have a hit, return true
131                      */
132                     assert e != e2;
133                     if (equivalentNodesForward(e.getToNode(), e2.getToNode())) {
134                         mergeNodesAndEdgesForward(n, e, e2);
135                         return true;
136                     }
137                 }
138             }
139             /*
140              * return false if we did not get a hit
141              */
142             return false;
143         }

146         /**
147          * nodes are equivalent forward if they have "from" edges from the same nodes, and have equivalent labels (Token,
148          * start/end times)
149          *
150          * @param n1
151          * @param n2
152          * @return true if n1 and n2 are "equivalent forwards"
153          */
154         protected boolean equivalentNodesForward(Node n1, Node n2) {
155     
156             assert lattice.hasNode(n1);
157             assert lattice.hasNode(n2);
158     
159             // do the labels match?
160             if (!equivalentNodeLabels(n1, n2)) return false;
161     
162             // if they have different number of "from" edges they are not equivalent
163             // or if there is a "from" edge with no match then the nodes are not
164             // equivalent
165             return n1.hasEquivalentEnteringEdges(n2);
166         }

169         /**
170          * given edges e1 and e2 from node n to nodes n1 and n2
171          * <p/>
172          * merge e1 and e2, that is, merge the scores of e1 and e2 create n' that is a merge of n1 and n2 add n' add edge e'
173          * from n to n'
174          * <p/>
175          * remove n1 and n2 and all associated edges
176          *
177          * @param n
178          * @param e1
179          * @param e2
180          */
181         protected void mergeNodesAndEdgesForward(Node n, Edge e1, Edge e2) {
182             assert lattice.hasNode(n);
183             assert lattice.hasEdge(e1);
184             assert lattice.hasEdge(e2);
185     
186             assert e1.getFromNode() == n;
187             assert e2.getFromNode() == n;
188     
189             Node n1 = e1.getToNode();
190             Node n2 = e2.getToNode();
191     
192             assert n1.hasEquivalentEnteringEdges(n1);
193             assert n1.getWord().equals(n2.getWord());
194     
195             // merge the scores of e1 and e2 into e1
196             e1.setAcousticScore(mergeAcousticScores
197                     (e1.getAcousticScore(), e2.getAcousticScore()));
198             e1.setLMScore(mergeLanguageScores(e1.getLMScore(),
199                     e2.getLMScore()));
200     
201             // add n2's edges to n1
202             for (Edge e : n2.getLeavingEdges()) {
203                 e2 = n1.getEdgeToNode(e.getToNode());
204                 if (e2 == null) {
205                     lattice.addEdge(n1, e.getToNode(),
206                             e.getAcousticScore(), e.getLMScore());
207                 } else {
208                     // if we got here then n1 and n2 had edges to the same node
209                     // choose the edge with best score
210                     e2.setAcousticScore
211                             (mergeAcousticScores
212                                     (e.getAcousticScore(), e2.getAcousticScore()));
213                     e2.setLMScore(mergeLanguageScores(e.getLMScore(),
214                             e2.getLMScore()));
215                 }
216             }
217     
218             // remove n2 and all associated edges
219             lattice.removeNodeAndEdges(n2);
220         }

223         /**
224          * Minimize the Lattice deterministic, so that no node has multiple incoming edges from equivalent nodes.
225          * <p/>
226          * Given two edges from equivalent nodes to a single nodes, replace with one edge from one node with incoming edges
227          * that are a union of the incoming edges of the old two nodes.
228          * <p/>
229          * A --> B --> C X --> B' --/
230          * <p/>
231          * where B and B' are equivalent.
232          * <p/>
233          * is replaced with
234          * <p/>
235          * A --> B" --> C X --/
236          * <p/>
237          * where B" is the merge of B and B'
238          * <p/>
239          * Note that equivalent nodes must have the same outgoing edges. For example
240          * <p/>
241          * A --> X \ \ \ A' --> B
242          * <p/>
243          * A and A' would not be equivalent because the outgoing edges are different
244          */
245         protected void optimizeBackward() {
246             //System.err.println("*** Optimizing backward ***");
247     
248             boolean moreChanges = true;
249             while (moreChanges) {
250                 moreChanges = false;
251                 // search for a node that can be optimized
252                 // note that we use getCopyOfNodes to avoid concurrent changes to nodes
253                 for (Node n : lattice.getCopyOfNodes()) {
254                     // we are iterating down a list of node before optimization
255                     // previous iterations may have removed nodes from the list
256                     // therefore we have to check that the node stiff exists
257                     if (lattice.hasNode(n)) {
258                         moreChanges |= optimizeNodeBackward(n);
259                     }
260                 }
261             }
262         }

265         /**
266          * Look for 2 entering edges from equivalent nodes.  Replace the edges with one edge to one new node that is a merge
267          * of the equivalent nodes Nodes are equivalent if they have equivalent to edges, and the same label. Merged nodes
268          * have a union of entering and leaving edges
269          *
270          * @param n
271          * @return true if Node n required optimizing backwards
272          */
273         protected boolean optimizeNodeBackward(Node n) {
274             List<Edge> enteringEdges = new ArrayList<Edge>(n.getEnteringEdges());
275             for (int j = 0; j < enteringEdges.size(); j++) {
276                 Edge e = enteringEdges.get(j);
277                 for (int k = j + 1; k < n.getEnteringEdges().size(); k++) {
278                     Edge e2 = enteringEdges.get(k);
279     
280                     /*
281                      * If these are not the same edge, and they point to
282                      * equivalent nodes, we have a hit, return true
283                      */
284                     assert e != e2;
285                     if (equivalentNodesBackward(e.getFromNode(),
286                             e2.getFromNode())) {
287                         mergeNodesAndEdgesBackward(n, e, e2);
288                         return true;
289                     }
290                 }
291             }
292             /*
293              * return false if we did not get a hit
294              */
295             return false;
296         }

299         /**
300          * nodes are equivalent backward if they have "to" edges to the same nodes, and have equivalent labels (Token,
301          * start/end times)
302          *
303          * @param n1
304          * @param n2
305          * @return true if n1 and n2 are "equivalent backwards"
306          */
307         protected boolean equivalentNodesBackward(Node n1, Node n2) {
308     
309             assert lattice.hasNode(n1);
310             assert lattice.hasNode(n2);
311     
312             // do the labels match?
313             if (!equivalentNodeLabels(n1, n2)) return false;
314     
315             // if they have different number of "to" edges they are not equivalent
316             // or if there is a "to" edge with no match then the nodes are not equiv
317             return n1.hasEquivalentLeavingEdges(n2);
318         }

321         /**
322          * Is the contents of these Node equivalent?
323          *
324          * @param n1
325          * @param n2
326          * @return true if n1 and n2 have "equivalent labels"
327          */
328         protected boolean equivalentNodeLabels(Node n1, Node n2) {
329             return (n1.getWord().equals(n2.getWord()) &&
330                     (n1.getBeginTime() == n2.getBeginTime() &&
331                             n1.getEndTime() == n2.getEndTime()));
332         }

389         /** Remove all Nodes that have no Edges to them (but not <s>) */
390         protected void removeHangingNodes() {
391             for (Node n : lattice.getCopyOfNodes()) {
392                 if (lattice.hasNode(n)) {
393                     if (n == lattice.getInitialNode()) {
394     
395                     } else if (n == lattice.getTerminalNode()) {
396     
397                     } else {
398                         if (n.getLeavingEdges().isEmpty()
399                                 || n.getEnteringEdges().isEmpty()) {
400                             lattice.removeNodeAndEdges(n);
401                             removeHangingNodes();
402                             return;
403                         }
404                     }
405                 }
406             }
407         }

410         /**
411          * Provides a single method to merge acoustic scores, so that changes to how acoustic score are merged can be made
412          * at one point only.
413          *
414          * @param score1 the first acoustic score
415          * @param score2 the second acoustic score
416          * @return the merged acoustic score
417          */
418         private double mergeAcousticScores(double score1, double score2) {
419             // return lattice.getLogMath().addAsLinear(score1, score2);
420             return Math.max(score1, score2);
421         }

426          * at one point only.
427          *
428          * @param score1 the first language score
429          * @param score2 the second language score
430          * @return the merged language score
431          */
432         private double mergeLanguageScores(double score1, double score2) {
433             // return lattice.getLogMath().addAsLinear(score1, score2);
434             return Math.max(score1, score2);
