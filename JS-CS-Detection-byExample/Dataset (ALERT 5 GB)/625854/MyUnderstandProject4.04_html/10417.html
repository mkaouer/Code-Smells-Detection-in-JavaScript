13     package edu.cmu.sphinx.linguist;
14     
15     import edu.cmu.sphinx.linguist.dictionary.Word;
16     
17     import java.util.ArrayList;
18     import java.util.List;
19     
20     
21     /**
22      * This class can be used to keep track of a word sequence.  This class is an immutable class. It can never be modified
23      * once it is created (except, perhaps for transient, cached things such as a precalculated hashcode).
24      */
25     
26     public final class WordSequence {
27     
28         private final Word[] words;
29         private transient int hashCode = -1;
30     
31         /** an empty word sequence, that is, it has no words. */
32         public final static WordSequence EMPTY = new WordSequence(0);
33     
34     
35         /**
36          * Constructs a word sequence with the given depth.
37          *
38          * @param size the maxium depth of the word history
39          */
40         private WordSequence(int size) {
41             words = new Word[size];
42         }
43     
44     
45         /**
46          * Constructs a word sequence with the given word IDs
47          *
48          * @param words the word IDs of the word sequence
49          */
50         public WordSequence(Word[] words) {
51             this.words = words.clone();
52             check();
53         }
54     
55     
56         /**
57          * Constructs a word sequence from the list of words
58          *
59          * @param list the list of words
60          */
61         public WordSequence(List<Word> list) {
62             this.words = list.toArray(new Word[list.size()]);
63             check();
64         }
65     
66     
67         private void check() {
68             for (Word word : words)
69                 if (word == null)
70                     throw new Error("WordSequence should not have null Words.");
71         }
72     
73     
74         /**
75          * Returns a new word sequence with the given word added to the sequence
76          *
77          * @param word    the word to add to the sequence
78          * @param maxSize the maximum size of the generated sequence
79          * @return a new word sequence with the word added (but trimmed to maxSize).
80          */
81         public WordSequence addWord(Word word, int maxSize) {
82             if (maxSize <= 0) {
83                 return EMPTY;
84             }
85             int nextSize = ((size() + 1) > maxSize) ? maxSize : (size() + 1);
86             WordSequence next = new WordSequence(nextSize);
87             int nextIndex = nextSize - 1;
88             int thisIndex = size() - 1;
89             next.words[nextIndex--] = word;
90     
91             while (nextIndex >= 0 && thisIndex >= 0) {
92                 next.words[nextIndex--] = this.words[thisIndex--];
93             }
94             next.check();
95     
96             return next;
97         }
98     
99     
100         /**
101          * Returns the oldest words in the sequence (the newest word is omitted)
102          *
103          * @return the oldest words in the sequence, with the newest word omitted
104          */
105         public WordSequence getOldest() {
106             WordSequence next = EMPTY;
107     
108             if (size() >= 1) {
109                 next = new WordSequence(words.length - 1);
110                 System.arraycopy(this.words, 0, next.words, 0, next.words.length);
111             }
112             return next;
113         }
114     
115     
116         /**
117          * Returns the newest words in the sequence (the old word is omitted)
118          *
119          * @return the newest words in the sequence with the oldest word omitted
120          */
121         public WordSequence getNewest() {
122             WordSequence next = EMPTY;
123     
124             if (size() >= 1) {
125                 next = new WordSequence(words.length - 1);
126                 System.arraycopy(this.words, 1, next.words, 0, next.words.length);
127             }
128             return next;
129         }
130     
131     
132         /**
133          * Returns a word sequence that is no longer than the given size, that is filled in with the newest words from this
134          * sequence
135          *
136          * @param maxSize the maximum size of the sequence
137          * @return a new word sequence, trimmed to maxSize.
138          */
139         public WordSequence trim(int maxSize) {
140             if (maxSize <= 0 || size() == 0) {
141                 return EMPTY;
142             } else if (maxSize == size()) {
143                 return this;
144             } else {
145                 if (maxSize > size()) {
146                     maxSize = size();
147                 }
148                 WordSequence next = new WordSequence(maxSize);
149                 int thisIndex = words.length - 1;
150                 int nextIndex = next.words.length - 1;
151     
152                 for (int i = 0; i < maxSize; i++) {
153                     next.words[nextIndex--] = this.words[thisIndex--];
154                 }
155                 return next;
156             }
157         }
158     
159     
160         /**
161          * Returns the nth word in this sequence
162          *
163          * @param n which word to return
164          * @return the nth word in this sequence
165          */
166         public Word getWord(int n) {
167             if (n >= words.length) {
168                 throw new ArrayIndexOutOfBoundsException(n);
169             }
170             return words[n];
171         }
172     
173     
174         /**
175          * Returns the number of words in this sequence
176          *
177          * @return the number of words
178          */
179         public int size() {
180             return words.length;
181         }
182     
183     
184         /**
185          * Returns a string representation of this word sequence. The format is: [ID_0][ID_1][ID_2].
186          *
187          * @return the string
188          */
189         @Override
190         public String toString() {
191             StringBuilder sb = new StringBuilder();
192             for (Word word : words)
193                 sb.append('[').append(word).append(']');
194             return sb.toString();
195         }
196     
197     
198         /**
199          * Calculates the hashcode for this object
200          *
201          * @return a hascode for this object
202          */
203         @Override
204         public int hashCode() {
205             if (hashCode == -1) {
206                 int code = 123;
207                 for (int i = 0; i < words.length; i++) {
208                     code += words[i].hashCode() * (i + 1);
209                 }
210                 hashCode = code;
211             }
212             return hashCode;
213         }
214     
215     
216         /**
217          * compares the given object to see if it is identical to this WordSequence
218          *
219          * @param o the object to compare this to
220          * @return true if the given object is equal to this object
221          */
222         @Override
223         public boolean equals(Object o) {
224             if (this == o) {
225                 return true;
226             } else if (o instanceof WordSequence) {
227                 WordSequence other = (WordSequence) o;
228                 if (words.length == other.words.length) {
229                     for (int i = 0; i < words.length; i++) {
230                         if (!words[i].equals(other.words[i])) {
231                             return false;
232                         }
233                     }
234                     return true;
235                 }
236             }
237             return false;
238         }
239     
240     
241         /**
242          * @return a subsequence with both <code>startIndex</code> and <code>stopIndex</code> exclusive.
243          */
244         public WordSequence getSubSequence(int startIndex, int stopIndex) {
245             List<Word> subseqWords = new ArrayList<Word>();
246     
247             for (int i = startIndex; i < stopIndex; i++) {
248                 subseqWords.add(getWord(i));
249             }
250     
251             return new WordSequence(subseqWords);
252         }
253     
254     
255         /**
256          * @return the words of the <code>WordSequence</code>.
257          */
258         public Word[] getWords() {
259             return getSubSequence(0, size()).words; //create a copy to keep the class immutable
260         }
261     }
