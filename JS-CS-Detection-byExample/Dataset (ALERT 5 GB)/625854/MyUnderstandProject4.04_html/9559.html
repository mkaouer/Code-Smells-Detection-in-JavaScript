218         /**
219          * Sets the current recognizer to the one with the given name.
220          * 
221          * @param recognizerName
222          *                name of the recognizer to use
223          */
224         public void setDecoder(String recognizerName) throws IOException {
225     
226             liveFrame.setDecoderComboBoxEnabled(false);
227             liveFrame.setNewTestFileButtonEnabled(false);
228     
229             String changeMessage = "Changing to " + recognizerName + 
230                 " recognizer ...\n";
231             info(changeMessage);
232             liveFrame.setMessage(changeMessage);
233             liveFrame.setSpeakButtonEnabled(false);
234             liveFrame.setNextButtonEnabled(false);
235             liveFrame.setPlayButtonEnabled(false);
236             liveFrame.setClearButtonEnabled(false);
237     
238             if (currentRecognizer != null) {
239                 currentRecognizer.deallocate();
240                 currentRecognizer = null;
241             }
242     
243             LiveRecognizer nextRecognizer =
244                     recognizers.get(recognizerName);
245             if (nextRecognizer.allocate()) {
246                 // if the decoder switch is successful
247                 currentRecognizer = nextRecognizer;
248                 liveFrame.setTestFile(currentRecognizer.getTestFile());
249                 if (!handsFree) {
250                     liveFrame.setReferenceLabel
251                         (currentRecognizer.getNextReference());
252                 }  else {
253                     liveFrame.setReferenceLabel("");
254                 }
255                 liveFrame.setRecognitionLabel("");
256                 liveFrame.setSpeakButtonEnabled(true);
257                 liveFrame.setNextButtonEnabled(true);
258                 liveFrame.setPlayButtonEnabled(true);
259                 liveFrame.setClearButtonEnabled(true);
260                 liveFrame.setMessage(recognizerName + ", press \"Speak\" to start");
261     
262             } else {
263                 liveFrame.setMessage("Error trying to use " + recognizerName);
264             }
265             info("... done changing\n");
266     
267             liveFrame.setDecoderComboBoxEnabled(true);
268             liveFrame.setNewTestFileButtonEnabled(true);
269         }

309         /**
310          * Parses the given list of decoders, and create a SynthesizerProperties
311          * for each of them. Also creates all the LiveDecoders.
312          * 
313          * @param decoderListFile
314          *                a file listing all the available decoders
315          */
316         private void parseDecoderListFile(String decoderListFile)
317                 throws  IOException,  LineUnavailableException {
318     
319             info("Parsing file " + decoderListFile + ' ');
320     
321             Properties properties = new Properties();
322             properties.load(new FileInputStream(decoderListFile));
323     
324             String decoderLine = properties.getProperty("decoders");
325             String[] recognizerNames = decoderLine.split("\\s+");
326     
327             for (String name : recognizerNames) {
328     
329                 // name of the recognizer
330                 String recognizerName = properties.getProperty(name + ".name");
331                 if (recognizerName == null) {
332                     throw new NullPointerException("No name for recognizer " + name);
333                 }
334     
335                 // cofnig file
336                 String configFile = properties.getProperty(name + ".configFile");
337                 if (configFile == null) {
338                     throw new NullPointerException("No config file for recognizer " + name);
339                 }
340     
341                 // transcript file
342                 String testFile = properties.getProperty(name + ".testFile");
343     
344     
345                 // random order
346                 String randomOrder = properties.getProperty(name + ".randomOrder");
347                 boolean randomReference = Boolean.parseBoolean(randomOrder);
348                 // add the name of the decoder to the recognizerNameList
349                 recognizerNameList.addElement(recognizerName);
350     
351                 recognizers.put(recognizerName, new LiveRecognizer(recognizerName,
352                     configFile, testFile, randomReference));
353                 info(".");
354             }
355     
356             info("done parsing " + decoderListFile + '\n');
357         }

360         /**
361          * Updates the LiveFrame with the statistics in the given NISTAlign.
362          * 
363          * @param aligner
364          *                the NISTAlign to get statistics from
365          */
366         private void updateLiveFrame(NISTAlign aligner) {
367             liveFrame.setRecognitionLabel(aligner.getHypothesis());
368     
369             float wordAccuracy = (aligner.getTotalWordAccuracy() * 100);
370             liveFrame.setWordAccuracyLabel(wordAccuracy + "%");
371     
372             float sentenceAccuracy = (aligner.getTotalSentenceAccuracy() * 100);
373             liveFrame.setSentenceAccuracyLabel(sentenceAccuracy + "%");
374     
375             String speedLabel = "N/A";
376             String cumulativeSpeedLabel = "N/A";
377     
378             speedLabel = (timeFormat.format(currentRecognizer.getSpeed()) + " X RT");
379             cumulativeSpeedLabel = (timeFormat.format(currentRecognizer.getCumulativeSpeed()) + " X RT");
380     
381             
382             liveFrame.setSpeedLabel(speedLabel);
383             liveFrame.setCumulativeSpeedLabel(cumulativeSpeedLabel);
384         }

397         /**
398          * Does decoding in a separate thread so that it does not block the calling
399          * thread. It will automatically update the GUI components once the
400          * decoding is completed. This is analogous to the "Control" components in
401          * the MVC model.
402          */
403         class DecodingThread extends Thread {
404     
405             /**
406              * Constructs a DecodingThread.
407              */
408             public DecodingThread() {
409                 super("Decoding");
410             }
411     
412             /**
413              * Implements the run() method of this thread.
414              */
415             public void run() {
416                 Microphone microphone = currentRecognizer.getMicrophone();
417                 Recognizer recognizer = currentRecognizer.getRecognizer();
418                 
419                 if (handsFree) {
420                     while (microphone.hasMoreData()) {
421                         // just sleep for 500ms so that it won't appear to
422                         // be flipping through so quickly
423                         try {
424                             Thread.sleep(500);
425                         } catch (InterruptedException ie) {
426                             ie.printStackTrace();
427                         }
428                         String nextReference = getNextReference();
429                         liveFrame.setReferenceLabel(nextReference);
430                         liveFrame.setRecognitionLabel("");
431                         recognizer.recognize(nextReference);
432                     }
433                     liveFrame.setDecoderComboBoxEnabled(true);
434                 } else {
435                     lastResult = recognizer.recognize(liveFrame.getReference());
436     
437                     if (epMode) {
438                         microphone.stopRecording();
439                         liveFrame.setMessage("Speaker turned off.");
440                     }
441     
442                 }
443                 liveFrame.setSpeakButtonEnabled(true);
444                 liveFrame.setStopButtonEnabled(false);
445                 liveFrame.setNextButtonEnabled(true);
446                 liveFrame.setPlayButtonEnabled(true);
447                 liveFrame.setDecoderComboBoxEnabled(true);
448             }
449         }

459         /**
460          * A recognizer container, associates a recognizer name, config file
461          * test file utterance, along with the statistics and microphone control
462          */
463         class LiveRecognizer {
464             private String name;
465             private String testFile;
466             private String configName;
467             private ConfigurationManager cm;
468             private Recognizer recognizer;
469             private Microphone microphone;
470             private SpeedTracker speedTracker;
471             private NISTAlign aligner;
472             private boolean allocated;
473             private List<String> referenceList;
474             private Iterator<String> iterator;
475             private boolean randomReferenceOrder;
476     
477             /**
478              * Creates a new live recognizer
479              * @param name the name of the recognizer
480              * @param configName the config file for the recognizer
481              * @param testFile the test utterance file
482              * @param randomReferenceOrder if true reference text is
483              * presented in random order.
484              */
485             LiveRecognizer(String name, String configName, String testFile,
486                     boolean randomReferenceOrder)  {
487                 this.name = name;
488                 this.testFile = testFile;
489                 this.configName = configName;
490                 this.randomReferenceOrder = randomReferenceOrder;
491                 allocated = false;
492             }
493     
494             /**
495              * Gets the name of this recognizer
496              * @return the name of this recognizer
497              */
498             String getName() {
499                 return name;
500             }
501     
502             /**
503              * Gets the test file in use by this recognizer
504              * @return the test file
505              */
506             String getTestFile() {
507                 return testFile;
508             }
509     
510             /**
511              * Allocates this recognizer
512              * @return
513              */
514             boolean allocate() {
515                 try {
516                     if (!allocated) {
517                         URL url = new File(configName).toURI().toURL();
518                         cm = new ConfigurationManager(url);
519                         recognizer = (Recognizer) cm.lookup("recognizer");
520                         microphone = (Microphone) cm.lookup("microphone");
521                         speedTracker = (SpeedTracker) cm.lookup("speedTracker");
522                         aligner = ((AccuracyTracker) cm.lookup("accuracyTracker")).getAligner();
523                         recognizer.allocate();
524                         setTestFile(testFile);
525     
526                         recognizer.addResultListener(new ResultListener() {
527                                 public void newResult(Result result) {
528                                     if (!result.isFinal() && 
529                                         showPartialResults) {
530                                         showPartialResult(result);
531                                     }
532                                     if (result.isFinal()) {
533                                         updateLiveFrame
534                                             (currentRecognizer.getAligner());
535                                         System.out.println
536                                             (result.getBestToken().getWordUnitPath());
537                                     }
538                                 }
539     							public void newProperties(PropertySheet ps)
540     									throws PropertyException {
541     								return;
542     							}
543                             });
544                         allocated = true;
545                     }
546     
547                 } catch (PropertyException pe) {
548                     warn("Can't configure recognizer " + pe);
549                 } catch (IOException ioe) {
550                     warn("Can't allocate recognizer " + ioe);
551                 }
552                 return allocated;
553             }
554             
555             
556             /**
557              * Deallocates this recognizer
558              *
559              */
560             void deallocate() {
561                 if (allocated) {
562                     recognizer.deallocate();
563                     allocated = false;
564                 }
565             }

628             /**
629              * Returns the next utterance in the test file. If its at the last
630              * utterance already, it will cycle back to the first utterance.
631              * If there is no utterance in the file at all, it will return 
632              * an empty string.
633              *
634              * @return the next utterance in the test file; if no utterance,
635              *    it will return an empty string
636              */
637             public String getNextReference() {
638                 String next ="";
639                 if (randomReferenceOrder) {
640                     int index = (int) (Math.random() * referenceList.size());
641                     next = referenceList.get(index);
642                 } else {
643                     if (iterator == null || !iterator.hasNext()) {
644                         iterator = referenceList.listIterator();
645                     }
646                     if (iterator.hasNext()) {
647                         next = iterator.next();
648                     }
649                 }
650                 if (next == null) {
651                     next = "";
652                 }
653                 return next;
654             }
655             
656             /**
657              * Sets the file of test utterances to be the given file
658              * @param testFile the name of the test file
659              */
660             void setTestFile(String testFile)  {
661                 try {
662                     this.testFile = testFile;
663                     referenceList = new ArrayList<String>();
664                     BufferedReader reader = new BufferedReader(new FileReader(testFile));
665                     String line = null;
666                     while ((line = reader.readLine()) != null) {
667                         referenceList.add(line);
668                     }
669                     iterator = referenceList.listIterator();
670                     reader.close();
671                 } catch (FileNotFoundException e) {
672                     warn("Can't find  file " + e);
673                 } catch (IOException e) {
674                     warn("Can't read  file " + e);
675                 }
676             }
677         }
