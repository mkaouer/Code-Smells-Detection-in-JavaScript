13     package edu.cmu.sphinx.decoder.search;
14     
15     
16     /** A stack of tokens at a SentenceHMMState, allowing more than one path through the state. */
17     public interface TokenStack {
18     
19     
20         /**
21          * Adds the given Token to this TokenStack.
22          *
23          * @param token
24          * @return the replaced Token, or null if no Token was replaced
25          */
26         public Token add(Token token);
27     
28     
29         /**
30          * Returns true if the given score is higher than the lowest scoring token in this TokenStack.
31          *
32          * @param score the score the check
33          * @param frameNumber
34          * @return true if the given score is higher than the lowest scoring token, false otherwise
35          */
36         public boolean isInsertable(float score, int frameNumber);
37     
38     
39         /**
40          * Returns the capacity of this TokenStack, that is, the maximum number of elements allowed in this TokenStack.
41          *
42          * @return the capacity of this TokenStack
43          */
44         public int getCapacity();
45     
46     
47         /** Removes all of the Tokens from this TokenStack */
48         public void clear();
49     }
