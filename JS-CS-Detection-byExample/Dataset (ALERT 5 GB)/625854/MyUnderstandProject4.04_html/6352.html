27     /**
28      * Constructs a loop of all the context-independent phones. This loop is used in the static flat linguist for detecting
29      * out-of-grammar utterances. A 'phoneInsertionProbability' will be added to the score each time a new phone is entered.
30      * To obtain the all-phone search graph loop, simply called the method {@link #getSearchGraph() getSearchGraph}.
31      * <p/>
32      * For futher details of this approach cf. 'Modeling Out-of-vocabulary Words for Robust Speech Recognition', Brazzi,
33      * 2000, Proc. ICSLP
34      */
35     public class CIPhoneLoop {
36     
37         public final AcousticModel model;
38         private final float logPhoneInsertionProbability;
39         public final float logOne = LogMath.getLogOne();
40     
41     
42         /**
43          * Creates the CIPhoneLoop with the given acoustic model and phone insertion probability
44          *
45          * @param model                        the acoustic model
46          * @param logPhoneInsertionProbability the insertion probability
47          */
48         public CIPhoneLoop(AcousticModel model,
49                            float logPhoneInsertionProbability) {
50             this.model = model;
51             this.logPhoneInsertionProbability =
52                     logPhoneInsertionProbability;
53         }
54     
55     
56         /**
57          * Creates a new loop of all the context-independent phones.
58          *
59          * @return the phone loop search graph
60          */
61         public SearchGraph getSearchGraph() {
62             return new PhoneLoopSearchGraph();
63         }
64     
65     
66         protected class PhoneLoopSearchGraph implements SearchGraph {
67     
68             protected final Map<String, SearchState> existingStates;
69             protected final SentenceHMMState firstState;
70     
71     
72             /** Constructs a phone loop search graph. */
73             public PhoneLoopSearchGraph() {
74                 existingStates = new HashMap<String, SearchState>();
75                 firstState = new UnknownWordState();
76                 SentenceHMMState branchState = new BranchOutState(firstState);
77                 attachState(firstState, branchState, logOne, logOne, logOne);
78     
79                 SentenceHMMState lastState = new LoopBackState(firstState);
80                 lastState.setFinalState(true);
81                 attachState(lastState, branchState, logOne, logOne, logOne);
82     
83                 for (Iterator<Unit> i = model.getContextIndependentUnitIterator(); i.hasNext();) {
84                     UnitState unitState = new UnitState(i.next(), HMMPosition.UNDEFINED);
85     
86                     // attach unit state to the branch out state
87                     attachState(branchState, unitState, logOne, logOne,
88                             logPhoneInsertionProbability);
89     
90                     HMM hmm = model.lookupNearestHMM
91                             (unitState.getUnit(), unitState.getPosition(), false);
92                     HMMState initialState = hmm.getInitialState();
93                     HMMStateState hmmTree = new HMMStateState(unitState, initialState);
94                     addStateToCache(hmmTree);
95     
96                     // attach first HMM state to the unit state
97                     attachState(unitState, hmmTree, logOne, logOne, logOne);
98     
99                     // expand the HMM tree
100                     HMMStateState finalState = expandHMMTree(unitState, hmmTree);
101     
102                     // attach final state of HMM tree to the loopback state
103                     attachState(finalState, lastState, logOne, logOne, logOne);
104                 }
105             }
106     
107     
108             /**
109              * Retrieves initial search state
110              *
111              * @return the set of initial search state
112              */
113             @Override
114             public SearchState getInitialState() {
115                 return firstState;
116             }
117     
118     
119             /**
120              * Returns the number of different state types maintained in the search graph
121              *
122              * @return the number of different state types
123              */
124             @Override
125             public int getNumStateOrder() {
126                 return 5;
127             }
128     
129     
130             /**
131              * Checks to see if a state that matches the given state already exists
132              *
133              * @param state the state to check
134              * @return true if a state with an identical signature already exists.
135              */
136             private SentenceHMMState getExistingState(SentenceHMMState state) {
137                 return (SentenceHMMState) existingStates.get(state.getSignature());
138             }
139     
140     
141             /**
142              * Adds the given state to the cache of states
143              *
144              * @param state the state to add
145              */
146             protected void addStateToCache(SentenceHMMState state) {
147                 existingStates.put(state.getSignature(), state);
148             }
149     
150     
151             /**
152              * Expands the given hmm state tree
153              *
154              * @param parent the parent of the tree
155              * @param tree   the tree to expand
156              * @return the final state in the tree
157              */
158             protected HMMStateState expandHMMTree(UnitState parent,
159                                                 HMMStateState tree) {
160                 HMMStateState retState = tree;
161                 for (HMMStateArc arc : tree.getHMMState().getSuccessors()) {
162                     HMMStateState newState;
163                     if (arc.getHMMState().isEmitting()) {
164                         newState = new HMMStateState
165                             (parent, arc.getHMMState());
166                     } else {
167                         newState = new NonEmittingHMMState
168                             (parent, arc.getHMMState());
169                     }
170                     SentenceHMMState existingState = getExistingState(newState);
171                     float logProb = arc.getLogProbability();
172                     if (existingState != null) {
173                         attachState(tree, existingState, logProb, logOne, logOne);
174                     } else {
175                         attachState(tree, newState, logProb, logOne, logOne);
176                         addStateToCache(newState);
177                         retState = expandHMMTree(parent, newState);
178                     }
179                 }
180                 return retState;
181             }
182     
183     
184             protected void attachState(SentenceHMMState prevState,
185                                        SentenceHMMState nextState,
186                                        float logAcousticProbability,
187                                        float logLanguageProbability,
188                                        float logInsertionProbability) {
189                 SentenceHMMStateArc arc = new SentenceHMMStateArc
190                         (nextState,
191                                 logAcousticProbability,
192                                 logLanguageProbability,
193                                 logInsertionProbability);
194                 prevState.connect(arc);
195             }
196         }
197     }

199     class UnknownWordState extends SentenceHMMState implements WordSearchState {
200     
201         @Override
202         public Pronunciation getPronunciation() {
203             return Word.UNKNOWN.getPronunciations()[0];
204         }
205     
206     
207         @Override
208         public int getOrder() {
209             return 0;
210         }
211     
212     
213         @Override
214         public String getName() {
215             return "UnknownWordState";
216         }
217     
218     
219         /**
220          * Returns true if this UnknownWordState indicates the start of a word. Returns false if this UnknownWordState
221          * indicates the end of a word.
222          *
223          * @return true if this UnknownWordState indicates the start of a word, false if this UnknownWordState indicates the
224          *         end of a word
225          */
226         @Override
227         public boolean isWordStart() {
228             return true;
229         }
230     }
