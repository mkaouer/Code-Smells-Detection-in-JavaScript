13     package edu.cmu.sphinx.decoder.search;
14     
15     import edu.cmu.sphinx.util.props.PropertyException;
16     import edu.cmu.sphinx.util.props.PropertySheet;
17     import edu.cmu.sphinx.util.props.S4Boolean;
18     import edu.cmu.sphinx.util.props.S4ComponentList;
19     
20     import java.util.Iterator;
21     import java.util.List;
22     import java.util.NoSuchElementException;
23     import java.util.logging.Logger;
24     
25     /**
26      * A list of ActiveLists. Different token types are placed in different lists.
27      * <p/>
28      * This class is not thread safe and should only be used by a single thread.
29      */
30     public class SimpleActiveListManager implements ActiveListManager {
31     
32         /**
33          * This property is used in the Iterator returned by the getNonEmittingListIterator() method. When the
34          * Iterator.next() method is called, this property determines whether the lists prior to that returned by next() are
35          * empty (they should be empty). If they are not empty, an Error will be thrown.
36          */
37         @S4Boolean(defaultValue = false)
38         public static final String PROP_CHECK_PRIOR_LISTS_EMPTY = "checkPriorListsEmpty";
39         
40         /** The property that defines the name of the active list factory to be used by this search manager. */
41         @S4ComponentList(type = ActiveListFactory.class)
42         public final static String PROP_ACTIVE_LIST_FACTORIES = "activeListFactories";
43     
44         // --------------------------------------
45         // Configuration data
46         // --------------------------------------
47         private Logger logger;
48         private boolean checkPriorLists;
49         private List<ActiveListFactory> activeListFactories;
50         private ActiveList[] currentActiveLists;
51     
52     
53         /**
54          * 
55          * @param activeListFactories
56          * @param checkPriorLists
57          */
58         public SimpleActiveListManager(List<ActiveListFactory> activeListFactories, boolean checkPriorLists) {
59             this.logger = Logger.getLogger( getClass().getName() );
60     
61             this.activeListFactories = activeListFactories;
62             this.checkPriorLists = checkPriorLists;
63         }
64     
65         public SimpleActiveListManager() {
66             
67         }
68     
69         /*
70         * (non-Javadoc)
71         *
72         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
73         */
74         @Override
75         public void newProperties(PropertySheet ps) throws PropertyException {
76             logger = ps.getLogger();
77     
78             activeListFactories = ps.getComponentList(PROP_ACTIVE_LIST_FACTORIES, ActiveListFactory.class);
79             checkPriorLists = ps.getBoolean(PROP_CHECK_PRIOR_LISTS_EMPTY);
80         }
81     
82     
83         /*
84         * (non-Javadoc)
85         *
86         * @see edu.cmu.sphinx.decoder.search.ActiveListManager#setNumStateOrder(java.lang.Class[])
87         */
88         @Override
89         public void setNumStateOrder(int numStateOrder) {
90             // check to make sure that we have the correct
91             // number of active list factories for the given searc states
92             currentActiveLists = new ActiveList[numStateOrder];
93     
94             if (activeListFactories.isEmpty()) {
95                 logger.severe("No active list factories configured");
96                 throw new Error("No active list factories configured");
97             }
98             if (activeListFactories.size() != currentActiveLists.length) {
99                 logger.warning("Need " + currentActiveLists.length +
100                         " active list factories, found " +
101                         activeListFactories.size());
102             }
103             createActiveLists();
104         }
105     
106     
107         /**
108          * Creates the emitting and non-emitting active lists. When creating the non-emitting active lists, we will look at
109          * their respective beam widths (eg, word beam, unit beam, state beam).
110          */
111         private void createActiveLists() {
112             int nlists = activeListFactories.size();
113             for (int i = 0; i < currentActiveLists.length; i++) {
114                 int which = i;
115                 if (which >= nlists) {
116                     which = nlists - 1;
117                 }
118                 ActiveListFactory alf = activeListFactories.get(which);
119                 currentActiveLists[i] = alf.newInstance();
120             }
121         }
122     
123     
124         /**
125          * Adds the given token to the list
126          *
127          * @param token the token to add
128          */
129         @Override
130         public void add(Token token) {
131             ActiveList activeList = findListFor(token);
132             if (activeList == null) {
133                 throw new Error("Cannot find ActiveList for "
134                         + token.getSearchState().getClass());
135             }
136             activeList.add(token);
137         }
138     
139     
140         /**
141          * Given a token find the active list associated with the token type
142          *
143          * @param token
144          * @return the active list
145          */
146         private ActiveList findListFor(Token token) {
147             return currentActiveLists[token.getSearchState().getOrder()];
148         }
149     
150     
151         /**
152          * Replaces an old token with a new token
153          *
154          * @param oldToken the token to replace (or null in which case, replace works like add).
155          * @param newToken the new token to be placed in the list.
156          */
157         @Override
158         public void replace(Token oldToken, Token newToken) {
159             ActiveList activeList = findListFor(oldToken);
160             assert activeList != null;
161             activeList.replace(oldToken, newToken);
162         }
163     
164     
165         /**
166          * Returns the emitting ActiveList, and removes it from this manager.
167          *
168          * @return the emitting ActiveList
169          */
170         @Override
171         public ActiveList getEmittingList() {
172             ActiveList list = currentActiveLists[currentActiveLists.length - 1];
173             currentActiveLists[currentActiveLists.length - 1] = list.newInstance();
174             return list;
175         }
176     
177     
178         /**
179          * Returns an Iterator of all the non-emitting ActiveLists. The iteration order is the same as the search state
180          * order.
181          *
182          * @return an Iterator of non-emitting ActiveLists
183          */
184         @Override
185         public Iterator<ActiveList> getNonEmittingListIterator() {
186             return (new NonEmittingListIterator());
187         }
188     
189     
190         private class NonEmittingListIterator implements Iterator<ActiveList> {
191     
192             private int listPtr;
193     
194     
195             public NonEmittingListIterator() {
196                 listPtr = -1;
197             }
198     
199     
200             @Override
201             public boolean hasNext() {
202                 return listPtr + 1 < currentActiveLists.length - 1;
203             }
204     
205     
206             @Override
207             public ActiveList next() throws NoSuchElementException {
208                 listPtr++;
209     
210                 if (listPtr >= currentActiveLists.length) {
211                     throw new NoSuchElementException();
212                 }
213                 if (checkPriorLists) {
214                     checkPriorLists();
215                 }
216                 return currentActiveLists[listPtr];
217             }
218     
219     
220             /** Check that all lists prior to listPtr is empty. */
221             private void checkPriorLists() {
222                 for (int i = 0; i < listPtr; i++) {
223                     ActiveList activeList = currentActiveLists[i];
224                     if (activeList.size() > 0) {
225                         throw new Error("At while processing state order"
226                                 + listPtr + ", state order " + i + " not empty");
227                     }
228                 }
229             }
230     
231     
232             @Override
233             public void remove() {
234                 currentActiveLists[listPtr] =
235                         currentActiveLists[listPtr].newInstance();
236             }
237         }
238     
239     
240         /** Outputs debugging info for this list manager */
241         @Override
242         public void dump() {
243             for (ActiveList al : currentActiveLists) {
244                 dumpList(al);
245             }
246         }
247     
248     
249         /**
250          * Dumps out debugging info for the given active list
251          *
252          * @param al the active list to dump
253          */
254         private void dumpList(ActiveList al) {
255             System.out.println("GBT " + al.getBestToken() + " size: " + al.size());
256         }
257     
258     }
