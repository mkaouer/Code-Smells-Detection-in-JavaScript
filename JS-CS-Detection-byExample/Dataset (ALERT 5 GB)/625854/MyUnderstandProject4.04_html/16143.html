8     package com.sun.speech.engine;
9     
10     import java.util.EventObject;
11     
12     import javax.speech.SpeechEvent;
13     
14     /**
15      * Wraps an arbitrary event object (from <code>EventObject</code>) in a <code>SpeechEvent</code> so that it can be
16      * dispatched through the speech event dispatch mechanism. One use of this is in the <code>BaseEngineProperties</code>
17      * class that needs to wrap and issue <code>PropertyChangeEvents</code>.
18      *
19      * @see SpeechEventUtilities
20      * @see EventObject
21      */
22     @SuppressWarnings("serial")
23     public class SpeechEventWrapper extends SpeechEvent {
24     
25         /** Use an id that won't be confused with JSAPI event ids. */
26         protected final static int WRAPPER_ID = -25468;
27     
28         /** The wrapped event. */
29         protected EventObject eventObject;
30     
31     
32         /**
33          * Class constructor.
34          *
35          * @param e the <code>EventObject</code> to wrap.
36          */
37         public SpeechEventWrapper(EventObject e) {
38             super(e.getSource(), WRAPPER_ID);
39             eventObject = e;
40         }
41     
42     
43         /**
44          * Gets the wrapped event.
45          *
46          * @return the event that was passed to the constructor
47          */
48         public EventObject getEventObject() {
49             return eventObject;
50         }
51     }
