13     package edu.cmu.sphinx.linguist.acoustic.tiedstate;
14     
15     import edu.cmu.sphinx.linguist.acoustic.HMM;
16     import edu.cmu.sphinx.linguist.acoustic.HMMPosition;
17     import edu.cmu.sphinx.linguist.acoustic.Unit;
18     
19     import java.util.*;
20     import java.util.logging.Logger;
21     
22     
23     /**
24      * Manages HMMs. This HMMManager groups {@link edu.cmu.sphinx.linguist.acoustic.HMM HMMs} together by their {@link
25      * edu.cmu.sphinx.linguist.acoustic.HMMPosition position} with the word.
26      */
27     public class HMMManager implements Iterable<HMM> {
28     
29         private final List<HMM> allHMMs = new ArrayList<HMM>();
30         private final Map<HMMPosition, Map<Unit, HMM>> hmmsPerPosition = new EnumMap<HMMPosition, Map<Unit, HMM>>(HMMPosition.class);
31     
32         public HMMManager () {
33             for (HMMPosition pos : HMMPosition.values())
34                 hmmsPerPosition.put(pos, new HashMap<Unit, HMM>());
35         }
36     
37         /**
38          * Put an HMM into this manager
39          *
40          * @param hmm the hmm to manage
41          */
42         public void put(HMM hmm) {
43             hmmsPerPosition.get(hmm.getPosition()).put(hmm.getUnit(), hmm);
44             allHMMs.add(hmm);
45         }
46     
47     
48         /**
49          * Retrieves an HMM by position and unit
50          *
51          * @param position the position of the HMM
52          * @param unit     the unit that this HMM represents
53          * @return the HMM for the unit at the given position or null if no HMM at the position could be found
54          */
55         public HMM get(HMMPosition position, Unit unit) {
56             return hmmsPerPosition.get(position).get(unit);
57         }
58     
59     
60         /**
61          * Gets an iterator that iterates through all HMMs
62          *
63          * @return an iterator that iterates through all HMMs
64          */
65         @Override
66         public Iterator<HMM> iterator() {
67             return allHMMs.iterator();
68         }
69     
70     
71         /**
72          * Returns the number of HMMS in this manager
73          *
74          * @return the number of HMMs
75          */
76         private int getNumHMMs() {
77             int count = 0;
78     
79             for (Map<Unit, HMM> map : hmmsPerPosition.values()) {
80                 if (map != null) {
81                     count += map.size();
82                 }
83             }
84             return count;
85         }
86     
87     
88         /**
89          * Log information about this manager
90          *
91          * @param logger logger to use for this logInfo
92          */
93         public void logInfo(Logger logger) {
94             logger.info("HMM Manager: " + getNumHMMs() + " hmms");
95         }
96     }
