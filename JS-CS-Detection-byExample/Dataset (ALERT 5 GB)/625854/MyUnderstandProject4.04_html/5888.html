106         @Override
107         public Data getData() throws DataProcessingException {
108             Data data = getPredecessor().getData();
109     
110             if (data instanceof DataStartSignal)
111                 sampleRate = ((DataStartSignal) data).getSampleRate();
112     
113             if (data instanceof DataStartSignal || (data instanceof SpeechStartSignal && captureUtts)) {
114                 baos = new ByteArrayOutputStream();
115                 dos = new DataOutputStream(baos);
116             }
117     
118     
119             if ((data instanceof DataEndSignal && !captureUtts) || (data instanceof SpeechEndSignal && captureUtts)) {
120             	
121                 String wavName;
122                 if (isCompletePath)
123                     wavName = dumpFilePath;
124                 else
125                     wavName = dumpFilePath + getNextFreeIndex(dumpFilePath) + ".wav";
126     
127                 writeFile(wavName);
128     
129                 isInSpeech = false;
130             }
131     
132             if (data instanceof SpeechStartSignal)
133                 isInSpeech = true;
134     
135             if ((data instanceof DoubleData || data instanceof FloatData) && (isInSpeech || !captureUtts)) {
136                 DoubleData dd = data instanceof DoubleData ? (DoubleData) data : DataUtil.FloatData2DoubleData((FloatData) data);
137                 double[] values = dd.getValues();
138     
139                 for (double value : values) {
140                     try {
141                         dos.writeShort(new Short((short) value));
142                     } catch (IOException e) {
143                         e.printStackTrace();
144                     }
145                 }
146             }
147     
148             return data;
149         }

171         private static AudioFileFormat.Type getTargetType(String extension) {
172             AudioFileFormat.Type[] typesSupported = AudioSystem.getAudioFileTypes();
173     
174             for (AudioFileFormat.Type aTypesSupported : typesSupported) {
175                 if (aTypesSupported.getExtension().equals(extension)) {
176                     return aTypesSupported;
177                 }
178             }
179     
180             return null;
181         }

184         /**
185          * Converts a big-endian byte array into an array of doubles. Each consecutive bytes in the byte array are converted
186          * into a double, and becomes the next element in the double array. The size of the returned array is
187          * (length/bytesPerValue). Currently, only 1 byte (8-bit) or 2 bytes (16-bit) samples are supported.
188          *
189          * @param values
190          * @param bytesPerValue the number of bytes per value
191          * @param signedData    whether the data is signed
192          * @return a double array, or <code>null</code> if byteArray is of zero length
193          * @throws ArrayIndexOutOfBoundsException
194          */
195         public static byte[] valuesToBytes(double[] values, int bytesPerValue, boolean signedData)
196                 throws ArrayIndexOutOfBoundsException {
197     
198             byte[] byteArray = new byte[bytesPerValue * values.length];
199     
200             int byteArInd = 0;
201     
202             for (double value : values) {
203                 int val = (int) value;
204     
205     
206                 for (int j = bytesPerValue - 1; j >= 0; j++) {
207                     byteArray[byteArInd + j] = (byte) (val & 0xff);
208                     val = val >> 8;
209                 }
210     
211                 byteArInd += bytesPerValue;
212             }
213     
214             return byteArray;
215         }

218         public static AudioInputStream convertDoublesToAudioStream(double[] values, int sampleRate) {
219             ByteArrayOutputStream baos = new ByteArrayOutputStream();
220             DataOutputStream dos = new DataOutputStream(baos);
221     
222             for (double value : values) {
223                 try {
224                     dos.writeShort(new Short((short) value));
225                 } catch (IOException e) {
226                     e.printStackTrace();
227                 }
228             }
229     
230             AudioFormat wavFormat = new AudioFormat(sampleRate, 16, 1, true, true);
231             byte[] abAudioData = baos.toByteArray();
232             ByteArrayInputStream bais = new ByteArrayInputStream(abAudioData);
233     
234             return new AudioInputStream(bais, wavFormat, abAudioData.length / wavFormat.getFrameSize());
235         }

242         public static void writeWavFile(double[] signal, int sampleRate, File targetFile) {
243             AudioInputStream ais = WavWriter.convertDoublesToAudioStream(signal, sampleRate);
244             AudioFileFormat.Type outputType = getTargetType("wav");

253         /**
254         * Writes the current stream to disc; override this method if you want to take 
255         * additional action on file writes
256         *
257         * @param wavName name of the file to be written
258         */
259         protected void writeFile(String wavName) {
260             AudioFormat wavFormat = new AudioFormat(sampleRate, bitsPerSample, 1, isSigned, true);
261             AudioFileFormat.Type outputType = getTargetType("wav");
262     
263             byte[] abAudioData = baos.toByteArray();
264             ByteArrayInputStream bais = new ByteArrayInputStream(abAudioData);
265             AudioInputStream ais = new AudioInputStream(bais, wavFormat, abAudioData.length / wavFormat.getFrameSize());
266     
267             File outWavFile = new File(wavName);
268     
269             if (AudioSystem.isFileTypeSupported(outputType, ais)) {
270                 try {
271                     AudioSystem.write(ais, outputType, outWavFile);
272                 } catch (IOException e) {
273                     e.printStackTrace();
274                 }
275             }   	
276         }
