30     /**
31      * This class is used mainly to facilitate scanning all the folders of sphinx system, and to filter classes that are
32      * Configurable (implements 'edu.cmu.sphinx.util.props.Configurable' in one of the ancestors) and are not of Interface
33      * type.
34      * <p/>
35      * This class is used by ModelBuilder, and contains only static methods
36      *
37      * @author Ariani
38      */
39     public class ClassFinder {
40     
41         private static final String COMMON_SPHINX_PACKAGE = "edu";
42     
43         private static final Logger logger = Logger.getLogger("ClassFinder");
44         
45         /**
46          * find a File, open it and return the content as BufferedReader
47          *
48          * @param filename name of file to be found
49          * @return BufferedReader with file content
50          */
51         public static BufferedReader findFile(String filename) throws
52                 ClassNotFoundException, FileNotFoundException {
53             File file = getStartingDir(filename);
54             BufferedReader br = new BufferedReader(new FileReader(file));
55             return br;
56         }

59         /**
60          * find a list of folder names that are directly under 'startDir', not recursive
61          *
62          * @param startDir String name of directory to start the search from
63          * @return list of folders under 'startDir'
64          * @throws ClassNotFoundException the class we're looking for is not found
65          * @throws FileNotFoundException  the folder does not exist
66          */
67         public static List<String> findFolder(String startDir) throws
68                 ClassNotFoundException, FileNotFoundException {
69             List<String> folderList = new ArrayList<String>();
70     
71             File tempDir = getStartingDir(startDir);
72             validateDirectory(tempDir);
73             File[] myFilesDirs = tempDir.listFiles();
74     
75             for (File file : myFilesDirs) {
76                 if (file.isDirectory()) {
77                     // if it's directory, add the name to result list
78                     folderList.add(file.getName());
79                 }
80             }
81             return folderList;        // list of String folderName 
82         }

85         /**
86          * find a list of java classes inside 'startDir', that meets the ancestor and fields requirement for the model
87          * recursively goes down to the bottom level
88          *
89          * @param startDir     String name directory to list
90          * @param startPackage String preceding package names
91          * @param classList    Set of .class files e.g. "nextDir.nextclass", that passes the filter, and will be in the
92          *                     model startDir and '.class' extension not included as package name
93          * @throws ClassNotFoundException, FileNotFoundException
94          */
95         public static void findClasses(String startDir, String startPackage, Set<Class<?>> classList)
96                 throws ClassNotFoundException, FileNotFoundException {
97             File tempDir = getStartingDir(startDir);
98             validateDirectory(tempDir);
99     
100             File[] myFilesDirs = tempDir.listFiles();
101             String tempFile = null;
102     
103             for (File file : myFilesDirs) {
104                 // go through each file and directory
105                 tempFile = file.getName();
106                 if (file.isDirectory()) {
107                     // if it's directory, go deeper recursively
108                     findClasses(startDir + '/' + tempFile, startPackage + '.' + tempFile, classList);
109                 } else if (tempFile.endsWith(".class")) {
110                     // removes the .class extension
111                     // if it's a file, check if it's a java .class file
112                     logger.log(Level.FINE, "Checking class" + tempFile);
113                     validateFile(file);
114                     try {
115                         String classname = startPackage + '.' + tempFile.substring(0, tempFile.length() - 6);
116                         Class<?> addclass = Class.forName(classname);
117                         if (filterClass(addclass)) {
118                             // if it's an interface, shouldn't display because interface cannot be instantiated
119                             if (!addclass.isInterface() && !Modifier.isAbstract(addclass.getModifiers())) {
120                                 classList.add(addclass);
121                             	logger.log(Level.FINE, "Adding class" + tempFile);
122                             }
123                         }
124     
125                     } catch (NoClassDefFoundError e) {
126                         System.err.flush();
127                         System.err.println("error loading " + tempFile + " in " + startPackage);
128                         System.err.flush();
129                     }
130                 }
131             }
132         }

135         /**
136          * check this class, if one of its ancestor classes or interfaces implements 'edu.cmu.sphinx.util.props.Configurable'
137          *
138          * @param c class to be examined
139          * @return boolean true if the class passes the filter
140          */
141         private static boolean filterClass(Class<?> c) {
142     
143             String interfaceName;
144             Class<?> superclass;
145     
146             //check the implemented interfaces
147             Class<?>[] theInterfaces = c.getInterfaces();
148     
149             for (Class<?> cls : theInterfaces) {
150                 interfaceName = cls.getName();
151                 logger.log(Level.FINE, "The class " + c.getName() + " has interface " + interfaceName);
152                 if (interfaceName.equalsIgnoreCase("edu.cmu.sphinx.util.props.Configurable")) {
153                     return true;
154                 } else if (interfaceName.startsWith(COMMON_SPHINX_PACKAGE)) {
155                     //check the implemented interface only if they starts with 'edu'
156                     //so, java library classes are not traversed
157                     if (filterClass(cls)) {
158                         return true;
159                     }
160                 }
161             }
162     
163             // check the ancestor classes
164             superclass = c.getSuperclass();
165             return superclass != null && superclass.getName().startsWith(COMMON_SPHINX_PACKAGE) && filterClass(superclass);
166         }

169         /**
170          * find directory
171          *
172          * @param name directory/file to be found
173          * @return File of directory/file if found
174          * @throws ClassNotFoundException if the dirname/file name is invalid or not found
175          */
176         private static File getStartingDir(String name) throws ClassNotFoundException {
177     
178             // Get a File object for the directory
179             File file = null;
180             try {
181     
182                 URL url = Thread.currentThread().getContextClassLoader().
183                         getResource(name.replace('.', '/'));
184                 file = new File(url.getFile());
185     
186             } catch (NullPointerException x) {
187                 throw new ClassNotFoundException(name + " cannot be found; " +
188                         "it does not appear to be a valid file/directory");
189             }
190     
191             if (file.exists()) {
192                 return file;
193             } else {
194                 throw new ClassNotFoundException(name + " does not appear to be a valid file/dir");
195             }
196         }
197     
198     
199         /** Directory is valid if it exists, does not represent a file, and can be read. */
200         static private void validateDirectory(File aDirectory) throws FileNotFoundException {
201             if (aDirectory == null) {
202                 throw new IllegalArgumentException("Directory should not be null.");
203             }
204             if (!aDirectory.exists()) {
205                 throw new FileNotFoundException("Directory does not exist: " + aDirectory);
206             }
207             if (!aDirectory.isDirectory()) {
208                 throw new IllegalArgumentException("Is not a directory: " + aDirectory);
209             }
210             if (!aDirectory.canRead()) {
211                 throw new IllegalArgumentException("Directory cannot be read: " + aDirectory);
212             }
213         }
214     
215     
216         /** Class is valid if it exists,  and can be read. */
217         static private void validateFile(File aFile) throws FileNotFoundException {
218             if (aFile == null) {
219                 throw new IllegalArgumentException("Directory should not be null.");
220             }
221             if (!aFile.exists()) {
222                 throw new FileNotFoundException("Directory does not exist: " + aFile);
223             }
224             if (!aFile.canRead()) {
225                 throw new IllegalArgumentException("Directory cannot be read: " + aFile);
226             }
227         }
