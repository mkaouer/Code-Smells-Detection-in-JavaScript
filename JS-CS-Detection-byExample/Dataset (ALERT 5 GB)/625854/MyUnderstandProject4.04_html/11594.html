13     package edu.cmu.sphinx.research.parallel;
14     
15     import edu.cmu.sphinx.decoder.search.Token;
16     
17     /** Prunes an ActiveList of ParallelTokens based on their FeatureScore. */
18     public class FeatureScorePruner extends TokenScorePruner {
19     
20         /**
21          * Returns the score that we use to compare this ParallelToken with other ParallelTokens.
22          *
23          * @param token the ParallelToken to compare
24          * @return the score we use for comparison
25          */
26         protected float getTokenScore(Token token) {
27             ParallelToken parallelToken = (ParallelToken) token;
28             return parallelToken.getFeatureScore();
29         }
30     }
