13     package edu.cmu.sphinx.tools.audio;
14     
15     import javax.sound.sampled.AudioFormat;
16     import javax.sound.sampled.AudioSystem;
17     import javax.sound.sampled.DataLine;
18     import javax.sound.sampled.SourceDataLine;
19     
20     /** Plays an AudioData in a separate thread. */
21     public class AudioPlayer extends Thread {
22     
23         private final AudioData audio;
24         private SourceDataLine line;
25         private int selectionStart;
26         private int selectionEnd;
27     
28     
29         /** Creates a new AudioPlayer for the given AudioData. */
30         public AudioPlayer(AudioData audio) {
31             this.audio = audio;
32             selectionStart = 0;
33             selectionEnd = audio.getAudioData().length;
34         }
35     
36     
37         /** Notifies the AudioPlayer thread to play the audio. */
38         public void play(int selectionStart, int selectionEnd) {
39             synchronized (audio) {
40                 this.selectionStart = selectionStart;
41                 this.selectionEnd = selectionEnd;
42                 audio.notify();
43             }
44         }
45     
46     
47         /** Plays the AudioData in a separate thread. */
48         @Override
49         public void run() {
50             while (true) {
51                 try {
52                     synchronized (audio) {
53                         audio.wait();
54                         AudioFormat format = audio.getAudioFormat();
55                         short[] data = audio.getAudioData();
56                         int start = Math.max(0, selectionStart);
57                         int end = selectionEnd;
58                         if (end == -1) {
59                             end = data.length;
60                         }
61     
62                         DataLine.Info info =
63                                 new DataLine.Info(SourceDataLine.class,
64                                         format);
65                         line = (SourceDataLine) AudioSystem.getLine(info);
66                         line.open(format);
67                         line.start();
68     
69                         byte[] frame = new byte[2];
70                         for (int i = start;
71                              i < end && i < data.length; i++) {
72                             Utils.toBytes(data[i], frame, true);
73                             line.write(frame, 0, frame.length);
74                         }
75     
76                         line.drain();
77                         line.close();
78                         line = null;
79                     }
80                 } catch (Exception e) {
81                     e.printStackTrace();
82                     break;
83                 }
84             }
85         }
86     }
