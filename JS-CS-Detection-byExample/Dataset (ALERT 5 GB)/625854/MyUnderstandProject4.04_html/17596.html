8     package edu.cmu.sphinx.tools.tags;
9     
10     import org.mozilla.javascript.Scriptable;
11     
12     /**
13      * An ObjectTagsParser is an ActionTagsParser that handles registration of Java object instances.  With this class, an
14      * application can use the put method to register a Java object instance with the parser.  Once this has been done,
15      * ECMAScript action tags can reference and manipulate that object by name.
16      *
17      * @see #put
18      */
19     public class ObjectTagsParser extends ActionTagsParser {
20     
21         /** Create a new ObjectParser. */
22         public ObjectTagsParser() {
23         }
24     
25     
26         /**
27          * Put the given object with the given name in the global namespace of the parser.  If the name already exists in
28          * the global namespace, it is replaced with the new object.
29          *
30          * @param name   the name of the object
31          * @param object the object
32          */
33         public void put(String name, Object object) {
34             put(global, name, object);
35         }
36     
37     
38         /**
39          * Put the given object with the given name in the given scope. If the name already exists, it is replaced with the
40          * new object.
41          *
42          * @param scope  the scope
43          * @param name   the name of the object
44          * @param object the object
45          */
46         protected void put(Scriptable scope, String name, Object object) {
47             scope.put(name, scope, object);
48         }
49     }
