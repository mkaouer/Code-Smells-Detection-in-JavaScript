158             if (wasLogConfigured)
159                 return;
160     
161             wasLogConfigured = true;

163             // Allow others to override the logging settings.
164             if (System.getProperty("java.util.logging.config.class") != null
165                     || System.getProperty("java.util.logging.config.file") != null) {
166                 return;
167             }
168             // apply the log level (if defined) for the root logger (because we're using package based logging now
169     
170             String cmPrefix = getLogPrefix(cm);
171             Logger cmRootLogger = Logger.getLogger(cmPrefix.substring(0, cmPrefix.length() - 1));
172     
173             // we need to determine the root-level here, because the logManager will reset it
174             Level rootLevel = Logger.getLogger("").getLevel();
175     
176             configureLogger(cmRootLogger);
177     
178     
179             String level = cm.getGlobalProperty(GLOBAL_COMMON_LOGLEVEL);
180             if (level == null)
181                 level = Level.WARNING.getName();
182     
183             cmRootLogger.setLevel(Level.parse(level));
184     
185             // restore the old root logger level
186             Logger.getLogger("").setLevel(rootLevel);
187         }

263          * <p/>

266         public static String toXML(ConfigurationManager cm) {
267             StringBuilder sb = new StringBuilder();
268             sb.append("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n");
269             sb.append("\n<!--    Sphinx-4 Configuration file--> \n\n");
270     
271             sb.append("<config>");
272     
273             Pattern pattern = Pattern.compile("\\$\\{(\\w+)\\}");
274     
275             Map<String, String> globalProps = cm.getGlobalProperties();
276             for (Map.Entry<String, String> entry : globalProps.entrySet()) {
277                 String propName = entry.getKey();
278     
279                 Matcher matcher = pattern.matcher(propName);
280                 propName = matcher.matches() ? matcher.group(1) : propName;
281     
282                 sb.append("\n\t<property name=\"").append(propName).append("\" value=\"").append(entry.getValue()).append("\"/>");
283             }
284     
285             for (String instanceName : cm.getComponentNames())
286                 sb.append("\n\n").append(propSheet2XML(instanceName, cm.getPropertySheet(instanceName)));
287     
288             sb.append("\n</config>");
289             return sb.toString();
290         }

293         private static String propSheet2XML(String instanceName, PropertySheet ps) {
294             StringBuilder sb = new StringBuilder();
295             sb.append("\t<component name=\"").append(instanceName).append("\" type=\"").append(ps.getConfigurableClass().getName()).append("\">");
296     
297             for (String propName : ps.getRegisteredProperties()) {
298                 String predec = "\n\t\t<property name=\"" + propName + "\" ";
299                 if (ps.getRawNoReplacement(propName) == null)
300                     continue;  // if the property was net defined within the xml-file
301     
302                 switch (ps.getType(propName)) {
303     
304                     case COMPONENT_LIST:
305                         sb.append("\n\t\t<propertylist name=\"").append(propName).append("\">");
306                         List<String> compNames = toStringList(ps.getRawNoReplacement(propName));
307                         for (String compName : compNames)
308                             sb.append("\n\t\t\t<item>").append(compName).append("</item>");
309                         sb.append("\n\t\t</propertylist>");
310                         break;
311                     default:
312                         sb.append(predec).append("value=\"").append(ps.getRawNoReplacement(propName)).append("\"/>");
313                 }
314             }
315     
316             sb.append("\n\t</component>\n\n");
317             return sb.toString();
318         }

387         /**
388          * Applies the system properties to the raw property map. System properties should be of the form
389          * compName[paramName]=paramValue
390          * <p/>
391          * List types cannot currently be set from system properties.
392          *
393          * @param rawMap the map of raw property values
394          * @param global global properties
395          * @throws PropertyException if an attempt is made to set a parameter for an unknown component.
396          */
397         static void applySystemProperties(Map<String, RawPropertyData> rawMap, Map<String, String> global)
398                 throws PropertyException {
399             Properties props = System.getProperties();
400             for (Enumeration<?> e = props.keys(); e.hasMoreElements();) {
401                 String param = (String) e.nextElement();
402                 String value = props.getProperty(param);
403     
404                 // search for parameters of the form component[parameter]=value
405                 // these go in the property sheet for the component
406                 int lb = param.indexOf('[');
407                 int rb = param.indexOf(']');
408     
409                 if (lb > 0 && rb > lb) {
410                     String compName = param.substring(0, lb);
411                     String paramName = param.substring(lb + 1, rb);
412                     RawPropertyData rpd = rawMap.get(compName);
413                     if (rpd != null) {
414                         rpd.add(paramName, value);
415                     } else {
416                         throw new InternalConfigurationException(compName, param,
417                                 "System property attempting to set parameter "
418                                         + " for unknown component " + compName
419                                         + " (" + param + ')');
420                     }
421                 }
422     
423                 // look for parameters of the form foo=bar
424                 // these go in the global map
425     
426                 else if (param.indexOf('.') == -1) {
427                     global.put(param, value);
428                 }
429             }
430         }

433         /**
434          * Renames a given <code>Configurable</code>. The configurable component named <code>oldName</code> is assumed to be
435          * registered to the CM. Renaming does not only affect the configurable itself but possibly global property values
436          * and properties of other components.
437          */
438         static void renameComponent(ConfigurationManager cm, String oldName, String newName) {
439             assert cm != null;
440             assert oldName != null && newName != null;
441             if (cm.getPropertySheet(oldName) == null) {
442                 throw new RuntimeException("no configurable (to be renamed) named " + oldName + " is contained in the CM");
443             }
444     
445             // this iteration is a little hacky. It would be much better to maintain the links to a configurable in a special table
446             for (String instanceName : cm.getComponentNames()) {
447                 PropertySheet propSheet = cm.getPropertySheet(instanceName);
448     
449                 for (String propName : propSheet.getRegisteredProperties()) {
450                     if (propSheet.getRawNoReplacement(propName) == null)
451                         continue;  // if the property was net defined within the xml-file
452     
453                     switch (propSheet.getType(propName)) {
454     
455                         case COMPONENT_LIST:
456                             List<String> compNames = toStringList(propSheet.getRawNoReplacement(propName));
457                             for (int i = 0; i < compNames.size(); i++) {
458                                 String compName = compNames.get(i);
459                                 if (compName.equals(oldName)) {
460                                     compNames.set(i, newName);
461                                 }
462                             }
463     
464                             break;
465                         case COMPONENT:
466                             if (propSheet.getRawNoReplacement(propName).equals(oldName)) {
467                                 propSheet.setRaw(propName, newName);
468                             }
469                     }
470                 }
471             }
472     
473             PropertySheet ps = cm.getPropertySheet(oldName);
474             ps.setInstanceName(newName);
475     
476             // it might be possible that the component is the value of a global property
477             for (Map.Entry<String, String> entry : cm.getGlobalProperties().entrySet()) {
478                 if (entry.getValue().equals(oldName))
479                     cm.setGlobalProperty(entry.getKey(), newName);
480             }
481         }

492         public static URL getResource(String name, PropertySheet ps) throws PropertyException {
493     
494             String location = ps.getString(name);
495             if (location == null) {

517         public static URL resourceToURL(String location) throws MalformedURLException, ClassNotFoundException {
518             Matcher jarMatcher = jarPattern.matcher(location);
519             if (jarMatcher.matches()) {
520                 String resourceName = jarMatcher.group(1);
521                 return ConfigurationManagerUtils.class.getResource(resourceName);
522             } else {
523                 if (location.indexOf(':') == -1) {
524                     location = "file:" + location;
525                 }
526                 return new URL(location);
527             }

542         public static boolean isImplementingInterface(Class<?> aClass, Class<?> interfaceClass) {
543             assert interfaceClass.isInterface();
544     
545             Class<?> superClass = aClass.getSuperclass();
546             if (superClass != null && isImplementingInterface(superClass, interfaceClass))
547                 return true;
548     
549             for (Class<?> curInterface : aClass.getInterfaces()) {
550                 if (curInterface.equals(interfaceClass) || isImplementingInterface(curInterface, interfaceClass))
551                     return true;
552             }
553     
554             return false;
555         }

559             while (aClass != null && !aClass.equals(Object.class)) {
560                 aClass = aClass.getSuperclass();
561     
562                 if (aClass != null && aClass.equals(possibleSuperclass))
563                     return true;
564             }
565     
566             return false;
567         }

628         /**
629          * Attempts to set the value of an arbitrary component-property. If the property-name is ambiguous  with respect to
630          * the given <code>ConfiguratioManager</code> an extended syntax (componentName->propName) can be used to access the
631          * property.
632          * <p/>
633          * Beside component properties it is also possible to modify the class of a configurable, but this is only allowed if
634          * the configurable under question has not been instantiated yet. Furthermore the user has to ensure to set all
635          * mandatory component properties.
636          */
637         public static void setProperty(ConfigurationManager cm, String propName, String propValue) {
638             assert propValue != null;
639     
640             Map<String, List<PropertySheet>> allProps = listAllsPropNames(cm);
641             Set<String> configurableNames = cm.getComponentNames();
642     
643             if (!allProps.containsKey(propName) && !propName.contains("->") && !configurableNames.contains(propName))
644                 throw new RuntimeException("No property or configurable '" + propName + "' in configuration '" + cm.getConfigURL() + "'!");
645     
646             // if a configurable-class should be modified
647             if (configurableNames.contains(propName)) {
648                 try {
649                     final Class<? extends Configurable> confClass = Class.forName(propValue).asSubclass(Configurable.class);
650                     ConfigurationManagerUtils.setClass(cm.getPropertySheet(propName), confClass);
651                 } catch (ClassNotFoundException e) {
652                     throw new RuntimeException(e);
653                 }
654     
655                 return;
656             }
657     
658             if (!propName.contains("->") && allProps.get(propName).size() > 1) {
659                 throw new RuntimeException("Property-name '" + propName + "' is ambiguous with respect to configuration '"
660                         + cm.getConfigURL() + "'. Use 'componentName->propName' to disambiguate your request.");
661             }
662     
663             String componentName;
664     
665             // if disambiguation syntax is used find the correct PS first
666             if (propName.contains("->")) {
667                 String[] splitProp = propName.split("->");
668                 componentName = splitProp[0];
669                 propName = splitProp[1];
670             } else {
671                 componentName = allProps.get(propName).get(0).getInstanceName();
672             }
673     
674             setProperty(cm, componentName, propName, propValue);
675         }

677         public static void setProperty(ConfigurationManager cm, String componentName, String propName, String propValue) {
678     
679             // now set the property
680             PropertySheet ps = cm.getPropertySheet(componentName);
681             if (ps == null)
682                 throw new RuntimeException("Component '" + propName + "' is not registered to this system configuration '");
683     
684             // set the value to null if the string content is 'null
685             if (propValue.equals("null"))
686                 propValue = null;
687     
688             switch (ps.getType(propName)) {
689                 case BOOLEAN:
690                     ps.setBoolean(propName, Boolean.valueOf(propValue));
691                     break;
692                 case DOUBLE:
693                     ps.setDouble(propName, new Double(propValue));
694                     break;
695                 case INT:
696                     ps.setInt(propName, new Integer(propValue));
697                     break;
698                 case STRING:
699                     ps.setString(propName, propValue);
700                     break;
701                 case COMPONENT:
702                     ps.setComponent(propName, propValue, null);
703                     break;
704                 case COMPONENT_LIST:
705                     List<String> compNames = new ArrayList<String>();
706                     for (String component : propValue.split(";")) {
707                         compNames.add(component.trim());
708                     }
709     
710                     ps.setComponentList(propName, compNames, null);
711                     break;
712                 default:
713                     throw new RuntimeException("unknown property-type");
714             }
715         }

729          * Returns the not yet instantiated components registered to this configuration manager.
730          */
731         public static Collection<String> getNonInstaniatedComps(ConfigurationManager cm) {
732             Collection<String> nonInstComponents = new ArrayList<String>();
733     
734             for (String compName : cm.getComponentNames()) {
735                 if (!cm.getPropertySheet(compName).isInstanciated())
736                     nonInstComponents.add(compName);
737             }
738             return nonInstComponents;
739         }

750         public static List<String> toStringList(Object obj) {
751             List<String> result = new ArrayList<String>();
752             if (!(obj instanceof List<?>))
753                 return null;
754             for (Object o : (List<?>) obj) {
755                 if (o instanceof String) {
756                     result.add((String) o);
757                 }
758             }
759             return result;
