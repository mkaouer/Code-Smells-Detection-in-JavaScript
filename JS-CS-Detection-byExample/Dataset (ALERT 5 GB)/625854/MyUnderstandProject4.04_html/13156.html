61         /**
62          * Returns the latest begin time of all nodes in the given cluster.
63          *
64          * @param cluster the cluster to examine
65          * @return the latest begin time
66          */
67         public int getLatestBeginTime(List<Node> cluster) {
68             if (cluster.isEmpty()) {
69                 return -1;
70             }
71             int startTime = 0;
72             for (Node n : cluster) {
73                 if (n.getBeginTime() > startTime) {
74                     startTime = n.getBeginTime();
75                 }
76             }
77             return startTime;
78         }

101         /**
102          * Perform one inter word clustering step of the algorithm
103          *
104          * @param clusters the current cluster set
105          */
106         protected boolean interWordClusterStep(List<Cluster> clusters) {
107             Cluster toBeMerged1 = null;
108             Cluster toBeMerged2 = null;
109             double maxSim = Double.NEGATIVE_INFINITY;
110             ListIterator<Cluster> i = clusters.listIterator();
111             while (i.hasNext()) {
112                 Cluster c1 = i.next();
113                 if (!i.hasNext()) {
114                     break;
115                 }
116                 ListIterator<Cluster> j = clusters.listIterator(i.nextIndex());
117                 while (j.hasNext()) {
118                     Cluster c2 = j.next();
119                     double sim = interClusterDistance(c1, c2);
120                     if (sim > maxSim && hasOverlap(c1, c2)) {
121                         maxSim = sim;
122                         toBeMerged1 = c1;
123                         toBeMerged2 = c2;
124                     }
125                 }
126             }
127             if (toBeMerged1 != null) {
128                 clusters.remove(toBeMerged2);
129                 toBeMerged1.add(toBeMerged2);
130                 return true;
131             }
132             return false;
133         }

136         /**
137          * Find the string edit distance between to lists of objects. 
138          * Objects are compared using .equals() 
139          * TODO: could be moved to a general utility class
140          *
141          * @param p1 the first list
142          * @param p2 the second list
143          * @return the string edit distance between the two lists
144          */
145         protected static int stringEditDistance(List<?> p1, List<?> p2) {
146             if (p1.isEmpty()) {
147                 return p2.size();
148             }
149             if (p2.isEmpty()) {
150                 return p1.size();
151             }
152             int[][] distances = new int[p1.size() + 1][p2.size() + 1];
153             for (int i = 0; i <= p1.size(); i++) {
154                 distances[i][0] = i;
155             }
156             for (int j = 0; j <= p2.size(); j++) {
157                 distances[0][j] = j;
158             }
159             for (int i = 1; i <= p1.size(); i++) {
160                 for (int j = 1; j <= p2.size(); j++) {
161                     int min = Math.min(distances[i - 1][j - 1]
162                             + (p1.get(i - 1).equals(p2.get(j - 1)) ? 0 : 1),
163                             distances[i - 1][j] + 1);
164                     min = Math.min(min, distances[i][j - 1] + 1);
165                     distances[i][j] = min;
166                 }
167             }
168             return distances[p1.size()][p2.size()];
169         }

172         /**
173          * Compute the phonetic similarity of two lattice nodes, based on the string edit distance between their most likely
174          * pronunciations. TODO: maybe move to Node.java?
175          *
176          * @param n1 the first node
177          * @param n2 the second node
178          * @return the phonetic similarity, between 0 and 1
179          */
180         protected double computePhoneticSimilarity(Node n1, Node n2) {
181             Pronunciation p1 = n1.getWord().getMostLikelyPronunciation();
182             Pronunciation p2 = n2.getWord().getMostLikelyPronunciation();
183             double sim = stringEditDistance(Arrays.asList(p1.getUnits()),
184                     Arrays.asList(p2.getUnits()));
185             sim /= (double) (p1.getUnits().length + p2.getUnits().length);
186             return 1 - sim;
187         }

190         /**
191          * Calculate the distance between two clusters
192          *
193          * @param c1 the first cluster
194          * @param c2 the second cluster
195          * @return the inter cluster similarity, or Double.NEGATIVE_INFINITY if these clusters should never be clustered
196          *         together.
197          */
198         protected double interClusterDistance(Cluster c1, Cluster c2) {
199             if (areClustersInRelation(c1, c2)) {
200                 return Double.NEGATIVE_INFINITY;
201             }
202             float totalSim = LogMath.getLogZero();
203             float wordPairCount = (float) 0.0;
204             HashSet<String> wordsSeen1 = new HashSet<String>();
205     
206             for (Node node1 : c1.getElements()) {
207                 String word1 = node1.getWord().getSpelling();
208                 if (wordsSeen1.contains(word1)) {
209                     continue;
210                 }
211                 wordsSeen1.add(word1);
212                 HashSet<String> wordsSeen2 = new HashSet<String>();
213                 for (Node node2 : c2.getElements()) {
214                     String word2 = node2.getWord().getSpelling();
215                     if (wordsSeen2.contains(word2)) {
216                         continue;
217                     }
218                     wordsSeen2.add(word2);
219                     float sim = (float) computePhoneticSimilarity(node1, node2);
220                     sim = lattice.getLogMath().linearToLog(sim);
221                     sim += wordSubClusterProbability(c1, word1);
222                     sim += wordSubClusterProbability(c2, word2);
223                     totalSim = lattice.getLogMath().addAsLinear(totalSim, sim);
224                     wordPairCount++;
225                 }
226             }
227             return totalSim - lattice.getLogMath().logToLinear(wordPairCount);
228         }

231         /**
232          * Check whether these to clusters stand in a relation to each other. Two clusters are related if a member of one is
233          * an ancestor of a member of the other cluster.
234          *
235          * @param cluster1 the first cluster
236          * @param cluster2 the second cluster
237          * @return true if the clusters are related
238          */
239         protected boolean areClustersInRelation(Cluster cluster1, Cluster cluster2) {
240             for (Node n1: cluster1.getElements()) {
241                 for (Node n2: cluster2.getElements()) {
242                     if (n1.hasAncestralRelationship(n2)) {
243                         return true;
244                     }
245                 }
246             }
247             return false;
248         }

251         /**
252          * Calculate the distance between two clusters, forcing them to have the same words in them, and to not be related
253          * to each other.
254          *
255          * @param cluster1 the first cluster
256          * @param cluster2 the second cluster
257          * @return The intra-cluster distance, or Double.NEGATIVE_INFINITY if the clusters should never be clustered
258          *         together.
259          */
260         protected double intraClusterDistance(Cluster cluster1, Cluster cluster2) {
261             double maxSim = Double.NEGATIVE_INFINITY;
262             for (Node node1 : cluster1.getElements()) {
263                 for (Node node2 : cluster2.getElements()) {
264                     if (!node1.getWord().getSpelling().equals(node2.getWord().getSpelling())) {
265                         return Double.NEGATIVE_INFINITY;
266                     }
267                     if (node1.hasAncestralRelationship(node2)) {
268                         return Double.NEGATIVE_INFINITY;
269                     }
270                     double overlap = getOverlap(node1, node2);
271                     if (overlap > 0.0) {
272                         overlap = lattice.getLogMath().logToLinear((float) overlap);
273                         overlap += node1.getPosterior() + node2.getPosterior();
274                         if (overlap > maxSim) {
275                             maxSim = overlap;
276                         }
277                     }
278                 }
279             }
280             return maxSim;
281         }

294         /**
295          * Perform a step of the intra word clustering stage
296          *
297          * @param clusters the current list of clusters
298          * @return did two clusters get merged?
299          */
300         protected boolean intraWordClusterStep(List<Cluster> clusters) {
301             Cluster toBeMerged1 = null;
302             Cluster toBeMerged2 = null;
303             double maxSim = Double.NEGATIVE_INFINITY;
304             for (ListIterator<Cluster> i = clusters.listIterator(); i.hasNext();) {
305                 Cluster c1 = i.next();
306                 if (!i.hasNext()) {
307                     break;
308                 }
309                 for (ListIterator<Cluster> j = clusters.listIterator(i.nextIndex()); j.hasNext();) {
310                     Cluster c2 = j.next();
311                     double sim = intraClusterDistance(c1, c2);
312                     if (sim > maxSim) {
313                         maxSim = sim;
314                         toBeMerged1 = c1;
315                         toBeMerged2 = c2;
316                     }
317                 }
318             }
319             if (toBeMerged1 != null) {
320                 clusters.remove(toBeMerged2);
321                 toBeMerged1.add(toBeMerged2);
322                 return true;
323             }
324             return false;
325         }

328         /**
329          * Turn the lattice contained in this sausage maker into a sausage object.
330          *
331          * @return the sausage producing by collapsing the lattice.
332          */
333         public Sausage makeSausage() {
334             List<Cluster> clusters = new ArrayList<Cluster>(lattice.nodes.size());
335             for (Node n : lattice.nodes.values()) {
336                 n.cacheDescendants();
337                 Cluster bucket = new Cluster(n);
338                 clusters.add(bucket);
339             }
340             intraWordCluster(clusters);
341             interWordCluster(clusters);
342             clusters = topologicalSort(clusters);
343             return sausageFromClusters(clusters);
344         }

347         /** @see edu.cmu.sphinx.result.ConfidenceScorer#score(edu.cmu.sphinx.result.Result) */
348         @Override
349         public ConfidenceResult score(Result result) {
350             lattice = new Lattice(result);
351             LatticeOptimizer lop = new LatticeOptimizer(lattice);
352             lop.optimize();
353             lattice.computeNodePosteriors(languageWeight);
354             return makeSausage();
355         }

358         /**
359          * Topologically sort the clusters. Note that this is a brute force sort by removing the min cluster from the list
360          * of clusters, since Collections.sort() does not work in all cases.
361          *
362          * @param clusters the list of clusters to be topologically sorted
363          * @return a topologically sorted list of clusters
364          */
365         private List<Cluster> topologicalSort(List<Cluster> clusters) {
366             Comparator<Cluster> comparator = new ClusterComparator();
367             List<Cluster> sorted = new ArrayList<Cluster>(clusters.size());
368             while (!clusters.isEmpty()) {
369                 Cluster cluster = Collections.min(clusters, comparator);
370                 clusters.remove(cluster);
371                 sorted.add(cluster);
372             }
373             return sorted;
374         }
