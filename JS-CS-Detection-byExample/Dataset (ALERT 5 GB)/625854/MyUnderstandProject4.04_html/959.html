13     package edu.cmu.sphinx.demo.zipcity;
14     
15     import edu.cmu.sphinx.frontend.util.Microphone;
16     import edu.cmu.sphinx.recognizer.Recognizer;

18     import edu.cmu.sphinx.result.Result;
19     import edu.cmu.sphinx.util.props.ConfigurationManager;

22     import java.io.IOException;
23     import java.net.URL;
24     import java.util.*;
25     
26     /** Manages the speech recognition for zip city */
27     public class ZipRecognizer implements Runnable {
28     
29         private Microphone microphone;
30         private Recognizer recognizer;
31         private List<ZipListener> zipListeners = new ArrayList<ZipListener>();
32     
33     
34         /**
35          * Creates the ZipRecognizer.
36          *
37          * @throws IOException if an error occurs while loading resources
38          */
39         public ZipRecognizer() throws IOException {
40             try {
41                 URL url = this.getClass().getResource("zipcity.config.xml");
42                 if (url == null) {
43                     throw new IOException("Can't find zipcity.config.xml");
44                 }
45                 ConfigurationManager cm = new ConfigurationManager(url);
46                 recognizer = (Recognizer) cm.lookup("recognizer");
47                 microphone = (Microphone) cm.lookup("microphone");
48             } catch (PropertyException e) {
49                 throw new IOException("Problem configuring ZipRecognizer " + e);
50             }
51         }
52     
53     
54         /** Turns on the microphone and starts recognition */
55         public boolean microphoneOn() {
56             if (microphone.getAudioFormat() == null) {
57                 return false;
58             } else {
59                 new Thread(this).start();
60                 return true;
61             }
62         }
63     
64     
65         /** Turns off the microphone, ending the current recognition in progress */
66         public void microphoneOff() {
67             microphone.stopRecording();
68         }
69     
70     
71         /** Allocates resources necessary for recognition. */
72         public void startup() throws IOException {
73             recognizer.allocate();
74         }
75     
76     
77         /** Releases recognition resources */
78         public void shutdown() {
79             microphoneOff();
80             if (recognizer.getState() == State.ALLOCATED) {
81                 recognizer.deallocate();
82             }
83         }
84     
85     
86         /** Performs a single recognition */
87         public void run() {
88             microphone.clear();
89             microphone.startRecording();
90             Result result = recognizer.recognize();
91             microphone.stopRecording();
92             if (result != null) {
93                 String resultText = result.getBestFinalResultNoFiller();
94                 if (!resultText.isEmpty()) {
95                     String zip = convertResultToZip(resultText);
96                     fireListeners(zip);
97                 } else {
98                     fireListeners(null);
99                 }
100             }
101         }
102     
103     
104         /**
105          * Converts a string of the form "one two three four five" to a string of the form "12345"
106          *
107          * @param zipstring the zip string
108          * @return the zip string in a digits form
109          */
110         private String convertResultToZip(String zipstring) {
111             StringBuilder sb = new StringBuilder();
112             StringTokenizer st = new StringTokenizer(zipstring);
113             while (st.hasMoreTokens()) {
114                 String word = st.nextToken();
115                 String digit = lookupDigit(word);
116                 if (digit == null) {
117                     throw new Error("Can't map " + word + " to a digit.");
118                 }
119                 sb.append(digit);
120             }
121             return sb.toString();
122         }
123     
124     
125         private static Map<String, String> digitMap = new HashMap<String, String>();
126     
127     
128         static {
129             digitMap.put("oh", "0");
130             digitMap.put("zero", "0");
131             digitMap.put("one", "1");
132             digitMap.put("two", "2");
133             digitMap.put("three", "3");
134             digitMap.put("four", "4");
135             digitMap.put("five", "5");
136             digitMap.put("six", "6");
137             digitMap.put("seven", "7");
138             digitMap.put("eight", "8");
139             digitMap.put("nine", "9");
140         }
141     
142     
143         /**
144          * looks up the digit for a word
145          *
146          * @param word the digit word
147          * @return digit the digit form of the word (or null)
148          */
149         private String lookupDigit(String word) {
150             return digitMap.get(word);
151         }
152     
153     
154         /**
155          * Adds a listener that is called whenever a new zip code is recognized
156          *
157          * @param zipListener the zip code listener
158          */
159         public synchronized void addZipListener(ZipListener zipListener) {
160             zipListeners.add(zipListener);
161         }
162     
163     
164         /**
165          * Removes a previously added zip listener
166          *
167          * @param zipListener the zip code listener
168          */
169         public synchronized void removeZipListener(ZipListener zipListener) {
170             zipListeners.remove(zipListener);
171         }
172     
173     
174         /**
175          * Invoke all added zip listeners
176          *
177          * @param zipcode the recognized zip code
178          */
179         private synchronized void fireListeners(String zipcode) {
180             for (ZipListener zl : zipListeners)
181                 zl.notify(zipcode);
182         }
183     }
184     
185     
186     /** An interface for zip listeners */
187     interface ZipListener {
188     
189         /** Invoked when a new zip code is recognized */
190         void notify(String zipcode);
191     }
