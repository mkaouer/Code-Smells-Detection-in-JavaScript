15          * @param dp
16          * @param predecClass
17          */
18         public static <T extends DataProcessor> T getFrontEndProcessor(DataProcessor dp, Class<T> predecClass) {
19             while (!predecClass.isInstance(dp)) {
20                 if (dp instanceof FrontEnd)
21                     dp = ((FrontEnd) dp).getLastDataProcessor();
22                 else
23                     dp = dp.getPredecessor();
24     
25                 if (dp == null)
26                     return null;
27             }
28     
29     
30             return predecClass.cast(dp);
31         }
