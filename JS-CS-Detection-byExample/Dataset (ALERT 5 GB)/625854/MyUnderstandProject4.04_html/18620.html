119         /**
120          * Starts listening for zipcodes
121          */
122         private void startListening() {
123             if (zipRecognizer.microphoneOn()) {
124                 setMessage("Speak a zip code ...");
125             } else {
126                 setMessage(
127                 "Sorry, can't find the microphone on your computer.");
128             }
129         }

131         /**
132          * Perform any needed initializations and then enable the 'speak'
133          * button, allowing recognition to proceed
134          */
135         public void go() {
136             try {
137                 setMessage("Loading zip codes...");
138                 zipDB = new ZipDatabase();
139     
140                 setMessage("Loading recognizer...(apologies to Alaska and Hawaii)");
141                 zipRecognizer = new ZipRecognizer();
142     
143                 setMessage("Starting recognizer...");
144                 zipRecognizer.startup();
145     
146                 zipRecognizer.addZipListener(new ZipListener() {
147                     public void notify(String zip) {
148                         updateMap(zip);
149                         if (continuousMode) {
150                             zipRecognizer.microphoneOn();
151                         }
152                     }
153                 });
154     
155                 displayMap();
156                 if (continuousMode) {
157                     startListening();
158                 } else {
159                     setMessage("ZipCity Version 2.0 - Ready");
160                 }
161             } catch (Throwable e) {
162                 setMessage("Error: " + e.getMessage());
163             }
164         }

166         /**
167          * Update the display with the new zip code information. The
168          * zip info is retrieved and if available disabled on the gui
169          *
170          * @param zip the zip code (in the form XXXX)
171          */
172         private void updateMap(final String zip) {
173             final ZipInfo zipInfo = zipDB.lookup(zip);
174             SwingUtilities.invokeLater(new Runnable() {
175                 public void run() {
176                     if (zip == null) {
177                         setMessage("I didn't understand what you said");
178                     } else {
179                         if (zipInfo == null) {
180                             setMessage("Can't find " + zip  + " in the database");
181                         } else {
182                             String location = zipInfo.getCity() + ", " 
183                                 + zipInfo.getState();
184                             setMessage(location);
185                             currentInfo = zipInfo;
186                         }
187                     }
188                     mapPanel.repaint();
189                     speakButton.setEnabled(true);
190                 }
191             });
192         }

296         /**
297          * A panel that draws a map of the US based upon the
298          * zip code database
299          */
300         class MapPanel extends JPanel {
301     
302             private final static float MIN_LAT = 24.0f;
303             private final static float MAX_LAT = 52f;
304             private final static float MAX_LONG = 126f;
305             private final static float MIN_LONG = 65f;
306             private final static float RANGE_LAT = MAX_LAT - MIN_LAT;
307             private final static float RANGE_LONG = MAX_LONG - MIN_LONG;
308             private final static int NORMAL_SIZE = 2;
309             private final static int HIGHLIGHT_SIZE = 10;
310     
311             private final static float DEFAULT_LAT = 25;
312             private final static float DEFAULT_LONG = 120;
313     
314             private final static int WIDTH_OFFSET = 10;
315             private final static int MARGIN = 10;
316             private Graphics g; 
317             private Dimension size;
318     
319     
320             /**
321              * Creates the map panel
322              */
323             MapPanel() {
324                 setFont(labelFont);
325                 setBackground(backgroundColor);
326             }
327     
328     
329             /**
330              * Updates the map
331              */
332             public void paintComponent(Graphics graphics) {
333                 g = graphics;
334                 size = getSize();
335                 super.paintComponent(g);
336                 if (zipDB != null) {
337                     g.setColor(NORM_COLOR);
338                     for (ZipInfo zi : zipDB) {
339                         plot(zi, false);
340                     }
341                     if (currentInfo != null) {
342                         g.setColor(HIGHLIGHT_COLOR);
343                         plot(currentInfo, true);
344                     }
345     
346                 }
347             }
348     
349             /**
350              * Plots a zip info with given size
351              *
352              * @param zi the zip info to plot
353              * @param highlight if true this info should be highlighted
354              *
355              */
356             void plot(ZipInfo zi, boolean highlight) {
357                 //System.out.println("ll " + zi.getLongitude() + " " +
358                 //       zi.getLatitude());
359                 if (highlight) {
360                     drawZipInfo(zi);
361                 } else {
362                     int x = mapx(zi.getLongitude());
363                     int y = mapy(zi.getLatitude());
364                     plot(x,y, NORMAL_SIZE);
365                 }
366             }
367     
368             /**
369              * Draws the zip info on the map
370              *
371              * @param zi the zip info
372              */
373             void drawZipInfo( ZipInfo zi) {
374                 int x, y;
375                 String label = zi.getCity() + ", " + zi.getState() + ' ' +
376                     zi.getZip();
377                 Dimension d = getStringDimension(g, label);
378     
379                 if (zi.getLatitude() < MIN_LAT || zi.getLatitude() > MAX_LAT 
380                         || zi.getLongitude() < MIN_LONG ||
381                         zi.getLongitude() > MAX_LONG) {
382                     x = mapx(DEFAULT_LONG);
383                     y = mapy(DEFAULT_LAT);
384                 } else {
385                     x = mapx(zi.getLongitude());
386                     y = mapy(zi.getLatitude());
387                     g.fillOval(x, y, HIGHLIGHT_SIZE, HIGHLIGHT_SIZE);
388                 }
389                 int xpos = x - WIDTH_OFFSET;
390                 int ypos = y - d.height / 2;
391     
392                 if (xpos + d.width + MARGIN > size.width) {
393                     xpos -= (xpos + d.width - size.width) + MARGIN;
394                 }
395                 g.drawString(label, xpos, ypos);
396             }
397     
398             /**
399              * plot a point with the given pixel size
400              *
401              * @param x the x position of the point
402              * @param y the y position of the point
403              * @param size the size of the point
404              */
405             void plot(int x, int y, int size) {
406                  // g.fillOval(x, y, size, size);
407                  g.fillRect(x, y, size, size);
408             }
409     
410             /**
411              * Maps the given x position to the map panel
412              * @param x the x position
413              *
414              * @return the x position mapped onto the map panel
415              */
416             private int mapx(float x) {
417                 return size.width - 
418                     (int) (size.width * (x - MIN_LONG) / RANGE_LONG);
419             }
420     
421             /**
422              * Maps the given y position to the map panel
423              * @param x the y position
424              *
425              * @return the y position mapped onto the map panel
426              */
427             private int mapy(float y)  {
428                 return size.height - 
429                     (int) (size.height * (y - MIN_LAT) / RANGE_LAT);
430             }
431     
432     
433             /**
434              * Gets the width and height in pixels of the given string
435              *
436              * @param g the current graphics context
437              * @param s the string of interest
438              * @return the dimension, in pixels of the string
439              */
440             private Dimension getStringDimension(Graphics g, String s) {
441                 FontMetrics fm = g.getFontMetrics();
442                 Rectangle2D r2d = fm.getStringBounds(s, g);
443                 return new Dimension((int) (r2d.getWidth() + .5),
444                                      (int) (r2d.getHeight() + .5));
445             }
446         }

492         /**
493          * The main program for zip city.  Creates the ZipCity frame,
494          * displays it, and  runs it.
495          *
496          * Usage:
497          *
498          *   java -mx200m ZipCity [-continuous]
499          *
500          * @param args program arguments 
501          */
502         public static void main(String[] args) {
503             boolean continuous = false;
504     
505             for (String arg : args) {
506                 if (arg.equals("-continuous")) {
507                     continuous = true;
508                 } else {
509                     usage();
510                     System.exit(0);
511                 }
512             }
513     
514             ZipCity zipCity = new ZipCity(continuous);
515             zipCity.setVisible(true);
516             zipCity.go();
517         }
