33     /** A SphinxRecognizer provides access to Sphinx speech recognition capabilities. */
34     public class SphinxRecognizer extends BaseRecognizer {
35         /** Msecs to sleep before the status of the recognizer is checked again. */
36         private static final long SLEEP_MSEC = 50;
37     
38         /** The encapsulated recognizer. */
39         private Recognizer recognizer;
40     
41         /** The input device. */
42         private DataProcessor dataProcessor;
43     
44         /** The grammar manager. */
45         private JSGFGrammar grammar;
46     
47         /** The result listener. */
48         private final Sphinx4ResultListener resultListener;
49     
50         /**
51          * The decoding thread. It points either to the single decoding thread
52          * or is <code>null</code> if no recognition thread is started.
53          */
54         private RecognitionThread recognitionThread;
55     
56         /**
57          * Creates the default sphinx recognizer.
58          * @exception EngineException
59          *            if the recognizer could not be created
60          */
61         public SphinxRecognizer() throws EngineException {
62             this(new SphinxRecognizerModeDesc());
63         }
64     
65     
66         /**
67          * Creates a sphinx 4 recognizer that matches the given mode descriptor
68          *
69          * @param modeDesc the mode descriptor describing the type of recognizer to create
70          * @exception EngineException
71          *            if the recognizer could not be created
72          */
73         public SphinxRecognizer(SphinxRecognizerModeDesc modeDesc)
74             throws EngineException {
75             String config = modeDesc.getConfigFile();
76             URL url = SphinxRecognizer.class.getResource(config);
77             if (url == null) {
78                 throw new EngineException(
79                         "Unable to load configuration from resource '" + config
80                         + "'!");
81             }
82     
83             try {
84                 final ConfigurationManager configuration =
85                     new ConfigurationManager(url);
86     
87                 recognizer = (Recognizer) configuration.lookup("recognizer");
88                 dataProcessor = (DataProcessor) configuration.lookup("microphone");
89                 grammar = (JSGFGrammar) configuration.lookup("jsgfGrammar");
90             } catch (Exception ex) {
91                 throw new EngineException(ex.getMessage());
92             }
93     
94             resultListener = new Sphinx4ResultListener(this);
95         }
96     
97         /**
98          * Called from the <code>resume</code> method.
99          */
100         @Override
101         protected void handleResume() {
102             if (recognizer == null) {
103                 return;
104             }
105     
106             if (recognitionThread != null) {
107                 return;
108             }
109     
110             recognitionThread = new RecognitionThread(this);
111             recognitionThread.start();
112         }
113     
114         /**
115          * Called from the <code>pause</code> method.
116          */
117         @Override
118         protected void handlePause() {
119             if (recognitionThread == null) {
120                 return;
121             }
122     
123             stopRecognitionThread();
124             if (dataProcessor instanceof Microphone) {
125                 final Microphone microphone = (Microphone) dataProcessor;
126                 microphone.stopRecording();
127             }
128         }
129     
130     
131         /**
132          * Called from the <code>allocate</code> method.
133          *
134          * @throws EngineException if problems are encountered
135          */
136         @Override
137         protected void handleAllocate()
138                 throws EngineException {
139             if (recognizer == null) {
140                 throw new EngineException(
141                         "cannot allocate: no recognizer created!");
142             }
143     
144             recognizer.allocate();
145     
146             final RuleGrammar[] grammars = listRuleGrammars();
147             for (int i = 0; i < grammars.length; i++) {
148                 deleteRuleGrammar(grammars[i]);
149             }
150     
151             recognizer.addResultListener(resultListener);
152     
153             setEngineState(CLEAR_ALL_STATE, ALLOCATED);
154         }
155     
156         /**
157          * {@inheritDoc}
158          * Sphinx4 has different concepts for loading grammars. Since the JSAPI 1.0
159          * way does not work properly, this workaround is needed. It simply adds
160          * the rules of the loaded grammar to the existing rules.
161          * @since 0.7.2
162          */
163         @Override
164         public RuleGrammar loadJSGF(Reader reader) throws GrammarException,
165                 IOException, EngineStateError {
166             final RuleGrammar loadedGrammar = super.loadJSGF(reader);
167             final RuleGrammar ruleGrammar = grammar.getRuleGrammar();
168             final String[] loadedRuleNames = loadedGrammar.listRuleNames();
169             for (String name : loadedRuleNames) {
170                 final Rule rule = loadedGrammar.getRule(name);
171                 ruleGrammar.setRule(name, rule, true);
172                 ruleGrammar.setEnabled(name, true);
173             }
174             grammar.commitChanges();
175             return loadedGrammar;
176         }
177     
178         /**
179          * Called from the <code>deallocate</code> method.
180          *
181          * @throws EngineException if this <code>Engine</code> cannot be
182          *   deallocated.
183          */
184         @Override
185         protected void handleDeallocate()
186                 throws EngineException {
187             if (recognizer == null) {
188                 throw new EngineException(
189                         "cannot deallocate: no recognizer created!");
190             }
191             // Stop the decoder thread.
192             stopRecognitionThread();
193     
194             // Deallocate the recognizer and wait until it stops recognizing.
195             recognizer.deallocate();
196             while (recognizer.getState() == Recognizer.State.RECOGNIZING) {
197                 try {
198                     Thread.sleep(SLEEP_MSEC);
199                 } catch (InterruptedException e) {
200                     throw new EngineException(e.getMessage());
201                 }
202             }
203             recognizer.resetMonitors();
204             recognizer.removeResultListener(resultListener);
205         }
206     
207         /**
208          * Load a RuleGrammar and its imported grammars from a URL containing
209          * JSGF text.
210          * From javax.speech.recognition.Recognizer.
211          * @param url the base URL containing the JSGF grammar file.
212          * @param name the name of the JSGF grammar to load.
213          * @return Loaded grammar.
214          *
215          * @exception GrammarException
216          *            Error in the grammar.
217          * @exception IOException
218          *            Error reading the grammar.
219          */
220         @Override
221         public RuleGrammar loadJSGF(final URL url, final String name)
222                 throws GrammarException, IOException {
223             grammar.loadJSGF(name);
224             return super.loadJSGF(url, name);
225         }
226     
227         /**
228          * Selector for the data processor (the microphone).
229          * @return The used data processor.
230          */
231         DataProcessor getDataProcessor() {
232             return dataProcessor;
233         }
234     
235         /**
236          * Selector for the wrapped sphinx4 recognizer.
237          * @return Recognizer
238          */
239         Recognizer getRecognizer() {
240             return recognizer;
241         }
242     
243         /**
244          * Stop the recognition thread and wait until it is terminated.
245          */
246         private void stopRecognitionThread() {
247             if (recognitionThread == null) {
248                 return;
249             }
250     
251             recognitionThread.stopRecognition();
252     
253             final long maxSleepTime = 5000;
254             long sleepTime = 0;
255     
256             while (recognitionThread.isAlive() && (sleepTime < maxSleepTime)) {
257                 try {
258                     Thread.sleep(SLEEP_MSEC);
259                     sleepTime += SLEEP_MSEC;
260                 } catch (InterruptedException e) {
261                     return;
262                 }
263             }
264     
265             recognitionThread = null;
266         }
267     
268         /**
269          * Get the current rule grammar.
270          * @return Active grammar.
271          */
272         RuleGrammar getRuleGrammar() {
273             return grammar.getRuleGrammar();
274         }
275     }
