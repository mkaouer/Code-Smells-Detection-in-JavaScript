31     /**
32      * Loads a tied-state acoustic model generated by the Sphinx-3 trainer.
33      * <p>
34      * The acoustic model is stored as a directory specified by a URL.
35      * The dictionary and language model files are not required to be in the
36      * package. You can specify their locations separately. 
37      * </p>
38      * <p>Configuration file should set mandatory property of component:
39      * <ul>
40      * <li><b>location</b> - this specifies the directory where the actual model
41      * data files are. You can use <b>resource:</b> prefix to refer to files packed
42      * inside jar or any other URI scheme.
43      * </ul>
44      * The actual model data files are named "mdef", "means", "variances",
45      * "transition_matrices", "mixture_weights" for binary versions, or with suffix
46      * ".ascii" for the ASCII versions.
47      * </p>
48      * <p>
49      * If model has some layout that is different default generated by SphinxTrain, you may 
50      * specify additional properties like <b>dataLocation</b> to set the path to the binary 
51      * files, <b>mdef</b> to set the path to the model definition file.
52      * </p>
53      * <p>
54      * As an example, lets look at the Wall Street Journal acoustic model JAR file,
55      * which is located at the <code>sphinx4/lib</code> directory. If you run
56      * <code>"jar tf lib/WSJ_8gau_13dCep_16k_40mel_130Hz_6800Hz.jar"</code>, you
57      * will find that its internal structure looks roughly like:
58      * <p/>
59      * 
60      * <pre>
61      * WSJ_8gau_13dCep_16k_40mel_130Hz_6800Hz.jar
62      *   |
63      *   +- license.terms
64      *   +- cd_continuous_8gau
65      *   |   |
66      *   |   +- means
67      *   |   +- variances
68      *   |   +- mixture_weights
69      *   |   +- transition_matrices
70      *   |
71      *   +- dict
72      *   |   |
73      *   |   +- alpha.dict
74      *   |   +- cmudict.0.6d
75      *   |   +- digits.dict
76      *   |   +- fillerdict
77      *   |
78      *   +- etc
79      *       |
80      *       +- WSJ_clean_13dCep_16k_40mel_130Hz_6800Hz.4000.mdef
81      *       +- WSJ_clean_13dCep_16k_40mel_130Hz_6800Hz.ci.mdef
82      *       +- variables.def
83      * </pre>
84      * <p/>
85      * <p>
86      *  So the configuration to load this model should look like
87      *  <pre>
88      *  &lt;component name="wsjLoader" type="edu.cmu.sphinx.linguist.acoustic.tiedstate.Sphinx3Loader"&gt;
89      *     &lt;property name="logMath" value="logMath"/&gt;
90      *        &lt;property name="unitManager" value="unitManager"/&gt;
91      *        &lt;property name="location" value="resource:/WSJ_8gau_13dCep_16k_40mel_130Hz_6800Hz"/&gt;
92      *        &lt;property name="modelDefinition" value="etc/WSJ_clean_13dCep_16k_40mel_130Hz_6800Hz.4000.mdef"/&gt;
93      *        &lt;property name="dataLocation" value="cd_continuous_8gau/"/&gt;
94      *    &lt;/component&gt;
95      *  </pre>
96      * </p>
97      * <p>
98      * For more details on using SphinxTrain models in sphinx4, see <a href="../../../../../../../doc/UsingSphinxTrainModels.html">documentation</a>
99      * </p>
100      */

337         /**
338          * Loads the AcousticModel from a directory in the file system.
339          * 
340          * @param modelDef
341          *            the name of the acoustic modelDef; if null we just load from the
342          *            default location
343          * @throws java.io.IOException
344          */
345         private void loadModelFiles(String modelDef) throws IOException, URISyntaxException {
346     
347             logger.config("Loading Sphinx3 acoustic model: " + modelDef);
348             logger.config("    modelName: " + this.model);
349             logger.config("    dataLocation   : " + dataLocation);
350     
351             if (binary) {
352                 meansPool = loadDensityFileBinary(dataLocation + "means",
353                         -Float.MAX_VALUE);
354                 variancePool = loadDensityFileBinary(dataLocation + "variances",
355                         varianceFloor);
356                 mixtureWeightsPool = loadMixtureWeightsBinary(dataLocation
357                         + "mixture_weights", mixtureWeightFloor);
358                 matrixPool = loadTransitionMatricesBinary(dataLocation
359                         + "transition_matrices");
360                 transformMatrix = loadTransformMatrix(dataLocation + "feature_transform");
361             } else {
362                 meansPool = loadDensityFileAscii(dataLocation + "means.ascii",
363                         -Float.MAX_VALUE);
364                 variancePool = loadDensityFileAscii(dataLocation + "variances.ascii",
365                         varianceFloor);
366                 mixtureWeightsPool = loadMixtureWeightsAscii(dataLocation
367                         + "mixture_weights.ascii", mixtureWeightFloor);
368                 matrixPool = loadTransitionMatricesAscii(dataLocation
369                         + "transition_matrices.ascii");
370             }
371             senonePool = createSenonePool(distFloor, varianceFloor);
372             // load the HMM modelDef file
373             InputStream modelStream = getDataStream(this.model);
374             if (modelStream == null) {
375                 throw new IOException("can't find modelDef " + this.model);
376             }
377             loadHMMPool(useCDUnits, modelStream, this.model);
378         }

385         /**
386          * Creates the senone pool from the rest of the pools.
387          * 
388          * @param distFloor
389          *            the lowest allowed score
390          * @param varianceFloor
391          *            the lowest allowed variance
392          * @return the senone pool
393          */
394         private Pool<Senone> createSenonePool(float distFloor, float varianceFloor) {
395             Pool<Senone> pool = new Pool<Senone>("senones");
396             int numMixtureWeights = mixtureWeightsPool.size();
397     
398             int numMeans = meansPool.size();
399             int numVariances = variancePool.size();
400             int numGaussiansPerSenone = mixtureWeightsPool.getFeature(
401                     NUM_GAUSSIANS_PER_STATE, 0);
402             int numSenones = mixtureWeightsPool.getFeature(NUM_SENONES, 0);
403             int whichGaussian = 0;
404     
405             logger.fine("NG " + numGaussiansPerSenone);
406             logger.fine("NS " + numSenones);
407             logger.fine("NMIX " + numMixtureWeights);
408             logger.fine("NMNS " + numMeans);
409             logger.fine("NMNS " + numVariances);
410     
411             assert numGaussiansPerSenone > 0;
412             assert numMixtureWeights == numSenones;
413             assert numVariances == numSenones * numGaussiansPerSenone;
414             assert numMeans == numSenones * numGaussiansPerSenone;
415     
416             for (int i = 0; i < numSenones; i++) {
417                 MixtureComponent[] mixtureComponents = new MixtureComponent[numGaussiansPerSenone];
418                 for (int j = 0; j < numGaussiansPerSenone; j++) {
419                     mixtureComponents[j] = new MixtureComponent(logMath, meansPool
420                             .get(whichGaussian), meanTransformationMatrixPool
421                             .get(0), meanTransformationVectorPool.get(0),
422                             variancePool.get(whichGaussian),
423                             varianceTransformationMatrixPool.get(0),
424                             varianceTransformationVectorPool.get(0), distFloor,
425                             varianceFloor);
426     
427                     whichGaussian++;
428                 }
429     
430                 Senone senone = new GaussianMixture(logMath, mixtureWeightsPool
431                         .get(i), mixtureComponents, i);
432     
433                 pool.put(i, senone);
434             }
435             return pool;
436         }

438         /**
439          * Loads the sphinx3 density file, a set of density arrays are created and
440          * placed in the given pool.
441          * 
442          * @param path
443          *            the name of the data
444          * @param floor
445          *            the minimum density allowed
446          * @return a pool of loaded densities
447          * @throws FileNotFoundException
448          *             if a file cannot be found
449          * @throws IOException
450          *             if an error occurs while loading the data
451          */
452         private Pool<float[]> loadDensityFileAscii(String path, float floor)
453                 throws IOException, URISyntaxException {
454             logger.fine("Loading density file from: " + path);
455             InputStream inputStream = getDataStream(path);
456             if (inputStream == null) {
457                 throw new FileNotFoundException("Error trying to read file " + path);
458             }
459     
460             // 'false' argument refers to EOL is insignificant
461             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream,
462                     '#', false);
463             Pool<float[]> pool = new Pool<float[]>(path);
464             est.expectString("param");
465             int numStates = est.getInt("numStates");
466             int numStreams = est.getInt("numStreams");
467             int numGaussiansPerState = est.getInt("numGaussiansPerState");
468             pool.setFeature(NUM_SENONES, numStates);
469             pool.setFeature(NUM_STREAMS, numStreams);
470             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
471             for (int i = 0; i < numStates; i++) {
472                 est.expectString("mgau");
473                 est.expectInt("mgau index", i);
474                 est.expectString("feat");
475                 est.expectInt("feat index", 0);
476                 for (int j = 0; j < numGaussiansPerState; j++) {
477                     est.expectString("density");
478                     est.expectInt("densityValue", j);
479                     float[] density = new float[vectorLength];
480                     for (int k = 0; k < vectorLength; k++) {
481                         density[k] = est.getFloat("val");
482                         if (density[k] < floor) {
483                             density[k] = floor;
484                         }
485                     }
486                     int id = i * numGaussiansPerState + j;
487                     pool.put(id, density);
488                 }
489             }
490             est.close();
491             return pool;
492         }

494         /**
495          * Loads the sphinx3 density file, a set of density arrays are created and
496          * placed in the given pool.
497          * 
498          * @param path
499          *            the name of the data
500          * @param floor
501          *            the minimum density allowed
502          * @return a pool of loaded densities
503          * @throws FileNotFoundException
504          *             if a file cannot be found
505          * @throws IOException
506          *             if an error occurs while loading the data
507          */
508         private Pool<float[]> loadDensityFileBinary(String path, float floor)
509                 throws IOException, URISyntaxException {
510             Properties props = new Properties();
511             int blockSize = 0;
512     
513             DataInputStream dis = readS3BinaryHeader(path, props);
514     
515             String version = props.getProperty("version");
516     
517             if (version == null || !version.equals(DENSITY_FILE_VERSION)) {
518                 throw new IOException("Unsupported version in " + path);
519             }
520     
521             String checksum = props.getProperty("chksum0");
522             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
523             resetChecksum();
524     
525             int numStates = readInt(dis);
526             int numStreams = readInt(dis);
527             int numGaussiansPerState = readInt(dis);
528     
529             int[] vectorLength = new int[numStreams];
530             for (int i = 0; i < numStreams; i++) {
531                 vectorLength[i] = readInt(dis);
532             }
533     
534             int rawLength = readInt(dis);
535     
536             logger.fine("Number of states " + numStates);
537             logger.fine("Number of streams " + numStreams);
538             logger.fine("Number of gaussians per state " + numGaussiansPerState);
539             logger.fine("Vector length " + vectorLength.length);
540             logger.fine("Raw length " + rawLength);
541     
542             for (int i = 0; i < numStreams; i++) {
543                 blockSize += vectorLength[i];
544             }
545     
546             assert rawLength == numGaussiansPerState * blockSize * numStates;
547             assert numStreams == 1;
548     
549             Pool<float[]> pool = new Pool<float[]>(path);
550             pool.setFeature(NUM_SENONES, numStates);
551             pool.setFeature(NUM_STREAMS, numStreams);
552             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
553     
554             for (int i = 0; i < numStates; i++) {
555                 for (int j = 0; j < numStreams; j++) {
556                     for (int k = 0; k < numGaussiansPerState; k++) {
557                         float[] density = readFloatArray(dis, vectorLength[j]);
558                         Utilities.floorData(density, floor);
559                         pool.put(i * numGaussiansPerState + k, density);
560                     }
561                 }
562             }
563     
564             validateChecksum(dis, doCheckSum);
565             
566             dis.close();
567             return pool;
568         }

570         /**
571          * Reads the S3 binary header from the given location + path. Adds header
572          * information to the given set of properties.
573          * 
574          * @param path
575          *            the name of the file
576          * @param props
577          *            the properties
578          * @return the input stream positioned after the header
579          * @throws IOException
580          *             on error
581          */
582         protected DataInputStream readS3BinaryHeader(String path, Properties props)
583                 throws IOException, URISyntaxException {
584     
585             InputStream inputStream = getDataStream(path);
586     
587             if (inputStream == null) {
588                 throw new IOException("Can't open " + path);
589             }
590             DataInputStream dis = new DataInputStream(new BufferedInputStream(
591                     inputStream));
592             String id = readWord(dis);
593             if (!id.equals("s3")) {
594                 throw new IOException("Not proper s3 binary file " + path);
595             }
596             String name;
597             while ((name = readWord(dis)) != null) {
598                 if (!name.equals("endhdr")) {
599                     String value = readWord(dis);
600                     props.setProperty(name, value);
601                 } else {
602                     break;
603                 }
604             }
605             int byteOrderMagic = dis.readInt();
606             if (byteOrderMagic == BYTE_ORDER_MAGIC) {
607                 logger.fine("Not swapping " + path);
608                 swap = false;
609             } else if (Utilities.swapInteger(byteOrderMagic) == BYTE_ORDER_MAGIC) {
610                 logger.fine("Swapping  " + path);
611                 swap = true;
612             } else {
613                 throw new IOException("Corrupted S3 file " + path);
614             }
615             return dis;
616         }

618         /**
619          * Reads the next word (text separated by whitespace) from the given stream.
620          * 
621          * @param dis
622          *            the input stream
623          * @return the next word
624          * @throws IOException
625          *             on error
626          */
627         String readWord(DataInputStream dis) throws IOException {
628             StringBuilder sb = new StringBuilder();
629             char c;
630             // skip leading whitespace
631             do {
632                 c = readChar(dis);
633             } while (Character.isWhitespace(c));
634             // read the word
635             do {
636                 sb.append(c);
637                 c = readChar(dis);
638             } while (!Character.isWhitespace(c));
639             return sb.toString();
640         }

642         /**
643          * Reads a single char from the stream.
644          * 
645          * @param dis
646          *            the stream to read
647          * @return the next character on the stream
648          * @throws IOException
649          *             if an error occurs
650          */
651         private char readChar(DataInputStream dis) throws IOException {
652             return (char) dis.readByte();
653         }

655         /* Stores checksum during loading */
656         private long calculatedCheckSum = 0;
657         
658         /**
659          * Resets the checksum before loading a new chunk of data
660          */
661         private void resetChecksum() {
662             calculatedCheckSum = 0;
663         }
664         
665         /**
666          *  Validates checksum in the stream
667          *  
668          *  @param dis input stream
669          *  @param doCheckSum validates
670          *  @throws IOException on error
671          **/
672         private void validateChecksum(DataInputStream dis, boolean doCheckSum) throws IOException {
673             if (!doCheckSum)
674                 return;
675             int oldCheckSum = (int)calculatedCheckSum;
676             int checkSum = readInt(dis);
677             if (checkSum != oldCheckSum) {
678                 throw new IOException("Invalid checksum "
679                         + Long.toHexString(calculatedCheckSum) + " must be "
680                         + Integer.toHexString(checkSum));
681             }
682         }

684         /**
685          * Read an integer from the input stream, byte-swapping as necessary.
686          * 
687          * @param dis
688          *            the input stream
689          * @return an integer value
690          * @throws IOException
691          *             on error
692          */
693         protected int readInt(DataInputStream dis) throws IOException {
694             int val;
695             if (swap) {
696                 val = Utilities.readLittleEndianInt(dis);
697             } else {
698                 val = dis.readInt();
699             }
700             calculatedCheckSum = ((calculatedCheckSum << 20 | calculatedCheckSum >> 12) + val) & 0xFFFFFFFFL;
701             return val;
702         }

704         /**
705          * Read a float from the input stream, byte-swapping as necessary.
706          * 
707          * @param dis
708          *            the input stream
709          * @return a floating pint value
710          * @throws IOException
711          *             on error
712          */
713         protected float readFloat(DataInputStream dis) throws IOException {
714             int val;
715             if (swap) {
716                 val = Utilities.readLittleEndianInt(dis);
717             } else {
718                 val = dis.readInt();
719             }
720             calculatedCheckSum = ((calculatedCheckSum << 20 | calculatedCheckSum >> 12) + val) & 0xFFFFFFFFL;
721             return Float.intBitsToFloat(val);
722         }

746         /**
747          * Loads the sphinx3 density file, a set of density arrays are created and
748          * placed in the given pool.
749          * 
750          * @param useCDUnits
751          *            if true, loads also the context dependent units
752          * @param inputStream
753          *            the open input stream to use
754          * @param path
755          *            the path to a density file
756          * @throws FileNotFoundException
757          *             if a file cannot be found
758          * @throws IOException
759          *             if an error occurs while loading the data
760          */
761         protected void loadHMMPool(boolean useCDUnits, InputStream inputStream,
762                 String path) throws IOException {
763             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream,
764                     '#', false);
765     
766             logger.fine("Loading HMM file from: " + path);
767     
768             est.expectString(MODEL_VERSION);
769     
770             int numBase = est.getInt("numBase");
771             est.expectString("n_base");
772     
773             int numTri = est.getInt("numTri");
774             est.expectString("n_tri");
775     
776             int numStateMap = est.getInt("numStateMap");
777             est.expectString("n_state_map");
778     
779             int numTiedState = est.getInt("numTiedState");
780             est.expectString("n_tied_state");
781     
782             int numContextIndependentTiedState = est
783                     .getInt("numContextIndependentTiedState");
784             est.expectString("n_tied_ci_state");
785     
786             int numTiedTransitionMatrices = est.getInt("numTiedTransitionMatrices");
787             est.expectString("n_tied_tmat");
788     
789             int numStatePerHMM = numStateMap / (numTri + numBase);
790     
791             assert numTiedState == mixtureWeightsPool.getFeature(NUM_SENONES, 0);
792             assert numTiedTransitionMatrices == matrixPool.size();
793     
794             // Load the base phones
795             for (int i = 0; i < numBase; i++) {
796                 String name = est.getString();
797                 String left = est.getString();
798                 String right = est.getString();
799                 String position = est.getString();
800                 String attribute = est.getString();
801                 int tmat = est.getInt("tmat");
802     
803                 int[] stid = new int[numStatePerHMM - 1];
804     
805                 for (int j = 0; j < numStatePerHMM - 1; j++) {
806                     stid[j] = est.getInt("j");
807                     assert stid[j] >= 0 && stid[j] < numContextIndependentTiedState;
808                 }
809                 est.expectString("N");
810     
811                 assert left.equals("-");
812                 assert right.equals("-");
813                 assert position.equals("-");
814                 assert tmat < numTiedTransitionMatrices;
815     
816                 Unit unit = unitManager.getUnit(name, attribute.equals(FILLER));
817                 contextIndependentUnits.put(unit.getName(), unit);
818     
819                 if (logger.isLoggable(Level.FINE)) {
820                     logger.fine("Loaded " + unit);
821                 }
822     
823                 // The first filler
824                 if (unit.isFiller() && unit.getName().equals(SILENCE_CIPHONE)) {
825                     unit = UnitManager.SILENCE;
826                 }
827     
828                 float[][] transitionMatrix = matrixPool.get(tmat);
829                 SenoneSequence ss = getSenoneSequence(stid);
830     
831                 HMM hmm = new SenoneHMM(unit, ss, transitionMatrix, HMMPosition
832                         .lookup(position));
833                 hmmManager.put(hmm);
834             }
835     
836             // Load the context dependent phones. If the useCDUnits
837             // property is false, the CD phones will not be created, but
838             // the values still need to be read in from the file.
839     
840             String lastUnitName = "";
841             Unit lastUnit = null;
842             int[] lastStid = null;
843             SenoneSequence lastSenoneSequence = null;
844     
845             for (int i = 0; i < numTri; i++) {
846                 String name = est.getString();
847                 String left = est.getString();
848                 String right = est.getString();
849                 String position = est.getString();
850                 String attribute = est.getString();
851                 int tmat = est.getInt("tmat");
852     
853                 int[] stid = new int[numStatePerHMM - 1];
854     
855                 for (int j = 0; j < numStatePerHMM - 1; j++) {
856                     stid[j] = est.getInt("j");
857                     assert stid[j] >= numContextIndependentTiedState
858                             && stid[j] < numTiedState;
859                 }
860                 est.expectString("N");
861     
862                 assert !left.equals("-");
863                 assert !right.equals("-");
864                 assert !position.equals("-");
865                 assert attribute.equals("n/a");
866                 assert tmat < numTiedTransitionMatrices;
867     
868                 if (useCDUnits) {
869                     Unit unit;
870                     String unitName = (name + ' ' + left + ' ' + right);
871     
872                     if (unitName.equals(lastUnitName)) {
873                         unit = lastUnit;
874                     } else {
875                         Unit[] leftContext = new Unit[1];
876                         leftContext[0] = contextIndependentUnits.get(left);
877     
878                         Unit[] rightContext = new Unit[1];
879                         rightContext[0] = contextIndependentUnits.get(right);
880     
881                         Context context = LeftRightContext.get(leftContext,
882                                 rightContext);
883                         unit = unitManager.getUnit(name, false, context);
884                     }
885                     lastUnitName = unitName;
886                     lastUnit = unit;
887     
888                     if (logger.isLoggable(Level.FINE)) {
889                         logger.fine("Loaded " + unit);
890                     }
891     
892                     float[][] transitionMatrix = matrixPool.get(tmat);
893     
894                     SenoneSequence ss = lastSenoneSequence;
895                     if (ss == null || !sameSenoneSequence(stid, lastStid)) {
896                         ss = getSenoneSequence(stid);
897                     }
898                     lastSenoneSequence = ss;
899                     lastStid = stid;
900     
901                     HMM hmm = new SenoneHMM(unit, ss, transitionMatrix, HMMPosition
902                             .lookup(position));
903                     hmmManager.put(hmm);
904                 }
905             }
906     
907             est.close();
908         }

910         /**
911          * Returns true if the given senone sequence IDs are the same.
912          * 
913          * @return true if the given senone sequence IDs are the same, false
914          *         otherwise
915          */
916         protected boolean sameSenoneSequence(int[] ssid1, int[] ssid2) {
917             if (ssid1.length == ssid2.length) {
918                 for (int i = 0; i < ssid1.length; i++) {
919                     if (ssid1[i] != ssid2[i]) {
920                         return false;
921                     }
922                 }
923                 return true;
924             } else {
925                 return false;
926             }
927         }

944         /**
945          * Loads the mixture weights.
946          * 
947          * @param path
948          *            the path to the mixture weight file
949          * @param floor
950          *            the minimum mixture weight allowed
951          * @return a pool of mixture weights
952          * @throws FileNotFoundException
953          *             if a file cannot be found
954          * @throws IOException
955          *             if an error occurs while loading the data
956          */
957         private Pool<float[]> loadMixtureWeightsAscii(String path, float floor)
958                 throws IOException, URISyntaxException {
959             logger.fine("Loading mixture weights from: " + path);
960             InputStream inputStream = getDataStream(path);
961             if (inputStream == null) {
962                 throw new FileNotFoundException("Error trying to read file " + path);
963             }
964     
965             Pool<float[]> pool = new Pool<float[]>(path);
966             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream,
967                     '#', false);
968             est.expectString("mixw");
969             int numStates = est.getInt("numStates");
970             int numStreams = est.getInt("numStreams");
971             int numGaussiansPerState = est.getInt("numGaussiansPerState");
972             pool.setFeature(NUM_SENONES, numStates);
973             pool.setFeature(NUM_STREAMS, numStreams);
974             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
975             for (int i = 0; i < numStates; i++) {
976                 est.expectString("mixw");
977                 est.expectString("[" + i);
978                 est.expectString("0]");
979                 // float total = est.getFloat("total");
980                 float[] logMixtureWeight = new float[numGaussiansPerState];
981                 for (int j = 0; j < numGaussiansPerState; j++) {
982                     float val = est.getFloat("mixwVal");
983                     if (val < floor) {
984                         val = floor;
985                     }
986                     logMixtureWeight[j] = val;
987                 }
988                 logMath.linearToLog(logMixtureWeight);
989                 pool.put(i, logMixtureWeight);
990             }
991             est.close();
992             return pool;
993         }

995         /**
996          * Loads the mixture weights (Binary).
997          * 
998          * @param path
999          *            the path to the mixture weight file
1000          * @param floor
1001          *            the minimum mixture weight allowed
1002          * @return a pool of mixture weights
1003          * @throws FileNotFoundException
1004          *             if a file cannot be found
1005          * @throws IOException
1006          *             if an error occurs while loading the data
1007          */
1008         private Pool<float[]> loadMixtureWeightsBinary(String path, float floor)
1009                 throws IOException, URISyntaxException {
1010             logger.fine("Loading mixture weights from: " + path);
1011     
1012             Properties props = new Properties();
1013     
1014             DataInputStream dis = readS3BinaryHeader(path, props);
1015     
1016             String version = props.getProperty("version");
1017     
1018             if (version == null || !version.equals(MIXW_FILE_VERSION)) {
1019                 throw new IOException("Unsupported version in " + path);
1020             }
1021     
1022             String checksum = props.getProperty("chksum0");
1023             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
1024             resetChecksum();
1025     
1026             Pool<float[]> pool = new Pool<float[]>(path);
1027     
1028             int numStates = readInt(dis);
1029             int numStreams = readInt(dis);
1030             int numGaussiansPerState = readInt(dis);
1031             int numValues = readInt(dis);
1032     
1033             assert numValues == numStates * numStreams * numGaussiansPerState;
1034             assert numStreams == 1;
1035     
1036             pool.setFeature(NUM_SENONES, numStates);
1037             pool.setFeature(NUM_STREAMS, numStreams);
1038             pool.setFeature(NUM_GAUSSIANS_PER_STATE, numGaussiansPerState);
1039     
1040             for (int i = 0; i < numStates; i++) {
1041                 float[] logMixtureWeight = readFloatArray(dis, numGaussiansPerState);
1042                 Utilities.normalize(logMixtureWeight);
1043                 Utilities.floorData(logMixtureWeight, floor);
1044                 logMath.linearToLog(logMixtureWeight);
1045                 pool.put(i, logMixtureWeight);
1046             }
1047             
1048             validateChecksum(dis, doCheckSum);
1049             
1050             dis.close();
1051             return pool;
1052         }

1054         /**
1055          * Loads the transition matrices.
1056          * 
1057          * @param path
1058          *            the path to the transitions matrices
1059          * @return a pool of transition matrices
1060          * @throws FileNotFoundException
1061          *             if a file cannot be found
1062          * @throws IOException
1063          *             if an error occurs while loading the data
1064          */
1065         protected Pool<float[][]> loadTransitionMatricesAscii(String path)
1066                 throws IOException, URISyntaxException {
1067             logger.fine("Loading transition matrices from: " + path);
1068     
1069             InputStream inputStream = getDataStream(path);
1070             if (inputStream == null) {
1071                 throw new FileNotFoundException("Error trying to read file " + path);
1072             }
1073     
1074             Pool<float[][]> pool = new Pool<float[][]>(path);
1075             ExtendedStreamTokenizer est = new ExtendedStreamTokenizer(inputStream,
1076                     '#', false);
1077             est.expectString("tmat");
1078             int numMatrices = est.getInt("numMatrices");
1079             int numStates = est.getInt("numStates");
1080             logger.fine("with " + numMatrices + " and " + numStates
1081                     + " states, in " + (sparseForm ? "sparse" : "dense") + " form");
1082     
1083             // read in the matrices
1084             for (int i = 0; i < numMatrices; i++) {
1085                 est.expectString("tmat");
1086                 est.expectString("[" + i + ']');
1087                 float[][] tmat = new float[numStates][numStates];
1088                 for (int j = 0; j < numStates; j++) {
1089                     for (int k = 0; k < numStates; k++) {
1090                         // the last row is just zeros, so we just do
1091                         // the first (numStates - 1) rows
1092                         if (j < numStates - 1) {
1093                             if (sparseForm) {
1094                                 if (k == j || k == j + 1) {
1095                                     tmat[j][k] = est.getFloat("tmat value");
1096                                 }
1097                             } else {
1098                                 tmat[j][k] = est.getFloat("tmat value");
1099                             }
1100                         }
1101                         tmat[j][k] = logMath.linearToLog(tmat[j][k]);
1102                         if (logger.isLoggable(Level.FINE)) {
1103                             logger.fine("tmat j " + j + " k " + k + " tm "
1104                                     + tmat[j][k]);
1105                         }
1106                     }
1107                 }
1108                 pool.put(i, tmat);
1109             }
1110             est.close();
1111             return pool;
1112         }

1114         /**
1115          * Loads the transition matrices (Binary).
1116          * 
1117          * @param path
1118          *            the path to the transitions matrices
1119          * @return a pool of transition matrices
1120          * @throws FileNotFoundException
1121          *             if a file cannot be found
1122          * @throws IOException
1123          *             if an error occurs while loading the data
1124          */
1125         protected Pool<float[][]> loadTransitionMatricesBinary(String path)
1126                 throws IOException, URISyntaxException {
1127             logger.fine("Loading transition matrices from: " + path);
1128     
1129             Properties props = new Properties();
1130             DataInputStream dis = readS3BinaryHeader(path, props);
1131     
1132             String version = props.getProperty("version");
1133     
1134             if (version == null || !version.equals(TMAT_FILE_VERSION)) {
1135                 throw new IOException("Unsupported version in " + path);
1136             }
1137     
1138             String checksum = props.getProperty("chksum0");
1139             boolean doCheckSum = (checksum != null && checksum.equals("yes"));
1140             resetChecksum();
1141     
1142             Pool<float[][]> pool = new Pool<float[][]>(path);
1143     
1144             int numMatrices = readInt(dis);
1145             int numRows = readInt(dis);
1146             int numStates = readInt(dis);
1147             int numValues = readInt(dis);
1148     
1149             assert numValues == numStates * numRows * numMatrices;
1150     
1151             for (int i = 0; i < numMatrices; i++) {
1152                 float[][] tmat = new float[numStates][];
1153                 // last row should be zeros
1154                 tmat[numStates - 1] = new float[numStates];
1155                 logMath.linearToLog(tmat[numStates - 1]);
1156     
1157                 for (int j = 0; j < numRows; j++) {
1158                     tmat[j] = readFloatArray(dis, numStates);
1159                     Utilities.nonZeroFloor(tmat[j], 0f);
1160                     Utilities.normalize(tmat[j]);
1161                     logMath.linearToLog(tmat[j]);
1162                 }
1163                 pool.put(i, tmat);
1164             }
1165     
1166             validateChecksum(dis, doCheckSum);
1167     
1168             dis.close();
1169             return pool;
1170         }

1189             try {
1190                 dis = readS3BinaryHeader(path, props);

1194                 return null;
