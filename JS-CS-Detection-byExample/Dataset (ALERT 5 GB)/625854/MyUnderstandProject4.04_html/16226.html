14     package edu.cmu.sphinx.jsapi;
15     
16     import java.util.Locale;
17     
18     import javax.speech.Engine;
19     import javax.speech.EngineCreate;
20     import javax.speech.EngineException;
21     import javax.speech.recognition.RecognizerModeDesc;
22     
23     /**
24      * Provides information about a specific operating mode of a sphinx recognition
25      * engine.
26      */
27     public class SphinxRecognizerModeDesc extends RecognizerModeDesc
28             implements EngineCreate {
29     
30         /** Name of the sphinx configuration file if any. */
31         private final String configFile;
32     
33         /**
34          * Constructs a new object. 
35          */
36         public SphinxRecognizerModeDesc() {
37             super("Sphinx 4",      // engine name
38                   null,            // mode name
39                   Locale.US,
40                   Boolean.FALSE,   // running?
41                   Boolean.TRUE,    // dictationGrammarSupported
42                   null);           // profile[]
43             configFile = "/sphinx4.config.xml";
44         }
45     
46         /**
47          * Retrieves the configuration file.
48          * @return the name of the configuration file,
49          *          <code>/sphinx4.config.xml</code> if none specified
50          */
51         public String getConfigFile() {
52             return configFile;
53         }
54     
55         /**
56          * Constructs a new object.
57          * @param config name of the configuration file as a resource. 
58          */
59         public SphinxRecognizerModeDesc(String config) {
60             super("Sphinx 4",      // engine name
61                   null,            // mode name
62                   Locale.US,
63                   Boolean.FALSE,   // running?
64                   Boolean.TRUE,    // dictationGrammarSupported
65                   null);           // profile[]
66             configFile = config;
67         }
68     
69         /**
70          * Creates an engine with the properties specified by this
71          * SphinxRecognizerModeDesc
72          *
73          * @return a SphinxRecognizer
74          * @throws IllegalArgumentException if the properties of the SphinxRecognizerModeDesc do not refer to a known engine
75          *                                  or engine mode.
76          * @throws EngineException          if the engine could not be properly created.
77          * @throws SecurityException        if the caller does not have permission to create an engine.
78          */
79         public Engine createEngine()
80                 throws IllegalArgumentException, EngineException, SecurityException {
81             return new SphinxRecognizer(this);
82         }
83     }
