6     /**
7      * Provides a simple object cache.
8      *
9      * <p>Object stored in cache must properly implement {@link Object#hashCode hashCode} and {@link Object#equals equals}.
10      *
11      * <p><strong>Note that this class is not synchronized.</strong>
12      * If multiple threads access a cache concurrently, and at least one of
13      * the threads modifies the cache, it <i>must</i> be synchronized externally.
14      * This is typically accomplished by synchronizing on some object that
15      * naturally encapsulates the cache.
16      */
17     public class Cache<T> {
18     
19         private Map<T, T> map = new HashMap<T, T>();
20     
21         private int hits = 0;
22     
23         /**
24          * Puts the given object in the cache if it is not already present.
25          *
26          * <p>If the object is already cached, than the instance that exists in the cached is returned.
27          * Otherwise, it is placed in the cache and null is returned.
28          *
29          * @param object object to cache
30          * @return the cached object or null if the given object was not already cached
31          */
32         public T cache(T object) {
33             T result = map.get(object);
34             if (result == null) {
35                 map.put(object, object);
36             } else {
37                 hits++;
38             }
39             return result;
40         }
41     
42         /**
43          * Returns the number of cache hits, which is the number of times {@link #cache} was called
44          * and returned an object that already existed in the cache.
45          *
46          * @return the number of cache hits
47          */
48         public int getHits() {
49             return hits;
50         }
51     
52         /**
53          * Returns the number of cache misses, which is the number of times {@link #cache} was called
54          * and returned null (after caching the object), effectively representing the size of the cache. 
55          *
56          * @return the number of cache misses
57          */
58         public int getMisses() {
59             return map.size();
60         }
61     
62     }
