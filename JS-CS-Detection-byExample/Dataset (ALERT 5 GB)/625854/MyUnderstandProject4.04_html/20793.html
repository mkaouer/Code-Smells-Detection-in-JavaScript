13     package edu.cmu.sphinx.frontend.filter;
14     
15     import edu.cmu.sphinx.frontend.*;
16     import edu.cmu.sphinx.util.props.PropertyException;
17     import edu.cmu.sphinx.util.props.PropertySheet;
18     import edu.cmu.sphinx.util.props.S4Boolean;
19     import edu.cmu.sphinx.util.props.S4Double;
20     
21     import static java.lang.Math.max;
22     import static java.lang.Math.min;
23     import java.util.Random;
24     
25     /**
26      * Implements a dither for the incoming packet. The small random noise is added
27      * to the signal to avoid floating point errors and prevent the energy from
28      * being zero. <p/> Other {@link Data}objects are passed along unchanged through
29      * this Dither processor. <p/>
30      */
31     public class Dither extends BaseDataProcessor {
32     
33         /** The maximal value which could be added/substracted to/from the signal*/
34         @S4Double(defaultValue = 2.0)
35         public static final String PROP_MAX_DITHER = "maxDither";
36         private double ditherMax;
37     
38         /** The maximal value of dithered values. */
39         @S4Double(defaultValue = Double.MAX_VALUE)
40         public static final String PROP_MAX_VAL = "upperValueBound";
41         private double maxValue;
42     
43         /** The minimal value of dithered values. */
44         @S4Double(defaultValue = -Double.MAX_VALUE)
45         public static final String PROP_MIN_VAL = "lowerValueBound";
46         private double minValue;
47     
48     
49         /** The property about using random seed or not */
50         @S4Boolean(defaultValue = false)
51         public static final String PROP_USE_RANDSEED = "useRandSeed";
52         private boolean useRandSeed;
53     
54         public Dither(  double ditherMax, boolean useRandSeed, double maxValue, double minValue) {
55             initLogger();
56     
57             this.ditherMax = ditherMax;
58             this.useRandSeed = useRandSeed;
59     
60             this.maxValue = maxValue;
61             this.minValue = minValue;
62         }
63     
64         public Dither( ) {
65     
66         }
67     
68         @Override
69         public void newProperties(PropertySheet ps) throws PropertyException {
70             super.newProperties(ps);
71     
72             ditherMax = ps.getDouble(PROP_MAX_DITHER);
73             useRandSeed = ps.getBoolean(PROP_USE_RANDSEED);
74     
75             maxValue = ps.getDouble(PROP_MAX_VAL);
76             minValue = ps.getDouble(PROP_MIN_VAL);
77         }
78     
79         @Override
80         public void initialize() {
81             super.initialize();
82         }
83     
84     
85         /**
86          * Returns the next DoubleData object, which is a dithered version of the input
87          *
88          * @return the next available DoubleData object, or null if no Data is available
89          * @throws edu.cmu.sphinx.frontend.DataProcessingException
90          *          if a data processing error occurred
91          */
92         @Override
93         public Data getData() throws DataProcessingException {
94             Data input = getPredecessor().getData(); // get the spectrum
95             if (input != null && ditherMax != 0) {
96                 if (input instanceof DoubleData || input instanceof FloatData) {
97                     input = process(input);
98                 }
99             }
100             return input;
101         }
102     
103     
104         /**
105          * Process data, adding dither
106          *
107          * @param input a frame
108          * @return processed frame
109          * @throws IllegalArgumentException
110          */
111         private DoubleData process(Data input) throws IllegalArgumentException {
112             DoubleData output;
113             Random r;
114             if (useRandSeed)
115                 r = new Random();
116             else
117                 r = new Random(12345);
118     
119             assert input instanceof DoubleData;
120             double[] inFeatures;
121     
122             DoubleData doubleData = (DoubleData) input;
123             inFeatures = doubleData.getValues();
124             double[] outFeatures = new double[inFeatures.length];
125             for (int i = 0; i < inFeatures.length; ++i) {
126                 outFeatures[i] = r.nextFloat() * 2 * ditherMax - ditherMax + inFeatures[i];
127                 outFeatures[i] = max(min(outFeatures[i], maxValue), minValue);
128             }
129     
130             output = new DoubleData(outFeatures, doubleData.getSampleRate(),
131                     doubleData.getCollectTime(),
132                     doubleData.getFirstSampleNumber());
133     
134             return output;
135         }
136     }
