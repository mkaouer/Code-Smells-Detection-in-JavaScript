20     package edu.cmu.sphinx.tools.gui.writer;
21     import edu.cmu.sphinx.tools.gui.ConfigProperties;
22     
23     import java.io.BufferedWriter;
24     import java.io.FileWriter;
25     import java.io.IOException;
26     import java.io.PrintWriter;
27     import java.io.File;
28     
29     /**
30      * This class is a type of <code>GUIWriter</code>
31      * that write output to .config.xml format
32      *
33      * @author Ariani
34      */
35     public class XMLConfigWriter implements GUIWriter{
36     
37          /* this class uses Singleton pattern for the constructor */
38           private XMLConfigWriter() {}
39     
40           private static class XMLWriterHolder {
41               private static final XMLConfigWriter instance = new XMLConfigWriter();
42           }
43           
44           /** 
45            * Singleton constructor of <code>XMLConfigWriter</code> ; 
46            * only one instance is created for the whole system
47            *
48            * @return reference to <code>XMLConfigWriter</code>
49            */
50           public static XMLConfigWriter getInstance(){
51               return XMLWriterHolder.instance;
52           }
53     
54           /**
55            * This method is inherited from <code>GUIWriter</code> interface.
56            * Will write output to .config.xml format 
57            *
58            * @param configProp Holds the property name-values
59            * @param fFile Output file
60            * @throws GUIWriterException writing error
61            */
62         @Override
63         public boolean writeOutput(ConfigProperties configProp, File fFile) throws GUIWriterException{
64             if (fFile == null || fFile.getName().trim().isEmpty()) /* no filename for output */
65             {
66                 throw new GUIWriterException
67                         ("No output filename specified",GUIWriterException.EXCEPTION_NO_FILENAME);
68             }
69             else /* start writing to config file */
70             {
71                 
72                 try{
73                     ConfigConverter cc = ConfigConverter.getInstance();
74                     StringBuilder sb = cc.writeOutput(configProp);
75                     PrintWriter pw = new PrintWriter(new BufferedWriter(new FileWriter(fFile)));                
76                     pw.print(sb);
77                     pw.flush();
78                     pw.close();
79                     return true;
80                 }catch(IOException e)
81                 {
82                     throw new GUIWriterException
83                             ("IOException thrown during file write",GUIWriterException.EXCEPTION_IO);
84                 }
85                 
86             }        
87         }
88         
89         /**
90            * This method is inherited from <code>GUIWriter</code> interface.
91            * Will return configuration values for output 
92            *
93            * @param configProp Holds the property name-values
94            * @return String configuration value as text
95            * @throws GUIWriterException writing error
96            */
97        @Override
98        public String getOutput(ConfigProperties configProp) throws GUIWriterException{
99             ConfigConverter cc = ConfigConverter.getInstance();
100             StringBuilder sb = cc.writeOutput(configProp);
101             return sb.toString();
102        }
103         
104     }
