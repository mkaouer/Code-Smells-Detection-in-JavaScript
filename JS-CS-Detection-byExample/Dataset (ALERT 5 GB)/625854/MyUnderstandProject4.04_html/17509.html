13     package edu.cmu.sphinx.decoder.search;
14     
15     
16     /**
17      * Partitions a list of tokens according to the token score.
18      */
19     public class Partitioner {
20     
21         /**
22          * Partitions sub-array of tokens around the rth token.
23          *
24          * @param tokens the token array to partition
25          * @param p      the starting index of the subarray
26          * @param r      the pivot and the ending index of the subarray, inclusive
27          * @return the index (after partitioning) of the element around which the array is partitioned
28          */
29         private int partitions(Token[] tokens, int p, int r) {
30     
31             Token pivot = tokens[r];
32             int i = p - 1;
33             for (int j = p; j < r; j++) {
34                 Token current = tokens[j];
35                 if (current.getScore() >= pivot.getScore()) {
36                     i++;
37                     setToken(tokens, j, tokens[i]);
38                     setToken(tokens, i, current);
39                 }
40             }
41             i++;
42             setToken(tokens, r, tokens[i]);
43             setToken(tokens, i, pivot);
44             return i;
45         }
46     
47     
48         /**
49          * Partitions sub-array of tokens around the x-th token by selecting the midpoint of the token array as the pivot.
50          *
51          * @param tokens the token array to partition
52          * @param p      the starting index of the subarray
53          * @param r      the ending index of the subarray, inclusive
54          * @return the index of the element around which the array is partitioned
55          */
56         private int midPointPartition(Token[] tokens, int p, int r) {
57             int i = (p + r) >>> 1;
58             Token temp = tokens[r];
59             setToken(tokens, r, tokens[i]);
60             setToken(tokens, i, temp);
61             return partitions(tokens, p, r);
62         }
63     
64     
65         /**
66          * Partitions the given array of tokens in place, so that the highest scoring n token will be at the beginning of
67          * the array, not in any order.
68          *
69          * @param tokens the array of tokens to partition
70          * @param size   the number of tokens to partition
71          * @param n      the number of tokens in the final partition
72          * @return the index of the last element in the partition
73          */
74         public int partition(Token[] tokens, int size, int n) {
75             if (tokens.length > n) {
76                 return midPointSelect(tokens, 0, size - 1, n);
77             } else {
78                 int r = -1;
79                 float lowestScore = Float.MAX_VALUE;
80                 for (int i = 0; i < tokens.length; i++) {
81                     Token current = tokens[i];
82                     float currentScore = current.getScore();
83                     if (currentScore <= lowestScore) {
84                         lowestScore = currentScore;
85                         r = i; // "r" is the returned index
86                     }
87                 }
88     
89                 // exchange tokens[r] <=> last token,
90                 // where tokens[r] has the lowest score
91                 int last = size - 1;
92                 if (last >= 0) {
93                     Token lastToken = tokens[last];
94                     setToken(tokens, last, tokens[r]);
95                     setToken(tokens, r, lastToken);
96                 }
97     
98                 // return the last index
99                 return last;
100             }
101         }
102     
103     
104         private void setToken(Token[] list, int index, Token token) {
105             list[index] = token;
106             token.setLocation(index);
107         }
108     
109     
110         /**
111          * Selects the token with the ith largest token score.
112          *
113          * @param tokens the token array to partition
114          * @param p      the starting index of the subarray
115          * @param r      the ending index of the subarray, inclusive
116          * @param i      the token with the i-th largest score
117          * @return the index of the token with the ith largest score
118          */
119         private int midPointSelect(Token[] tokens, int p, int r, int i) {
120             if (p == r) {
121                 return p;
122             }
123             int q = midPointPartition(tokens, p, r);
124             int k = q - p + 1;
125             if (i == k) {
126                 return q;
127             } else if (i < k) {
128                 return midPointSelect(tokens, p, q - 1, i);
129             } else {
130                 return midPointSelect(tokens, q + 1, r, i - k);
131             }
132         }
133     }
