123         /*
124          * Actual implementation of the class-based LM:
125          * P=P(W|C1)*P(C1|C2,C3..)
126          *
127          * @see edu.cmu.sphinx.linguist.language.ngram.LanguageModel#getProbability(edu.cmu.sphinx.linguist.WordSequence)
128          */
129         @Override
130         public float getProbability(WordSequence wordSequence) {
131             Word[] classes = new Word[wordSequence.size()];
132             float wordToClassProb = 0;
133             for (int i = 0; i < classes.length; i++) {
134                 Word sourceWord = wordSequence.getWord(i);
135                 ClassProbability classProbability = classMap.getClassProbability(sourceWord.getSpelling());
136                 classes[i] = (classProbability == null ? sourceWord : classMap.getClassAsWord(classProbability.getClassName()));
137                 if (i == classes.length - 1) {
138                     if (classProbability != null) {
139                         // the first word of the word sequence is a class
140                         wordToClassProb = classProbability.getLogProbability();
141                     }
142                 }
143             }
144             float classBasedProbability = classLM.getProbability(new WordSequence(classes));
145             return classBasedProbability + wordToClassProb;
146         }

178         /**
179          * Converts a vocabulary of the class LM to a word vocabulary.
180          */
181         private void makeVocabulary() {
182             vocabulary = new HashSet<String>();
183             for (String name : classLM.getVocabulary()) {
184                 Set<String> wordsInClass = classMap.getWordsInClass(name);
185                 if (wordsInClass == null) {
186                     // 'name' not a class
187                     vocabulary.add(name);
188                 } else {
189                     vocabulary.addAll(wordsInClass);
190                 }
191             }
192         }
