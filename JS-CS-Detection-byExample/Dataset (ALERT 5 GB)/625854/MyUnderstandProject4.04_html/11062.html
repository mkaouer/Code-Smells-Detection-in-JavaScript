214         @Override
215         @SuppressWarnings("unchecked")
216         public void allocate() throws IOException {
217             TimerPool.getTimer(this, "LM Load").start();
218             
219             // create the log file if specified
220             if (ngramLogFile != null)
221                 logFile = new PrintWriter(new FileOutputStream(ngramLogFile));
222             
223             loader = new BinaryLoader(format, location, applyLanguageWeightAndWip,
224                     logMath, languageWeight, wip, unigramWeight);
225             
226             maxNGramCacheSize = new int[loader.getMaxDepth()];
227             maxNGramCacheSize[1] = maxBigramCacheTmp;
228             
229             for (int i = 2; i < loader.getMaxDepth(); i++)
230     			maxNGramCacheSize[i] = maxNGramCacheTmp;
231             
232             unigramIDMap = new HashMap<Word, UnigramProbability>();
233             unigrams = loader.getUnigrams();
234             loadedNGramBuffers = new Map[loader.getMaxDepth()];
235             NGramCache = new LRUCache[loader.getMaxDepth()];
236             NGramProbTable = new float[loader.getMaxDepth()][];
237             NGramBackoffTable = new float[loader.getMaxDepth()][];
238             NGramSegmentTable = new int[loader.getMaxDepth()][];
239             NGramHit = new int[loader.getMaxDepth()];
240             NGramMisses = new int[loader.getMaxDepth()];
241     
242             for (int i = 1; i <= loader.getMaxDepth(); i++) {
243             	loadedNGramBuffers[i - 1] = new HashMap<WordSequence, NGramBuffer>();
244               NGramCache[i - 1] = new LRUCache<WordSequence, NGramProbability>(maxNGramCacheSize[i - 1]);
245     
246               if (i >= 2) 
247                 NGramProbTable[i - 1] = loader.getNGramProbabilities(i);
248                 
249                 if (i > 2) {
250                 	NGramBackoffTable[i - 1] = loader.getNGramBackoffWeights(i);
251                     NGramSegmentTable[i - 1] = loader.getNGramSegments(i);
252                 }
253             }
254             
255             NMaxCache = new LRUCache<WordSequence, Float>(maxNGramCacheSize[loader.getMaxDepth() - 1]);
256             buildUnigramIDMap(dictionary);
257             loadedBigramBuffers = new NGramBuffer[unigrams.length];
258     
259             if (maxDepth <= 0 || maxDepth > loader.getMaxDepth())
260                 maxDepth = loader.getMaxDepth();
261     
262             for (int i = 1; i <= loader.getMaxDepth(); i++)
263             	logger.info(Integer.toString(i) + "-grams: " + loader.getNumberNGrams(i));
264     
265             if (fullSmear) {
266                 System.out.println("Full Smear");
267                 try {
268                     System.out.println("... Reading ...");
269                     readSmearInfo("smear.dat");
270                     System.out.println("... Done ");
271                 } catch (IOException e) {
272                     System.out.println("... " + e);
273                     System.out.println("... Calculating");
274                     buildSmearInfo();
275                     System.out.println("... Writing");
276                     // writeSmearInfo("smear.dat");
277                     System.out.println("... Done");
278                 }
279             }
280             
281             TimerPool.getTimer(this,"LM Load").stop();
282         }

296         /** Builds the map from unigram to unigramID. Also finds the startWordID and endWordID.
297          * @param dictionary
298          * */
299         private void buildUnigramIDMap(Dictionary dictionary) {
300             int missingWords = 0;
301             String[] words = loader.getWords();
302             for (int i = 0; i < words.length; i++) {
303                 Word word = dictionary.getWord(words[i]);
304     
305                 if (word == null) {
306                     logger.info("Missing word: " + words[i]);
307                     missingWords++;
308                 }
309                 
310                 unigramIDMap.put(word, unigrams[i]);
311                 
312                 if (logger.isLoggable(Level.FINE))
313                     logger.fine("Word: " + word);
314             }
315     
316             if (missingWords > 0)
317                 logger.warning("Dictionary is missing " + missingWords + " words that are contained in the language model.");
318         }

341         /** Clears the various N-gram caches. */
342         private void clearCache() {
343             for (int i = 0; i < loadedBigramBuffers.length; i++) {
344                 NGramBuffer buffer = loadedBigramBuffers[i];
345                 
346                 if (buffer != null) {
347                     if (!buffer.getUsed())
348                         loadedBigramBuffers[i] = null; // free the BigramBuffer
349                     else
350                         buffer.setUsed(false);
351                 }
352             }
353     
354             loadedBigramBuffers = new NGramBuffer[unigrams.length];
355             
356             for (int i = 2; i <= loader.getMaxDepth(); i++) {
357                 loadedNGramBuffers[i - 1] = new HashMap<WordSequence, NGramBuffer>();
358             	logger.info("LM Cache: " + Integer.toString(i) + "g " + NGramCache[i - 1].size());
359             	
360             	if (clearCacheAfterUtterance) {
361             		NGramCache[i - 1] = new LRUCache<WordSequence, NGramProbability>(maxNGramCacheSize[i - 1]);
362             		NMaxCache = new LRUCache<WordSequence, Float>(maxNGramCacheSize[loader.getMaxDepth()]);
363             	}
364             }
365         }

382          * Gets the ngram probability of the word sequence represented by the word list
383          *
384          * @param wordSequence the word sequence
385          * @return the probability of the word sequence. Probability is in logMath log base
386          */
387         @Override
388         public float getProbability(WordSequence wordSequence) {
389             int numberWords = wordSequence.size();
390             
391             if (numberWords <= maxDepth) {
392             	if (numberWords >= 2) {
393                 if (logFile != null) 
394                   logFile.println(wordSequence.toString().replace("][", " ") + " : " + Float.toString(getNGramProbability(wordSequence)));
395             		return getNGramProbability(wordSequence);
396               }
397             	else {
398                 if (logFile != null) 
399                   logFile.println(wordSequence.toString().replace("][", " ") + " : " + Float.toString(getUnigramProbability(wordSequence)));
400                 return getUnigramProbability(wordSequence);
401               }
402             }
403             
404             throw new Error("Unsupported NGram: " + wordSequence.size());
405         }

408         /**
409          * Returns the NGram probability of the given NGram.
410          *
411          * @param wordSequence the NGram word sequence
412          * @return the NGram probability
413          */
414         private float getNGramProbability(WordSequence wordSequence) {
415         	int numberWords = wordSequence.size();
416         	Word firstWord = wordSequence.getWord(0);
417         	Float floatProbability = null;
418         	NGramProbability nGProbability = null;  
419     
420             if (loader.getNumberNGrams(numberWords) == 0 || !hasUnigram(firstWord))
421                 return getNGramProbability(wordSequence.getNewest());
422     
423             if (numberWords == maxDepth)
424             	floatProbability = NMaxCache.get(wordSequence);
425             
426             if (floatProbability == null) {
427                 float score = 0.0f;
428     
429                 nGProbability = findNGram(wordSequence);
430     
431                 if (nGProbability != null) {
432                     NGramHit[numberWords - 1]++;
433                     score = NGramProbTable[numberWords - 1][nGProbability.getProbabilityID()];
434                 } 
435                 else {
436                     NGramMisses[numberWords - 1]++;
437                     
438                     if (wordSequence.size() == 2) {
439                       UnigramProbability unigramProb = getUnigram(wordSequence.getWord(0));
440                       score = unigramProb.getLogBackoff() + getProbability(wordSequence.getNewest());
441                     }
442                     else {
443                       NGramProbability nMinus1Gram = findNGram(wordSequence.getOldest());
444                     
445                       if (nMinus1Gram != null)
446                         score = NGramBackoffTable[numberWords - 1][nMinus1Gram.getBackoffID()] + getProbability(wordSequence.getNewest());
447                       else
448                           score = getProbability(wordSequence.getNewest());
449                    }
450                 }
451                 
452                 floatProbability = score;
453                 
454                 if (numberWords == maxDepth)
455                 	NMaxCache.put(wordSequence, floatProbability);
456             } 
457     
458             return floatProbability.floatValue();
459         }

462         /**
463          * Finds or loads the NGram probability of the given NGram.
464          *
465          * @param wordSequence the NGram to load
466          * @return a NGramProbability of the given NGram
467          */
468         private NGramProbability findNGram(WordSequence wordSequence) {
469         	int numberWords = wordSequence.size();
470             NGramProbability nGram = null; 
471             WordSequence oldest = wordSequence.getOldest();
472             NGramBuffer nGramBuffer = loadedNGramBuffers[numberWords - 1].get(oldest);
473     
474             if (nGramBuffer == null) {
475             	nGramBuffer = getNGramBuffer(oldest);
476     
477                 if (nGramBuffer != null)
478                     loadedNGramBuffers[numberWords - 1].put(oldest, nGramBuffer);
479             }
480     
481             if (nGramBuffer != null) {
482                 int nthWordID = getWordID(wordSequence.getWord(numberWords - 1));
483                 nGram = nGramBuffer.findNGram(nthWordID);
484             }
485     
486             return nGram;
487         }

490         /**
491          * Loads into a buffer all the NGram followers of the given N-1Gram.
492          * 
493          * @param ws the N-1Gram to find followers
494          * 
495          * @return a NGramBuffer of all the NGram followers of the given sequence
496          */
497         private NGramBuffer loadNGramBuffer(WordSequence ws) { 
498         	int firstWordID = getWordID(ws.getWord(0));
499         	int firstCurrentNGramEntry = 0;
500         	int numberNGrams = 0;
501         	int size = 0;
502     		long position = 0;
503     		int orderBuffer = ws.size() + 1;
504     		NGramBuffer currentBuffer = null;
505     		NGramBuffer nMinus1Buffer = null;
506     
507     		firstCurrentNGramEntry = unigrams[firstWordID].getFirstBigramEntry(); 
508     		numberNGrams = getNumberBigramFollowers(firstWordID) + 1;
509     		
510     		if (numberNGrams == 1) // 1 means that there is no bigram starting with firstWordID
511     			return null;
512             
513     		if (orderBuffer == 2)  {
514     			size = numberNGrams * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField();
515     			position = (long) (loader.getNGramOffset(orderBuffer) + (firstCurrentNGramEntry * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField()));
516     		}
517         
518         
519     		else { // only for ws.size() >= 2
520     			int lastWordId = getWordID(ws.getWord(ws.size() - 1));
521     			nMinus1Buffer = getNGramBuffer(ws.getOldest());
522     			int index = nMinus1Buffer.findNGramIndex(lastWordId);
523     
524     			if (index == -1)
525     				return null;
526     
527     			int firstNMinus1GramEntry = nMinus1Buffer.getFirstNGramEntry();
528     			firstCurrentNGramEntry = getFirstNGramEntry(nMinus1Buffer.getNGramProbability(index), firstNMinus1GramEntry, orderBuffer);
529     			int firstNextNGramEntry = getFirstNGramEntry(nMinus1Buffer.getNGramProbability(index + 1), firstNMinus1GramEntry, orderBuffer);
530     			numberNGrams = firstNextNGramEntry - firstCurrentNGramEntry;
531     			
532     			if (numberNGrams == 0)
533     				return null;
534     			
535     			if (loader.getMaxDepth() != orderBuffer) 
536     				numberNGrams++;
537     
538     			size = numberNGrams * ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * loader.getBytesPerField();
539     			position = (long) loader.getNGramOffset(orderBuffer) + (long) firstCurrentNGramEntry * (long) ((loader.getMaxDepth() == orderBuffer) ? BYTES_PER_NMAXGRAM : BYTES_PER_NGRAM) * (long) loader.getBytesPerField();
540     		}
541     
542     		try {
543     			byte[] buffer = loader.loadBuffer(position, size);
544     			
545     			if (loader.getMaxDepth() == orderBuffer) {
546     				currentBuffer = new NMaxGramBuffer(buffer, numberNGrams, loader.getBigEndian(), is32bits(), orderBuffer, firstCurrentNGramEntry);
547     			}
548     			else
549     				currentBuffer = new NGramBuffer(buffer, numberNGrams, loader.getBigEndian(), is32bits(), orderBuffer, firstCurrentNGramEntry);
550     		} 
551     		catch (IOException ioe) {
552     			ioe.printStackTrace();
553     			throw new Error("Error loading " + orderBuffer + "-Grams.");
554     		}
555     
556     		return currentBuffer;
557         }

560         /**
561          * Returns the NGrams of the given word sequence
562          *
563          * @param wordSequence the word sequence 
564          * 					   from which to get the buffer
565          * @return the NGramBuffer of the word sequence
566          */
567         private NGramBuffer getNGramBuffer(WordSequence wordSequence) {
568         	NGramBuffer nGramBuffer = null;
569         	int order = wordSequence.size();
570         	
571         	if (order > 1)
572         		nGramBuffer = loadedNGramBuffers[order - 1].get(wordSequence); // better when using containsKey
573     
574         	if (nGramBuffer == null) {
575         		nGramBuffer = loadNGramBuffer(wordSequence);
576     
577         		if (nGramBuffer != null) 
578         			loadedNGramBuffers[order - 1].put(wordSequence, nGramBuffer); // optimizable by adding an 'empty' nGramBuffer
579         	}
580     
581         	return nGramBuffer;
582         }

585         /**
586          * Returns the index of the first NGram entry of the given N-1Gram
587          *
588          * @param nMinus1Gram            the N-1Gram which first NGram entry we're looking for
589          * @param firstNMinus1GramEntry  the index of the first N-1Gram entry of the N-1Gram in question
590          * @param n						 the order of the NGram
591          * @return the index of the first NGram entry of the given N-1Gram
592          */
593         private int getFirstNGramEntry(NGramProbability nMinus1Gram, int firstNMinus1GramEntry, int n) {
594             int firstNGramEntry = NGramSegmentTable[n - 1][(firstNMinus1GramEntry + nMinus1Gram.getWhichFollower()) 
595                                                       >> loader.getLogNGramSegmentSize()] 
596                                                       + nMinus1Gram.getFirstNPlus1GramEntry();
597             
598             return firstNGramEntry;
599         }

684         @Override
685         public float getSmear(WordSequence wordSequence) {
686             float smearTerm = 1.0f;
687             
688             if (fullSmear) {
689                 smearCount++;
690                 int length = wordSequence.size();
691                 
692                 if (length == 1) {
693                     int wordID = getWordID(wordSequence.getWord(0));
694                     smearTerm = unigramSmearTerm[wordID];
695                 } 
696                 else if (length >= 2) {
697                     int size = wordSequence.size();
698                     int wordID1 = getWordID(wordSequence.getWord(size - 2));
699                     int wordID2 = getWordID(wordSequence.getWord(size - 1));
700                     Float st = getSmearTerm(wordID1, wordID2);
701                     
702                     if (st == null)
703                         smearTerm = unigramSmearTerm[wordID2];
704                     else {
705                         smearTerm = st;
706                         smearBigramHit++;
707                     }
708                 }
709     
710                 if (smearCount % 100000 == 0)
711                     System.out.println("Smear hit: " + smearBigramHit + " tot: " + smearCount);
712             }
713             
714             if (fullSmear && logger.isLoggable(Level.FINE))
715                 logger.fine("SmearTerm: " + smearTerm);
716     
717             return smearTerm;
718         }

869         private void buildSmearInfo() throws IOException {
870             double S0 = 0;
871             double R0 = 0;
872     
873             bigramSmearMap = new HashMap<Long, Float>();
874     
875             double[] ugNumerator = new double[unigrams.length];
876             double[] ugDenominator = new double[unigrams.length];
877             double[] ugAvgLogProb = new double[unigrams.length];
878     
879             unigramSmearTerm = new float[unigrams.length];
880     
881             for (UnigramProbability unigram : unigrams) {
882                 float logp = unigram.getLogProbability();
883                 double p = logMath.logToLinear(logp);
884                 S0 += p * logp;
885                 R0 += p * logp * logp;
886             }
887     
888             System.out.println("R0 S0 " + R0 + ' ' + S0);
889     
890             for (int i = 0; i < loadedBigramBuffers.length; i++) {
891                 NGramBuffer bigram = getBigramBuffer(i);
892                 
893                 if (bigram == null) {
894                     unigramSmearTerm[i] = LogMath.getLogOne();
895                     continue;
896                 }
897     
898                 ugNumerator[i] = 0.0;
899                 ugDenominator[i] = 0.0;
900                 ugAvgLogProb[i] = 0.0;
901     
902                 float logugbackoff = unigrams[i].getLogBackoff();
903                 double ugbackoff = logMath.logToLinear(logugbackoff);
904     
905                 for (int j = 0; j < bigram.getNumberNGrams(); j++) {
906                     int wordID = bigram.getWordID(j);
907                     NGramProbability bgProb = bigram.getNGramProbability(j);
908     
909                     float logugprob = unigrams[wordID].getLogProbability();
910                     float logbgprob = NGramProbTable[1][bgProb.getProbabilityID()];
911     
912                     double ugprob = logMath.logToLinear(logugprob);
913                     double bgprob = logMath.logToLinear(logbgprob);
914     
915                     double backoffbgprob = ugbackoff * ugprob;
916                     double logbackoffbgprob = logMath.linearToLog(backoffbgprob);
917     
918                     ugNumerator[i] += (bgprob * logbgprob
919                             - backoffbgprob * logbackoffbgprob) * logugprob;
920     
921                     ugDenominator[i] += (bgprob - backoffbgprob) * logugprob;
922                     // dumpProbs(ugNumerator, ugDenominator, i, j, logugprob,
923     				//		logbgprob, ugprob, bgprob, backoffbgprob,
924     				//		logbackoffbgprob);
925                 }
926                 
927                 ugNumerator[i] += ugbackoff * (logugbackoff * S0 + R0);
928                 ugAvgLogProb[i] = ugDenominator[i] + ugbackoff * S0;
929                 ugDenominator[i] += ugbackoff * R0;
930     
931                 // System.out.println("n/d " + ugNumerator[i] + " " +
932                 //                     ugDenominator[i]);
933     
934                 unigramSmearTerm[i] = (float) (ugNumerator[i] / ugDenominator[i]);
935                 /// unigramSmearTerm[i] = 
936                 //   logMath.linearToLog(ugNumerator[i] / ugDenominator[i]);
937                 //  System.out.println("ugs " + unigramSmearTerm[i]);
938             }
939     
940             for (int i = 0; i < loadedBigramBuffers.length; i++) {
941                 System.out.println("Processed " + i
942                         + " of " + loadedBigramBuffers.length);
943                 NGramBuffer bigram = getBigramBuffer(i);
944                 
945                 if (bigram == null)
946                     continue;
947     
948                 for (int j = 0; j < bigram.getNumberNGrams(); j++) {
949                     float smearTerm;
950                     NGramProbability bgProb = bigram.getNGramProbability(j);
951                     float logbgbackoff = NGramBackoffTable[2][bgProb.getBackoffID()];
952                     double bgbackoff = logMath.logToLinear(logbgbackoff);
953                     int k = bigram.getWordID(j);
954                     NGramBuffer trigram = loadTrigramBuffer(i, k);
955     
956                     if (trigram == null)
957                         smearTerm = unigramSmearTerm[k];
958                     else {
959                         double bg_numerator = 0;
960                         double bg_denominator = 0;
961                         for (int l = 0; l < trigram.getNumberNGrams(); l++) {
962                             int m = trigram.getWordID(l);
963                             float logtgprob
964                                     = NGramProbTable[2][trigram.getProbabilityID(l)];
965                             double tgprob = logMath.logToLinear(logtgprob);
966                             float logbgprob = getBigramProb(k, m);
967                             double bgprob = logMath.logToLinear(logbgprob);
968                             float logugprob = unigrams[m].getLogProbability();
969                             double backofftgprob = bgbackoff * bgprob;
970                             double logbackofftgprob
971                                     = logMath.linearToLog(backofftgprob);
972     
973                             bg_numerator += (tgprob * logtgprob
974                                     - backofftgprob * logbackofftgprob) * logugprob;
975     
976                             bg_denominator += (tgprob - backofftgprob)
977                                     * logugprob * logugprob;
978                         }
979                         
980                         bg_numerator += bgbackoff * (logbgbackoff * ugAvgLogProb[k] - ugNumerator[k]);
981                         bg_denominator += bgbackoff * ugDenominator[k];
982                         // bigram.ugsmear = bg_numerator / bg_denominator;
983                         smearTerm = (float) (bg_numerator / bg_denominator);
984                         smearTermCount++;
985                     }
986                     
987                     putSmearTerm(i, k, smearTerm);
988                 }
989             }
990             
991             System.out.println("Smear count is " + smearTermCount);
992         }

1059          * @throws IOException if an inconsistent file is found or on any general I/O error

1063

1069                 throw new IOException("Bad unigram length in " + filename);
1070

1072             unigramSmearTerm = new float[unigrams.length];
1073             System.out.println("Reading " + unigrams.length);
1074             
1075             for (int i = 0; i < unigrams.length; i++)
1076                 unigramSmearTerm[i] = in.readFloat();
1077     
1078             for (int i = 0; i < unigrams.length; i++) {
1079                 System.out.println("Processed " + i
1080                         + " of " + loadedBigramBuffers.length);
1081                 int numBigrams = in.readInt();
1082                 NGramBuffer bigram = getBigramBuffer(i);
1083

1085                     throw new IOException("Bad ngrams for unigram " + i + " Found " + numBigrams + " expected " + bigram.getNumberNGrams());
