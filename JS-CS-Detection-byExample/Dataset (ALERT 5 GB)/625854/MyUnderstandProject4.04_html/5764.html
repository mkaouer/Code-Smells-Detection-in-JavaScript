28     /**
29      * A demonstration of how to use JSGF grammars in sphinx-4. This
30      * program demonstrates how an application can cycle through a number of JSGF
31      * grammars as well as how an application can build up grammar rules
32      * directly
33      *
34      */
35     public class JSGFDemo {
36         private Recognizer recognizer;
37         private JSGFGrammar jsgfGrammarManager;
38         private Microphone microphone;
39     
40     
41         /**
42          * Creates a new JSGFDemo. The demo will retrieve the
43          * configuration from the jsgf.config.xml file found in the
44          * classpath for the JSGFDemo. The jsgf.config.xml should at the
45          * minimum define a recognizer with the name 'recognizer', a
46          * JSGFGrammar named 'jsgfGrammar' and a Microphone named
47          * 'microphone'.
48          *
49          * @throws IOException if an I/O error occurs
50          * @throws PropertyException if a property configuration occurs
51          * @throws InstantiationException if a problem occurs while
52          * creating any of the recognizer components.
53          */
54         public JSGFDemo() throws 
55                 IOException, PropertyException, InstantiationException {
56     
57             URL url = JSGFDemo.class.getResource("jsgf.config.xml");
58             ConfigurationManager cm = new ConfigurationManager(url);
59     
60             // retrive the recognizer, jsgfGrammar and the microphone from
61             // the configuration file.
62     
63             recognizer = (Recognizer) cm.lookup("recognizer");
64             jsgfGrammarManager = (JSGFGrammar) cm.lookup("jsgfGrammar");
65             microphone = (Microphone) cm.lookup("microphone");
66         }
67     
68     
69         /**
70          * Executes the demo. The demo will cycle through four separate
71          * JSGF grammars,a 'movies' grammar, a 'news' grammar a 'books'
72          * grammar and a 'music' grammar.  The news, books and movies
73          * grammars are loaded from the corresponding JSGF grammar file,
74          * while the music grammar is loaded from a file, but then
75          * augmented via RuleGrammar.setRule.
76          */
77         public void execute() throws IOException, GrammarException  {
78             System.out.println("JSGF Demo Version 1.0\n");
79     
80             System.out.print(" Loading recognizer ...");
81             recognizer.allocate();
82             System.out.println(" Ready");
83     
84             if (microphone.startRecording()) {
85                 loadAndRecognize("movies");
86                 loadAndRecognize("news");
87                 loadAndRecognize("books");
88                 loadAndRecognizeMusic();
89             } else {
90                 System.out.println("Can't start the microphone");
91             }
92     
93             System.out.print("\nDone. Cleaning up ...");
94             recognizer.deallocate();
95     
96             System.out.println(" Goodbye.\n");
97             System.exit(0);
98         }
99     
100         /**
101          * Load the grammar with the given grammar name and start
102          * recognizing speech using the grammar.  Spoken utterances will
103          * be echoed to the terminal.  This method will return when the
104          * speaker utters the exit phrase for the grammar. The exit phrase
105          * is a phrase in the grammar with the word 'exit' as a tag.
106          *
107          * @throws IOException if an I/O error occurs
108          * @throws GrammarException if a grammar format error is detected
109          */
110         private void loadAndRecognize(String grammarName) throws
111                 IOException, GrammarException  {
112             jsgfGrammarManager.loadJSGF(grammarName);
113             dumpSampleSentences(grammarName);
114             recognizeAndReport();
115         }
116     
117         /**
118          * Performs recognition with the currently loaded grammar.
119          * Recognition for potentially multiple utterances until an 'exit'
120          * tag is returned.
121          *
122          * @htrows GrammarException if an error in the JSGF grammar is
123          * encountered
124          */
125         private void recognizeAndReport() throws GrammarException {
126             boolean done = false;
127     
128     
129             while (!done)  {
130                 Result result = recognizer.recognize();
131                 String bestResult = result.getBestFinalResultNoFiller();
132                 RuleGrammar ruleGrammar = jsgfGrammarManager.getRuleGrammar();
133                 RuleParse ruleParse = ruleGrammar.parse(bestResult, null);
134                 if (ruleParse != null) {
135                     System.out.println("\n  " + bestResult + '\n');
136                     done = isExit(ruleParse);
137                 } 
138             }
139         }
140     
141         /**
142          * Searches through the tags of the rule parse for an 'exit' tag.
143          *
144          * @return true if an 'exit' tag is found
145          */
146         private boolean isExit(RuleParse ruleParse) {
147             String[] tags = ruleParse.getTags();
148     
149             for (int i = 0; tags != null && i < tags.length; i++) {
150                 if (tags[i].trim().equals("exit")) {
151                     return true;
152                 }
153             }
154             return  false;
155         }

157         /**
158          * Loads the music grammar and augments it with a set of rules for
159          * playing the top dozen or so movie songs of all time.
160          *
161          * @throws IOException if an I/O error occurs
162          * @throws GrammarException if a grammar format error is detected
163          */
164         private void loadAndRecognizeMusic() throws
165                 IOException, GrammarException  {
166             jsgfGrammarManager.loadJSGF("music");
167             RuleGrammar ruleGrammar = jsgfGrammarManager.getRuleGrammar();
168             addRule(ruleGrammar, "song1", "listen to over the rainbow");
169             addRule(ruleGrammar, "song2", "listen to as time goes by");
170             addRule(ruleGrammar, "song3", "listen to singing in the rain");
171             addRule(ruleGrammar, "song4", "listen to moon river");
172             addRule(ruleGrammar, "song5", "listen to white christmas");
173             addRule(ruleGrammar, "song6", "listen to mrs robinson");
174             addRule(ruleGrammar, "song7", "listen to when you wish upon a star");
175             addRule(ruleGrammar, "song8", "listen to the way we were");
176             addRule(ruleGrammar, "song9", "listen to staying alive");
177             addRule(ruleGrammar, "song10", "listen to the sound of music");
178             addRule(ruleGrammar, "song11", "listen to the man that got away");
179             addRule(ruleGrammar, "song12", 
180                         "listen to diamonds are a girl's best friend");
181             jsgfGrammarManager.commitChanges();
182             dumpSampleSentences("music");
183             recognizeAndReport();
184         }
185     
186         /**
187          * Adds the given jsgf as a rule to a grammar
188          *
189          * @param ruleGrammar the grammar to receive the new rule
190          * @param ruleName the name of the rule
191          * @param jsgf the definition for the new rule
192          */
193         private void addRule(RuleGrammar ruleGrammar, 
194                     String ruleName, String jsgf) throws GrammarException {
195             Rule newRule = ruleGrammar.ruleForJSGF(jsgf);
196             ruleGrammar.setRule(ruleName, newRule, true);
197             ruleGrammar.setEnabled(ruleName, true);
198         }
