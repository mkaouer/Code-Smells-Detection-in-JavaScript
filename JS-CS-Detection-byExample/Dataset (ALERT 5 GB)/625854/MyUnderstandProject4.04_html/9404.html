11     package edu.cmu.sphinx.instrumentation;
12     
13     import edu.cmu.sphinx.result.*;
14     import edu.cmu.sphinx.util.NISTAlign;
15     import edu.cmu.sphinx.util.props.*;
16     import edu.cmu.sphinx.recognizer.Recognizer;
17     
18     /**
19      * Tracks and reports recognition accuracy using the "confidenceScorer" component specified in the ConfigurationManager.
20      * The "confidenceScorer" component is typically configured to be edu.cmu.sphinx.result.SausageMaker.
21      *
22      * @see edu.cmu.sphinx.result.SausageMaker
23      */
24     public class BestConfidenceAccuracyTracker extends AccuracyTracker {
25     
26         /** Defines the class to use for confidence scoring. */
27         @S4Component(type = ConfidenceScorer.class)
28         public final static String PROP_CONFIDENCE_SCORER = "confidenceScorer";
29     
30         /** The confidence scorer */
31         protected ConfidenceScorer confidenceScorer;
32     
33         public BestConfidenceAccuracyTracker(ConfidenceScorer confidenceScorer, Recognizer recognizer, boolean showSummary, boolean showDetails, boolean showResults, boolean showAlignedResults, boolean showRawResults) {
34             super( recognizer, showSummary, showDetails, showResults, showAlignedResults, showRawResults);
35             this.confidenceScorer = confidenceScorer;
36         }
37     
38         public BestConfidenceAccuracyTracker() {
39         }
40     
41         /*
42         * (non-Javadoc)
43         *
44         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
45         */
46         @Override
47         public void newProperties(PropertySheet ps) throws PropertyException {
48             super.newProperties(ps);
49             confidenceScorer = (ConfidenceScorer)ps.getComponent(PROP_CONFIDENCE_SCORER);
50         }
51     
52         /*
53         * (non-Javadoc)
54         *
55         * @see edu.cmu.sphinx.decoder.ResultListener#newResult(edu.cmu.sphinx.result.Result)
56         */
57         @Override
58         public void newResult(Result result) {
59             NISTAlign aligner = getAligner();
60             String ref = result.getReferenceText();
61             if (result.isFinal() && (ref != null)) {
62                 try {
63                     Path bestPath = null;
64                     String hyp = "";
65                     if (result.getBestFinalToken() != null) {
66                         ConfidenceResult confidenceResult =
67                                 confidenceScorer.score(result);
68                         bestPath = confidenceResult.getBestHypothesis();
69                         hyp = bestPath.getTranscriptionNoFiller();
70                     }
71                     aligner.align(ref, hyp);
72                     if (bestPath != null) {
73                         showDetails(bestPath.getTranscription());
74                     } else {
75                         showDetails("");
76                     }
77                 } catch (Exception e) {
78                     e.printStackTrace();
79                 }
80             }
81         }
82     }
