52          * @param url The location of the configuration file.

55             configURL = url;
56             
57             try {
58                 rawPropertyMap = new SaxLoader(url, globalProperties).load();
59             } catch (IOException e) {
60                 throw new RuntimeException(e);
61             }
62     
63             ConfigurationManagerUtils.applySystemProperties(rawPropertyMap, globalProperties);
64             ConfigurationManagerUtils.configureLogger(this);
65     
66             // we can't configure the configuration manager with itself so we

69             if (showCreations != null)
70                 this.showCreations = "true".equals(showCreations);

74         /**
75          * Returns the property sheet for the given object instance
76          *
77          * @param instanceName the instance name of the object
78          * @return the property sheet for the object.
79          */
80         public PropertySheet getPropertySheet(String instanceName) {
81             if (!symbolTable.containsKey(instanceName)) {
82                 // if it is not in the symbol table, so construct
83                 // it based upon our raw property data
84                 RawPropertyData rpd = rawPropertyMap.get(instanceName);
85                 if (rpd != null) {
86                     String className = rpd.getClassName();
87                     try {
88                         Class<?> cls = Class.forName(className);
89                         
90                         // now load the property-sheet by using the class annotation
91                         PropertySheet propertySheet = new PropertySheet(cls.asSubclass(Configurable.class), instanceName, this, rpd);
92     
93                         symbolTable.put(instanceName, propertySheet);
94     
95                     } catch (ClassNotFoundException e) {
96                         System.err.println("class not found !" + e);
97                     } catch (ClassCastException e) {
98                         System.err.println("can not cast class !" + e);
99                     } catch (ExceptionInInitializerError e) {
100                         System.err.println("couldn't load class !" + e);
101                     }
102                 }
103             }
104     
105             return symbolTable.get(instanceName);
106         }

110          * Gets all instances that are of the given type.
111          *
112          * @param type the desired type of instance
113          * @return the set of all instances
114          */
115         public Collection<String> getInstanceNames(Class<? extends Configurable> type) {
116             Collection<String> instanceNames = new ArrayList<String>();
117     
118             for (PropertySheet ps : symbolTable.values()) {
119                 if (!ps.isInstanciated())
120                     continue;
121     
122                 if (ConfigurationManagerUtils.isDerivedClass(ps.getConfigurableClass(), type))
123                     instanceNames.add(ps.getInstanceName());
124             }
125     
126             return instanceNames;
127         }

145          * @return the component, or null if a component was not found.

150         public Configurable lookup(String instanceName) throws InternalConfigurationException {
151             // apply all new properties to the model
152             instanceName = getStrippedComponentName(instanceName);
153     
154             PropertySheet ps = getPropertySheet(instanceName);
155             if (ps == null)

160     
161             return ps.getOwner();

165         /**
166          * Returns a <code>Configurable</code> instance of a given type <code>C</code>, if such a component (or a derived
167          * one) is registered to this <code>ConfigurationManager</code> instance, and there is one and only match.
168          * <p/>
169          * This is a convenience method that allows to access a system configuration without knowing the instance names of
170          * registered components.
171          *
172          * @param <C> A component type
173          * @return The <code>Configurable</code> instance of null if there is no matching <code>Configurable</code>.
174          * @throws IllegalArgumentException if more than one component of the given type is registered to this
175          *                                  ConfigurationManager.
176          */
177         public <C extends Configurable> C lookup(Class<C> confClass) {
178             List<PropertySheet> matchPropSheets = getPropSheets(confClass);
179             if (matchPropSheets.isEmpty())
180                 return null;
181     
182             if (matchPropSheets.size() > 1)
183                 throw new IllegalArgumentException("more than one match to type " + confClass + ": " + matchPropSheets);
184     
185             return confClass.cast(lookup(matchPropSheets.get(0).getInstanceName()));
186         }

193         public List<PropertySheet> getPropSheets(Class<? extends Configurable> confClass) {
194             List<PropertySheet> psCol = new ArrayList<PropertySheet>();

196             for (PropertySheet ps : symbolTable.values()) {

220          * Registers a new configurable to this configuration manager.
221          *
222          * @param confClass The class of the configurable to be instantiated and to be added to this configuration manager
223          *                  instance.
224          * @param name      The desired  lookup-name of the configurable
225          * @param props     The properties to be used for component configuration
226          * @throws IllegalArgumentException if the there's already a component with the same <code>name</code> registered to
227          *                                  this configuration manager instance.
228          */
229         public void addConfigurable(Class<? extends Configurable> confClass, String name, Map<String, Object> props) {
230             if (name == null) // use the class name as default if no name is given
231                 name = confClass.getName();
232     
233             if (symbolTable.containsKey(name))
234                 throw new IllegalArgumentException("tried to override existing component name : " + name);
235     
236             PropertySheet ps = getPropSheetInstanceFromClass(confClass, props, name, this);
237             symbolTable.put(name, ps);
238             rawPropertyMap.put(name, new RawPropertyData(name, confClass.getName()));
239     
240             for (ConfigurationChangeListener changeListener : changeListeners)
241                 changeListener.componentAdded(this, ps);
242         }

266             PropertySheet ps = getPropertySheet(oldName);
267     
268             if (ps == null) {
269                 throw new RuntimeException("no configurable (to be renamed) named " + oldName + " is contained in the CM");

271     
272             ConfigurationManagerUtils.renameComponent(this, oldName, newName);
273     
274             symbolTable.remove(oldName);
275             symbolTable.put(newName, ps);

302         /**
303          * Adds a subconfiguration to this instance by registering all subCM-components and all its global properties.
304          *
305          * @param subCM                The subconfiguration that should be  to this instance
306          * @param doOverrideComponents If <code>true</code> non-instantiated components will be overridden by elements of
307          *                             subCM even if already being registered to this CM-instance. The same holds for global
308          *                             properties.
309          * @throws RuntimeException if an already instantiated component in this instance is redefined in subCM.
310          */
311         public void addSubConfiguration(ConfigurationManager subCM, boolean doOverrideComponents) {
312             Collection<String> compNames = getComponentNames();
313     
314             for (String componentName : subCM.getComponentNames()) {
315                 if (compNames.contains(componentName)) {
316                     if (doOverrideComponents && !getPropertySheet(componentName).isInstanciated()) {
317                         PropertySheet ps = subCM.getPropertySheet(componentName);
318                         symbolTable.put(componentName, ps);
319                         rawPropertyMap.put(componentName, new RawPropertyData(componentName, ps.getConfigurableClass().getSimpleName()));
320     
321                     } else {
322                         throw new RuntimeException(componentName + " is already registered to system configuration");
323                     }
324                 }
325             }
326     
327             for (String globProp : subCM.globalProperties.keySet()) {
328                 // the second test is necessary because system-props will be global-props in both CMs
329                 if (globalProperties.containsKey(globProp) && !System.getProperties().containsKey(globProp)) {
330                     if (!doOverrideComponents)
331                         throw new RuntimeException(globProp + " is already registered as global property");
332                 }
333             }
334     
335             globalProperties.putAll(subCM.globalProperties);
336     
337             // correct the reference to the configuration manager
338             for (PropertySheet ps : subCM.symbolTable.values()) {
339                 ps.setCM(this);
340             }
341     
342             symbolTable.putAll(subCM.symbolTable);
343             rawPropertyMap.putAll(subCM.rawPropertyMap);
344         }

379         /**
380          * Sets a global property.
381          *
382          * @param propertyName The name of the global property.
383          * @param value        The new value of the global property. If the value is <code>null</code> the property becomes
384          *                     removed.
385          */
386         public void setGlobalProperty(String propertyName, String value) {
387             if (value == null)
388                 globalProperties.remove(propertyName);
389             else
390                 globalProperties.put(propertyName, value);
391     
392             // update all component configurations because they might be affected by the change
393             for (String instanceName : getInstanceNames(Configurable.class)) {
394                 PropertySheet ps = getPropertySheet(instanceName);
395                 if (ps.isInstanciated())
396                     try {
397                         ps.getOwner().newProperties(ps);
398                     } catch (PropertyException e) {
399                         e.printStackTrace();
400                     }
401             }
402         }

458         /**
459          * Test whether the given configuration manager instance equals this instance in terms of same configuration. This
460          * This equals implementation does not care about instantiation of components.
461          */
462         @Override
463         public boolean equals(Object obj) {
464             if (!(obj instanceof ConfigurationManager))
465                 return false;
466     
467             ConfigurationManager cm = (ConfigurationManager) obj;
468     
469             Set<String> thisCompNames = getComponentNames();
470             if (!thisCompNames.equals(cm.getComponentNames()))
471                 return false;
472     
473             // make sure that all components are the same
474             for (String instanceName : thisCompNames) {
475                 PropertySheet myPropSheet = getPropertySheet(instanceName);
476                 PropertySheet otherPropSheet = cm.getPropertySheet(instanceName);
477     
478                 if (!otherPropSheet.equals(myPropSheet))
479                     return false;
480             }
481     
482             // make sure that both configuration managers have the same set of global properties
483             return cm.getGlobalProperties().equals(getGlobalProperties());
484         }

545         /**
546          * Instantiates the given <code>targetClass</code> and instruments it using default properties or the properties
547          * given by the <code>defaultProps</code>.
548          */
549         private static PropertySheet getPropSheetInstanceFromClass(Class<? extends Configurable> targetClass, Map<String, Object> defaultProps, String componentName, ConfigurationManager cm) {
550             RawPropertyData rpd = new RawPropertyData(componentName, targetClass.getName());
551     
552             for (Map.Entry<String, Object> entry : defaultProps.entrySet()) {
553                 Object property = entry.getValue();
554     
555                 if (property instanceof Class<?>)
556                     property = ((Class<?>) property).getName();
557     
558                 rpd.getProperties().put(entry.getKey(), property);
559             }
560     
561             return new PropertySheet(targetClass, componentName, cm, rpd);
562         }
