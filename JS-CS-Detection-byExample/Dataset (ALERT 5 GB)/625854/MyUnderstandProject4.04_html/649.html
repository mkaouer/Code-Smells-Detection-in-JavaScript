360         /**
361          * Loads the language model from the given file.
362          *
363          * @param location the file containing the language model
364          * @throws java.io.IOException
365          */
366         private void loadBinary(File location) throws IOException {
367     
368             DataInputStream stream = new DataInputStream
369                     (new BufferedInputStream(new FileInputStream(location)));
370     
371             // read standard header string-size; set bigEndian flag
372             readHeader(stream);
373     
374             // +1 is the sentinel unigram at the end
375             unigrams = readUnigrams(stream, numberNGrams[0] + 1, bigEndian);
376     
377             skipNGrams(stream);
378     
379             // Read the NGram prob & bow tables
380             for (int i = 1; i < maxNGram; i++) {      	
381             	if (numberNGrams[i] > 0) {
382             		if (i == 1) {
383             			NGramProbTable[i] = readFloatTable(stream, bigEndian);
384             		}
385             		else {
386             			NGramBackoffTable[i] = readFloatTable(stream, bigEndian);
387             			NGramProbTable[i] = readFloatTable(stream, bigEndian);
388             			
389             			int nMinus1gramSegmentSize = 1 << logNGramSegmentSize;
390             			int NGramSegTableSize = ((numberNGrams[i - 1] + 1) / nMinus1gramSegmentSize) + 1;
391             			
392             			NGramSegmentTable[i] = readIntTable(stream, bigEndian, NGramSegTableSize);
393             		}
394             	}
395             }
396             
397             // read word string names
398             int wordsStringLength = readInt(stream, bigEndian);
399             if (wordsStringLength <= 0) {
400                 throw new Error("Bad word string size: " + wordsStringLength);
401             }
402     
403             // read the string of all words
404             this.words = readWords(stream, wordsStringLength, numberNGrams[0]);
405     
406             // A voir
407             if (startWordID > -1) {
408                 UnigramProbability unigram = unigrams[startWordID];
409                 unigram.setLogProbability(MIN_PROBABILITY);
410             }
411             if (endWordID > -1) {
412                 UnigramProbability unigram = unigrams[endWordID];
413                 unigram.setLogBackoff(MIN_PROBABILITY);
414             }
415     
416             applyUnigramWeight();
417             
418             if (applyLanguageWeightAndWip) {
419             	for (int i = 0; i <= maxNGram; i++) {
420             		applyLanguageWeight(NGramProbTable[i], languageWeight);
421             		applyWip(NGramProbTable[i], wip);
422             		
423             		if (i > 1) {
424             			applyLanguageWeight(NGramBackoffTable[i], languageWeight);
425             		}
426             	}
427             }
428     
429             stream.close();
430             file = new RandomAccessFile(location, "r");
431         }

434         /**
435          * Reads the LM file header
436          *
437          * @param stream the data stream of the LM file
438          * @throws java.io.IOException
439          */
440         private void readHeader(DataInputStream stream) throws IOException {
441             int headerLength = readInt(stream, bigEndian);
442     
443             if ((headerLength != DARPA_TG_HEADER.length() + 1) && (headerLength != DARPA_QG_HEADER.length() + 1) && (headerLength != DARPA_NG_HEADER.length() - 1)) { // not big-endian
444                 headerLength = Utilities.swapInteger(headerLength);
445                 if (headerLength == (DARPA_TG_HEADER.length() + 1) || headerLength == (DARPA_QG_HEADER.length() + 1) || headerLength == (DARPA_NG_HEADER.length() - 1)) {
446                     bigEndian = false;
447                     // System.out.println("Little-endian");
448                 } else {
449                     throw new Error("Bad binary LM file magic number: "
450                             + headerLength + ", not an LM dumpfile?");
451                 }
452             } else {
453                 // System.out.println("Big-endian");
454             }
455     
456             // read and verify standard header string
457             String header = readString(stream, headerLength - 1);
458             readByte(stream); // read the '\0'
459     
460             if (!header.equals(DARPA_TG_HEADER) & !header.equals(DARPA_QG_HEADER) & !Pattern.matches(DARPA_NG_HEADER, header)) {
461                 throw new Error("Bad binary LM file header: " + header);
462             }
463             else {
464             	if (header.equals(DARPA_TG_HEADER))
465             		maxNGram = 3;
466             	else if (header.equals(DARPA_QG_HEADER))
467             		maxNGram = 4;
468             	else {
469             		Pattern p = Pattern.compile("\\d");
470             		Matcher m = p.matcher(header);
471             		maxNGram = Integer.parseInt(m.group());
472             	}
473             }
474             
475             // read LM filename string size and string
476             int fileNameLength = readInt(stream, bigEndian);
477             bytesRead += (long)stream.skipBytes(fileNameLength);
478     
479             numberNGrams = new int[maxNGram];
480             NGramOffset = new long[maxNGram];
481             NGramProbTable = new float[maxNGram][];
482             NGramBackoffTable = new float[maxNGram][];
483             NGramSegmentTable = new int[maxNGram][];
484             
485             numberNGrams[0] = 0;
486             logNGramSegmentSize = LOG2_NGRAM_SEGMENT_SIZE;
487     
488             // read version number, if present. it must be <= 0.
489     
490             int version = readInt(stream, bigEndian);
491             // System.out.println("Version: " + version);
492     
493             bytesPerField = 2;
494             
495             if (version <= 0) { // yes, its the version number
496                 readInt(stream, bigEndian); // read and skip timestamp
497     
498                 // Means we are going 32 bits.
499                 if (version <= -3)
500                 	bytesPerField = 4;
501                 
502                 // read and skip format description
503                 int formatLength;
504                 for (; ;) {
505                     if ((formatLength = readInt(stream, bigEndian)) == 0) {
506                         break;
507                     }
508                     bytesRead += (long)stream.skipBytes(formatLength);
509                 }
510     
511                 // read log NGram segment size if present
512                 // only for 16 bits version 2 LM
513                 if (version == -2) {
514                     logNGramSegmentSize = readInt(stream, bigEndian);
515                     if (logNGramSegmentSize < 1 || logNGramSegmentSize > 15) {
516                         throw new Error("log2(bg_seg_sz) outside range 1..15");
517                     }
518                 }
519     
520                 numberNGrams[0] = readInt(stream, bigEndian);
521             } else {
522             	numberNGrams[0] = version;
523             }
524     
525             if (numberNGrams[0] <= 0) {
526                 throw new Error("Bad number of unigrams: " + numberNGrams[0]
527                         + ", must be > 0.");
528             }
529     
530             for (int i = 1; i < maxNGram; i++) {
531                 if ((numberNGrams[i] = readInt(stream, bigEndian)) < 0) {
532                     throw new Error("Bad number of " + String.valueOf(i) + "-grams: " + numberNGrams[i]);
533                 }
534             }
535         }

538         /**
539          * Skips the NGrams of the LM.
540          *
541          * @param stream the source of data
542          * @throws java.io.IOException
543          */
544         private void skipNGrams(DataInputStream stream) throws IOException {
545     		NGramOffset[1] = bytesRead;
546     		int bytesToSkip = (numberNGrams[1] + 1) * LargeNGramModel.BYTES_PER_NGRAM * getBytesPerField();
547     		stream.skipBytes(bytesToSkip);
548     		bytesRead += bytesToSkip;
549     		
550         	for (int i = 2; i < maxNGram; i++) {
551         		if (numberNGrams[i] > 0 && i < maxNGram - 1) {
552         			NGramOffset[i] = bytesRead;
553         			long bToSkip = (long)(numberNGrams[i] + 1) * (long)LargeNGramModel.BYTES_PER_NGRAM * (long)getBytesPerField();
554         			stream.skip(bToSkip);
555         			bytesRead += bToSkip;
556         		}
557         		else if (numberNGrams[i] > 0 && i == maxNGram - 1) {
558         			NGramOffset[i] = bytesRead;
559         			long bToSkip = (long)(numberNGrams[i]) * (long)LargeNGramModel.BYTES_PER_NMAXGRAM * (long)getBytesPerField();
560         			stream.skip(bToSkip);
561         			bytesRead += bToSkip;
562         		}
563         	}
564         }

567         /** Apply the unigram weight to the set of unigrams */
568         private void applyUnigramWeight() {
569             float logUnigramWeight = logMath.linearToLog(unigramWeight);
570             float logNotUnigramWeight = logMath.linearToLog(1.0f - unigramWeight);
571             float logUniform = logMath.linearToLog(1.0f / (numberNGrams[0]));
572     
573             float logWip = logMath.linearToLog(wip);
574     
575             float p2 = logUniform + logNotUnigramWeight;
576     
577             for (int i = 0; i < numberNGrams[0]; i++) {
578                 UnigramProbability unigram = unigrams[i];
579     
580                 float p1 = unigram.getLogProbability();
581     
582                 if (i != startWordID) {
583                     p1 += logUnigramWeight;
584                     p1 = logMath.addAsLinear(p1, p2);
585                 }
586     
587                 if (applyLanguageWeightAndWip) {
588                     p1 = p1 * languageWeight + logWip;
589                     unigram.setLogBackoff(unigram.getLogBackoff() * languageWeight);
590                 }
591     
592                 unigram.setLogProbability(p1);
593             }
594         }

597         /** Apply the language weight to the given array of probabilities.
598          */
599         private void applyLanguageWeight(float[] logProbabilities,
600                                          float languageWeight) {
601             for (int i = 0; i < logProbabilities.length; i++) {
602                 logProbabilities[i] = logProbabilities[i] * languageWeight;
603             }
604         }

607         /** Apply the WIP to the given array of probabilities.
608         */
609         private void applyWip(float[] logProbabilities, double wip) {
610             float logWip = logMath.linearToLog(wip);
611             for (int i = 0; i < logProbabilities.length; i++) {
612                 logProbabilities[i] = logProbabilities[i] + logWip;
613             }
614         }

617         /**
618          * Reads the probability table from the given DataInputStream.
619          *
620          * @param stream    the DataInputStream from which to read the table
621          * @param bigEndian true if the given stream is bigEndian, false otherwise
622          * @throws java.io.IOException
623          */
624         private float[] readFloatTable(DataInputStream stream, boolean bigEndian)
625                 throws IOException {
626     
627             int numProbs = readInt(stream, bigEndian);
628             if (numProbs <= 0 || numProbs > MAX_PROB_TABLE_SIZE) {
629                 throw new Error("Bad probabilities table size: " + numProbs);
630             }
631     
632             float[] probTable = new float[numProbs];
633     
634             for (int i = 0; i < numProbs; i++) {
635             	//probTable[i] = readFloat(stream, bigEndian);
636                 probTable[i] = logMath.log10ToLog(readFloat(stream, bigEndian));
637             }
638     
639             return probTable;
640         }

643         /**
644          * Reads a table of integers from the given DataInputStream.
645          *
646          * @param stream    the DataInputStream from which to read the table
647          * @param bigEndian true if the given stream is bigEndian, false otherwise
648          * @param tableSize the size of the NGram segment table
649          * @return the NGram segment table, which is an array of integers
650          * @throws java.io.IOException
651          */
652         private int[] readIntTable(DataInputStream stream, boolean bigEndian,
653                                    int tableSize) throws IOException {
654             int numSegments = readInt(stream, bigEndian);
655             if (numSegments != tableSize) {
656                 throw new Error("Bad NGram seg table size: " + numSegments);
657             }
658             int[] segmentTable = new int[numSegments];
659             for (int i = 0; i < numSegments; i++) {
660                 segmentTable[i] = readInt(stream, bigEndian);
661             }
662             return segmentTable;
663         }

666         /**
667          * Read in the unigrams in the given DataInputStream.
668          *
669          * @param stream         the DataInputStream to read from
670          * @param numberUnigrams the number of unigrams to read
671          * @param bigEndian      true if the DataInputStream is big-endian, false otherwise
672          * @return an array of UnigramProbability index by the unigram ID
673          * @throws java.io.IOException
674          */
675         private UnigramProbability[] readUnigrams(DataInputStream stream,
676                                                   int numberUnigrams, boolean bigEndian) throws IOException {
677     
678             UnigramProbability[] unigrams = new UnigramProbability[numberUnigrams];
679     
680             for (int i = 0; i < numberUnigrams; i++) {
681     
682                 // read unigram ID, unigram probability, unigram backoff weight
683                 int unigramID = readInt(stream, bigEndian);
684                 
685                 /* sphinx3_lm_convert doesn't store ID's in unigrams */
686                 if (unigramID < 1) 
687                 	unigramID = i;
688     
689                 // if we're not reading the sentinel unigram at the end,
690                 // make sure that the unigram IDs are consecutive
691                 if (i != (numberUnigrams - 1)) {
692                     assert (unigramID == i);
693                 }
694     
695                 float unigramProbability = readFloat(stream, bigEndian);
696                 float unigramBackoff = readFloat(stream, bigEndian);
697                 int firstBigramEntry = readInt(stream, bigEndian);
698     
699                 float logProbability = logMath.log10ToLog(unigramProbability);
700                 float logBackoff = logMath.log10ToLog(unigramBackoff);
701     
702                 unigrams[i] = new UnigramProbability(unigramID, logProbability,
703                         logBackoff, firstBigramEntry);
704             }
705     
706             return unigrams;
707         }

710         /**
711          * Reads a byte from the given DataInputStream.
712          *
713          * @param stream the DataInputStream to read from
714          * @return the byte read
715          * @throws java.io.IOException
716          */
717         private byte readByte(DataInputStream stream) throws IOException {
718             bytesRead++;
719             return stream.readByte();

762          /**
763           * Reads a string of the given length from the given DataInputStream. It is assumed that the DataInputStream
764           * contains 8-bit chars.
765           *
766           * @param stream the DataInputStream to read from
767           * @param length the number of characters in the returned string
768           * @return a string of the given length from the given DataInputStream
769           * @throws java.io.IOException
770           */
771          private String readString(DataInputStream stream, int length)
772                  throws IOException {
773              StringBuilder builder = new StringBuilder();
774              byte[] bytes = new byte[length];
775              bytesRead += stream.read(bytes);
776      
777              for (int i = 0; i < length; i++) {
778                  builder.append((char) bytes[i]);
779              }
780              return builder.toString();
781          }

784          /**
785           * Reads a series of consecutive Strings from the given stream.
786           *
787           * @param stream         the DataInputStream to read from
788           * @param length         the total length in bytes of all the Strings
789           * @param numberUnigrams the number of String to read
790           * @return an array of the Strings read
791           * @throws java.io.IOException
792           */
793          private String[] readWords(DataInputStream stream, int length,
794                                           int numberUnigrams) throws IOException {
795              String[] words = new String[numberUnigrams];
796              byte[] bytes = new byte[length];
797              bytesRead += stream.read(bytes);
798      
799              int s = 0;
800              int wordStart = 0;
801              for (int i = 0; i < length; i++) {
802                  char c = (char) (bytes[i] & 0xFF);
803                  bytesRead++;
804                  if (c == '\0') {
805                      // if its the end of a string, add it to the 'words' array
806                      words[s] = new String(bytes, wordStart, i - wordStart).toLowerCase();
807                      wordStart = i + 1;
808                      if (words[s].equals(Dictionary.SENTENCE_START_SPELLING)) {
809                          startWordID = s;
810                      } else if (words[s].equals(Dictionary.SENTENCE_END_SPELLING)) {
811                          endWordID = s;
812                      }
813                      s++;
814                  }
815              }
816              assert (s == numberUnigrams);
817              return words;
818          }
