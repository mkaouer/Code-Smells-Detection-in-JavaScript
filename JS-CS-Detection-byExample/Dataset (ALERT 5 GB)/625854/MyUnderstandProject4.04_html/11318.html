88         /** Reads and verifies a driver file.
89          *  @param fileName
90          */
91         private static List<URL> readDriver(String fileName) {
92             File inputFile = new File(fileName);
93             List<URL> driverFiles = null;
94     
95             try {
96                 if (!inputFile.isFile() || !inputFile.canRead())
97                     throw new IllegalArgumentException("file to read is not valid");
98     
99                 BufferedReader bf = new BufferedReader(new FileReader(inputFile));
100     
101                 driverFiles = new ArrayList<URL>();
102     
103                 String line;
104                 while ((line = bf.readLine()) != null && line.trim().length() != 0) {
105                     File file = new File(line);
106     
107                     assert file.isFile() : "file " + file + " does not exist!";
108                     driverFiles.add(file.toURI().toURL());
109                 }
110     
111                 bf.close();
112             } catch (IOException e) {
113                 e.printStackTrace();
114             }
115     
116             assert driverFiles != null;
117             return driverFiles;
118         }

138         /**
139          * The work of the concatenating of the audio files are done here. The idea here is to turn the list of audio files
140          * into an Enumeration, and then fed it to a SequenceInputStream, giving the illusion that the audio files are
141          * concatenated, but only logically.
142          */
143         class InputStreamEnumeration implements Enumeration<AudioInputStream> {
144     
145             private URL lastFile;
146             final Iterator<URL> fileIt;
147     
148     
149             InputStreamEnumeration(List<URL> files) throws IOException {
150                 fileIt = new ArrayList<URL>(files).iterator();
151             }
152     
153     
154             /**
155              * Tests if this enumeration contains more elements.
156              *
157              * @return true if and only if this enumeration object contains at least one more element to provide; false
158              *         otherwise.
159              */
160             @Override
161             public boolean hasMoreElements() {
162                 if (nextFile == null) {
163                     nextFile = readNext();
164                 }
165                 return (nextFile != null);
166             }
167     
168     
169             /**
170              * Returns the next element of this enumeration if this enumeration object has at least one more element to
171              * provide.
172              *
173              * @return the next element of this enumeration.
174              */
175             @Override
176             public AudioInputStream nextElement() {
177                 AudioInputStream stream = null;
178                 if (lastFile == null) {
179                     nextFile = readNext();
180                 }
181     
182                 if (nextFile != null) {
183                     try {
184                         AudioInputStream ais = AudioSystem.getAudioInputStream(nextFile);
185     
186                         // test wether all files in the stream have the same format
187                         AudioFormat format = ais.getFormat();
188                         if (!isInitialized) {
189                             isInitialized = true;
190     
191                             bigEndian = format.isBigEndian();
192                             sampleRate = (int) format.getSampleRate();
193                             signedData = format.getEncoding().equals(AudioFormat.Encoding.PCM_SIGNED);
194                             bytesPerValue = format.getSampleSizeInBits() / 8;
195                         }
196     
197                         if (format.getSampleRate() != sampleRate || format.getChannels() != 1 || format.isBigEndian() != bigEndian) {
198                             throw new RuntimeException("format mismatch for subsequent files");
199                         }
200     
201                         stream = ais;
202                         // System.out.println(nextFile);
203     
204                         logger.finer("Strating processing of '" + lastFile.getFile() + '\'');
205                         for (AudioFileProcessListener fl : fileListeners)
206                             fl.audioFileProcStarted(new File(nextFile.getFile()));
207     
208                         lastFile = nextFile;
209                         nextFile = null;
210                     } catch (IOException ioe) {
211                         ioe.printStackTrace();
212                         throw new Error("Cannot convert " + nextFile + " to a FileInputStream");
213                     } catch (UnsupportedAudioFileException e) {
214                         e.printStackTrace();
215                     }
216                 }
217     
218                 return stream;
219             }
220     
221     
222             /**
223              * Returns the name of next audio file
224              *
225              * @return the name of the appropriate audio file
226              */
227             public URL readNext() {
228                 if (lastFile != null) {
229                     logger.finest("Finished processing of '" + lastFile.getFile() + '\'');
230                     for (AudioFileProcessListener fl : fileListeners)
231                         fl.audioFileProcFinished(new File(lastFile.getFile()));
232     
233                     lastFile = null;
234                 }
235     
236                 if (fileIt.hasNext())
237                     lastFile = fileIt.next();
238     
239                 return lastFile;
240             }
241         }
