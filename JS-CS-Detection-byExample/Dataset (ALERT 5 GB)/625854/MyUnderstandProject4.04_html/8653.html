13     package edu.cmu.sphinx.linguist.acoustic.tiedstate;
14     
15     import edu.cmu.sphinx.frontend.Data;
16     import edu.cmu.sphinx.linguist.acoustic.HMM;
17     import edu.cmu.sphinx.linguist.acoustic.HMMState;
18     import edu.cmu.sphinx.linguist.acoustic.HMMStateArc;
19     import edu.cmu.sphinx.util.LogMath;
20     import edu.cmu.sphinx.util.Utilities;
21     
22     import java.util.ArrayList;
23     import java.util.List;
24     
25     /** Represents a single state in an HMM */
26     public class SenoneHMMState implements HMMState {
27     
28         private final SenoneHMM hmm;
29         private final int state;
30         HMMStateArc[] arcs;
31         private final boolean isEmitting;
32         private Senone senone;
33     
34         private static int objectCount;
35     
36     
37         /**
38          * Constructs a SenoneHMMState
39          *
40          * @param hmm   the hmm for this state
41          * @param which the index for this particular state
42          */
43         SenoneHMMState(SenoneHMM hmm, int which) {
44             this.hmm = hmm;
45             this.state = which;
46             this.isEmitting = ((hmm.getTransitionMatrix().length - 1) != state);
47             if (isEmitting) {
48                 SenoneSequence ss = hmm.getSenoneSequence();
49                 senone = ss.getSenones()[state];
50             }
51             Utilities.objectTracker("HMMState", objectCount++);
52         }
53     
54     
55         /**
56          * Gets the HMM associated with this state
57          *
58          * @return the HMM
59          */
60         @Override
61         public HMM getHMM() {
62             return hmm;
63         }
64     
65     
66         /**
67          * Gets the state
68          *
69          * @return the state
70          */
71         @Override
72         public int getState() {
73             return state;
74         }
75     
76     
77         /**
78          * Gets the score for this HMM state
79          *
80          * @param feature the feature to be scored
81          * @return the acoustic score for this state.
82          */
83         @Override
84         public float getScore(Data feature) {
85             return senone.getScore(feature);
86         }
87     
88     
89         /**
90          * Gets the scores for each mixture component in this HMM state
91          *
92          * @param feature the feature to be scored
93          * @return the acoustic scores for the components of this state.
94          */
95         public float[] calculateComponentScore(Data feature) {
96             hmm.getSenoneSequence();
97             return senone.calculateComponentScore(feature);
98         }
99     
100     
101         /**
102          * Gets the senone for this HMM state
103          *
104          * @return the senone for this state.
105          */
106         public Senone getSenone() {
107             return senone;
108         }
109     
110     
111         /**
112          * Determines if two HMMStates are equal
113          *
114          * @param other the state to compare this one to
115          * @return true if the states are equal
116          */
117         @Override
118         public boolean equals(Object other) {
119             if (this == other) {
120                 return true;
121             } else if (!(other instanceof SenoneHMMState)) {
122                 return false;
123             } else {
124                 SenoneHMMState otherState = (SenoneHMMState) other;
125                 return this.hmm == otherState.hmm &&
126                         this.state == otherState.state;
127             }
128         }
129     
130     
131         /**
132          * Returns the hashcode for this state
133          *
134          * @return the hashcode
135          */
136         @Override
137         public int hashCode() {
138             return hmm.hashCode() + state;
139         }
140     
141     
142         /**
143          * Determines if this HMMState is an emittting state
144          *
145          * @return true if the state is an emitting state
146          */
147         // TODO: We may have non-emitting entry states as well.
148         @Override
149         public final boolean isEmitting() {
150             return isEmitting;
151         }
152     
153     
154         /**
155          * Retrieves the state of successor states for this state
156          *
157          * @return the set of successor state arcs
158          */
159         @Override
160         public HMMStateArc[] getSuccessors() {
161             if (arcs == null) {
162                 List<HMMStateArc> list = new ArrayList<HMMStateArc>();
163                 float[][] transitionMatrix = hmm.getTransitionMatrix();
164     
165                 for (int i = 0; i < transitionMatrix.length; i++) {
166                     if (transitionMatrix[state][i] > LogMath.getLogZero()) {
167                         HMMStateArc arc = new HMMStateArc(hmm.getState(i),
168                                 transitionMatrix[state][i]);
169                         list.add(arc);
170                     }
171                 }
172                 arcs = list.toArray(new HMMStateArc[list.size()]);
173             }
174             return arcs;
175         }
176     
177     
178         /**
179          * Determines if this state is an exit state of the HMM
180          *
181          * @return true if the state is an exit state
182          */
183         @Override
184         public boolean isExitState() {
185             // return (hmm.getTransitionMatrix().length - 1) == state;
186             return !isEmitting;
187         }
188     
189     
190         /**
191          * returns a string representation of this object
192          *
193          * @return a string representation
194          */
195         @Override
196         public String toString() {
197             return "HMMS " + hmm + " state " + state;
198         }
199     }
