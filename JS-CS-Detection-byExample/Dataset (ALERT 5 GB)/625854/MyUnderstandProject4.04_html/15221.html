8     package edu.cmu.sphinx.jsgf.rule;
9     
10     import java.util.List;
11     
12     public class JSGFRuleSequence extends JSGFRule {
13     	protected List<JSGFRule> rules;
14     
15     	public JSGFRuleSequence() {
16     		setRules(null);
17     	}
18     
19     	public JSGFRuleSequence(List<JSGFRule> rules) {
20     		setRules(rules);
21     	}
22     
23     	public void append(JSGFRule rule) {
24     		if (rules == null) {
25     			throw new NullPointerException("null rule to append");
26     		}
27     		rules.add(rule);
28     	}
29     
30     	public List<JSGFRule> getRules() {
31     		return rules;
32     	}
33     
34     	public void setRules(List<JSGFRule> rules) {
35     		this.rules = rules;
36     	}
37     
38     	public String toString() {
39     		if (rules.size() == 0) {
40     			return "<NULL>";
41     		}
42     		StringBuilder sb = new StringBuilder();
43     
44     		for (int i = 0; i < rules.size(); ++i) {
45     			if (i > 0)
46     				sb.append(' ');
47     
48     			JSGFRule r = rules.get(i);
49     			if ((r instanceof JSGFRuleAlternatives) || (r instanceof JSGFRuleSequence))
50     				sb.append("( ").append(r).append(" )");
51     			else {
52     				sb.append(r);
53     			}
54     		}
55     		return sb.toString();
56     	}
57     }
