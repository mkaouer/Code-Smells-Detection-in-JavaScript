27      * User: Peter Wolf Date: Nov 10, 2005 Time: 2:42:06 PM Copyright 2005, Peter Wolf
28      * <p/>
29      * Runs a NIST corpus as used by the GALE project.  The inputs are a CTL file, and a REF file.  The output is a CTM
30      * file.
31      * <p/>
32      * A CTL file contains a list of utterances to decode. The format is
33      * <p/>
34      * <utterance file> <start offset> <end offset> <utterance name>
35      * <p/>
36      * The <utterance file> is a base to which the property "dataDirectory" is prepended, and ".raw" is appended.  The
37      * utterance file should be raw PCM that agrees with the "bitsPerSample", "channelCount", "samplesPerSecond", and
38      * "framesPerSecond" properties.
39      * <p/>
40      * The <start offset> and <end offset> are specified in frames, where
41      * <p/>
42      * bytesPerFrame = (bitsPerSample/8)*channelCount*samplesPerSecond/framesPerSecond
43      * <p/>
44      * The <utterance name> should be a unique string.  For example "<utterance file>_<start offset>_<end offset>".
45      * <p/>
46      * A REF file contains the correct transcripts of the utterances specified in the CTL file.  Each line should be of the
47      * form
48      * <p/>
49      * <ASCII transcript> (<utterance name>)
50      * <p/>
51      * The output is a "processed" CTM file.  It is used by the NIST tools to compute the performance on the copus.  The
52      * format is not documented because it is currently a hack to get the Dry Run going.  We need to think more about it. If
53      * you want to use this tool talk to Peter Wolf, or Arthur Chan.
54      */

185     
186             int startOffset;
187             int endOffset;

189             byte[] data;

194                 return file;

200     
201             CTLUtterance(String ctl, String ref) throws CTLException {
202                 /*
203                     example line:
204                     20040422_150000_NTDTV.80Hz-6400Hz 64155 65103 20040422_150000_NTDTV_64155-65103_spk8
205                 */
206                 this.ref = ref;
207                 String[] fields = ctl.split(" ");
208                 if (fields.length != 4) throw new CTLException("CTL Syntax Error: " + ctl);
209                 startOffset = Integer.parseInt(fields[1]);
210                 endOffset = Integer.parseInt(fields[2]);
211                 name = fields[3];
212                 data = new byte[(endOffset - startOffset) * bytesPerFrame];
213                 int i = fields[0].indexOf('.');
214                 file = fields[0];
215                 if (i >= 0) {
216                     file = file.substring(0, i);
217                 }
218                 file = dataDir + '/' + file + ".raw";
219                 try {
220                     InputStream dataStream = new FileInputStream(file);
221                     dataStream.skip(startOffset * bytesPerFrame);
222                     if (dataStream.read(data) != data.length) {
223                         throw new CTLException("Unable to read " + data.length + " bytes of utterance " + name);
224                     }
225                 }
226                 catch (IOException e) {
227                     throw new CTLException("Unable to read utterance " + name + ": " + e.getMessage());
228                 }
229             }

306         protected void setInputStream(CTLUtterance utt) throws IOException {
307             for (DataProcessor dataSource : inputDataProcessors) {
308                 if (dataSource instanceof StreamDataSource) {
309                     ((StreamDataSource) dataSource).setInputStream(utt.getInputStream(), utt.getName());
310                 } else if (dataSource instanceof StreamCepstrumSource) {
311                     boolean isBigEndian = Utilities
312                             .isCepstraFileBigEndian(utt.getName());
313                     StreamCepstrumSource cepstrumSource =
314                             (StreamCepstrumSource) dataSource;
315                     cepstrumSource.setInputStream(utt.getInputStream(), isBigEndian);
316                 }
317             }
318         }

321         public void decode() {
322     
323             try {
324                 utteranceId = 0;
325                 DataOutputStream ctm = new DataOutputStream(new FileOutputStream(ctmFile));
326                 recognizer.allocate();
327     
328                 for (Iterator<CTLUtterance> i = new CTLIterator(); i.hasNext();) {
329                     CTLUtterance utt = i.next();
330                     setInputStream(utt);
331                     Result result = recognizer.recognize();
332                     System.out.println("Utterance " + utteranceId + ": " + utt.getName());
333                     System.out.println("Reference: " + utt.getRef());
334                     System.out.println("Result   : " + result);
335                     logger.info("Utterance " + utteranceId + ": " + utt.getName());
336                     logger.info("Result   : " + result);
337                     handleResult(ctm, utt, result);
338                     utteranceId++;
339                 }
340     
341                 recognizer.deallocate();
342             } catch (IOException io) {
343                 logger.severe("I/O error during decoding: " + io.getMessage());
344             }
345             logger.info("BatchCTLDecoder: " + utteranceId + " utterances decoded");
346         }

354         private int dumpBestPath(DataOutputStream out, CTLUtterance utt, Token token) throws IOException {
355     
356             if (token == null) return 0;
357     
358             Token pred = token.getPredecessor();
359             int startFrame = dumpBestPath(out, utt, pred);
360             if (token.isWord()) {
361     
362                 int endFrame = token.getFrameNumber();
363     
364                 WordSearchState wordState = (WordSearchState) token.getSearchState();
365                 Word word = wordState.getPronunciation().getWord();
366                 String spelling = word.getSpelling();
367                 if (!spelling.startsWith("<")) {
368                     String[] names = utt.name.split("_");
369                     out.write((names[0] + '_' + names[1] + '_' + names[2]
370                             + " 1 " + (utt.startOffset + startFrame) / 100.0 + ' ' + (endFrame - startFrame) / 100.0 + ' ').getBytes());
371                     out.write(hex2Binary(spelling));
372                     out.write(" 0.700000\n".getBytes());
373                 }
374                 return endFrame;
375             }
376             return startFrame;
377         }

380         static public byte[] hex2Binary(String spelling) {
381             byte[] bin = new byte[spelling.length() / 2];
382             for (int i = 0; i < spelling.length(); i += 2) {
383                 int i0 = hexToByte(spelling.charAt(i));
384                 int i1 = hexToByte(spelling.charAt(i + 1));
385                 bin[i / 2] = (byte) (i1 + (16 * i0));
386             }
387             return bin;
388         }

391         static private int hexToByte(char c) {
392             switch (c) {
393                 case '0':
394                     return 0;
395                 case '1':
396                     return 1;
397                 case '2':
398                     return 2;
399                 case '3':
400                     return 3;
401                 case '4':
402                     return 4;
403                 case '5':
404                     return 5;
405                 case '6':
406                     return 6;
407                 case '7':
408                     return 7;
409                 case '8':
410                     return 8;
411                 case '9':
412                     return 9;
413                 case 'a':
414                     return 10;
415                 case 'b':
416                     return 11;
417                 case 'c':
418                     return 12;
419                 case 'd':
420                     return 13;
421                 case 'e':
422                     return 14;
423                 case 'f':
424                     return 15;
425                 default:
426                     throw new Error("Bad hex char " + c);
427             }
