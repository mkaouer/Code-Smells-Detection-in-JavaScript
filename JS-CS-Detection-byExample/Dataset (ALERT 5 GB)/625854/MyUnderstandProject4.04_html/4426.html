163          * @param card the card that the button will be created for

165          * @return the button
166          */
167         private JToggleButton createCardButton(final Card card) {
168             JToggleButton button = getCardButton(card);
169     
170             if (button == null) {
171                 button = new JToggleButton("Card " + card.getID(), 
172                          new ImageIcon(card.getDefaultImageFile()));
173                 button.setSelectedIcon(new ImageIcon(card.getImageFile()));
174                 button.setBorderPainted(true);
175                 button.setContentAreaFilled(false);
176                 button.setHorizontalAlignment(SwingConstants.CENTER);
177                 button.setVerticalAlignment(SwingConstants.CENTER);
178                 button.setVerticalTextPosition(SwingConstants.BOTTOM);
179                 button.setHorizontalTextPosition(SwingConstants.CENTER);
180                 button.addActionListener(
181                     new ActionListener() {
182                         public void actionPerformed(ActionEvent e) {
183                             JToggleButton source = (JToggleButton) e.getSource();
184                             handleCards(card, source.isSelected());
185                         }
186                 });
187                 buttonMap.put(card, button);
188             }
189             return button;
190         }

379         /**
380          * Process the given results
381          *
382          * @param text the spoken text
383          * @param tag the tag associated with the text
384          */
385         void processResults(final String text, final String tag) {
386             try {
387                 SwingUtilities.invokeLater(
388                         new Runnable() {
389                         public void run() {
390                             if (tag != null) {
391                                 if (tag.equals("new_game")) {
392                                     startNewGame(game);
393                                 } else {
394                                     Card card = game.getCard(tag);
395                                     if (card != null) {
396                                         JToggleButton button = getCardButton(card);
397                                         button.setSelected(!button.isSelected());
398                                         handleCards(card, button.isSelected());
399                                     }
400                                 }
401                             } else {
402                                 speak(noInputPrompt);
403                             }
404                             resultsTextField.setText(text);
405                             speakButton.setSelected(false);
406                         }
407                     });
408             } catch (Exception ie) {
409                 ie.printStackTrace();
410             }
411         }

414         /**
415          * Handle card turnovers
416          *
417          * @param card the card that was affected
418          @ @param isSelected if <code>true</code> card was selected.
419          */
420         private void handleCards(Card card, boolean isSelected) {
421             card.setSelected(isSelected);
422     
423             if (!isSelected) {
424                 game.unsetMatch(card);
425             }
426     
427             updateAllCards();
428             if (game.getNumSelected() >= 2) {
429                 if (game.processMatches()) {
430                     if (game.hasWon()) {
431                         speak(victoryPrompt);
432                     } else {
433                         speak(goodGuessPrompt);
434                     }
435                 } else {
436                     speak(badGuessPrompt);
437                     pauseAndTurnOver();
438                 }
439             }
440         }
441     
442         /**
443          * Wait a bit, turn over the unmatched cards
444          */
445         private void pauseAndTurnOver() {
446             Timer timer =new Timer(2000,  
447                     new ActionListener() {
448                        public void actionPerformed(ActionEvent evt) {
449                            game.turnGuessedCards();
450                            updateAllCards();
451                        }
452                     });
453     
454             timer.setRepeats(false);
455             timer.start();
456         }
