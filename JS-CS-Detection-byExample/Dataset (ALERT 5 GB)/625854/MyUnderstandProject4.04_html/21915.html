23     /** A parallel acoustic scorer that is capable of scoring multiple feature streams. */
24     public class ParallelAcousticScorer extends SimpleAcousticScorer {
25     
26         @Override
27         public void newProperties(PropertySheet ps) throws PropertyException {
28             super.newProperties(ps);
29         }
30     
31         /**
32          * Scores the given set of Tokens. All Tokens in the given list are assumed to belong to the same acoustic model.
33          *
34          * @param scoreableList a list containing StateToken objects to be scored
35          * @return the best scoring scorable, or null if there are no more frames to score
36          */
37         @Override
38         public Data calculateScores(List<? extends Scoreable> scoreableList) {
39             frontEnd = getFrontEnd(scoreableList);
40             return super.calculateScores(scoreableList);
41         }
42     
43         /**
44          * Returns the acoustic model name of the Tokens in the given list .
45          *
46          * @return the acoustic model name of the Tokens
47          */
48         private FrontEnd getFrontEnd(List<? extends Scoreable> activeList) {
49            Scoreable scoreable =  activeList.isEmpty() ? null : activeList.get(0);
50            if (scoreable == null || !(scoreable instanceof ParallelToken))
51         	   throw new RuntimeException("Scorer doesn't support anything except list of ParallelTokens");
52            return ((ParallelToken)scoreable).getFeatureStream().getFrontEnd();
53         }
54     }
