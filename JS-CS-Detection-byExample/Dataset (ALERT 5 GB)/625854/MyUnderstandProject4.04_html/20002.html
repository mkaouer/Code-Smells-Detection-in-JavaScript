15     package edu.cmu.sphinx.tools.gui.reader;
16     
17     /**
18      * This class is thrown when there is error during reading operations
19      *
20      * @author Ariani
21      */
22     public class GUIReaderException extends java.lang.Exception {
23         
24         public final static int EXCEPTION_OTHER = 1;
25         public final static int EXCEPTION_IO = 2;
26         public final static int EXCEPTION_NO_FILENAME = 3;
27         public final static int EXCEPTION_FILE_NOT_FOUND = 4;
28         public final static int EXCEPTION_INVALID_TYPE = 5;
29         
30         private final int _mode;
31         
32         /**
33          * Constructs an instance of <code>GUIReaderException</code> with the specified detail message.
34          *
35          * @param msg the detail message.
36          * @param mode the type of error that occurs
37          */
38         public GUIReaderException(String msg, int mode) {
39             super(msg);
40             _mode = mode;
41         }
42         
43         /**
44          * @return Mode of this exception, which explains the type of exception 
45          */
46         public int getErrorCode(){
47             return _mode;
48         }
49     }
