13     package edu.cmu.sphinx.trainer;

16     /**
17      * Indicates stages during training. The numbers prepended to the names indicate the order in which they are performed.
18      * Consecutive numbers were avoided to allow for future expansion of alternative stages. The value of the numbers has no
19      * bearing on the importance or duration of each stage, it is just an ordering method.
20      */
21     public class Stage {
22     
23         private String name;
24     
25         /** Initialization stage. */
26         public static final Stage _00_INITIALIZATION =
27                 new Stage("_00_INITIALIZATION");
28     
29         /** Context independent model training stage. */
30         public static final Stage _10_CI_TRAIN = new Stage("_10_CI_TRAIN");
31     
32         /** Untied context dependent model training stage. */
33         public static final Stage _20_UNTIED_CD_TRAIN =
34                 new Stage("_20_UNTIED_CD_TRAIN");
35     
36         /** State pruning stage. */
37         public static final Stage _30_STATE_PRUNING =
38                 new Stage("_30_STATE_PRUNING");
39     
40         /** Tied context dependent model training stage. */
41         public static final Stage _40_TIED_CD_TRAIN =
42                 new Stage("_40_TIED_CD_TRAIN");
43     
44         /** Copy models, possibly changing format. */
45         public static final Stage _90_CP_MODEL =
46                 new Stage("_90_CP_MODEL");
47     
48     
49         /** Constructs a Stage with the given name. */
50         protected Stage(String name) {
51             this.name = name;
52         }
53     
54     
55         /**
56          * Returns true if the given Stage is equal to this Stage.
57          *
58          * @param stage the Stage to compare
59          * @return true if they are the same, false otherwise
60          */
61         public boolean equals(Stage stage) {
62             if (stage != null) {
63                 return toString().equals(stage.toString());
64             } else {
65                 return false;
66             }
67         }
68     
69     
70         /**
71          * Returns true if the given String is equal to this Stage's name.
72          *
73          * @param stage the Stage to compare
74          * @return true if they are the same, false otherwise
75          */
76         public boolean equals(String stage) {
77             if (stage != null) {
78                 return toString().equals(stage);
79             } else {
80                 return false;
81             }
82         }
83     
84     
85         /**
86          * Returns the name of this Stage.
87          *
88          * @return the name of this Stage.
89          */
90         public String toString() {
91             return name;
92         }
93     
94     }
