79     JNIEnv* getAttachedEnv(JavaVM *jvm){
80     	JNIEnv *localEnv = NULL;
81     	int envErr = 0;
82     
83     	/* get a local java env */
84     	envErr = (*jvm)->AttachCurrentThread( jvm, (void**)&localEnv, NULL );
85     
86     	if ( envErr != 0 ){
87     		if ( (*localEnv)->ExceptionCheck( localEnv ) == JNI_TRUE ){
88     		  (*localEnv)->ExceptionDescribe( localEnv );
89     		}
90     	}
91     
92     	if((*localEnv)->ExceptionOccurred(localEnv)){
93     	    (*localEnv)->ExceptionDescribe(localEnv);
94         }
95     
96     	if (localEnv == NULL) {
97             printf("ERROR: failed to get ENV pointer in pushContext");
98     		//  S4JniExample_destroyJVM(jvm);
99     
100             return (JNIEnv*) NULL;
101         }
102     
103     	return localEnv;
104     }

113     jobject S4JniExample_createTranscriber(JavaVM *jvm){
114     	JNIEnv *env = getAttachedEnv(jvm);
115     	
116     	jclass cls;
117     	jmethodID mid;
118     	jobject transcriber;
119     
120     	cls = (*env)->FindClass(env, TRANSCRIBER_CLASS);
121         if (cls == NULL) {
122             S4JniExample_destroyJVM(jvm);
123         }
124     
125         jstring wavFileString = (*env)->NewStringUTF(env, "foobar.wav");
126     
127     	mid = (*env)->GetMethodID(env, cls, "<init>", "()V");
128     	if (mid == NULL) {
129             S4JniExample_destroyJVM(jvm);
130         }
131     
132     	transcriber = (*env)->NewObject(env, cls, mid, wavFileString);
133     	printf("instantiated Transcriber instance\n");
134     	
135     	//mid = (*env)->GetMethodID(env, cls, "main", "([Ljava/lang/String;)V");
136     	mid = (*env)->GetStaticMethodID(env, cls, "main", "([Ljava/lang/String;)V");
137     	
138     	if (mid == NULL) {
139             S4JniExample_destroyJVM(jvm);
140         }
141     
142     	printf("calling main...\n");
143     	(*env)->CallStaticObjectMethod(env, cls, mid, NULL);
144     	printf("done\n");
