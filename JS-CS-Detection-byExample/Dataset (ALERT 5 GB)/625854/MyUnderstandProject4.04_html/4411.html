14     package tests.live;
15     
16     import javax.sound.sampled.AudioFormat;
17     import javax.sound.sampled.AudioSystem;
18     import javax.sound.sampled.DataLine;
19     import javax.sound.sampled.LineUnavailableException;
20     import javax.sound.sampled.SourceDataLine;
21     
22     
23     /**
24      * Plays audio data to the System's audio device.
25      */
26     public class AudioPlayer {
27     
28         private SourceDataLine line;
29     
30     
31         /**
32          * Constructs a default AudioPlayer.
33          */
34         public AudioPlayer() {}
35     
36     
37         /**
38          * Plays the given byte array audio to the System's audio device.
39          *
40          * @param audio the audio data to play
41          */
42         public void play(byte[] audio, AudioFormat format) {
43             if (openLine(format)) {
44                 line.start();
45                 line.write(audio, 0, audio.length);
46                 line.drain();
47                 line.stop();
48                 line.close();
49                 line = null;
50             }
51         }
52     
53         
54         /**
55          * Opens the audio
56          *
57          * @param format the format for the audio
58          *
59          * @return true if the audio line is successfully opened
60          *         false if the audio line cannot be opened
61          *
62          * @throws UnsupportedOperationException if the line cannot be opened with
63          *     the given format
64          */
65         private boolean openLine(AudioFormat format) {
66             if (line != null) {
67                 line.close();
68             }
69             DataLine.Info info = new DataLine.Info(SourceDataLine.class, format);
70             try {
71                 line = (SourceDataLine) AudioSystem.getLine(info);
72                 line.open(format);
73                 return true;
74             } catch(LineUnavailableException lue) {
75                 System.err.println("Can't open line " + lue);
76                 return false;
77             }   
78         }
79     }
