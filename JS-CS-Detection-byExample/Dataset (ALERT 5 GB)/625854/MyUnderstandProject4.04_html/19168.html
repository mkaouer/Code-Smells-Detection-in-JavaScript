15     package edu.cmu.sphinx.tools.gui.util;
16     
17     import edu.cmu.sphinx.util.props.PropertyType;
18     
19     /**
20      * This class holds information about a <code>Configurable</code> property that 
21      * is owned by a <code>Configurable</code> class. The information includes
22      * name of property, default value, and description
23      *
24      * @author Ariani
25      * @see ConfigurableComponent
26      */
27     public class ConfigurableProperty {
28         
29         private final String _name;
30         private final String _default;
31         private final PropertyType _type;
32         private final String _classtype;
33         private final String _desc;
34         private final String _original_name;
35             
36         /** 
37          * Creates a new instance of ConfigurableProperty 
38          *
39          * @param name Name of property
40          * @param defaultval Default value
41          * @param type Property type
42          * @param desc Description about this property  
43          * @param original Original name of property in the source code   
44          */
45         public ConfigurableProperty(String name, String defaultval, PropertyType type, String desc,
46                 String original) {
47             _name = name;
48             _desc = desc;
49             _type = type;
50             _default = defaultval;      
51             _original_name = original;
52             _classtype = null;
53         
54             System.out.println(" ***** not component property " + name + '_' + defaultval + '_' +
55                     type + '_' + desc + '_' + original);
56         }
57         
58         /** 
59          * Creates a new instance of ConfigurableProperty, and this property needs
60          * a certain class type as value
61          *
62          * @param name Name of property
63          * @param defaultval Default value
64          * @param type Property type
65          * @param desc Description about this property  
66          * @param original Original name of property in the source code   
67          * @param class_type The type of class for this property
68          */
69         public ConfigurableProperty(String name, String defaultval, PropertyType type, String desc,
70                 String original, String class_type) {
71             _name = name;
72             _desc = desc;
73             _type = type;
74             _default = defaultval;      
75             _original_name = original;
76             _classtype = class_type;
77             
78             System.out.println(" ***** component property " + name + '_' + defaultval + '_' +
79                     type + '_' + desc + '_' + original + '_' + class_type);
80         }
81         
82         /**
83          * This method is used to desribe the values of the <code>ConfigurableProperty</code>
84          * It is used mainly for debugging and testing purposes
85          *
86          * @return Description of this property
87          */
88         @Override
89         public String toString(){
90             StringBuilder output = new StringBuilder("\nProperty name: ").append(_name).append(' ');
91             if (_default != null && !_default.isEmpty())
92                 output.append(" Default value : ").append(_default).append(' ');
93             if (!_desc.isEmpty())
94                 output.append("\nDescription: ").append(_desc).append(' ');
95             output.append("Original name : ").append(_original_name).append('\n');
96             return output.toString();
97         }
98     
99         /**
100          * @return Property name
101          */
102         public String getName(){
103             return _name;
104         }
105         
106         /**
107          * @return Default value of property
108          */
109         public String getDefault(){
110             return _default;
111         }
112         
113         /**
114          * @return Type of property
115          */
116         public PropertyType getType(){
117             if (_type == null)
118                 return null;
119             else
120                 return _type;
121         }
122         
123         /**
124          * @return Type of class that's required for this property
125          */
126         public String getClassType(){
127             return _classtype;      
128         }
129         
130         /**
131          * @return original property name inside the source code
132          */
133         public String getOriginalName(){
134             return _original_name;
135         }
136         
137         /**
138          * @return Description of property
139          */
140         public String getDesc(){
141             return _desc;
142         }
143     }
