87         protected void createBuffers() {
88             // the option false or true refers to whether the buffer is in

97         /** Create buffers for a given pool. */
98         private Pool<Buffer> create1DPoolBuffer(Pool<float[]> pool, boolean isLog) {
99             Pool<Buffer> bufferPool = new Pool<Buffer>(pool.getName());
100     
101             for (int i = 0; i < pool.size(); i++) {
102                 float[] element = pool.get(i);
103                 indexMap.put(element, i);
104                 Buffer buffer = new Buffer(element.length, isLog, i);
105                 bufferPool.put(i, buffer);
106             }
107             return bufferPool;
108         }

110         /** Create buffers for a given pool. */
111         private Pool<Buffer[]> create2DPoolBuffer(Pool<float[][]> pool, boolean isLog) {
112             Pool<Buffer[]> bufferPool = new Pool<Buffer[]>(pool.getName());
113     
114             for (int i = 0; i < pool.size(); i++) {
115                 float[][] element = pool.get(i);
116                 indexMap.put(element, i);
117                 int poolSize = element.length;
118                 Buffer[] bufferArray = new Buffer[poolSize];
119                 for (int j = 0; j < poolSize; j++) {
120                     bufferArray[j] = new Buffer(element[j].length, isLog, j);
121                 }
122                 bufferPool.put(i, bufferArray);
123             }
124             return bufferPool;
125         }

137         /**
138          * Accumulate the TrainerScore into the buffers.
139          *
140          * @param index     the current index into the TrainerScore vector
141          * @param score     the TrainerScore for the current frame
142          * @param nextScore the TrainerScore for the next time frame
143          */
144         protected void accumulate(int index, TrainerScore[] score, TrainerScore[] nextScore) {
145             int senoneID;
146             TrainerScore thisScore = score[index];
147     
148             // We should be doing this just once per utterance...
149             // currentLogLikelihood = thisScore.getLogLikelihood();
150     
151             // Since we're scaling, the loglikelihood disappears...
152             currentLogLikelihood = 0;
153             // And the total becomes the sum of (-) scaling factors
154             logLikelihood -= score[0].getScalingFactor();
155     
156             SenoneHMMState state = (SenoneHMMState) thisScore.getState();
157             if (state == null) {
158                 // We only care about the case "all models"
159                 senoneID = thisScore.getSenoneID();
160                 if (senoneID == TrainerAcousticModel.ALL_MODELS) {
161                     accumulateMean(senoneID, score[index]);
162                     accumulateVariance(senoneID, score[index]);
163                     accumulateMixture(senoneID, score[index]);
164                     accumulateTransition(senoneID, index, score, nextScore);
165                 }
166             } else {
167                 // If state is non-emitting, we presume there's only one
168                 // transition out of it. Therefore, we only accumulate
169                 // data for emitting states.
170                 if (state.isEmitting()) {
171                     senoneID = senonePool.indexOf(state.getSenone());
172                     // accumulateMean(senoneID, score[index]);
173                     // accumulateVariance(senoneID, score[index]);
174                     accumulateMixture(senoneID, score[index]);
175                     accumulateTransition(senoneID, index, score, nextScore);
176                 }
177             }
178         }

180         /** Accumulate the means. */
181         private void accumulateMean(int senone, TrainerScore score) {
182             if (senone == TrainerAcousticModel.ALL_MODELS) {
183                 for (int i = 0; i < senonePool.size(); i++) {
184                     accumulateMean(i, score);
185                 }
186             } else {
187                 GaussianMixture gaussian = (GaussianMixture)senonePool.get(senone);
188                 MixtureComponent[] mix = gaussian.getMixtureComponents();
189                 for (int i = 0; i < mix.length; i++) {
190                     float[] mean = mix[i].getMean();
191                     // int indexMean = meansPool.indexOf(mean);
192                     int indexMean = indexMap.get(mean);
193                     assert indexMean >= 0;
194                     assert indexMean == senone;
195                     Buffer buffer = meansBufferPool.get(indexMean);
196                     float[] feature = ((FloatData) score.getData()).getValues();
197                     double[] data = new double[feature.length];
198                     float prob = score.getComponentGamma()[i];
199                     prob -= currentLogLikelihood;
200                     double dprob = logMath.logToLinear(prob);
201                     // prob = (float) logMath.logToLinear(prob);
202                     for (int j = 0; j < data.length; j++) {
203                         data[j] = feature[j] * dprob;
204                     }
205                     buffer.accumulate(data, dprob);
206                 }
207             }
208         }

211         /** Accumulate the variance. */
212         private void accumulateVariance(int senone, TrainerScore score) {
213             if (senone == TrainerAcousticModel.ALL_MODELS) {
214                 for (int i = 0; i < senonePool.size(); i++) {
215                     accumulateVariance(i, score);
216                 }
217             } else {
218                 GaussianMixture gaussian = (GaussianMixture)senonePool.get(senone);
219                 MixtureComponent[] mix = gaussian.getMixtureComponents();
220                 for (int i = 0; i < mix.length; i++) {
221                     float[] mean = mix[i].getMean();
222                     float[] variance = mix[i].getVariance();
223                     // int indexVariance = variancePool.indexOf(variance);
224                     int indexVariance = indexMap.get(variance);
225                     Buffer buffer = varianceBufferPool.get(indexVariance);
226                     float[] feature = ((FloatData) score.getData()).getValues();
227                     double[] data = new double[feature.length];
228                     float prob = score.getComponentGamma()[i];
229                     prob -= currentLogLikelihood;
230                     double dprob = logMath.logToLinear(prob);
231                     for (int j = 0; j < data.length; j++) {
232                         data[j] = (feature[j] - mean[j]);
233                         data[j] *= data[j] * dprob;
234                     }
235                     buffer.accumulate(data, dprob);
236                 }
237             }
238         }

240         /** Accumulate the mixture weights. */
241         private void accumulateMixture(int senone, TrainerScore score) {
242             // The index into the senone pool and the mixture weight pool
243             // is the same
244             if (senone == TrainerAcousticModel.ALL_MODELS) {
245                 for (int i = 0; i < senonePool.size(); i++) {
246                     accumulateMixture(i, score);
247                 }
248             } else {
249                 Buffer buffer = mixtureWeightsBufferPool.get(senone);
250                 float[] mixw = mixtureWeightsPool.get(senone);
251                 for (int i = 0; i < mixw.length; i++) {
252                     float prob = score.getComponentGamma()[i];
253                     prob -= currentLogLikelihood;
254                     buffer.logAccumulate(prob, i, logMath);
255                 }
256             }
257         }

259         /**
260          * Accumulate transitions from a given state.
261          *
262          * @param indexScore the current index into the TrainerScore
263          * @param score      the score information
264          * @param nextScore  the score information for the next frame
265          */
266         private void accumulateStateTransition(int indexScore, TrainerScore[] score, TrainerScore[] nextScore) {
267             HMMState state = score[indexScore].getState();
268             if (state == null) {
269                 // Non-emitting state
270                 return;
271             }
272             int indexState = state.getState();
273             SenoneHMM hmm = (SenoneHMM) state.getHMM();
274             float[][] matrix = hmm.getTransitionMatrix();
275     
276             // Find the index for current matrix in the transition matrix pool
277             // int indexMatrix = matrixPool.indexOf(matrix);
278             int indexMatrix = indexMap.get(matrix);
279     
280             // Find the corresponding buffer
281             Buffer[] bufferArray = matrixBufferPool.get(indexMatrix);
282     
283             // Let's concentrate on the transitions *from* the current state
284             float[] vector = matrix[indexState];
285     
286             for (int i = 0; i < vector.length; i++) {
287                 // Make sure this is a valid transition
288                 if (vector[i] != LogMath.getLogZero()) {
289     
290                     // We're assuming that if the states have position "a"
291                     // and "b" in the HMM, they'll have positions "k+a"
292                     // and "k+b" in the graph, that is, their relative
293                     // position is the same.
294     
295                     // Distance between current state and "to" state in
296                     // the HMM
297                     int dist = i - indexState;
298     
299                     // "to" state in the graph
300                     int indexNextScore = indexScore + dist;
301     
302                     // Make sure the next state is non-emitting (the last
303                     // in the HMM), or in the same HMM.
304                     assert ((nextScore[indexNextScore].getState() == null) ||
305                             (nextScore[indexNextScore].getState().getHMM() == hmm));
306                     float alpha = score[indexScore].getAlpha();
307                     float beta = nextScore[indexNextScore].getBeta();
308                     float transitionProb = vector[i];
309                     float outputProb = nextScore[indexNextScore].getScore();
310                     float prob = alpha + beta + transitionProb + outputProb;
311                     prob -= currentLogLikelihood;
312                     // i is the index into the next state.
313                     bufferArray[indexState].logAccumulate(prob, i, logMath);
314                     /*
315             if ((indexMatrix == 0) && (i == 2)) {
316                 //    	    System.out.println("Out: " + outputProb);
317                     //	    	    bufferArray[indexState].dump();
318             }
319                 */
320                 }
321             }
322         }

324         /**
325          * Accumulate transitions from a given state.
326          *
327          * @param indexState the state index
328          * @param hmm        the HMM
329          * @param value      the value to accumulate
330          */
331         private void accumulateStateTransition(int indexState, SenoneHMM hmm, float value) {
332             // Find the transition matrix in this hmm
333             float[][] matrix = hmm.getTransitionMatrix();
334     
335             // Find the vector with transitions from the current state to
336             // other states.
337             float[] stateVector = matrix[indexState];
338     
339             // Find the index of the current transition matrix in the
340             // transition matrix pool.
341             // int indexMatrix = matrixPool.indexOf(matrix);
342             int indexMatrix = indexMap.get(matrix);
343     
344             // Find the buffer for the transition matrix.
345             Buffer[] bufferArray = matrixBufferPool.get(indexMatrix);
346     
347             // Accumulate for the transitions from current state
348             for (int i = 0; i < stateVector.length; i++) {
349                 // Make sure we're not trying to accumulate in an invalid
350                 // transition.
351                 if (stateVector[i] != LogMath.getLogZero()) {
352                     bufferArray[indexState].logAccumulate(value, i, logMath);
353                 }
354             }
355         }

357         /** Accumulate the transition probabilities. */
358         private void accumulateTransition(int indexHmm, int indexScore, TrainerScore[] score, TrainerScore[] nextScore) {
359             if (indexHmm == TrainerAcousticModel.ALL_MODELS) {
360                 // Well, special case... we want to add an amount to all
361                 // the states in all models
362                 for (HMM hmm : hmmManager) {
363                     for (int j = 0; j < hmm.getOrder(); j++) {
364                         accumulateStateTransition(j, (SenoneHMM)hmm, score[indexScore].getScore());
365                     }
366                 }
367             } else {
368                 // For transition accumulation, we don't consider the last
369                 // time frame, since there's no transition from there to
370                 // anywhere...
371                 if (nextScore != null) {
372                     accumulateStateTransition(indexScore, score, nextScore);
373                 }
374             }
375         }

396          * Normalize a single buffer pool.
397          *
398          * @param pool the buffer pool to normalize
399          */
400         private void normalizePool(Pool<Buffer> pool) {
401             assert pool != null;
402             for (int i = 0; i < pool.size(); i++) {
403                 Buffer buffer = pool.get(i);
404                 if (buffer.wasUsed()) {
405                     buffer.normalize();
406                 }
407             }

425         /**
426          * Normalize a 2D buffer pool in log scale. Typically, this is the case with the transition matrix, which also needs
427          * a mask for values that are allowed, and therefor have to be updated, or not allowed, and should be ignored.
428          *
429          * @param pool     the buffer pool to normalize
430          * @param maskPool pool containing a mask with zero/non-zero values.
431          */
432         private void logNormalize2DPool(Pool<Buffer[]> pool, Pool<float[][]> maskPool) {
433             assert pool != null;
434             for (int i = 0; i < pool.size(); i++) {
435                 Buffer[] bufferArray = pool.get(i);
436                 float[][] mask = maskPool.get(i);
437                 for (int j = 0; j < bufferArray.length; j++) {
438                     if (bufferArray[j].wasUsed()) {
439                         bufferArray[j].logNormalizeNonZero(mask[j]);
440                     }
441                 }
442             }
443         }

455          * Copy one vector onto another.

457          * @param in  the source vector
458          * @param out the destination vector

460         private void copyVector(float[] in, float[] out) {
461             assert in.length == out.length;
462             System.arraycopy(in, 0, out, 0, in.length);
463         }

465         /** Update the means. */
466         private void updateMeans() {
467             assert meansPool.size() == meansBufferPool.size();
468             for (int i = 0; i < meansPool.size(); i++) {
469                 float[] means = meansPool.get(i);
470                 Buffer buffer = meansBufferPool.get(i);
471                 if (buffer.wasUsed()) {
472                     float[] meansBuffer = buffer.getValues();
473                     copyVector(meansBuffer, means);
474                 } else {
475                     logger.info("Senone " + i + " not used.");
476                 }
477             }
478         }
479     
480         /** Update the variances. */
481         private void updateVariances() {
482             assert variancePool.size() == varianceBufferPool.size();
483             for (int i = 0; i < variancePool.size(); i++) {
484                 float[] means = meansPool.get(i);
485                 float[] variance = variancePool.get(i);
486                 Buffer buffer = varianceBufferPool.get(i);
487                 if (buffer.wasUsed()) {
488                     float[] varianceBuffer = buffer.getValues();
489                     assert means.length == varianceBuffer.length;
490                     for (int j = 0; j < means.length; j++) {
491                         varianceBuffer[j] -= means[j] * means[j];
492                         if (varianceBuffer[j] < varianceFloor) {
493                             varianceBuffer[j] = varianceFloor;
494                         }
495                     }
496                     copyVector(varianceBuffer, variance);
497                 }
498             }
499         }

501         /** Recompute the precomputed values in all mixture components. */
502         private void recomputeMixtureComponents() {
503             for (int i = 0; i < senonePool.size(); i++) {
504                 GaussianMixture gMix = (GaussianMixture) senonePool.get(i);
505                 MixtureComponent[] mixComponent = gMix.getMixtureComponents();
506                 for (MixtureComponent component : mixComponent) {
507                     component.precomputeDistance();
508                 }
509             }
510         }

512         /** Update the mixture weights. */
513         private void updateMixtureWeights() {
514             assert mixtureWeightsPool.size() == mixtureWeightsBufferPool.size();
515             for (int i = 0; i < mixtureWeightsPool.size(); i++) {
516                 float[] mixtureWeights = mixtureWeightsPool.get(i);
517                 Buffer buffer = mixtureWeightsBufferPool.get(i);
518                 if (buffer.wasUsed()) {
519                     if (buffer.logFloor(logMixtureWeightFloor)) {
520                         buffer.logNormalizeToSum(logMath);
521                     }
522                     float[] mixtureWeightsBuffer = buffer.getValues();
523                     copyVector(mixtureWeightsBuffer, mixtureWeights);
524                 }
525             }
526         }
527     
528         /** Update the transition matrices. */
529         private void updateTransitionMatrices() {
530             assert matrixPool.size() == matrixBufferPool.size();
531             for (int i = 0; i < matrixPool.size(); i++) {
532                 float[][] matrix = matrixPool.get(i);
533                 Buffer[] bufferArray = matrixBufferPool.get(i);
534                 for (int j = 0; j < matrix.length; j++) {
535                     Buffer buffer = bufferArray[j];
536                     if (buffer.wasUsed()) {
537                         for (int k = 0; k < matrix[j].length; k++) {
538                             float bufferValue = buffer.getValue(k);
539                             if (bufferValue != LogMath.getLogZero()) {
540                                 assert matrix[j][k] != LogMath.getLogZero();
541                                 if (bufferValue < logTransitionProbabilityFloor) {
542                                     buffer.setValue(k, logTransitionProbabilityFloor);
543                                 }
544                             }
545                         }
546                         buffer.logNormalizeToSum(logMath);
547                         copyVector(buffer.getValues(), matrix[j]);
548                     }
549                 }
550             }
551         }
