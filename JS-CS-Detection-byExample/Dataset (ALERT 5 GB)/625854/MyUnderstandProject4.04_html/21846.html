56          * Class constructor.

62             engine.addEngineListener(getEngineListener());

66         /**
67          * Creates the engine listener if necessary, and then returns it. There should be only one.
68          *

72             if (engineListener == null) {
73                 engineListener = new EngineMonitorEngineListener();
74             }
75             return engineListener;
76         }

87             }

92         /**
93          * Gets the panel containing the labels for representing the current engine state.
94          *
95          * @return the panel containing the labels for representing the current engine state.
96          */
97         public Component getStatePanel() {
98             if (statePanel == null) {
99                 JPanel newStatePanel = new JPanel();
100                 newStatePanel.setLayout(new GridLayout(1, 2));
101     
102                 JPanel engineStatePanel = new JPanel();
103                 engineStatePanel.setLayout(new GridLayout(4, 2));
104                 engineStatePanel.setBorder(
105                         BorderFactory.createTitledBorder("Engine State:"));
106     
107                 deallocatedLabel = new JLabel("DEALLOCATED");
108                 allocatingResourcesLabel = new JLabel("ALLOCATING_RESOURCES");
109                 allocatedLabel = new JLabel("ALLOCATED");
110                 deallocatingResourcesLabel = new JLabel("DEALLOCATING_RESOURCES");
111                 pausedLabel = new JLabel("PAUSED");
112                 resumedLabel = new JLabel("RESUMED");
113     
114                 engineStatePanel.add(deallocatedLabel);
115                 engineStatePanel.add(pausedLabel);
116                 engineStatePanel.add(allocatedLabel);
117                 engineStatePanel.add(resumedLabel);
118                 engineStatePanel.add(deallocatingResourcesLabel);
119                 engineStatePanel.add(new JLabel(""));
120                 engineStatePanel.add(allocatingResourcesLabel);
121                 newStatePanel.add(engineStatePanel);
122                 statePanel = newStatePanel;
123             }
124             return statePanel;
125         }

127     
128         /**
129          * Handles an event from the engine.
130          *
131          * @param e the event from the engine
132          */
133         protected void handleEvent(EngineEvent e) {
134             if (eventPanel != null) {
135                 eventPanel.addText(new Date() + ": " + e + '\n');
136                 eventPanel.addText("   Old state: " + engineStateString(e.getOldEngineState()) + '\n');
137                 eventPanel.addText("   New state: " + engineStateString(e.getNewEngineState()) + '\n');
138             }
139             updateGUIComponents();
140         }
141     
142     
143         /** Checks the current state of the engine and makes sure the GUI components reflect this state accurately. */
144         protected void updateGUIComponents() {
145             updateEngineStateComponents();
146         }
147     
148     
149         /** Checks the current state of the engine and makes sure the GUI components reflect this state accurately. */
150         protected void updateEngineStateComponents() {
151             if (statePanel != null) {
152                 deallocatedLabel.setEnabled(
153                         engine.testEngineState(Engine.DEALLOCATED));
154                 allocatingResourcesLabel.setEnabled(
155                         engine.testEngineState(Engine.ALLOCATING_RESOURCES));
156                 allocatedLabel.setEnabled(
157                         engine.testEngineState(Engine.ALLOCATED));
158                 deallocatingResourcesLabel.setEnabled(
159                         engine.testEngineState(Engine.DEALLOCATING_RESOURCES));
160                 pausedLabel.setEnabled(
161                         engine.testEngineState(Engine.PAUSED));
162                 resumedLabel.setEnabled(
163                         engine.testEngineState(Engine.RESUMED));
164             }
165         }
166     
167     
168         /**
169          * Returns a <code>String</code> representing the <code>state</code>.
170          *
171          * @param state the state to turn into a <code>String</code>
172          * @return a <code>String</code> representing the <code>state</code>
173          */
174         protected String engineStateString(long state) {
175             StringBuilder sb = new StringBuilder();
176     
177             if ((state & Engine.DEALLOCATED) != 0)
178                 appendBuilder(sb, "DEALLOCATED");
179             if ((state & Engine.ALLOCATING_RESOURCES) != 0)
180                 appendBuilder(sb, "ALLOCATING_RESOURCES");
181             if ((state & Engine.ALLOCATED) != 0)
182                 appendBuilder(sb, "ALLOCATED");
183             if ((state & Engine.DEALLOCATING_RESOURCES) != 0)
184                 appendBuilder(sb, "DEALLOCATING_RESOURCES");
185     
186             if ((state & Engine.PAUSED) != 0)
187                 appendBuilder(sb, "PAUSED");
188             if ((state & Engine.RESUMED) != 0)
189                 appendBuilder(sb, "RESUMED");
190     
191             return sb.toString();
192         }
193     
194     
195         /**
196          * Adds a <code>String</code> to a <code>StringBuilder</code>,
197          * with each <code>String</code> being separated by a ":".
198          *
199          * @param b the <code>StringBuilder</code> to which to append <code>s</code to
200          * @param s the <code>String</code> to append to <code>b</code>
201          */
202         protected void appendBuilder(StringBuilder b, String s) {
203             if (b.length() > 0)
204                 b.append(':');
205             b.append(s);
206         }
