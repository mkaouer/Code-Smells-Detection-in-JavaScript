20     /**
21      * Compares a reference result strings to actual result strings and keeps track of statistics with regard to the
22      * strings
23      */
24     public class ResultAnalyzer {
25     
26         private final static DecimalFormat percent = new DecimalFormat("%.0");
27         private int numSentences;
28         private int numRefWords;
29         private int numHypWords;
30         private int numMatchingWords;
31         private int numMatchingSentences;
32         private int recognitionErrors;
33         private int insertionErrors;
34         private int deletionErrors;
35     
36         private final boolean verbose;
37     
38         private StringBuffer hypOutput;
39         private StringBuffer refOutput;
40     
41         private final List<Misrecognition> mismatchedUtterances;

44         /**
45          * Creates a result analyzer
46          *
47          * @param verbose if true output comparisons as they are made
48          */
49         public ResultAnalyzer(boolean verbose) {
50             this.verbose = verbose;
51             this.mismatchedUtterances = new LinkedList<Misrecognition>();
52         }
53

55         /**
56          * Compare the hypothesis to the reference string collecting statistics on it. If verbose was set to true,
57          * statistics of the match sent to stdout.
58          *
59          * @param ref the reference string
60          * @param hyp the hypothesis string
61          * @return true if the reference and  hypothesis match
62          */
63         public boolean analyze(String ref, String hyp) {
64             List<String> refList = stringToList(ref);
65             List<String> hypList = stringToList(hyp);
66             String filteredRef = toString(refList);
67             String filteredHyp = toString(hypList);
68             boolean match = false;
69     
70             hypOutput = new StringBuffer();
71             refOutput = new StringBuffer();
72     
73             numRefWords += refList.size();
74             numHypWords += hypList.size();
75             numSentences++;
76     
77             while (!refList.isEmpty() || !hypList.isEmpty()) {
78     
79                 if (refList.isEmpty()) {
80                     addInsert(refList, hypList);
81                 } else if (hypList.isEmpty()) {
82                     addDeletion(refList, hypList);
83                 } else if (!refList.get(0).equals(hypList.get(0))) {
84                     processMismatch(refList, hypList);
85                 } else {
86                     addMatch(refList, hypList);
87                 }
88             }
89     
90             if (filteredHyp.equals(filteredRef)) {
91                 numMatchingSentences++;
92                 match = true;
93             } else {
94                 mismatchedUtterances.add(new Misrecognition(ref, hyp));
95             }
96     
97             if (verbose) {
98                 System.out.println();
99                 System.out.println("REF:       " + filteredRef);
100                 System.out.println("HYP:       " + filteredHyp);
101                 System.out.println("ALIGN_REF: " + refOutput);
102                 System.out.println("ALIGN_HYP: " + hypOutput);
103                 System.out.println();
104                 showResults();
105             }
106     
107             return match;
108         }

111         /**
112          * Returns the accuracy
113          *
114          * @return the accuracy between 0.0 and 1.0
115          */
116         public float getWordAccuracy() {
117             if (numMatchingWords == 0 || numRefWords == 0) {
118                 return 0;
119             } else {
120                 return ((float) numMatchingWords) / ((float) numRefWords);
121             }
122         }

125         /**
126          * Returns the sentence accuracy
127          *
128          * @return the accuracy between 0.0 and 1.0
129          */
130         public float getSentenceAccuracy() {
131             if (numMatchingSentences == 0 || numSentences == 0) {
132                 return 0;
133             } else {
134                 return ((float) numMatchingSentences) / ((float) numSentences);
135             }
136         }

140         public String getHypothesis() {
141             if (hypOutput == null) {
142                 return null;
143             } else {
144                 return hypOutput.toString().trim();
145             }
146         }

150         public void reset() {
151             numSentences = 0;
152             numRefWords = 0;
153             numHypWords = 0;
154             numMatchingWords = 0;
155             numMatchingSentences = 0;
156             recognitionErrors = 0;
157             insertionErrors = 0;
158             deletionErrors = 0;
159             mismatchedUtterances.clear();
160         }

163         /**
164          * convert the list of words back to a space separated string
165          *
166          * @param list the list of words
167          * @return a space separated string
168          */
169         private String toString(List<String> list) {
170             if (list.isEmpty())
171                 return "";        
172             StringBuilder sb = new StringBuilder();
173             for (String s : list)
174                 sb.append(s).append(' ');
175             sb.setLength(sb.length() - 1);
176             return sb.toString();
177         }

180         /**
181          * Add an insertion error corresponding to the first item on the hypList
182          *
183          * @param refList the list of reference words
184          * @param hypList the list of hypothesis  words
185          */
186         private void addInsert(List<String> refList, List<String> hypList) {
187             insertionErrors++;
188             String word = hypList.remove(0);
189     
190             refOutput.append(' ').append(pad(word.length()));
191             hypOutput.append(' ').append(word.toUpperCase());
192         }
193     
194     
195         /**
196          * Add a deletion error corresponding to the first item on the refList
197          *
198          * @param refList the list of reference words
199          * @param hypList the list of hypothesis  words
200          */
201         private void addDeletion(List<String> refList, List<String> hypList) {
202             deletionErrors++;
203             String word = refList.remove(0);
204     
205             refOutput.append(' ').append(word.toUpperCase());
206             hypOutput.append(' ').append(pad(word.length()));
207         }
208     
209     
210         /**
211          * Add a recognition error
212          *
213          * @param refList the list of reference words
214          * @param hypList the list of hypothesis  words
215          */
216         private void addRecognitionError(List<String> refList, List<String> hypList) {
217             recognitionErrors++;
218             String ref = refList.remove(0);
219             String hyp = hypList.remove(0);
220             int length = Math.max(ref.length(), hyp.length());
221     
222             refOutput.append(' ').append(pad(ref.toUpperCase(), length));
223             hypOutput.append(' ').append(pad(hyp.toUpperCase(), length));
224         }
225     
226     
227         /**
228          * Add a match
229          *
230          * @param refList the list of reference words
231          * @param hypList the list of hypothesis  words
232          */
233         private void addMatch(List<String> refList, List<String> hypList) {
234             numMatchingWords++;
235             String ref = refList.remove(0);
236             String hyp = hypList.remove(0);
237             refOutput.append(' ').append(ref);
238             hypOutput.append(' ').append(hyp);
239         }

242         /**
243          * Process a mismatch by seeing which type of error is most likely
244          *
245          * @param refList the list of reference words
246          * @param hypList the list of hypothesis  words
247          */
248         private void processMismatch(List<String> refList, List<String> hypList) {
249             int deletionMatches = countMatches(
250                     refList, 1, hypList, 0);
251             int insertMatches = countMatches(
252                     refList, 0, hypList, 1);
253             int normalMatches = countMatches(refList, 0, hypList, 0);
254     
255             if (deletionMatches > insertMatches &&
256                     deletionMatches > normalMatches) {
257                 addDeletion(refList, hypList);
258             } else if (insertMatches > deletionMatches &&
259                     insertMatches > normalMatches) {
260                 addInsert(refList, hypList);
261             } else {
262                 addRecognitionError(refList, hypList);
263             }
264         }

267         /**
268          * Counts the number of matches between the two lists starting at the respective indexes
269          *
270          * @param refList  the list of reference words
271          * @param refIndex the starting point in the ref list
272          * @param hypList  the list of hypothesis  words
273          * @param hypIndex the starting point in the hyp list
274          * @return the number of matching words
275          */
276         private int countMatches(List<String> refList, int refIndex,
277                                  List<String> hypList, int hypIndex) {
278             int match = 0;
279     
280             while (refIndex < refList.size() && hypIndex < hypList.size()) {
281                 String ref = refList.get(refIndex++);
282                 String hyp = hypList.get(hypIndex++);
283                 if (ref.equals(hyp)) {
284                     match++;
285                 }
286             }
287             return match;
288         }

291         /**
292          * Returns a string of "*" of the given length
293          *
294          * @param length the length of the resulting string
295          * @return the string
296          */
297         private String pad(int length) {
298             StringBuilder result = new StringBuilder(length);
299             for (int i = 0; i < length; i++) {
300                 result.append('*');
301             }
302             return result.toString();
303         }
304     
305     
306         /**
307          * Pads the given string with spaces to the given length
308          *
309          * @param s      the string to pad
310          * @param length the length of the resulting string
311          * @return the padded string
312          */
313         private String pad(String s, int length) {
314             StringBuilder result = new StringBuilder(length);
315             result.append(s);
316             for (int i = s.length(); i < length; i++) {
317                 result.append(' ');
318             }
319             return result.toString();
320         }

323         /**
324          * Converts the given string to a list
325          *
326          * @param s the string to convert
327          * @return a list, one word per item with silences removed
328          */
329         private List<String> stringToList(String s) {
330             List<String> list = new LinkedList<String>();
331             StringTokenizer st = new StringTokenizer(s);
332     
333             while (st.hasMoreTokens()) {
334                 String word = st.nextToken();
335                 list.add(word);
336             }
337             return list;
338         }

344                     (mismatchedUtterances.size() + " sentence errors");

352         /** Shows the results for this analyzer */
353         public void showResults() {
354             if (numSentences > 0) {
355                 int totalErrors = recognitionErrors
356                         + insertionErrors + deletionErrors;
357                 float wordErrorRate = (recognitionErrors + insertionErrors
358                         + deletionErrors) / (float) numRefWords;
359     
360                 System.out.print("   Accuracy: " +
361                         percent.format(getWordAccuracy()));
362                 System.out.println("    Errors: " + totalErrors +
363                         "  (Sub: " + recognitionErrors +
364                         "  Ins: " + insertionErrors +
365                         "  Del: " + deletionErrors + ')');
366                 System.out.println("   Words: " + numRefWords +
367                         "   Matches: " + numMatchingWords +
368                         "    WER: " + percent.format(wordErrorRate));
369                 System.out.println("   Sentences: " + numSentences +
370                         "   Matches: " + numMatchingSentences +
371                         "   SentenceAcc: " +
372                         percent.format(getSentenceAccuracy()));
373             }
374         }

403     /** Represents the reference and hypothesis of a misrecognized utterance. */
404     class Misrecognition {
405     
406         private final String reference;
407         private final String hypothesis;
408     
409     
410         /**
411          * Constructs a Misrecognition.
412          *
413          * @param reference  the reference utterance
414          * @param hypothesis the hypothesis utterance
415          */
416         public Misrecognition(String reference, String hypothesis) {
417             this.reference = reference;
418             this.hypothesis = hypothesis;
419         }
420     
421     
422         /**
423          * Returns the reference.
424          *
425          * @return the reference string
426          */
427         public String getReference() {
428             return "REF: " + reference;
429         }
430     
431     
432         /**
433          * Returns the hypothesis string.
434          *
435          * @return the hypothesis string
436          */
437         public String getHypothesis() {
438             return "HYP: " + hypothesis;
439         }
