13     package edu.cmu.sphinx.linguist.acoustic.tiedstate.trainer;
14     
15     import edu.cmu.sphinx.linguist.acoustic.tiedstate.Loader;
16     import edu.cmu.sphinx.linguist.acoustic.tiedstate.Saver;
17     import edu.cmu.sphinx.linguist.acoustic.tiedstate.TiedStateAcousticModel;
18     import edu.cmu.sphinx.linguist.acoustic.tiedstate.Sphinx3Loader;
19     import edu.cmu.sphinx.util.TimerPool;
20     import edu.cmu.sphinx.util.props.*;
21     
22     import java.io.FileNotFoundException;
23     import java.io.IOException;
24     import java.util.logging.Logger;
25     
26     /** Represents the generic interface to the Acoustic Model for sphinx4 */
27     public class TrainerAcousticModel extends TiedStateAcousticModel {
28     
29         /** Prefix for acoustic model SphinxProperties. */
30         public final static String PROP_PREFIX = "edu.cmu.sphinx.linguist.acoustic.";
31     
32     
33         @S4Component(type = Loader.class)
34         public static final String LOADER = "loader";
35         private Loader loader;
36     
37         @S4Component(type = Saver.class)
38         public static final String SAVER = "saver";
39         private Saver saver;
40     
41     
42         @S4Double(defaultValue = 0.0001f)
43         public final static String PROP_VARIANCE_FLOOR = "varianceFloor";
44     
45         /** Mixture component score floor. */
46         @S4Double(defaultValue = 0.0)
47         public final static String PROP_MC_FLOOR = "MixtureComponentScoreFloor";
48     
49         /** Mixture weight floor. */
50         @S4Double(defaultValue = 1e-7f)
51         public final static String PROP_MW_FLOOR = "mixtureWeightFloor";
52     
53         /**
54          * The save format for the acoustic model data. Current supported formats are:
55          * <p/>
56          * sphinx3.ascii sphinx3.binary sphinx4.ascii sphinx4.binary
57          */
58         @S4String(defaultValue = "sphinx3.binary")
59         public final static String PROP_FORMAT_SAVE = PROP_PREFIX + "format.save";
60     
61     
62         /** Flag indicating all models should be operated on. */
63         public final static int ALL_MODELS = -1;
64     
65         /** The logger for this class */
66         private Logger logger;
67     
68         /** The pool manager */
69         private HMMPoolManager hmmPoolManager;
70         public String saveFormat;
71     
72     
73         @Override
74         public void newProperties(PropertySheet ps) throws PropertyException {
75             super.newProperties(ps);
76             logger = ps.getLogger();
77     
78             loader = (Loader) ps.getComponent(LOADER);
79             saver = (Saver) ps.getComponent(SAVER);
80     
81             hmmPoolManager = new HMMPoolManager(loader);
82             loadTimer = TimerPool.getTimer(this, TIMER_LOAD);
83             saveFormat = ps.getString(PROP_FORMAT_SAVE);
84     
85             logInfo();
86         }
87     
88     
89         /**
90          * Saves the acoustic model with a given name and format
91          *
92          * @param name the name of the acoustic model
93          * @throws IOException           if the model could not be loaded
94          * @throws FileNotFoundException if the model does not exist
95          */
96         public void save(String name) throws IOException {
97             saver.save(name, true);
98             logger.info("saved models with " + saver);
99         }
100     
101     
102         /**
103          * Loads the acoustic models. This has to be explicitly requested in this class.
104          *
105          * @throws IOException           if the model could not be loaded
106          * @throws FileNotFoundException if the model does not exist
107          */
108         public void load() throws IOException, FileNotFoundException {
109             loadTimer.start();
110     //        super.load();
111             loadTimer.stop();
112             logInfo();
113             hmmPoolManager = new HMMPoolManager(loader);
114         }
115     
116     
117         /** Resets the buffers. */
118         public void resetBuffers() {
119             // Resets the buffers and associated variables.
120             hmmPoolManager.resetBuffers();
121         }
122     
123     
124         /**
125          * Accumulate the current TrainerScore into the buffers.
126          *
127          * @param index            the current index into the TrainerScore vector
128          * @param trainerScore     the TrainerScore in the current frame
129          * @param nextTrainerScore the TrainerScore in the next frame
130          */
131         public void accumulate(int index, TrainerScore[] trainerScore,
132                                TrainerScore[] nextTrainerScore) {
133             hmmPoolManager.accumulate(index, trainerScore, nextTrainerScore);
134         }
135     
136     
137         /**
138          * Accumulate the current TrainerScore into the buffers.
139          *
140          * @param index        the current index into the TrainerScore vector
141          * @param trainerScore the TrainerScore
142          */
143         public void accumulate(int index, TrainerScore[] trainerScore) {
144             hmmPoolManager.accumulate(index, trainerScore);
145         }
146     
147     
148         /** Update the log likelihood. This should be called at the end of each utterance. */
149         public void updateLogLikelihood() {
150             hmmPoolManager.updateLogLikelihood();
151         }
152     
153     
154         /**
155          * Normalize the buffers and update the models.
156          *
157          * @return the log likelihood for the whole training set
158          */
159         public float normalize() {
160             float logLikelihood = hmmPoolManager.normalize();
161             hmmPoolManager.update();
162             return logLikelihood;
163         }
164     
165     
166     }
