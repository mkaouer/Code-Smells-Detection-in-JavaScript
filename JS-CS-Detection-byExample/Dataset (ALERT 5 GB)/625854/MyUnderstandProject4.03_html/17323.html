13     package edu.cmu.sphinx.result;
14     
15     import edu.cmu.sphinx.frontend.Data;
16     import edu.cmu.sphinx.linguist.dictionary.Pronunciation;
17     import edu.cmu.sphinx.util.LogMath;
18     
19     
20     /**
21      * Represents a word in a recognition result. This is designed specifically for obtaining confidence scores. All scores
22      * are maintained in LogMath log base.
23      */
24     public interface WordResult {
25     
26         /**
27          * Gets the total score for this word.
28          *
29          * @return the score for the word (in LogMath log base)
30          */
31         public double getScore();
32     
33     
34         /**
35          * Returns a log confidence score for this WordResult. Use the getLogMath().logToLinear() method to convert the log
36          * confidence score to linear. The linear value should be between 0.0 and 1.0 (inclusive) for this word.
37          *
38          * @return a log confidence score which linear value is between 0.0 and 1.0 (inclusive)
39          */
40         public double getConfidence();
41     
42     
43         /**
44          * Returns the log math of the scores.
45          *
46          * @return the log math of the scores
47          */
48         public LogMath getLogMath();
49     
50     
51         /**
52          * Gets the pronunciation for this word.
53          *
54          * @return the pronunciation for the word
55          */
56         public Pronunciation getPronunciation();
57     
58     
59         /**
60          * Gets the starting frame number for the word
61          *
62          * @return the starting frame number for the word
63          */
64         public int getStartFrame();
65     
66     
67         /**
68          * Gets the ending frame number for the word
69          *
70          * @return the ending frame number for the word
71          */
72         public int getEndFrame();
73     
74     
75         /**
76          * Gets the feature frames associated with this word
77          *
78          * @return the set of feature frames associated with this word, or null if the frames are not available.
79          */
80         public Data[] getDataFrames();
81     
82     
83         /** Returns a string representation of this object */
84         public String toString();
85     
86     
87         /**
88          * Does this word result represent a filler token?
89          *
90          * @return true if this is a filler
91          */
92         public boolean isFiller();
93     }
