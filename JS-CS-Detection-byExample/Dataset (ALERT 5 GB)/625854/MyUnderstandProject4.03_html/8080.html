47         /**
48          * Scores the given set of states.
49          *
50          * @param scoreableList A list containing scoreable objects to be scored
51          * @return The best scoring scoreable, or <code>null</code> if there are no more features to score
52          */
53         public Data calculateScores(List<? extends Scoreable> scoreableList) {
54     
55         	try {
56                 Data data = getNextData();
57                 while (data instanceof Signal) {
58                     if (data instanceof SpeechEndSignal || data instanceof DataEndSignal) {
59                         return data;
60                     }
61     
62                     data = getNextData();
63                 }
64     
65                 if (data == null)
66                 	return null;
67     
68                 if (scoreableList.size() <= 0)
69                 	return null;
70                 
71                 // convert the data to FloatData if not yet done
72                 if (data instanceof DoubleData)
73                     data = DataUtil.DoubleData2FloatData((DoubleData) data);
74     
75                 Data bestToken = doScoring(scoreableList, data);
76     
77                 // apply optional score normalization
78                 if (scoreNormalizer != null && bestToken instanceof Token)
79                     bestToken = scoreNormalizer.normalize(scoreableList, (Token)bestToken);
80     
81                 return bestToken;
82             } catch (DataProcessingException dpe) {
83                 dpe.printStackTrace();
84                 return null;
85             }
86         }

89         private Data getNextData() {
90             Data data = frontEnd.getData();
91     
92             // reconfigure the scorer for the coming data stream
93             if (data instanceof DataStartSignal) {
94                 handleDataStartSignal((DataStartSignal) data);
95             }
96     
97             if (data instanceof DataEndSignal)
98                 handleDataEndSignal((DataEndSignal) data);
99     
100             return data;
101         }

104         /** Handles the first element in a feature-stream. */
105         protected void handleDataStartSignal(DataStartSignal dataStartSignal) {
106             Map<String, Object> dataProps = dataStartSignal.getProps();
107     
108             if (dataProps.containsKey(DataStartSignal.SPEECH_TAGGED_FEATURE_STREAM))
109                 useSpeechSignals = (Boolean) dataProps.get(DataStartSignal.SPEECH_TAGGED_FEATURE_STREAM);
110             else
111                 useSpeechSignals = false;
112         }
113     
114     
115         /** Handles the last element in a feature-stream. */
116         protected void handleDataEndSignal(DataEndSignal dataEndSignal) {
117             // we don't treat the end-signal here, but extending classes might do
118         }

121         public void startRecognition() {
122             if (useSpeechSignals == null) {
123                 Data firstData = getNextData();
124                 
125                 if (firstData == null)
126             	    return;
127                 
128                 assert firstData instanceof DataStartSignal :
129                         "The first element in an sphinx4-feature stream must be a DataStartSignal but was a " + firstData.getClass().getSimpleName();
130             }
131     
132             if (!useSpeechSignals)
133                 return;
134     
135             Data data = getNextData();
136             while (!(data instanceof SpeechStartSignal)) {
137                 if (data == null) {
138                     break;
139                 }
140     
141                 data = getNextData();
142             }
143         }
