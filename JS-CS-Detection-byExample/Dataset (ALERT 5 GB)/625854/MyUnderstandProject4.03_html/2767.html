35     /**
36      * Decodes a batch file containing a list of files to decode. The files can be either audio files or cepstral files, but
37      * defaults to audio files. The audio data should be 16-bit, 16kHz, PCM-linear data. Since this classes makes use of
38      * Java Sound, it supports all the audio file formats that are supported by Java Sound. If the audio file does not
39      * correspond to a format supported by Java Sound, it is treated as a raw audio file (i.e., one without a header). Audio
40      * file formats differ in the endian order of the audio data. Therefore, it is important to specify it correctly in the
41      * configuration of the <a href="../../frontend/util/StreamDataSource.html">StreamDataSource</a>. Note that in the ideal
42      * situation, the audio format of the data should be passed into the StreamDataSource, so that no extra configuration is
43      * needed. This will be fixed in future releases.
44      * <p/>
45      * To run this BatchModeRecognizer:
46      * <pre>
47      * java BatchModeRecognizer &lt;xmlConfigFile&gt; &lt;batchFile&gt;
48      * </pre>
49      * where <code>xmlConfigFile</code> is an XML-based configuration file and <code>batchFile</code> is a file listing all
50      * the files to decode and transcript of those files. For information about the configuration file, refer to the
51      * document <a href="../../util/props/doc-files/ConfigurationManagement.html"> Sphinx-4 Configuration Management</a>.
52      * For information about the batch file, refer to the <a href="../../../../../../index.html#batch_files"> batch file
53      * description</a>.
54      * <p/>
55      * This class will send recognition results to the logger if the log level is set to INFO.
56      */
57     public class BatchModeRecognizer implements Configurable {

151         /** Decodes the batch of audio files */
152         public void decode(String batchFile) throws IOException {
153             BatchItem batchItem;
154             int count = 0;
155             try {
156                 recognizer.allocate();
157                 setBatchFile(batchFile);
158     
159                 batchManager.start();
160                 logger.info("BatchDecoder: decoding files in "
161                         + batchManager.getFilename());
162     
163                 while (count < utteranceId &&
164                         (batchItem = batchManager.getNextItem()) != null) {
165                     setInputStream(batchItem.getFilename());
166                     Result result = recognizer.recognize(batchItem.getTranscript());
167                     logger.info("File  : " + batchItem.getFilename());
168                     logger.info("Result: " + result);
169                     count++;
170                 }
171                 batchManager.stop();
172                 recognizer.deallocate();
173             } catch (IOException io) {
174                 logger.severe("I/O error during decoding: " + io.getMessage());
175                 throw io;
176             }
177             logger.info("BatchDecoder: " + count + " files decoded");
178         }

181         /**
182          * Sets the input stream to the given filename
183          *
184          * @param filename the filename to set the input stream to
185          * @throws IOException if an error occurs
186          */
187         void setInputStream(String filename) throws IOException {
188             for (Object inputDataProcessor : inputDataProcessors) {
189                 DataProcessor dataSource = (DataProcessor) inputDataProcessor;
190                 InputStream is;
191                 try {
192                     File file = new File(filename);
193                     logger.info
194                             (AudioSystem.getAudioFileFormat(file).toString());
195                     is = AudioSystem.getAudioInputStream(file);
196                 } catch (UnsupportedAudioFileException uafe) {
197                     logger.info
198                             ("Reading " + filename + " as raw audio file.");
199                     is = new FileInputStream(filename);
200                 }
201                 if (dataSource instanceof StreamDataSource) {
202                     ((StreamDataSource) dataSource).setInputStream(is, filename);
203                 } else if (dataSource instanceof StreamCepstrumSource) {
204                     boolean isBigEndian = Utilities
205                             .isCepstraFileBigEndian(filename);
206                     StreamCepstrumSource cepstrumSource =
207                             (StreamCepstrumSource) dataSource;
208                     cepstrumSource.setInputStream(is, isBigEndian);
209                 }
210             }
211         }

509         public void shell(String batchfile) throws IOException {
510             try {
511                 CommandInterpreter ci = new CommandInterpreter();
512                 ci.setPrompt("s4> ");
513                 addCommands(ci);
514                 setBatchFile(batchfile);
515                 recognizer.allocate();
516                 ci.run();
517                 batchManager.stop();
518                 if (recognizer.getState() == RecognizerState.READY) {
519                     recognizer.deallocate();
520                 }
521             } catch (IOException io) {
522                 logger.severe("I/O error during decoding: " + io.getMessage());
523                 throw io;
524             }
525         }

610         private static void setProperty(String componentName, String propName, String propValue, ConfigurationManager cm) {
611     
612             PropertySheet ps = cm.getPropertySheet(componentName);
613             try {
614                 Proxy wrapper = ps.getProperty(propName, Object.class).getAnnotation();
615                 if (wrapper instanceof S4Component) {
616                     ps.setComponent(propName, propValue, cm.lookup(propValue));
617                 } else if (wrapper instanceof S4Boolean)
618                     ps.setBoolean(propName, Boolean.valueOf(propValue));
619                 else if (wrapper instanceof S4Integer)
620                     ps.setInt(propName, Integer.valueOf(propValue));
621                 else if (wrapper instanceof S4String)
622                     ps.setString(propName, propValue);
623                 else if (wrapper instanceof S4Double)
624                     ps.setDouble(propName, Double.valueOf(propValue));
625                 else if (wrapper instanceof S4ComponentList)
626                     throw new RuntimeException("to set component lists please use PS.setComponentList()");
627     //                   ps.setComponentList(propName, null, cm.lookup(propValue));
628             } catch (PropertyException e) {
629                 e.printStackTrace();
630             }
631         }
