6     /**
7      * Removes excessive non-speech-segments from a speech stream. Compared with <code>NonSpeechDatatFilter</code> this
8      * component does not remove all non-speech frames. It just reduces the non-speech parts to a user defined length.
9      *
10      * @see SpeechMarker
11      * @see NonSpeechDataFilter
12      */
13     public class ExcessiveNonSpeechPruner extends BaseDataProcessor {
14     
15         /**
16          * The Sphinx Property for the maximum amount of (subsequent) none-speech time (in ms) to be preserved in the speech
17          * stream.
18          */
19         @S4Integer(defaultValue = Integer.MAX_VALUE)
20         public static final String PROP_MAX_NON_SPEECH_TIME_MS = "maxNonSpeechTimeMs";
21     
22         private int maxNonSpeechTime;
23         private boolean inSpeech;
24         private int nonSpeechCounter = 0;
25     
26     
27         /**
28          * Returns the processed Data output.
29          *
30          * @return an Data object that has been processed by this DataProcessor
31          * @throws edu.cmu.sphinx.frontend.DataProcessingException
32          *          if a data processor error occurs
33          */
34         public Data getData() throws DataProcessingException {
35             Data data = getPredecessor().getData();
36     
37             if (data instanceof SpeechEndSignal || data instanceof DataStartSignal) {
38                 inSpeech = false;
39                 nonSpeechCounter = 0;
40             } else if (data instanceof SpeechStartSignal) {
41                 inSpeech = true;
42             } else if (data instanceof DoubleData || data instanceof FloatData) {
43                 if (!inSpeech) {
44                     nonSpeechCounter += getAudioTime(data);
45                     if (nonSpeechCounter >= maxNonSpeechTime)
46                         data = getData();
47                 }
48             }
49     
50             return data;
51         }
52     
53     
54         /**
55          * Returns the amount of audio data in milliseconds in the given SpeechClassifiedData object.
56          *
57          * @param data the SpeechClassifiedData object
58          * @return the amount of audio data in milliseconds
59          */
60         public int getAudioTime(Data data) {
61             if (data instanceof DoubleData) {
62                 DoubleData audio = (DoubleData) data;
63                 return (int) ((audio.getValues().length * 1000.0f / audio.getSampleRate()));
64             } else if (data instanceof FloatData) {
65                 FloatData audio = (FloatData) data;
66                 return (int) ((audio.getValues().length * 1000.0f / audio.getSampleRate()));
67             }
68     
69             assert false;
70             return -1;
71         }

78         */
79         public void newProperties(PropertySheet ps) throws PropertyException {
