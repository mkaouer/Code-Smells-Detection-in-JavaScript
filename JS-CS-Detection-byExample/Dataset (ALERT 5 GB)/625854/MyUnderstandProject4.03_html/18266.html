13     package edu.cmu.sphinx.result;
14     
15     import edu.cmu.sphinx.frontend.Data;
16     import edu.cmu.sphinx.linguist.acoustic.Unit;
17     
18     /**
19      * Contains statistics about a frame.
20      * <p/>
21      * Note that all scores are maintained in LogMath log base
22      */
23     public abstract class FrameStatistics {
24     
25         /**
26          * Gets the frame number
27          *
28          * @return the frame number
29          */
30         public abstract int getFrameNumber();
31     
32     
33         /**
34          * Gets the feature associated with this frame
35          *
36          * @return the feature associated with the frame or null if the feature is not available
37          */
38         public abstract Data getData();
39     
40     
41         /**
42          * Gets the best score for this frame
43          *
44          * @return the best score for this frame in the LogMath log domain
45          */
46         public abstract float getBestScore();
47     
48     
49         /**
50          * Gets the unit that had the best score for this frame
51          *
52          * @return the unit with the best score
53          */
54         public abstract Unit getBestUnit();
55     
56     
57         /**
58          * Gets the best scoring hmm state for this frame
59          *
60          * @return the best scoring state
61          */
62         public abstract int getBestState();
63     }
