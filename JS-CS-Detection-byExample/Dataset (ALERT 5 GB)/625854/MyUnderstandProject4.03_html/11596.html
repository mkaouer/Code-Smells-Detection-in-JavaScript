13     package edu.cmu.sphinx.util;
14     
15     import java.io.BufferedReader;
16     import java.io.FileReader;
17     import java.io.IOException;
18     import java.util.ArrayList;
19     import java.util.List;
20     
21     
22     /** Provides a set of utilities methods for manipulating batch files. */
23     public class BatchFile {
24     
25         /**
26          * Returns a List of the lines in a batch file.
27          *
28          * @param batchFile the batch file to read
29          * @return a List of the lines in a batch file
30          */
31         public static List<String> getLines(String batchFile) throws IOException {
32             return getLines(batchFile, 0);
33         }
34     
35     
36         /**
37          * Returns a List of the lines in a batch file.
38          *
39          * @param batchFile the batch file to read
40          * @param skip      the number of lines to skip between items
41          * @return a List of the lines in a batch file
42          */
43         public static List<String> getLines(String batchFile, int skip) throws IOException {
44             int curCount = skip;
45             List<String> list = new ArrayList<String>();
46             BufferedReader reader = new BufferedReader(new FileReader(batchFile));
47     
48             String line = null;
49     
50             while ((line = reader.readLine()) != null) {
51                 if (line.length() > 0) {
52                     if (++curCount >= skip) {
53                         list.add(line);
54                         curCount = 0;
55                     }
56                 }
57             }
58             reader.close();
59             return list;
60         }
61     
62     
63         /**
64          * Returns the file name portion of a line in a batch file. This is the portion of the line before the first space.
65          *
66          * @return the file name portion of a line in a batch file.
67          */
68         public static String getFilename(String batchFileLine) {
69             int firstSpace = batchFileLine.indexOf(" ");
70             return batchFileLine.substring(0, firstSpace).trim();
71         }
72     
73     
74         /**
75          * Returns the reference string portion of a line in a batch file. This is the portion of the line after the first
76          * space
77          *
78          * @return the reference string portion of a line in a batch file.
79          */
80         public static String getReference(String batchFileLine) {
81             int firstSpace = batchFileLine.indexOf(" ");
82             return batchFileLine.substring(firstSpace + 1).trim();
83         }
84     }
