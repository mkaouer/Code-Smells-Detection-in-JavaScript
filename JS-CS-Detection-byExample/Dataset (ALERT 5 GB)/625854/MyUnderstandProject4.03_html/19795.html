14     /** Adds a uniformly distribued dithering to a signal. */
15     public class Dither extends BaseDataProcessor {
16     
17         /** The maximal value which could be added/substracted to/from the signal*/
18         @S4Double(defaultValue = 0.0)
19         public static final String PROP_MAX_DITHER = "maxDither";
20         private double ditherMax;
21     
22         /** The maximal value of dithered values. */
23         @S4Double(defaultValue = Double.MAX_VALUE)
24         public static final String PROP_MAX_VAL = "upperValueBound";
25         private double maxValue;
26     
27         /** The minimal value of dithered values. */
28         @S4Double(defaultValue = -Double.MAX_VALUE)
29         public static final String PROP_MIN_VAL = "lowerValueBound";
30         private double minValue;

32     
33         /** The name of the sphinx property about using random seed or not */
34         @S4Boolean(defaultValue = false)
35         public static final String PROP_USE_RANDSEED = "useRandSeed";
36         private boolean useRandSeed;

44             useRandSeed = ps.getBoolean(PROP_USE_RANDSEED);

57         /**
58          * Returns the next DoubleData object, which is a dithered version of the input
59          *
60          * @return the next available DoubleData object, or null if no Data is available
61          * @throws edu.cmu.sphinx.frontend.DataProcessingException
62          *          if a data processing error occurred
63          */
64         public Data getData() throws DataProcessingException {
65             Data input = getPredecessor().getData(); // get the spectrum
66             getTimer().start();
67             if (input != null && ditherMax != 0) {
68                 if (input instanceof DoubleData || input instanceof FloatData) {
69                     input = process(input);
70                 }
71             }
72     
73             getTimer().stop();
74             return input;
75         }

78         /**
79          * Process data, adding dither
80          *
81          * @param input a MelSpectrum frame
82          * @return a mel Cepstrum frame
83          */
84         private DoubleData process(Data input) throws IllegalArgumentException {
85             DoubleData output;
86             Random r;
87             if (useRandSeed)
88                 r = new Random();
89             else
90                 r = new Random(12345);
91     
92             assert input instanceof DoubleData;
93             double[] inFeatures;
94     
95             DoubleData doubleData = (DoubleData) input;
96             inFeatures = doubleData.getValues();
97             double[] outFeatures = new double[inFeatures.length];
98             for (int i = 0; i < inFeatures.length; ++i) {
99                 outFeatures[i] = r.nextFloat() * 2 * ditherMax - ditherMax + inFeatures[i];
100                 outFeatures[i] = max(min(outFeatures[i], maxValue), minValue);
101             }
102     
103             output = new DoubleData(outFeatures, doubleData.getSampleRate(),
104                     doubleData.getCollectTime(),
105                     doubleData.getFirstSampleNumber());
106     
107             return output;
108         }
