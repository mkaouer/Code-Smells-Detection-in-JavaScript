23     /**
24      * A StreamDataSource converts data from an InputStream into Data objects. One would call {@link
25      * #setInputStream(InputStream,String) setInputStream}to set the input stream, and call {@link #getData}to obtain the
26      * Data object.
27      */
28     @SuppressWarnings({"UnnecessaryLocalVariable"})
29     public class StreamDataSource extends BaseDataProcessor {
30     
31         /** SphinxProperty for the sample rate. */
32         @S4Integer(defaultValue = 16000)
33         public static final String PROP_SAMPLE_RATE = "sampleRate";
34     
35         /** SphinxProperty for the number of bytes to read from the InputStream each time. */
36         @S4Integer(defaultValue = 3200)
37         public static final String PROP_BYTES_PER_READ = "bytesPerRead";
38     
39         /** SphinxProperty for the number of bits per value. */
40         @S4Integer(defaultValue = 16)
41         public static final String PROP_BITS_PER_SAMPLE = "bitsPerSample";
42     
43         /** The SphinxProperty specifying whether the input data is big-endian. */
44         @S4Boolean(defaultValue = true)
45         public static final String PROP_BIG_ENDIAN_DATA = "bigEndianData";
46     
47         /** The SphinxProperty specifying whether the input data is signed. */
48         @S4Boolean(defaultValue = true)
49         public static final String PROP_SIGNED_DATA = "signedData";
50     
51         private InputStream dataStream;
52         protected int sampleRate;
53         private int bytesPerRead;
54         private int bytesPerValue;
55         private long totalValuesRead;
56         private boolean bigEndian;
57         private boolean signedData;
58         private boolean streamEndReached = false;
59         private boolean utteranceEndSent = false;
60         private boolean utteranceStarted = false;
61         protected int bitsPerSample;
62     
63     
64         /*
65         * (non-Javadoc)
66         *
67         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
68         */
69         public void newProperties(PropertySheet ps) throws PropertyException {
70             super.newProperties(ps);
71             sampleRate = ps.getInt(PROP_SAMPLE_RATE);
72             bytesPerRead = ps.getInt(PROP_BYTES_PER_READ);
73             bitsPerSample = ps.getInt(PROP_BITS_PER_SAMPLE);
74     
75             if (bitsPerSample % 8 != 0) {
76                 throw new Error("StreamDataSource: bits per sample must be a " +
77                         "multiple of 8.");
78             }
79             
80             bytesPerValue = bitsPerSample / 8;
81             bigEndian = ps.getBoolean(PROP_BIG_ENDIAN_DATA);
82             signedData = ps.getBoolean(PROP_SIGNED_DATA);
83             if (bytesPerRead % 2 == 1) {
84                 bytesPerRead++;
85             }
86         }

99         /**
100          * Sets the InputStream from which this StreamDataSource reads.
101          *
102          * @param inputStream the InputStream from which audio data comes
103          * @param streamName  the name of the InputStream
104          */
105         public void setInputStream(InputStream inputStream, String streamName) {
106             dataStream = inputStream;
107             streamEndReached = false;
108             utteranceEndSent = false;
109             utteranceStarted = false;
110             totalValuesRead = 0;
111         }
112     
113     
114         /**
115          * Reads and returns the next Data from the InputStream of StreamDataSource, return null if no data is read and end
116          * of file is reached.
117          *
118          * @return the next Data or <code>null</code> if none is available
119          * @throws DataProcessingException if there is a data processing error
120          */
121         public Data getData() throws DataProcessingException {
122             getTimer().start();
123             Data output = null;
124             if (streamEndReached) {
125                 if (!utteranceEndSent) {
126                     // since 'firstSampleNumber' starts at 0, the last
127                     // sample number should be 'totalValuesRead - 1'
128                     output = new DataEndSignal(getDuration());
129                     utteranceEndSent = true;
130                 }
131             } else {
132                 if (!utteranceStarted) {
133                     utteranceStarted = true;
134                     output = new DataStartSignal(sampleRate);
135                 } else {
136                     if (dataStream != null) {
137                         output = readNextFrame();
138                         if (output == null) {
139                             if (!utteranceEndSent) {
140                                 output = new DataEndSignal(getDuration());
141                                 utteranceEndSent = true;
142                             }
143                         }
144                     }
145                 }
146             }
147             getTimer().stop();
148             return output;
149         }

152         /**
153          * Returns the next Data from the input stream, or null if there is none available
154          *
155          * @return a Data or null
156          * @throws java.io.IOException
157          */
158         private Data readNextFrame() throws DataProcessingException {
159             // read one frame's worth of bytes
160             int read;
161             int totalRead = 0;
162             final int bytesToRead = bytesPerRead;
163             byte[] samplesBuffer = new byte[bytesPerRead];
164             long collectTime = System.currentTimeMillis();
165             long firstSample = totalValuesRead;
166             try {
167                 do {
168                     read = dataStream.read(samplesBuffer, totalRead, bytesToRead
169                             - totalRead);
170                     if (read > 0) {
171                         totalRead += read;
172                     }
173                 } while (read != -1 && totalRead < bytesToRead);
174                 if (totalRead <= 0) {
175                     closeDataStream();
176                     return null;
177                 }
178                 // shrink incomplete frames
179                 totalValuesRead += (totalRead / bytesPerValue);
180                 if (totalRead < bytesToRead) {
181                     totalRead = (totalRead % 2 == 0)
182                             ? totalRead + 2
183                             : totalRead + 3;
184                     byte[] shrinkedBuffer = new byte[totalRead];
185                     System
186                             .arraycopy(samplesBuffer, 0, shrinkedBuffer, 0,
187                                     totalRead);
188                     samplesBuffer = shrinkedBuffer;
189                     closeDataStream();
190                 }
191             } catch (IOException ioe) {
192                 ioe.printStackTrace();
193                 throw new DataProcessingException("Error reading data");
194             }
195             // turn it into an Data object
196             double[] doubleData;
197             if (bigEndian) {
198                 doubleData = DataUtil.bytesToValues(samplesBuffer, 0, totalRead,
199                         bytesPerValue, signedData);
200             } else {
201                 doubleData = DataUtil.littleEndianBytesToValues(samplesBuffer, 0,
202                         totalRead, bytesPerValue, signedData);
203             }
204             return new DoubleData(doubleData, sampleRate, collectTime, firstSample);
205         }
206     
207     
208         private void closeDataStream() throws IOException {
209             streamEndReached = true;
210             if (dataStream != null) {
211                 dataStream.close();
212             }
213         }
214     
215     
216         /**
217          * Returns the duration of the current data stream in milliseconds.
218          *
219          * @return the duration of the current data stream in milliseconds
220          */
221         private long getDuration() {
222             return (long) (((double) totalValuesRead / (double) sampleRate) * 1000.0);
223         }
