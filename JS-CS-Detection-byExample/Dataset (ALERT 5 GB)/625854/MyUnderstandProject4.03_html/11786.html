13     package edu.cmu.sphinx.decoder;
14     
15     import edu.cmu.sphinx.result.Result;
16     import edu.cmu.sphinx.util.props.PropertyException;
17     import edu.cmu.sphinx.util.props.PropertySheet;
18     import edu.cmu.sphinx.util.props.S4Integer;
19     
20     /** The primary decoder class */
21     public class Decoder extends AbstractDecoder {
22     
23         /** The sphinx property name for the number of features to recognize at once. */
24         @S4Integer(defaultValue = 100000)
25         public final static String PROP_FEATURE_BLOCK_SIZE = "featureBlockSize";
26         private int featureBlockSize;
27     
28     
29         @Override
30         public void newProperties(PropertySheet ps) throws PropertyException {
31             super.newProperties(ps);
32     
33             featureBlockSize = ps.getInt(PROP_FEATURE_BLOCK_SIZE);
34         }
35     
36     
37         /**
38          * Decode frames until recognition is complete.
39          *
40          * @param referenceText the reference text (or null)
41          * @return a result
42          */
43         public Result decode(String referenceText) {
44             searchManager.startRecognition();
45             Result result;
46             do {
47                 result = searchManager.recognize(featureBlockSize);
48                 if (result != null) {
49                     result.setReferenceText(referenceText);
50                     fireResultListeners(result);
51                 }
52             } while (result != null && !result.isFinal());
53             searchManager.stopRecognition();
54             return result;
55         }
56     }
