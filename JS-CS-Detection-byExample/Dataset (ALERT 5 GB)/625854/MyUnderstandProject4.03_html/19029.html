15         public static <T extends DataProcessor> T getFrontEndProcessor(DataProcessor dp, Class<T> predecClass) {
16             while (!predecClass.isInstance(dp)) {
17                 if (dp instanceof FrontEnd)
18                     dp = ((FrontEnd) dp).getLastDataProcessor();
19                 else
20                     dp = dp.getPredecessor();
21     
22                 if (dp == null)
23                     return null;
24             }
25     
26     
27             return (T) dp;
28         }
