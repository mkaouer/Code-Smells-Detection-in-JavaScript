15     package edu.cmu.sphinx.result;
16     
17     import edu.cmu.sphinx.frontend.Data;
18     import edu.cmu.sphinx.linguist.dictionary.Pronunciation;
19     import edu.cmu.sphinx.linguist.dictionary.Word;
20     import edu.cmu.sphinx.util.LogMath;
21     
22     /**
23      * Represents a single word result with associated scoring and timing information.
24      *
25      * @author pgorniak
26      */
27     public class SimpleWordResult implements WordResult {
28     
29         Word word;
30         int startFrame;
31         int endFrame;
32         double score;
33         double confidence;
34         LogMath logMath;
35     
36     
37         /**
38          * Construct a word result from a string and a confidence score. Note that
39          *
40          * @param w          the word
41          * @param confidence the confidence for this word
42          */
43         public SimpleWordResult(String w, double confidence, LogMath logMath) {
44             Pronunciation[] pros = {Pronunciation.UNKNOWN};
45             this.word = new Word(w, pros, false);
46             this.confidence = confidence;
47             this.score = LogMath.getLogZero();
48             this.logMath = logMath;
49         }
50     
51     
52         /**
53          * Construct a word result with full information.
54          *
55          * @param w          the word object to store
56          * @param sf         word start time
57          * @param ef         word end time
58          * @param score      score of the word
59          * @param confidence confidence (posterior) of the word
60          */
61         public SimpleWordResult(Word w, int sf, int ef, double score,
62                                 double confidence, LogMath logMath) {
63             this.word = w;
64             this.startFrame = sf;
65             this.endFrame = ef;
66             this.score = score;
67             this.confidence = confidence;
68             this.logMath = logMath;
69         }
70     
71     
72         /**
73          * Construct a WordResult using a Node object and a confidence (posterior). This does not use the posterior stored
74          * in the Node object, just its word, start and end. TODO: score is currently set to zero
75          *
76          * @param node       the node to extract information from
77          * @param confidence the confidence (posterior) to assign
78          */
79         public SimpleWordResult(Node node, double confidence, LogMath logMath) {
80             this(node.getWord(), node.getBeginTime(), node.getEndTime(),
81                     LogMath.getLogZero(), confidence, logMath);
82         }
83     
84     
85         /** @see edu.cmu.sphinx.result.WordResult#getScore() */
86         public double getScore() {
87             return score;
88         }
89     
90     
91         /** @see edu.cmu.sphinx.result.WordResult#getConfidence() */
92         public double getConfidence() {
93             if (confidence > LogMath.getLogOne()) {
94                 return LogMath.getLogOne();
95             } else {
96                 return confidence;
97             }
98         }
99     
100     
101         /** @see edu.cmu.sphinx.result.WordResult#getLogMath() */
102         public LogMath getLogMath() {
103             return logMath;
104         }
105     
106     
107         /** @see edu.cmu.sphinx.result.WordResult#getPronunciation() */
108         public Pronunciation getPronunciation() {
109             return word.getMostLikelyPronunciation();
110         }
111     
112     
113         /** @see edu.cmu.sphinx.result.WordResult#getStartFrame() */
114         public int getStartFrame() {
115             return startFrame;
116         }
117     
118     
119         /** @see edu.cmu.sphinx.result.WordResult#getEndFrame() */
120         public int getEndFrame() {
121             return endFrame;
122         }
123     
124     
125         /**
126          * Return this WordResult as a string.
127          *
128          * @return the word stored here as a string
129          */
130         public String toString() {
131             return word.toString();
132         }
133     
134     
135         /** @see edu.cmu.sphinx.result.WordResult#getDataFrames() */
136         public Data[] getDataFrames() {
137             // TODO not yet implemented
138             return null;
139         }
140     
141     
142         /** @see edu.cmu.sphinx.result.WordResult#isFiller() */
143         public boolean isFiller() {
144             return word.isFiller() || word.toString().equals("<noop>");
145         }
146     }
