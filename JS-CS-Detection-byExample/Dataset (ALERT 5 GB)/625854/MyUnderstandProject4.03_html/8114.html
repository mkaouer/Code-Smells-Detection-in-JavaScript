15     /**
16      * This component concatenate the cepstrum from the sequence of frames according to the window size.
17      * It's not supposed to give high accuracy alone, but combined with LDA transform it can give the same
18      * or even better results than conventional delta and delta-delta coefficients. The idea is that
19      * delta-delta computation is also a matrix multiplication thus using automatically generated
20      * with LDA/MLLT matrix we can gain better results.
21      * The model for this feature extractor should be trained with SphinxTrain with 1s_c feature type and
22      * with cepwin option enabled. Don't forget to set the window size accordingly.
23      */
24     public class ConcatFeatureExtractor extends AbstractFeatureExtractor {
25     
26         /**
27          * Computes the next feature. Advances the pointers as well.
28          *
29          * @return the feature Data computed
30          */
31         protected Data computeNextFeature() {
32             DoubleData currentCepstrum = cepstraBuffer[currentPosition];
33             float[] feature = new float[(window * 2 + 1) * currentCepstrum.getValues().length];
34             int j = 0;
35             for (int k = -window; k <= window; k++) {
36             	int position = (currentPosition + k + cepstraBufferSize) % cepstraBufferSize;
37             	double[] buffer = cepstraBuffer[position].getValues();
38                 for (int l = 0; l < buffer.length; l++) {
39                     feature[j++] = (float) buffer[l];
40                 }
41             }
42             currentPosition = (currentPosition + 1) % cepstraBufferSize ;
43     
44             return (new FloatData(feature,
45                     currentCepstrum.getSampleRate(),
46                     currentCepstrum.getCollectTime(),
47                     currentCepstrum.getFirstSampleNumber()));
48         }
49     }
