9     package edu.cmu.sphinx.demo.jsapi.tags;
10     
11     import java.util.Vector;
12     
13     public class Pizza implements OrderItem {
14         public Vector toppings = new Vector();
15     
16         public Pizza() {
17         }
18     
19         public void addTopping(String topping) {
20             toppings.add(topping);
21         }
22     
23         public String toString() {
24             int numToppings = toppings.size();
25             if (numToppings == 0) {
26                 return "plain pizza.";
27             } else {
28                 StringBuffer sb = new StringBuffer("pizza with ");
29                 for (int i = 0; i < numToppings; i++) {
30                     sb.append(toppings.elementAt(i));
31                     if (numToppings > 1) {
32                         if (i == (numToppings - 2)) {
33                             sb.append(" and ");
34                         } else if (i < (numToppings - 1)) {
35                             sb.append(", ");
36                         }
37                     }
38                 }
39                 sb.append(".");
40                 return sb.toString();
41             }
42         }
43     }
