12     package edu.cmu.sphinx.frontend.feature;
13     
14     import edu.cmu.sphinx.frontend.*;
15     import edu.cmu.sphinx.frontend.endpoint.*;
16     import edu.cmu.sphinx.util.props.*;
17     
18     import java.util.*;

20     /**
21      * Abstract base class for windowed feature extractors like DeltasFeatureExtractor, ConcatFeatureExtractor
22      * or S3FeatureExtractor. The main purpose of this it to collect window size cepstra frames in a buffer
23      * and let the extractor compute the feature frame with them.
24      */
25     public abstract class AbstractFeatureExtractor extends BaseDataProcessor {
26     
27         /** The name of the SphinxProperty for the window of the DeltasFeatureExtractor. */
28         @S4Integer(defaultValue = 3)
29         public static final String PROP_FEATURE_WINDOW = "windowSize";
30     
31         private int bufferPosition;
32         private Signal pendingSignal;
33         private List<Data> outputQueue;
34     
35         protected int cepstraBufferEdge;
36         protected int window;
37         protected int currentPosition;
38         protected int cepstraBufferSize;
39         protected DoubleData[] cepstraBuffer;
40     
41         /*
42         * (non-Javadoc)
43         *
44         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
45         */
46         public void newProperties(PropertySheet ps) throws PropertyException {
47             super.newProperties(ps);
48             window = ps.getInt(PROP_FEATURE_WINDOW);
49         }
50     
51     
52         /*
53         * (non-Javadoc)
54         *
55         * @see edu.cmu.sphinx.frontend.DataProcessor#initialize(edu.cmu.sphinx.frontend.CommonConfig)
56         */
57         public void initialize() {
58             super.initialize();
59             cepstraBufferSize = 256;
60             cepstraBuffer = new DoubleData[cepstraBufferSize];
61             cepstraBufferEdge = cepstraBufferSize - (window * 2 + 2);
62             outputQueue = new Vector<Data>();
63             reset();
64         }
65     
66     
67         /** Resets the DeltasFeatureExtractor to be ready to read the next segment of data. */
68         private void reset() {
69             bufferPosition = 0;
70             currentPosition = 0;
71         }
72     
73     
74         /**
75          * Returns the next Data object produced by this DeltasFeatureExtractor.
76          *
77          * @return the next available Data object, returns null if no Data is available
78          * @throws DataProcessingException if there is a data processing error
79          */
80         public Data getData() throws DataProcessingException {
81             if (outputQueue.size() == 0) {
82                 Data input = getNextData();
83                 if (input != null) {
84                     if (input instanceof DoubleData) {
85                         addCepstrum((DoubleData) input);
86                         computeFeatures(1);
87                     } else if (input instanceof DataStartSignal) {
88                         pendingSignal = null;
89                         outputQueue.add(input);
90                         Data start = getNextData();
91                         int n = processFirstCepstrum(start);
92                         computeFeatures(n);
93                         if (pendingSignal != null) {
94                             outputQueue.add(pendingSignal);
95                         }
96                     } else if (input instanceof DataEndSignal || input instanceof SpeechEndSignal) {
97                         // when the DataEndSignal is right at the boundary
98                         int n = replicateLastCepstrum();
99                         computeFeatures(n);
100                         outputQueue.add(input);
101                     }
102                 }
103             }
104             if (outputQueue.size() > 0) {
105                 Data feature = outputQueue.remove(0);
106                 return feature;
107             } else {
108                 return null;
109             }
110         }
111     
112     
113         private Data getNextData() throws DataProcessingException {
114             Data d = getPredecessor().getData();
115             while (d != null && !(d instanceof DoubleData || d instanceof DataEndSignal || d instanceof DataStartSignal || d instanceof SpeechEndSignal)) {
116                 outputQueue.add(d);
117                 d = getPredecessor().getData();
118             }
119     
120             return d;
121         }
122     
123     
124         /**
125          * Replicate the given cepstrum Data object into the first window+1 number of frames in the cepstraBuffer. This is
126          * the first cepstrum in the segment.
127          *
128          * @param cepstrum the Data to replicate
129          * @return the number of Features that can be computed
130          */
131         private int processFirstCepstrum(Data cepstrum)
132                 throws DataProcessingException {
133             if (cepstrum instanceof DataEndSignal) {
134                 outputQueue.add(cepstrum);
135                 return 0;
136             } else if (cepstrum instanceof DataStartSignal) {
137                 throw new Error("Too many UTTERANCE_START");
138             } else {
139                 // At the start of an utterance, we replicate the first frame
140                 // into window+1 frames, and then read the next "window" number
141                 // of frames. This will allow us to compute the delta-
142                 // double-delta of the first frame.
143                 Arrays.fill(cepstraBuffer, 0, window + 1, cepstrum);
144                 bufferPosition = window + 1;
145                 bufferPosition %= cepstraBufferSize;
146                 currentPosition = window;
147                 currentPosition %= cepstraBufferSize;
148                 int numberFeatures = 1;
149                 pendingSignal = null;
150                 for (int i = 0; i < window; i++) {
151                     Data next = getNextData();
152                     if (next != null) {
153                         if (next instanceof DoubleData) {
154                             // just a cepstra
155                             addCepstrum((DoubleData) next);
156                         } else if (next instanceof DataEndSignal || next instanceof SpeechEndSignal) {
157                             // end of segment cepstrum
158                             pendingSignal = (Signal) next;
159                             replicateLastCepstrum();
160                             numberFeatures += i;
161                             break;
162                         } else if (next instanceof DataStartSignal) {
163                             throw new Error("Too many UTTERANCE_START");
164                         }
165                     }
166                 }
167                 return numberFeatures;
168             }
169         }
170     
171     
172         /**
173          * Adds the given DoubleData object to the cepstraBuffer.
174          *
175          * @param cepstrum the DoubleData object to add
176          */
177         private void addCepstrum(DoubleData cepstrum) {
178             cepstraBuffer[bufferPosition++] = cepstrum;
179             bufferPosition %= cepstraBufferSize;
180         }
181     
182     
183         /**
184          * Replicate the last frame into the last window number of frames in the cepstraBuffer.
185          *
186          * @return the number of replicated Cepstrum
187          */
188         private int replicateLastCepstrum() {
189             DoubleData last;
190             if (bufferPosition > 0) {
191                 last = cepstraBuffer[bufferPosition - 1];
192             } else if (bufferPosition == 0) {
193                 last = cepstraBuffer[cepstraBuffer.length - 1];
194             } else {
195                 throw new Error("BufferPosition < 0");
196             }
197             for (int i = 0; i < window; i++) {
198                 addCepstrum(last);
199             }
200             return window;
201         }
202     
203     
204         /**
205          * Converts the Cepstrum data in the cepstraBuffer into a FeatureFrame.
206          *
207          * @param totalFeatures the number of Features that will be produced
208          */
209         private void computeFeatures(int totalFeatures) {
210             getTimer().start();
211             if (totalFeatures == 1) {
212                 computeFeature();
213             } else {
214                 // create the Features
215                 for (int i = 0; i < totalFeatures; i++) {
216                     computeFeature();
217                 }
218             }
219             getTimer().stop();
220         }
221     
222     
223         /** Computes the next Feature. */
224         private void computeFeature() {
225             Data feature = computeNextFeature();
226             outputQueue.add(feature);
227         }
228     
229     
230         /**
231          * Computes the next feature. Advances the pointers as well.
232          *
233          * @return the feature Data computed
234          */
235         protected abstract Data computeNextFeature();
236     }
