33     /**
34      * A extended form of the {@link edu.cmu.sphinx.linguist.flat.FlatLinguist FlatLinguist} that creates a search graph
35      * with states for multiple feature streams.
36      * <p/>
37      * The only difference in the topology of the search graph created by this linguist and the FlatLinguist is that at a
38      * certain level (either the unit or the state level), the graph splits into two parallel branches of states, one for
39      * each feature stream. Moreover, at the end of that same level, the multiple streams are again merged.
40      */
41     public class ParallelSimpleLinguist extends FlatLinguist {

48         @S4String(defaultValue = "unit")

52         @S4ComponentList(type = FeatureStream.class)
53         public static final String PROP_FEATURE_STREAMS = "featureStreams";

164             /**
165              * Expands the unit into a set of HMMStates. If the unit is a silence unit add an optional loopback to the
166              * tail.
167              *
168              * @param unit the unit to expand
169              * @return the head of the hmm tree
170              */
171             protected SentenceHMMState expandUnit(UnitState unit) {
172                 SentenceHMMState tail = null;
173     
174                 if (tieLevel.equals("unit")) {
175                     tail = getTiedHMMs(unit);
176                 } else if (tieLevel.equals("state")) {
177                     tail = getTiedHMMStates(unit);
178                 }
179     
180                 // if the unit is a silence unit add a loop back from the
181                 // tail silence unit
182                 if (unit.getUnit().isSilence()) {
183                     // add the loopback, but don't expand it // anymore
184                     attachState(tail, unit,
185                             LogMath.getLogOne(),
186                             LogMath.getLogOne(),
187                             getLogSilenceInsertionProbability());
188                 }
189                 return tail;
190             }

193             /**
194              * Expands the given UnitState into the set of associated parallel HMMs.
195              *
196              * @param unitState the UnitState to expand
197              * @return the last SentenceHMMState from the expansion
198              */
199             private SentenceHMMState getTiedHMMs(UnitState unitState) {
200     
201                 SentenceHMMState combineState = new CombineState
202                         (unitState.getParent(), unitState.getWhich());
203     
204                 // create an HMM branch for each acoustic model            
205                 for (Object featureStream : featureStreams) {
206     
207                     FeatureStream stream = (FeatureStream) featureStream;
208                     AcousticModel model = stream.getAcousticModel();
209     
210                     HMM hmm = model.lookupNearestHMM
211                             (unitState.getUnit(), unitState.getPosition(), false);
212     
213                     ParallelHMMStateState firstHMMState =
214                             new ParallelHMMStateState(unitState, stream,
215                                     hmm.getInitialState(),
216                                     tokenStackCapacity);
217     
218                     // Color.GREEN indicates an in-feature-stream state
219                     firstHMMState.setColor(Color.GREEN);
220     
221                     // attach first HMMStateState to the splitState
222                     attachState(unitState, firstHMMState,
223                             logOne,
224                             logOne,
225                             logOne);
226     
227                     // expand the HMM and connect the lastState w/ the combineState
228                     Map hmmStates = new HashMap();
229                     hmmStates.put(firstHMMState.getHMMState(), firstHMMState);
230     
231                     SentenceHMMState lastState =
232                             expandParallelHMMTree(firstHMMState, stream, hmmStates);
233     
234                     attachState(lastState, combineState,
235                             logOne,
236                             logOne,
237                             logOne);
238                 }
239     
240                 return combineState;
241             }
242     
243     
244             /**
245              * Expands the given HMM tree into the full set of HMMStateStates.
246              *
247              * @param hmmStateState  the first state of the HMM tree
248              * @param stream         the FeatureStream of the relevant acoustic model
249              * @param expandedStates the map of HMMStateStates
250              * @return the last state of the expanded tree
251              */
252             private SentenceHMMState expandParallelHMMTree
253                     (ParallelHMMStateState hmmStateState, FeatureStream stream,
254                      Map expandedStates) {
255     
256                 SentenceHMMState lastState = hmmStateState;
257     
258                 HMMState hmmState = hmmStateState.getHMMState();
259                 HMMStateArc[] arcs = hmmState.getSuccessors();
260     
261                 for (HMMStateArc arc : arcs) {
262     
263                     HMMState nextHmmState = arc.getHMMState();
264     
265                     if (nextHmmState == hmmState) {
266     
267                         // this is a self-transition
268                         attachState(hmmStateState, hmmStateState, arc.getLogProbability(), logOne, logOne);
269     
270                         lastState = hmmStateState;
271                     } else {
272     
273                         // transition to the next state
274                         ParallelHMMStateState nextState;
275     
276                         if (expandedStates.containsKey(nextHmmState)) {
277                             nextState = (ParallelHMMStateState)
278                                     expandedStates.get(nextHmmState);
279                         } else {
280                             nextState = new ParallelHMMStateState
281                                     (hmmStateState.getParent(), stream,
282                                             nextHmmState, tokenStackCapacity);
283                             expandedStates.put(nextHmmState, nextState);
284                         }
285     
286                         // Color.GREEN indicates an in-feature-stream state
287                         nextState.setColor(Color.GREEN);
288     
289                         attachState(hmmStateState, nextState,
290                                 arc.getLogProbability(),
291                                 logOne,
292                                 logOne);
293     
294                         lastState = expandParallelHMMTree
295                                 (nextState, stream, expandedStates);
296                     }
297                 }
298     
299                 return lastState;
300             }

303             /**
304              * Expands the given UnitState into the set of associated HMMStateStates that tie at the state level.
305              *
306              * @param unitState the UnitState to expand
307              * @return the last SentenceHMMState from the expansion
308              */
309             private SentenceHMMState getTiedHMMStates(UnitState unitState) {
310                 HMM[] hmms = new HMM[featureStreams.size()];
311     
312                 SentenceHMMState lastState;
313     
314                 int s = 0;
315                 // create an HMM branch for each feature stream
316                 for (Iterator<FeatureStream> i = featureStreams.iterator(); i.hasNext(); s++) {
317                     FeatureStream stream = (FeatureStream) i.next();
318                     hmms[s] = stream.getAcousticModel().lookupNearestHMM
319                             (unitState.getUnit(), unitState.getPosition(), false);
320                 }
321     
322                 lastState = getHMMTiedStates(hmms, unitState);
323     
324                 return lastState;
325             }

328             /**
329              * Converts the given HMMs into a network of SentenceHMMStates tied at the state level.
330              *
331              * @param hmms      the HMMs to convert
332              * @param unitState the UnitState that corresponds to these HMMs
333              * @return the last SentenceHMMState from the expansion
334              */
335             private SentenceHMMState getHMMTiedStates(HMM[] hmms,
336                                                       UnitState unitState) {
337     
338                 SentenceHMMState lastState = new CombineState(unitState, 0);
339     
340                 HMMStateArc[] arcs = new HMMStateArc[featureStreams.size()];
341     
342                 //
343                 // In this for loop, we connect the unitState to the 
344                 // ParallelHMMState created from the first HMMState of each of 
345                 // the HMMs. We then connect each of the ParallelHMMStates
346                 // to a combining state. Lastly, if the HMMState has a
347                 // self-transition, we create a transition from the combining
348                 // state to the ParallelHMMState, using the self-transition
349                 // probability.
350                 //
351                 for (int i = 0; i < hmms.length; i++) {
352                     HMMState hmmState = hmms[i].getInitialState();
353                     FeatureStream stream = (FeatureStream) featureStreams.get(i);
354     
355                     ParallelHMMStateState firstHMMState = new ParallelHMMStateState
356                             (unitState, stream, hmmState, tokenStackCapacity);
357     
358                     // Color.GREEN indicates an in-feature-stream state
359                     firstHMMState.setColor(Color.GREEN);
360     
361                     // connect previous last state to this HMMState
362                     attachState(unitState, firstHMMState,
363                             logOne,
364                             logOne,
365                             logOne);
366     
367                     // connect this HMMState to the next combining state
368                     attachState(firstHMMState, lastState,
369                             logOne,
370                             logOne,
371                             logOne);
372     
373                     HMMStateArc selfTransition = getSelfTransition(hmmState);
374     
375                     if (selfTransition != null) {
376                         // connect the next combining state to this HMMState
377                         attachState(lastState, firstHMMState,
378                                 selfTransition.getLogProbability(),
379                                 logOne,
380                                 logOne);
381                     }
382     
383                     arcs[i] = getTransitionToNextState(hmmState);
384                 }
385     
386                 // 
387                 // We then start with the second HMMState of each HMM, and do
388                 // the same thing as the previous for loop: connect each
389                 // ParallelHMMState (created from the HMMState) to a combining
390                 // state, and if that HMMState has a self transition, connect
391                 // that combining state to the ParallelHMMState using the
392                 // self transition probability
393                 // 
394                 for (int i = 1; i <= hmms[0].getOrder(); i++) {
395     
396                     SentenceHMMState combineState = new CombineState(unitState, i);
397     
398                     for (int a = 0; a < arcs.length; a++) {
399                         HMMStateArc arc = arcs[a];
400                         HMMState hmmState = arc.getHMMState();
401                         FeatureStream stream = (FeatureStream) featureStreams.get(a);
402     
403                         ParallelHMMStateState hmmStateState =
404                                 new ParallelHMMStateState
405                                         (unitState, stream, hmmState, tokenStackCapacity);
406     
407                         // Color.GREEN indicates an in-feature-stream state
408                         hmmStateState.setColor(Color.GREEN);
409     
410                         // connect lastState and this HMMStateState
411                         attachState(lastState, hmmStateState,
412                                 arc.getLogProbability(),
413                                 logOne,
414                                 logOne);
415     
416                         // connect this HMMStateState and the combineState
417                         attachState(hmmStateState, combineState,
418                                 logOne,
419                                 logOne,
420                                 logOne);
421     
422                         // connect the self-transition
423                         HMMStateArc selfTransition = getSelfTransition(hmmState);
424     
425                         if (selfTransition != null) {
426                             // connect the next combining state to this HMMState
427                             attachState(combineState, hmmStateState,
428                                     selfTransition.getLogProbability(),
429                                     logOne,
430                                     logOne);
431                         }
432                         arcs[a] = getTransitionToNextState(hmmState);
433                     }
434     
435                     lastState = combineState;
436                 }
437     
438                 return lastState;
439             }

442             /** Returns the self-transitioning HMMStateArc of the given HMMState. */
443             private HMMStateArc getSelfTransition(HMMState hmmState) {
444                 HMMStateArc[] arcs = hmmState.getSuccessors();
445                 for (HMMStateArc arc : arcs) {
446                     HMMState nextHmmState = arc.getHMMState();
447                     if (nextHmmState == hmmState) {
448                         return arc;
449                     }
450                 }
451                 return null;
452             }
453     
454     
455             /** Returns the HMMStateArc that transitioin to the next HMMState. */
456             private HMMStateArc getTransitionToNextState(HMMState hmmState) {
457                 HMMStateArc[] arcs = hmmState.getSuccessors();
458                 for (HMMStateArc arc : arcs) {
459                     HMMState nextHmmState = arc.getHMMState();
460                     if (nextHmmState != hmmState) {
461                         return arc;
462                     }
463                 }
464                 return null;
465             }
