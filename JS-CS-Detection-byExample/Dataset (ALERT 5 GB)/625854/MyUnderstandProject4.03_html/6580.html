95         private void paintVUMeter(Graphics g) {
96             int level = (int) ((vu.getRmsDB() / vu.getMaxDB()) * numberOfLights);
97             int peak = (int) ((vu.getPeakDB() / vu.getMaxDB()) * numberOfLights);
98     
99             assert level >= 0;
100             assert level < numberOfLights;
101     
102     //        if (level == lastLevel) return;
103             lastLevel = level;
104     //
105             Dimension sz = getSize();
106             int w = sz.width;
107             int h = (sz.height / numberOfLights);
108     //
109     //        assert h > 2;
110     //        assert w > 2;
111     
112             g.setColor(Color.BLACK);
113             g.fillRect(0, 0, sz.width - 1, sz.height - 1);
114     //
115             for (int i = 0; i < level; i++) {
116                 setLevelColor(i, g);
117                 g.fillRect(1, sz.height - (i * h) + 1, w - 2, h - 2);
118             }
119     //
120             setLevelColor(peak, g);
121             g.fillRect(1, sz.height - (peak * h) + 1, w - 2, h - 2);
122     
123         }

126         private void setLevelColor(int i, Graphics g) {
127             if (i < greenLevel)
128                 g.setColor(Color.BLUE);
129             else if (i < yellowLevel)
130                 g.setColor(Color.GREEN);
131             else if (i < redLevel)
132                 g.setColor(Color.YELLOW);
133             else
134                 g.setColor(Color.RED);
135         }
