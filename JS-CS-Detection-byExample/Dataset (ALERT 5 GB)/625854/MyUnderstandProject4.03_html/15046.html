9     package edu.cmu.sphinx.research.distributed.server;
10     
11     import java.io.IOException;
12     import java.net.ServerSocket;
13     import java.net.Socket;
14     
15     
16     /**
17      * A bare-bones BaseServer containing a ServerSocket waiting for connection requests. Subclasses should implement the
18      * <code>spawnProtocolHandler</code> method.
19      */
20     public abstract class BaseServer implements Runnable {
21     
22         /** The port number to listen on. It is the value specified by the System property "port". */
23         private int port = Integer.parseInt
24                 (System.getProperty("port", String.valueOf(52703)));
25     
26     
27         /**
28          * Implements the run() method of Runnable interface. It starts a ServerSocket, listens for connections, and spawns
29          * a handler for each connection.
30          */
31         public void run() {
32             ServerSocket ss;
33     
34             try {
35                 ss = new ServerSocket(getPort());
36                 System.out.println("Waiting on " + ss);
37             } catch (IOException ioe) {
38                 System.out.println("Can't open socket on port " + getPort());
39                 ioe.printStackTrace();
40                 return;
41             }
42     
43             while (true) {
44                 try {
45                     Socket socket = ss.accept();
46                     System.out.println("... new socket connection");
47                     spawnProtocolHandler(socket);
48                 } catch (IOException ioe) {
49                     System.err.println("Could not accept socket " + ioe);
50                     ioe.printStackTrace();
51                     break;
52                 }
53             }
54     
55             try {
56                 ss.close();
57             } catch (IOException ioe) {
58                 System.err.println("Could not close server socket " + ioe);
59                 ioe.printStackTrace();
60             }
61         }
62     
63     
64         /**
65          * Returns the port number.
66          *
67          * @return the port number
68          */
69         public int getPort() {
70             return port;
71         }
72     
73     
74         /**
75          * This method is called after a connection request is made to this BaseServer. The <code>Socket</code> created as a
76          * result of the connection request is passed to this method.
77          *
78          * @param socket the socket
79          */
80         protected abstract void spawnProtocolHandler(Socket socket);
81     }
