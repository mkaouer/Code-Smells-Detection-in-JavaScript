14     import edu.cmu.sphinx.frontend.BaseDataProcessor;
15     import edu.cmu.sphinx.frontend.Data;
16     import edu.cmu.sphinx.frontend.DataProcessingException;
17     import edu.cmu.sphinx.frontend.DoubleData;
18     import edu.cmu.sphinx.util.props.*;

20     /**
21      * Applies a logarithm and then a Discrete Cosine Transform (DCT) to the input data. The input data is normally the mel
22      * spectrum. It has been proven that, for a sequence of real numbers, the discrete cosine transform is equivalent to the
23      * discrete Fourier transform. Therefore, this class corresponds to the last stage of converting a signal to cepstra,
24      * defined as the inverse Fourier transform of the logarithm of the Fourier transform of a signal. The property {@link
25      * #PROP_CEPSTRUM_LENGTH}refers to the dimensionality of the coefficients that are actually returned, defaulting to
26      * 13. When the input is mel-spectrum, the vector returned is the MFCC (Mel-Frequency
27      * Cepstral Coefficient) vector, where the 0-th element is the energy value.
28      */
29     public class DiscreteCosineTransform extends BaseDataProcessor {
30     
31         /** The name of the Sphinx Property for the number of filters in the filterbank. */
32         @S4Integer(defaultValue = 40)
33         public static final String PROP_NUMBER_FILTERS = "numberFilters";
34     
35         /** The name of the sphinx property for the size of the ceptrum */
36         @S4Integer(defaultValue = 13)
37         public static final String PROP_CEPSTRUM_LENGTH = "cepstrumLength";
38     
39         protected int cepstrumSize; // size of a Cepstrum
40         protected int numberMelFilters; // number of mel-filters
41         protected double[][] melcosine;
42     
43     
44         @Override
45         public void newProperties(PropertySheet ps) throws PropertyException {
46             super.newProperties(ps);
47     
48             numberMelFilters = ps.getInt(PROP_NUMBER_FILTERS);
49             cepstrumSize = ps.getInt(PROP_CEPSTRUM_LENGTH);
50         }
51     
52     
53         @Override
54         public void initialize() {
55             super.initialize();
56         }
57     
58     
59         /**
60          * Returns the next DoubleData object, which is the mel cepstrum of the input frame. Signals are returned
61          * unmodified.
62          *
63          * @return the next available DoubleData melcepstrum, or Signal object, or null if no Data is available
64          * @throws DataProcessingException if a data processing error occurred
65          */
66         public Data getData() throws DataProcessingException {
67             Data input = getPredecessor().getData(); // get the spectrum
68             getTimer().start();
69             if (input != null && input instanceof DoubleData) {
70                 input = process((DoubleData) input);
71             }
72             getTimer().stop();
73             return input;
74         }
75     
76     
77         /**
78          * Process data, creating the mel cepstrum from an input spectrum frame.
79          *
80          * @param input a MelSpectrum frame
81          * @return a mel Cepstrum frame
82          */
83         private DoubleData process(DoubleData input)
84                 throws IllegalArgumentException {
85             double[] melspectrum = input.getValues();
86     
87             if (melcosine == null) {
88                 numberMelFilters = melspectrum.length;
89                 computeMelCosine();
90     
91             } else if (melspectrum.length != numberMelFilters) {
92                 throw new IllegalArgumentException
93                         ("MelSpectrum size is incorrect: melspectrum.length == " +
94                                 melspectrum.length + ", numberMelFilters == " +
95                                 numberMelFilters);
96             }
97             // first compute the log of the spectrum
98             for (int i = 0; i < melspectrum.length; ++i) {
99                 if (melspectrum[i] > 0) {
100                     melspectrum[i] = Math.log(melspectrum[i]);
101                 } else {
102                     // in case melspectrum[i] isn't greater than 0
103                     // instead of trying to compute a log we just
104                     // assign a very small number
105                     melspectrum[i] = -1.0e+5;
106                 }
107             }
108     
109             double[] cepstrum;
110     
111             // create the cepstrum by apply the melcosine filter
112             cepstrum = applyMelCosine(melspectrum);
113     
114             return new DoubleData(cepstrum, input.getSampleRate(),
115                     input.getCollectTime(),
116                     input.getFirstSampleNumber());
117         }
118     
119     
120         /** Compute the MelCosine filter bank. */
121         private void computeMelCosine() {
122             melcosine = new double[cepstrumSize][numberMelFilters];
123             double period = (double) 2 * numberMelFilters;
124             for (int i = 0; i < cepstrumSize; i++) {
125                 double frequency = 2 * Math.PI * i / period;
126                 for (int j = 0; j < numberMelFilters; j++) {
127                     melcosine[i][j] = Math.cos(frequency * (j + 0.5));
128                 }
129             }
130         }
131     
132     
133         /**
134          * Apply the MelCosine filter to the given melspectrum.
135          *
136          * @param melspectrum the MelSpectrum data
137          * @return MelCepstrum data produced by apply the MelCosine filter to the MelSpectrum data
138          */
139         protected double[] applyMelCosine(double[] melspectrum) {
140             // create the cepstrum
141             double[] cepstrum = new double[cepstrumSize];
142             double period = (double) numberMelFilters;
143             double beta = 0.5;
144             // apply the melcosine filter
145             for (int i = 0; i < cepstrum.length; i++) {
146                 if (numberMelFilters > 0) {
147                     double[] melcosine_i = melcosine[i];
148                     int j = 0;
149                     cepstrum[i] += (beta * melspectrum[j] * melcosine_i[j]);
150                     for (j = 1; j < numberMelFilters; j++) {
151                         cepstrum[i] += (melspectrum[j] * melcosine_i[j]);
152                     }
153                     cepstrum[i] /= period;
154                 }
155             }
156             
157             return cepstrum;
158         }
