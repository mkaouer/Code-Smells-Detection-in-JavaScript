97         public void setActivationMode(int mode) throws IllegalArgumentException {
98             if ((mode != GLOBAL)

440         /**
441          * Dispatch a SpeechEvent.  This is a method from SpeechEventDispatcher. The dispatcher should notify all listeners
442          * of the speech event from this method.
443          */
444         public void dispatchSpeechEvent(SpeechEvent event) {
445             switch (event.getId()) {
446                 case GrammarEvent.GRAMMAR_ACTIVATED:
447                     fireGrammarActivated((GrammarEvent) event);
448                     break;
449                 case GrammarEvent.GRAMMAR_CHANGES_COMMITTED:
450                     fireGrammarChangesCommitted((GrammarEvent) event);
451                     break;
452                 case GrammarEvent.GRAMMAR_DEACTIVATED:
453                     fireGrammarDeactivated((GrammarEvent) event);
454                     break;
455     
456                 case ResultEvent.AUDIO_RELEASED:
457                     fireAudioReleased((ResultEvent) event);
458                     break;
459                 case ResultEvent.GRAMMAR_FINALIZED:
460                     fireGrammarFinalized((ResultEvent) event);
461                     break;
462                 case ResultEvent.RESULT_ACCEPTED:
463                     fireResultAccepted((ResultEvent) event);
464                     break;
465                 case ResultEvent.RESULT_CREATED:
466                     fireResultCreated((ResultEvent) event);
467                     break;
468                 case ResultEvent.RESULT_REJECTED:
469                     fireResultRejected((ResultEvent) event);
470                     break;
471                 case ResultEvent.RESULT_UPDATED:
472                     fireResultUpdated((ResultEvent) event);
473                     break;
474                 case ResultEvent.TRAINING_INFO_RELEASED:
475                     fireTrainingInfoReleased((ResultEvent) event);
476                     break;
477             }
478         }
