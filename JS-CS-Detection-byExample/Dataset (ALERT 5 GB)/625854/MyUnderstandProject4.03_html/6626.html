13     package edu.cmu.sphinx.linguist.language.ngram.large;
14     
15     
16     /**
17      * Represents a word ID (second word of a bigram), a bigram probability ID, a backoff probability ID, and the location
18      * of the first trigram entry.
19      */
20     class BigramProbability {
21     
22         private int which;
23         private int wordID;
24         private int probabilityID;
25         private int backoffID;
26         private int firstTrigramEntry;
27     
28     
29         /**
30          * Constructs a BigramProbability
31          *
32          * @param which             which follower of the first word is this bigram
33          * @param wordID            the ID of the second word in a bigram
34          * @param probabilityID     the index into the probability array
35          * @param backoffID         the index into the backoff probability array
36          * @param firstTrigramEntry the first trigram entry
37          */
38         public BigramProbability(int which, int wordID,
39                                  int probabilityID, int backoffID,
40                                  int firstTrigramEntry) {
41             this.which = which;
42             this.wordID = wordID;
43             this.probabilityID = probabilityID;
44             this.backoffID = backoffID;
45             this.firstTrigramEntry = firstTrigramEntry;
46         }
47     
48     
49         /**
50          * Returns which follower of the first word is this bigram
51          *
52          * @return which follower of the first word is this bigram
53          */
54         public int getWhichFollower() {
55             return which;
56         }
57     
58     
59         /**
60          * Returns the second word ID of this bigram
61          *
62          * @return the second word ID
63          */
64         public int getWordID() {
65             return wordID;
66         }
67     
68     
69         /**
70          * Returns the bigram probability ID.
71          *
72          * @return the bigram probability ID
73          */
74         public int getProbabilityID() {
75             return probabilityID;
76         }
77     
78     
79         /**
80          * Returns the backoff weight ID.
81          *
82          * @return the backoff weight ID
83          */
84         public int getBackoffID() {
85             return backoffID;
86         }
87     
88     
89         /**
90          * Returns the index of the first trigram entry.
91          *
92          * @return the index of the first trigram entry
93          */
94         public int getFirstTrigramEntry() {
95             return firstTrigramEntry;
96         }
97     }
