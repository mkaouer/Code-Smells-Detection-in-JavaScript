10     /**
11      * Computes the linear predictive model using the Levinson-Durbin algorithm. Linear prediction assumes that a signal can
12      * be model as a linear combination of previous samples, that is, the current sample x[i] can be modeled as:
13      * <p/>
14      * <pre> x[i] = a[0] + a[1] * x[i - 1] + a[2] * x[i - 2] + ... </pre>
15      * <p/>
16      * The summation on the right hand side of the equation involves a finite number of terms. The number of previous
17      * samples used is the order of the linear prediction.
18      * <p/>
19      * This class also provides a method to compute LPC cepstra, that is, the cepstra computed from LPC coefficients, as
20      * well as a method to compute the bilinear transformation of the LPC
21      */
22     public class LinearPredictor {
23     
24         private int order;
25         private int cepstrumOrder;
26         private double[] reflectionCoeffs;
27         private double[] ARParameters;
28         private double alpha;
29         private double[] cepstra;
30         private double[] bilinearCepstra;

39             this.order = order;

47         }

50         /**
51          * Routine to compute Linear Prediction Coefficients for a frame of speech. Returns the energy of the frame (alpha
52          * in the Levinson recursion) Assumes the following sign convention:<br> prediction(x[t]) = Sum_i {Ar[i] * x[t-i]}
53          *
54          * @param autocor
55          */
56         public double[] getARFilter(double[] autocor) {
57             /* No signal */
58             if (autocor[0] == 0) {
59                 return null;
60             }
61             reflectionCoeffs = new double[order + 1];
62             ARParameters = new double[order + 1];
63             double[] backwardPredictor = new double[order + 1];
64     
65             alpha = autocor[0];
66             reflectionCoeffs[1] = -autocor[1] / autocor[0];
67             ARParameters[0] = 1.0;
68             ARParameters[1] = reflectionCoeffs[1];
69             alpha *= (1 - reflectionCoeffs[1] * reflectionCoeffs[1]);
70     
71             for (int i = 2; i <= order; i++) {
72                 for (int j = 1; j < i; j++) {
73                     backwardPredictor[j] = ARParameters[i - j];
74                 }
75                 reflectionCoeffs[i] = 0;
76                 for (int j = 0; j < i; j++) {
77                     reflectionCoeffs[i] -= ARParameters[j] * autocor[i - j];
78                 }
79                 reflectionCoeffs[i] /= alpha;
80     
81                 for (int j = 1; j < i; j++) {
82                     ARParameters[j] += reflectionCoeffs[i] * backwardPredictor[j];
83                 }
84                 ARParameters[i] = reflectionCoeffs[i];
85                 alpha *= (1 - reflectionCoeffs[i] * reflectionCoeffs[i]);
86                 if (alpha <= 0.0) {
87                     return null;
88                 }
89             }
90             return ARParameters;
91         }

94         /**
95          * Computes AR parameters from a given set of reflection coefficients.
96          *
97          * @param RC       double array of reflection coefficients. The RC array must begin at 1 (RC[0] is a dummy value)
98          * @param lpcorder AR order desired
99          * @return AR parameters
100          */
101         public double[] reflectionCoeffsToARParameters(double[] RC, int lpcorder) {
102             double[][] tmp = new double[lpcorder + 1][lpcorder + 1];
103     
104             order = lpcorder;
105             int RCorder = RC.length;
106             reflectionCoeffs = new double[RCorder];
107     
108             System.arraycopy(RC, 0, reflectionCoeffs, 0, RCorder);
109     
110             for (int i = 1; i <= lpcorder; i++) {
111                 for (int m = 1; m < i; m++) {
112                     tmp[i][m] = tmp[i - 1][m] - RC[i] * tmp[i - 1][i - m];
113                 }
114                 tmp[i][i] = RC[i];
115             }
116             ARParameters[0] = 1;
117             for (int m = 1; m <= lpcorder; m++) {
118                 ARParameters[m] = tmp[m][m];
119             }
120             return ARParameters;
121         }

124         /**
125          * Computes LPC Cepstra from the AR predictor parameters and alpha using a recursion invented by Oppenheim et al.
126          * The literature shows the optimal value of cepstral order to be:
127          * <p/>
128          * <pre>0.75 * LPCorder <= ceporder <= 1.25 * LPCorder</pre>
129          *
130          * @param ceporder is the order of the LPC cepstral vector to be computed.
131          * @return LPC cepstra
132          */
133         public double[] getData(int ceporder) {
134             int i;
135             double sum;
136     
137             if (ceporder <= 0) {
138                 return null;
139             }
140     
141             cepstrumOrder = ceporder;
142             cepstra = new double[cepstrumOrder];
143     
144             cepstra[0] = Math.log(alpha);
145             if (cepstrumOrder == 1) {
146                 return cepstra;
147             }
148     
149             cepstra[1] = -ARParameters[1];
150     
151             for (i = 2; i < Math.min(cepstrumOrder, order + 1); i++) {
152                 sum = (double) i * ARParameters[i];
153                 for (int j = 1; j < i; j++) {
154                     sum += ARParameters[j] * cepstra[i - j] * (double) (i - j);
155                 }
156                 cepstra[i] = -sum / (double) i;
157             }
158             for (; i < cepstrumOrder; i++) { // Only if cepstrumOrder > order+1
159                 sum = 0;
160                 for (int j = 1; j <= order; j++) {
161                     sum += ARParameters[j] * cepstra[i - j] * (double) (i - j);
162                 }
163                 cepstra[i] = -sum / (double) i;
164             }
165             return cepstra;
166         }

169         /**
170          * Computes a bi-linear frequency warped version of the LPC cepstrum from the LPC cepstrum. The recursive algorithm
171          * used is defined in Oppenheim's paper in Proceedings of IEEE, June 1972 The program has been written using g[x,y]
172          * = g_o[x,-y] where g_o is the array used by Oppenheim. To handle the reversed array index the recursion has been
173          * done DOWN the array index.
174          *
175          * @param warp          is the warping coefficient. For 16KHz speech 0.6 is good valued.
176          * @param nbilincepstra is the number of bilinear cepstral values to be computed from the linear frequency
177          *                      cepstrum.
178          * @return a bi-linear frequency warped version of the LPC cepstrum
179          */
180         public double[] getBilinearCepstra(double warp, int nbilincepstra) {
181             double[][] g = new double[nbilincepstra][cepstrumOrder];
182     
183             // Make a local copy as this gets destroyed
184             double[] lincep = new double[cepstrumOrder];
185             System.arraycopy(cepstra, 0, lincep, 0, cepstrumOrder);
186     
187             bilinearCepstra[0] = lincep[0];
188             lincep[0] = 0;
189             g[0][cepstrumOrder - 1] = lincep[cepstrumOrder - 1];
190             for (int i = 1; i < nbilincepstra; i++) {
191                 g[i][cepstrumOrder - 1] = 0;
192             }
193     
194             for (int i = cepstrumOrder - 2; i >= 0; i--) {
195                 g[0][i] = warp * g[0][i + 1] + lincep[i];
196                 g[1][i] = (1 - warp * warp) * g[0][i + 1] + warp * g[1][i + 1];
197                 for (int j = 2; j < nbilincepstra; j++) {
198                     g[j][i] = warp * (g[j][i + 1] - g[j - 1][i]) + g[j - 1][i + 1];
199                 }
200             }
201     
202             for (int i = 1; i <= nbilincepstra; i++) {
203                 bilinearCepstra[i] = g[i][0];
204             }
205     
206             return bilinearCepstra;
207         }
