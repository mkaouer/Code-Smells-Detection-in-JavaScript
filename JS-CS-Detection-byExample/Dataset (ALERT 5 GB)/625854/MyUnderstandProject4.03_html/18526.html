29     /**
30      * A simple Dialog demo showing a simple speech application built using Sphinx-4 that uses the DialogManager.
31      * <p/>
32      * This demo uses a DialogManager to manage a set of dialog states. Each dialog state potentially has its own grammar.
33      */
34     public class Dialog {
35     
36         /** Main method for running the Dialog demo. */
37         public static void main(String[] args) {
38             try {
39                 URL url;
40                 if (args.length > 0) {
41                     url = new File(args[0]).toURI().toURL();
42                 } else {
43                     url = Dialog.class.getResource("dialog.config.xml");
44                 }
45                 ConfigurationManager cm = new ConfigurationManager(url);
46     
47                 DialogManager dialogManager = (DialogManager)
48                         cm.lookup("dialogManager");
49     
50                 Recognizer weatherRecognizer = (Recognizer)
51                         cm.lookup("weatherRecognizer");
52     
53     
54                 System.out.println("\nWelcome to the Sphinx-4 Dialog Demo "
55                         + " - Version 1.0\n");
56     
57                 dialogManager.addNode("menu", new MyBehavior());
58                 dialogManager.addNode("email", new MyBehavior());
59                 dialogManager.addNode("games", new MyBehavior());
60                 dialogManager.addNode("news", new MyBehavior());
61                 dialogManager.addNode("music", new MyMusicBehavior());
62                 dialogManager.addNode("movies", new MyBehavior());
63                 dialogManager.addNode("phone", new MyBehavior());
64                 dialogManager.addNode("books", new MyBehavior());
65                 dialogManager.addNode("weather", new WeatherNode(weatherRecognizer));
66     
67                 dialogManager.setInitialNode("menu");
68     
69                 System.out.println("Loading dialogs ...");
70     
71                 dialogManager.allocate();
72                 System.out.println("Loading weather recognizer ...");
73                 weatherRecognizer.allocate();
74     
75     
76                 System.out.println("Running  ...");
77     
78                 dialogManager.go();
79     
80                 System.out.println("Cleaning up  ...");
81     
82                 dialogManager.deallocate();
83     
84             } catch (IOException e) {
85                 System.err.println("Problem when loading Dialog: " + e);
86             } catch (PropertyException e) {
87                 System.err.println("Problem configuring Dialog: " + e);
88             }
89             System.exit(0);
90         }
91     }

94     /**
95      * Defines the standard behavior for a node. The standard behavior is: <ul> <li> On entry the set of sentences that can
96      * be spoken is displayed. <li> On recognition if a tag returned contains the prefix 'dialog_' it indicates that control
97      * should transfer to another dialog node. </ul>
98      */
99     class MyBehavior extends NewGrammarDialogNodeBehavior {
100     
101         private Collection sampleUtterances;
102     
103     
104         /** Executed when we are ready to recognize */
105         public void onReady() {
106             super.onReady();
107             help();
108         }
109     
110     
111         /**
112          * Displays the help message for this node. Currently we display the name of the node and the list of sentences that
113          * can be spoken.
114          */
115         protected void help() {
116             System.out.println(" ======== " + getGrammarName() + " =======");
117             dumpSampleUtterances();
118             System.out.println(" =================================");
119         }
120     
121     
122         /**
123          * Executed when the recognizer generates a result. Returns the name of the next dialog node to become active, or
124          * null if we should stay in this node
125          *
126          * @param result the recongition result
127          * @return the name of the next dialog node or null if control should remain in the current node.
128          */
129         public String onRecognize(Result result) throws GrammarException {
130             String tag = super.onRecognize(result);
131     
132             if (tag != null) {
133                 System.out.println("\n "
134                         + result.getBestFinalResultNoFiller() + "\n");
135                 if (tag.equals("exit")) {
136                     System.out.println("Goodbye! Thanks for visiting!\n");
137                     System.exit(0);
138                 }
139                 if (tag.equals("help")) {
140                     help();
141                 } else if (tag.equals("stats")) {
142                     TimerPool.dumpAll();
143                 } else if (tag.startsWith("goto_")) {
144                     return tag.replaceFirst("goto_", "");
145                 } else if (tag.startsWith("browse")) {
146                     execute(tag);
147                 }
148             } else {
149                 System.out.println("\n Oops! didn't hear you.\n");
150             }
151             return null;
152         }
153     
154     
155         /**
156          * execute the given command
157          *
158          * @param cmd the command to execute
159          */
160         private void execute(String cmd) {
161             try {
162                 Runtime.getRuntime().exec(cmd);
163             } catch (IOException e) {
164                 // if we can't run the command, just fall back to
165                 // a non-working demo.
166             }
167         }
168     
169     
170         /**
171          * Collects the set of possible utterances.
172          * <p/>
173          * TODO: Note the current implementation just generates a large set of random utterances and tosses away any
174          * duplicates. There's no guarantee that this will generate all of the possible utterances. (yep, this is a hack)
175          *
176          * @return the set of sample utterances
177          */
178         private Collection collectSampleUtterances() {
179             Set set = new HashSet();
180             for (int i = 0; i < 100; i++) {
181                 String s = getGrammar().getRandomSentence();
182                 if (!set.contains(s)) {
183                     set.add(s);
184                 }
185             }
186     
187             List sampleList = new ArrayList(set);
188             Collections.sort(sampleList);
189             return sampleList;
190         }
191     
192     
193         /** Dumps out the set of sample utterances for this node */
194         private void dumpSampleUtterances() {
195             if (sampleUtterances == null) {
196                 sampleUtterances = collectSampleUtterances();
197             }
198     
199             for (Iterator i = sampleUtterances.iterator(); i.hasNext();) {
200                 System.out.println("  " + i.next());
201             }
202         }
203     
204     
205         /** Indicated that the grammar has changed and the collection of sample utterances should be regenerated. */
206         protected void grammarChanged() {
207             sampleUtterances = null;
208         }
209     }

211     /**
212      * An extension of the standard node behavior for music. This node will add rules to the grammar based upon the contents
213      * of the music.txt file. This provides an example of how to extend a grammar directly from code as opposed to writing
214      * out a JSGF file.
215      */
216     class MyMusicBehavior extends MyBehavior {
217     
218         private List songList = new ArrayList();
219     
220     
221         /** Creates a music behavior */
222         MyMusicBehavior() {
223             try {
224                 InputStream is = Dialog.class.getResourceAsStream("playlist.txt");
225                 BufferedReader br = new BufferedReader(new InputStreamReader(is));
226                 String song;
227     
228                 while ((song = br.readLine()) != null) {
229                     if (song.length() > 0) {
230                         songList.add(song);
231                     }
232                 }
233                 br.close();
234             } catch (IOException ioe) {
235                 System.err.println("Can't get playlist");
236             }
237         }
238     
239     
240         /** Executed when we enter this node. Displays the active grammar */
241         public void onEntry() throws IOException {
242             super.onEntry();
243     
244             // now lets add our custom songs from the play list
245             // First, get the JSAPI RuleGrammar
246     
247             RuleGrammar ruleGrammar = getGrammar().getRuleGrammar();
248     
249             // now lets add a rule for each song in the play list
250     
251             String ruleName = "song";
252             int count = 1;
253     
254             try {
255                 for (Iterator i = songList.iterator(); i.hasNext();) {
256                     String song = (String) i.next();
257                     String newRuleName = ruleName + count;
258                     Rule newRule = ruleGrammar.ruleForJSGF("listen to " + song
259                             + " { " + newRuleName + " }");
260                     ruleGrammar.setRule(newRuleName, newRule, true);
261                     ruleGrammar.setEnabled(newRuleName, true);
262                     count++;
263                 }
264             } catch (GrammarException ge) {
265                 System.out.println("Trouble with the grammar " + ge);
266                 throw new IOException("Can't add rules for playlist " + ge);
267             }
268             // now lets commit the changes
269             getGrammar().commitChanges();
270             grammarChanged();
271         }
272     
273     
274         /**
275          * Dumps out the rules of the given rule grammar, used for debugging.
276          *
277          * @param rg the rule grammar to dump
278          */
279         private void dumpGrammar(RuleGrammar rg) {
280             String[] ruleNames = rg.listRuleNames();
281     
282             for (int i = 0; i < ruleNames.length; i++) {
283                 String enabled = rg.isEnabled(ruleNames[i]) ? "ON" : "OFF";
284                 System.out.println("Rule: " + ruleNames[i] + " " +
285                         rg.getRule(ruleNames[i]) + " " + enabled);
286             }
287         }
288     }

290     /**
291      * Defines the behavior for a weather node. The weather node allows the user to dictate a weather forecast.  To do this
292      * we can't use a JSGF grammar since JSGF grammars are not appropriate for dictation, so instead we have a special
293      * weather node that will install a 'weather recognizer' as the current recognizer when this node is active.  The
294      * weather recognizer is configured to use a language model suitable for weather forecasts.
295      */
296     
297     class WeatherNode extends DialogNodeBehavior {
298     
299         private Recognizer weatherRecognizer;
300         private Recognizer savedRecognizer;
301         private DialogManager dialogManager;
302     
303     
304         /**
305          * Creates the WeatherNode
306          *
307          * @param recognizer the weather recognizer
308          */
309         public WeatherNode(Recognizer recognizer) {
310             this.weatherRecognizer = recognizer;
311         }
312     
313     
314         /**
315          * Called during the initialization phase
316          *
317          * @param node the dialog node that the behavior is attached to
318          */
319         public void onInit(DialogManager.DialogNode node) {
320             super.onInit(node);
321             dialogManager = node.getDialogManager();
322         }
323     
324     
325         /** Called when this node becomes the active node */
326         public void onEntry() throws IOException {
327             savedRecognizer = dialogManager.getRecognizer();
328             dialogManager.setRecognizer(weatherRecognizer);
329             System.out.println("  Give your best imitation of ");
330             System.out.println("  a british weather forecaster");
331             System.out.println();
332             System.out.println(" Say 'forecast over' when you are done.");
333             System.out.println();
334         }
335     
336     
337         /** Called when this node is ready to perform recognition */
338         public void onReady() {
339             trace("Ready " + getName());
340         }
341     
342     
343         /*
344         * Called with the recognition results. Should return a string
345         * representing the name of the next node.
346         *
347         * @return a tag indicating where to go next. An empty string
348         * indicates that control should stay in this node.
349         */
350         public String onRecognize(Result result) throws GrammarException {
351             trace("Recognize result: " + result.getBestFinalResultNoFiller());
352             String forecast = result.getBestFinalResultNoFiller();
353     
354             System.out.println("Forecast: " + forecast);
355     
356             if ("forecast over".equals(forecast)) {
357                 return "menu";
358             }
359             return "";
360         }
361     
362     
363         /** Called when this node is no lnoger the active node */
364         public void onExit() {
365             System.out.println();
366             dialogManager.setRecognizer(savedRecognizer);
367         }
368     }
