91         /**
92          * Gets the  set of hmm nodes associated with the given end node
93          *
94          * @param endNode the end node
95          * @return an array of associated hmm nodes
96          */
97         public HMMNode[] getHMMNodes(EndNode endNode) {
98             HMMNode[] results = endNodeMap.get(endNode.getKey());
99             if (results == null) {
100                 // System.out.println("Filling cache for " + endNode.getKey()
101                 //        + " size " + endNodeMap.size());
102                 Map<HMM, HMMNode> resultMap = new HashMap<HMM, HMMNode>();
103                 Unit baseUnit = endNode.getBaseUnit();
104                 Unit lc = endNode.getLeftContext();
105                 for (Iterator<Unit> i = entryPoints.iterator(); i.hasNext();) {
106                     Unit rc = i.next();
107                     HMM hmm = getHMM(baseUnit, lc, rc, HMMPosition.END);
108                     HMMNode hmmNode = resultMap.get(hmm);
109                     if (hmmNode == null) {
110                         hmmNode = new HMMNode(hmm, LogMath.getLogOne());
111                         resultMap.put(hmm, hmmNode);
112                     }
113                     hmmNode.addRC(rc);
114                     for (Node node : endNode.getSuccessors()) {
115                         WordNode wordNode = (WordNode) node;
116                         hmmNode.addSuccessor(wordNode);
117                     }
118                 }
119     
120                 // cache it
121                 results = (HMMNode[]) resultMap.values().toArray(
122                         new HMMNode[resultMap.size()]);
123                 endNodeMap.put(endNode.getKey(), results);
124             }
125     
126             // System.out.println("GHN: " + endNode + " " + results.length);
127             return results;
128         }

149         /** Compiles the vocabulary into an HMM Tree */
150         private void compile() {
151             collectEntryAndExitUnits();
152             entryPointTable = new EntryPointTable(entryPoints);
153             addWords();
154             entryPointTable.createEntryPointMaps();
155             freeze();
156         }

160         void dumpTree() {
161             System.out.println("Dumping Tree ...");

163             dumpTree(0, getInitialNode(), dupNode);
164             System.out.println("... done Dumping Tree");

168         /**
169          * Dumps the tree
170          *
171          * @param level   the level of the dump
172          * @param node    the root of the tree to dump
173          * @param dupNode map of visited nodes
174          */
175         private void dumpTree(int level, Node node, Map<Node, Node> dupNode) {
176             if (dupNode.get(node) == null) {
177                 dupNode.put(node, node);
178                 System.out.println(Utilities.pad(level * 1) + node);
179                 if (!(node instanceof WordNode)) {
180                     for (Node nextNode : node.getSuccessors()) {
181                         dumpTree(level + 1, nextNode, dupNode);
182                     }
183                 }
184             }
185         }

188         /** Collects all of the entry and exit points for the vocabulary. */
189         private void collectEntryAndExitUnits() {
190             Collection<Word> words = getAllWords();
191             for (Iterator<Word> i = words.iterator(); i.hasNext();) {
192                 Word word = i.next();
193                 for (int j = 0; j < word.getPronunciations().length; j++) {
194                     Pronunciation p = word.getPronunciations()[j];
195                     Unit first = p.getUnits()[0];
196                     Unit last = p.getUnits()[p.getUnits().length - 1];
197                     entryPoints.add(first);
198                     exitPoints.add(last);
199                 }
200             }
201     
202             if (debug) {
203                 System.out.println("Entry Points: " + entryPoints.size());
204                 System.out.println("Exit Points: " + exitPoints.size());
205             }
206         }

232         /**
233          * Adds a single word to the lex tree
234          *
235          * @param word the word to add
236          */
237         private void addWord(Word word) {
238             float prob = getWordUnigramProbability(word);
239             Pronunciation[] pronunciations = word.getPronunciations();
240             for (int i = 0; i < pronunciations.length; i++) {
241                 addPronunciation(pronunciations[i], prob);
242             }
243         }

246         /**
247          * Adds the given pronunciation to the lex tree
248          *
249          * @param pronunciation the pronunciation
250          * @param probability   the unigram probability
251          */
252         private void addPronunciation(Pronunciation pronunciation,
253                                       float probability) {
254             Unit baseUnit;
255             Unit lc;
256             Unit rc;
257             Node curNode;
258             WordNode wordNode;
259     
260             Unit[] units = pronunciation.getUnits();
261             baseUnit = units[0];
262             EntryPoint ep = entryPointTable.getEntryPoint(baseUnit);
263     
264             ep.addProbability(probability);
265     
266             if (units.length > 1) {
267                 curNode = ep.getNode();
268                 lc = baseUnit;
269                 for (int i = 1; i < units.length - 1; i++) {
270                     baseUnit = units[i];
271                     rc = units[i + 1];
272                     HMM hmm = getHMM(baseUnit, lc, rc, HMMPosition.INTERNAL);
273                     if (hmm == null) {
274                         lm.getLogger().severe("Missing HMM for unit " + baseUnit.getName() + " with lc=" + lc.getName() + " rc=" + rc.getName());
275                     } else {
276                         curNode = curNode.addSuccessor(hmm, probability);
277                     }
278                     lc = baseUnit;          // next lc is this baseUnit
279                 }
280     
281                 // now add the last unit as an end unit
282                 baseUnit = units[units.length - 1];
283                 EndNode endNode = new EndNode(baseUnit, lc, probability);
284                 curNode = curNode.addSuccessor(endNode, probability);
285                 wordNode = curNode.addSuccessor(pronunciation, probability);
286                 if (wordNode.getWord() == dictionary.getSentenceEndWord()) {
287                     sentenceEndWordNode = wordNode;
288                 }
289             } else {
290                 ep.addSingleUnitWord(pronunciation);
291             }
292         }

295         /**
296          * Retrieves an HMM for a unit in context. If there is no direct match, the nearest match will be used.  Note that
297          * we are currently only dealing with, at most, single unit left and right contexts.
298          *
299          * @param base the base CI unit
300          * @param lc   the left context
301          * @param rc   the right context
302          * @param pos  the position of the base unit within the word
303          * @return the HMM. (This should never return null)
304          */
305         private HMM getHMM(Unit base, Unit lc, Unit rc, HMMPosition pos) {
306             int id = -1;
307             int bid = hmmPool.getID(base);
308             int lid = hmmPool.getID(lc);
309             int rid = hmmPool.getID(rc);
310     
311             if (!hmmPool.isValidID(bid)) {
312                 lm.getLogger().severe("Bad HMM Unit: " + base.getName());
313                 return null;
314             }
315             if (!hmmPool.isValidID(lid)) {
316                 lm.getLogger().severe("Bad HMM Unit: " + lc.getName());
317                 return null;
318             }
319             if (!hmmPool.isValidID(rid)) {
320                 lm.getLogger().severe("Bad HMM Unit: " + rc.getName());
321                 return null;
322             }
323             id = hmmPool.buildID(bid, lid, rid);
324             if (id < 0) {
325                 lm.getLogger().severe("Unable to build HMM Unit ID for " + base.getName() + " lc=" + lc.getName() + " rc=" + rc.getName());
326                 return null;
327             }
328             HMM hmm = hmmPool.getHMM(id, pos);
329             if (hmm == null) {
330                 lm.getLogger().severe("Missing HMM Unit for " + base.getName() + " lc=" + lc.getName() + " rc=" + rc.getName());
331             }
332     
333             return hmm;
334         }

337         /**
338          * Gets the unigram probability for the given word
339          *
340          * @param word the word
341          * @return the unigram probability for the word.
342          */
343         private float getWordUnigramProbability(Word word) {
344             float prob = LogMath.getLogOne();
345             if (!word.isFiller()) {
346                 Word[] wordArray = new Word[1];
347                 wordArray[0] = word;
348                 prob = lm.getProbability((new WordSequence(wordArray)));
349                 // System.out.println("gwup: " + word + " " + prob);
350                 prob *= languageWeight;
351             }
352             return prob;
353         }

356         /**
357          * Returns the entire set of words, including filler words
358          *
359          * @return the set of all words (as Word objects)
360          */
361         private Set<Word> getAllWords() {
362             if (allWords == null) {
363                 allWords = new HashSet<Word>();
364                 for (String spelling : lm.getVocabulary()) {
365                     Word word = dictionary.getWord(spelling);
366                     if (word != null) {
367                         allWords.add(word);
368                     }
369                 }
370     
371                 if (addFillerWords) {
372                     Word[] fillerWords = dictionary.getFillerWords();
373                     for (int i = 0; i < fillerWords.length; i++) {
374                         allWords.add(fillerWords[i]);
375                     }
376                 } else if (addSilenceWord) {
377                     allWords.add(dictionary.getSilenceWord());
378                 }
379             }
380             return allWords;
381         }

402              *
403              * @param entryPointCollection the set of possible entry points

405             EntryPointTable(Collection<Unit> entryPointCollection) {
406                 entryPoints = new HashMap<Unit, EntryPoint>();
407                 for (Iterator<Unit> i = entryPointCollection.iterator(); i.hasNext();) {
408                     Unit unit = i.next();
409                     entryPoints.put(unit, new EntryPoint(unit));

422             }

429                     ep.createEntryPointMap();
430                 }
431             }

439                 }

447                     ep.dump();
448                 }

453         /** Manages a single entry point. */
454         class EntryPoint {
455     
456             Unit baseUnit;
457             Node baseNode;      // second units and beyond start here
458             Map<Unit, Node> unitToEntryPointMap;
459             List<Pronunciation> singleUnitWords;
460             int nodeCount = 0;
461             Set<Unit> rcSet;
462             float totalProbability;
463     
464     
465             /**
466              * Creates an entry point for the given usnit
467              *
468              * @param baseUnit the EntryPoint is created for this unit
469              */
470             EntryPoint(Unit baseUnit) {
471                 this.baseUnit = baseUnit;
472                 this.baseNode = new Node(LogMath.getLogZero());
473                 this.unitToEntryPointMap = new HashMap<Unit, Node>();
474                 this.singleUnitWords = new ArrayList<Pronunciation>();
475                 this.totalProbability = LogMath.getLogZero();
476             }
477     
478     
479             /**
480              * Given a left context get a node that represents a single set of entry points into this unit
481              *
482              * @param leftContext the left context of interest
483              * @return the node representing the entry point
484              */
485             Node getEntryPointsFromLeftContext(Unit leftContext) {
486                 return unitToEntryPointMap.get(leftContext);
487             }
488     
489     
490             /**
491              * Accumulates the probability for this entry point
492              *
493              * @param probability a new  probability
494              */
495             void addProbability(float probability) {
496                 if (probability > totalProbability) {
497                     totalProbability = probability;
498                 }
499             }
500     
501     
502             /**
503              * Returns the probability for all words reachable from this node
504              *
505              * @return the log probability
506              */
507             float getProbability() {
508                 return totalProbability;
509             }
510     
511     
512             /** Once we have built the full entry point we can eliminate some fields */
513             void freeze() {
514                 for (Iterator<Node> i = unitToEntryPointMap.values().iterator();
515                      i.hasNext();) {
516                     Node node = i.next();
517                     node.freeze();
518                 }
519                 singleUnitWords = null;
520                 rcSet = null;
521             }
522     
523     
524             /**
525              * Gets the base node for this entry point
526              *
527              * @return the base node
528              */
529             Node getNode() {
530                 return baseNode;
531             }
532     
533     
534             /**
535              * Adds a one-unit word to this entry point. Such single unit words need to be dealt with specially.
536              *
537              * @param p the pronunciation of the single unit word
538              */
539             void addSingleUnitWord(Pronunciation p) {
540                 singleUnitWords.add(p);

544             /**
545              * Gets the set of possible right contexts that we can transition to from this entry point
546              *
547              * @return the set of possible transition points.
548              */
549             private Collection<Unit> getEntryPointRC() {
550                 if (rcSet == null) {
551                     rcSet = new HashSet<Unit>();
552                     for (Node node : baseNode.getSuccessors()) {
553                         UnitNode unitNode = (UnitNode) node;
554                         rcSet.add(unitNode.getBaseUnit());
555                     }
556                 }
557                 return rcSet;
558             }
559     
560     
561             /**
562              * Creates the entry point map for this entry point.  The entry point map is represented by the
563              * unitToEntryPointMap. It contains a node for each possible left context. The node successors point to the
564              * following hmm nodes (usually associated with the next units that can follow from this entry point.
565              */
566             void createEntryPointMap() {
567                 for (Iterator<Unit> i = exitPoints.iterator(); i.hasNext();) {
568                     Unit lc = i.next();
569                     Node epNode = new Node(LogMath.getLogZero());
570                     for (Iterator<Unit> j = getEntryPointRC().iterator(); j.hasNext();) {
571                         Unit rc = j.next();
572                         HMM hmm = getHMM(baseUnit, lc, rc, HMMPosition.BEGIN);
573                         Node addedNode = epNode.addSuccessor(hmm, getProbability());
574                         nodeCount++;
575                         connectEntryPointNode(addedNode, rc);
576                     }
577                     connectSingleUnitWords(lc, epNode);
578                     unitToEntryPointMap.put(lc, epNode);
579                 }
580             }
581     
582     
583             /**
584              * An alternate version of createEntryPointMap that compresses common hmms across all entry points, not just
585              * those shaing the same left context.  This really doesn't speed things up in the least bit, so it is not worth
586              * the effort.
587              */
588             void createEntryPointMap_alternateVersion() {
589                 HashMap<HMM, Node> map = new HashMap<HMM, Node>();
590                 for (Iterator<Unit> i = exitPoints.iterator(); i.hasNext();) {
591                     Unit lc = i.next();
592                     Node epNode = new Node(LogMath.getLogZero());
593                     for (Iterator<Unit> j = getEntryPointRC().iterator(); j.hasNext();) {
594                         Unit rc = j.next();
595                         HMM hmm = getHMM(baseUnit, lc, rc, HMMPosition.BEGIN);
596                         Node addedNode;
597     
598                         if ((addedNode = map.get(hmm)) == null) {
599                             addedNode = epNode.addSuccessor(hmm, getProbability());
600                             map.put(hmm, addedNode);
601                         } else {
602                             epNode.putSuccessor(hmm, addedNode);
603                         }
604     
605                         nodeCount++;
606                         connectEntryPointNode(addedNode, rc);
607                     }
608                     connectSingleUnitWords(lc, epNode);
609                     unitToEntryPointMap.put(lc, epNode);
610                 }
611             }
612     
613     
614             /**
615              * Connects the single unit words associated with this entry point.   The singleUnitWords list contains all
616              * single unit pronunciations that have as their sole unit, the unit associated with this entry point. Entry
617              * points for these words are added to the epNode for all possible left (exit) and right (entry) contexts.
618              *
619              * @param lc     the left context
620              * @param epNode the entry point node
621              */
622             private void connectSingleUnitWords(Unit lc, Node epNode) {
623                 if (singleUnitWords.size() > 0) {
624                     for (Iterator<Unit> i = entryPoints.iterator(); i.hasNext();) {
625                         Unit rc = i.next();
626                         HMM hmm = getHMM(baseUnit, lc, rc, HMMPosition.SINGLE);
627                         HMMNode tailNode = (HMMNode)
628                                 epNode.addSuccessor(hmm, getProbability());
629                         WordNode wordNode;
630                         tailNode.addRC(rc);
631                         nodeCount++;
632     
633                         for (int j = 0; j < singleUnitWords.size(); j++) {
634                             Pronunciation p = singleUnitWords.get(j);
635     
636     
637                             if (p.getWord() == dictionary.getSentenceStartWord()) {
638                                 initialNode = new InitialWordNode(p, tailNode);
639                             } else {
640                                 float prob = getWordUnigramProbability(p.getWord());
641                                 wordNode = tailNode.addSuccessor(p, prob);
642                                 if (p.getWord() ==
643                                         dictionary.getSentenceEndWord()) {
644                                     sentenceEndWordNode = wordNode;
645                                 }
646                             }
647                             nodeCount++;
648                         }
649                     }
650                 }
651             }
652     
653     
654             /**
655              * Connect the entry points that match the given rc to the given epNode
656              *
657              * @param epNode add matching successors here
658              * @param rc     the next unit
659              */
660             private void connectEntryPointNode(Node epNode, Unit rc) {
661                 for (Node node : baseNode.getSuccessors()) {
662                     UnitNode successor = (UnitNode) node;
663                     if (successor.getBaseUnit() == rc) {
664                         epNode.addSuccessor(successor);
665                     }
666                 }
667             }

670             /** Dumps the entry point */
671             void dump() {
672                 System.out.println("EntryPoint " + baseUnit + " RC Followers: "
673                         + getEntryPointRC().size());
674                 int count = 0;
675                 Collection<Unit> rcs = getEntryPointRC();
676                 System.out.print("    ");
677                 for (Iterator<Unit> i = rcs.iterator(); i.hasNext();) {
678                     Unit rc = i.next();
679                     System.out.print(Utilities.pad(rc.getName(), 4));
680                     if (count++ >= 12) {
681                         count = 0;
682                         System.out.println();
683                         System.out.print("    ");
684                     }
685                 }
686                 System.out.println();
687             }
688         }

693     /** Represents a node in the HMM Tree */
694     
695     // For large vocabularies we may create millions of these objects,
696     // therefore they are extremely space sensitive. So we want to make
697     // these objects as small as possible.  The requirements for these
698     // objects when building the tree of nodes are very different from once
699     // we have built it. When building, we need to easily add successor
700     // nodes and quickly identify duplicate children nodes. After the tree
701     // is built we just need to quickly identify successors.  We want the
702     // flexibility of a map to manage successors at startup, but we don't
703     // want the space penalty (at least 5 32 bit fields per map), instead
704     // we'd like an array.  To support this dual mode, we manage the
705     // successors in an Object which can either be a Map or a List
706     // depending upon whether the node has been frozen or not.
707     class Node {
708     
709         private static int nodeCount = 0;
710         private static int successorCount = 0;

774         /**
775          * Gets the successor map for this node
776          *
777          * @return the successor map
778          */
779         @SuppressWarnings({"unchecked"})
780         private Map<Object, Node> getSuccessorMap() {
781             if (successors == null) {
782                 successors = new HashMap<Object, Node>(4);
783             }
784     
785             assert successors instanceof Map;
786             return (Map<Object, Node>) successors;
787         }

789     
790         /** Freeze the node. Convert the successor map into an array list */
791         void freeze() {
792             if (successors instanceof Map<?,?>) {
793                 Map<Object, Node> map = getSuccessorMap();
794                 List<Node> frozenSuccessors = new ArrayList<Node>(map.values().size());
795                 successors = null; // avoid recursive death spiral
796                 for (Iterator<Node> i = map.values().iterator(); i.hasNext();) {
797                     Node node = i.next();
798                     frozenSuccessors.add(node);
799                     node.freeze();
800                 }
801                 successors = frozenSuccessors;
802                 successorCount += frozenSuccessors.size();
803             }
804         }

813         /**
814          * Adds a child node holding an hmm to the successor.  If a node similar to the child has already been added, we use
815          * the previously added node, otherwise we add this. Also, we record the base unit of the child in the set of right
816          * context
817          *
818          * @param hmm the hmm to add
819          * @return the node that holds the hmm (new or old)
820          */
821         Node addSuccessor(HMM hmm, float probability) {
822             Node child = null;
823             Node matchingChild = getSuccessor(hmm);
824             if (matchingChild == null) {
825                 child = new HMMNode(hmm, probability);
826                 putSuccessor(hmm, child);
827             } else {
828                 if (matchingChild.getUnigramProbability() < probability) {
829                     matchingChild.setUnigramProbability(probability);
830                 }
831                 child = matchingChild;
832             }
833             return child;
834         }

1110     /** A node that represents an HMM in the hmm tree */
1111     class HMMNode extends UnitNode {
1112     
1113         private HMM hmm;
1114     
1115         // There can potentially be a large number of nodes (millions),
1116         // therefore it is important to conserve space as much as
1117         // possible.  While building the HMMNodes, we keep right contexts
1118         // in a set to allow easy pruning of duplicates.  Once the tree is
1119         // entirely built, we no longer need to manage the right contexts
1120         // as a set, a simple array will do. The freeze method converts
1121         // the set to the array of units.  This rcSet object holds the set
1122         // during construction and the array after the freeze.
1123     
1124         private Object rcSet;
1125     
1126     
1127         /**
1128          * Creates the node, wrapping the given hmm
1129          *
1130          * @param hmm the hmm to hold
1131          */
1132         HMMNode(HMM hmm, float probablilty) {
1133             super(probablilty);
1134             this.hmm = hmm;
1135     
1136             Unit base = getBaseUnit();
1137     
1138             int type = SIMPLE_UNIT;
1139             if (base.isSilence()) {
1140                 type = SILENCE_UNIT;
1141             } else if (base.isFiller()) {
1142                 type = FILLER_UNIT;
1143             } else if (hmm.getPosition().isWordBeginning()) {
1144                 type = WORD_BEGINNING_UNIT;
1145             }
1146             setType(type);
1147         }
1148     
1149     
1150         /**
1151          * Returns the base unit for this hmm node
1152          *
1153          * @return the base unit
1154          */
1155         Unit getBaseUnit() {
1156             // return hmm.getUnit().getBaseUnit();
1157             return hmm.getBaseUnit();
1158         }
1159     
1160     
1161         /**
1162          * Returns the hmm for this node
1163          *
1164          * @return the hmm
1165          */
1166         HMM getHMM() {
1167             return hmm;
1168         }
1169     
1170     
1171         HMMPosition getPosition() {
1172             return hmm.getPosition();
1173         }
1174     
1175     
1176         HMM getKey() {
1177             return getHMM();
1178         }
1179     
1180     
1181         /**
1182          * Returns a string representation for this object
1183          *
1184          * @return a string representation
1185          */
1186         public String toString() {
1187             return "HMMNode " + hmm + " p " + getUnigramProbability();
1188         }
1189     
1190     
1191         /**
1192          * Adds a right context to the set of possible right contexts for this node. This is typically only needed for hmms
1193          * at the ends of words.
1194          *
1195          * @param rc the right context.
1196          */
1197         void addRC(Unit rc) {
1198             getRCSet().add(rc);
1199         }
1200     
1201     
1202         /** Freeze this node. Convert the set into an array to reduce memory overhead */
1203         @SuppressWarnings({"unchecked"})
1204         void freeze() {
1205             super.freeze();
1206             if (rcSet instanceof Set) {
1207                 Set<Unit> set = (Set<Unit>) rcSet;
1208                 rcSet = set.toArray(new Unit[set.size()]);
1209             }
1210         }
1211     
1212     
1213         /**
1214          * Gets the rc as a set. If we've already been frozen it is an error
1215          *
1216          * @return the set of right contexts
1217          */
1218         @SuppressWarnings({"unchecked"})
1219         private Set<Unit> getRCSet() {
1220             if (rcSet == null) {
1221                 rcSet = new HashSet<Unit>();
1222             }
1223     
1224             assert rcSet instanceof HashSet;
1225             return (Set<Unit>) rcSet;
1226         }
1227     
1228     
1229         /**
1230          * returns the set of right contexts for this node
1231          *
1232          * @return the set of right contexts
1233          */
1234         Unit[] getRC() {
1235             if (rcSet instanceof HashSet<?>) {
1236                 freeze();
1237             }
1238             return (Unit[]) rcSet;
1239         }
1240     }
