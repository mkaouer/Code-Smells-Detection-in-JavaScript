13     package edu.cmu.sphinx.linguist.acoustic;
14     
15     /** Represents a unit of speech. Units may represent phones, words or any other suitable unit */
16     public class Unit {
17     
18         private String name;
19         private boolean filler = false;
20         private boolean silence = false;
21         private Context context = null;
22         private int baseID;
23         private Unit baseUnit;
24     
25         private volatile String key = null;
26     
27     
28         /**
29          * Constructs a context dependent  unit.  Constructors are package private, use the UnitManager to create and access
30          * units.
31          *
32          * @param name   the name of the unit
33          * @param filler <code>true</code> if the unit is a filler unit
34          * @param id     the base id for the unit
35          */
36         Unit(String name, boolean filler, int id) {
37             this.name = name;
38             this.filler = filler;
39             this.context = Context.EMPTY_CONTEXT;
40             this.baseID = id;
41             this.baseUnit = this;
42             if (name.equals(UnitManager.SILENCE_NAME)) {
43                 silence = true;
44             }
45         }
46     
47     
48         /**
49          * Constructs a context dependent  unit.  Constructors are package private, use the UnitManager to create and access
50          * units.
51          *
52          * @param baseUnit the base id for the unit
53          * @param filler   <code>true</code> if the unit is a filler unit
54          * @param context  the context for this unit
55          */
56         Unit(Unit baseUnit, boolean filler, Context context) {
57             this.baseUnit = baseUnit;
58             this.filler = filler;
59             this.context = context;
60             this.name = baseUnit.getName();
61             this.baseID = baseUnit.getBaseID();
62             if (name.equals(UnitManager.SILENCE_NAME)) {
63                 silence = true;
64             }
65         }
66     
67     
68         /**
69          * Determines if this unit is context dependent
70          *
71          * @return true if the unit is context dependent
72          */
73         public boolean isContextDependent() {
74             return getContext() != Context.EMPTY_CONTEXT;
75         }
76     
77     
78         /**
79          * Returns the context for this unit
80          *
81          * @return the context for this unit (or null if context independent
82          */
83         public Context getContext() {
84             return context;
85         }
86     
87     
88         /**
89          * Determines if this unit is a filler unit
90          *
91          * @return <code>true</code> if the unit is a filler unit
92          */
93         public boolean isFiller() {
94             return filler;
95         }
96     
97     
98         /**
99          * Determines if this unit is the silence unit
100          *
101          * @return true if the unit is the silence unit
102          */
103         public boolean isSilence() {
104             return silence;
105         }
106     
107     
108         /**
109          * Checks to see of an object is equal to this unit
110          *
111          * @param o the object to check
112          * @return true if the objects are equal
113          */
114         public boolean equals(Object o) {
115             if (this == o) {
116                 return true;
117             } else if (o instanceof Unit) {
118                 Unit otherUnit = (Unit) o;
119                 return getKey().equals(otherUnit.getKey());
120             } else {
121                 return false;
122             }
123         }
124     
125     
126         /**
127          * calculates a hashCode for this unit. Since we defined an equals for Unit, we must define a hashCode as well
128          *
129          * @return the hashcode for this object
130          */
131         public int hashCode() {
132             return getKey().hashCode();
133         }
134     
135     
136         /**
137          * Converts to a string
138          *
139          * @return string version
140          */
141         public String toString() {
142             if (key == null) {
143                 if (context == Context.EMPTY_CONTEXT) {
144                     key = (filler ? "*" : "") + name;
145                 } else {
146                     key = (filler ? "*" : "") + name + "[" + context + "]";
147                 }
148             }
149             return key;
150         }
151     
152     
153         /**
154          * Gets the name for this unit
155          *
156          * @return the name for this unit
157          */
158         public String getName() {
159             return name;
160         }
161     
162     
163         /**
164          * Gets the base ID for this unit
165          *
166          * @return the id
167          */
168         public int getBaseID() {
169             return baseID;
170         }
171     
172     
173         /** gets the key for this unit */
174         private String getKey() {
175             return toString();
176         }
177     
178     
179         /**
180          * Gets the  base unit associated with this HMM
181          *
182          * @return the unit associated with this HMM
183          */
184         public Unit getBaseUnit() {
185             return baseUnit;
186         }

188     
189         /**
190          * Checks to see if the given unit with associated contexts is a partial match for this unit.   Zero, One or both
191          * contexts can be null. A null context matches any context
192          *
193          * @param name    the name of the unit
194          * @param context the  context to match against
195          * @return true if this unit matches the name and non-null context
196          */
197         public boolean isPartialMatch(String name, Context context) {
198             return (getName().equals(name) &&
199                     context.isPartialMatch(this.context));
200         }
201     
202     
203         /**
204          * Creates and returns an empty context with the given size. The context is padded with SIL filler
205          *
206          * @param size the size of the context
207          * @return the context
208          */
209     
210         public static Unit[] getEmptyContext(int size) {
211             Unit[] context = new Unit[size];
212     
213             for (int i = 0; i < context.length; i++) {
214                 context[i] = UnitManager.SILENCE;
215             }
216             return context;
217         }
218     
219     
220         /**
221          * Checks to see that there is 100% overlap in the given contexts
222          *
223          * @param a context to check for a match
224          * @param b context to check for a match
225          * @return <code>true</code> if the contexts match
226          */
227         public static boolean isContextMatch(Unit[] a, Unit[] b) {
228             if (a == null || b == null) {
229                 return a == b;
230             } else if (a.length != b.length) {
231                 return false;
232             } else {
233                 for (int i = 0; i < a.length; i++) {
234                     if (!a[i].getName().equals(b[i].getName())) {
235                         return false;
236                     }
237                 }
238                 return true;
239             }
240         }
241     }
