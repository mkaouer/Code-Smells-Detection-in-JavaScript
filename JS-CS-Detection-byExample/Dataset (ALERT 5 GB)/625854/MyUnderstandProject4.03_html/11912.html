13     package edu.cmu.sphinx.linguist.language.ngram.large;
14     
15     
16     /** Represents a probability, a backoff probability, and the location of the first bigram entry. */
17     class UnigramProbability {
18     
19         private int wordID;
20         private float logProbability;
21         private float logBackoff;
22         private int firstBigramEntry;
23     
24     
25         /**
26          * Constructs a UnigramProbability
27          *
28          * @param wordID           the id of the word
29          * @param logProbability   the probability
30          * @param logBackoff       the backoff probability
31          * @param firstBigramEntry the first bigram entry
32          */
33         public UnigramProbability(int wordID, float logProbability,
34                                   float logBackoff, int firstBigramEntry) {
35             this.wordID = wordID;
36             this.logProbability = logProbability;
37             this.logBackoff = logBackoff;
38             this.firstBigramEntry = firstBigramEntry;
39         }
40     
41     
42         /**
43          * Returns a string representation of this object
44          *
45          * @return the string form of this object
46          */
47         public String toString() {
48             return "Prob: " + logProbability + " " + logBackoff;
49         }
50     
51     
52         /**
53          * Returns the word ID of this unigram
54          *
55          * @return the word ID of this unigram
56          */
57         public int getWordID() {
58             return wordID;
59         }
60     
61     
62         /**
63          * Returns the log probability of this unigram.
64          *
65          * @return the log probability of this unigram
66          */
67         public float getLogProbability() {
68             return logProbability;
69         }
70     
71     
72         /**
73          * Returns the log backoff weight of this unigram
74          *
75          * @return the log backoff weight of this unigram
76          */
77         public float getLogBackoff() {
78             return logBackoff;
79         }
80     
81     
82         /**
83          * Returns the index of the first bigram entry of this unigram.
84          *
85          * @return the index of the first bigram entry of this unigram
86          */
87         public int getFirstBigramEntry() {
88             return firstBigramEntry;
89         }
90     
91     
92         /**
93          * Sets the log probability of this unigram
94          *
95          * @param logProbability the new log probability of this unigram
96          */
97         public void setLogProbability(float logProbability) {
98             this.logProbability = logProbability;
99         }
100     
101     
102         /**
103          * Sets the log backoff weight.
104          *
105          * @param logBackoff the new log backoff weight
106          */
107         public void setLogBackoff(float logBackoff) {
108             this.logBackoff = logBackoff;
109         }
110     }
