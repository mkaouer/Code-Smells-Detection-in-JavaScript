5     /**
6      * {@code FrontEnd} element that asserts the audio-stream to be continuous. This is often a mandatory property for
7      * frontend setups. The component operates on the acoustic data level and needs to plugged into the frontend
8      * before the actual feature extraction starts.
9      * <p/>
10      * This component can help to debug new VAD implementations, where it has been shown that data-blocks easily get lost.
11      *
12      * @author Holger Brandl
13      */
14     public class AudioContinuityTester extends BaseDataProcessor {
15     
16         long lastSampleNum = -1;
17     
18     
19         public Data getData() throws DataProcessingException {
20             Data d = getPredecessor().getData();
21     
22             assert isAudioStreamContinuous(d) : "audio stream is not continuous";
23     
24             return d;
25         }
26     
27     
28         private boolean isAudioStreamContinuous(Data input) {
29             if (input instanceof DoubleData) {
30                 DoubleData d = (DoubleData) input;
31                 if (lastSampleNum != -1 && lastSampleNum != d.getFirstSampleNumber()) {
32                     return false;
33                 }
34     
35                 lastSampleNum = d.getFirstSampleNumber() + d.getValues().length;
36     
37             } else if (input instanceof DataStartSignal)
38                 lastSampleNum = -1;
39     
40             return true;
41         }
