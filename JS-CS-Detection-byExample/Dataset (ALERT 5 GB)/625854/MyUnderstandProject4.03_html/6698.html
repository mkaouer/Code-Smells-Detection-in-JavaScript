26     /** A parallel acoustic scorer that is capable of scoring multiple feature streams. */
27     public class ParallelAcousticScorer extends AbstractScorer {
28     
29     
30         @Override
31         public void newProperties(PropertySheet ps) throws PropertyException {
32             super.newProperties(ps);
33         }
34     
35     
36         /**
37          * Scores the given set of Tokens. All Tokens in the given list are assumed to belong to the same acoustic model.
38          *
39          * @param scoreableList a list containing StateToken objects to be scored
40          * @return the best scoring scorable, or null if there are no more frames to score
41          */
42         public Data calculateScores(List<? extends Scoreable> scoreableList) {
43             frontEnd = getFrontEnd(scoreableList);
44             return super.calculateScores(scoreableList);
45         }
46     
47     
48         protected Data doScoring(List<? extends Scoreable> scoreableList, Data data) {
49             float logMaxScore = -Float.MAX_VALUE;
50             Scoreable bestScoreable = null;
51     
52             for (Scoreable scoreable : scoreableList) {
53                 float logScore = scoreable.calculateScore(data);
54                 if (logScore > logMaxScore) {
55                     logMaxScore = logScore;
56                     bestScoreable = scoreable;
57                 }
58             }
59             return bestScoreable;
60         }
61     
62     
63         /**
64          * Returns the acoustic model name of the Tokens in the given list .
65          *
66          * @return the acoustic model name of the Tokens
67          */
68         private FrontEnd getFrontEnd(List activeList) {
69             if (activeList.size() > 0) {
70                 Iterator i = activeList.iterator();
71                 if (i.hasNext()) {
72                     ParallelToken token = (ParallelToken) i.next();
73                     return token.getFeatureStream().getFrontEnd();
74                 }
75             }
76             return null;
77         }
