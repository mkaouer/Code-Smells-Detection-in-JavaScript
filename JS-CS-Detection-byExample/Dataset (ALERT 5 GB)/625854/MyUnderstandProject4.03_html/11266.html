27     /**
28      * A test program for jsgf grammars. This program will generate a number of
29      * random sentences from a JSGF grammar and attempt to validate them
30      * via a rule parse.  A count of valid and invalid sentences is
31      * reported after the run.  
32      *
33      * This test will detect and reports only certain grammar failures.
34      * It will detect when a JSGFGrammar generates invalid sentences. It
35      * will not detect the case where a valid sentence can never be
36      * generated by the JSGFGrammar.
37      *
38      * Note that there is a bug in the Rule Grammar parse where rules with
39      * nested recursion cause a stack overflow error.
40      *
41      * Usage:  JSGFTest config.xml  count
42      */
43     public class JSGFTest {
44         /**
45          * Main method for running the HelloDigits demo.
46          */
47         public static void main(String[] args) {
48             int match = 0;
49             int noMatch = 0;
50             if (args.length != 2) {
51                 System.out.println("Usage: JSGFTest config.xml count");
52                 System.exit(0);
53             }
54     
55             try {
56                 URL url = new File(args[0]).toURI().toURL();
57                 int count = Integer.parseInt(args[1]);
58                 ConfigurationManager cm = new ConfigurationManager(url);
59     
60     	    JSGFGrammar jsgfGrammar = (JSGFGrammar) cm.lookup("jsgfGrammar");
61     
62                 jsgfGrammar.allocate();
63     
64                 RuleGrammar rg = jsgfGrammar.getRuleGrammar();
65                 for (int i = 0; i < count; i++) {
66                     String sentence = jsgfGrammar.getRandomSentence();
67                     // System.out.println("Sentence is : " + sentence);
68                     RuleParse rp = rg.parse(sentence, null);
69                     if (rp == null) {
70                         noMatch++;
71                         System.out.println("Fail: " + sentence);
72                     } else {
73                         match++;
74                     }
75                 }
76                 System.out.println("JSGFTest: Matches: " + match 
77                         + " Failures: " + noMatch);
78             } catch (IOException e) {
79                 System.err.println("Problem when loading JSGFTest: " + e);
80                 e.printStackTrace();
81             } catch (PropertyException e) {
82                 System.err.println("Problem configuring JSGFTest: " + e);
83                 e.printStackTrace();
84             } catch (GrammarException e) {
85                 System.err.println("Problem parsing result: " + e);
86                 e.printStackTrace();
87             }
88         }
89     }
