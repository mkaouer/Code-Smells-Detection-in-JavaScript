13     package edu.cmu.sphinx.tools.audio;
14     
15     import javax.swing.*;
16     import java.awt.*;
17     import java.awt.event.ActionEvent;
18     import java.awt.event.ActionListener;
19     import java.awt.event.WindowAdapter;
20     import java.awt.event.WindowEvent;
21     
22     /** Creates a dialog that prompts for a filename. */
23     public class FilenameDialog extends JDialog {
24     
25         String action;
26         JTextField filename;
27     
28     
29         /**
30          * Class constructor.
31          *
32          * @param parent the parent of this dialog
33          * @param modal  if true, this dialog box is modal
34          * @param title  the title for the login box
35          */
36         public FilenameDialog(Frame parent,
37                               boolean modal,
38                               String title) {
39             super(parent, modal);
40             setTitle(title);
41             addWindowListener(new WindowAdapter() {
42                 public void windowClosing(WindowEvent event) {
43                     setVisible(false);
44                 }
45             });
46             createFilenamePanel();
47             pack();
48         }
49     
50     
51         /** Creates the filename panel. */
52         void createFilenamePanel() {
53             Container contentPane = getContentPane();
54             GridBagLayout gridBag = new GridBagLayout();
55             GridBagConstraints constraints;
56             Insets insets;
57     
58             contentPane.setLayout(gridBag);
59             filename = new JTextField(12);
60             JLabel filenameLabel = new JLabel("Filename:");
61             filenameLabel.setLabelFor(filename);
62     
63             insets = new Insets(12, 12, 0, 0);  // top, left, bottom, right
64             constraints = new GridBagConstraints(
65                     0, 0, 1, 1,                     // x, y, width, height
66                     0.0, 0.0,                       // weightx, weighty
67                     GridBagConstraints.WEST,        // anchor
68                     GridBagConstraints.NONE,        // fill
69                     insets,                         // insets
70                     0, 0);                          // ipadx, ipady
71             gridBag.setConstraints(filenameLabel, constraints);
72             contentPane.add(filenameLabel);
73     
74             insets = new Insets(12, 7, 0, 12);  // top, left, bottom, right
75             constraints = new GridBagConstraints(
76                     1, 0, 1, 1,                     // x, y, width, height
77                     1.0, 1.0,                       // weightx, weighty
78                     GridBagConstraints.WEST,        // anchor
79                     GridBagConstraints.HORIZONTAL,  // fill
80                     insets,                         // insets
81                     0, 0);                          // ipadx, ipady
82             gridBag.setConstraints(filename, constraints);
83             contentPane.add(filename);
84     
85             /* BUTTON PANEL
86              */
87             JButton okButton = new JButton("Save");
88             okButton.addActionListener(new ActionListener() {
89                 public void actionPerformed(ActionEvent event) {
90                     setVisible(false);
91                 }
92             });
93     
94             insets = new Insets(0, 12, 12, 12); // top, left, bottom, right
95             constraints = new GridBagConstraints(
96                     0, 2, 2, 1,                     // x, y, width, height
97                     1.0, 1.0,                       // weightx, weighty
98                     GridBagConstraints.EAST,        // anchor
99                     GridBagConstraints.NONE,        // fill
100                     insets,                         // insets
101                     0, 0);                          // ipadx, ipady
102             gridBag.setConstraints(okButton, constraints);
103             contentPane.add(okButton);
104     
105             getRootPane().setDefaultButton(okButton);
106         }
107     
108     
109         /**
110          * Gets the user ID.
111          *
112          * @return the user ID
113          */
114         public String getFilename() {
115             return filename.getText();
116         }
117     
118     
119         /** Debug and example use. */
120         public static void main(String args[]) {
121             JFrame frame = new JFrame();
122             frame.setTitle("Debug");
123             frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
124             frame.pack();
125             frame.setVisible(false);
126     
127             FilenameDialog dialog = new FilenameDialog(frame, true, "Save as...");
128     
129             System.out.println("Showing dialog...");
130             dialog.setVisible(true);
131     
132             String filename = dialog.getFilename();
133             System.out.println("Filename: " + filename
134                     + " (length = " + filename.length() + ")");
135             System.exit(0);
136         }
137     }
