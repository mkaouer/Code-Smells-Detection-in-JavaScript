13     package edu.cmu.sphinx.util;
14     
15     /** Implements complex types and arythmetics */
16     public class Complex {
17     
18         /** The real part of a complex number. */
19         private double real;
20     
21         /** The imaginary part of a complex number. */
22         private double imaginary;
23     
24     
25         /** Create a default complex number */
26         public Complex() {
27             reset();
28         }
29     
30     
31         /** Create a complex number from a real one */
32         public Complex(double real) {
33             set(real, 0.0f);
34         }
35     
36     
37         /** Create a complex number from the real and imaginary parts */
38         public Complex(double real, double imaginary) {
39             set(real, imaginary);
40         }
41     
42     
43         /**
44          * Returns the real part of this Complex number.
45          *
46          * @return the real part
47          */
48         public double getReal() {
49             return real;
50         }
51     
52     
53         /**
54          * Returns the imaginary part of this Complex number.
55          *
56          * @return the imaginary part
57          */
58         public double getImaginary() {
59             return imaginary;
60         }
61     
62     
63         /** Sets both the real and imaginary parts of this complex number to zero. */
64         public void reset() {
65             this.real = 0.0f;
66             this.imaginary = 0.0f;
67         }
68     
69     
70         /**
71          * Sets the real and imaginary parts of this complex number.
72          *
73          * @param real      the value of the real part
74          * @param imaginary the value of the imaginary part
75          */
76         public void set(double real, double imaginary) {
77             this.real = real;
78             this.imaginary = imaginary;
79         }
80     
81     
82         /**
83          * Method to add two complex numbers.
84          *
85          * @param a the first element to be added
86          * @param b the second element to be added
87          */
88         public void addComplex(Complex a, Complex b) {
89             this.real = a.real + b.real;
90             this.imaginary = a.imaginary + b.imaginary;
91         }
92     
93     
94         /**
95          * Method to subtract two complex numbers.
96          *
97          * @param a the element we subtract from
98          * @param b the element to be subtracted
99          */
100         public void subtractComplex(Complex a, Complex b) {
101             this.real = a.real - b.real;
102             this.imaginary = a.imaginary - b.imaginary;
103         }
104     
105     
106         /**
107          * Method to multiply two complex numbers.
108          *
109          * @param a the first element to multiply
110          * @param b the second element to multiply
111          */
112         public void multiplyComplex(Complex a, Complex b) {
113             this.real = a.real * b.real - a.imaginary * b.imaginary;
114             this.imaginary = a.real * b.imaginary + a.imaginary * b.real;
115         }
116     
117     
118         /**
119          * Method to divide two complex numbers. To divide two complexes, we multiply by the complex conjugate of the
120          * denominator, thus resulting in a real number in the denominator.
121          *
122          * @param a the numerator
123          * @param b the denominator
124          */
125         public void divideComplex(Complex a, Complex b) {
126             this.real = a.real * b.real + a.imaginary * b.imaginary;
127             this.imaginary = a.imaginary * b.real - a.real * b.imaginary;
128             this.scaleComplex(this, b.squaredMagnitudeComplex());
129         }
130     
131     
132         /**
133          * Method to scale a complex number by a real one. The input complex number is modified in place.
134          *
135          * @param a the complex number
136          * @param b the real scaling factor
137          */
138         public void scaleComplex(Complex a, double b) {
139             this.real = a.real / b;
140             this.imaginary = a.imaginary / b;
141         }
142     
143     
144         /**
145          * Method to compute the squared magnitude of a complex number.
146          *
147          * @return the squared magnitude of the complex number
148          */
149         public double squaredMagnitudeComplex() {
150             double squaredMag;
151             squaredMag = this.real * this.real + this.imaginary * this.imaginary;
152             return squaredMag;
153         }
154     
155     
156         /** Returns this complex number as a string in the format (real, imaginary). */
157         public String toString() {
158             return ("(" + this.real + ", " + this.imaginary + ")");
159         }
160     }
