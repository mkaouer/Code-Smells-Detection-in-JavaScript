12     package edu.cmu.sphinx.demo.confidence;
13     
14     import edu.cmu.sphinx.frontend.util.Microphone;
15     import edu.cmu.sphinx.recognizer.Recognizer;
16     import edu.cmu.sphinx.result.*;
17     import edu.cmu.sphinx.util.props.ConfigurationManager;
18     import edu.cmu.sphinx.util.props.PropertyException;
19     
20     import java.io.File;
21     import java.io.IOException;
22     import java.net.URL;
23     import java.text.DecimalFormat;
24     
25     public class Confidence {
26     
27         private static DecimalFormat format = new DecimalFormat("#.#####");
28     
29     
30         /** Main method for running the Confidence demo. */
31         public static void main(String[] args) {
32             try {
33                 URL configURL;
34     
35                 if (args.length > 0) {
36                     configURL = new File(args[0]).toURI().toURL();
37                 } else {
38                     configURL = Confidence.class.getResource("config.xml");
39                 }
40     
41                 System.out.println("Loading Recognizer...\n");
42     
43                 ConfigurationManager cm = new ConfigurationManager(configURL);
44                 Recognizer recognizer = (Recognizer) cm.lookup("recognizer");
45     
46                 /* allocate the resource necessary for the recognizer */
47                 recognizer.allocate();
48     
49                 printInstructions();
50     
51                 Microphone microphone = (Microphone) cm.lookup("microphone");
52     
53                 if (microphone.startRecording()) {
54     
55                     while (true) {
56     
57                         System.out.println
58                                 ("Start speaking. Press Ctrl-C to quit.\n");
59     
60                         /* decode the audio file */
61                         Result result = recognizer.recognize();
62     
63                         if (result != null) {
64     
65                             ConfidenceScorer cs = (ConfidenceScorer) cm.lookup
66                                     ("confidenceScorer");
67                             ConfidenceResult cr = cs.score(result);
68                             Path best = cr.getBestHypothesis();
69     
70                             /* confidence of the best path */
71                             System.out.println(best.getTranscription());
72                             System.out.println
73                                     ("     (confidence: " +
74                                             format.format(best.getLogMath().logToLinear
75                                                     ((float) best.getConfidence()))
76                                             + ")");
77                             System.out.println();
78     
79                             /*
80                             * print out confidence of individual words
81                             * in the best path
82                             */
83                             WordResult[] words = best.getWords();
84                             for (int i = 0; i < words.length; i++) {
85                                 WordResult wr = words[i];
86                                 printWordConfidence(wr);
87                             }
88                             System.out.println();
89                         }
90                     }
91                 } else {
92                     System.out.println("Cannot start microphone.");
93                     recognizer.deallocate();
94                     System.exit(1);
95                 }
96             } catch (IOException e) {
97                 System.err.println("Problem when loading Confidence: " + e);
98                 e.printStackTrace();
99             } catch (PropertyException e) {
100                 System.err.println("Problem configuring Confidence: " + e);
101                 e.printStackTrace();
102             }
103         }
104     
105     
106         /** Prints out what to say for this demo. */
107         private static void printInstructions() {
108             System.out.println
109                     ("Sample sentences:\n" +
110                             "\n" +
111                             "the green one right in the middle\n" +
112                             "the purple one on the lower right side\n" +
113                             "the closest purple one on the far left side\n" +
114                             "the only one left on the left\n" +
115                             "\n" +
116                             "Refer to the file confidence.test for a complete list.\n");
117         }
118     
119     
120         /**
121          * Prints out the word and its confidence score.
122          *
123          * @param wr the WordResult to print
124          */
125         private static void printWordConfidence(WordResult wr) {
126             String word = wr.getPronunciation().getWord().getSpelling();
127     
128             System.out.print(word);
129     
130             /* pad spaces between the word and its score */
131             int entirePadLength = 10;
132             if (word.length() < entirePadLength) {
133                 for (int i = word.length(); i < entirePadLength; i++) {
134                     System.out.print(" ");
135                 }
136             }
137     
138             System.out.println
139                     (" (confidence: " +
140                             format.format
141                                     (wr.getLogMath().logToLinear((float) wr.getConfidence())) + ")");
142         }
143     }
