51         public static RuleParse[] mparse(String inputTokens[], Recognizer R, RuleGrammar G, String ruleName) {
52             RuleParser rp = new RuleParser();
53             rp.theRec = R;
54             String rNames[];
55             Rule startRule = null;
56             if (ruleName != null) {
57                 rNames = new String[1];
58                 rNames[0] = ruleName;
59             } else {
60                 rNames = G.listRuleNames();
61             }
62             Vector p = null;
63             int j = 0;
64             Vector t = new Vector();
65             for (j = 0; j < rNames.length; j++) {
66                 if ((ruleName == null) && !(G.isEnabled(rNames[j]))) {
67                     continue;
68                 }
69                 startRule = G.getRule(rNames[j]);
70                 if (startRule == null) {
71                     System.out.println("BAD RULENAME " + rNames[j]);
72                     continue;
73                 }
74                 p = rp.parse(G, startRule, inputTokens, 0);
75                 if ((p != null) && (p.size() != 0)) {
76                     for (int i = 0; i < p.size(); i++) {
77                         tokenPos tp = (tokenPos) p.elementAt(i);
78                         if (tp.getPos() == inputTokens.length) {
79                             RuleName rn = new RuleName(rNames[j]);
80                             t.addElement(new RuleParse(rn, (Rule) tp));
81                         }
82                     }
83                 }
84             }
85             if (t.size() == 0) {
86                 return null;
87             }
88             RuleParse rpa[] = new RuleParse[t.size()];
89             t.copyInto(rpa);
90             return rpa;
91         }

94         /*
95         * parse routine called recursively while traversing the Rule structure
96         * in a depth first manner. Returns a list of valid parses.
97         */
98         private Vector parse(RuleGrammar G, Rule r, String input[], int iPos) {
99     
100             //System.out.println("PARSE " + r.getClass().getName() + " " 
101             //+ iPos + " " + r);
102     
103             /*
104              * RULE REFERENCES
105              */
106             if (r instanceof RuleName) {
107                 RuleName rn = (RuleName) r;
108                 if (rn.getFullGrammarName() == null) {
109                     rn.setRuleName(G.getName() + "." + rn.getSimpleRuleName());
110                 }
111                 String simpleName = rn.getSimpleRuleName();
112                 if (simpleName.equals("VOID")) return null;
113                 if (simpleName.equals("NULL")) {
114                     Vector p = new Vector();
115                     jsgfRuleParse rp1 = new jsgfRuleParse(rn, RuleName.NULL);
116                     rp1.setPos(iPos);
117                     p.addElement(rp1);
118                     return p;
119                 }
120                 Rule ruleref = G.getRule(simpleName);
121                 if (rn.getFullGrammarName() != G.getName()) {
122                     ruleref = null;
123                 }
124                 if (ruleref == null) {
125                     String gname = rn.getFullGrammarName();
126                     //System.out.println("gname=" + gname);
127                     if ((gname != null) && (gname.length() > 0)) {
128                         RuleGrammar RG1 = theRec.getRuleGrammar(gname);
129                         if (RG1 != null) {
130                             //System.out.println("simpleName=" + simpleName);
131                             ruleref = RG1.getRule(simpleName);
132                             //System.out.println("ruleRef=" + ruleref);
133                             G = RG1;
134                         } else {
135                             System.out.println("ERROR: UNKNOWN GRAMMAR " + gname);
136                             //Thread.dumpStack();
137                         }
138                     }
139                     if (ruleref == null) {
140                         System.out.println("ERROR: UNKNOWN RULE NAME " + rn.getRuleName() + " " + rn);
141                         //Thread.dumpStack();
142                         return null;
143                     }
144                 }
145                 Vector p = parse(G, ruleref, input, iPos);
146                 if (p == null) {
147                     return null;
148                 }
149                 Vector t = new Vector();
150                 for (int j = 0; j < p.size(); j++) {
151                     tokenPos tp = (tokenPos) p.elementAt(j);
152                     if (tp instanceof emptyToken) {
153                         t.addElement(tp);
154                         continue;
155                     }
156                     Rule ar[] = new Rule[1];
157                     ar[0] = (Rule) p.elementAt(j);
158                     try {
159                         jsgfRuleParse rulep = new jsgfRuleParse(rn, ar[0]);
160                         rulep.setPos(tp.getPos());
161                         t.addElement(rulep);
162                     } catch (IllegalArgumentException e) {
163                         System.out.println("ERROR " + e);
164                     }
165                 }
166                 return t;
167             }
168     
169             /*
170              * LITERAL TOKENS
171              */
172             if (r instanceof RuleToken) {
173                 if (iPos >= input.length) {
174                     return null;
175                 }
176                 RuleToken rt = (RuleToken) r;
177                 //System.out.println(rt.getText() + " ?= " + input[iPos]);
178                 // TODO: what about case sensitivity ??????
179                 String tText = rt.getText().toLowerCase();
180                 if (tText.equals(input[iPos]) || (input[iPos].equals("%")) || (input[iPos].equals("*"))) {
181                     Vector v = new Vector();
182                     jsgfRuleToken tok = new jsgfRuleToken(rt.getText());
183                     tok.setPos(iPos + 1);
184                     v.addElement(tok);
185                     if (input[iPos].equals("*")) {
186                         jsgfRuleToken tok2 = new jsgfRuleToken(rt.getText());
187                         tok2.setPos(iPos);
188                         v.addElement(tok2);
189                     }
190                     return v;
191                 } else {
192                     if (tText.indexOf(' ') < 0) {
193                         return null;
194                     }
195                     if (!tText.startsWith(input[iPos])) {
196                         return null;
197                     }
198                     String ta[] = tokenize(tText);
199                     int j = 0;
200                     while (true) {
201                         if (j >= ta.length) {
202                             break;
203                         }
204                         if (iPos >= input.length) {
205                             return null;
206                         }
207                         if (!ta[j].equals(input[iPos])) {
208                             return null;
209                         }
210                         iPos++;
211                         j++;
212                     }
213                     Vector v = new Vector();
214                     jsgfRuleToken tok = new jsgfRuleToken(rt.getText());
215                     tok.setPos(iPos);
216                     v.addElement(tok);
217                     return v;
218                 }
219             }
220     
221             /*
222             * ALTERNATIVES
223             */
224             if (r instanceof RuleAlternatives) {
225                 RuleAlternatives ra = (RuleAlternatives) r;
226                 Rule rar[] = ra.getRules();
227                 Vector alts = new Vector();
228                 for (int i = 0; i < rar.length; i++) {
229                     Vector p = parse(G, rar[i], input, iPos);
230                     if (p == null) {
231                         continue;
232                     }
233                     for (int j = 0; j < p.size(); j++) {
234                         alts.addElement(p.elementAt(j));
235                     }
236                 }
237                 return alts;
238             }
239     
240             /*
241             * RULESEQUENCE
242             */
243             if (r instanceof RuleSequence) {
244                 RuleSequence rs = (RuleSequence) r;
245                 Rule rarry[] = rs.getRules();
246                 if ((rarry == null) || (rarry.length == 0)) {
247                     return null;
248                 }
249                 Vector p = parse(G, rarry[0], input, iPos);
250                 if (p == null) {
251                     return null;
252                 }
253                 Vector res = new Vector();
254                 //System.out.println("seq sz" + p.size());
255                 for (int j = 0; j < p.size(); j++) {
256                     //System.out.println("seq  " + p.elementAt(j));
257                     tokenPos tp = (tokenPos) p.elementAt(j);
258                     Rule rule0 = (Rule) p.elementAt(j);
259                     int nPos = tp.getPos();
260                     if (rarry.length == 1) {
261                         if (rule0 instanceof emptyToken) {
262                             res.addElement(rule0);
263                             continue;
264                         }
265                         jsgfRuleSequence rp = null;
266                         Rule ra[] = new Rule[1];
267                         ra[0] = rule0;
268                         try {
269                             rp = new jsgfRuleSequence(ra);
270                             rp.setPos(tp.getPos());
271                             res.addElement(rp);
272                         }
273                         catch (IllegalArgumentException e) {
274                             System.out.println(e);
275                         }
276                         continue;
277                     }
278                     Rule nra[] = new Rule[rarry.length - 1];
279                     System.arraycopy(rarry, 1, nra, 0, nra.length);
280                     RuleSequence nrs = new RuleSequence(nra);
281                     //System.out.println("2parse " + nPos + nrs);
282                     Vector q = parse(G, nrs, input, nPos);
283                     if (q == null) {
284                         continue;
285                     }
286                     //System.out.println("2 seq sz " + p.size());
287                     for (int k = 0; k < q.size(); k++) {
288                         //System.out.println("2 seq  " + q.elementAt(k));
289                         Rule r1 = (Rule) q.elementAt(k);
290                         Rule ra[] = null;
291                         tokenPos tp1 = (tokenPos) q.elementAt(k);
292                         //System.out.println("rule0 " + rule0);
293                         //System.out.println("r1 " + r1);
294                         if (r1 instanceof emptyToken) {
295                             res.addElement(rule0);
296                             continue;
297                         }
298                         if (rule0 instanceof emptyToken) {
299                             res.addElement(r1);
300                             continue;
301                         }
302                         if (r1 instanceof RuleSequence) {
303                             RuleSequence r2 = (RuleSequence) r1;
304                             Rule r2r[] = r2.getRules();
305                             ra = new Rule[r2r.length + 1];
306                             ra[0] = rule0;
307                             System.arraycopy(r2r, 0, ra, 1, r2r.length);
308                         } else {
309                             ra = new Rule[2];
310                             ra[0] = rule0;
311                             ra[1] = r1;
312                         }
313                         jsgfRuleSequence rp = null;
314                         try {
315                             rp = new jsgfRuleSequence(ra);
316                             rp.setPos(tp1.getPos());
317                             res.addElement(rp);
318                         } catch (IllegalArgumentException e) {
319                             System.out.println(e);
320                         }
321                     }
322                 }
323                 return res;
324             }
325     
326             /*
327             * TAGS
328             */
329             if (r instanceof RuleTag) {
330                 RuleTag rtag = (RuleTag) r;
331                 String theTag = rtag.getTag();
332                 //System.out.println("tag="+theTag);
333                 Vector p = parse(G, rtag.getRule(), input, iPos);
334                 if (p == null) {
335                     return null;
336                 }
337                 Vector t = new Vector();
338                 for (int j = 0; j < p.size(); j++) {
339                     tokenPos tp = (tokenPos) p.elementAt(j);
340                     if (tp instanceof emptyToken) {
341                         t.addElement(tp);
342                         continue;
343                     }
344                     ;
345                     jsgfRuleTag tag = new jsgfRuleTag((Rule) tp, theTag);
346                     tag.setPos(tp.getPos());
347                     t.addElement(tag);
348                 }
349                 return t;
350             }
351     
352             //
353             // RULECOUNT (e.g. [], *, or + )
354             //
355             if (r instanceof RuleCount) {
356                 RuleCount rc = (RuleCount) r;
357                 int rcount = rc.getCount();
358                 emptyToken empty = new emptyToken();
359                 empty.setPos(iPos);
360                 Vector p = parse(G, rc.getRule(), input, iPos);
361                 if (p == null) {
362                     if (rcount == RuleCount.ONCE_OR_MORE) return null;
363                     Vector v = new Vector();
364                     v.addElement(empty);
365                     return v;
366                 }
367                 if (rcount != RuleCount.ONCE_OR_MORE) {
368                     p.addElement(empty);
369                 }
370                 if (rcount == RuleCount.OPTIONAL) {
371                     return p;
372                 }
373                 for (int m = 2; m <= input.length - iPos; m++) {
374                     Rule ar[] = new Rule[m];
375                     for (int n = 0; n < m; n++) {
376                         ar[n] = rc.getRule();
377                     }
378                     RuleSequence rs1 = new RuleSequence(ar);
379                     Vector q = parse(G, rs1, input, iPos);
380                     if (q == null) {
381                         return p;
382                     }
383                     for (int z = 0; z < q.size(); z++) {
384                         p.addElement(q.elementAt(z));
385                     }
386                 }
387                 return p;
388             }
389     
390             System.out.println("ERROR UNKNOWN OBJECT " + r);
391             return null;
392         }

395         /*
396         * tokenize a string
397         */
398         static String[] tokenize(String text) {
399             StringTokenizer st = new StringTokenizer(text);
400             int size = st.countTokens();
401             String res[] = new String[size];
402             int i = 0;
403             while (st.hasMoreTokens()) {
404                 res[i++] = st.nextToken().toLowerCase();
405             }
406             return res;
407         }
