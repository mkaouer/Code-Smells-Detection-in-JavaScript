13     package edu.cmu.sphinx.jsapi;
14     
15     import javax.speech.EngineCentral;
16     import javax.speech.EngineList;
17     import javax.speech.EngineModeDesc;
18     
19     
20     /**
21      * Provides a list of SphinxRecognizerModeDesc objects that define the available operating modes of the sphinx
22      * recognition engine.  An instance of this SphinxEngineCentral object is registered with the javax.speeech.Central
23      * class.  When requested by the Central class, this object provides a list of EngineModeDesc objects that describes the
24      * available operating modes of the engine.
25      */
26     public class SphinxEngineCentral implements EngineCentral {
27     
28         static private SphinxRecognizerModeDesc sphinxModeDesc
29                 = new SphinxRecognizerModeDesc();
30     
31     
32         /**
33          * Create an EngineList containin and EngineModeDesc object for each mode of operation of the Sphinx speech engine.
34          *
35          * @param require describes the constraints to be placed on engines placed on the engine list. null matches all
36          *                engines. Note that require is guaranteed to be of type RecognizerModeDesc.
37          * @return a list of RecognizerModeDesc objects describing the available sphinx recognition engines that match
38          *         <code> require </code>. Returns <code> null </code> if no engines match.
39          */
40         public EngineList createEngineList(EngineModeDesc require) {
41             if (require == null || sphinxModeDesc.match(require)) {
42                 EngineList el = new EngineList();
43                 el.addElement(sphinxModeDesc);
44                 return el;
45             }
46             return null;
47         }
48     }
