44         protected GrammarNode createGrammar() {
45             // TODO: FlatLinguist requires the initial grammar node
46             // to contain a single silence. We'll do that for now,
47             // but once the FlatLinguist is fixed, this should be
48             // returned to its former method of creating an empty
49             // initial grammar node
50             //          initialNode = createGrammarNode(initialID, false);
51     
52             initialNode = null;
53             finalNode = createGrammarNode(true);
54             try {
55                 LineNumberReader in = new LineNumberReader(new FileReader(refFile));
56                 String line;
57                 while (!(line = in.readLine()).equals("")) {
58                     int i = line.indexOf('(');
59                     String uttName = line.substring(i + 1, line.indexOf(')'));
60                     String transcript = line.substring(0, i).trim();
61                     if (!transcript.equals("")) {
62                         initialNode = createGrammarNode(Dictionary.SILENCE_SPELLING);
63                         createForcedAlignerGrammar(initialNode, finalNode, transcript);
64                         grammars.put(uttName, initialNode);
65                         currentUttName = uttName;
66                     }
67                 }
68             } catch (FileNotFoundException e) {
69                 throw new Error(e);
70             } catch (IOException e) {
71                 throw new Error(e);
72             } catch (NoSuchMethodException e) {
73                 throw new Error(e);
74             }
75             return initialNode;
76         }
77     
78     
79         public GrammarNode getInitialNode() {
80             return initialNode;
81         }

83     
84         public void setUtterance(String utteranceName) {
