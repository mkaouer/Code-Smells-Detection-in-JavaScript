13     package edu.cmu.sphinx.result;
14     
15     import edu.cmu.sphinx.util.LogMath;
16     
17     
18     /**
19      * Represents a path of words through the recognition result lattice.
20      * <p/>
21      * All scores are maintained in the logMath log domain
22      */
23     public interface Path {
24     
25         /**
26          * Gets the total score for this path. Scores are in the LogMath log domain
27          *
28          * @return the score for the path in the LogMath log domaain.
29          */
30         public double getScore();
31     
32     
33         /**
34          * Returns a log confidence score for this path. Use the getLogMath().logToLinear() method to convert the log
35          * confidence score to linear. The linear value should be between 0.0 and 1.0 (inclusive).
36          *
37          * @return a log confidence score which linear value is between 0.0 and 1.0 (inclusive)
38          */
39         public double getConfidence();
40     
41     
42         /**
43          * Returns the log math for the scores of this path.
44          *
45          * @return the log math
46          */
47         public LogMath getLogMath();
48     
49     
50         /**
51          * Gets the ordered set of words for this path
52          *
53          * @return an array containing zero or more words
54          */
55         public WordResult[] getWords();
56     
57     
58         /**
59          * Gets the transcription of the path.
60          *
61          * @return the transcription of the path.
62          */
63         public String getTranscription();
64     
65     
66         /** Returns a string representation of this object */
67         public String toString();
68     }
