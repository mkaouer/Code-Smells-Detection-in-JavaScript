82         /**
83          * Blocks the calling thread until this <code>Engine</code> is in a specified state.
84          * <p/>
85          * <p>All state bits specified in the <code>state</code> parameter must be set in order for the method to return, as
86          * defined for the <code>testEngineState</code> method.  If the <code>state</code> parameter defines an unreachable
87          * state (e.g. <code>PAUSED | RESUMED</code>) an exception is thrown.
88          * <p/>
89          * <p>The <code>waitEngineState</code> method can be called successfully in any <code>Engine</code> state.
90          *
91          * @param state a bitmask of the state to wait for
92          * @throws InterruptedException     if another thread has interrupted this thread.
93          * @throws IllegalArgumentException if the specified state is unreachable
94          * @see #testEngineState
95          * @see #getEngineState
96          */
97         public void waitEngineState(long state)
98                 throws InterruptedException, IllegalArgumentException {
99             synchronized (engineStateLock) {
100                 while (!testEngineState(state))
101                     engineStateLock.wait();
102             }
103         }

106         /**
107          * Returns <code>true</code> if this state of this <code>Engine</code> matches the specified state.
108          * <p/>
109          * <p>The test performed is not an exact match to the current state.  Only the specified states are tested.  For
110          * example the following returns true only if the <code>Synthesizer</code> queue is empty, irrespective of the
111          * pause/resume and allocation states.
112          * <p/>
113          * <PRE> if (synth.testEngineState(Synthesizer.QUEUE_EMPTY)) ... </PRE>
114          * <p/>
115          * <p>The <code>testEngineState</code> method is equivalent to:
116          * <p/>
117          * <PRE> if ((engine.getEngineState() & state) == state) </PRE>
118          * <p/>
119          * <p>The <code>testEngineState</code> method can be called successfully in any <code>Engine</code> state.
120          *
121          * @param state a bitmask of the states to test for
122          * @return <code>true</code> if this <code>Engine</code> matches <code>state</code>; otherwise <code>false</code>
123          * @throws IllegalArgumentException if the specified state is unreachable
124          */
125         public boolean testEngineState(long state)
126                 throws IllegalArgumentException {
127             return ((getEngineState() & state) == state);
128         }

131         /**
132          * Updates this <code>Engine</code> state by clearing defined bits, then setting other specified bits.
133          *
134          * @return a length-2 array with old and new state values.
135          */
136         protected long[] setEngineState(long clear, long set) {
137             long states[] = new long[2];
138             synchronized (engineStateLock) {
139                 states[0] = engineState;
140                 engineState = engineState & (~clear);
141                 engineState = engineState | set;
142                 states[1] = engineState;
143                 engineStateLock.notifyAll();
144             }
145             return states;
146         }

149         /**
150          * Allocates the resources required for this <code>Engine</code> and puts it into the <code>ALLOCATED</code> state.
151          * When this method returns successfully the <code>ALLOCATED</code> bit of this <code>Engine</code> state is set,
152          * and the <code>testEngineState(Engine.ALLOCATED)</code> method returns <code>true</code>.
153          * <p/>
154          * <p>During the processing of the method, this <code>Engine</code> is temporarily in the
155          * <code>ALLOCATING_RESOURCES</code> state.
156          *
157          * @throws EngineException  if this <code>Engine</code> cannot be allocated
158          * @throws EngineStateError if this <code>Engine</code> is in the <code>DEALLOCATING_RESOURCES</code> state
159          * @see #deallocate
160          */
161         public void allocate() throws EngineException, EngineStateError {
162             if (testEngineState(ALLOCATED)) {
163                 return;
164             }
165     
166             long[] states = setEngineState(CLEAR_ALL_STATE, ALLOCATING_RESOURCES);
167             postEngineAllocatingResources(states[0], states[1]);
168     
169             handleAllocate();
170         }

183          * Frees the resources of this <code>Engine</code> that were acquired during allocation and during operation and
184          * return this <code>Engine</code> to the <code>DEALLOCATED</code>.  When this method returns the
185          * <code>DEALLOCATED</code> bit of this <code>Engine</code> state is set so the
186          * <code>testEngineState(Engine.DEALLOCATED)</code> method returns <code>true</code>.
187          * <p/>

189          * <code>DEALLOCATING_RESOURCES</code> state.
190          * <p/>
191          * <p>A deallocated engine can be re-started with a subsequent call to <code>allocate</code>.
192          *

194          * @throws EngineStateError if this <code>Engine</code> is in the <code>ALLOCATING_RESOURCES</code> state

604         /**
605          * Convenience method that throws an <code>EngineStateError</code> if any of the bits in the passed state are set in
606          * the <code>state</code>.
607          *
608          * @param state the <code>Engine</code> state to check
609          * @throws EngineStateError if any of the bits in the passed state are set in the <code>state</code>
610          */
611         protected void checkEngineState(long state) throws EngineStateError {
612             long currentState = getEngineState();
613             if ((currentState & state) != 0) {
614                 throw new EngineStateError
615                         ("Invalid EngineState: expected=("
616                                 + stateToString(state) + ") current state=("
617                                 + stateToString(currentState) + ")");
618             }
619         }

622         /**
623          * Returns a <code>String</code> of the names of all the <code>Engine</code> states in the given <code>Engine</code>
624          * state.
625          *
626          * @param state the bitmask of states
627          * @return a <code>String</code> containing the names of all the states set in <code>state</code>
628          */
629         protected String stateToString(long state) {
630             StringBuffer buf = new StringBuffer();
631             if ((state & Engine.DEALLOCATED) != 0)
632                 buf.append(" DEALLOCATED ");
633             if ((state & Engine.ALLOCATING_RESOURCES) != 0)
634                 buf.append(" ALLOCATING_RESOURCES ");
635             if ((state & Engine.ALLOCATED) != 0)
636                 buf.append(" ALLOCATED ");
637             if ((state & Engine.DEALLOCATING_RESOURCES) != 0)
638                 buf.append(" DEALLOCATING_RESOURCES ");
639             if ((state & Engine.PAUSED) != 0)
640                 buf.append(" PAUSED ");
641             if ((state & Engine.RESUMED) != 0)
642                 buf.append(" RESUMED ");
643             return buf.toString();
644         }

647         /**
648          * Dispatches a <code>SpeechEvent</code>. The dispatcher should notify all <code>EngineListeners</code> from this
649          * method.  The <code>SpeechEvent</code> was added via the various post methods of this class.
650          *
651          * @param event the <code>SpeechEvent</code> to dispatch
652          * @see #postEngineAllocatingResources
653          * @see #postEngineAllocated
654          * @see #postEngineDeallocatingResources
655          * @see #postEngineDeallocated
656          * @see #postEnginePaused
657          * @see #postEngineResumed
658          */
659         public void dispatchSpeechEvent(SpeechEvent event) {
660             switch (event.getId()) {
661                 case EngineEvent.ENGINE_ALLOCATED:
662                     fireEngineAllocated((EngineEvent) event);
663                     break;
664                 case EngineEvent.ENGINE_ALLOCATING_RESOURCES:
665                     fireEngineAllocatingResources((EngineEvent) event);
666                     break;

669                     break;

671                     fireEngineDeallocatingResources((EngineEvent) event);

678                     break;

680                     fireEngineResumed((EngineEvent) event);
681                     break;
682             }
