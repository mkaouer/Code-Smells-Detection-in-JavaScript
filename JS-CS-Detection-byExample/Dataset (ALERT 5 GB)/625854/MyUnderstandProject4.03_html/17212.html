70         /**
71          * Converts a byte array into a short array. Since a byte is 8-bits, and a short is 16-bits, the returned short
72          * array will be half in length than the byte array. If the length of the byte array is odd, the length of the short
73          * array will be <code>(byteArray.length - 1)/2</code>, i.e., the last byte is discarded.
74          *
75          * @param byteArray a byte array
76          * @param offset    which byte to start from
77          * @param length    how many bytes to convert
78          * @return a short array, or <code>null</code> if byteArray is of zero length
79          * @throws java.lang.ArrayIndexOutOfBoundsException
80          *
81          */
82         public static short[] byteToShortArray
83                 (byte[] byteArray, int offset, int length)
84                 throws ArrayIndexOutOfBoundsException {
85     
86             if (0 < length && (offset + length) <= byteArray.length) {
87                 int shortLength = length / 2;
88                 short[] shortArray = new short[shortLength];
89                 int temp;
90                 for (int i = offset, j = 0; j < shortLength;
91                      j++, temp = 0x00000000) {
92                     temp = byteArray[i++] << 8;
93                     temp |= 0x000000FF & byteArray[i++];
94                     shortArray[j] = (short) temp;
95                 }
96                 return shortArray;
97             } else {
98                 throw new ArrayIndexOutOfBoundsException
99                         ("offset: " + offset + ", length: " + length
100                                 + ", array length: " + byteArray.length);
101             }
102         }

105         /**
106          * Converts a big-endian byte array into an array of doubles. Each consecutive bytes in the byte array are converted
107          * into a double, and becomes the next element in the double array. The size of the returned array is
108          * (length/bytesPerValue). Currently, only 1 byte (8-bit) or 2 bytes (16-bit) samples are supported.
109          *
110          * @param byteArray     a byte array
111          * @param offset        which byte to start from
112          * @param length        how many bytes to convert
113          * @param bytesPerValue the number of bytes per value
114          * @param signedData    whether the data is signed
115          * @return a double array, or <code>null</code> if byteArray is of zero length
116          * @throws java.lang.ArrayIndexOutOfBoundsException
117          *
118          */
119         public static final double[] bytesToValues(byte[] byteArray,
120                                                    int offset,
121                                                    int length,
122                                                    int bytesPerValue,
123                                                    boolean signedData)
124                 throws ArrayIndexOutOfBoundsException {
125     
126             if (0 < length && (offset + length) <= byteArray.length) {
127                 assert (length % bytesPerValue == 0);
128                 double[] doubleArray = new double[length / bytesPerValue];
129     
130                 int i = offset;
131     
132                 for (int j = 0; j < doubleArray.length; j++) {
133                     int val = (int) byteArray[i++];
134                     if (!signedData) {
135                         val &= 0xff; // remove the sign extension
136                     }
137                     for (int c = 1; c < bytesPerValue; c++) {
138                         int temp = (int) byteArray[i++] & 0xff;
139                         val = (val << 8) + temp;
140                     }
141     
142                     doubleArray[j] = (double) val;
143                 }
144     
145                 return doubleArray;
146             } else {
147                 throw new ArrayIndexOutOfBoundsException
148                         ("offset: " + offset + ", length: " + length
149                                 + ", array length: " + byteArray.length);
150             }
151         }

154         /**
155          * Converts a little-endian byte array into an array of doubles. Each consecutive bytes of a float are converted
156          * into a double, and becomes the next element in the double array. The number of bytes in the double is specified
157          * as an argument. The size of the returned array is (data.length/bytesPerValue).
158          *
159          * @param data          a byte array
160          * @param offset        which byte to start from
161          * @param length        how many bytes to convert
162          * @param bytesPerValue the number of bytes per value
163          * @param signedData    whether the data is signed
164          * @return a double array, or <code>null</code> if byteArray is of zero length
165          * @throws java.lang.ArrayIndexOutOfBoundsException
166          *
167          */
168         public static final double[] littleEndianBytesToValues(byte[] data,
169                                                                int offset,
170                                                                int length,
171                                                                int bytesPerValue,
172                                                                boolean signedData)
173                 throws ArrayIndexOutOfBoundsException {
174     
175             if (0 < length && (offset + length) <= data.length) {
176                 assert (length % bytesPerValue == 0);
177                 double[] doubleArray = new double[length / bytesPerValue];
178     
179                 int i = offset + bytesPerValue - 1;
180     
181                 for (int j = 0; j < doubleArray.length; j++) {
182                     int val = (int) data[i--];
183                     if (!signedData) {
184                         val &= 0xff; // remove the sign extension
185                     }
186                     for (int c = 1; c < bytesPerValue; c++) {
187                         int temp = (int) data[i--] & 0xff;
188                         val = (val << 8) + temp;
189                     }
190     
191                     // advance 'i' to the last byte of the next value
192                     i += (bytesPerValue * 2);
193     
194                     doubleArray[j] = (double) val;
195                 }
196     
197                 return doubleArray;
198     
199             } else {
200                 throw new ArrayIndexOutOfBoundsException
201                         ("offset: " + offset + ", length: " + length
202                                 + ", array length: " + data.length);
203             }
204         }

208          * Convert the two bytes starting at the given offset to a short.
209          *
210          * @param byteArray the byte array
211          * @param offset    where to start
212          * @return a short
213          * @throws java.lang.ArrayIndexOutOfBoundsException
214          *
215          */
216         public static short bytesToShort(byte[] byteArray, int offset)
217                 throws ArrayIndexOutOfBoundsException {
218             short result = (short)
219                     ((byteArray[offset++] << 8) |
220                             (0x000000FF & byteArray[offset]));
221             return result;

259         /**
260          * Returns the given double array as a string. The dump will be in the form:
261          * <pre>data.length data[0] data[1] ... data[data.length-1]</pre>where
262          * <code>data[i]</code> is formatted by the method <code>Util.formatDouble(data[i], 10, 5)</code>.
263          *
264          * @param data   the double array to dump
265          * @param format either HEXADECIMAL, SCIENTIFIC or DECIMAL
266          * @return a string representation of the double array
267          */
268         private static String doubleArrayToString(double[] data, int format) {
269             String dump = String.valueOf(data.length);
270     
271             for (int i = 0; i < data.length; i++) {
272                 if (format == DECIMAL) {
273                     dump += (" " + formatDouble
274                             (data[i], decimalIntegerDigits,
275                                     decimalFractionDigits));
276                 } else if (format == HEXADECIMAL) {
277                     long binary = Double.doubleToRawLongBits(data[i]);
278                     dump += (" 0x" + Long.toHexString(binary));
279                 } else if (format == SCIENTIFIC) {
280                     dump += (" " + Utilities.doubleToScientificString
281                             (data[i], doubleScientificFractionDigits));
282                 }
283             }
284             return dump;
285         }

334         /**
335          * Returns a formatted string of the given number, with the given numbers of digit space for the integer and
336          * fraction parts. If the integer part has less than <code>integerDigits</code> digits, spaces will be prepended to
337          * it. If the fraction part has less than <code>fractionDigits</code>, spaces will be appended to it. Therefore,
338          * <code>formatDouble(12345.6789, 6, 6)</code> will give
339          * the string <pre>" 12345.6789  "</pre> (one space before 1, two spaces
340          * after 9).
341          *
342          * @param number         the number to format
343          * @param integerDigits  the length of the integer part
344          * @param fractionDigits the length of the fraction part
345          * @return a formatted number
346          */
347         public static String formatDouble(double number, int integerDigits,
348                                           int fractionDigits) {
349             String formatter = "0.";
350             for (int i = 0; i < fractionDigits; i++) {
351                 formatter += "0";
352             }
353     
354             format.applyPattern(formatter);
355             String formatted = format.format(number);
356     
357             // pad preceding spaces before the number
358             int dotIndex = formatted.indexOf('.');
359             if (dotIndex == -1) {
360                 formatted += ".";
361                 dotIndex = formatted.length() - 1;
362             }
363             String result = "";
364             for (int i = dotIndex; i < integerDigits; i++) {
365                 result += " ";
366             }
367             result += formatted;
368             return result;
369         }

425         /**
426          * Returns a native audio format that has the same encoding, endianness and sample size as the given format, and a
427          * sample rate that is greater than or equal to the given sample rate.
428          *
429          * @param format the desired format
430          * @param mixer  if non-null, use this Mixer; otherwise use AudioSystem
431          * @return a suitable native audio format
432          */
433         public static AudioFormat getNativeAudioFormat(AudioFormat format,
434                                                        Mixer mixer) {
435             Line.Info[] lineInfos;
436     
437             if (mixer != null) {
438                 lineInfos = mixer.getTargetLineInfo
439                         (new Line.Info(TargetDataLine.class));
440             } else {
441                 lineInfos = AudioSystem.getTargetLineInfo
442                         (new Line.Info(TargetDataLine.class));
443             }
444     
445             AudioFormat nativeFormat = null;
446     
447             // find a usable target line
448             for (int i = 0; i < lineInfos.length; i++) {
449     
450                 AudioFormat[] formats =
451                         ((TargetDataLine.Info) lineInfos[i]).getFormats();
452     
453                 for (int j = 0; j < formats.length; j++) {
454     
455                     // for now, just accept downsampling, not checking frame
456                     // size/rate (encoding assumed to be PCM)
457     
458                     AudioFormat thisFormat = formats[j];
459                     if (thisFormat.getEncoding() == format.getEncoding()
460                             && thisFormat.isBigEndian() == format.isBigEndian()
461                             && thisFormat.getSampleSizeInBits() ==
462                             format.getSampleSizeInBits()
463                             && thisFormat.getSampleRate() >= format.getSampleRate()) {
464                         nativeFormat = thisFormat;
465                         break;
466                     }
467                 }
468                 if (nativeFormat != null) {
469                     //no need to look through remaining lineinfos
470                     break;
471                 }
472             }
473             return nativeFormat;
474         }
