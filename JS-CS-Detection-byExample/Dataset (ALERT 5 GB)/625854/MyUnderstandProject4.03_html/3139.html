21     /**
22      * Filters an input power spectrum through a PLP filterbank. The filters in the filterbank are placed in the frequency
23      * axis so as to mimic the critical band, representing different perceptual effect at different frequency bands. The
24      * filter outputs are also scaled for equal loudness preemphasis. The filter shapes are defined by the {@link PLPFilter}
25      * class. Like the {@link MelFrequencyFilterBank}, this filter bank has characteristics defined by the {@link
26      * #PROP_NUMBER_FILTERS number of filters}, the {@link #PROP_MIN_FREQ minimum frequency}, and the {@link #PROP_MAX_FREQ
27      * maximum frequency}. Unlike the {@link MelFrequencyFilterBank}, the minimum and maximum frequencies here refer to the
28      * <b>center</b> frequencies of the filters located at the leftmost and rightmost positions, and not to the edges.
29      * Therefore, this filter bank spans a frequency range that goes beyond the limits suggested by the minimum and maximum
30      * frequencies.
31      *
32      * @author <a href="mailto:rsingh@cs.cmu.edu">rsingh</a>
33      * @version 1.0
34      * @see PLPFilter
35      */
36     @SuppressWarnings({"UnnecessaryLocalVariable"})
37     public class PLPFrequencyFilterBank extends BaseDataProcessor {
38     
39         /** The name of the Sphinx Property for the number of filters in the filterbank. */
40         @S4Integer(defaultValue = 32)
41         public static final String PROP_NUMBER_FILTERS = "numberFilters";
42     
43         /** The name of the Sphinx Property for the center frequency of the lowest filter in the filterbank. */
44         @S4Integer(defaultValue = 130)
45         public static final String PROP_MIN_FREQ = "minimumFrequency";
46     
47         /** The name of the Sphinx Property for the center frequency of the highest filter in the filterbank. */
48         @S4Integer(defaultValue = 3600)
49         public static final String PROP_MAX_FREQ = "maximumFrequency";

51         private int sampleRate;
52         private int numberFftPoints;
53         private int numberFilters;
54         private double minFreq;
55         private double maxFreq;
56         private PLPFilter[] criticalBandFilter;
57         private double[] equalLoudnessScaling;

61         * (non-Javadoc)

79         /**
80          * Build a PLP filterbank with the parameters given. The center frequencies of the PLP filters will be uniformly
81          * spaced between the minimum and maximum analysis frequencies on the Bark scale. on the Bark scale.
82          *
83          * @throws IllegalArgumentException
84          */
85         private void buildCriticalBandFilterbank() throws IllegalArgumentException {
86             double minBarkFreq;
87             double maxBarkFreq;
88             double deltaBarkFreq;
89             double nyquistFreq;
90             double centerFreq;
91             int numberDFTPoints = (numberFftPoints >> 1) + 1;
92             double[] DFTFrequencies;
93     
94             /* This is the same class of warper called by PLPFilter.java */
95             FrequencyWarper bark = new FrequencyWarper();
96     
97             this.criticalBandFilter = new PLPFilter[numberFilters];
98     
99             if (numberFftPoints == 0) {
100                 throw new IllegalArgumentException("Number of FFT points is zero");
101             }
102             if (numberFilters < 1) {
103                 throw new IllegalArgumentException("Number of filters illegal: "
104                         + numberFilters);
105             }
106     
107             DFTFrequencies = new double[numberDFTPoints];
108             nyquistFreq = sampleRate / 2;
109             for (int i = 0; i < numberDFTPoints; i++) {
110                 DFTFrequencies[i] = (double) i * nyquistFreq /
111                         (double) (numberDFTPoints - 1);
112             }
113     
114             /**
115              * Find center frequencies of filters in the Bark scale
116              * translate to linear frequency and create PLP filters
117              * with these center frequencies.
118              *
119              * Note that minFreq and maxFreq specify the CENTER FREQUENCIES
120              * of the lowest and highest PLP filters
121              */
122     
123     
124             minBarkFreq = bark.hertzToBark(minFreq);
125             maxBarkFreq = bark.hertzToBark(maxFreq);
126     
127             if (numberFilters < 1) {
128                 throw new IllegalArgumentException("Number of filters illegal: "
129                         + numberFilters);
130             }
131             deltaBarkFreq = (maxBarkFreq - minBarkFreq) / (numberFilters + 1);
132     
133             for (int i = 0; i < numberFilters; i++) {
134                 centerFreq = bark.barkToHertz(minBarkFreq + i * deltaBarkFreq);
135                 criticalBandFilter[i] = new PLPFilter(DFTFrequencies, centerFreq);
136             }
137         }
138     
139     
140         /**
141          * This function return the equal loudness preemphasis factor at any frequency. The preemphasis function is given
142          * by
143          * <p/>
144          * E(w) = (w^2+56.8e6)*w^4/((w^2+6.3e6)^2(w^2+0.38e9)(w^6+9.58e26))
145          * <p/>
146          * where w is frequency in radians/second
147          */
148         private double loudnessScalingFunction(double freq) {
149             double freqsquared = freq * freq;
150             double freqfourth = freqsquared * freqsquared;
151             double freqsixth = freqfourth * freqsquared;
152     
153             double numerator = (freqsquared + 56.8e6) * freqfourth;
154             double denominator = Math.pow((freqsquared + 6.3e6), 2) *
155                     (freqsquared + 0.38e9) *
156                     (freqsixth + 9.58e26);
157     
158             return numerator / denominator;
159         }
160     
161     
162         /** Create an array of equal loudness preemphasis scaling terms for all the filters */
163         private void buildEqualLoudnessScalingFactors() {
164             double centerFreq;
165     
166             equalLoudnessScaling = new double[numberFilters];
167             for (int i = 0; i < numberFilters; i++) {
168                 centerFreq = criticalBandFilter[i].centerFreqInHz * 2.0 * Math.PI;
169                 equalLoudnessScaling[i] = loudnessScalingFunction(centerFreq);
170             }
171         }

174         /**
175          * Process data, creating the power spectrum from an input audio frame.
176          *
177          * @param input input power spectrum
178          * @return PLP power spectrum
179          * @throws java.lang.IllegalArgumentException
180          *
181          */
182         private DoubleData process(DoubleData input) throws
183                 IllegalArgumentException {
184     
185             double[] in = input.getValues();
186     
187             if (criticalBandFilter == null ||
188                     sampleRate != input.getSampleRate()) {
189                 numberFftPoints = (in.length - 1) * 2;
190                 sampleRate = input.getSampleRate();
191                 buildCriticalBandFilterbank();
192                 buildEqualLoudnessScalingFactors();
193     
194             } else if (in.length != ((numberFftPoints >> 1) + 1)) {
195                 throw new IllegalArgumentException
196                         ("Window size is incorrect: in.length == " + in.length +
197                                 ", numberFftPoints == " + ((numberFftPoints >> 1) + 1));
198             }
199     
200             double[] outputPLPSpectralArray = new double[numberFilters];
201     
202             /**
203              * Filter input power spectrum
204              */
205             for (int i = 0; i < numberFilters; i++) {
206                 // First compute critical band filter output
207                 outputPLPSpectralArray[i] = criticalBandFilter[i].filterOutput(in);
208                 // Then scale it for equal loudness preemphasis
209                 outputPLPSpectralArray[i] *= equalLoudnessScaling[i];
210             }
211     
212             DoubleData output = new DoubleData
213                     (outputPLPSpectralArray, input.getSampleRate(),
214                             input.getCollectTime(), input.getFirstSampleNumber());
215     
216             return output;
217         }

220         /**
221          * Reads the next Data object, which is the power spectrum of an audio input frame. However, it can also be other
222          * Data objects like a Signal, which is returned unmodified.
223          *
224          * @return the next available Data object, returns null if no Data object is available
225          * @throws DataProcessingException if there is a data processing error
226          */
227         public Data getData() throws DataProcessingException {
228     
229             Data input = getPredecessor().getData();
230     
231             getTimer().start();
232     
233             if (input != null) {
234                 if (input instanceof DoubleData) {
235                     input = process((DoubleData) input);
236                 }
237             }
238     
239             getTimer().stop();
240     
241             return input;
242         }
