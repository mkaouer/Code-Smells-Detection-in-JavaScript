241         /**
242          * Performs the recognition for the given number of frames.
243          *
244          * @param nFrames the number of frames to recognize
245          * @return the current result
246          */
247         public Result recognize(int nFrames) {
248             boolean done = false;
249             Result result = null;
250             streamEnd = false;
251     
252             try {
253     
254                 for (int i = 0; i < nFrames && !done; i++) {
255                     // System.out.println("Frame " + currentFrameNumber);
256                     // score the emitting list
257     
258                     // tokenTracker.startFrame();
259                     activeList = activeListManager.getEmittingList();
260                     if (activeList != null) {
261                         do {
262                             currentFrameNumber++;
263                             done = !scoreTokens();
264                         } while (!done
265                                 && (growSkipInterval > 1 &&
266                                 ((currentFrameNumber % growSkipInterval) == 0)));
267                         if (!done) {
268                             bestTokenMap = createBestTokenMap();
269                             // prune and grow the emitting list
270                             pruneBranches();
271     
272                             resultList = new LinkedList<Token>();
273                             growEmittingBranches();
274                             // prune and grow the non-emitting lists
275                             // activeListManager.dump();
276                             growNonEmittingLists();
277                         }
278                     }
279                     // tokenTracker.stopFrame();
280                 }
281             } catch (OutOfMemoryError e) {
282                 done = true;
283                 activeList = null;
284                 System.gc();
285                 System.out.println("OutOfMemoryError: Aborting recognition");
286             }
287             
288             if (!streamEnd) {
289             	result = new Result(loserManager, activeList, resultList,
290             					    currentFrameNumber, done, logMath);
291             }
292     
293             // tokenTypeTracker.show();
294             if (showTokenCount) {
295                 showTokenCount();
296             }
297             return result;
298         }

358         /**
359          * Goes through the active list of tokens and expands each token, finding the set of successor tokens until all the
360          * successor tokens are emitting tokens.
361          */
362         protected void growBranches() {
363             growTimer.start();
364             float relativeBeamThreshold = activeList.getBeamThreshold();
365             if (logger.isLoggable(Level.FINE)) {
366                 logger.fine("Frame: " + currentFrameNumber
367                         + " thresh : " + relativeBeamThreshold + " bs "
368                         + activeList.getBestScore() + " tok "
369                         + activeList.getBestToken());
370             }
371             for (Token token : activeList.getTokens()) {
372                 if (token.getScore() >= relativeBeamThreshold && skewPrune(token)) {
373                     collectSuccessorTokens(token);
374                 }
375             }
376             growTimer.stop();
377     
378             // activeListManager.dump();
379         }

382         /**
383          * Grows the emitting branches. This version applies a simple acoustic lookahead based upon the rate of change in
384          * the current acoustic score.
385          */
386         protected void growEmittingBranches() {
387             if (acousticLookaheadFrames > 0F) {
388                 growTimer.start();
389                 float bestScore = -Float.MAX_VALUE;
390                 for (Token t : activeList.getTokens()) {
391                     float score = t.getScore() + t.getAcousticScore()
392                             * acousticLookaheadFrames;
393                     if (score > bestScore) {
394                         bestScore = score;
395                     }
396                     t.setWorkingScore(score);
397                 }
398                 float relativeBeamThreshold = bestScore + relativeBeamWidth;
399     
400                 for (Token t : activeList.getTokens()) {
401                     if (t.getWorkingScore() >= relativeBeamThreshold) {
402                         collectSuccessorTokens(t);
403                     }
404                 }
405                 growTimer.stop();
406             } else {
407                 growBranches();
408             }
409         }

412         /** Grow the non-emitting ActiveLists, until the tokens reach an emitting state. */
413         private void growNonEmittingLists() {
414             for (Iterator<ActiveList> i = activeListManager.getNonEmittingListIterator(); i
415                     .hasNext();) {
416                 activeList = i.next();
417                 if (activeList != null) {
418                     i.remove();
419                     pruneBranches();
420                     growBranches();
421                 }
422             }

427          * Calculate the acoustic scores for the active list. The active list should contain only emitting tokens.
428          *
429          * @return <code>true</code> if there are more frames to score, otherwise, false
430          */
431         protected boolean scoreTokens() {
432             boolean moreTokens;
433             scoreTimer.start();
434             Data data = scorer.calculateScores(activeList.getTokens());
435             scoreTimer.stop();
436     
437             Token bestToken = null;
438             if (data instanceof Token) {
439                 bestToken = (Token)data;
440             } else if (data == null) {
441                 streamEnd = true;
442             }
443      
444             moreTokens = (bestToken != null);
445             activeList.setBestToken(bestToken);
446     
447             // monitorWords(activeList);
448             monitorStates(activeList);
449     
450             // System.out.println("BEST " + bestToken);
451     
452             curTokensScored.value += activeList.size();
453             totalTokensScored.value += activeList.size();
454     
455             return moreTokens;
456

499         /**
500          * Gets the best token for this state
501          *
502          * @param state the state of interest
503          * @return the best token
504          */
505         protected Token getBestToken(SearchState state) {
506             Object key = getStateKey(state);
507     
508             if (!wantTokenStacks) {
509                 return (Token) bestTokenMap.get(key);
510             } else {
511                 // new way... if the heap for this state isn't full return
512                 // null, otherwise return the worst scoring token
513                 TokenHeap th = (TokenHeap) bestTokenMap.get(key);
514                 Token t;
515     
516                 if (th == null) {
517                     return null;
518                 } else if ((t = th.get(state)) != null) {
519                     return t;
520                 } else if (!th.isFull()) {
521                     return null;
522                 } else {
523                     return th.getSmallest();
524                 }
525             }
526         }

531          *

535         protected void setBestToken(Token token, SearchState state) {

550         /**
551          * Find the best token to use as a predecessor token given a candidate predecessor. The predecessor is the most
552          * recent word token unless keepAllTokens is set, in which case, the predecessor is always the candidate
553          * predecessor.
554          *
555          * @param token the token of interest
556          * @return the immediate successor word token
557          */
558         protected Token getWordPredecessor(Token token) {
559             if (keepAllTokens) {
560                 return token;
561             } else {
562                 float logAcousticScore = 0.0f;
563                 float logLanguageScore = 0.0f;
564     
565                 while (token != null && !token.isWord()) {
566                     logAcousticScore += token.getAcousticScore();
567                     logLanguageScore += token.getLanguageScore();
568                     token = token.getPredecessor();
569                 }
570                 if (buildWordLattice) {
571                     return new Token(logAcousticScore, logLanguageScore, token);
572                 } else {
573                     return token;
574                 }
575             }
576         }

580          * Returns the state key for the given state. If we are using token stacks then the key will be related to the
581          * search state and the word history (depending on the type of token stacks), otherwise, the key will simply be the
582          * state itself. Currently we are not using token stacks so the state is the key.
583          *
584          * @param state the state to get the key for
585          * @return the key for the given state
586          */
587         private Object getStateKey(SearchState state) {
588             if (!wantTokenStacks) {
589                 return state;
590             } else {
591                 if (state.isEmitting()) {
592                     return new SinglePathThroughHMMKey(((HMMSearchState) state));
593                     // return ((HMMSearchState) state).getHMMState().getHMM();
594                 } else {
595                     return state;
596                 }
597             }

601         /** Checks that the given two states are in legitimate order. */
602         private void checkStateOrder(SearchState fromState, SearchState toState) {
603             if (fromState.getOrder() == numStateOrder - 1) {
604                 return;
605             }
606     
607             if (fromState.getOrder() > toState.getOrder()) {
608                 throw new Error("IllegalState order: from "
609                         + fromState.getClass().getName() + " "
610                         + fromState.toPrettyString()
611                         + " order: " + fromState.getOrder()
612                         + " to "
613                         + toState.getClass().getName() + " "
614                         + toState.toPrettyString()
615                         + " order: " + toState.getOrder());
616             }
617         }

620         /**
621          * Collects the next set of emitting tokens from a token and accumulates them in the active or result lists
622          *
623          * @param token the token to collect successors from be immediately expanded are placed. Null if we should always
624          *              expand all nodes.
625          */
626         protected void collectSuccessorTokens(Token token) {
627     
628             // tokenTracker.add(token);
629             // tokenTypeTracker.add(token);
630     
631             // If this is a final state, add it to the final list
632     
633             if (token.isFinal()) {
634                 resultList.add(getWordPredecessor(token));
635                 return;
636             }
637     
638             // if this is a non-emitting token and we've already 
639             // visited the same state during this frame, then we
640             // are in a grammar loop, so we don't continue to expand.
641             // This check only works properly if we have kept all of the
642             // tokens (instead of skipping the non-word tokens).
643             // Note that certain linguists will never generate grammar loops
644             // (lextree linguist for example). For these cases, it is perfectly
645             // fine to disable this check by setting keepAllTokens to false
646     
647             if (!token.isEmitting() && (keepAllTokens && isVisited(token))) {
648                 return;
649             }
650     
651             SearchState state = token.getSearchState();
652             SearchStateArc[] arcs = state.getSuccessors();
653             Token predecessor = getWordPredecessor(token);
654     
655             // For each successor
656             // calculate the entry score for the token based upon the
657             // predecessor token score and the transition probabilities
658             // if the score is better than the best score encountered for
659             // the SearchState and frame then create a new token, add
660             // it to the lattice and the SearchState.
661             // If the token is an emitting token add it to the list,
662             // othewise recursively collect the new tokens successors.
663     
664             for (SearchStateArc arc : arcs) {
665                 SearchState nextState = arc.getState();
666     
667                 if (checkStateOrder) {
668                     checkStateOrder(state, nextState);
669                 }
670     
671                 // We're actually multiplying the variables, but since
672                 // these come in log(), multiply gets converted to add
673                 float logEntryScore = token.getScore() + arc.getProbability();
674     
675                 Token bestToken = getBestToken(nextState);
676                 boolean firstToken = bestToken == null;
677     
678                 if (firstToken || bestToken.getScore() < logEntryScore) {
679                     Token newBestToken = new Token(predecessor, nextState,
680                             logEntryScore, arc.getLanguageProbability(), arc
681                                     .getInsertionProbability(), currentFrameNumber);
682                     tokensCreated.value++;
683     
684                     setBestToken(newBestToken, nextState);
685                     if (firstToken) {
686                         activeListAdd(newBestToken);
687                     } else {
688     //                    System.out.println("Replacing " + bestToken + " with " + newBestToken);
689                         activeListReplace(bestToken, newBestToken);
690                         if (buildWordLattice && newBestToken.isWord()) {
691     
692                             // Move predecessors of bestToken to precede
693                             // newBestToken, bestToken will be garbage collected.
694                             loserManager.changeSuccessor(newBestToken, bestToken);
695                             loserManager.addAlternatePredecessor(newBestToken,
696                                     bestToken.getPredecessor());
697                         }
698                     }
699                 } else {
700                     if (buildWordLattice && nextState instanceof WordSearchState) {
701                         if (predecessor != null) {
702                             loserManager.addAlternatePredecessor(bestToken,

1007     /**
1008      * A quick and dirty token heap that allows us to perform token stack experiments. It is not very efficient. We will
1009      * likely replace this with something better once we figure out how we want to prune things.
1010      */
1011     
1012     class TokenHeap {
1013     
1014         Token[] tokens;
1015         int curSize = 0;
1016     
1017     
1018         /**
1019          * Creates a token heap with the maximum size
1020          *
1021          * @param maxSize the maximum size of the heap
1022          */
1023         TokenHeap(int maxSize) {
1024             tokens = new Token[maxSize];
1025         }
1026     
1027     
1028         /**
1029          * Adds a token to the heap
1030          *
1031          * @param token the token to add
1032          */
1033         void add(Token token) {
1034             // first, if an identical state exists, replace
1035             // it.
1036     
1037             if (!tryReplace(token)) {
1038                 if (curSize < tokens.length) {
1039                     tokens[curSize++] = token;
1040                 } else if (token.getScore() > tokens[curSize - 1].getScore()) {
1041                     tokens[curSize - 1] = token;
1042                 }
1043             }
1044             fixupInsert();
1045         }
1046     
1047     
1048         /**
1049          * Returns the smallest scoring token on the heap
1050          *
1051          * @return the smallest scoring token
1052          */
1053         Token getSmallest() {
1054             if (curSize == 0) {
1055                 return null;
1056             } else {
1057                 return tokens[curSize - 1];
1058             }
1059         }
1060     
1061     
1062         /**
1063          * Determines if the heap is ful
1064          *
1065          * @return <code>true</code> if the heap is full
1066          */
1067         boolean isFull() {
1068             return curSize == tokens.length;
1069         }
1070     
1071     
1072         /**
1073          * Checks to see if there is already a token t on the heap that has the same search state. If so, this token
1074          * replaces that one
1075          *
1076          * @param t the token to try to add to the heap
1077          * @return <code>true</code> if the token was added
1078          */
1079         private boolean tryReplace(Token t) {
1080             for (int i = 0; i < curSize; i++) {
1081                 if (t.getSearchState().equals(tokens[i].getSearchState())) {
1082                     assert t.getScore() > tokens[i].getScore();
1083                     tokens[i] = t;
1084                     return true;
1085                 }
1086             }
1087             return false;
1088         }
1089     
1090     
1091         /** Orders the heap after an insert */
1092         private void fixupInsert() {
1093             Arrays.sort(tokens, 0, curSize - 1, Token.COMPARATOR);
1094         }
1095     
1096         /**
1097          * returns a token on the heap that matches the given search state
1098          *
1099          * @param s the search state
1100          * @return the token that matches, or null
1101          */
1102         Token get(SearchState s) {
1103             for (int i = 0; i < curSize; i++) {
1104                 if (tokens[i].getSearchState().equals(s)) {
1105                     return tokens[i];
1106                 }
1107             }
1108             return null;
1109         }
