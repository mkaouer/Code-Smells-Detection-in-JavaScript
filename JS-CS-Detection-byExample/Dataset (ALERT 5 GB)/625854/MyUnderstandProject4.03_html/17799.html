10     package edu.cmu.sphinx.frontend.feature;
11     
12     import edu.cmu.sphinx.frontend.*;
13     import edu.cmu.sphinx.linguist.acoustic.tiedstate.*;
14     import edu.cmu.sphinx.util.props.*;
15     
16     /**
17      * Implements an LDA transformation. The dimension of the feature stream is
18      * reduced with a matrix transform thus reducing speed and increasing accuracy.
19      * This component requires special model trained with LDA/MLLT transform.
20      */
21     public class LDA extends BaseDataProcessor {
22     
23     	/** The name of the transform matrix file */
24     	@S4Component(type = Loader.class)
25     	public final static String PROP_LOADER = "loader";
26     
27     	float[][] transformMatrix;
28     	protected Loader loader;
29     	
30     	int rows;
31     	int values;
32     
33     	/*
34     	 * (non-Javadoc)
35     	 * 
36     	 * @see
37     	 * edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util
38     	 * .props.PropertySheet)
39     	 */
40     	public void newProperties(PropertySheet ps) throws PropertyException {
41     		super.newProperties(ps);
42     		
43     		loader = (Loader)ps.getComponent(PROP_LOADER);
44     		
45     		try {
46     			loader.load();
47     		} catch (Exception e) {
48     			e.printStackTrace();
49     		}
50     		transformMatrix = loader.getTransformMatrix();
51     		
52     		if (transformMatrix == null)
53     			throw new RuntimeException("Model doesn't include transformation matrix");
54     	                
55     		rows = transformMatrix.length;
56     		values = transformMatrix[0].length;
57     	}
58     
59     	/**
60     	 * Returns the next Data object being processed by this LDA, or if it is a
61     	 * Signal, it is returned without modification.
62     	 * 
63     	 * @return the next available Data object, returns null if no Data object is
64     	 *         available
65     	 * @throws DataProcessingException
66     	 *             if there is a processing error
67     	 * @see Data
68     	 */
69     	public Data getData() throws DataProcessingException {
70     		Data input = getPredecessor().getData();
71     		Data output;
72     		getTimer().start();
73     		if (input != null && input instanceof FloatData) {
74     			FloatData inputData = (FloatData) input;
75     			float[] in = inputData.getValues();
76     			float[] out = new float[rows];
77     			
78     			assert in.length == values; 
79     
80     			for (int i = 0; i < rows; i++) {
81     				out [i] = 0;
82     				float transform_matrix_i[] = transformMatrix[i];
83     				for (int j = 0; j < values; j++) {
84     					out[i] += in[j] * transform_matrix_i[j];
85     				}
86     			}
87     			output = new FloatData(out, inputData.getSampleRate(), inputData
88     					.getCollectTime(), inputData.getFirstSampleNumber());
89     		} else {
90     			output = input;
91     		}
92     		getTimer().stop();
93     		return output;
94     	}
95     }
