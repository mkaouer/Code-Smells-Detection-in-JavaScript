25     /**
26      * Computes the Discrete Fourier Transform (FT) of an input sequence, using Fast Fourier Transform (FFT). Fourier
27      * Transform is the process of analyzing a signal into its frequency components. In speech, rather than analyzing the
28      * signal over its entrie duration, we analyze one <b>window</b> of audio data. This window is the product of applying a
29      * sliding Hamming window to the signal. Moreover, since the amplitude is a lot more important than the phase for speech
30      * recognition, this class returns the power spectrum of that window of data instead of the complex spectrum. Each value
31      * in the returned spectrum represents the strength of that particular frequency for that window of data.
32      * <p/>
33      * By default, the number of FFT points is the closest power pf 2 that is equal to or larger than the number of samples
34      * in the incoming window of data. The FFT points can also be set by the user with the property defined by {@link
35      * #PROP_NUMBER_FFT_POINTS}. The length of the returned power spectrum is the number of FFT points, divided by 2, plus
36      * 1. Since the input signal is real, the FFT is symmetric, and the information contained in the whole vector is already
37      * present in its first half.
38      * <p/>
39      * Note that each call to {@link #getData() getData} only returns the spectrum of one window of data. To display the
40      * spectrogram of the entire original audio, one has to collect all the spectra from all the windows generated from the
41      * original data. A spectrogram is a two dimensional representation of three dimensional information. The horizontal
42      * axis represents time. The vertical axis represents the frequency. If we slice the spectrogram at a given time, we get
43      * the spectrum computed as the short term Fourier transform of the signal windowed around that time stamp. The
44      * intensity of the spectrum for each time frame is given by the color in the graph, or by the darkness in a gray scale
45      * plot. The spectrogram can be thought of as a view from the top of a surface generated by concatenating the spectral
46      * vectors obtained from the windowed signal.
47      * <p/>
48      * For example, Figure 1 below shows the audio signal of the utterance "one three nine oh", and Figure 2 shows its
49      * spectrogram, produced by putting together all the spectra returned by this FFT. Frequency is on the vertical axis,
50      * and time is on the horizontal axis. The darkness of the shade represents the strength of that frequency at that point
51      * in time: <p> <br><img src="doc-files/139o.jpg"> <br><b>Figure 1: The audio signal of the utterance "one three nine
52      * oh".</b> <p> <br><img src="doc-files/139ospectrum.jpg"> <br><b>Figure 2: The spectrogram of the utterance "one three
53      * nine oh" in Figure 1.</b>
54      */

108         /** Initialize all the data structures necessary for computing FFT. */
109         private void initializeFFT() {
110             /**
111              * Number of points in the FFT. By default, the value is 512,
112              * which means that we compute 512 values around a circle in the
113              * complex plane. Complex conjugate pairs will yield the same
114              * power, therefore the power produced by indices 256 through
115              * 511 are symmetrical with the ones between 1 and 254. Therefore,
116              * we need only return values between 0 and 255.
117              */
118             computeLogBase2(numberFftPoints);
119             createWeightFft(numberFftPoints, invert);
120             initComplexArrays();
121             weightFftTimesFrom2 = new Complex();
122             tempComplex = new Complex();
123         }
124

126         /** Initialize all the Complex arrays that will be necessary for FFT. */
127         private void initComplexArrays() {
128     
129             inputFrame = new Complex[numberFftPoints];
130             from = new Complex[numberFftPoints];
131             to = new Complex[numberFftPoints];
132     
133             for (int i = 0; i < numberFftPoints; i++) {
134                 inputFrame[i] = new Complex();
135                 from[i] = new Complex();
136                 to[i] = new Complex();
137             }
138         }

141         /**
142          * Process data, creating the power spectrum from an input frame.
143          *
144          * @param input the input frame
145          * @return a DoubleData that is the power spectrum of the input frame
146          * @throws java.lang.IllegalArgumentException
147          *
148          */
149         private DoubleData process(DoubleData input)
150                 throws IllegalArgumentException {
151     
152             /**
153              * Create complex input sequence equivalent to the real
154              * input sequence.
155              * If the number of points is less than the window size,
156              * we incur in aliasing. If it's greater, we pad the input
157              * sequence with zeros.
158              */
159             double[] in = input.getValues();
160     
161             if (numberFftPoints < in.length) {
162                 int i = 0;
163                 for (; i < numberFftPoints; i++) {
164                     inputFrame[i].set(in[i], 0.0f);
165                 }
166                 for (; i < in.length; i++) {
167                     tempComplex.set(in[i], 0.0f);
168                     inputFrame[i % numberFftPoints].addComplex
169                             (inputFrame[i % numberFftPoints], tempComplex);
170                 }
171             } else {
172                 int i = 0;
173                 for (; i < in.length; i++) {
174                     inputFrame[i].set(in[i], 0.0f);
175                 }
176                 for (; i < numberFftPoints; i++) {
177                     inputFrame[i].reset();
178                 }
179             }
180     
181             /**
182              * Create output sequence.
183              */
184             double[] outputSpectrum = new double[(numberFftPoints >> 1) + 1];
185     
186             /**
187              * Start Fast Fourier Transform recursion
188              */
189             recurseFft(inputFrame, outputSpectrum, numberFftPoints, invert);
190     
191             /**
192              * Return the power spectrum
193              */
194             DoubleData output = new DoubleData
195                     (outputSpectrum, input.getSampleRate(),
196                             input.getCollectTime(),
197                             input.getFirstSampleNumber());
198     
199             return output;
200         }

203         /**
204          * Make sure the number of points in the FFT is a power of 2 by computing its log base 2 and checking for
205          * remainders.
206          *
207          * @param numberFftPoints number of points in the FFT
208          * @throws java.lang.IllegalArgumentException
209          *
210          */
211         private void computeLogBase2(int numberFftPoints)
212                 throws IllegalArgumentException {
213             this.logBase2NumberFftPoints = 0;
214             for (int k = numberFftPoints; k > 1;
215                  k >>= 1, this.logBase2NumberFftPoints++) {
216                 if (((k % 2) != 0) || (numberFftPoints < 0)) {
217                     throw new IllegalArgumentException("Not a power of 2: "
218                             + numberFftPoints);
219                 }
220             }
221         }

224         /**
225          * Initializes the <b>weightFft[]</b> vector. <p><b>weightFft[k] = w ^ k</b></p> where: <p><b>w = exp(-2 * PI * i /
226          * N)</b></p> <p><b>i</b> is a complex number such that <b>i * i = -1</b> and <b>N</b> is the number of points in
227          * the FFT. Since <b>w</b> is complex, this is the same as</p> <p><b>Re(weightFft[k]) = cos ( -2 * PI * k /
228          * N)</b></p> <p><b>Im(weightFft[k]) = sin ( -2 * PI * k / N)</b></p>
229          *
230          * @param numberFftPoints number of points in the FFT
231          * @param invert          whether it's direct (false) or inverse (true) FFT
232          */
233         private void createWeightFft(int numberFftPoints, boolean invert) {
234             /**
235              * weightFFT will have numberFftPoints/2 complex elements.
236              */
237             weightFft = new Complex[numberFftPoints >> 1];
238     
239             /**
240              * For the inverse FFT,
241              * w = 2 * PI / numberFftPoints;
242              */
243             double w = -2 * Math.PI / numberFftPoints;
244             if (invert) {
245                 w = -w;
246             }
247     
248             for (int k = 0; k < (numberFftPoints >> 1); k++) {
249                 weightFft[k] = new Complex(Math.cos(w * k), Math.sin(w * k));
250             }
251         }

254         /**
255          * Reads the next DoubleData object, which is a data frame from which we'll compute the power spectrum. Signal
256          * objects just pass through unmodified.
257          *
258          * @return the next available power spectrum DoubleData object, or null if no Spectrum object is available
259          * @throws DataProcessingException if there is a processing error
260          */
261         public Data getData() throws DataProcessingException {
262     
263             Data input = getPredecessor().getData();
264     
265             getTimer().start();
266     
267             if ((input != null) && (input instanceof DoubleData)) {
268                 DoubleData data = (DoubleData) input;
269                 if (!isNumberFftPointsSet) {
270                     /*
271                      * If numberFftPoints is not set by the user,
272                      * figure out the numberFftPoints and initialize the
273                      * data structures appropriately.
274                      */
275                     if (numberDataPoints != data.getValues().length) {
276                         numberDataPoints = data.getValues().length;
277                         numberFftPoints = getNumberFftPoints(numberDataPoints);
278                         initializeFFT();
279                     }
280                 } else {
281                     /*
282                      * Warn if the user-set numberFftPoints is not ideal.
283                      */
284                     if (numberDataPoints != data.getValues().length) {
285                         numberDataPoints = data.getValues().length;
286                         int idealFftPoints = getNumberFftPoints(numberDataPoints);
287                         if (idealFftPoints != numberFftPoints) {
288                             logger.warning("User set numberFftPoints (" +
289                                     numberFftPoints + ") is not ideal (" +
290                                     idealFftPoints + ")");
291                         }
292                     }
293                 }
294                 input = process(data);
295             }
296     
297             // At this point - or in the call immediatelly preceding
298             // this -, we should have created a cepstrum frame with
299             // whatever data came last, even if we had less than
300             // window size of data.
301     
302             getTimer().stop();
303     
304             return input;
305         }

308         /**
309          * Returns the ideal number of FFT points given the number of samples. The ideal number of FFT points is the closest
310          * power of 2 that is equal to or larger than the number of samples in the incoming window.
311          *
312          * @param numberSamples the number of samples in the incoming window
313          * @return the closest power of 2 that is equal to or larger than the number of samples in the incoming window
314          */
315         private final static int getNumberFftPoints(int numberSamples) {
316             int fftPoints = 1;
317     
318             while (fftPoints < numberSamples) {
319                 fftPoints *= 2;
320                 if (fftPoints < 1 || fftPoints > Integer.MAX_VALUE) {
321                     throw new Error("Invalid # of FFT points: " + fftPoints);
322                 }
323             }
324             return fftPoints;
325         }

328         /**
329          * Establish the recursion. The FFT computation will be computed by as a recursion. Each stage in the butterfly will
330          * be fully computed during recursion. In fact, we use the mechanism of recursion only because it's the simplest way
331          * of switching the "input" and "output" vectors. The output of a stage is the input to the next stage. The
332          * butterfly computes elements in place, but we still need to switch the vectors. We could copy it (not very
333          * efficient...) or, in C, switch the pointers. We can avoid the pointers by using recursion.
334          *
335          * @param input           input sequence
336          * @param output          output sequence
337          * @param numberFftPoints number of points in the FFT
338          * @param invert          whether it's direct (false) or inverse (true) FFT
339          */
340         private void recurseFft(Complex[] input,
341                                 double[] output,
342                                 int numberFftPoints,
343                                 boolean invert) {
344     
345             double divisor;
346     
347             /**
348              * The direct and inverse FFT are essentially the same
349              * algorithm, except for two difference: a scaling factor of
350              * "numberFftPoints" and the signal of the exponent in the
351              * weightFft vectors, defined in the method
352              * <code>createWeightFft</code>.
353              */
354     
355             if (!invert) {
356                 divisor = 1.0;
357             } else {
358                 divisor = (double) numberFftPoints;
359             }
360     
361             /**
362              * Initialize the "from" and "to" variables.
363              */
364             for (int i = 0; i < numberFftPoints; i++) {
365                 to[i].reset();
366                 from[i].scaleComplex(input[i], divisor);
367             }
368     
369             /**
370              * Repeat the recursion log2(numberFftPoints) times,
371              * i.e., we have log2(numberFftPoints) butterfly stages.
372              */
373             butterflyStage(from, to, numberFftPoints, numberFftPoints >> 1);
374     
375             /**
376              * Compute energy ("float") for each frequency point
377              * from the fft ("complex")
378              */
379             if ((this.logBase2NumberFftPoints & 1) == 0) {
380                 for (int i = 0; i <= (numberFftPoints >> 1); i++) {
381                     output[i] = from[i].squaredMagnitudeComplex();
382                 }
383             } else {
384                 for (int i = 0; i <= (numberFftPoints >> 1); i++) {
385                     output[i] = to[i].squaredMagnitudeComplex();
386                 }
387             }
388             return;
389         }

392         /**
393          * Compute one stage in the FFT butterfly. The name "butterfly" appears because this method computes elements in
394          * pairs, and a flowgraph of the computation (output "0" comes from input "0" and "1" and output "1" comes from
395          * input "0" and "1") resembles a butterfly.
396          * <p/>
397          * We repeat <code>butterflyStage</code> for <b>log_2(numberFftPoints)</b> stages, by calling the recursion with the
398          * argument <code>currentDistance</code> divided by 2 at each call, and checking if it's still > 0.
399          *
400          * @param from            the input sequence at each stage
401          * @param to              the output sequence
402          * @param numberFftPoints the total number of points
403          * @param currentDistance the "distance" between elements in the butterfly
404          */
405         private void butterflyStage(Complex[] from,
406                                     Complex[] to,
407                                     int numberFftPoints,
408                                     int currentDistance) {
409             int ndx1From;
410             int ndx2From;
411             int ndx1To;
412             int ndx2To;
413             int ndxWeightFft;
414     
415             if (currentDistance > 0) {
416     
417                 int twiceCurrentDistance = 2 * currentDistance;
418     
419                 for (int s = 0; s < currentDistance; s++) {
420                     ndx1From = s;
421                     ndx2From = s + currentDistance;
422                     ndx1To = s;
423                     ndx2To = s + (numberFftPoints >> 1);
424                     ndxWeightFft = 0;
425                     while (ndxWeightFft < (numberFftPoints >> 1)) {
426                         /**
427                          * <b>weightFftTimesFrom2 = weightFft[k]   </b>
428                          * <b>                      *from[ndx2From]</b>
429                          */
430                         weightFftTimesFrom2.multiplyComplex
431                                 (weightFft[ndxWeightFft], from[ndx2From]);
432                         /**
433                          * <b>to[ndx1To] = from[ndx1From]       </b>
434                          * <b>             + weightFftTimesFrom2</b>
435                          */
436                         to[ndx1To].addComplex
437                                 (from[ndx1From], weightFftTimesFrom2);
438                         /**
439                          * <b>to[ndx2To] = from[ndx1From]       </b>
440                          * <b>             - weightFftTimesFrom2</b>
441                          */
442                         to[ndx2To].subtractComplex
443                                 (from[ndx1From], weightFftTimesFrom2);
444                         ndx1From += twiceCurrentDistance;
445                         ndx2From += twiceCurrentDistance;
446                         ndx1To += currentDistance;
447                         ndx2To += currentDistance;
448                         ndxWeightFft += currentDistance;
449                     }
450                 }
451     
452                 /**
453                  * This call'd better be the last call in this block, so when
454                  * it returns we go straight into the return line below.
455                  *
456                  * We switch the <i>to</i> and <i>from</i> variables,
457                  * the total number of points remains the same, and
458                  * the <i>currentDistance</i> is divided by 2.
459                  */
460                 butterflyStage(to, from, numberFftPoints, (currentDistance >> 1));
461             }
462             return;
463         }
