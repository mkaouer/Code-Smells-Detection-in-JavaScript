39         /** a token comparator that is used to order tokens in descending order */
40         public final static Comparator<Token> COMPARATOR = new Comparator<Token>() {
41             public int compare(Token t1, Token t2) {
42                 if (t1.getScore() > t2.getScore()) {
43                     return -1;
44                 } else if (t1.getScore() == t2.getScore()) {
45                     return 0;
46                 } else {
47                     return 1;
48                 }
49             }

91         /**
92          * Constructs a new token that continues the search from the current token. If predessorClasses is null or if the
93          * class of the state is a member of predecessorClasses, the predecessor of the new token is set to the current
94          * token.  Otherwise it is set to the predecessor of the current token.  This behavior is used to save memory when
95          * building lattices.
96          *
97          * @param state                   the SentenceHMMState associated with this token
98          * @param logTotalScore           the total entry score for this token (in LogMath log base)
99          * @param logLanguageScore        the language score associated with this token (in LogMath log base)
100          * @param logInsertionProbability the insertion probabilty  associated with this token (in LogMath log base)
101          * @param frameNumber             the frame number associated with this token
102          */
103         public Token child(SearchState state,
104                            float logTotalScore,
105                            float logLanguageScore,
106                            float logInsertionProbability,
107                            int frameNumber) {
108             if ((predecessorClasses == null) ||
109                     predecessorClasses.contains(state.getClass())) {
110                 return new Token(this, state,
111                         logTotalScore, logLanguageScore,
112                         logInsertionProbability, frameNumber);
113             } else {
114                 return new Token(predecessor, state,
115                         logTotalScore, logLanguageScore,
116                         logInsertionProbability, frameNumber);
117             }
118         }

405         /**
406          * dumps a branch of tokens
407          *
408          * @param includeHMMStates if true include all sentence hmm states
409          */
410         public void dumpTokenPath(boolean includeHMMStates) {
411             Token token = this;
412             List<Token> list = new ArrayList<Token>();
413     
414             while (token != null) {
415                 list.add(token);
416                 token = token.getPredecessor();
417             }
418             for (int i = list.size() - 1; i >= 0; i--) {
419                 token = list.get(i);
420                 if (includeHMMStates ||
421                         (!(token.getSearchState() instanceof HMMSearchState))) {
422                     System.out.println("  " + token);
423                 }
424             }
425             System.out.println();
426         }

429         /**
430          * Returns the string of words leading up to this token.
431          *
432          * @param wantFiller         if true, filler words are added
433          * @param wantPronunciations if true append [ phoneme phoneme ... ] after each word
434          * @return the word path
435          */
436         public String getWordPath(boolean wantFiller, boolean wantPronunciations) {
437             StringBuffer sb = new StringBuffer();
438             Token token = this;
439     
440             while (token != null) {
441                 if (token.isWord()) {
442                     WordSearchState wordState =
443                             (WordSearchState) token.getSearchState();
444                     Pronunciation pron = wordState.getPronunciation();
445                     Word word = wordState.getPronunciation().getWord();
446     
447     //                System.out.println(token.getFrameNumber() + " " + word + " " + token.logLanguageScore + " " + token.logAcousticScore);
448     
449                     if (wantFiller || !word.isFiller()) {
450                         if (wantPronunciations) {
451                             sb.insert(0, "]");
452                             Unit[] u = pron.getUnits();
453                             for (int i = u.length - 1; i >= 0; i--) {
454                                 if (i < u.length - 1) sb.insert(0, ",");
455                                 sb.insert(0, u[i].getName());
456                             }
457                             sb.insert(0, "[");
458                         }
459                         sb.insert(0, word.getSpelling());
460                         sb.insert(0, " ");
461                     }
462                 }
463                 token = token.getPredecessor();
464             }
465             return sb.toString().trim();
466         }

489         /**
490          * Returns the string of words and units for this token, with embedded silences.
491          *
492          * @return the string of words and units
493          */
494         public String getWordUnitPath() {
495             StringBuffer sb = new StringBuffer();
496             Token token = this;
497     
498             while (token != null) {
499                 SearchState searchState = token.getSearchState();
500                 if (searchState instanceof WordSearchState) {
501                     WordSearchState wordState = (WordSearchState) searchState;
502                     Word word = wordState.getPronunciation().getWord();
503                     sb.insert(0, " " + word.getSpelling());
504                 } else if (searchState instanceof UnitSearchState) {
505                     UnitSearchState unitState = (UnitSearchState) searchState;
506                     Unit unit = unitState.getUnit();
507                     sb.insert(0, " " + unit.getName());
508                 }
509                 token = token.getPredecessor();
510             }
511             return sb.toString().trim();
512         }
