9     /**
10      * A simple converter which converts <code>DoubleData</code> to <code>FloatData</code> and vv (depending on its
11      * configuration). All remaining <code>Data</code>s will pass this processor unchanged.
12      *
13      * @author Holger Brandl
14      */
15     
16     public class DataConverter extends BaseDataProcessor {
17     
18         public static final String CONVERT_D2F = "d2f";
19         public static final String CONVERT_F2D = "f2d";
20     
21         @S4String(defaultValue = "d2f", range = {CONVERT_D2F, CONVERT_F2D})
22         public static final String PROP_CONVERSION_MODE = "conversionMode";
23         private String convMode;
24     
25     
26         /*
27         * (non-Javadoc)
28         *
29         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
30         */
31         @Override
32         public void newProperties(PropertySheet ps) throws PropertyException {
33             super.newProperties(ps);
34     
35             convMode = ps.getString(PROP_CONVERSION_MODE);
36         }
37     
38     
39         public Data getData() throws DataProcessingException {
40             Data d = getPredecessor().getData();
41     
42             if (d instanceof DoubleData && convMode.equals(CONVERT_D2F)) {
43                 DoubleData dd = (DoubleData) d;
44                 d = new FloatData(MatrixUtils.double2float(dd.getValues()), dd.getSampleRate(),
45                         dd.getCollectTime(), dd.getFirstSampleNumber());
46             } else if (d instanceof FloatData && convMode.equals(CONVERT_F2D)) {
47                 FloatData fd = (FloatData) d;
48                 d = new DoubleData(MatrixUtils.float2double(fd.getValues()), fd.getSampleRate(),
49                         fd.getCollectTime(), fd.getFirstSampleNumber());
50             }
51     
52             return d;
53         }
54     }
