50         /** Dumps the search space hmm in GDL format */
51         public void run() {
52             try {
53                 FileOutputStream fos = new FileOutputStream(filename);
54                 PrintStream out = new PrintStream(fos);
55                 SearchState firstState =
56                         getLinguist().getSearchGraph().getInitialState();
57                 dumpSearchGraph(out, firstState);
58                 out.close();
59             } catch (FileNotFoundException fnfe) {
60                 System.out.println("Can't dump to file " + filename + " " + fnfe);
61             }
62         }

127         /**
128          * Dumps the search graph
129          *
130          * @param out           place to dump the output
131          * @param startingState the initial state of the search space
132          */
133         private void dumpSearchGraph(PrintStream out, SearchState startingState) {
134             List queue = new LinkedList();
135             Set visitedStates = new HashSet();
136             startDump(out);
137             queue.add(new StateLevel(startingState, 0));
138             while (queue.size() > 0) {
139                 StateLevel stateLevel = (StateLevel) queue.remove(0);
140                 int level = stateLevel.getLevel();
141                 SearchState state = stateLevel.getState();
142                 // equalCheck(state);
143                 if (!visitedStates.contains(state.getSignature())) {
144                     visitedStates.add(state.getSignature());
145                     startDumpNode(out, state, level);
146                     SearchStateArc[] arcs = state.getSuccessors();
147                     for (int i = arcs.length - 1; i >= 0; i--) {
148                         SearchState nextState = arcs[i].getState();
149                         dumpArc(out, state, arcs[i], level);
150                         if (depthFirst) {
151                             // if depth first, its a stack
152                             queue.add(0, new StateLevel(nextState, level + 1));
153                         } else {
154                             queue.add(new StateLevel(nextState, level + 1));
155                         }
156                     }
157                     endDumpNode(out, state, level);
158                 }
159             }
160             endDump(out);
161         }

164         Map eqStates = new HashMap();
165         Map eqSigs = new HashMap();
166     
167     
168         /**
169          * This is a bit of test/debugging code that ensures that the states that have equal signatures are also considered
170          * to be object.equals and vice versa.. This method will dump out any states where this contract is not true
171          *
172          * @param state the state to check
173          */
174         private void equalCheck(SearchState state) {
175             SearchState eqState = (SearchState) eqStates.get(state);
176             SearchState eqSig = (SearchState) eqSigs.get(state.getSignature());
177             if ((eqState == null || eqSig == null) && !(eqState == eqSig)) {
178                 System.out.println("Missing one: ");
179                 System.out.println("  state val: " + state);
180                 System.out.println("  state sig: " + state.getSignature());
181                 System.out.println("  eqState val: " + eqState);
182                 System.out.println("  eqSig val: " + eqSig);
183                 if (eqState != null) {
184                     System.out.println("   eqState sig: " + eqState.getSignature());
185                 }
186                 if (eqSig != null) {
187                     System.out.println("   eqSig sig: " + eqSig.getSignature());
188                 }
189             }
190             if (eqState == null) {
191                 eqStates.put(state, state);
192                 eqState = state;
193             }
194             if (eqSig == null) {
195                 eqSigs.put(state.getSignature(), state);
196                 eqSig = state;
197             }
198             if (!eqState.getSignature().equals(state.getSignature())) {
199                 System.out.println("Sigs mismatch for: ");
200                 System.out.println("  state sig: " + state.getSignature());
201                 System.out.println("  eqSig sig: " + eqSig.getSignature());
202                 System.out.println("  state val: " + state);
203                 System.out.println("  eqSig val: " + eqSig);
204             }
205             if (!eqState.equals(state)) {
206                 System.out.println("obj mismatch for: ");
207                 System.out.println("  state sig: " + state.getSignature());
208                 System.out.println("  eqSig sig: " + eqSig.getSignature());
209                 System.out.println("  state val: " + state);
210                 System.out.println("  eqSig val: " + eqSig);
211             }
212         }

215     /** A class for bundling together a SearchState and its level. */
216     
217     class StateLevel {
218     
219         private int level;
220         private SearchState state;
221     
222     
223         /**
224          * Constructs a StateLevel from its primitive components.
225          *
226          * @param state the state to be bundled in the StateLevel
227          * @param level the level of the state
228          */
229         StateLevel(SearchState state, int level) {
230             this.state = state;
231             this.level = level;
232         }
233     
234     
235         /**
236          * Returns the state
237          *
238          * @return the state
239          */
240         SearchState getState() {
241             return state;
242         }
243     
244     
245         /**
246          * Returns the level
247          *
248          * @return the level.
249          */
250         int getLevel() {
251             return level;
252         }
253     
254     
255         /**
256          * Returns a string representation of the object
257          *
258          * @return a string representation
259          */
260         public String toString() {
261             return "" + level + " " + state.getSignature() + " 1 "
262                     + getTypeLabel(state);
263         }
264     
265     
266         /**
267          * Retrieves a type label for a state
268          *
269          * @param state the state of interest
270          * @return a label for the type of state (one of Unit, Word, HMM or other
271          */
272         public String getTypeLabel(SearchState state) {
273             if (state instanceof UnitSearchState) {
274                 return "Unit";
275             }
276             if (state instanceof WordSearchState) {
277                 return "Word";
278             }
279             if (state instanceof HMMSearchState) {
280                 return "HMM";
281             }
282             return "other";
283         }
