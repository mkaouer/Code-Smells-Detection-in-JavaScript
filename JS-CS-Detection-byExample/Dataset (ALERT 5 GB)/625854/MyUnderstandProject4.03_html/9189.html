13     package edu.cmu.sphinx.tools.feature;
14     
15     import edu.cmu.sphinx.frontend.*;
16     import edu.cmu.sphinx.frontend.util.StreamDataSource;
17     import edu.cmu.sphinx.util.props.ConfigurationManager;
18     import edu.cmu.sphinx.util.props.PropertyException;
19     
20     import java.io.*;
21     import java.net.URL;
22     import java.util.Iterator;
23     import java.util.LinkedList;
24     import java.util.List;
25     import java.util.logging.Logger;
26     
27     /**
28      * This program takes in an audio file, does frontend signal processing to it, and then dumps the resulting Feature into
29      * a separate file.
30      * <p/>
31      * This program takes three arguments: 1. propsFile - the Sphinx properties file 2. audioFile - the name of the audio
32      * file 3. outputFile - the name of the output file
33      */
34     public class FeatureFileDumper {
35     
36         private FrontEnd frontEnd;
37         private List allFeatures;
38         // Initialize this to an invalid number
39         private int featureLength = -1;
40     
41         /** The logger for this class */
42         private static Logger logger = Logger
43                 .getLogger("edu.cmu.sphinx.tools.feature.FeatureFileDumper");
44     
45     
46         /**
47          * Constructs a FeatureFileDumper.
48          *
49          * @param cm             the Sphinx configuration manager
50          * @param inputAudioFile the input audio file
51          */
52         public FeatureFileDumper(ConfigurationManager cm, String frontEndName,
53                                  String inputAudioFile)
54                 throws FileNotFoundException, IOException {
55             try {
56     
57                 frontEnd = (FrontEnd) cm.lookup(frontEndName);
58                 StreamDataSource audioSource = (StreamDataSource) cm
59                         .lookup("streamDataSource");
60                 audioSource.setInputStream(new FileInputStream(inputAudioFile),
61                         "audio");
62                 allFeatures = new LinkedList();
63                 getAllFeatures();
64                 logger.info("Frames: " + allFeatures.size());
65             } catch (Exception e) {
66                 e.printStackTrace();
67             }
68         }
69     
70     
71         /** Retrieve all Features from the frontend, and cache all those with actual feature data. */
72         private void getAllFeatures() {
73             /*
74              * Run through all the data and produce feature.
75              */
76             try {
77                 assert (allFeatures != null);
78                 Data feature = frontEnd.getData();
79                 while (!(feature instanceof DataEndSignal)) {
80                     if (feature instanceof DoubleData) {
81                         double[] featureData = ((DoubleData) feature).getValues();
82                         if (featureLength < 0) {
83                             featureLength = featureData.length;
84                             logger.info("Feature length: " + featureLength);
85                         }
86                         allFeatures.add(featureData);
87                     } else if (feature instanceof FloatData) {
88                         float[] featureData = ((FloatData) feature).getValues();
89                         if (featureLength < 0) {
90                             featureLength = featureData.length;
91                             logger.info("Feature length: " + featureLength);
92                         }
93                         allFeatures.add(featureData);
94                     }
95                     feature = frontEnd.getData();
96                 }
97             } catch (Exception e) {
98                 e.printStackTrace();
99             }
100         }
101     
102     
103         /**
104          * Returns the total number of data points that should be written to the output file.
105          *
106          * @return the total number of data points that should be written
107          */
108         private int getNumberDataPoints() {
109             return (allFeatures.size() * featureLength);
110         }
111     
112     
113         /**
114          * Dumps the feature to the given binary output.
115          *
116          * @param outputFile the binary output file
117          */
118         public void dumpBinary(String outputFile) throws IOException {
119             DataOutputStream outStream = new DataOutputStream(new FileOutputStream(
120                     outputFile));
121             outStream.writeInt(getNumberDataPoints());
122     
123             for (Iterator i = allFeatures.iterator(); i.hasNext();) {
124                 Object data = i.next();
125                 if (data instanceof double[]) {
126                     double[] feature = (double[]) data;
127                     for (int d = 0; d < feature.length; d++) {
128                         outStream.writeFloat((float) feature[d]);
129                     }
130                 } else if (data instanceof float[]) {
131                     float[] feature = (float[]) data;
132                     for (int d = 0; d < feature.length; d++) {
133                         outStream.writeFloat(feature[d]);
134                     }
135                 }
136             }
137     
138             outStream.close();
139         }
140     
141     
142         /**
143          * Dumps the feature to the given ascii output file.
144          *
145          * @param outputFile the ascii output file
146          */
147         public void dumpAscii(String outputFile) throws IOException {
148             PrintStream ps = new PrintStream(new FileOutputStream(outputFile), true);
149             ps.print(getNumberDataPoints());
150             ps.print(' ');
151     
152             for (Iterator i = allFeatures.iterator(); i.hasNext();) {
153                 Object data = i.next();
154                 if (data instanceof double[]) {
155                     double[] feature = (double[]) data;
156                     for (int d = 0; d < feature.length; d++) {
157                         ps.print(feature[d]);
158                         ps.print(' ');
159                     }
160                 } else if (data instanceof float[]) {
161                     float[] feature = (float[]) data;
162                     for (int d = 0; d < feature.length; d++) {
163                         ps.print(feature[d]);
164                         ps.print(' ');
165                     }
166                 }
167             }
168     
169             ps.close();
170         }
171     
172     
173         /**
174          * Main program for this dumper.
175          * <p/>
176          * This program takes three arguments: 1. propsFile - the Sphinx properties file 2. audioFile - the name of the
177          * audio file 3. outputFile - the name of the binary output file
178          */
179         public static void main(String[] argv) {
180     
181             if (argv.length < 3) {
182                 System.out
183                         .println("Usage: FeatureFileDumper "
184                                 + "configFile frontendName inputFile outputFile [(binary|ascii)]");
185                 System.exit(1);
186             }
187             String configFile = argv[0];
188             String frontEndName = argv[1];
189             String inputFile = argv[2];
190             String outputFile = argv[3];
191             String format = "binary";
192             if (argv.length == 5) {
193                 format = argv[4];
194             }
195             logger.info("Input file: " + inputFile);
196             logger.info("Output file: " + outputFile);
197             logger.info("Format: " + format);
198     
199             try {
200                 URL url = new File(configFile).toURI().toURL();
201                 ConfigurationManager cm = new ConfigurationManager(url);
202                 FeatureFileDumper dumper = new FeatureFileDumper(cm, frontEndName,
203                         inputFile);
204                 if (format.equals("binary")) {
205                     dumper.dumpBinary(outputFile);
206                 } else if (format.equals("ascii")) {
207                     dumper.dumpAscii(outputFile);
208                 } else {
209                     System.out.println("ERROR: unknown output format: " + format);
210                 }
211             } catch (IOException ioe) {
212                 System.err.println("I/O Error " + ioe);
213             } catch (PropertyException p) {
214                 System.err.println("Bad configuration " + p);
215             }
216         }
217     }
