13     package edu.cmu.sphinx.demo.wavfile;
14     
15     import edu.cmu.sphinx.frontend.util.AudioFileDataSource;
16     import edu.cmu.sphinx.recognizer.Recognizer;
17     import edu.cmu.sphinx.result.Result;
18     import edu.cmu.sphinx.util.props.ConfigurationManager;
19     
20     import java.io.File;
21     import java.net.MalformedURLException;
22     import java.net.URL;
23     
24     
25     /** A simple Sphinx-4 application that decodes a .WAV file containing connnected-digits audio data. */
26     public class WavFile {
27     
28         public static void main(String[] args) throws MalformedURLException {
29             URL audioFileURL;
30             URL configURL;
31     
32             // use defaults that are loaded from the WavFile.jar or use values provided as arguments to main
33             if (args.length == 2) {
34                 configURL = new File(args[0]).toURI().toURL();
35                 audioFileURL = new File(args[1]).toURI().toURL();
36             } else {
37                 audioFileURL = WavFile.class.getResource("12345.wav");
38                 configURL = WavFile.class.getResource("config.xml");
39             }
40     
41             System.out.println("Loading Recognizer as defined in '" + configURL.toString() + "'...\n");
42             ConfigurationManager cm = new ConfigurationManager(configURL);
43     
44             // look up the recognizer (which will also lookup all its dependencies
45             Recognizer recognizer = (Recognizer) cm.lookup("recognizer");
46             recognizer.allocate();
47     
48             // configure the audio input for the recognizer
49             AudioFileDataSource dataSource = (AudioFileDataSource) cm.lookup("audioFileDataSource");
50             dataSource.setAudioFile(audioFileURL, null);
51     
52             // decode the audio file.
53             System.out.println("Decoding " + audioFileURL);
54             Result result = recognizer.recognize();
55     
56             System.out.println("Result: " + (result != null ? result.getBestFinalResultNoFiller() : null));
57         }
58     }
