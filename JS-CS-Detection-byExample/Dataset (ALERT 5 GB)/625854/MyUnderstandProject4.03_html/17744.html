27     /**
28      * Implements a level tracking endpointer invented by Bent Schmidt Nielsen.
29      * <p/>
30      * <p>This endpointer is composed of three main steps. <ol> <li>classification of audio into speech and non-speech
31      * <li>inserting SPEECH_START and SPEECH_END signals around speech <li>removing non-speech regions </ol>
32      * <p/>
33      * <p>The first step, classification of audio into speech and non-speech, uses Bent Schmidt Nielsen's algorithm. Each
34      * time audio comes in, the average signal level and the background noise level are updated, using the signal level of
35      * the current audio. If the average signal level is greater than the background noise level by a certain threshold
36      * value (configurable), then the current audio is marked as speech. Otherwise, it is marked as non-speech.
37      * <p/>
38      * <p>The second and third step of this endpointer are documented in the classes {@link SpeechMarker SpeechMarker} and
39      * {@link NonSpeechDataFilter NonSpeechDataFilter}.
40      *
41      * @see SpeechMarker
42      */
43     public class SpeechClassifier extends BaseDataProcessor {
44     
45         /** A property specifying the endpointing frame length in milliseconds. */
46         @S4Integer(defaultValue = 10)
47         public static final String PROP_FRAME_LENGTH_MS = "frameLengthInMs";
48     
49         /** A property specifying the minimum signal level used to update the background signal level. */
50         @S4Double(defaultValue = 0)
51         public static final String PROP_MIN_SIGNAL = "minSignal";
52     
53         /**
54          * A property specifying the threshold. If the current signal level is greater than the background level by
55          * this threshold, then the current signal is marked as speech. Therefore, a lower threshold will make the
56          * endpointer more sensitive, that is, mark more audio as speech. A higher threshold will make the endpointer less
57          * sensitive, that is, mark less audio as speech.
58          */
59         @S4Double(defaultValue = 10)
60         public static final String PROP_THRESHOLD = "threshold";
61     
62         /** A property specifying the adjustment. */
63         @S4Double(defaultValue = 0.003)
64         public static final String PROP_ADJUSTMENT = "adjustment";
65     
66         protected double averageNumber = 1;
67         protected double adjustment;
68         protected double level;               // average signal level
69         protected double background;          // background signal level
70         protected double minSignal;           // minimum valid signal level
71         protected double threshold;
72         protected float frameLengthSec;
73         protected boolean isSpeech;
74     
75         protected List<Data> outputQueue = new LinkedList<Data>();
76     
77         @Override
78         public void newProperties(PropertySheet ps) throws PropertyException {
79             super.newProperties(ps);
80             int frameLengthMs = ps.getInt(PROP_FRAME_LENGTH_MS);
81             frameLengthSec = ((float) frameLengthMs) / 1000.f;
82     
83             adjustment = ps.getDouble(PROP_ADJUSTMENT);
84             threshold = ps.getDouble(PROP_THRESHOLD);
85             minSignal = ps.getDouble(PROP_MIN_SIGNAL);
86     
87             logger = ps.getLogger();
88             //logger.setLevel(Level.FINEST);
89      
90             initialize();
91         }
92     
93     
94         /** Initializes this LevelTracker endpointer and DataProcessor predecessor. */
95         public void initialize() {
96             super.initialize();
97             reset();
98         }
99     
100     
101         /** Resets this LevelTracker to a starting state. */
102         protected void reset() {
103             level = 0;
104             background = 100;
105         }
106     
107     
108         /**
109          * Returns the logarithm base 10 of the root mean square of the given samples.
110          *
111          * @param samples the samples
112          * @return the calculated log root mean square in log 10
113          */
114         public static double logRootMeanSquare(double[] samples) {
115             assert samples.length > 0;
116             double sumOfSquares = 0.0f;
117             for (int i = 0; i < samples.length; i++) {
118                 double sample = samples[i];
119                 sumOfSquares += sample * sample;
120             }
121             double rootMeanSquare = Math.sqrt
122                     (sumOfSquares / samples.length);
123             rootMeanSquare = Math.max(rootMeanSquare, 1);
124             return (LogMath.log10((float) rootMeanSquare) * 20);
125         }
126     
127     
128         /**
129          * Classifies the given audio frame as speech or not, and updates the endpointing parameters.
130          *
131          * @param audio the audio frame
132          */
133         protected void classify(DoubleData audio) {
134             double current = logRootMeanSquare(audio.getValues());
135             // System.out.println("current: " + current);
136             isSpeech = false;
137             if (current >= minSignal) {
138                 level = ((level * averageNumber) + current) / (averageNumber + 1);
139                 if (current < background) {
140                     background = current;
141                 } else {
142                     background += (current - background) * adjustment;
143                 }
144                 if (level < background) {
145                     level = background;
146                 }
147                 isSpeech = (level - background > threshold);
148             }
149     
150             SpeechClassifiedData labeledAudio = new SpeechClassifiedData(audio, isSpeech);
151     
152             if (logger.isLoggable(Level.FINEST)) {
153                 String speech = "";
154                 if (labeledAudio.isSpeech())
155                     speech = "*";
156     
157                 System.out.println("Bkg: " + background + ", level: " + level +
158                         ", current: " + current + " " + speech);
159             }
160     
161             outputQueue.add(labeledAudio);
162         }
163     
164     
165         /**
166          * Returns the next Data object.
167          *
168          * @return the next Data object, or null if none available
169          * @throws DataProcessingException if a data processing error occurs
170          */
171         public Data getData() throws DataProcessingException {
172             if (outputQueue.size() == 0) {
173                 Data audio = getPredecessor().getData();
174     
175                 if (audio instanceof DataStartSignal)
176                     reset();
177     
178                 if (audio != null) {
179                     if (audio instanceof DoubleData) {
180                         DoubleData data = (DoubleData) audio;
181                         if (data.getValues().length >
182                                 ((int) (frameLengthSec * data.getSampleRate()))) {
183                             throw new Error
184                                     ("Length of data frame is " +
185                                             data.getValues().length +
186                                             " samples, but the expected frame is <= " +
187                                             (frameLengthSec * data.getSampleRate()));
188                         }
189                         classify(data);
190                     } else {
191                         outputQueue.add(audio);
192                     }
193                 }
194             }
195             if (outputQueue.size() > 0) {
196                 Data audio = outputQueue.remove(0);
197                 return audio;
198             } else {
199                 return null;
200             }
201         }
202         
203         public boolean isSpeech() {
204         	return isSpeech;
205         }
206     }
