15     import edu.cmu.sphinx.frontend.*;
16     import edu.cmu.sphinx.frontend.endpoint.SpeechEndSignal;
17     import edu.cmu.sphinx.frontend.endpoint.SpeechStartSignal;
18     import edu.cmu.sphinx.result.Result;

21     /**
22      * A decoder which does not use the common pull-principle of S4 but processes only one single frame on each call of
23      * <code>decode()</code>. When using this decoder, make sure that the <code>AcousticScorer</code> used by the
24      * <code>SearchManager</code> can access some buffered <code>Data</code>s.
25      */
26     public class FrameDecoder extends AbstractDecoder implements DataProcessor {
27     
28         private DataProcessor predecessor;
29     
30         private boolean isRecognizing;
31         private Result result;
32     
33     
34         /**
35          * Decode a single frame.
36          *
37          * @param referenceText the reference text (or null)
38          * @return a result
39          */
40         public Result decode(String referenceText) {
41             return searchManager.recognize(1);
42         }
43     
44     
45         public Data getData() throws DataProcessingException {
46             Data d = getPredecessor().getData();
47     
48             if (isRecognizing && (d instanceof FloatData || d instanceof DoubleData || d instanceof SpeechEndSignal)) {
49                 result = decode(null);
50     
51                 if (result != null) {
52                     fireResultListeners(result);
53                     result = null;
54                 }
55             }
56     
57             // we also trigger recogntion on a DataEndSignal to allow threaded scorers to shut down correctly
58             if (d instanceof DataEndSignal) {
59                 searchManager.stopRecognition();
60             }
61     
62             if (d instanceof SpeechStartSignal) {
63                 searchManager.startRecognition();
64                 isRecognizing = true;
65                 result = null;
66             }
67     
68             if (d instanceof SpeechEndSignal) {
69                 searchManager.stopRecognition();
70     
71                 //fire results which were not yet final
72                 if (result != null)
73                     fireResultListeners(result);
74     
75                 isRecognizing = false;
76             }
77     
78             return d;
79         }
80     
81     
82         public DataProcessor getPredecessor() {
83             return predecessor;
84         }
85     
86     
87         public void setPredecessor(DataProcessor predecessor) {
88             this.predecessor = predecessor;
89         }
90     
91     
92         public void initialize() {
93         }
94     
95     }
