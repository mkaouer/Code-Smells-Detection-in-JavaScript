13     package edu.cmu.sphinx.linguist.flat;
14     
15     import edu.cmu.sphinx.linguist.acoustic.LeftRightContext;
16     import edu.cmu.sphinx.linguist.acoustic.Unit;
17     
18     /** A ContextBucket manages left context history when an SentenceHMM is being built */
19     class ContextBucket {
20     
21         private int maxSize;
22         private Unit[] units;
23     
24     
25         /**
26          * Creates a context bucket with the given maximum size
27          *
28          * @param maxSize the maximum size of the context bucket
29          */
30         ContextBucket(int maxSize) {
31             units = new Unit[0];
32             this.maxSize = maxSize;
33         }
34     
35     
36         /**
37          * Creates a new context bucket by appending the unit to the context in the previous context bucket
38          *
39          * @param prev the previous context bucket
40          * @param unit the unit to add to this context
41          */
42         ContextBucket(ContextBucket prev, Unit unit) {
43             int newSize;
44             Unit[] prevUnits = prev.getUnits();
45     
46             this.maxSize = prev.maxSize;
47             newSize = Math.min(prevUnits.length + 1, maxSize);
48             this.units = new Unit[newSize];
49     
50             if (newSize > 0) {
51                 units[0] = unit;
52     
53                 System.arraycopy(prevUnits, 0, units, 1, newSize);
54             }
55         }
56     
57     
58         /**
59          * Returns the set of units for this context bucket
60          *
61          * @return the set of units
62          */
63         Unit[] getUnits() {
64             return units;
65         }
66     
67     
68         /**
69          * Returns the string representation for this StatePath
70          *
71          * @return the string representation
72          */
73         public String toString() {
74             return LeftRightContext.getContextName(units);
75         }
76     }
