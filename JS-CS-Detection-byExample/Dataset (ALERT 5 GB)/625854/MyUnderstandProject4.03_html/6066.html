26     import java.awt.image.FilteredImageSource;
27     import java.awt.image.ImageFilter;
28     import java.awt.image.ReplicateScaleFilter;
29     import java.util.ArrayList;

31     /** Converts a set of log magnitude Spectrum data into a graphical representation. */
32     public class SpectrogramPanel extends JPanel {
33     
34         /** Where the spectrogram will live. */
35         protected BufferedImage spectrogram = null;
36     
37         /** A scaled version of the spectrogram image. */
38         protected Image scaledSpectrogram = null;
39     
40         /** The zooming factor. */
41         protected float zoom = 1.0f;
42     
43         /** Offset factor - what will be subtracted from the image to adjust for noise level. */
44         protected double offsetFactor;
45     
46         /** The audio data. */
47         protected AudioData audio;
48     
49     
50         /** The frontEnd (the source of features */
51         protected FrontEnd frontEnd;
52     
53         /** The source of audio (the first stage of the frontend) */
54         protected StreamDataSource dataSource;
55     
56     
57         /** Creates a new <code>JPanel</code> with a double buffer and a flow layout. */
58         public SpectrogramPanel() {
59         }
60     
61     
62         /**
63          * Creates a new SpectrogramPanel for the given AudioData.
64          *
65          * @param frontEnd   the front end to use
66          * @param dataSource the source of audio
67          * @param audioData  the AudioData
68          */
69         public SpectrogramPanel(FrontEnd frontEnd,
70                                 StreamDataSource dataSource, AudioData audioData) {
71             audio = audioData;
72             this.frontEnd = frontEnd;
73             this.dataSource = dataSource;
74             audio.addChangeListener(new ChangeListener() {
75                 public void stateChanged(ChangeEvent event) {
76                     computeSpectrogram();
77                 }
78             });
79         }
80     
81     
82         /** Actually creates the Spectrogram image. */
83         protected void computeSpectrogram() {
84             try {
85                 AudioDataInputStream is = new AudioDataInputStream(audio);
86                 dataSource.setInputStream(is, "live audio");
87     
88                 /* Run through all the spectra one at a time and convert
89                  * them to an log intensity value.
90                  */
91                 ArrayList<double[]> intensitiesList = new ArrayList<double[]>();
92                 double maxIntensity = Double.MIN_VALUE;
93                 Data spectrum = frontEnd.getData();
94     
95                 while (!(spectrum instanceof DataEndSignal)) {
96                     if (spectrum instanceof DoubleData) {
97                         double[] spectrumData = ((DoubleData) spectrum).getValues();
98                         double[] intensities = new double[spectrumData.length];
99                         for (int i = 0; i < intensities.length; i++) {
100                             /*
101     			 * A very small intensity is, for all intents
102                              * and purposes, the same as 0.
103                              */
104                             intensities[i] = Math.max(Math.log(spectrumData[i]),
105                                     0.0);
106                             if (intensities[i] > maxIntensity) {
107                                 maxIntensity = intensities[i];
108                             }
109                         }
110                         intensitiesList.add(intensities);
111                     }
112                     spectrum = frontEnd.getData();
113                 }
114                 is.close();
115     
116                 int width = intensitiesList.size();
117                 int height = (intensitiesList.get(0)).length;
118                 int maxYIndex = height - 1;
119                 Dimension d = new Dimension(width, height);
120     
121                 setMinimumSize(d);
122                 setMaximumSize(d);
123                 setPreferredSize(d);
124     
125                 /* Create the image for displaying the data.
126                  */
127                 spectrogram = new BufferedImage(width,
128                         height,
129                         BufferedImage.TYPE_INT_RGB);
130     
131                 /* Set scaleFactor so that the maximum value, after removing
132                 * the offset, will be 0xff.
133                 */
134                 double scaleFactor = ((0xff + offsetFactor) / maxIntensity);
135     
136                 for (int i = 0; i < width; i++) {
137                     double[] intensities = intensitiesList.get(i);
138                     for (int j = maxYIndex; j >= 0; j--) {
139     
140                         /* Adjust the grey value to make a value of 0 to mean
141                         * white and a value of 0xff to mean black.
142                         */
143                         int grey = (int) (intensities[j] * scaleFactor
144                                 - offsetFactor);
145                         grey = Math.max(grey, 0);
146                         grey = 0xff - grey;
147     
148                         /* Turn the grey into a pixel value.
149                         */
150                         int pixel = ((grey << 16) & 0xff0000)
151                                 | ((grey << 8) & 0xff00)
152                                 | (grey & 0xff);
153     
154                         spectrogram.setRGB(i, maxYIndex - j, pixel);
155                     }
156                 }
157                 ImageFilter scaleFilter =
158                         new ReplicateScaleFilter((int) (zoom * width), height);
159                 scaledSpectrogram =
160                         createImage(new FilteredImageSource(spectrogram.getSource(),
161                                 scaleFilter));
162                 Dimension sz = getSize();
163                 repaint(0, 0, 0, sz.width - 1, sz.height - 1);
164             } catch (Exception e) {
165                 e.printStackTrace();
166             }
167         }
168     
169     
170         /**
171          * Updates the offset factor used to calculate the greyscale values from the intensities.  This also calculates and
172          * populates all the greyscale values in the image.
173          *
174          * @param offsetFactor the offset factor used to calculate the greyscale values from the intensities; this is used
175          *                     to adjust the level of background noise that shows up in the image
176          */
177         public void setOffsetFactor(double offsetFactor) {
178             this.offsetFactor = offsetFactor;
179             computeSpectrogram();
180         }
181     
182     
183         /** Zoom the image, preparing for new display. */
184         protected void zoomSet(float zoom) {
185             this.zoom = zoom;
186             if (spectrogram != null) {
187                 int width = spectrogram.getWidth();
188                 int height = spectrogram.getHeight();
189     
190                 ImageFilter scaleFilter =
191                         new ReplicateScaleFilter((int) (zoom * width), height);
192                 scaledSpectrogram =
193                         createImage(new FilteredImageSource(spectrogram.getSource(),
194                                 scaleFilter));
195                 repaint();
196             }
197         }
198     
199     
200         /**
201          * Paint the component.  This will be called by AWT/Swing.
202          *
203          * @param g The <code>Graphics</code> to draw on.
204          */
205         public void paint(Graphics g) {
206             /**
207              * Fill in the whole image with white.
208              */
209             Dimension sz = getSize();
210     
211             g.setColor(Color.WHITE);
212             g.fillRect(0, 0, sz.width - 1, sz.height - 1);
213     
214             if (spectrogram != null) {
215     
216                 g.drawImage(scaledSpectrogram, 0, 0, null);
217             }
218         }
219     }
