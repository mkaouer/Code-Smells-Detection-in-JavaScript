13     package edu.cmu.sphinx.recognizer;
14     
15     import edu.cmu.sphinx.decoder.Decoder;
16     import edu.cmu.sphinx.decoder.ResultProducer;
17     import edu.cmu.sphinx.decoder.ResultListener;
18     import edu.cmu.sphinx.instrumentation.Resetable;
19     import edu.cmu.sphinx.result.Result;
20     import edu.cmu.sphinx.util.props.*;
21     
22     import java.util.ArrayList;
23     import java.util.Collections;
24     import java.util.Iterator;
25     import java.util.List;
26     
27     /**
28      * The Sphinx-4 recognizer. This is the main entry point for Sphinx-4. Typical usage of a recognizer is like so:
29      * <p/>
30      * <pre><code>
31      *  public void recognizeDigits() {
32      *      URL digitsConfig = new URL("file:./digits.xml");
33      *      ConfigurationManager cm = new ConfigurationManager(digitsConfig);
34      *      Recognizer sphinxDigitsRecognizer
35      *          = (Recognizer) cm.lookup("digitsRecognizer&quot");
36      *      boolean done = false;
37      *      Result result;
38      * <p/>
39      *      sphinxDigitsRecognizer.allocate();
40      * <p/>
41      *     // echo spoken digits, quit when 'nine' is spoken
42      * <p/>
43      *      while (!done) {
44      *           result = sphinxDigitsRecognizer.recognize();
45      *           System.out.println(&quot;Result: &quot; + result);
46      *           done = result.toString().equals(&quot;nine&quot;);
47      *      }
48      * <p/>
49      *      sphinxDigitsRecognizer.deallocate();
50      *   }
51      * </code></pre>
52      * <p/>
53      * Note that some Recognizer methods may throw an IllegalStateException if the recognizer is not in the proper state
54      */
55     public class Recognizer implements Configurable, ResultProducer {
56     
57         /** Property name for the decoder to be used by this recognizer. */
58         @S4Component(type = Decoder.class)
59         public final static String PROP_DECODER = "decoder";
60     
61         /** Property name for the set of monitors for this recognizer */
62         @S4ComponentList(type = StateListener.class)
63         public final static String PROP_MONITORS = "monitors";
64     
65     
66         private String name;
67         private Decoder decoder;
68         private RecognizerState currentState = RecognizerState.DEALLOCATED;
69     
70         private List<StateListener> stateListeners = Collections.synchronizedList(new ArrayList<StateListener>());
71         private List monitors;
72     
73     
74         /* (non-Javadoc)
75         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
76         */
77         public void newProperties(PropertySheet ps) throws PropertyException {
78             decoder = (Decoder) ps.getComponent(PROP_DECODER);
79             monitors = ps.getComponentList(PROP_MONITORS);
80     
81             name = ps.getInstanceName();
82         }
83     
84     
85         /**
86          * Performs recognition for the given number of input frames, or until a 'final' result is generated. This method
87          * should only be called when the recognizer is in the <code>allocated</code> state.
88          *
89          * @param referenceText what was actually spoken
90          * @return a recognition result
91          * @throws IllegalStateException if the recognizer is not in the <code>ALLOCATED</code> state
92          */
93         public Result recognize(String referenceText) throws IllegalStateException {
94             Result result = null;
95             checkState(RecognizerState.READY);
96             try {
97                 setState(RecognizerState.RECOGNIZING);
98                 result = decoder.decode(referenceText);
99             } finally {
100                 setState(RecognizerState.READY);
101             }
102             return result;
103         }
104     
105     
106         /**
107          * Performs recognition for the given number of input frames, or until a 'final' result is generated. This method
108          * should only be called when the recognizer is in the <code>allocated</code> state.
109          *
110          * @return a recognition result
111          * @throws IllegalStateException if the recognizer is not in the <code>ALLOCATED</code> state
112          */
113         public Result recognize() throws IllegalStateException {
114             return recognize(null);
115         }
116     
117     
118         /**
119          * Checks to ensure that the recognizer is in the given state.
120          *
121          * @param desiredState the state that the recognizer should be in
122          * @throws IllegalStateException if the recognizer is not in the desired state.
123          */
124         private void checkState(RecognizerState desiredState) {
125             if (currentState != desiredState) {
126                 throw new IllegalStateException("Expected state " + desiredState
127                         + " actual state " + currentState);
128             }
129         }
130     
131     
132         /**
133          * sets the current state
134          *
135          * @param newState the new state
136          */
137         private void setState(RecognizerState newState) {
138             currentState = newState;
139             synchronized (stateListeners) {
140                 for (Iterator<StateListener> i = stateListeners.iterator(); i.hasNext();) {
141                     StateListener stateListener = i.next();
142                     stateListener.statusChanged(currentState);
143                 }
144             }
145         }
146     
147     
148         /**
149          * Allocate the resources needed for the recognizer. Note this method make take some time to complete. This method
150          * should only be called when the recognizer is in the <code> deallocated </code> state.
151          *
152          * @throws IllegalStateException if the recognizer is not in the <code>DEALLOCATED</code> state
153          */
154         public void allocate() throws IllegalStateException {
155             checkState(RecognizerState.DEALLOCATED);
156             setState(RecognizerState.ALLOCATING);
157             decoder.allocate();
158             setState(RecognizerState.ALLOCATED);
159             setState(RecognizerState.READY);
160         }
161     
162     
163         /**
164          * Deallocates the recognizer. This method should only be called if the recognizer is in the <code> allocated
165          * </code> state.
166          *
167          * @throws IllegalStateException if the recognizer is not in the <code>ALLOCATED</code> state
168          */
169         public void deallocate() throws IllegalStateException {
170             checkState(RecognizerState.READY);
171             setState(RecognizerState.DEALLOCATING);
172             decoder.deallocate();
173             setState(RecognizerState.DEALLOCATED);
174         }
175     
176     
177         /**
178          * Retrieves the recognizer state. This method can be called in any state.
179          *
180          * @return the recognizer state
181          */
182         public RecognizerState getState() {
183             return currentState;
184         }
185     
186     
187         /** Resets the monitors monitoring this recognizer */
188         public void resetMonitors() {
189             for (Iterator i = monitors.iterator(); i.hasNext();) {
190                 Object o = i.next();
191                 if (o instanceof Resetable) {
192                     Resetable r = (Resetable) o;
193                     r.reset();
194                 }
195             }
196         }
197     
198     
199         /**
200          * Adds a result listener to this recognizer. A result listener is called whenever a new result is generated by the
201          * recognizer. This method can be called in any state.
202          *
203          * @param resultListener the listener to add
204          */
205         public void addResultListener(ResultListener resultListener) {
206             decoder.addResultListener(resultListener);
207         }
208     
209     
210         /**
211          * Adds a status listener to this recognizer. The status listener is called whenever the status of the recognizer
212          * changes. This method can be called in any state.
213          *
214          * @param stateListener the listener to add
215          */
216         public void addStateListener(StateListener stateListener) {
217             stateListeners.add(stateListener);
218         }
219     
220     
221         /**
222          * Removes a previously added result listener. This method can be called in any state.
223          *
224          * @param resultListener the listener to remove
225          */
226         public void removeResultListener(ResultListener resultListener) {
227             decoder.removeResultListener(resultListener);
228         }
229     
230     
231         /**
232          * Removes a previously added state listener. This method can be called in any state.
233          *
234          * @param stateListener the state listener to remove
235          */
236         public void removeStateListener(StateListener stateListener) {
237             stateListeners.remove(stateListener);
238         }
239     
240     
241         /* (non-Javadoc)
242         * @see java.lang.Object#toString()
243         */
244         public String toString() {
245             return "Recognizer: " + name + " State: " + currentState;
246         }
247     
248     }
