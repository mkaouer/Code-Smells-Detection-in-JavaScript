354         /**
355          * Enters speaking (i.e., recording) mode, sets the various GUI objects
356          * to the correct state.
357          */
358         public void enterSpeakingMode() {
359             debugMessage("Entering speaking mode.");
360             setGUISpeakingState(true);
361             setMessage("Wait...");
362             
363             // update GUI states
364             setRecognitionLabel("");
365     
366             // start recording
367             debugMessage("Turning speaker on.");
368             if (live.startRecording()) {
369                 debugMessage("Speaker turned on.");
370                 setMessage("OK, start speaking...");
371                 live.decode();
372             } else {
373                 debugMessage("Can't turn speaker on.");
374                 setMessage("Error opening the audio device");
375                 setGUISpeakingState(false);
376             }
377         }

489          * Shows a FileDialog, and updates the testFileField if a file
490          * is selected.
491          */
492         private void chooseTestFile() {
493             JFileChooser chooser = new JFileChooser(live.getCurrentDirectory());
494             int returnValue = chooser.showDialog(this, "Select");
495             if (returnValue == JFileChooser.APPROVE_OPTION) {
496                 String testFile = chooser.getSelectedFile().getName();

498                 live.setTestFile(testFile);

781         /**
782          * Changes the font size of all components in this JFrame.
783          * 
784          * @param fontSize the new font size
785          */
786         private void setGlobalFontSize(int fontSize) {
787             if (globalFont == null) {
788                 globalFont = getFont();
789             }
790             globalFont = new Font(globalFontFace, Font.PLAIN, fontSize);
791     
792             String[] guis = {"Button.font", "ComboBox.font", "Frame.font",
793                              "Label.font", "List.font",
794                              "Menu.font", "MenuItem.font",
795                              "TextArea.font", "TextField.font",
796                              "TitledBorder.font", "ToggleButton.font",
797                              "ToolTip.font"};
798             
799             for (int i = 0; i < guis.length; i++) {
800                 UIManager.put(guis[i], globalFont);
801             }
802     
803             setFont(globalFont);
804

809         /**
810          * A ComboBox that shows which decoder we are using.
811          */
812         class DecoderComboBox extends JComboBox {
813     
814             private int lastIndex;
815     
816             /**
817              * Constructs a DecoderComboBox with the given ComboBoxModel.
818              *
819              * @param model the ComboBoxModel that contains a list
820              *    of all the decoder name
821              */
822             public DecoderComboBox(ComboBoxModel model) {
823                 super(model);
824                 lastIndex = -1;
825                 addActionListener(new ComboBoxActionListener());
826             }
827     
828             class ComboBoxActionListener implements ActionListener {
829                 
830                 /**
831                  * Invoked when an action is performed on this DecoderComboBox.
832                  *
833                  * @param e the ActionEvent performed
834                  */
835                 public void actionPerformed(ActionEvent e) {
836                     if (getSelectedIndex() != lastIndex) {
837                         Thread t = new Thread() {
838                             public void run() {
839                                 String decoderName = (String) getSelectedItem();
840                                 try {
841                                     live.setDecoder(decoderName);
842                                     lastIndex = getSelectedIndex();
843                                 } catch (IOException ioe) {
844                                     setMessage("Error switching  to " 
845                                             + decoderName);
846                                 }
847                             }
848                         };
849                         t.start();
850                     }
851                 }
852             }
853         }
854     }
