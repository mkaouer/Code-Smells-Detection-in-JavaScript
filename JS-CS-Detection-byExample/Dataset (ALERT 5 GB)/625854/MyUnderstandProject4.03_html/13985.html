8      * Allows to modify the gain of an audio-signal.  If the gainFactor is 1 the signal passes this
9      * <code>DataProcessor</code> unchanged.
10      *
11      * @author Holger Brandl
12      */
13     public class GainControlProcessor extends BaseDataProcessor {
14     
15         @S4Double(defaultValue = 1.0)
16         public static final String GAIN_FACTOR = "gainFactor";
17     
18         private double gainFactor;
19     
20     
21         /*
22         * (non-Javadoc)
23         *
24         * @see edu.cmu.sphinx.util.props.Configurable#newProperties(edu.cmu.sphinx.util.props.PropertySheet)
25         */
26         public void newProperties(PropertySheet ps) throws PropertyException {
27             super.newProperties(ps);
28     
29             gainFactor = ps.getDouble(GAIN_FACTOR);
30         }
31     
32     
33         public Data getData() throws DataProcessingException {
34             Data data = getPredecessor().getData();
35     
36             if (data instanceof FloatData) {
37                 float[] values = ((FloatData) data).getValues();
38                 if (gainFactor != 1.0) {
39                     // apply the gain-factor
40                     for (int i = 0; i < values.length; i++) {
41                         values[i] *= gainFactor;
42     
43                     }
44                 }
45     
46             } else if (data instanceof DoubleData) {
47                 double[] values = ((DoubleData) data).getValues();
48                 if (gainFactor != 1.0) {
49                     // apply the gain-factor
50                     for (int i = 0; i < values.length; i++) {
51                         values[i] *= gainFactor;
52     
53                     }
54                 }
55             }
56     
57             return data;
58         }
59     
60     
61         public double getGainFactor() {
62             return gainFactor;
63         }
64     
65

67             this.gainFactor = gainFactor;
