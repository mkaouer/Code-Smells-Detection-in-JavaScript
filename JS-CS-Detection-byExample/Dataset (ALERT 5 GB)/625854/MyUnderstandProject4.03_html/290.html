14     package edu.cmu.sphinx.demo.jsapi.cardmatch;
15     
16     
17     /**
18      * An interface that represents a spoken prompt
19      */
20     public class SequencePrompt implements Prompt {
21         private String[] prompts;
22         private int curPrompt = 0;
23     
24         /**
25          * Creates a sequence prompt
26          *
27          * @param an array of prompts
28          *
29          */
30         public SequencePrompt(String[] prompts) {
31             this.prompts = prompts;
32         }
33     
34         /**
35          * Gets the next text to be spoken
36          */
37         public String getText() {
38             String prompt = prompts[curPrompt++];
39             if (curPrompt >= prompts.length) {
40                 curPrompt = 0;
41             }
42             return prompt;
43         }
44     
45     
46         /**
47          * Resets the prompt to its initial state
48          */
49         public void reset() {
50             curPrompt = 0;
51         }
52     }
