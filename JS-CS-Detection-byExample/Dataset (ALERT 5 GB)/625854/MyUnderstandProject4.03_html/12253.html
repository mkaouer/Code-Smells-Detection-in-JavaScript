26     /**
27      * This is a helper class to convert from ConfigProperties to String format 
28      * that's ready to be written to .config.XML file
29      *
30      * @author Ariani
31      */
32     public class ConfigConverter {
33         
34         /* this class uses Singleton pattern for the constructor */
35         private ConfigConverter() {}
36         
37         private static class ConfigConverterHolder {
38           private static ConfigConverter instance = new ConfigConverter();
39         }
40           
41         /**
42          * Get reference to the <code>ConfigConverter</code> 
43          * 
44          * @return <code>ConfigConverter</code>
45          */
46         public static ConfigConverter getInstance(){
47           return ConfigConverterHolder.instance;
48         }

50         private final static String XML_HEADING = "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n";
51         private final static String XML_SUBHEAD = "\n" +
52             "<!--    Sphinx-4 Configuration file--> \n" +
53             "<!--  ******************************************************** -->\n\n";
54         private final static String XML_ROOT = "<config>\n";
55         private final static String XML_ROOT_CLOSE = "</config>\n";
56         private final static String XML_GLOBAL = "\n" +
57             "<!--  *******************************************-->\n"+ 
58             "<!--  frequently tuned properties                -->\n" +
59             "<!-- ********************************************-->\n"; 
60         private final static String XML_CONF =  "\n" +
61          "<!--  ********************************************************-->\n"+
62          "<!--  component       configuration                           -->\n"+
63          "<!--  ********************************************************-->\n\n"; 
64     
65         private final static String XML_PROP = "\t <property name=\"";
66         private final static String XML_COMPONENT = "<component name=\"";
67         private final static String XML_COMPONENT_CLOSE = "</component>\n\n";
68         private final static String XML_TYPE = " type=\"";
69         private final static String XML_VALUE = " value=\"";
70         private final static String XML_PROPLIST = "\t <propertylist name=\"";
71         private final static String XML_PROPLIST_CLOSE = "\t </propertylist>\n";
72         private final static String XML_ITEM = "\t\t <item>";
73         private final static String XML_ITEM_CLOSE = "</item>\n";
74         
75         private final static String XML_CLOSE_ELEMENT = "\" /> \n";
76         private final static String XML_CLOSE_TAG = "\" > \n";
77         private final static String XML_QUOTE = "\"";
78                 
79         // a helper method to write the information from a single RawPropertyData 
80         private void writeComponent(StringBuffer sb, RawPropertyData rpd){
81             String name = rpd.getName();
82             String classname = rpd.getClassName();
83             Map properties = rpd.getProperties();
84             if (name != null && classname != null) // if the rpd is valid
85             {
86                 sb = sb.append(XML_COMPONENT+name+XML_QUOTE+XML_TYPE+classname+XML_CLOSE_TAG);
87                 
88                 for(Iterator it = properties.entrySet().iterator(); it.hasNext();)
89                 {
90                     Map.Entry entry = (Map.Entry) it.next();            
91                     String propName = (String)entry.getKey();
92                     List propList;
93                     String propVal;
94     
95                     if(propName != null && (entry.getValue() != null))
96                     {
97                         if (entry.getValue() instanceof String)
98                         {
99                             propVal = (String)entry.getValue();
100                             sb = sb.append(XML_PROP+propName+XML_QUOTE+XML_VALUE+propVal+XML_CLOSE_ELEMENT);
101                         }
102                         else // value is a list
103                         {
104                             sb = sb.append(XML_PROPLIST+propName+XML_CLOSE_TAG);
105                             propList = (List)entry.getValue();
106                             // iterate the propertyList
107                             for (Iterator it2= propList.iterator(); it2.hasNext();)
108                             {
109                                 String item = (String) it2.next();            
110     
111                                 if(item != null && !(item.trim().equalsIgnoreCase("")))
112                                 {
113                                     sb = sb.append(XML_ITEM+item+XML_ITEM_CLOSE);                
114                                 }
115     
116                             }
117                             sb = sb.append(XML_PROPLIST_CLOSE);
118                         }
119                     }
120                 }
121                 sb = sb.append(XML_COMPONENT_CLOSE);
122             }
123         }
124         
125         /**
126          * Write the property name and values into a StringBuffer
127          * 
128          * @param cp <code>ConfigProperties</code> that keeps the property name-values
129          * @return StringBuffer to write the output
130          */
131         public StringBuffer writeOutput(ConfigProperties cp) {
132     
133             StringBuffer sb = new StringBuffer();
134             Map global = cp.getGlobal();
135             Map property = cp.getProperty();
136     
137             sb = sb.append(XML_HEADING + XML_SUBHEAD + XML_ROOT + '\n');
138             sb = sb.append(XML_GLOBAL + '\n');
139             
140             // iterate the global properties
141             if (cp != null && sb != null) {
142                 if (global != null) {
143                     for (Iterator it = global.entrySet().iterator(); it.hasNext();) {
144                         Map.Entry entry = (Map.Entry) it.next();
145                         String name = (String) entry.getKey();
146                         if (name != null) {
147                             String value = (String) entry.getValue();
148                             sb = sb.append(XML_PROP + name + XML_QUOTE + XML_VALUE + value + XML_CLOSE_ELEMENT);
149                         }
150     
151                     }
152                 }
153                 sb = sb.append(XML_CONF);
154     
155                 if (property != null) {
156                     for (Iterator it = property.entrySet().iterator(); it.hasNext();) {
157                         Map.Entry entry = (Map.Entry) it.next();
158                         String name = (String) entry.getKey();
159                         RawPropertyData rpd = (RawPropertyData) entry.getValue();
160                         if (name != null && rpd != null) {
161                             writeComponent(sb, rpd);
162                         }
163                     }
164                 }
165             }
166             sb = sb.append(XML_ROOT_CLOSE + '\n');
167             return sb;
168         }
169     }
