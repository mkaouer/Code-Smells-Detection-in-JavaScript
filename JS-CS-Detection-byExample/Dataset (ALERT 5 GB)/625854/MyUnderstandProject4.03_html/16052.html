168         /**
169          * This method would return the source code for the specified class name
170          *
171          * @param classname Name of class to be searched
172          */
173         public String getSourceCode(String classname) {
174             String text = "";
175             if (_source_path == null) {
176                 return "== Source code not available ==";
177             }
178             try {
179                 String completename = classname.trim().replace('.', '/');
180                 if (!_source_path.trim().endsWith("/")) // add  '/' at the end of path
181                     completename = _source_path.trim().concat("/" + completename);
182                 else
183                     completename = _source_path.trim().concat(completename);
184     
185                 //add the .java at the end of class name
186                 if (!completename.endsWith(".java"))
187                     completename = completename.concat(".java");
188     
189                 BufferedReader br = new BufferedReader(new FileReader(completename));
190                 String thisline;
191                 while ((thisline = br.readLine()) != null) {
192                     text = text.concat(thisline + '\n');
193                 }
194                 br.close();
195     
196             } catch (FileNotFoundException e) {
197                 return null;
198             } catch (IOException e) {
199                 return null;
200             }
201             return text;
202         }

268         /**
269          * Load values in ConfigProperties into the model
270          *
271          * @param cp <code>ConfigProperty</code> that holds the new property values
272          */
273         private void loadCurrentValues(ConfigProperties cp) {
274             Map rpdMap = cp.getOtherProp(); // get map that has classname as key
275     
276             // iterate through the property map and retrieve the values to initialize the model
277             for (Object o : rpdMap.entrySet()) {
278                 Map.Entry entry = (Map.Entry) o;
279                 String classname = (String) entry.getKey();
280                 for (Object o1 : ((Map) entry.getValue()).entrySet()) {
281                     Map.Entry nextentry = (Map.Entry) o1;
282                     RawPropertyData rpd = (RawPropertyData) nextentry.getValue();
283                     if (_classes.containsKey(classname)) {
284                         // copy the value from rpd
285                         ConfigurableComponent cc = (ConfigurableComponent) _classes.get(classname);
286                         Map propertyMap = rpd.getProperties();
287                         checkProperty(propertyMap, cc);  // check if rpd properties are valid
288                         addIncompleteProps(propertyMap, cc); // add incomplete properties into rpd
289                         cc.addConfigurationProp(rpd); // add rpd into the model
290     
291                     } // end _classes contain this classname
292     
293                 } // end for
294             } // end for
295         } // end loadCurrentValues

298         /**
299          * add the properties that are in the Configurable Component model, but not in the configurablecomponent yet, by
300          * adding them as default this function should never be public
301          */
302         private void addIncompleteProps(Map propertyMap, ConfigurableComponent checker) {
303             Map completePropMap = checker.getPropertyMap();
304             for (Object o : completePropMap.entrySet()) {
305                 Map.Entry propentry = (Map.Entry) o;
306                 String propname = (String) propentry.getKey();
307     
308                 // if it doesn't exist yet
309                 if (!propertyMap.containsKey(propname)) {
310                     ConfigurableProperty prop = (ConfigurableProperty) propentry.getValue();
311                     String defaultVal = prop.getDefault();
312                     if (defaultVal != null && !defaultVal.trim().equalsIgnoreCase("")) {
313                         propertyMap.put(propname, defaultVal);
314                         //  System.out.println("***** add prop "+propname);
315                     } else {
316                         propertyMap.put(propname, null);
317                         //  System.out.println("***** add null prop "+propname);
318                     }
319                 }
320             }
321         }

325          * check if the set of configuration to be loaded corresponds to the property listed in the sphinx model. if it
326          * doesn't exist in sphinx model, delete it this method should always be private

328         private void checkProperty(Map propertyMap, ConfigurableComponent checker) {
329             for (Object o : propertyMap.entrySet()) {
330                 Map.Entry propentry = (Map.Entry) o;
331                 String propname = (String) propentry.getKey();

333                 if (!checker.containsProperty(propname)) {
334                     propertyMap.remove(propname);
335                     // delete because it doesn't exist in sphinx model               
336                 }
337             } // end for

376         /**
377          * Refresh the Sphinx model
378          *
379          * @throws ConfigurableUtilException when there is an error during re-reading of Sphinx system
380          */
381         public void refresh() throws ConfigurableUtilException {
382             _classes.clear();
383             _groups.clear();
384     
385             //init config data
386             clear_config();
387             try {
388                 read_config_prop();
389             } catch (ConfigurableUtilException e) {
390                 // the util.conf configuration read is not successful
391                 // probably the file does not exist 
392                 // (must be in the same dir as GUIMainDriver.class)
393                 // so - no features that requires the source code    
394                 System.err.println("Error:Configuration file not loaded successfully\n" +
395                         "Error:Features that require access to source code will not be available");
396             }
397             read_classpath();
398     
399             // scan again and refresh the model
400             scan(_folder_path, _package_path);
401         }

421         // read the configuration file that keeps the path to
422         // where Sphinx system source code files and compiled .class files
423         // should be located
424         private void read_config_prop() throws ConfigurableUtilException {
425             try {
426                 InputStream in = this.getClass().getResourceAsStream(CONFIG_PATH);
427                 //FileInputStream in = new FileInputStream(CONFIG_PATH);
428                 Properties configuration = new Properties();
429                 configuration.load(in);   //load the properties from configuration file           
430                 in.close();
431                 _source_path = (String) configuration.get(SOURCE_PROP);
432                 System.out.println("source path: " + _source_path);
433                 _classes_path = (String) configuration.get(CLASSES_PROP);
434     
435             } catch (FileNotFoundException fe) {
436                 throw new ConfigurableUtilException("Configuration file " + CONFIG_PATH +
437                         " not found", ConfigurableUtilException.UTIL_INIT);
438             } catch (IOException ie) {
439                 throw new ConfigurableUtilException("Configuration " +
440                         "File I/O Error", ConfigurableUtilException.UTIL_INIT);
441             } catch (java.lang.NullPointerException e) {
442                 throw new ConfigurableUtilException("Configuration " +
443                         "File Load Error", ConfigurableUtilException.UTIL_INIT);
444             }
445         }

448         /**
449          * Does the scanning starting from the given directory and create the list of configurable classes
450          *
451          * @param startDir     the root directory of Sphinx
452          * @param startPackage the starting package of Sphinx
453          * @throws <code>ConfigurableUtilException</code>
454          *          when there is an error while scanning folders and building the model
455          */
456         private void scan(String startDir, String startPackage)
457                 throws ConfigurableUtilException {
458             try {
459                 // get back a set of folder names that is directly under startDir
460                 List myFolders = ClassFinder.findFolder(startDir);
461     
462                 Set myClasses;
463                 Set thisgroup;
464                 String tempName;
465                 ConfigurableComponent tempcc;
466     
467                 // get the list of classes from each group
468                 for (Object myFolder : myFolders) {
469                     thisgroup = new HashSet();
470     
471                     tempName = (String) myFolder;
472                     //   System.out.println("Folder "+tempName);
473     
474                     myClasses = new HashSet(); // classes from this section
475                     ClassFinder.findClasses(startDir + '/' + tempName,
476                             startPackage + '.' + tempName, myClasses);
477     
478                     // get the classes for this section and build the model
479                     for (Object myClass : myClasses) {
480                         try {
481                             tempcc = createcomponent((Class) myClass,
482                                     startPackage + '.' + tempName);
483                             _classes.put(tempcc.getName(), tempcc);
484                             thisgroup.add(tempcc);
485                         } catch (IllegalAccessException e) {
486                             throw new ConfigurableUtilException
487                                     ("IllegalAccessException while creating configurable component",
488                                             ConfigurableUtilException.UTIL_BUILDER);
489                         }
490                     }
491     
492                     // add the group to the Map with the section name as key    
493                     _groups.put(startPackage + '.' + tempName, thisgroup);
494     
495                 }
496     
497             } catch (ClassNotFoundException e) {
498                 throw new ConfigurableUtilException("class not found " + e.getMessage(),
499                         ConfigurableUtilException.UTIL_BUILDER);
500             } catch (FileNotFoundException e) {
501                 throw new ConfigurableUtilException("file not found " + e.getMessage(),
502                         ConfigurableUtilException.UTIL_BUILDER);
503             } catch (IOException e) {
504                 throw new ConfigurableUtilException("Classpath loader error " + e.getMessage(),
505                         ConfigurableUtilException.UTIL_PATHHACKER);
506             }
507     
508         }//end scan method

511         /**
512          * Create a new ConfigurableComponent based on this <code>Class</code>
513          *
514          * @param c     Class that the ConfigurableComponent should be based on
515          * @param group The name of the group/section this component belongs to
516          * @return <code>ConfigurableComponent</code> instance
517          */
518         private ConfigurableComponent createcomponent(Class c, String group)
519                 throws ConfigurableUtilException, IllegalAccessException {
520             ConfigurableComponent cc =
521                     new ConfigurableComponent(group, c, c.getName(), "");
522             System.out.println("***** create ConfigurableComponent " +
523                     c.getName() + " for " + group);
524     
525             // check all the public fields
526             // if any of them is with modifier 'public static final' and starts with 'PROP_'        
527             Field[] publicFields = c.getFields();
528             for (int i = 0; i < publicFields.length; i++) {
529                 int m = publicFields[i].getModifiers();
530                 String fieldname = publicFields[i].getName();
531                 //System.out.println("*** checking field "+fieldname);
532                 if (Modifier.isPublic(m) && Modifier.isStatic(m) && Modifier.isFinal(m)) {
533                     if (fieldname.startsWith("PROP_")) {
534                         // Name of property
535     		    // String propname = new String((String) publicFields[i].get(null));
536                         // Create the configurable property
537                         // System.out.println("***** create ConfigurableProperty " + 
538                         // 			  (String)publicFields[i].get(null));
539                         ConfigurableProperty cp = createProperty(c, publicFields[i]);
540                         cc.addProperty(cp);
541                     }
542                 }
543             }
544             return cc;
545         }//end createcomponent method

604             private static void processAnnotation() throws IllegalAccessException {
605     
606                 Annotation[] annotations = _field.getAnnotations();
607     
608                 // check each annotation belongs to this field
609                 for (Annotation annotation : annotations) {
610                     //search for the super common annotation of all s4 properties
611                     Annotation[] superAnnotations = annotation.annotationType().getAnnotations();
612                     for (Annotation superAnnotation : superAnnotations) {
613     
614                         // if this annotation belongs to a sphinx4 configurable property, 
615                         // then only we get its info
616                         if (superAnnotation instanceof S4Property) {
617                             System.out.println("*** s4 property member");
618     
619                             // get info from the s4 property annotation
620                             System.out.println(_field.getName() + " : *** " + annotation.annotationType().getName());
621                             if (annotation instanceof S4Double) {
622                                 _type = PropertyType.DOUBLE;
623                                 _defval = Double.toString(((S4Double) annotation).defaultValue());
624                             } else if (annotation instanceof S4Integer) {
625                                 _type = PropertyType.INT;
626                                 _defval = Integer.toString(((S4Integer) annotation).defaultValue());
627                             } else if (annotation instanceof S4Component) {
628                                 _type = PropertyType.COMPONENT;
629                                 _classtype = ((S4Component) annotation).type().getName();
630                             } else if (annotation instanceof S4String) {
631                                 _type = PropertyType.STRING;
632                                 _defval = ((S4String) annotation).defaultValue();
633                             } else if (annotation instanceof S4Boolean) {
634                                 _type = PropertyType.BOOLEAN;
635                                 _defval = String.valueOf(((S4Boolean) annotation).defaultValue());
636                             } else if (annotation instanceof S4ComponentList) {
637                                 _type = PropertyType.COMPONENT_LIST;
638                                 _classtype = ((S4ComponentList) annotation).type().getName();
639                             }
640     //                        System.out.println(" *** Property type : " + _type);
641     //                        System.out.println(" *** class type : " + _classtype );
642     //                        System.out.println(" *** default value : " + _defval);
643                         }
644                     }
645                 }
646             } // end processAnnotation

649             private static ConfigurableProperty createProperty(Field f, Class c) throws IllegalAccessException {
650                 _class = c;
651                 _field = f;
652                 _fieldname = f.getName();
653                 _type = null;
654                 _defval = null;
655                 _classtype = null;
656     
657                 ConfigurableProperty cp;
658                 String propname = (String) _field.get(null); //name of configurable property
659     
660                 //javadoc comment of the property
661                 String field_comment = JavadocExtractor.getJavadocComment(_class.getName(),
662                         _classes_path, _source_path, _fieldname);
663                 if (field_comment == null) // no comment
664                     field_comment = "";
665                 System.out.println(" Comment *** : " + field_comment);
666     
667                 processAnnotation();
668     
669                 if (_defval == null)
670                     _defval = "";
671     
672                 if (_type != null &&
673                         (_type == PropertyType.COMPONENT || _type == PropertyType.COMPONENT_LIST)) {
674                     // it's a class component type, 
675                     // _classtype info should exist if it is specified in sphinx4 source code                                    
676                     cp = new ConfigurableProperty
677                             (propname, _defval, _type, field_comment, _fieldname, _classtype);
678                     // System.out.println("with type "+proptype+" and class type : " + classtype);
679     
680                 } else { // it's a native java type
681                     cp = new ConfigurableProperty
682                             (propname, _defval, _type, field_comment, _fieldname);
683                     // System.out.println("with type "+proptype);
684                 }
685                 return cp;
686             } // end createProperty
