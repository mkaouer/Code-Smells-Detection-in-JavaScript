30     /**
31      * Maps the phones from one phoneset to another to use dictionary from the one
32      * acoustic mode with another one. The mapping file is specified with a mapList
33      * property. The contents should look like
34      * 
35      * <pre>
36      * AX AH
37      * IX IH
38      * </pre>
39      */
40     public class MappingDictionary extends FastDictionary implements Dictionary {
41     
42     	@S4String(mandatory = true, defaultValue = "")
43     	public static final String PROP_MAP_FILE = "mapFile";
44     
45     	private URL mappingFile;
46     	private Map<String,String> mapping = new HashMap<String,String>();
47     				
48     	@Override
49     	public void newProperties(PropertySheet ps) throws PropertyException {
50     		super.newProperties(ps);
51     	   
52     		mappingFile = ConfigurationManagerUtils.getResource(PROP_MAP_FILE, ps);
53     	}
54     
55        /*
56         * (non-Javadoc)
57         *
58         * @see edu.cmu.sphinx.linguist.dictionary.Dictionary#allocate()
59         */
60     	@Override
61         public void allocate() throws IOException {
62         		super.allocate();
63         		if (mappingFile.getFile() != "") 
64         				loadMapping (mappingFile.openStream());
65         }
66     	/**
67     	 * Gets a context independent unit. There should only be one instance of any
68     	 * CI unit
69     	 * 
70     	 * @param name
71     	 *            the name of the unit
72     	 * @param isFiller
73     	 *            if true, the unit is a filler unit
74     	 * @return the unit
75     	 */
76     	protected Unit getCIUnit(String name, boolean isFiller) {
77     		if (mapping.containsKey(name)) {
78     			name = mapping.get(name);
79     		}
80     		return unitManager.getUnit(name, isFiller, Context.EMPTY_CONTEXT);
81     	}
82     
83     	protected void loadMapping(InputStream inputStream) throws IOException {
84     		InputStreamReader isr = new InputStreamReader(inputStream);
85     		BufferedReader br = new BufferedReader(isr);
86     		String line;
87     		while ((line = br.readLine()) != null) {
88                 StringTokenizer st = new StringTokenizer(line);
89                 if (st.countTokens() != 2) {
90                 	throw new IOException ("Wrong file format");
91                 }
92     			mapping.put(st.nextToken(), st.nextToken());
93     		}
94     		br.close();
95     		isr.close();
96     		inputStream.close();
97     	}
98     }
