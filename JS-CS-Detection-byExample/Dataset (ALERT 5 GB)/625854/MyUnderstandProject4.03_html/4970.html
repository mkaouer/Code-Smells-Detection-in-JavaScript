12     package edu.cmu.sphinx.result;

14     import edu.cmu.sphinx.util.props.Configurable;
15     
16     /**
17      * <p/>
18      * Computes confidences for a Result. Typically, one is interested in the confidence of the best path of a result, as
19      * well as the confidence of each word in the best path of a result. To obtain this information, one should do the
20      * following: </p>
21      * <p/>
22      * <pre>
23      * <p/>
24      * ConfidenceScorer scorer = (ConfidenceScorer) ... // obtain scorer from configuration manager
25      * <p/>
26      * Result result = recognizer.recognize();
27      * ConfidenceResult confidenceResult = scorer.score(result);
28      * <p/>
29      * // confidence for best path
30      * Path bestPath = confidenceResult.getBestHypothesis();
31      * double pathConfidence = bestPath.getConfidence();
32      * <p/>
33      * // confidence for each word in best path
34      * WordResult[] words = bestPath.getWords();
35      * for (int i = 0; i < words.length; i++) {
36      *     WordResult wordResult = (WordResult) words[i];
37      *     double wordConfidence = wordResult.getConfidence();
38      * }
39      * <p/>
40      * </pre>
41      * <p/>
42      * <p/>
43      * Note that different ConfidenceScorers have different definitions for the 'best path', and therefore their
44      * <code>getBestHypothesis</code> methods will return different things. The {@link
45      * edu.cmu.sphinx.result.MAPConfidenceScorer} returns the highest scoring path. On the other hand, the {@link
46      * edu.cmu.sphinx.result.SausageMaker} returns the path where all the words have the highest confidence in their
47      * corresponding time slot. </p>
48      */
49     public interface ConfidenceScorer extends Configurable {
50     
51         /**
52          * Computes confidences for a Result and returns a ConfidenceResult, a compact representation of all the hypothesis
53          * contained in the result together with their per-word and per-path confidences.
54          *
55          * @param result the result to compute confidences for
56          * @return a confidence result
57          */
58         public ConfidenceResult score(Result result);
59     }
