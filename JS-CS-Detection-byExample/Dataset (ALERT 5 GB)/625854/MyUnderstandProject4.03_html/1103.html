24     /**
25      * <p/>
26      * Provides recognition lattice results. Lattices are created from {@link edu.cmu.sphinx.result.Result Results} which
27      * can be partial or final. </p>
28      * <p/>
29      * Lattices describe all theories considered by the Recognizer that have not been pruned out.  Lattices are a directed
30      * graph containing {@link edu.cmu.sphinx.result.Node Nodes} and {@link edu.cmu.sphinx.result.Edge Edges}. A Node that
31      * corresponds to a theory that a word was spoken over a particular period of time.  An Edge that corresponds to the
32      * score of one word following another.  The usual result transcript is the sequence of Nodes though the Lattice with
33      * the best scoring path. Lattices are a useful tool for analyzing "alternate results". </p>
34      * <p/>
35      * A Lattice can be created from a Result that has a full token tree (with its corresponding
36      * AlternativeHypothesisManager). Currently, only the {@link edu.cmu.sphinx.decoder.search.WordPruningBreadthFirstSearchManager}
37      * has an AlternativeHypothesisManager. Furthermore, the lattice construction code currently only works for linguists
38      * where the {@link edu.cmu.sphinx.linguist.WordSearchState} returns false on the <code>isWordStart</code> method, i.e.,
39      * where the word states appear at the end of the word in the linguist. <i>Therefore, lattices should only be created
40      * from Result from the {@link edu.cmu.sphinx.linguist.lextree.LexTreeLinguist} and the {@link
41      * edu.cmu.sphinx.decoder.search.WordPruningBreadthFirstSearchManager}. </i> </p>
42      * <p/>
43      * Lattices can also be created from a collapsed {@link edu.cmu.sphinx.decoder.search.Token} tree and its
44      * AlternativeHypothesisManager. This is what 'collapsed' means. Normally, between two word tokens is a series of tokens
45      * for other types of states, such as unit or HMM states. Using 'W' for word tokens, 'U' for unit tokens, 'H' for HMM
46      * tokens, a token chain can look like: </p>
47      * <pre>
48      * W - U - H - H - H - H - U - H - H - H - H - W
49      * </pre>
50      * <p/>
51      * Usually, HMM tokens contains acoustic scores, and word tokens contains language scores. If we want to know the total
52      * acoustic and language scores between any two words, it is unnecessary to keep around the unit and HMM tokens.
53      * Therefore, all their acoustic and language scores are 'collapsed' into one token, so that it will look like: </p>
54      * <pre>
55      * W - P - W
56      * </pre>
57      * <p/>
58      * where 'P' is a token that represents the path between the two words, and P contains the acoustic and language scores
59      * between the two words. It is this type of collapsed token tree that the Lattice class is expecting. Normally, the
60      * task of collapsing the token tree is done by the {@link edu.cmu.sphinx.decoder.search.WordPruningBreadthFirstSearchManager}.
61      * A collapsed token tree can look like: </p>
62      * <pre>
63      *                             "cat" - P - &lt;/s&gt;
64      *                            /
65      *                           P
66      *                          /
67      * &lt;s&gt; - P - "a" - P - "big"
68      *                          \
69      *                           P
70      *                            \
71      *                             "dog" - P - &lt;/s&gt;
72      * </pre>
73      * <p/>
74      * When a Lattice is constructed from a Result, the above collapsed token tree together with the alternate hypothesis of
75      * "all" instead of "a", will be converted into a Lattice that looks like the following:
76      * <pre>
77      *       "a"           "cat"
78      *     /     \        /     \
79      * &lt;s&gt;          "big"         - &lt;/s&gt;
80      *     \     /        \     /
81      *      "all"          "dog"
82      * </pre>
83      * <p/>
84      * Initially, a lattice can have redundant nodes, i.e., nodes referring to the same word and that originate from the
85      * same parent node. These nodes can be collapsed using the {@link LatticeOptimizer}. </p>
86      */

113         /**
114          * Create a Lattice from a Result.
115          * <p/>
116          * The Lattice is created from the Token tree referenced by the Result. The Lattice is then optimized to all
117          * collapse equivalent paths.
118          *
119          * @param result the result to convert into a lattice
120          */
121         public Lattice(Result result) {
122             this(result.getLogMath());
123             visitedWordTokens = new HashSet<Token>();
124             loserManager = result.getAlternateHypothesisManager();
125             if (loserManager != null) {
126                 loserManager.purge();
127             }
128     
129             Iterator<Token> tokenIter;
130             if (result.getBestFinalToken() != null) {
131                 tokenIter = result.getResultTokens().iterator();
132             } else {
133                 tokenIter = result.getActiveTokens().iterator();
134             }
135             while (tokenIter.hasNext()) {
136                 Token token = (Token) tokenIter.next();
137                 while (token != null && !token.isWord()) {
138                     token = token.getPredecessor();
139                 }
140                 assert token != null && token.getWord().isSentenceEndWord();
141                 if (terminalNode == null) {
142                     terminalNode = new Node(getNodeID(result.getBestToken()),
143                             token.getWord(), -1, -1);
144                     initialNode = terminalNode; 
145                     addNode(terminalNode);
146                 }
147                 collapseWordToken(token);
148             }
149         }

152         /**
153          * Returns the node corresponding to the given word token.
154          *
155          * @param token the token which we want a node of
156          * @return the node of the given token
157          */
158         private Node getNode(Token token) {
159             if (token.getWord().isSentenceEndWord()) {
160                 return terminalNode;
161             }
162             Node node = nodes.get(getNodeID(token));
163             if (node == null) {
164                 WordSearchState wordState =
165                         (WordSearchState) token.getSearchState();
166     
167                 int startFrame = -1;
168                 int endFrame = -1;
169     
170                 if (wordState.isWordStart()) {
171                     startFrame = token.getFrameNumber();
172                 } else {
173                     endFrame = token.getFrameNumber();
174                 }
175     
176                 node = new Node(getNodeID(token), token.getWord(),
177                         startFrame, endFrame);
178                 addNode(node);
179             }
180             return node;
181         }

184         /**
185          * Collapse the given word-ending token. This means collapsing all the unit
186          * and HMM tokens that correspond to the word represented by this token into
187          * an edge of the lattice.
188          * 
189          * @param token
190          *            the word-ending token to collapse
191          */
192         private void collapseWordToken(Token token) {
193             assert token != null;
194             if (visitedWordTokens.contains(token)) {
195                 return;
196             }
197             visitedWordTokens.add(token);
198     
199             collapseWordPath(getNode(token), token.getPredecessor(), token
200                     .getAcousticScore(), token.getLanguageScore());
201     
202             if (loserManager != null
203                     && loserManager.hasAlternatePredecessors(token)) {
204                 for (Token loser : loserManager.getAlternatePredecessors(token)) {
205                     collapseWordPath(getNode(token), loser, token
206                             .getAcousticScore(), token.getLanguageScore());
207                 }
208             }
209         }
210     
211         /**
212          * @param parentWordNode
213          *            the 'toNode' of the returned edge
214          * @param token
215          *            the predecessor token of the token represented by the
216          *            parentWordNode
217          * @param acousticScore
218          *            the acoustic score until and including the parent of token
219          * @param languageScore
220          *            the language score until and including the parent of token
221          */
222         private void collapseWordPath(Node parentWordNode, Token token,
223                 float acousticScore, float languageScore) {
224     
225             if (token == null)
226                 return;
227     
228             if (token.isWord()) {
229                 /*
230                  * If this is a word, create a Node for it, and then create an edge
231                  * from the Node to the parentWordNode
232                  */
233                 Node fromNode = getNode(token);
234                 addEdge(fromNode, parentWordNode, (double) acousticScore,
235                         (double) languageScore);
236                 if (token.getPredecessor() != null) {
237                     /* Collapse the token sequence ending in this token. */
238                     collapseWordToken(token);
239                 } else {
240                     /* we've reached the sentence start token */
241                     assert token.getWord().isSentenceStartWord();
242                     initialNode = fromNode;
243                 }
244                 return;
245             }
246     
247             /*
248              * If a non-word token, just add the acoustic and language scores to the
249              * current totals, and then move on to the predecessor token. Fast
250              * forward through the not so interesting states to save stack space.
251              */
252             while (true) {
253                 acousticScore += token.getAcousticScore();
254                 languageScore += token.getLanguageScore();
255                 Token preToken = token.getPredecessor();
256     
257                 if (preToken == null)
258                     return;
259     
260                 if (preToken.isWord()
261                         || (loserManager != null && loserManager
262                                 .hasAlternatePredecessors(token)))
263                     break;
264                 token = preToken;
265             }
266     
267             collapseWordPath(parentWordNode, token.getPredecessor(), acousticScore,
268                     languageScore);
269     
270             /* Traverse the path(s) for the loser token(s). */
271             if (loserManager != null
272                     && loserManager.hasAlternatePredecessors(token)) {
273                 for (Token loser : loserManager.getAlternatePredecessors(token)) {
274                     collapseWordPath(parentWordNode, loser, acousticScore,
275                             languageScore);
276                 }
277             }
278         }

292         /**
293          * Create a Lattice from a LAT file.  LAT files are created by the method Lattice.dump()
294          *
295          * @param fileName
296          */
297         public Lattice(String fileName) {
298             try {
299                 System.err.println("Loading from " + fileName);
300     
301                 // load the nodes
302                 LineNumberReader in = new LineNumberReader(new FileReader(fileName));
303                 String line;
304                 while ((line = in.readLine()) != null) {
305                     StringTokenizer tokens = new StringTokenizer(line);
306                     if (tokens.hasMoreTokens()) {
307                         String type = tokens.nextToken();
308     
309                         if (type.equals("edge:")) {
310                             Edge.load(this, tokens);
311                         } else if (type.equals("node:")) {
312                             Node.load(this, tokens);
313                         } else if (type.equals("initialNode:")) {
314                             setInitialNode(getNode(tokens.nextToken()));
315                         } else if (type.equals("terminalNode:")) {
316                             setTerminalNode(getNode(tokens.nextToken()));
317                         } else if (type.equals("logBase:")) {
318                             logBase = Double.parseDouble(tokens.nextToken());
319                         } else {
320                             throw new Error("SYNTAX ERROR: " + fileName +
321                                     "[" + in.getLineNumber() + "] " + line);
322                         }
323                     }
324                 }
325                 in.close();
326             } catch (Exception e) {
327                 throw new Error(e.toString());
328             }
329         }

332         /**
333          * Add an edge from fromNode to toNode.  This method creates the Edge object and does all the connecting
334          *
335          * @param fromNode
336          * @param toNode
337          * @param acousticScore
338          * @param lmScore
339          * @return the new Edge
340          */
341         public Edge addEdge(Node fromNode, Node toNode,
342                             double acousticScore, double lmScore) {
343             Edge e = new Edge(fromNode, toNode, acousticScore, lmScore);
344             fromNode.addLeavingEdge(e);
345             toNode.addEnteringEdge(e);
346             edges.add(e);
347             return e;
348         }

581          * @param out
582          * @throws IOException

593             out.println("terminalNode: " + terminalNode.getId());
594             out.println("logBase: " + logMath.getLogBase());
595             out.flush();
596         }

613         /**
614          * Remove a Node and all Edges connected to it.  Also remove those Edges from all connected Nodes.
615          *
616          * @param n
617          */
618         protected void removeNodeAndEdges(Node n) {
619     
620             //System.err.println("Removing node " + n + " and associated edges");
621             for (Edge e : n.getLeavingEdges()) {
622                 e.getToNode().removeEnteringEdge(e);
623                 //System.err.println( "\tRemoving " + e );
624                 edges.remove(e);
625             }
626             for (Edge e : n.getEnteringEdges()) {
627                 e.getFromNode().removeLeavingEdge(e);
628                 //System.err.println( "\tRemoving " + e );
629                 edges.remove(e);
630             }
631             //System.err.println( "\tRemoving " + n );
632             nodes.remove(n.getId());
633     
634             assert checkConsistency();
635         }

638         /**
639          * Remove a Node and cross connect all Nodes with Edges to it.
640          * <p/>
641          * For example given
642          * <p/>
643          * Nodes A, B, X, M, N Edges A-->X, B-->X, X-->M, X-->N
644          * <p/>
645          * Removing and cross connecting X would result in
646          * <p/>
647          * Nodes A, B, M, N Edges A-->M, A-->N, B-->M, B-->N
648          *
649          * @param n
650          */
651         protected void removeNodeAndCrossConnectEdges(Node n) {
652             System.err.println("Removing node " + n + " and cross connecting edges");
653             for (Edge ei : n.getEnteringEdges()) {
654                 for (Edge ej : n.getLeavingEdges()) {
655                     addEdge(ei.getFromNode(), ej.getToNode(),
656                             ei.getAcousticScore(), ei.getLMScore());
657                 }
658             }
659             removeNodeAndEdges(n);
660     
661             assert checkConsistency();
662         }

756         protected List<String> allPathsFrom(String path, Node n) {
757             String p = path + " " + n.getWord();
758             List<String> l = new LinkedList<String>();
759             if (n == terminalNode) {
760                 l.add(p);
761             } else {
762                 for (Edge e : n.getLeavingEdges()) {
763                     l.addAll(allPathsFrom(p, e.getToNode()));
764                 }
765             }
766             return l;

770         boolean checkConsistency() {
771             for (Node n : nodes.values()) {
772                 for (Edge e : n.getEnteringEdges()) {
773                     if (!hasEdge(e)) {
774                         throw new Error("Lattice has NODE with missing FROM edge: "
775                                 + n + "," + e);
776                     }
777                 }
778                 for (Edge e : n.getLeavingEdges()) {
779                     if (!hasEdge(e)) {
780                         throw new Error("Lattice has NODE with missing TO edge: " +
781                                 n + "," + e);
782                     }
783                 }
784             }
785             for (Iterator<Edge> i = edges.iterator(); i.hasNext();) {
786                 Edge e = i.next();
787                 if (!hasNode(e.getFromNode())) {
788                     throw new Error("Lattice has EDGE with missing FROM node: " +
789                             e);
790                 }
791                 if (!hasNode(e.getToNode())) {
792                     throw new Error("Lattice has EDGE with missing TO node: " + e);
793                 }
794                 if (!e.getToNode().hasEdgeFromNode(e.getFromNode())) {
795                     throw new Error("Lattice has EDGE with TO node with no corresponding FROM edge: " + e);
796                 }
797                 if (!e.getFromNode().hasEdgeToNode(e.getToNode())) {
798                     throw new Error("Lattice has EDGE with FROM node with no corresponding TO edge: " + e);
799                 }
800             }
801             return true;
802         }

805         protected void sortHelper(Node n, List<Node> sorted, Set<Node> visited) {
806             if (visited.contains(n)) {
807                 return;
808             }
809             visited.add(n);
810             if (n == null) {
811                 throw new Error("Node is null");
812             }
813             for (Edge e : n.getLeavingEdges()) {
814                 sortHelper(e.getToNode(), sorted, visited);
815             }
816             sorted.add(n);
817         }

834          * Compute the utterance-level posterior for every node in the lattice, i.e. the probability that this node occurs
835          * on any path through the lattice. Uses a forward-backward algorithm specific to the nature of non-looping
836          * left-to-right lattice structures.
837          * <p/>
838          * Node posteriors can be retrieved by calling getPosterior() on Node objects.
839          *
840          * @param languageModelWeight the language model weight that was used in generating the scores in the lattice
841          */
842         public void computeNodePosteriors(float languageModelWeight) {
843             computeNodePosteriors(languageModelWeight, false);
844         }

847         /**
848          * Compute the utterance-level posterior for every node in the lattice, i.e. the probability that this node occurs
849          * on any path through the lattice. Uses a forward-backward algorithm specific to the nature of non-looping
850          * left-to-right lattice structures.
851          * <p/>
852          * Node posteriors can be retrieved by calling getPosterior() on Node objects.
853          *
854          * @param languageModelWeight   the language model weight that was used in generating the scores in the lattice
855          * @param useAcousticScoresOnly use only the acoustic scores to compute the posteriors, ignore the language weight
856          *                              and scores
857          */
858         public void computeNodePosteriors(float languageModelWeight,
859                                           boolean useAcousticScoresOnly) {
860             if (initialNode == null)
861                     return;
862             //forward
863             initialNode.setForwardScore(LogMath.getLogOne());
864             initialNode.setViterbiScore(LogMath.getLogOne());
865             List<Node> sortedNodes = sortNodes();
866             assert sortedNodes.get(0) == initialNode;
867             for (Node currentNode : sortedNodes) {
868                 for (Edge edge : currentNode.getLeavingEdges()) {
869                     double forwardProb = edge.getFromNode().getForwardScore();
870                     double edgeScore = computeEdgeScore
871                             (edge, languageModelWeight, useAcousticScoresOnly);
872                     forwardProb += edgeScore;
873                     edge.getToNode().setForwardScore
874                             (logMath.addAsLinear
875                                     ((float) forwardProb,
876                                             (float) edge.getToNode().getForwardScore()));
877                     double vs = edge.getFromNode().getViterbiScore() +
878                             edgeScore;
879                     if (edge.getToNode().getBestPredecessor() == null ||
880                             vs > edge.getToNode().getViterbiScore()) {
881                         edge.getToNode().setBestPredecessor(currentNode);
882                         edge.getToNode().setViterbiScore(vs);
883                     }
884                 }
885             }
886     
887             //backward
888             terminalNode.setBackwardScore(LogMath.getLogOne());
889             assert sortedNodes.get(sortedNodes.size() - 1) == terminalNode;
890             ListIterator<Node> n = sortedNodes.listIterator(sortedNodes.size() - 1);
891             while (n.hasPrevious()) {
892                 Node currentNode = (Node) n.previous();
893                 Collection<Edge> currentEdges = currentNode.getLeavingEdges();
894                 for (Edge edge : currentEdges) {
895                     double backwardProb = edge.getToNode().getBackwardScore();
896                     backwardProb += computeEdgeScore
897                             (edge, languageModelWeight, useAcousticScoresOnly);
898                     edge.getFromNode().setBackwardScore
899                             (logMath.addAsLinear((float) backwardProb,
900                                     (float) edge.getFromNode().getBackwardScore()));
901                 }
902             }
903     
904             //inner
905             double normalizationFactor = terminalNode.getForwardScore();
906             for (Iterator<Node> i = nodes.values().iterator(); i.hasNext();) {
907                 Node node = i.next();
908                 node.setPosterior((node.getForwardScore() +
909                         node.getBackwardScore()) - normalizationFactor);
910             }
911         }

914         /**
915          * Retrieves the MAP path from this lattice. Only works once computeNodePosteriors has been called.
916          *
917          * @return a list of nodes representing the MAP path.
918          */
919         public List<Node> getViterbiPath() {
920             LinkedList<Node> path = new LinkedList<Node>();
921             Node n = terminalNode;
922             while (n != initialNode) {
923                 path.addFirst(n);
924                 n = n.getBestPredecessor();
925             }
926             path.addFirst(initialNode);
927             return path;
928         }

932          * Computes the score of an edge.
933          *
934          * @param edge                the edge which score we want to compute
935          * @param languageModelWeight the language model weight to use
936          * @return the score of an edge
937          */
938         private double computeEdgeScore(Edge edge, float languageModelWeight,
939                                         boolean useAcousticScoresOnly) {
940             if (useAcousticScoresOnly) {
941                 return edge.getAcousticScore();
942             } else {
943                 return (edge.getAcousticScore() + edge.getLMScore()) / languageModelWeight;
944             }

960         /**
961          * Returns true if the two lattices starting at the given two nodes are equivalent. It recursively checks all the
962          * child nodes until these two nodes until there are no more child nodes.
963          *
964          * @param n1 starting node of the first lattice
965          * @param n2 starting node of the second lattice
966          * @return true if the two lattices are equivalent
967          */
968         private boolean checkNodesEquivalent(Node n1, Node n2) {
969             assert n1 != null && n2 != null;
970     
971             boolean equivalent = n1.isEquivalent(n2);
972             if (equivalent) {
973                 Collection<Edge> leavingEdges = n1.getCopyOfLeavingEdges();
974                 Collection<Edge> leavingEdges2 = n2.getCopyOfLeavingEdges();
975     
976                 System.out.println("# edges: " + leavingEdges.size() + " " +
977                         leavingEdges2.size());
978     
979                 for (Edge edge : leavingEdges) {
980                 	/* find an equivalent edge from n2 for this edge */
981                     Edge e2 = n2.findEquivalentLeavingEdge(edge);
982     
983                     if (e2 == null) {
984                         System.out.println
985                                 ("Equivalent edge not found, lattices not equivalent.");
986                         return false;
987                     } else {
988                         if (!leavingEdges2.remove(e2)) {
989                             /*
990                              * if it cannot be removed, then the leaving edges
991                              * are not the same
992                              */
993                             System.out.println
994                                     ("Equivalent edge already matched, lattices not equivalent.");
995                             return false;
996                         } else {
997                             /* recursively check the two child nodes */
998                             equivalent &= checkNodesEquivalent
999                                     (edge.getToNode(), e2.getToNode());
1000                             if (equivalent == false) {
1001                                 return false;
1002                             }
1003                         }
1004                     }
1005                 }
1006                 if (leavingEdges2.size() != 0) {
1007                     System.out.println("One lattice has too many edges.");
1008                     return false;
1009                 }
1010             }
1011             return equivalent;
1012         }
