15     package edu.cmu.sphinx.tools.gui.writer;
16     
17     /**
18      * This exception is thrown when there is an error in the writing process to output
19      *
20      * @author Ariani
21      */
22     public class GUIWriterException extends java.lang.Exception {
23         
24         public final static int EXCEPTION_OTHER = 1;
25         public final static int EXCEPTION_IO = 2;
26         public final static int EXCEPTION_NO_FILENAME = 3;
27         public final static int EXCEPTION_INVALID_TYPE = 4;
28                 
29         private int _exceptionMode;
30         
31        
32         /**
33          * Constructs an instance of <code>GUIWriterException</code> with the 
34          * specified detail message and mode as the type of error
35          *
36          * @param msg the detail message.
37          * @param mode Type of error
38          */
39         public GUIWriterException(String msg, int mode) {
40             super(msg);
41             _exceptionMode = mode;
42             
43         }
44         
45         /**
46          * @return Mode of this exception, which explains the type of exception 
47          */
48         public int getErrorCode(){
49             return _exceptionMode;
50         }
51     }
