13     package edu.cmu.sphinx.linguist.acoustic;
14     
15     import edu.cmu.sphinx.frontend.Data;
16     
17     /** Represents a single state in an HMM */
18     public interface HMMState {
19     
20         /**
21          * Gets the HMM associated with this state
22          *
23          * @return the HMM
24          */
25         public HMM getHMM();
26     
27     
28         /**
29          * Gets the state
30          *
31          * @return the state
32          */
33         public int getState();
34     
35     
36         /**
37          * Gets the score for this HMM state
38          *
39          * @param data the data to be scored
40          * @return the acoustic score for this state.
41          */
42         public float getScore(Data data);
43     
44     
45         /**
46          * Determines if this HMMState is an emittting state
47          *
48          * @return true if the state is an emitting state
49          */
50         public boolean isEmitting();
51     
52     
53         /**
54          * Retrieves the state of successor states for this state
55          *
56          * @return the set of successor state arcs
57          */
58         public HMMStateArc[] getSuccessors();
59     
60     
61         /**
62          * Determines if this state is an exit state of the HMM
63          *
64          * @return true if the state is an exit state
65          */
66         public boolean isExitState();
67     }
