50     package org.jpos.iso.packager;
51     
52     import java.util.BitSet;
53     import java.util.Hashtable;
54     import java.util.Vector;
55     
56     import org.jpos.iso.ISOBitMap;
57     import org.jpos.iso.ISOComponent;
58     import org.jpos.iso.ISOException;
59     import org.jpos.iso.ISOUtil;
60     import org.jpos.util.LogEvent;
61     import org.jpos.util.Logger;
62     
63     /**
64      * GenericSubFieldPackager
65      * Used to pack composite SubFields from the GenericPackager
66      *
67      * @author Eoin Flood
68      * @see GenericPackager
69      *
70      * This class is basically the same as Base1SubFieldPackager except that it extends
71      * GenericPackager which means that parameters such as emitBitmap, maxvalidField and 
72      * bitmapField can be specified in the GenericPackager xml config file.
73      */
74     
75     public class GenericSubFieldPackager extends GenericPackager 
76     {
77         public GenericSubFieldPackager() throws ISOException
78         {
79             super();
80         }
81     
82         public int unpack (ISOComponent m, byte[] b) throws ISOException 
83         {
84             LogEvent evt = new LogEvent (this, "unpack");
85             try 
86             {
87                 if (m.getComposite() != m) 
88                     throw new ISOException ("Can't call packager on non Composite");
89                 if (logger != null)  // save a few CPU cycle if no logger available
90                     evt.addMessage (ISOUtil.hexString (b));
91     
92                 int consumed=0;
93                 ISOBitMap bitmap = new ISOBitMap (-1);
94     
95                 BitSet bmap = null;
96                 int maxField = fld.length;
97                 if (emitBitMap()) 
98                 {
99                     consumed += getBitMapfieldPackager().unpack(bitmap,b,consumed);
100                     bmap = (BitSet) bitmap.getValue();
101                     m.set (bitmap);
102                     maxField = bmap.size();
103                 }
104                 for (int i=getFirstField(); i<maxField && consumed < b.length; i++) 
105                 {
106                     if (bmap == null || bmap.get(i)) 
107                     {
108                         ISOComponent c = fld[i].createComponent(i);
109                         consumed += fld[i].unpack (c, b, consumed);
110                         m.set(c);
111                     }
112                 }
113                 if (b.length != consumed) 
114                 {
115                     evt.addMessage (
116                     "WARNING: unpack len=" +b.length +" consumed=" +consumed);
117                 }
118                 return consumed;
119             } 
120             catch (ISOException e) 
121             {
122                 evt.addMessage (e);
123                 throw e;
124             } 
125             catch (Exception e)
126             {
127                 evt.addMessage (e);
128                 throw new ISOException (e);
129             }
130             finally 
131             {
132                 Logger.log (evt);
133             }
134         }
135     
136         /**
137          * Pack the subfield into a byte array
138          */ 
139     
140         public byte[] pack (ISOComponent m) throws ISOException 
141         {
142             LogEvent evt = new LogEvent (this, "pack");
143             try 
144             {
145                 ISOComponent c;
146                 Vector v = new Vector();
147                 Hashtable fields = m.getChildren();
148                 int len = 0;
149                 byte[] b;
150     
151                 if (emitBitMap()) 
152                 {
153                     // BITMAP (-1 in HashTable)
154                     c = (ISOComponent) fields.get (new Integer (-1));
155                     b = getBitMapfieldPackager().pack(c);
156                     len += b.length;
157                     v.addElement (b);
158                 }
159     
160                 for (int i=getFirstField(); i<=m.getMaxField(); i++) 
161                 {
162                     if ((c = (ISOComponent) fields.get (new Integer (i))) != null) 
163                     {
164                         try 
165                         {
166                             b = fld[i].pack(c);
167                             len += b.length;
168                             v.addElement (b);
169                         } 
170                         catch (Exception e) 
171                         {
172                             evt.addMessage ("error packing subfield "+i);
173                             evt.addMessage (c);
174                             evt.addMessage (e);
175                         }
176                     }
177                 }
178                 int k = 0;
179                 byte[] d = new byte[len];
180                 for (int i=0; i<v.size(); i++) 
181                 {
182                     b = (byte[]) v.elementAt(i);
183                     for (int j=0; j<b.length; j++)
184                     d[k++] = b[j];
185                 }
186                 if (logger != null)  // save a few CPU cycle if no logger available
187                     evt.addMessage (ISOUtil.hexString (d));
188                 return d;
189             } 
190             catch (ISOException e) 
191             {
192                 evt.addMessage (e);
193                 throw e;
194             } 
195             catch (Exception e)
196             {
197                 evt.addMessage (e);
198                 throw new ISOException (e);
199             }
200             finally 
201             {
202                 Logger.log(evt);
203             }
204         }
205     }
