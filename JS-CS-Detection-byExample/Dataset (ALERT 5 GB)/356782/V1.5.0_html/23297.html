50     package org.jpos.iso;
51     
52     import java.io.InputStream;
53     import java.io.PrintStream;
54     import java.util.BitSet;
55     
56     import org.jpos.iso.packager.XMLPackager;
57     
58     /**
59      * implements <b>Leaf</b> for Bitmap field
60      *
61      * @author apr@cs.com.uy
62      * @version $Id$
63      * @see ISOComponent
64      */
65     public class ISOBitMap extends ISOComponent implements Cloneable {
66         protected int fieldNumber;
67         protected BitSet value;
68     
69         /**
70          * @param n - the FieldNumber
71          */
72         public ISOBitMap (int n) {
73             fieldNumber = n;
74         }
75         /**
76          * @param n - fieldNumber
77          * @param v - field value (Bitset)
78          * @see BitSet
79          */
80         public ISOBitMap (int n, BitSet v) {
81             fieldNumber = n;
82             value = v;
83         }
84         /**
85          * changes this Component field number<br>
86          * Use with care, this method does not change
87          * any reference held by a Composite.
88          * @param fieldNumber new field number
89          */
90         public void setFieldNumber (int fieldNumber) {
91             this.fieldNumber = fieldNumber;
92         }
93         /**
94          * not available on Leaf - always throw ISOException
95          * @exception ISOException
96          */
97         public byte[] pack() throws ISOException {
98             throw new ISOException ("Not available on Leaf");
99         }
100         /**
101          * not available on Leaf - always throw ISOException
102          * @exception ISOException
103          */
104         public int unpack(byte[] b) throws ISOException {
105             throw new ISOException ("Not available on Leaf");
106         }
107         /**
108          * not available on Leaf - always throw ISOException
109          * @exception ISOException
110          */
111         public void unpack(InputStream in) throws ISOException {
112             throw new ISOException ("Not available on Leaf");
113         }
114         /**
115          * @return Object representing this field number
116          */
117         public Object getKey() {
118             return new Integer(fieldNumber);
119         }
120         /**
121          * @return Object representing this field value
122          */
123         public Object getValue() {
124             return value;
125         }
126         /**
127          * @param obj - Object representing this field value
128          * @exception ISOException
129          */
130         public void setValue(Object obj) throws ISOException {
131             value = (BitSet) obj;
132         }
133         /**
134          * dump this field to PrintStream. The output is sorta
135          * XML, intended to be easily parsed.
136          * @param p - print stream
137          * @param indent - optional indent string
138          */
139         public void dump (PrintStream p, String indent) {
140             p.println (indent +"<"+XMLPackager.ISOFIELD_TAG + " " +
141                 XMLPackager.ID_ATTR +"=\""+XMLPackager.TYPE_BITMAP+"\" "+
142                 XMLPackager.VALUE_ATTR +"=\"" +value+"\" "+
143                 XMLPackager.TYPE_ATTR +"=\"" + XMLPackager.TYPE_BITMAP+ "\"/>"
144             );
145         }
146     }
