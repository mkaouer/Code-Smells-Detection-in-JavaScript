50     package org.jpos.core;
51     
52     
53     /**
54      * SubConfiguration objects lets childs objects access attributes
55      * in its parent object with a given prefix, for example "child.".
56      * Child objects can access properties with their prefix removed.
57      *
58      * @author <a href="mailto:alcarraz@iie.edu.uy">Andr?s Alcarraz</a>
59      * @version $Id$
60      */
61     public class SubConfiguration implements Configuration{
62         /** Parent Configuration */
63         protected Configuration cfg;
64         /** prefix identifying the child parameters */
65         protected String prefix;
66         /** Creates a new empty SubConfiguration object */
67         public SubConfiguration() {
68             super();
69         }
70         /**
71          * Creates a new SubConfiguration from its parent's Configuration 
72          * and the a given <code>prefix</code>.
73          * @param cfg Parent's Configuration 
74          * @param prefix prefix identifying child parameters.
75          */
76         public SubConfiguration(Configuration cfg, String prefix) {
77             super ();
78             this.cfg = cfg;
79             this.prefix = prefix;
80         }
81         /**
82          * Sets the container configuration.
83          * @param newCfg New container configuration.
84          */
85         public void setConfiguration(Configuration newCfg){
86             cfg=newCfg;
87         }
88         /**
89          * Sets the prefix that identifies the parameters of the child object
90          * inside the parent configuration.
91          * @param newPrefix New prefix
92          */
93         public void setPrefix(String newPrefix){
94             prefix = newPrefix;
95         }
96         public String get(String propertyName){
97             return cfg.get(prefix + propertyName);
98         }
99         public String[] getAll(String propertyName){
100             return cfg.getAll(prefix + propertyName);
101         }
102         public String get(String propertyName, String defaultValue){
103             return cfg.get(prefix + propertyName, defaultValue);
104         }
105         public boolean getBoolean(String propertyName){
106             return cfg.getBoolean(prefix + propertyName);
107         }
108         public boolean getBoolean(String propertyName, boolean defaultValue){
109             return cfg.getBoolean(prefix + propertyName, defaultValue);
110         }
111         public double getDouble(String propertyName){
112             return cfg.getDouble(prefix + propertyName);
113         }
114         public double getDouble(String propertyName, double defaultValue){
115             return cfg.getDouble(prefix + propertyName, defaultValue);
116         }
117         public long getLong(String propertyName){
118             return cfg.getLong(prefix + propertyName);
119         }
120         public long getLong(String propertyName, long defaultValue){
121             return cfg.getLong(prefix + propertyName, defaultValue);
122         }
123         public int getInt(String propertyName){
124             return cfg.getInt(prefix + propertyName);
125         }
126         public int getInt(String propertyName, int defaultValue){
127             return cfg.getInt(prefix + propertyName, defaultValue);
128         }
129         public void put (String name, Object value) {
130             cfg.put (prefix + name, value);
131         }
132         /**
133          * Creates a new object, it takes the class from the value of the property
134          * <code>propertyName</code>
135          * @param propertyName Property whose value is the class name of  
136          * the object being created.
137          */
138         public Object getObject (String propertyName) throws ConfigurationException{
139             try{
140                 Class[] paramTypes = {};
141                 Object[] params = {};
142                 Object ret = 
143                     Class.forName (get (propertyName)).newInstance();
144                 if(ret instanceof Configurable) 
145                     ((Configurable)ret).setConfiguration(this);
146                 return ret;
147             } catch (Exception e){
148                 throw new ConfigurationException ("Error trying to create an " 
149                                                  + "object from property " 
150                                                  + prefix + propertyName, e
151                                                  );
152             }
153         }
154     }
