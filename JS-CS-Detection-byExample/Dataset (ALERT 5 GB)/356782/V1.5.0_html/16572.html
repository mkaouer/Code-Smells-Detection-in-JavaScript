75     /** This is a log listener that reads its actions from Bean Shell scripts.
76      * You can define many scripts, and the order in wich they are called, also you
77      * can define scripts whose name depends on the event being processed, and the
78      * realm of the object that generated it.
79      * This way you can set a configuration like this:
80      * <log-listener class="org.jpos.bsh.BSHLoglistener">
81      *    <property name="source" value="general.bsh"/>
82      *    <property name="source" value="tag_$tag.bsh"/> <!--this is to handle
83      * specific tags-->
84      *    <property name="source" value="realm_$realm.bsh"/> <!-- to handle specific
85      * realms-->
86      *    <property name="source" value="tag_$tag_realm_$realm.bsh"/> <!-- to handle
87      * specific tags from specific realms-->
88      * </log-listener>
89      * <!-- the following lines are for html doc-->
90      * <br>
91      * <pre>
92      * <font color="#008080">&lt;</font><font color="#008080">log-listener</font><font color="#008080"> </font><font color="#2e8b57"><b>class</b></font>=<font color="#ff00ff">&quot;org.jpos.bsh.BSHLoglistener&quot;</font><font color="#008080">&gt;</font>
93      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;general.bsh&quot;</font><font color="#008080">/&gt;</font>
94      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;tag_$tag.bsh&quot;</font><font color="#008080">/&gt;</font> <font color="#0000ff">&lt;!</font><font color="#0000ff">--this is to handle specific tags--</font><font color="#0000ff">&gt;</font>
95      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;realm_$realm.bsh&quot;</font><font color="#008080">/&gt;</font> <font color="#0000ff">&lt;!</font><font color="#0000ff">-- to handle specific realms--</font><font color="#0000ff">&gt;</font>
96      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;tag_$tag_realm_$realm.bsh&quot;</font><font color="#008080">/&gt;</font> <font color="#0000ff">&lt;!</font><font color="#0000ff">-- to handle specific tags from specific realms--</font><font color="#0000ff">&gt;</font>
97      * <font color="#008080">&lt;/log-listener&gt;</font>
98      * </pre>
99      * <!-- end of lines for html doc-->
100      * If a source with the given name is not found, or it canbe read, it is not processed, so this lets
101      * you change what is processed in real time, if you put a file called
102      * tag_SystemMonitor_realm_monitor.bsh it will be executed whenever the system
103      * monitor is run.<BR>
104      * If you want to filter an event so that the remaining log listeners don't see
105      * it, you have to set event = null in your script.<br>
106      * <table border=1 color="black">
107      * <caption>Other Configuration Options: </caption>
108      * <th><td> Name                </td><td>Type       </td><td>Description</td></th>
109      * <tr><td> filter-by-default   </td><td>boolean    </td><td>
110      *  If true, filter the events if no script is processed, this way you can put a
111      * Log Listener that filters every thing unless you put some script file, even if
112      * it is empty, you also can dynamically change what kind of message are filtered
113      * by placing empty files with names like tag_SystemMonitor_realm_monitor.bsh in
114      * the previous example, if you "touch" a file with this name these events will
115      * begin to be processed.</TD></TR>
116      * <TR><TD> preload-scripts     </TD><TD>boolean    </TD><TD>If true scripts a
117      * loaded once, and kept in memory, being realoaded only if they are touched. This
118      * is good when you have lots of RAM memory but ou have troubles with
119      * speed</TD></TR>
120      * <TR><TD> save-name-space     </TD><TD>boolean    </TD><TD>If true the namespace
121      * of the script instance will be saved so that in the next event you can access
122      * them from the script, by default it's off, this property is overriden if the
123      * script exposes a boolean variable named saveNameSpace</TD></TR>
124      * <TR><TD> reload              </TD><TD>long       </TD><TD>this property is used
125      * if the preload-script property is true, is the time in milliseconds between
126      * updates in the script, during this time BSHLogListener will not check if the
127      * script source was modified or deleted on disk </TD></TR>
128      * </TABLE>
129      */

142         protected static String[] replace(String[] src, String[] patterns, String[] to){
143             String[] ret = new String[src.length];
144             for(int i=0; i<src.length; i++){
145                 StringBuffer buff = new StringBuffer(2*src[i].length());
146                 int begin=0,end=0;
147                 //begin is the position of the next pattern, end is the end of the last pattern
148                 while ((begin = src[i].indexOf('$',end))>=0 && begin<src[i].length()){
149                     buff.append(src[i].substring(end, begin));
150                     boolean patternFound = false;
151                     for(int k=0; k<patterns.length && !patternFound ; k++){
152                         if(patternFound = (src[i].indexOf(patterns[k], begin) == begin+1)){
153                             buff.append(to[k]);
154                             end = begin + patterns[k].length() + 1;
155                         }
156                     }
157                     if(!patternFound) end = begin + 1;
158                 }
159                 buff.append(src[i].substring(end));
160                 //if(buff.length()==0) ret[i] = src[i];
161                 ret[i] = buff.toString();
162             }
163             return ret;
164         }
165         public LogEvent log(org.jpos.util.LogEvent ev) {
166             LogEvent ret = ev;
167             boolean processed = false;
168             try{
169                 String[] sources = replace(cfg.getAll("source"), patterns, new String[] {ev.tag, ev.getRealm()});
170                 for(int i=0; i<sources.length && ret != null; i++){
171                     try{
172                         Interpreter bsh = new Interpreter();
173                         BSHLogListener.ScriptInfo info = getScriptInfo(sources[i]);
174                         NameSpace ns = (info!=null)?info.getNameSpace():null;
175                         if(ns!=null) bsh.setNameSpace(ns);
176                         bsh.set("event", ret);
177                         bsh.set("cfg", cfg);
178                         File f = new File(sources[i]);
179                         if(!cfg.getBoolean("preload-scripts")){
180                             if(f.exists() && f.canRead() && f.isFile()){
181                                 //if(f.lastModified())
182                                 processed = true;
183                                 bsh.eval(new java.io.FileReader(f));
184                             }
185                         }else{
186                             if(info == null) scripts.put(sources[i], info=new ScriptInfo());
187                             String code;
188                             if(System.currentTimeMillis() > info.getLastCheck() + cfg.getLong("reload")){
189                                 info.setLastCheck(System.currentTimeMillis());
190                                 if(f.exists() && f.canRead() && f.isFile()){
191                                     if(info.getLastModified() != f.lastModified()) {
192                                         info.setLastModified(f.lastModified());
193                                         info.setCode(loadCode(f));
194                                     }
195                                 }else{
196                                     info.setCode(null);
197                                 }
198                             }
199                             if(info.getCode() != null){
200                                 processed = true;
201                                 bsh.eval(new StringReader(info.getCode()));
202                             }else scripts.remove(sources[i]);
203                         }
204                         ret = (LogEvent)bsh.get("event");
205                         Object saveNS = bsh.get("saveNameSpace");
206                         boolean saveNameSpace = 
207                             (saveNS instanceof Boolean)?((Boolean)saveNS).booleanValue():cfg.getBoolean("save-name-space");
208                         if(saveNameSpace) {
209                             if(info!=null) info.setNameSpace(bsh.getNameSpace());
210                             else scripts.put(sources[i], new ScriptInfo(bsh.getNameSpace()));
211                         }else if (info!=null) info.setNameSpace(null);
212                     }catch(Exception e){
213                         ret.addMessage(e);
214                     }
215                 }
216                 return (!processed && cfg.getBoolean("filter-by-default"))?null:ret;
217             }catch(Exception e){
218                 ret.addMessage(e);
219                 return ret;
220             }
221         }
222         protected String loadCode(File f) throws IOException{
223             FileReader r = new FileReader(f);
224             StringBuffer buf = new StringBuffer((int)f.length());
225             char[] content = new char[(int)f.length()];
226             int l;
227             while((l=r.read(content))!=-1) buf.append(content,0,l);
228             return buf.toString();
229         }

232             return (ScriptInfo)scripts.get(filename);
233         }
234         protected void addScriptInfo(String filename, String code, long lastModified){
