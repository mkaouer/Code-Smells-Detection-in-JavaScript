19     /**
20      * send e-mail with selected LogEvents to operator account
21      * <b>Configuration properties</b>
22      * <pre>
23      *    jpos.operator.from=jpos
24      *    jpos.operator.to=operator@foo.bar
25      *    jpos.operator.subject.prefix=[jPOS]
26      *    jpos.operator.tags="Operator ISORequest SystemMonitor"
27      *    jpos.operator.delay=10000
28      *    jpos.mail.smtp.host=localhost
29      * </pre>
30      *
31      * @author apr@cs.com.uy
32      * @version $Id$
33      */
34     public class OperatorLogListener 
35         implements LogListener, Configurable, Runnable
36     {
37         Configuration cfg;
38         BlockingQueue queue;
39         boolean logExceptions;
40     
41         public OperatorLogListener () {
42             super();
43             queue = new BlockingQueue();
44         }
45         public OperatorLogListener (Configuration cfg) {
46             super();
47             this.cfg = cfg;
48             queue = new BlockingQueue();
49             new Thread(this).start();
50         }
51         public void setConfiguration (Configuration cfg) 
52             throws ConfigurationException
53         {
54             this.cfg = cfg;
55             assertProperty ("jpos.operator.to");
56             assertProperty ("jpos.operator.subject.prefix");
57             assertProperty ("jpos.operator.tags");
58             assertProperty ("jpos.operator.delay");
59             assertProperty ("jpos.mail.smtp.host");
60     
61             logExceptions = cfg.get ("jpos.operator.tags").
62                 indexOf ("exception") >= 0;
63     
64             new Thread(this).start();
65         }
66         public void run() {
67             Thread.currentThread().setName ("OperatorLogListener");
68             int delay = cfg.getInt ("jpos.operator.delay");
69             try {
70                 ISOUtil.sleep (2500);   // initial delay
71                 for (;;) {
72                     try {
73                         LogEvent ev[] = new LogEvent[1];
74                         if (queue.pending() > 0) {
75                             ev = new LogEvent [queue.pending()];
76                             for (int i=0; i < ev.length; i++)
77                                 ev[i] = (LogEvent) queue.dequeue();
78                         } else 
79                             ev[0] = (LogEvent) queue.dequeue();
80                         sendMail (ev);
81                         if (delay > 0)
82                             Thread.sleep (delay);
83                     } catch (InterruptedException e) { }
84                 }
85             } catch (BlockingQueue.Closed e) { }
86         }
87         private void sendMail (LogEvent[] ev) {
88             String from    = cfg.get ("jpos.operator.from", "jpos-logger");
89             String[] to    = cfg.getAll ("jpos.operator.to");
90             String[] cc    = cfg.getAll ("jpos.operator.cc");
91             String[] bcc   = cfg.getAll ("jpos.operator.bcc");
92             String subject = cfg.get ("jpos.operator.subject.prefix");
93             if (ev.length > 1) 
94                 subject = subject + ev.length + " events";
95             else
96                 subject = subject + ev[0].getRealm() + " - " +ev[0].tag;
97     
98             // create some properties and get the default Session
99             Properties props = System.getProperties();
100             props.put("mail.smtp.host", cfg.get ("jpos.mail.smtp.host", 
101                     "localhost"));
102             
103             Session session = Session.getDefaultInstance(props, null);
104             session.setDebug(false);
105             
106             try {
107                 // create a message
108                 MimeMessage msg = new MimeMessage(session);
109                 msg.setFrom (new InternetAddress(from));
110     
111                 InternetAddress[] address = new InternetAddress[to.length];
112                 for (int i=0; i<to.length; i++) 
113                     address[i] = new InternetAddress (to[i]);
114                 msg.setRecipients (Message.RecipientType.TO, getAddress (to));
115                 msg.setRecipients (Message.RecipientType.CC, getAddress (cc));
116                 msg.setRecipients (Message.RecipientType.BCC, getAddress (bcc));
117                 msg.setSubject(subject);
118                 Multipart mp = new MimeMultipart();
119     
120                 for(int i=0; i<ev.length; i++) {
121                     ByteArrayOutputStream buf = new ByteArrayOutputStream();
122                     PrintStream p = new PrintStream (buf);
123                     ev[i].dump (p, "");
124                     p.close();
125             
126                     // create and fill the first message part
127                     MimeBodyPart mbp = new MimeBodyPart();
128                     mbp.setText(buf.toString());
129                     mbp.setFileName (ev[i].tag + "_" + i + ".txt");
130                     mp.addBodyPart(mbp);
131                 }
132                 msg.setContent(mp);
133                 msg.setSentDate(new Date());
134                 Transport.send(msg);
135             } catch (MessagingException mex) {
136                 mex.printStackTrace();
137                 Exception ex = null;
138                 if ((ex = mex.getNextException()) != null) {
139                     ex.printStackTrace();
140                 }
141             }
142         }
143         private boolean checkOperatorTag(LogEvent ev) {
144             String tags = cfg.get ("jpos.operator.tags");
145     
146             return (tags.indexOf (ev.tag) >= 0) || (logExceptions && hasException (ev));
147         }
148         private boolean hasException (LogEvent evt) {
149             Iterator iter = evt.getPayLoad().iterator();
150             while (iter.hasNext()) {
151                 if (iter.next() instanceof Throwable)
152                     return true;
153             }
154             return false;
155         }
156         private InternetAddress[] getAddress (String[] s) throws AddressException {
157             InternetAddress[] address = new InternetAddress[s.length];
158             for (int i=0; i<s.length; i++) 
159                 address[i] = new InternetAddress (s[i]);
160             return address;
161         }
162         public synchronized LogEvent log (LogEvent ev) {
163             if (checkOperatorTag(ev))
164                 queue.enqueue (ev);
165             return ev;
166         }
167         private void assertProperty (String propName) throws ConfigurationException
168         {
169             if (cfg.get (propName) == null)
170                 throw new ConfigurationException 
171                     (propName + " property not present");
172         }
