7     package org.jpos.util;
8     
9     import java.util.Iterator;
10     
11     /**
12      * A specific log listener that filters all LogEvents that doesn't
13      * contain any exception.
14      * @author Alejandro Revilla
15      * @version $Revision$ $Date$
16      */
17     public class ExceptionLogFilter implements LogListener {
18         public ExceptionLogFilter () {
19             super();
20         }
21         public synchronized LogEvent log (LogEvent evt) {
22             Iterator iter = evt.getPayLoad().iterator();
23             while (iter.hasNext()) {
24                 if (iter.next() instanceof Throwable)
25                     return evt;
26             }
27             return null;
28         }
29     }
