24     public class TSpace extends TimerTask implements LocalSpace {
25         protected Map entries;
26         protected TSpace sl;    // space listeners
27         public static final long GCDELAY = 60*1000;
28     
29         public TSpace () {
30             super();
31             entries = new HashMap ();
32             DefaultTimer.getTimer().schedule (this, GCDELAY, GCDELAY);
33         }
34         public synchronized void out (Object key, Object value) {
35             getList (key).add (value);
36             this.notifyAll ();
37             if (sl != null)
38                 notifyListeners(key, value);
39         }
40         public void out (Object key, Object value, long timeout) {
41             if (timeout > 0) {
42                 value   = new Expirable (value, 
43                    System.currentTimeMillis() + timeout);
44             }
45             synchronized (this) {
46                 getList (key).add (value);
47                 this.notifyAll ();
48             }
49             if (sl != null)
50                 notifyListeners(key, value);
51         }
52         public synchronized Object rdp (Object key) {
53             if (key instanceof Template)
54                 return getObject ((Template) key, false);
55             return getHead (key, false);
56         }
57         public synchronized Object inp (Object key) {
58             if (key instanceof Template)
59                 return getObject ((Template) key, true);
60             return getHead (key, true);
61         }
62         public synchronized Object in (Object key) {
63             Object obj;
64             while ((obj = inp (key)) == null) {
65                 try {
66                     this.wait ();
67                 } catch (InterruptedException e) { }
68             }
69             return obj;
70         }
71         public synchronized Object in  (Object key, long timeout) {
72             Object obj;
73             long now = System.currentTimeMillis();
74             long end = now + timeout;
75             while ((obj = inp (key)) == null && 
76                     ((now = System.currentTimeMillis()) < end))
77             {
78                 try {
79                     this.wait (end - now);
80                 } catch (InterruptedException e) { }
81             }
82             return obj;
83         }
84         public synchronized Object rd  (Object key) {
85             Object obj;
86             while ((obj = rdp (key)) == null) {
87                 try {
88                     this.wait ();
89                 } catch (InterruptedException e) { }
90             }
91             return obj;
92         }
93         public synchronized Object rd  (Object key, long timeout) {
94             Object obj;
95             long now = System.currentTimeMillis();
96             long end = now + timeout;
97             while ((obj = rdp (key)) == null && 
98                     ((now = System.currentTimeMillis()) < end))
99             {
100                 try {
101                     this.wait (end - now);
102                 } catch (InterruptedException e) { }
103             }
104             return obj;
105         }
106         public void run () {
107             gc();
108         }
109         public void gc () {
110             Object[] keys;
111             synchronized (this) {
112                 keys = entries.keySet().toArray();
113             }
114             for (int i=0; i<keys.length; i++) {
115                 rdp (keys[i]);
116                 Thread.yield ();
117             }
118             if (sl != null) {
119                 sl.gc();
120                 synchronized (this) {
121                     if (sl != null && sl.isEmpty())
122                         sl = null;
123                 }
124             }
125         }
126         public synchronized int size (Object key) {
127             int size = 0;
128             List l = (List) entries.get (key);
129             if (l != null) 
130                 size = l.size();
131             return size;
132         }
133         public synchronized void addListener (Object key, SpaceListener listener) {
134             getSL().out (key, listener);
135         }
136         public synchronized void addListener 
137             (Object key, SpaceListener listener, long timeout) 
138         {
139             getSL().out (key, listener, timeout);
140         }
141         public synchronized void removeListener 
142             (Object key, SpaceListener listener) 
143         {
144             if (sl != null) {
145                 sl.inp (new ObjectTemplate (key, listener));
146             }
147         }
148         public boolean isEmpty() {
149             return entries.isEmpty();
150         }
151         public Set getKeySet() {
152             return entries.keySet();
153         }
154         public String getKeysAsString () {
155             StringBuffer sb = new StringBuffer();
156             Object[] keys;
157             synchronized (this) {
158                 keys = entries.keySet().toArray();
159             }
160             for (int i=0; i<keys.length; i++) {
161                 if (i > 0)
162                     sb.append (' ');
163                 sb.append (keys[i]);
164             }
165             return sb.toString();
166         }
167         public void notifyListeners (Object key, Object value) {
168             Object[] listeners = null;
169             synchronized (this) {
170                 if (sl == null)
171                     return;
172                 List l = (List) sl.entries.get (key);
173                 if (l != null)
174                     listeners = l.toArray();
175             }
176             if (listeners != null) {
177                 for (int i=0; i<listeners.length; i++) {
178                     Object o = listeners[i];
179                     if (o instanceof Expirable)
180                         o = ((Expirable)o).getValue();
181                     if (o instanceof SpaceListener)
182                         ((SpaceListener) o).notify (key, value);
183                 }
184             }
185         }
186         private List getList (Object key) {
187             List l = (List) entries.get (key);
188             if (l == null) 
189                 entries.put (key, l = new LinkedList());
190             return l;
191         }
192         private Object getHead (Object key, boolean remove) {
193             Object obj = null;
194             List l = (List) entries.get (key);
195             while (obj == null && l != null && l.size() > 0) {
196                 obj = l.get(0);
197                 if (obj instanceof Expirable) {
198                     obj = ((Expirable) obj).getValue();
199                     if (obj == null) {
200                         l.remove (0);
201                         if (l.size() == 0) {
202                             entries.remove (key);
203                         }
204                     }
205                 }
206             }
207             if (obj != null && remove) {
208                 l.remove (0);
209                 if (l.size() == 0) {
210                     entries.remove (key);
211                 }
212             }
213             return obj;
214         }
215         private Object getObject (Template tmpl, boolean remove) {
216             Object obj = null;
217             List l = (List) entries.get (tmpl.getKey());
218             if (l == null)
219                 return obj;
220     
221             Iterator iter = l.iterator();
222             while (iter.hasNext()) {
223                 obj = iter.next();
224                 if (obj instanceof Expirable) {
225                     obj = ((Expirable) obj).getValue();
226                     if (obj == null) {
227                         iter.remove();
228                         continue;
229                     }
230                 }
231                 if (tmpl.equals (obj)) {
232                     if (remove)
233                         iter.remove();
234                     break;
235                 } else
236                     obj = null;
237             }
238             return obj;
239         }
240         private TSpace getSL() {
241             synchronized (this) {
242                 if (sl == null)
243                     sl = new TSpace();
244             }
245             return sl;
246         }
247         static class Expirable implements Comparable {
248             Object value;
249             long expires;
250     
251             public Expirable (Object value, long expires) {
252                 super();
253                 this.value = value;
254                 this.expires = expires;
255             }
256             public boolean isExpired () {
257                 return expires < System.currentTimeMillis ();
258             }
259             public String toString() {
260                 return getClass().getName() 
261                     + "@" + Integer.toHexString(hashCode())
262                     + ",value=" + value.toString()
263                     + ",expired=" + isExpired ();
264             }
265             public Object getValue() {
266                 return isExpired() ? null : value;
267             }
268             public int compareTo (Object obj) {
269                 Expirable other = (Expirable) obj;
270                 long otherExpires = other.expires;
271                 if (otherExpires == expires)
272                     return 0;
273                 else if (expires < otherExpires)
274                     return -1;
275                 else 
276                     return 1;
277             }
278         }
