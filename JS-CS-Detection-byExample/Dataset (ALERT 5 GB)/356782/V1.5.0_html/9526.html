50     package org.jpos.iso.channel;
51     
52     import java.io.IOException;
53     import java.net.ServerSocket;
54     
55     import org.jpos.iso.BaseChannel;
56     import org.jpos.iso.ISOException;
57     import org.jpos.iso.ISOMsg;
58     import org.jpos.iso.ISOPackager;
59     import org.jpos.iso.ISOUtil;
60     import org.jpos.util.LogEvent;
61     import org.jpos.util.Logger;
62     
63     /**
64      * Implements an ISOChannel able to exchange messages with
65      * ACI's BASE24 over a TCP link, modified from BASE24ISOChannel 
66      * by Victor A. Salaman (salaman@teknos.com) .<br>
67      * An instance of this class exchanges messages by means of an
68      * intermediate 'port server' as described in the
69      * <a href="/doc/javadoc/overview-summary.html">Overview</a> page.
70      * @author apr@cs.com.uy
71      * @author salaman@teknos.com
72      *
73      * @version $Id$
74      *
75      * @see ISOMsg
76      * @see ISOException
77      * @see ISOChannel
78      */
79     
80     public class BASE24TCPChannel extends BaseChannel {
81         /**
82          * Public constructor (used by Class.forName("...").newInstance())
83          */
84         public BASE24TCPChannel () {
85             super();
86         }
87         /**
88          * Construct client ISOChannel
89          * @param host  server TCP Address
90          * @param port  server port number
91          * @param p     an ISOPackager
92          * @see ISOPackager
93          */
94         public BASE24TCPChannel (String host, int port, ISOPackager p) {
95             super(host, port, p);
96         }
97         /**
98          * Construct server ISOChannel
99          * @param p     an ISOPackager
100          * @see ISOPackager
101          * @exception IOException
102          */
103         public BASE24TCPChannel (ISOPackager p) throws IOException {
104             super(p);
105         }
106         /**
107          * constructs a server ISOChannel associated with a Server Socket
108          * @param p     an ISOPackager
109          * @param serverSocket where to accept a connection
110          * @exception IOException
111          * @see ISOPackager
112          */
113         public BASE24TCPChannel (ISOPackager p, ServerSocket serverSocket) 
114             throws IOException
115         {
116             super(p, serverSocket);
117         }
118         /**
119          * @param m the Message to send (in this case it is unused)
120          * @param len   message len (ignored)
121          * @exception IOException
122          */
123         protected void sendMessageTrailler(ISOMsg m, int len) throws IOException {
124             serverOut.write (3);
125         }
126         protected void sendMessageLength(int len) throws IOException {
127             len++;  // one byte trailler
128             serverOut.write (len >> 8);
129             serverOut.write (len);
130         }
131         protected int getMessageLength() throws IOException, ISOException {
132             int l = 0;
133             byte[] b = new byte[2];
134             Logger.log (new LogEvent (this, "get-message-length"));
135             while (l == 0) {
136                 serverIn.readFully(b,0,2);
137                 l = ((((int)b[0])&0xFF) << 8) | (((int)b[1])&0xFF);
138                 if (l == 0) {
139                     serverOut.write(b);
140                     serverOut.flush();
141                 }
142             }
143             Logger.log (new LogEvent (this, "got-message-length", Integer.toString(l)));
144             return l - 1;   // trailler length
145         }
146         protected void getMessageTrailler() throws IOException {
147             Logger.log (new LogEvent (this, "get-message-trailler"));
148             byte[] b = new byte[1];
149             serverIn.readFully(b,0,1);
150             Logger.log (new LogEvent (this, "got-message-trailler", ISOUtil.hexString(b)));
151         }
152     }
