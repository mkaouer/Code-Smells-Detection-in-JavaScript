73     /**
74      * QSP Router implements ISORequestListener
75      * and forward all incoming messages to the given
76      * destination MUX according to the conditions, handling back responses
77      *
78      * @author <a href="mailto:tzymail@163.com">Zhiyu Tang</a>
79      * @version $Revision$ $Date$
80      * @see org.jpos.iso.ISORequestListener
81      */
82     public class Router
83         implements ISORequestListener, LogSource, NodeConfigurable
84     {
85       Logger logger;
86       String realm;
87       String condition[] = null;
88       String type[] = null;
89       int[] timeout = null;
90       Hashtable dest = null;
91       boolean[] bounce = null;
92       int condNum;
93     
94       ThreadPool pool;
95     
96       public Router () {
97         super();
98         pool = new ThreadPool (1, 100);
99         dest = new Hashtable();
100       }
101     
102       public void setLogger (Logger logger, String realm) {
103         this.logger = logger;
104         this.realm  = realm;
105       }
106     
107       public String getRealm () {
108         return realm;
109       }
110     
111       public Logger getLogger() {
112         return logger;
113       }
114     
115        /**
116         * Destination can be a Channel or a MUX. If Destination is a Channel
117         * then timeout applies (used on ISORequest to get a Response).
118         * <ul>
119         * <li>destination-mux
120         * <li>destination-channel
121         * <li>timeout
122         * <li>bounce
123         * </ul>
124         * @param node Configuration
125         */
126       public void setConfiguration (Node node)
127           throws ConfigurationException
128       {
129         NodeList nodes = node.getChildNodes();
130         condNum = nodes.getLength();
131         int j = 0;
132         for( int i=0 ; i < condNum; i++ ){
133           if (nodes.item(i).getNodeName().equals ("router"))
134             j++;
135         }
136     
137         condition = new String[j];
138         type = new String[j];
139         timeout = new int[j];
140         bounce = new boolean[j];
141         j = 0;
142         for( int i=0 ; i < condNum; i++ ){
143           if (nodes.item(i).getNodeName().equals ("router")) {
144             condition[j] = nodes.item(i).getAttributes().
145                    getNamedItem ("switch").getNodeValue();
146             type[j] = nodes.item(i).getAttributes().
147                getNamedItem ("type").getNodeValue();
148             String destination = nodes.item(i).getAttributes().
149                           getNamedItem ("destination").getNodeValue();
150             Node attrnode = nodes.item(i).getAttributes().
151                             getNamedItem ("timeout");
152             if( attrnode != null ){
153               timeout[j] = Integer.parseInt( attrnode.getNodeValue() );
154             }
155             else
156               timeout[j] = 0;
157             attrnode =  nodes.item(i).getAttributes().getNamedItem ("bounce");
158             if( attrnode != null ){
159               if( attrnode.getNodeValue().equals("true") )
160                 bounce[j] = true;
161             }else
162               bounce[j] = false;
163             try {
164               dest.put( String.valueOf(j) ,
165                         NameRegistrar.get (type[j]+"."+destination));
166             } catch (NotFoundException e) {
167               throw new ConfigurationException (e);
168             }
169             j++;
170           }
171         }
172         condNum = j;
173       }
174     
175         /**
176          * hook used to optional bounce an unanswered message
177          * to its source channel
178          * @param s message source
179          * @param m unanswered message
180          * @exception ISOException
181          * @exception IOException
182          */
183       protected void processNullResponse (ISOSource s, ISOMsg m,
184                                           LogEvent evt,boolean bounce)
185           throws ISOException, IOException
186       {
187         if ( bounce ) {
188           ISOMsg c = (ISOMsg) m.clone();
189           c.setResponseMTI();
190           if (c.hasField (39))
191             c.unset (39);
192             s.send (c);
193             evt.addMessage ("<bounced/>");
194         } else
195           evt.addMessage ("<null-response/>");
196       }

198     
199       private int getDestination( LogEvent evt , ISOMsg m  , JepUtil jeputil)
200           throws ISOException
201       {
202         for( int i = 0 ; i < condNum ; i++ )
203         {
204            if( jeputil.getResultBoolean(m , condition[i]) )
205              return i;
206         }
207         return -1;
208       }
209     
210       protected class Process implements Runnable {
211         ISOSource source;
212         ISOMsg m;
213         JepUtil jeputil;
214     
215         Process (ISOSource source, ISOMsg m) {
216           super();
217           this.source = source;
218           this.m = m;
219           jeputil = new JepUtil();
220         }
221         
222         public void run (){
223           LogEvent evt = new LogEvent ( Router.this,
224                     "Router");
225           try {
226             ISOMsg c = (ISOMsg) m.clone();
227             evt.addMessage (c);
228             int direction = getDestination( evt, c , jeputil );
229             if( direction == -1 ) return;
230             String key = String.valueOf( direction );
231             Object destobj = dest.get(key);
232     
233             if ( type[direction].equals ("mux") ) {
234                 ISOMUX destMux = (ISOMUX)( destobj );
235                 if ( timeout[direction] > 0) {
236                     ISOMsg response = null;
237                     if ( destMux.isConnected()) {
238                         ISORequest req = new ISORequest (c);
239                         destMux.queue (req);
240                         evt.addMessage ("<queued/>");
241                         response = req.getResponse (timeout[direction]);
242                     } else
243                         evt.addMessage ("<mux-not-connected/>");
244                     if (response != null) {
245                         evt.addMessage ("<got-response/>");
246                         evt.addMessage (response);
247                         response.setHeader (c.getISOHeader());
248                         source.send(response);
249                     } else {
250                         processNullResponse (source, m, evt, bounce[direction]);
251                     }
252                 } else {
253                             evt.addMessage ("<sent-through-mux/>");
254                             destMux.send (c);
255                         }
256                     } else if (type[direction].equals ("channel")) {
257                         evt.addMessage ("<sent-to-channel/>");
258                         ISOChannel destChannel = (ISOChannel)( destobj );
259                         destChannel.send (c);
260                     }
261                 } catch (ISOException e) {
262                     evt.addMessage (e);
263                 } catch (IOException e) {
264                     evt.addMessage (e);
265                 }
266                 Logger.log (evt);
267             }
268         }
269     
270         public boolean process (ISOSource source, ISOMsg m) {
271             pool.execute (new Process (source, m));
272             return true;
273         }
274     }
