65     public class Test implements Runnable, ReConfigurable, LogSource, Loggeable {
66         Configuration cfg;
67         Logger logger;
68         String realm;
69         public ConnectionPool connectionPool = null;
70     
71     
72         public void run() {
73           NameRegistrar.register (getRealm(), this);
74           LogEvent evt = new LogEvent(this, getRealm(), this);
75           logger.log (evt);
76     
77           //this is where you do the work..
78     
79           //The following code is a simple example to show you the pooling effect.
80           //In most cases, you would be spawning separate threads and sharing
81           //the pool amongst all the threads instead.
82     
83           //You can get connections from the pool,
84           // using connectionPool.getConnection() method.
85           //if wait-if-busy = false, a SQLException is thrown when the
86           //pool limit is reached.
87           //if wait-if-busy = true, then the getConnection() will block, till a
88           //connection is available.
89     
90           java.sql.Connection conn1 = null;
91           java.sql.Connection conn2 = null;
92           java.sql.Connection conn3 = null;
93           java.sql.Connection conn4 = null;
94           try {
95             conn1 = connectionPool.getConnection();
96             conn2 = connectionPool.getConnection();
97             conn3 = connectionPool.getConnection();
98             conn4 = connectionPool.getConnection();
99           } catch(java.sql.SQLException e) {
100             //If waitIfBusy is false, then SQLException is thrown on reaching the pool limit
101             logger.log(new LogEvent(this, getRealm(), e));
102          } finally {
103             //free all connections
104             if(connectionPool != null) {
105               if(conn1 != null) connectionPool.free(conn1);
106               if(conn2 != null) connectionPool.free(conn2);
107               if(conn3 != null) connectionPool.free(conn3);
108               if(conn4 != null) connectionPool.free(conn4);
109             }
110           }
111         }
112     
113        /**
114         * implements [Re]Configurable
115         * @param cfg Configuration object (called by QSP)
116         */
117         public void setConfiguration (Configuration cfg)
118             throws ConfigurationException
119         {
120             this.cfg = cfg;
121     
122     
123             /** Setup Connection Pooling
124               */
125             try {
126     //          connectionPool = (ConnectionPool)
127     //              NameRegistrar.get ("connection.pool."+cfg.get("connection-pool"));
128               connectionPool = ConnectionPool.getConnectionPool(cfg.get("connection-pool"));
129             } catch (Exception e) {
130               throw new ConfigurationException(e);
131             }
132         }
