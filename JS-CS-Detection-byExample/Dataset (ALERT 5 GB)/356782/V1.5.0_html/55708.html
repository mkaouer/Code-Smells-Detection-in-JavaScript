90     package org.jpos.iso.gui;
91     
92     import java.awt.BorderLayout;
93     import java.awt.Dimension;
94     import java.awt.FlowLayout;
95     import java.awt.Insets;
96     import java.awt.event.MouseAdapter;
97     import java.awt.event.MouseEvent;
98     import java.awt.event.MouseListener;
99     import java.util.Observable;
100     import java.util.Observer;
101     
102     import javax.swing.BorderFactory;
103     import javax.swing.DefaultListModel;
104     import javax.swing.JFrame;
105     import javax.swing.JLabel;
106     import javax.swing.JList;
107     import javax.swing.JPanel;
108     import javax.swing.JScrollPane;
109     import javax.swing.ListModel;
110     
111     import org.jpos.iso.ISOException;
112     import org.jpos.iso.ISOMsg;
113     import org.jpos.iso.ISORequestListener;
114     
115     public class ISORequestListenerPanel extends JPanel implements Observer {
116         /**
117          * @serial
118          */
119         DefaultListModel log;
120         /**
121          * @serial
122          */
123         String symbolicName;
124         public static final int LOG_CAPACITY = 250;
125     
126         public ISORequestListenerPanel (
127             ISORequestListener requestListener,
128             String symbolicName)
129         {
130             super();
131             this.symbolicName = symbolicName;
132             setLayout(new FlowLayout());
133             setBorder(BorderFactory.createRaisedBevelBorder());
134             log = new DefaultListModel();
135             add(createPanel());
136             if (requestListener instanceof Observable)
137                 ((Observable)requestListener).addObserver(this);
138         }
139         public final String getSymbolicName() {
140             return symbolicName;
141         }
142         public final ListModel getLog() {
143             return log;
144         }
145     
146         public void update(Observable o, Object arg) {
147             ISORequestListener l = (ISORequestListener) o;
148             if (arg != null && arg instanceof ISOMsg) {
149                 ISOMsg m = (ISOMsg) arg;
150                 try {
151                     String mti = (String) m.getValue(0);
152                     int imti   = Integer.parseInt(mti);
153                     log.addElement(m);
154                     if (log.getSize() > LOG_CAPACITY) 
155                         log.remove(0);
156                 } catch (ISOException e) { }
157             }
158         }
159     
160         private JPanel createPanel() {
161             JPanel A = new JPanel() {
162                 public Insets getInsets() {
163                     return new Insets(10,10,10,10);
164                 }
165             };
166     
167             A.setLayout(new BorderLayout());
168     
169             JLabel l = new JLabel(symbolicName);
170             A.add(l, BorderLayout.NORTH);
171     
172             final JList logList = new JList(log);
173             logList.setPrototypeCellValue("9999 99999999 999999");
174     
175             MouseListener mouseListener = new MouseAdapter() {
176                 public void mouseClicked(MouseEvent e) {
177                     ISOMsg m = (ISOMsg) logList.getSelectedValue();
178                     if (m != null) {
179                         JFrame f = new JFrame(m.toString());
180                         ISOMsgPanel p = new ISOMsgPanel(m);
181                         f.getContentPane().add(p);
182                         f.pack();
183                         f.show();
184                     }
185                 }
186             };
187             logList.addMouseListener(mouseListener);
188     
189             JScrollPane scrollPane = new JScrollPane(logList);
190             scrollPane.setPreferredSize(new Dimension(170,200));
191             A.add(scrollPane, BorderLayout.SOUTH);
192             return A;
193         }
194     }
