52     package isomsg;
53     
54     import java.util.Date;
55     import java.io.PrintStream;
56     import org.jpos.util.Logger;
57     import org.jpos.util.LogEvent;
58     import org.jpos.util.SimpleLogSource;
59     import org.jpos.util.SimpleLogListener;
60     
61     import org.jpos.iso.ISOMsg;
62     import org.jpos.iso.ISOUtil;
63     import org.jpos.iso.ISODate;
64     import org.jpos.iso.ISOField;
65     import org.jpos.iso.ISOException;
66     
67     public class Test extends SimpleLogSource {
68         int[] mask  = { 0, 3, 7 };
69         int[] mask1 = { 0, 3, 7, 41, 42 };
70     
71         public Test (Logger logger, String realm) {
72             setLogger (logger, realm);
73         }
74         private ISOMsg createMessage(int[] mask) throws ISOException 
75         {
76             Date d = new Date();
77             ISOMsg m = new ISOMsg("0800");
78             m.set (new ISOField (3,  "000000"));
79             m.set (new ISOField (7,  ISODate.getDateTime(d)));
80     
81             // alternate way of setting fields
82             m.set (11, "000001");
83             m.set (12, ISODate.getTime(d));
84             m.set (13, ISODate.getDate(d));
85             return mask == null ? m : (ISOMsg) m.clone(mask);
86         }
87         public void test() {
88             LogEvent evt = new LogEvent (this, "Test");
89             try {
90                 ISOMsg a = createMessage(null);
91                 evt.addMessage ("Original Message A (full):");
92                 evt.addMessage (a);
93     
94                 ISOMsg b = (ISOMsg) a.clone();
95                 b.set (new ISOField (41, "12345678"));
96                 b.set (new ISOField (42, "123456789012345"));
97                 evt.addMessage 
98                     ("Message B is message A's clone plus fields 41 and 42");
99                 evt.addMessage (b);
100     
101                 ISOMsg c = createMessage (mask);
102                 evt.addMessage ("Message C just fields 0, 3 and 7");
103                 evt.addMessage (c);
104     
105                 ISOMsg d = (ISOMsg) c.clone();
106                 d.merge (b);
107                 evt.addMessage ("Message D is C merged with B");
108                 evt.addMessage (d);
109     
110                 ISOMsg e = (ISOMsg) d.clone(mask1);
111                 evt.addMessage (
112                     "Message E == message (D and mask1) (clone(int[])");
113                 evt.addMessage (e);
114             } catch (ISOException ex) {
115                 evt.addMessage (ex);
116             }
117             Logger.log (evt);
118         }
119         public void testNested() {
120             LogEvent evt = new LogEvent (this, "TestNested");
121             try {
122                 ISOMsg m      = createMessage(null);
123                 ISOMsg inner  = new ISOMsg (127); // goes at outter field 127
124                 inner.set (new ISOField (0,"001"));
125                 inner.set (new ISOField (2,"002"));
126                 inner.set (new ISOField (3,"003"));
127                 m.set (inner);
128                 evt.addMessage (m);
129             } catch (ISOException ex) {
130                 evt.addMessage (ex);
131             }
132             Logger.log (evt);
133         }
134         public void testRenumber() {
135             LogEvent evt = new LogEvent (this, "TestRenumber");
136             try {
137                 ISOMsg m      = createMessage(null);
138                 ISOMsg inner  = new ISOMsg (127); // goes at outter field 127
139                 inner.set (new ISOField (0,"001"));
140                 inner.set (new ISOField (2,"002"));
141                 inner.set (new ISOField (3,"003"));
142                 m.set (inner);
143                 evt.addMessage (m);
144                 Logger.log (evt);
145     
146                 m.move (127, 126);
147                 evt.addMessage ("Field 127 renumbered to 126");
148             } catch (ISOException ex) {
149                 evt.addMessage (ex);
150             }
151             Logger.log (evt);
152         }
153         public static void main (String args[]) {
154             Logger logger = new Logger();
155             logger.addListener (new SimpleLogListener (System.out));
156     
157             Test t = new Test (logger, "Test");
158             t.test();
159             t.testNested();
160             t.testRenumber();
161         }
162     }
