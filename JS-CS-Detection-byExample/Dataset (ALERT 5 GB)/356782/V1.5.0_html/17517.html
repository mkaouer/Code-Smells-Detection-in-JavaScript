49     package org.jpos.iso;
50     
51     import java.io.IOException;
52     import java.io.InputStream;
53     
54     /**
55      * @author joconnor
56      * @version $Revision$ $Date$
57      */
58     public class ISOFilledStringFieldPackager extends ISOFieldPackager
59     {
60         private Interpreter interpreter;
61         private Padder padder;
62         private Prefixer prefixer;
63     
64         /**
65          * Constructs a default ISOFilledStringFieldPackager. There is no padding,
66          * no length prefix and a literal interpretation. The set methods must be called to
67          * make this useful.
68          */
69         public ISOFilledStringFieldPackager()
70         {
71             super();
72             this.padder = NullPadder.INSTANCE;
73             this.interpreter = LiteralInterpreter.INSTANCE;
74             this.prefixer = NullPrefixer.INSTANCE;
75         }
76     
77         /**
78          * Constructs an ISOFilledStringFieldPackager with a specific Padder, Interpreter and Prefixer.
79          * The length and description should be set with setLength() and setDescription methods.
80          * @param padder The type of padding used.
81          * @param interpreter The interpreter used to encode the field.
82          * @param prefixer The type of length prefixer used to encode this field.
83          */
84         public ISOFilledStringFieldPackager(Padder padder, Interpreter interpreter, Prefixer prefixer)
85         {
86             super();
87             this.padder = padder;
88             this.interpreter = interpreter;
89             this.prefixer = prefixer;
90         }
91     
92         /**
93          * Creates an ISOFilledStringFieldPackager.
94          * @param maxLength The maximum length of the field in characters or bytes depending on the datatype.
95          * @param description The description of the field. For human readable output.
96          * @param interpreter The interpreter used to encode the field.
97          * @param padder The type of padding used.
98          * @param prefixer The type of length prefixer used to encode this field.
99          */
100         public ISOFilledStringFieldPackager(int maxLength, String description, Padder padder,
101                                       Interpreter interpreter, Prefixer prefixer)
102         {
103             super(maxLength, description);
104             this.padder = padder;
105             this.interpreter = interpreter;
106             this.prefixer = prefixer;
107         }
108     
109         /**
110          * Sets the Padder.
111          * @param padder The padder to use during packing and unpacking.
112          */
113         public void setPadder(Padder padder)
114         {
115             this.padder = padder;
116         }
117     
118         /**
119          * Sets the Interpreter.
120          * @param interpreter The interpreter to use in packing and unpacking.
121          */
122         public void setInterpreter(Interpreter interpreter)
123         {
124             this.interpreter = interpreter;
125         }
126     
127         /**
128          * Sets the length prefixer.
129          * @param prefixer The length prefixer to use during packing and unpacking.
130          */
131         public void setPrefixer(Prefixer prefixer)
132         {
133             this.prefixer = prefixer;
134         }
135     
136         /**
137          * Returns the prefixer's packed length and the interpreter's packed length.
138     	 * @see org.jpos.iso.ISOFieldPackager#getMaxPackedLength()
139     	 */
140         public int getMaxPackedLength()
141         {
142             return prefixer.getPackedLength() + interpreter.getPackedLength(getLength());
143         }
144     
145         /** Create a nice readable message for errors */
146         private String makeExceptionMessage(ISOComponent c, String operation) {
147             Object fieldKey = "unknown";
148             if (c != null)
149             {
150                 try
151                 {
152                     fieldKey = c.getKey();
153                 } catch (Exception ignore)
154                 {
155                 }
156             }
157             return this.getClass().getName() + ": Problem " + operation + " field " + fieldKey;
158         }
159     
160         /**
161     	 * Convert the component into a byte[].
162     	 */
163         public byte[] pack(ISOComponent c) throws ISOException
164         {
165             try
166             {
167                 String data = (String)c.getValue();
168                 if (data.length() > getLength())
169                 {
170                     throw new ISOException("Field length " + data.length() + " too long. Max: " + getLength());
171                 }
172                 String paddedData = padder.pad(data, getLength());
173                 byte[] rawData = new byte[prefixer.getPackedLength()
174                         + interpreter.getPackedLength(paddedData.length())];
175                 prefixer.encodeLength(data.length(), rawData);
176                 interpreter.interpret(paddedData, rawData, prefixer.getPackedLength());
177                 return rawData;
178             } catch(Exception e)
179             {
180                 throw new ISOException(makeExceptionMessage(c, "packing"), e);
181             }
182         }
183     
184         /**
185          * Unpacks the byte array into the component.
186          * @param c The component to unpack into.
187          * @param b The byte array to unpack.
188          * @param offset The index in the byte array to start unpacking from.
189          * @return The number of bytes consumed unpacking the component.
190          */
191         public int unpack(ISOComponent c, byte[] b, int offset) throws ISOException
192         {
193             try
194             {
195                 int len = prefixer.decodeLength(b, offset);
196                 if (len == -1)
197                 {
198                     // The prefixer doesn't know how long the field is, so use
199         			// maxLength instead
200                     len = getLength();
201                 }
202                 int lenLen = prefixer.getPackedLength();
203                 String unpacked = interpreter.uninterpret(b, offset + lenLen, len);
204                 c.setValue(unpacked);
205                 return lenLen + interpreter.getPackedLength(getLength());
206             } catch(Exception e)
207             {
208                 throw new ISOException(makeExceptionMessage(c, "unpacking"), e);
209             }
210         }
211     
212         /**
213          * Unpack the input stream into the component.
214          * @param c  The Component to unpack into.
215          * @param in Input stream where the packed bytes come from.
216          * @exception IOException Thrown if there's a problem reading the input stream.
217          */
218         public void unpack (ISOComponent c, InputStream in) 
219             throws IOException, ISOException
220         {
221             try
222             {
223                 int lenLen = prefixer.getPackedLength ();
224                 int len;
225                 if (lenLen == 0)
226                 {
227                     len = getLength();
228                 } else
229                 {
230                     len = prefixer.decodeLength (readBytes (in, lenLen), 0);
231                 }
232                 int packedLen = interpreter.getPackedLength(len);
233                 String unpacked = interpreter.uninterpret(readBytes (in, packedLen), 0, len);
234                 c.setValue(unpacked);
235                 in.skip(interpreter.getPackedLength(getLength()) - packedLen);
236             } catch(ISOException e)
237             {
238                 throw new ISOException(makeExceptionMessage(c, "unpacking"), e);
239             }
240         }
241     
242         /**
243          * Checks the length of the data against the maximum, and throws an IllegalArgumentException.
244          * This is designed to be called from field Packager constructors and the setLength()
245          * method.
246          * @param len The length of the data for this field packager.
247          * @param maxLength The maximum length allowed for this type of field packager.
248          *          This depends on the prefixer that is used.
249          * @throws IllegalArgumentException If len > maxLength.
250          */
251         protected void checkLength(int len, int maxLength) throws IllegalArgumentException
252         {
253             if (len > maxLength)
254             {
255                 throw new IllegalArgumentException("Length " + len + " too long for " + getClass().getName());
256             }
257         }
258     }
