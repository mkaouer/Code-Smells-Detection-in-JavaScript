17     /**
18      * Tiny Space implementation
19      *
20      * @author Alejandro Revilla

22      * @since 1.4.7
23      */
24     public class TinySpace implements Space, Serializable {
25         protected Map map = new HashMap ();
26     
27         public TinySpace () {
28             super();
29         }
30         public void out (Object key, Object value) {
31             synchronized (this) {
32                 Object v = map.get (key);
33                 if (v == null)
34                     map.put (key, value);
35                 else if (v instanceof Data)
36                     ((Data)v).add (value);
37                 else {
38                     Data data = new Data();
39                     data.add (v);
40                     data.add (value);
41                     map.put (key, data);
42                 }
43                 this.notifyAll ();
44             }
45         }
46         public void out (Object id, Object value, long timeout) {
47             out (id, new LeasedReference (value, timeout));
48         }
49         public synchronized Object rdp (Object key) {
50             Object obj = map.get (key);
51             if (obj instanceof Data) 
52                 obj = ((Data) obj).get (key);
53             else if (obj instanceof LeasedReference) {
54                 obj = ((LeasedReference)obj).get ();
55             }
56             return obj;
57         }
58         public synchronized Object inp (Object key) {
59             Object obj = map.get (key);
60             if (obj instanceof Data) {
61                 Data data = (Data) obj;
62                 obj = data.remove ();
63                 if (data.isEmpty ())
64                     map.remove (key);
65             } else if (obj != null) {
66                 if (obj instanceof LeasedReference) {
67                     obj = ((LeasedReference)obj).get ();
68                 }
69                 map.remove (key);
70             }
71     
72             return obj;
73         }
74         public synchronized Object in (Object key) {
75             Object obj;
76             while ((obj = inp (key)) == null) {
77                 try {
78                     this.wait ();
79                 } catch (InterruptedException e) { }
80             }
81             return obj;
82         }
83         public synchronized Object in  (Object key, long timeout) {
84             Object obj;
85             long now = System.currentTimeMillis();
86             long end = now + timeout;
87             while ((obj = inp (key)) == null && 
88                     ((now = System.currentTimeMillis()) < end))
89             {
90                 try {
91                     this.wait (end - now);
92                 } catch (InterruptedException e) { }
93             }
94             return obj;
95         }
96         public synchronized Object rd  (Object key) {
97             Object obj;
98             while ((obj = rdp (key)) == null) {
99                 try {
100                     this.wait ();
101                 } catch (InterruptedException e) { }
102             }
103             return obj;
104         }
105         public synchronized Object rd  (Object key, long timeout) {
106             Object obj;
107             long now = System.currentTimeMillis();
108             long end = now + timeout;
109             while ((obj = rdp (key)) == null && 
110                     ((now = System.currentTimeMillis()) < end))
111             {
112                 try {
113                     this.wait (end - now);
114                 } catch (InterruptedException e) { }
115             }
116             return obj;
117         }
118         protected static final class Data {
119             LinkedList data;
120     
121             protected Data () {
122                 super();
123                 data = new LinkedList ();
124             }
125             protected Data (Object value) {
126                 this ();
127                 add (value);
128             }
129             protected void add (Object value) {
130                 data.add (value);
131             }
132             protected Object get (Object value) {
133                 Object obj = null;
134                 while (size() > 0) {
135                     obj = data.getFirst();
136                     if (obj instanceof LeasedReference) {
137                         obj = ((LeasedReference)obj).get ();
138                         if (obj == null) {
139                             data.removeFirst ();
140                             continue;
141                         }
142                     }
143                     break;
144                 }
145                 return obj;
146             }
147             protected int size () {
148                 return data.size ();
149             }
150             protected Object remove () {
151                 Object obj = null;
152                 while (size() > 0) {
153                     obj = data.removeFirst();
154                     if (obj instanceof LeasedReference) {
155                         LeasedReference ref = (LeasedReference) obj;
156                         obj = ref.get ();
157                         if (obj == null) {
158                             continue;
159                         }
160                         ref.discard ();
161                     }
162                     break;
163                 }
164                 return obj;
165             }
166             protected boolean isEmpty () {
167                 return data.isEmpty ();
168             }
169         }
