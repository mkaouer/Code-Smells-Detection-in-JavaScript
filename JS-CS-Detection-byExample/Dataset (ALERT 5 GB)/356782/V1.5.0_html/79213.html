27     /** A TransactionParticipant whichï¿½s prepare, commit and abort methods can be 
28      *  specified through beanshell scripts. <BR>
29      *
30      *  To indicate what code to execute for any of the methods just add an element 
31      *  named 'prepare', 'commit' or 'abort' contained in that of the participant. <BR>
32      *
33      *  See BSHMethod for details on the syntax of these elemets. The value to return 
34      *  in the prepare method should be stored in the script variable named "result".
35      *  None of these tags are mandatory. <BR>
36      *
37      *  You can subclass BSHTransactionParticipant and override the default... 
38      *  methods. That way you can provide default behaviour for a participant and 
39      *  override it at deploy time through scripts. 
40      * 
41      * @see BSHMethod
42      * @author  AMarques
43      */
44     public class BSHTransactionParticipant extends SimpleLogSource implements TransactionParticipant, XmlConfigurable {
45         
46         protected BSHMethod prepareMethod;
47         protected BSHMethod commitMethod;
48         protected BSHMethod abortMethod;
49         
50         /** Creates a new instance of BSHTransactionParticipant */
51         public BSHTransactionParticipant() {
52         }
53         
54         public void abort(long id, java.io.Serializable context) {
55             LogEvent ev = new LogEvent(this, "abort");
56             if (abortMethod != null) {
57                 try {
58                     executeMethod(abortMethod, id, context, ev, "");
59                 } catch (Exception ex) {
60                     ev.addMessage(ex);
61                 }
62             } else {
63                 defaultAbort(id, context, ev);
64             }
65             Logger.log(ev);
66         }
67         
68         protected void defaultAbort(long id, Serializable context, LogEvent ev) {}
69         
70         public void commit(long id, java.io.Serializable context) {
71             LogEvent ev = new LogEvent(this, "commit");
72             if (commitMethod != null) {
73                 try {
74                     executeMethod(commitMethod, id, context, ev, "");
75                 } catch (Exception ex) {
76                     ev.addMessage(ex);
77                 }
78             } else {
79                 defaultCommit(id, context, ev);
80             }
81             Logger.log(ev);
82         }
83         
84         protected void defaultCommit(long id, Serializable context, LogEvent ev) {}
85         
86         public int prepare(long id, java.io.Serializable context) {
87             LogEvent ev = new LogEvent(this, "prepare");
88             int result = ABORTED | READONLY;
89             if (prepareMethod != null) {
90                 try {
91                     result = ((Integer) executeMethod(prepareMethod, id, context, ev, "result")).intValue();
92                 } catch (Exception ex) {
93                     ev.addMessage(ex);
94                 }
95             } else {
96                 result = defaultPrepare(id, context, ev);
97             }
98             ev.addMessage("result", Integer.toBinaryString(result));
99             Logger.log(ev);
100             return result;
101         }
102         
103         protected int defaultPrepare(long id, Serializable context, LogEvent ev) {
104             return PREPARED | READONLY;
105         }
106         
107         public void setConfiguration(Element e) throws ConfigurationException {
108     	try {
109                 prepareMethod = BSHMethod.createBshMethod(e.getChild("prepare"));
110                 commitMethod = BSHMethod.createBshMethod(e.getChild("commit"));
111                 abortMethod = BSHMethod.createBshMethod(e.getChild("abort"));
112             } catch (Exception ex) {
113                 throw new ConfigurationException(ex.getMessage(), ex);
114             }
115         }
116         
117         protected Object executeMethod(BSHMethod m, long id, Serializable context, LogEvent evt, String resultName) 
118         throws EvalError, FileNotFoundException, IOException {
119             Map params = new HashMap();
120             params.put("context", context);
121             params.put("id", new Long(id));
122             params.put("evt", evt);
123             return m.execute(params, resultName);
124         }
