8     package org.jpos.iso.channel;
9     
10     import java.io.IOException;
11     import java.util.List;
12     import java.util.Vector;
13     
14     import org.jpos.core.Configurable;
15     import org.jpos.core.Configuration;
16     import org.jpos.core.ConfigurationException;
17     import org.jpos.iso.ISOChannel;
18     import org.jpos.iso.ISOClientSocketFactory;
19     import org.jpos.iso.ISOException;
20     import org.jpos.iso.ISOMsg;
21     import org.jpos.iso.ISOPackager;
22     import org.jpos.util.LogEvent;
23     import org.jpos.util.LogSource;
24     import org.jpos.util.Logger;
25     import org.jpos.util.NameRegistrar;
26     
27     public class ChannelPool implements ISOChannel, LogSource, Configurable {
28         boolean usable = true;
29         String name = "";
30         protected Logger logger;
31         protected String realm;
32         Configuration cfg = null;
33         List pool;
34         ISOChannel current;
35     
36         public ChannelPool () {
37             super ();
38             pool = new Vector ();
39         }
40         public void setPackager(ISOPackager p) {
41             // nothing to do
42         }
43         public synchronized void connect () throws IOException {
44             IOException ioe = null;
45             current = null;
46             LogEvent evt = new LogEvent (this, "connect");
47             evt.addMessage ("pool-size=" + Integer.toString (pool.size()));
48             for (int i=0; i<pool.size(); i++) {
49                 try {
50                     evt.addMessage ("pool-" + Integer.toString (i));
51                     ISOChannel c = (ISOChannel) pool.get (i);
52                     c.connect ();
53                     if (c.isConnected()) {
54                         current = c;
55                         usable = true;
56                         break;
57                     }
58                 } catch (IOException e) {
59                     evt.addMessage (e);
60                 }
61             }
62             if (current == null)
63                 evt.addMessage ("connect failed");
64             Logger.log (evt);
65             if (current == null) {
66                 throw new IOException ("unable to connect");
67             }
68         }
69         public synchronized void disconnect () throws IOException {
70             current = null;
71             LogEvent evt = new LogEvent (this, "disconnect");
72             for (int i=0; i<pool.size(); i++) {
73                 try {
74                     ISOChannel c = (ISOChannel) pool.get (i);
75                     c.disconnect ();
76                 } catch (IOException e) {
77                     evt.addMessage (e);
78                 }
79             }
80             Logger.log (evt);
81         }
82         public synchronized void reconnect() throws IOException {
83             disconnect ();
84             connect ();
85         }
86         public synchronized boolean isConnected() {
87             try {
88                 return getCurrent().isConnected ();
89             } catch (IOException e) {
90                 return false;
91             }
92         }
93         public ISOMsg receive() throws IOException, ISOException {
94             return getCurrent().receive ();
95         }
96         public void send (ISOMsg m) throws IOException, ISOException {
97             getCurrent().send (m);
98         }
99         public void setUsable(boolean b) {
100             this.usable = b;
101         }
102         public void setName (String name) {
103             this.name = name;
104             NameRegistrar.register ("channel."+name, this);
105         }
106         public String getName() {
107             return this.name;
108         }
109         public ISOPackager getPackager () {
110             return (ISOPackager) null;
111         }
112         public void setLogger (Logger logger, String realm) {
113             this.logger = logger;
114             this.realm  = realm;
115         }
116         public String getRealm () {
117             return realm;
118         }
119         public Logger getLogger() {
120             return logger;
121         }
122         public synchronized void setConfiguration (Configuration cfg) 
123             throws ConfigurationException
124         {
125             this.cfg = cfg;
126             String channelName[] = cfg.getAll ("channel");
127             for (int i=0; i<channelName.length; i++) {
128                 try {
129                     addChannel (channelName[i]);
130                 } catch (NameRegistrar.NotFoundException e) {
131                     throw new ConfigurationException (e);
132                 }
133             }
134         }
135         public void addChannel (ISOChannel channel) {
136             pool.add (channel);
137         }
138         public void addChannel (String name) 
139             throws NameRegistrar.NotFoundException
140         {
141             pool.add ((ISOChannel) NameRegistrar.get ("channel."+name));
142         }
143         public void removeChannel (ISOChannel channel) {
144             pool.remove (channel);
145         }
146         public void removeChannel (String name) throws NameRegistrar.NotFoundException {
147             pool.remove ((ISOChannel) NameRegistrar.get ("channel."+name));
148         }
149         public int size() {
150             return pool.size();
151         }
152         public synchronized ISOChannel getCurrent () throws IOException {
153             if (current == null)
154                 connect();
155             else if (!usable)
156                 reconnect();
157     
158             return current;
159         }
160     }
