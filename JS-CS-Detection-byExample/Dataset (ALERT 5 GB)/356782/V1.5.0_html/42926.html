16     /**
17      * Request adds to TinySpace a few helper methods suitable for
18      * exchanging requests and responses over an outter [Transient|Tiny]Space.
19      *
20      * @author Alejandro Revilla
21      * @version $Revision$ $Date$
22      * @since 1.4.7
23      */
24     public class Request extends TinySpace {
25         public static String REQUEST  = "$REQ";
26         public static String RESPONSE = "$RESP";
27         public static String ERROR    = "$ERR";
28         
29         public Request () {
30             super();
31         }
32         public Request (Object value) {
33             out (REQUEST, value);
34         }
35     
36         public Object getResponse () {
37             return rd (RESPONSE);
38         }
39         public Object getResponse (long timeout) {
40             return rd (RESPONSE, timeout);
41         }
42         public void setResponse (Object o) {
43             out (RESPONSE, o);
44         }
45     
46         public Object getRequest () {
47             return rd (REQUEST);
48         }
49         public Object getRequest (long timeout) {
50             return rd (REQUEST, timeout);
51         }
52     
53         public void addError (Object o) {
54             out (ERROR, o);
55         }
56         public Object[] getErrors () {
57             return (Object[]) SpaceUtil.inpAll(this, ERROR);
58         }
59         public Object getError () {
60             return inp (ERROR);
61         }
62     }
