70     /**
71      * Configure mux
72      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
73      * @version $Revision$ $Date$
74      */
75     public class ConfigMux implements QSPReConfigurator {
76         public void config (QSP qsp, Node node) throws ConfigurationException
77         {
78             String name = node.getAttributes().getNamedItem ("name").getNodeValue();
79             LogEvent evt = new LogEvent (qsp, "config-mux", name);
80             Logger logger = ConfigLogger.getLogger (node);
81             String realm  = ConfigLogger.getRealm (node);
82             ISOChannel channel = ConfigChannel.getChildChannel (node);
83             ISOMUX mux = null;
84             try{
85                 if(node.getAttributes().getNamedItem("class")==null) 
86                     mux = new ISOMUX (channel, logger, realm);
87                 else{ 
88                     Class muxClass = Class.forName(node.getAttributes().getNamedItem("class").getNodeValue());
89                     Class[] argTypes = {ISOChannel.class, Logger.class, String.class};
90                     Object[] args = {channel, logger, realm};
91                     mux = (ISOMUX)muxClass.getConstructor(argTypes).newInstance(args);
92                 }
93             }catch(Exception e){
94                 throw new ConfigurationException("error trying to construct ISOMUX", e);
95             }
96             boolean connect = node.getAttributes()
97                             .getNamedItem("connect").getNodeValue().equals("yes");
98             evt.addMessage ("MUX "+name+"/"+channel.getName());
99             mux.setName (name);
100             mux.setConnect (connect);
101             if (mux instanceof Configurable) {
102                 Configuration cfg = new SimpleConfiguration (
103                     ConfigUtil.addProperties (node, new Properties(), evt)
104                 );
105                 ((Configurable)mux).setConfiguration (cfg);
106             }
107             new Thread (mux).start();
108             try {
109                 qsp.registerMBean (mux, "type=mux,name="+name);
110             } catch (NotCompliantMBeanException e) {
111                 evt.addMessage (e.getMessage());
112             } catch (Exception e) {
113                 evt.addMessage (e);
114                 throw new ConfigurationException (e);
115             } finally {
116                 Logger.log (evt);
117             }
118         }
119         public void reconfig (QSP qsp, Node node) throws ConfigurationException
120         {
121             String name = node.getAttributes().getNamedItem ("name").getNodeValue();
122             LogEvent evt = new LogEvent (qsp, "re-config-mux", name);
123             boolean connect = node.getAttributes()
124                             .getNamedItem("connect").getNodeValue().equals("yes");
125             try {
126                 ISOMUX mux = ISOMUX.getMUX (name);
127                 mux.setLogger (ConfigLogger.getLogger (node),
128                                ConfigLogger.getRealm (node));
129                 mux.setConnect (connect);
130     
131                 if (mux instanceof ReConfigurable) {
132                     Configuration cfg = new SimpleConfiguration (
133                         ConfigUtil.addProperties (node, new Properties(), evt)
134                     );
135                     ((Configurable)mux).setConfiguration (cfg);
136                 }
137             } catch (NotFoundException e) {
138                 evt.addMessage (e);
139                 throw new ConfigurationException (e);
140             } finally {
141                 Logger.log (evt);
142             }
143         }
144     
145         public static ISOMUX getMUX (Node node) {
146             Node n = node.getAttributes().getNamedItem ("name");
147             if (n != null)
148                 try {
149                     return ISOMUX.getMUX (n.getNodeValue());
150                 } catch (NotFoundException e) { }
151             return null;
152         }
153     }
