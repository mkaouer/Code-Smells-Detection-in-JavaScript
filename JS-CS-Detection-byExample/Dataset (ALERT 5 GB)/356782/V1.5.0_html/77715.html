8     package org.jpos.q2;
9     
10     import java.lang.reflect.Constructor;
11     import java.lang.reflect.Method;
12     
13     /**
14      * @author <a href="mailto:alwyn@smart.com.ph">Alwyn Schoeman</a>
15      * @version $Revision$ $Date$
16      */
17     public class QBeanSupportMBeanDescription extends mx4j.MBeanDescriptionAdapter {
18         public String getConstructorDescription (Constructor ctor) {
19             return "Default Service Constructor";
20         }
21     
22         public String getAttributeDescription (String attribute) {
23             if (attribute.equals("Loader"))
24                 return "Classloader";
25             if (attribute.equals("LoaderURLS"))
26                 return "ClassLoader URLs";
27             if (attribute.equals("Logger"))
28                 return "Logger service";
29             if (attribute.equals("Modified"))
30                 return "Configuration modified?";
31             if (attribute.equals("Name"))
32                 return "Service Name";
33             if (attribute.equals("Persist"))
34                 return "XML Descriptor";
35             if (attribute.equals("Server"))
36                 return "MBeanServer Object";
37             if (attribute.equals("State"))
38                 return "Runtime Status";
39             if (attribute.equals("StateAsString"))
40                 return "Runtime Status";
41             return super.getAttributeDescription (attribute);
42         }
43     
44         public String getOperationDescription(Method operation) {
45             if (operation.getName().equals("init"))
46                 return "Initialise service";
47             if (operation.getName().equals("start"))
48                 return "Start service";
49             if (operation.getName().equals("stop"))
50                 return "Stop service";
51             if (operation.getName().equals("destroy"))
52                 return "Destroy service";
53             if (operation.getName().equals("shutdownQ2"))
54                 return "Shutdown and undeploy EVERYTHING";
55             return super.getOperationDescription (operation);
56         }
57     }
