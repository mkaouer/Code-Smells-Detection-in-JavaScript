34         public void testUnpack() throws Exception {
35     
36     		TLVList tlvList=new TLVList();
37     		tlvList.unpack(data);
38     
39     		TLVMsg  m1=tlvList.find(0x84);
40     		TLVMsg  m2=tlvList.findNextTLV();
41     		TLVMsg  m3=tlvList.findNextTLV();
42     		TLVMsg  m4=tlvList.find(0x9F70);
43     
44             TestUtils.assertEquals(aid1,m1.getValue());
45             TestUtils.assertEquals(aid2,m2.getValue());
46             TestUtils.assertEquals(aid3,m3.getValue());

53             tlv.append(0x84,aid2);
54             tlv.append(0x84,aid3);
55             tlv.append(0x9F70,atc);
56             TestUtils.assertEquals(data, tlv.pack());
