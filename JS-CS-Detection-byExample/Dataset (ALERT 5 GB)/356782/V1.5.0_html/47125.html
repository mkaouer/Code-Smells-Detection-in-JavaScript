65         /*
66          * @return tag + length + value of the TLV Message
67          */
68         public byte[] getTLV() {
69             String hexVal = Integer.toHexString(tag);
70             byte[] bTag = ISOUtil.hex2byte(hexVal);
71             byte[] bLen = getL();
72             if (value != null) {
73                 int tLength = bTag.length + bLen.length + value.length;
74                 byte[] out = new byte[tLength];
75                 System.arraycopy(bTag, 0, out, 0, bTag.length);
76                 System.arraycopy(bLen, 0, out, bTag.length, bLen.length);
77                 System.arraycopy(value, 0, out, bTag.length + bLen.length,
78                         value.length);
79                 return out;
80             } else {//Length can be 0
81                 int tLength = bTag.length + bLen.length;
82                 byte[] out = new byte[tLength];
83                 System.arraycopy(bTag, 0, out, 0, bTag.length);
84                 System.arraycopy(bLen, 0, out, bTag.length, bLen.length);
85                 return out;
86     
87             }
88         }

90         /**
91          * Value up to 127 can be encoded in single byte and multiple bytes are
92          * required for length bigger than 127
93          * 
94          * @return encoded length
95          */
96         public byte[] getL() {
97     
98             if (value == null)
99                 return new byte[1];
100             int ix = 0;
101             int tmp = value.length;
102             int bytes = 0;
103             // if Length is greater less than 127
104             //set the 8bit as 0 indicating next 7 bits is the length
105             //of the message
106             //if length is more than 127 then, set the first bit as 1 indicating
107             //next 7 bits will indicate the length of following bytes used for
108             // length
109     
110             while (tmp != 0) {
111                 tmp = tmp >> 8;
112                 bytes++;
113             }
114     
115             /* If value can be encoded on one byte */
116             if (bytes <= 1 && value.length <= 127) {
117                 byte[] rBytes = new byte[bytes];
118                 rBytes[0] = (byte) value.length;
119                 return rBytes;
120             }
121             //else {
122             /* Value to be encoded on multiple bytes */
123             //we need 1 byte to indicate the length
124             byte[] rBytes = new byte[1 + bytes];
125             rBytes[0] = (byte) (0x80 | bytes);
126     
127             int mask = 0xFF;
128             tmp = value.length;
129             while (ix < bytes) {
130                 /* Mask off 8 bits of the value at a time */
131                 rBytes[(bytes - ix)] = (byte) (tmp & mask);
132                 bytes--;
133                 /* Shift value right 8 bits, effectively removing them */
134                 tmp = (tmp >> 8);
135             }
136             //}
137             return rBytes;
138         }
