7     package org.jpos.transaction;
8     
9     import java.io.Serializable;
10     
11     /**
12      * 2 phase commit participant
13      * @author apr
14      * @since 1.4.7
15      */
16     public interface TransactionParticipant extends TransactionConstants {
17         /**
18          * Called by TransactionManager in preparation for a transaction
19          * @param id the Transaction identifier
20          * @param context transaction context
21          * @return PREPARED or ABORTED (| NO_JOIN | READONLY)
22          */
23         public int  prepare (long id, Serializable context);
24     
25         /**
26          * Called by TransactionManager upon transaction commit.
27          * Warning: implementation should be able to handle multiple calls
28          * with the same transaction id (rare crash recovery)
29          *
30          * @param id the Transaction identifier
31          * @param context transaction context
32          */
33         public void commit  (long id, Serializable context);
34     
35         /**
36          * Called by TransactionManager upon transaction commit.
37          * Warning: implementation should be able to handle multiple calls
38          * with the same transaction id (rare crash recovery)
39          *
40          * @param id the Transaction identifier
41          * @param context transaction context
42          */
43         public void abort   (long id, Serializable context);
44     }
