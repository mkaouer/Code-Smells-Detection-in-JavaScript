191         private void recalcMaxField() {
192             maxField = 0;
193             for (Enumeration e = fields.keys(); e.hasMoreElements(); ) {
194                 Object obj = e.nextElement();
195                 if (obj instanceof Integer) 
196                     maxField = Math.max (maxField, ((Integer)obj).intValue());
197             }
198             maxFieldDirty = false;
199         }

271         /**
272          * setup BitMap
273          * @exception ISOException
274          */
275         public void recalcBitMap () throws ISOException {
276             if (!dirty)
277                 return;
278     
279             if (getMaxField() > 128) {
280                 BitSet bmap=new BitSet(64);
281                 for (int i=1; i<=64; i++)
282                     if(((ISOComponent) fields.get(new Integer (i+128))) != null) 
283                         bmap.set (i);
284                 set (new ISOBitMap (65, bmap));
285             }
286                     
287             BitSet bmap = new BitSet (getMaxField() > 64 ? 128 : 64);
288                     int tmpMaxField=maxField > 128 ? 128 : maxField;
289     
290             for (int i=1; i<=tmpMaxField; i++)
291                 if (((ISOComponent) fields.get (new Integer (i))) != null) 
292                     bmap.set (i);
293             set (new ISOBitMap (-1, bmap));
294                     
295             dirty = false;
296         }

331          * dump the message to a PrintStream. The output is sorta
332          * XML, intended to be easily parsed.
333          * <br>

339         public void dump (PrintStream p, String indent) {
340             ISOComponent c;
341             p.print (indent + "<" + XMLPackager.ISOMSG_TAG);
342             switch (direction) {
343                 case INCOMING:
344                     p.print (" direction=\"incoming\"");
345                     break;
346                 case OUTGOING:
347                     p.print (" direction=\"outgoing\"");
348                     break;
349             }
350             if (fieldNumber != -1)
351                 p.print (" "+XMLPackager.ID_ATTR +"=\""+fieldNumber +"\"");
352             p.println (">");
353             String newIndent = indent + "  ";
354     
355             if (header instanceof Loggeable)
356                 ((Loggeable) header).dump (p, newIndent);
357     
358             for (int i=0; i<=maxField; i++) {
359                 if ((c = (ISOComponent) fields.get (new Integer (i))) != null)
360                     c.dump (p, newIndent);
361                 //

366                 //    c.dump (p, newIndent);
367                 // }
368                 //
369             }
370     
371             p.println (indent + "</" + XMLPackager.ISOMSG_TAG+">");
372         }

440             try {
441                 ISOMsg m = (ISOMsg) super.clone();
442                 m.fields = (Hashtable) fields.clone();
443                 if (header != null)
444                     m.header = (ISOHeader) header.clone();

451         /**
452          * Partially clone an ISOMsg
453          * @param fields int array of fields to go
454          * @return new ISOMsg instance
455          */
456         public Object clone(int[] fields) {
457             try {
458                 ISOMsg m = (ISOMsg) super.clone();
459                 m.fields = new Hashtable();
460                 for (int i=0; i<fields.length; i++) {
461                     if (hasField(fields[i])) {
462                         try {
463                             m.set (getComponent(fields[i]));
464                         } catch (ISOException e) { 
465                             // it should never happen
466                         }
467                     }
468                 }
469                 return m;
470             } catch (CloneNotSupportedException e) {
471                 throw new InternalError();
472             }
473         }

477          * please note that received fields take precedence over 
478          * existing ones (simplifying card agent message creation

495         public String toString() {
496             StringBuffer s = new StringBuffer();
497             if (isIncoming())
498                 s.append("<-- ");
499             else if (isOutgoing())
500                 s.append("--> ");
501             else
502                 s.append("    ");
503     
504             try {
505                 s.append((String) getValue(0));
506                 if (hasField(11)) {
507                     s.append(' ');
508                     s.append((String) getValue(11));
509                 }
510                 if (hasField(41)) {
511                     s.append(' ');
512                     s.append((String) getValue(41));
513                 }
514             } catch (ISOException e) { }
515             return s.toString();
516         }

590         /**
591          * sets an appropiate response MTI.
592          *
593          * i.e. 0100 becomes 0110<br>
594          * i.e. 0201 becomes 0210<br>
595          * i.e. 1201 becomes 1210<br>
596          * @exception ISOException on MTI not set or it is not a request
597          */
598         public void setResponseMTI() throws ISOException {
599             if (!isRequest())
600                 throw new ISOException ("not a request - can't set response MTI");
601     
602             String mti = getMTI();
603             char c1 = mti.charAt(3);
604             char c2 = '0';
605             switch (c1)
606             {
607                 case '0' :
608                 case '1' : c2='0';break;
609                 case '2' :
610                 case '3' : c2='2';break;
611                 case '4' :
612                 case '5' : c2='4';break;
613     
614             }
615             set (new ISOField (0,
616                 mti.substring(0,2)
617                 +(Character.getNumericValue(getMTI().charAt (2))+1) + c2
618                 )
619             );
620         }

631         protected void writeHeader (ObjectOutput out) throws IOException {
632             int len = header.getLength();
633             if (len > 0) {

656         public void writeExternal (ObjectOutput out) throws IOException {
657             out.writeByte (0);  // reserved for future expansion (version id)
658             out.writeShort (fieldNumber);
659     
660             if (header != null) 
661                 writeHeader (out);
662             if (direction > 0)
663                 writeDirection (out);
664     
665             for (Enumeration e = fields.elements(); e.hasMoreElements(); ) {
666                 ISOComponent c = (ISOComponent) e.nextElement();
667                 if (c instanceof ISOMsg) {
668                     out.writeByte ('M');
669                     ((Externalizable) c).writeExternal (out);
670                 }
671                 else if (c instanceof ISOBinaryField) {
672                     out.writeByte ('B');
673                     ((Externalizable) c).writeExternal (out);
674                 }
675                 else if (c instanceof ISOField) {
676                     out.writeByte ('F');
677                     ((Externalizable) c).writeExternal (out);
678                 }
679             }
680             out.writeByte ('E');
681         }
682     
683         public void readExternal  (ObjectInput in) 
684             throws IOException, ClassNotFoundException
685         {
686             in.readByte();  // ignore version for now
687             fieldNumber = in.readShort();
688             byte fieldType;
689             ISOComponent c;
690             try {
691                 while ((fieldType = in.readByte()) != 'E') {
692                     c = null;
693                     switch (fieldType) {
694                         case 'F':
695                             c = new ISOField ();
696                             break;
697                         case 'B':
698                             c = new ISOBinaryField ();
699                             break;
700                         case 'M':
701                             c = new ISOMsg ();
702                             break;
703                         case 'H':
704                             readHeader (in);
705                             break;
706                         case 'D':
707                             readDirection (in);
708                             break;
709                         default:
710                             throw new IOException ("malformed ISOMsg");
711                     }
712                     if (c != null) {
713                         ((Externalizable)c).readExternal (in);
714                         set (c);
715                     }
716                 }
717             }
718             catch (ISOException e) {
719                 throw new IOException (e.getMessage());
720             }
721         }
