30          * unpack a message
31          * @param buf - raw message
32          */
33         public void unpack(byte[] buf) throws ISOException {
34             ByteBuffer buffer=ByteBuffer.wrap(buf);
35             TLVMsg currentNode;
36             while (hasNext(buffer)) {    
37                 currentNode = getTLVMsg(buffer);
38                 append(currentNode);
39             }
40         }

71         public void deleteByTag(int tag) {
72             int i = 0;
73             TLVMsg tlv2;
74     
75             while (i < tags.size()) {
76                 tlv2 = (TLVMsg) (tags.elementAt(i));

78                     tags.removeElement(tlv2);

81             }
82            }

88         public TLVMsg find(int tag) {
89             int i = 0;
90             tagToFind = tag;
91             while (i < tags.size()) {
92                 TLVMsg tlv = (TLVMsg) (tags.elementAt(i));
93                 if (tlv.getTag() == tag) {

96                 }

150         /**
151          * pack the TLV message (BER-TLV Encoding)
152          * @return the packed message
153          */
154         public byte[] pack() {
155             int i = 0;
156             TLVMsg tlv;
157             ByteBuffer buffer=ByteBuffer.allocate(400);
158             while(i<tags.size()) {
159                 tlv=(TLVMsg)(tags.elementAt(i));
160                 buffer.put(tlv.getTLV());
161                 i++;
162             }
163             byte[] b=new byte[buffer.position()];
164             buffer.flip();
165             buffer.get(b);
166             return b;

168         }

170         /*
171          * Read next TLV Message from stream and return it 
172          *@return TLVMsg 
173          */
174         private TLVMsg getTLVMsg(ByteBuffer buffer) throws ISOException {
175             int tag = getTAG(buffer);
176             byte[] arrValue=null;
177             //Get Length
178             int length = getValueLength(buffer);
179             if(length >buffer.remaining()) throw new ISOException("BAD FORMAT");
180             
181             if(length>0) {
182                 arrValue= new byte[length];
183                 buffer.get(arrValue);
184             }
185         
186             TLVMsg tlv = new TLVMsg(tag, arrValue);
187             return tlv;
188         }

190         /*
191          * Check Existance of next TLV Field
192          * @param buffer  ByteBuffer containing TLV data
193          */
194         private  boolean hasNext(ByteBuffer buffer) {
195                   return buffer.hasRemaining();
196         }
197         
198         /* 
199          * Return the next TAG
200          * @return tag
201          */
202         private int getTAG(ByteBuffer buffer) {
203             int b;
204             int tag = 0x0000;
205             b = buffer.get() & 0xff;
206             // Skip padding chars
207             if (b == 0xFF || b == 0x00) {
208                 do {
209                     b = buffer.get() & 0xff;    
210                 } while ((b == 0xFF || b == 0x00) && hasNext(buffer));
211             }
212             // Get first byte of Tag Identifier
213             tag = b;
214             // Get rest of Tag identifier if required
215             if ((b & 0x1F) == 0x1F) {
216                 do {
217                     tag <<= 8;
218                     b = buffer.get();
219                      tag |= b;
220                     
221                 } while ((b & 0x80) == 0x80);
222             }
223             return tag;
224         }
225         
226         /*
227          * Read length bytes and return the int value
228          * @return value length
229          */
230         private int getValueLength(ByteBuffer buffer) {
231             int length = 0;
232             int count = 0;
233             byte b;
234             b = buffer.get();
235             count = b;
236             if ((count & 0x80) == 0x80) {
237                 // check first byte for more bytes to follow
238                 count -= 0x80;
239                 for (length = 0; count > 0; count--) {
240                     length <<= 8;
241                     b = buffer.get();
242                     length |= b;
243                 }
244             } else {
245                 length = count;
246             }
247             return length;
248         }
