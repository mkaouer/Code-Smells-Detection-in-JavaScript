116     /**
117      * Called from ISOChannelPanel when you click on it's ISOMeter.
118      * It enable field and header visualization by means of visual
119      * components such as JTable
120      *
121      * @see ISOChannelPanel
122      * @see ISORequestListenerPanel
123      */
124     public class ISOMsgPanel extends JPanel {
125         /**
126          * @serial
127          */
128         ISOMsg m;
129         /**
130          * @serial
131          */
132         Vector validFields;
133         public ISOMsgPanel(ISOMsg m, boolean withDump) {
134             super();
135             this.m = m;
136             setLayout(new BorderLayout());
137             setBorder(BorderFactory.createRaisedBevelBorder());
138             setValidFields();
139             add(createISOMsgTable(), BorderLayout.CENTER);
140             if (withDump)
141                 add(createISOMsgDumpPanel(), BorderLayout.SOUTH);
142         }
143         public ISOMsgPanel(ISOMsg m) {
144             this(m, true);
145         }
146         private void setValidFields() {
147             validFields = new Vector();
148             for (int i=0; i<=128; i++)
149                 if (m.hasField(i))
150                     validFields.addElement(new Integer(i));
151         }
152         private JComponent createISOMsgTable() {
153             TableModel dataModel = new AbstractTableModel() {
154                 public int getColumnCount() {
155                     return 3;
156                 }
157                 public int getRowCount() {
158                     return validFields.size();
159                 }
160                 public String getColumnName(int columnIndex) {
161                     switch (columnIndex) {
162                         case 0 :
163                             return "#";
164                         case 1 :
165                             return "Content";
166                         case 2 :
167                             return "Description";
168                         default:
169                             return "";
170                     }
171                 }
172                 public Object getValueAt(int row, int col) {
173                     switch (col) {
174                         case 0 :
175                             return ((Integer)validFields.elementAt(row));
176                         case 1 :
177                             try {
178                                 int index =
179                                 ((Integer)validFields.elementAt(row)).intValue();
180     
181                                 Object obj = m.getValue(index);
182                                 if (obj instanceof String) 
183                                     return obj.toString();
184                                 else if (obj instanceof byte[])
185                                     return ISOUtil.hexString((byte[]) obj);
186                                 else if (obj instanceof ISOMsg)
187                                     return "<ISOMsg>";
188                             } catch (ISOException e) {
189                                 e.printStackTrace();
190                             }   
191                             break;
192                         case 2 :
193                             int i=((Integer)validFields.elementAt(row)).intValue();
194                             ISOPackager p = m.getPackager();
195                             return p.getFieldDescription(m,i);
196                     }
197                     return "<???>";
198                 }
199             };
200             JTable table = new JTable(dataModel);
201             table.getColumnModel().getColumn(0).setPreferredWidth(10);
202             table.setPreferredScrollableViewportSize(
203                 new Dimension (500,table.getRowCount()*table.getRowHeight()));
204             table.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
205     
206             ListSelectionModel rowSM = table.getSelectionModel();
207             rowSM.addListSelectionListener(new ListSelectionListener() {
208                 public void valueChanged(ListSelectionEvent e) {
209                     if (e.getValueIsAdjusting())
210                         return;
211     
212                     ListSelectionModel lsm =
213                         (ListSelectionModel)e.getSource();
214                     if (!lsm.isSelectionEmpty()) {
215                         int selectedRow = lsm.getMinSelectionIndex();
216                         try {
217                             int index = ((Integer)
218                                 validFields.elementAt(selectedRow)).intValue();
219     
220                             Object obj = m.getValue(index);
221                             if (obj instanceof ISOMsg) {
222                                 ISOMsg sm = (ISOMsg) obj;
223                                 JFrame f = new JFrame("ISOMsg field "+index);
224                                 ISOMsgPanel p = new ISOMsgPanel(sm, false);
225                                 f.getContentPane().add(p);
226                                 f.pack();
227                                 f.show();
228                             }
229                         } catch (ISOException ex) {
230                             ex.printStackTrace();
231                         }   
232                     }
233                 }
234             });
235     
236             JScrollPane scrollpane = new JScrollPane(table);
237             return scrollpane;
238         }
239         JComponent createISOMsgDumpPanel() {
240             JPanel p = new JPanel();
241             JTextArea t = new JTextArea(3,20);
242     
243             p.setLayout(new BorderLayout());
244             p.setBackground(Color.white);
245             p.setBorder(BorderFactory.createLoweredBevelBorder());
246             p.add(new JLabel("Dump", SwingConstants.LEFT),
247                 BorderLayout.NORTH);
248             t.setFont(new Font ("Helvetica", Font.PLAIN, 8));
249             t.setLineWrap(true);
250             try {
251                 StringBuffer buf = new StringBuffer();
252                 if (m.getHeader() != null) 
253                     buf.append("--[Header]--\n" 
254                         +ISOUtil.hexString(m.getHeader()) + "\n--[Msg]--\n");
255                 byte[] b = m.pack();
256                 buf.append (ISOUtil.hexString(b));
257                 t.setText(buf.toString());
258             } catch (ISOException e) {
259                 t.setText(e.toString());
260                 t.setForeground(Color.red);
261             }
262             p.add(t, BorderLayout.CENTER);
263             return p;
264         }
