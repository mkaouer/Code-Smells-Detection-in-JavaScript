8     package org.jpos.q2.qbean;
9     
10     import org.jdom.Element;
11     import org.jpos.q2.QBeanSupport;
12     
13     import bsh.BshClassManager;
14     import bsh.Interpreter;
15     import bsh.UtilEvalError;
16     
17     public class BSH extends QBeanSupport implements Runnable {
18         Interpreter bsh;
19         public void initService() {
20             bsh = new Interpreter ();
21             BshClassManager bcm = bsh.getClassManager();
22             try {
23                 bcm.setClassPath(getServer().getLoader().getURLs());
24             } catch (UtilEvalError e) {
25                 e.printStackTrace();
26             }
27             bcm.setClassLoader(getServer().getLoader());
28         }
29         public void startService() {
30             new Thread (this).start ();
31         }
32         public void run () {
33             Element config = getPersist();
34             try {
35                 bsh.set  ("qbean", this);
36                 bsh.set  ("log", getLog());
37                 bsh.eval (config.getText());
38                 String source = config.getAttributeValue ("source");
39                 if (source != null)
40                     bsh.source (source);
41             } catch (Throwable e) {
42                 getLog().warn (e);
43             }
44         }
45     }
