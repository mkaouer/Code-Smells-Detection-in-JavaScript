26     /**
27      * ISO Server wrapper.
28      *
29      * @author Alwyn Schoeman
30      * @version $Revision$ $Date$
31      * @jmx:mbean description="ISOServer wrapper"
32      *                  extends="org.jpos.q2.QBeanSupportMBean"
33      */
34     
35     public class QServer
36         extends QBeanSupport
37         implements QServerMBean 
38     {
39         private int port = 0;
40         private int maxSessions = 100;
41         private int minSessions = 1;
42         private String channelString, packagerString, socketFactoryString;
43         private ISOChannel channel = null;
44         private ISOPackager packager = null;
45         private ISOServer server;
46     
47         public QServer () {
48             super ();
49         }
50     
51         private void newChannel () throws ConfigurationException {
52             Element persist = getPersist ();
53             Element e = persist.getChild ("channel");
54             if (e == null)
55                 throw new ConfigurationException ("channel element missing");
56     
57             ChannelAdaptor adaptor = new ChannelAdaptor ();
58             channel = adaptor.newChannel (e, getFactory ());
59         }
60     
61         private void initServer () 
62             throws ConfigurationException
63         {
64             if (port == 0)
65                 throw new ConfigurationException ("Port value not set");
66             newChannel();
67             if (channel == null)
68                 throw new ConfigurationException ("ISO Channel is null");
69     
70             if (!(channel instanceof ServerChannel)) {
71                 throw new ConfigurationException (channelString + 
72                       "does not implement ServerChannel");
73             }
74     
75             ThreadPool pool = null;
76             pool = new ThreadPool (minSessions ,maxSessions);
77             pool.setLogger (log.getLogger(), getName() + ".pool");
78     
79             server = new ISOServer (port, (ServerChannel) channel, pool);
80             server.setLogger (log.getLogger(), getName() + ".server");
81             server.setName (getName ());
82             if (socketFactoryString != null) {
83                 ISOServerSocketFactory sFac = (ISOServerSocketFactory) getFactory().newInstance(socketFactoryString);
84                 if (sFac != null && sFac instanceof LogSource) {
85                     ((LogSource) sFac).setLogger(log.getLogger(),getName() + ".socket-factory");
86                 }
87                 server.setSocketFactory(sFac);
88             }
89             getFactory().setConfiguration (server, getPersist());     
90             addListeners ();
91             new Thread (server).start();
92         }
93         public void startService () {
94             try {
95                 initServer ();
96             } catch (Exception e) {
97                 getLog().warn ("error starting service", e);
98             }
99         }
100         public void stopService () {
101             NameRegistrar.unregister ("server." + getName());
102             server.shutdown ();
103         }

194         }    
195         private void addListeners () 
196             throws ConfigurationException
197         {
198             QFactory factory = getFactory ();
199             Iterator iter = getPersist().getChildren (
200                 "request-listener"
201             ).iterator();
202             while (iter.hasNext()) {
203                 Element l = (Element) iter.next();
204                 ISORequestListener listener = (ISORequestListener) 
205                     factory.newInstance (l.getAttributeValue ("class"));
206                 factory.setLogger        (listener, l);
207                 factory.setConfiguration (listener, l);
208                 server.addISORequestListener (listener);
209             }
210         }
