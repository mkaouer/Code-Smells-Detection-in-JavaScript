7     package org.jpos.bsh;
8     
9     import java.io.Serializable;
10     import org.jpos.core.ConfigurationException;
11     import org.jpos.transaction.GroupSelector;
12     import org.jpos.util.Log;
13     import org.jpos.util.LogEvent;
14     import org.jpos.util.Logger;
15     
16     /**
17      * @author  AMarques
18      */
19     public class BSHGroupSelector extends BSHTransactionParticipant implements GroupSelector {
20         
21         protected BSHMethod selectMethod;
22         
23         public void setConfiguration(org.jdom.Element e) throws ConfigurationException {
24             super.setConfiguration(e);
25             try {
26                 selectMethod = BSHMethod.createBshMethod(e.getChild("select"));
27             } catch (Exception ex) {
28                 throw new ConfigurationException(ex.getMessage(), ex);
29             }
30         }    
31         
32         public String select(long id, java.io.Serializable context) {
33             LogEvent ev = new LogEvent(this, "select");
34             String result = null;
35             if (selectMethod != null) {
36                 try {
37                     result = (String) executeMethod(selectMethod, id, context, ev, "result");
38                 } catch (Exception ex) {
39                     ev.addMessage(ex);
40                 }
41             } 
42             if (result == null) {
43                 result = defaultSelect(id, context);
44             }
45             ev.addMessage("result", result);
46             Logger.log(ev);
47             return result;
48         }
49         
50         public String defaultSelect(long id, Serializable context) {
51             return "";
52         }
53             
54     }
