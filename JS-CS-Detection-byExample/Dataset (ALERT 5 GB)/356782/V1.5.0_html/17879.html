124         /**
125          * Validate that component is not zero-filled.
126          */
127         public ISOComponent validate ( ISOComponent f ) throws ISOException {
128             ISOField c = (ISOField)f;
129             try {
130                 /** numeric **/
131                 c = (ISOField)super.validate( c );
132                 /** positive **/
133                 if (ISOUtil.isZero( (String)c.getValue() ) ){
134                     ISOVError e = new ISOVError(
135                             "Invalid Value Error. It can not be zero-filled. (Current value: " +
136                             (String)c.getValue()+ ") ",
137                             getRejCode( ISOVError.ERR_INVALID_VALUE ) );
138                     if ( c instanceof ISOVField )
139                         ((ISOVField)c).addISOVError( e );
140                     else
141                         c = new ISOVField( c, e );
142                     if ( breakOnError )
143                         throw new ISOVException ( "Error on field " + ((Integer)c.getKey()).intValue(), c );
144                 }
145                 return c;
146             }
147             catch (Exception ex) {
148                 /** This catch is useful in case of error-dependencies. If
149                  an error take place in super, and it imply second in child. **/
150                 if ( ex instanceof ISOVException ) throw (ISOVException)ex;
151                 return c;
152             }
153         }
