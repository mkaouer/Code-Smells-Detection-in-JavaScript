77     /**
78      * <p>
79      * JCESecurityModule is an implementation of a security module in software.
80      * It doesn't require any hardware device to work.<br>
81      * JCESecurityModule also implements the SMAdapter, so you can view it: either
82      * as a self contained security module adapter that doesn't need a security module
83      * or a security module that plugs directly to jpos, so doesn't need
84      * a separate adapter.<br>
85      * It relies on Java(tm) Cryptography Extension (JCE), hence its name.<br>
86      * JCESecurityModule relies on the JCEHandler class to do the low level JCE work.
87      * </p>
88      * <p>
89      * WARNING: This version of JCESecurityModule is meant for testing purposes and
90      * NOT for life operation, since the Local Master Keys are stored in CLEAR on
91      * the system's disk. Comming versions of JCESecurity Module will rely on
92      * java.security.KeyStore for a better protection of the Local Master Keys.
93      * </p>
94      * @author Hani Samuel Kirollos
95      * @version $Revision$ $Date$
96      */

188             EncryptedPIN pinUnderLmk = null;
189             // read inputs
190             String accountNumber = pinUnderKd1.getAccountNumber();
191             // Use FORMAT00 for encrypting PIN under LMK

203             pinUnderLmk = new EncryptedPIN(translatedPINBlock, destinationPINBlockFormat,
204                     accountNumber);
205             return  pinUnderLmk;

309         /**
310          * Forms a key from 3 clear components and returns it encrypted under its corresponding LMK
311          * The corresponding LMK is determined from the keyType
312          * @param keyLength e.g. LENGTH_DES, LENGTH_DES3_2, LENGTH_DES3_3, ..
313          * @param keyType possible values are those defined in the SecurityModule inteface. e.g., ZMK, TMK,...
314          * @param clearComponent1HexString HexString containing the first component
315          * @param clearComponent2HexString HexString containing the second component
316          * @param clearComponent3HexString HexString containing the second component
317          * @return forms an SecureDESKey from two clear components
318          * @throws SMException
319          */
320         SecureDESKey formKEYfromThreeClearComponents (short keyLength, String keyType,
321                 String clearComponent1HexString, String clearComponent2HexString, String clearComponent3HexString) throws SMException {
322             SecureDESKey secureDESKey;
323             SimpleMsg[] cmdParameters =  {
324                 new SimpleMsg("parameter", "Key Length", keyLength),
325                 new SimpleMsg("parameter", "Key Type", keyType),
326                 new SimpleMsg("parameter", "Clear Componenent 1", clearComponent1HexString),
327                 new SimpleMsg("parameter", "Clear Componenent 2", clearComponent2HexString),
328                 new SimpleMsg("parameter", "Clear Componenent 3", clearComponent3HexString)
329             };
330             LogEvent evt = new LogEvent(this, "s-m-operation");
331             evt.addMessage(new SimpleMsg("command", "Form Key from Three Clear Components", cmdParameters));
332             try {
333                 byte[] clearComponent1 = ISOUtil.hex2byte(clearComponent1HexString);
334                 byte[] clearComponent2 = ISOUtil.hex2byte(clearComponent2HexString);
335                 byte[] clearComponent3 = ISOUtil.hex2byte(clearComponent3HexString);
336                 byte[] clearKeyBytes = ISOUtil.xor(ISOUtil.xor(clearComponent1, clearComponent2),
337                         clearComponent3);
338                 Key clearKey = null;
339                 clearKey = jceHandler.formDESKey(keyLength, clearKeyBytes);
340                 secureDESKey = encryptToLMK(keyLength, keyType, clearKey);
341                 evt.addMessage(new SimpleMsg("result", "Formed Key", secureDESKey));
342             } catch (JCEHandlerException e) {
343                 evt.addMessage(e);
344                 throw  e;
345             } finally {
346                 Logger.log(evt);
347             }
348             return  secureDESKey;
349         }

395         /**
396          * Calculates the clear PIN Block
397          * @param pin as entered by the card holder on the PIN entry device
398          * @param pinBlockFormat
399          * @param accountNumber (the 12 right-most digits of the account number excluding the check digit)
400          * @return The clear PIN Block
401          * @throws SMException
402          *
403          */
404         private byte[] calculatePINBlock (String pin, byte pinBlockFormat, String accountNumber) throws SMException {
405             byte[] pinBlock = null;
406             if (pin.length() > MAX_PIN_LENGTH)
407                 throw  new SMException("Invalid PIN length: " + pin.length());
408             if (accountNumber.length() != 12)
409                 throw  new SMException("Invalid Account Number: " + accountNumber + ". The length of the account number must be 12 (the 12 right-most digits of the account number excluding the check digit)");
410             switch (pinBlockFormat) {
411                 case FORMAT00: // same as FORMAT01
412                 case FORMAT01:
413                     {
414                         // Block 1
415                         String block1 = null;
416                         byte[] block1ByteArray;
417                         switch (pin.length()) {
418                             // pin length then pad with 'F'
419                             case 4:
420                                 block1 = "04" + pin + "FFFFFFFFFF";
421                                 break;
422                             case 5:
423                                 block1 = "05" + pin + "FFFFFFFFF";
424                                 break;
425                             case 6:
426                                 block1 = "06" + pin + "FFFFFFFF";
427                                 break;
428                             default:
429                                 throw  new SMException("Unsupported PIN Length: " +
430                                         pin.length());
431                         }
432                         block1ByteArray = ISOUtil.hex2byte(block1);
433                         // Block 2
434                         String block2;
435                         byte[] block2ByteArray = null;
436                         block2 = "0000" + accountNumber;
437                         block2ByteArray = ISOUtil.hex2byte(block2);
438                         // pinBlock
439                         pinBlock = ISOUtil.xor(block1ByteArray, block2ByteArray);
440                     }
441                     ;
442                     break;
443                 case FORMAT03: 
444                     {
445                         if(pin.length() < 4 || pin.length() > 12) 
446                             throw new SMException("Unsupported PIN Length: " + 
447                                     pin.length());
448                         pinBlock = ISOUtil.hex2byte (
449                             pin + "FFFFFFFFFFFFFFFF".substring(pin.length(),16)
450                         );
451                     }
452                     break;
453                 default:
454                     throw  new SMException("Unsupported PIN format: " + pinBlockFormat);
455             }
456             return  pinBlock;
457         }

459         /**
460          * Calculates the clear pin (as entered by card holder on the pin entry device)
461          * givin the clear PIN block
462          * @param pinBlock clear PIN Block
463          * @param pinBlockFormat
464          * @param accountNumber
465          * @return the pin
466          * @throws SMException
467          */
468         private String calculatePIN (byte[] pinBlock, byte pinBlockFormat, String accountNumber) throws SMException {
469             String pin = null;
470             int pinLength;
471             if (accountNumber.length() != 12)
472                 throw  new SMException("Invalid Account Number: " + accountNumber + ". The length of the account number must be 12 (the 12 right-most digits of the account number excluding the check digit)");
473             switch (pinBlockFormat) {
474                 case FORMAT00: // same as format 01
475                 case FORMAT01:
476                     {
477                         // Block 2
478                         String block2;
479                         block2 = "0000" + accountNumber;
480                         byte[] block2ByteArray = ISOUtil.hex2byte(block2);
481                         // get Block1
482                         byte[] block1ByteArray = ISOUtil.xor(pinBlock, block2ByteArray);
483                         pinLength = Math.abs (block1ByteArray[0]);
484                         if (pinLength > MAX_PIN_LENGTH)
485                             throw  new SMException("PIN Block Error");
486                         // get pin
487                         String pinBlockHexString = ISOUtil.hexString(block1ByteArray);
488                         pin = pinBlockHexString.substring(2, pinLength
489                                 + 2);
490                         String pad = pinBlockHexString.substring(pinLength + 2);
491                         pad = pad.toUpperCase();
492                         int i = pad.length();
493                         while (--i >= 0)
494                             if (pad.charAt(i) != 'F')
495                                 throw new SMException("PIN Block Error");
496                     }
497                     break;
498                 case FORMAT03: 
499                     {
500                         String block1 = ISOUtil.hexString(pinBlock);
501                         int len = block1.indexOf('F');
502                         if(len == -1) len = 12;
503                         int i = block1.length();
504                         pin = block1.substring(0, len);
505     
506                         while(--i >= len) 
507                             if(block1.charAt(i) != 'F') 
508                                 throw new SMException("PIN Block Error");
509                         while(--i >= 0) 
510                             if(pin.charAt(i) >= 'A') 
511                                 throw new SMException("PIN Block Error");
512     
513                         if(pin.length() < 4 || pin.length() > 12) 
514                             throw new SMException("Unsupported PIN Length: " + 
515                                     pin.length());
516                     }
517                     break;
518                 default:
519                     throw  new SMException("Unsupported PIN Block format: " + pinBlockFormat);
520             }
521             return  pin;
522         }

524         /**
525          * Initializes the JCE Security Module
526          * @param jceProviderClassName
527          * @param lmkFile Local Master Keys File used by JCE Security Module to store the LMKs
528          * @param lmkRebuild if set to true, the lmkFile gets overwritten with newly generated keys (WARNING: this would render all your previously stored SecureKeys unusable)
529          * @throws SMException
530          */
531         private void init (String jceProviderClassName, String lmkFile, boolean lmkRebuild) throws SMException {
532             File lmk = new File(lmkFile);
533             try {
534                 keyTypeToLMKIndex = new Hashtable();
535                 keyTypeToLMKIndex.put(SMAdapter.TYPE_ZMK, new Integer(0));
536                 keyTypeToLMKIndex.put(SMAdapter.TYPE_ZPK, new Integer(1));
537                 keyTypeToLMKIndex.put(SMAdapter.TYPE_PVK, new Integer(2));
538                 keyTypeToLMKIndex.put(SMAdapter.TYPE_TPK, new Integer(2));
539                 keyTypeToLMKIndex.put(SMAdapter.TYPE_TMK, new Integer(2));
540                 keyTypeToLMKIndex.put(SMAdapter.TYPE_TAK, new Integer(3));
541                 keyTypeToLMKIndex.put(PINLMKIndex, new Integer(4));
542                 keyTypeToLMKIndex.put(SMAdapter.TYPE_CVK, new Integer(5));
543                 keyTypeToLMKIndex.put(SMAdapter.TYPE_ZAK, new Integer(8));
544                 keyTypeToLMKIndex.put(SMAdapter.TYPE_BDK, new Integer(9));
545                 Provider provider = null;
546                 LogEvent evt = new LogEvent(this, "jce-provider");
547                 try {
548                     if ((jceProviderClassName == null) || (jceProviderClassName.compareTo("")
549                             == 0)) {
550                         evt.addMessage("No JCE Provider specified. Attempting to load default provider (SunJCE).");
551                         jceProviderClassName = "com.sun.crypto.provider.SunJCE";
552                     }
553                     provider = (Provider)Class.forName(jceProviderClassName).newInstance();
554                     Security.addProvider(provider);
555                     evt.addMessage("name", provider.getName());
556                 } catch (Exception e) {
557                     evt.addMessage(e);
558                     throw  new SMException("Unable to load jce provider whose class name is: "
559                             + jceProviderClassName);
560                 } finally {
561                     Logger.log(evt);
562                 }
563                 jceHandler = new JCEHandler(provider);
564                 // Load Local Master Keys
565                 Properties lmkProps = new Properties();
566                 if (lmkRebuild) {
567                     // Creat new LMK file
568                     evt = new LogEvent(this, "local-master-keys");
569                     evt.addMessage("Rebuilding new Local Master Keys in file: \"" +
570                             lmk.getCanonicalPath() + "\".");
571                     Logger.log(evt);
572                     // Generate New random Local Master Keys
573                     generateLMK();
574                     // Write the new Local Master Keys to file
575                     writeLMK(lmk);
576                     evt = new LogEvent(this, "local-master-keys");
577                     evt.addMessage("Local Master Keys built successfully in file: \""
578                             + lmk.getCanonicalPath() + "\".");
579                     Logger.log(evt);
580                 }
581                 if (!lmk.exists()) {
582                     // LMK File does not exist
583                     throw  new SMException("Error loading Local Master Keys, file: \""
584                             + lmk.getCanonicalPath() + "\" does not exist." + " Please specify a valid LMK file, or rebuild a new one.");
585                 }
586                 else {
587                     // Read LMK from file
588                     readLMK(lmk);
589                     evt = new LogEvent(this, "local-master-keys");
590                     evt.addMessage("Loaded successfully from file: \"" + lmk.getCanonicalPath()
591                             + "\"");
592                     Logger.log(evt);
593                 }
594             } catch (Exception e) {
595                 if (e instanceof SMException) {
596                     throw  (SMException)e;
597                 }
598                 else {
599                     throw  new SMException(e);
600                 }
601             }
602         }

631         /**
632          * reads (loads) LMK's from lmkFile
633          * @param lmkFile
634          * @exception SMException
635          */
636         private void readLMK (File lmkFile) throws SMException {
637             LMK = new SecretKey[0x0f];
638             try {
639                 Properties lmkProps = new Properties();
640                 FileInputStream in = new FileInputStream(lmkFile);
641                 lmkProps.load(in);
642                 in.close();
643                 byte[] lmkData;
644                 for (int i = 0x00; i < 0x0f; i++) {
645                     lmkData = ISOUtil.hex2byte(lmkProps.getProperty("LMK0x0" +
646                             Integer.toHexString(i)));
647                     // provider-independent method
648                     LMK[i] = new SecretKeySpec(lmkData, JCEHandler.ALG_TRIPLE_DES);
649                 }
650             } catch (Exception e) {
651                 throw  new SMException("Can't read Local Master Keys from file: " +
652                         lmkFile, e);
653             }
654         }

656         /**
657          * Writes a newly generated LMK's to lmkFile
658          * @param lmkFile
659          * @exception SMException
660          */
661         private void writeLMK (File lmkFile) throws SMException {
662             Properties lmkProps = new Properties();
663             try {
664                 for (int i = 0x00; i < 0x0f; i++) {
665                     lmkProps.setProperty("LMK0x0" + Integer.toHexString(i), ISOUtil.hexString(LMK[i].getEncoded()));
666                 }
667                 FileOutputStream out = new FileOutputStream(lmkFile);
668                 lmkProps.store(out, "Local Master Keys");
669                 out.close();
670             } catch (Exception e) {
671                 throw  new SMException("Can't write Local Master Keys to file: " + lmkFile,
672                         e);
673             }
674         }

676         /**
677          * gets the suitable LMK for the key type
678          * @param keyType
679          * @return the LMK secret key for the givin key type
680          * @throws SMException
681          */
682         private SecretKey getLMK (String keyType) throws SMException {
683             //int lmkIndex = keyType;
684             if (!keyTypeToLMKIndex.containsKey(keyType)) {
685                 throw  new SMException("Unsupported key type: " + keyType);
686             }
687             int lmkIndex = ((Integer)keyTypeToLMKIndex.get(keyType)).intValue();
688             SecretKey lmk = null;
689             try {
690                 lmk = LMK[lmkIndex];
691             } catch (Exception e) {
692                 throw  new SMException("Invalid key code: " + "LMK0x0" + Integer.toHexString(lmkIndex));
693             }
694             return  lmk;
695         }
