7     package org.jpos.transaction;
8     
9     import java.io.Serializable;
10     
11     /**
12      * ContextRecovery can be implemented by a TransactionParticipant in
13      * order to customize the activation of a persisted context.
14      *
15      * @author apr
16      * @since 1.4.7
17      * @see TransactionParticipant
18      */
19     public interface ContextRecovery {
20         /**
21          * Give participant the chance to "activate" a previously 
22          * persisted context.
23          *
24          * @param id the Transaction identifier
25          * @param context transaction context (as persisted by TransactionManager)
26          * @param commit true if transaction is committing
27          * @return activated context
28          */
29         public Serializable recover (long id, Serializable context, boolean commit);
30     }
