40     package iso;
41     
42     import junit.framework.TestCase;
43     
44     import org.jpos.iso.LeftPadder;
45     
46     /**
47      * @author joconnor
48      */
49     public class LeftPadderTest extends TestCase {
50         private LeftPadder padder;
51         
52         protected void setUp() {
53             padder = LeftPadder.ZERO_PADDER;
54         }
55         
56         public void testPaddingNeeded() throws Exception {
57             assertEquals("000123", padder.pad("123", 6));
58         }
59     
60         public void testNoPaddingNeeded() throws Exception {
61             assertEquals("123", padder.pad("123", 3));
62         }
63     
64         public void testPadLengthTooShort() throws Exception {
65             try {
66                 padder.pad("123", 2);
67                 fail("Padding a string longer than the available width should throw an exception");
68             } catch (Exception asIExpected) {
69             }
70         }
71     
72         public void testUnpad() throws Exception {
73             assertEquals("123", padder.unpad("000123"));
74         }
75     
76         public void testUnpadAllPadding() throws Exception {
77             assertEquals("", padder.unpad("000"));
78         }
79     
80         public void testReversability() throws Exception {
81             String origin = "Abc";
82             assertEquals(origin, padder.unpad(padder.pad(origin, 6)));
83         }
84     }
