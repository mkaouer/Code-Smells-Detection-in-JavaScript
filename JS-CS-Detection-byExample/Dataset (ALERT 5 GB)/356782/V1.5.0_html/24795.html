50     package org.jpos.iso;
51     
52     import java.io.EOFException;
53     import java.io.IOException;
54     import java.io.InputStream;
55     import java.io.ObjectOutput;
56     
57     /**
58      * base class for the various IF*.java Field Packagers
59      * Implements "FlyWeight" pattern
60      *
61      * @author apr@cs.com.uy
62      * @version $Id$
63      *
64      * @see IFA_AMOUNT
65      * @see IFA_BINARY
66      * @see IFA_BITMAP
67      * @see IFA_FLLCHAR
68      * @see IFA_FLLNUM
69      * @see IFA_LLCHAR
70      * @see IFA_LLLBINARY
71      * @see IFA_LLLCHAR
72      * @see IFA_LLLNUM
73      * @see IFA_LLNUM
74      * @see IFA_NUMERIC
75      * @see IFB_AMOUNT
76      * @see IFB_BINARY
77      * @see IFB_BITMAP
78      * @see IFB_LLBINARY
79      * @see IFB_LLCHAR
80      * @see IFB_LLHBINARY
81      * @see IFB_LLHCHAR
82      * @see IFB_LLHECHAR
83      * @see IFB_LLHNUM
84      * @see IFB_LLLBINARY
85      * @see IFB_LLLCHAR
86      * @see IFB_LLLNUM
87      * @see IFB_LLNUM
88      * @see IFB_NUMERIC
89      * @see IF_CHAR
90      * @see IF_ECHAR
91      */
92     public abstract class ISOFieldPackager {
93         private int len;
94         private String description;
95         protected boolean pad;
96     
97         /**
98          * Default Constructor
99          */
100         public ISOFieldPackager()
101         {
102             this.len = -1;
103             this.description = null;
104         }
105     
106         /**
107          * @param len - field Len
108          * @param description - details
109          */
110         public ISOFieldPackager(int len, String description) {
111             this.len = len;
112             this.description = description;
113         }
114         public String getDescription() {
115             return description;
116         }
117         public void setDescription(String description) {
118             this.description = description;
119         }
120         public int getLength() {
121             return len;
122         }
123         public void setLength(int len) {
124             this.len = len;
125         }
126     
127         public void setPad(boolean pad) {
128             this.pad = pad;
129         }
130     
131         public abstract int getMaxPackedLength();
132     
133         public ISOComponent createComponent(int fieldNumber) {
134             return new ISOField (fieldNumber);
135         }
136         /**
137          * @param c - a component
138          * @return packed component
139          * @exception ISOException
140          */
141         public abstract byte[] pack (ISOComponent c) throws ISOException;
142     
143         /**
144          * @param c - the Component to unpack
145          * @param b - binary image
146          * @param offset - starting offset within the binary image
147          * @return consumed bytes
148          * @exception ISOException
149          */
150         public abstract int unpack (ISOComponent c, byte[] b, int offset)
151             throws ISOException;
152     
153         /**
154          * @param c  - the Component to unpack
155          * @param in - input stream
156          * @exception ISOException
157          */
158         public void unpack (ISOComponent c, InputStream in) 
159             throws IOException, ISOException
160         {
161             unpack (c, readBytes (in, getMaxPackedLength ()), 0);
162         }
163         /**
164          * @param c   - the Component to unpack
165          * @param out - output stream
166          * @exception ISOException
167          * @exception IOException
168          */
169         public void pack (ISOComponent c, ObjectOutput out) 
170             throws IOException, ISOException
171         {
172             out.write (pack (c));
173         }
174     
175         protected byte[] readBytes (InputStream in, int l) throws IOException {
176             byte[] b = new byte [l];
177             int n = 0;
178             while (n < l) {
179                 int count = in.read(b, n, l - n);
180                 if (count < 0)
181                     throw new EOFException();
182                 n += count;
183             }
184             return b;
185         }
186     }
