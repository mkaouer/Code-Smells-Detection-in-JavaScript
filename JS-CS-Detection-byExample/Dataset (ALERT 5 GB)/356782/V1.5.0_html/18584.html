50     package org.jpos.apps.qsp.config;
51     
52     import javax.management.NotCompliantMBeanException;
53     
54     import org.jpos.apps.qsp.QSP;
55     import org.jpos.apps.qsp.QSPConfigurator;
56     import org.jpos.core.ConfigurationException;
57     import org.jpos.util.LogEvent;
58     import org.jpos.util.Logger;
59     import org.jpos.util.NameRegistrar;
60     import org.jpos.util.ThreadPool;
61     import org.jpos.util.NameRegistrar.NotFoundException;
62     import org.w3c.dom.NamedNodeMap;
63     import org.w3c.dom.Node;
64     
65     /**
66      * Configure ThreadPool
67      * @author <A href=mailto:alcarraz@fing.edu.uy>Andr&eacute;s Alcarraz</A>
68      * @version $Revision$ $Date$
69      */
70     public class ConfigThreadPool implements QSPConfigurator {
71         public void config (QSP qsp, Node node) throws ConfigurationException
72         {
73             LogEvent evt = new LogEvent (qsp, "config-thread-pool");
74             try{
75                 NamedNodeMap atts = node.getAttributes();
76                 int initialSize = Integer.parseInt(atts.getNamedItem("initial-size").getNodeValue());
77                 int maxSize = Integer.parseInt(atts.getNamedItem("max-size").getNodeValue());
78                 ThreadPool threadPool = new ThreadPool (initialSize, maxSize);
79                 threadPool.setLogger (
80                     ConfigLogger.getLogger (node),
81                     ConfigLogger.getRealm (node)
82                 );
83                 evt.addMessage (threadPool);
84                 String name = atts.getNamedItem ("name").getNodeValue();
85                 NameRegistrar.register ("thread.pool."+name,threadPool);
86                 try{
87                     qsp.registerMBean(threadPool, "type=thread-pool,name=" + name);
88                 }catch (NotCompliantMBeanException e){
89                     evt.addMessage(e.getMessage());
90                 }
91             } catch (Exception e){
92                 evt.addMessage(e);
93             } finally{
94                 Logger.log (evt);
95             }
96         }
97         public static ThreadPool getThreadPool (Node node) {
98             Node n = node.getAttributes().getNamedItem ("thread-pool");
99             if (n != null) {
100                 try {
101                     return ThreadPool.getThreadPool (n.getNodeValue ());
102                 } catch (NotFoundException e) { }
103             }
104             return null;
105         }
106     }
