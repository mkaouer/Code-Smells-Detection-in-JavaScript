8     package org.jpos.iso;
9     
10     import java.io.IOException;
11     
12     /**
13      * allows the transmision and reception of ISO 8583 Messages
14      *
15      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
16      * @author <a href="mailto:alwynschoeman@yahoo.com">Alwyn Schoeman</a>
17      * @version $Revision$ $Date$
18      */
19     public interface ISOChannel extends ISOSource {
20         public static final int CONNECT      = 0;
21         public static final int TX           = 1;
22         public static final int RX           = 2;
23         public static final int SIZEOF_CNT   = 3;
24     
25         /**
26          * Associate a packager with this channel
27          * @param p     an ISOPackager
28          */
29         public void setPackager(ISOPackager p);
30     
31         /**
32          * Connects ISOChannel 
33          * @exception IOException
34          */
35         public void connect () throws IOException;
36     
37         /**
38          * disconnects ISOChannel
39          * @exception IOException
40          */
41         public void disconnect () throws IOException;
42     
43         /**
44          * Reconnect channel
45          * @exception IOException
46          */
47         public void reconnect() throws IOException;
48     
49         /**
50          * @return true if Channel is connected and usable
51          */
52         public boolean isConnected();
53     
54         /**
55          * Receives an ISOMsg
56          * @return the Message received
57          * @exception IOException
58          * @exception ISOException
59          */
60         public ISOMsg receive() throws IOException, ISOException;
61     
62         /**
63          * sends an ISOMsg over the TCP/IP session
64          * @param m the Message to be sent
65          * @exception IOException
66          * @exception ISOException
67          */
68         public void send (ISOMsg m) throws IOException, ISOException;
69     
70         /**
71          * @param b - usable state
72          */
73         public void setUsable(boolean b);
74     
75         /**
76          * associates this ISOChannel with a name on NameRegistrar
77          * @param name name to register
78          * @see NameRegistrar
79          */
80         public void setName (String name);
81     
82        /**
83         * @return this ISOChannel's name ("" if no name was set)
84         */
85         public String getName();
86     
87        /**
88         * @return current packager
89         */
90         public ISOPackager getPackager();
91         
92     }
