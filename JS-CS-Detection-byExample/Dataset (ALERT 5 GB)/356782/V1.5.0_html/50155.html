50     package org.jpos.util;
51     
52     import java.io.ByteArrayOutputStream;
53     import java.io.PrintStream;
54     
55     import org.apache.log4j.Level;
56     import org.apache.log4j.xml.DOMConfigurator;
57     import org.jpos.core.Configuration;
58     import org.jpos.core.ConfigurationException;
59     import org.jpos.core.ReConfigurable;
60     
61     /**
62      * @author Eoin P. FLood</a>
63      * @version $Revision$ $Date$
64      * @see org.jpos.util.LogListener
65      * @see org.jpos.util.LogEvent
66      * @since jPOS 1.3
67      *
68      * This class acts as a simple bridge between jPOS's logging system
69      * and log4j.
70      * The jPOS <code>realm</code> is used as the log4j <code>Logger</code>
71      * and messages are by default logged with the DEBUG level. This can
72      * be changed by calling <code>setLevel</code>
73      */
74     
75     public class Log4JListener implements LogListener, ReConfigurable
76     {
77         private Level _level;
78         /** 
79          * Create a new Log4JListener with DEBUG level.
80          */
81         public Log4JListener ()
82         {
83             setLevel (Level.DEBUG_INT);
84         }
85     
86         public void setLevel (int level) {
87             _level = Level.toLevel (level);
88         }
89     
90         public void setLevel (String level) {
91             _level = Level.toLevel (level);
92         }
93     
94         public void close() 
95         {
96         }
97     
98         /**
99          * Expects the following properties:
100          * <ul>
101          *  <li>config   - Configuration file path
102          *  <li>priority - Log4J priority (debug, info, warn, error)
103          *  <li>watch    - interval (in ms) to monitor XML config file for changes 
104          * </ul>
105          */
106         public void setConfiguration (Configuration cfg) 
107             throws ConfigurationException
108         {
109             DOMConfigurator.configureAndWatch (
110                 cfg.get ("config"), cfg.getLong ("watch")
111             );
112             setLevel (cfg.get ("priority"));
113         }
114     
115         public synchronized LogEvent log (LogEvent ev) {
116     
117             org.apache.log4j.Logger logger = org.apache.log4j.Logger.getLogger (
118                 ev.getRealm().replace('/', ':')
119             );
120             if (logger.isEnabledFor ( _level)) {
121                 ByteArrayOutputStream w = new ByteArrayOutputStream ();
122                 PrintStream p = new PrintStream (w);
123                 ev.dump (p, "");
124                 logger.log (_level, w.toString());
125             }
126             return ev;
127         }
128     }
