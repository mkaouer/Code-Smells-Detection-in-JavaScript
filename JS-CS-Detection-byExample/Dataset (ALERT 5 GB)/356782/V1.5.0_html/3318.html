87         public void out (Object key, Object value) {
88             List listeners;
89             synchronized (this) {
90                 Data data = (Data) map.get (key);
91                 if (data == null) 
92                     map.put (key, (data = new Data (key)));
93                 data.add (value);
94                 listeners = data.getListeners();
95                 this.notifyAll ();
96             }
97             if (listeners != null) {
98                 Iterator iter = listeners.iterator();
99                 while (iter.hasNext()) {
100                     ((SpaceListener)iter.next()).notify (key, value);
101                 }
102             }
103         }
104         public void out (Object id, Object value, long timeout) {
105             LeasedReference ref = new LeasedReference (value, timeout);
106             out (id, ref);
107         }

126         public synchronized Object in (Object key) {
127             Object obj;
128             while ((obj = inp (key)) == null) {
129                 try {
130                     this.wait ();
131                 } catch (InterruptedException e) { }
132             }
133             return obj;
134         }
135         public synchronized Object in  (Object key, long timeout) {
136             Object obj;
137             long now = System.currentTimeMillis();
138             long end = now + timeout;
139             while ((obj = inp (key)) == null && 
140                     ((now = System.currentTimeMillis()) < end))
141             {
142                 try {
143                     this.wait (end - now);
144                 } catch (InterruptedException e) { }
145             }
146             return obj;
147         }

187         protected static final class Data {
188             LinkedList data;
189             LinkedList stored;
190             LinkedList listeners;
191             File dir;
192     
193             protected Data (Object name) {
194                 super();
195                 data = new LinkedList ();
196                 stored = new LinkedList ();
197                 listeners = null;
198                 String n = "space" + File.separatorChar;
199                     if (name instanceof String)
200                         n += name;
201                     else
202                         n += "H" + Integer.toString(name.hashCode());
203                 dir = new File (n);
204                 dir.mkdirs ();
205                 FilenameFilter filter = new FilenameFilter() {
206                     public boolean accept(File f, String name) {
207                         return name.toUpperCase().startsWith("S");
208                     }
209                 };
210                 File file[] = dir.listFiles (filter);
211                 Arrays.sort (file);
212                 for (int i=0; file.length > i; i++) {
213                     if (cacheSize > data.size()) {
214                         Object value = readValue (file[i].getAbsolutePath ());
215                         if (value == null) {
216                             (new File (file[i].getAbsolutePath ())).delete ();
217                         } else {
218                             stored.add(file[i].getAbsolutePath ());
219                             data.add (value);
220                         }
221                     } else
222                         stored.add(file[i].getAbsolutePath ());
223                 }
224             }
225             private Object readValue (String f) {
226                 Object value = null;
227                 ObjectInputStream fin = null;
228                 try {
229                     fin = new ObjectInputStream (
230                           new BufferedInputStream (
231                           new FileInputStream (f)));
232                     value = fin.readObject ();
233                 } catch (Exception e) { 
234                     throw new SpaceError (e);
235                 } finally {
236                     if (fin != null)
237                         try {
238                             fin.close();
239                         } catch (Exception e) { 
240                             throw new SpaceError (e);
241                         }
242                 }
243                 return value;
244             }
245             
246             protected File createTempFile (String prefex, File dir) {
247                 long t = System.currentTimeMillis();
248                 File f = null;
249                 do {
250                     String fn = prefex + Long.toHexString(t);
251                     t = t + 1;
252                     f = new File (dir, fn);
253                 } while (f.exists());
254                 return f;
255             }
256             
257             protected void add (Object value) {
258                 File f = null;
259                 FileOutputStream fos = null;
260                 try {
261                     f = createTempFile ("S", dir);
262                     fos = new FileOutputStream (f);
263                     ObjectOutputStream fout = new ObjectOutputStream (
264                                               new BufferedOutputStream (fos));
265                     fout.writeObject (value);
266                     fout.flush();
267                     fos.getFD().sync();
268                 } catch (Exception e) {
269                     throw new SpaceError (e);
270                 } finally {
271                     if (fos != null)
272                         try {
273                             fos.close();
274                         } catch (Exception e) { 
275                             throw new SpaceError (e);
276                         };
277                 }
278                 stored.add(f.getAbsolutePath());
279                 /* fill cache */
280                 if (cacheSize > data.size ())
281                     if ((data.size() + 1) == stored.size ())
282                         data.add (value);
283             }
284             protected Object get (Object value) {
285                 Object obj = null;
286                 while (size() > 0) {
287                     obj = getFirst();
288                     if (obj instanceof LeasedReference)
289                         obj = ((LeasedReference)obj).get ();
290                     if (obj == null) {
291                         data.removeFirst ();
292                         File f = new File ((String) stored.removeFirst ());
293                         f.delete ();
294                         continue;
295                     }
296                     break;
297                 }
298                 return obj;
299             }
300             protected Object getFirst () {
301                 Object object = null;
302                 if (data.size() > 0) {
303                     object = data.getFirst ();
304                 } else if (stored.size() > 0) {
305                     object = readValue ((String) stored.getFirst ());
306                 }
307                 return object;
308             }
309             protected int size () {
310                 return stored.size ();
311             }
312             protected Object remove () {
313                 Object obj = null;
314                 File f = null;
315                 while (size() > 0) {
316                     obj = getFirst ();
317                     if (data.size () > 0) 
318                         data.removeFirst();
319                     f = new File ((String) stored.removeFirst ());
320                     f.delete ();
321                     if (obj instanceof LeasedReference) {
322                         LeasedReference ref = (LeasedReference) obj;
323                         obj = ref.get ();
324                         if (obj == null) {
325                             continue;
326                         }
327                         ref.discard ();
328                     }
329                     break;
330                 }
331                 return obj;
332             }
333             protected boolean isEmpty () {
334                 return stored.isEmpty () && listeners == null;
335             }
336             protected void addListener (SpaceListener l) {
337                 if (listeners == null)
338                     listeners = new LinkedList ();
339                 listeners.add (l);
340             }
341             protected void removeListener (SpaceListener l) {
342                 if (listeners != null) {
343                     listeners.remove (l);
344                     if (listeners.isEmpty ())
345                         listeners = null;
346                 }
347             }
348             protected List getListeners () {
349                 return listeners;
350             }

352         public static final LocalSpace getSpace () {

355         public static final LocalSpace getSpace (String spaceName) {
356             String key = "jpos:pSpace/"+spaceName;
357             Space sp   = TransientSpace.getSpace();
358             Object obj = sp.rdp (key);
359             if (obj == null) {
360                 synchronized (PersistentSpace.class) {
361                     obj = sp.rdp (key);
362                     if (obj == null) {
363                         obj = new PersistentSpace ();
364                         sp.out (key, obj);
365                     }
366                 }
367             } 
368             return (LocalSpace) obj;
369         }

377         public String getKeys () {
378             StringBuffer sb = new StringBuffer ();
379             Iterator iter = map.keySet().iterator ();
380             boolean first = true;
381             while (iter.hasNext()) {
382                 if (!first)
383                     sb.append (' ');
384                 else
385                     first = false;
386                 sb.append (iter.next().toString ());
387             }
388             return sb.toString ();
389         }
