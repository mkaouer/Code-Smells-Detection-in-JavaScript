228             throws IOException, SocketException
229         {
230             this.socket = socket;
231             applyTimeout();
232             setLogger(getLogger(), getOriginalRealm() + 
233                 "/" + socket.getInetAddress().getHostAddress() + ":" 
234                 + socket.getPort()
235             );
236             serverIn = new DataInputStream (
237                 new BufferedInputStream (socket.getInputStream ())
238             );
239             serverOut = new DataOutputStream(
240                 new BufferedOutputStream(socket.getOutputStream(), 2048)

242             usable = true;
243             cnt[CONNECT]++;
244             setChanged();
245             notifyObservers();
246         }

418         /**
419          * sends an ISOMsg over the TCP/IP session
420          * @param m the Message to be sent
421          * @exception IOException
422          * @exception ISOException
423          * @exception ISOFilter.VetoException;
424          */
425         public void send (ISOMsg m) 
426             throws IOException, ISOException, VetoException
427         {
428             LogEvent evt = new LogEvent (this, "send");
429             evt.addMessage (m);
430             try {
431                 if (!isConnected())
432                     throw new ISOException ("unconnected ISOChannel");
433                 m.setDirection(ISOMsg.OUTGOING);
434                 m = applyOutgoingFilters (m, evt);
435                 m.setDirection(ISOMsg.OUTGOING); // filter may have drop this info
436                 m.setPackager (getDynamicPackager(m));
437                 byte[] b = m.pack();
438                 synchronized (serverOut) {
439                     sendMessageLength(b.length + getHeaderLength());
440                     sendMessageHeader(m, b.length);
441                     sendMessage (b, 0, b.length);
442                     sendMessageTrailler(m, b);
443                     serverOut.flush ();
444                 }
445                 cnt[TX]++;
446                 setChanged();
447                 notifyObservers(m);
448             } catch (VetoException e) {
449                 evt.addMessage (e);
450                 throw e;
451             } catch (ISOException e) {
452                 evt.addMessage (e);
453                 throw e;
454             } catch (IOException e) {
455                 evt.addMessage (e);
456                 throw e;
457             } catch (Exception e) {
458                 evt.addMessage (e);
459                 throw new ISOException ("unexpected exception", e);
460             } finally {
461                 Logger.log (evt);
462             }
463         }

487         /**
488          * Waits and receive an ISOMsg over the TCP/IP session
489          * @return the Message received
490          * @exception IOException
491          * @exception ISOException
492          */
493         public ISOMsg receive() throws IOException, ISOException {
494             byte[] b=null;
495             byte[] header=null;
496             LogEvent evt = new LogEvent (this, "receive");
497             ISOMsg m = createMsg ();
498             m.setSource (this);
499             try {
500                 if (!isConnected())
501                     throw new ISOException ("unconnected ISOChannel");
502     
503                 synchronized (serverIn) {
504                     int len  = getMessageLength();
505                     int hLen = getHeaderLength();
506     
507                     if (len == -1) {
508                         if (hLen > 0) {
509                             header = readHeader(hLen);
510                         }
511                         b = streamReceive();
512                     }
513                     else if (len > 0 && len <= 10000) {
514                         if (hLen > 0) {
515                             // ignore message header (TPDU)
516                             // Note header length is not necessarily equal to hLen (see VAPChannel)
517                             header = readHeader(hLen);
518                             len -= header.length;
519                         }
520                         b = new byte[len];
521                         serverIn.readFully(b, 0, len);
522                         getMessageTrailler();
523                     }
524                     else
525                         throw new ISOException(
526                             "receive length " +len + " seems strange");
527                 }
528                 m.setPackager (getDynamicPackager(b));
529                 m.setHeader (getDynamicHeader(header));
530                 if (b.length > 0 && !shouldIgnore (header))  // Ignore NULL messages
531                     m.unpack (b);
532                 m.setDirection(ISOMsg.INCOMING);
533                 m = applyIncomingFilters (m, header, b, evt);
534                 m.setDirection(ISOMsg.INCOMING);
535                 evt.addMessage (m);
536                 cnt[RX]++;
537                 setChanged();
538                 notifyObservers(m);
539             } catch (ISOException e) {
540                 evt.addMessage (e);
541                 if (header != null) {
542                     evt.addMessage ("--- header ---");
543                     evt.addMessage (ISOUtil.hexdump (header));
544                 }
545                 if (b != null) {
546                     evt.addMessage ("--- data ---");
547                     evt.addMessage (ISOUtil.hexdump (b));
548                 }
549                 throw e;
550             } catch (EOFException e) {
551                 if (socket != null)
552                     socket.close ();
553                 evt.addMessage ("<peer-disconnect/>");
554                 throw e;
555             } catch (InterruptedIOException e) {
556                 if (socket != null)
557                     socket.close ();
558                 evt.addMessage ("<io-timeout/>");
559                 throw e;
560             } catch (IOException e) { 
561                 if (socket != null)
562                     socket.close ();
563                 if (usable) 
564                     evt.addMessage (e);
565                 throw e;
566             } catch (Exception e) { 
567                 evt.addMessage (m);
568                 evt.addMessage (e);
569                 throw new ISOException ("unexpected exception", e);
570             } finally {
571                 Logger.log (evt);
572             }
573             return m;
574         }

583         /**
584          * disconnects the TCP/IP session. The instance is ready for
585          * a reconnection. There is no need to create a new ISOChannel<br>
586          * @exception IOException
587          */
588         public void disconnect () throws IOException {
589             LogEvent evt = new LogEvent (this, "disconnect");
590             if (serverSocket != null) 
591                 evt.addMessage ("local port "+serverSocket.getLocalPort()
592                     +" remote host "+serverSocket.getInetAddress());
593             else
594                 evt.addMessage (host+":"+port);
595             try {
596                 usable = false;
597                 setChanged();
598                 notifyObservers();
599                 if (serverIn != null) {
600                     try {
601                         serverIn.close();
602                     } catch (IOException ex) { evt.addMessage (ex); }
603                     serverIn  = null;
604                 }
605                 if (serverOut != null) {
606                     try {
607                         serverOut.close();
608                     } catch (IOException ex) { evt.addMessage (ex); }
609                     serverOut = null;
610                 }
611                 if (socket != null) {
612                     try {
613                         socket.setSoLinger (true, 0);
614                     } catch (SocketException e) {
615                         // safe to ignore - can be closed already
616                     }
617                     socket.close ();
618                 }
619                 Logger.log (evt);
620             } catch (IOException e) {
621                 evt.addMessage (e);
622                 Logger.log (evt);
623                 throw e;
624             }
625             socket = null;
626         }

739         protected ISOMsg applyOutgoingFilters (ISOMsg m, LogEvent evt) 
740             throws VetoException
741         {
742             Iterator iter  = outgoingFilters.iterator();
743             while (iter.hasNext())
744                 m = ((ISOFilter) iter.next()).filter (this, m, evt);
745             return m;
746         }
747         protected ISOMsg applyIncomingFilters (ISOMsg m, LogEvent evt) 
748             throws VetoException 
749         {
750             return applyIncomingFilters (m, null, null, evt);

752         protected ISOMsg applyIncomingFilters (ISOMsg m, byte[] header, byte[] image, LogEvent evt) 
753             throws VetoException
754         {
755             Iterator iter  = incomingFilters.iterator();
756             while (iter.hasNext()) {
757                 ISOFilter f = (ISOFilter) iter.next ();
758                 if (image != null && (f instanceof RawIncomingFilter))
759                     m = ((RawIncomingFilter)f).filter (this, m, header, image, evt);
760                 else
761                     m = f.filter (this, m, evt);
762             }
763             return m;
764         }
765        /**
766         * Implements Configurable<br>
767         * Properties:<br>
768         * <ul>
769         * <li>host - destination host (if ClientChannel)
770         * <li>port - port number      (if ClientChannel)
771         * </ul>
772         * (host not present indicates a ServerChannel)
773         *
774         * @param cfg Configuration
775         * @throws ConfigurationException

777         public void setConfiguration (Configuration cfg)
778             throws ConfigurationException 
779         {
780             String h    = cfg.get    ("host");
781             int port    = cfg.getInt ("port");
782             if (h != null && h.length() > 0) {
783                 if (port == 0)
784                     throw new ConfigurationException 
785                         ("invalid port for host '"+h+"'");
786                 setHost (h, port);
787             }
788             overrideHeader = cfg.getBoolean ("override-header", false);
789             if (socketFactory != this && socketFactory instanceof Configurable)
790                 ((Configurable)socketFactory).setConfiguration (cfg);
791             try {
792                 setTimeout (cfg.getInt ("timeout"));
793             } catch (SocketException e) {
794                 throw new ConfigurationException (e);
