3     package genericpackager;
4     
5     import java.io.*;
6     import java.util.*;
7     
8     import org.jpos.iso.*;
9     import org.jpos.iso.packager.GenericPackager;
10     
11     public class Test
12     {
13      /**
14          * Test harness for GenericPackager
15          *
16          * <pre>
17          * Takes 2 arguments
18          * args[0] = xml field description file
19          * args[1] = file containing a hex dump of the message to parse
20          * </pre>
21          */
22         public static void main(String[] args) throws Exception
23         {
24             if (args.length != 2)
25             {
26                 System.out.println ("Usage: GenericPackager packager datafile");
27                 System.out.println ("       Where packager is an XML packager description");
28                 System.out.println ("       And   datafile is a hex dump of an ISOMessage");
29                 System.exit(1);
30             }
31             GenericPackager p = new GenericPackager(args[0]);
32     
33             InputStream in = new FileInputStream (args[1]);
34     
35             int nbytes = in.available();
36     
37             // Read in input file and strip any training white space
38             byte[] hexbytes = new byte[nbytes];
39             in.read (hexbytes, 0, nbytes);
40             while (Character.isWhitespace ((char)hexbytes[nbytes-1]))
41                 nbytes--;
42     
43             // Convert it to a byte array
44             byte b[] = ISOUtil.hex2byte (hexbytes,0,nbytes/2);
45     
46             // And upack it into an ISOMsg
47             ISOMsg msg = new ISOMsg();
48             msg.setPackager (p);
49             msg.unpack (b);
50     
51             msg.dump (new PrintStream(System.out), "");
52     
53             // Now Re-Pack the message
54             byte b2[] = msg.pack();
55     
56             // The byte arrays should be the same!
57             if (!Arrays.equals (b, b2))
58             {
59                 System.out.println ("Error Re-Packing Message");
60                 System.out.println (new String (ISOUtil.hexString(b)));
61                 System.out.println (new String (ISOUtil.hexString(b2)));
62             }
63             else 
64                 System.out.println ("Message re-packed ok");
65         }
66     
67     }
