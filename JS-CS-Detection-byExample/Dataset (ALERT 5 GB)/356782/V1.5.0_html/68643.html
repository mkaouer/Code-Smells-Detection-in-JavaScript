40     package org.jpos.iso;
41     
42     /**
43      * Implementations convert Strings into byte arrays and vice versa.
44      * 
45      * @author joconnor
46      * @version $Revision$ $Date$
47      */
48     public interface Interpreter
49     {
50         /**
51     	 * Converts the string data into a different interpretation. Standard
52     	 * interpretations are ASCII, EBCDIC, BCD and LITERAL.
53     	 * 
54     	 * @param data
55     	 *            The data to be interpreted.
56     	 * @return The interpreted data.
57     	 */
58         void interpret(String data, byte[] b, int offset);
59     
60         /**
61     	 * Converts the byte array into a String. This reverses the interpret
62     	 * method.
63     	 * 
64     	 * @param rawData
65     	 *            The interpreted data.
66     	 * @param offset
67     	 *            The index in rawData to start interpreting at.
68     	 * @param length
69     	 *            The number of data units to interpret.
70     	 * @return The uninterpreted data.
71     	 */
72         String uninterpret(byte[] rawData, int offset, int length);
73     
74         /**
75     	 * Returns the number of bytes required to interpret a String of length
76     	 * nDataUnits.
77     	 */
78         int getPackedLength(int nDataUnits);
79     }
