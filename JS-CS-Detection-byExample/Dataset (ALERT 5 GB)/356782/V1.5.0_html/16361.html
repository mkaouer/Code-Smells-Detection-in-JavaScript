50         public void initChannel () throws ConfigurationException {
51             Element persist = getPersist ();
52             sp = grabSpace (persist.getChild ("space")); 
53             Element e = persist.getChild ("channel");
54             if (e == null)
55                 throw new ConfigurationException ("channel element missing");
56     
57             in      = persist.getChildTextTrim ("in");
58             out     = persist.getChildTextTrim ("out");
59     
60             String s = persist.getChildTextTrim ("reconnect-delay");
61             delay    = s != null ? Long.parseLong (s) : 10000; // reasonable default
62     
63             channel = newChannel (e, getFactory());
64             
65             String socketFactoryString = getSocketFactory();
66             if (socketFactoryString != null && channel instanceof FactoryChannel) {
67                 ISOClientSocketFactory sFac = (ISOClientSocketFactory) getFactory().newInstance(socketFactoryString);
68                 if (sFac != null && sFac instanceof LogSource) {
69                     ((LogSource) sFac).setLogger(log.getLogger(),getName() + ".socket-factory");
70                 }
71                 getFactory().setConfiguration (sFac, e);
72                 ((FactoryChannel)channel).setSocketFactory(sFac);
73             }
74             ready   = getName() + ".ready";
75     
76             reconnect = getName() + ".reconnect";
77         }
78         public void startService () {
79             try {
80                 initChannel ();
81                 new Thread (new Sender ()).start ();
82                 new Thread (new Receiver ()).start ();
83             } catch (Exception e) {
84                 getLog().warn ("error starting service", e);
85             }
86         }
87         public void stopService () {
88             try {
89                 sp.out (in, new Object());
90                 if (channel.isConnected())
91                     disconnect();
92             } catch (Exception e) {
93                 getLog().warn ("error disconnecting from remote host", e);
94             }
95         }
96         public void destroyService () {
97             NameRegistrar.unregister ("channel." + getName ());
98         }

187         public ISOChannel newChannel (Element e, QFactory f) 
188             throws ConfigurationException
189         {
190             String channelName  = e.getAttributeValue ("class");
191             String packagerName = e.getAttributeValue ("packager");
192     
193             ISOChannel channel   = (ISOChannel) f.newInstance (channelName);
194             ISOPackager packager = null;
195             if (packagerName != null) {
196                 packager = (ISOPackager) f.newInstance (packagerName);
197                 channel.setPackager (packager);
198                 f.setConfiguration (packager, e);
199             }
200             QFactory.invoke (channel, "setHeader", e.getAttributeValue ("header"));
201             f.setLogger        (channel, e);
202             f.setConfiguration (channel, e);
203     
204             if (channel instanceof FilteredChannel) {
205                 addFilters ((FilteredChannel) channel, e, f);
206             }
207             if (getName () != null)
208                 channel.setName (getName ());
209             return channel;
210         }
211     
212         private void addFilters (FilteredChannel channel, Element e, QFactory fact) 
213             throws ConfigurationException
214         {
215             Iterator iter = e.getChildren ("filter").iterator();
216             while (iter.hasNext()) {
217                 Element f = (Element) iter.next();
218                 String clazz = f.getAttributeValue ("class");
219                 ISOFilter filter = (ISOFilter) fact.newInstance (clazz);
220                 fact.setLogger        (filter, f);
221                 fact.setConfiguration (filter, f);
222                 String direction = f.getAttributeValue ("direction");
223                 if (direction == null)
224                     channel.addFilter (filter);
225                 else if ("incoming".equalsIgnoreCase (direction))
226                     channel.addIncomingFilter (filter);
227                 else if ("outgoing".equalsIgnoreCase (direction))
228                     channel.addOutgoingFilter (filter);
229                 else if ("both".equalsIgnoreCase (direction)) {
230                     channel.addIncomingFilter (filter);
231                     channel.addOutgoingFilter (filter);
232                 }
233             }
234         }

236         public class Sender implements Runnable {
237             public Sender () {
238                 super ();
239             }
240             public void run () {
241                 Thread.currentThread().setName ("channel-sender-" + in);
242                 while (running ()){
243                     try {
244                         checkConnection ();
245                         Object o = sp.in (in, delay);
246                         if (o instanceof ISOMsg)
247                             channel.send ((ISOMsg) o);
248                     } catch (Exception e) { 
249                         getLog().warn ("channel-sender-"+in, e.getMessage ());
250                         disconnect ();
251                         ISOUtil.sleep (1000);
252                     }
253                 }
254             }
255         }
256         public class Receiver implements Runnable {
257             public Receiver () {
258                 super ();
259             }
260             public void run () {
261                 Thread.currentThread().setName ("channel-receiver-"+out);
262                 while (running()) {
263                     try {
264                         sp.rd (ready);
265                         ISOMsg m = channel.receive ();
266                         sp.out (out, m);
267                     } catch (Exception e) { 
268                         if (running()) {
269                             getLog().warn ("channel-receiver-"+out, e);
270                             disconnect ();
271                             sp.out (in, new Object()); // wake-up Sender
272                             ISOUtil.sleep(1000);
273                             sp.out (reconnect, new Object(), delay);
274                         }
275                     }
276                 }
277             }
278         }
279         protected void checkConnection () {
280             while (running() && 
281                     sp.rdp (reconnect) != null)
282             {
283                 ISOUtil.sleep(1000);
284             }
285             while (running() && !channel.isConnected ()) {
286                 while (sp.inp (ready) != null)
287                     ;
288                 try {
289                     channel.connect ();
290                 } catch (IOException e) {
291                     getLog().warn ("check-connection", e.getMessage ());
292                 }
293                 if (!channel.isConnected ())
294                     ISOUtil.sleep (delay);
295             }
296             if (running() && (sp.rdp (ready) == null))
297                 sp.out (ready, new Date());
298         }
299         protected synchronized void disconnect () {
300             try {
301                 while (sp.inp (ready) != null)
302                     ;
303                 channel.disconnect ();
304             } catch (IOException e) {
305                 getLog().warn ("disconnect", e);
306             }
307         }
