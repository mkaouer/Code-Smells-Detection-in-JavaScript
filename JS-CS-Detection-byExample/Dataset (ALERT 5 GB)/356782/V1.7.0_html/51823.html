19     package org.jpos.space;
20     
21     import org.jpos.core.Configurable;
22     import org.jpos.core.Configuration;
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.iso.ISOChannel;
25     import org.jpos.iso.ISOMsg;
26     import org.jpos.iso.ISORequestListener;
27     import org.jpos.iso.ISOSource;
28     import org.jpos.util.*;
29     import org.jpos.util.NameRegistrar.NotFoundException;
30     
31     
32     /**
33      * Integrates legacy (pre 2.0) ISORequestListeners to the new Space 
34      * based architecture
35      * @author Alejandro Revilla
36      * @version $Revision$ $Date$
37      */
38     public class ISORequestListenerAdaptor extends SimpleLogSource 
39             implements Configurable, ISORequestListener
40     {
41         Space sp;
42         Configuration cfg;
43         ISOChannel channel;
44         SpaceMUX mux;
45         ThreadPool pool;
46         long timeout;
47         public ISORequestListenerAdaptor () {
48             super ();
49             sp = TransientSpace.getSpace ();
50         }
51         public void setConfiguration (Configuration cfg) 
52             throws ConfigurationException 
53         {
54             this.cfg = cfg;
55             try {
56                 mux = (SpaceMUX) NameRegistrar.get (cfg.get ("mux"));
57             } catch (NotFoundException e) {
58                 throw new ConfigurationException (e);
59             }
60             timeout = cfg.getLong ("timeout", 120000);
61             pool    = new ThreadPool (1, cfg.getInt ("pool", 100));
62         }
63     
64         public class Processor implements Runnable {
65             ISOSource source;
66             ISOMsg m;
67             public Processor (ISOSource source, ISOMsg m) {
68                 super ();
69                 this.source = source;
70                 this.m      = m;
71             }
72             public void run () {
73                 try {
74                     ISOMsg resp = mux.request (m, timeout);
75                     if (resp != null)
76                         source.send (resp);
77                 } catch (Exception e) {
78                     Logger.log (
79                         new LogEvent (
80                             ISORequestListenerAdaptor.this, 
81                             "iso-request-listener-adaptor", e
82                         )
83                     );
84                 }
85             }
86         }
87     
88         public boolean process (ISOSource source, ISOMsg m) {
89             pool.execute (new Processor (source, m));
90             return false;   // let other listeners do their work too
91         }
92     }
