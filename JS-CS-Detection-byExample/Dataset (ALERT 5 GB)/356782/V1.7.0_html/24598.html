19     package org.jpos.iso;
20     
21     /**
22      * Implements Hex Interpreter. The Hex digits are stored in ASCII.
23      * 
24      * @author joconnor
25      * @version $Revision$ $Date$
26      */
27     public class AsciiHexInterpreter implements BinaryInterpreter
28     {
29         /** An instance of this Interpreter. Only one needed for the whole system */
30         public static final AsciiHexInterpreter INSTANCE = new AsciiHexInterpreter();
31     
32         /** 0-15 to ASCII hex digit lookup table. */
33         private static final byte[] HEX_ASCII = new byte[] {
34                   0x30, 0x31, 0x32, 0x33, 0x34, 0x35, 0x36, 0x37,
35                   0x38, 0x39, 0x41, 0x42, 0x43, 0x44, 0x45, 0x46
36         };
37     
38         /**
39          * Converts the binary data into ASCII hex digits.
40          * @see org.jpos.iso.BinaryInterpreter#interpret(byte[], byte[], int)
41          */
42         public void interpret(byte[] data, byte[] b, int offset)
43         {
44             for (int i = 0; i < data.length; i++) {
45                 b[offset + i * 2] = HEX_ASCII[(data[i] & 0xF0) >> 4];
46                 b[offset + i * 2 + 1] = HEX_ASCII[data[i] & 0x0F];
47             }
48         }
49     
50         /**
51          * Converts the ASCII hex digits into binary data.
52          * @see org.jpos.iso.BinaryInterpreter#uninterpret(byte[], int, int)
53          */
54         public byte[] uninterpret(byte[] rawData, int offset, int length)
55         {
56             byte[] ret = new byte[length];
57             for (int i = 0; i < length; i++)
58             {
59                 byte hi = rawData[offset + i * 2];
60                 byte lo = rawData[offset + i * 2 + 1];
61                 int h = hi > 0x40 ? 10 + hi - 0x41 : hi - 0x30;
62                 int l = lo > 0x40 ? 10 + lo - 0x41 : lo - 0x30;
63                 ret[i] = (byte)(h << 4 | l);
64             }
65             return ret;
66         }
67     
68         /**
69          * Returns double nBytes because the hex representation of 1 byte needs 2 hex digits.
70          * 
71          * @see org.jpos.iso.BinaryInterpreter#getPackedLength(int)
72          */
73         public int getPackedLength(int nBytes)
74         {
75             return nBytes * 2;
76         }
77     }
