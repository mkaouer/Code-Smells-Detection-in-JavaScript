19     package org.jpos.iso;
20     
21     /**
22      * Implements the Padder interface for padding strings and byte arrays on the
23      * Right. The difference between this and RightPadder is that this truncates the data
24      * during packing, instead of throwing an exception.
25      * 
26      * @author jonathan.oconnor@xcom.de
27      * @version $Revision$ $Date$
28      */
29     public class RightTPadder extends RightPadder
30     {
31         /**
32     	 * A padder for padding spaces on the right. This is very common in
33     	 * alphabetic fields.
34     	 */
35         public static final RightTPadder SPACE_PADDER = new RightTPadder(' ');
36     
37         /**
38     	 * Creates a Right Truncating Padder with a specific pad character.
39     	 * 
40     	 * @param pad
41     	 *            The padding character. For binary padders, the pad character
42     	 *            is truncated to lower order byte.
43     	 */
44         public RightTPadder(char pad)
45         {
46             super(pad);
47         }
48     
49         /**
50     	 * @see org.jpos.iso.Padder#pad(java.lang.String, int,
51     	 *      char)
52     	 */
53         public String pad(String data, int maxLength) throws ISOException
54         {
55             if (data.length() > maxLength)
56             {
57                 return super.pad(data.substring(0,maxLength), maxLength);
58             } else
59             {
60                 return super.pad(data, maxLength);
61             }
62         }
63     }
