19     package org.jpos.space;
20     
21     import java.util.Set;
22     
23     /**
24      * @author Kris, Bharavi, Alejandro
25      * @version $Revision$ $Date$
26      */
27     public interface LocalSpace<K,V> extends Space<K,V> {
28         /**
29          * add a SpaceListener associated with a given key
30          * @param key Entry's key
31          * @param listener a SpaceListener
32          */
33         public void addListener    (Object key, SpaceListener listener);
34     
35         /**
36          * add a SpaceListener associated with a given key for a given 
37          * period of time.
38          * <b>Warning: not supported by all space implementations.</b>
39          * @param key Entry's key
40          * @param listener a SpaceListener
41          * @param timeout 
42          */
43         public void addListener (Object key, SpaceListener listener, long timeout);
44     
45         /**
46          * removes a SpaceListener associated with a given key
47          * @param key Entry's key
48          * @param listener the SpaceListener
49          */
50         public void removeListener (Object key, SpaceListener listener);
51     
52         /**
53          * @return Set containing all keys in Space
54          */
55         public Set getKeySet ();
56     
57         /**
58          * @return number of entries in a given key
59          */
60         public int size (Object key);
61     }
