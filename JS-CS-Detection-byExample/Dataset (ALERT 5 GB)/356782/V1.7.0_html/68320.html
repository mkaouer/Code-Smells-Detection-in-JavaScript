19     package org.jpos.q2.cli;
20     
21     import org.jpos.iso.ISOUtil;
22     import org.jpos.q2.CLI;
23     import org.jpos.util.Loggeable;
24     import org.jpos.util.NameRegistrar;
25     
26     import java.util.Iterator;
27     import java.util.Map;
28     
29     public class SHOWNR implements CLI.Command {
30         public void exec (CLI cli, String[] args) throws Exception {
31             boolean all = args.length > 1 && "-a".equals (args[1]);
32             int i = 1;
33             if (all)
34                 i++;
35             if (args.length > i)
36                 showOne (cli, args[i], all);
37             else
38                 showAll (cli, all);
39         }
40         private void showOne (CLI cli, String name, boolean detail) {
41             try {
42                 Object obj = NameRegistrar.get (name);
43                 cli.println (name + " : " + obj.toString());
44                 if (detail && obj instanceof Loggeable) {
45                     ((Loggeable)obj).dump (cli.getOutputStream(), "   ");
46                     cli.getOutputStream().flush();
47                 }
48             } catch (NameRegistrar.NotFoundException e) {
49                 cli.println ("Object not found in NameRegistrar");
50             }
51         }
52         private void showAll (CLI cli, boolean detail) {
53             NameRegistrar nr = NameRegistrar.getInstance();
54             int maxw = 0;
55             Iterator iter = nr.getMap().entrySet().iterator();
56             while (iter.hasNext()) {
57                 Map.Entry entry = (Map.Entry) iter.next ();
58                 maxw = Math.max (maxw, entry.getKey().toString().length());
59             }
60             iter = nr.getMap().entrySet().iterator();
61             maxw++;
62             while (iter.hasNext()) {
63                 Map.Entry entry = (Map.Entry) iter.next ();
64                 cli.println (
65                     ISOUtil.strpad (entry.getKey().toString(), maxw) + 
66                     entry.getValue().toString()
67                 );
68                 if (detail && entry.getValue() instanceof Loggeable) {
69                     ((Loggeable)entry.getValue()).dump (cli.getOutputStream(), "   ");
70                     cli.getOutputStream().flush();
71                 }
72             }
73         }
74     }
