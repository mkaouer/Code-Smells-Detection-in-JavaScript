43         public void out (Object key, Object value) {
44             List listeners;
45             synchronized (this) {
46                 Data data = (Data) map.get (key);
47                 if (data == null) 
48                     map.put (key, (data = new Data (key)));
49                 data.add (value);
50                 listeners = data.getListeners();
51                 this.notifyAll ();
52             }
53             if (listeners != null) {
54                 Iterator iter = listeners.iterator();
55                 while (iter.hasNext()) {
56                     ((SpaceListener)iter.next()).notify (key, value);
57                 }
58             }
59         }

91         public synchronized Object in  (Object key, long timeout) {
92             Object obj;
93             long now = System.currentTimeMillis();
94             long end = now + timeout;
95             while ((obj = inp (key)) == null && 
96                     ((now = System.currentTimeMillis()) < end))
97             {
98                 try {
99                     this.wait (end - now);

150         protected static final class Data {
151             LinkedList data;
152             LinkedList stored;
153             LinkedList listeners;
154             File dir;
155     
156             protected Data (Object name) {
157                 super();
158                 data = new LinkedList ();
159                 stored = new LinkedList ();
160                 listeners = null;
161                 String n = "space" + File.separatorChar;
162                     if (name instanceof String)
163                         n += name;
164                     else
165                         n += "H" + Integer.toString(name.hashCode());
166                 dir = new File (n);
167                 dir.mkdirs ();
168                 FilenameFilter filter = new FilenameFilter() {
169                     public boolean accept(File f, String name) {
170                         return name.toUpperCase().startsWith("S");
171                     }
172                 };
173                 File file[] = dir.listFiles (filter);
174                 Arrays.sort (file);
175                 for (int i=0; file.length > i; i++) {
176                     if (cacheSize > data.size()) {
177                         Object value = readValue (file[i].getAbsolutePath ());
178                         if (value == null) {
179                             (new File (file[i].getAbsolutePath ())).delete ();
180                         } else {
181                             stored.add(file[i].getAbsolutePath ());
182                             data.add (value);
183                         }
184                     } else
185                         stored.add(file[i].getAbsolutePath ());
186                 }
187             }
188             private Object readValue (String f) {
189                 Object value = null;
190                 ObjectInputStream fin = null;
191                 try {
192                     fin = new ObjectInputStream (
193                           new BufferedInputStream (
194                           new FileInputStream (f)));
195                     value = fin.readObject ();
196                 } catch (Exception e) { 
197                     throw new SpaceError (e);
198                 } finally {
199                     if (fin != null)
200                         try {
201                             fin.close();
202                         } catch (Exception e) { 
203                             throw new SpaceError (e);
204                         }
205                 }
206                 return value;
207             }
208             
209             protected File createTempFile (String prefex, File dir) {
210                 long t = System.currentTimeMillis();
211                 File f = null;
212                 do {
213                     String fn = prefex + Long.toHexString(t);
214                     t = t + 1;
215                     f = new File (dir, fn);
216                 } while (f.exists());
217                 return f;
218             }
219             
220             protected void add (Object value) {
221                 File f = null;
222                 FileOutputStream fos = null;
223                 try {
224                     f = createTempFile ("S", dir);
225                     fos = new FileOutputStream (f);
226                     ObjectOutputStream fout = new ObjectOutputStream (
227                                               new BufferedOutputStream (fos));
228                     fout.writeObject (value);
229                     fout.flush();
230                     fos.getFD().sync();
231                 } catch (Exception e) {
232                     throw new SpaceError (e);
233                 } finally {
234                     if (fos != null) {
235                         try {
236                             fos.close();
237                         } catch (Exception e) { 
238                             throw new SpaceError (e);
239                         }
240                     }
241                 }
242                 stored.add(f.getAbsolutePath());
243                 /* fill cache */
244                 if (cacheSize > data.size ())
245                     if ((data.size() + 1) == stored.size ())
246                         data.add (value);
247             }
248             protected Object get (Object value) {
249                 Object obj = null;
250                 while (size() > 0) {
251                     obj = getFirst();
252                     if (obj instanceof LeasedReference)
253                         obj = ((LeasedReference)obj).get ();
254                     if (obj == null) {
255                         data.removeFirst ();
256                         File f = new File ((String) stored.removeFirst ());
257                         f.delete ();
258                         continue;
259                     }
260                     break;
261                 }
262                 return obj;
263             }
264             protected Object getFirst () {
265                 Object object = null;
266                 if (data.size() > 0) {
267                     object = data.getFirst ();
268                 } else if (stored.size() > 0) {
269                     object = readValue ((String) stored.getFirst ());
270                 }
271                 return object;
272             }
273             protected int size () {
274                 return stored.size ();
275             }
276             protected Object remove () {
277                 Object obj = null;
278                 File f = null;
279                 while (size() > 0) {
280                     obj = getFirst ();
281                     if (data.size () > 0) 
282                         data.removeFirst();
283                     f = new File ((String) stored.removeFirst ());
284                     f.delete ();
285                     if (obj instanceof LeasedReference) {
286                         LeasedReference ref = (LeasedReference) obj;
287                         obj = ref.get ();
288                         if (obj == null) {
289                             continue;
290                         }
291                         ref.discard ();
292                     }
293                     break;
294                 }
295                 return obj;
296             }
297             protected boolean isEmpty () {
298                 return stored.isEmpty () && listeners == null;
299             }
300             protected void addListener (SpaceListener l) {
301                 if (listeners == null)
302                     listeners = new LinkedList ();
303                 listeners.add (l);
304             }
305             protected void removeListener (SpaceListener l) {
306                 if (listeners != null) {
307                     listeners.remove (l);
308                     if (listeners.isEmpty ())
309                         listeners = null;
310                 }
311             }
312             protected List getListeners () {
313                 return listeners;
314             }

319         public static final LocalSpace getSpace (String spaceName) {
320             String key = "jpos:pSpace/"+spaceName;
321             Space sp   = TransientSpace.getSpace();
322             Object obj = sp.rdp (key);
323             if (obj == null) {
324                 synchronized (PersistentSpace.class) {
325                     obj = sp.rdp (key);
326                     if (obj == null) {
327                         obj = new PersistentSpace ();
328                         sp.out (key, obj);
329                     }
330                 }
331             } 
332             return (LocalSpace) obj;
333         }
