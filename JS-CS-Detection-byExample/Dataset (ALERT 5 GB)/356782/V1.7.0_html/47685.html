23     /**
24      * Validator for no zero-filled fields.
25      * <p>Title: jPOS</p>
26      * <p>Description: Java Framework for Financial Systems</p>
27      * <p>Copyright: Copyright (c) 2000 jPOS.org.  All rights reserved.</p>
28      * <p>Company: www.jPOS.org</p>
29      * @author Jose Eduardo Leon
30      * @version 1.0
31      */
32     public class IVA_ALPHANUMNOZERO extends IVA_ALPHANUM {
33     
34         public IVA_ALPHANUMNOZERO() {
35             super();
36         }
37     
38         public IVA_ALPHANUMNOZERO( String Description ) {
39             super( Description );
40         }
41     
42         public IVA_ALPHANUMNOZERO( int minLen, int maxLen, String Description ) {
43             super( minLen, maxLen, Description );
44         }
45     
46         public IVA_ALPHANUMNOZERO( int maxLen, String Description ) {
47             super( maxLen, Description );
48         }
49     
50         public IVA_ALPHANUMNOZERO( boolean breakOnError, String Description ) {
51             this( Description );
52             this.breakOnError = breakOnError;
53         }
54     
55         public IVA_ALPHANUMNOZERO( boolean breakOnError, int maxLen, String Description ) {
56             this( maxLen, Description );
57             this.breakOnError = breakOnError;
58         }
59     
60         public IVA_ALPHANUMNOZERO( boolean breakOnError, int minLen, int maxLen, String Description ) {
61             this( minLen, maxLen, Description );
62             this.breakOnError = breakOnError;
63         }
64     
65         /**
66          * Validate that the component is not zero-filled.
67          */
68         public ISOComponent validate ( ISOComponent f ) throws ISOException {
69             ISOField c = (ISOField)f;
70             try {
71                 /** alphanum validations **/
72                 c = (ISOField)super.validate( c );
73                 /** no zero... **/
74                 if ( ISOUtil.isZero( (String)c.getValue() ) ){
75                     ISOVError e = new ISOVError( "Invalid Value Error. It can not be zero-filled. (Current value: "+
76                             (String)c.getValue()+") ", getRejCode( ISOVError.ERR_INVALID_VALUE ) );
77                     if ( c instanceof ISOVField )
78                         ((ISOVField)c).addISOVError( e );
79                     else
80                         c = new ISOVField( c, e );
81                     if ( breakOnError )
82                         throw new ISOVException ( "Error on field " + ((Integer)c.getKey()).intValue(), c );
83                 }
84                 return c;
85             }
86             catch (Exception ex) {
87                 if ( ex instanceof ISOVException ) throw (ISOVException) ex;
88                 return c;
89             }
90         }
91     }
