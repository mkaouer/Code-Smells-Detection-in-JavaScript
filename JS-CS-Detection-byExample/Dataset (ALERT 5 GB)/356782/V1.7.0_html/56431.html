19     package org.jpos.q2.iso;
20     
21     import org.jpos.core.Configurable;
22     import org.jpos.core.Configuration;
23     import org.jpos.q2.QBeanSupport;
24     import org.jpos.q2.QFactory;
25     import org.jpos.util.DirPoll;
26     import org.jpos.util.LogSource;
27     import org.jpos.util.ThreadPool;
28     
29     /**
30      * DirPoll Adaptor
31      *
32      * @author Alejandro Revilla
33      * @version $Revision$ $Date$
34      * @jmx:mbean description="DirPoll adaptor QBean"
35      *                  extends="org.jpos.q2.QBeanSupportMBean"
36      */
37     public class DirPollAdaptor 
38         extends QBeanSupport
39         implements DirPollAdaptorMBean
40     {
41         String path, priorities, processorClass;
42         int poolSize;
43         long pollInterval;
44         protected DirPoll dirPoll;
45         public DirPollAdaptor () {
46             super ();
47             poolSize = 1;
48             pollInterval = 1000;
49         }
50     
51         protected void initService () throws Exception {
52             QFactory factory = getServer().getFactory();
53             dirPoll  = new DirPoll ();
54             dirPoll.setPath (getPath ());
55             dirPoll.setThreadPool (new ThreadPool (1, poolSize));
56             dirPoll.setPollInterval (pollInterval);
57             if (priorities != null) 
58                 dirPoll.setPriorities (priorities);
59             dirPoll.setLogger (getLog().getLogger(), getLog().getRealm ());
60             Configuration cfg = factory.getConfiguration (getPersist());
61             dirPoll.setConfiguration (cfg);
62             dirPoll.createDirs ();
63             Object dpp = factory.newInstance (getProcessor());
64             if (dpp instanceof LogSource) {
65                 ((LogSource) dpp).setLogger (
66                     getLog().getLogger(), getLog().getRealm ()
67                 );
68                 dirPoll.setProcessor (dpp);
69             }
70             if (dpp instanceof Configurable) {
71                 ((Configurable) dpp).setConfiguration (cfg);
72             }
73         }
74         protected void startService () throws Exception {
75             new Thread (dirPoll).start ();
76         }
77     
78         protected void stopService () throws Exception {
79             dirPoll.destroy ();
80         }
81     
82     
83         /**
84          * @jmx:managed-attribute description="Base path"
85          */
86         public synchronized void setPath (String path) {
87             this.path = path;
88             setModified (true);
89         }
90     
91         /**
92          * @jmx:managed-attribute description="thread pool size"
93          */
94         public synchronized void setPoolSize (int size) {
95             this.poolSize = size;
96             setModified (true);
97         }
98         /**
99          * @jmx:managed-attribute description="thread pool size"
100          */
101         public int getPoolSize () {
102             return poolSize;
103         }
104     
105         /**
106          * @jmx:managed-attribute description="Base path"
107          */
108         public String getPath () {
109             return path == null ? "." : path;
110         }
111     
112         /**
113          * @jmx:managed-attribute description="poll time in millis"
114          */
115         public synchronized void setPollInterval (long pollInterval) {
116             this.pollInterval = pollInterval;
117             setModified (true);
118         }
119         /**
120          * @jmx:managed-attribute description="poll time in millis"
121          */
122         public long getPollInterval () {
123             return pollInterval;
124         }
125         /**
126          * @jmx:managed-attribute description="priorities"
127          */
128         public synchronized void setPriorities (String priorities) {
129             this.priorities = priorities;
130             setModified (true);
131         }
132         /**
133          * @jmx:managed-attribute description="priorities"
134          */
135         public String getPriorities () {
136             return priorities;
137         }
138     
139         /**
140          * @jmx:managed-attribute description="processor class"
141          */
142         public synchronized void setProcessor (String processor) {
143             this.processorClass = processor;
144             setModified (true);
145         }
146         /**
147          * @jmx:managed-attribute description="processor class"
148          */
149         public String getProcessor() {
150             return processorClass;
151         }
152     }
