19     package org.jpos.ui.factory;
20     
21     import org.jdom.Element;
22     import org.jpos.ui.UI;
23     import org.jpos.ui.UIFactory;
24     
25     import javax.swing.*;
26     import java.awt.*;
27     
28     /**
29      * @author Alejandro Revilla
30      *
31      * creates a panel with a Border layout
32      * i.e:
33      * <pre>
34      *  &lt;border-layout&gt;
35      *   &lt;north&gt;...&lt;/north&gt;
36      *   &lt;south&gt;...&lt;/south&gt;
37      *   &lt;east&gt;...&lt;/east&gt;
38      *   &lt;west&gt;...&lt;/west&gt;
39      *   &lt;center&gt;...&lt;/center&gt;
40      *  &lt;/border-layout&gt;
41      * </pre>
42      * @see org.jpos.ui.UIFactory
43      */
44     public class BorderLayoutFactory implements UIFactory {
45         UI ui;
46     
47         public JComponent create (UI ui, Element e) {
48             this.ui    = ui;
49             JPanel p = new JPanel (new BorderLayout ());
50     
51             add (p, e.getChild ("north"),  BorderLayout.NORTH);
52             add (p, e.getChild ("south"),  BorderLayout.SOUTH);
53             add (p, e.getChild ("east"),   BorderLayout.EAST);
54             add (p, e.getChild ("west"),   BorderLayout.WEST);
55             add (p, e.getChild ("center"), BorderLayout.CENTER);
56     
57             return p;
58         }
59     
60         private void add (JPanel p, Element e, String location) {
61             if (e != null)
62                 p.add (ui.create (e), location);
63         }
64     }
