19     package org.jpos.iso;
20     
21     /**
22      * ISOFieldPackager Binary Amount
23      *
24      * @author apr@cs.com.uy
25      * @version $Id$
26      * @see ISOComponent
27      */
28     public class IFB_AMOUNT extends ISOFieldPackager {
29         private BCDInterpreter interpreter;
30         
31         public IFB_AMOUNT() {
32             super();
33             interpreter = BCDInterpreter.LEFT_PADDED;
34         }
35         /**
36          * @param len - field len
37          * @param description symbolic descrption
38          */
39         public IFB_AMOUNT(int len, String description, boolean pad) {
40             super(len, description);
41             this.pad = pad;
42             interpreter = pad ? BCDInterpreter.LEFT_PADDED : BCDInterpreter.RIGHT_PADDED;
43         }
44         
45         public void setPad(boolean pad)
46         {
47             this.pad = pad;
48             interpreter = pad ? BCDInterpreter.LEFT_PADDED : BCDInterpreter.RIGHT_PADDED;
49         }
50     
51         /**
52          * @param c - a component
53          * @return packed component
54          * @exception ISOException
55          */
56         public byte[] pack (ISOComponent c) throws ISOException {
57             String s = (String) c.getValue();
58             String amount = ISOUtil.zeropad(s.substring(1), getLength()-1);
59             byte[] b   = new byte[1 + (getLength() >> 1)];
60             b[0] = (byte) s.charAt(0);
61             interpreter.interpret(amount, b, 1);
62             return b;
63         }
64         /**
65          * @param c - the Component to unpack
66          * @param b - binary image
67          * @param offset - starting offset within the binary image
68          * @return consumed bytes
69          * @exception ISOException
70          */
71         public int unpack (ISOComponent c, byte[] b, int offset)
72             throws ISOException
73         {
74             String d = (new String(b, offset, 1)) 
75                         + interpreter.uninterpret(b, offset + 1, getLength() - 1);
76             c.setValue(d);
77             return 1 + ((getLength()) >> 1);
78         }
79         public int getMaxPackedLength() {
80             return 1 + ((getLength()) >> 1);
81         }
82     }
