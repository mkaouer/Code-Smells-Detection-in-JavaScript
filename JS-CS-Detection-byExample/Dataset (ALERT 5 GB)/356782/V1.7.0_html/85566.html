19     package org.jpos.util;
20     
21     import java.io.PrintStream;
22     
23     /**
24      * Periodically dumps Thread and memory usage
25      * @author apr@cs.com.uy
26      * @version $Id$
27      * @see Logger
28      */
29     public class SystemMonitor implements Runnable, LogSource, Loggeable
30     {
31         private Logger logger = null;
32         private String realm  = null;
33         private int sleepTime = 0;
34         private int delay     = 0;
35         private Thread thread = null;
36     
37         /**
38          * noargs constructor
39          */
40         public SystemMonitor () {
41             super();
42         }
43         /**
44          * @param sleepTime sleep
45          * @param logger current logger
46          * @param realm  instance realm
47          */
48         public SystemMonitor (int sleepTime, Logger logger, String realm) {
49             setLogger (logger, realm);
50             this.sleepTime = sleepTime;
51             startThread();
52         }
53         
54         private void startThread() {
55             if (thread != null)
56                 thread.interrupt();
57             else if (sleepTime > 0) {
58                 thread = new Thread(this,"SystemMonitor");
59                 thread.setPriority (Thread.MIN_PRIORITY);
60                 thread.start();
61             }
62         }
63     
64         /**
65          * @param sleepTime new sleepTime;
66          */
67         public void setSleepTime (int sleepTime) {
68             this.sleepTime = sleepTime;
69             startThread();
70         }
71     
72         void dumpThreads (ThreadGroup g, PrintStream p, String indent) {
73             Thread[] list = new Thread[g.activeCount()+5];
74             int nthreads = g.enumerate(list);
75             for (int i=0; i<nthreads; i++) 
76                 p.println (indent + list[i]);
77         }
78     
79         public void showThreadGroup (ThreadGroup g, PrintStream p, String indent) {
80             if (g.getParent() != null)
81                 showThreadGroup (g.getParent(), p, indent + "  ");
82             else
83                 dumpThreads (g, p, indent + "    ");
84         }
85     
86         public void run() {
87             for (;;) {
88                 Logger.log (new LogEvent (this, "SystemMonitor", this));
89                 try {
90                     long expected = System.currentTimeMillis() + sleepTime;
91                     Thread.sleep (sleepTime);
92                     delay = (int) (System.currentTimeMillis() - expected);
93                 } catch (InterruptedException e) { }
94             }
95         }
96         public void dump (PrintStream p, String indent) {
97             String newIndent = indent + "  ";
98             Runtime r = Runtime.getRuntime();
99             p.println (indent + "--- memory ---");
100             p.println (newIndent+" freeMemory="+r.freeMemory());
101             p.println (newIndent+"totalMemory="+r.totalMemory());
102             p.println (newIndent+"inUseMemory="+(r.totalMemory()-r.freeMemory()));
103             p.println ("");
104             p.println (indent + "--- threads ---");
105             p.println (newIndent+"      delay="+delay+" ms");
106             p.println (newIndent+"    threads="+Thread.activeCount());
107             showThreadGroup (Thread.currentThread().getThreadGroup(), p, newIndent);
108             p.println ("");
109             NameRegistrar.getInstance().dump (p, indent);
110         }
111         public void setLogger (Logger logger, String realm) {
112             this.logger = logger;
113             this.realm  = realm;
114         }
115         public String getRealm () {
116             return realm;
117         }
118         public Logger getLogger() {
119             return logger;
120         }
121     }
