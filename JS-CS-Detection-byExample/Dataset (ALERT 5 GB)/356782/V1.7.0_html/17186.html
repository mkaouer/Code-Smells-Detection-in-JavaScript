19     package org.jpos.q2.cli;
20     
21     import org.jpos.q2.CLI;
22     
23     import java.io.IOException;
24     
25     public class SHUTDOWN implements CLI.Command {
26         public void exec (CLI cli, String[] args) throws IOException {
27             boolean shutdown = false;
28             if (args.length == 2 && "--force".equals (args[1]))
29                 shutdown = true;
30             else 
31                 shutdown = cli.confirm ("Confirm shutdown (Yes/No) ? ");
32     
33             if (shutdown) {
34                 cli.println ("Shutting down.");
35                 cli.getQ2().shutdown();
36             }
37             else 
38                 cli.println ("Q2 will continue running.");
39         }
40     }
