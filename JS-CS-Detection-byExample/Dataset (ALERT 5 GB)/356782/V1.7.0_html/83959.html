19     package  org.jpos.security.jceadapter;
20     
21     import org.jpos.core.ConfigurationException;
22     import org.jpos.core.SimpleConfiguration;
23     import org.jpos.iso.ISOUtil;
24     import org.jpos.security.SMException;
25     import org.jpos.security.SecureDESKey;
26     import org.jpos.util.Logger;
27     import org.jpos.util.SimpleLogListener;
28     
29     import java.util.Properties;
30     
31     
32     /**
33      * A simple application for sending critical commands to the JCE Security Module.
34      * The functionalities available from this console, are not available programmatically (via API's),
35      * for security reasons, because most of them involve clear (non encrypted) keys.
36      * Those commands are package protected in the JCE Security Module.
37      * @author Hani Samuel Kirollos
38      * @version $Revision$ $Date$
39      */
40     public class Console {
41     
42         public Console () {
43         }
44     
45         /**
46          * @param args
47          */
48         public static void main (String[] args) {
49             JCESecurityModule sm = new JCESecurityModule();
50             Logger logger = new Logger();
51             logger.addListener(new SimpleLogListener(System.out));
52             sm.setLogger(logger, "jce-security-module");
53             Properties cfgProps = new Properties();
54             SimpleConfiguration cfg = new SimpleConfiguration(cfgProps);
55             String commandName = null;
56             String[] commandParams = new String[10];                 // 10 is Maximum number of paramters for a command
57             System.out.println("Welcome to JCE Security Module console commander!");
58             if (args.length == 0) {
59                 System.out.println("Usage: Console [-options] command [commandparameters...]");
60                 System.out.println("\nwhere options include:");
61                 System.out.println("    -lmk <filename>");
62                 System.out.println("                  to specify the Local Master Keys file");
63                 System.out.println("    -rebuildlmk   to rebuild new Local Master Keys");
64                 System.out.println("                  WARNING: old Local Master Keys gets overwritten");
65                 System.out.println("    -jce <provider classname>");
66                 System.out.println("                  to specify a JavaTM Cryptography Extension 1.2.1 provider");
67                 System.out.println("\nWhere command include: ");
68                 System.out.println("    GC <keyLength>");
69                 System.out.println("                  to generate a clear key component.");
70                 System.out.println("    FK <keyLength> <keyType> <component1> <component2> <component3>");
71                 System.out.println("                  to form a key from three clear components.");
72                 System.out.println("                  and returns the key encrypted under LMK");
73                 System.out.println("                  Odd parity is be forced before encryption under LMK");
74                 System.out.println("    CK <keyLength> <keyType> <KEYunderLMK>");
75                 System.out.println("                  to generate a key check value for a key encrypted under LMK.");
76                 System.out.println("    IK <keyLength> <keyType> <KEYunderKEK> ");
77                 System.out.println("       <kekLength> <kekType> <KEKunderLMK> <KEKcheckValue>");
78                 System.out.println("                  to import a key from encryption under KEK (eg. ZMK,TMK) to encryption under LMK");
79                 System.out.println("                  Odd parity is be forced before encryption under LMK");
80                 System.out.println("    KE <keyLength> <keyType> <KEYunderLMK> <KEYcheckValue> ");
81                 System.out.println("       <kekLength> <kekType> <KEKunderLMK> <KEKcheckValue> ");
82                 System.out.println("                  to translate (export) a key from encryption under LMK");
83                 System.out.println("                  to encryption under KEK (eg. ZMK,TMK)");
84             }
85             else {
86                 int argsCounter = 0;
87                 for (int j = 0; j < 10; j++) {
88                     if ((argsCounter < args.length) &&
89                         (args[argsCounter].toLowerCase().compareTo("-lmk") == 0)
90                     ) {
91                         argsCounter++;
92                         cfgProps.setProperty("lmk", args[argsCounter++]);
93                     }
94                     if ((argsCounter < args.length) &&
95                         (args[argsCounter].toLowerCase().compareTo("-jce") == 0)
96                     ) {
97                         argsCounter++;
98                         cfgProps.setProperty("provider", args[argsCounter++]);
99                     }
100                     if ((argsCounter < args.length) &&
101                         (args[argsCounter].toLowerCase().compareTo("-rebuildlmk") == 0)
102                     ) {
103                         argsCounter++;
104                         cfgProps.setProperty("rebuildlmk", "true");
105                     }
106                 }
107                 if (argsCounter < args.length) {
108                     commandName = args[argsCounter++];
109                     int i = 0;
110                     while (argsCounter < args.length) {
111                         commandParams[i++] = args[argsCounter++];
112                     }
113                 }
114                 // Configure JCE Security Module
115                 try {
116                     sm.setConfiguration(cfg);
117                 } catch (ConfigurationException e) {
118                     e.printStackTrace();
119                     System.exit(0);
120                 }
121                 // Execute Command
122                 if (commandName != null) {
123                     try {
124                         short keyLength = (short)Integer.parseInt(commandParams[0]);
125                         if (commandName.toUpperCase().compareTo("GC") == 0) {
126                             String clearKeyComponenetHexString = sm.generateClearKeyComponent(keyLength);
127                         }
128                         else if (commandName.toUpperCase().compareTo("FK") == 0) {
129                             SecureDESKey KEYunderLMK = sm.formKEYfromThreeClearComponents(keyLength,
130                                     commandParams[1].toUpperCase(), commandParams[2], commandParams[3], commandParams[4]);
131                         }
132                         else if (commandName.toUpperCase().compareTo("CK") == 0) {
133                             byte[] keyCheckValue = sm.generateKeyCheckValue(
134                                     new SecureDESKey(keyLength,commandParams[1].toUpperCase(), commandParams[2],""));
135                         }
136                         else if (commandName.toUpperCase().compareTo("IK") == 0) {
137                             SecureDESKey KEKunderLMK = new SecureDESKey((short)Integer.parseInt(commandParams[3]),
138                                     commandParams[4].toUpperCase(), commandParams[5], commandParams[6]);
139                             sm.importKey(keyLength, commandParams[1].toUpperCase(),
140                                     ISOUtil.hex2byte(commandParams[2]), KEKunderLMK, true);
141                         }
142                         else if (commandName.toUpperCase().compareTo("KE") == 0) {
143                             SecureDESKey KEKunderLMK = new SecureDESKey((short)Integer.parseInt(commandParams[4]),
144                                     commandParams[5].toUpperCase(), commandParams[6], commandParams[7]);
145                             SecureDESKey KEYunderLMK = new SecureDESKey(keyLength, commandParams[1].toUpperCase(),
146                                     commandParams[2], commandParams[3] );
147                             sm.exportKey(KEYunderLMK, KEKunderLMK);
148                         }
149                         else {
150                             System.err.println("Unknown command: " + commandName);
151                         }
152                     } catch (SMException e) {
153                     //e.printStackTrace();
154                     } catch (java.lang.NumberFormatException e) {
155                         System.err.println("Invalid KeyLength");
156                     }
157                 }
158                 else {
159                     System.err.println("No command specified");
160                 }
161             }
162         }
163     }
