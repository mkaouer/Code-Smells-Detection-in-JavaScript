37     public class VISA1Packager 
38         extends SimpleLogSource implements ISOPackager, VISA1ResponseFilter
39     {
40         public static final byte[] FS = { (byte)'\034' };
41         int[] sequence;
42         int respField;
43         String badResultCode;
44         String okPattern;
45         VISA1ResponseFilter filter;
46     
47         /**
48          * @param sequence array of fields that go to VISA1 request
49          * @param respField where to put response
50          * @param badResultCode (i.e. "05")
51          * @param okPattern (i.e. "AUT. ")
52          */
53         public VISA1Packager 
54             (int[] sequence, int respField, String badResultCode, String okPattern)
55         { 
56             super();
57             this.sequence      = sequence;
58             this.respField     = respField;
59             this.badResultCode = badResultCode;
60             this.okPattern     = okPattern;
61             setVISA1ResponseFilter (this);
62         }
63         public void setVISA1ResponseFilter (VISA1ResponseFilter filter) {
64             this.filter = filter;
65         }
66         protected int handleSpecialField35 (ISOMsg m, List l) 
67             throws ISOException
68         {
69             int len = 0;
70             byte[] entryMode = new byte[1];
71             if (m.hasField (35)) {
72                 entryMode[0] = (byte) '\001';
73                 byte[] value = m.getString(35).getBytes();
74                 l.add (entryMode);
75                 l.add (value);
76                 l.add (FS);
77                 len += value.length+2;
78             } else if (m.hasField (2) && m.hasField (14)) {
79                 entryMode[0] = (byte) '\000';
80                 String simulatedTrack2 = m.getString(2) + "=" + m.getString(14);
81                 l.add (entryMode);
82                 l.add (simulatedTrack2.getBytes());
83                 l.add (FS);
84                 len += simulatedTrack2.length()+2;
85             }
86             return len;
87         }
88         public byte[] pack (ISOComponent c) throws ISOException
89         {
90             LogEvent evt = new LogEvent (this, "pack");
91             try {
92                 if (!(c instanceof ISOMsg))
93                     throw new ISOException
94                         ("Can't call VISA1 packager on non ISOMsg");
95             
96                 ISOMsg m = (ISOMsg) c;
97     
98                 int len  = 0;
99                 List<byte[]> l = new ArrayList();
100                 for (int i=0; i<sequence.length; i++) {
101                     int fld = sequence[i];
102                     if (fld == 35) 
103                         len += handleSpecialField35 (m, l);
104                     else if (m.hasField(fld)) {
105                         byte[] value;
106                         if (fld == 4) {
107                             long amt = Long.valueOf(m.getString(4));
108                             value = ISOUtil.formatAmount (amt,12).trim().getBytes();
109                         }
110                         else
111                             value = m.getString(fld).getBytes();
112                         l.add(value);
113                         len += value.length;
114                         if (i < (sequence.length-1)) {
115                             l.add(FS);
116                             len++;
117                         }
118                     }
119                 }
120     
121                 int k = 0;
122                 byte[] d = new byte[len];
123                 for (byte[] b :l) {
124                     System.arraycopy(b, 0, d, k, b.length);
125                     k += b.length;
126                 }
127                 if (logger != null)  // save a few CPU cycle if no logger available
128                     evt.addMessage (ISOUtil.dumpString (d));
129                 return d;
130             } catch (ISOException e) {
131                 evt.addMessage (e);
132                 throw e;
133             } finally {
134                 Logger.log(evt);
135             }
136         }

138         public String guessAutNumber (String s) {
139             StringBuffer buf = new StringBuffer();
140             for (int i=0; i<s.length(); i++)
141                 if (Character.isDigit(s.charAt(i))) 
142                     buf.append (s.charAt(i));
143             if (buf.length() == 0)
144                 return null;
145     
146             while (buf.length() > 6)
147                 buf.deleteCharAt(0);
148             while (buf.length() < 6)
149                 buf.insert(0, "0");
150             
151             return buf.toString();
152         }
153     
154         public int unpack (ISOComponent m, byte[] b) throws ISOException
155         {
156             String response = new String (b);
157             m.set (new ISOField (respField, response));
158             m.set (new ISOField (39, badResultCode));
159             if (response.startsWith (okPattern)) {
160                 String autNumber = filter.guessAutNumber (response);
161                 if (autNumber != null) {
162                     m.set (new ISOField (39, "00"));
163                     m.set (new ISOField (38, autNumber));
164                 }
165             }
166             return b.length;
167         }
168         public void unpack (ISOComponent m, InputStream in) throws ISOException {
169             throw new ISOException ("not implemented");
170         }

172         {

174         }
