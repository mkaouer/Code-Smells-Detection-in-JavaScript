22     /**
23      * IF_UNUSED acts as a filler for unused elements of a message packager.
24      * Packing or unpacking with this packager throws an ISOException.
25      * Use IF_NOP if you don't want an exception thrown.
26      * 
27      * @author jonathan.oconnor@xcom.de
28      * @version $Id$
29      * @see ISOComponent
30      * @see IF_NOP
31      */
32     public class IF_UNUSED extends ISOFieldPackager {
33         /**
34          * @param len - field len
35          * @param description symbolic descrption
36          */
37         public IF_UNUSED () {
38             super(0, "<dummy>");
39         }
40         /**
41          * @param c - a component
42          * @return packed component
43          * @exception ISOException
44          */
45         public byte[] pack (ISOComponent c) throws ISOException {
46             throw new ISOException("IF_UNUSED: Packager should not pack field " + c.getKey());
47         }
48         /**
49          * @param c - the Component to unpack
50          * @param b - binary image
51          * @param offset - starting offset within the binary image
52          * @return consumed bytes
53          */
54         public int unpack (ISOComponent c, byte[] b, int offset) throws ISOException {
55             throw new ISOException("IF_UNUSED: Packager should not unpack field " + c.getKey());
56         }
57         public int getMaxPackedLength() {
58             return 0;
59         }
60     }
