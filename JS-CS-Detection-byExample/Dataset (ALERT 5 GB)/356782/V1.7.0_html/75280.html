43         public synchronized void out (K key, V value) {
44             if (key == null || value == null)
45                 throw new NullPointerException ("key=" + key + ", value=" + value);
46             getList (key).add (value);
47             this.notifyAll ();
48             if (sl != null)
49                 notifyListeners(key, value);
50         }
51         public void out (K key, V value, long timeout) {
52             if (key == null || value == null)
53                 throw new NullPointerException ("key=" + key + ", value=" + value);
54             Object v = value;
55             if (timeout > 0) {
56                 v = new Expirable (value, System.currentTimeMillis() + timeout);
57             }
58             synchronized (this) {
59                 getList (key).add (v);
60                 this.notifyAll ();
61             }
62             if (sl != null)
63                 notifyListeners(key, value);
64         }
65         public synchronized V rdp (Object key) {

75         public synchronized V in (Object key) {
76             Object obj;
77             while ((obj = inp (key)) == null) {
78                 try {
79                     this.wait ();

84         public synchronized V in  (Object key, long timeout) {
85             Object obj;
86             long now = System.currentTimeMillis();
87             long end = now + timeout;
88             while ((obj = inp (key)) == null && 
89                     ((now = System.currentTimeMillis()) < end))
90             {
91                 try {
92                     this.wait (end - now);
93                 } catch (InterruptedException e) { }

126         public void gc () {
127             Object[] keys;
128             synchronized (this) {
129                 keys = entries.keySet().toArray();
130             }
131             for (int i=0; i<keys.length; i++) {
132                 rdp (keys[i]);
133                 Thread.yield ();
134             }
135             if (sl != null) {
136                 synchronized (this) {
137                     if (sl != null && sl.isEmpty())
138                         sl = null;
139                 }
140             }
141         }

197         public void notifyListeners (Object key, Object value) {
198             Object[] listeners = null;
199             synchronized (this) {
200                 if (sl == null)
201                     return;
202                 List l = (List) sl.entries.get (key);
203                 if (l != null)
204                     listeners = l.toArray();
205             }
206             if (listeners != null) {
207                 for (int i=0; i<listeners.length; i++) {
208                     Object o = listeners[i];
209                     if (o instanceof Expirable)
210                         o = ((Expirable)o).getValue();
211                     if (o instanceof SpaceListener)
212                         ((SpaceListener) o).notify (key, value);
213                 }
214             }
215         }

274         public boolean existAny (K[] keys, long timeout) {
275             long now = System.currentTimeMillis();
276             long end = now + timeout;
277             while (((now = System.currentTimeMillis()) < end)) {
278                 if (existAny (keys))
279                     return true;
280                 synchronized (this) {
281                     try {
282                         wait (end - now);
283                     } catch (InterruptedException e) { }
284                 }
285             }
286             return false;
287         }

302         private List getList (Object key) {
303             List l = (List) entries.get (key);
304             if (l == null) 
305                 entries.put (key, l = new LinkedList());
306             return l;
307         }
308         private Object getHead (Object key, boolean remove) {
309             Object obj = null;
310             List l = (List) entries.get (key);
311             while (obj == null && l != null && l.size() > 0) {
312                 obj = l.get(0);
313                 if (obj instanceof Expirable) 
314                     obj = ((Expirable) obj).getValue();
315                 if (obj == null) {
316                     l.remove (0);
317                     if (l.size() == 0) {
318                         entries.remove (key);
319                     }
320                 }
321             }
322             if (obj != null && remove) {
323                 l.remove (0);
324                 if (l.size() == 0) {
325                     entries.remove (key);
326                 }
327             }
328             return obj;
329         }
330         private Object getObject (Template tmpl, boolean remove) {
331             Object obj = null;
332             List l = (List) entries.get (tmpl.getKey());
333             if (l == null)
334                 return obj;
335     
336             Iterator iter = l.iterator();
337             while (iter.hasNext()) {
338                 obj = iter.next();
339                 if (obj instanceof Expirable) {
340                     obj = ((Expirable) obj).getValue();
341                     if (obj == null) {
342                         iter.remove();
343                         continue;
344                     }
345                 }
346                 if (tmpl.equals (obj)) {
347                     if (remove)
348                         iter.remove();
349                     break;
350                 } else
351                     obj = null;
352             }
353             return obj;
354         }

362         static class Expirable implements Comparable {
363             Object value;
364             long expires;
365     
366             public Expirable (Object value, long expires) {
367                 super();
368                 this.value = value;
369                 this.expires = expires;
370             }
371             public boolean isExpired () {
372                 return expires < System.currentTimeMillis ();
373             }
374             public String toString() {
375                 return getClass().getName() 
376                     + "@" + Integer.toHexString(hashCode())
377                     + ",value=" + value.toString()
378                     + ",expired=" + isExpired ();
379             }
380             public Object getValue() {
381                 return isExpired() ? null : value;
382             }
383             public int compareTo (Object obj) {
384                 Expirable other = (Expirable) obj;
385                 long otherExpires = other.expires;
386                 if (otherExpires == expires)
387                     return 0;
388                 else if (expires < otherExpires)
389                     return -1;
390                 else 
391                     return 1;
392             }
393         }
