19     package org.jpos.space;

21     import org.jpos.core.Configurable;
22     import org.jpos.core.Configuration;
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.iso.ISOChannel;
25     import org.jpos.iso.ISOMsg;
26     import org.jpos.iso.ISOUtil;
27     import org.jpos.util.LogEvent;
28     import org.jpos.util.Logger;
29     import org.jpos.util.NameRegistrar;
30     import org.jpos.util.SimpleLogSource;
31     
32     import java.io.IOException;
33     
34     
35     /**
36      * Integrates legacy (pre 2.0) ISOChannels to new Space based architecture
37      * @author Alejandro Revilla
38      * @version $Revision$ $Date$
39      */
40     public class ISOChannelAdaptor 
41             extends SimpleLogSource 
42             implements Configurable
43     {
44         Space sp;
45         Configuration cfg;
46         ISOChannel channel;
47         String to, from, ready;
48         public ISOChannelAdaptor () {
49             super ();
50             sp = TransientSpace.getSpace ();
51         }
52         public void setConfiguration (Configuration cfg) 
53             throws ConfigurationException 
54         {
55             this.cfg = cfg;
56             try {
57                 to   = get ("to");
58                 from = get ("from");
59                 channel = (ISOChannel) NameRegistrar.get (
60                     "channel." + get ("channel")
61                 );
62                 ready  = channel.toString() + ".ready";
63                 new Thread (new Sender ()).start ();
64                 new Thread (new Receiver ()).start ();
65             } catch (NameRegistrar.NotFoundException e) {
66                 throw new ConfigurationException (e);
67             }
68         }
69         public class Sender implements Runnable {
70             public Sender () {
71                 super ();
72             }
73             public void run () {
74                 Thread.currentThread().setName ("channel-sender-" + to);
75                 for (;;) {
76                     try {
77                         checkConnection ();
78                         Object o = sp.in (to, 10000);
79                         if (o instanceof ISOMsg)
80                             channel.send ((ISOMsg) o);
81                     } catch (Exception e) { 
82                         Logger.log (
83                             new LogEvent (
84                                 ISOChannelAdaptor.this, "channel-sender"+to, e
85                             )
86                         );
87                         ISOUtil.sleep (1000);
88                     }
89                 }
90             }
91         }
92         public class Receiver implements Runnable {
93             public Receiver () {
94                 super ();
95             }
96             public void run () {
97                 Thread.currentThread().setName ("channel-receiver-"+from);
98                 for (;;) {
99                     try {
100                         sp.rd (ready);
101                         ISOMsg m = channel.receive ();
102                         sp.out (from, m);
103                     } catch (Exception e) { 
104                         Logger.log (
105                             new LogEvent (
106                                 ISOChannelAdaptor.this, "channel-receiver-"+from, e
107                             )
108                         );
109                         disconnect ();
110                     }
111                 }
112             }
113         }
114         protected void checkConnection () {
115             try {
116                 while (!channel.isConnected ()) {
117                     while (sp.inp (ready) != null)
118                         ;
119                     channel.connect ();
120                     if (!channel.isConnected ())
121                         ISOUtil.sleep (10000);
122                 }
123                 sp.out (ready, new Object ());
124             } catch (IOException e) {
125                 Logger.log (
126                     new LogEvent (
127                         ISOChannelAdaptor.this, "check-connection", e
128                     )
129                 );
130                 ISOUtil.sleep (10000);
131             }
132         }
133         protected void disconnect () {
134             try {
135                 while (sp.inp (ready) != null)
136                     ;
137                 channel.disconnect ();
138             } catch (IOException e) {
139                 Logger.log (
140                     new LogEvent (
141                         ISOChannelAdaptor.this, "disconnect", e
142                     )
143                 );
144             }
145         }
146         private String get (String prop) throws ConfigurationException
147         {
148             String value = cfg.get (prop);
149             if (value == null)
150                 throw new ConfigurationException ("null property "+prop);
151             return value;
152         }
153     }
