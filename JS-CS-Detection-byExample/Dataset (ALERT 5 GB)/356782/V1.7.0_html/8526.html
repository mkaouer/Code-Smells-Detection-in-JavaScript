19     package org.jpos.q2.iso;
20     
21     import org.jdom.Element;
22     import org.jpos.core.Configurable;
23     import org.jpos.iso.ISOUtil;
24     import org.jpos.q2.QBeanSupport;
25     import org.jpos.q2.QFactory;
26     
27     import java.util.Calendar;
28     import java.util.Date;
29     import java.util.GregorianCalendar;
30     
31     /**
32      * DailyTask Adaptor
33      *
34      * @author Alejandro Revilla
35      * @version $Revision$ $Date$
36      */
37     public class DailyTaskAdaptor extends QBeanSupport implements Runnable {
38         Runnable task;
39         Thread thisThread = null;
40     
41         public DailyTaskAdaptor () {
42             super ();
43         }
44     
45         protected void initService () throws Exception {
46             QFactory factory = getServer().getFactory();
47             Element e = getPersist ();
48             task = (Runnable) factory.newInstance (e.getChildTextTrim ("class"));
49             factory.setLogger (task, e);
50         }
51         protected void startService () throws Exception {
52             if (task instanceof Configurable) {
53                 Element e = getPersist ();
54                 QFactory factory = getServer().getFactory();
55                 ((Configurable)task).setConfiguration (
56                     factory.getConfiguration (e)
57                 );
58             }
59             (thisThread = new Thread(this)).start();
60         }
61         protected void stopService () throws Exception {
62             if (thisThread != null)
63                 thisThread.interrupt();
64         }
65         public void run () {
66             while (running()) {
67                 waitUntilStartTime();
68                 if (running()) {
69                     Thread taskThread = new Thread(task);
70                     taskThread.setDaemon (true);
71                     taskThread.start();
72                     ISOUtil.sleep (1000);
73                 }
74             }
75         }
76         public Date getWhen() {
77             String s = cfg.get ("start")+":00:00";
78             int hh = Integer.parseInt(s.substring (0, 2));
79             int mm = Integer.parseInt(s.substring (3, 5));
80             int ss = Integer.parseInt(s.substring (6, 8));
81     
82             Date now = new Date();
83             Calendar cal = new GregorianCalendar();
84     
85             cal.setTime (now);
86             cal.set (Calendar.HOUR_OF_DAY, hh);
87             cal.set (Calendar.MINUTE, mm);
88             cal.set (Calendar.SECOND, ss);
89     
90             Date when = cal.getTime();
91             if (when.before(now)) 
92                 when = new Date(when.getTime() + 24*60*60*1000);
93     
94             return when;
95         }
96         protected void waitUntilStartTime() {
97             Date when = getWhen();
98             while (running()) {
99                 Date now = new GregorianCalendar().getTime();
100                 if (now.before (when)) {
101                     long sleepTime = when.getTime() - now.getTime();
102                     if (sleepTime <= 0) {
103                         ISOUtil.sleep(1000);
104                         continue;
105                     }
106                     getLog().info ("sleeping",
107                         (sleepTime/1000) + " secs until " + when.toString()
108                     );
109                     try {
110                         Thread.sleep (sleepTime);
111                     } catch (InterruptedException e) { 
112                         when = getWhen();
113                     }
114                 } else
115                     break;
116             }
117         }
118     }
