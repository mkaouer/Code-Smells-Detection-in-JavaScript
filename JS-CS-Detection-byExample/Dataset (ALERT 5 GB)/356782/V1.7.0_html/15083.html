19     package org.jpos.iso.filter;
20     
21     import org.jpos.core.Configurable;
22     import org.jpos.core.Configuration;
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.iso.*;
25     import org.jpos.util.LogEvent;
26     
27     import java.security.MessageDigest;
28     import java.security.NoSuchAlgorithmException;
29     import java.util.Arrays;
30     import java.util.StringTokenizer;
31     
32     /**
33      * Computes an MD5 based Message Authentication Code
34      * on outgoing messages and checks that MAC on incoming
35      * ones.
36      *
37      * @author Alejandro P. Revilla
38      * @version $Revision$ $Date$
39      * @since 1.2.8
40      * @see org.jpos.iso.ISOFilter
41      */
42     public class MD5Filter implements ISOFilter, Configurable {
43         String key;
44         int[] fields;
45     
46         public MD5Filter() {
47             super();
48         }
49        /**
50         * @param cfg
51         * <ul>
52         * <li>key    - initial key
53         * <li>fields - Space separated field list
54         * </ul>
55         */
56         public void setConfiguration (Configuration cfg) 
57             throws ConfigurationException 
58         {
59             key = cfg.get ("key");
60             String fieldList = cfg.get ("fields");
61             if (fieldList == null)
62                 throw new ConfigurationException ("'fields' property not present");
63     
64             StringTokenizer st = new StringTokenizer (fieldList);
65             int f[] = new int[st.countTokens()];
66     
67             for (int i=0; i<f.length; i++) 
68                 f[i] = Integer.parseInt (st.nextToken());
69     
70             fields = f;
71         }
72         public void setFields (int[] fields) {
73             this.fields = fields;
74         }
75         /**
76          * factory method
77          * @param m current ISOMsg
78          * @return key fields associated with this ISOMsg
79          */
80         public int[] getFields (ISOMsg m) {
81             return fields;
82         }
83         public ISOMsg filter (ISOChannel channel, ISOMsg m, LogEvent evt) 
84             throws VetoException
85         {
86             if (key == null || fields == null)
87                 throw new VetoException ("MD5Filter not configured");
88             try {
89                 MessageDigest md = MessageDigest.getInstance("MD5");
90                 md.update (getKey());
91                 int[] f = getFields (m);
92                 for (int i=0; i<f.length; i++) {
93                     int fld = f[i];
94                     if (m.hasField (fld)) {
95                         ISOComponent c = m.getComponent (fld);
96                         if (c instanceof ISOBinaryField)
97                             md.update ((byte[]) c.getValue());
98                         else
99                             md.update (((String)c.getValue()).getBytes());
100                     }
101                 }
102                 byte[] digest = md.digest();
103                 if (m.getDirection() == ISOMsg.OUTGOING) {
104                     m.set (new ISOBinaryField ( 64, digest, 0, 8));
105                     m.set (new ISOBinaryField (128, digest, 8, 8));
106                 } else {
107                     byte[] rxDigest = new byte[16];
108                     if (m.hasField (64))
109                         System.arraycopy (
110                             (byte[]) m.getValue(64), 0, rxDigest, 0, 8
111                         );
112                     if (m.hasField (128))
113                         System.arraycopy (
114                             (byte[]) m.getValue(128), 0, rxDigest, 8, 8
115                         );
116                     if (!Arrays.equals (digest, rxDigest)) {
117                         evt.addMessage (m);
118                         evt.addMessage ("MAC expected: "
119                             +ISOUtil.hexString (digest));
120                         evt.addMessage ("MAC received: "
121                             +ISOUtil.hexString (rxDigest));
122                         throw new VetoException ("invalid MAC");
123                     }
124                     m.unset  (64);
125                     m.unset (128);
126                 }
127             } catch (NoSuchAlgorithmException e) {
128                 throw new VetoException (e);
129             } catch (ISOException e) {
130                 throw new VetoException (e);
131             }
132             return m;
133         }
134         /**
135          * hook for custom key storage (i.e. crypto cards)
136          */
137         protected byte[] getKey() {
138             return key.getBytes();
139         }
140     }
