33     /** This is a log listener that reads its actions from Bean Shell scripts.
34      * You can define many scripts, and the order in wich they are called, also you
35      * can define scripts whose name depends on the event being processed, and the
36      * realm of the object that generated it.
37      * This way you can set a configuration like this:
38      * <log-listener class="org.jpos.bsh.BSHLoglistener">
39      *    <property name="source" value="general.bsh"/>
40      *    <property name="source" value="tag_$tag.bsh"/> <!--this is to handle
41      * specific tags-->
42      *    <property name="source" value="realm_$realm.bsh"/> <!-- to handle specific
43      * realms-->
44      *    <property name="source" value="tag_$tag_realm_$realm.bsh"/> <!-- to handle
45      * specific tags from specific realms-->
46      * </log-listener>
47      * <!-- the following lines are for html doc-->
48      * <br>
49      * <pre>
50      * <font color="#008080">&lt;</font><font color="#008080">log-listener</font><font color="#008080"> </font><font color="#2e8b57"><b>class</b></font>=<font color="#ff00ff">&quot;org.jpos.bsh.BSHLoglistener&quot;</font><font color="#008080">&gt;</font>
51      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;general.bsh&quot;</font><font color="#008080">/&gt;</font>
52      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;tag_$tag.bsh&quot;</font><font color="#008080">/&gt;</font> <font color="#0000ff">&lt;!</font><font color="#0000ff">--this is to handle specific tags--</font><font color="#0000ff">&gt;</font>
53      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;realm_$realm.bsh&quot;</font><font color="#008080">/&gt;</font> <font color="#0000ff">&lt;!</font><font color="#0000ff">-- to handle specific realms--</font><font color="#0000ff">&gt;</font>
54      *       <font color="#008080">&lt;</font><font color="#008080">property</font><font color="#008080"> </font><font color="#2e8b57"><b>name</b></font>=<font color="#ff00ff">&quot;source&quot;</font><font color="#008080"> </font><font color="#2e8b57"><b>value</b></font>=<font color="#ff00ff">&quot;tag_$tag_realm_$realm.bsh&quot;</font><font color="#008080">/&gt;</font> <font color="#0000ff">&lt;!</font><font color="#0000ff">-- to handle specific tags from specific realms--</font><font color="#0000ff">&gt;</font>
55      * <font color="#008080">&lt;/log-listener&gt;</font>
56      * </pre>
57      * <!-- end of lines for html doc-->
58      * If a source with the given name is not found, or it canbe read, it is not processed, so this lets
59      * you change what is processed in real time, if you put a file called
60      * tag_SystemMonitor_realm_monitor.bsh it will be executed whenever the system
61      * monitor is run.<BR>
62      * If you want to filter an event so that the remaining log listeners don't see
63      * it, you have to set event = null in your script.<br>
64      * <table border=1 color="black">
65      * <caption>Other Configuration Options: </caption>
66      * <th><td> Name                </td><td>Type       </td><td>Description</td></th>
67      * <tr><td> filter-by-default   </td><td>boolean    </td><td>
68      *  If true, filter the events if no script is processed, this way you can put a
69      * Log Listener that filters every thing unless you put some script file, even if
70      * it is empty, you also can dynamically change what kind of message are filtered
71      * by placing empty files with names like tag_SystemMonitor_realm_monitor.bsh in
72      * the previous example, if you "touch" a file with this name these events will
73      * begin to be processed.</TD></TR>
74      * <TR><TD> preload-scripts     </TD><TD>boolean    </TD><TD>If true scripts a
75      * loaded once, and kept in memory, being realoaded only if they are touched. This
76      * is good when you have lots of RAM memory but ou have troubles with
77      * speed</TD></TR>
78      * <TR><TD> save-name-space     </TD><TD>boolean    </TD><TD>If true the namespace
79      * of the script instance will be saved so that in the next event you can access
80      * them from the script, by default it's off, this property is overriden if the
81      * script exposes a boolean variable named saveNameSpace</TD></TR>
82      * <TR><TD> reload              </TD><TD>long       </TD><TD>this property is used
83      * if the preload-script property is true, is the time in milliseconds between
84      * updates in the script, during this time BSHLogListener will not check if the
85      * script source was modified or deleted on disk </TD></TR>
86      * </TABLE>
87      */
88     public class BSHLogListener implements org.jpos.util.LogListener, org.jpos.core.Configurable {
89         /**Holds the configuration for this object*/
90         protected Configuration cfg;
91         protected static final String[] patterns = {"tag", "realm"};
92         protected Map<String,ScriptInfo> scripts = new Hashtable<String,ScriptInfo>();
93         /** Creates a new instance of BSHLogListener */
94         public BSHLogListener() {
95         }

97         public void setConfiguration(org.jpos.core.Configuration cfg) {
98             this.cfg = cfg;
99         }
100         protected static String[] replace(String[] src, String[] patterns, String[] to){
101             String[] ret = new String[src.length];
102             for(int i=0; i<src.length; i++){
103                 StringBuffer buff = new StringBuffer(2*src[i].length());
104                 int begin, end=0;
105                 //begin is the position of the next pattern, end is the end of the last pattern
106                 while ((begin = src[i].indexOf('$',end))>=0 && begin<src[i].length()){
107                     buff.append(src[i].substring(end, begin));
108                     boolean patternFound = false;
109                     for(int k=0; k<patterns.length && !patternFound ; k++){
110                         if(patternFound = (src[i].indexOf(patterns[k], begin) == begin+1)){
111                             buff.append(to[k]);
112                             end = begin + patterns[k].length() + 1;
113                         }
114                     }
115                     if(!patternFound) end = begin + 1;
116                 }
117                 buff.append(src[i].substring(end));
118                 //if(buff.length()==0) ret[i] = src[i];
119                 ret[i] = buff.toString();
120             }
121             return ret;
122         }
123         public LogEvent log(org.jpos.util.LogEvent ev) {
124             LogEvent ret = ev;
125             boolean processed = false;
126             try{
127                 String[] sources = replace(cfg.getAll("source"), patterns, new String[] {ev.getTag(), ev.getRealm()});
128                 for(int i=0; i<sources.length && ret != null; i++){
129                     try{
130                         Interpreter bsh = new Interpreter();
131                         BSHLogListener.ScriptInfo info = getScriptInfo(sources[i]);
132                         NameSpace ns = (info!=null)?info.getNameSpace():null;
133                         if(ns!=null) bsh.setNameSpace(ns);
134                         bsh.set("event", ret);
135                         bsh.set("cfg", cfg);
136                         File f = new File(sources[i]);
137                         if(!cfg.getBoolean("preload-scripts")){
138                             if(f.exists() && f.canRead() && f.isFile()){
139                                 //if(f.lastModified())
140                                 processed = true;
141                                 bsh.eval(new java.io.FileReader(f));
142                             }
143                         }else{
144                             if(info == null) scripts.put(sources[i], info=new ScriptInfo());
145                             if(System.currentTimeMillis() > info.getLastCheck() + cfg.getLong("reload")){
146                                 info.setLastCheck(System.currentTimeMillis());
147                                 if(f.exists() && f.canRead() && f.isFile()){
148                                     if(info.getLastModified() != f.lastModified()) {
149                                         info.setLastModified(f.lastModified());
150                                         info.setCode(loadCode(f));
151                                     }
152                                 }else{
153                                     info.setCode(null);
154                                 }
155                             }
156                             if(info.getCode() != null){
157                                 processed = true;
158                                 bsh.eval(new StringReader(info.getCode()));
159                             }else scripts.remove(sources[i]);
160                         }
161                         ret = (LogEvent)bsh.get("event");
162                         Object saveNS = bsh.get("saveNameSpace");
163                         boolean saveNameSpace = 
164                             (saveNS instanceof Boolean)? (Boolean) saveNS :cfg.getBoolean("save-name-space");
165                         if(saveNameSpace) {
166                             if(info!=null) info.setNameSpace(bsh.getNameSpace());
167                             else scripts.put(sources[i], new ScriptInfo(bsh.getNameSpace()));
168                         }else if (info!=null) info.setNameSpace(null);
169                     }catch(Exception e){
170                         ret.addMessage(e);
171                     }
172                 }
173                 return (!processed && cfg.getBoolean("filter-by-default"))?null:ret;
174             }catch(Exception e){
175                 ret.addMessage(e);
176                 return ret;
177             }
178         }
179         protected String loadCode(File f) throws IOException{
180             FileReader r = new FileReader(f);
181             StringBuffer buf = new StringBuffer((int)f.length());
182             char[] content = new char[(int)f.length()];
183             int l;
184             while((l=r.read(content))!=-1) buf.append(content,0,l);
185             r.close();
186             return buf.toString();
187         }
188         
189         protected ScriptInfo getScriptInfo(String filename){
190             return scripts.get(filename);
191         }
192         protected void addScriptInfo(String filename, String code, long lastModified){
193             scripts.put(filename, new ScriptInfo(code, lastModified));
194         }
