19     package org.jpos.iso.channel;
20     
21     import org.jpos.core.Configurable;
22     import org.jpos.core.Configuration;
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.iso.ISOChannel;
25     import org.jpos.iso.ISOException;
26     import org.jpos.iso.ISOMsg;
27     import org.jpos.iso.ISOPackager;
28     import org.jpos.util.LogEvent;
29     import org.jpos.util.LogSource;
30     import org.jpos.util.Logger;
31     import org.jpos.util.NameRegistrar;
32     
33     import java.io.IOException;
34     import java.util.List;
35     import java.util.Vector;
36     
37     public class ChannelPool implements ISOChannel, LogSource, Configurable, Cloneable {
38         boolean usable = true;
39         String name = "";
40         protected Logger logger;
41         protected String realm;
42         Configuration cfg = null;
43         List pool;
44         ISOChannel current;
45     
46         public ChannelPool () {
47             super ();
48             pool = new Vector ();
49         }
50         public void setPackager(ISOPackager p) {
51             // nothing to do
52         }
53         public synchronized void connect () throws IOException {
54             current = null;
55             LogEvent evt = new LogEvent (this, "connect");
56             evt.addMessage ("pool-size=" + Integer.toString (pool.size()));
57             for (int i=0; i<pool.size(); i++) {
58                 try {
59                     evt.addMessage ("pool-" + Integer.toString (i));
60                     ISOChannel c = (ISOChannel) pool.get (i);
61                     c.connect ();
62                     if (c.isConnected()) {
63                         current = c;
64                         usable = true;
65                         break;
66                     }
67                 } catch (IOException e) {
68                     evt.addMessage (e);
69                 }
70             }
71             if (current == null)
72                 evt.addMessage ("connect failed");
73             Logger.log (evt);
74             if (current == null) {
75                 throw new IOException ("unable to connect");
76             }
77         }
78         public synchronized void disconnect () throws IOException {
79             current = null;
80             LogEvent evt = new LogEvent (this, "disconnect");
81             for (int i=0; i<pool.size(); i++) {
82                 try {
83                     ISOChannel c = (ISOChannel) pool.get (i);
84                     c.disconnect ();
85                 } catch (IOException e) {
86                     evt.addMessage (e);
87                 }
88             }
89             Logger.log (evt);
90         }
91         public synchronized void reconnect() throws IOException {
92             disconnect ();
93             connect ();
94         }
95         public synchronized boolean isConnected() {
96             try {
97                 return getCurrent().isConnected ();
98             } catch (IOException e) {
99                 return false;
100             }
101         }
102         public ISOMsg receive() throws IOException, ISOException {
103             return getCurrent().receive ();
104         }
105         public void send (ISOMsg m) throws IOException, ISOException {
106             getCurrent().send (m);
107         }
108         public void send (byte[] b) throws IOException, ISOException {
109             getCurrent().send (b);
110         }
111         public void setUsable(boolean b) {
112             this.usable = b;
113         }
114         public void setName (String name) {
115             this.name = name;
116             NameRegistrar.register ("channel."+name, this);
117         }
118         public String getName() {
119             return this.name;
120         }
121         public ISOPackager getPackager () {
122             return (ISOPackager) null;
123         }
124         public void setLogger (Logger logger, String realm) {
125             this.logger = logger;
126             this.realm  = realm;
127         }
128         public String getRealm () {
129             return realm;
130         }
131         public Logger getLogger() {
132             return logger;
133         }
134         public synchronized void setConfiguration (Configuration cfg) 
135             throws ConfigurationException
136         {
137             this.cfg = cfg;
138             String channelName[] = cfg.getAll ("channel");
139             for (int i=0; i<channelName.length; i++) {
140                 try {
141                     addChannel (channelName[i]);
142                 } catch (NameRegistrar.NotFoundException e) {
143                     throw new ConfigurationException (e);
144                 }
145             }
146         }
147         public void addChannel (ISOChannel channel) {
148             pool.add (channel);
149         }
150         public void addChannel (String name) 
151             throws NameRegistrar.NotFoundException
152         {
153             pool.add ((ISOChannel) NameRegistrar.get ("channel."+name));
154         }
155         public void removeChannel (ISOChannel channel) {
156             pool.remove (channel);
157         }
158         public void removeChannel (String name) throws NameRegistrar.NotFoundException {
159             pool.remove ((ISOChannel) NameRegistrar.get ("channel."+name));
160         }
161         public int size() {
162             return pool.size();
163         }
164         public synchronized ISOChannel getCurrent () throws IOException {
165             if (current == null)
166                 connect();
167             else if (!usable)
168                 reconnect();
169     
170             return current;
171         }
172         
173         public Object clone(){
174           try {
175             return (ChannelPool)super.clone();
176           } catch (CloneNotSupportedException e) {
177             throw new InternalError();
178           }
179         }
180     }
