19     package  org.jpos.security;
20     
21     import org.jpos.iso.ISOUtil;
22     import org.jpos.util.Loggeable;
23     
24     import java.io.PrintStream;
25     import java.io.Serializable;
26     
27     
28     /**
29      * <p>
30      * The PIN (Personal Identification Number), is used to authenticate card
31      * holders. A user enters his/her PIN on the pin-pad (also called pin entry device)
32      * of a terminal (whether ATM or POS). The terminal forms the PIN Block, which
33      * is a mix of the PIN and the account number.<br>
34      * In a typical environment, the PIN Block (not the PIN) gets encrypted and sent
35      * to the acquirer. This Encrypted PIN Block is the typical content of the
36      * PIN Data ISO Field (Field 52).
37      * This class represents an encrypted PIN, no matter by whom it is encrypted.
38      * Typically a PIN is encrypted under one of these three:<br>
39      * 1- Under a terminal PIN key (like TPK or DUKPT)<br>
40      * 2- Under an Interchange PIN key (like ZPK)<br>
41      * 3- Under the the security module itself (i.e. under LMK)<br>
42      * This class knows nothing about, who encrypted it.
43      * </p>
44      * <p>
45      * This class represents an encrypted PIN using:<br>
46      * 1- The PIN Block (encrypted)<br>
47      * 2- The account number (the 12 right-most digits of the account number excluding the check digit)<br>
48      * 3- The PIN Block Format<br>
49      * </p>
50      * <p>
51      * The PIN Block Format specifies how the clear pin (as entered by the card holder)
52      * and the account number get mixed to form the PIN Block.<br>
53      * </p>
54      * @author Hani Samuel Kirollos
55      * @version $Revision$ $Date$
56      * @see SMAdapter
57      */
58     public class EncryptedPIN
59             implements Serializable, Loggeable {
60     
61         private static final long serialVersionUID = -9117335317030664867L;
62         /**
63          * Account Number (the 12 right-most digits of the account number excluding the check digit)
64          */
65         String accountNumber;
66         /**
67          * This is the ByteArray holding the PIN Block
68          * The PIN Block can be either clear or encrypted
69          * It is typically DES or 3DES encrypted, with length 8 bytes.
70          */
71         byte[] pinBlock;
72         /**
73          * The PIN Block Format
74          * value -1 means no block format defined
75          */
76         byte pinBlockFormat;
77     
78         public EncryptedPIN () {
79         }
80     
81         /**
82          *
83          * @param pinBlock
84          * @param pinBlockFormat
85          * @param accountNumber (also functions correctly, if the complete account number with the check digit is passed)
86          */
87         public EncryptedPIN (byte[] pinBlock, byte pinBlockFormat, String accountNumber) {
88             setPINBlock(pinBlock);
89             setPINBlockFormat(pinBlockFormat);
90             setAccountNumber(accountNumber);
91         }
92     
93         /**
94          *
95          * @param pinBlockHexString the PIN Block represented as a HexString instead of a byte[]
96          * @param pinBlockFormat
97          * @param accountNumber (also functions correctly, if the complete account number with the check digit is passed)
98          */
99         public EncryptedPIN (String pinBlockHexString, byte pinBlockFormat, String accountNumber) {
100             this(ISOUtil.hex2byte(pinBlockHexString), pinBlockFormat, accountNumber);
101         }
102     
103     
104         /**
105          * dumps PIN basic information
106          * @param p a PrintStream usually supplied by Logger
107          * @param indent indention string, usually suppiled by Logger
108          * @see org.jpos.util.Loggeable
109          */
110         public void dump (PrintStream p, String indent) {
111             String inner = indent + "  ";
112             p.print(indent + "<encrypted-pin");
113             p.print(" format=\"0" + getPINBlockFormat() + "\"");
114             p.println(">");
115             p.println(inner + "<pin-block>" + ISOUtil.hexString(getPINBlock()) + "</pin-block>");
116             p.println(inner + "<account-number>" + getAccountNumber() + "</account-number>");
117             p.println(indent + "</encrypted-pin>");
118         }
119     
120         /**
121          *
122          * @param pinBlock
123          */
124         public void setPINBlock (byte[] pinBlock) {
125             this.pinBlock = pinBlock;
126         }
127     
128         /**
129          *
130          * @return pinBlock
131          */
132         public byte[] getPINBlock () {
133             return  pinBlock;
134         }
135     
136         /**
137          *
138          * @param pinBlockFormat
139          */
140         public void setPINBlockFormat (byte pinBlockFormat) {
141             this.pinBlockFormat = pinBlockFormat;
142         }
143     
144         /**
145          *
146          * @return PIN Block Format
147          */
148         public byte getPINBlockFormat () {
149             return  this.pinBlockFormat;
150         }
151     
152         /**
153          * Sets the 12 right-most digits of the account number excluding the check digit
154          * @param accountNumber (also functions correctly, if the complete account number with the check digit is passed)
155          */
156         public void setAccountNumber (String accountNumber) {
157             this.accountNumber = extractAccountNumberPart(accountNumber);
158         }
159     
160         /**
161          * @return accountNumber (the 12 right-most digits of the account number excluding the check digit)
162          */
163         public String getAccountNumber () {
164             return  accountNumber;
165         }
166     
167         /**
168          * This method extracts the 12 right-most digits of the account number,
169          * execluding the check digit.
170          * @param accountNumber (PAN) consists of the BIN (Bank Identification Number), accountNumber
171          * and a check digit.
172          * @return the 12 right-most digits of the account number, execluding the check digit.
173          */
174         public static String extractAccountNumberPart (String accountNumber) {
175             String accountNumberPart = null;
176             if (accountNumber.length() > 12)
177                 accountNumberPart = accountNumber.substring(accountNumber.length() -
178                         13, accountNumber.length() - 1);
179             else
180                 accountNumberPart = accountNumber;
181             return  accountNumberPart;
182         }

184     
185     }
