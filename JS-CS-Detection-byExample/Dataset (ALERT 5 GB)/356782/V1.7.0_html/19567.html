19     package org.jpos.core;
20     
21     import java.io.BufferedInputStream;
22     import java.io.FileInputStream;
23     import java.io.FileNotFoundException;
24     import java.io.IOException;
25     import java.util.List;
26     import java.util.Properties;
27     
28     /**
29      * @author apr@cs.com.uy
30      * @version $Id$
31      * @since jPOS 1.1
32      */
33     public class SimpleConfiguration implements Configuration {
34         private Properties props;
35     
36         public SimpleConfiguration () {
37             props = new Properties();
38         }
39         public SimpleConfiguration (Properties props) {
40             this.props = props;
41         }
42         public SimpleConfiguration (String filename) 
43             throws FileNotFoundException, IOException
44         {
45             props = new Properties();
46             load (filename);
47         }
48         public String get (String name, String def) {
49             Object obj = props.get (name);
50             if (obj instanceof List) {
51                 List l = (List) obj;
52                 obj = (l.size() > 0) ? l.get(0) : null;
53             }
54             return (obj instanceof String) ? ((String) obj) : def;
55         }
56         public String[] getAll (String name) {
57             String[] ret;
58             Object obj = props.get (name);
59             if (obj instanceof String[]) {
60                 ret = (String[]) obj;
61             } else if (obj instanceof String) {
62                 ret = new String[1];
63                 ret[0] = (String) obj;
64             } else
65                 ret = new String[0];
66     
67             return ret;
68         }
69         public int[] getInts (String name) {
70             String[] ss = getAll (name);
71             int[] ii = new int[ss.length];
72             for (int i=0; i<ss.length; i++)
73                 ii[i] = Integer.parseInt(ss[i].trim());
74             return ii;
75         }
76         public long[] getLongs (String name) {
77             String[] ss = getAll (name);
78             long[] ll = new long[ss.length];
79             for (int i=0; i<ss.length; i++)
80                 ll[i] = Long.parseLong(ss[i].trim());
81             return ll;
82         }
83         public double[] getDoubles (String name) {
84             String[] ss = getAll (name);
85             double[] dd = new double[ss.length];
86             for (int i=0; i<ss.length; i++)
87                 dd[i] = Double.valueOf(ss[i].trim()).doubleValue();
88             return dd;
89         }
90         public boolean[] getBooleans (String name) {
91             String[] ss = getAll (name);
92             boolean[] bb = new boolean[ss.length];
93             for (int i=0; i<ss.length; i++)
94                 bb[i] = ss[i].equalsIgnoreCase("true") || ss[i].equalsIgnoreCase("yes");
95             return bb;
96         }
97         public String get (String name) {
98             return get (name, "");
99         }
100         public int getInt (String name) {
101             return Integer.parseInt(props.getProperty(name, "0").trim());
102         }
103         public int getInt (String name, int def) {
104             return Integer.parseInt(
105                 props.getProperty (name, Integer.toString (def)).trim());
106         }
107         public long getLong (String name) {
108             return Long.parseLong(props.getProperty(name, "0").trim());
109         }
110         public long getLong (String name, long def) {
111             return Long.parseLong (
112                 props.getProperty (name, Long.toString (def)).trim());
113         }
114         public double getDouble(String name) {
115             return Double.valueOf(
116                 props.getProperty(name,"0.00").trim()).doubleValue();
117         }
118         public double getDouble(String name, double def) {
119             return Double.valueOf(
120                 props.getProperty(name,Double.toString(def)).trim()).doubleValue();
121         }
122         public boolean getBoolean (String name) {
123             String v = get (name, "false").trim();
124             return v.equalsIgnoreCase("true") || v.equalsIgnoreCase("yes");
125         }
126         public boolean getBoolean (String name, boolean def) {
127             String v = get (name);
128             return v.length() == 0 ? def :
129                 (v.equalsIgnoreCase("true") || v.equalsIgnoreCase("yes"));
130         }
131         public void load(String filename) 
132             throws FileNotFoundException, IOException
133         {
134             FileInputStream fis = new FileInputStream(filename);
135             props.load(new BufferedInputStream(fis));
136             fis.close();
137         }
138         public void put (String name, Object value) {
139             props.put (name, value);
140         }
141     }
