19     package org.jpos.q2.qbean;
20     
21     import org.jpos.q2.Q2;
22     import org.jpos.q2.QBeanSupport;
23     import org.jpos.space.LocalSpace;
24     import org.jpos.space.Space;
25     import org.jpos.space.SpaceFactory;
26     
27     import javax.management.ObjectName;
28     import java.util.Set;
29     
30     /**
31      * Space Adaptor
32      *
33      * @author Alejandro Revilla
34      * @version $Revision$ $Date$
35      * @jmx:mbean description="Space QBean"
36      *                  extends="org.jpos.q2.QBeanSupportMBean"
37      */
38     
39     public class SpaceAdaptor 
40         extends QBeanSupport
41         implements SpaceAdaptorMBean
42     {
43         private Space sp = null;
44         private String spaceName = null;
45         private ObjectName objectName = null;
46     
47         public SpaceAdaptor () {
48             super ();
49         }
50     
51         protected void startService () throws Exception {
52             if (spaceName == null) 
53                 sp = SpaceFactory.getSpace ();
54             else 
55                 sp = SpaceFactory.getSpace (spaceName);
56     
57             objectName = new ObjectName (Q2.QBEAN_NAME + 
58                 getName() + ",space=" +
59                 (spaceName != null ? spaceName : "default")
60             );
61             getServer().getMBeanServer().registerMBean (sp, objectName);
62         }
63     
64         protected void stopService () throws Exception {
65             getServer().getMBeanServer().unregisterMBean (objectName);
66         }
67     
68         /**
69          * @jmx:managed-attribute description="Space Name"
70          */
71         public synchronized void setSpaceName (String spaceName) {
72             this.spaceName = spaceName;
73             setAttr (getAttrs (), "spaceName", spaceName);
74             setModified (true);
75         }
76     
77         /**
78          * @jmx:managed-attribute description="Space Name"
79          */
80         public String getSpaceName () {
81             return spaceName;
82         }
83         
84         /**
85          * @jmx:managed-attribute description="Space Keys"
86          */
87         public Set getKeys () {
88             if (sp instanceof LocalSpace) 
89                 return ((LocalSpace)sp).getKeySet ();
90             return null;
91         }
92     }
