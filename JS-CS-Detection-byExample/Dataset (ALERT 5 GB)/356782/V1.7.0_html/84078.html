32     /**
33      * DirPoll operates on a set of directories which defaults to
34      * <ul>

172         /**
173          * @param priorities blank separated list of extensions
174          */
175         public void setPriorities (String priorities) {
176             StringTokenizer st = new StringTokenizer (priorities);
177             Vector v = new Vector();
178             while (st.hasMoreTokens()) {
179                 String ext = st.nextToken();
180                 v.addElement (ext.equals ("*") ? "" : ext);
181             }
182             if (v.size() == 0)
183                 v.addElement ("");
184             synchronized (this) {
185                 prio = v;
186             }
187         }

191         //--------------------------------------- FilenameFilter implementation
192         public boolean accept(File dir, String name) {
193             boolean result;
194             String ext = currentPriority >= 0 ? 
195                 ((String) prio.elementAt(currentPriority)) : null;
196             if (ext != null && !name.endsWith(ext))
197                 return false;
198             File f = new File (dir, name);
199             if (acceptZeroLength){
200                  result = f.isFile();
201             } else {
202                  result = f.isFile() && (f.length() > 0);
203             }
204             return result;
205         }
206         //--------------------------------------------- Runnable implementation
207     
208         public void run() { 
209             Thread.currentThread().setName ("DirPoll-"+basePath);
210             if (prio.size() == 0)
211                 addPriority("");
212             while (!shutdown) {
213                 synchronized (this) {
214                     if (paused) {
215                         try {
216                             wait();
217                             paused = false;
218                         } catch (InterruptedException e) {
219                         }
220                     }
221                 }
222     
223                 try {
224                     File f;
225                     synchronized (this) {
226                         f = scan();
227                     }
228                     if (f != null) {
229                         getPool().execute (new ProcessorRunner (f));
230                         Thread.yield(); // be nice
231                     }
232                     else
233                         Thread.sleep(pollInterval);
234                 } catch (InterruptedException e) { 
235                 } catch (Throwable e) {
236                     Logger.log (new LogEvent (this, "dirpoll", e));
237                     try {
238                         Thread.sleep(pollInterval * 10);    // anti hog
239                     } catch (InterruptedException ex) { }
240                 }
241             }   
242         }
243         public void destroy () {
244             shutdown = true;
245         }

262         private byte[] readRequest (File f) throws IOException {
263             byte[] b = new byte[(int) f.length()];
264             FileInputStream in = new FileInputStream(f);
265             in.read(b);
266             in.close();
267             return b;
268         }
269         private void writeResponse (String requestName, byte[] b) 
270             throws IOException
271         {
272             if (responseSuffix != null) {
273                 int pos = requestName.lastIndexOf ('.');
274                 if (pos > 0)
275                     requestName = requestName.substring (0, pos) + responseSuffix;
276             }
277     
278             File tmp = new File(tmpDir, requestName);
279             FileOutputStream out = new FileOutputStream(tmp);
280             out.write(b);
281             out.close();
282             moveTo (tmp, responseDir);
283         }
284     
285         private File moveTo(File f, File dir) throws IOException {
286             File destination = new File(dir, f.getName());
287             if (!f.renameTo(destination))
288                 throw new IOException("Unable to move"+f.getName());
289             return destination;
290         }
291     
292         private void store(File f, File destinationDirectory) throws IOException {
293             String storedFilename = f.getName();
294             if (shouldTimestampArchive)
295                 storedFilename = f.getName() + "." + new SimpleDateFormat(archiveDateFormat).format(new Date());
296             File destination = new File(destinationDirectory, storedFilename);
297             if (!f.renameTo(destination))
298                 throw new IOException("Unable to archive " + "'" + f.getName() + "' in directory " + destinationDirectory);
299         }
300         
301         private File scan() {
302             for (currentPriority=0; 
303                 currentPriority < prio.size(); currentPriority++)
304             {
305                 String files[] = requestDir.list(this);
306                 if (files != null && files.length > 0)
307                     return new File(requestDir, files[0]);
308             }
309             return null;
310         }
311     
312         private synchronized ThreadPool getPool() {
313             if (pool == null)
314                 pool = new ThreadPool (1, 10);
315             return pool;
316         }

320             /**
321              * @param name request name
322              * @param request request image
323              * @return response (or null)
324              */
325             public byte[] process(String name, byte[] request) 
326                 throws DirPollException;

328         public interface FileProcessor {
329             /**
330              * @param name request File
331              * @throws org.jpos.util.DirPoll.DirPollException on errors
332              */
333             public void process (File name) throws DirPollException;
334         }
335         public class ProcessorRunner implements Runnable {
336             File request;
337             LogEvent logEvent;
338             public ProcessorRunner (File request) throws IOException {
339                 this.request = moveTo (request, runDir);
340                 this.logEvent = null;
341             }
342             public void run() {
343                 LogEvent evt = 
344                     new LogEvent (
345                         DirPoll.this, "dirpoll", request.getName()
346                     );
347                 try {
348                     if (processor == null) 
349                         throw new DirPollException 
350                             ("null processor - nothing to do");
351                     else if (processor instanceof Processor) {
352                         byte[] resp = ((Processor) processor).process (
353                             request.getName(), readRequest (request)
354                         );
355                         if (resp != null) 
356                             writeResponse (request.getName(), resp);
357                     } else if (processor instanceof FileProcessor) 
358                         ((FileProcessor) processor).process (request);
359     
360                     if (shouldArchive) {
361                         store(request, archiveDir);
362                     } else {
363                         if (!request.delete ())
364                             throw new DirPollException 
365                                 ("error: can't unlink request " + request.getName());                    
366                     }
367     
368                 } catch (Throwable e) {
369                     logEvent = evt;
370                     evt.addMessage (e);
371                     try {
372                         store (request, badDir);
373                     } catch (IOException _e) {
374                         evt.addMessage ("Can't move to "+badDir.getPath());
375                         evt.addMessage (_e);
376                     }
377                 } finally {
378                     if (logEvent != null) 
379                         Logger.log (logEvent);
380                 }
381             }
382         }

398         public void pause() {
399             synchronized (this) {
400                 if (!paused) {
401                     paused = true;
402                     // Wake up the run() method from sleeping and tell it to pause
403                     notify();
404                 }
405             }
406         }
407     
408         public void unpause() {
409             synchronized (this) {
410                 if (paused) {
411                     paused = false;
412                     // Wake up the wait()ing thread from being paused
413                     notify();
414                     // The run() method will reset the paused flag
415                 }
416             }
417         }
418         
419         public boolean isPaused() {
420             synchronized (this) {
421                 return paused;
422             }
