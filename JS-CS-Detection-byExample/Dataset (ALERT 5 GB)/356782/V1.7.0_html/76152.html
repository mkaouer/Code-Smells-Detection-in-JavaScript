19     package org.jpos.q2.qbean;
20     
21     import org.jdom.Element;
22     import org.jpos.core.ConfigurationException;
23     import org.jpos.q2.QBeanSupport;
24     import org.jpos.util.NameRegistrar;
25     
26     import java.util.*;
27     
28     public class QBeanFactory extends QBeanSupport implements QBeanFactoryMBean {
29       
30         private static Map beanMap = new WeakHashMap();
31         private Map privateList; // list of beans in the config file
32     
33         public void initService() throws Exception {
34             Element e = getPersist();
35             privateList = new HashMap();
36             List beans = e.getChildren("bean");
37             for (Iterator i = beans.iterator(); i.hasNext();) {
38                 Element bean = (Element) i.next();
39                 String id = bean.getAttributeValue("id");
40                 privateList.put(id, bean);
41             }
42     
43         }
44     
45         public void start() {
46             super.start();
47             NameRegistrar.register("QBeanFactory." + getName(), this);
48         }
49     
50         public void stop() {
51             super.stop();
52             NameRegistrar.unregister("QBeanFactory." + getName());
53         }
54         // Support methods for bean life cycle
55         public void startService() {
56             Iterator keys = privateList.keySet().iterator();
57             while (keys.hasNext()) {
58                 Element bean = (Element) privateList.get(keys.next());
59                 String id = bean.getAttributeValue("id");
60                 String lazy = bean.getAttributeValue("lazy");
61                 // Load the bean only when lazy="false". default value true
62                 if (lazy != null && lazy.equalsIgnoreCase("false")) {
63                     getBean(id);
64                 }
65             }
66         }
67     
68         public void stopService() {
69             Iterator keys = privateList.keySet().iterator();
70             while (keys.hasNext()) {
71                 Element bean = (Element) privateList.get(keys.next());
72                 String id = bean.getAttributeValue("id");// id and key are same
73                 Object beanInstance=beanMap.remove(id);
74                 String stopMethod = bean.getAttributeValue("stop-method");
75                 if(beanInstance!=null && stopMethod!=null){
76                     try{
77                         getFactory().invoke(beanInstance, stopMethod, null);
78                     }catch(Exception e){
79                         log.warn(e);
80                     }
81                 }
82             }
83         }
84     
85         public Object getBean(String id) {
86     
87             // Hack to support a global ref bean across multiple
88             // deployment files. We may need to support lazy loading here but now
89             // not supported
90             Element bean = (Element) privateList.get(id);
91             if (bean == null) {
92                 return beanMap.get(id);
93             }
94             // Need to create a new instance when singleton="false"
95             // It is true by default
96             String singleton = bean.getAttributeValue("singleton");
97            try{ 
98             if(singleton != null && singleton.equals("false")){// create new instance every time
99                 return newBean(bean,false);
100             }else{
101                 return newBean(bean,true); 
102             }
103             }catch(Exception e){
104                 log.fatal(e);
105                 return null;
106             }
107            
108         }
109         private Object newBean(Element bean,boolean useCache) throws ConfigurationException{
110             String id = bean.getAttributeValue("id");
111             
112             if(useCache && beanMap.containsKey(id)){
113                 return beanMap.get(id);
114             }
115             
116             String className = bean.getAttributeValue("class");
117             Object beanInstance = getFactory().newInstance(className);
118             List propertyList = bean.getChildren("property");
119             for (Iterator props = propertyList.iterator(); props.hasNext();) {
120                 Element propertyElement = (Element) props.next();
121                 String pName = propertyElement.getAttributeValue("name");
122                 String pValue = propertyElement.getAttributeValue("value");
123                 String pRef =  propertyElement.getAttributeValue("ref");
124                 String methodName = "set"
125                         + Character.toUpperCase(pName.charAt(0))
126                         + pName.substring(1);
127                 if (pValue == null) {
128                     getFactory()
129                             .invoke(beanInstance, methodName, getBean(pRef));
130                 } else {
131                     getFactory().invoke(beanInstance, methodName, pValue);
132                 }
133             }
134             String startMethod = bean.getAttributeValue("start-method");
135             if(startMethod!=null){
136                 getFactory().invoke(beanInstance, startMethod, null);
137             }
138             if(useCache)//indication for a singleton 
139               beanMap.put(id, beanInstance);   
140             // Return the bean instance
141             return beanInstance;
142         }

144     }
