19     package org.jpos.bsh;
20     
21     import bsh.Interpreter;
22     import org.jdom.Element;
23     import org.jpos.ui.UI;
24     
25     import javax.swing.*;
26     
27     /**
28      * @author Alejandro Revilla
29      *
30      * UI with BSH script support
31      */
32     public class BSHUI extends UI {
33         protected JComponent doScript (JComponent component, Element e) {
34             try {
35                 Interpreter bsh = new Interpreter ();
36                 bsh.set ("component", component);
37                 bsh.set ("config", e);
38                 bsh.set ("log", getLog());
39                 bsh.set ("ui", this);
40                 bsh.eval ("import java.awt.*;");
41                 bsh.eval ("import java.awt.event.*;");
42                 bsh.eval ("import javax.swing.*;");
43                 bsh.eval ("import org.jpos.ui.*;");
44                 bsh.eval (e.getText ());
45                 return (JComponent) bsh.get ("component");
46             } catch (Exception ex) {
47                 warn (ex);
48                 return new JLabel (ex.toString ());
49             }
50         }
51     }
