47     /**
48      * General purpose, Field Separator delimited message.
49      * 
50      * <h1>How to use</h1>
51      * <p>
52      * The message format (or schema) is defined in xml files containing a schema element, with an optional id attribute, and multiple
53      * field elements. A field element is made up of the following attributes:
54      * <dl>
55      * <dt>id</dt>
56      * <dd>The name of the field. This is used in calls to {@link FSDMsg#set(String, String)}. It should be unique amongst the fields in an FSDMsg.</dd>
57      * <dt>length</dt>
58      * <dd>The maximum length of the data allowed in this field. Fixed length fields will be padded to this length. A zero length is allowed, and can
59      * be useful to define extra separator characters in the message.</dd>
60      * <dt>type</dt>
61      * <dd>The type of the included data, including an optional separator for marking the end of the field and the beginning of the next one. The data type
62      * is defined by the first char of the type, and the separator is defined by the following chars. If a field separator is specified, then no
63      * padding is done on values for this field.
64      * </dd>
65      * <dt>key</dt>
66      * <dd>If this optional attribute has a value of "true", then fields from another schema, specified by the value, are appended to this schema.</dd>
67      * <dt>separator</dt>
68      * <dd>An optional attribute containing the separator for the field. This is the preferred method of specifying the separator. See the list of optional</dd>
69      * </dl>
70      * <p>
71      * Possible types are:
72      * <dl>
73      * <dt>A</dt><dd>Alphanumeric. Padding if any is done with spaces to the right.</dd>
74      * <dt>B</dt><dd>Binary. Padding, if any, is done with zeros to the left.</dd>
75      * <dt>K</dt><dd>Constant. The value is specified by the field content. No padding is done.</dd>
76      * <dt>N</dt><dd>Numeric. Padding, if any, is done with zeros to the left.</dd>
77      * </dl>
78      * </p>
79      * <p>
80      * Supported field separators are:
81      * <dl>
82      * <dt>FS</dt><dd>Field separator using '034' as the separator.</dd>
83      * <dt>US</dt><dd>Field separator using '037' as the separator.</dd>
84      * <dt>GS</dt><dd>Group separator using '035' as the separator.</dd>
85      * <dt>RS</dt><dd>Row separator using '036' as the separator.</dd>
86      * <dt>PIPE</dt><dd>Field separator using '|' as the separator.</dd>
87      * <dt>EOF</dt><dd>End of File - no separator character is emitted, but also no padding is done. Also if the end of file is reached
88      * parsing a message, then no exception is thrown.</dd>
89      * <dt>DS</dt><dd>A dummy separator. This is similar to EOF, but the message stream must not end before it is allowed.</dd>
90      * <dt>EOM</dt><dd>End of message separator. This reads all bytes available in the stream.  
91      * </dl>
92      * </p>
93      * <p>
94      * Key fields allow you to specify a tree of possible message formats. The key fields are the fork points of the tree.
95      * Multiple key fields are supported. It is also possible to have more key fields specified in appended schemas.
96      * </p>
97      * 
98      * @author Alejandro Revila
99      * @author Mark Salter
100      * @author Dave Bergert
101      * @since 1.4.7
102      */

230         protected String get (String id, String type, int length, String defValue, String separator) 
231             throws ISOException
232         {
233             String value = (String) fields.get (id);
234             if (value == null)
235                 value = defValue == null ? "" : defValue;
236     
237             type   = type.toUpperCase ();
238     
239             switch (type.charAt (0)) {
240                 case 'N':
241                     if (isSeparated(separator)) {
242                         // Leave value unpadded.
243                     } else {
244                         value = ISOUtil.zeropad (value, length);
245                     }
246                     break;
247                 case 'A':
248                     if (isSeparated(separator)) {
249                         // Leave value unpadded.
250                     } else {
251                         value = ISOUtil.strpad (value, length);
252                     }
253                     if (value.length() > length)
254                         value = value.substring(0,length);
255                     break;
256                 case 'K':
257                     if (defValue != null)
258                         value = defValue;
259                     break;
260                 case 'B':
261                     try {
262                         if ((length << 1) >= value.length()) {
263                             if (isSeparated(separator)) {
264                                 // Convert but do not pad if this field ends with a
265                                 // separator
266                                 value = new String(ISOUtil.hex2byte(value), "ISO8859_1");
267                             } else {
268                                 value = new String(ISOUtil.hex2byte(ISOUtil.zeropad(
269                                         value, length << 1).substring(0, length << 1)),
270                                         "ISO8859_1");
271                             }
272                         } else {
273                             throw new RuntimeException("field content=" + value
274                                     + " is too long to fit in field " + id
275                                     + " whose length is " + length);
276                         }
277                     } catch (UnsupportedEncodingException e) {
278                         // ISO8859_1 is supported
279                     }
280                 break;
281             }
282             return (isSeparated(separator)) ? ISOUtil.blankUnPad(value) : value;
283         }

285         private boolean isSeparated(String separator) {
286             /*
287              * if type's last two characters appear in our Map of separators,
288              * return true
289              */
290             if (separator != null) {
291                 if (separators.containsKey (separator)) {
292                     return true;
293                 } else {
294                     if (isDummySeparator (separator)) { 
295                         return true;
296                     }
297                     
298                     throw new RuntimeException("FSDMsg.isSeparated(String) found that "+separator+" has not been defined as a separator!");
299                 }
300             }
301             return false;
302     
303         }

320         private String getSeparatorType(String type) {
321             if (type.length() > 2) {
322                 return type.substring(1);
323             }
324             return null;
325         }

327         private char getSeparator(String separator) {
328             if (separators.containsKey(separator)) {
329                 return ((Character)separators.get(separator)).charValue();
330             } else {
331                 if (isDummySeparator (separator)) {
332                     // Dummy separator type, return 0 to indicate nothing to add.
333                     return 0;
334                 }
335             }
336             
337             throw new RuntimeException("getSeparator called on separator="+separator+" which does not resolve to a known separator.");
338         }

340         protected void pack (Element schema, StringBuffer sb) 
341             throws JDOMException, MalformedURLException, IOException, ISOException
342         {
343             String keyOff = "";
344             String defaultKey = "";
345             Iterator iter = schema.getChildren("field").iterator();
346             while (iter.hasNext()) {
347                 Element elem = (Element) iter.next ();
348                 String id    = elem.getAttributeValue ("id");
349                 int length   = Integer.parseInt (elem.getAttributeValue ("length"));
350                 String type  = elem.getAttributeValue ("type");
351                 // For backward compatibility, look for a separator at the end of the type attribute, if no separator has been defined.
352                 String separator = elem.getAttributeValue ("separator");
353                 if (type != null && separator == null) {
354                 	separator = getSeparatorType (type);
355                 }
356                 boolean key  = "true".equals (elem.getAttributeValue ("key"));
357                 Map properties = key ? loadProperties(elem) : Collections.EMPTY_MAP;
358                 String defValue = elem.getText();
359                 // If properties were specified, then the defValue contains lots of \n and \t in it. It should just be set to the empty string, or null.
360                 if (!properties.isEmpty()) {
361                 	defValue = defValue.replace("\n", "").replace("\t", "").replace("\r", "");
362                 }
363                 String value = get (id, type, length, defValue, separator);
364                 sb.append (value);
365                 
366                 if (isSeparated(separator)) {
367                     char c = getSeparator(separator);
368                     if (c > 0)
369                         sb.append(c);
370                 }
371                 if (key) {
372                     keyOff = keyOff + normalizeKeyValue(value, properties);
373                     defaultKey += elem.getAttributeValue ("default-key");
374                 }
375             }
376             if (keyOff.length() > 0) 
377                 pack (getSchema (getId (schema), keyOff, defaultKey), sb);
378         }

380         private Map loadProperties(Element elem) {
381         	Map props = new HashMap ();
382         	Iterator iter = elem.getChildren ("property").iterator ();
383         	while (iter.hasNext ()) {
384         		Element prop = (Element) iter.next ();
385         		String name = prop.getAttributeValue ("name");
386         		String value = prop.getAttributeValue ("value");
387         		props.put (name, value);
388         	}
389     	    return props;
390         }

392     	private String normalizeKeyValue(String value, Map properties) {
393         	if (properties.containsKey(value)) {
394         		return (String) properties.get(value);
395         	}
396         	return ISOUtil.normalize(value);
397         }

399         protected void unpack (InputStream is, Element schema) 
400             throws IOException, JDOMException, MalformedURLException  
401         
402         {
403             Iterator iter = schema.getChildren("field").iterator();
404             String keyOff = "";
405             String defaultKey = "";
406             while (iter.hasNext()) {
407                 Element elem = (Element) iter.next();
408     
409                 String id    = elem.getAttributeValue ("id");
410                 int length   = Integer.parseInt (elem.getAttributeValue ("length"));
411                 String type  = elem.getAttributeValue ("type").toUpperCase();
412                 String separator = elem.getAttributeValue ("separator");
413                 if (type != null && separator == null) {
414                 	separator = getSeparatorType (type);
415                 }
416                 boolean key  = "true".equals (elem.getAttributeValue ("key"));
417                 Map properties = key ? loadProperties(elem) : Collections.EMPTY_MAP;
418                 String value = readField (
419                     is, id, length, type, separator );
420                 
421                 if (key) {
422                     keyOff = keyOff + normalizeKeyValue(value, properties);
423                     defaultKey += elem.getAttributeValue ("default-key");
424                 }
425                 if ("K".equals(type) && !value.equals (elem.getText()))
426                     throw new IllegalArgumentException (
427                         "Field "+id 
428                            + " value='"     +value
429                            + "' expected='" + elem.getText () + "'"
430                     );
431             }
432             if (keyOff.length() > 0) {
433                 unpack (is, 
434                     getSchema (getId (schema), keyOff, defaultKey)
435                 );
436             }
437         }

442         protected String read (InputStream is, int len, String type, String separator) 
443             throws IOException 
444         {
445             StringBuffer sb = new StringBuffer();
446             byte[] b = new byte[1];
447             boolean expectSeparator = isSeparated(separator);
448             boolean separated = expectSeparator;
449     
450             if (EOM_SEPARATOR.equals(separator)) {
451             	// Grab what's left. is.available() should work because it is normally a ByteArrayInputStream
452             	byte[] rest = new byte[is.available()];
453             	is.read(rest, 0, rest.length);
454             	for (int i = 0; i < rest.length; i++) {
455             		sb.append((char) (rest[i] & 0xff));
456             	}
457             } else if (isDummySeparator(separator)) {
458                 /*
459                  * No need to look for a seperator, that is not there! Try and take
460                  * len bytes from the is.
461                  */
462                 for (int i = 0; i < len; i++) {
463                     if (is.read(b) < 0) {
464                         break; // end of stream indicates end of field?
465                     }
466                     sb.append((char) (b[0] & 0xff));
467                 }
468             } else {
469     
470                 for (int i = 0; i < len; i++) {
471                     if (is.read(b) < 0) {
472                         if (!"EOF".equals(separator))
473                             throw new EOFException();
474                         else {
475                             separated = false;
476                             break;
477                         }
478                     }
479                     if (expectSeparator && (b[0] == getSeparator(separator))) {
480                         separated = false;
481                         break;
482                     }
483                     sb.append((char) (b[0] & 0xff));
484                 }
485     
486                 if (separated && !"EOF".equals(separator)) {
487                     if (is.read(b) < 0) {
488                         throw new EOFException();
489                     }
490                 }
491             }
492             return sb.toString();
493         }

504             return fieldValue;
505         }

549         public Element toXML () {
550             Element e = new Element ("message");
551             if (header != null) {
552                 e.addContent (
553                     new Element ("header")
554                         .setText (getHexHeader ())
555                 );
556             }
557             Iterator iter = fields.keySet().iterator();
558             while (iter.hasNext()) {
559                 String fieldName = (String) iter.next();
560                 Element inner = new Element (fieldName);
561                 inner.addContent (ISOUtil.normalize ((String) fields.get (fieldName)));
562                 e.addContent (inner);
563             }
564             return e;
565         }

574         protected Element getSchema (String prefix, String suffix, String defSuffix)
575             throws JDOMException, IOException {
576             StringBuilder sb = new StringBuilder (basePath);
577             sb.append (prefix);
578             prefix = sb.toString(); // little hack, we'll reuse later with defSuffix
579             sb.append (suffix);
580             sb.append (".xml");
581             String uri = sb.toString ();
582     
583             Space sp = SpaceFactory.getSpace();
584             Element schema = (Element) sp.rdp (uri);
585             if (schema == null) {
586                 SAXBuilder builder = new SAXBuilder ();
587                 URL url = new URL (uri);
588                 File f = new File(url.getFile());
589                 if (f.exists()) {
590                     schema = builder.build (url).getRootElement ();
591                 } else if (defSuffix != null) {
592                     sb = new StringBuilder (prefix);
593                     sb.append (defSuffix);
594                     sb.append (".xml");
595                     url = new URL (sb.toString());
596                     f = new File (url.getFile());
597                     if (f.exists()) {
598                         schema = builder.build (url).getRootElement ();
599                     }
600                 }
601                 if (schema == null){
602                     throw new RuntimeException(f.getCanonicalPath().toString() + " not found");
603                 }
604                 sp.out (uri, schema);
605             }
606             return schema;
607         }

619         public void dump (PrintStream p, String indent) {
620             String inner = indent + "  ";
621             p.println (indent + "<fsdmsg schema='" + basePath + baseSchema  + "'>");
622             if (header != null) {
623                 append (p, "header", getHexHeader(), inner);
624             }
625             Iterator iter = fields.keySet().iterator();
626             while (iter.hasNext()) {
627                 String f = (String) iter.next();
628                 String v = ((String) fields.get (f));
629                 append (p, f, v, inner);
630             }
631             p.println (indent + "</fsdmsg>");
632         }
