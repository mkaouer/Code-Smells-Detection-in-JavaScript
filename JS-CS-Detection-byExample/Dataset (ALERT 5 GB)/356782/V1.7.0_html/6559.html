19     package org.jpos.transaction;
20     
21     import org.jpos.util.Loggeable;
22     
23     import java.io.PrintStream;
24     import java.util.Iterator;
25     import java.util.List;
26     import java.util.TimerTask;
27     
28     public class PausedTransaction implements Loggeable {
29         private long id;
30         private List members;
31         private Iterator iter;
32         private boolean aborting;
33         private TransactionManager txnmgr;
34         private boolean resumed;
35         private TimerTask expirationMonitor;
36         public PausedTransaction (
37                 TransactionManager txnmgr,
38                 long id, List members, Iterator iter, boolean aborting, TimerTask expirationMonitor) 
39         {
40             super();
41             this.txnmgr = txnmgr;
42             this.id = id;
43             this.members = members;
44             this.iter = iter;
45             this.aborting = aborting;
46             this.expirationMonitor = expirationMonitor;
47         }
48         public long id() {
49             return id;
50         }
51         public List members() {
52             return members;
53         }
54         public Iterator iterator() {
55             return iter;
56         }
57         public void dump (PrintStream p, String indent) {
58             p.println (indent + "id: " + id
59                     + (isAborting() ? " (aborting)" : ""));
60     
61         }
62         public boolean isAborting() {
63             return aborting;
64         }
65         public void forceAbort() {
66             this.aborting = true;
67         }
68         public TransactionManager getTransactionManager() {
69             return txnmgr;
70         }
71         public void setResumed (boolean resumed) {
72             this.resumed = resumed;
73         }
74         public boolean isResumed() {
75             return resumed;
76         }
77         public synchronized void cancelExpirationMonitor() {
78             if (expirationMonitor != null)
79                 expirationMonitor.cancel();
80         }
81     }
