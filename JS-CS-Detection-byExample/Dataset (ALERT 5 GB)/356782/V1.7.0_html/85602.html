19     package org.jpos.ui.factory;
20     
21     import org.jdom.Element;
22     import org.jpos.ui.UI;
23     import org.jpos.ui.UIFactory;
24     
25     import javax.swing.*;
26     import javax.swing.event.ChangeEvent;
27     import javax.swing.event.ChangeListener;
28     import java.awt.*;
29     import java.awt.event.ActionEvent;
30     import java.awt.event.ActionListener;
31     import java.util.ArrayList;
32     import java.util.Iterator;
33     import java.util.List;
34     
35     /**
36      * @author Alejandro Revilla
37      *
38      * creates a tabbed pane
39      * i.e:
40      * <pre>
41      *  &lt;tabbed-pane font="xxx"&gt;
42      *   &lt;pane title="xxx" icon="optional/icon/file.jpg"
43      *            action="yyy" command="zzz"&gt;
44      *   ...
45      *   ...
46      *   &lt;/pane&gt;
47      *  &lt;/tabbed-pane&gt;
48      * </pre>
49      * @see org.jpos.ui.UIFactory
50      */
51     public class JTabbedPaneFactory implements UIFactory, ChangeListener {
52         UI ui;
53         List actions = new ArrayList();
54         JTabbedPane p;
55     
56         public JComponent create (UI ui, Element e) {
57             this.ui    = ui;
58             p = new JTabbedPane ();
59     
60             String font = e.getAttributeValue ("font");
61             if (font != null) 
62                 p.setFont (Font.decode (font));
63     
64             Iterator iter = e.getChildren ("pane").iterator();
65             while (iter.hasNext()) {
66                 add (p, (Element) iter.next ());
67             }
68             p.addChangeListener(this);
69             return p;
70         }
71     
72         private void add (JTabbedPane p, Element e) {
73             if (e != null) {
74                 Icon icon = null;
75                 String iconFile = e.getAttributeValue ("icon");
76                 if (iconFile != null)
77                     icon = new ImageIcon (iconFile);
78                 p.addTab (e.getAttributeValue ("title"), icon, ui.create (e));
79     
80                 String action[] = new String[2];
81                 action[0]=e.getAttributeValue ("command");
82                 action[1]=e.getAttributeValue ("action");
83                 actions.add(action);
84             }
85         }
86     
87         public void stateChanged (ChangeEvent e) {
88             try {
89                 String action[] = new String[2];
90                 action = (String[]) actions.get(p.getSelectedIndex());
91                 Object al = ui.get (action[1]);
92                 if (al instanceof ActionListener) {
93                     ActionEvent ae = new ActionEvent(this,0,action[0]);
94                     ((ActionListener) al).actionPerformed(ae);
95                 }
96             } catch (Exception f) {
97                 f.printStackTrace();
98             }
99         }
100     }
