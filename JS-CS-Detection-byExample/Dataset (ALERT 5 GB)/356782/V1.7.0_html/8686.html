21     import org.jdom.Element;
22     import org.jpos.ui.UI;
23     import org.jpos.ui.UIFactory;
24     
25     import javax.swing.*;
26     import java.awt.*;
27     import java.util.Iterator;
28     
29     /**
30      * @author Alejandro Revilla
31      *
32      * creates a panel with a grid layout
33      * i.e:
34      * <pre>
35      *  &lt;grid rows="yy" columns="xx"&gt;
36      *   &lt;cell&gt;...&lt;/cell&gt;
37      *   &lt;cell&gt;...&lt;/cell&gt;
38      *   &lt;cell&gt;...&lt;/cell&gt;
39      *   ...
40      *   ...
41      *  &lt;/grid&gt;
42      * </pre>
43      * @see org.jpos.ui.UIFactory
44      */
45     public class GridLayoutFactory implements UIFactory {
46         public JComponent create (UI ui, Element e) {
47             JComponent p;
48             try {
49                 int rows = Integer.parseInt (e.getAttributeValue  ("rows", "1"));
50                 int cols = Integer.parseInt (e.getAttributeValue ("columns", "1"));
51                 p = new JPanel (new GridLayout (rows, cols));
52                 Iterator iter = e.getChildren ("cell").iterator ();
53                 while (iter.hasNext())
54                     p.add (ui.create ((Element) iter.next ()));
55             } catch (Exception ex) {
56                 p = new JLabel (ex.getMessage());
57             }
58             return p;
59         }
60     }
