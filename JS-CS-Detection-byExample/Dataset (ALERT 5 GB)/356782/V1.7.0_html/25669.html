19     package org.jpos.iso.packager;
20     
21     import org.jpos.iso.*;
22     import org.jpos.util.LogEvent;
23     import org.jpos.util.Logger;
24     
25     import java.util.BitSet;
26     import java.util.ArrayList;
27     import java.util.List;
28     import java.util.Map;
29     
30     /**
31      * GenericSubFieldPackager
32      * Used to pack composite SubFields from the GenericPackager
33      *
34      * @author Eoin Flood
35      * @see GenericPackager
36      *
37      * This class is basically the same as Base1SubFieldPackager except that it extends
38      * GenericPackager which means that parameters such as emitBitmap, maxvalidField and 
39      * bitmapField can be specified in the GenericPackager xml config file.
40      */
41     
42     public class GenericSubFieldPackager extends GenericPackager 
43     {
44         public GenericSubFieldPackager() throws ISOException
45         {
46             super();
47         }
48     
49         @Override
50         public int unpack (ISOComponent m, byte[] b) throws ISOException 
51         {
52             LogEvent evt = new LogEvent (this, "unpack");
53             try 
54             {
55                 if (m.getComposite() != m) 
56                     throw new ISOException ("Can't call packager on non Composite");
57                 if (logger != null)  // save a few CPU cycle if no logger available
58                     evt.addMessage (ISOUtil.hexString (b));
59     
60                 int consumed=0;
61                 ISOBitMap bitmap = new ISOBitMap (-1);
62     
63                 BitSet bmap = null;
64                 int maxField = fld.length;
65                 if (emitBitMap()) 
66                 {
67                     consumed += getBitMapfieldPackager().unpack(bitmap,b,consumed);
68                     bmap = (BitSet) bitmap.getValue();
69                     m.set (bitmap);
70                     maxField = bmap.size();
71                 }
72                 for (int i=getFirstField(); i<maxField && consumed < b.length; i++) 
73                 {
74                     if (bmap == null || bmap.get(i)) 
75                     {
76                         if (fld[i] != null) {
77                             ISOComponent c = fld[i].createComponent(i);
78                             consumed += fld[i].unpack (c, b, consumed);
79                             m.set(c);
80                         }
81                     }
82                 }
83                 if (b.length != consumed) 
84                 {
85                     evt.addMessage (
86                     "WARNING: unpack len=" +b.length +" consumed=" +consumed);
87                 }
88                 return consumed;
89             } 
90             catch (ISOException e) 
91             {
92                 evt.addMessage (e);
93                 throw e;
94             } 
95             catch (Exception e)
96             {
97                 evt.addMessage (e);
98                 throw new ISOException (e);
99             }
100             finally 
101             {
102                 Logger.log (evt);
103             }
104         }
105     
106         /**
107          * Pack the subfield into a byte array
108          */ 
109     
110         public byte[] pack (ISOComponent m) throws ISOException 
111         {
112             LogEvent evt = new LogEvent (this, "pack");
113             try 
114             {
115                 ISOComponent c;
116                 List<byte[]> l = new ArrayList();
117                 Map fields = m.getChildren();
118                 int len = 0;
119     
120                 if (emitBitMap()) 
121                 {
122                     // BITMAP (-1 in HashTable)
123                     c = (ISOComponent) fields.get (new Integer(-1));
124                     byte[] b = getBitMapfieldPackager().pack(c);
125                     len += b.length;
126                     l.add(b);
127                 }
128     
129                 for (int i=getFirstField(); i<=m.getMaxField(); i++) 
130                 {
131                     c = (ISOComponent) fields.get (new Integer(i));
132                     if (c == null && !emitBitMap())
133                         c = new ISOField (i, "");
134                     if (c != null) {
135                         try 
136                         {
137                             byte[] b = fld[i].pack(c);
138                             len += b.length;
139                             l.add(b);
140                         } 
141                         catch (Exception e) 
142                         {
143                             evt.addMessage ("error packing subfield "+i);
144                             evt.addMessage (c);
145                             evt.addMessage (e);
146                             throw e;
147                         }
148                     }
149                 }
150                 int k = 0;
151                 byte[] d = new byte[len];
152                 for (byte[] b :l) {
153                     System.arraycopy(b, 0, d, k, b.length);
154                     k += b.length;
155                 }
156                 if (logger != null)  // save a few CPU cycle if no logger available
157                     evt.addMessage (ISOUtil.hexString (d));
158                 return d;
159             } 
160             catch (ISOException e) 
161             {
162                 evt.addMessage (e);
163                 throw e;
164             } 
165             catch (Exception e)
166             {
167                 evt.addMessage (e);
168                 throw new ISOException (e);
169             }
170             finally 
171             {
172                 Logger.log(evt);
173             }
174         }
175     }
