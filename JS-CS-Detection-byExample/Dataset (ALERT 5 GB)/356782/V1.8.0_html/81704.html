19     package org.jpos.bsh;
20     
21     import bsh.Interpreter;
22     import org.jpos.core.Configurable;
23     import org.jpos.core.Configuration;
24     import org.jpos.core.ConfigurationException;
25     import org.jpos.iso.ISOMsg;
26     import org.jpos.iso.ISORequestListener;
27     import org.jpos.iso.ISOSource;
28     import org.jpos.util.Log;
29     
30     import java.util.Arrays;
31     import java.util.HashSet;
32     
33     /**
34      * BSHRequestListener - BeanShell based request listener
35      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
36      * @version $Revision$ $Date$
37      */
38     public class BSHRequestListener extends Log 
39         implements ISORequestListener, Configurable
40     {
41         protected static final String MTI_MACRO = "$mti";
42         protected HashSet whitelist;
43         protected String[] bshSource;
44         Configuration cfg;
45         public BSHRequestListener () {
46             super();
47         }
48        /**
49         * @param cfg
50         * <ul>
51         *  <li>whitelist - supported message types (example: "1100,1220")
52         *  <li>source - BSH script(s) to run (can be more than one)
53         * </ul>
54         */
55         public void setConfiguration (Configuration cfg) 
56             throws ConfigurationException
57         {
58             this.cfg = cfg;
59             bshSource = cfg.getAll ("source");
60             String[] mti = cfg.get ("whitelist", "*").split(",");
61             whitelist = new HashSet( Arrays.asList(mti) );
62         }
63     
64         public boolean process (ISOSource source, ISOMsg m) {
65             try{
66                 String mti = m.getMTI ();
67                 if (!whitelist.contains(mti) && !whitelist.contains("*"))
68                     mti = "unsupported";
69     
70                 for (int i=0; i<bshSource.length; i++) {
71                     try {
72                         Interpreter bsh = new Interpreter ();
73                         bsh.set ("source", source);
74                         bsh.set ("message", m);
75                         bsh.set ("log", this);
76                         bsh.set ("cfg", cfg);
77     
78                         int idx = bshSource[i].indexOf(MTI_MACRO);
79                         String script;
80     
81                         if (idx >= 0) {
82                             // replace $mti with the actual value in script file name
83                             script = bshSource[i].substring(0, idx) + mti +
84                                 bshSource[i].substring(idx + MTI_MACRO.length());
85                         } else {
86                             script = bshSource[i];
87                         }
88     
89                         bsh.source (script);
90                     } catch (Exception e) {
91                         warn (e);
92                     }
93                 }
94             }catch (Exception e){
95                 warn(e);
96                 return false;
97             }
98             return true;
99         }
100     }
