40     package org.jpos.q2.qbean;
41     
42     import java.util.Iterator;
43     
44     import org.jdom.Element;
45     import org.jpos.iso.ISOUtil;
46     import org.jpos.iso.ISOUtilCrypto;
47     import org.jpos.q2.QBeanSupport;
48     
49     import bsh.BshClassManager;
50     import bsh.Interpreter;
51     import bsh.UtilEvalError;
52     
53     public class BSH extends QBeanSupport implements Runnable {
54         Interpreter bsh;
55         public void initService() {
56             bsh = new Interpreter ();
57             BshClassManager bcm = bsh.getClassManager();
58             try {
59                 bcm.setClassPath(getServer().getLoader().getURLs());
60             } catch (UtilEvalError e) {
61                 e.printStackTrace();
62             }
63             bcm.setClassLoader(getServer().getLoader());
64         }
65         public void startService() {
66             new Thread (this).start ();
67         }
68         public void run () {
69             Element config = getPersist();
70             try {
71                 bsh.set  ("qbean", this);
72                 bsh.set  ("log", getLog());
73                 Iterator i = config.getChildren("property").iterator();
74                 while (i.hasNext()) {
75                     Element e = (Element)i.next();
76                     bsh.set(e.getAttributeValue("name"),e.getAttributeValue("value"));
77                 }
78                 String code = config.getText();
79                 String source = config.getAttributeValue ("source");
80                 if (!code.trim().equals("")) {
81                     bsh.eval(code);
82                 }
83                 else {
84                     if (source != null) {
85                         bsh.source (source);
86                     }
87                     else {
88                         code=config.getChildText("source");
89                         bsh.eval(code);
90                     }
91                     
92                 }
93             } catch (Throwable e) {
94                 getLog().warn (e);
95             }
96         }
97     }
