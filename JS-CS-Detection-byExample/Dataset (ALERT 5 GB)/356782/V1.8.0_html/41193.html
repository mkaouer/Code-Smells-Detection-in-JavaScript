219         /**
220          * pad to the left
221          * @param s - original string
222          * @param len - desired len
223          * @param c - padding char
224          * @return padded string
225          */
226         public static String padleft(String s, int len, char c)
227             throws ISOException 
228         {
229             s = s.trim();
230             if (s.length() > len)
231                 throw new ISOException("invalid len " +s.length() + "/" +len);
232             StringBuffer d = new StringBuffer (len);
233             int fill = len - s.length();
234             while (fill-- > 0)
235                 d.append (c);
236             d.append(s);
237             return d.toString();
238         }

314         /**
315          * converts to BCD
316          * @param s - the number
317          * @param padLeft - flag indicating left/right padding
318          * @param d The byte array to copy into.
319          * @param offset Where to start copying into.
320          * @return BCD representation of the number

322         public static byte[] str2bcd(String s, boolean padLeft, byte[] d, int offset) {
323             int len = s.length();
324             int start = (((len & 1) == 1) && padLeft) ? 1 : 0;
325             for (int i=start; i < len+start; i++) 
326                 d [offset + (i >> 1)] |= (s.charAt(i-start)-'0') << ((i & 1) == 1 ? 0 : 4);
327             return d;
328         }

347         public static byte[] str2bcd(String s, boolean padLeft, byte fill) {
348             int len = s.length();

353                 d [i >> 1] |= (s.charAt(i-start)-'0') << ((i & 1) == 1 ? 0 : 4);
354             return d;
355         }
356         /**
357          * converts a BCD representation of a number to a String
358          * @param b - BCD representation
359          * @param offset - starting offset
360          * @param len - BCD field len
361          * @param padLeft - was padLeft packed?
362          * @return the String representation of the number
363          */
364         public static String bcd2str(byte[] b, int offset,
365                             int len, boolean padLeft)
366         {
367             StringBuffer d = new StringBuffer(len);
368             int start = (((len & 1) == 1) && padLeft) ? 1 : 0;
369             for (int i=start; i < len+start; i++) {
370                 int shift = ((i & 1) == 1 ? 0 : 4);
371                 char c = Character.forDigit (
372                     ((b[offset+(i>>1)] >> shift) & 0x0F), 16);
373                 if (c == 'd')
374                     c = '=';
375                 d.append (Character.toUpperCase (c));
376             }
377             return d.toString();
378         }

747          * @param s source String
748          * @param canonical true if we want to normalize \r and \n as well
749          * @return normalized string suitable for XML Output
750          */
751         public static String normalize (String s, boolean canonical) {
752             StringBuffer str = new StringBuffer();
753     
754             int len = (s != null) ? s.length() : 0;
755             for (int i = 0; i < len; i++) {
756                 char ch = s.charAt(i);
757                 switch (ch) {
758                     case '<': 
759                         str.append("&lt;");
760                         break;
761                     case '>': 
762                         str.append("&gt;");
763                         break;
764                     case '&': 
765                         str.append("&amp;");
766                         break;
767                     case '"': 
768                         str.append("&quot;");
769                         break;
770                     case '\r':
771                     case '\n': 
772                         if (canonical) {
773                             str.append("&#");
774                             str.append(Integer.toString((int) (ch & 0xFF)));
775                             str.append(';');
776                             break;
777                         }
778                         // else, default append char
779                     default: 
780                         if (ch < 0x20) {
781                             str.append("&#");
782                             str.append(Integer.toString((int) (ch & 0xFF)));
783                             str.append(';');
784                         }
785                         else if (ch > 0xff00) {
786                             str.append((char) (ch & 0xFF));
787                         } else
788                             str.append(ch);
789                 }
790             }
791             return (str.toString());
792         }

801         /**
802          * Protects PAN, Track2, CVC (suitable for logs).
803          *
804          * <pre>
805          * "40000101010001" is converted to "400001____0001"
806          * "40000101010001=020128375" is converted to "400001____0001=0201_____"
807          * "123" is converted to "___"
808          * </pre>
809          * @param s string to be protected 
810          * @return 'protected' String
811          */
812         public static String protect (String s) {
813             StringBuffer sb = new StringBuffer();
814             int len   = s.length();
815             int clear = len > 6 ? 6 : 0;
816             int lastFourIndex = -1;
817             if (clear > 0) {
818                 lastFourIndex = s.indexOf ('=') - 4;
819                 if (lastFourIndex < 0) {
820                     lastFourIndex = s.indexOf ('^') - 4;
821                     if (lastFourIndex < 0) 
822                         lastFourIndex = len - 4;
823                 }
824             }
825             for (int i=0; i<len; i++) {
826                 if (s.charAt(i) == '=')
827                     clear = 1;  // use clear=5 to keep the expiration date
828                 else if (s.charAt(i) == '^') {
829                     lastFourIndex = 0;
830                     clear = len - i;
831                 }
832                 else if (i == lastFourIndex)
833                     clear = 4;
834                 sb.append (clear-- > 0 ? s.charAt(i) : '_');
835             }
836             s = sb.toString();
837             try {
838             //Addresses Track1 Truncation
839                 int charCount = s.replaceAll("[^\\^]", "").length();
840                 if (charCount == 2 ) {
841                     s = s.substring(0, s.lastIndexOf("^")+1);
842                     s = ISOUtil.padright(s, len, '_');
843                 }
844             } catch (ISOException e){
845                 //cannot PAD - should never get here 
846             }
847             return s;        
848         }

1072         public static int parseInt (String s, int radix) throws NumberFormatException {
1073             int length = s.length();
1074             if (length > 9)
1075                 throw new NumberFormatException ("Number can have maximum 9 digits");
1076             int result = 0;
1077             int index = 0;
1078             int digit = Character.digit (s.charAt(index++), radix);
1079             if (digit == -1)
1080                 throw new NumberFormatException ("String contains non-digit");
1081             result = digit;

1087                 result += digit;

1244                         sb.append (' ');
1245                         sb.append (ascii.toString());
1246                         sb.append (lineSep);
1247                         hex   = new StringBuffer ();

1250                 }
1251             }

1253                 while (hex.length () < 49)
1254                     hex.append (' ');
