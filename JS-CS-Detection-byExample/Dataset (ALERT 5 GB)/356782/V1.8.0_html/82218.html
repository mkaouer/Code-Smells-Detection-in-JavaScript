19     package org.jpos.space;
20     
21     /**
22      * Intercepts space operations.
23      *
24      * @author Alejandro Revilla
25      * @since  1.4.7
26      */
27     public class SpaceInterceptor implements Space {
28         protected Space sp;
29         public SpaceInterceptor (Space sp) {
30             super();
31             this.sp = sp;
32         }
33         public void out (Object key, Object value) {
34             sp.out (key, value);
35         }
36         public void out (Object key, Object value, long timeout) {
37             sp.out (key, value, timeout);
38         }
39         public void push (Object key, Object value) {
40             sp.push (key, value);
41         }
42         public void push (Object key, Object value, long timeout) {
43             sp.push (key, value, timeout);
44         }
45         public void put (Object key, Object value) {
46             sp.put (key, value);
47         }
48         public void put (Object key, Object value, long timeout) {
49             sp.put (key, value, timeout);
50         }    
51         public Object in  (Object key) {
52             return sp.in (key);
53         }
54         public Object rd  (Object key) {
55             return sp.rd (key);
56         }
57         public Object in  (Object key, long timeout) {
58             return sp.in (key, timeout);
59         }
60         public Object rd  (Object key, long timeout) {
61             return sp.rd (key, timeout);
62         }
63         public Object inp (Object key) {
64             return sp.inp (key);
65         }
66         public Object rdp (Object key) {
67             return sp.rdp (key);
68         }
69         public boolean existAny (Object[] keys) {
70             return sp.existAny (keys);
71         }
72         public boolean existAny (Object[] keys, long timeout) {
73             return sp.existAny (keys, timeout);
74         }
75     }
