43         public synchronized void out (K key, V value) {
44             if (key == null || value == null)
45                 throw new NullPointerException ("key=" + key + ", value=" + value);
46             getList (key).add (value);
47             this.notifyAll ();
48             if (sl != null)
49                 notifyListeners(key, value);
50         }
51         public void out (K key, V value, long timeout) {
52             if (key == null || value == null)
53                 throw new NullPointerException ("key=" + key + ", value=" + value);
54             Object v = value;
55             if (timeout > 0) {
56                 v = new Expirable (value, System.currentTimeMillis() + timeout);
57             }
58             synchronized (this) {
59                 getList (key).add (v);
60                 this.notifyAll ();
61             }
62             if (sl != null)
63                 notifyListeners(key, value);
64         }
65         public synchronized V rdp (Object key) {
66             if (key instanceof Template)
67                 return (V) getObject ((Template) key, false);
68             return (V) getHead (key, false);
69         }

72                 return (V) getObject ((Template) key, true);

75         public synchronized V in (Object key) {
76             Object obj;
77             while ((obj = inp (key)) == null) {
78                 try {
79                     this.wait ();
80                 } catch (InterruptedException e) { }
81             }
82             return (V) obj;
83         }
84         public synchronized V in  (Object key, long timeout) {
85             Object obj;
86             long now = System.currentTimeMillis();
87             long end = now + timeout;
88             while ((obj = inp (key)) == null && 
89                     ((now = System.currentTimeMillis()) < end))
90             {
91                 try {
92                     this.wait (end - now);
93                 } catch (InterruptedException e) { }
94             }

96         }

119         public void run () {

126         public void gc () {
127             Object[] keys;
128             synchronized (this) {
129                 keys = entries.keySet().toArray();
130             }
131             for (int i=0; i<keys.length; i++) {
132                 rdp (keys[i]);
133                 Thread.yield ();
134             }
135             if (sl != null) {
136                 synchronized (this) {
137                     if (sl != null && sl.isEmpty())
138                         sl = null;
139                 }
140             }
141         }

193         public void notifyListeners (Object key, Object value) {
194             Object[] listeners = null;
195             synchronized (this) {
196                 if (sl == null)
197                     return;
198                 List l = (List) sl.entries.get (key);
199                 if (l != null)
200                     listeners = l.toArray();
201             }
202             if (listeners != null) {
203                 for (int i=0; i<listeners.length; i++) {
204                     Object o = listeners[i];
205                     if (o instanceof Expirable)
206                         o = ((Expirable)o).getValue();
207                     if (o instanceof SpaceListener)
208                         ((SpaceListener) o).notify (key, value);
209                 }
210             }
211         }

270         public boolean existAny (K[] keys, long timeout) {
271             long now = System.currentTimeMillis();
272             long end = now + timeout;
273             while (((now = System.currentTimeMillis()) < end)) {
274                 if (existAny (keys))
275                     return true;
276                 synchronized (this) {
277                     try {
278                         wait (end - now);
279                     } catch (InterruptedException e) { }
280                 }
281             }
282             return false;
283         }

298         private List getList (Object key) {
299             List l = (List) entries.get (key);
300             if (l == null) 
301                 entries.put (key, l = new LinkedList());
302             return l;
303         }
304         private Object getHead (Object key, boolean remove) {
305             Object obj = null;
306             List l = (List) entries.get (key);
307             while (obj == null && l != null && l.size() > 0) {
308                 obj = l.get(0);
309                 if (obj instanceof Expirable) 
310                     obj = ((Expirable) obj).getValue();
311                 if (obj == null) {
312                     l.remove (0);
313                     if (l.size() == 0) {
314                         entries.remove (key);
315                     }
316                 }
317             }
318             if (obj != null && remove) {
319                 l.remove (0);
320                 if (l.size() == 0) {
321                     entries.remove (key);
322                 }
323             }
324             return obj;
325         }
326         private Object getObject (Template tmpl, boolean remove) {
327             Object obj = null;
328             List l = (List) entries.get (tmpl.getKey());
329             if (l == null)
330                 return obj;
331     
332             Iterator iter = l.iterator();
333             while (iter.hasNext()) {
334                 obj = iter.next();
335                 if (obj instanceof Expirable) {
336                     obj = ((Expirable) obj).getValue();
337                     if (obj == null) {
338                         iter.remove();
339                         continue;
340                     }
341                 }
342                 if (tmpl.equals (obj)) {
343                     if (remove)
344                         iter.remove();
345                     break;
346                 } else
347                     obj = null;
348             }
349             return obj;
350         }

358         static class Expirable implements Comparable {
359             Object value;
360             long expires;
361     
362             public Expirable (Object value, long expires) {
363                 super();
364                 this.value = value;
365                 this.expires = expires;
366             }
367             public boolean isExpired () {
368                 return expires < System.currentTimeMillis ();
369             }
370             public String toString() {
371                 return getClass().getName() 
372                     + "@" + Integer.toHexString(hashCode())
373                     + ",value=" + value.toString()
374                     + ",expired=" + isExpired ();
375             }
376             public Object getValue() {
377                 return isExpired() ? null : value;
378             }
379             public int compareTo (Object obj) {
380                 Expirable other = (Expirable) obj;
381                 long otherExpires = other.expires;
382                 if (otherExpires == expires)
383                     return 0;
384                 else if (expires < otherExpires)
385                     return -1;
386                 else 
387                     return 1;
388             }
389         }
