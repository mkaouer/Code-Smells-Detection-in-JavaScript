19     package org.jpos.iso;
20     
21     import junit.framework.TestCase;
22     
23     /**
24      * @author joconnor
25      */
26     public class IFB_LLLBINARYTest extends TestCase
27     {
28         public void testPack() throws Exception
29         {
30             ISOBinaryField field = new ISOBinaryField(12, new byte[] {0x12, 0x34});
31             IFB_LLLBINARY packager = new IFB_LLLBINARY(100, "Should be 1234");
32             TestUtils.assertEquals(new byte[] {0x00, 0x02, 0x12, 0x34}, packager.pack(field));
33         }
34     
35         public void testUnpack() throws Exception
36         {
37             byte[] raw = new byte[] {0x00, 0x02, 0x12, 0x34};
38             IFB_LLLBINARY packager = new IFB_LLLBINARY(100, "Should be 1234");
39             ISOBinaryField field = new ISOBinaryField(12);
40             packager.unpack(field, raw, 0);
41             TestUtils.assertEquals(new byte[] {0x12, 0x34}, (byte[])field.getValue());
42         }
43     
44         public void testReversability() throws Exception
45         {
46             byte[] origin = new byte[] {0x12, 0x34, 0x56, 0x78};
47             ISOBinaryField f = new ISOBinaryField(12, origin);
48             IFB_LLLBINARY packager = new IFB_LLLBINARY(100, "Should be 12345678");
49     
50             ISOBinaryField unpack = new ISOBinaryField(12);
51             packager.unpack(unpack, packager.pack(f), 0);
52             TestUtils.assertEquals(origin, (byte[])unpack.getValue());
53         }
54         
55         public void testPackGreaterThan16() throws Exception
56         {
57             ISOBinaryField field = new ISOBinaryField(1, new byte[] {0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07 ,0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f, 0x10, 0x11});
58             IFB_LLLBINARY packager = new IFB_LLLBINARY(1, "Should be 1234");
59             TestUtils.assertEquals(new byte[] {0x00, 0x17, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07 ,0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f, 0x10, 0x11}, packager.pack(field));
60         }
61     
62         public void testUnpackGreaterThan16() throws Exception
63         {
64             byte[] raw = new byte[] {0x00, 0x17, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07 ,0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f, 0x10, 0x11};
65             IFB_LLLBINARY packager = new IFB_LLLBINARY(1, "Should be 17 bytes 01 through 11");
66             ISOBinaryField field = new ISOBinaryField(1);
67             packager.unpack(field, raw, 0);
68             TestUtils.assertEquals(new byte[] {0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07 ,0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f, 0x10, 0x11}, (byte[])field.getValue());
69         }
70     
71         public void testReversabilityGreaterThan16() throws Exception
72         {
73             byte[] origin = new byte[] {0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07 ,0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f, 0x10, 0x11};
74             ISOBinaryField f = new ISOBinaryField(1, origin);
75             IFB_LLLBINARY packager = new IFB_LLLBINARY(1, "Should be x'0102030405060708090a0b0c0d0e0f1011'");
76     
77             ISOBinaryField unpack = new ISOBinaryField(12);
78             packager.unpack(unpack, packager.pack(f), 0);
79             TestUtils.assertEquals(origin, (byte[])unpack.getValue());
80         }
