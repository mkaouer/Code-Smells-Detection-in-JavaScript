63         public JESpace(String name, String path) throws SpaceError {
64             super();
65             try {
66                 EnvironmentConfig envConfig = new EnvironmentConfig();
67                 StoreConfig storeConfig = new StoreConfig();
68     
69                 envConfig.setAllowCreate (true);
70                 envConfig.setTransactional (true);
71                 envConfig.setTxnTimeout(30000000); // 30 seconds
72                 storeConfig.setAllowCreate (true);
73                 storeConfig.setTransactional (true);
74     
75                 File dir = new File(path);
76                 dir.mkdirs();
77     
78                 dbe = new Environment (dir, envConfig);
79                 store = new EntityStore (dbe, name, storeConfig);
80                 pIndex = store.getPrimaryIndex (Long.class, Ref.class);
81                 gcpIndex = store.getPrimaryIndex (Long.class, GCRef.class);
82                 sIndex = store.getSecondaryIndex (pIndex, String.class, "key");
83                 gcsIndex = store.getSecondaryIndex (gcpIndex, Long.class, "expires");
84                 scheduleGC();
85             } catch (Exception e) {
86                 throw new SpaceError (e);
87             }
88         }

93         public void out (K key, V value, long timeout) {
94             Transaction txn = null;
95             try {
96                 txn = dbe.beginTransaction (null, null);
97                 Ref ref = new Ref(key.toString(), value, timeout);
98                 pIndex.put (ref);
99                 if (timeout > 0L)
100                     gcpIndex.putNoReturn (
101                         new GCRef (ref.getId(), ref.getExpiration())
102                     );
103                 txn.commit();
104                 txn = null;
105                 synchronized (this) {
106                     notifyAll ();
107                 }
108             } catch (Exception e) {
109                 throw new SpaceError (e);
110             } finally {
111                 if (txn != null)
112                     abort (txn);
113             }
114         }
115         public void push (K key, V value, long timeout) {
116             Transaction txn = null;
117             try {
118                 txn = dbe.beginTransaction (null, null);
119                 Ref ref = new Ref(key.toString(), value, timeout);
120                 pIndex.put (ref);
121                 pIndex.delete (ref.getId());
122                 ref.reverseId();
123                 pIndex.put (ref);
124                 txn.commit();
125                 txn = null;
126                 synchronized (this) {
127                     notifyAll ();
128                 }
129             } catch (Exception e) {
130                 throw new SpaceError (e);
131             } finally {
132                 if (txn != null)

134             }
135         }

239         public void gc () throws DatabaseException {
240             Transaction txn = null;
241             EntityCursor<GCRef> cursor = null;
242             try {
243                 if (!gcSem.tryAcquire())
244                     return;
245                 txn = dbe.beginTransaction (null, null);
246                 cursor = gcsIndex.entities (
247                     txn, 0L, true, System.currentTimeMillis(), false, null
248                 );
249                 for (GCRef gcRef: cursor) {
250                     pIndex.delete (gcRef.getId());
251                     cursor.delete ();
252                 }
253                 cursor.close();
254                 cursor = null;
255                 txn.commit();
256                 txn = null;
257             } finally {
258                 if (cursor != null)
259                     cursor.close();
260                 if (txn != null)
261                     abort (txn);
262                 gcSem.release();
263             }
264         }
265         public void scheduleGC() {
266             gcTimerTask  = new TimerTask() {
267                 public void run() {
268                     try {
269                         gc();
270                     } catch (DatabaseException e) {
271                         warn (e);
272                     }
273                 }
274             };
275             DefaultTimer.getTimer().schedule(gcTimerTask, GC_DELAY, GC_DELAY);
276         }
277         public void close () throws DatabaseException {
278             gcSem.acquireUninterruptibly();
279             gcTimerTask.cancel();
280             store.close ();
281             dbe.close();
282         }

285         {
286             JESpace sp = (JESpace) spaceRegistrar.get (name);

296         private Object getObject (Object key, boolean remove) throws DatabaseException {
297             Transaction txn = null;
298             EntityCursor<Ref> cursor = null;
299             Template tmpl = null;
300             if (key instanceof Template) {
301                 tmpl = (Template) key;
302                 key  = tmpl.getKey();
303             }
304             try {
305                 txn = dbe.beginTransaction (null, null);
306                 cursor = sIndex.subIndex(key.toString()).entities(txn, null);
307                 for (Ref ref : cursor) {
308                     if (ref.isActive()) {
309                         if (tmpl != null && !tmpl.equals (ref.getValue()))
310                             continue;
311                         if (remove) {
312                             cursor.delete();
313                             if (ref.hasExpiration()) 
314                                 gcpIndex.delete (txn, ref.getId());
315                         }
316                         cursor.close(); cursor = null;
317                         txn.commit(); txn = null;
318                         return ref.getValue();
319                     }
320                     else {
321                         cursor.delete();
322                         if (ref.hasExpiration()) 
323                             gcpIndex.delete (txn, ref.getId());
324                     }
325                 }
326                 cursor.close(); cursor = null;
327                 txn.commit(); txn = null;
328                 return null;
329             } finally {
330                 if (cursor != null)
331                     cursor.close ();
332                 if (txn != null)
333                     txn.abort();
334             }
335         }
336         private void abort (Transaction txn) throws SpaceError {
337             try {
338                 txn.abort();
339             } catch (DatabaseException e) {
340                 throw new SpaceError (e);
341             }
342         }

344         @Entity
345         public static class Ref {
346             @PrimaryKey(sequence="Ref.id")
347             private long id;
348     
349             @SecondaryKey(relate= Relationship.MANY_TO_ONE)
350             private String key;
351     
352             private long expires;
353             private Object value;
354     
355             public Ref () {
356                 super();
357             }
358             public Ref (String key, Object value, long timeout) {
359                 this.key = key;
360                 this.value =  serialize (value);
361                 if (timeout > 0L)
362                     this.expires = System.currentTimeMillis() + timeout;
363             }
364             public long getId() {
365                 return id;
366             }
367             public void reverseId() {
368                 this.id = -this.id;
369             }
370             public boolean isExpired () {
371                 return expires > 0L && expires < System.currentTimeMillis ();
372             }
373             public boolean isActive () {
374                 return !isExpired();
375             }
376             public Object getKey () {
377                 return key;
378             }
379             public Object getValue () {
380                 return deserialize(value);
381             }
382             public long getExpiration () {
383                 return expires;
384             }
385             public boolean hasExpiration () {
386                 return expires > 0L;
387             }
388             private boolean isPersistent (Class c) {
389                 return
390                     c.isPrimitive() ||
391                     c.isAnnotationPresent(Entity.class) ||
392                     c.isAnnotationPresent(Persistent.class);
393             }
394             private Object serialize (Object obj) {
395                 Class cls = obj.getClass();
396                 if (isPersistent (cls))
397                     return obj;
398     
399                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
400                 try {
401                     ObjectOutputStream os = new ObjectOutputStream(baos);
402                     os.writeObject(obj);
403                     obj = baos.toByteArray();
404                 } catch (IOException e) {
405                     throw new SpaceError (e);
406                 }
407                 return obj;
408             }
409             private Object deserialize (Object obj) {
410                 Class cls = obj.getClass();
411                 if (isPersistent (cls))
412                     return obj;
413     
414                 ByteArrayInputStream bais = new ByteArrayInputStream((byte[]) obj);
415                 try {
416                     ObjectInputStream is = new ObjectInputStream (bais);
417                     return is.readObject();
418                 } catch (Exception e) {
419                     throw new SpaceError (e);
420                 }
421     
422             }
423         }

425         public void dump(PrintStream p, String indent) {
426             Transaction txn = null;
427             EntityCursor<Ref> cursor = null;
428             int count = 0;
429             try {
430                 txn = dbe.beginTransaction (null, null);
431                 cursor = sIndex.entities(txn, null);
432                 String key = null;
433                 int keyCount = 0;
434                 for (Ref ref : cursor) {
435                     if (ref.getKey().equals(key)) {
436                         keyCount++;
437                     } else {
438                         if (key != null) {
439                             dumpKey (p, indent, key, keyCount);
440                             count++;
441                         }
442                         keyCount = 1;
443                         key = ref.getKey().toString();
444                     }
445                 }
446                 if (key != null) {
447                     dumpKey (p, indent, key, keyCount);
448                     count++;
449                 }
450                 p.println(indent+"<keycount>"+count+"</keycount>");
451                 cursor.close(); cursor = null;
452                 txn.commit(); txn = null;
453             } catch (IllegalStateException e) {
454                 //Empty Cursor
455                 p.println(indent+"<keycount>0</keycount>");
456             } finally {
457                 if (cursor != null)
458                     cursor.close ();
459                 if (txn != null)
460                     txn.abort();
461             }
462         }
463     
464         private void dumpKey (PrintStream p, String indent, String key, int count) {
465             if (count > 0)
466                 p.printf ("%s<key size='%d'>%s</key>\n", indent, count, key);
467             else
468                 p.printf ("%s<key>%s</key>\n", indent, key);
469         }
