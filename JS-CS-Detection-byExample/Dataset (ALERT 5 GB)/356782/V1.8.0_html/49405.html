19     package org.jpos.util;
20     
21     /**
22      * @author Alejandro P. Revilla
23      * @version $Revision$ $Date$
24      */
25     public class Asserter {
26         public Asserter() {
27             super();
28         }
29         public Asserter check (Object obj) throws AssertFailedException {
30             if (obj == null)
31                 throw new AssertFailedException();
32             return this;
33         }
34         public Asserter check (Object obj1, Object obj2) 
35             throws AssertFailedException
36         {
37             check(obj1).check(obj2);
38             if (!obj1.equals(obj2))
39                 throw new AssertFailedException();
40             return this;
41         }
42         public Asserter check (boolean b) throws AssertFailedException {
43             if (!b)
44                 throw new AssertFailedException();
45             return this;
46         }
47     }
