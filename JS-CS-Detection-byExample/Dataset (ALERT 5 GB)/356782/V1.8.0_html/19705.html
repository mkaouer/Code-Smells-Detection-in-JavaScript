19     package org.jpos.util;
20     
21     import java.util.Iterator;
22     
23     /**
24      * A specific log listener that filters all LogEvents that doesn't
25      * contain any exception.
26      * @author Alejandro Revilla
27      * @version $Revision$ $Date$
28      */
29     public class ExceptionLogFilter implements LogListener {
30         public ExceptionLogFilter () {
31             super();
32         }
33         public synchronized LogEvent log (LogEvent evt) {
34             Iterator iter = evt.getPayLoad().iterator();
35             while (iter.hasNext()) {
36                 if (iter.next() instanceof Throwable)
37                     return evt;
38             }
39             return null;
40         }
41     }
