19     package org.jpos.transaction.participant;
20     
21     import bsh.EvalError;
22     import bsh.Interpreter;
23     import org.jdom.Element;
24     
25     import java.io.FileNotFoundException;
26     import java.io.FileReader;
27     import java.io.IOException;
28     import java.util.Collection;
29     import java.util.HashMap;
30     import java.util.Iterator;
31     import java.util.Map;

33     /** This is a utility class that makes it a bit easier to work with beanshell 
34      *  scripts.
35      *  Check out the execute methods.
36      *
37      * @author  AMarques
38      */
39     public class BSHMethod {
40         
41         private String bshData;
42         private boolean source;
43         
44         
45         /** Creates a BSHMethod from a JDom Element.
46          *
47          *  The element is not requiered to have a specific name. 
48          *  
49          *  If the given element defines an attribute named 'file' the file 
50          *  specified as its value will be sourced by the created BSHMethod.
51          *
52          *  If the 'file' attribute is specified, a 'cache'
53          *  attribute may be specified as well which can take the values true|false
54          *  and indicates wether to load the script to memory or to read from the 
55          *  file for every script evaluation.
56          *
57          *  If the 'file' attibute is not specified then the text contained by the 
58          *  element is set to be evaluated by the new BSHMethod. 
59          *  <pre>
60          *  Example 1 : 
61          *          &lt;prepare>
62          *                  import org.jpos.iso.*;
63          *  		import org.jpos.transaction.*;
64          *
65          *			msg = context.get("txnRequest");
66          *			BaseChannel.getChannel("loop-channel").send(msg);
67          *			result=TransactionConstants.PREPARED | TransactionConstants.READONLY;
68          *		&lt;/prepare>
69          *
70          *  Example 2 :
71          *          &lt;routing file='cfg\files\routing1.bsh' cache='false'/>
72          *  </pre>
73          */ 
74         public static BSHMethod createBshMethod(Element e) throws FileNotFoundException, IOException {
75             if (e == null) {
76                 return null;
77             }
78             String file = e.getAttributeValue("file");
79             String bsh;
80             if (file != null) {
81                 boolean cache = false;
82                 String cacheAtt = e.getAttributeValue("cache");
83                 if (cacheAtt != null) {
84                     cache = cacheAtt.equalsIgnoreCase("true"); 
85                 }
86                 if (!cache) {
87                     return new BSHMethod(file, true);
88                 } else {
89                     bsh = "";
90                     FileReader f = new FileReader(file);
91                     int c;
92                     while ( (c = f.read()) != -1) {
93                         bsh += (char) c; 
94                     }
95                     f.close();
96                     return new BSHMethod(bsh, false);
97                 }
98             } else {
99                 bsh = e.getTextTrim();
100                 if (bsh == null || bsh.equals("")) {
101                     return null;
102                 }
103                 return new BSHMethod(bsh, false);
104             }
105         }
106             
107         /** Creates a BSHMethod.
108          *  @param bshData - May either be the file to source or the script itself to
109          *                  evaluate.
110          *  @param source - If true indicates that the bshData passed is a file to 
111          *                  source. Otherwise the string itself is evaluated.
112          */
113         public BSHMethod(String bshData, boolean source) {
114             this.bshData = bshData;
115             this.source = source; 
116         }
117     
118         /** Sets the given arguments to the Interpreter, evaluates the script and 
119          *  returns the object stored on the variable named resultName.
120          *
121          *  @param arguments    Parameters to set to the Interpreter. For every 
122          *                      Map.Entry (key, value), interpreter.set(key, value)
123          *                      is called. All keys must be Strings.
124          *  @param returnName   The names of the variables wich`s content is to be 
125          *                      returned.
126          */
127         public Object execute(Map arguments, String resultName) throws EvalError, FileNotFoundException, IOException {
128             Interpreter i = initInterpreter(arguments);
129             return i.get(resultName);
130         }
131         
132         /** Sets the given arguments to the Interpreter, evaluates the script and 
133          *  returns a map that has the Strings of the returnNames collection as keys
134          *  and the objects stored in the variables thus named as values.
135          *  
136          *  @param arguments    Parameters to set to the Interpreter. For every 
137          *                      Map.Entry (key, value), interpreter.set(key, value)
138          *                      is called. All keys must be Strings.
139          *  @param returnNames  Collection of Strings. The names of the variables 
140          *                      wich`s contents are to be returned.
141          */
142         public Map execute(Map arguments, Collection returnNames) throws EvalError, FileNotFoundException, IOException {
143             Interpreter i = initInterpreter(arguments);
144             Map result = new HashMap();
145             String rName;
146             for (Iterator it = returnNames.iterator(); it.hasNext(); ) {
147                 rName = (String) it.next();
148                 result.put(rName, i.get(rName));
149             }
150             return result;
151         }
152         
153         protected Interpreter initInterpreter(Map arguments) throws EvalError, FileNotFoundException, IOException {
154             Interpreter i = new Interpreter();
155             Map.Entry entry;
156             for (Iterator it = arguments.entrySet().iterator(); it.hasNext(); ) {
157                 entry = (Map.Entry) it.next();
158                 i.set((String) entry.getKey(), entry.getValue());
159             }
160             if (source) {
161                 i.source(bshData);
162             } else {
163                 i.eval(bshData);
164             }
165             return i;
166         }
167     
168         public String toString() {
169             return bshData;
170         }
171     }
