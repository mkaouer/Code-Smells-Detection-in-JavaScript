19     package org.jpos.ui.factory;
20     
21     import org.jdom.Element;
22     import org.jpos.ui.UI;
23     import org.jpos.ui.UIFactory;
24     
25     import javax.swing.*;
26     import javax.swing.event.HyperlinkEvent;
27     import javax.swing.event.HyperlinkListener;
28     import javax.swing.text.html.HTMLDocument;
29     import javax.swing.text.html.HTMLFrameHyperlinkEvent;
30     
31     /**
32      * @author Alejandro Revilla
33      *
34      * Creates an html browser/editor
35      * i.e:
36      * <pre>
37      *  &lt;html editable="false" follow-links="true" scrollable="true"&gt;
38      *    http://jpos.org
39      *  &lt;/html&gt;
40      * </pre>
41      * @see org.jpos.ui.UIFactory
42      */
43     public class HtmlFactory implements UIFactory {
44         public JComponent create (UI ui, Element e) {
45             try {
46                 JEditorPane editorPane = new JEditorPane (e.getText());
47                 editorPane.setEditable (
48                     "true".equals (e.getAttributeValue ("editable"))
49                 );
50                 if ("true".equals (e.getAttributeValue ("follow-links")))
51                     editorPane.addHyperlinkListener (new Listener ());
52                 return editorPane;
53             } catch (Exception ex) {
54                 return new JLabel (ex.getMessage());
55             }
56         }
57         class Listener implements HyperlinkListener {
58             public void hyperlinkUpdate(HyperlinkEvent e) {
59                 if (e.getEventType() == HyperlinkEvent.EventType.ACTIVATED) {
60                     JEditorPane pane = (JEditorPane) e.getSource();
61                     if (e instanceof HTMLFrameHyperlinkEvent) {
62                         HTMLFrameHyperlinkEvent  evt = (HTMLFrameHyperlinkEvent)e;
63                         HTMLDocument doc = (HTMLDocument)pane.getDocument();
64                         doc.processHTMLFrameHyperlinkEvent(evt);
65                     } else {
66                         try {
67                             pane.setPage(e.getURL());
68                         } catch (Throwable t) {
69                             t.printStackTrace();
70                         }
71                     }
72                 }
73             }
74         }
75     }
