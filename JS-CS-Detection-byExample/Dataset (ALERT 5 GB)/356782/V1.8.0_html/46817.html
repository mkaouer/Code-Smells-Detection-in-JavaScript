19     package org.jpos.iso.channel;
20     
21     import org.jpos.iso.*;
22     import org.jpos.util.LogEvent;
23     import org.jpos.util.Logger;
24     
25     import java.io.IOException;
26     import java.net.ServerSocket;
27     
28     /**
29      * Talks with TCP based NCCs
30      * Sends [LEN][TPDU][ISOMSG]
31      * (len=2 bytes HEX)
32      *
33      * @author Mladen Mrkic <mmrkic@arius.co.yu>
34      * @version $Revision: 2706 $ $Date: 2009-03-05 11:24:43 +0000 (Thu, 05 Mar 2009) $
35      * @see ISOMsg
36      * @see ISOException
37      * @see ISOChannel
38      */
39     public class BCDChannel extends BaseChannel {
40         /**
41          * Public constructor 
42          */
43         public BCDChannel () {
44             super();
45         }
46         /**
47          * Construct client ISOChannel
48          * @param host  server TCP Address
49          * @param port  server port number
50          * @param p     an ISOPackager
51          * @param TPDU  an optional raw header (i.e. TPDU)
52          * @see ISOPackager
53          */
54         public BCDChannel (String host, int port, ISOPackager p, byte[] TPDU) {
55             super(host, port, p);
56             this.header = TPDU;
57         }
58         /**
59          * Construct server ISOChannel
60          * @param p     an ISOPackager
61          * @param TPDU  an optional raw header (i.e. TPDU)
62          * @exception IOException
63          * @see ISOPackager
64          */
65         public BCDChannel (ISOPackager p, byte[] TPDU) throws IOException {
66             super(p);
67             this.header = TPDU;
68         }
69         /**
70          * constructs server ISOChannel associated with a Server Socket
71          * @param p     an ISOPackager
72          * @param TPDU  an optional raw header (i.e. TPDU)
73          * @param serverSocket where to accept a connection
74          * @exception IOException
75          * @see ISOPackager
76          */
77         public BCDChannel (ISOPackager p, byte[] TPDU, ServerSocket serverSocket) 
78             throws IOException
79         {
80             super(p, serverSocket);
81             this.header = TPDU;
82         }
83         protected void sendMessageLength(int len) throws IOException {
84             try {
85                 serverOut.write (                                                                                                         
86                     ISOUtil.str2bcd (                                                                                                     
87                         ISOUtil.zeropad (Integer.toString (len), 4), true
88                     )
89                 );
90             } 
91             catch (ISOException e) {
92                 Logger.log (new LogEvent (this, "send-message-length", e));
93             }
94         }
95         protected int getMessageLength() throws IOException, ISOException {
96             byte[] b = new byte[2];
97             serverIn.readFully(b,0,2);
98             return Integer.parseInt (
99                 ISOUtil.bcd2str (b, 0, 4, true)
100             );
101         }
102         protected void sendMessageHeader(ISOMsg m, int len) throws IOException { 
103             byte[] h = m.getHeader();
104             if (h != null) {
105                 if (h.length == 5) {
106                     // swap src/dest address
107                     byte[] tmp = new byte[2];
108                     System.arraycopy (h,   1, tmp, 0, 2);
109                     System.arraycopy (h,   3,   h, 1, 2);
110                     System.arraycopy (tmp, 0,   h, 3, 2);
111                 }
112             }
113             else
114                 h = header ;
115             if (h != null) 
116                 serverOut.write(h);
117         }
118         /**
119          * New QSP compatible signature (see QSP's ConfigChannel)
120          * @param header String as seen by QSP
121          */
122         public void setHeader (String header) {
123             super.setHeader (ISOUtil.str2bcd(header, false));
124         }
125     }
