19     package org.jpos.iso;
20     
21     import java.util.LinkedList;
22     import java.util.ListIterator;
23     
24     /**
25      * Wrapper class resulting from process of validating an ISOField
26      * instance. Contains details of the original field and validation-error
27      * details too. Normally in validation process when an error is detected
28      * by validator in some field, then the field is replaced by an instance
29      * of this class, containning error details.
30      * <p>Title: jPOS</p>
31      * <p>Description: Java Framework for Financial Systems</p>
32      * <p>Copyright: Copyright (c) 2000 jPOS.org.  All rights reserved.</p>
33      * <p>Company: www.jPOS.org</p>
34      * @author Jose Eduardo Leon
35      * @version 1.0
36      */
37     public class ISOVField extends ISOField implements ISOVErrorList {
38     
39         private static final long serialVersionUID = -2503711799295775875L;
40     
41         /**
42          * Creates the vfield.
43          * @param Source original field instance.
44          */
45         public ISOVField( ISOField Source ) {
46             super();
47             this.fieldNumber = Source.fieldNumber;
48             this.value = Source.value;
49         }
50     
51         public ISOVField( ISOField Source, ISOVError FirstError ) {
52             this( Source );
53             this.errors.addLast( FirstError );
54         }
55     
56         public boolean addISOVError(ISOVError Error) {
57             return errors.add( Error );
58         }
59     
60         public ListIterator errorListIterator() {
61             return errors.listIterator();
62         }
63     
64         /** list of errors **/
65         protected LinkedList errors = new LinkedList(  );
66     }
