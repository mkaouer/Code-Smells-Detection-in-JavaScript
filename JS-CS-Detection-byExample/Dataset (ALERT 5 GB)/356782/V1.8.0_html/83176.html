19     package org.jpos.iso;
20     
21     import java.io.IOException;
22     import java.io.InputStream;
23     import java.util.BitSet;
24     
25     /**
26      * ASCII packaged Bitmap
27      *
28      * @author apr@cs.com.uy
29      * @version $Id$
30      * @see ISOComponent
31      * @see ISOBitMapPackager
32      */
33     public class IFA_BITMAP extends ISOBitMapPackager {
34         public IFA_BITMAP() {
35             super();
36         }
37         /**
38          * @param len - field len
39          * @param description symbolic descrption
40          */
41         public IFA_BITMAP(int len, String description) {
42             super(len, description);
43         }
44         /**
45          * @param c - a component
46          * @return packed component
47          * @exception ISOException
48          */
49         public byte[] pack (ISOComponent c) throws ISOException {
50             byte[] b = ISOUtil.bitSet2byte ((BitSet) c.getValue());
51             return ISOUtil.hexString(b).getBytes();
52         }
53     
54         public int getMaxPackedLength() {
55             return getLength() >> 2;
56         }
57         /**
58          * @param c - the Component to unpack
59          * @param b - binary image
60          * @param offset - starting offset within the binary image
61          * @return consumed bytes
62          * @exception ISOException
63          */
64         public int unpack (ISOComponent c, byte[] b, int offset)
65             throws ISOException
66         {
67             int len;
68             BitSet bmap = ISOUtil.hex2BitSet (b, offset, getLength() << 3);
69             c.setValue(bmap);
70             len = (bmap.get(1) == true) ? 128 : 64; /* changed by Hani */
71             if (getLength() > 16 && bmap.get(65))
72                 len = 192;
73             return (len >> 2);
74         }
75         public void unpack (ISOComponent c, InputStream in) 
76             throws IOException, ISOException
77         {
78             BitSet bmap = ISOUtil.hex2BitSet (new BitSet (64), readBytes (in, 16), 0);
79             if (getLength() > 8 && bmap.get (1)) {
80                 ISOUtil.hex2BitSet (bmap, readBytes (in, 16), 64);
81             }
82             c.setValue(bmap);
83         }
84     }
