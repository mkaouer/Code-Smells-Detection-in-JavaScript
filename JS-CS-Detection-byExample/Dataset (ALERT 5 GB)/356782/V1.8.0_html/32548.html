19     package org.jpos.iso;
20     
21     import org.jpos.util.LogEvent;
22     
23     /**
24      * An ISOFilter has the oportunity to modify an incoming or
25      * outgoing ISOMsg that is about to go thru an ISOChannel.
26      * It also has the chance to Veto by throwing an Exception
27      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
28      * @version $Revision$ $Date$
29      */
30     public interface ISOFilter {
31         public class VetoException extends ISOException {
32     
33             private static final long serialVersionUID = -4640160572663583113L;
34             public VetoException () {
35                 super();
36             }
37             public VetoException (String detail) {
38                 super(detail);
39             }
40             public VetoException (Exception nested) {
41                 super(nested);
42             }
43             public VetoException (String detail, Exception nested) {
44                 super(detail, nested);
45             }
46         }
47         /**
48          * @param channel current ISOChannel instance
49          * @param m ISOMsg to filter
50          * @param evt LogEvent
51          * @return an ISOMsg (possibly parameter m)
52          * @throws VetoException
53          */
54         public ISOMsg filter (ISOChannel channel, ISOMsg m, LogEvent evt) 
55             throws VetoException;
56     }
