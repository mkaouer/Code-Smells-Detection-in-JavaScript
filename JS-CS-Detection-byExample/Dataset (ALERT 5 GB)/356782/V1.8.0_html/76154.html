52         /**
53          * protected constructor.
54          * @param name Space Name
55          * @param filename underlying JDBM filename
56          * @see SpaceFactory().getSpace()
57          */
58         protected JDBMSpace (String name, String filename) {
59             super();
60             this.name = name;
61             try {
62                 Properties props = new Properties();
63                 props.put (RecordManagerOptions.CACHE_SIZE, "512");
64                 recman = RecordManagerFactory.createRecordManager (filename, props);
65                 long recid = recman.getNamedObject ("space");
66                 if (recid != 0) {
67                     htree = HTree.load (recman, recid);
68                 } else {
69                     htree = HTree.createInstance (recman);
70                     recman.setNamedObject ("space", htree.getRecid());
71                 }
72                 recman.commit ();
73             } catch (IOException e) {
74                 throw new SpaceError (e);
75             }
76             DefaultTimer.getTimer().schedule (this, GCDELAY, GCDELAY);
77         }

176         /**
177          * Write a new entry into the Space
178          * The entry will timeout after the specified period
179          * @param key Entry's key
180          * @param value Object value
181          * @param timeout entry timeout in millis
182          */
183         public void out (K key, V value, long timeout) {
184             if (key == null || value == null)
185                 throw new NullPointerException ("key=" + key + ", value=" + value);
186             try {
187                 synchronized (this) {
188                     long recid = recman.insert (value);
189     
190                     long expiration = timeout == -1 ? Long.MAX_VALUE :
191                         (System.currentTimeMillis() + timeout);
192                     Ref dataRef = new Ref (recid, expiration);
193                     long dataRefRecId = recman.insert (dataRef, refSerializer);
194     
195                     Head head = (Head) htree.get (key);
196                     if (head == null) {
197                         head = new Head ();
198                         head.first = dataRefRecId;
199                         head.last  = dataRefRecId;
200                         head.count = 1;
201                     } else {
202                         long previousLast = head.last;
203                         Ref lastRef   = 
204                             (Ref) recman.fetch (previousLast, refSerializer);
205                         lastRef.next      = dataRefRecId;
206                         head.last         = dataRefRecId;
207                         head.count++;
208                         recman.update (previousLast, lastRef, refSerializer);
209                     }
210                     htree.put (key, head);
211                     if (autoCommit) {
212                         recman.commit ();
213                         this.notifyAll ();
214                     }
215                 }
216             } catch (IOException e) {
217                 throw new SpaceError (e);
218             }
219         }

223         /**
224          * Write a new entry into the Space at the head of a queue
225          * The entry will timeout after the specified period
226          * @param key Entry's key
227          * @param value Object value
228          * @param timeout entry timeout in millis
229          */
230         public void push (Object key, Object value, long timeout) {
231             if (key == null || value == null)
232                 throw new NullPointerException ("key=" + key + ", value=" + value);
233             try {
234                 synchronized (this) {
235                     long recid = recman.insert (value);
236                     long expiration = timeout == -1 ? Long.MAX_VALUE :
237                         (System.currentTimeMillis() + timeout);
238                     Ref dataRef = new Ref (recid, expiration);
239     
240                     Head head = (Head) htree.get (key);
241                     if (head == null) {
242                         head = new Head ();
243                         head.first = head.last = recman.insert (dataRef, refSerializer);
244                     } else {
245                         dataRef.next = head.first;
246                         head.first   = recman.insert (dataRef, refSerializer);
247                     }
248                     head.count++;
249                     htree.put (key, head);
250                     if (autoCommit) {
251                         recman.commit ();
252                         this.notifyAll ();
253                     }
254                 }
255             } catch (IOException e) {
256                 throw new SpaceError (e);
257             }
258         }

260          * Read probe reads an entry from the space if one exists, 
261          * return null otherwise.
262          * @param key Entry's key
263          * @return value or null
264          */
265         public synchronized V rdp (Object key) {
266             try {
267                 if (key instanceof Template) 
268                     return (V) getObject ((Template) key, false);
269     
270                 Object obj = null;
271                 Ref ref = getFirst (key, false);
272                 if (ref != null) 
273                     obj = recman.fetch (ref.recid);
274                 if (autoCommit)
275                     recman.commit ();
276                 return (V) obj;
277             } catch (IOException e) {
278                 throw new SpaceError (e);
279             }
280         }

295                 if (ref != null) {

297                     recman.delete (ref.recid);
298                 }
299                 if (autoCommit)
300                     recman.commit ();
301                 return (V) obj;
302             } catch (IOException e) {
303                 throw new SpaceError (e);

306         public synchronized V in (Object key) {
307             Object obj;
308             while ((obj = inp (key)) == null) {
309                 try {
310                     this.wait ();
311                 } catch (InterruptedException ignored) { }
312             }

321             Object obj;
322             long now = System.currentTimeMillis();
323             long end = now + timeout;
324             while ((obj = inp (key)) == null && 
325                     ((now = System.currentTimeMillis()) < end))

327                 try {
328                     this.wait (end - now);

388         public boolean existAny (Object[] keys, long timeout) {
389             long now = System.currentTimeMillis();
390             long end = now + timeout;
391             while (((now = System.currentTimeMillis()) < end)) {
392                 if (existAny (keys))
393                     return true;
394                 synchronized (this) {
395                     try {

412         private void purge (Object key) throws IOException {
413             Head head = (Head) htree.get (key);
414             Ref previousRef = null;
415             if (head != null) {
416                 for (long recid = head.first; recid >= 0; ) {
417                     Ref r = (Ref) recman.fetch (recid, refSerializer);
418                     if (r.isExpired ()) {
419                         recman.delete (r.recid);
420                         recman.delete (recid);
421                         head.count--;
422                         if (previousRef == null) {
423                             head.first = r.next;
424                         } else {
425                             previousRef.next = r.next;
426                             recman.update (
427                                 head.last, previousRef, refSerializer
428                             );
429                         }
430                     } else {
431                         previousRef   = r;
432                         head.last     = recid;
433                     }
434                     recid = r.next;
435                 }
436                 if (head.first == -1)  {
437                     htree.remove (key);
438                 }
439                 else {
440                     htree.put (key, head);
441                 }
442             }
443         }

451         /**
452          * garbage collector.
453          * removes expired entries
454          */
455         public void gc () {
456             final String GCKEY = "GC$" + Integer.toString (hashCode());
457             final long TIMEOUT = 24 * 3600 * 1000;
458             Object obj;
459             try {
460                 synchronized (this) {
461                     // avoid concurrent gc
462                     if (rdp (GCKEY) != null) 
463                         return;
464                     ((Space)this).out (GCKEY, Boolean.TRUE, TIMEOUT);  
465                 }
466                 FastIterator iter = htree.keys ();
467     
468                 try {
469                     while ( (obj = iter.next()) != null) {
470                         ((Space)this).out (GCKEY, obj, TIMEOUT);
471                         Thread.yield ();
472                     }
473                 } catch (ConcurrentModificationException e) {
474                     // ignore, we may have better luck on next try
475                 }
476                 while ( (obj = inp (GCKEY)) != null) {
477                     synchronized (this) {
478                         purge (obj);
479                         recman.commit ();
480                     }
481                     Thread.yield ();
482                 }    
483             } catch (IOException e) {
484                 throw new SpaceError (e);
485             }
486         }

503         private Ref getFirst (Object key, boolean remove) throws IOException {
504             Head head = (Head) htree.get (key);
505             Ref ref = null;
506             if (head != null) {
507                 long recid;
508                 for (recid = head.first; recid >= 0; ) {
509                     Ref r = (Ref) recman.fetch (recid, refSerializer);
510                     if (r.isExpired ()) {
511                         recman.delete (r.recid);
512                         recman.delete (recid);
513                         recid = r.next;
514                         head.count--;
515                     } else  {
516                         ref = r;
517                         if (remove) {
518                             recman.delete (recid);
519                             recid = ref.next;
520                             head.count--;
521                         }
522                         break;
523                     }
524                 } 
525                 if (head.first != recid) {
526                     if (recid < 0)
527                         htree.remove (key);
528                     else {
529                         head.first = recid;
530                         htree.put (key, head);
531                     }
532                 }
533             }
534             return ref;
535         }
536         private void unlinkRef
537             (long recid, Head head, Ref r, Ref previousRef, long previousRecId) 
538             throws IOException
539         {
540             recman.delete (r.recid);
541             recman.delete (recid);
542             head.count--;
543             if (previousRef == null)
544                 head.first = r.next;
545             else {
546                 previousRef.next = r.next;
547                 recman.update (
548                     previousRecId, previousRef, refSerializer
549                 );
550             }
551         }
552         private Object getObject (Template tmpl, boolean remove) 
553             throws IOException 
554         {
555             Object obj = null;
556             Object key = tmpl.getKey();
557             Head head = (Head) htree.get (key);
558             Ref previousRef = null;
559             long previousRecId = 0;
560             int unlinkCount = 0;
561             if (head != null) {
562                 for (long recid = head.first; recid >= 0; ) {
563                     Ref r = (Ref) recman.fetch (recid, refSerializer);
564                     if (r.isExpired ()) {
565                         unlinkRef (recid, head, r, previousRef, previousRecId);
566                         unlinkCount++;
567                     } else  {
568                         Object o = recman.fetch (r.recid);
569                         if (o != null) {
570                             if (tmpl.equals (o)) {
571                                 obj = o;
572                                 if (remove) {
573                                     unlinkRef (
574                                         recid, head, r, previousRef, previousRecId
575                                     );
576                                     unlinkCount++;
577                                 }
578                                 break;
579                             }
580                         }
581                         previousRef = r;
582                         previousRecId = recid;
583                     }
584                     recid = r.next;
585                 } 
586                 if (unlinkCount > 0) {
587                     if (head.first == -1)  {
588                         htree.remove (key);
589                     }
590                     else {
591                         htree.put (key, head);
592                     }
593                 }
594             }
595             return obj;
596         }
597         static class Head implements Externalizable {
598             public long first;
599             public long last;
600             public long count;
601             static final long serialVersionUID = 2L;
602     
603             public Head () {
604                 super ();
605                 first = -1;
606                 last  = -1;
607             }
608             public void writeExternal (ObjectOutput out) throws IOException {
609                 out.writeLong (first);
610                 out.writeLong (last);
611                 out.writeLong (count);
612             }
613             public void readExternal (ObjectInput in) throws IOException {
614                 first = in.readLong ();
615                 last  = in.readLong ();
616                 count = in.readLong ();
617             }
618             public String toString() {
619                 return getClass().getName() 
620                     + "@" + Integer.toHexString(hashCode())
621                     + ":[first=" + first 
622                     + ",last=" + last
623                     + "]";
624             }

626         static class Ref implements Serializer {
627             long recid;
628             long expires;
629             long next;
630             static final long serialVersionUID = 1L;
631     
632             public Ref () {
633                 super();
634             }
635             public Ref (long recid, long expires) {
636                 super();
637                 this.recid   = recid;
638                 this.expires = expires;
639                 this.next    = -1;
640             }
641     
642             public boolean isExpired () {
643                 return expires < System.currentTimeMillis ();
644             }
645             public String toString() {
646                 return getClass().getName() 
647                     + "@" + Integer.toHexString(hashCode())
648                     + ":[recid=" + recid
649                     + ",next=" + next
650                     + ",expired=" + isExpired ()
651                     + "]";
652             }
653             public byte[] serialize (Object obj) 
654                 throws IOException
655             {
656                 Ref d = (Ref) obj;
657     
658                 byte[] buf = new byte [24];
659                 putLong (buf, 0, d.recid);
660                 putLong (buf, 8, d.next);
661                 putLong (buf,16, d.expires);
662                 return buf;
663             }
664             public Object deserialize (byte[] serialized) 
665                 throws IOException
666             {
667                 Ref d = new Ref ();
668                 d.recid   = getLong (serialized,  0);
669                 d.next    = getLong (serialized,  8);
670                 d.expires = getLong (serialized, 16);
671                 return d;
672             }
673         }
674         static void putLong (byte[] b, int off, long val) {
675             b[off+7] = (byte) (val);
676             b[off+6] = (byte) (val >>>  8);
677             b[off+5] = (byte) (val >>> 16);
678             b[off+4] = (byte) (val >>> 24);
679             b[off+3] = (byte) (val >>> 32);
680             b[off+2] = (byte) (val >>> 40);
681             b[off+1] = (byte) (val >>> 48);
682             b[off] = (byte) (val >>> 56);
683         }
684         static long getLong (byte[] b, int off) {
685             return (b[off+7] & 0xFFL) +
686                    ((b[off+6] & 0xFFL) << 8) +
687                    ((b[off+5] & 0xFFL) << 16) +
688                    ((b[off+4] & 0xFFL) << 24) +
689                    ((b[off+3] & 0xFFL) << 32) +
690                    ((b[off+2] & 0xFFL) << 40) +
691                    ((b[off+1] & 0xFFL) << 48) +
692                    ((b[off] & 0xFFL) << 56);
693         }
