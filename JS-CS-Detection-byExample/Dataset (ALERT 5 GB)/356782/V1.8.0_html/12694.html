19     package org.jpos.iso;
20     
21     import junit.framework.TestCase;
22     
23     /**
24      * @author joconnor
25      * @author Mark Salter
26      */
27     public class IFEPE_LLCHARTest extends TestCase
28     {
29         public void testPack() throws Exception
30         {
31             ISOField field = new ISOField(92, "123");
32             IFEPE_LLCHAR packager = new IFEPE_LLCHAR(92, "Should be 0592123");
33             TestUtils.assertEquals(new byte[] {(byte)0xF9, (byte)0xF2,(byte)0xf0, (byte)0xF3,  (byte)0xF1, (byte)0xF2, (byte)0xF3},
34                                 packager.pack(field));
35         }
36     
37         public void testUnpack() throws Exception
38         {
39             byte[] raw = new byte[] {(byte)0xF9, (byte)0xF2,(byte)0xF0, (byte)0xF3,  (byte)0xF1, (byte)0xF2, (byte)0xF3};
40             IFEPE_LLCHAR packager = new IFEPE_LLCHAR(10, "Should be 0592123");
41             ISOField field = new ISOField();
42             int len = packager.unpack(field, raw, 0);
43             assertEquals(raw.length, len);
44             String v = (String)field.getValue();
45             assertEquals("123", (String) field.getValue());
46             assertEquals(92, field.fieldNumber);    // Derived from TAG!
47         }
48     
49         public void testReversability() throws Exception
50         {
51             String origin = "Abc123:.-";
52             ISOField f = new ISOField(92, origin);
53             IFEPE_LLCHAR packager = new IFEPE_LLCHAR(92, "Should be Abc123:.-");
54     
55             ISOField unpack = new ISOField();
56             int len = packager.unpack(unpack, packager.pack(f), 0);
57             assertEquals(origin.length()+2+2, len);
58             assertEquals(origin, (String) unpack.getValue());
59             assertEquals(92,unpack.fieldNumber);    // Derived from TAG!
60         }
61         
62         public void testReversability2() throws Exception
63         {
64             String origin = "P";
65             ISOField f = new ISOField(87, origin);
66             IFEPE_LLCHAR packager = new IFEPE_LLCHAR(87, "Should be P");
67     
68             ISOField unpack = new ISOField();
69             byte[] packed = packager.pack(f);
70             int len = packager.unpack(unpack, packed , 0);
71             assertEquals(origin.length()+2+2, len);
72             assertEquals(origin, unpack.getValue());
73             assertEquals(87,unpack.fieldNumber);    // Derived from TAG!
74         }
75     }
