19     package org.jpos.q2.qbean;
20     
21     import org.jpos.q2.QBeanSupport;
22     
23     /**
24     * QBean for starting and stopping scripts or programs.
25     * <pre>
26     * Example xml:
27     *  &lt;exec class="org.jpos.q2.qbean.QExec"&gt;
28     *    &lt;attr name="start"&gt;YOUR PATH TO PROGRAM&lt;/attr&gt;
29     *    &lt;attr name="shutdown"&gt;YOUR PATH TO PROGRAM&lt;/attr&gt;
30     *  &lt;/exec&gt;
31     * </pre>
32     * @author Alwyn Schoeman
33     * @version $Revision$ $Date$
34     * @jmx:mbean description="QExec QBean" extends="org.jpos.q2.QBeanSupportMBean"
35     */ 
36     
37     public class QExec extends QBeanSupport implements QExecMBean {
38         String startScript;
39         String shutdownScript;
40     
41         public void initService () throws Exception {
42         }
43     
44         public void startService () throws Exception {
45             Runtime.getRuntime().exec (startScript);
46         }
47     
48         public void stopService () throws Exception {
49             Runtime.getRuntime().exec (shutdownScript);
50         }
51     
52         /**
53          * @jmx:managed-attribute description="Program startup script"
54          */
55         public void setStartScript (String scriptPath) {
56             startScript = scriptPath;
57         }
58     
59         /**
60          * @jmx:managed-attribute description="Program startup script"
61          */
62         public String getStartScript () {
63             return startScript;
64         }
65     
66         /**
67          * @jmx:managed-attribute description="Program shutdown script"
68          */
69         public void setShutdownScript (String scriptPath) {
70             shutdownScript = scriptPath;
71         }
72     
73         /**
74          * @jmx:managed-attribute description="Program shutdown script"
75          */
76         public String getShutdownScript () {
77             return shutdownScript;
78         }
79     }
