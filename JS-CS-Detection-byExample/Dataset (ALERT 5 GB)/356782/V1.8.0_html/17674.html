19     package org.jpos.iso.packager;
20     
21     import org.jpos.iso.*;
22     import org.jpos.util.LogEvent;
23     import org.jpos.util.Logger;
24     
25     import java.util.BitSet;
26     import java.util.ArrayList;
27     import java.util.List;
28     import java.util.Map;
29     
30     /**
31      * ISO 8583 v1987 BINARY Packager 
32      * customized for VISA Base1 subfields
33      * 
34      *
35      * @author <a href="mailto:eoin.flood@orbiscom.com">Eoin Flood</a>
36      * @version $Id$
37      * @see ISOPackager
38      * @see ISOBasePackager
39      * @see ISOComponent
40      */
41     
42     public class Base1SubFieldPackager extends ISOBasePackager 
43     {
44         // These methods are identical to ISOBasePackager
45         // except that fld[1] has been replaced with fld[0]
46         // and a secondard bitmap is not allowed
47     
48         protected boolean emitBitMap()
49         {
50             return (fld[0] instanceof ISOBitMapPackager);
51         }
52     
53         protected int getFirstField()
54         {
55             return (fld[0] instanceof ISOBitMapPackager) ? 1 : 0;
56         }
57     
58         protected ISOFieldPackager getBitMapfieldPackager() 
59         {
60             return fld[0];
61         }
62     
63         /**
64          * Unpack a packed subfield into
65          * its corresponding ISOComponent
66          */
67     
68         public int unpack (ISOComponent m, byte[] b) throws ISOException 
69         {
70             LogEvent evt = new LogEvent (this, "unpack");
71             try 
72             {
73                 if (m.getComposite() != m) 
74                     throw new ISOException ("Can't call packager on non Composite");
75                 if (logger != null)  // save a few CPU cycle if no logger available
76                     evt.addMessage (ISOUtil.hexString (b));
77     
78                 int consumed=0;
79                 ISOBitMap bitmap = new ISOBitMap (-1);
80     
81                 BitSet bmap = null;
82                 int maxField = fld.length;
83                 if (emitBitMap()) 
84                 {
85                     consumed += getBitMapfieldPackager().unpack(bitmap,b,consumed);
86                     bmap = (BitSet) bitmap.getValue();
87                     m.set (bitmap);
88                     maxField = bmap.size();
89                 }
90                 for (int i=getFirstField(); i<maxField && consumed < b.length; i++) 
91                 {
92                     if (bmap == null || bmap.get(i)) 
93                     {
94                         ISOComponent c = fld[i].createComponent(i);
95                         consumed += fld[i].unpack (c, b, consumed);
96                         m.set(c);
97                     }
98                 }
99                 if (b.length != consumed) 
100                 {
101                     evt.addMessage (
102                     "WARNING: unpack len=" +b.length +" consumed=" +consumed);
103                 }
104                 return consumed;
105             } 
106             catch (ISOException e) 
107             {
108                 evt.addMessage (e);
109                 throw e;
110             } 
111             finally 
112             {
113                 Logger.log (evt);
114             }
115         }
116     
117         /**
118          * Pack the subfield into a byte array
119          */ 
120     
121         public byte[] pack (ISOComponent m) throws ISOException 
122         {
123             LogEvent evt = new LogEvent (this, "pack");
124             try 
125             {
126                 ISOComponent c;
127                 List<byte[]> l = new ArrayList();
128                 Map fields = m.getChildren();
129                 int len = 0;
130     
131                 if (emitBitMap()) 
132                 {
133                     // BITMAP (-1 in HashTable)
134                     c = (ISOComponent) fields.get (new Integer(-1));
135                     byte[] b = getBitMapfieldPackager().pack(c);
136                     len += b.length;
137                     l.add(b);
138                 }
139     
140                 for (int i=getFirstField(); i<=m.getMaxField(); i++) 
141                 {
142                     if ((c = (ISOComponent) fields.get (new Integer(i))) != null)
143                     {
144                         try 
145                         {
146                             byte[] b = fld[i].pack(c);
147                             len += b.length;
148                             l.add(b);
149                         } 
150                         catch (Exception e) 
151                         {
152                             evt.addMessage ("error packing field "+i);
153                             evt.addMessage (c);
154                             evt.addMessage (e);
155                         }
156                     }
157                 }
158                 int k = 0;
159                 byte[] d = new byte[len];
160                 for (byte[] b :l) {
161                     System.arraycopy(b, 0, d, k, b.length);
162                     k += b.length;
163                 }
164                 if (logger != null)  // save a few CPU cycle if no logger available
165                     evt.addMessage (ISOUtil.hexString (d));
166                 return d;
167             } 
168             catch (ISOException e) 
169             {
170                 evt.addMessage (e);
171                 throw e;
172             } 
173             finally 
174             {
175                 Logger.log(evt);
176             }
177         }
178     }
