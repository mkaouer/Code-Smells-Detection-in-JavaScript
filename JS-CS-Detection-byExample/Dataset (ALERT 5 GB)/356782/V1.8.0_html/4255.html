33     public class IVA_ALPHANUMNOZERO_NOBLANK extends IVA_ALPHANUMNOZERO {
34     
35         public IVA_ALPHANUMNOZERO_NOBLANK() {
36             super();
37         }
38     
39         public IVA_ALPHANUMNOZERO_NOBLANK( String Description ) {
40             super( Description );
41         }
42     
43         public IVA_ALPHANUMNOZERO_NOBLANK( int minLen, int maxLen, String Description ) {
44             super( minLen, maxLen, Description );
45         }
46     
47         public IVA_ALPHANUMNOZERO_NOBLANK( int maxLen, String Description ) {
48             super( maxLen, Description );
49         }
50     
51         public IVA_ALPHANUMNOZERO_NOBLANK( boolean breakOnError, String Description ) {
52             this( Description );
53             this.breakOnError = breakOnError;
54         }
55     
56         public IVA_ALPHANUMNOZERO_NOBLANK( boolean breakOnError, int maxLen, String Description ) {
57             this( maxLen, Description );
58             this.breakOnError = breakOnError;
59         }
60     
61         public IVA_ALPHANUMNOZERO_NOBLANK( boolean breakOnError, int minLen, int maxLen, String Description ) {
62             this( minLen, maxLen, Description );
63             this.breakOnError = breakOnError;
64         }
65     
66         /**
67          * Validate that the component is not blank-filled.
68          */
69         public ISOComponent validate ( ISOComponent f ) throws ISOException {
70             ISOField c = (ISOField)f;
71             try {
72                 /** no zero... **/
73                 c = (ISOField)super.validate( c );
74                 /** no blank **/
75                 if ( ISOUtil.isBlank( (String)c.getValue() ) ){
76                     ISOVError e = new ISOVError( "Invalid Value Error. It can not be blank-filled. (Current value: "+
77                             (String)c.getValue()+") ", getRejCode( ISOVError.ERR_INVALID_VALUE ) );
78                     if ( c instanceof ISOVField )
79                         ((ISOVField)c).addISOVError( e );
80                     else
81                         c = new ISOVField( c, e );
82                     if ( breakOnError )
83                         throw new ISOVException ( "Error on field " + ((Integer)c.getKey()).intValue(), c );
84                 }
85                 return c;
86             }
87             catch (Exception ex) {
88                 if ( ex instanceof ISOVException ) throw (ISOVException)ex;
89                 return c;
90             }
91         }
92     }
