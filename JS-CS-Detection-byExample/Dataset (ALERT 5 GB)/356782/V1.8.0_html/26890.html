26     /**
27      * Tiny Space implementation
28      *
29      * @author Alejandro Revilla
30      * @version $Revision$ $Date$
31      * @since 1.4.7
32      * @deprecated org.jpos.space.TSpace is the new default lightweight space
33      */
34     public class TinySpace implements Space, Serializable {
35         protected Map map = new HashMap ();
36         private static final long serialVersionUID = -5216796586015661708L;
37     
38         public TinySpace () {
39             super();
40         }
41         public void out (Object key, Object value) {
42             synchronized (this) {
43                 Object v = map.get (key);
44                 if (v == null)
45                     map.put (key, value);
46                 else if (v instanceof Data)
47                     ((Data)v).add (value);
48                 else {
49                     Data data = new Data();
50                     data.add (v);
51                     data.add (value);
52                     map.put (key, data);
53                 }
54                 this.notifyAll ();
55             }
56         }
57         public void out (Object id, Object value, long timeout) {
58             out (id, new LeasedReference (value, timeout));
59         }
60         public synchronized Object rdp (Object key) {
61             Object obj = map.get (key);
62             if (obj instanceof Data) 
63                 obj = ((Data) obj).get (key);
64             else if (obj instanceof LeasedReference) {
65                 obj = ((LeasedReference)obj).get ();
66             }
67             return obj;
68         }
69         public synchronized Object inp (Object key) {
70             Object obj = map.get (key);
71             if (obj instanceof Data) {
72                 Data data = (Data) obj;
73                 obj = data.remove ();
74                 if (data.isEmpty ())
75                     map.remove (key);
76             } else if (obj != null) {
77                 if (obj instanceof LeasedReference) {
78                     obj = ((LeasedReference)obj).get ();
79                 }
80                 map.remove (key);
81             }
82     
83             return obj;
84         }
85         public synchronized Object in (Object key) {
86             Object obj;
87             while ((obj = inp (key)) == null) {
88                 try {
89                     this.wait ();
90                 } catch (InterruptedException e) { }
91             }
92             return obj;
93         }
94         public synchronized Object in  (Object key, long timeout) {
95             Object obj;
96             long now = System.currentTimeMillis();
97             long end = now + timeout;
98             while ((obj = inp (key)) == null && 
99                     ((now = System.currentTimeMillis()) < end))
100             {
101                 try {
102                     this.wait (end - now);
103                 } catch (InterruptedException e) { }
104             }
105             return obj;
106         }
107         public synchronized Object rd  (Object key) {
108             Object obj;
109             while ((obj = rdp (key)) == null) {
110                 try {
111                     this.wait ();
112                 } catch (InterruptedException e) { }
113             }
114             return obj;
115         }
116         public synchronized Object rd  (Object key, long timeout) {
117             Object obj;
118             long now = System.currentTimeMillis();
119             long end = now + timeout;
120             while ((obj = rdp (key)) == null && 
121                     ((now = System.currentTimeMillis()) < end))
122             {
123                 try {
124                     this.wait (end - now);
125                 } catch (InterruptedException e) { }
126             }
127             return obj;
128         }
129         public boolean existAny (Object[] keys) {
130             for (int i=0; i<keys.length; i++) {
131                 if (rdp (keys[i]) != null)
132                     return true;
133             }
134             return false;
135         }
136         public boolean existAny (Object[] keys, long timeout) {
137             long now = System.currentTimeMillis();
138             long end = now + timeout;
139             while (((now = System.currentTimeMillis()) < end)) {
140                 if (existAny (keys))
141                     return true;
142                 synchronized (this) {
143                     try {
144                         wait (end - now);
145                     } catch (InterruptedException e) { }
146                 }
147             }
148             return false;
149         }
150         public void push (Object id, Object value) {
151             throw new SpaceError ("Unsupported operation");
152         }
153         public void push (Object id, Object value, long timeout) {
154             throw new SpaceError ("Unsupported operation");
155         }
156         public void put (Object key, Object value) {
157             throw new SpaceError ("Unsupported operation");
158         }
159         public void put (Object key, Object value, long timeout) {
160             throw new SpaceError ("Unsupported operation");
161         }
162         protected static final class Data {
163             LinkedList data;
164     
165             protected Data () {
166                 super();
167                 data = new LinkedList ();
168             }
169             protected Data (Object value) {
170                 this ();
171                 add (value);
172             }
173             protected void add (Object value) {
174                 data.add (value);
175             }
176             protected Object get (Object value) {
177                 Object obj = null;
178                 while (size() > 0) {
179                     obj = data.getFirst();
180                     if (obj instanceof LeasedReference) {
181                         obj = ((LeasedReference)obj).get ();
182                         if (obj == null) {
183                             data.removeFirst ();
184                             continue;
185                         }
186                     }
187                     break;
188                 }
189                 return obj;
190             }
191             protected int size () {
192                 return data.size ();
193             }
194             protected Object remove () {
195                 Object obj = null;
196                 while (size() > 0) {
197                     obj = data.removeFirst();
198                     if (obj instanceof LeasedReference) {
199                         LeasedReference ref = (LeasedReference) obj;
200                         obj = ref.get ();
201                         if (obj == null) {
202                             continue;
203                         }
204                         ref.discard ();
205                     }
206                     break;
207                 }
208                 return obj;
209             }
210             protected boolean isEmpty () {
211                 return data.isEmpty ();
212             }
213         }
214     }
