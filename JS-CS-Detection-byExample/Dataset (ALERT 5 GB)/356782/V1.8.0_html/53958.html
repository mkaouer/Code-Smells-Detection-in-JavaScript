52         public void test02ByteBitmap() throws Exception
53         {
54     
55             ISOComponent c = new ISOBitMap(1);
56             int consumed = twoBytes.unpack(c, inBytes,0);
57             assertEquals("2 characters should be consumed irrespective of 2nd, 3rd or any bitmap indicators",2,consumed);
58             assertEquals("2 characters can only result in a bitmap holding fields up to 8",8,((BitSet)c.getValue()).length() - 1);
59     
60             outBytes = twoBytes.pack(c);
61             assertEquals("Pack of two bytes must produce a result 2 bytes long",2,outBytes.length);
62             assertEquals("2 byte bitmap pack should reflect unpack", ISOUtil.hexString(inBytes,0,2),ISOUtil.hexString(outBytes));
63         }

65         public void test16ByteBitmap() throws Exception
66         {
67     
68             ISOComponent c = new ISOBitMap(1);
69             int consumed = sixteenBytes.unpack(c, inBytes,0);
70             assertEquals("16 characters should be consumed irrespective of 2nd and 3rd bitmap indicators",16,consumed);
71             assertEquals("16 characters can only result in a bitmap holding fields up to 64",64,((BitSet)c.getValue()).length() - 1);
72     
73             outBytes = sixteenBytes.pack(c);
74             assertEquals("Pack of sixteen bytes must produce a result 16 bytes long",16,outBytes.length);
75             assertEquals("16 byte bitmap pack should reflect unpack", ISOUtil.hexString(inBytes,0,16),ISOUtil.hexString(outBytes));
76             
77             try {
78                 outBytes = twoBytes.pack(c);
79                 fail("Pack of bitmap with fields outside of 2 byte range should result in ISOException");
80             } catch (Exception e) {

83             }
