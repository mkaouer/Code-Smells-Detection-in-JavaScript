19     package org.jpos.q2.cli;
20     
21     import org.jpos.q2.CLI;
22     
23     public class SLEEP implements CLI.Command {
24         public void exec (CLI cli, String[] args) throws Exception {
25             if (args.length > 1) {
26                 Thread.sleep (Long.parseLong (args[1])*1000);
27             } else {
28                 cli.println ("Usage: sleep number-of-seconds");
29             }
30         }
31     }
