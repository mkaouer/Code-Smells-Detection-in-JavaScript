19     package org.jpos.space;
20     
21     
22     /**
23      * <p><b>Space</b> uses concepts described in the Linda Coordination Language 
24      * that eases the implementation of other jPOS components (such as 
25      * Channels, Muxes, etc.), but it is not by any means an attempt to provide
26      * a full implementation.</p>
27      *
28      * <p>jPOS's Space is basically a Map where each entry is a LinkedList 
29      * of values that can be used as a BlockingQueue</p>
30      *
31      * <p>One can place entries on a queue by calling Space.out, take them
32      * by calling Space.in and read (without taking) by calling Space.rd</p>
33      *
34      * @author Alejandro Revilla
35      * @version $Revision$ $Date$
36      * @since 2.0
37      * @see LeasedReference
38      * @see TransientSpace
39      * @see SpaceError
40      * @see <a href="http://www.cs.yale.edu/Linda/linda-lang.html">The Linda Coordination Language</a>
41      */
42     
43     public interface Space<K,V> {
44         /**
45          * Write a new entry into the Space
46          * @param key Entry's key
47          * @param value Object value
48          */
49         public void out (K key, V value);
50     
51         /**
52          * Write a new entry into the Space, with an timeout value
53          * @param key Entry's key
54          * @param value Object value
55          * @param timeout timeout value
56          */
57         public void out (K key, V value, long timeout);
58     
59         /**
60          * Take an entry from the space, waiting forever until one exists.
61          * @param key Entry's key
62          * @return value
63          */
64         public V in  (Object key);
65     
66         /**
67          * Read an entry from the space, waiting forever until one exists.
68          * @param key Entry's key
69          * @return value
70          */
71         public V rd  (Object key);
72     
73         /**
74          * Take an entry from the space, waiting a limited amount of time
75          * until one exists.
76          * @param key Entry's key
77          * @param timeout millis to wait
78          * @return value or null
79          */
80         public V in  (Object key, long timeout);
81     
82     
83         /**
84          * Read an entry from the space, waiting a limited amount of time
85          * until one exists.
86          * @param key Entry's key
87          * @param timeout millis to wait
88          * @return value or null
89          */
90         public V rd  (Object key, long timeout);
91     
92     
93         /**
94          * In probe takes an entry from the space if one exists, 
95          * return null otherwise.
96          * @param key Entry's key
97          * @return value or null
98          */
99         public V inp (Object key);
100     
101     
102         /**
103          * Read probe reads an entry from the space if one exists, 
104          * return null otherwise.
105          * @param key Entry's key
106          * @return value or null
107          */
108         public V rdp (Object key);
109     
110         /**
111          * Write a new entry at the head of a queue.
112          * @param key Entry's key
113          * @param value Object value
114          */
115         public void push (K key, V value);
116     
117         /**
118          * Write a new entry at the head of the queue with a timeout value
119          * @param key Entry's key
120          * @param value Object value
121          * @param timeout timeout value
122          */
123         public void push (K key, V value, long timeout);
124     
125         /**
126          * @param keys array of keys to check
127          * @return true if one or more keys are available in the space
128          */
129         public boolean existAny (K[] keys);
130     
131         /**
132          * @param keys array of keys to check
133          * @param timeout to wait for any of the entries to become available
134          * @return true if one or more keys are available in the space
135          */
136         public boolean existAny (K[] keys, long timeout);
137     
138         /**
139          * Write a single entry at the head of the queue discarding the other entries
140          * @param key Entry's key
141          * @param value Object value
142     
143          */
144         public void put (K key, V value);
145     
146         /**
147          * Write a single entry at the head of the queue discarding the other entries, with timeout.
148          * @param key Entry's key
149          * @param value Object value
150          * @param timeout timeout value
151          */
152         public void put (K key, V value, long timeout);
153     
154     }
