19     package org.jpos.iso;
20     
21     import org.jpos.util.Loggeable;
22     
23     import java.io.PrintStream;
24     import java.io.PrintWriter;
25     
26     /**
27      * Signals that an ISO exception of some sort has occurred. 
28      *
29      * @author  <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
30      * @version $Revision$ $Date$
31      */
32     public class ISOException extends Exception implements Loggeable {
33     
34         private static final long serialVersionUID = -777216335204861186L;
35         /**
36          * @serial
37          */
38         Throwable nested = null;
39     
40         /**
41          * Constructs an <code>ISOException</code> with no detail message. 
42          */
43         public ISOException() {
44             super();
45         }
46     
47         /**
48          * Constructs an <code>ISOException</code> with the specified detail 
49          * message. 
50          *
51          * @param   s   the detail message.
52          */
53         public ISOException(String s) {
54             super(s);
55         }
56     
57         /**
58          * Constructs an <code>ISOException</code> with a nested
59          * exception
60          * @param nested another exception
61          */
62         public ISOException (Throwable nested) {
63             super(nested.toString());
64             this.nested = nested;
65         }
66     
67         /**
68          * Constructs an <code>ISOException</code> with a detail Message nested
69          * exception
70          * @param   s   the detail message.
71          * @param nested another exception
72          */
73         public ISOException (String s, Throwable nested) {
74             super(s);
75             this.nested = nested;
76         }
77     
78         /**
79          * @return nested exception (may be null)
80          */
81         public Throwable getNested() {
82             return nested;
83         }
84     
85         protected String getTagName() {
86             return "iso-exception";
87         }
88         public void dump (PrintStream p, String indent) {
89             String inner = indent + "  ";
90             p.println (indent + "<"+getTagName()+">");
91             p.println (inner  + getMessage());
92             if (nested != null) {
93                 if (nested instanceof ISOException) 
94                     ((ISOException)nested).dump (p, inner);
95                 else {
96                     p.println (inner + "<nested-exception>");
97                     p.print   (inner);
98                     nested.printStackTrace (p);
99                     p.println (inner + "</nested-exception>");
100                 }
101             }
102             p.print (inner);
103             printStackTrace (p);
104             p.println (indent + "</"+getTagName()+">");
105         }
106         public String toString() {
107             StringBuffer buf = new StringBuffer (super.toString());
108             if (nested != null)
109                 buf.append (" (" + nested.toString() +")");
110             return buf.toString();
111         }
112     
113         public void printStackTrace() {
114             super.printStackTrace();
115             if (nested != null) {
116                 System.err.print("Nested:");
117                 nested.printStackTrace();
118             }
119         }
120     
121         public void printStackTrace(PrintStream ps) {
122             super.printStackTrace(ps);
123             if (nested != null) {
124                 ps.print("Nested:");
125                 nested.printStackTrace(ps);
126             }
127         }
128     
129         public void printStackTrace(PrintWriter pw) {
130             super.printStackTrace(pw);
131             if (nested != null) {
132                 pw.print("Nested:");
133                 nested.printStackTrace(pw);
134             }
135         }
136     }
