19     package org.jpos.iso.packager;
20     
21     import java.io.IOException;
22     import org.jpos.iso.ISOComponent;
23     import org.jpos.iso.ISOException;
24     import org.jpos.iso.ISOMsg;
25     import org.jpos.iso.ISOPackager;
26     import org.jpos.util.Log;
27     
28     import java.io.InputStream;
29     
30     public class FSDPackager extends Log implements ISOPackager {
31         public byte[] pack (ISOComponent m) throws ISOException {
32             return m.pack ();
33         }
34         public int unpack (ISOComponent m, byte[] b) throws ISOException {
35             return m.unpack (b);
36         }
37         public void unpack (ISOComponent m, InputStream in) throws ISOException {
38             try {
39                 m.unpack (in);
40             } catch (IOException e) {
41                 throw new ISOException (e);
42             }
43         }
44         public String getFieldDescription(ISOComponent m, int fldNumber) {
45             return null;
46         }
47         public ISOMsg createISOMsg() {
48             return new ISOMsg();
49         }
50         public String getDescription () {
51             return getClass().getName();
52         }
53     }
