19     package org.jpos.iso.gui;
20     
21     import org.jpos.iso.ISOException;
22     import org.jpos.iso.ISOMsg;
23     import org.jpos.iso.ISOPackager;
24     import org.jpos.iso.ISOUtil;
25     
26     import javax.swing.*;
27     import javax.swing.event.ListSelectionEvent;
28     import javax.swing.event.ListSelectionListener;
29     import javax.swing.table.AbstractTableModel;
30     import javax.swing.table.TableModel;
31     import java.awt.*;
32     import java.util.Vector;
33     
34     /**
35      * Called from ISOChannelPanel when you click on it's ISOMeter.
36      * It enable field and header visualization by means of visual
37      * components such as JTable
38      *
39      * @see ISOChannelPanel
40      * @see ISORequestListenerPanel
41      */
42     public class ISOMsgPanel extends JPanel {
43     
44         private static final long serialVersionUID = 7779880613544725704L;
45         /**
46          * @serial
47          */
48         ISOMsg m;
49         /**
50          * @serial
51          */
52         Vector validFields;
53         public ISOMsgPanel(ISOMsg m, boolean withDump) {
54             super();
55             this.m = m;
56             setLayout(new BorderLayout());
57             setBorder(BorderFactory.createRaisedBevelBorder());
58             setValidFields();
59             add(createISOMsgTable(), BorderLayout.CENTER);
60             if (withDump)
61                 add(createISOMsgDumpPanel(), BorderLayout.SOUTH);
62         }
63         public ISOMsgPanel(ISOMsg m) {
64             this(m, false);
65         }
66         private void setValidFields() {
67             validFields = new Vector();
68             for (int i=0; i<=m.getMaxField(); i++)
69                 if (m.hasField(i))
70                     validFields.addElement(new Integer(i));
71         }
72         private JComponent createISOMsgTable() {
73             TableModel dataModel = new AbstractTableModel() {
74     
75                 private static final long serialVersionUID = 8917029825751856951L;
76                 public int getColumnCount() {
77                     return 3;
78                 }
79                 public int getRowCount() {
80                     return validFields.size();
81                 }
82                 public String getColumnName(int columnIndex) {
83                     switch (columnIndex) {
84                         case 0 :
85                             return "#";
86                         case 1 :
87                             return "Content";
88                         case 2 :
89                             return "Description";
90                         default:
91                             return "";
92                     }
93                 }
94                 public Object getValueAt(int row, int col) {
95                     switch (col) {
96                         case 0 :
97                             return ((Integer)validFields.elementAt(row));
98                         case 1 :
99                             try {
100                                 int index =
101                                 ((Integer)validFields.elementAt(row)).intValue();
102     
103                                 Object obj = m.getValue(index);
104                                 if (obj instanceof String) 
105                                     return obj.toString();
106                                 else if (obj instanceof byte[])
107                                     return ISOUtil.hexString((byte[]) obj);
108                                 else if (obj instanceof ISOMsg)
109                                     return "<ISOMsg>";
110                             } catch (ISOException e) {
111                                 e.printStackTrace();
112                             }   
113                             break;
114                         case 2 :
115                             int i=((Integer)validFields.elementAt(row)).intValue();
116                             ISOPackager p = m.getPackager();
117                             return p.getFieldDescription(m,i);
118                     }
119                     return "<???>";
120                 }
121             };
122             JTable table = new JTable(dataModel);
123             table.getColumnModel().getColumn(0).setPreferredWidth(10);
124             table.setPreferredScrollableViewportSize(
125                 new Dimension (500,table.getRowCount()*table.getRowHeight()));
126             table.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
127     
128             ListSelectionModel rowSM = table.getSelectionModel();
129             rowSM.addListSelectionListener(new ListSelectionListener() {
130                 public void valueChanged(ListSelectionEvent e) {
131                     if (e.getValueIsAdjusting())
132                         return;
133     
134                     ListSelectionModel lsm =
135                         (ListSelectionModel)e.getSource();
136                     if (!lsm.isSelectionEmpty()) {
137                         int selectedRow = lsm.getMinSelectionIndex();
138                         try {
139                             int index = ((Integer)
140                                 validFields.elementAt(selectedRow)).intValue();
141     
142                             Object obj = m.getValue(index);
143                             if (obj instanceof ISOMsg) {
144                                 ISOMsg sm = (ISOMsg) obj;
145                                 JFrame f = new JFrame("ISOMsg field "+index);
146                                 ISOMsgPanel p = new ISOMsgPanel(sm, false);
147                                 f.getContentPane().add(p);
148                                 f.pack();
149                                 f.show();
150                             }
151                         } catch (ISOException ex) {
152                             ex.printStackTrace();
153                         }   
154                     }
155                 }
156             });
157     
158             return new JScrollPane(table);
159         }
160         JComponent createISOMsgDumpPanel() {
161             JPanel p = new JPanel();
162             JTextArea t = new JTextArea(3,20);
163     
164             p.setLayout(new BorderLayout());
165             p.setBackground(Color.white);
166             p.setBorder(BorderFactory.createLoweredBevelBorder());
167             p.add(new JLabel("Dump", SwingConstants.LEFT),
168                 BorderLayout.NORTH);
169             t.setFont(new Font ("Helvetica", Font.PLAIN, 8));
170             t.setLineWrap(true);
171             try {
172                 StringBuffer buf = new StringBuffer();
173                 if (m.getHeader() != null) 
174                     buf.append("--[Header]--\n" 
175                         +ISOUtil.hexString(m.getHeader()) + "\n--[Msg]--\n");
176                 byte[] b = m.pack();
177                 buf.append (ISOUtil.hexString(b));
178                 t.setText(buf.toString());
179             } catch (ISOException e) {
180                 t.setText(e.toString());
181                 t.setForeground(Color.red);
182             }
183             p.add(t, BorderLayout.CENTER);
184             return p;
185         }
186     }
