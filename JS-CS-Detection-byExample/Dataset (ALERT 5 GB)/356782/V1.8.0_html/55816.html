19     package org.jpos.util;
20     
21     import java.io.PrintStream;
22     import java.util.Collection;
23     import java.util.Iterator;
24     import java.util.LinkedHashMap;
25     
26     /**
27      * Simple Profiler
28      * @author Alejandro P. Revilla
29      * @author David D. Bergert
30      * @version $Id$
31      */
32     public class Profiler implements Loggeable {
33         long start, partial;
34         LinkedHashMap events;
35         public static final int TO_MILLIS = 1000000;
36     
37         public Profiler () {
38             super();
39             reset();
40         }
41         /**
42          * reset timers
43          */
44         public void reset() {
45             start = partial = System.nanoTime();
46             events = new LinkedHashMap();
47         }
48         /**
49          * mark checkpoint
50          * @param detail checkpoint information
51          */
52         @SuppressWarnings("unchecked")
53         public synchronized void checkPoint (String detail) {
54             long now = System.nanoTime();
55             Entry e = new Entry();
56             e.setDurationInNanos((now - partial));
57             e.setTotalDurationInNanos((now - start));
58             if (events.containsKey(detail)) {
59                 for (int i=1; ;i++) {
60                     String d = detail + "-" + i;
61                     if (!events.containsKey (d)) {
62                         detail = d;
63                         break;
64                     }
65                 }
66             }
67             e.setEventName(detail);
68             events.put (detail, e);
69             partial = now;
70         }
71         /**
72          * @return total elapsed time since last reset
73          */
74         public long getElapsed() {
75             return System.nanoTime() - start;
76         }
77         /**
78          * @return parcial elapsed time since last reset
79          */
80         public long getPartial() {
81             return System.nanoTime() - partial;
82         }
83         public void dump (PrintStream p, String indent) {
84             String inner = indent + "  ";
85             partial = start;
86             if (!events.containsKey("end"))
87                 checkPoint ("end");
88             Collection c = events.values();
89             Iterator iter = c.iterator();
90             p.println (indent + "<profiler>");
91             while (iter.hasNext()) 
92                 p.println (inner + iter.next().toString());
93             p.println (indent + "</profiler>");
94         }
95         public Entry getEntry(String eventName) {
96              return (Entry)events.get(eventName);         
97         }
98     
99         public class Entry  {
100             String  eventName;
101             long    duration;
102             long    totalDuration;          
103             public Entry() {
104                eventName     = "";
105                duration      = 0L;
106                totalDuration = 0L;        
107             }
108             public void setEventName (String myEvent) {
109                 this.eventName = myEvent;
110             }
111             public String getEventName () {
112                 return eventName;
113             }    
114             public void setDurationInNanos (long duration) {
115                 this.duration = duration;
116             }
117             public long getDuration () {
118                 return duration / TO_MILLIS;
119             }
120             public long getDurationInNanos() {
121                 return duration;
122             }
123             public void setTotalDurationInNanos (long totalDuration) {
124                 this.totalDuration = totalDuration;
125             }
126             public long getTotalDuration () {
127                 return totalDuration / TO_MILLIS;
128             }
129             public long getTotalDurationInNanos () {
130                 return totalDuration;
131             }
132             public String toString()  {
133                 StringBuffer sb = new StringBuffer (eventName);
134                 sb.append (" [");
135                 sb.append (getDuration());
136                 sb.append ('.');
137                 sb.append ((duration % TO_MILLIS) / 100000);
138                 sb.append ('/');
139                 sb.append (getTotalDuration ());
140                 sb.append ('.');
141                 sb.append ((totalDuration % TO_MILLIS) / 100000);
142                 sb.append (']');
143                 return sb.toString();
144             }            
145         }
146     }
