19     package org.jpos.ui.factory;
20     
21     import org.jdom.Element;
22     import org.jpos.ui.UI;
23     import org.jpos.ui.UIFactory;
24     
25     import javax.swing.*;
26     import javax.swing.border.EmptyBorder;
27     import java.awt.*;
28     
29     /**
30      * @author Alejandro Revilla
31      *
32      * Creates a JLabel
33      * i.e:
34      * <pre>
35      *  &lt;label font="arial-bold-16"&gt;Hello UI&lt;/label&gt;
36      * </pre>
37      * @see org.jpos.ui.UIFactory
38      */
39     public class JLabelFactory implements UIFactory {
40         public JComponent create (UI ui, Element e) {
41             JLabel label = new JLabel (e.getText());
42             String font = e.getAttributeValue ("font");
43             if (font != null) 
44                 label.setFont (Font.decode (font));
45             label.setHorizontalAlignment(JLabel.CENTER);
46             label.setBorder(new EmptyBorder(3, 3, 3, 3));
47             return label;
48         }
49     }
