19     package org.jpos.iso.channel;
20     
21     import java.io.BufferedReader;
22     import java.io.EOFException;
23     import java.io.IOException;
24     import java.io.InputStreamReader;
25     import java.net.ServerSocket;
26     import java.net.Socket;
27     import java.util.Vector;
28     
29     import org.jpos.iso.BaseChannel;
30     import org.jpos.iso.ISOChannel;
31     import org.jpos.iso.ISOException;
32     import org.jpos.iso.ISOMsg;
33     import org.jpos.iso.ISOPackager;
34     import org.jpos.iso.ISOUtil;
35     
36     /**
37      * Implements an ISOChannel suitable to be used to connect to an X.25 PAD. 
38      * It waits a limited amount of time to decide when a packet is ready
39      * to be unpacked.
40      *
41      * This channel is based on PADChannel version 1.4. The new version
42      * seems to have some problems dealing with ETXs (we're working on it).
43      * Use this version _only_ if you have problems with current PADChannel
44      * as it will be deprecated some time in the future.
45      *
46      * @author  <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
47      * @version $Id$
48      *
49      * @see ISOMsg
50      * @see ISOException
51      * @see ISOChannel
52      */
53     public class X25Channel extends BaseChannel {
54         BufferedReader reader = null;
55         protected byte[] header;
56         /**
57          * No-args constructor
58          */
59         public X25Channel () {
60             super();
61         }
62         /**
63          * Constructs client ISOChannel
64          * @param host  server TCP Address
65          * @param port  server port number
66          * @param p     an ISOPackager
67          * @see ISOPackager
68          */
69         public X25Channel (String host, int port, ISOPackager p) {
70             super(host, port, p);
71         }
72         /**
73          * Construct server ISOChannel
74          * @param p     an ISOPackager
75          * @see ISOPackager
76          * @exception IOException
77          */
78         public X25Channel (ISOPackager p) throws IOException {
79             super(p);
80         }
81         /**
82          * constructs a server ISOChannel associated with a Server Socket
83          * @param p     an ISOPackager
84          * @param serverSocket where to accept a connection
85          * @exception IOException
86          * @see ISOPackager
87          */
88         public X25Channel (ISOPackager p, ServerSocket serverSocket) 
89             throws IOException
90         {
91             super(p, serverSocket);
92         }
93         /**
94          * @return a byte array with the received message
95          * @exception IOException
96          */
97         protected byte[] streamReceive() throws IOException {
98             int c, k=0, len = 1;
99             Vector v = new Vector();
100     
101             c = serverIn.read();
102             if (c == -1)
103                 throw new EOFException ("connection closed");
104             byte[] b = new byte[1];
105             b[0] = (byte) c;
106             v.addElement (b);
107     
108             // Wait for packets until timeout
109             while ((c = serverIn.available()) > 0) {
110                 b = new byte[c];
111                 if (serverIn.read (b) != c)
112                     throw new EOFException ("connection closed");
113                 v.addElement (b);
114                 len += c;
115                 try {
116                     Thread.sleep (50);
117                 } catch (InterruptedException e) { }
118             }
119     
120             byte[] d = new byte[len];
121             for (int i=0; i<v.size(); i++) {
122                 b = (byte[]) v.elementAt(i);
123                 System.arraycopy (b, 0, d, k, b.length);
124                 k += b.length;
125             }
126             return d;
127         }
128         protected void connect (Socket socket) throws IOException {
129             super.connect (socket);
130             reader = new BufferedReader (new InputStreamReader (serverIn));
131         }
132         public void disconnect () throws IOException {
133             super.disconnect ();
134             reader = null;
135         }
136         protected int getHeaderLength() { 
137             return header != null ? header.length : 0;
138         }
139         public void setHeader (byte[] header) {
140             this.header = header;
141         }
142         /**
143          * @param header Hex representation of header
144          */
145         public void setHeader (String header) {
146             setHeader (
147                 ISOUtil.hex2byte (header.getBytes(), 0, header.getBytes().length / 2)
148             );
149         }
150         public byte[] getHeader () {
151             return header;
152         }
153         protected void sendMessageHeader(ISOMsg m, int len) throws IOException { 
154             if (m.getHeader() != null)
155                 serverOut.write(m.getHeader());
156             else if (header != null) 
157                 serverOut.write(header);
158         }
159     }
