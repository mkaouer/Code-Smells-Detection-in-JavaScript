29     /**
30      * Tiny Space implementation
31      *
32      * @author Alejandro Revilla
33      * @version $Revision$ $Date$
34      * @since 1.4.7
35      */
36     public class TinySpace implements Space, Serializable {
37         protected Map map = new HashMap ();
38     
39         public TinySpace () {
40             super();
41         }
42         public void out (Object key, Object value) {
43             synchronized (this) {
44                 Object v = map.get (key);
45                 if (v == null)
46                     map.put (key, value);
47                 else if (v instanceof Data)
48                     ((Data)v).add (value);
49                 else {
50                     Data data = new Data();
51                     data.add (v);
52                     data.add (value);
53                     map.put (key, data);
54                 }
55                 this.notifyAll ();
56             }
57         }
58         public void out (Object id, Object value, long timeout) {
59             out (id, new LeasedReference (value, timeout));
60         }
61         public synchronized Object rdp (Object key) {
62             Object obj = map.get (key);
63             if (obj instanceof Data) 
64                 obj = ((Data) obj).get (key);
65             else if (obj instanceof LeasedReference) {
66                 obj = ((LeasedReference)obj).get ();
67             }
68             return obj;
69         }
70         public synchronized Object inp (Object key) {
71             Object obj = map.get (key);
72             if (obj instanceof Data) {
73                 Data data = (Data) obj;
74                 obj = data.remove ();
75                 if (data.isEmpty ())
76                     map.remove (key);
77             } else if (obj != null) {
78                 if (obj instanceof LeasedReference) {
79                     obj = ((LeasedReference)obj).get ();
80                 }
81                 map.remove (key);
82             }
83     
84             return obj;
85         }
86         public synchronized Object in (Object key) {
87             Object obj;
88             while ((obj = inp (key)) == null) {
89                 try {
90                     this.wait ();
91                 } catch (InterruptedException e) { }
92             }
93             return obj;
94         }
95         public synchronized Object in  (Object key, long timeout) {
96             Object obj;
97             long now = System.currentTimeMillis();
98             long end = now + timeout;
99             while ((obj = inp (key)) == null && 
100                     ((now = System.currentTimeMillis()) < end))
101             {
102                 try {
103                     this.wait (end - now);
104                 } catch (InterruptedException e) { }
105             }
106             return obj;
107         }
108         public synchronized Object rd  (Object key) {
109             Object obj;
110             while ((obj = rdp (key)) == null) {
111                 try {
112                     this.wait ();
113                 } catch (InterruptedException e) { }
114             }
115             return obj;
116         }
117         public synchronized Object rd  (Object key, long timeout) {
118             Object obj;
119             long now = System.currentTimeMillis();
120             long end = now + timeout;
121             while ((obj = rdp (key)) == null && 
122                     ((now = System.currentTimeMillis()) < end))
123             {
124                 try {
125                     this.wait (end - now);
126                 } catch (InterruptedException e) { }
127             }
128             return obj;
129         }
130         public void push (Object id, Object value) {
131             throw new SpaceError ("Unsupported operation");
132         }
133         public void push (Object id, Object value, long timeout) {
134             throw new SpaceError ("Unsupported operation");
135         }
136         protected static final class Data {
137             LinkedList data;
138     
139             protected Data () {
140                 super();
141                 data = new LinkedList ();
142             }
143             protected Data (Object value) {
144                 this ();
145                 add (value);
146             }
147             protected void add (Object value) {
148                 data.add (value);
149             }
150             protected Object get (Object value) {
151                 Object obj = null;
152                 while (size() > 0) {
153                     obj = data.getFirst();
154                     if (obj instanceof LeasedReference) {
155                         obj = ((LeasedReference)obj).get ();
156                         if (obj == null) {
157                             data.removeFirst ();
158                             continue;
159                         }
160                     }
161                     break;
162                 }
163                 return obj;
164             }
165             protected int size () {
166                 return data.size ();
167             }
168             protected Object remove () {
169                 Object obj = null;
170                 while (size() > 0) {
171                     obj = data.removeFirst();
172                     if (obj instanceof LeasedReference) {
173                         LeasedReference ref = (LeasedReference) obj;
174                         obj = ref.get ();
175                         if (obj == null) {
176                             continue;
177                         }
178                         ref.discard ();
179                     }
180                     break;
181                 }
182                 return obj;
183             }
184             protected boolean isEmpty () {
185                 return data.isEmpty ();
186             }
187         }
188     }
