19     package  org.jpos.security;
20     
21     import java.io.Serializable;
22     
23     import org.jpos.util.Loggeable;
24     
25     
26     /**
27      * Represents a key that cannot be used except by your security module and for
28      * performing the operations allowed by the security module for this type 
29      * of keys.
30      *
31      * So, a SecureKey can be safely stored in a clear file or database.
32      *
33      * <p>
34      * A SecureKey is typically a key encrypted under one of the secret keys that are
35      * protected by the security module itself (Local Master Keys --LMK for short).<br>
36      * </p>
37      * <p>
38      * SecureKey just holds:<br>
39      * 1- Secure Key: a byte[] holding the key in the secure proprietary format
40      *    of your security module. This is typically the clear key encrypted under LMK.<br>
41      * 2- Key Type: identifies what this key can be used for (e.g. TYPE_ZPK
42      *    (Zone PIN Key), TYPE_ZMK (Zone Master Key)...<BR>
43      * 3- Key Length (in bits): also called key size. e.g. LENGTH_DES, LENGTH_DES3_2KEY,...etc.
44      *    This is not necessarily deducible from the length of the byte[] holding
45      *    the secure key bytes, since encryption under LMK is proprietary to the
46      *    security module.
47      * </p>
48      * <p>
49      * NOTE: The security of SecureKey is totally dependent on the security of
50      * the used security module.
51      * </p>
52      * @author Hani S. Kirollos
53      * @version $Revision$ $Date$
54      * @see SMAdapter SMAdapter
55      *
56      */
57     public abstract class SecureKey
58             implements Serializable, Loggeable {
59         /**
60          * Secure Key Bytes
61          */
62         protected byte[] keyBytes = null;
63         /**
64          * This is the bit length of the key
65          * This can be: LENGTH_DES, LENGTH_DES3_2KEY, ...
66          */
67         protected short keyLength;
68         /**
69          * Key Type is useful for stating what this key can be used for.
70          * The value of Key Type specifies whether this encryped key is a
71          * TYPE_TMK (Terminal Master Key), TYPE_ZPK (Zone PIN Key)....<BR>
72          */
73         protected String keyType;
74     
75         /**
76          * Optional key name
77          */
78         protected String keyName;
79     
80         /**
81          * Sets the secure key bytes
82          * @param keyBytes byte[] representing the secured key bytes
83          */
84         public void setKeyBytes (byte[] keyBytes) {
85             this.keyBytes = keyBytes;
86         }
87     
88         /**
89          * @return The byte[] holding the secure key Bytes
90          */
91         public byte[] getKeyBytes () {
92             return  keyBytes;
93         }
94     
95         /**
96          * Sets the length of the key (in bits) (when it was still clear).
97          * This might be different than the bit length of the secureKeyBytes.
98          * @param keyLength
99          */
100         public void setKeyLength (short keyLength) {
101             this.keyLength = keyLength;
102         }
103     
104         /**
105          * @return The Length of the secure key (when it was still clear)
106          */
107         public short getKeyLength () {
108             return  keyLength;
109         }
110     
111         /**
112          * Key Type is useful for stating what this key can be used for.
113          * The value of Key Type specifies whether this secure key is a
114          * TYPE_TMK (Terminal Master Key), TYPE_ZPK (Zone PIN Key)....<BR>
115          * @param keyType
116          */
117         public void setKeyType (String keyType) {
118             this.keyType = keyType;
119         }
120     
121         /**
122          * Key Type is useful for stating what this key can be used for.
123          * The value of Key Type specifies whether this secure key is a
124          * TYPE_TMK (Terminal Master Key), TYPE_ZPK (Zone PIN Key)....<BR>
125          * @return keyType
126          */
127         public String getKeyType () {
128             return  this.keyType;
129         }
130     
131         /**
132          * optional key name
133          */
134         public String getKeyName() {
135             return this.keyName;
136         }
137         /**
138          * optional key name
139          * @param keyName string
140          */
141         public void setKeyName (String keyName) {
142             this.keyName = keyName;
143         }
144     }
