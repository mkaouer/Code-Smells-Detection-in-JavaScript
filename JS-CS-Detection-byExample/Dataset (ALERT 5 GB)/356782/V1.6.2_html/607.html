19     package  org.jpos.security.jceadapter;
20     
21     import java.util.Properties;
22     
23     import org.jpos.core.ConfigurationException;
24     import org.jpos.core.SimpleConfiguration;
25     import org.jpos.security.SMException;
26     import org.jpos.security.SecureDESKey;
27     import org.jpos.util.Logger;
28     import org.jpos.util.SimpleLogListener;
29     
30     
31     /**
32      * A simple application for sending critical commands to the JCE Security Module.
33      * The functionalities available from this console, are not available programmatically (via API's),
34      * for security reasons, because most of them involve clear (non encrypted) keys.
35      * Those commands are package protected in the JCE Security Module.
36      * @author Hani Samuel Kirollos
37      * @version $Revision$ $Date$
38      */
39     public class Console {
40     
41         public Console () {
42         }
43     
44         /**
45          * @param args
46          */
47         public static void main (String[] args) {
48             JCESecurityModule sm = new JCESecurityModule();
49             Logger logger = new Logger();
50             logger.addListener(new SimpleLogListener(System.out));
51             sm.setLogger(logger, "jce-security-module");
52             Properties cfgProps = new Properties();
53             SimpleConfiguration cfg = new SimpleConfiguration(cfgProps);
54             String commandName = null;
55             String[] commandParams = new String[7];                 // 7 is Maximum number of paramters for a command
56             System.out.println("Welcome to JCE Security Module console commander!");
57             if (args.length == 0) {
58                 System.out.println("Usage: Console [-options] command [commandparameters...]");
59                 System.out.println("\nwhere options include:");
60                 System.out.println("    -lmk <filename>");
61                 System.out.println("                  to specify the Local Master Keys file");
62                 System.out.println("    -rebuildlmk   to rebuild new Local Master Keys");
63                 System.out.println("                  WARNING: old Local Master Keys gets overwritten");
64                 System.out.println("    -jce <provider classname>");
65                 System.out.println("                  to specify a JavaTM Cryptography Extension 1.2.1 provider");
66                 System.out.println("\nWhere command include: ");
67                 System.out.println("    GC <keyLength>");
68                 System.out.println("                  to generate a clear key component.");
69                 System.out.println("    FK <keyLength> <keyType> <component1> <component2> <component2>");
70                 System.out.println("                  to form a key from three clear components.");
71                 System.out.println("                  and returns the key encrypted under LMK");
72                 System.out.println("                  Odd parity is be forced before encryption under LMK");
73                 System.out.println("    CK <keyLength> <keyType> <KEYunderLMK>");
74                 System.out.println("                  to generate a key check value for a key encrypted under LMK.");
75             }
76             else {
77                 int argsCounter = 0;
78                 for (int j = 0; j < 10; j++) {
79                     if ((argsCounter < args.length) &&
80                         (args[argsCounter].toLowerCase().compareTo("-lmk") == 0)
81                     ) {
82                         argsCounter++;
83                         cfgProps.setProperty("lmk", args[argsCounter++]);
84                     }
85                     if ((argsCounter < args.length) &&
86                         (args[argsCounter].toLowerCase().compareTo("-jce") == 0)
87                     ) {
88                         argsCounter++;
89                         cfgProps.setProperty("jce", args[argsCounter++]);
90                     }
91                     if ((argsCounter < args.length) &&
92                         (args[argsCounter].toLowerCase().compareTo("-rebuildlmk") == 0)
93                     ) {
94                         argsCounter++;
95                         cfgProps.setProperty("rebuildlmk", "true");
96                     }
97                 }
98                 if (argsCounter < args.length) {
99                     commandName = args[argsCounter++];
100                     int i = 0;
101                     while (argsCounter < args.length) {
102                         commandParams[i++] = args[argsCounter++];
103                     }
104                 }
105                 // Configure JCE Security Module
106                 try {
107                     sm.setConfiguration(cfg);
108                 } catch (ConfigurationException e) {
109                     e.printStackTrace();
110                     System.exit(0);
111                 }
112                 // Execute Command
113                 if (commandName != null) {
114                     try {
115                         short keyLength = (short)Integer.parseInt(commandParams[0]);
116                         if (commandName.toUpperCase().compareTo("GC") == 0) {
117                             String clearKeyComponenetHexString = sm.generateClearKeyComponent(keyLength);
118                         }
119                         else if (commandName.toUpperCase().compareTo("FK") == 0) {
120                             SecureDESKey KEYunderLMK = sm.formKEYfromThreeClearComponents(keyLength,
121                                     commandParams[1].toUpperCase(), commandParams[2], commandParams[3], commandParams[4]);
122                         }
123                         else if (commandName.toUpperCase().compareTo("CK") == 0) {
124                             SecureDESKey KEYunderLMK = sm.generateKeyCheckValue(keyLength,
125                                     commandParams[1].toUpperCase(), commandParams[2]);
126                         }
127                         else {
128                             System.err.println("Unknown command: " + commandName);
129                         }
130                     } catch (SMException e) {
131                     //e.printStackTrace();
132                     } catch (java.lang.NumberFormatException e) {
133                         System.err.println("Invalid KeyLength");
134                     }
135                 }
136                 else {
137                     System.err.println("No command specified");
138                 }
139             }
140         }
141     }
