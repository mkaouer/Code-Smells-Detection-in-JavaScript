19     package org.jpos.iso.packager;
20     
21     import java.util.BitSet;
22     import java.util.Hashtable;
23     import java.util.Vector;
24     
25     import org.jpos.iso.ISOBasePackager;
26     import org.jpos.iso.ISOBitMap;
27     import org.jpos.iso.ISOBitMapPackager;
28     import org.jpos.iso.ISOComponent;
29     import org.jpos.iso.ISOException;
30     import org.jpos.iso.ISOFieldPackager;
31     import org.jpos.iso.ISOPackager;
32     import org.jpos.iso.ISOUtil;
33     import org.jpos.util.LogEvent;
34     import org.jpos.util.Logger;
35     
36     /**
37      * ISO 8583 v1987 BINARY Packager 
38      * customized for VISA Base1 subfields
39      * 
40      *
41      * @author <a href="mailto:eoin.flood@orbiscom.com">Eoin Flood</a>
42      * @version $Id$
43      * @see ISOPackager
44      * @see ISOBasePackager
45      * @see ISOComponent
46      */
47     
48     public class Base1SubFieldPackager extends ISOBasePackager 
49     {
50         // These methods are identical to ISOBasePackager
51         // except that fld[1] has been replaced with fld[0]
52         // and a secondard bitmap is not allowed
53     
54         protected boolean emitBitMap()
55         {
56             return (fld[0] instanceof ISOBitMapPackager);
57         }
58     
59         protected int getFirstField()
60         {
61             return (fld[0] instanceof ISOBitMapPackager) ? 1 : 0;
62         }
63     
64         protected ISOFieldPackager getBitMapfieldPackager() 
65         {
66             return fld[0];
67         }
68     
69         /**
70          * Unpack a packed subfield into
71          * its corresponding ISOComponent
72          */
73     
74         public int unpack (ISOComponent m, byte[] b) throws ISOException 
75         {
76             LogEvent evt = new LogEvent (this, "unpack");
77             try 
78             {
79                 if (m.getComposite() != m) 
80                     throw new ISOException ("Can't call packager on non Composite");
81                 if (logger != null)  // save a few CPU cycle if no logger available
82                     evt.addMessage (ISOUtil.hexString (b));
83     
84                 int consumed=0;
85                 ISOBitMap bitmap = new ISOBitMap (-1);
86     
87                 BitSet bmap = null;
88                 int maxField = fld.length;
89                 if (emitBitMap()) 
90                 {
91                     consumed += getBitMapfieldPackager().unpack(bitmap,b,consumed);
92                     bmap = (BitSet) bitmap.getValue();
93                     m.set (bitmap);
94                     maxField = bmap.size();
95                 }
96                 for (int i=getFirstField(); i<maxField && consumed < b.length; i++) 
97                 {
98                     if (bmap == null || bmap.get(i)) 
99                     {
100                         ISOComponent c = fld[i].createComponent(i);
101                         consumed += fld[i].unpack (c, b, consumed);
102                         m.set(c);
103                     }
104                 }
105                 if (b.length != consumed) 
106                 {
107                     evt.addMessage (
108                     "WARNING: unpack len=" +b.length +" consumed=" +consumed);
109                 }
110                 return consumed;
111             } 
112             catch (ISOException e) 
113             {
114                 evt.addMessage (e);
115                 throw e;
116             } 
117             finally 
118             {
119                 Logger.log (evt);
120             }
121         }
122     
123         /**
124          * Pack the subfield into a byte array
125          */ 
126     
127         public byte[] pack (ISOComponent m) throws ISOException 
128         {
129             LogEvent evt = new LogEvent (this, "pack");
130             try 
131             {
132                 ISOComponent c;
133                 Vector v = new Vector();
134                 Hashtable fields = m.getChildren();
135                 int len = 0;
136                 byte[] b;
137     
138                 if (emitBitMap()) 
139                 {
140                     // BITMAP (-1 in HashTable)
141                     c = (ISOComponent) fields.get (new Integer (-1));
142                     b = getBitMapfieldPackager().pack(c);
143                     len += b.length;
144                     v.addElement (b);
145                 }
146     
147                 for (int i=getFirstField(); i<=m.getMaxField(); i++) 
148                 {
149                     if ((c = (ISOComponent) fields.get (new Integer (i))) != null) 
150                     {
151                         try 
152                         {
153                             b = fld[i].pack(c);
154                             len += b.length;
155                             v.addElement (b);
156                         } 
157                         catch (Exception e) 
158                         {
159                             evt.addMessage ("error packing field "+i);
160                             evt.addMessage (c);
161                             evt.addMessage (e);
162                         }
163                     }
164                 }
165                 int k = 0;
166                 byte[] d = new byte[len];
167                 for (int i=0; i<v.size(); i++) 
168                 {
169                     b = (byte[]) v.elementAt(i);
170                     for (int j=0; j<b.length; j++)
171                     d[k++] = b[j];
172                 }
173                 if (logger != null)  // save a few CPU cycle if no logger available
174                     evt.addMessage (ISOUtil.hexString (d));
175                 return d;
176             } 
177             catch (ISOException e) 
178             {
179                 evt.addMessage (e);
180                 throw e;
181             } 
182             finally 
183             {
184                 Logger.log(evt);
185             }
186         }
187     }
