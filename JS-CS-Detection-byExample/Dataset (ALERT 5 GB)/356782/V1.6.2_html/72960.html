133         public ISOMeter(ISOChannelPanel parent) {
134             super();
135             this.parent = parent;
136     
137             fontBig   = new Font ("Helvetica", Font.ITALIC, mass*3/4);
138             fontSmall = new Font ("Helvetica", Font.PLAIN, 10);
139             yPoints = new int[width];
140             xPoints = new int[width];
141             for (int i=0; i<width; i++) {
142                 xPoints[i] = i;
143                 yPoints[i] = mass;
144             }
145             positiveText  = null;
146             negativeText  = null;
147             positiveCounter = negativeCounter = "";
148             connected     = false;
149     
150             MouseListener mouseListener = new MouseAdapter() {
151                 public void mouseClicked(MouseEvent e) {
152                     showLogList();
153                 }
154             };
155             addMouseListener(mouseListener);
156         }

158         public void start() {
159             Thread t = new Thread (this,"ISOMeter");
160             t.setPriority (Thread.NORM_PRIORITY-1);
161             t.setName ("ISOMeter");
162             t.start();
163         }

165         public void showLogList() {
166             JFrame f = new JFrame(parent.getSymbolicName());
167             f.getContentPane().add(createLogList());
168             f.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
169             f.validate();
170             f.pack();
171             f.setSize(width,width+50);
172             f.show();
173         }
174     
175         public JComponent createLogList() {
176             final JList logList = new JList(parent.getLog());
177             JPanel A = new JPanel();
178             A.setLayout(new BorderLayout());
179     
180             MouseListener mouseListener = new MouseAdapter() {
181                 public void mouseClicked(MouseEvent e) {
182                     ISOMsg m = (ISOMsg) logList.getSelectedValue();
183                     if (m != null) {
184                         JFrame f = new JFrame(
185                             parent.getSymbolicName()+":"+m.toString());
186                         ISOMsgPanel p = new ISOMsgPanel(m);
187                         f.getContentPane().add(p);
188                         f.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
189                         f.pack();
190                         f.show();
191                     }
192                 }
193             };
194             logList.addMouseListener(mouseListener);
195     
196             logList.setPrototypeCellValue("9999 99999999 999999");
197             JScrollPane scrollPane = new JScrollPane(logList);
198             A.add(scrollPane, BorderLayout.CENTER);
199             return A;
200         }

202         public void setValue(int val) {
203             int y = mass - (val*height/2000);
204             yPoints[width-1] = y;
205             continueScroll = width;
206             scroll();
207         }
208     
209         public void setScroll (boolean scroll) {
210             this.scroll = scroll;
211         }
212         public void setRefresh (int refreshPanel) {
213             if (refreshPanel > 0)
214                 this.refreshPanel = refreshPanel;
215         }
216         public void setConnected(boolean connected) {
217             if (this.connected != connected)
218                 if (!scroll)
219                     if (connected)
220                         continueScroll = width;
221                     else
222                         continueScroll = 1;
223             this.connected = connected;
224         }
225         public void setPositiveCounter(String s) {
226             positiveCounter = s;
227         }
228         public void setNegativeCounter(String s){
229             negativeCounter = s;
230         }
231         public void setValue(int val, String textString) {
232             setValue(val);
233             if (val < 0) {
234                 negativeText = textString;
235                 lastNegative = 0;
236             }
237             else {
238                 positiveText = textString;
239                 lastPositive = 0;
240             }
241         }
242         public void paint (Graphics g) {
243             plot();
244             g.drawImage (im, 0, 0, null);
245         }
246         public Dimension getPreferredSize() {
247             return new Dimension(width, height);
248         }
249         private void scroll() {
250             for (int i=0; i<width-1; i++) 
251                 yPoints[i] = yPoints[i+1];
252             if (continueScroll > 0)
253                 continueScroll--;
254         }
255         public void plot() {
256            if (im == null) {
257                 im = createImage(width, height);
258                 img = im.getGraphics ();
259                 img.setColor (Color.black);
260                 img.fillRoundRect (0, 0, width, height, 10, 10);
261                 img.clipRect (0, 0, width, height);
262                 plotGrid();
263     
264                 /* save a copy of the image */
265                 imb = createImage(width, height);
266                 imbCopy = imb.getGraphics ();
267                 imbCopy.drawImage (im, 0, 0, this);
268             }
269             img.drawImage (imb, 0, 0, this);
270             if (continueScroll > 0)
271                 scroll();
272             plotText(positiveText, lastPositive++, 3, mass-3);
273             plotText(negativeText, lastNegative++, 3, height-3);
274             plotCounters(positiveCounter, negativeCounter);
275             img.setColor (connected ? Color.green : Color.red);
276             img.drawPolyline(xPoints, yPoints, width);
277         }
278         private void plotGrid() {
279             img.setColor(Color.blue);
280             for (int i=0; i<width; i++)
281                 if (i % 20 == 0) 
282                     img.drawLine(i,0,i,height);
283             for (int i=-1000; i<1000; i+= 200) {
284                 int y = mass + (i*height/2000);
285                 img.drawLine(0,y,width,y);
286             }
287         }
288         private void plotText(String t, int l, int x, int y) {
289             if (t != null && l < 20) {
290                 img.setColor(Color.lightGray);
291                 img.setFont(fontBig);
292                 img.drawString (t, x, y);
293             }
294         }
295         private void plotCounters(String p, String n) {
296             img.setColor(Color.lightGray);
297             img.setFont(fontSmall);
298             img.drawString (p, width-45, 13);
299             img.drawString (n, width-45, height-3);
300         }
301         public void run () {
302             for (;;) {
303                 if (continueScroll > 0)
304                     repaint();
305                 try { 
306                     Thread.sleep(refreshPanel);
307                 } catch (InterruptedException e) { }
308             }
309         }
310         public void update (Graphics g) {
311             paint (g);
312         }
313     }
