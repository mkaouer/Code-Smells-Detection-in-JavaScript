33         public void testSimpleOut() {
34             sp.out ("testSimpleOut_Key", "ABC");
35             sp.out ("testSimpleOut_Key", "XYZ");
36     
37             assertEquals ("ABC", sp.rdp ("testSimpleOut_Key"));
38             assertEquals ("ABC", sp.inp ("testSimpleOut_Key"));
39             assertEquals ("XYZ", sp.rdp ("testSimpleOut_Key"));
40             assertEquals ("XYZ", sp.inp ("testSimpleOut_Key"));
41             assertNull (sp.rdp ("Test"));
42             assertNull (sp.inp ("Test"));
43         }
44         public void testNullEntry() {
45             try {
46                 sp.out ("testNull", null);
47             } catch (NullPointerException e) {
48                 assertNull ("Verify null entry (rdp)", sp.rdp ("testNull"));
49                 assertNull ("Verify null entry (inp)", sp.inp ("testNull"));
50                 return;

54         public void testExpiration () {
55             sp.out ("testExpiration_Key", "ABC", 50);
56             assertEquals ("ABC", sp.rdp ("testExpiration_Key"));
57             try {
58                 Thread.sleep (60);
59             } catch (InterruptedException e) { }
60             assertNull ("ABC", sp.rdp ("testExpiration_Key"));

62         public void testOutRdpInpRdp() throws Exception {
63             Object o = new Boolean (true);
64             String k = "testOutRdpInpRdp_Key";
65             sp.out (k, o);
66             assertTrue (o.equals (sp.rdp (k)));
67             assertTrue (o.equals (sp.rd  (k)));
68             assertTrue (o.equals (sp.rd  (k, 1000)));
69             assertTrue (o.equals (sp.inp (k)));
70             assertNull (sp.rdp (k));
71             assertNull (sp.rd  (k, 100));
72         }
73         public void testMultiKeyLoad() throws Exception {
74             String s = "The quick brown fox jumped over the lazy dog";
75             Profiler prof = new Profiler ();
76             for (int i=0; i<COUNT; i++) {
77                 sp.out ("testMultiKeyLoad_Key" + Integer.toString (i), s, 60000);
78             }
79             // prof.dump (System.err, "MultiKeyLoad out >");
80             prof = new Profiler ();
81             for (int i=0; i<COUNT; i++) {
82                 assertTrue (s.equals (sp.in ("testMultiKeyLoad_Key" + Integer.toString (i))));
83             }
84             // prof.dump (System.err, "MultiKeyLoad in  >");
85         }

155         public void testNotify() {
156             sp.addListener ("TestDelayNotify_Key", this, 500);
157             sp.out ("TestNotify_Key", "ABCCBA");
158             assertNull (notifiedValue);
159     
160             sp.addListener ("TestNotify_Key", this);
161             sp.out ("TestNotify_Key", "ABCCBA");
162             assertEquals (notifiedValue, "ABCCBA");
163     
164             sp.out ("TestNotify_Key", "012345");
165             assertEquals (notifiedValue, "012345");
166     
167             assertEquals (sp.inp ("TestNotify_Key"), "ABCCBA");
168             assertEquals (sp.inp ("TestNotify_Key"), "ABCCBA");
169             assertEquals (sp.inp ("TestNotify_Key"), "012345");
170     
171             sp.out ("TestDelayNotify_Key", "OLD");
172             assertEquals (notifiedValue, "OLD");
173             try {
174                 Thread.sleep (600);
175             } catch (InterruptedException e) { }
176             sp.out ("TestDelayNotify_Key", "NEW");
177             assertEquals (notifiedValue, "OLD");  // still OLD
178             assertEquals (sp.inp ("TestDelayNotify_Key"), "OLD");
179             assertEquals (sp.inp ("TestDelayNotify_Key"), "NEW");
180         }
