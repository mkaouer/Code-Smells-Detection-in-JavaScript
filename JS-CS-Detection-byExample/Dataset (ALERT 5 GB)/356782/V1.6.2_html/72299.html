19     package org.jpos.q2.cli;
20     
21     import org.jpos.q2.CLI;
22     import java.util.Map;
23     import java.util.Iterator;
24     import org.jpos.iso.ISOUtil;
25     import org.jpos.util.Loggeable;
26     import org.jpos.util.NameRegistrar;
27     
28     public class SHOWNR implements CLI.Command {
29         public void exec (CLI cli, String[] args) throws Exception {
30             boolean all = args.length > 1 && "-a".equals (args[1]);
31             int i = 1;
32             if (all)
33                 i++;
34             if (args.length > i)
35                 showOne (cli, args[i], all);
36             else
37                 showAll (cli, all);
38         }
39         private void showOne (CLI cli, String name, boolean detail) {
40             try {
41                 Object obj = NameRegistrar.get (name);
42                 cli.println (name + " : " + obj.toString());
43                 if (detail && obj instanceof Loggeable) {
44                     ((Loggeable)obj).dump (cli.getOutputStream(), "   ");
45                     cli.getOutputStream().flush();
46                 }
47             } catch (NameRegistrar.NotFoundException e) {
48                 cli.println ("Object not found in NameRegistrar");
49             }
50         }
51         private void showAll (CLI cli, boolean detail) {
52             NameRegistrar nr = NameRegistrar.getInstance();
53             int maxw = 0;
54             Iterator iter = nr.getMap().entrySet().iterator();
55             while (iter.hasNext()) {
56                 Map.Entry entry = (Map.Entry) iter.next ();
57                 maxw = Math.max (maxw, entry.getKey().toString().length());
58             }
59             iter = nr.getMap().entrySet().iterator();
60             maxw++;
61             while (iter.hasNext()) {
62                 Map.Entry entry = (Map.Entry) iter.next ();
63                 cli.println (
64                     ISOUtil.strpad (entry.getKey().toString(), maxw) + 
65                     entry.getValue().toString()
66                 );
67                 if (detail && entry.getValue() instanceof Loggeable) {
68                     ((Loggeable)entry.getValue()).dump (cli.getOutputStream(), "   ");
69                     cli.getOutputStream().flush();
70                 }
71             }
72         }
73     }
