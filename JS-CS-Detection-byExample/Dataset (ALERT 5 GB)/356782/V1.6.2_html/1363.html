19     package org.jpos.iso;
20     
21     /**
22      * This interpreter does no conversion and leaves the input the same as the output.
23      * @author jonathan.oconnor@xcom.de
24      */
25     public class LiteralBinaryInterpreter implements BinaryInterpreter
26     {
27         /**
28          * The only instance of this interpreter.
29          */
30         public static final LiteralBinaryInterpreter INSTANCE = new LiteralBinaryInterpreter();
31     
32         /**
33          * Private constructor so we don't allow multiple instances.
34          */
35         private LiteralBinaryInterpreter()
36         {
37         }
38     
39         /**
40          * Copies the input to the output.
41          * @see org.jpos.iso.BinaryInterpreter#interpret(byte[], byte[], int)
42          */
43         public void interpret(byte[] data, byte[] b, int offset)
44         {
45             System.arraycopy(data, 0, b, offset, data.length);
46         }
47     
48         /**
49          * Copies the data out of the byte array.
50          * @see org.jpos.iso.BinaryInterpreter#uninterpret(byte[], int, int)
51          */
52         public byte[] uninterpret(byte[] rawData, int offset, int length)
53         {
54             byte[] ret = new byte[length];
55             System.arraycopy(rawData, offset, ret, 0, length);
56             return ret;
57         }
58     
59         /**
60          * Returns nBytes because we are not doing any conversion.
61          * @see org.jpos.iso.BinaryInterpreter#getPackedLength(int)
62          */
63         public int getPackedLength(int nBytes)
64         {
65             // TODO Auto-generated method stub
66             return nBytes;
67         }
68     }
