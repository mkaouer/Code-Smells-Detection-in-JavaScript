127         public void start () 
128             throws MalformedObjectNameException,
129                    InstanceAlreadyExistsException,
130                    NotCompliantMBeanException,
131                    MBeanRegistrationException
132         {
133             /*
134              * The following code determines whether a MBeanServer exists already.
135              * If so then the first one in the list is used.  I have not yet find a way to
136              * interrogate the server for information other than MBeans so to pick a
137              * specific one would be difficult.
138              */
139             ArrayList mbeanServerList = MBeanServerFactory.findMBeanServer(null);
140             if (mbeanServerList.size() == 0) {
141                 server  = MBeanServerFactory.createMBeanServer (JMX_NAME);
142             } else {
143                 server = (MBeanServer) mbeanServerList.get(0);
144             }
145             final ObjectName loaderName = new ObjectName (Q2_CLASS_LOADER);
146             try {
147                 loader = (QClassLoader) java.security.AccessController.doPrivileged(
148                     new java.security.PrivilegedAction() {
149                         public Object run() {
150                             return new QClassLoader (server, libDir, loaderName, mainClassLoader);
151                         }
152                     }
153                 );
154                 server.registerMBean (loader, loaderName);
155                 loader = loader.scan(false);
156             } catch (Throwable t) {
157                 if (log != null)
158                     log.error ("initial-scan", t);
159                 else
160                     t.printStackTrace();
161             }
162             factory = new QFactory (loaderName, this);
163             initSystemLogger ();
164             addShutdownHook ();
165             q2Thread = Thread.currentThread ();
166             q2Thread.setContextClassLoader (loader);
167             if (cli != null)
168                 cli.start();
169             while (!shutdown) {
170                 try {
171                     boolean forceNewClassLoader = scan ();
172                     QClassLoader oldClassLoader = loader;
173                     loader = loader.scan (forceNewClassLoader);
174                     if (loader != oldClassLoader) {
175                         oldClassLoader = null;
176                         System.gc();  // force a GC
177                         log.info (
178                           "new classloader ["
179                           + Integer.toString(loader.hashCode(),16)
180                           + "] has been created"
181                         );
182                     }
183                     deploy ();
184                     checkModified ();
185                     relax (SCAN_INTERVAL);
186                 } catch (Throwable t) {
187                     log.error ("start", t);
188                     relax ();
189                 }
190             }
191             undeploy ();
192             try {
193                 server.unregisterMBean (loaderName);
194             } catch (InstanceNotFoundException e) {
195                 log.error (e);
196             }
197             if (exit && !shuttingDown)
198                 System.exit (0);
199         }

240         private boolean scan () {
241                 boolean rc = false;
242                 File file[] = deployDir.listFiles (this);
243                 // Arrays.sort (file); --apr not required - we use TreeMap
244                 if (file == null) {
245                     // Shutting down might be best, how to trigger from within?
246                     throw new Error("Deploy directory \""+deployDir.getAbsolutePath()+"\" is not available");
247                 } else {
248                     for (int i=0; i<file.length; i++) {
249                         if (register (file[i]))
250                             rc = true;
251                     }
252                 }
253                 return rc;
254         }

256         private void deploy () {
257             List startList = new ArrayList ();
258             Iterator iter = dirMap.entrySet().iterator();
259             try {
260                 while (iter.hasNext() && !shutdown) {
261                     Map.Entry entry = (Map.Entry) iter.next();
262                     File   f        = (File)   entry.getKey ();
263                     QEntry qentry   = (QEntry) entry.getValue ();
264                     long deployed   = qentry.getDeployed ();
265                     if (deployed == 0) {
266                         if (deploy (f)) {
267                             if (qentry.isQBean ())
268                                 startList.add (qentry.getInstance());
269                             qentry.setDeployed (f.lastModified ());
270                         } else {
271                             // deploy failed, clean up.
272                             iter.remove();
273                         }
274                     } else if (deployed != f.lastModified ()) {
275                         undeploy (f);
276                         iter.remove ();
277                         loader.forceNewClassLoaderOnNextScan();
278                     }
279                 }
280                 iter = startList.iterator();
281                 while (iter.hasNext ()) {
282                     start ((ObjectInstance) iter.next ());
283                 }
284             }
285             catch (Exception e){
286                 log.error ("deploy", e);
287             }
288         }

290         private void undeploy () {
291             Object[] set = dirMap.entrySet().toArray ();
292             int l = set.length;
293     
294             while (l-- > 0) {
295                 Map.Entry entry = (Map.Entry) set[l];
296                 File   f  = (File) entry.getKey ();
297                 undeploy (f);
298             }
299         }

301         private void addShutdownHook () {
302             Runtime.getRuntime().addShutdownHook (
303                 new Thread ("Q2-ShutdownHook") {
304                     public void run () {
305                         shuttingDown = true;
306                         shutdown = true;
307                         if (q2Thread != null) {
308                             log.info ("shutting down (hook)");
309                             try {
310                                 q2Thread.join (SHUTDOWN_TIMEOUT);
311                             } catch (InterruptedException e) { 
312                             } catch (NullPointerException e) {
313                                 // on thin Q2 systems where shutdown is very fast, 
314                                 // q2Thread can become null between the upper if and
315                                 // the actual join. Not a big deal so we ignore the
316                                 // exception.
317                             }
318                         }
319                     }
320                 }
321             );
322         }

324         private void checkModified () {
325             Iterator iter = dirMap.entrySet().iterator();
326             while (iter.hasNext()) {
327                 Map.Entry entry = (Map.Entry) iter.next();
328                 File   f        = (File)   entry.getKey ();
329                 QEntry qentry   = (QEntry) entry.getValue ();
330                 if (qentry.isQBean() && qentry.isQPersist()) {
331                     ObjectName name = qentry.getObjectName ();
332                     if (getState (name) == QBean.STARTED && isModified (name)) {
333                         qentry.setDeployed (persist (f, name));
334                     }
335                 }
336             }
337         }

365         private long persist (File f, ObjectName name) {
366             long deployed = f.lastModified ();
367             try {
368                 Element e = (Element) server.getAttribute (name, "Persist");
369                 if (e != null) {
370                     XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
371                     Document doc = new Document ();
372                     e.detach();
373                     doc.setRootElement (e);
374                     File tmp = new File (f.getAbsolutePath () + ".tmp");
375                     FileWriter writer = new FileWriter (tmp);
376                     out.output (doc, writer);
377                     writer.close ();
378                     f.delete();
379                     tmp.renameTo (f);
380                     deployed = f.lastModified ();
381                 }
382             } catch (Exception ex) {
383                 log.warn ("persist", ex);
384             }
385             return deployed;
386         }

388         private void undeploy (File f) {
389             QEntry qentry = (QEntry) dirMap.get (f);
390             try {
391                 if (log != null)
392                     log.trace ("undeploying:" + f.getCanonicalPath());
393     
394                 Object obj      = qentry.getObject ();
395                 ObjectName name = qentry.getObjectName ();
396                 factory.destroyQBean (this, name, obj);
397                 if (log != null)
398                     log.info ("undeployed:" + f.getCanonicalPath());
399             } catch (Exception e) {
400                 getLog().warn ("undeploy", e);
401             }
402         }

404         private boolean register (File f) {
405             boolean rc = false;
406             if (f.isDirectory()) {
407                 File file[] = f.listFiles (this);
408                 for (int i=0; i<file.length; i++) {
409                     if (register (file[i]))
410                         rc = true;
411                 }
412             } else if (dirMap.get (f) == null) {
413                 dirMap.put (f, new QEntry ());
414                 rc = true;
415             }
416             return rc;
417         }

419         private boolean deploy (File f) {
420             try {
421                 if (log != null)
422                     log.info ("deploy:" + f.getCanonicalPath());
423                 QEntry qentry = (QEntry) dirMap.get (f);
424                 SAXBuilder builder = new SAXBuilder ();
425                 Document doc = decrypt (builder.build (f));
426     
427                 Object obj = factory.instantiate (this, doc.getRootElement ());
428                 qentry.setObject (obj);
429     
430                 ObjectInstance instance = factory.createQBean (
431                     this, doc.getRootElement(), obj
432                 );
433                 qentry.setInstance (instance);
434             } 
435             catch (InstanceAlreadyExistsException e) {
436                /*
437                 * Ok, the file we tried to deploy, holds an object
438                 *  that already has been deployed.
439                 *  
440                 * Rename it out of the way.
441                 * 
442                 */
443                 tidyFileAway(f,DUPLICATE_EXTENSION);
444                 getLog().warn ("deploy", e);
445                 return false;
446             }
447             catch (Exception e) {
448                 getLog().warn ("deploy", e);
449                 tidyFileAway(f,ERROR_EXTENSION);
450                 // This will also save deploy error repeats...
451                 return false;
452             } 
453             catch (Error e) {
454                 getLog().warn ("deploy", e);
455                 tidyFileAway(f,ENV_EXTENSION);
456                 // This will also save deploy error repeats...
457                 return false;
458             }
459             return true ;
460         }

505         private void parseCmdLine (String[] args) {
506             CommandLineParser parser = new PosixParser ();
507     
508             Options options = new Options ();
509             options.addOption ("v","version", false, "Q2's version");
510             options.addOption ("d","deploydir", true, "Deployment directory");
511             options.addOption ("r","recursive", false, "Deploy subdirectories recursively");
512             options.addOption ("h","help", false, "Usage information");
513             options.addOption ("C","config", true, "Configuration bundle");
514             options.addOption ("e","encrypt", true, "Encrypt configuration bundle");
515             options.addOption ("i","cli", false, "Command Line Interface");
516             options.addOption ("c","command", true, "Command to execute");
517     
518             try {
519                 CommandLine line = parser.parse (options, args);
520                 if (line.hasOption ("v")) {
521                     System.out.println (
522                         "Q2 version: " +Q2_VERSION + "(" + Q2_REVISION + ")"
523                     );
524                     System.exit (0);
525                 } 
526                 if (line.hasOption ("h")) {
527                     HelpFormatter helpFormatter = new HelpFormatter ();
528                     helpFormatter.printHelp ("Q2", options);
529                     System.exit (0);
530                 } 
531                 if (line.hasOption ("c")) {
532                     cli = new CLI(this, line.getOptionValue("c"), line.hasOption("i"));
533                 } else if (line.hasOption ("i")) 
534                     cli = new CLI(this, null, true);
535     
536                 String dir = DEFAULT_DEPLOY_DIR;
537                 if (line.hasOption ("d")) {
538                     dir = line.getOptionValue ("d");
539                 }
540                 recursive = line.hasOption ("r");
541                 this.deployDir  = new File (dir);
542                 if (line.hasOption ("C"))
543                     deployBundle (new File (line.getOptionValue ("C")), false);
544                 if (line.hasOption ("e"))
545                     deployBundle (new File (line.getOptionValue ("e")), true);
546             } catch (MissingArgumentException e) {
547                 System.out.println ("ERROR: " + e.getMessage());
548                 System.exit (1);
549             } catch (Exception e) {
550                 e.printStackTrace ();
551                 System.exit (1);
552             }
553         }
554         private void deployBundle (File bundle, boolean encrypt) 
555             throws JDOMException, IOException, 
556                     ISOException, GeneralSecurityException
557         {
558             SAXBuilder builder = new SAXBuilder ();
559             Document doc = builder.build (bundle);
560             Iterator iter = doc.getRootElement().getChildren ().iterator ();
561             for (int i=0; iter.hasNext (); i += 5) {
562                 deployElement ((Element) iter.next (), i, encrypt);
563             }
564         }
565         private void deployElement (Element e, int i, boolean encrypt) 
566             throws ISOException, IOException, GeneralSecurityException
567         {
568             e = ((Element) e.clone ());
569     
570             XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
571             Document doc = new Document ();
572             doc.setRootElement (e);
573             File qbean = new File (
574                 deployDir, ISOUtil.zeropad (Integer.toString (i),3) + "_" 
575                 + e.getName () + ".xml"
576             );
577             FileWriter writer = new FileWriter (qbean);
578             if (encrypt)
579                 doc = encrypt (doc);
580             out.output (doc, writer);
581             writer.close ();
582         }

584         private byte[] dodes (byte[] data, int mode) 
585            throws GeneralSecurityException
586         {
587             Cipher cipher = Cipher.getInstance ("DES");
588             cipher.init (mode, new SecretKeySpec(getKey(), "DES"));
589             return cipher.doFinal (data);
590         }
591         protected byte[] getKey() {
592             return "CAFEBABE".getBytes();
593         }
594         protected Document encrypt (Document doc)
595             throws GeneralSecurityException, IOException
596         {
597             ByteArrayOutputStream os = new ByteArrayOutputStream ();
598             OutputStreamWriter writer = new OutputStreamWriter (os);
599             XMLOutputter out = new XMLOutputter (Format.getPrettyFormat());
600             out.output (doc, writer);
601             writer.close ();
602     
603             byte[] crypt = dodes (os.toByteArray(), Cipher.ENCRYPT_MODE);
604     
605             Document secureDoc = new Document ();
606             Element root = new Element (PROTECTED_QBEAN);
607             secureDoc.setRootElement (root);
608             Element secureData = new Element ("data");
609             root.addContent (secureData);
610     
611             secureData.setText (
612                 ISOUtil.hexString (crypt)
613             );
614             return secureDoc;
615         }
616     
617         protected Document decrypt (Document doc) 
618             throws GeneralSecurityException, IOException, JDOMException
619         {
620             Element root = doc.getRootElement ();
621             if (PROTECTED_QBEAN.equals (root.getName ())) {
622                 Element data = root.getChild ("data");
623                 if (data != null) {
624                     ByteArrayInputStream is = new ByteArrayInputStream (
625                         dodes (
626                             ISOUtil.hex2byte (data.getTextTrim()),
627                             Cipher.DECRYPT_MODE)
628                     );
629                     SAXBuilder builder = new SAXBuilder ();
630                     doc = builder.build (is);
631                 }
632             }
633             return doc;
634         }

636         private void tidyFileAway (File f, String extension) {
637             File rename = new File(f.getAbsolutePath()+"."+extension);
638             while (rename.exists()){
639                 rename = new File(rename.getAbsolutePath()+"."+extension);
640             }
641             getLog().warn("Tidying "+f.getAbsolutePath()+" out of the way, by adding ."+extension,"It will be called: "+rename.getAbsolutePath()+" see log above for detail of problem.");
642             f.renameTo(rename);
643         }
