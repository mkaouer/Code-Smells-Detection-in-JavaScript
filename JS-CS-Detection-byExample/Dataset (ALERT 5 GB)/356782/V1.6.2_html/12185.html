49         private ISOMsg getMsg (String message) throws Exception {
50             FileInputStream fis = new FileInputStream (PREFIX + message + ".xml");
51             byte[] b = new byte[fis.available()];
52             fis.read (b);
53             ISOMsg m = new ISOMsg ();
54             m.setPackager (xmlPackager);
55             m.unpack (b);
56             fis.close();
57             return m;
58         }
59         private byte[] getImage (String message) throws Exception {
60             FileInputStream fis = new FileInputStream (PREFIX + message + ".bin");
61             byte[] b = new byte[fis.available()];
62             fis.read (b);
63             fis.close();
64             return b;
65         }
66     
67         private void writeImage (String message, byte[] b) throws Exception {
68             FileOutputStream fos = new FileOutputStream (PREFIX + message + ".run");
69             fos.write (b);
70             fos.close();
71         }

126         private void doTest (ISOPackager packager, String msg, String img)
127             throws Exception
128         {
129             // Logger logger = new Logger();
130             // logger.addListener (new SimpleLogListener (System.out));
131             // packager.setLogger (logger, msg + "-m");
132     
133             ISOMsg m = getMsg (msg);
134             m.setPackager (packager);
135             byte[] p = m.pack();
136             ByteArrayOutputStream out = new ByteArrayOutputStream ();
137             m.pack (out);
138     
139             assertTrue (Arrays.equals (out.toByteArray(), p));
140     
141             writeImage (img, p);
142     
143             byte[] b = getImage (img);
144             TestUtils.assertEquals(b, p);
145     
146             ISOMsg m1 = new ISOMsg ();
147             // packager.setLogger (logger, msg + "-m1");
148             m1.setPackager (packager);
149             m1.unpack (b);
150             TestUtils.assertEquals(b, m1.pack());
151     
152             ISOMsg m2 = new ISOMsg ();
153             m2.setPackager (packager);
154             // packager.setLogger (logger, msg + "-m2");
155             m2.unpack (new ByteArrayInputStream (out.toByteArray()));
156             TestUtils.assertEquals(b, m2.pack());
157         }
