108         /**
109          * @param m message to send
110          * @param timeout amount of time in millis to wait for a response
111          * @return response or null
112          */
113         public ISOMsg request (ISOMsg m, long timeout) throws ISOException {
114             String key = getKey (m);
115             String req = key + ".req";
116             sp.out (req, m);
117             m.setDirection(0);
118             if (timeout > 0)
119                 sp.out (out, m, timeout);
120             else
121                 sp.out (out, m);
122     
123             ISOMsg resp = null;
124             try {
125                 synchronized (this) { tx++; rxPending++; }
126     
127                 for (;;) {
128                     resp = (ISOMsg) sp.rd (key, timeout);
129                     if (shouldIgnore (resp)) 
130                         continue;
131                     sp.inp (key);
132                     break;
133                 } 
134                 if (resp == null && sp.inp (req) == null) {
135                     // possible race condition, retry for a few extra seconds
136                     resp = (ISOMsg) sp.in (key, 10000);
137                 }
138                 synchronized (this) {
139                     if (resp != null) 
140                         rx++;
141                     else {
142                         rxExpired++;
143                         if (m.getDirection() != ISOMsg.OUTGOING)
144                             txExpired++;
145                     }
146                 }
147             } finally {
148                 synchronized (this) { rxPending--; }
149             }
150             return resp;
151         }
152         public void notify (Object k, Object value) {
153             Object obj = sp.inp (k);
154             if (obj instanceof ISOMsg) {
155                 ISOMsg m = (ISOMsg) obj;
156                 try {
157                     String key = getKey (m);
158                     String req = key + ".req";
159                     Object r = sp.inp (req);
160                     if (r != null) {
161                         if (r instanceof AsyncRequest) {
162                             ((AsyncRequest) r).responseReceived (m);
163                         } else {
164                             sp.out (key, m);
165                         }
166                         return;
167                     }
168                 } catch (ISOException e) { 
169                     getLog().warn ("notify", e);
170                 }
171                 processUnhandled (m);
172             }
173         }

175         public String getKey (ISOMsg m) throws ISOException {
176             StringBuffer sb = new StringBuffer (out);
177             sb.append ('.');
178             sb.append (mapMTI(m.getMTI()));
179             for (int i=0; i<key.length; i++) {
180                 int f = key[i];
181                 String v = m.getString(f);
182                 if (v != null) {
183                     if (f == 41) 
184                         v = ISOUtil.zeropad (v.trim(), 16); // BIC ANSI to ISO hack
185                     else if (f == 11) 
186                         v = ISOUtil.zeropad (v.trim(), m.getMTI().charAt(0)=='2' ? 12 : 6); 
187                     sb.append (v);
188                 }
189             }
190             return sb.toString();
191         }
192         private String mapMTI (String mti) {
193             StringBuffer sb = new StringBuffer();
194             if (mti != null && mti.length() == 4) {
195                 for (int i=0; i<mtiMapping.length; i++) {
196                     int c = mti.charAt (i) - '0';
197                     if (c >= 0 && c < 10) 
198                         sb.append (mtiMapping[i].charAt(c));
199                 }
200             }
201             return sb.toString();
202         }

249         public void request (ISOMsg m, long timeout, ISOResponseListener rl, Object handBack)
250             throws ISOException 
251         {
252             String key = getKey (m);
253             String req = key + ".req";
254             m.setDirection(0);
255             AsyncRequest ar = new AsyncRequest (rl, handBack);
256             synchronized (ar) {
257                 if (timeout > 0)
258                     DefaultTimer.getTimer().schedule (ar, timeout);
259             }
260             sp.out (req, ar, timeout);
261             sp.out (out, m, timeout);
262         }
263         private void addListeners () 
264             throws ConfigurationException
265         {
266             QFactory factory = getFactory ();
267             Iterator iter = getPersist().getChildren (
268                 "request-listener"
269             ).iterator();
270             while (iter.hasNext()) {
271                 Element l = (Element) iter.next();
272                 ISORequestListener listener = (ISORequestListener) 
273                     factory.newInstance (l.getAttributeValue ("class"));
274                 factory.setLogger        (listener, l);
275                 factory.setConfiguration (listener, l);
276                 addISORequestListener (listener);
277             }
278         }

288         public String getCountersAsString () {
289             StringBuffer sb = new StringBuffer();
290             append (sb, "tx=", tx);
291             append (sb, ", rx=", rx);
292             append (sb, ", tx_expired=", txExpired);
293             append (sb, ", tx_pending=", sp.size(out));
294             append (sb, ", rx_expired=", rxExpired);
295             append (sb, ", rx_pending=", rxPending);
296             append (sb, ", rx_unhandled=", rxUnhandled);
297             append (sb, ", rx_forwarded=", rxForwarded);
298             sb.append (", connected=");
299             sb.append (Boolean.toString(isConnected()));
300             return sb.toString();
301         }
302         protected void processUnhandled (ISOMsg m) {
303             ISOSource source = m.getSource ();
304             if (source != null) {
305                 Iterator iter = listeners.iterator();
306                 if (iter.hasNext())
307                     synchronized (this) { rxForwarded++; }
308                 while (iter.hasNext())
309                     if (((ISORequestListener)iter.next()).process (source, m))
310                         return;
311             }
312             if (unhandled != null) {
313                 synchronized (this) { rxUnhandled++; }
314                 sp.out (unhandled, m, 120000);
315             }
316         }
317         private LocalSpace grabSpace (Element e) 
318             throws ConfigurationException
319         {
320             String uri = e != null ? e.getText() : "";
321             Space sp = SpaceFactory.getSpace (uri);
322             if (sp instanceof LocalSpace) {
323                 return (LocalSpace) sp;
324             }
325             throw new ConfigurationException ("Invalid space " + uri);
326         }
327         public boolean isConnected() {
328             if (ready != null && ready.length > 0) {
329                 for (int i=0; i<ready.length; i++)
330                     if (sp.rdp (ready[i]) != null)
331                         return true;
332                 return false;
333             }
334             else
335                 return true;
336         }

340         private String[] toStringArray (String s) {
341             String[] ready = null;
342             if (s != null && s.length() > 0) {
343                 StringTokenizer st = new StringTokenizer (s);
344                 ready = new String[st.countTokens()];
345                 for (int i=0; st.hasMoreTokens(); i++)
346                     ready[i] = st.nextToken();
347             }
348             return ready;
349         }
350         private int[] toIntArray (String s) 
351             throws ConfigurationException
352         {
353             if (s == null || s.length() == 0)
354                 s = "41, 11";
355             try {
356                 int[] k = null;
357                 StringTokenizer st = new StringTokenizer (s, ", ");
358                 k = new int[st.countTokens()];
359                 for (int i=0; st.hasMoreTokens(); i++)
360                     k[i] = Integer.parseInt(st.nextToken());
361                 return k;
362             } catch (NumberFormatException e) {
363                 throw new ConfigurationException (e);
364             }
365         }
366         private boolean shouldIgnore (ISOMsg m) {
367             if (m != null && ignorerc != null 
368                 && ignorerc.length() > 0 && m.hasField(39))
369             {
370                 return ignorerc.indexOf(m.getString(39)) >= 0;
371             }
372             return false;
373         }

378         public class AsyncRequest extends TimerTask {
379             ISOResponseListener rl;
380             Object handBack;
381             public AsyncRequest (ISOResponseListener rl, Object handBack) {
382                 super();
383                 this.rl = rl;
384                 this.handBack = handBack;
385             }
386             public void responseReceived (ISOMsg response) {
387                 cancel();
388                 ISOResponseListener _rl;
389                 synchronized (this) {
390                     _rl = rl;
391                     rl = null;
392                 }
393                 if (_rl != null)
394                     _rl.responseReceived (response, handBack);
395             }
396             public void run() {
397                 cancel();
398                 ISOResponseListener _rl;
399                 synchronized (this) {
400                     _rl = rl;
401                     rl = null;
402                 }
403                 if (_rl != null)
404                     _rl.expired(handBack);
405             }
406         }
