23     import java.awt.event.ActionListener;

26     import java.util.Map;
27     import java.util.StringTokenizer;

29     import javax.swing.JComponent;
30     import javax.swing.JTree;
31     import javax.swing.event.TreeSelectionEvent;
32     import javax.swing.event.TreeSelectionListener;
33     import javax.swing.tree.DefaultMutableTreeNode;
34     
35     import org.jdom.Element;
36     import org.jpos.ui.UI;
37     import org.jpos.ui.UIFactory;
38     
39     /**
40      * @author Alejandro Revilla
41      *
42      * Creates a JTree
43      * i.e:
44      * <pre>
45      *  &lt;tree&gt;
46      *   &lt;node&gt;
47      *    ...
48      *   &lt;/node&gt;
49      *   ...
50      *   ...
51      *   &lt;node&gt;
52      *    ...
53      *   &lt;/node&gt;
54      *  &lt;/tree&gt;
55      * </pre>
56      * @see org.jpos.ui.UIFactory
57      */
58     public class JTreeFactory implements UIFactory {
59         public JComponent create (UI ui, Element e) {
60             final UI parentUI = ui;
61             final Map map = new HashMap ();
62             final JTree tree = new JTree (getNode (e, map));
63             String font = e.getAttributeValue ("font");
64             if (font != null) 
65                 tree.setFont (Font.decode (font));
66             tree.setRootVisible (e.getTextTrim().length() > 0);
67     
68             tree.addTreeSelectionListener (
69                 new TreeSelectionListener() {
70                     public void valueChanged(TreeSelectionEvent evt) {
71                         DefaultMutableTreeNode node = 
72                             (DefaultMutableTreeNode) 
73                                 tree.getLastSelectedPathComponent();
74                         if (node != null) {
75                             String s = (String) map.get (node);
76                             if (s != null) {
77                                 StringTokenizer st = new StringTokenizer (s);
78                                 String action  = st.nextToken ();
79                                 String command = null;
80                                 if (st.hasMoreTokens ())
81                                     command = st.nextToken ();
82     
83                                 ActionListener al = 
84                                     (ActionListener) parentUI.get(action);
85                                 if (al != null) {
86                                     al.actionPerformed (
87                                         new ActionEvent (node, 0, command)
88                                     );
89                                 }
90                                 // System.out.println ("ACTION: " + action);
91                                 // System.out.println ("COMMAND: " + command);
92                             }
93                         }
94                     }
95                 }
96             );
97             return tree;
98         }
99     
100         private DefaultMutableTreeNode getNode (Element e, Map map) { 
101             DefaultMutableTreeNode node = new DefaultMutableTreeNode (
102                     e.getTextTrim()
103             );
104             String action = e.getAttributeValue ("action");
105             if (action != null) {
106                 String command = e.getAttributeValue ("command");
107                 if (command != null)
108                     action = action + " " + command;
109     
110                 map.put (node, action);
111             }
112             Iterator iter = e.getChildren().iterator();
113             while (iter.hasNext()) {
114                 node.add (getNode ((Element) iter.next (), map));
115             }
116             return node;
117         }
118     }
