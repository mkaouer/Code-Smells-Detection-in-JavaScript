19     package org.jpos.util;
20     
21     import java.io.ByteArrayOutputStream;
22     import java.io.IOException;
23     import java.io.PrintStream;
24     import java.sql.SQLException;
25     import java.util.Iterator;
26     import java.util.Date;
27     import java.util.Vector;
28     
29     import org.jdom.Element;
30     import org.jdom.output.Format;
31     import org.jdom.output.XMLOutputter;
32     
33     /**
34      * @author apr@cs.com.uy
35      * @version $Id$
36      * @serial
37      */
38     public class LogEvent {
39        /**
40         * @serial
41         */
42         LogSource source;
43        /**
44         * @serial
45         */
46         public String tag;
47        /**
48         * @serial
49         */
50         public Vector payLoad;
51         
52         public LogEvent () {
53             super();
54             this.tag = "info";
55             this.payLoad = new Vector(1);
56         }
57         public LogEvent (String tag) {
58             super();
59             this.tag = tag;
60             this.payLoad = new Vector(1);
61         }
62         public LogEvent (String tag, Object msg) {
63             super();
64             this.tag     = tag;
65             this.payLoad = new Vector(1);
66             addMessage(msg);
67         }
68         public LogEvent (LogSource source, String tag) {
69             super ();
70             this.source  = source;
71             this.tag     = tag;
72             this.payLoad = new Vector();
73         }
74         public LogEvent (LogSource source, String tag, Object msg) {
75             super ();
76             this.source  = source;
77             this.tag     = tag;
78             this.payLoad = new Vector(1);
79             addMessage(msg);
80         }
81         public void addMessage (Object msg) {
82             this.payLoad.addElement(msg);
83         }
84         public void addMessage (String tagname, String message) {
85             this.payLoad.addElement("<"+tagname+">"+message+"</"+tagname+">");
86         }
87         /**
88          * @return log source (may be null)
89          */
90         public LogSource getSource() {
91             return source;
92         }
93         /**
94          * @param source a LogSource
95          */
96         public void setSource(LogSource source) {
97             this.source = source;
98         }
99         protected String dumpHeader (PrintStream p, String indent) {
100             Date d = new Date();
101             p.println (indent + "<log realm=\"" +getRealm()+ "\" at=\""+d.toString()
102                +"." + d.getTime() % 1000 + "\">"
103             );
104             return indent + "  ";
105         }
106         protected void dumpTrailer (PrintStream p, String indent) {
107             p.println (indent + "</log>");
108         }
109         public void dump (PrintStream p, String outer) {
110             String indent = dumpHeader (p, outer);
111             if (payLoad.size() == 0) {
112                 if (tag != null)
113                     p.println (indent + "<" + tag + "/>");
114             }
115             else {
116                 String newIndent;
117                 if (tag != null) {
118                     p.println (indent + "<" + tag + ">");
119                     newIndent = indent + "  ";
120                 } else
121                     newIndent = "";
122                 Iterator i = payLoad.iterator();
123                 while (i.hasNext()) {
124                     Object o = i.next();
125                     if (o instanceof Loggeable) 
126                         ((Loggeable)o).dump (p, newIndent);
127                     else if (o instanceof SQLException) {
128                         SQLException e = (SQLException) o;
129                         p.println (newIndent + "<SQLException>"
130                                     +e.getMessage() + "</SQLException>");
131                         p.println (newIndent + "<SQLState>"
132                                     +e.getSQLState() + "</SQLState>");
133                         p.println (newIndent + "<VendorError>"
134                                     +e.getErrorCode() + "</VendorError>");
135                         ((Throwable)o).printStackTrace (p);
136                     }
137                     else if (o instanceof Throwable) {
138                         p.println (newIndent + "<exception name=\""
139                             + ((Throwable)o).getMessage() + "\">");
140                         p.print (newIndent);
141                         ((Throwable)o).printStackTrace (p);
142                         p.println (newIndent + "</exception>");
143                     }
144                     else if (o instanceof Object[]) {
145                         Object [] oa = (Object[]) o;
146                         p.print (newIndent + "[");
147                         for (int j=0; j<oa.length; j++) {
148                             if (j>0)
149                                 p.print (",");
150                             p.print (oa[j].toString());
151                         }
152                         p.println ("]");
153                     } 
154                     else if (o instanceof Element) {
155                         p.println ("");
156                         p.println (newIndent + "<![CDATA[");
157                         XMLOutputter out = new XMLOutputter (Format.getPrettyFormat ());
158                         out.getFormat().setLineSeparator ("\n");
159                         try {
160                             out.output ((Element) o, p);
161                         } catch (IOException ex) {
162                             ex.printStackTrace (p);
163                         }
164                         p.println ("");
165                         p.println (newIndent + "]]>");
166                     }
167                     else if (o != null) {
168                         p.println (newIndent + o.toString());
169                     } else {
170                         p.println (newIndent + "null");
171                     }
172                 }
173                 if (tag != null)
174                     p.println (indent + "</" + tag + ">");
175             }
176             dumpTrailer (p, outer);
177         }
178         public String getRealm() {
179             return source != null ? source.getRealm() : "";
180         }
181         public Vector getPayLoad() {
182             return payLoad;
183         }
184         public String toString() {
185             ByteArrayOutputStream baos = new ByteArrayOutputStream();
186             PrintStream p = new PrintStream (baos);
187             dump (p, "");
188             return baos.toString();
189         }
190     }
