19     package org.jpos.iso;
20     
21     import java.io.IOException;
22     
23     /**
24      * allows the transmision and reception of ISO 8583 Messages
25      *
26      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
27      * @author <a href="mailto:alwynschoeman@yahoo.com">Alwyn Schoeman</a>
28      * @version $Revision$ $Date$
29      */
30     public interface ISOChannel extends ISOSource {
31         public static final int CONNECT      = 0;
32         public static final int TX           = 1;
33         public static final int RX           = 2;
34         public static final int SIZEOF_CNT   = 3;
35     
36         /**
37          * Associate a packager with this channel
38          * @param p     an ISOPackager
39          */
40         public void setPackager(ISOPackager p);
41     
42         /**
43          * Connects ISOChannel 
44          * @exception IOException
45          */
46         public void connect () throws IOException;
47     
48         /**
49          * disconnects ISOChannel
50          * @exception IOException
51          */
52         public void disconnect () throws IOException;
53     
54         /**
55          * Reconnect channel
56          * @exception IOException
57          */
58         public void reconnect() throws IOException;
59     
60         /**
61          * @return true if Channel is connected and usable
62          */
63         public boolean isConnected();
64     
65         /**
66          * Receives an ISOMsg
67          * @return the Message received
68          * @exception IOException
69          * @exception ISOException
70          */
71         public ISOMsg receive() throws IOException, ISOException;
72     
73         /**
74          * sends an ISOMsg over the TCP/IP session
75          * @param m the Message to be sent
76          * @exception IOException
77          * @exception ISOException
78          */
79         public void send (ISOMsg m) throws IOException, ISOException;
80     
81         /**
82          * @param b - usable state
83          */
84         public void setUsable(boolean b);
85     
86         /**
87          * associates this ISOChannel with a name on NameRegistrar
88          * @param name name to register
89          * @see NameRegistrar
90          */
91         public void setName (String name);
92     
93        /**
94         * @return this ISOChannel's name ("" if no name was set)
95         */
96         public String getName();
97     
98        /**
99         * @return current packager
100         */
101         public ISOPackager getPackager();
102     
103        /**
104         * Expose channel clonning interface
105         */
106         public Object clone();
107         
108     }
