19     package org.jpos.util;
20     
21     import java.io.PrintStream;
22     import java.util.Hashtable;
23     import java.util.Iterator;
24     import java.util.Map;
25     
26     /**
27      * Allow runtime binding of jPOS's components (ISOChannels, Logger, MUXes, etc)
28      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
29      * @version $Revision$ $Date$
30      */
31     public class NameRegistrar implements Loggeable {
32         private static NameRegistrar instance = new NameRegistrar();
33         private Map registrar;
34     
35         public static class NotFoundException extends Exception {
36     
37             private static final long serialVersionUID = 8744022794646381475L;
38             public NotFoundException() {
39                 super();
40             }
41             public NotFoundException(String detail) {
42                 super(detail);
43             }
44         }
45     
46         private NameRegistrar() {
47             super();
48             registrar = new Hashtable();
49         }
50         public static Map getMap() {
51             return getInstance().registrar;
52         }
53         /**
54          * @return singleton instance
55          */
56         public static NameRegistrar getInstance() {
57             return instance;
58         }
59         /**
60          * register object
61          * @param key - key with which the specified value is to be associated.
62          * @param value - value to be associated with the specified key
63          */
64         public static void register (String key, Object value) {
65             Map map = getMap();
66             synchronized (map) {
67                 map.put (key, value);
68             }
69         }
70         /**
71          * @param key key whose mapping is to be removed from registrar.
72          */
73         public static void unregister (String key) {
74             Map map = getMap();
75             synchronized (map) {
76                 map.remove (key);
77             }
78         }
79         /**
80          * @param key key whose associated value is to be returned.
81          * @throws NotFoundException if key not present in registrar
82          */
83         public static Object get (String key) throws NotFoundException {
84             Object obj = getMap().get(key);
85             if (obj == null)
86                 throw new NotFoundException (key);
87             return obj;
88         }
89         /**
90          * @param key key whose associated value is to be returned.
91          * @throws NotFoundException if key not present in registrar
92          */
93         public static Object getIfExists (String key) {
94             return getMap().get(key);
95         }
96     
97         public void dump (PrintStream p, String indent) {
98             dump(p,indent,false);
99         }
100         public void dump (PrintStream p, String indent, boolean detail) {
101             String inner = indent + "  ";
102             p.println (indent + "--- name-registrar ---");
103             synchronized (registrar) {
104                 Iterator iter = registrar.entrySet().iterator();
105                 while (iter.hasNext()) {
106                     Map.Entry entry = (Map.Entry) iter.next ();
107                     Object obj = entry.getValue();
108                     p.println (inner + 
109                         entry.getKey().toString() + ": " +
110                         obj.getClass().getName()
111                     );
112                     if ((detail == true) && (obj instanceof Loggeable)) {
113                         ((Loggeable)obj).dump(p, inner+"  ");
114                     }
115                 }
116             }
117         }
118     }
