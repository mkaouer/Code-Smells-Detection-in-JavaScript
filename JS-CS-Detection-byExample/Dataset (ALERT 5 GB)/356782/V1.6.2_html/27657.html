39     public class TSpace extends TimerTask implements LocalSpace, Loggeable {
40         protected Map entries;
41         protected TSpace sl;    // space listeners
42         public static final long GCDELAY = 60*1000;
43     
44         public TSpace () {
45             super();
46             entries = new HashMap ();
47             DefaultTimer.getTimer().schedule (this, GCDELAY, GCDELAY);
48         }
49         public synchronized void out (Object key, Object value) {
50             if (key == null || value == null)
51                 throw new NullPointerException ("key=" + key + ", value=" + value);
52             getList (key).add (value);
53             this.notifyAll ();
54             if (sl != null)
55                 notifyListeners(key, value);
56         }
57         public void out (Object key, Object value, long timeout) {
58             if (key == null || value == null)
59                 throw new NullPointerException ("key=" + key + ", value=" + value);
60             if (timeout > 0) {
61                 value   = new Expirable (value, 
62                    System.currentTimeMillis() + timeout);
63             }
64             synchronized (this) {
65                 getList (key).add (value);
66                 this.notifyAll ();
67             }
68             if (sl != null)
69                 notifyListeners(key, value);
70         }
71         public synchronized Object rdp (Object key) {
72             if (key instanceof Template)
73                 return getObject ((Template) key, false);
74             return getHead (key, false);
75         }
76         public synchronized Object inp (Object key) {
77             if (key instanceof Template)
78                 return getObject ((Template) key, true);
79             return getHead (key, true);
80         }
81         public synchronized Object in (Object key) {
82             Object obj;
83             while ((obj = inp (key)) == null) {
84                 try {
85                     this.wait ();
86                 } catch (InterruptedException e) { }
87             }
88             return obj;
89         }
90         public synchronized Object in  (Object key, long timeout) {
91             Object obj;
92             long now = System.currentTimeMillis();
93             long end = now + timeout;
94             while ((obj = inp (key)) == null && 
95                     ((now = System.currentTimeMillis()) < end))
96             {
97                 try {
98                     this.wait (end - now);
99                 } catch (InterruptedException e) { }
100             }
101             return obj;
102         }

112         public synchronized Object rd  (Object key, long timeout) {
113             Object obj;

124         }
125         public void run () {
126             gc();
127         }
128         public void gc () {
129             Object[] keys;
130             synchronized (this) {
131                 keys = entries.keySet().toArray();
132             }
133             for (int i=0; i<keys.length; i++) {
134                 rdp (keys[i]);
135                 Thread.yield ();
136             }
137             if (sl != null) {
138                 synchronized (this) {
139                     if (sl != null && sl.isEmpty())
140                         sl = null;
141                 }
142             }
143         }

145             int size = 0;

150         }

199         public void notifyListeners (Object key, Object value) {
200             Object[] listeners = null;
201             synchronized (this) {
202                 if (sl == null)
203                     return;
204                 List l = (List) sl.entries.get (key);
205                 if (l != null)
206                     listeners = l.toArray();
207             }
208             if (listeners != null) {
209                 for (int i=0; i<listeners.length; i++) {
210                     Object o = listeners[i];
211                     if (o instanceof Expirable)
212                         o = ((Expirable)o).getValue();
213                     if (o instanceof SpaceListener)
214                         ((SpaceListener) o).notify (key, value);
215                 }
216             }
217         }
218         public synchronized void push (Object key, Object value) {
219             if (key == null || value == null)
220                 throw new NullPointerException ("key=" + key + ", value=" + value);
221             getList (key).add (0, value);
222             this.notifyAll ();
223             if (sl != null)
224                 notifyListeners(key, value);
225         }
226         public void push (Object key, Object value, long timeout) {
227             if (key == null || value == null)
228                 throw new NullPointerException ("key=" + key + ", value=" + value);
229             if (timeout > 0) {
230                 value   = new Expirable (value, 
231                    System.currentTimeMillis() + timeout);
232             }
233             synchronized (this) {
234                 getList (key).add (0, value);
235                 this.notifyAll ();
236             }
237             if (sl != null)
238                 notifyListeners(key, value);
239         }
240         private List getList (Object key) {
241             List l = (List) entries.get (key);
242             if (l == null) 
243                 entries.put (key, l = new LinkedList());
244             return l;
245         }
246         private Object getHead (Object key, boolean remove) {
247             Object obj = null;
248             List l = (List) entries.get (key);
249             while (obj == null && l != null && l.size() > 0) {
250                 obj = l.get(0);
251                 if (obj instanceof Expirable) 
252                     obj = ((Expirable) obj).getValue();
253                 if (obj == null) {
254                     l.remove (0);
255                     if (l.size() == 0) {
256                         entries.remove (key);
257                     }
258                 }
259             }
260             if (obj != null && remove) {
261                 l.remove (0);
262                 if (l.size() == 0) {
263                     entries.remove (key);
264                 }
265             }
266             return obj;
267         }
268         private Object getObject (Template tmpl, boolean remove) {
269             Object obj = null;
270             List l = (List) entries.get (tmpl.getKey());
271             if (l == null)
272                 return obj;
273     
274             Iterator iter = l.iterator();
275             while (iter.hasNext()) {
276                 obj = iter.next();
277                 if (obj instanceof Expirable) {
278                     obj = ((Expirable) obj).getValue();
279                     if (obj == null) {
280                         iter.remove();
281                         continue;
282                     }
283                 }
284                 if (tmpl.equals (obj)) {
285                     if (remove)
286                         iter.remove();
287                     break;
288                 } else
289                     obj = null;
290             }
291             return obj;
292         }
293         private TSpace getSL() {
294             synchronized (this) {
295                 if (sl == null)
296                     sl = new TSpace();
297             }
298             return sl;
299         }
300         static class Expirable implements Comparable {
301             Object value;
302             long expires;
303     
304             public Expirable (Object value, long expires) {
305                 super();
306                 this.value = value;
307                 this.expires = expires;
308             }
309             public boolean isExpired () {
310                 return expires < System.currentTimeMillis ();
311             }
312             public String toString() {
313                 return getClass().getName() 
314                     + "@" + Integer.toHexString(hashCode())
315                     + ",value=" + value.toString()
316                     + ",expired=" + isExpired ();
317             }
318             public Object getValue() {
319                 return isExpired() ? null : value;
320             }
321             public int compareTo (Object obj) {
322                 Expirable other = (Expirable) obj;
323                 long otherExpires = other.expires;
324                 if (otherExpires == expires)
325                     return 0;
326                 else if (expires < otherExpires)
327                     return -1;
328                 else 
329                     return 1;
330             }
331         }
