64     /**
65      * link together a whole transaction, the requesting ISOMsg with the
66      * corresponding response in a form suitable to be queued to an
67      * <b>ISOMUX</b>
68      *
69      * @author apr@cs.com.uy
70      * @version $Id$
71      * @see ISOMsg
72      * @see ISOException
73      * @see ISOChannel
74      * @see ISOMUX
75      * @serial
76      */
77     public class ISORequest implements LogSource, Loggeable {
78         private ISOMsg request, response;
79         private long requestTime, txTime, responseTime;
80         private boolean expired;
81         private Logger logger = null;
82         private String realm = null;
83     
84         /**
85          * creates an ISORequest suitable to be queued to an ISOMUX
86          * @param m - the Request Message
87          */
88         public ISORequest (ISOMsg m) {
89             request = m;
90             Date d = new Date();
91             requestTime = d.getTime();
92             txTime = 0;
93             expired = false;
94         }
95         /**
96          * checks if the message has expired
97          */
98         public boolean isExpired() {
99             return expired;
100         }
101         /**
102          * force this message as expired
103          * (used internally by ISOMUX)
104          */
105         public void setExpired(boolean b) {
106             expired = b;
107         }
108         /**
109          * queue the message to a MUX.
110          *
111          * Warning: do not queue the same message to more than one MUX.
112          * @param mux - an ISOMUX
113          */
114         public void queue (ISOMUX mux) {
115             mux.queue(this);
116         }
117         /**
118          * @return the request message
119          */
120         public ISOMsg getRequest() {
121             return request;
122         }
123         /**
124          * MUX calls setTransmitted when chances are really good
125          */
126         public void setTransmitted () {
127             txTime = System.currentTimeMillis();
128         }
129         /**
130          * @return true if this request was actually sent thru channel
131          */
132         public boolean isTransmitted() {
133             return txTime != 0;
134         }
135             
136         /**
137          * wait for a response to arrive. 
138          * ISOMUX will notify this object when the response message is ready.
139          *
140          * @param timeout   timeout in milliseconds. 0 blocks forever
141          * @return ISOMsg or null
142          */
143         public ISOMsg getResponse(int timeout) {
144             synchronized (this) {
145                 if (response == null) {
146                     try {
147                         if (timeout > 0)
148                             wait(timeout);
149                         else
150                             wait();
151                     } catch (InterruptedException e) { }
152                 }
153                 setExpired (response == null);
154             }
155             Logger.log (new LogEvent (this, "ISORequest", this));
156             return response;
157         }
158         public void dump (PrintStream p, String indent) {
159             String newIndent = indent + "  ";
160             p.println (indent + "<request" +
161                 (expired ? " expired=\"true\">" : ">"));
162             request.dump (p, newIndent);
163             p.println (indent + "</request>");
164             if (response != null) {
165                 p.println (indent + "<response elapsed=\""
166                     +(responseTime-requestTime)+ "\">");
167                 response.dump (p, newIndent);
168                 p.println (indent + "</response>");
169             }
170         }
171     
172         /**
173          * called by ISOMUX to set the response and notify this
174          * possibly waiting object.
175          */
176         public void setResponse(ISOMsg m) {
177             Date d = new Date();
178             responseTime = d.getTime();
179             synchronized (this) {
180                 response = m;
181                 this.notify();
182             }
183         }
184         public void setLogger (Logger logger, String realm) {
185             this.logger = logger;
186             this.realm  = realm;
187         }
188         public String getRealm () {
189             return realm;
190         }
191         public Logger getLogger() {
192             return logger;
193         }
194         /**
195          * @return time in milliseconds of how long it took to get a Response
196          */
197         public long getResponseTime () {
198             return (responseTime-requestTime);
199         }
200     }
