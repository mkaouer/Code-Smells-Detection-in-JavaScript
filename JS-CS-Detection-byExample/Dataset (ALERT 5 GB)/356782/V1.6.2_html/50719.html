39     public class QBeanSupport 
40         implements QBean, QPersist, QBeanSupportMBean, Configurable 
41     {
42         Element persist;
43         int state;
44         Q2 server;
45         boolean modified;
46         String name;
47         protected Log log;
48         protected Configuration cfg;
49         public static final String stateString[] = {
50             "Stopped", "Stopping", "Starting", "Started", "Failed", "Destroyed"
51         };
52             
53         public QBeanSupport () {
54             super();
55             setLogger (Q2.LOGGER_NAME);
56             state = -1;
57         }
58         public void setServer (Q2 server) {
59             this.server = server;
60         }
61         public Q2 getServer () {
62             return server;
63         }
64         public QFactory getFactory () {
65             return getServer().getFactory ();
66         }
67         public void setName (String name) {
68             if (this.name == null) 
69                 this.name = name;
70             log.setRealm (name);
71             setModified (true);
72         }
73         public void setLogger (String loggerName) {
74             log = Log.getLog (loggerName, getClass().getName());
75             setModified (true);
76         }
77     
78         public String getLogger () {
79             return log.getLogger().getName();
80         }
81         public Log getLog () {
82             return log;
83         }
84         public String getName () {
85             return name;
86         }
87         public void init () {
88             if (state == -1) {
89                 setModified (false);
90                 try {
91                     initService();
92                     state = QBean.STOPPED;
93                 } catch (Throwable t) {
94                     log.warn ("init", t);
95                 }
96             }
97         }
98         public synchronized void start() {
99             if (state != QBean.DESTROYED && 
100                 state != QBean.STOPPED   && 
101                 state != QBean.FAILED)
102                return;
103     
104             this.state = QBean.STARTING;
105     
106             try {
107                startService();
108             } catch (Throwable t) {
109                state = QBean.FAILED;
110                log.warn ("start", t);
111                return;
112             }
113             state = QBean.STARTED;
114         }
115         public synchronized void stop () {
116             if (state != QBean.STARTED)
117                return;
118             state = QBean.STOPPING;
119             try {
120                stopService();
121             } catch (Throwable t) {
122                state = QBean.FAILED;
123                log.warn ("stop", t);
124                return;
125             }
126             state = QBean.STOPPED;
127         }
128         public void destroy () {
129             if (state == QBean.DESTROYED)
130                return;
131             if (state != QBean.STOPPED)
132                stop();
133     
134             try {
135                destroyService();
136             }
137             catch (Throwable t) {
138                log.warn ("destroy", t);
139             }
140             state = QBean.DESTROYED;
141         }
142         public void shutdownQ2 () {
143             getServer().shutdown ();
144         }
145         public int getState () {
146             return state;
147         }
148         public URL[] getLoaderURLS() {
149             return server.getLoader().getURLs();
150         }    
151         public QClassLoader getLoader() {
152             return server.getLoader();
153         }
154         public String getStateAsString () {
155             return state >= 0 ? stateString[state] : "Unknown";
156         }
157         public void setState (int state) {
158             this.state = state;
159         }
160         public void setPersist (Element persist) {
161             this.persist = persist ;
162         }
163         public synchronized Element getPersist () {
164             setModified (false);
165             return persist;
166         }
167         public synchronized void setModified (boolean modified) {
168             this.modified = modified;
169         }
170         public synchronized boolean isModified () {
171             return modified;
172         }
173         public boolean running () {
174             return state == QBean.STARTING || state == QBean.STARTED;
175         }
176         protected void initService()    throws Exception {}
177         protected void startService()   throws Exception {}
178         protected void stopService()    throws Exception {}
179         protected void destroyService() throws Exception {}
180     
181         protected Element createElement (String name, Class mbeanClass) {
182             Element e = new Element (name);
183             Element classPath = persist != null ?
184                 persist.getChild ("classpath") : null;
185             if (classPath != null)
186                 e.addContent (classPath);
187             e.setAttribute ("class", getClass().getName());
188             if (!e.getName().equals (getName ()))
189                 e.setAttribute ("name", getName());
190             String loggerName = getLogger();
191             if (loggerName != null)
192                 e.setAttribute ("logger", loggerName);
193     
194             try {
195                 BeanInfo info = Introspector.getBeanInfo (mbeanClass);
196                 PropertyDescriptor[] desc = info.getPropertyDescriptors();
197                 for (int i=0; i<desc.length; i++) {
198                     if (desc[i].getWriteMethod() != null) {
199                         Method read = desc[i].getReadMethod();
200                         Object obj  = read.invoke (this, new Object[] { } );
201                         String type = read.getReturnType().getName();
202                         if ("java.lang.String".equals (type))
203                             type = null;
204     
205                         addAttr (e, desc[i].getName(), obj, type);
206                     }
207                 }
208             } catch (Exception ex) {
209                 log.warn ("get-persist", ex);
210             } 
211             return e;
212         }
213         protected void addAttr (Element e, String name, Object obj, String type) {
214             String value = obj == null ? "null" : obj.toString();
215             Element attr = new Element ("attr");
216             attr.setAttribute ("name", name);
217             if (type != null)
218                 attr.setAttribute ("type", type);
219             attr.setText (value);
220             e.addContent (attr);
221         }
222         protected Iterator getAttrs () {
223             return getPersist().getChildren ("attr").iterator();
224         }
225         protected Iterator getAttrs (String parent) {
226             return getPersist().getChild (parent).
227                 getChildren ("attr").iterator();
228         }
229         protected void setAttr (Iterator attrs, String name, Object obj) {
230             String value = obj == null ? "null" : obj.toString ();
231             while (attrs.hasNext ()) {
232                 Element e = (Element) attrs.next ();
233                 if (name.equals (e.getAttributeValue ("name")))  {
234                     e.setText (value);
235                     break;
236                 }
237             }
238         }
239         protected Iterator getProperties (String parent) {
240             return getPersist().getChild (parent).
241                    getChildren ("property").iterator();
242         }
243         protected void setProperty (Iterator props, String name, String value) {
244             while (props.hasNext()) {
245                 Element e = (Element) props.next();
246                 if (name.equals (e.getAttributeValue ("name"))) {
247                     e.setAttribute ("value", value);
248                     break;
249                 }
250             }
251         }
252         protected String getProperty (Iterator props, String name) {
253             while (props.hasNext()) {
254                 Element e = (Element) props.next();
255                 if (name.equals (e.getAttributeValue ("name"))) {
256                     return e.getAttributeValue ("value");
257                 }
258             }
259             return null;
260         }
261         public void setConfiguration (Configuration cfg)
262             throws ConfigurationException
263         {
264             this.cfg = cfg;
265         }
266         public Configuration getConfiguration () {
267             return cfg;
268         }
