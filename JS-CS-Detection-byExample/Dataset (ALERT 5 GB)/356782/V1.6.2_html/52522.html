19     package org.jpos.util;
20     
21     import org.jpos.iso.ISOUtil;
22     import java.util.Arrays;
23     import junit.framework.*;
24     
25     public class FSDMsgTestCase extends TestCase {
26         FSDMsg msg;
27     
28         public void setUp() throws Exception {
29             msg = new FSDMsg ("file:../test/org/jpos/util/msg-");
30         }
31         public void testLeadingBlanks () throws Exception {
32             msg.set ("testafs", "   123");
33             assertEquals ("Leading blanks",
34                 ISOUtil.hex2byte ("2020203132331C"),
35                 msg.pack().getBytes()
36             );
37         }
38         public void testTraillingBlanks () throws Exception {
39             msg.set ("testafs", "123   ");
40             assertEquals ("Trailing blanks",
41                 ISOUtil.hex2byte ("3132331C"),
42                 msg.pack().getBytes()
43             );
44         }
45         public void testMixedBlanks () throws Exception {
46             msg.set ("testafs", "  123 ");
47             assertEquals ("Mixed blanks",
48                 ISOUtil.hex2byte ("20203132331C"),
49                 msg.pack().getBytes()
50             );
51         }
52         public void testFinalField() throws Exception {
53             msg.set ("testafs", "  123 ");
54             msg.set ("finalfield", "ABC");
55             assertEquals ("Final Field",
56                 ISOUtil.hex2byte ("20203132331C414243"),
57                 msg.pack().getBytes()
58             );
59     
60             FSDMsg m = new FSDMsg ("file:../test/org/jpos/util/msg-");
61             m.unpack (ISOUtil.hex2byte ("20203132331C414243"));
62             assertEquals ("Final Field",
63                 ISOUtil.hex2byte ("20203132331C414243"),
64                 m.pack().getBytes()
65             );
66         }
67         public void assertEquals (String msg, byte[] b1, byte[] b2) {
68             assertTrue (msg, Arrays.equals (b1, b2));
69         }
70     }
