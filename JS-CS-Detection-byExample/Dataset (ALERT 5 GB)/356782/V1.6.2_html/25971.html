49     import java.util.Hashtable;
50     import java.util.Map;
51     
52     public class SimpleLockManager implements LockManager {
53         Map locks;
54     
55         public SimpleLockManager () {
56             locks = new Hashtable();
57         }
58     
59         public class SimpleTicket implements Ticket {
60             String resourceName;
61             long expiration;
62             public SimpleTicket (String resourceName, long duration) {
63                 this.resourceName = resourceName;
64                 this.expiration = System.currentTimeMillis() + duration;
65             }
66             public boolean renew (long duration) {
67                 if (!isExpired()) {
68                     this.expiration = System.currentTimeMillis() + duration;
69                     return true;
70                 }
71                 return false;
72             }
73             public long getExpiration() {
74                 return expiration;
75             }
76             public boolean isExpired() {
77                 return System.currentTimeMillis() > expiration;
78             }
79             public String getResourceName () {
80                 return resourceName;
81             }
82             public void cancel() {
83                 expiration = 0;
84                 locks.remove (resourceName);
85                 synchronized (this) {
86                     notify();
87                 }
88             }
89             public String toString() {
90                 return super.toString() 
91                     + "[" + resourceName + "/" +isExpired() + "/"
92                     + (expiration - System.currentTimeMillis()) + "ms left]";
93             }
94         }
95         public Ticket lock (String resourceName, long duration, long wait)
96         {
97             long maxWait = System.currentTimeMillis() + wait;
98     
99             while (System.currentTimeMillis() < maxWait) {
100                 Ticket t = null;
101                 synchronized (this) {
102                     t = (Ticket) locks.get (resourceName);
103                     if (t == null) {
104                         t = new SimpleTicket (resourceName, duration);
105                         locks.put (resourceName, t);
106                         return t;
107                     } 
108                     else if (t.isExpired()) {
109                         t.cancel();
110                         continue;
111                     }
112                 }
113                 synchronized (t) {
114                     try {
115                         t.wait (Math.min (1000, wait));
116                     } catch (InterruptedException e) { }
117                 }
118             }
119             return null;
120         }
121     }
