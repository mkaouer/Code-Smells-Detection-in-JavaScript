162         protected String get (String id, String type, int length, String defValue) 
163             throws ISOException
164         {
165             String value = (String) fields.get (id);
166             if (value == null)
167                 value = defValue == null ? "" : defValue;
168     
169             type   = type.toUpperCase ();
170     
171             switch (type.charAt (0)) {
172                 case 'N':
173                     if (isSeparated(type)) {
174                         // Leave value unpadded.
175                     } else {
176                         value = ISOUtil.zeropad (value, length);
177                     }
178                     break;
179                 case 'A':
180                     if (isSeparated(type)) {
181                         // Leave value unpadded.
182                     } else {
183                         value = ISOUtil.strpad (value, length);
184                     }
185                     if (value.length() > length)
186                         value = value.substring(0,length);
187                     break;
188                 case 'K':
189                     if (defValue != null)
190                         value = defValue;
191                     break;
192                 case 'B':
193                     try {
194                         if ((length << 1) >= value.length()) {
195                             value = new String (
196                                     ISOUtil.hex2byte (ISOUtil.zeropad(value,length << 1).substring (0, length << 1)),
197                                     "ISO8859_1");
198                         } else {
199                             throw new RuntimeException("field content="+value+" is too long to fit in field "+id+" whose length is "+length);
200                         }
201                     } catch (UnsupportedEncodingException e) {
202                         // ISO8859_1 is supported
203                     }
204                     break;
205             }
206             return (isSeparated(type)) ? ISOUtil.blankUnPad(value) : value;
207         }

209         private boolean isSeparated(String type) {
210             /*
211              * if type's last two characters appear in our Map of separators,
212              * return true
213              */
214             if (type.length() > 2) {
215                 if (separators.containsKey(getSeparatorType(type))) {
216                     return true;
217                 } else {
218                     throw new RuntimeException("FSDMsg.isSeparated(String) found that type of "+type+" is used, but "+getSeparatorType(type)+" has not been defined as a separator!");
219                 }
220             }
221             return false;
222             
223         }
224         
225         private boolean isBinary(String type) {
226             /*
227              * if type's first digit is a 'B' return true
228              */
229             return type.startsWith("B");
230         }

232         public boolean isSeparator(byte b) {
233             return separators.containsValue(new Character((char)b));
234         }

236         private String getSeparatorType(String type) {
237             if (type.length() > 2) {
238                 return type.substring(1);
239             }
240             return null;
241         }
242         
243         private char getSeparator(String type) {
244             if (type.length() > 2) {
245                 return ((Character)separators.get(getSeparatorType(type))).charValue();
246             }
247             
248             throw new RuntimeException("getSeparator called on for type="+type+" which does not resolve to a known separator.");
249         }

251         protected void pack (Element schema, StringBuffer sb) 
252             throws JDOMException, MalformedURLException, IOException, ISOException
253         {
254             String keyOff = "";
255             Iterator iter = schema.getChildren("field").iterator();
256             while (iter.hasNext()) {
257                 Element elem = (Element) iter.next ();
258                 String id    = elem.getAttributeValue ("id");
259                 int length   = Integer.parseInt (elem.getAttributeValue ("length"));
260                 String type  = elem.getAttributeValue ("type");
261                 boolean key  = "true".equals (elem.getAttributeValue ("key"));
262                 String defValue = elem.getText();
263                 String value = get (id, type, length, defValue);
264                 sb.append (value);
265                 
266                 if (isSeparated(type)) {
267                     char c = getSeparator(type);
268                     if (c > 0)
269                         sb.append(c);
270                 }
271                 if (key) 
272                     keyOff = keyOff + value;
273             }
274             if (keyOff.length() > 0) 
275                 pack (getSchema (getId (schema) + keyOff), sb);
276         }

278         protected void unpack (InputStream is, Element schema) 
279             throws IOException, JDOMException, MalformedURLException  
280         
281         {
282             Iterator iter = schema.getChildren("field").iterator();
283             String keyOff = "";
284             while (iter.hasNext()) {
285                 Element elem = (Element) iter.next();
286     
287                 String id    = elem.getAttributeValue ("id");
288                 int length   = Integer.parseInt (elem.getAttributeValue ("length"));
289                 String type  = elem.getAttributeValue ("type").toUpperCase();
290                 boolean key  = "true".equals (elem.getAttributeValue ("key"));
291                 String value = readField (
292                     is, id, length, type );
293                 
294                 if (key)
295                     keyOff = keyOff + value;
296     
297                 if ("K".equals(type) && !value.equals (elem.getText()))
298                     throw new IllegalArgumentException (
299                         "Field "+id 
300                            + " value='"     +value
301                            + "' expected='" + elem.getText () + "'"
302                     );
303             }
304             if (keyOff.length() > 0) {
305                 unpack (is, 
306                     getSchema (getId (schema) + keyOff)
307                 );
308             }
309         }
310         private String getId (Element e) {
311             String s = e.getAttributeValue ("id");
312             return s == null ? "" : s;
313         }
314         protected String read (InputStream is, int len, String type) 
315             throws IOException 
316         {
317             StringBuffer sb = new StringBuffer();
318             byte[] b = new byte[1];
319             boolean expectSeparator = isSeparated(type);
320             boolean separated = expectSeparator;
321     
322             for (int i=0; i<len; i++) {
323                 if (is.read (b) < 0) {
324                     if (!type.endsWith("EOF"))
325                         throw new EOFException ();
326                     else {
327                         separated = false;
328                         break;
329                     }
330                 }
331                 if (expectSeparator && (b[0] == getSeparator(type))) {
332                     separated = false;
333                     break;
334                 }
335                 sb.append ((char) (b[0] & 0xff));
336             }
337             if (separated && !type.endsWith("EOF")) {
338                 if (is.read (b) < 0) {
339                     throw new EOFException ();
340                 }
341             }
342             return sb.toString ();
343         }
344         protected String readField 
345             (InputStream is, String fieldName, int len, String type) 
346             throws IOException
347         {
348             String fieldValue = read (is, len, type);
349             
350             if (isBinary(type))
351                 fieldValue = ISOUtil.hexString (fieldValue.getBytes ("ISO8859_1"));
352             fields.put (fieldName, fieldValue);
353     //         System.out.println ("++++ "+fieldName + ":" + fieldValue + " " + type + "," + isBinary(type));
354             return fieldValue;
355         }

399         public Element toXML () {
400             Element e = new Element ("message");
401             if (header != null) {
402                 e.addContent (
403                     new Element ("header")
404                         .setText (getHexHeader ())
405                 );
406             }
407             Iterator iter = fields.keySet().iterator();
408             while (iter.hasNext()) {
409                 String fieldName = (String) iter.next();
410                 Element inner = new Element (fieldName);
411                 inner.addContent (ISOUtil.normalize ((String) fields.get (fieldName)));
412                 e.addContent (inner);
413             }
414             return e;
415         }
416         protected Element getSchema () 
417             throws JDOMException, MalformedURLException, IOException {
418             return getSchema (baseSchema);

420         protected Element getSchema (String message) 
421             throws JDOMException, MalformedURLException, IOException {
422             StringBuffer sb = new StringBuffer (basePath);
423             sb.append (message);
424             sb.append (".xml");
425             String uri = sb.toString ();
426     
427             Space sp = SpaceFactory.getSpace();
428             Element schema = (Element) sp.rdp (uri);
429             if (schema == null) {
430                 synchronized (FSDMsg.class) {
431                     schema = (Element) sp.rdp (uri);
432                     if (schema == null) {
433                         SAXBuilder builder = new SAXBuilder ();
434                         URL url = new URL (uri);
435                         File f = new File(url.getFile());
436                         if (f.exists()) {
437                             schema = builder.build (url).getRootElement ();
438                         } else {
439                             throw new RuntimeException(f.getCanonicalPath().toString() + " not found");
440                         }
441                     }
442                     sp.out (uri, schema);
443                 }
444             } 
445             return schema;
446         }

453         public void dump (PrintStream p, String indent) {
454             String inner = indent + "  ";
455             p.println (indent + "<fsdmsg schema='" + basePath + baseSchema  + "'>");
456             if (header != null) {
457                 append (p, "header", getHexHeader(), inner);
458             }
459             Iterator iter = fields.keySet().iterator();
460             while (iter.hasNext()) {
461                 String f = (String) iter.next();
462                 String v = ((String) fields.get (f));
463                 append (p, f, v, inner);
464             }
465             p.println (indent + "</fsdmsg>");
466         }
467         private void append (PrintStream p, String f, String v, String indent) {
468             p.println (indent + f + ": '" + v + "'");
469         }
