19     package org.jpos.q2.cli;
20     
21     import java.util.Date;
22     import org.jpos.iso.ISOUtil;
23     import org.jpos.iso.ISOException;
24     import org.jpos.q2.CLI;
25     
26     public class UPTIME implements CLI.Command {
27         public void exec (CLI cli, String[] args) throws Exception {
28             StringBuffer sb = new StringBuffer();
29             long uptime = cli.getQ2().getUptime();
30             int ms = (int) (uptime % 1000);
31             uptime /= 1000;
32             int dd = (int) (uptime/86400);
33             uptime -= (dd * 86400);
34             int hh = (int) (uptime/3600);
35             uptime -= (hh * 3600);
36             int mm = (int) (uptime/60);
37             uptime -= (mm * 60);
38             int ss = (int) uptime;
39             if (dd > 0) {
40                 sb.append (Long.toString(dd));
41                 sb.append ("d ");
42             }
43             sb.append (zeropad (hh, 2));
44             sb.append (':');
45             sb.append (zeropad (mm, 2));
46             sb.append (':');
47             sb.append (zeropad (ss, 2));
48             sb.append ('.');
49             sb.append (zeropad (ms, 3));
50             cli.println (sb.toString());
51         }
52         private String zeropad (int i, int l) throws ISOException {
53             return ISOUtil.zeropad (Integer.toString (i), l);
54         }
55     }
