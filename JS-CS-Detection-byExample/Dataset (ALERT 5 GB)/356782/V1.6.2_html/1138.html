19     package org.jpos.util;
20     
21     /**
22      * ThroughputControl can be used to limit the throughput 
23      * of a system to a maximum number of transactions in 
24      * a given period of time.
25      */
26     public class ThroughputControl {
27         private int[] period;
28         private int[] max;
29         private int[] cnt;
30         private long[] start;
31         private long[] sleep;
32     
33         /**
34          * @param maxTransactions ditto
35          * @param periodInMillis ditto
36          */
37         public ThroughputControl (int maxTransactions, int periodInMillis) {
38             this (new int[] { maxTransactions },
39                   new int[] { periodInMillis });
40         }
41         /**
42          * @param maxTransactions ditto
43          * @param periodInMillis ditto
44          */
45         public ThroughputControl (int[] maxTransactions, int[] periodInMillis) {
46             super();
47             int l = maxTransactions.length;
48             period = new int[l];
49             max = new int[l];
50             cnt = new int[l];
51             start = new long[l];
52             sleep = new long[l];
53             for (int i=0; i<l; i++) {
54                 this.max[i]    = maxTransactions[i];
55                 this.period[i] = periodInMillis[i];
56                 this.sleep[i]  = Math.min(Math.max (periodInMillis[i]/10, 500L),50L);;
57                 this.start[i]  = System.currentTimeMillis();
58             }
59         }
60     
61         /**
62          * control should be called on every transaction.
63          * it may sleep for a while in order to control the system throughput
64          * 
65          * @return aprox sleep time or zero if no sleep
66          */
67         public long control() {
68             boolean delayed = false;
69             long init = System.currentTimeMillis();
70             for (int i=0; i<cnt.length; i++) {
71                 synchronized (this) {
72                     cnt[i]++;
73                 }
74                 do {
75                     if (cnt[i] > max[i]) {
76                         delayed = true;
77                         try { 
78                             Thread.sleep (sleep[i]); 
79                         } catch (InterruptedException e) { }
80                     }
81                     synchronized (this) {
82                         long now = System.currentTimeMillis();
83                         if (now - start[i] > period[i]) {
84                             long elapsed = now - start[i];
85                             int  allowed = (int) ((elapsed * max[i]) / period[i]);
86                             start[i] = now;
87                             cnt[i] = Math.max (cnt[i] - allowed, 0);
88                         }
89                     }
90                 } while (cnt[i] > max[i]);
91             }
92             return delayed ? (System.currentTimeMillis() - init) : 0L;
93         }
94     }
