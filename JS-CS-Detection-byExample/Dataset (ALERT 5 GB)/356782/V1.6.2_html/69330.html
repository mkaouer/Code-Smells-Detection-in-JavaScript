22     import java.net.ServerSocket;
23     
24     import org.jpos.iso.BaseChannel;
25     import org.jpos.iso.ISOChannel;

27     import org.jpos.iso.ISOMsg;
28     import org.jpos.iso.ISOPackager;
29     import org.jpos.iso.ISOUtil;
30     import org.jpos.util.LogEvent;
31     import org.jpos.util.Logger;

33     /**
34      * Implements an ISOChannel able to exchange messages with
35      * ACI's BASE24 over a TCP link, modified from BASE24ISOChannel 
36      * by Victor A. Salaman (salaman@teknos.com) .<br>
37      * An instance of this class exchanges messages by means of an
38      * intermediate 'port server' as described in the
39      * <a href="/doc/javadoc/overview-summary.html">Overview</a> page.
40      * @author apr@cs.com.uy
41      * @author salaman@teknos.com
42      *
43      * @version $Id$
44      *
45      * @see ISOMsg
46      * @see ISOException
47      * @see ISOChannel
48      */
49     
50     public class BASE24TCPChannel extends BaseChannel {
51         /**
52          * Public constructor (used by Class.forName("...").newInstance())
53          */
54         public BASE24TCPChannel () {
55             super();
56         }
57         /**
58          * Construct client ISOChannel
59          * @param host  server TCP Address
60          * @param port  server port number
61          * @param p     an ISOPackager
62          * @see ISOPackager
63          */
64         public BASE24TCPChannel (String host, int port, ISOPackager p) {
65             super(host, port, p);
66         }
67         /**
68          * Construct server ISOChannel
69          * @param p     an ISOPackager
70          * @see ISOPackager
71          * @exception IOException
72          */
73         public BASE24TCPChannel (ISOPackager p) throws IOException {
74             super(p);
75         }
76         /**
77          * constructs a server ISOChannel associated with a Server Socket
78          * @param p     an ISOPackager
79          * @param serverSocket where to accept a connection
80          * @exception IOException
81          * @see ISOPackager
82          */
83         public BASE24TCPChannel (ISOPackager p, ServerSocket serverSocket) 
84             throws IOException
85         {
86             super(p, serverSocket);
87         }
88         /**
89          * @param m the Message to send (in this case it is unused)
90          * @param len   message len (ignored)
91          * @exception IOException
92          */
93         protected void sendMessageTrailler(ISOMsg m, int len) throws IOException {
94             serverOut.write (3);
95         }
96         protected void sendMessageLength(int len) throws IOException {
97             len++;  // one byte trailler
98             serverOut.write (len >> 8);
99             serverOut.write (len);
100         }
101         protected int getMessageLength() throws IOException, ISOException {
102             int l = 0;
103             byte[] b = new byte[2];
104             Logger.log (new LogEvent (this, "get-message-length"));
105             while (l == 0) {
106                 serverIn.readFully(b,0,2);
107                 l = ((((int)b[0])&0xFF) << 8) | (((int)b[1])&0xFF);
108                 if (l == 0) {
109                     serverOut.write(b);
110                     serverOut.flush();
111                 }
112             }
113             Logger.log (new LogEvent (this, "got-message-length", Integer.toString(l)));
114             return l - 1;   // trailler length
115         }
116         protected void getMessageTrailler() throws IOException {
117             Logger.log (new LogEvent (this, "get-message-trailler"));
118             byte[] b = new byte[1];
119             serverIn.readFully(b,0,1);
120             Logger.log (new LogEvent (this, "got-message-trailler", ISOUtil.hexString(b)));
121         }
122     }
