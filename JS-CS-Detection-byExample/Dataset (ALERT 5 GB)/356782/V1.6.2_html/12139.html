35     
36     /**
37      * Integrates legacy (pre 2.0) ISORequestListeners to the new Space 
38      * based architecture
39      * @author Alejandro Revilla
40      * @version $Revision$ $Date$
41      */
42     public class ISORequestListenerAdaptor extends SimpleLogSource 
43             implements Configurable, ISORequestListener
44     {
45         Space sp;
46         Configuration cfg;
47         ISOChannel channel;
48         SpaceMUX mux;
49         ThreadPool pool;
50         long timeout;
51         public ISORequestListenerAdaptor () {
52             super ();
53             sp = TransientSpace.getSpace ();
54         }
55         public void setConfiguration (Configuration cfg) 
56             throws ConfigurationException 
57         {
58             this.cfg = cfg;
59             try {
60                 mux = (SpaceMUX) NameRegistrar.get (cfg.get ("mux"));
61             } catch (NotFoundException e) {
62                 throw new ConfigurationException (e);
63             }
64             timeout = cfg.getLong ("timeout", 120000);
65             pool    = new ThreadPool (1, cfg.getInt ("pool", 100));
66         }
67     
68         public class Processor implements Runnable {
69             ISOSource source;
70             ISOMsg m;
71             public Processor (ISOSource source, ISOMsg m) {
72                 super ();
73                 this.source = source;
74                 this.m      = m;
75             }
76             public void run () {
77                 try {
78                     ISOMsg resp = mux.request (m, timeout);
79                     if (resp != null)
80                         source.send (resp);
81                 } catch (Exception e) {
82                     Logger.log (
83                         new LogEvent (
84                             ISORequestListenerAdaptor.this, 
85                             "iso-request-listener-adaptor", e
86                         )
87                     );
88                 }
89             }
90         }
91     
92         public boolean process (ISOSource source, ISOMsg m) {
93             pool.execute (new Processor (source, m));
94             return false;   // let other listeners do their work too
95         }
96     }
