19     package org.jpos.q2.security;
20     
21     import org.jdom.Element;
22     import org.jpos.q2.Q2;
23     import org.jpos.q2.QFactory;
24     import org.jpos.q2.QBeanSupport;
25     import org.jpos.security.SMAdapter;
26     import org.jpos.util.NameRegistrar;
27     
28     /**
29      * SMAdaptor Adaptor
30      *
31      * <b>Sample Configuration</b>
32      * <pre>
33      * &lt;s-m-adaptor class="org.jpos.q2.security.SMAdaptor" logger="Q2"&gt;
34      *  &lt;attr name="impl"&gt;org.jpos.security.jceadapter.JCESecurityModule&lt;/attr&gt;
35      * 
36      *  &lt;property name="provider" value="com.sun.crypto.provider.SunJCE" /&gt;
37      * &lt;property name="lmk" value="path/to/your/lmk" /&gt;
38      * &lt;/s-m-adaptor&gt;
39      * </pre>
40      * 
41      * @author Hani Kirollos
42      * @author Alejandro Revilla
43      * @version $Revision$ $Date$
44      * @jmx:mbean description="SMAdaptor QBean"
45      *                extends="org.jpos.q2.QBeanSupportMBean"
46      */
47     public class SMAdaptor extends QBeanSupport implements SMAdaptorMBean
48     {
49         String clazz;
50         SMAdapter sm;
51         public SMAdaptor () {
52             super ();
53         }
54         protected void initService () throws Exception {
55             Element e = getPersist ();
56             QFactory factory = getServer().getFactory();
57             sm = (SMAdapter) factory.newInstance (getImpl ());
58             factory.setLogger  (sm, e);
59             factory.setConfiguration (sm, e);
60         }
61     
62         /**
63          * @jmx:managed-attribute description="Implementation class name"
64          */
65         public void setImpl (String clazz) {
66             this.clazz = clazz;
67         }
68         /**
69          * @jmx:managed-attribute description="Implementation class name"
70          */
71         public String getImpl() {
72             return clazz;
73         }
74         protected void startService () throws Exception {
75             NameRegistrar.register (getName (), sm);
76         }
77         protected void stopService () throws Exception {
78             NameRegistrar.unregister (getName ());
79         }
80     }
