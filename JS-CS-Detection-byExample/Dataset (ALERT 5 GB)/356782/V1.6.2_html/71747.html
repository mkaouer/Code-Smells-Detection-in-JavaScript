19     package org.jpos.iso;
20     
21     import junit.framework.TestCase;
22     
23     /**
24      * @author joconnor
25      */
26     public class IFB_LLLHCHARTest extends TestCase
27     {
28         public void testPack() throws Exception
29         {
30             ISOField field = new ISOField(10, "ABCDEFGHIJ");
31             IFB_LLLHCHAR packager = new IFB_LLLHCHAR(10, "Should be ABCDEFGHIJ");
32             TestUtils.assertEquals(new byte[] {0x00, 0x0A, 0x41, 0x42, 0x43, 0x44,
33                                                 0x45, 0x46, 0x47, 0x48, 0x49, 0x4A},
34                 packager.pack(field));
35         }
36     
37         public void testPackagerTooLong() throws Exception
38         {
39             try
40             {
41                 new IFB_LLLHCHAR(65536, "Too long for this");
42                 fail("65536 is too long and should have thrown an exception");
43             } catch (Exception ignored)
44             {
45             }
46         }
47     
48         public void testPackTooMuch() throws Exception
49         {
50             ISOField field = new ISOField(10, "ABCDEFGHIJ");
51             IFB_LLLHCHAR packager = new IFB_LLLHCHAR(5, "Should be ABCDEFGHIJ");
52             try
53             {
54                 packager.pack(field);
55                 fail("field is too long and should have thrown an exception");
56             } catch (Exception ignored)
57             {
58             }
59         }
60     
61         public void testUnpack() throws Exception
62         {
63             byte[] raw = new byte[] {0x00, 0x0A, 0x41, 0x42, 0x43, 0x44,
64             0x45, 0x46, 0x47, 0x48, 0x49, 0x4A};
65             IFB_LLLHCHAR packager = new IFB_LLLHCHAR(10, "Should be ABCDEFGHIJ");
66             ISOField field = new ISOField(12);
67             packager.unpack(field, raw, 0);
68             assertEquals("ABCDEFGHIJ", (String) field.getValue());
69         }
70     
71         public void testReversability() throws Exception
72         {
73             String origin = "Abc123:.-";
74             ISOField f = new ISOField(12, origin);
75             IFB_LLLHCHAR packager = new IFB_LLLHCHAR(10, "Should be Abc123:.-");
76     
77             ISOField unpack = new ISOField(12);
78             packager.unpack(unpack, packager.pack(f), 0);
79             assertEquals(origin, (String) unpack.getValue());
80         }
81     }
