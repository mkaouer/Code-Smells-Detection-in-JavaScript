19     package org.jpos.core;
20     
21     import java.io.PrintStream;
22     import java.math.BigDecimal;
23     
24     import org.jpos.util.Loggeable;
25     
26     /**
27      * @author apr@cs.com.uy
28      * @version $Id$
29      * @since jPOS 1.1
30      *
31      * @see CardTransaction
32      * @see CardAgentLookup
33      * @see CardHolder
34      * @serial
35      */
36     public class AuthorizationTransaction
37         implements CardTransaction, Loggeable
38     {
39         /**
40          * @serial
41          */
42         public CardHolder cardHolder;
43         /**
44          * @serial
45          */
46         public BigDecimal amount;
47         /**
48          * @serial
49          */
50         public Integer currency;
51         /**
52          * @serial
53          */
54         public String rrn;
55         /**
56          * @serial
57          */
58         public String terminal;
59         /**
60          * @serial
61          */
62         public String purchasePlan;
63         /**
64          * @serial
65          */
66         public byte numberOfPayments;
67         /**
68          * @serial
69          */
70         protected String action;
71         /**
72          * @serial
73          */
74         protected String[] args = { "" };
75     
76         private transient CardTransactionResponse response;
77     
78         public AuthorizationTransaction() {
79             cardHolder = null;
80             amount     = null;
81             rrn        = null;
82             terminal   = null;
83             currency   = null;
84             purchasePlan = null;
85             numberOfPayments = 1;
86             action     = "authorize";
87         }
88         public void setCardHolder (CardHolder cardHolder) {
89             this.cardHolder = cardHolder;
90         }
91         public CardHolder getCardHolder() {
92             return cardHolder;
93         }
94         public void setAmount (BigDecimal amount) {
95             this.amount = amount;
96         }
97         public BigDecimal getAmount() {
98             return amount;
99         }
100         public void setCurrency (Integer currency) {
101             this.currency = currency;
102         }
103         public void setCurrency (int m) {
104             this.currency = new Integer (m);
105         }
106         public Integer getCurrency() {
107             return currency;
108         }
109         public void setRRN (String rrn) {
110             this.rrn = rrn;
111         }
112         public String getRRN() {
113             return rrn;
114         }
115         public void setTerminal (String terminal) {
116             this.terminal = terminal;
117         }
118         public String getTerminal () {
119             return terminal;
120         }
121         public void setAction (String action) {
122             this.action = action;
123         }
124         public void setArgs (String[] args) {
125             this.args = args;
126         }
127         public String getAction() {
128             return action;
129         }
130         public String[] getArgs() {
131             return args;
132         }
133         public void setPurchasePlan(String purchasePlan) {
134             this.purchasePlan = purchasePlan;
135         }
136         public String getPurchasePlan () {
137             return purchasePlan;
138         }
139         public void setNumberOfPayments(int n) {
140             this.numberOfPayments = (byte) n;
141         }
142         public int getNumberOfPayments() {
143             return (int) numberOfPayments;
144         }
145         public void dump (PrintStream p, String indent) {
146             String inner = indent + "  ";
147             p.println (indent + "<" + getTagName() + ">");
148             p.println (inner  + "<action>" + action + "</action>");
149             getCardHolder().dump (p, inner);
150             p.println (inner  + "<amount>"      + amount      + "</amount>");
151             p.println (inner  + "<currency>"    + currency     + "</currency>");
152             p.println (inner  + "<terminal>"    + terminal     + "</terminal>");
153             p.println (inner  + "<purchasePlan>"+ purchasePlan + "</purchasePlan>");
154             p.println (inner  + "<numberOfPayments>" + numberOfPayments + 
155                                "</numberOfPayments>"
156             );
157             if (response != null && response instanceof Loggeable)
158                 ((Loggeable)response).dump (p, inner);
159             p.println (indent + "</" + getTagName() + ">");
160         }
161         protected String getTagName() {
162             return "AuthorizationTransaction";
163         }
164         public void setResponse (CardTransactionResponse response) {
165             this.response = response;
166         }
167         public CardTransactionResponse getResponse () {
168             return response;
169         }
170     }
