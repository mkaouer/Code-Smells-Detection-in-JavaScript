19     package org.jpos.q2;
20     
21     import java.io.IOException;
22     import java.io.PrintStream;
23     import java.util.List;
24     import java.util.LinkedList;
25     import java.util.StringTokenizer;
26     import org.jpos.iso.ISOUtil;
27     import jline.ConsoleReader;
28     import jline.Completor;
29     import jline.ClassNameCompletor;
30     import jline.SimpleCompletor;
31     import jline.ArgumentCompletor;
32     
33     public class CLI extends Thread 
34         implements SimpleCompletor.SimpleCompletorFilter 
35     {
36         Q2 q2;
37         ConsoleReader reader;
38         PrintStream out;
39         Completor completor;
40         String line = null;
41         boolean keepRunning = false;
42     
43         private CLI() { }
44     
45         public CLI (Q2 q2, String line, boolean keepRunning) throws IOException {
46             super();
47             this.q2 = q2;
48             reader = new ConsoleReader();
49             reader.setBellEnabled(true);
50             out = System.out;
51             this.line = line;
52             this.keepRunning = keepRunning;
53             initCompletors();
54             setName ("Q2-CLI");
55         }
56     
57         public void run() {
58             try {
59                 while (q2.running()) {
60                     if (line == null) {
61                         reader.addCompletor(completor);
62                         line = reader.readLine("q2> ");
63                         reader.removeCompletor(completor);
64                     }
65                     if (line != null) {
66                         StringTokenizer st = new StringTokenizer (line, ";");
67                         while (st.hasMoreTokens()) {
68                             String n = st.nextToken();
69                             execCommand (n);
70                         }
71                         line = null;
72                     }
73                     if (!keepRunning)
74                         break;
75                 }
76             } catch (IOException e) {
77                 e.printStackTrace (out);
78                 out.flush();
79             }
80         }
81         public void print (String s) {
82             try {
83                 reader.printString (s);
84             } catch (IOException e) {
85                 e.printStackTrace (out);
86                 out.flush();
87             }
88         }
89         public void println (String s) {
90             try {
91                 reader.printString (s);
92                 reader.printNewline ();
93             } catch (IOException e) {
94                 e.printStackTrace (out);
95                 out.flush();
96             }
97         }
98         public boolean confirm (String prompt) throws IOException {
99             return "yes".equalsIgnoreCase (reader.readLine (prompt));
100         }
101         public Q2 getQ2() {
102             return q2;
103         }
104         public ConsoleReader getConsoleReader() {
105             return reader;
106         }
107         public PrintStream getOutputStream () {
108             return out;
109         }
110         public String filter (String element) {
111             return element.startsWith ("org.jpos.q2.cli.") ? element.substring(16).toLowerCase() : null;
112         }
113         private void execCommand (String line) throws IOException {
114             String args[] = parseCommand (line);
115             if (args.length == 0)
116                 return;
117             String command = args[0].toUpperCase();
118             String msg = null;
119             String className = command;
120     
121             if (command.indexOf (".") < 0)
122                 className = "org.jpos.q2.cli."+command;
123             try {
124                 if (className != null) {
125                     Object cmd = q2.getFactory().newInstance (className);
126                     if (cmd instanceof Command) {
127                         try {
128                             args[0] = ISOUtil.unPadLeft(line, ' '); // full line
129                             ((Command) cmd).exec (this, args);
130                         } catch (Exception ex) {
131                             ex.printStackTrace (out);
132                             out.flush();
133                         }
134                     }
135                 }
136             } catch (Exception ex) {
137                 println ("Invalid command '" + command + "'");
138             }
139         }
140         private String[] parseCommand (String line) throws IOException {
141             if (line == null)
142                 return new String[0];
143     
144             StringTokenizer st = new StringTokenizer(line);
145             String[] args = new String[st.countTokens()];
146             for (int i=0; st.hasMoreTokens(); i++)
147                 args[i] = new String (st.nextToken());
148             return args;
149         }
150         private void initCompletors() throws IOException {
151             List l = new LinkedList();
152             l.add(new ClassNameCompletor(this));
153             completor = new ArgumentCompletor(l);
154         }
155         public interface Command {
156             public void exec (CLI cli, String[] args) throws Exception;
157         }
158     }
