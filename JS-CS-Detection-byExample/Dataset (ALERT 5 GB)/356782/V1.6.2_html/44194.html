19     package org.jpos.iso.filter;
20     
21     import org.jpos.core.Configuration;
22     import org.jpos.core.Configurable;
23     import org.jpos.iso.ISOChannel;
24     import org.jpos.iso.ISOFilter;
25     import org.jpos.iso.ISOMsg;
26     import org.jpos.util.LogEvent;
27     import org.jpos.util.ThroughputControl;
28     
29     public class ThroughputControlFilter implements ISOFilter, Configurable {
30         ThroughputControl tc;
31         public ThroughputControlFilter () {
32             super();
33             tc = null;
34         }
35     
36        /**
37         * @param cfg
38         * <ul>
39         *  <li>transactions</li>
40         *  <li>period (in millis)</li>
41         * </ul>
42         */
43         public void setConfiguration (Configuration cfg) {
44             tc = new ThroughputControl (cfg.getInts ("transactions"), cfg.getInts ("period"));
45         }
46     
47         public ISOMsg filter (ISOChannel channel, ISOMsg m, LogEvent evt) 
48         {
49             if (tc != null) {
50                 long delay = tc.control ();
51                 if (delay > 0L)
52                     evt.addMessage ("ThroughputControl=" + delay);
53             }
54             return m;
55         }
56     }
