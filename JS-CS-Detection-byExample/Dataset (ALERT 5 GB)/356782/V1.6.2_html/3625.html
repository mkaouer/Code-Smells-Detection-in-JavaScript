19     package org.jpos.iso.channel;
20     
21     import java.io.IOException;
22     import java.util.List;
23     import java.util.Vector;
24     
25     import org.jpos.core.Configurable;
26     import org.jpos.core.Configuration;
27     import org.jpos.core.ConfigurationException;
28     import org.jpos.iso.ISOChannel;
29     import org.jpos.iso.ISOException;
30     import org.jpos.iso.ISOMsg;
31     import org.jpos.iso.ISOPackager;
32     import org.jpos.util.LogEvent;
33     import org.jpos.util.LogSource;
34     import org.jpos.util.Logger;
35     import org.jpos.util.NameRegistrar;
36     
37     public class ChannelPool implements ISOChannel, LogSource, Configurable, Cloneable {
38         boolean usable = true;
39         String name = "";
40         protected Logger logger;
41         protected String realm;
42         Configuration cfg = null;
43         List pool;
44         ISOChannel current;
45     
46         public ChannelPool () {
47             super ();
48             pool = new Vector ();
49         }
50         public void setPackager(ISOPackager p) {
51             // nothing to do
52         }
53         public synchronized void connect () throws IOException {
54             current = null;
55             LogEvent evt = new LogEvent (this, "connect");
56             evt.addMessage ("pool-size=" + Integer.toString (pool.size()));
57             for (int i=0; i<pool.size(); i++) {
58                 try {
59                     evt.addMessage ("pool-" + Integer.toString (i));
60                     ISOChannel c = (ISOChannel) pool.get (i);
61                     c.connect ();
62                     if (c.isConnected()) {
63                         current = c;
64                         usable = true;
65                         break;
66                     }
67                 } catch (IOException e) {
68                     evt.addMessage (e);
69                 }
70             }
71             if (current == null)
72                 evt.addMessage ("connect failed");
73             Logger.log (evt);
74             if (current == null) {
75                 throw new IOException ("unable to connect");
76             }
77         }
78         public synchronized void disconnect () throws IOException {
79             current = null;
80             LogEvent evt = new LogEvent (this, "disconnect");
81             for (int i=0; i<pool.size(); i++) {
82                 try {
83                     ISOChannel c = (ISOChannel) pool.get (i);
84                     c.disconnect ();
85                 } catch (IOException e) {
86                     evt.addMessage (e);
87                 }
88             }
89             Logger.log (evt);
90         }
91         public synchronized void reconnect() throws IOException {
92             disconnect ();
93             connect ();
94         }
95         public synchronized boolean isConnected() {
96             try {
97                 return getCurrent().isConnected ();
98             } catch (IOException e) {
99                 return false;
100             }
101         }
102         public ISOMsg receive() throws IOException, ISOException {
103             return getCurrent().receive ();
104         }
105         public void send (ISOMsg m) throws IOException, ISOException {
106             getCurrent().send (m);
107         }
108         public void setUsable(boolean b) {
109             this.usable = b;
110         }
111         public void setName (String name) {
112             this.name = name;
113             NameRegistrar.register ("channel."+name, this);
114         }
115         public String getName() {
116             return this.name;
117         }
118         public ISOPackager getPackager () {
119             return (ISOPackager) null;
120         }
121         public void setLogger (Logger logger, String realm) {
122             this.logger = logger;
123             this.realm  = realm;
124         }
125         public String getRealm () {
126             return realm;
127         }
128         public Logger getLogger() {
129             return logger;
130         }
131         public synchronized void setConfiguration (Configuration cfg) 
132             throws ConfigurationException
133         {
134             this.cfg = cfg;
135             String channelName[] = cfg.getAll ("channel");
136             for (int i=0; i<channelName.length; i++) {
137                 try {
138                     addChannel (channelName[i]);
139                 } catch (NameRegistrar.NotFoundException e) {
140                     throw new ConfigurationException (e);
141                 }
142             }
143         }
144         public void addChannel (ISOChannel channel) {
145             pool.add (channel);
146         }
147         public void addChannel (String name) 
148             throws NameRegistrar.NotFoundException
149         {
150             pool.add ((ISOChannel) NameRegistrar.get ("channel."+name));
151         }
152         public void removeChannel (ISOChannel channel) {
153             pool.remove (channel);
154         }
155         public void removeChannel (String name) throws NameRegistrar.NotFoundException {
156             pool.remove ((ISOChannel) NameRegistrar.get ("channel."+name));
157         }
158         public int size() {
159             return pool.size();
160         }
161         public synchronized ISOChannel getCurrent () throws IOException {
162             if (current == null)
163                 connect();
164             else if (!usable)
165                 reconnect();
166     
167             return current;
168         }
169         
170         public Object clone(){
171           try {
172             return (ChannelPool)super.clone();
173           } catch (CloneNotSupportedException e) {
174             throw new InternalError();
175           }
176         }
177     }
