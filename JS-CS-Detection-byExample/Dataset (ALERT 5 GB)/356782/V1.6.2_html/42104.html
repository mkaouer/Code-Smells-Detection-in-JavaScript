135         public void setConfiguration(Configuration cfg) 
136             throws ConfigurationException 
137         {
138             setVetoUnmatched(cfg.getBoolean("veto-unmatched"));
139             setSpace(cfg.get("space",""));
140             setKeyPrefix(cfg.get("key-prefix", ""+hashCode()));
141             setTimeout(cfg.getLong("timeout", TIMEOUT));
142             setOverwriteOriginalFields(cfg.getBoolean("overwrite-original-fields", true));
143             int dir = cfg.get("direction","incoming").equals("incoming") ? 
144                 ISOMsg.INCOMING : ISOMsg.OUTGOING;
145             setMatchDirection(dir);
146             
147             setKey(ISOUtil.toIntArray(cfg.get("key", "11 41")));
148             setSavedFields(ISOUtil.toIntArray(cfg.get("saved-fields", "")));
149             setIgnoredFields(ISOUtil.toIntArray(cfg.get("ignored-fields", "")));
150         }

155         public ISOMsg filter(ISOChannel iSOChannel, ISOMsg m, LogEvent evt) 
156             throws ISOFilter.VetoException 
157         {
158             int[] key = getKey();
159             StringBuffer b = new StringBuffer(getKeyPrefix());
160             for(int i = 0; i < key.length; i++)
161                 b.append("|"+m.getString(key[i]));
162             String skey = b.toString();
163             if(m.getDirection() == getMatchDirection()){
164                 int[] savedFields = getSavedFields();
165                 ISOMsg saved = (ISOMsg)(
166                     (savedFields != null && savedFields.length != 0) ?
167                         m.clone(savedFields) : m.clone());
168                 int[] ignoredFields = getIgnoredFields();
169                 if (ignoredFields != null) saved.unset(ignoredFields);
170                 getSpace().out(skey, saved, getTimeout());
171                 return m;
172             } else {
173                 ISOMsg saved = (ISOMsg)getSpace().inp(skey);
174                 if (saved == null && isVetoUnmatched()) 
175                     throw new VetoException("unmatched iso message");
176                 else if(saved != null) {
177                     if (!isOverwriteOriginalFields()) m.merge(saved);
178                     else {
179                         saved.merge(m);
180                         m = saved;
181                     }
182                 }
183                 return m;
184             }
185         }
