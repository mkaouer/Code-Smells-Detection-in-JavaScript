43     public class VISA1Packager 
44         extends SimpleLogSource implements ISOPackager, VISA1ResponseFilter
45     {
46         public static final byte[] FS = { (byte)'\034' };
47         int[] sequence;
48         int respField;
49         String badResultCode;
50         String okPattern;
51         VISA1ResponseFilter filter;
52     
53         /**
54          * @param sequence array of fields that go to VISA1 request
55          * @param respField where to put response
56          * @param badResultCode (i.e. "05")
57          * @param okPattern (i.e. "AUT. ")
58          */
59         public VISA1Packager 
60             (int[] sequence, int respField, String badResultCode, String okPattern)
61         { 
62             super();
63             this.sequence      = sequence;
64             this.respField     = respField;
65             this.badResultCode = badResultCode;
66             this.okPattern     = okPattern;
67             setVISA1ResponseFilter (this);
68         }
69         public void setVISA1ResponseFilter (VISA1ResponseFilter filter) {
70             this.filter = filter;
71         }
72         protected int handleSpecialField35 (ISOMsg m, Vector v) 
73             throws ISOException
74         {
75             int len = 0;
76             byte[] entryMode = new byte[1];
77             if (m.hasField (35)) {
78                 entryMode[0] = (byte) '\001';
79                 byte[] value = ((String)m.getValue(35)).getBytes();
80                 v.addElement (entryMode);
81                 v.addElement (value);
82                 v.addElement (FS);
83                 len += value.length+2;
84             } else if (m.hasField (2) && m.hasField (14)) {
85                 entryMode[0] = (byte) '\000';
86                 String simulatedTrack2 = 
87                     (String) m.getValue(2) + "=" + (String) m.getValue(14);
88                 v.addElement (entryMode);
89                 v.addElement (simulatedTrack2.getBytes());
90                 v.addElement (FS);
91                 len += simulatedTrack2.length()+2;
92             }
93             return len;
94         }
95         public byte[] pack (ISOComponent c) throws ISOException
96         {
97             LogEvent evt = new LogEvent (this, "pack");
98             try {
99                 if (!(c instanceof ISOMsg))
100                     throw new ISOException
101                         ("Can't call VISA1 packager on non ISOMsg");
102             
103                 ISOMsg m = (ISOMsg) c;
104     
105                 int len  = 0;
106                 Vector v = new Vector();
107                 for (int i=0; i<sequence.length; i++) {
108                     int fld = sequence[i];
109                     if (fld == 35) 
110                         len += handleSpecialField35 (m, v);
111                     else if (m.hasField(fld)) {
112                         byte[] value;
113                         if (fld == 4) {
114                             long l = Long.parseLong (((String)m.getValue(4)));
115                             value = ISOUtil.formatAmount (l,12).trim().getBytes();
116                         }
117                         else
118                             value = ((String)m.getValue(fld)).getBytes();
119                         v.addElement (value);
120                         len += value.length;
121                         if (i < (sequence.length-1)) {
122                             v.addElement (FS);
123                             len++;
124                         }
125                     }
126                 }
127     
128                 int k = 0;
129                 byte[] d = new byte[len];
130                 for (int i=0; i<v.size(); i++) {
131                     byte[] b = (byte[]) v.elementAt(i);
132                     for (int j=0; j<b.length; j++)
133                         d[k++] = b[j];
134                 }
135                 if (logger != null)  // save a few CPU cycle if no logger available
136                     evt.addMessage (ISOUtil.dumpString (d));
137                 return d;
138             } catch (ISOException e) {
139                 evt.addMessage (e);
140                 throw e;
141             } finally {
142                 Logger.log(evt);
143             }
144         }
145     
146         public String guessAutNumber (String s) {
147             StringBuffer buf = new StringBuffer();
148             for (int i=0; i<s.length(); i++)
149                 if (Character.isDigit(s.charAt(i))) 
150                     buf.append (s.charAt(i));
151             if (buf.length() == 0)
152                 return null;
153     
154             while (buf.length() > 6)
155                 buf.deleteCharAt(0);
156             while (buf.length() < 6)
157                 buf.insert(0, "0");
158             
159             return buf.toString();
160         }
161     
162         public int unpack (ISOComponent m, byte[] b) throws ISOException
163         {
164             String response = new String (b);
165             m.set (new ISOField (respField, response));
166             m.set (new ISOField (39, badResultCode));
167             if (response.startsWith (okPattern)) {
168                 String autNumber = filter.guessAutNumber (response);
169                 if (autNumber != null) {
170                     m.set (new ISOField (39, "00"));
171                     m.set (new ISOField (38, autNumber));
172                 }
173             }
174             return b.length;
175         }
176         public void unpack (ISOComponent m, InputStream in) throws ISOException {
177             throw new ISOException ("not implemented");
178         }
179         public String getFieldDescription(ISOComponent m, int fldNumber)
180         {
181             return "VISA 1 fld "+fldNumber;
182         }
183         public ISOMsg createISOMsg() {
184             return new ISOMsg();
185         }
