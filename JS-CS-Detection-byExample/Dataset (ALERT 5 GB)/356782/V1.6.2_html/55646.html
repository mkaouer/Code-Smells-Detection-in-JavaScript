19     package org.jpos.ui.action;
20     
21     import java.awt.event.ActionEvent;
22     import java.awt.event.ActionListener;
23     
24     import org.jdom.Element;
25     import org.jpos.ui.UI;
26     import org.jpos.ui.UIAware;
27     
28     public class Exit implements ActionListener, UIAware {
29         public UI ui;
30         public int exitCode = 0;
31     
32         public Exit () {
33             super();
34         }
35         public void setUI (UI ui, Element e) {
36             this.ui = ui;
37         }
38         public void actionPerformed (ActionEvent ev) {
39             ui.dispose ();
40             try {
41                 exitCode = Integer.parseInt(ev.getActionCommand());
42             } catch (Exception e) { }
43             new Thread() {
44                 public void run() {
45                     System.exit (exitCode);
46                 }
47             }.start();
48         }
49     }
