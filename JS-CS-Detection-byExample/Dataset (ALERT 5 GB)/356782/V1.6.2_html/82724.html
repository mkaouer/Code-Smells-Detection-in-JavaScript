27     /**
28      * ISOFieldPackager ASCII variable len CHAR 
29      * suitable for EuroPay subfield 48<br>
30      * <code>
31      * Format LLTT....
32      * Where LL is the 2 digit field length
33      *       TT is the 2 digit field number (Tag)
34      *       is the field content   
35      * </code>
36      *
37      * @author <a href="mailto:eoin.flood@orbiscom.com">Eoin Flood</a>
38      * @version $Id$
39      * @see ISOComponent
40      */
41     public class IFEP_LLCHAR extends ISOFieldPackager {
42         public IFEP_LLCHAR() {
43             super();
44         }
45         /**
46          * @param len - field len
47          * @param description symbolic descrption
48          */
49         public IFEP_LLCHAR (int len, String description) {
50             super(len, description);
51         }
52         /**
53          * @param c - a component
54          * @return packed component
55          * @exception ISOException
56          */
57         public byte[] pack (ISOComponent c) throws ISOException {
58             int len;
59             String s = (String) c.getValue();
60         
61             if ((len=s.length()) > getLength() || len>97)   // paranoia settings
62                 throw new ISOException (
63                     "invalid len "+len +" packing LLEPCHAR field "
64                     +(Integer) c.getKey()
65                 );
66     
67             return (
68                 ISOUtil.zeropad(Integer.toString(len+2), 2) 
69                +ISOUtil.zeropad(((Integer) c.getKey()).toString(), 2) 
70                +s
71             ).getBytes();
72         }
73     
74         /**
75          * @param c - the Component to unpack
76          * @param b - binary image
77          * @param offset - starting offset within the binary image
78          * @return consumed bytes
79          * @exception ISOException
80          */
81         public int unpack (ISOComponent c, byte[] b, int offset)
82             throws ISOException
83         {
84             int len = Integer.parseInt(new String(b, offset, 2));
85             if (!(c instanceof ISOField))
86                 throw new ISOException 
87                     (c.getClass().getName() + " is not an ISOField");
88     
89             ((ISOField)c).setFieldNumber (
90                 Integer.parseInt(new String(b, offset+2, 2))
91             );
92             c.setValue (new String (b, offset+4, len-2));
93             return len + 2;
94         }
95         public void unpack (ISOComponent c, InputStream in) 
96             throws IOException, ISOException
97         {
98     
99             if (!(c instanceof ISOField))
100                 throw new ISOException 
101                     (c.getClass().getName() + " is not an ISOField");
102     
103             int len   = Integer.parseInt(new String(readBytes (in, 2)));
104             int fldno = Integer.parseInt(new String(readBytes (in, 2)));
105             ((ISOField)c).setFieldNumber (fldno);
106             c.setValue (new String (readBytes (in, len-2)));
107         }
108     
109         public int getMaxPackedLength() {
110             return getLength() + 2;
111         }
112     }
