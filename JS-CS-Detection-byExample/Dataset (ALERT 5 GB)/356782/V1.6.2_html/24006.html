22     /**
23      * @author apr@cs.com.uy
24      * @version $Id$
25      * @since jPOS 1.1
26      *
27      * CardAgents relies on a Configuration object to provide
28      * runtime configuration parameters such as merchant number, etc.
29      */
30     public interface Configuration {
31         public String get       (String propertyName);
32         /**
33          * @param propertyName  ditto
34          * @param propertyValue ditto
35          * @return all properties with a given name (or a zero length string)
36          */
37         public String[] getAll  (String propertyName);
38         public int[] getInts  (String propertyName);
39         public long[] getLongs (String propertyName);
40         public double[] getDoubles (String propertyName);
41         public boolean[] getBooleans (String propertyName);
42         public String get       (String propertyName, String defaultValue);
43         public int getInt       (String propertyName);
44         public int getInt       (String propertyName, int defaultValue);
45         public long getLong     (String propertyName);
46         public long getLong     (String propertyName, long defaultValue);
47         public double getDouble (String propertyName);
48         public double getDouble (String propertyName, double defaultValue);
49         public boolean getBoolean (String propertyName);
50         public boolean getBoolean (String propertyName, boolean defaultValue);
51         /**
52          * @param name the Property name
53          * @param value typically a String, but could be a String[] too
54          */
55         public void put (String name, Object value);
56     }
