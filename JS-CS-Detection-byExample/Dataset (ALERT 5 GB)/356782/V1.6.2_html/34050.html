19     package org.jpos.util;
20     
21     import java.util.Vector;
22     
23     import org.jpos.core.Configurable;
24     import org.jpos.core.Configuration;
25     import org.jpos.core.ConfigurationException;
26     import org.jpos.iso.ISOException;
27     import org.jpos.iso.ISOField;
28     import org.jpos.iso.ISOMsg;
29     import org.jpos.iso.ISOUtil;
30     
31     /**
32      * Protects selected fields from LogEvents.
33      *
34      * ProtectedLogListener acts like a filter for Event logs,
35      * it should be defined _before_ other standard LogListeners
36      * such as SimpleLogListener or RotateLogListeners.<br>
37      * i.e.
38      * <pre>
39      * <logger name="qsp">
40      *   <log-listener class="org.jpos.util.SimpleLogListener"/>
41      *   <log-listener class="org.jpos.util.ProtectedLogListener">
42      *     <property name="protect" value="2 35 45 55" />
43      *     <property name="wipe"    value="48" />
44      *   </log-listener>
45      *   <log-listener class="org.jpos.util.RotateLogListener">
46      *     <property name="file" value="/tmp/qsp.log" />
47      *     <property name="window" value="86400" />
48      *     <property name="copies" value="5" />
49      *     <property name="maxsize" value="1000000" />
50      *   </log-listener>
51      * </logger>
52      * </pre>
53      * 
54      * Order is important. In the previous example SimpleLogListener
55      * will dump unprotected LogEvents while RotateLogListener will
56      * dump protected ones (for selected fields)
57      *
58      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
59      * @version $Revision$ $Date$
60      * @see org.jpos.core.Configurable
61      * @since jPOS 1.3.3
62      */
63     public class ProtectedLogListener implements LogListener, Configurable
64     {
65         String[] protectFields = null;
66         String[] wipeFields    = null;
67         Configuration cfg   = null;
68         public static final String WIPED = "[WIPED]";
69         public static final byte[] BINARY_WIPED = ISOUtil.hex2byte ("AA55AA55");
70     
71         public ProtectedLogListener () {
72             super();
73         }
74     
75        /**
76         * Configure this ProtectedLogListener<br>
77         * Properties:<br>
78         * <ul>
79         *  <li>[protect]   blank separated list of fields to be protected
80         *  <li>[wipe]      blank separated list of fields to be wiped
81         * </ul>
82         * @param cfg Configuration 
83         * @throws ConfigurationException
84         */
85         public void setConfiguration (Configuration cfg)
86             throws ConfigurationException
87         {
88             this.cfg = cfg;
89             protectFields = ISOUtil.toStringArray (cfg.get ("protect", ""));
90             wipeFields    = ISOUtil.toStringArray (cfg.get ("wipe", ""));
91         }
92         public synchronized LogEvent log (LogEvent ev) {
93             Vector payLoad = ev.getPayLoad();
94             int size = payLoad.size();
95             for (int i=0; i<size; i++) {
96                 Object obj = payLoad.elementAt (i);
97                 if (obj instanceof ISOMsg) {
98                     ISOMsg m = (ISOMsg) ((ISOMsg) obj).clone();
99                     try {
100                         checkProtected (m);
101                         checkHidden (m);
102                     } catch (ISOException e) { 
103                         ev.addMessage (e);
104                     }
105                     payLoad.setElementAt (m, i);
106                 }
107             }
108             return ev;
109         }
110         private void checkProtected (ISOMsg m) throws ISOException {
111             for (int i=0; i<protectFields.length; i++) {
112                 String f = protectFields[i];
113                 Object v = null;
114                 try {
115                     v = m.getValue (f);
116                 } catch (ISOException e) {
117                     // ignore error
118                 }
119                 if (v != null) {
120                     if (v instanceof String)
121                         m.set (f, ISOUtil.protect ((String) v));
122                     else
123                         m.set (f, BINARY_WIPED);
124                 }
125             }
126         }
127         private void checkHidden (ISOMsg m) throws ISOException {
128             for (int i=0; i<wipeFields.length; i++) {
129                 String f = wipeFields[i];
130                 Object v = null;
131                 try {
132                     v = m.getValue (f);
133                 } catch (ISOException e) {
134                     // ignore error
135                 }
136                 if (v != null) {
137                     if (v instanceof String)
138                         m.set (f, WIPED);
139                     else
140                         m.set (f, BINARY_WIPED);
141                 }
142             }
143         }
144     }
