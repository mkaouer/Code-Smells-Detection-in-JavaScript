19     package org.jpos.space;
20     
21     import java.io.Serializable;
22     import java.rmi.RemoteException;
23     import java.rmi.registry.LocateRegistry;
24     import java.rmi.registry.Registry;
25     import java.rmi.server.ExportException;
26     import java.rmi.server.RemoteObject;
27     import java.rmi.server.RemoteRef;
28     import java.rmi.server.RemoteStub;
29     import java.rmi.server.UnicastRemoteObject;
30     import java.util.Set;
31     
32     import javax.naming.InitialContext;
33     import javax.naming.NamingException;
34     
35     import org.jpos.core.Configuration;
36     import org.jpos.core.ConfigurationException;
37     import org.jpos.core.ReConfigurable;
38     
39     /**
40      * RMI Space Proxy 
41      * @author Alejandro Revilla
42      * @author Niclas Hedhman
43      * @version $Revision$ $Date$
44      * @since 1.4.9
45      */
46     public class SpaceProxy implements RemoteSpace, ReConfigurable {
47         Space sp;
48         Configuration cfg;
49         private RemoteRef ref;
50         private RemoteStub stub;
51         public SpaceProxy () throws RemoteException {
52             super();
53             sp = SpaceFactory.getSpace ();
54             startService ();
55         }
56         public SpaceProxy (String spaceUri) throws RemoteException {
57             super ();
58             sp = SpaceFactory.getSpace (spaceUri);
59             startService ();
60         }
61     
62         private void startService () throws RemoteException 
63         {
64             try {
65                 LocateRegistry.createRegistry (Registry.REGISTRY_PORT);
66             } catch (ExportException e) {
67                 // registry already exists
68             }
69             stub = UnicastRemoteObject.exportObject (this);
70             ref  = stub.getRef();
71         }
72     
73         public void out (Serializable key, Serializable value) 
74             throws RemoteException
75         {
76             sp.out (key, value);
77         }
78         public void out (Serializable key, Serializable value, long timeout)
79             throws RemoteException
80         {
81             sp.out (key, value, timeout);
82         }
83         public Serializable in (Serializable key)
84             throws RemoteException
85         {
86             return (Serializable) sp.in (key);
87         }
88         public Serializable rd  (Serializable key)
89             throws RemoteException
90         {
91             return (Serializable) sp.rd (key);
92         }
93         public Serializable in  (Serializable key, long timeout)
94             throws RemoteException
95         {
96             return (Serializable) sp.in (key, timeout);
97         }
98         public Serializable rd  (Serializable key, long timeout)
99             throws RemoteException
100         {
101             return (Serializable) sp.rd (key, timeout);
102         }
103         public Serializable inp (Serializable key)
104             throws RemoteException
105         {
106             return (Serializable) sp.inp (key);
107         }
108         public Serializable rdp (Serializable key)
109             throws RemoteException
110         {
111             return (Serializable) sp.rdp (key);
112         }
113         public void shutdown() {
114             try {
115                 if (UnicastRemoteObject.unexportObject (this, false)) 
116                     return;
117                 Thread.sleep (5000);
118                 UnicastRemoteObject.unexportObject (this, true);
119             } catch (Exception e) {
120                 // nothing to do .. we're shutting down ...
121             }
122         }
123         public void setConfiguration (Configuration cfg) 
124             throws ConfigurationException
125         {
126             this.cfg = cfg;
127             try {
128                 InitialContext ctx = new InitialContext ();
129                 ctx.rebind (cfg.get ("name"), stub);
130             } catch (NamingException e) {
131                 throw new ConfigurationException (e);
132             }
133         }
134         public Set getKeySet () {
135             if (sp instanceof LocalSpace) 
136                 return ((LocalSpace)sp).getKeySet ();
137             else
138                 return null;
139         }
140         public String toString() {
141             if (ref == null)
142                 return getClass().getName() + "[<unexported>]";
143             else
144                 return getClass().getName() + "[" + ref.remoteToString() + "]";
145         }
146         public int hashCode() {
147             if (ref == null )
148                 return super.hashCode();
149             else
150                 return ref.remoteHashCode();
151         }
152         public boolean equals (Object obj) {
153             if (obj instanceof RemoteObject) {
154                 if (ref == null) {
155                     return obj == this;
156                 } else {
157                     RemoteRef otherRef = ((RemoteObject)obj).getRef();
158                     return ref.remoteEquals (otherRef);
159                 }
160             } else if (obj != null) {
161                 return obj.equals (this);
162             } else {
163                 return false;
164             }
165         }
166     }
