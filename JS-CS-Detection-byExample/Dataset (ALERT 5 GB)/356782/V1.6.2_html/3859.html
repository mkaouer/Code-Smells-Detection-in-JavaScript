19     package  org.jpos.security;
20     
21     
22     /**
23      * A class that implements the SecurityModuleAdapter interface would act as an
24      * adapter to the real security module device (by communicating with it using
25      * its proprietary protocol).
26      *
27      * But application programmers will be communicating
28      * with the security module using this simple interface.
29      *
30      * @todo support for PIN Verification API's and RSA
31      * @author Hani S. Kirollos
32      * @version $Revision$ $Date$
33      */
34     public interface SMAdapter {
35         /**
36          * DES Key Length <code>LENGTH_DES</code> = 64.
37          */
38         public static final short LENGTH_DES = 64;
39         /**
40          * Triple DES (2 keys) <code>LENGTH_DES3_2KEY</code> = 128.
41          */
42         public static final short LENGTH_DES3_2KEY = 128;
43         /**
44          * Triple DES (3 keys) <code>LENGTH_DES3_3KEY</code> = 192.
45          */
46         public static final short LENGTH_DES3_3KEY = 192;
47         /**
48          * ZMK: Zone Master Key is a DES (or Triple-DES) key-encryption key which is distributed
49          * manually in order that further keys can be exchanged automatically.
50          */
51         public static final String TYPE_ZMK = "ZMK";
52     
53         /**
54          * ZPK: Zone PIN Key.
55          *
56          * is a DES (or Triple-DES) data-encrypting key which is distributed
57          * automatically and is used to encrypt PINs for transfer between
58          * communicating parties (e.g. between acquirers and issuers).
59          */
60         public static final String TYPE_ZPK = "ZPK";
61     
62         /**
63          * TMK: Terminal Master Key.
64          *
65          * is a  DES (or Triple-DES) key-encrypting key which is distributed
66          * manually, or automatically under a previously installed TMK. It is
67          * used to distribute data-encrypting keys, whithin a local network,
68          * to an ATM or POS terminal or similar.
69          */
70         public static final String TYPE_TMK = "TMK";
71     
72         /**
73          * TPK: Terminal PIN Key.
74          *
75          * is a  DES (or Triple-DES) data-encrypting key which is used
76          * to encrypt PINs for transmission, within a local network,
77          * between the terminal and the terminal data acquirer.
78          */
79         public static final String TYPE_TPK = "TPK";
80     
81         /**
82          * TAK: Terminal Authentication Key.
83          *
84          * is a  DES (or Triple-DES) data-encrypting key which is used to
85          * generate and verify a Message Authentication Code (MAC) when data
86          * is transmitted, within a local network, between the terminal and
87          * the terminal data acquirer.
88          */
89         public static final String TYPE_TAK = "TAK";
90     
91         /**
92          * PVK: PIN Verification Key.
93          * is a  DES (or Triple-DES) data-encrypting key which is used to
94          * generate and verify PIN verification data and thus verify the
95          * authenticity of a PIN.
96          */
97         public static final String TYPE_PVK = "PVK";
98     
99         /**
100          * CVK: Card Verification Key.
101          *
102          * is similar for PVK but for card information instead of PIN
103          */
104         public static final String TYPE_CVK = "CVK";
105     
106         /**
107          * BDK: Base Derivation Key.
108          * is a  Triple-DES key-encryption key used to derive transaction
109          * keys in DUKPT (see ANSI X9.24)
110          */
111         public static final String TYPE_BDK = "BDK";
112     
113         /**
114          * ZAK: Zone Authentication Key.
115          *
116          * a  DES (or Triple-DES) data-encrypting key that is distributed
117          * automatically, and is used to generate and verify a Message
118          * Authentication Code (MAC) when data is transmitted between
119          * communicating parties (e.g. between acquirers and issuers)
120          */
121         public static final String TYPE_ZAK = "ZAK";
122     
123         /**
124          * PIN Block Format adopted by ANSI (ANSI X9.8) and is one of
125          * two formats supported by the ISO (ISO 95641 - format 0).
126          */
127         public static final byte FORMAT01 = (byte)01;
128     
129         /**
130          * PIN Block Format 02 supports Douctel ATMs.
131          */
132         public static final byte FORMAT02 = (byte)02;
133     
134         /**
135              * PIN Block Format 03 is the Diabold Pin Block format.
136              */
137         public static final byte FORMAT03 = (byte)03;
138     
139         /**
140          * PIN Block Format 04 is the PIN block format adopted
141          * by the PLUS network.
142          */
143         public static final byte FORMAT04 = (byte)04;
144     
145         /**
146          * PIN Block Format 05 is the ISO 9564-1 Format 1 PIN Block.
147          */
148         public static final byte FORMAT05 = (byte)05;
149         /**
150          * Proprietary PIN Block format.
151          *
152          * Most Security Modules use a proprietary PIN Block format
153          * when encrypting the PIN under the LMK of the Security Module
154          * hence this format (FORMAT00).
155          *
156          * <p>
157          * This is not a standard format, every Security Module would
158          * interpret FORMAT00 differently.
159          *
160          * So, no interchange would accept PIN Blocks from other interchanges
161          * using this format. It is useful only when working with PIN's inside
162          * your own interchange.
163          * </p>
164          */
165         public static final byte FORMAT00 = (byte)00;
166     
167         /**
168          * Generates a random DES Key.
169          *
170          * @param keyType type of the key to be generated (TYPE_ZMK, TYPE_TMK...etc)
171          * @param keyLength bit length of the key to be generated (LENGTH_DES, LENGTH_DES3_2KEY...)
172          * @return the random key secured by the security module<BR>
173          * @throws SMException
174          */
175         public SecureDESKey generateKey (short keyLength, String keyType) throws SMException;
176     
177     
178     
179         /**
180          * Imports a key from encryption under a KEK (Key-Encrypting Key)
181          * to protection under the security module.
182          *
183          * @param keyLength bit length of the key to be imported (LENGTH_DES, LENGTH_DES3_2KEY...etc)
184          * @param keyType type of the key to be imported (TYPE_ZMK, TYPE_TMK...etc)
185          * @param encryptedKey key to be imported encrypted under KEK
186          * @param kek the key-encrypting key
187          * @param checkParity if true, the key is not imported unless it has adjusted parity
188          * @return imported key secured by the security module
189          * @throws SMException if the parity of the imported key is not adjusted AND checkParity = true
190          */
191         public SecureDESKey importKey (short keyLength, String keyType, byte[] encryptedKey,
192                 SecureDESKey kek, boolean checkParity) throws SMException;
193     
194     
195     
196         /**
197          * Exports secure key to encryption under a KEK (Key-Encrypting Key).
198          * @param key the secure key to be exported
199          * @param kek the key-encrypting key
200          * @return the exported key (key encrypted under kek)
201          * @throws SMException
202          */
203         public byte[] exportKey (SecureDESKey key, SecureDESKey kek) throws SMException;
204     
205         /**
206          * Encrypts a clear pin under LMK.
207          *
208          * CAUTION: The use of clear pin presents a significant security risk
209          * @param pin clear pin as entered by card holder
210          * @param accountNumber The 12 right-most digits of the account number excluding the check digit. Should also function correctly if the complete account number, including the check digit is passed.
211          * @return PIN under LMK
212          * @throws SMException
213          */
214         public EncryptedPIN encryptPIN (String pin, String accountNumber) throws SMException;
215     
216         /**
217          * Decrypts an Encrypted PIN (under LMK).
218          * CAUTION: The use of clear pin presents a significant security risk
219          * @param pinUnderLmk
220          * @return clear pin as entered by card holder
221          * @throws SMException
222          */
223         public String decryptPIN (EncryptedPIN pinUnderLmk) throws SMException;
224     
225         /**
226          * Imports a PIN from encryption under KD (Data Key)
227          * to encryption under LMK.
228          *
229          * @param pinUnderKd1 the encrypted PIN
230          * @param kd1 Data Key under which the pin is encrypted
231          * @return pin encrypted under LMK
232          * @throws SMException
233          */
234         public EncryptedPIN importPIN (EncryptedPIN pinUnderKd1, SecureDESKey kd1) throws SMException;
235     
236     
237     
238         /**
239          * Translates a PIN from encrytion under KD1 to encryption under KD2.
240          *
241          * @param pinUnderKd1 pin encrypted under KD1
242          * @param kd1 Data Key (also called session key) under which the pin is encrypted
243          * @param kd2 the destination Data Key 2 under which the pin will be encrypted
244          * @param destinationPINBlockFormat the PIN Block Format of the exported encrypted PIN
245          * @return pin encrypted under KD2
246          * @throws SMException
247          */
248         public EncryptedPIN translatePIN (EncryptedPIN pinUnderKd1, SecureDESKey kd1,
249                 SecureDESKey kd2, byte destinationPINBlockFormat) throws SMException;
250     
251     
252     
253         /**
254          * Imports a PIN from encryption under a transaction key to encryption
255          * under LMK.
256          *
257          * The transaction key is derived from the Key Serial Number and the Base Derivation Key using DUKPT (Derived Unique Key per Transaction). See ANSI X9.24 for more information.
258          * @param pinUnderDuk pin encrypted under a transaction key
259          * @param ksn Key Serial Number (also called Key Name, in ANSI X9.24) needed to derive the transaction key
260          * @param bdk Base Derivation Key, used to derive the transaction key underwhich the pin is encrypted
261          * @return pin encrypted under LMK
262          * @throws SMException
263          */
264         public EncryptedPIN importPIN (EncryptedPIN pinUnderDuk, KeySerialNumber ksn,
265                 SecureDESKey bdk) throws SMException;
266     
267     
268     
269         /**
270          * Translates a PIN from encryption under a transaction key to
271          * encryption under a KD (Data Key).
272          *
273          * The transaction key is derived from the Key Serial Number and the Base Derivation Key using DUKPT (Derived Unique Key per Transaction). See ANSI X9.24 for more information.
274          * @param pinUnderDuk pin encrypted under a DUKPT transaction key
275          * @param ksn Key Serial Number (also called Key Name, in ANSI X9.24) needed to derive the transaction key
276          * @param bdk Base Derivation Key, used to derive the transaction key underwhich the pin is encrypted
277          * @param kd2 the destination Data Key (also called session key) under which the pin will be encrypted
278          * @param destinationPINBlockFormat the PIN Block Format of the translated encrypted PIN
279          * @return pin encrypted under kd2
280          * @throws SMException
281          */
282         public EncryptedPIN translatePIN (EncryptedPIN pinUnderDuk, KeySerialNumber ksn,
283                 SecureDESKey bdk, SecureDESKey kd2, byte destinationPINBlockFormat) throws SMException;
284     
285     
286     
287         /**
288          * Exports a PIN from encryption under LMK to encryption under a KD
289          * (Data Key).
290          *
291          * @param pinUnderLmk pin encrypted under LMK
292          * @param kd2 the destination data key (also called session key) under which the pin will be encrypted
293          * @param destinationPINBlockFormat the PIN Block Format of the exported encrypted PIN
294          * @return pin encrypted under kd2
295          * @throws SMException
296          */
297         public EncryptedPIN exportPIN (EncryptedPIN pinUnderLmk, SecureDESKey kd2, byte destinationPINBlockFormat) throws SMException;
298     
299     
300     
301         /**
302          * Generates CBC-MAC (Cipher Block Chaining Message Authentication Code)
303          * for some data.
304          *
305          * @param data the data to be MACed
306          * @param kd the key used for MACing
307          * @return the MAC
308          * @throws SMException
309          */
310         public byte[] generateCBC_MAC (byte[] data, SecureDESKey kd) throws SMException;
311     }
