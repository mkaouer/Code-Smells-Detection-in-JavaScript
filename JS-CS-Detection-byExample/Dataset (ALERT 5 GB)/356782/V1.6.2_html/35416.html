19     package  org.jpos.util;
20     import java.io.PrintStream;
21     
22     import org.jpos.iso.ISOUtil;
23     
24     /**
25      * <p>
26      * A simple general purpose loggeable message.
27      * </p>
28      * @author Hani S. Kirollos
29      * @version $Revision$ $Date$
30      */
31         public class SimpleMsg
32                 implements Loggeable {
33             String tagName;
34             String msgName;
35             Object msgContent;
36     
37             public SimpleMsg (String tagName, String msgName, Object msgContent) {
38                 this.tagName = tagName;
39                 this.msgName = msgName;
40                 this.msgContent = msgContent;
41             }
42     
43             public SimpleMsg (String tagName, String msgName, byte[] msgContent) {
44                 this(tagName, msgName, ISOUtil.hexString(msgContent));
45             }
46     
47             public SimpleMsg (String tagName, String msgName, boolean msgContent) {
48                 this(tagName, msgName, new Boolean(msgContent));
49             }
50     
51             public SimpleMsg (String tagName, String msgName, short msgContent) {
52                 this(tagName, msgName, new Short(msgContent));
53             }
54     
55             public SimpleMsg (String tagName, String msgName, int msgContent) {
56                 this(tagName, msgName, new Integer(msgContent));
57             }
58     
59             public SimpleMsg (String tagName, String msgName, long msgContent) {
60                 this(tagName, msgName, new Long(msgContent));
61             }
62     
63             /**
64              * dumps message
65              * @param p a PrintStream usually supplied by Logger
66              * @param indent indention string, usually suppiled by Logger
67              * @see org.jpos.util.Loggeable
68              */
69             public void dump (PrintStream p, String indent) {
70                 String inner = indent + "  ";
71                 p.print(indent + "<" + tagName);
72                 p.print(" name=\"" + msgName + "\"");
73                 p.println(">");
74                 if (msgContent instanceof SimpleMsg[]) {
75                     // dump sub messages
76                     for (int i = 0; i < ((SimpleMsg[])msgContent).length; i++)
77                         ((SimpleMsg[])msgContent)[i].dump(p, inner);
78                 }
79                 else if (msgContent instanceof Loggeable)
80                     ((Loggeable)msgContent).dump(p, inner);
81                 else
82                     p.println(inner + msgContent.toString());
83                 p.println(indent + "</" + tagName + ">");
84             }
85         }
