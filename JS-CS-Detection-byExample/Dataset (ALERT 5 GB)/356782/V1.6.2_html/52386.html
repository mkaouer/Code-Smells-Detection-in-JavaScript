19     package org.jpos.util;
20     
21     import java.util.Timer;
22     
23     /**
24      * Timer accesor singleton
25      * @author Alejandro P. Revilla
26      * @version $Id$
27      */
28     
29     public class DefaultTimer {
30         private static Timer defaultTimer = null;
31     
32         private DefaultTimer() { }
33     
34         public static Timer getTimer() {
35             if (defaultTimer == null) {
36                 synchronized (DefaultTimer.class) {
37                     if (defaultTimer == null) 
38                         defaultTimer = new Timer(true);
39                 }
40             }
41             return defaultTimer;
42         }
43     }
