33     public class QMUXTestCase extends TestCase implements ISOResponseListener {
34         Q2 q2;
35         Space sp;
36         MUX mux;
37         boolean expiredCalled;
38         ISOMsg responseMsg;
39         Object receivedHandback;
40     
41         public void setUp () throws Exception {
42             sp = SpaceFactory.getSpace();
43             q2 = new Q2(new String[] { "-d", "../test/org/jpos/q2/iso" });
44             expiredCalled=false;
45             new Thread(q2).start();
46             Thread.sleep (2000L);
47             try {
48                 mux = (MUX) NameRegistrar.get ("mux.mux");
49             } catch (NameRegistrar.NotFoundException e) { 
50                 fail ("MUX not found");
51             }
52             receivedHandback = null;
53         }
54         public void testExpiredMessage() throws Exception {
55             mux.request (createMsg("000001"), 500L, this, "Handback One");
56             assertFalse ("expired called too fast", expiredCalled);
57             assertNotNull ("Space doesn't contain message key", 
58                 sp.rdp ("send.0800000000029110001000001.req")
59             );
60             Thread.sleep (1000L);
61             assertTrue ("expired has not been called after 1 second", expiredCalled);
62             assertNull ("Cleanup failed, Space still contains message key", 
63                 sp.rdp ("send.0800000000029110001000001.req")
64             );
65             assertEquals ("Handback One not received", "Handback One", receivedHandback);
66         }
67         public void testAnsweredMessage() throws Exception {
68             mux.request (createMsg("000002"), 500L, this, "Handback Two");
69             assertFalse ("expired called too fast", expiredCalled);
70             ISOMsg m = (ISOMsg) sp.in ("send", 500L);
71             assertNotNull ("Message not received by pseudo-channel", m);
72             assertNotNull ("Space doesn't contain message key", 
73                 sp.rdp ("send.0800000000029110001000002.req")
74             );
75             m.setResponseMTI();
76             sp.out ("receive", m);
77             Thread.sleep (100L);
78             assertNotNull ("Response not received", responseMsg);
79             Thread.sleep (1000L);
80             assertFalse ("Response received but expired was called", expiredCalled);
81             assertNull ("Cleanup failed, Space still contains message key", 
82                 sp.rdp ("send.0800000000029110001000002.req")
83             );
84             assertEquals ("Handback Two not received", "Handback Two", receivedHandback);
85         }
86         public void tearDown() throws Exception {
87             Thread.sleep (1500); // let the thing run
88             q2.shutdown(true);
89         }
90         private ISOMsg createMsg(String stan) throws ISOException {
91             ISOMsg m = new ISOMsg("0800");
92             m.set (11, stan);
93             m.set (41, "29110001"); // our favourite test terminal
94             return m;
95         }
96         public void responseReceived (ISOMsg m, Object handBack) {
97             responseMsg = m;
98             receivedHandback = handBack;
99         }
100         public void expired (Object handBack) {
101             expiredCalled=true;
102             receivedHandback = handBack;
103         }
104         
105         public void testMTIMapping() throws ISOException{
106             String[] requests  = new String[] {"0100","0101","0400","0401","0420","0800","1100","1800","1804","1820","1820","1200","1220"};
107             String[] responses = new String[] {"0110","0110","0410","0410","0430","0810","1110","1810","1814","1824","1830","1210","1230"};
108                     
109             assertEquals("Request/Response string arrays must hold same number of entries",requests.length,responses.length);
110             
111             ISOMsg request = new ISOMsg();
112             ISOMsg response = new ISOMsg();
113             
114             for (int i = 0; i<requests.length;i++) {
115                 request.setMTI(requests[i]);
116                 response.setMTI(responses[i]);
117                 assertEquals(((QMUX)mux).getKey(request),((QMUX)mux).getKey(response));    
118             }
119      
120         }
121     }
