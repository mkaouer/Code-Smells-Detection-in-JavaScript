86         private String [] attributeNames = {  
87     		    "class",
88     		    "packager",
89     	   	    "header",
90     		    "host",
91     		    "port",
92     		    "timeout",
93     		    "name"  // used here to update digest
94     		};
95     
96         public void config (QSP qsp, Node node) throws ConfigurationException
97         {
98     	ISOChannel channel;
99     	LogEvent evt = new LogEvent (qsp, "config-channel");
100     	String name = node.getAttributes().getNamedItem ("name").getNodeValue();
101     	try {
102     	    channel = getChannel (name);
103     	} catch (NameRegistrar.NotFoundException e) {
104     	    channel = createChannel (name, node, evt);
105     	    channel.setName (name);
106     	}
107     	Logger.log (evt);
108         }
109         public void reconfig (QSP qsp, Node node) throws ConfigurationException
110         {
111     	String name = node.getAttributes().getNamedItem ("name").getNodeValue();
112     	LogEvent evt = new LogEvent (qsp, "re-config-channel", name);
113     	try {
114     	    ISOChannel channel = getChannel (name);
115     
116     	    byte[] digest = (byte[])NameRegistrar.get (name + ".digest");
117     	    byte[] previousDigest = new byte [digest.length];
118     	    System.arraycopy (digest, 0, previousDigest, 0, digest.length);
119     	    Properties props = ConfigUtil.addAttributes (
120     		node, attributeNames, null, evt
121     	    );
122     	    digest = (byte[]) props.get (ConfigUtil.DIGEST_PROPERTY);
123     	    NameRegistrar.register (name + ".digest", digest);
124     
125     	    if (channel instanceof LogSource)
126     		((LogSource)channel).setLogger (
127     		    ConfigLogger.getLogger (node),
128     		    ConfigLogger.getRealm  (node)
129     		);
130     	    Configuration cfg = new SimpleConfiguration (
131     		ConfigUtil.addProperties (node, props, evt)
132     	    );
133     	    if (channel instanceof ReConfigurable) {
134     		((Configurable)channel).setConfiguration (cfg);
135     	    }
136     	    if (Arrays.equals (previousDigest, digest)) {
137     		evt.addMessage ("<unchanged/>");
138     		return;
139     	    }
140     	    evt.addMessage ("<modified/>");
141     
142     	    if (cfg.get ("timeout", null) != null)
143     		ConfigUtil.invoke (
144     		    channel, "setTimeout", new Integer(cfg.getInt ("timeout"))
145     		);
146     
147     	    ConfigUtil.invoke (channel, "setHeader", cfg.get ("header", null));
148     	} catch (NameRegistrar.NotFoundException e) {
149     	    evt.addMessage (e);
150     	} finally {
151     	    Logger.log (evt);
152     	}
153         }

155         private ISOChannel createChannel (String name, Node node, LogEvent evt)
156     	throws ConfigurationException
157         {
158     	Properties props = ConfigUtil.addAttributes
159     	    (node, attributeNames, null, evt);
160     
161     	NameRegistrar.register 
162     	    (name + ".digest", props.get (ConfigUtil. DIGEST_PROPERTY));
163     
164     	Logger logger = ConfigLogger.getLogger (node);
165     	String realm  = ConfigLogger.getRealm  (node);
166     
167     	try {
168     	    ISOChannel channel = newChannel (
169     		props.getProperty ("class"),
170     		props.getProperty ("packager"), 
171     		logger, realm
172     	    );
173     	    Configuration cfg = new SimpleConfiguration (
174     		ConfigUtil.addProperties (node, props, evt)
175     	    );
176     	    if (channel instanceof Configurable) {
177     		((Configurable)channel).setConfiguration (cfg);
178     	    }
179     
180     	    if (cfg.get ("timeout", null) != null)
181     		ConfigUtil.invoke (
182     		    channel, "setTimeout", new Integer(cfg.getInt ("timeout"))
183     		);
184     
185     	    ConfigUtil.invoke (channel, "setHeader", cfg.get ("header", null));
186     
187     	    if (cfg.get ("connect").equals ("yes")) {
188     		try {
189     		    channel.connect();
190     		} catch (IOException e) {
191     		    evt.addMessage (e);
192     		}
193     	    }
194     
195     	    JPanel panel = ConfigControlPanel.getPanel (node);
196     	    if (panel != null) 
197     		panel.add (new ISOChannelPanel (channel, name));
198     
199     	    return channel;
200     	} catch (ConfigurationException e) {
201     	    throw e;
202     	} catch (ISOException e) {
203     	    throw new ConfigurationException ("error creating channel", e);
204     	} catch (NullPointerException e) {
205     	    throw new ConfigurationException ("error creating channel", e);
206     	}
207         }
208     
209         private ISOChannel newChannel
210     	(String channelName, String packagerName, Logger logger, String realm)
211     	throws ConfigurationException
212         {
213             ISOChannel channel  = null;
214     	ISOPackager packager= null;
215             try {
216                 Class c = Class.forName(channelName);
217                 if (c != null) {
218                     channel = (ISOChannel) c.newInstance();
219     		if (packagerName != null) {
220     		    Class p = Class.forName(packagerName);
221     		    packager = (ISOPackager) p.newInstance();
222     		    channel.setPackager(packager);
223     		}
224     		if (logger != null && (channel instanceof LogSource))
225     		    ((LogSource) channel) .
226     			setLogger (logger, realm + ".channel");
227                 }
228             } catch (ClassNotFoundException e) {
229     	    throw new ConfigurationException ("newChannel:"+channelName, e);
230             } catch (InstantiationException e) {
231     	    throw new ConfigurationException ("newChannel:"+channelName, e);
232             } catch (IllegalAccessException e) {
233     	    throw new ConfigurationException ("newChannel:"+channelName, e);
234     	}
235             return channel;
236         }

238         public static ISOChannel getChannel (String name)
239     	throws NameRegistrar.NotFoundException
240         {
241     	return (ISOChannel) NameRegistrar.get ("channel."+name);
242         }
243         public static ISOChannel getChannel (Node node) {
244     	Node n = node.getAttributes().getNamedItem ("name");
245     	if (n != null)
246     	    try {
247     		return ConfigChannel.getChannel (n.getNodeValue());
248     	    } catch (NameRegistrar.NotFoundException e) { }
249     	return null;
250         }
251         public static ISOChannel getChildChannel (Node node) 
252     	throws ConfigurationException
253         {
254     	ISOChannel channel = null;
255     	NodeList childs = node.getChildNodes();
256     	for (int i=0; i<childs.getLength() && channel == null; i++) {
257     	    Node n = childs.item(i);
258     	    if (n.getNodeName().equals ("channel"))
259     		channel = ConfigChannel.getChannel (n);
260     	}
261     
262     	if (channel == null)
263     	    throw new ConfigurationException ("could not find channel");
264     
265     	return channel;
266         }
