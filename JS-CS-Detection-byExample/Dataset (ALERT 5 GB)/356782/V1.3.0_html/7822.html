224             for (int i=0; i<e.length; i++)
225                 a[i] = EBCDIC2ASCII[((int)e[i])&0xFF];

242         /**
243          * pad to the left
244          * @param s - original string
245          * @param len - desired len
246          * @param c - padding char
247          * @return padded string
248          */
249         public static String padleft(String s, int len, char c)
250             throws ISOException 
251         {
252             s = s.trim();
253             if (s.length() > len)
254                 throw new ISOException("invalid len " +s.length() + "/" +len);
255             StringBuffer d = new StringBuffer (len);
256             int fill = len - s.length();
257             while (fill-- > 0)
258                 d.append (c);
259             d.append(s);
260             return d.toString();
261         }

291         /**
292          * converts to BCD
293          * @param s - the number
294          * @param padLeft - flag indicating left/right padding
295          * @return BCD representation of the number
296          */
297         public static byte[] str2bcd(String s, boolean padLeft) {
298             int len = s.length();
299             byte[] d = new byte[ (len+1) >> 1 ];
300             int start = (((len & 1) == 1) && padLeft) ? 1 : 0;
301             for (int i=start; i < len+start; i++) 
302                 d [i >> 1] |= (s.charAt(i-start)-'0') << ((i & 1) == 1 ? 0 : 4);
303             return d;
304         }
305         /**
306          * converts a BCD representation of a number to a String
307          * @param b - BCD representation
308          * @param offset - starting offset
309          * @param len - BCD field len
310          * @param padLeft - was padLeft packed?
311          * @return the String representation of the number
312          */
313         public static String bcd2str(byte[] b, int offset,
314                             int len, boolean padLeft)
315         {
316             StringBuffer d = new StringBuffer(len);
317             int start = (((len & 1) == 1) && padLeft) ? 1 : 0;
318             for (int i=start; i < len+start; i++) {
319                 int shift = ((i & 1) == 1 ? 0 : 4);
320                 char c = Character.forDigit (
321                     ((b[offset+(i>>1)] >> shift) & 0x0F), 16);
322     	    if (c == 'd')
323     		c = '=';
324                 d.append (Character.toUpperCase (c));
325             }
326             return d.toString();
327         }

359     		    case '\000': d.append ("{NULL}"); break;
360     		    case '\001': d.append ("{SOH}");  break;
361     		    case '\002': d.append ("{STX}");  break;
362     		    case '\003': d.append ("{ETX}");  break;
363     		    case '\004': d.append ("{EOT}");  break;
364     		    case '\005': d.append ("{ENQ}");  break;
365     		    case '\006': d.append ("{ACK}");  break;
366     		    case '\007': d.append ("{BEL}");  break;
367     		    case '\020': d.append ("{DLE}");  break;
368     		    case '\025': d.append ("{NAK}");  break;
369     		    case '\026': d.append ("{SYN}");  break;
370     		    case '\034': d.append ("{FS}");  break;
371     		    default:
372     			char hi = Character.forDigit ((b[i] >> 4) & 0x0F, 16);
373     			char lo = Character.forDigit (b[i] & 0x0F, 16);
374     			d.append('[');
375     			d.append(Character.toUpperCase(hi));
376     			d.append(Character.toUpperCase(lo));
377     			d.append(']');
378     			break;
379     		}
380     	    }
381     	    else
382     		d.append (c);
383     
384             }
385             return d.toString();
386         }

428         {
429             int len = (b.size() >> 3) << 3;
430             len = (len > 128) ? 128: len;
431             byte[] d = new byte[len >> 3];
432             for (int i=0; i<len; i++) 
433                 if (b.get(i+1)) 
434                     d[i >> 3] |= (0x80 >> (i % 8));
435             if (len>64)
436                 d[0] |= 0x80;
437             return d;
438         }

440         /**
441          * Converts a binary representation of a Bitmap field
442          * into a Java BitSet
443          * @param b - binary representation
444          * @param offset - staring offset
445          * @param bitZeroMeansExtended - true for ISO-8583
446          * @return java BitSet object
447          */
448         public static BitSet byte2BitSet 
449             (byte[] b, int offset, boolean bitZeroMeansExtended)
450         {
451             int len = bitZeroMeansExtended ?
452                 ((b[offset] & 0x80) == 0x80 ? 128 : 64) : 64;
453     
454             BitSet bmap = new BitSet (len);
455             for (int i=0; i<len; i++) 
456                 if (((b[offset + (i >> 3)]) & (0x80 >> (i % 8))) > 0)
457                     bmap.set(i+1);
458             return bmap;
459         }

462          * Converts an ASCII representation of a Bitmap field
463          * into a Java BitSet
464          * @param b - binary representation
465          * @param offset - staring offset
466          * @param bitZeroMeansExtended - true for ISO-8583
467          * @return java BitSet object
468          */
469         public static BitSet hex2BitSet 
470             (byte[] b, int offset, boolean bitZeroMeansExtended)
471         {
472             int len = bitZeroMeansExtended ?
473               ((Character.digit((char)b[offset],16) & 0x08) == 8 ? 128 : 64) :
474               64;
475             BitSet bmap = new BitSet (len);
476             for (int i=0; i<len; i++) {
477                 int digit = Character.digit((char)b[offset + (i >> 2)], 16);
478                 if ((digit & (0x08 >> (i%4))) > 0)
479                     bmap.set(i+1);
480             }
481             return bmap;
482         }

484         /**
485          * @param   b       source byte array
486          * @param   offset  starting offset
487          * @param   len     number of bytes in destination (procesa len*2 de source)
488          * @return  byte[len]
489          */
490         public static byte[] hex2byte (byte[] b, int offset, int len) {
491             byte[] d = new byte[len];
492             for (int i=0; i<len*2; i++) {
493                 int shift = i%2 == 1 ? 0 : 4;
494                 d[i>>1] |= Character.digit((char) b[offset+i], 16) << shift;
495             }
496             return d;
497         }

500          * format double value
501          * @param amount    the amount
502          * @param fieldLen  the field len
503          * @return a String of fieldLen characters (right justified)
504          */
505         public static String formatDouble(double d, int len) {
506             String prefix = Long.toString((long) d);
507             String suffix = Integer.toString (
508                 (int) ((Math.round(d * 100f)) % 100) );
509             try {
510                 prefix = ISOUtil.padleft(prefix,len-3,' ');
511                 suffix = ISOUtil.zeropad(suffix, 2);
512             } catch (ISOException e) {
513                 e.printStackTrace();
514             }
515             return prefix + "." + suffix;
516         }

534         /**
535          * XML normalizer
536          * @param s source String
537          * @param canonical true if we want to normalize \r and \n as well
538          * @return normalized string suitable for XML Output
539          */
540         public static String normalize (String s, boolean canonical) {
541             StringBuffer str = new StringBuffer();
542     
543             int len = (s != null) ? s.length() : 0;
544             for (int i = 0; i < len; i++) {
545                 char ch = s.charAt(i);
546                 switch (ch) {
547                     case '<': 
548                         str.append("&lt;");
549                         break;
550                     case '>': 
551                         str.append("&gt;");
552                         break;
553                     case '&': 
554                         str.append("&amp;");
555                         break;
556                     case '"': 
557                         str.append("&quot;");
558                         break;
559                     case '\r':
560                     case '\n': 
561                         if (canonical) {
562                             str.append("&#");
563                             str.append(Integer.toString(ch));
564                             str.append(';');
565                             break;
566                         }
567                         // else, default append char
568                     default: 
569                         str.append(ch);
570                 }
571             }
572             return (str.toString());
573         }
574         /**
575          * XML normalizer (default canonical)
576          * @param s source String
577          * @return normalized string suitable for XML Output
578          */
579         public static String normalize (String s) {
580     	return normalize (s, true);
581         }
