84     package org.jpos.iso.gui;
85     
86     import java.util.Observer;
87     import java.util.Observable;
88     import java.awt.*;
89     import java.awt.event.*;
90     import javax.swing.*;
91     import javax.swing.event.*;
92     
93     import org.jpos.iso.*;
94     
95     public class ISORequestListenerPanel extends JPanel implements Observer {
96         /**
97          * @serial
98          */
99         DefaultListModel log;
100         /**
101          * @serial
102          */
103         String symbolicName;
104         public static final int LOG_CAPACITY = 250;
105     
106         public ISORequestListenerPanel (
107             ISORequestListener requestListener,
108             String symbolicName)
109         {
110             super();
111             this.symbolicName = symbolicName;
112             setLayout(new FlowLayout());
113             setBorder(BorderFactory.createRaisedBevelBorder());
114             log = new DefaultListModel();
115             add(createPanel());
116     	if (requestListener instanceof Observable)
117     	    ((Observable)requestListener).addObserver(this);
118         }
119         public final String getSymbolicName() {
120             return symbolicName;
121         }
122         public final ListModel getLog() {
123             return log;
124         }
125     
126         public void update(Observable o, Object arg) {
127             ISORequestListener l = (ISORequestListener) o;
128             if (arg != null && arg instanceof ISOMsg) {
129                 ISOMsg m = (ISOMsg) arg;
130                 try {
131                     String mti = (String) m.getValue(0);
132                     int imti   = Integer.parseInt(mti);
133                     log.addElement(m);
134                     if (log.getSize() > LOG_CAPACITY) 
135                         log.remove(0);
136                 } catch (ISOException e) { }
137             }
138         }
139     
140         private JPanel createPanel() {
141             JPanel A = new JPanel() {
142                 public Insets getInsets() {
143                     return new Insets(10,10,10,10);
144                 }
145             };
146     
147             A.setLayout(new BorderLayout());
148     
149             JLabel l = new JLabel(symbolicName);
150             A.add(l, BorderLayout.NORTH);
151     
152             final JList logList = new JList(log);
153             logList.setPrototypeCellValue("9999 99999999 999999");
154     
155             MouseListener mouseListener = new MouseAdapter() {
156                 public void mouseClicked(MouseEvent e) {
157                     ISOMsg m = (ISOMsg) logList.getSelectedValue();
158                     if (m != null) {
159                         JFrame f = new JFrame(m.toString());
160                         ISOMsgPanel p = new ISOMsgPanel(m);
161                         f.getContentPane().add(p);
162                         f.pack();
163                         f.show();
164                     }
165                 }
166             };
167             logList.addMouseListener(mouseListener);
168     
169             JScrollPane scrollPane = new JScrollPane(logList);
170             scrollPane.setPreferredSize(new Dimension(170,200));
171             A.add(scrollPane, BorderLayout.SOUTH);
172             return A;
173         }
174     }
