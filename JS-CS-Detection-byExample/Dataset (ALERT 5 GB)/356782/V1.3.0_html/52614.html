50     package org.jpos.iso;
51     
52     import org.jpos.iso.*;
53     
54     /*
55      * BASE1 Header
56      * <pre>
57      *   0 hlen;         Fld  1: Header Length        1B      (Byte     0)
58      *   1 hformat;      Fld  2: Header Format        8N,bit  (Byte     1)
59      *   2 format;       Fld  3: Text Format          1B      (Byte     2)
60      *   3 len[2];       Fld  4: Total Message Length 2B      (Byte  3- 4)
61      *   5 dstId[3];     Fld  5: Destination Id       6N,BCD  (Byte  5- 7)
62      *   8 srcId[3];     Fld  6: Source Id            6N,BCD  (Byte  8-10)
63      *  11 rtCtl;        Fld  7: Round-Trip Ctrl Info 8N,bit  (Byte    11)
64      *  12 flags[2];     Fld  8: BASE I Flags        16N,bit  (Byte 12-13)
65      *  14 status[3];    Fld  9: Message Status Flags 24bits  (Byte 14-16)
66      *  17 batchNbr;     Fld 10: Batch Number        1B       (Byte    17)
67      *  18 reserved[3];  Fld 11: Reserved            3B       (Byte 18-20)
68      *  21 userInfo;     Fld 12: User Info           1B       (Byte    21)
69      * </pre>
70      *
71      */
72     public class BASE1Header {
73         public static final int LENGTH = 22;
74         private byte[] header;
75     
76         public BASE1Header(String source, String destination) {
77             super();
78             header = new byte[LENGTH];
79             header[0] = LENGTH; // hlen
80             setHFormat(1);
81             setFormat(2);
82             setSource(source);
83             setDestination(destination);
84         }
85         public BASE1Header(byte[] header) {
86             this.header = header;
87         }
88         public byte[] getBytes() {
89             return header;
90         }
91         public int getHLen() {
92             return (int) (header[0] & 0xFF);
93         }
94         public void setHFormat(int hformat) {
95             header[1] = (byte) hformat;
96         }
97         public void setRtCtl(int i) {
98             header[11] = (byte) i;
99         }
100         public void setFlags(int i) {
101             header[12] = (byte) ((i >> 8) & 0xFF);
102             header[13] = (byte) (i & 0xFF);
103         }
104         public void setStatus(int i) {
105             header[14] = (byte) ((i >> 16) & 0xFF);
106             header[15] = (byte) ((i >> 8) & 0xFF);
107             header[16] = (byte) (i & 0xFF);
108         }
109         public void setBatchNumber(int i) {
110             header[17] = (byte) (i & 0xFF);
111         }
112         public void setFormat(int format) {
113             header[2] = (byte) format;
114         }
115         public void setLen(int len) {
116             len += LENGTH;
117             header[3]  = (byte) ((len >> 8) & 0xff);
118             header[4]  = (byte) (len        & 0xff);
119         }
120         public void setDestination(String dest) {
121             byte[] d = ISOUtil.str2bcd(dest, true);
122             System.arraycopy(d, 0, header, 5, 3);
123         }
124         public void setSource(String src) {
125             byte[] d = ISOUtil.str2bcd(src, true);
126             System.arraycopy(d, 0, header, 8, 3);
127         }
128         public byte[] getSource() {
129             byte[] b = new byte[3];
130             System.arraycopy(header, 8, b, 0, 3);
131             return b;
132         }
133         public byte[] getDestination() {
134             byte[] b = new byte[3];
135             System.arraycopy(header, 5, b, 0, 3);
136             return b;
137         }
138     
139         public void swapDirection() {
140     	if (header != null && header.length >= LENGTH) {
141     	    byte[] source = new byte[3];
142     	    System.arraycopy(header, 8, source, 0, 3);
143     	    System.arraycopy(header, 5, header, 8, 3);
144     	    System.arraycopy(source, 0, header, 5, 3);
145     	}
146         }
147         public boolean isRejected() {
148             return (header[16] & 0x80) == 0x80;
149         }
150         // first bit of fld 13 of header == 1 && len >=26
151         // indica un nuevo header de 22 (Pagina 4-2 Vol-1 VIP System
152         // Technical Reference General Requirements
153     };
