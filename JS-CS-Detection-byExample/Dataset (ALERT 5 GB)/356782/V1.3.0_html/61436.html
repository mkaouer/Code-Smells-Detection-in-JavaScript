50     package org.jpos.iso;
51     
52     /**
53      * ISOFieldPackager CHARACTERS (ASCII & BINARY)
54      * EBCDIC version of IF_CHAR
55      * @author apr@cs.com.uy
56      * @version $Id$
57      * @see IF_CHAR
58      * @see ISOComponent
59      * @deprecated Use IFE_CHAR instead
60      */
61     public class IF_ECHAR extends ISOFieldPackager {
62         /**
63          * @param len - field len
64          * @param description symbolic descrption
65          */
66         public IF_ECHAR(int len, String description) {
67             super(len, description);
68         }
69         /**
70          * @param c - a component
71          * @return packed component
72          * @exception ISOException
73          */
74         public byte[] pack (ISOComponent c) throws ISOException {
75             String s = (ISOUtil.strpad ((String) c.getValue(), getLength()));
76             return ISOUtil.asciiToEbcdic(s);
77         }
78         /**
79          * @param c - the Component to unpack
80          * @param b - binary image
81          * @param offset - starting offset within the binary image
82          * @return consumed bytes
83          * @exception ISOException
84          */
85         public int unpack (ISOComponent c, byte[] b, int offset)
86             throws ISOException
87         {
88             c.setValue(ISOUtil.ebcdicToAscii(b, offset, getLength()));
89             return getLength();
90         }
91         public int getMaxPackedLength() {
92             return getLength();
93         }
94     }
