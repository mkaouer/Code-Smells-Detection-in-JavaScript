50     package tpl;
51     
52     import java.sql.*;
53     import java.util.*;
54     import org.jpos.tpl.*;
55     
56     /**
57      * HotCard Peer
58      */
59     public class HCPeer implements PersistentPeer {
60         PersistentEngine engine = null;
61     
62         public HCPeer () {
63     	super();
64         }
65         public HCPeer (PersistentEngine engine) {
66     	super();
67     	setPersistentEngine (engine);
68         }
69     
70         // =============================================================
71         // PersistentPeer implementation
72         // =============================================================
73         public void setPersistentEngine (PersistentEngine engine) {
74     	this.engine = engine;
75         }
76         public void create (Object obj) throws SQLException {
77     	HC hc = (HC) obj;
78     	engine.executeUpdate (getCreateSql (hc));
79         }
80         public void load (Object obj) throws SQLException {
81     	// nothing to do, 'pan' field is already loaded
82         }
83         public void update (Object obj) throws SQLException {
84     	HC hc = (HC) obj;
85     	engine.executeUpdate (getUpdateSql (hc));
86         }
87         public void remove (Object obj) throws SQLException {
88     	HC hc = (HC) obj;
89     	engine.executeUpdate (getRemoveSql (hc.getPan()));
90         }
91     
92         // =============================================================
93         // Finders
94         // =============================================================
95         public HC findByPan (String pan) 
96     	throws	SQLException,
97     		NotFoundException
98         {
99     	ResultSet rs = null;
100     	try {
101     	    rs = engine.executeQuery (getSelectSql (pan));
102     	    if (!rs.next()) 
103     		throw new 
104     		    NotFoundException (pan + " not found");
105     	    return new HC (pan);
106     	} finally {
107     	    if (rs != null)
108     		rs.close();
109     	}
110         }
111     
112         public Collection findByRange (String initialPan, String finalPan) 
113     	throws SQLException
114         {
115     	Collection c = new ArrayList();
116     	ResultSet rs = null;
117     	try {
118     	    rs = engine.executeQuery (getSelectRangeSql (initialPan, finalPan));
119     	    while (rs.next()) {
120     		HC hc = new HC();
121     		load (hc, rs);
122     		c.add (hc);
123     	    }
124     	} finally {
125     	    if (rs != null)
126     		rs.close();
127     	}
128     	return c;
129         }
130     
131         // =============================================================
132         // private helper methods
133         // =============================================================
134         private String getCreateSql (HC o) {
135     	return "INSERT INTO nf (pan) VALUES ('" + o.getPan() + "')";
136         }
137         private String getSelectSql (String pan) {
138     	return "SELECT (pan) FROM nf WHERE pan = '" + pan + "'";
139         }
140         private String getRemoveSql (String pan) {
141     	return "DELETE FROM nf WHERE pan = '" + pan + "'";
142         }
143         private String getUpdateSql (HC o) {
144     	// sin sentido en HC, for demo purposes...
145     	return "UPDATE nf SET pan = '" + o.getPan() + 
146     	    "' WHERE pan='" + o.getPan()+ "'";
147         }
148         private String getSelectRangeSql (String initialPan, String finalPan) {
149     	return "SELECT (pan) FROM nf WHERE pan >= '" + initialPan + "'"
150     	+ " AND pan <= '" + finalPan + "'";
151         }
152         private void load (Object obj, ResultSet rs) throws SQLException {
153     	HC hc = (HC) obj;
154     	hc.setPan (rs.getString(1));
155         }
156     }
