125     /**
126      * provides base functionality for the actual packagers
127      *
128      * @author apr@cs.com.uy
129      * @version $Id$
130      * @see org.jpos.iso.packager.ISO87APackager
131      * @see org.jpos.iso.packager.ISO87BPackager
132      */
133     public abstract class ISOBasePackager implements ISOPackager, LogSource {
134         protected ISOFieldPackager[] fld;
135     
136         protected Logger logger = null;
137         protected String realm = null;
138     
139         protected void setFieldPackager (ISOFieldPackager[] fld) {
140             this.fld = fld;
141         }
142         /**
143          * @return true if BitMap have to be emited
144          */
145         protected boolean emitBitMap () {
146     	return (fld[1] instanceof ISOBitMapPackager);
147         }
148         /**
149          * usually 2 for normal fields, 1 for bitmap-less
150          * or ANSI X9.2 
151          * @return first valid field
152          */
153         protected int getFirstField() {
154     	if (!(fld[0] instanceof ISOMsgFieldPackager))
155     	    return (fld[1] instanceof ISOBitMapPackager) ? 2 : 1;
156     	return 0;
157         }
158         /**
159          * @param   m   the Component to pack
160          * @return      Message image
161          * @exception ISOException
162          */
163         public byte[] pack (ISOComponent m) throws ISOException {
164     	LogEvent evt = new LogEvent (this, "pack");
165     	try {
166     	    if (m.getComposite() != m) 
167     		throw new ISOException ("Can't call packager on non Composite");
168     
169     	    ISOComponent c;
170     	    Vector v = new Vector();
171     	    Hashtable fields = m.getChildren();
172     	    int len = 0;
173     
174     	    // MTI (field 0)
175     	    c = (ISOComponent) fields.get (new Integer (0));
176     	    byte[] b;
177     	    if (c instanceof ISOField) {
178     		b = fld[0].pack(c);
179     		len += b.length;
180     		v.addElement (b);
181     	    }
182     
183     	    if (emitBitMap()) {
184     		// BITMAP (-1 in HashTable)
185     		c = (ISOComponent) fields.get (new Integer (-1));
186     		b = getBitMapfieldPackager().pack(c);
187     		len += b.length;
188     		v.addElement (b);
189     	    }
190     
191     	    // if Field 1 is a BitMap then we are packing an
192     	    // ISO-8583 message so next field is fld#2.
193     	    // else we are packing an ANSI X9.2 message, first field is 1
194     	    int tmpMaxField=Math.min (m.getMaxField(), 128);
195     
196     	    for (int i=getFirstField(); i<=tmpMaxField; i++) {
197     		if ((c=(ISOComponent) fields.get (new Integer (i))) != null)
198     		{
199     		    try {
200     			b = fld[i].pack(c);
201     			len += b.length;
202     			v.addElement (b);
203     		    } catch (Exception e) {
204     			evt.addMessage ("error packing field "+i);
205     			evt.addMessage (c);
206     			evt.addMessage (e);
207     		    }
208     		}
209     	    }
210     	
211     	    if(m.getMaxField()>128 && fld.length > 128) {
212     		for (int i=1; i<=64; i++) {
213     		    if ((c = (ISOComponent) 
214     			fields.get (new Integer (i+128))) != null)
215     		    {
216     			try {
217     			    b = fld[i+128].pack(c);
218     			    len += b.length;
219     			    v.addElement (b);
220     			} catch (Exception e) {
221     			    evt.addMessage ("error packing field "+(i+128));
222     			    evt.addMessage (c);
223     			    evt.addMessage (e);
224     			}
225     		    }
226     		}
227     	    }
228     
229     	    int k = 0;
230     	    byte[] d = new byte[len];
231     	    for (int i=0; i<v.size(); i++) {
232     		b = (byte[]) v.elementAt(i);
233     		for (int j=0; j<b.length; j++)
234     		    d[k++] = b[j];
235     	    }
236     	    if (logger != null)	 // save a few CPU cycle if no logger available
237     		evt.addMessage (ISOUtil.hexString (d));
238     	    return d;
239     	} catch (ISOException e) {
240     	    evt.addMessage (e);
241     	    throw e;
242     	} finally {
243     	    Logger.log(evt);
244     	}
245         }
246     
247         /**
248          * @param   m   the Container of this message
249          * @param   b   ISO message image
250          * @return      consumed bytes
251          * @exception ISOException
252          */
253         public int unpack (ISOComponent m, byte[] b) throws ISOException {
254     	LogEvent evt = new LogEvent (this, "unpack");
255     	try {
256     	    if (m.getComposite() != m) 
257     		throw new ISOException ("Can't call packager on non Composite");
258     	    if (logger != null)	 // save a few CPU cycle if no logger available
259     		evt.addMessage (ISOUtil.hexString (b));
260     
261     	    int consumed = 0;
262     	    if (!(fld[0] instanceof ISOMsgFieldPackager) &&
263                     !(fld[0] instanceof ISOBitMapPackager))
264                 {
265     		ISOComponent mti     = new ISOField (0);
266     		consumed  = fld[0].unpack(mti, b, 0);
267     		m.set (mti);
268     	    }
269     	    BitSet bmap = null;
270     	    int maxField = fld.length;
271     	    if (emitBitMap()) {
272     		ISOBitMap bitmap = new ISOBitMap (-1);
273     		consumed += getBitMapfieldPackager().unpack(bitmap,b,consumed);
274     		bmap = (BitSet) bitmap.getValue();
275     		if (logger != null)
276     		    evt.addMessage ("<bitmap>"+bmap.toString()+"</bitmap>");
277     		m.set (bitmap);
278     		maxField = Math.min(maxField, bmap.size());
279     	    }
280     		
281     	    for (int i=getFirstField(); i<maxField; i++) {
282     		if (bmap == null || bmap.get(i)) {
283     		    ISOComponent c = fld[i].createComponent(i);
284     		    consumed += fld[i].unpack (c, b, consumed);
285     		    if (logger != null) {
286     			evt.addMessage ("<unpack fld=\"" + i 
287     			    +"\" packager=\""
288     			    +fld[i].getClass().getName()+ "\">");
289     			if (c.getValue() instanceof ISOMsg)
290     			    evt.addMessage (c.getValue());
291     			else
292     			    evt.addMessage ("  <value>" 
293     				+c.getValue().toString()
294     				+ "</value>");
295     			evt.addMessage ("</unpack>");
296     		    }
297     		    m.set(c);
298     		}
299     	    }
300     	    if (bmap != null && bmap.get(65) && fld.length > 128 &&
301     		fld[65] instanceof ISOBitMapPackager)
302     	    {
303     		bmap= (BitSet) 
304     		    ((ISOComponent) m.getChildren().get 
305     			(new Integer(65))).getValue();
306     		for (int i=1; i<64; i++) {
307     		    if (bmap == null || bmap.get(i)) {
308     			ISOComponent c = fld[i+128].createComponent(i);
309     			consumed += fld[i+128].unpack (c, b, consumed);
310     			if (logger != null) {
311     			    evt.addMessage ("<unpack fld=\"" + i+128
312     				+"\" packager=\""
313     				+fld[i+128].getClass().getName()+ "\">");
314     			    evt.addMessage ("  <value>" 
315     				+c.getValue().toString()
316     				+ "</value>");
317     			    evt.addMessage ("</unpack>");
318     			}
319     			m.set(c);
320     		    }
321     		}
322     	    }
323     
324     	    if (b.length != consumed) {
325     		evt.addMessage (
326     		    "WARNING: unpack len=" +b.length +" consumed=" +consumed
327     		);
328     	    }
329     	    return consumed;
330     	} catch (ISOException e) {
331     	    evt.addMessage (e);
332     	    throw e;
333     	} finally {
334     	    Logger.log (evt);
335     	}
336         }
337         /**
338          * @param   m   the Container (i.e. an ISOMsg)
339          * @param   fldNumber the Field Number
340          * @return  Field Description
341          */
342         public String getFieldDescription(ISOComponent m, int fldNumber) {
343             return fld[fldNumber].getDescription();
344         }
345         /**
346          * @return 128 for ISO-8583, should return 64 for ANSI X9.2
347          */
348         protected int getMaxValidField() {
349             return 128;
350         }
351         /**
352          * @return suitable ISOFieldPackager for Bitmap
353          */
354         protected ISOFieldPackager getBitMapfieldPackager() {
355             return fld[1];
356         }
357         public void setLogger (Logger logger, String realm) {
358     	this.logger = logger;
359     	this.realm  = realm;
360         }
361         public String getRealm () {
362     	return realm;
363         }
364         public Logger getLogger() {
365     	return logger;
366         }
367     }
