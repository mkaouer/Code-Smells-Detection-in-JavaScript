72     /**
73      * Configure DirPoll node
74      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
75      * @version $Revision$ $Date$
76      */
77     public class ConfigDirPoll implements QSPReConfigurator {
78         public static final String NAMEREGISTRAR_PREFIX = "qsp.dirpoll.";
79     
80         public void config (QSP qsp, Node node) throws ConfigurationException
81         {
82     	LogEvent evt = new LogEvent (qsp, "config-dir-poll");
83     	try {
84     	    DirPoll dp = new DirPoll();
85     	    configDirPoll (qsp, node, dp, evt);
86     
87     	    // non-reconfigurable attributes
88     	    dp.setThreadPool (new ThreadPool (1, getPoolSize (node, evt)));
89     
90     	    String name = ConfigUtil.getAttribute (node, "name", null);
91     	    if (name != null)
92     		NameRegistrar.register (NAMEREGISTRAR_PREFIX+name, dp);
93     
94     	    Thread thread = new Thread(dp);
95     	    thread.setName ("qsp-dir-poll-"+name);
96     	    thread.start();
97     	} finally {
98     	    Logger.log (evt);
99     	}
100         }
101     
102         public void reconfig (QSP qsp, Node node) throws ConfigurationException
103         {
104     	Node nameNode = node.getAttributes().getNamedItem ("name");
105     	if (nameNode == null)
106     	    return; 
107     
108     	String name   = nameNode.getNodeValue();
109     	LogEvent evt = new LogEvent (qsp, "re-config-dir-poll");
110     	evt.addMessage ("<name>" + name + "</name>");
111     	try {
112     	    DirPoll dp = (DirPoll) 
113     		NameRegistrar.get (NAMEREGISTRAR_PREFIX + name);
114     	    if (dp != null)
115     		configDirPoll (qsp, node, dp, evt);
116     	} catch (NameRegistrar.NotFoundException e) {
117     	    evt.addMessage ("<dir-poll-not-found/>");
118     	}
119     	Logger.log (evt);
120         }
121         // ---------------------------------------------------- private helpers
122         private void configureProcessor 
123     	(Configurable processor, Node node, LogEvent evt)
124     	throws ConfigurationException
125         {
126     	processor.setConfiguration (new SimpleConfiguration (
127     		ConfigUtil.addProperties (node, null, evt)
128     	    )
129     	);
130         }
131         private int getPoolSize(Node node, LogEvent evt) {
132     	int i = ConfigUtil.getAttributeAsInt (node, "poolsize", 10);
133     	evt.addMessage ("<poolsize>"+i+"</poolsize>");
134     	return i;
135         }
136         private int getPollInterval (Node node, LogEvent evt) {
137     	int i = ConfigUtil.getAttributeAsInt (node, "interval", 1000);
138     	evt.addMessage ("<interval>"+i+"</interval>");
139     	return i;
140         }
141         private String getPriorities (Node node, LogEvent evt) {
142     	String s = ConfigUtil.getAttribute (node, "priorities", "");
143     	evt.addMessage ("<priorities>"+s+"</priorities>");
144     	return s;
145         }
146         private String getPath (Node node, LogEvent evt) { 
147     	String s = ConfigUtil.getAttribute (node, "path", "");
148     	evt.addMessage ("<path>"+s+"</path>");
149     	return s;
150         }
151         private void configDirPoll (QSP qsp, Node node, DirPoll dp, LogEvent evt)
152     	throws ConfigurationException
153         {
154     	dp.setPath (getPath (node, evt));
155     	dp.setLogger (
156     	    ConfigLogger.getLogger (node),
157     	    ConfigLogger.getRealm  (node)
158     	);
159     	dp.setPollInterval (getPollInterval (node, evt));
160     	dp.setPriorities (getPriorities (node, evt)); 
161     
162     	String className = ConfigUtil.getAttribute (node, "processor", null);
163     	Object processor = ConfigUtil.newInstance (className);
164     
165     	if (!(processor instanceof DirPoll.Processor))
166     	    throw new ConfigurationException (
167     		"invalid class "+className
168     		+" does not implement DirPoll.Processor");
169     	dp.setProcessor ((DirPoll.Processor) processor);
170     	evt.addMessage ("<processor>"+className+"</processor>");
171     	if (ConfigUtil.getAttribute (node, "create", "no").equals ("yes")) {
172     	    evt.addMessage ("<create-dirs/>");
173     	    dp.createDirs();
174     	}
175     	if (processor instanceof LogSource)
176     	    ((LogSource)processor).setLogger (dp.getLogger(), dp.getRealm());
177     
178     	if (processor instanceof Configurable)
179     	    configureProcessor ((Configurable) processor, node, evt);
180         }
181     }
