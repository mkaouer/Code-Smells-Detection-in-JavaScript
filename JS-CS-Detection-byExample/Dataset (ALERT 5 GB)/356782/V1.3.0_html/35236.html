50     package org.jpos.iso;
51     
52     import java.util.Vector;
53     import java.util.Observable;
54     import java.util.Collection;
55     import java.util.Iterator;
56     import org.jpos.util.LogEvent;
57     import org.jpos.iso.ISOFilter.VetoException;
58     
59     /**
60      * Filtered Channel Base
61      * 
62      * @author <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
63      * @version $Revision$ $Date$
64      * @see FilteredChannel
65      */
66     
67     public abstract class FilteredBase extends Observable 
68         implements FilteredChannel
69     {
70         protected Vector incomingFilters, outgoingFilters;
71     
72         public FilteredBase () {
73     	super();
74     	incomingFilters = new Vector();
75     	outgoingFilters = new Vector();
76         }
77     
78         /**
79          * @param filter filter to add
80          * @param direction ISOMsg.INCOMING, ISOMsg.OUTGOING, 0 for both
81          */
82         private void addFilter (ISOFilter filter, int direction) {
83     	switch (direction) {
84     	    case ISOMsg.INCOMING :
85     		incomingFilters.add (filter);
86     		break;
87     	    case ISOMsg.OUTGOING :
88     		outgoingFilters.add (filter);
89     		break;
90     	    case 0 :
91     		incomingFilters.add (filter);
92     		outgoingFilters.add (filter);
93     		break;
94     	}
95         }
96         /**
97          * @param filter incoming filter to add
98          */
99         public void addIncomingFilter (ISOFilter filter) {
100     	addFilter (filter, ISOMsg.INCOMING);
101         }
102         /**
103          * @param filter outgoing filter to add
104          */
105         public void addOutgoingFilter (ISOFilter filter) {
106     	addFilter (filter, ISOMsg.OUTGOING);
107         }
108     
109         /**
110          * @param filter filter to add (both directions, incoming/outgoing)
111          */
112         public void addFilter (ISOFilter filter) {
113     	addFilter (filter, 0);
114         }
115         /**
116          * @param filter filter to remove
117          * @param direction ISOMsg.INCOMING, ISOMsg.OUTGOING, 0 for both
118          */
119         private void removeFilter (ISOFilter filter, int direction) {
120     	switch (direction) {
121     	    case ISOMsg.INCOMING :
122     		incomingFilters.remove (filter);
123     		break;
124     	    case ISOMsg.OUTGOING :
125     		outgoingFilters.remove (filter);
126     		break;
127     	    case 0 :
128     		incomingFilters.remove (filter);
129     		outgoingFilters.remove (filter);
130     		break;
131     	}
132         }
133         /**
134          * @param filter filter to remove (both directions)
135          */
136         public void removeFilter (ISOFilter filter) {
137     	removeFilter (filter, 0);
138         }
139         /**
140          * @param filter incoming filter to remove
141          */
142         public void removeIncomingFilter (ISOFilter filter) {
143     	removeFilter (filter, ISOMsg.INCOMING);
144         }
145         /**
146          * @param filter outgoing filter to remove
147          */
148         public void removeOutgoingFilter (ISOFilter filter) {
149     	removeFilter (filter, ISOMsg.OUTGOING);
150         }
151         protected ISOMsg applyOutgoingFilters (ISOMsg m, LogEvent evt) 
152     	throws VetoException
153         {
154     	Iterator iter  = outgoingFilters.iterator();
155     	while (iter.hasNext()) {
156     	    m.setDirection(ISOMsg.OUTGOING);
157     	    m = ((ISOFilter) iter.next()).filter (this, m, evt);
158     	}
159     	m.setDirection(ISOMsg.OUTGOING);
160     	setChanged ();
161     	notifyObservers (m);
162     	return m;
163         }
164         protected ISOMsg applyIncomingFilters (ISOMsg m, LogEvent evt) 
165     	throws VetoException
166         {
167     	Iterator iter  = incomingFilters.iterator();
168     	while (iter.hasNext()) {
169     	    m.setDirection(ISOMsg.INCOMING);
170     	    m = ((ISOFilter) iter.next()).filter (this, m, evt);
171     	}
172     	m.setDirection(ISOMsg.INCOMING);
173     	setChanged ();
174     	notifyObservers (m);
175     	return m;
176         }
177         public Collection getIncomingFilters() {
178     	return incomingFilters;
179         }
180         public Collection getOutgoingFilters() {
181     	return outgoingFilters;
182         }
183         public void setIncomingFilters (Collection filters) {
184     	incomingFilters = new Vector (filters);
185         }
186         public void setOutgoingFilters (Collection filters) {
187     	outgoingFilters = new Vector (filters);
188         }
189     }
