50     package org.jpos.iso.channel;
51     
52     import java.io.*;
53     import java.util.*;
54     import java.net.ServerSocket;
55     import java.net.Socket;
56     import java.net.SocketException;
57     import org.jpos.iso.*;
58     import org.jpos.iso.packager.XMLPackager;
59     
60     /**
61      * Implements an ISOChannel suitable to be used to connect to an X.25 PAD. 
62      * It waits a limited amount of time to decide when a packet is ready
63      * to be unpacked.
64      *
65      * @author  <a href="mailto:apr@cs.com.uy">Alejandro P. Revilla</a>
66      * @version $Id$
67      *
68      * @see ISOMsg
69      * @see ISOException
70      * @see ISOChannel
71      */
72     public class PADChannel extends BaseChannel {
73         BufferedReader reader = null;
74         protected byte[] header;
75         /**
76          * No-args constructor
77          */
78         public PADChannel () {
79             super();
80         }
81         /**
82          * Constructs client ISOChannel
83          * @param host  server TCP Address
84          * @param port  server port number
85          * @param p     an ISOPackager
86          * @see ISOPackager
87          */
88         public PADChannel (String host, int port, ISOPackager p) {
89             super(host, port, p);
90         }
91         /**
92          * Construct server ISOChannel
93          * @param p     an ISOPackager
94          * @see ISOPackager
95          * @exception IOException
96          */
97         public PADChannel (ISOPackager p) throws IOException {
98             super(p);
99         }
100         /**
101          * constructs a server ISOChannel associated with a Server Socket
102          * @param p     an ISOPackager
103          * @param serverSocket where to accept a connection
104          * @exception IOException
105          * @see ISOPackager
106          */
107         public PADChannel (ISOPackager p, ServerSocket serverSocket) 
108             throws IOException
109         {
110             super(p, serverSocket);
111         }
112         /**
113          * @return a byte array with the received message
114          * @exception IOException
115          */
116         protected byte[] streamReceive() throws IOException {
117     	int c, k=0, len = 1;
118     	Vector v = new Vector();
119     
120     	c = serverIn.read();
121     	if (c == -1)
122     	    throw new EOFException ("connection closed");
123     	byte[] b = new byte[1];
124     	b[0] = (byte) c;
125     	v.addElement (b);
126     
127     	// Wait for packets until timeout
128     	while ((c = serverIn.available()) > 0) {
129     	    b = new byte[c];
130     	    if (serverIn.read (b) != c)
131     		throw new EOFException ("connection closed");
132     	    v.addElement (b);
133     	    len += c;
134     	    try {
135     		Thread.sleep (50);
136     	    } catch (InterruptedException e) { }
137     	}
138     
139     	byte[] d = new byte[len];
140     	for (int i=0; i<v.size(); i++) {
141     	    b = (byte[]) v.elementAt(i);
142     	    System.arraycopy (b, 0, d, k, b.length);
143     	    k += b.length;
144     	}
145     	return d;
146         }
147         protected void connect (Socket socket) throws IOException {
148     	super.connect (socket);
149     	reader = new BufferedReader (new InputStreamReader (serverIn));
150         }
151         public void disconnect () throws IOException {
152     	super.disconnect ();
153     	reader = null;
154         }
155         protected int getHeaderLength() { 
156             return header != null ? header.length : 0;
157         }
158         public void setHeader (byte[] header) {
159     	this.header = header;
160         }
161         /**
162          * @param header Hex representation of header
163          */
164         public void setHeader (String header) {
165     	setHeader (
166     	    ISOUtil.hex2byte (header.getBytes(), 0, header.getBytes().length)
167     	);
168         }
169         public byte[] getHeader () {
170     	return header;
171         }
172         protected void sendMessageHeader(ISOMsg m, int len) throws IOException { 
173     	if (m.getHeader() != null)
174                 serverOut.write(m.getHeader());
175             else if (header != null) 
176                 serverOut.write(header);
177         }
178     }
