37         static final int[] NUMBER_OF_TILES = new int[ZOOMLEVEL_MAX - ZOOMLEVEL_MIN + 1];
38         static final int[] NUMBER_OF_PIXELS = new int[ZOOMLEVEL_MAX - ZOOMLEVEL_MIN + 1];
39         static {
40             for (int z = ZOOMLEVEL_MIN; z <= ZOOMLEVEL_MAX; z++) {
41                 NUMBER_OF_TILES[z] = 1 << z;
42                 NUMBER_OF_PIXELS[z] = TILE_SIZE * 1 << z;
43             }

75         /**
76          * Calculate the tile for a Geopoint based on the Spherical Mercator.
77          *
78          * @see <a
79          *      href="http://developers.cloudmade.com/projects/tiles/examples/convert-coordinates-to-tile-numbers">Cloudmade</a>
80          */
81         private static int calcX(final Geopoint origin, final int zoomlevel) {
82             // The cut of the fractional part instead of rounding to the nearest integer is intentional and part of the algorithm
83             return (int) ((origin.getLongitude() + 180.0) / 360.0 * NUMBER_OF_TILES[zoomlevel]);
84         }
85     
86         /**
87          * Calculate the tile for a Geopoint based on the Spherical Mercator.
88          *
89          */
90         private static int calcY(final Geopoint origin, final int zoomlevel) {
91     
92             // double latRad = Math.toRadians(origin.getLatitude());
93             // return (int) ((1 - (Math.log(Math.tan(latRad) + (1 / Math.cos(latRad))) / Math.PI)) / 2 * numberOfTiles);
94     
95             // Optimization from Bing
96             double sinLatRad = Math.sin(Math.toRadians(origin.getLatitude()));
97             // The cut of the fractional part instead of rounding to the nearest integer is intentional and part of the algorithm
98             return (int) ((0.5 - Math.log((1 + sinLatRad) / (1 - sinLatRad)) / (4 * Math.PI)) * NUMBER_OF_TILES[zoomlevel]);
99         }

109         /**
110          * Calculate latitude/longitude for a given x/y position in this tile.
111          *
112          * @see <a
113          *      href="http://developers.cloudmade.com/projects/tiles/examples/convert-coordinates-to-tile-numbers">Cloudmade</a>
114          */
115         @NonNull
116         public Geopoint getCoord(UTFGridPosition pos) {
117     
118             double pixX = tileX * TILE_SIZE + pos.x * 4;
119             double pixY = tileY * TILE_SIZE + pos.y * 4;
120     
121             double lonDeg = ((360.0 * pixX) / NUMBER_OF_PIXELS[this.zoomLevel]) - 180.0;
122             double latRad = Math.atan(Math.sinh(Math.PI * (1 - 2 * pixY / NUMBER_OF_PIXELS[this.zoomLevel])));
123             return new Geopoint(Math.toDegrees(latRad), lonDeg);
124         }

131         /**
132          * Calculates the maximum possible zoom level where the supplied points
133          * are covered by at least by the supplied number of
134          * adjacent tiles on the east/west axis.
135          * This criterion can be exactly met for even numbers of tiles
136          * while it may result in one more tile as requested for odd numbers
137          * of tiles.
138          *
139          * The order of the points (left/right) is irrelevant.
140          *
141          * @param left
142          *            First point
143          * @param right
144          *            Second point
145          * @return
146          */
147         public static int calcZoomLon(final Geopoint left, final Geopoint right, final int numberOfTiles) {
148     
149             int zoom = (int) Math.floor(
150                     Math.log(360.0 * numberOfTiles / (2.0 * Math.abs(left.getLongitude() - right.getLongitude())))
151                             / Math.log(2)
152             );
153     
154             Tile tileLeft = new Tile(left, zoom);
155             Tile tileRight = new Tile(right, zoom);
156     
157             if (Math.abs(tileLeft.tileX - tileRight.tileX) < (numberOfTiles - 1)) {
158                 zoom += 1;
159             }
160     
161             return Math.min(zoom, ZOOMLEVEL_MAX);
162         }
163     
164         /**
165          * Calculates the maximum possible zoom level where the supplied points
166          * are covered by at least by the supplied number of
167          * adjacent tiles on the north/south axis.
168          * This criterion can be exactly met for even numbers of tiles
169          * while it may result in one more tile as requested for odd numbers
170          * of tiles.
171          *
172          * The order of the points (bottom/top) is irrelevant.
173          *
174          * @param bottom
175          *            First point
176          * @param top
177          *            Second point
178          * @return
179          */
180         public static int calcZoomLat(final Geopoint bottom, final Geopoint top, final int numberOfTiles) {
181     
182             int zoom = (int) Math.ceil(
183                     Math.log(2.0 * Math.PI * numberOfTiles / (
184                             Math.abs(
185                                     asinh(tanGrad(bottom.getLatitude()))
186                                             - asinh(tanGrad(top.getLatitude()))
187                                     ) * 2.0)
188                             ) / Math.log(2)
189                     );
190     
191             Tile tileBottom = new Tile(bottom, zoom);
192             Tile tileTop = new Tile(top, zoom);
193     
194             if (Math.abs(tileBottom.tileY - tileTop.tileY) > (numberOfTiles - 1)) {
195                 zoom -= 1;
196             }
197     
198             return Math.min(zoom, ZOOMLEVEL_MAX);
199         }
200     
201         private static double tanGrad(double angleGrad) {
202             return Math.tan(angleGrad / 180.0 * Math.PI);
203         }

205         /**
206          * Calculates the inverted hyperbolic sine
207          * (after Bronstein, Semendjajew: Taschenbuch der Mathematik)
208          *
209          * @param x
210          * @return
211          */
212         private static double asinh(double x) {
213             return Math.log(x + Math.sqrt(x * x + 1.0));
214         }

269         /**
270          * Calculate needed tiles for the given viewport.
271          * You can define the minimum number of tiles on the longer axis
272          * and/or the minimum zoom level.
273          *
274          * @param viewport
275          * @param tilesOnAxis
276          * @param minZoom
277          * @return
278          */
279         protected static Set<Tile> getTilesForViewport(final Viewport viewport, final int tilesOnAxis, final int minZoom) {
280             Set<Tile> tiles = new HashSet<Tile>();
281             int zoom = Math.max(
282                     Math.min(Tile.calcZoomLon(viewport.bottomLeft, viewport.topRight, tilesOnAxis),
283                             Tile.calcZoomLat(viewport.bottomLeft, viewport.topRight, tilesOnAxis)),
284                     minZoom);
285     
286             Tile tileBottomLeft = new Tile(viewport.bottomLeft, zoom);
287             Tile tileTopRight = new Tile(viewport.topRight, zoom);
288     
289             int xLow = Math.min(tileBottomLeft.getX(), tileTopRight.getX());
290             int xHigh = Math.max(tileBottomLeft.getX(), tileTopRight.getX());
291     
292             int yLow = Math.min(tileBottomLeft.getY(), tileTopRight.getY());
293             int yHigh = Math.max(tileBottomLeft.getY(), tileTopRight.getY());
294     
295             for (int xNum = xLow; xNum <= xHigh; xNum++) {
296                 for (int yNum = yLow; yNum <= yHigh; yNum++) {
297                     tiles.add(new Tile(xNum, yNum, zoom));
298                 }
299             }
300     
301             return tiles;
302         }
