20         /**
21          * Extract the text from a HTML based string. This is similar to what HTML.fromHtml(...) does, but this method also
22          * removes the embedded images instead of replacing them by a small rectangular representation character.
23          *
24          * @param html
25          * @return
26          */
27         public static String extractText(CharSequence html) {
28             if (StringUtils.isBlank(html)) {
29                 return StringUtils.EMPTY;
30             }
31             String result = html.toString();
32     
33             // recognize images in textview HTML contents
34             if (html instanceof Spanned) {
35                 Spanned text = (Spanned) html;
36                 Object[] styles = text.getSpans(0, text.length(), Object.class);
37                 ArrayList<Pair<Integer, Integer>> removals = new ArrayList<Pair<Integer, Integer>>();
38                 for (Object style : styles) {
39                     if (style instanceof ImageSpan) {
40                         int start = text.getSpanStart(style);
41                         int end = text.getSpanEnd(style);
42                         removals.add(Pair.of(start, end));
43                     }
44                 }
45     
46                 // sort reversed and delete image spans
47                 Collections.sort(removals, new Comparator<Pair<Integer, Integer>>() {
48     
49                     @Override
50                     public int compare(Pair<Integer, Integer> lhs, Pair<Integer, Integer> rhs) {
51                         return rhs.getRight().compareTo(lhs.getRight());
52                     }
53                 });
54                 result = text.toString();
55                 for (Pair<Integer, Integer> removal : removals) {
56                     result = result.substring(0, removal.getLeft()) + result.substring(removal.getRight());
57                 }
58             }
59     
60             // now that images are gone, do a normal html to text conversion
61             return Html.fromHtml(result).toString().trim();
62         }
