60         public void merge(final Waypoint old) {
61             if (StringUtils.isBlank(prefix)) {
62                 setPrefix(old.prefix);
63             }
64             if (StringUtils.isBlank(lookup)) {
65                 lookup = old.lookup;
66             }
67             if (StringUtils.isBlank(name)) {
68                 setName(old.name);
69             }
70             if (StringUtils.isBlank(latlon) || latlon.startsWith("?")) { // there are waypoints containing "???"
71                 latlon = old.latlon;
72             }
73             if (coords == null) {
74                 coords = old.coords;
75             }
76             if (StringUtils.isBlank(note)) {
77                 note = old.note;
78             }
79             if (note != null && old.note != null) {
80                 if (old.note.length() > note.length()) {
81                     note = old.note;
82                 }
83             }
84             if (id < 0) {
85                 id = old.id;
86             }
87             visited = old.visited;
88         }

90         public static void mergeWayPoints(final List<Waypoint> newPoints, final List<Waypoint> oldPoints, final boolean forceMerge) {
91             // Build a map of new waypoints for faster subsequent lookups
92             final Map<String, Waypoint> newPrefixes = new HashMap<String, Waypoint>(newPoints.size());
93             for (final Waypoint waypoint : newPoints) {
94                 newPrefixes.put(waypoint.getPrefix(), waypoint);
95             }
96     
97             // Copy user modified details of the old waypoints over the new ones
98             for (final Waypoint oldWaypoint : oldPoints) {
99                 final String prefix = oldWaypoint.getPrefix();
100                 if (newPrefixes.containsKey(prefix)) {
101                     newPrefixes.get(prefix).merge(oldWaypoint);
102                 } else if (oldWaypoint.isUserDefined() || forceMerge) {
103                     // personal note waypoints should always be taken from the new list only
104                     if (!isPersonalNoteWaypoint(oldWaypoint)) {
105                         newPoints.add(oldWaypoint);
106                     }
107                 }
108             }
109         }

124         private int computeOrder() {
125             switch (waypointType) {
126                 case PARKING:
127                     return -1;
128                 case TRAILHEAD:
129                     return 1;
130                 case STAGE: // puzzles and stages with same value
131                     return 2;
132                 case PUZZLE:
133                     return 2;
134                 case FINAL:
135                     return 3;
136                 case OWN:
137                     return 4;
138                 default:
139                     return 0;
140             }
141         }

294         /**
295          * Detect coordinates in the personal note and convert them to user defined waypoints. Works by rule of thumb.
296          *
297          * @param initialNote Note content
298          * @return a collection of found waypoints
299          */
300         public static Collection<Waypoint> parseWaypointsFromNote(@NonNull final String initialNote) {
301             final List<Waypoint> waypoints = new LinkedList<Waypoint>();
302             final Pattern COORDPATTERN = Pattern.compile("\\b[nNsS]{1}\\s*\\d"); // begin of coordinates
303     
304             String note = initialNote;
305             MatcherWrapper matcher = new MatcherWrapper(COORDPATTERN, note);
306             int count = 1;
307             while (matcher.find()) {
308                 try {
309                     final Geopoint point = new Geopoint(note.substring(matcher.start()));
310                     // Coords must have non zero latitude and longitude and at least one part shall have fractional degrees.
311                     if (point.getLatitudeE6() != 0 && point.getLongitudeE6() != 0 &&
312                             ((point.getLatitudeE6() % 1000) != 0 || (point.getLongitudeE6() % 1000) != 0)) {
313                         final String name = CgeoApplication.getInstance().getString(R.string.cache_personal_note) + " " + count;
314                         final String potentialWaypointType = note.substring(Math.max(0, matcher.start() - 15));
315                         final Waypoint waypoint = new Waypoint(name, parseWaypointType(potentialWaypointType), false);
316                         waypoint.setCoords(point);
317                         waypoints.add(waypoint);
318                         count++;
319                     }
320                 } catch (final Geopoint.ParseException e) {
321                     // ignore
322                 }
323     
324                 note = note.substring(matcher.start() + 1);
325                 matcher = new MatcherWrapper(COORDPATTERN, note);
326             }
327             return waypoints;
328         }

330         /**
331          * Detect waypoint types in the personal note text. It works by rule of thumb only.
332          */
333         private static WaypointType parseWaypointType(final String input) {
334             final String lowerInput = StringUtils.substring(input, 0, 20).toLowerCase(Locale.getDefault());
335             for (final WaypointType wpType : WaypointType.values()) {
336                 if (lowerInput.contains(wpType.getL10n().toLowerCase(Locale.getDefault()))) {
337                     return wpType;
338                 }
339                 if (lowerInput.contains(wpType.id)) {
340                     return wpType;
341                 }
342                 if (lowerInput.contains(wpType.name().toLowerCase(Locale.US))) {
343                     return wpType;
344                 }
345             }
346             return WaypointType.WAYPOINT;
347         }
