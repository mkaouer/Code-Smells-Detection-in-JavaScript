16             logDate.setTimeInMillis(date);
17             logDate.set(Calendar.SECOND, 0);
18             logDate.set(Calendar.MINUTE, 0);
19             logDate.set(Calendar.HOUR_OF_DAY, 0);
20             final Calendar today = Calendar.getInstance();

23             today.set(Calendar.HOUR_OF_DAY, 0);
24             return (int) Math.round((today.getTimeInMillis() - logDate.getTimeInMillis()) / 86400000d);
25         }
