33         private class LogcatAsyncTask extends AsyncTask<Integer, String, Void> {
34             // TestRunner and silence others
35             private static final String CMD = "logcat -v brief TestRunner:I cgeo:I *:S";
36             private BufferedReader mReader;
37             private Process mProc;
38     
39             public LogcatAsyncTask() {
40                 try {
41                     mProc = Runtime.getRuntime().exec(CMD);
42                     mReader = new BufferedReader(new InputStreamReader(
43                             mProc.getInputStream()));
44                 } catch (Exception e) {
45                     Log.e(TAG, "Creating proc", e);
46                 }
47             }
48     
49             @Override
50             protected void onProgressUpdate(String... values) {
51                 final String line = values[0];
52                 final boolean isAtBottom = scrollView.isAtBottom();
53                 if (!TextUtils.isEmpty(line)) {
54                     logView.append(Html.fromHtml("<font color=\"" + color(line) + "\">" + line + "</font><br/>"));
55                     if (isAtBottom) {
56                         scrollView.scrollTo(0, logView.getBottom());
57                     }
58                 }
59             }
60     
61             private String color(String line) {
62                 switch (line.charAt(0)) {
63                     case 'E':
64                         return "red";
65                     case 'W':
66                         return "#FFA500";
67                     case 'D':
68                         return "blue";
69                     default:
70                         return "white";
71                 }
72             }
73     
74             @Override
75             protected Void doInBackground(Integer... params) {
76                 final long timeout = System.currentTimeMillis() + params[0];
77                 try {
78                     do {
79                         Thread.sleep(50);
80                         publishProgress(mReader.readLine());
81                     } while (System.currentTimeMillis() < timeout);
82                 } catch (InterruptedException e) {
83                     publishProgress("ERROR: " + e);
84                 } catch (IOException e) {
85                     publishProgress("ERROR: " + e);
86                 } finally {
87                     publishProgress("END");
88                     mProc.destroy();
89                 }
90                 return null;
91             }
92         }

131         private void runTestsInternally() {
132             final String pn = getPackageName().replaceFirst(".test$", "");
133             final InstrumentationInfo info = getInstrumentationInfo(pn);
134             if (info == null) {
135                 Toast.makeText(this,
136                         "Cannot find instrumentation for " + pn, Toast.LENGTH_SHORT)
137                         .show();
138                 return;
139             }
140             final ComponentName cn = new ComponentName(info.packageName,
141                     info.name);
142             if (startInstrumentation(cn, null, null)) {
143                 logCatTask = new LogcatAsyncTask();
144                 logCatTask.execute(TIMEOUT);
145             }
146             else {
147                 Toast.makeText(this,

149                         .show();
