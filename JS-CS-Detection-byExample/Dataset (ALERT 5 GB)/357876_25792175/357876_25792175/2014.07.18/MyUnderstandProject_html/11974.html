53         public static List<String> getOfflineMaps() {
54             final String directoryPath = Settings.getMapFileDirectory();
55             if (StringUtils.isBlank(directoryPath)) {
56                 return Collections.emptyList();
57             }
58     
59             File directory = new File(directoryPath);
60             if (directory.isDirectory()) {
61                 try {
62                     ArrayList<String> mapFileList = new ArrayList<String>();
63                     final File[] files = directory.listFiles();
64                     if (ArrayUtils.isNotEmpty(files)) {
65                         for (File file : files) {
66                             if (file.getName().endsWith(".map")) {
67                                 if (MapsforgeMapProvider.isValidMapFile(file.getAbsolutePath())) {
68                                     mapFileList.add(file.getAbsolutePath());
69                                 }
70                             }
71                         }
72                         Collections.sort(mapFileList, String.CASE_INSENSITIVE_ORDER);
73                     }
74                     return mapFileList;
75                 } catch (Exception e) {
76                     Log.e("MapsforgeMapProvider.getOfflineMaps: ", e);
77                 }
78             }
79             return Collections.emptyList();
80         }

82         public static boolean isValidMapFile(String mapFileIn) {
83     
84             if (StringUtils.isEmpty(mapFileIn)) {
85                 return false;
86             }
87     
88             MapDatabase mapDB = new MapDatabase();
89             FileOpenResult result = mapDB.openFile(new File(mapFileIn));
90             mapDB.closeFile();
91     
92             boolean isValid = result.isSuccess();
93     
94             if (!isValid) {
95                 isValid = isMapfile024(mapFileIn);
96             }
97     
98             return isValid;
99         }

101         private static boolean isMapfile024(String mapFileIn) {
102             return mapFileIn != null && org.mapsforge.android.mapsold.MapDatabase.isValidMapFile(mapFileIn);

146         /**
147          * Offline maps use the hash of the filename as ID. That way changed files can easily be detected. Also we do no
148          * longer need to differentiate between internal map sources and offline map sources, as they all just have an
149          * numerical ID (based on the hash code).

151         public static final class OfflineMapSource extends MapsforgeMapSource {
152     
153             private final String fileName;
154     
155             public OfflineMapSource(final String fileName, MapProvider mapProvider, final String name, final MapGeneratorInternal generator) {
156                 super(fileName, mapProvider, name, generator);
157                 this.fileName = fileName;
158             }
159     
160             @Override
161             public boolean isAvailable() {
162                 return isValidMapFile(fileName);
163             }
164     
165             public String getFileName() {

170         public void updateOfflineMaps() {
171             MapProviderFactory.deleteOfflineMapSources();
172             final Resources resources = CgeoApplication.getInstance().getResources();
173             final List<String> offlineMaps = getOfflineMaps();
174             for (String mapFile : offlineMaps) {
175                 final String mapName = StringUtils.capitalize(StringUtils.substringBeforeLast(new File(mapFile).getName(), "."));
176                 registerMapSource(new OfflineMapSource(mapFile, this, mapName + " (" + resources.getString(R.string.map_source_osm_offline) + ")", MapGeneratorInternal.DATABASE_RENDERER));
177             }
