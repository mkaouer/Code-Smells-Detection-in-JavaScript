38         @Override
39         public void onCreate(final Bundle savedInstanceState) {
40             super.onCreate(savedInstanceState);
41             ButterKnife.inject(this);
42             // get parameters
43             final Bundle extras = getIntent().getExtras();
44             if (extras != null) {
45                 waypointId = extras.getInt(Intents.EXTRA_WAYPOINT_ID);

49         @Override
50         public void onUpdateGeoData(IGeoData geo) {
51             if (geo.getCoords() != null && waypoint != null && waypoint.getCoords() != null) {
52                 waypointDistance.setText(Units.getDistanceFromKilometers(geo.getCoords().distanceTo(waypoint.getCoords())));
53                 waypointDistance.bringToFront();
54             }
55         }

57         @Override
58         protected void init() {
59             super.init();
60             waypoint = DataStore.loadWaypoint(waypointId);
61             try {
62                 if (StringUtils.isNotBlank(waypoint.getName())) {
63                     setTitle(waypoint.getName());
64                 } else {
65                     setTitle(waypoint.getGeocode());
66                 }
67     
68                 actionBarTitle.setCompoundDrawablesWithIntrinsicBounds(getResources().getDrawable(waypoint.getWaypointType().markerId), null, null, null);
69     
70                 details = new CacheDetailsCreator(this, waypointDetailsLayout);
71     
72                 //Waypoint geocode
73                 details.add(R.string.cache_geocode, waypoint.getPrefix() + waypoint.getGeocode().substring(2));
74                 details.addDistance(waypoint, waypointDistance);
75                 waypointDistance = details.getValueView();
76                 details.add(R.string.waypoint_note, waypoint.getNote());
77     
78                 buttonEdit.setOnClickListener(new OnClickListener() {
79     
80                     @Override
81                     public void onClick(View arg0) {
82                         EditWaypointActivity.startActivityEditWaypoint(WaypointPopup.this, cache, waypoint.getId());
83                         finish();
84                     }
85                 });
86     
87                 details = new CacheDetailsCreator(this, cacheDetailsLayout);
88                 details.add(R.string.cache_name, cache.getName());
89     
90                 addCacheDetails();
91     
92             } catch (Exception e) {
93                 Log.e("WaypointPopup.init", e);
94             }
95         }

105         @Override
106         protected void startDefaultNavigation2() {
107             if (waypoint == null || waypoint.getCoords() == null) {
108                 showToast(res.getString(R.string.cache_coordinates_no));
109                 return;
110             }
111             NavigationAppFactory.startDefaultNavigationApplication(2, this, waypoint);

115         public static void startActivity(final Context context, final int waypointId, final String geocode) {
116             final Intent popupIntent = new Intent(context, WaypointPopup.class);
117             popupIntent.putExtra(Intents.EXTRA_WAYPOINT_ID, waypointId);
118             popupIntent.putExtra(Intents.EXTRA_GEOCODE, geocode);
119             context.startActivity(popupIntent);
120         }
