49                                     statusIcon.setVisibility(View.VISIBLE);
50                                 } else {
51                                     Log.w("StatusHandler: could not find icon corresponding to @drawable/" + status.icon);
52                                     statusIcon.setVisibility(View.GONE);
53                                 }
54                             } else {
55                                 statusIcon.setVisibility(View.GONE);
56                             }
57     
58                             String message = status.message;
59                             if (status.messageId != null) {
60                                 final int messageId = res.getIdentifier(status.messageId, "string", packageName);
61                                 if (messageId != 0) {
62                                     message = res.getString(messageId);
63                                 }
64                             }
65     
66                             statusMessage.setText(message);
67                             statusGroup.setVisibility(View.VISIBLE);
68     
69                             if (status.url != null) {
70                                 statusGroup.setOnClickListener(new OnClickListener() {
71                                     @Override
72                                     public void onClick(final View v) {
73                                         startActivity(new Intent(Intent.ACTION_VIEW, Uri.parse(status.url)));
74                                     }
75                                 });
76                             } else {
77                                 statusGroup.setClickable(false);
78                             }
79                         }
80                     }, Schedulers.io());
