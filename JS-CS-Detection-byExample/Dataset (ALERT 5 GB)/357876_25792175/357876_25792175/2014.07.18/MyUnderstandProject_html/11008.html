114         @Override
115         public int getLatitudeSpan() {
116     
117             int span = 0;
118     
119             Projection projection = getProjection();
120     
121             if (projection != null && getHeight() > 0) {
122     
123                 GeoPoint low = projection.fromPixels(0, 0);
124                 GeoPoint high = projection.fromPixels(0, getHeight());
125     
126                 if (low != null && high != null) {
127                     span = Math.abs(high.latitudeE6 - low.latitudeE6);
128                 }
129             }
130     
131             return span;
132         }
133     
134         @Override
135         public int getLongitudeSpan() {
136     
137             int span = 0;
138     
139             Projection projection = getProjection();
140     
141             if (projection != null && getWidth() > 0) {
142                 GeoPoint low = projection.fromPixels(0, 0);
143                 GeoPoint high = projection.fromPixels(getWidth(), 0);
144     
145                 if (low != null && high != null) {
146                     span = Math.abs(high.longitudeE6 - low.longitudeE6);
147                 }
148             }
149     
150             return span;
151         }

179         @Override
180         public void setMapSource() {
181     
182             MapGeneratorInternal newMapType = MapGeneratorInternal.MAPNIK;
183             final MapSource mapSource = Settings.getMapSource();
184             if (mapSource instanceof MapsforgeMapSource) {
185                 newMapType = ((MapsforgeMapSource) mapSource).getGenerator();
186             }
187     
188             MapGenerator mapGenerator = MapGeneratorFactory.createMapGenerator(newMapType);
189     
190             // When swapping map sources, make sure we aren't exceeding max zoom. See bug #1535
191             final int maxZoom = mapGenerator.getZoomLevelMax();
192             if (getMapPosition().getZoomLevel() > maxZoom) {
193                 getController().setZoom(maxZoom);
194             }
195     
196             setMapGenerator(mapGenerator);
197             if (!mapGenerator.requiresInternetConnection()) {
198                 if (!new File(Settings.getMapFile()).exists()) {
199                     Toast.makeText(
200                             getContext(),
201                             getContext().getResources().getString(R.string.warn_nonexistant_mapfile),
202                             Toast.LENGTH_LONG)
203                             .show();
204                     return;
205                 }
206                 setMapFile(new File(Settings.getMapFile()));
207                 if (!Settings.isValidMapFile(Settings.getMapFile())) {
208                     Toast.makeText(
209                             getContext(),
210                             getContext().getResources().getString(R.string.warn_invalid_mapfile),
211                             Toast.LENGTH_LONG)
212                             .show();
213                 }
214             }
215             if (hasMapThemes()) {
216                 setMapTheme();
217             }
218         }
