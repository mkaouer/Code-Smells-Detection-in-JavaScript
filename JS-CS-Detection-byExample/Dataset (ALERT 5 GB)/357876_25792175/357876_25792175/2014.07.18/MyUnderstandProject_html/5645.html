106         /**
107          * The ViewPagerAdapter for scrolling through pages of the CacheDetailActivity.
108          */
109         private class ViewPagerAdapter extends PagerAdapter implements TitleProvider {
110     
111             @Override
112             public void destroyItem(ViewGroup container, int position, Object object) {
113                 if (position >= pageOrder.size()) {
114                     return;
115                 }
116                 final Page page = pageOrder.get(position);
117     
118                 // Store the state of the view if the page supports it
119                 PageViewCreator creator = viewCreators.get(page);
120                 if (creator != null) {
121                     @Nullable
122                     Bundle state = creator.getViewState();
123                     if (state != null) {
124                         viewStates.put(page, state);
125                     }
126                 }
127     
128                 container.removeView((View) object);
129             }
130     
131             @Override
132             public void finishUpdate(ViewGroup container) {
133             }
134     
135             @Override
136             public int getCount() {
137                 return pageOrder.size();
138             }
139     
140             @Override
141             public Object instantiateItem(ViewGroup container, int position) {
142     
143                 final Page page = pageOrder.get(position);
144     
145                 PageViewCreator creator = viewCreators.get(page);
146     
147                 if (null == creator && null != page) {
148                     creator = AbstractViewPagerActivity.this.createViewCreator(page);
149                     viewCreators.put(page, creator);
150                     viewStates.put(page, new Bundle());
151                 }
152     
153                 View view = null;
154     
155                 try {
156                     if (null != creator) {
157                         // Result from getView() is maybe cached, but it should be valid because the
158                         // creator should be informed about data-changes with notifyDataSetChanged()
159                         view = creator.getView();
160     
161                         // Restore the state of the view if the page supports it
162                         Bundle state = viewStates.get(page);
163                         if (state != null) {
164                             creator.setViewState(state);
165                         }
166     
167                         container.addView(view, 0);
168                     }
169                 } catch (Exception e) {
170                     Log.e("ViewPagerAdapter.instantiateItem ", e);
171                 }
172                 return view;
173             }
174     
175             @Override
176             public boolean isViewFromObject(View view, Object object) {
177                 return view == object;
178             }
179     
180             @Override
181             public void restoreState(Parcelable arg0, ClassLoader arg1) {
182             }
183     
184             @Override
185             public Parcelable saveState() {
186                 return null;
187             }
188     
189             @Override
190             public void startUpdate(ViewGroup arg0) {
191             }
192     
193             @Override
194             public int getItemPosition(Object object) {
195                 // We are doing the caching. So pretend that the view is gone.
196                 // The ViewPager will get it back in instantiateItem()
197                 return POSITION_NONE;
198             }

200             @Override
201             public String getTitle(int position) {
202                 final Page page = pageOrder.get(position);
203                 if (null == page) {
204                     return "";
205                 }
206                 return AbstractViewPagerActivity.this.getTitle(page);

211         /**
212          * Create the view pager. Call this from the {@link Activity#onCreate} implementation.
213          *
214          * @param startPageIndex
215          *            index of the page shown first
216          * @param pageSelectedListener
217          *            page selection listener or <code>null</code>
218          */
219         protected final void createViewPager(int startPageIndex, final OnPageSelectedListener pageSelectedListener) {
220             // initialize ViewPager
221             viewPager = (ViewPager) findViewById(R.id.viewpager);
222             viewPagerAdapter = new ViewPagerAdapter();
223             viewPager.setAdapter(viewPagerAdapter);
224     
225             titleIndicator = (TitlePageIndicator) findViewById(R.id.pager_indicator);
226             titleIndicator.setViewPager(viewPager);
227             if (pageSelectedListener != null) {
228                 titleIndicator.setOnPageChangeListener(new OnPageChangeListener() {
229                     @Override
230                     public void onPageSelected(int position) {
231                         pageSelectedListener.onPageSelected(position);
232                     }
233     
234                     @Override
235                     public void onPageScrolled(int position, float positionOffset, int positionOffsetPixels) {
236                     }
237     
238                     @Override
239                     public void onPageScrollStateChanged(int state) {
240                     }
241                 });
242             }
243     
244             // switch to entry page (last used or 2)
245             if (viewPagerAdapter.getCount() < startPageIndex) {
246                 for (int i = 0; i <= startPageIndex; i++) {
247                     // we can't switch to a page that is out of bounds, so we add null-pages
248                     pageOrder.add(null);
249                 }
250             }
251             viewPagerAdapter.notifyDataSetChanged();
252             viewPager.setCurrentItem(startPageIndex, false);
253         }

267         protected final void reinitializeViewPager() {
268     
269             // notify all creators that the data has changed
270             for (PageViewCreator creator : viewCreators.values()) {
271                 creator.notifyDataSetChanged();
272             }
273             // reset the stored view states of all pages
274             for (Bundle state : viewStates.values()) {
275                 state.clear();
276             }
277     
278             pageOrder.clear();
279             final Pair<List<? extends Page>, Integer> pagesAndIndex = getOrderedPages();
280             pageOrder.addAll(pagesAndIndex.getLeft());
281     
282             // Since we just added pages notifyDataSetChanged needs to be called before we possibly setCurrentItem below.
283             //  But, calling it will reset current item and we won't be able to tell if we would have been out of bounds
284             final int currentItem = getCurrentItem();
285     
286             // notify the adapter that the data has changed
287             viewPagerAdapter.notifyDataSetChanged();
288     
289             // switch to details page, if we're out of bounds
290             final int defaultPage = pagesAndIndex.getRight();
291             if (currentItem < 0 || currentItem >= viewPagerAdapter.getCount()) {
292                 viewPager.setCurrentItem(defaultPage, false);
293             }
294     
295             // notify the indicator that the data has changed
296             titleIndicator.notifyDataSetChanged();
297         }
