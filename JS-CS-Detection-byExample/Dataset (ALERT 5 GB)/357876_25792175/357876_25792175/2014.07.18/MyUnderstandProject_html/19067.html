99             assertThat(trackable.getName()).isEqualTo("Krtek - Der kleine Maulwurf");
100             final String goal = trackable.getGoal();
101             assertThat(goal).isNotNull();
102             assertThat(goal.startsWith("Bei meinem Besitzer auf der Couch")).isTrue();
103             assertThat(goal.endsWith("Geocachern zusammen fotografieren.")).isTrue();
104             assertEquals("Der kleine Maulwurf in etwas gr&ouml;&szlig;er :-)", trackable.getDetails());
105             assertThat(trackable.getGeocode()).isEqualTo("TB54VJJ");
106             assertEquals("Nordrhein-Westfalen, Germany", trackable.getOrigin());
107             assertThat(trackable.getOwner()).isEqualTo("Lineflyer");
108             // the icon url is manipulated during compression

110             assertThat(trackable.getImage().endsWith("img.geocaching.com/track/display/d9a475fa-da90-43ec-aec0-92afe26163e1.jpg")).isTrue();
111             assertThat(trackable.getOwnerGuid()).isEqualTo("d11a3e3d-7db0-4d43-87f2-7893238844a6");
112             assertThat(trackable.getSpottedGuid()).isNull();

114             assertThat(trackable.getReleased()).isNotNull();
