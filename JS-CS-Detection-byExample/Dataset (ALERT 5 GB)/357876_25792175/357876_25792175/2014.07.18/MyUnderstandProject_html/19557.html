123         public static String formatCacheInfoLong(Geocache cache, CacheListType cacheListType) {
124             final ArrayList<String> infos = new ArrayList<String>();
125             if (StringUtils.isNotBlank(cache.getGeocode())) {
126                 infos.add(cache.getGeocode());
127             }
128     
129             addShortInfos(cache, infos);
130     
131             if (cache.isPremiumMembersOnly()) {
132                 infos.add(CgeoApplication.getInstance().getString(R.string.cache_premium));
133             }
134             if (cacheListType != CacheListType.OFFLINE && cacheListType != CacheListType.HISTORY && cache.getListId() > 0) {
135                 infos.add(CgeoApplication.getInstance().getString(R.string.cache_offline));
136             }
137             return StringUtils.join(infos, Formatter.SEPARATOR);
138         }

140         public static String formatCacheInfoShort(Geocache cache) {
141             final ArrayList<String> infos = new ArrayList<String>();
142             addShortInfos(cache, infos);
143             return StringUtils.join(infos, Formatter.SEPARATOR);
144         }

146         private static void addShortInfos(Geocache cache, final ArrayList<String> infos) {
147             if (cache.hasDifficulty()) {
148                 infos.add("D " + String.format("%.1f", cache.getDifficulty()));
149             }
150             if (cache.hasTerrain()) {
151                 infos.add("T " + String.format("%.1f", cache.getTerrain()));
152             }
153     
154             // don't show "not chosen" for events and virtuals, that should be the normal case
155             if (cache.getSize() != CacheSize.UNKNOWN && cache.showSize()) {
156                 infos.add(cache.getSize().getL10n());
157             } else if (cache.isEventCache()) {
158                 final Date hiddenDate = cache.getHiddenDate();
159                 if (hiddenDate != null) {
160                     infos.add(Formatter.formatShortDate(hiddenDate.getTime()));
161                 }
162             }
163         }

165         public static String formatCacheInfoHistory(Geocache cache) {
166             final ArrayList<String> infos = new ArrayList<String>(3);
167             infos.add(StringUtils.upperCase(cache.getGeocode()));
168             infos.add(Formatter.formatDate(cache.getVisitedDate()));
169             infos.add(Formatter.formatTime(cache.getVisitedDate()));
170             return StringUtils.join(infos, Formatter.SEPARATOR);
171         }

173         public static String formatWaypointInfo(Waypoint waypoint) {
174             final List<String> infos = new ArrayList<String>(3);
175             WaypointType waypointType = waypoint.getWaypointType();
176             if (waypointType != WaypointType.OWN && waypointType != null) {
177                 infos.add(waypointType.getL10n());
178             }
179             if (Waypoint.PREFIX_OWN.equalsIgnoreCase(waypoint.getPrefix())) {
180                 infos.add(CgeoApplication.getInstance().getString(R.string.waypoint_custom));
181             } else {
182                 if (StringUtils.isNotBlank(waypoint.getPrefix())) {
183                     infos.add(waypoint.getPrefix());
184                 }
185                 if (StringUtils.isNotBlank(waypoint.getLookup())) {
186                     infos.add(waypoint.getLookup());
187                 }
188             }
189             return StringUtils.join(infos, Formatter.SEPARATOR);
190         }
