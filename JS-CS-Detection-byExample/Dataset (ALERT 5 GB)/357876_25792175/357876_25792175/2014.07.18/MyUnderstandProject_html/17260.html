7     /**
8      * Stream to measure progress of reading automatically.
9      * <p>
10      * The method @link ProgressInputStream#read(byte[]) does not need to be overridden as it delegates to @link
11      * ProgressInputStream#read(byte[], int, int) anyway.
12      * </p>
13      *
14      */
15     public class ProgressInputStream extends FilterInputStream {
16     
17         private int progress = 0;
18     
19         protected ProgressInputStream(InputStream in) {
20             super(in);
21         }
22     
23         @Override
24         public int read() throws IOException {
25             final int read = super.read();
26             if (read >= 0) {
27                 progress++;
28             }
29             return read;
30         }
31     
32         @Override
33         public int read(byte[] buffer, int offset, int count) throws IOException {
34             final int read = super.read(buffer, offset, count);
35             progress += read;
36             return read;
37         }
38     
39         int getProgress() {
40             return progress;
41         }
42     
43     }
