147         private static void setLocale(Locale locale, boolean metric) {
148             setLocale(locale, locale, metric);
149         }
150     
151         private static void setLocale(Locale locale1, Locale locale2, boolean metric) {
152             final Configuration config = new Configuration();
153             config.locale = locale1;
154             final Resources resources = CgeoApplication.getInstance().getResources();
155             resources.updateConfiguration(config, resources.getDisplayMetrics());
156     
157             Locale.setDefault(locale2);
158             TestSettings.setUseImperialUnits(!metric);
159         }
