31         public static boolean onMenuItemSelected(MenuItem item, CacheMenuHandler.ActivityInterface activityInterface, Geocache cache) {
32             assert activityInterface instanceof Activity;
33             final Activity activity = (Activity) activityInterface;
34             switch (item.getItemId()) {
35                 case R.id.menu_default_navigation:
36                     activityInterface.navigateTo();
37                     return true;
38                 case R.id.menu_navigate:
39                     activityInterface.showNavigationMenu();
40                     return true;
41                 case R.id.menu_caches_around:
42                     activityInterface.cachesAround();
43                     return true;
44                 case R.id.menu_show_in_browser:
45                     cache.openInBrowser(activity);
46                     return true;
47                 case R.id.menu_share:
48                     cache.shareCache(activity, res);
49                     return true;
50                 case R.id.menu_calendar:
51                     CalendarAddon.addToCalendarWithIntent(activity, cache);
52                     return true;
53                 default:
54                     return false;
55             }
56         }
57     
58         public static void onPrepareOptionsMenu(final Menu menu, final Geocache cache) {
59             //
60             if (cache == null) {
61                 return;
62             }
63             final boolean hasCoords = cache.getCoords() != null;
64             menu.findItem(R.id.menu_default_navigation).setVisible(hasCoords);
65             menu.findItem(R.id.menu_navigate).setVisible(hasCoords);
66             menu.findItem(R.id.menu_caches_around).setVisible(hasCoords && cache.supportsCachesAround());
67             menu.findItem(R.id.menu_calendar).setVisible(cache.canBeAddedToCalendar());
68             menu.findItem(R.id.menu_show_in_browser).setVisible(cache.canOpenInBrowser());
69     
70             menu.findItem(R.id.menu_default_navigation).setTitle(NavigationAppFactory.getDefaultNavigationApplication().getName());
71         }
72     
73         public static void addMenuItems(Activity activity, Menu menu, Geocache cache) {
74             activity.getMenuInflater().inflate(R.menu.cache_options, menu);
75             onPrepareOptionsMenu(menu, cache);
