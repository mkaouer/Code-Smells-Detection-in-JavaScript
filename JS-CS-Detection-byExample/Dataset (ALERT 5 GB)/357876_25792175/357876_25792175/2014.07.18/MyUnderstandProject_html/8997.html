12     public class SimpleCancellableHandler extends CancellableHandler {
13         public static final String MESSAGE_TEXT = "message_text";
14         protected final WeakReference<AbstractActivity> activityRef;
15         protected final WeakReference<Progress> progressDialogRef;
16     
17         public SimpleCancellableHandler(final AbstractActivity activity, final Progress progress) {
18             this.activityRef = new WeakReference<AbstractActivity>(activity);
19             this.progressDialogRef = new WeakReference<Progress>(progress);
20         }
21     
22         @Override
23         protected void handleRegularMessage(final Message msg) {
24             AbstractActivity activity = activityRef.get();
25             if (activity != null && msg.getData() != null && msg.getData().getString(MESSAGE_TEXT) != null) {
26                 activity.showToast(msg.getData().getString(MESSAGE_TEXT));
27             }
28             dismissProgress();
29         }
30     
31         @Override
32         protected void handleCancel(final Object extra) {
33             AbstractActivity activity = activityRef.get();
34             if (activity != null) {
35                 activity.showToast((String) extra);
36             }
37             dismissProgress();
38         }
39     
40         protected final void showToast(int resId) {
41             AbstractActivity activity = activityRef.get();
42             if (activity != null) {
43                 Resources res = activity.getResources();
44                 activity.showToast(res.getText(resId).toString());
45             }
46         }
47     
48         protected final void dismissProgress() {
49             Progress progressDialog = progressDialogRef.get();
50             if (progressDialog != null) {
51                 progressDialog.dismiss();
52             }
53         }
54     
55         protected final void setProgressMessage(final String txt) {
56             Progress progressDialog = progressDialogRef.get();
57             if (progressDialog != null) {
58                 progressDialog.setMessage(txt);
59             }
60         }
61     
62         protected final void finishActivity() {
63             AbstractActivity activity = activityRef.get();
64             if (activity != null) {
65                 activity.finish();
66             }
67     
68         }
69     
70         protected void updateStatusMsg(final int resId, final String msg) {
71             CacheDetailActivity activity = ((CacheDetailActivity) activityRef.get());
72             if (activity != null) {
73                 setProgressMessage(activity.getResources().getString(resId)
74                         + "\n\n"
75                         + msg);
76             }
77         }
