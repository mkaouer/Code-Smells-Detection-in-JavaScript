30         private void processAttributes(Context context, @Nullable AttributeSet attrs, int defStyle) {
31             if (attrs == null) {
32                 return;
33             }
34             TypedArray types = context.obtainStyledAttributes(attrs, getAttributeNames(),
35                     defStyle, 0);
36     
37             processAttributeValues(types);
38     
39             types.recycle();
40         }
41     
42         /**
43          * Evaluate the attributes which where requested in {@link AbstractAttributeBasedPrefence#getAttributeNames()}.
44          *

46          */

49         /**
50          * @return the names of the attributes you want to read in your preference implementation
51          */
