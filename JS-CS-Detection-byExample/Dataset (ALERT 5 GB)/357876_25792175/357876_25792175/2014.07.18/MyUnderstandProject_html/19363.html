52         public static void parseLoc(final SearchResult searchResult, final String fileContent) {
53             final Map<String, Geocache> cidCoords = parseCoordinates(fileContent);
54     
55             // save found cache coordinates
56             final HashSet<String> contained = new HashSet<String>();
57             for (String geocode : searchResult.getGeocodes()) {
58                 if (cidCoords.containsKey(geocode)) {
59                     contained.add(geocode);
60                 }
61             }
62             Set<Geocache> caches = DataStore.loadCaches(contained, LoadFlags.LOAD_CACHE_OR_DB);
63             for (Geocache cache : caches) {
64                 Geocache coord = cidCoords.get(cache.getGeocode());
65                 copyCoordToCache(coord, cache);
66             }
67         }

69         private static void copyCoordToCache(final Geocache coord, final Geocache cache) {
70             cache.setCoords(coord.getCoords());
71             cache.setDifficulty(coord.getDifficulty());
72             cache.setTerrain(coord.getTerrain());
73             cache.setSize(coord.getSize());
74             cache.setGeocode(coord.getGeocode());
75             cache.setReliableLatLon(true);
76             if (StringUtils.isBlank(cache.getName())) {
77                 cache.setName(coord.getName());
78             }
79             cache.setOwnerUserId(coord.getOwnerUserId());
80         }

82         static Map<String, Geocache> parseCoordinates(final String fileContent) {
83             final Map<String, Geocache> coords = new HashMap<String, Geocache>();
84             if (StringUtils.isBlank(fileContent)) {
85                 return coords;
86             }
87             // >> premium only
88     
89             final String[] points = fileContent.split("<waypoint>");
90     
91             // parse coordinates
92             for (String pointString : points) {
93                 final Geocache pointCoord = parseCache(pointString);
94                 if (StringUtils.isNotBlank(pointCoord.getGeocode())) {
95                     coords.put(pointCoord.getGeocode(), pointCoord);
96                 }
97             }
98     
99             Log.i("Coordinates found in .loc file: " + coords.size());
100             return coords;
101         }

103         public static Geopoint parsePoint(final String latitude, final String longitude) {
104             // the loc file contains the coordinates as plain floating point values, therefore avoid using the GeopointParser
105             try {
106                 return new Geopoint(Double.valueOf(latitude), Double.valueOf(longitude));
107             } catch (NumberFormatException e) {
108                 Log.e("LOC format has changed", e);
109             }
110             // fall back to parser, just in case the format changes
111             return new Geopoint(latitude, longitude);
112         }

118         @Override
119         public Collection<Geocache> parse(InputStream stream, CancellableHandler progressHandler) throws IOException, ParserException {
120             // TODO: progress reporting happens during reading stream only, not during parsing
121             String streamContent = readStream(stream, progressHandler).toString();
122             final Map<String, Geocache> coords = parseCoordinates(streamContent);
123             final List<Geocache> caches = new ArrayList<Geocache>();
124             for (Entry<String, Geocache> entry : coords.entrySet()) {
125                 Geocache coord = entry.getValue();
126                 if (StringUtils.isBlank(coord.getGeocode()) || StringUtils.isBlank(coord.getName())) {
127                     continue;
128                 }
129                 Geocache cache = new Geocache();
130                 cache.setReliableLatLon(true);
131                 copyCoordToCache(coord, cache);
132                 caches.add(cache);
133     
134                 fixCache(cache);
135                 cache.setType(CacheType.UNKNOWN); // type is not given in the LOC file
136                 cache.setListId(listId);
137                 cache.setDetailed(true);
138                 cache.store(null);
139             }
140             Log.i("Caches found in .loc file: " + caches.size());
141             return caches;
142         }

144         public static Geocache parseCache(final String pointString) {
145             final Geocache cache = new Geocache();
146             final MatcherWrapper matcherGeocode = new MatcherWrapper(patternGeocode, pointString);
147             if (matcherGeocode.find()) {
148                 cache.setGeocode(matcherGeocode.group(1).trim());
149             }
150     
151             final MatcherWrapper matcherName = new MatcherWrapper(patternName, pointString);
152             if (matcherName.find()) {
153                 final String name = matcherName.group(1).trim();
154                 String ownerName = StringUtils.trim(StringUtils.substringAfterLast(name, NAME_OWNER_SEPARATOR));
155                 if (StringUtils.isEmpty(cache.getOwnerUserId()) && StringUtils.isNotEmpty(ownerName)) {
156                     cache.setOwnerUserId(ownerName);
157                 }
158                 cache.setName(StringUtils.substringBeforeLast(name, NAME_OWNER_SEPARATOR).trim());
159             } else {
160                 cache.setName(cache.getGeocode());
161             }
162     
163             final MatcherWrapper matcherLat = new MatcherWrapper(patternLat, pointString);
164             final MatcherWrapper matcherLon = new MatcherWrapper(patternLon, pointString);
165             if (matcherLat.find() && matcherLon.find()) {
166                 cache.setCoords(parsePoint(matcherLat.group(1).trim(), matcherLon.group(1).trim()));
167             }
168     
169             final String difficulty = StringUtils.substringBetween(pointString, "<difficulty>", "</difficulty>");
170             final String terrain = StringUtils.substringBetween(pointString, "<terrain>", "</terrain>");
171             final String container = StringUtils.substringBetween(pointString, "<container>", "</container");
172             try {
173                 if (StringUtils.isNotBlank(difficulty)) {
174                     cache.setDifficulty(Float.parseFloat(difficulty.trim()));
175                 }
176     
177                 if (StringUtils.isNotBlank(terrain)) {
178                     cache.setTerrain(Float.parseFloat(terrain.trim()));
179                 }
180     
181                 if (StringUtils.isNotBlank(container)) {
182                     final int size = Integer.parseInt(container.trim());
183                     if (size >= 1 && size <= 8) {
184                         cache.setSize(SIZES[size - 1]);
185                     }
186                 }
187             } catch (NumberFormatException e) {
188                 Log.e("LocParser.parseCache", e);
189             }
190     
191             return cache;
192         }
