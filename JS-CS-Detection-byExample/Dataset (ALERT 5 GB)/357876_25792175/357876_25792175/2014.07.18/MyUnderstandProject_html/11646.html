109         private class LoadMapsThread extends Thread {
110     
111             @Override
112             public void run() {
113                 try {
114                     // try downloading 2 times
115                     for (int trials = 0; trials < 2; trials++) {
116                         for (int level = 1; level <= StaticMapsProvider.MAPS_LEVEL_MAX; level++) {
117                             try {
118                                 if (waypointId != null) {
119                                     final Geocache cache = DataStore.loadCache(geocode, LoadFlags.LOAD_CACHE_OR_DB);
120                                     final Bitmap image = StaticMapsProvider.getWaypointMap(geocode, cache.getWaypointById(waypointId), level);
121                                     if (image != null) {
122                                         maps.add(image);
123                                     }
124                                 } else {
125                                     final Bitmap image = StaticMapsProvider.getCacheMap(geocode, level);
126                                     if (image != null) {
127                                         maps.add(image);
128                                     }
129                                 }
130                             } catch (Exception e) {
131                                 Log.e("StaticMapsActivity.LoadMapsThread.run", e);
132                             }
133                         }
134                         if (!maps.isEmpty()) {
135                             break;
136                         }
137                     }
138     
139                     loadMapsHandler.sendMessage(Message.obtain());
140                 } catch (Exception e) {
141                     Log.e("StaticMapsActivity.LoadMapsThread.run", e);
142                 }
143             }
144         }

152         private boolean downloadStaticMaps() {
153             final Geocache cache = DataStore.loadCache(geocode, LoadFlags.LOAD_CACHE_OR_DB);
154             if (waypointId == null) {
155                 showToast(res.getString(R.string.info_storing_static_maps));
156                 StaticMapsProvider.storeCacheStaticMap(cache, true);
157                 return cache.hasStaticMap();
158             }
159             final Waypoint waypoint = cache.getWaypointById(waypointId);
160             if (waypoint != null) {
161                 showToast(res.getString(R.string.info_storing_static_maps));
162                 // refresh always removes old waypoint files
163                 StaticMapsProvider.removeWpStaticMaps(waypoint, geocode);
164                 StaticMapsProvider.storeWaypointStaticMap(cache, waypoint, true);
165                 return StaticMapsProvider.hasStaticMapForWaypoint(geocode, waypoint);
166             }
167             showToast(res.getString(R.string.err_detail_not_load_map_static));
168             return false;
169         }
