31         public void onCreate(Bundle savedInstanceState) {
32             super.onCreate(savedInstanceState);
33     
34             // get parameters
35             final Bundle extras = getIntent().getExtras();
36     
37             String geocode = null;
38             if (extras != null) {
39                 geocode = extras.getString(Intents.EXTRA_GEOCODE);
40                 imgType = (ImageType) extras.getSerializable(Intents.EXTRA_TYPE);
41             }
42     
43             if (extras == null || geocode == null) {
44                 showToast("Sorry, c:geo forgot for what cache you want to load spoiler images.");
45                 finish();
46                 return;
47             }
48     
49             // init
50             setTheme();
51             setContentView(R.layout.images_activity);
52             setTitle(res.getString(imgType.getTitle()));
53     
54             imagesList = new ImagesList(this, geocode);
55     
56             imageNames = extras.getParcelableArrayList(Intents.EXTRA_IMAGES);
57             if (CollectionUtils.isEmpty(imageNames)) {
58                 showToast(res.getString(R.string.warn_load_images));
59                 finish();
60                 return;
61             }
62     
63             offline = DataStore.isOffline(geocode, null) && (imgType == ImageType.SpoilerImages

66         }

85         private static void startActivity(final Context fromActivity, final String geocode, List<Image> logImages, ImageType imageType) {
86             final Intent logImgIntent = new Intent(fromActivity, ImagesActivity.class);
87             // if resuming our app within this activity, finish it and return to the cache activity
88             logImgIntent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_WHEN_TASK_RESET)
89                     .putExtra(Intents.EXTRA_GEOCODE, geocode)
90                     .putExtra(Intents.EXTRA_TYPE, imageType);
91     
92             // avoid forcing the array list as parameter type
93             final ArrayList<Image> arrayList = new ArrayList<Image>(logImages);
94             logImgIntent.putParcelableArrayListExtra(Intents.EXTRA_IMAGES, arrayList);
95             fromActivity.startActivity(logImgIntent);
96         }
