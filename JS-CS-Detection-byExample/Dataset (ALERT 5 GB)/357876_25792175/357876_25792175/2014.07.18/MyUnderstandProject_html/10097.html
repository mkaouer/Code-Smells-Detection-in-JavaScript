579                     if (matcherSpoilersInside.group(4) != null) {

587             }

591                 cache.setInventoryItems(0);
592

594                 if (matcherInventory.find()) {
595                     if (cache.getInventory() == null) {
596                         cache.setInventory(new ArrayList<Trackable>());
597                     }
598     
599                     if (matcherInventory.groupCount() > 1) {
600                         final String inventoryPre = matcherInventory.group(2);
601     
602                         if (StringUtils.isNotBlank(inventoryPre)) {
603                             final MatcherWrapper matcherInventoryInside = new MatcherWrapper(GCConstants.PATTERN_INVENTORYINSIDE, inventoryPre);
604     
605                             while (matcherInventoryInside.find()) {
606                                 if (matcherInventoryInside.groupCount() > 0) {

611                                     cache.getInventory().add(inventoryItem);
612                                     cache.setInventoryItems(cache.getInventoryItems() + 1);
613                                 }
614                             }
615                         }
616                     }
617                 }
618             } catch (final RuntimeException e) {
619                 // failed to parse cache inventory
620                 Log.w("GCParser.parseCache: Failed to parse cache inventory (2)");
621             }

624             try {

626                 if (null != countlogs) {
627                     final MatcherWrapper matcherLog = new MatcherWrapper(GCConstants.PATTERN_COUNTLOG, countlogs);
628     
629                     while (matcherLog.find()) {
630                         final String typeStr = matcherLog.group(1);
631                         final String countStr = getNumberString(matcherLog.group(2));
632     
633                         if (StringUtils.isNotBlank(typeStr)
634                                 && LogType.UNKNOWN != LogType.getByIconName(typeStr)

637                         }

639                 }
640             } catch (final NumberFormatException e) {
641                 // failed to parse logs
642                 Log.w("GCParser.parseCache: Failed to parse cache log count");
643             }

646             cache.setWaypoints(Collections.<Waypoint> emptyList(), false);
647     
648             // add waypoint for original coordinates in case of user-modified listing-coordinates
649             try {
650                 final String originalCoords = TextUtils.getMatch(page, GCConstants.PATTERN_LATLON_ORIG, false, null);
651     
652                 if (null != originalCoords) {
653                     final Waypoint waypoint = new Waypoint(CgeoApplication.getInstance().getString(R.string.cache_coordinates_original), WaypointType.ORIGINAL, false);
654                     waypoint.setCoords(new Geopoint(originalCoords));
655                     cache.addOrChangeWaypoint(waypoint, false);
656                     cache.setUserModifiedCoords(true);
657                 }
658             } catch (final Geopoint.GeopointException e) {
659             }
660     
661             int wpBegin = page.indexOf("<table class=\"Table\" id=\"ctl00_ContentBody_Waypoints\">");
662             if (wpBegin != -1) { // parse waypoints
663                 if (CancellableHandler.isCancelled(handler)) {
664                     return null;
665                 }
666                 CancellableHandler.sendLoadProgressDetail(handler, R.string.cache_dialog_loading_details_status_waypoints);
667     
668                 String wpList = page.substring(wpBegin);
669     
670                 int wpEnd = wpList.indexOf("</p>");
671                 if (wpEnd > -1 && wpEnd <= wpList.length()) {
672                     wpList = wpList.substring(0, wpEnd);
673                 }
674     
675                 if (!wpList.contains("No additional waypoints to display.")) {
676                     wpEnd = wpList.indexOf("</table>");
677                     wpList = wpList.substring(0, wpEnd);
678     
679                     wpBegin = wpList.indexOf("<tbody>");
680                     wpEnd = wpList.indexOf("</tbody>");
681                     if (wpBegin >= 0 && wpEnd >= 0 && wpEnd <= wpList.length()) {
682                         wpList = wpList.substring(wpBegin + 7, wpEnd);
683                     }
684     
685                     final String[] wpItems = wpList.split("<tr");
686     
687                     for (int j = 1; j < wpItems.length; j++) {
688                         String[] wp = wpItems[j].split("<td");
689     
690                         // waypoint name
691                         // res is null during the unit tests
692                         final String name = TextUtils.getMatch(wp[6], GCConstants.PATTERN_WPNAME, true, 1, CgeoApplication.getInstance().getString(R.string.waypoint), true);
693     
694                         // waypoint type
695                         final String resulttype = TextUtils.getMatch(wp[3], GCConstants.PATTERN_WPTYPE, null);
696     
697                         final Waypoint waypoint = new Waypoint(name, WaypointType.findById(resulttype), false);
698     
699                         // waypoint prefix
700                         waypoint.setPrefix(TextUtils.getMatch(wp[4], GCConstants.PATTERN_WPPREFIXORLOOKUPORLATLON, true, 2, waypoint.getPrefix(), false));
701     
702                         // waypoint lookup
703                         waypoint.setLookup(TextUtils.getMatch(wp[5], GCConstants.PATTERN_WPPREFIXORLOOKUPORLATLON, true, 2, waypoint.getLookup(), false));
704     
705                         // waypoint latitude and longitude
706                         latlon = Html.fromHtml(TextUtils.getMatch(wp[7], GCConstants.PATTERN_WPPREFIXORLOOKUPORLATLON, false, 2, "", false)).toString().trim();
707                         if (!StringUtils.startsWith(latlon, "???")) {
708                             waypoint.setLatlon(latlon);
709                             waypoint.setCoords(new Geopoint(latlon));
710                         }
711     
712                         j++;
713                         if (wpItems.length > j) {
714                             wp = wpItems[j].split("<td");
715                         }
716     
717                         // waypoint note
718                         waypoint.setNote(TextUtils.getMatch(wp[3], GCConstants.PATTERN_WPNOTE, waypoint.getNote()));
719     
720                         cache.addOrChangeWaypoint(waypoint, false);
721                     }
722                 }
723             }
724     
725             cache.parseWaypointsFromNote();
726     
727             // last check for necessary cache conditions
728             if (StringUtils.isBlank(cache.getGeocode())) {
729                 searchResult.setError(StatusCode.UNKNOWN_ERROR);
730                 return searchResult;
731             }
732     
733             cache.setDetailedUpdatedNow();
734             searchResult.addAndPutInCache(Collections.singletonList(cache));
735             DataStore.saveLogsWithoutTransaction(cache.getGeocode(), getLogsFromDetails(page, false));
736             return searchResult;
737         }

1414         /**
1415          * Parse a trackable HTML description into a Trackable object
1416          *
1417          * @param page
1418          *            the HTML page to parse, already processed through {@link TextUtils#replaceWhitespace}
1419          * @return the parsed trackable, or null if none could be parsed
1420          */
1421         static Trackable parseTrackable(final String page, final String possibleTrackingcode) {
1422             if (StringUtils.isBlank(page)) {
1423                 Log.e("GCParser.parseTrackable: No page given");
1424                 return null;
1425             }
1426     
1427             if (page.contains(GCConstants.ERROR_TB_DOES_NOT_EXIST) || page.contains(GCConstants.ERROR_TB_ARITHMETIC_OVERFLOW) || page.contains(GCConstants.ERROR_TB_ELEMENT_EXCEPTION)) {
1428                 return null;
1429             }
1430     
1431             final Trackable trackable = new Trackable();
1432     
1433             // trackable geocode
1434             trackable.setGeocode(TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_GEOCODE, true, StringUtils.upperCase(possibleTrackingcode)));
1435             if (trackable.getGeocode() == null) {
1436                 Log.e("GCParser.parseTrackable: could not figure out trackable geocode");
1437                 return null;
1438             }
1439     
1440             // trackable id
1441             trackable.setGuid(TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_GUID, true, trackable.getGuid()));
1442     
1443             // trackable icon
1444             trackable.setIconUrl(TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_ICON, true, trackable.getIconUrl()));
1445     
1446             // trackable name
1447             trackable.setName(Html.fromHtml(TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_NAME, true, "")).toString());
1448     
1449             // trackable type
1450             if (StringUtils.isNotBlank(trackable.getName())) {
1451                 trackable.setType(TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_TYPE, true, trackable.getType()));
1452             }
1453     
1454             // trackable owner name
1455             try {
1456                 final MatcherWrapper matcherOwner = new MatcherWrapper(GCConstants.PATTERN_TRACKABLE_OWNER, page);
1457                 if (matcherOwner.find() && matcherOwner.groupCount() > 0) {
1458                     trackable.setOwnerGuid(matcherOwner.group(1));
1459                     trackable.setOwner(matcherOwner.group(2).trim());
1460                 }
1461             } catch (final RuntimeException e) {
1462                 // failed to parse trackable owner name
1463                 Log.w("GCParser.parseTrackable: Failed to parse trackable owner name");
1464             }
1465     
1466             // trackable origin
1467             trackable.setOrigin(TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_ORIGIN, true, trackable.getOrigin()));
1468     
1469             // trackable spotted
1470             try {
1471                 final MatcherWrapper matcherSpottedCache = new MatcherWrapper(GCConstants.PATTERN_TRACKABLE_SPOTTEDCACHE, page);
1472                 if (matcherSpottedCache.find() && matcherSpottedCache.groupCount() > 0) {
1473                     trackable.setSpottedGuid(matcherSpottedCache.group(1));
1474                     trackable.setSpottedName(matcherSpottedCache.group(2).trim());
1475                     trackable.setSpottedType(Trackable.SPOTTED_CACHE);
1476                 }
1477     
1478                 final MatcherWrapper matcherSpottedUser = new MatcherWrapper(GCConstants.PATTERN_TRACKABLE_SPOTTEDUSER, page);
1479                 if (matcherSpottedUser.find() && matcherSpottedUser.groupCount() > 0) {
1480                     trackable.setSpottedGuid(matcherSpottedUser.group(1));
1481                     trackable.setSpottedName(matcherSpottedUser.group(2).trim());
1482                     trackable.setSpottedType(Trackable.SPOTTED_USER);
1483                 }
1484     
1485                 if (TextUtils.matches(page, GCConstants.PATTERN_TRACKABLE_SPOTTEDUNKNOWN)) {
1486                     trackable.setSpottedType(Trackable.SPOTTED_UNKNOWN);
1487                 }
1488     
1489                 if (TextUtils.matches(page, GCConstants.PATTERN_TRACKABLE_SPOTTEDOWNER)) {
1490                     trackable.setSpottedType(Trackable.SPOTTED_OWNER);
1491                 }
1492             } catch (final RuntimeException e) {
1493                 // failed to parse trackable last known place
1494                 Log.w("GCParser.parseTrackable: Failed to parse trackable last known place");
1495             }
1496     
1497             // released date - can be missing on the page
1498             final String releaseString = TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_RELEASES, false, null);
1499             if (releaseString != null) {
1500                 try {
1501                     trackable.setReleased(dateTbIn1.parse(releaseString));
1502                 } catch (ParseException e) {
1503                     if (trackable.getReleased() == null) {
1504                         try {
1505                             trackable.setReleased(dateTbIn2.parse(releaseString));
1506                         } catch (ParseException e1) {
1507                             Log.e("Could not parse trackable release " + releaseString);
1508                         }
1509                     }
1510                 }
1511             }
1512     
1513             // trackable distance
1514             final String distance = TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_DISTANCE, false, null);
1515             if (null != distance) {
1516                 try {
1517                     trackable.setDistance(DistanceParser.parseDistance(distance,
1518                             !Settings.isUseImperialUnits()));
1519                 } catch (final NumberFormatException e) {
1520                     Log.e("GCParser.parseTrackable: Failed to parse distance", e);
1521                 }
1522             }
1523     
1524             // trackable goal
1525             trackable.setGoal(convertLinks(TextUtils.getMatch(page, GCConstants.PATTERN_TRACKABLE_GOAL, true, trackable.getGoal())));
1526     
1527             // trackable details & image
1528             try {
1529                 final MatcherWrapper matcherDetailsImage = new MatcherWrapper(GCConstants.PATTERN_TRACKABLE_DETAILSIMAGE, page);
1530                 if (matcherDetailsImage.find() && matcherDetailsImage.groupCount() > 0) {
1531                     final String image = StringUtils.trim(matcherDetailsImage.group(3));
1532                     final String details = StringUtils.trim(matcherDetailsImage.group(4));
1533     
1534                     if (StringUtils.isNotEmpty(image)) {
1535                         trackable.setImage(image);
1536                     }
1537                     if (StringUtils.isNotEmpty(details) && !StringUtils.equals(details, "No additional details available.")) {
1538                         trackable.setDetails(convertLinks(details));
1539                     }
1540                 }
1541             } catch (final RuntimeException e) {
1542                 // failed to parse trackable details & image
1543                 Log.w("GCParser.parseTrackable: Failed to parse trackable details & image");
1544             }
1545             if (StringUtils.isEmpty(trackable.getDetails()) && page.contains(GCConstants.ERROR_TB_NOT_ACTIVATED)) {
1546                 trackable.setDetails(CgeoApplication.getInstance().getString(R.string.trackable_not_activated));
1547             }
1548     
1549             // trackable logs
1550             try {
1551                 final MatcherWrapper matcherLogs = new MatcherWrapper(GCConstants.PATTERN_TRACKABLE_LOG, page);
1552                 /*
1553                  * 1. Type (image)
1554                  * 2. Date
1555                  * 3. Author
1556                  * 4. Cache-GUID
1557                  * 5. <ignored> (strike-through property for ancient caches)
1558                  * 6. Cache-name
1559                  * 7. Log text
1560                  */
1561                 while (matcherLogs.find()) {
1562                     long date = 0;
1563                     try {
1564                         date = GCLogin.parseGcCustomDate(matcherLogs.group(2)).getTime();
1565                     } catch (final ParseException e) {
1566                     }
1567     
1568                     final LogEntry logDone = new LogEntry(
1569                             Html.fromHtml(matcherLogs.group(3)).toString().trim(),
1570                             date,
1571                             LogType.getByIconName(matcherLogs.group(1)),
1572                             matcherLogs.group(7).trim());
1573     
1574                     if (matcherLogs.group(4) != null && matcherLogs.group(6) != null) {
1575                         logDone.cacheGuid = matcherLogs.group(4);
1576                         logDone.cacheName = matcherLogs.group(6);
1577                     }
1578     
1579                     // Apply the pattern for images in a trackable log entry against each full log (group(0))
1580                     final String logEntry = matcherLogs.group(0);
1581                     final MatcherWrapper matcherLogImages = new MatcherWrapper(GCConstants.PATTERN_TRACKABLE_LOG_IMAGES, logEntry);
1582                     /*
1583                      * 1. Image URL
1584                      * 2. Image title
1585                      */
1586                     while (matcherLogImages.find()) {
1587                         final Image logImage = new Image(matcherLogImages.group(1), matcherLogImages.group(2));
1588                         logDone.addLogImage(logImage);
1589                     }
1590     
1591                     trackable.getLogs().add(logDone);
1592                 }
1593             } catch (final Exception e) {
1594                 // failed to parse logs
1595                 Log.w("GCParser.parseCache: Failed to parse cache logs", e);
1596             }
1597     
1598             // tracking code
1599             if (!StringUtils.equalsIgnoreCase(trackable.getGeocode(), possibleTrackingcode)) {
1600                 trackable.setTrackingcode(possibleTrackingcode);
1601             }
1602     
1603             if (CgeoApplication.getInstance() != null) {
1604                 DataStore.saveTrackable(trackable);
1605             }
1606     
1607             return trackable;
1608         }

1673             // for non logged in users the log book is not shown
1674             if (StringUtils.isBlank(rawResponse)) {
1675                 return logs;
1676             }
1677     
1678             try {
1679                 final JSONObject resp = new JSONObject(rawResponse);
1680                 if (!resp.getString("status").equals("success")) {
1681                     Log.e("GCParser.loadLogsFromDetails: status is " + resp.getString("status"));
1682                     return null;
1683                 }
1684     
1685                 final JSONArray data = resp.getJSONArray("data");
1686     
1687                 for (int index = 0; index < data.length(); index++) {
1688                     final JSONObject entry = data.getJSONObject(index);
1689     
1690                     // FIXME: use the "LogType" field instead of the "LogTypeImage" one.
1691                     final String logIconNameExt = entry.optString("LogTypeImage", ".gif");
1692                     final String logIconName = logIconNameExt.substring(0, logIconNameExt.length() - 4);
1693     
1694                     long date = 0;
1695                     try {
1696                         date = GCLogin.parseGcCustomDate(entry.getString("Visited")).getTime();
1697                     } catch (final ParseException e) {
1698                         Log.e("GCParser.loadLogsFromDetails: failed to parse log date.");
1699                     }
1700     
1701                     // TODO: we should update our log data structure to be able to record
1702                     // proper coordinates, and make them clickable. In the meantime, it is
1703                     // better to integrate those coordinates into the text rather than not
1704                     // display them at all.
1705                     final String latLon = entry.getString("LatLonString");
1706                     final String logText = (StringUtils.isEmpty(latLon) ? "" : (latLon + "<br/><br/>")) + TextUtils.removeControlCharacters(entry.getString("LogText"));
1707                     final LogEntry logDone = new LogEntry(
1708                             TextUtils.removeControlCharacters(entry.getString("UserName")),
1709                             date,
1710                             LogType.getByIconName(logIconName),
1711                             logText);
1712                     logDone.found = entry.getInt("GeocacheFindCount");
1713                     logDone.friend = friends;
1714     
1715                     final JSONArray images = entry.getJSONArray("Images");
1716                     for (int i = 0; i < images.length(); i++) {
1717                         final JSONObject image = images.getJSONObject(i);
1718                         final String url = "http://imgcdn.geocaching.com/cache/log/large/" + image.getString("FileName");
1719                         final String title = TextUtils.removeControlCharacters(image.getString("Name"));
1720                         final Image logImage = new Image(url, title);
1721                         logDone.addLogImage(logImage);
1722                     }
1723     
1724                     logs.add(logDone);
1725                 }
1726             } catch (final JSONException e) {
1727                 // failed to parse logs
1728                 Log.w("GCParser.loadLogsFromDetails: Failed to parse cache logs", e);
1729             }
1730     
1731             return logs;
1732         }

1735         public static List<LogType> parseTypes(String page) {
1736             if (StringUtils.isEmpty(page)) {
1737                 return Collections.emptyList();
1738             }
1739     
1740             final List<LogType> types = new ArrayList<LogType>();
1741     
1742             final MatcherWrapper typeBoxMatcher = new MatcherWrapper(GCConstants.PATTERN_TYPEBOX, page);
1743             if (typeBoxMatcher.find() && typeBoxMatcher.groupCount() > 0) {
1744                 final String typesText = typeBoxMatcher.group(1);
1745                 final MatcherWrapper typeMatcher = new MatcherWrapper(GCConstants.PATTERN_TYPE2, typesText);
1746                 while (typeMatcher.find()) {
1747                     if (typeMatcher.groupCount() > 1) {
1748                         try {
1749                             final int type = Integer.parseInt(typeMatcher.group(2));
1750                             if (type > 0) {
1751                                 types.add(LogType.getById(type));
1752                             }
1753                         } catch (final NumberFormatException e) {
1754                             Log.e("Error parsing log types", e);
1755                         }
1756                     }
1757                 }
1758             }
1759     
1760             // we don't support this log type
1761             types.remove(LogType.UPDATE_COORDINATES);
1762     
1763             return types;
1764         }

1766         public static List<TrackableLog> parseTrackableLog(final String page) {
1767             if (StringUtils.isEmpty(page)) {
1768                 return null;
1769             }
1770     
1771             String table = StringUtils.substringBetween(page, "<table id=\"tblTravelBugs\"", "</table>");
1772     
1773             // if no trackables are currently in the account, the table is not available, so return an empty list instead of null
1774             if (StringUtils.isBlank(table)) {
1775                 return Collections.emptyList();
1776             }
1777     
1778             table = StringUtils.substringBetween(table, "<tbody>", "</tbody>");
1779             if (StringUtils.isBlank(table)) {
1780                 Log.e("GCParser.parseTrackableLog: tbody not found on page");
1781                 return null;
1782             }
1783     
1784             final List<TrackableLog> trackableLogs = new ArrayList<TrackableLog>();
1785     
1786             final MatcherWrapper trackableMatcher = new MatcherWrapper(GCConstants.PATTERN_TRACKABLE, page);
1787             while (trackableMatcher.find()) {
1788                 if (trackableMatcher.groupCount() > 0) {
1789     
1790                     final String trackCode = trackableMatcher.group(1);
1791                     final String name = Html.fromHtml(trackableMatcher.group(2)).toString();
1792                     try {
1793                         final Integer ctl = Integer.valueOf(trackableMatcher.group(3));
1794                         final Integer id = Integer.valueOf(trackableMatcher.group(5));
1795                         if (trackCode != null && ctl != null && id != null) {
1796                             final TrackableLog entry = new TrackableLog(trackCode, name, id, ctl);
1797     
1798                             Log.i("Trackable in inventory (#" + entry.ctl + "/" + entry.id + "): " + entry.trackCode + " - " + entry.name);
1799                             trackableLogs.add(entry);
1800                         }
1801                     } catch (final NumberFormatException e) {
1802                         Log.e("GCParser.parseTrackableLog", e);
1803                     }
1804                 }
1805             }
1806     
1807             return trackableLogs;
1808         }
