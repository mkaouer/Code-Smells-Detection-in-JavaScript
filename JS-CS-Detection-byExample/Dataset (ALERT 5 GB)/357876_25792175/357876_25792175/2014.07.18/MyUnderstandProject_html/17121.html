155             final Geocache cache = new Geocache();
156             cache.setGeocode("GC2ZN4G");
157             // upload coordinates
158             GCParser.editModifiedCoordinates(cache, new Geopoint("N51 21.544", "E07 02.566"));
159             cache.dropSynchronous();
160             final String page = GCParser.requestHtmlPage(cache.getGeocode(), null, "n", "0");
161             final Geocache cache2 = GCParser.parseCacheFromText(page, null).getFirstCacheFromResult(LoadFlags.LOAD_CACHE_ONLY);
162             assertThat(cache2).isNotNull();
163             assert (cache2 != null); // eclipse bug
164             assertThat(cache2.hasUserModifiedCoords()).isTrue();
165             assertEquals(new Geopoint("N51 21.544", "E07 02.566"), cache2.getCoords());
166             // delete coordinates
167             GCParser.deleteModifiedCoordinates(cache2);
168             cache2.dropSynchronous();
169             final String page2 = GCParser.requestHtmlPage(cache.getGeocode(), null, "n", "0");
170             final Geocache cache3 = GCParser.parseCacheFromText(page2, null).getFirstCacheFromResult(LoadFlags.LOAD_CACHE_ONLY);
171             assertThat(cache3).isNotNull();
172             assert (cache3 != null); // eclipse bug
173             assertThat(cache3.hasUserModifiedCoords()).isFalse();
