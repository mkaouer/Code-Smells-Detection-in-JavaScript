47         final GeoDirHandler geoDirHandler = new GeoDirHandler() {
48             @Override
49             public void updateGeoDir(final IGeoData newGeo, final float newDirection) {
50                 position = newGeo.getCoords();
51                 direction = newDirection;
52                 // avoid any calculation, if the delay since the last output is not long enough
53                 final long now = System.currentTimeMillis();
54                 if (now - lastSpeechTime <= SPEECH_MINPAUSE_SECONDS * 1000) {
55                     return;
56                 }
57     
58                 // to speak, we want max pause to have elapsed or distance to geopoint to have changed by a given amount
59                 final float distance = position.distanceTo(target);
60                 if (now - lastSpeechTime <= SPEECH_MAXPAUSE_SECONDS * 1000) {
61                     if (Math.abs(lastSpeechDistance - distance) < getDeltaForDistance(distance)) {
62                         return;
63                     }
64                 }
65     
66                 final String text = TextFactory.getText(position, target, direction);
67                 if (StringUtils.isNotEmpty(text)) {
68                     lastSpeechTime = System.currentTimeMillis();
69                     lastSpeechDistance = distance;
70                     speak(text);
71                 }
72             }
73         };
74         /**
75          * remember when we talked the last time
76          */
77         private long lastSpeechTime = 0;
78         private float lastSpeechDistance = 0.0f;
79         private Geopoint target;
80         private Subscription initSubscription = Subscriptions.empty();

87         /**
88          * Return distance required to be moved based on overall distance.<br>
89          *
90          * @param distance
91          *            in km
92          * @return delta in km
93          */
94         private static float getDeltaForDistance(final float distance) {
95             if (distance > 1.0) {
96                 return 0.2f;
97             }
98             if (distance > 0.05) {
99                 return distance / 5.0f;
100             }
101             return 0f;
102         }

120         @Override
121         public void onInit(int status) {
122             // The text to speech system takes some time to initialize.
123             if (status != TextToSpeech.SUCCESS) {
124                 Log.e("Text to speech cannot be initialized.");
125                 return;
126             }
127     
128             Locale locale = Locale.getDefault();
129             if (Settings.isUseEnglish()) {
130                 locale = Locale.ENGLISH;
131             }
132     
133             int switchLocale = tts.setLanguage(locale);
134     
135             if (switchLocale == TextToSpeech.LANG_MISSING_DATA) {
136                 startingActivity.startActivity(new Intent(Engine.ACTION_INSTALL_TTS_DATA));
137                 return;
138             }
139             if (switchLocale == TextToSpeech.LANG_NOT_SUPPORTED) {
140                 Log.e("Current languge not supported by text to speech.");
141                 ActivityMixin.showToast(startingActivity, R.string.err_tts_lang_not_supported);
142                 return;
143             }
144     
145             initialized = true;
146     
147             initSubscription = geoDirHandler.start(GeoDirHandler.UPDATE_GEODIR);
148         }
149     
150         @Override
151         public int onStartCommand(Intent intent, int flags, int startId) {
152             if (intent != null) {
153                 target = intent.getParcelableExtra(EXTRA_TARGET_COORDS);
154             }
155             return START_NOT_STICKY; // service can be stopped by system, if under memory pressure
156         }
157     
158         private void speak(final String text) {
159             if (!initialized) {
160                 return;
161             }
162             tts.speak(text, TextToSpeech.QUEUE_FLUSH, null);
163         }
