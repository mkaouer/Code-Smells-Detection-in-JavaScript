28         public static void postTweetCache(String geocode, final @Nullable LogEntry logEntry) {
29             final Geocache cache = DataStore.loadCache(geocode, LoadFlags.LOAD_CACHE_OR_DB);
30             if (cache == null) {
31                 return;
32             }
33             postTweet(CgeoApplication.getInstance(), getStatusMessage(cache, logEntry), null);
34         }

36         public static void postTweetTrackable(String geocode, final @Nullable LogEntry logEntry) {
37             final Trackable trackable = DataStore.loadTrackable(geocode);
38             if (trackable == null) {
39                 return;
40             }
41             postTweet(CgeoApplication.getInstance(), getStatusMessage(trackable, logEntry), null);
42         }

44         private static void postTweet(final CgeoApplication app, final String statusIn, final Geopoint coords) {
45             if (app == null || !Settings.isUseTwitter() || !Settings.isTwitterLoginValid()) {
46                 return;
47             }
48     
49             try {
50                 final String status = shortenToMaxSize(statusIn);
51                 Parameters parameters = new Parameters("status", status);
52                 if (coords != null) {
53                     parameters.put(
54                             "lat", coords.format(Format.LAT_DECDEGREE_RAW),
55                             "long", coords.format(Format.LON_DECDEGREE_RAW),
56                             "display_coordinates", "true");
57                 }
58     
59                 OAuth.signOAuth("api.twitter.com", "/1.1/statuses/update.json", "POST", true, parameters, Settings.getTokenPublic(), Settings.getTokenSecret(), Settings.getKeyConsumerPublic(), Settings.getKeyConsumerSecret());
60                 final HttpResponse httpResponse = Network.postRequest("https://api.twitter.com/1.1/statuses/update.json", parameters);
61                 if (httpResponse != null) {
62                     if (httpResponse.getStatusLine().getStatusCode() == 200) {
63                         Log.i("Tweet posted");
64                     } else {
65                         Log.e("Tweet could not be posted. Reason: " + httpResponse.toString());
66                     }
67                 } else {
68                     Log.e("Tweet could not be posted. Reason: httpResponse Object is null");
69                 }
70             } catch (Exception e) {
71                 Log.e("Twitter.postTweet", e);
72             }
73         }

75         private static String shortenToMaxSize(final String status) {
76             String result = StringUtils.trim(status);
77             if (StringUtils.length(result) > MAX_TWEET_SIZE) {
78                 return StringUtils.substring(result, 0, MAX_TWEET_SIZE - 1) + 'â€¦';
79             }
80             return result;
81         }
82     
83         private static void appendHashTag(final StringBuilder status, final String tag) {
84             if (status.length() + HASH_PREFIX_WITH_BLANK.length() + tag.length() <= MAX_TWEET_SIZE) {
85                 final String tagWithPrefix = HASH_PREFIX_WITH_BLANK + tag;
86                 if (!StringUtils.contains(status, tagWithPrefix)) {
87                     status.append(tagWithPrefix);
88                 }
89             }
90         }

92         static String getStatusMessage(final @NonNull Geocache cache, final @Nullable LogEntry logEntry) {
93             return appendHashTags(LogTemplateProvider.applyTemplates(Settings.getCacheTwitterMessage(), new LogContext(cache, logEntry)));
94         }
95     
96         static String getStatusMessage(final @NonNull Trackable trackable, final @Nullable LogEntry logEntry) {
97             return appendHashTags(LogTemplateProvider.applyTemplates(Settings.getTrackableTwitterMessage(), new LogContext(trackable, logEntry)));
98         }
99     
100         private static String appendHashTags(final String status) {
101             StringBuilder builder = new StringBuilder(status);
102             appendHashTag(builder, "cgeo");
103             appendHashTag(builder, "geocaching");
104             return builder.toString();
105         }
