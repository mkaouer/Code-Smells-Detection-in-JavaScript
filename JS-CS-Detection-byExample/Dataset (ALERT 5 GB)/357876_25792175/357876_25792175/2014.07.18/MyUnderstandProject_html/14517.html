17     /**
18      * Preference which shows a dialog containing textual explanation. The dialog has two buttons, where one will open a
19      * hyper link with more detailed information.
20      * <p>
21      * The URL for the hyper link and the text are given as custom attributes in the preference XML definition.
22      * </p>
23      *
24      */
25     public class InfoPreference extends AbstractAttributeBasedPrefence {
26     
27         /**
28          * Content of the dialog, filled from preferences XML.
29          */
30         private String text;
31         /**
32          * URL for the second button, filled from preferences XML.
33          */
34         private String url;
35         /**
36          * text for the second button to open an URL, filled from preferences XML.
37          */
38         private String urlButton;
39     
40         private LayoutInflater inflater;
41     
42         public InfoPreference(Context context) {
43             super(context);
44             init(context);
45         }
46     
47         public InfoPreference(Context context, AttributeSet attrs) {
48             super(context, attrs);
49             init(context);
50         }
51     
52         public InfoPreference(Context context, AttributeSet attrs, int defStyle) {
53             super(context, attrs, defStyle);
54             init(context);
55         }
56     
57         private void init(Context context) {
58             inflater = ((Activity) context).getLayoutInflater();
59             setPersistent(false);
60         }
61     
62         @Override
63         protected int[] getAttributeNames() {
64             return new int[] { android.R.attr.text, R.attr.url, R.attr.urlButton };
65         }
66     
67         @Override
68         protected void processAttributeValues(TypedArray values) {
69             text = values.getString(0);
70             url = values.getString(1);
71             urlButton = values.getString(2);
72         }
73     
74         @Override
75         protected View onCreateView(ViewGroup parent) {
76     
77             // show popup when clicked
78             setOnPreferenceClickListener(new OnPreferenceClickListener() {
79     
80                 @Override
81                 public boolean onPreferenceClick(final Preference preference) {
82                     new UrlPopup(preference.getContext()).show(preference.getTitle().toString(), text, url, urlButton);
83                     // don't update the preference value
84                     return false;
85                 }
86             });
87     
88             return addInfoIcon(parent);
89         }
90     
91         /**
92          * Add an info icon at the left hand side of the preference.
93          *
94          * @param parent
95          * @return
96          */
97         private View addInfoIcon(ViewGroup parent) {
98             View preferenceView = super.onCreateView(parent);
99     
100             ImageView iconView = (ImageView) inflater.inflate(R.layout.preference_info_icon, parent, false);
101             LinearLayout frame = (LinearLayout) preferenceView.findViewById(android.R.id.widget_frame);
102             frame.setVisibility(View.VISIBLE);
103             frame.addView(iconView);
104     
105             return preferenceView;
106         }
