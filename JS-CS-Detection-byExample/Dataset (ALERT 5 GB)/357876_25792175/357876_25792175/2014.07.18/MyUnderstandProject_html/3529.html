34         /**
35          * Set the map zoom level to mapzoom-1 or maxZoomLevel, whichever is least
36          * mapzoom-1 is used to be compatible with Google Maps zoom levels
37          */
38         @Override
39         public void setZoom(int mapzoom) {
40             // Google Maps and OSM Maps use different zoom levels for the same view.
41             // All OSM Maps zoom levels are offset by 1 so they match Google Maps.
42             mapController.setZoom(Math.min(mapzoom - 1, maxZoomLevel));
43         }
44     
45         @Override
46         public void zoomToSpan(int latSpanE6, int lonSpanE6) {
47     
48             if (latSpanE6 != 0 && lonSpanE6 != 0) {
49                 // calculate zoomlevel
50                 int distDegree = Math.max(latSpanE6, lonSpanE6);
51                 int zoomLevel = (int) Math.floor(Math.log(360.0 * 1e6 / distDegree) / Math.log(2));
52                 mapController.setZoom(zoomLevel + 1);
53             }
54         }
