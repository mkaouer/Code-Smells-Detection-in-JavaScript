18         private String[] suggestions = EMPTY;

24         }
25     
26         @Override
27         public int getCount() {
28             return suggestions.length;
29         }
30     
31         @Override
32         public String getItem(int index) {
33             return suggestions[index];

36         @Override
37         public Filter getFilter() {
38             return new Filter() {
39     
40                 @Override
41                 protected FilterResults performFiltering(CharSequence constraint) {
42                     FilterResults filterResults = new FilterResults();
43                     if (constraint == null) {
44                         return filterResults;
45                     }
46                     String trimmed = StringUtils.trim(constraint.toString());
47                     if (StringUtils.length(trimmed) >= 2) {
48                         String[] newResults = suggestionFunction.call(trimmed);
49     
50                         // Assign the data to the FilterResults, but do not yet store in the global member.
51                         // Otherwise we might invalidate the adapter and cause an IllegalStateException.
52                         filterResults.values = newResults;
53                         filterResults.count = newResults.length;
54                     }
55                     return filterResults;
56                 }
57     
58                 @Override
59                 protected void publishResults(CharSequence constraint, FilterResults filterResults) {
60                     if (filterResults != null && filterResults.count > 0) {
61                         suggestions = (String[]) filterResults.values;
62                         notifyDataSetChanged();
63                     }
64                     else {
65                         notifyDataSetInvalidated();
66                     }
67                 }
68             };
69         }
