11         private final boolean isDetailed;

13         public OXGPXParser(int listIdIn, boolean isDetailed) {
14             super(listIdIn);

16         }
17     
18         @Override
19         protected void afterParsing(Geocache cache) {
20             cache.setUpdated(System.currentTimeMillis());
21             if (isDetailed) {
22                 cache.setDetailedUpdate(cache.getUpdated());
23                 cache.setDetailed(true);
24             }
25             removeTitleFromShortDescription(cache);
26         }
27     
28         /**
29          * The short description of OX caches contains "title by owner, type(T/D/Awesomeness)". That is a lot of
30          * duplication.
31          *
32          * @param cache
33          */
34         private static void removeTitleFromShortDescription(final @NonNull Geocache cache) {
35             cache.setShortDescription(StringUtils.trim(StringUtils.substringAfterLast(cache.getShortDescription(), ",")));
36         }
