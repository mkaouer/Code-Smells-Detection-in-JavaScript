253         /**
254          * Searches the view port on the live map with Strategy.AUTO
255          *
256          * @param viewport
257          *            Area to search
258          * @param tokens
259          *            Live map tokens
260          * @return
261          */
262         public static SearchResult searchByViewport(final Viewport viewport, final MapTokens tokens) {
263             int speed = (int) CgeoApplication.getInstance().currentGeo().getSpeed() * 60 * 60 / 1000; // in km/h
264             Strategy strategy = Settings.getLiveMapStrategy();
265             if (strategy == Strategy.AUTO) {
266                 strategy = speed >= 30 ? Strategy.FAST : Strategy.DETAILED;
267             }
268     
269             SearchResult result = searchByViewport(viewport, tokens, strategy);
270     
271             if (Settings.isDebug()) {
272                 StringBuilder text = new StringBuilder(Formatter.SEPARATOR).append(strategy.getL10n()).append(Formatter.SEPARATOR).append(Units.getSpeed(speed));
273                 result.setUrl(result.getUrl() + text);
274             }
275     
276             return result;
277         }

279         /**
280          * Searches the view port on the live map for caches.
281          * The strategy dictates if only live map information is used or if an additional
282          * searchByCoordinates query is issued.
283          *
284          * @param viewport
285          *            Area to search
286          * @param tokens
287          *            Live map tokens
288          * @param strategy
289          *            Strategy for data retrieval and parsing, @see Strategy
290          * @return
291          */
292         private static SearchResult searchByViewport(final Viewport viewport, final MapTokens tokens, Strategy strategy) {
293             Log.d("GCMap.searchByViewport" + viewport.toString());
294     
295             final SearchResult searchResult = new SearchResult();
296     
297             if (Settings.isDebug()) {
298                 searchResult.setUrl(viewport.getCenter().format(Format.LAT_LON_DECMINUTE));
299             }
300     
301             if (strategy.flags.contains(StrategyFlag.LOAD_TILES)) {
302                 final Set<Tile> tiles = Tile.getTilesForViewport(viewport);
303     
304                 if (Settings.isDebug()) {
305                     searchResult.setUrl(new StringBuilder().append(tiles.iterator().next().getZoomLevel()).append(Formatter.SEPARATOR).append(searchResult.getUrl()).toString());
306                 }
307     
308                 for (Tile tile : tiles) {
309     
310                     if (!Tile.cache.contains(tile)) {
311                         final Parameters params = new Parameters(
312                                 "x", String.valueOf(tile.getX()),
313                                 "y", String.valueOf(tile.getY()),
314                                 "z", String.valueOf(tile.getZoomLevel()),
315                                 "ep", "1",
316                                 "app", "cgeo");
317                         if (tokens != null) {
318                             params.put("k", tokens.getUserSession(), "st", tokens.getSessionToken());
319                         }
320                         if (Settings.isExcludeMyCaches()) { // works only for PM
321                             params.put("hf", "1", "hh", "1"); // hide found, hide hidden
322                         }
323                         // ect: exclude cache type (probably), comma separated list
324                         if (Settings.getCacheType() != CacheType.ALL) {
325                             params.put("ect", getCacheTypeFilter(Settings.getCacheType()));
326                         }
327                         if (tile.getZoomLevel() != 14) {
328                             params.put("_", String.valueOf(System.currentTimeMillis()));
329                         }
330     
331                         // The PNG must be requested first, otherwise the following request would always return with 204 - No Content
332                         Bitmap bitmap = Tile.requestMapTile(params);
333     
334                         // Check bitmap size
335                         if (bitmap != null && (bitmap.getWidth() != Tile.TILE_SIZE ||
336                                 bitmap.getHeight() != Tile.TILE_SIZE)) {
337                             bitmap.recycle();
338                             bitmap = null;
339                         }
340     
341                         String data = Tile.requestMapInfo(GCConstants.URL_MAP_INFO, params, GCConstants.URL_LIVE_MAP);
342                         if (StringUtils.isEmpty(data)) {
343                             Log.w("GCMap.searchByViewport: No data from server for tile (" + tile.getX() + "/" + tile.getY() + ")");
344                         } else {
345                             final SearchResult search = GCMap.parseMapJSON(data, tile, bitmap, strategy);
346                             if (CollectionUtils.isEmpty(search.getGeocodes())) {
347                                 Log.e("GCMap.searchByViewport: No cache parsed for viewport " + viewport);
348                             }
349                             else {
350                                 searchResult.addSearchResult(search);
351                             }
352                             Tile.cache.add(tile);
353                         }
354     
355                         // release native bitmap memory
356                         if (bitmap != null) {
357                             bitmap.recycle();
358                         }
359     
360                     }
361                 }
362     
363                 // Check for vanished found caches
364                 if (tiles.iterator().next().getZoomLevel() >= Tile.ZOOMLEVEL_MIN_PERSONALIZED) {
365                     searchResult.addFilteredGeocodes(DataStore.getCachedMissingFromSearch(searchResult, tiles, GCConnector.getInstance(), Tile.ZOOMLEVEL_MIN_PERSONALIZED - 1));
366                 }
367             }
368     
369             if (strategy.flags.contains(StrategyFlag.SEARCH_NEARBY) && Settings.isGCPremiumMember()) {
370                 final Geopoint center = viewport.getCenter();
371                 if ((lastSearchViewport == null) || !lastSearchViewport.contains(center)) {
372                     //FIXME We don't have a RecaptchaReceiver!?
373                     SearchResult search = GCParser.searchByCoords(center, Settings.getCacheType(), false, null);
374                     if (search != null && !search.isEmpty()) {
375                         final Set<String> geocodes = search.getGeocodes();
376                         lastSearchViewport = DataStore.getBounds(geocodes);
377                         searchResult.addGeocodes(geocodes);
378                     }
379                 }
380             }
381     
382             return searchResult;
383         }

385         /**
386          * Creates a list of caches types to filter on the live map (exclusion string)
387          * 
388          * @param typeToDisplay
389          *            - cache type to omit from exclusion list so it gets displayed
390          * @return
391          * 
392          *         cache types for live map filter:
393          *         2 = traditional, 9 = ape, 5 = letterbox
394          *         3 = multi
395          *         6 = event, 453 = mega, 13 = cito, 1304 = gps adventures
396          *         4 = virtual, 11 = webcam, 137 = earth
397          *         8 = mystery, 1858 = whereigo
398          */
399         private static String getCacheTypeFilter(CacheType typeToDisplay) {
400             Set<String> filterTypes = new HashSet<String>();
401             // Put all types in set, remove what should be visible in a second step
402             filterTypes.addAll(Arrays.asList("2", "9", "5", "3", "6", "453", "13", "1304", "4", "11", "137", "8", "1858"));
403             switch (typeToDisplay) {
404                 case TRADITIONAL:
405                     filterTypes.remove("2");
406                     break;
407                 case PROJECT_APE:
408                     filterTypes.remove("9");
409                     break;
410                 case LETTERBOX:
411                     filterTypes.remove("5");
412                     break;
413                 case MULTI:
414                     filterTypes.remove("3");
415                     break;
416                 case EVENT:
417                     filterTypes.remove("6");
418                     break;
419                 case MEGA_EVENT:
420                     filterTypes.remove("453");
421                     break;
422                 case CITO:
423                     filterTypes.remove("13");
424                     break;
425                 case GPS_EXHIBIT:
426                     filterTypes.remove("1304");
427                     break;
428                 case VIRTUAL:
429                     filterTypes.remove("4");
430                     break;
431                 case WEBCAM:
432                     filterTypes.remove("11");
433                     break;
434                 case EARTH:
435                     filterTypes.remove("137");
436                     break;
437                 case MYSTERY:
438                     filterTypes.remove("8");
439                     break;
440                 case WHERIGO:
441                     filterTypes.remove("1858");
442                     break;
443                 default:
444                     // nothing to remove otherwise
445             }
446     
447             return StringUtils.join(filterTypes.toArray(), ",");
448         }
