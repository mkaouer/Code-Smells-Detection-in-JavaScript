60                     alert.setTitle(R.string.init_signature_template_button);
61                     final ArrayList<LogTemplate> templates = LogTemplateProvider.getTemplates();
62                     String[] items = new String[templates.size()];
63                     for (int i = 0; i < templates.size(); i++) {
64                         items[i] = settingsActivity.getResources().getString(templates.get(i).getResourceId());

70                             dialog.dismiss();

76                 }
77             });
