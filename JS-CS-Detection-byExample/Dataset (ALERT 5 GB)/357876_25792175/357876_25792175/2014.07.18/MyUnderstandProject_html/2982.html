92                 @Override
93                 public void call(final Scheduler.Inner inner) {
94                     final float newAzimuthShown = smoothUpdate(northMeasured, azimuthShown);
95                     final float newCacheHeadingShown = smoothUpdate(cacheHeadingMeasured, cacheHeadingShown);
96                     if (Math.abs(AngleUtils.difference(azimuthShown, newAzimuthShown)) >= 2 ||
97                             Math.abs(AngleUtils.difference(cacheHeadingShown, newCacheHeadingShown)) >= 2) {

128         /**
129          * Update north and cache headings. This method may only be called on the UI thread.
130          *
131          * @param northHeading the north direction (rotation of the rose)
132          * @param cacheHeading the cache direction (extra rotation of the needle)
133          */
134         public void updateNorth(final float northHeading, final float cacheHeading) {
135             if (initialDisplay) {
136                 // We will force the compass to move brutally if this is the first
137                 // update since it is visible.
138                 azimuthShown = northHeading;
139                 cacheHeadingShown = cacheHeading;
140     
141                 // it may take some time to get an initial direction measurement for the device
142                 if (northHeading != 0.0) {
143                     initialDisplay = false;
144                 }
145             }
146             northMeasured = northHeading;
147             cacheHeadingMeasured = cacheHeading;
148         }

150         /**
151          * Compute the new value, moving by small increments.
152          *
153          * @param goal
154          *            the goal to reach
155          * @param actual
156          *            the actual value
157          * @return the new value
158          */
159         static protected float smoothUpdate(float goal, float actual) {
160             final double diff = AngleUtils.difference(actual, goal);
161     
162             double offset = 0;
163     
164             // If the difference is smaller than 1 degree, do nothing as it
165             // causes the arrow to vibrate. Round away from 0.
166             if (diff > 1.0) {
167                 offset = Math.ceil(diff / 10.0); // for larger angles, rotate faster
168             } else if (diff < 1.0) {
169                 offset = Math.floor(diff / 10.0);
170             }
171     
172             return AngleUtils.normalize((float) (actual + offset));
173         }

175         @Override
176         protected void onDraw(Canvas canvas) {
177     
178             final float azimuthTemp = azimuthShown;
179             final float azimuthRelative = AngleUtils.normalize(azimuthTemp - cacheHeadingShown);
180     
181             // compass margins
182             final int canvasCenterX = (compassRoseWidth / 2) + ((getWidth() - compassRoseWidth) / 2);
183             final int canvasCenterY = (compassRoseHeight / 2) + ((getHeight() - compassRoseHeight) / 2);
184     
185             super.onDraw(canvas);
186     
187             canvas.save();
188             canvas.setDrawFilter(setfil);
189     
190             int marginLeftTemp = (getWidth() - compassUnderlayWidth) / 2;
191             int marginTopTemp = (getHeight() - compassUnderlayHeight) / 2;
192     
193             canvas.drawBitmap(compassUnderlay, marginLeftTemp, marginTopTemp, null);
194     
195             marginLeftTemp = (getWidth() - compassRoseWidth) / 2;
196             marginTopTemp = (getHeight() - compassRoseHeight) / 2;
197     
198             canvas.save();
199             canvas.rotate(-azimuthTemp, canvasCenterX, canvasCenterY);
200             canvas.drawBitmap(compassRose, marginLeftTemp, marginTopTemp, null);
201             canvas.restore();
202     
203             marginLeftTemp = (getWidth() - compassArrowWidth) / 2;
204             marginTopTemp = (getHeight() - compassArrowHeight) / 2;
205     
206             canvas.save();
207             canvas.rotate(-azimuthRelative, canvasCenterX, canvasCenterY);
208             canvas.drawBitmap(compassArrow, marginLeftTemp, marginTopTemp, null);
209             canvas.restore();
210     
211             marginLeftTemp = (getWidth() - compassOverlayWidth) / 2;
212             marginTopTemp = (getHeight() - compassOverlayHeight) / 2;
213     
214             canvas.drawBitmap(compassOverlay, marginLeftTemp, marginTopTemp, null);
215     
216             canvas.setDrawFilter(remfil);
217             canvas.restore();
218         }

225         private int measureWidth(int measureSpec) {
226             final int specMode = MeasureSpec.getMode(measureSpec);
227             final int specSize = MeasureSpec.getSize(measureSpec);
228     
229             if (specMode == MeasureSpec.EXACTLY) {
230                 return specSize;
231             }
232     
233             final int desired = compassArrow.getWidth() + getPaddingLeft() + getPaddingRight();
234             if (specMode == MeasureSpec.AT_MOST) {
235                 return Math.min(desired, specSize);
236             }
237     
238             return desired;
239         }
240     
241         private int measureHeight(int measureSpec) {
242             // The duplicated code in measureHeight and measureWidth cannot be avoided.
243             // Those methods must be efficient, therefore we cannot extract the code differences and unify the remainder.
244             final int specMode = MeasureSpec.getMode(measureSpec);
245             final int specSize = MeasureSpec.getSize(measureSpec);
246     
247             if (specMode == MeasureSpec.EXACTLY) {
248                 return specSize;
249             }
250     
251             final int desired = compassArrow.getHeight() + getPaddingTop() + getPaddingBottom();
252             if (specMode == MeasureSpec.AT_MOST) {
253                 return Math.min(desired, specSize);
254             }
255     
256             return desired;
