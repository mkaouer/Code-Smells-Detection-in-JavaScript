226         private class ButtonClickListener implements View.OnClickListener {
227     
228             @Override
229             public void onClick(View view) {
230                 assert view instanceof Button;
231                 final Button button = (Button) view;
232                 final CharSequence text = button.getText();
233                 if (StringUtils.isBlank(text)) {
234                     return;
235                 }
236                 switch (text.charAt(0)) {
237                     case 'N':
238                         button.setText("S");
239                         break;
240                     case 'S':
241                         button.setText("N");
242                         break;
243                     case 'E':
244                         button.setText("W");
245                         break;
246                     case 'W':
247                         button.setText("E");
248                         break;
249                     default:
250                         break;
251                 }
252     
253                 // This will serve as a reminder to the user that the current coordinates might not be valid
254                 areCurrentCoordinatesValid(true);
255             }
256         }

258         private class TextChanged implements TextWatcher {
259     
260             private final EditText editText;
261     
262             public TextChanged(EditText editText) {
263                 this.editText = editText;
264             }
265     
266             @Override
267             public void afterTextChanged(Editable s) {
268                 /*
269                  * Max lengths, depending on currentFormat
270                  *
271                  * formatPlain = disabled
272                  * DEG MIN SEC SUB
273                  * formatDeg 2/3 5 - -
274                  * formatMin 2/3 2 3 -
275                  * formatSec 2/3 2 2 3
276                  */
277     
278                 if (currentFormat == CoordInputFormatEnum.Plain) {
279                     return;
280                 }
281     
282                 final int maxLength = getMaxLengthFromCurrentField(editText);
283                 if (s.length() == maxLength) {
284                     if (editText == eLatDeg) {
285                         eLatMin.requestFocus();
286                     } else if (editText == eLatMin) {
287                         if (eLatSec.getVisibility() == View.GONE) {
288                             eLonDeg.requestFocus();
289                         } else {
290                             eLatSec.requestFocus();
291                         }
292                     } else if (editText == eLatSec) {
293                         if (eLatSub.getVisibility() == View.GONE) {
294                             eLonDeg.requestFocus();
295                         } else {
296                             eLatSub.requestFocus();
297                         }
298                     } else if (editText == eLatSub) {
299                         eLonDeg.requestFocus();
300                     } else if (editText == eLonDeg) {
301                         eLonMin.requestFocus();
302                     } else if (editText == eLonMin) {
303                         if (eLonSec.getVisibility() == View.GONE) {
304                             eLatDeg.requestFocus();
305                         } else {
306                             eLonSec.requestFocus();
307                         }
308                     } else if (editText == eLonSec) {
309                         if (eLonSub.getVisibility() == View.GONE) {
310                             eLatDeg.requestFocus();
311                         } else {
312                             eLonSub.requestFocus();
313                         }
314                     } else if (editText == eLonSub) {
315                         eLatDeg.requestFocus();
316                     }
317                 }
318             }

321             public void beforeTextChanged(CharSequence s, int start, int count, int after) {
322             }
323     
324             @Override

326             }

330         private boolean areCurrentCoordinatesValid(final boolean signalError) {
331             try {
332                 Geopoint current = null;
333                 if (currentFormat == CoordInputFormatEnum.Plain) {
334                     current = new Geopoint(eLat.getText().toString(), eLon.getText().toString());
335                 } else {
336                     final String latDir = bLat.getText().toString();
337                     final String lonDir = bLon.getText().toString();
338                     final String latDeg = eLatDeg.getText().toString();
339                     final String lonDeg = eLonDeg.getText().toString();
340                     final String latDegFrac = eLatMin.getText().toString();
341                     final String lonDegFrac = eLonMin.getText().toString();
342                     final String latMin = eLatMin.getText().toString();
343                     final String lonMin = eLonMin.getText().toString();
344                     final String latMinFrac = eLatSec.getText().toString();
345                     final String lonMinFrac = eLonSec.getText().toString();
346                     final String latSec = eLatSec.getText().toString();
347                     final String lonSec = eLonSec.getText().toString();
348                     final String latSecFrac = eLatSub.getText().toString();
349                     final String lonSecFrac = eLonSub.getText().toString();
350     
351                     switch (currentFormat) {
352                         case Deg:
353                             current = new Geopoint(latDir, latDeg, latDegFrac, lonDir, lonDeg, lonDegFrac);
354                             break;
355                         case Min:
356                             current = new Geopoint(latDir, latDeg, latMin, latMinFrac, lonDir, lonDeg, lonMin, lonMinFrac);
357                             break;
358                         case Sec:
359                             current = new Geopoint(latDir, latDeg, latMin, latSec, latSecFrac, lonDir, lonDeg, lonMin, lonSec, lonSecFrac);
360                             break;
361                         case Plain:
362                             // This case has been handled above
363                     }
364                 }
365                 // The null check is necessary to keep FindBugs happy
366                 if (current != null && current.isValid()) {
367                     gp = current;
368                     return true;
369                 }
370             } catch (final Geopoint.ParseException e) {
371                 // Signaled and returned below
372             }
373             if (signalError) {
374                 context.showToast(context.getResources().getString(R.string.err_parse_lat_lon));
375             }
376             return false;
377         }

379         public int getMaxLengthFromCurrentField(final EditText editText) {
380             if (editText == eLonDeg || editText == eLatSub || editText == eLonSub) {
381                 return 3;
382             }
383             if ((editText == eLatMin || editText == eLonMin) && currentFormat == CoordInputFormatEnum.Deg) {
384                 return 5;
385             }
386             if ((editText == eLatSec || editText == eLonSec) && currentFormat == CoordInputFormatEnum.Min) {
387                 return 3;
388             }
389             return 2;
390         }

392         private class CoordinateFormatListener implements OnItemSelectedListener {
393     
394             @Override
395             public void onItemSelected(AdapterView<?> parent, View view, int pos, long id) {
396                 // Ignore first call, which comes from onCreate()
397                 if (currentFormat != null) {
398     
399                     // Start new format with an acceptable value: either the current one
400                     // entered by the user, else our current coordinates, else (0,0).
401                     if (!areCurrentCoordinatesValid(false)) {
402                         if (geo != null && geo.getCoords() != null) {
403                             gp = geo.getCoords();
404                         } else {
405                             gp = Geopoint.ZERO;
406                         }
407                     }
408                 }
409     
410                 currentFormat = CoordInputFormatEnum.fromInt(pos);
411                 Settings.setCoordInputFormat(currentFormat);
412                 updateGUI();
413             }
414     
415             @Override
416             public void onNothingSelected(AdapterView<?> arg0) {
417             }
418     
419         }
