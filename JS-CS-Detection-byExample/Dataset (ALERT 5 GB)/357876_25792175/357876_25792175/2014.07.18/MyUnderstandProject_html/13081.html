3     import java.util.LinkedHashMap;

6     /**
7      * Base class for caching objects. Don't mix up with a geocache !
8      *
9      * The LeastRecentlyUsedMap is basically a LinkedHashMap which can be configured to have certain modes of operation:
10      * <ul>
11      * <li> LRU_CACHE means that the elements are updated in the LinkedList on every get() access,
12      *      so the objects that are dropped are the ones that haven't been used the longest</li>
13      * <li> BOUNDED means that objects are updated only when they are put,
14      *      so the objects that are dropped are the ones that haven't been written the longest</li>
15      * </ul>
16      */
17     public abstract class LeastRecentlyUsedMap<K, V> extends LinkedHashMap<K, V> {
18     
19         private enum OperationModes {
20             LRU_CACHE, BOUNDED
21         }
22     
23         private static final long serialVersionUID = -5077882607489806620L;
24     
25         private final int maxEntries;
26         private final OperationModes opMode;
27         private RemoveHandler<V> removeHandler;
28     
29         // store the HashMap parameters for serialization, as we can't access the originals in the LinkedHashMap
30         final int initialCapacity;
31         final float loadFactor;
32     
33         protected LeastRecentlyUsedMap(int maxEntries, int initialCapacity, float loadFactor, OperationModes opMode) {
34         	super(initialCapacity, loadFactor, (opMode==OperationModes.LRU_CACHE));
35         	this.initialCapacity = initialCapacity;
36         	this.loadFactor = loadFactor;
37             this.maxEntries = maxEntries;
38             this.opMode = opMode;
39         }
40     
41         protected LeastRecentlyUsedMap(int maxEntries, OperationModes opMode) {
42         	this(maxEntries, 16, 0.75f, opMode);
43         }
44     
45         @Override
46         public V put(K key, V value) {
47         	// in case the underlying Map is not running with accessOrder==true, the map won't notice any changes
48         	// of existing keys, so for the normal BOUNDED mode we remove and put the value to get its order updated.
49         	if (opMode == OperationModes.BOUNDED && containsKey(key)) {
50                 // avoid trigger the remove notification
51                 final V oldVal = super.remove(key);
52         		put(key, value);
53         		return oldVal;
54         	}
55     
56             return super.put(key, value);
57         }
58     
59         @Override
60         protected boolean removeEldestEntry(Map.Entry<K, V> eldest) {
61             return size() > maxEntries;
62         }
63     
64         public int getMaxEntries() {
65     		return maxEntries;
66     	}
67     
68         @Override
69         public V remove(Object key) {
70     
71             V removed = super.remove(key);
72     
73             if (removed != null && removeHandler != null) {
74                 removeHandler.onRemove(removed);
75             }
76     
77             return removed;
78         }
79     
80         /**
81          * Sets a handler for remove notifications. Currently only one handler
82          * instance is supported
83          *
84          * @param removeHandler
85          *            The new handler to receive notifications or null to remove a handler
86          */
87         public void setRemoveHandler(RemoveHandler<V> removeHandler) {
88             this.removeHandler = removeHandler;
89         }

95                 super(maxEntries, initialCapacity, loadFactor, OperationModes.LRU_CACHE);
96             }
97     
98             public LruCache(int maxEntries) {
99                 super(maxEntries, OperationModes.LRU_CACHE);
100             }
101         }
102     
103         public static class Bounded<K, V> extends LeastRecentlyUsedMap<K, V> {
104     
105             private static final long serialVersionUID = -1476389304214398315L;
106     
107             public Bounded(int maxEntries, int initialCapacity, float loadFactor) {
108                 super(maxEntries, initialCapacity, loadFactor, OperationModes.BOUNDED);
109             }
110     
111             public Bounded(int maxEntries) {
112                 super(maxEntries, OperationModes.BOUNDED);
113             }
114         }

116         /**
117          * Interface for handlers that wish to get notified when items are
118          * removed from the LRUMap
119          *
120          * @param <V>
121          */
122         public interface RemoveHandler<V> {
123     
124             /**
125              * Method will be called on remove
126              *
127              * @param removed
128              *            Item that has been removed
129              */
130             void onRemove(V removed);
131     
132         }
