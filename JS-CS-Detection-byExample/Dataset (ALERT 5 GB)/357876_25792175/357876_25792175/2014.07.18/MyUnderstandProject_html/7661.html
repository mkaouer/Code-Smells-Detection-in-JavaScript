23         /**
24          * remember the last state of drawing so we can avoid repainting for very small changes

26         private float lastDrawnAzimuth;

28         /**
29          * bitmap shared by all instances of the view

33          * bitmap width

41          * view instance counter for bitmap recycling

43         private static int instances = 0;

45         /**
46          * pixel size of the square arrow bitmap
47          */
48         private static final int ARROW_BITMAP_SIZE = 21;
49         private static final PaintFlagsDrawFilter FILTER_SET = new PaintFlagsDrawFilter(0, Paint.FILTER_BITMAP_FLAG);
50         private static final float MINIMUM_ROTATION_DEGREES_FOR_REPAINT = 5;

61         @Override
62         public void onAttachedToWindow() {
63             if (instances++ == 0) {
64                 compassArrow = BitmapFactory.decodeResource(getResources(), Settings.isLightSkin() ? R.drawable.compass_arrow_mini_black : R.drawable.compass_arrow_mini_white);
65                 compassArrowWidth = compassArrow.getWidth();
66                 compassArrowHeight = compassArrow.getWidth();
67             }
68         }

71         public void onDetachedFromWindow() {
72             super.onDetachedFromWindow();
73             if (--instances == 0) {
74                 compassArrow.recycle();
75             }
76         }

85         }

90         }

92         public void updateCurrentCoords(@NonNull final Geopoint currentCoords) {
93             if (targetCoords == null) {
94                 return;
95             }
96     
97             heading = currentCoords.bearingTo(targetCoords);
98             updateDirection();
99         }
100     
101         private void updateDirection() {
102             if (compassArrow == null || compassArrow.isRecycled()) {
103                 return;
104             }
105     
106             azimuthRelative = AngleUtils.normalize(azimuth - heading);
107     
108             // avoid updates on very small changes, which are not visible to the user
109             float change = Math.abs(azimuthRelative - lastDrawnAzimuth);
110             if (change < MINIMUM_ROTATION_DEGREES_FOR_REPAINT) {
111                 return;
112             }
113     
114             // compass margins
115             final int marginLeft = (getWidth() - compassArrowWidth) / 2;
116             final int marginTop = (getHeight() - compassArrowHeight) / 2;
117     
118             invalidate(marginLeft, marginTop, (marginLeft + compassArrowWidth), (marginTop + compassArrowHeight));
119         }
120     
121         @Override
122         protected void onDraw(Canvas canvas) {
123             super.onDraw(canvas);
124     
125             lastDrawnAzimuth = azimuthRelative;
126     
127             // compass margins
128             final int canvasCenterX = getWidth() / 2;
129             final int canvasCenterY = getHeight() / 2;
130     
131             final int marginLeft = (getWidth() - compassArrowWidth) / 2;
132             final int marginTop = (getHeight() - compassArrowHeight) / 2;
133     
134             canvas.setDrawFilter(FILTER_SET);
135             canvas.rotate(-azimuthRelative, canvasCenterX, canvasCenterY);
136             canvas.drawBitmap(compassArrow, marginLeft, marginTop, null);
137         }

139         @Override
140         protected void onMeasure(int widthMeasureSpec, int heightMeasureSpec) {
141             setMeasuredDimension(measureWidth(widthMeasureSpec), measureHeight(heightMeasureSpec));
142         }

144         private int measureWidth(int measureSpec) {
145             final int specMode = MeasureSpec.getMode(measureSpec);
146             final int specSize = MeasureSpec.getSize(measureSpec);
147     
148             if (specMode == MeasureSpec.EXACTLY) {
149                 return specSize;
150             }
151     
152             int result = ARROW_BITMAP_SIZE + getPaddingLeft() + getPaddingRight();
153     
154             if (specMode == MeasureSpec.AT_MOST) {
155                 result = Math.min(result, specSize);
156             }
157     
158             return result;
159         }
160     
161         private int measureHeight(int measureSpec) {
162             final int specMode = MeasureSpec.getMode(measureSpec);
163             final int specSize = MeasureSpec.getSize(measureSpec);
164     
165             if (specMode == MeasureSpec.EXACTLY) {
166                 return specSize;
167             }
168     
169             int result = ARROW_BITMAP_SIZE + getPaddingTop() + getPaddingBottom();
170     
171             if (specMode == MeasureSpec.AT_MOST) {
172                 result = Math.min(result, specSize);
173             }
174     
175             return result;
176         }
