91                 info = extras.getString(EXTRAS_CACHE_INFO);
92     
93                 if (StringUtils.isNotBlank(name)) {
94                     if (StringUtils.isNotBlank(title)) {
95                         title += ": " + name;
96                     } else {
97                         title = name;
98                     }
99                 }
100             } else {
101                 Intent pointIntent = new Intent(this, NavigateAnyPointActivity.class);
102                 startActivity(pointIntent);

104                 finish();

110             setDestCoords();

178         @Override
179         public boolean onOptionsItemSelected(MenuItem item) {
180             int id = item.getItemId();
181             switch (id) {
182                 case R.id.menu_map:
183                     CGeoMap.startActivityCoords(this, dstCoords, null, null);
184                     return true;
185                 case R.id.menu_switch_compass_gps:
186                     boolean oldSetting = Settings.isUseCompass();
187                     Settings.setUseCompass(!oldSetting);
188                     invalidateOptionsMenuCompatible();
189                     return true;
190                 case R.id.menu_edit_destination:
191                     Intent pointIntent = new Intent(this, NavigateAnyPointActivity.class);
192                     startActivity(pointIntent);
193     
194                     finish();
195                     return true;
196                 case R.id.menu_tts_start:
197                     SpeechService.startService(this, dstCoords);
198                     return true;
199                 case R.id.menu_tts_stop:
200                     SpeechService.stopService(this);
201                     return true;
202                 default:
203                     if (LoggingUI.onMenuItemSelected(item, this, cache)) {
204                         return true;
205                     }
206                     int coordinatesIndex = id - COORDINATES_OFFSET;
207                     if (coordinatesIndex >= 0 && coordinatesIndex < coordinates.size()) {
208                         final IWaypoint coordinate = coordinates.get(coordinatesIndex);
209                         title = coordinate.getName();
210                         dstCoords = coordinate.getCoords();
211                         setTitle();
212                         setDestCoords();
213                         setCacheInfo();
214                         updateDistanceInfo(app.currentGeo());
215     
216                         Log.d("destination set: " + title + " (" + dstCoords + ")");
217                         return true;
218                     }
219             }
220             return false;
221         }

248         private void updateDistanceInfo(final IGeoData geo) {
249             if (geo.getCoords() == null || dstCoords == null) {
250                 return;
251             }
252     
253             cacheHeading = geo.getCoords().bearingTo(dstCoords);
254             distanceView.setText(Units.getDistanceFromKilometers(geo.getCoords().distanceTo(dstCoords)));
255             headingView.setText(Math.round(cacheHeading) + "");
256         }

258         private GeoDirHandler geoDirHandler = new GeoDirHandler() {
259             @Override
260             public void updateGeoDir(final IGeoData geo, final float dir) {
261                 try {
262                     if (geo.getCoords() != null) {
263                         if (geo.getSatellitesVisible() >= 0) {
264                             navSatellites.setText(res.getString(R.string.loc_sat) + ": " + geo.getSatellitesFixed() + "/" + geo.getSatellitesVisible());
265                         } else {
266                             navSatellites.setText("");
267                         }
268                         navType.setText(res.getString(geo.getLocationProvider().resourceId));
269     
270                         if (geo.getAccuracy() >= 0) {
271                             navAccuracy.setText("" + Units.getDistanceFromMeters(geo.getAccuracy()));
272                         } else {
273                             navAccuracy.setText(null);
274                         }
275     
276                         navLocation.setText(geo.getCoords().toString());
277     
278                         updateDistanceInfo(geo);
279                     } else {
280                         navType.setText(null);
281                         navAccuracy.setText(null);
282                         navLocation.setText(res.getString(R.string.loc_trying));
283                     }
284     
285                     updateNorthHeading(DirectionProvider.getDirectionNow(dir));
286                 } catch (RuntimeException e) {
287                     Log.w("Failed to LocationUpdater location.");
288                 }
289             }
290         };

292         private void updateNorthHeading(final float northHeading) {
293             if (compassView != null) {
294                 compassView.updateNorth(northHeading, cacheHeading);
295             }
296         }

298         public static void startActivity(final Context context, final String geocode, final String displayedName, final Geopoint coords, final Collection<IWaypoint> coordinatesWithType,
299                 final String info) {
300             coordinates.clear();
301             if (coordinatesWithType != null) {
302                 for (IWaypoint coordinate : coordinatesWithType) {
303                     if (coordinate != null) {
304                         coordinates.add(coordinate);
305                     }
306                 }
307             }
308     
309             final Intent navigateIntent = new Intent(context, CompassActivity.class);
310             navigateIntent.putExtra(EXTRAS_COORDS, coords);
311             navigateIntent.putExtra(EXTRAS_GEOCODE, geocode);
312             if (null != displayedName) {
313                 navigateIntent.putExtra(EXTRAS_NAME, displayedName);
314             }
315             navigateIntent.putExtra(EXTRAS_CACHE_INFO, info);
316             context.startActivity(navigateIntent);
317         }
