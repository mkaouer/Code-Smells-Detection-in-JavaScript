94         private static List<String> measure(int iterations, String fieldName, Pattern p1, Pattern p2) {
95     
96             final List<String> output = new ArrayList<String>();
97             output.add(fieldName + ":");
98     
99             for (MockedCache cache : MOCKED_CACHES) {
100                 String page = cache.getData();
101                 String result1 = TextUtils.getMatch(page, p1, true, "");
102                 String result2 = TextUtils.getMatch(page, p2, true, "");
103                 assertThat(result2).isEqualTo(result1);
104     
105                 long diff1, diff2;
106     
107                 output.add("Parsing " + cache.getGeocode() + " " + cache.getName());
108     
109                 diff1 = parse(page, p1, iterations);
110                 output.add("Time pattern 1:\t" + diff1 + " ms");
111     
112                 diff2 = parse(page, p2, iterations);
113                 output.add("Time pattern 2:\t" + diff2 + " ms");
114     
115                 float reduction = (float) diff2 * 100 / diff1;
116                 output.add("New runtime:\t" + String.format("%.1f", reduction) + "%\n");
117             }
118     
119             return output;
120     
121         }

123         private static long parse(String page, Pattern pattern, int iterations) {
124             final long start = System.currentTimeMillis();
125             for (int j = 0; j < iterations; j++) {
126                 TextUtils.getMatch(page, pattern, true, "");
127             }
128             return System.currentTimeMillis() - start;
