34         public static boolean isGoogleMapsInstalled() {
35             // Check if API key is available
36             if (StringUtils.isBlank(CgeoApplication.getInstance().getString(R.string.maps_api_key))) {
37                 Log.w("No Google API key available.");
38                 return false;
39             }
40     
41             // Check if API is available
42             try {
43                 Class.forName("com.google.android.maps.MapActivity");
44             } catch (ClassNotFoundException e) {
45                 return false;
46             }
47     
48             // Assume that Google Maps is available and working
49             return true;
50         }

56         public static boolean isSameActivity(@NonNull final MapSource source1, @NonNull final MapSource source2) {
57             final MapProvider provider1 = source1.getMapProvider();
58             final MapProvider provider2 = source2.getMapProvider();
59             return provider1 == provider2 && provider1.isSameActivity(source1, source2);

62         public static void addMapviewMenuItems(Menu menu) {
63             final SubMenu parentMenu = menu.findItem(R.id.menu_select_mapview).getSubMenu();
64     
65             final int currentSource = Settings.getMapSource().getNumericalId();
66             for (int i = 0; i < mapSources.size(); i++) {
67                 final MapSource mapSource = mapSources.get(i);
68                 final int id = mapSource.getNumericalId();
69                 parentMenu.add(R.id.menu_group_map_sources, id, i, mapSource.getName()).setCheckable(true).setChecked(id == currentSource);
70             }
71             parentMenu.setGroupCheckable(R.id.menu_group_map_sources, true, true);
72         }

74         /**
75          * Return a map source by id.
76          *
77          * @param id the map source id
78          * @return the map source, or <tt>null</tt> if <tt>id</tt> does not correspond to a registered map source

80         @Nullable
81         public static MapSource getMapSource(int id) {
82             for (MapSource mapSource : mapSources) {
83                 if (mapSource.getNumericalId() == id) {
84                     return mapSource;

86             }

88         }

90         /**

92          *
93          * @return the first map source in the collection, or <tt>null</tt> if there are none registered
94          */
95         public static MapSource getAnyMapSource() {
96             return mapSources.isEmpty() ? null : mapSources.get(0);

102

106     
107         /**
108          * remove offline map sources after changes of the settings
109          */
110         public static void deleteOfflineMapSources() {
111             final ArrayList<MapSource> deletion = new ArrayList<MapSource>();
112             for (MapSource mapSource : mapSources) {
113                 if (mapSource instanceof MapsforgeMapProvider.OfflineMapSource) {
114                     deletion.add(mapSource);
115                 }
116             }
117             mapSources.removeAll(deletion);
