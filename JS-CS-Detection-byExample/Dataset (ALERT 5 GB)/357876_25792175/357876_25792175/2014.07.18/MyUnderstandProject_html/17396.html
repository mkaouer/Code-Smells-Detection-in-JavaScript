41         final private static Map<String, CacheSize> FIND_BY_ID;
42         static {
43             final HashMap<String, CacheSize> mapping = new HashMap<String, CacheSize>();
44             for (final CacheSize cs : values()) {
45                 mapping.put(cs.id.toLowerCase(Locale.US), cs);
46                 mapping.put(cs.ocSize2.toLowerCase(Locale.US), cs);
47             }
48             // add medium as additional string for Regular
49             mapping.put("medium", CacheSize.REGULAR);
50             FIND_BY_ID = Collections.unmodifiableMap(mapping);
51         }

53         public static CacheSize getById(final String id) {
54             if (id == null) {
55                 return UNKNOWN;
56             }
57             // avoid String operations for performance reasons
58             final CacheSize result = CacheSize.FIND_BY_ID.get(id);
59             if (result != null) {
60                 return result;
61             }
62             // only if String was not found, normalize it
63             final CacheSize resultNormalized = CacheSize.FIND_BY_ID.get(id.toLowerCase(Locale.US).trim());
64             if (resultNormalized != null) {
65                 return resultNormalized;
66             }
67             return getByNumber(id);
68         }

70         /**
71          * Bad GPX files can contain the container size encoded as number.
72          *
73          * @param id
74          * @return
75          */
76         private static CacheSize getByNumber(final String id) {
77             try {
78                 final int numerical = Integer.parseInt(id);
79                 if (numerical != 0) {
80                     for (final CacheSize size : CacheSize.values()) {
81                         if (size.comparable == numerical) {
82                             return size;
83                         }
84                     }
85                 }
86             } catch (final NumberFormatException e) {
87                 // ignore, as this might be a number or not
88             }
89             return UNKNOWN;
90         }
