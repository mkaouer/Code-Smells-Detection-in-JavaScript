80         private void createStarImages(final ViewGroup starsContainer, final float value, final int max) {
81             final LayoutInflater inflater = LayoutInflater.from(activity);
82     
83             for (int i = 0; i < max; i++) {
84                 ImageView star = (ImageView) inflater.inflate(R.layout.star_image, null);
85                 if (value - i >= 0.75) {
86                     star.setImageResource(R.drawable.star_on);
87                 } else if (value - i >= 0.25) {
88                     star.setImageResource(R.drawable.star_half);
89                 } else {
90                     star.setImageResource(R.drawable.star_off);
91                 }
92                 starsContainer.addView(star);
93             }
94         }

96         public void addCacheState(Geocache cache) {
97             if (cache.isLogOffline() || cache.isArchived() || cache.isDisabled() || cache.isPremiumMembersOnly() || cache.isFound()) {
98                 final List<String> states = new ArrayList<String>(5);
99                 if (cache.isLogOffline()) {
100                     states.add(res.getString(R.string.cache_status_offline_log));
101                 }
102                 if (cache.isFound()) {
103                     states.add(res.getString(R.string.cache_status_found));
104                 }
105                 if (cache.isArchived()) {
106                     states.add(res.getString(R.string.cache_status_archived));
107                 }
108                 if (cache.isDisabled()) {
109                     states.add(res.getString(R.string.cache_status_disabled));
110                 }
111                 if (cache.isPremiumMembersOnly()) {
112                     states.add(res.getString(R.string.cache_status_premium));
113                 }
114                 add(R.string.cache_status, StringUtils.join(states, ", "));
115             }
116         }

147         public void addDistance(final Geocache cache, final TextView cacheDistanceView) {
148             Float distance = null;
149             if (cache.getCoords() != null) {
150                 final Geopoint currentCoords = CgeoApplication.getInstance().currentGeo().getCoords();
151                 if (currentCoords != null) {
152                     distance = currentCoords.distanceTo(cache);
153                 }
154             }
155             if (distance == null) {
156                 if (cache.getDistance() != null) {
157                     distance = cache.getDistance();
158                 }
159             }
160             String text = "--";
161             if (distance != null) {
162                 text = Units.getDistanceFromKilometers(distance);
163             }
164             else if (cacheDistanceView != null) {
165                 // if there is already a distance in cacheDistance, use it instead of resetting to default.
166                 // this prevents displaying "--" while waiting for a new position update (See bug #1468)
167                 text = cacheDistanceView.getText().toString();
168             }
169             add(R.string.cache_distance, text);
170         }

199         public TextView addHiddenDate(final @NonNull Geocache cache) {
200             final Date hiddenDate = cache.getHiddenDate();
201             if (hiddenDate == null) {
202                 return null;
203             }
204             final long time = hiddenDate.getTime();
205             if (time > 0) {
206                 String dateString = Formatter.formatFullDate(time);
207                 if (cache.isEventCache()) {
208                     dateString = DateUtils.formatDateTime(CgeoApplication.getInstance().getBaseContext(), time, DateUtils.FORMAT_SHOW_WEEKDAY) + ", " + dateString;
209                 }
210                 final TextView view = add(cache.isEventCache() ? R.string.cache_event : R.string.cache_hidden, dateString);
211                 view.setId(R.id.date);
212                 return view;
213             }
214             return null;
215         }
