227         private int getDefaultWaypointType() {
228             // potentially restore saved instance state
229             if (waypointTypeSelectorPosition >= 0) {
230                 return waypointTypeSelectorPosition;
231             }
232     
233             // when editing, use the existing type
234             if (waypoint != null) {
235                 return POSSIBLE_WAYPOINT_TYPES.indexOf(waypoint.getWaypointType());
236             }
237     
238             // make default for new waypoint depend on cache type
239             switch (cache.getType()) {
240                 case MYSTERY:
241                     return POSSIBLE_WAYPOINT_TYPES.indexOf(WaypointType.FINAL);
242                 case MULTI:
243                     return POSSIBLE_WAYPOINT_TYPES.indexOf(WaypointType.STAGE);
244                 default:
245                     return POSSIBLE_WAYPOINT_TYPES.indexOf(WaypointType.WAYPOINT);
246             }

317         private class SaveWaypointListener implements View.OnClickListener {
318     
319             @Override
320             public void onClick(View arg0) {
321                 final String bearingText = bearing.getText().toString();
322                 // combine distance from EditText and distanceUnit saved from Spinner
323                 final String distanceText = distanceView.getText().toString() + distanceUnits.get(distanceUnitSelector.getSelectedItemPosition());
324                 final String latText = buttonLat.getText().toString();
325                 final String lonText = buttonLon.getText().toString();
326     
327                 if (StringUtils.isBlank(bearingText) && StringUtils.isBlank(distanceText)
328                         && StringUtils.isBlank(latText) && StringUtils.isBlank(lonText)) {
329                     Dialogs.message(EditWaypointActivity.this, R.string.err_point_no_position_given_title, R.string.err_point_no_position_given);
330                     return;
331                 }
332     
333                 Geopoint coords;
334     
335                 if (StringUtils.isNotBlank(latText) && StringUtils.isNotBlank(lonText)) {
336                     try {
337                         coords = new Geopoint(latText, lonText);
338                     } catch (Geopoint.ParseException e) {
339                         showToast(res.getString(e.resource));
340                         return;
341                     }
342                 } else {
343                     final IGeoData geo = app.currentGeo();
344                     if (geo.getCoords() == null) {
345                         showToast(res.getString(R.string.err_point_curr_position_unavailable));
346                         return;
347                     }
348                     coords = geo.getCoords();
349                 }
350     
351                 if (StringUtils.isNotBlank(bearingText) && StringUtils.isNotBlank(distanceText)) {
352                     // bearing & distance
353                     double bearing;
354                     try {
355                         bearing = Double.parseDouble(bearingText);
356                     } catch (NumberFormatException e) {
357                         Dialogs.message(EditWaypointActivity.this, R.string.err_point_bear_and_dist_title, R.string.err_point_bear_and_dist);
358                         return;
359                     }
360     
361                     double distance;
362                     try {
363                         distance = DistanceParser.parseDistance(distanceText,
364                                 !Settings.isUseImperialUnits());
365                     } catch (NumberFormatException e) {
366                         showToast(res.getString(R.string.err_parse_dist));
367                         return;
368                     }
369     
370                     coords = coords.project(bearing, distance);
371                 }
372     
373                 // if no name is given, just give the waypoint its number as name
374                 final String givenName = waypointName.getText().toString().trim();
375                 final String name = StringUtils.isNotEmpty(givenName) ? givenName : res.getString(R.string.waypoint) + " " + (wpCount + 1);
376                 final String noteText = note.getText().toString().trim();
377                 final Geopoint coordsToSave = coords;
378                 final int selectedTypeIndex = waypointTypeSelector.getSelectedItemPosition();
379                 final WaypointType type = selectedTypeIndex >= 0 ? POSSIBLE_WAYPOINT_TYPES.get(selectedTypeIndex) : waypoint.getWaypointType();
380                 final boolean visited = visitedCheckBox.isChecked();
381                 final ProgressDialog progress = ProgressDialog.show(EditWaypointActivity.this, getString(R.string.cache), getString(R.string.waypoint_being_saved), true);
382                 final Handler finishHandler = new Handler() {
383     
384                     @Override
385                     public void handleMessage(Message msg) {
386                         // TODO: The order of showToast, progress.dismiss and finish is different in these cases. Why?
387                         switch (msg.what) {
388                             case UPLOAD_SUCCESS:
389                                 showToast(getString(R.string.waypoint_coordinates_has_been_modified_on_website, coordsToSave));
390                                 progress.dismiss();
391                                 finish();
392                                 break;
393                             case SUCCESS:
394                                 progress.dismiss();
395                                 finish();
396                                 break;
397                             case UPLOAD_START:
398                                 progress.setMessage(getString(R.string.waypoint_coordinates_uploading_to_website, coordsToSave));
399                                 break;
400                             case UPLOAD_ERROR:
401                                 progress.dismiss();
402                                 finish();
403                                 showToast(getString(R.string.waypoint_coordinates_upload_error));
404                                 break;
405                             case UPLOAD_NOT_POSSIBLE:
406                                 progress.dismiss();
407                                 finish();
408                                 showToast(getString(R.string.waypoint_coordinates_couldnt_be_modified_on_website));
409                                 break;
410                             case SAVE_ERROR:
411                                 progress.dismiss();
412                                 finish(); //TODO: should we close activity here ?
413                                 showToast(res.getString(R.string.err_waypoint_add_failed));
414                                 break;
415                             default:
416                                 throw new UnsupportedOperationException();
417                         }
418                     }
419                 };

421                 class SaveWptTask extends AsyncTask<Void, Void, Void> {
422     
423                     @Override
424                     protected Void doInBackground(Void... params) {
425                         final Waypoint waypoint = new Waypoint(name, type, own);
426                         waypoint.setGeocode(geocode);
427                         waypoint.setPrefix(prefix);
428                         waypoint.setLookup(lookup);
429                         waypoint.setCoords(coordsToSave);
430                         waypoint.setNote(noteText);
431                         waypoint.setVisited(visited);
432                         waypoint.setId(id);
433     
434                         Geocache cache = DataStore.loadCache(geocode, LoadFlags.LOAD_WAYPOINTS);
435                         if (cache == null) {
436                             finishHandler.sendEmptyMessage(SAVE_ERROR);
437                             return null;
438                         }
439                         Waypoint oldWaypoint = cache.getWaypointById(id);
440                         if (cache.addOrChangeWaypoint(waypoint, true)) {
441                             DataStore.saveCache(cache, EnumSet.of(SaveFlag.SAVE_DB));
442                             if (!StaticMapsProvider.hasAllStaticMapsForWaypoint(geocode, waypoint)) {
443                                 StaticMapsProvider.removeWpStaticMaps(oldWaypoint, geocode);
444                                 if (Settings.isStoreOfflineWpMaps()) {
445                                     StaticMapsProvider.storeWaypointStaticMap(cache, waypoint, false);
446                                 }
447                             }
448                             if (modifyLocal.isChecked() || modifyBoth.isChecked()) {
449                                 if (!cache.hasUserModifiedCoords()) {
450                                     final Waypoint origWaypoint = new Waypoint(CgeoApplication.getInstance().getString(R.string.cache_coordinates_original), WaypointType.ORIGINAL, false);
451                                     origWaypoint.setCoords(cache.getCoords());
452                                     cache.addOrChangeWaypoint(origWaypoint, false);
453                                     cache.setUserModifiedCoords(true);
454                                 }
455                                 cache.setCoords(waypoint.getCoords());
456                                 DataStore.saveChangedCache(cache);
457                             }
458                             if (modifyBoth.isChecked() && waypoint.getCoords() != null) {
459                                 finishHandler.sendEmptyMessage(UPLOAD_START);
460     
461                                 if (cache.supportsOwnCoordinates()) {
462                                     boolean result = uploadModifiedCoords(cache, waypoint.getCoords());
463                                     finishHandler.sendEmptyMessage(result ? SUCCESS : UPLOAD_ERROR);
464                                 } else {
465                                     showToast(getString(R.string.waypoint_coordinates_couldnt_be_modified_on_website));
466                                     finishHandler.sendEmptyMessage(UPLOAD_NOT_POSSIBLE);
467                                 }
468                             } else {
469                                 finishHandler.sendEmptyMessage(SUCCESS);
470                             }
471                         } else {
472                             finishHandler.sendEmptyMessage(SAVE_ERROR);
473                         }
474                         return null;
475                     }
476                 }
477                 new SaveWptTask().execute();
478             }
479         }
