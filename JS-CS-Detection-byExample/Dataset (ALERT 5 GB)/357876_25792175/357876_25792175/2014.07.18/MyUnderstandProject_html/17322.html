51         @Override
52         protected void addHeaderView() {
53             // adds the log counts
54             final Map<LogType, Integer> logCounts = getCache().getLogCounts();
55             if (logCounts != null) {
56                 final List<Entry<LogType, Integer>> sortedLogCounts = new ArrayList<Entry<LogType, Integer>>(logCounts.size());
57                 for (final Entry<LogType, Integer> entry : logCounts.entrySet()) {
58                     // it may happen that the label is unknown -> then avoid any output for this type
59                     if (entry.getKey() != LogType.PUBLISH_LISTING && entry.getKey().getL10n() != null) {
60                         sortedLogCounts.add(entry);
61                     }
62                 }
63     
64                 if (!sortedLogCounts.isEmpty()) {
65                     // sort the log counts by type id ascending. that way the FOUND, DNF log types are the first and most visible ones
66                     Collections.sort(sortedLogCounts, new Comparator<Entry<LogType, Integer>>() {
67     
68                         @Override
69                         public int compare(Entry<LogType, Integer> logCountItem1, Entry<LogType, Integer> logCountItem2) {
70                             return logCountItem1.getKey().compareTo(logCountItem2.getKey());
71                         }
72                     });
73     
74                     final ArrayList<String> labels = new ArrayList<String>(sortedLogCounts.size());
75                     for (final Entry<LogType, Integer> pair : sortedLogCounts) {
76                         labels.add(pair.getValue() + "Ã— " + pair.getKey().getL10n());
77                     }
78     
79                     final TextView countView = new TextView(activity);
80                     countView.setText(res.getString(R.string.cache_log_types) + ": " + StringUtils.join(labels, ", "));
81                     view.addHeaderView(countView, null, false);
82                 }
83             }
84         }
