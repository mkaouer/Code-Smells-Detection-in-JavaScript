44     /**
45      * A TitlePageIndicator is a PageIndicator which displays the title of left view
46      * (if exist), the title of the current select view (centered) and the title of
47      * the right view (if exist). When the user scrolls the ViewPager then titles are
48      * also scrolled.
49      */
50     public class TitlePageIndicator extends View implements PageIndicator {
51         /**
52          * Percentage indicating what percentage of the screen width away from
53          * center should the underline be fully faded. A value of 0.25 means that
54          * halfway between the center of the screen and an edge.
55          */
56         private static final float SELECTION_FADE_PERCENTAGE = 0.25f;
57     
58         /**
59          * Percentage indicating what percentage of the screen width away from
60          * center should the selected text bold turn off. A value of 0.05 means
61          * that 10% between the center and an edge.
62          */
63         private static final float BOLD_FADE_PERCENTAGE = 0.05f;
64     
65         public enum IndicatorStyle {
66             None(0), Triangle(1), Underline(2);
67     
68             public final int value;
69     
70             IndicatorStyle(int value) {
71                 this.value = value;
72             }
73     
74             public static IndicatorStyle fromValue(int value) {
75                 for (IndicatorStyle style : IndicatorStyle.values()) {
76                     if (style.value == value) {
77                         return style;
78                     }
79                 }
80                 return null;
81             }
82         }

289         /*
290          * (non-Javadoc)
291          *
292          * @see android.view.View#onDraw(android.graphics.Canvas)
293          */
294         @Override
295         protected void onDraw(Canvas canvas) {
296             super.onDraw(canvas);
297     
298             if (mViewPager == null) {
299                 return;
300             }
301             final int count = mViewPager.getAdapter().getCount();
302             if (count == 0) {
303                 return;
304             }
305     
306             //Calculate views bounds
307             ArrayList<RectF> bounds = calculateAllBounds(mPaintText);
308     
309             //Make sure we're on a page that still exists
310             if (mCurrentPage >= bounds.size()) {
311                 setCurrentItem(bounds.size()-1);
312             }
313     
314             final int countMinusOne = count - 1;
315             final float halfWidth = getWidth() / 2f;
316             final int left = getLeft();
317             final float leftClip = left + mClipPadding;
318             final int width = getWidth();
319             final int height = getHeight();
320             final int right = left + width;
321             final float rightClip = right - mClipPadding;
322     
323             int page = mCurrentPage;
324             float offsetPercent;
325             if (mCurrentOffset <= halfWidth) {
326                 offsetPercent = 1.0f * mCurrentOffset / width;
327             } else {
328                 page += 1;
329                 offsetPercent = 1.0f * (width - mCurrentOffset) / width;
330             }
331             final boolean currentSelected = (offsetPercent <= SELECTION_FADE_PERCENTAGE);
332             final boolean currentBold = (offsetPercent <= BOLD_FADE_PERCENTAGE);
333             final float selectedPercent = (SELECTION_FADE_PERCENTAGE - offsetPercent) / SELECTION_FADE_PERCENTAGE;
334     
335             //Verify if the current view must be clipped to the screen
336             RectF curPageBound = bounds.get(mCurrentPage);
337             float curPageWidth = curPageBound.right - curPageBound.left;
338             if (curPageBound.left < leftClip) {
339                 //Try to clip to the screen (left side)
340                 clipViewOnTheLeft(curPageBound, curPageWidth, left);
341             }
342             if (curPageBound.right > rightClip) {
343                 //Try to clip to the screen (right side)
344                 clipViewOnTheRight(curPageBound, curPageWidth, right);
345             }
346     
347             //Left views starting from the current position
348             if (mCurrentPage > 0) {
349                 for (int i = mCurrentPage - 1; i >= 0; i--) {
350                     RectF bound = bounds.get(i);
351                     //Is left side is outside the screen
352                     if (bound.left < leftClip) {
353                         float w = bound.right - bound.left;
354                         //Try to clip to the screen (left side)
355                         clipViewOnTheLeft(bound, w, left);
356                         //Except if there's an intersection with the right view
357                         RectF rightBound = bounds.get(i + 1);
358                         //Intersection
359                         if (bound.right + mTitlePadding > rightBound.left) {
360                             bound.left = rightBound.left - w - mTitlePadding;
361                             bound.right = bound.left + w;
362                         }
363                     }
364                 }
365             }
366             //Right views starting from the current position
367             if (mCurrentPage < countMinusOne) {
368                 for (int i = mCurrentPage + 1 ; i < count; i++) {
369                     RectF bound = bounds.get(i);
370                     //If right side is outside the screen
371                     if (bound.right > rightClip) {
372                         float w = bound.right - bound.left;
373                         //Try to clip to the screen (right side)
374                         clipViewOnTheRight(bound, w, right);
375                         //Except if there's an intersection with the left view
376                         RectF leftBound = bounds.get(i - 1);
377                         //Intersection
378                         if (bound.left - mTitlePadding < leftBound.right) {
379                             bound.left = leftBound.right + mTitlePadding;
380                             bound.right = bound.left + w;
381                         }
382                     }
383                 }
384             }
385     
386             //Now draw views
387             for (int i = 0; i < count; i++) {
388                 //Get the title
389                 RectF bound = bounds.get(i);
390                 //Only if one side is visible
391                 if ((bound.left > left && bound.left < right) || (bound.right > left && bound.right < right)) {
392                     final boolean currentPage = (i == page);
393                     //Only set bold if we are within bounds
394                     mPaintText.setFakeBoldText(currentPage && currentBold && mBoldText);
395     
396                     //Draw text as unselected
397                     mPaintText.setColor(mColorText);
398                     canvas.drawText(mTitleProvider.getTitle(i), bound.left, bound.bottom + mTopPadding, mPaintText);
399     
400                     //If we are within the selected bounds draw the selected text
401                     if (currentPage && currentSelected) {
402                         mPaintText.setColor(mColorSelected);
403                         mPaintText.setAlpha((int)((mColorSelected >>> 24) * selectedPercent));
404                         canvas.drawText(mTitleProvider.getTitle(i), bound.left, bound.bottom + mTopPadding, mPaintText);
405                     }
406                 }
407             }
408     
409             //Draw the footer line
410             mPath.reset();
411             mPath.moveTo(0, height - mFooterLineHeight / 2f);
412             mPath.lineTo(width, height - mFooterLineHeight / 2f);
413             mPath.close();
414             canvas.drawPath(mPath, mPaintFooterLine);
415     
416             switch (mFooterIndicatorStyle) {
417                 case Triangle:
418                     mPath.reset();
419                     mPath.moveTo(halfWidth, height - mFooterLineHeight - mFooterIndicatorHeight);
420                     mPath.lineTo(halfWidth + mFooterIndicatorHeight, height - mFooterLineHeight);
421                     mPath.lineTo(halfWidth - mFooterIndicatorHeight, height - mFooterLineHeight);
422                     mPath.close();
423                     canvas.drawPath(mPath, mPaintFooterIndicator);
424                     break;
425     
426                 case Underline:
427                     if (!currentSelected) {
428                         break;
429                     }
430     
431                     RectF underlineBounds = bounds.get(page);
432                     mPath.reset();
433                     mPath.moveTo(underlineBounds.left  - mFooterIndicatorUnderlinePadding, height - mFooterLineHeight);
434                     mPath.lineTo(underlineBounds.right + mFooterIndicatorUnderlinePadding, height - mFooterLineHeight);
435                     mPath.lineTo(underlineBounds.right + mFooterIndicatorUnderlinePadding, height - mFooterLineHeight - mFooterIndicatorHeight);
436                     mPath.lineTo(underlineBounds.left  - mFooterIndicatorUnderlinePadding, height - mFooterLineHeight - mFooterIndicatorHeight);
437                     mPath.close();
438     
439                     mPaintFooterIndicator.setAlpha((int)(0xFF * selectedPercent));
440                     canvas.drawPath(mPath, mPaintFooterIndicator);
441                     mPaintFooterIndicator.setAlpha(0xFF);
442                     break;
443     
444                 default:
445                     break;
446             }
447         }

449         @Override
450         public boolean onTouchEvent(@NonNull android.view.MotionEvent ev) {
451             if ((mViewPager == null) || (mViewPager.getAdapter().getCount() == 0)) {
452                 return false;
453             }
454     
455             final int action = ev.getAction();
456     
457             switch (action & MotionEventCompat.ACTION_MASK) {
458                 case MotionEvent.ACTION_DOWN:
459                     mActivePointerId = MotionEventCompat.getPointerId(ev, 0);
460                     mLastMotionX = ev.getX();
461                     break;
462     
463                 case MotionEvent.ACTION_MOVE: {
464                     final int activePointerIndex = MotionEventCompat.findPointerIndex(ev, mActivePointerId);
465                     final float x = MotionEventCompat.getX(ev, activePointerIndex);
466                     final float deltaX = x - mLastMotionX;
467     
468                     if (!mIsDragging) {
469                         if (Math.abs(deltaX) > mTouchSlop) {
470                             mIsDragging = true;
471                         }
472                     }
473     
474                     if (mIsDragging) {
475                         if (!mViewPager.isFakeDragging()) {
476                             mViewPager.beginFakeDrag();
477                         }
478     
479                         mLastMotionX = x;
480     
481                         mViewPager.fakeDragBy(deltaX);
482                     }
483     
484                     break;
485                 }
486     
487                 case MotionEvent.ACTION_CANCEL:
488                 case MotionEvent.ACTION_UP:
489                     if (!mIsDragging) {
490                         final int count = mViewPager.getAdapter().getCount();
491                         final int width = getWidth();
492                         final float halfWidth = width / 2f;
493                         final float sixthWidth = width / 6f;
494     
495                         if ((mCurrentPage > 0) && (ev.getX() < halfWidth - sixthWidth)) {
496                             mViewPager.setCurrentItem(mCurrentPage - 1);
497                             return true;
498                         } else if ((mCurrentPage < count - 1) && (ev.getX() > halfWidth + sixthWidth)) {
499                             mViewPager.setCurrentItem(mCurrentPage + 1);
500                             return true;
501                         }
502                     }
503     
504                     mIsDragging = false;
505                     mActivePointerId = INVALID_POINTER;
506                     if (mViewPager.isFakeDragging()) {
507                         mViewPager.endFakeDrag();
508                     }
509                     break;
510     
511                 case MotionEventCompat.ACTION_POINTER_DOWN: {
512                     final int index = MotionEventCompat.getActionIndex(ev);
513                     mLastMotionX = MotionEventCompat.getX(ev, index);
514                     mActivePointerId = MotionEventCompat.getPointerId(ev, index);
515                     break;
516                 }
517     
518                 case MotionEventCompat.ACTION_POINTER_UP:
519                     final int pointerIndex = MotionEventCompat.getActionIndex(ev);
520                     final int pointerId = MotionEventCompat.getPointerId(ev, pointerIndex);
521                     if (pointerId == mActivePointerId) {
522                         final int newPointerIndex = pointerIndex == 0 ? 1 : 0;
523                         mActivePointerId = MotionEventCompat.getPointerId(ev, newPointerIndex);
524                     }
525                     mLastMotionX = MotionEventCompat.getX(ev, MotionEventCompat.findPointerIndex(ev, mActivePointerId));
526                     break;
527             }
528     
529             return true;
530         }

532         /**
533          * Set bounds for the right textView including clip padding.
534          *
535          * @param curViewBound
536          *            current bounds.
537          * @param curViewWidth
538          *            width of the view.
539          */
540         private void clipViewOnTheRight(RectF curViewBound, float curViewWidth, int right) {
541             curViewBound.right = right - mClipPadding;
542             curViewBound.left = curViewBound.right - curViewWidth;
543         }

549          *            current bounds.

551          *            width of the view.
552          */
553         private void clipViewOnTheLeft(RectF curViewBound, float curViewWidth, int left) {
554             curViewBound.left = left + mClipPadding;
555             curViewBound.right = mClipPadding + curViewWidth;
556         }

558         /**
559          * Calculate views bounds and scroll them according to the current index
560          *
561          * @param paint
562          * @param currentIndex
563          * @return
564          */
565         private ArrayList<RectF> calculateAllBounds(Paint paint) {
566             ArrayList<RectF> list = new ArrayList<RectF>();
567             //For each views (If no values then add a fake one)
568             final int count = mViewPager.getAdapter().getCount();
569             final int width = getWidth();
570             final int halfWidth = width / 2;
571             for (int i = 0; i < count; i++) {
572                 RectF bounds = calcBounds(i, paint);
573                 float w = (bounds.right - bounds.left);
574                 float h = (bounds.bottom - bounds.top);
575                 bounds.left = (halfWidth) - (w / 2) - mCurrentOffset + ((i - mCurrentPage) * width);
576                 bounds.right = bounds.left + w;
577                 bounds.top = 0;
578                 bounds.bottom = h;
579                 list.add(bounds);
580             }
581     
582             return list;
583         }

585         /**
586          * Calculate the bounds for a view's title
587          *
588          * @param index
589          * @param paint
590          * @return
591          */
592         private RectF calcBounds(int index, Paint paint) {
593             //Calculate the text bounds
594             RectF bounds = new RectF();
595             bounds.right = paint.measureText(mTitleProvider.getTitle(index));
596             bounds.bottom = paint.descent() - paint.ascent();
597             return bounds;
598         }

600         @Override
601         public void setViewPager(ViewPager view) {
602             final PagerAdapter adapter = view.getAdapter();
603             if (adapter == null) {
604                 throw new IllegalStateException("ViewPager does not have adapter instance.");
605             }
606             if (!(adapter instanceof TitleProvider)) {
607                 throw new IllegalStateException("ViewPager adapter must implement TitleProvider to be used with TitlePageIndicator.");
608             }
609             mViewPager = view;
610             mViewPager.setOnPageChangeListener(this);
611             mTitleProvider = (TitleProvider)adapter;
612             invalidate();
613         }

683         /**
684          * Determines the width of this view
685          *
686          * @param measureSpec
687          *            A measureSpec packed into an int
688          * @return The width of the view, honoring constraints from measureSpec
689          */
690         private int measureWidth(int measureSpec) {
691             int specMode = MeasureSpec.getMode(measureSpec);
692             int specSize = MeasureSpec.getSize(measureSpec);
693     
694             if (specMode != MeasureSpec.EXACTLY) {
695                 throw new IllegalStateException(getClass().getSimpleName() + " can only be used in EXACTLY mode.");
696             }
697             return specSize;
698         }
699     
700         /**
701          * Determines the height of this view
702          *
703          * @param measureSpec
704          *            A measureSpec packed into an int
705          * @return The height of the view, honoring constraints from measureSpec
706          */
707         private int measureHeight(int measureSpec) {
708             float result;
709             int specMode = MeasureSpec.getMode(measureSpec);
710             int specSize = MeasureSpec.getSize(measureSpec);
711     
712             if (specMode == MeasureSpec.EXACTLY) {
713                 //We were told how big to be
714                 result = specSize;
715             } else {
716                 //Calculate the text bounds
717                 RectF bounds = new RectF();
718                 bounds.bottom = mPaintText.descent()-mPaintText.ascent();
719                 result = bounds.bottom - bounds.top + mFooterLineHeight + mFooterPadding + mTopPadding;
720                 if (mFooterIndicatorStyle != IndicatorStyle.None) {
721                     result += mFooterIndicatorHeight;
722                 }
723             }
724             return (int)result;
725         }
