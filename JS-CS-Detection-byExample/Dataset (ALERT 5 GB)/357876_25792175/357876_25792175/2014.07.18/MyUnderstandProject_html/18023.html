68         private final static Map<String, LogType> FIND_BY_ICONNAME;
69         private final static Map<String, LogType> FIND_BY_TYPE;
70         static {
71             final HashMap<String, LogType> mappingPattern = new HashMap<String, LogType>();
72             final HashMap<String, LogType> mappingType = new HashMap<String, LogType>();
73             for (LogType lt : values()) {
74                 if (lt.iconName != null) {
75                     mappingPattern.put(lt.iconName, lt);
76                 }
77                 mappingType.put(lt.type, lt);
78             }
79             FIND_BY_ICONNAME = Collections.unmodifiableMap(mappingPattern);
80             FIND_BY_TYPE = Collections.unmodifiableMap(mappingType);
81         }

83         public static LogType getById(final int id) {
84             for (LogType logType : values()) {
85                 if (logType.id == id) {
86                     return logType;
87                 }
88             }
89             return UNKNOWN;
90         }

92         public static LogType getByIconName(final String imageType) {
93             // Special case for post reviewer note, which appears sometimes as 18.png (in individual entries) or as 68.png
94             // (in logs counts).
95             if ("68".equals(imageType)) {
96                 return POST_REVIEWER_NOTE;
97             }
98             final LogType result = imageType != null ? LogType.FIND_BY_ICONNAME.get(imageType.toLowerCase(Locale.US).trim()) : null;
99             if (result == null) {
100                 return UNKNOWN;
101             }
102             return result;
103         }
104     
105         public static LogType getByType(final String type) {
106             final LogType result = type != null ? LogType.FIND_BY_TYPE.get(type.toLowerCase(Locale.US).trim()) : null;
107             if (result == null) {
108                 return UNKNOWN;
109             }
110             return result;
111         }
