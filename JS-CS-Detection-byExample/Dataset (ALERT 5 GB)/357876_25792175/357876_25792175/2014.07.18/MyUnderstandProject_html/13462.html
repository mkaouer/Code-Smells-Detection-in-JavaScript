23     package com.mapswithme.maps.api;
24     
25     import java.io.Serializable;
26     
27     /**
28      * POI wrapper object.
29      * Has it's <code>equals()</code> and <code>hashCode()</code> methods overloaded
30      * so could be used in Hash(Map/Set/etc) classes.
31      */
32     public final class MWMPoint implements Serializable
33     {
34       private static final long serialVersionUID = 1L;
35     
36       final private double mLat;
37       final private double mLon;
38       final private String mName;
39       private String mId;
40     
41       public MWMPoint(double lat, double lon, String name)
42       {
43         this(lat, lon, name, null);
44       }
45     
46       public MWMPoint(double lat, double lon, String name, String id)
47       {
48         this.mLat = lat;
49         this.mLon = lon;
50         this.mName = name;
51         this.mId = id;
52       }
53     
54       public double getLat()       { return mLat;   }
55       public double getLon()       { return mLon;   }
56       public String getName()      { return mName;  }
57       public String getId()        { return mId;    }
58     
59       /**
60        * Sets string ID for this point. Internally it is not used to distinguish point,
61        * it's purpose to help clients code to associate point with domain objects of their application.
62        * @param id
63        */
64       public void setId(String id) { mId = id; }
65     
66       @Override
67       public String toString()
68       {
69         return "MWMPoint [lat=" + mLat + ", lon=" + mLon + ", name=" + mName + ", id=" + mId + "]";
70       }
71     
72       @Override
73       public int hashCode()
74       {
75         final int prime = 31;
76         int result = 1;
77         long temp;
78         temp = Double.doubleToLongBits(mLat);
79         result = prime * result + (int) (temp ^ (temp >>> 32));
80         temp = Double.doubleToLongBits(mLon);
81         result = prime * result + (int) (temp ^ (temp >>> 32));
82         result = prime * result + ((mName == null) ? 0 : mName.hashCode());
83         return result;
84       }
85     
86       /**
87        *  Two point are considered
88        *  equal if they have they lat, lon, and name attributes equal.
89        */
90       @Override
91       public boolean equals(Object obj)
92       {
93         if (this == obj)
94           return true;
95         if (obj == null)
96           return false;
97         if (getClass() != obj.getClass())
98           return false;
99         final MWMPoint other = (MWMPoint) obj;
100         if (Double.doubleToLongBits(mLat) != Double.doubleToLongBits(other.mLat))
101           return false;
102         if (Double.doubleToLongBits(mLon) != Double.doubleToLongBits(other.mLon))
103           return false;
104     
105         return mName == null ? other.mName == null : mName.equals(other.mName);
106       }
107     }
