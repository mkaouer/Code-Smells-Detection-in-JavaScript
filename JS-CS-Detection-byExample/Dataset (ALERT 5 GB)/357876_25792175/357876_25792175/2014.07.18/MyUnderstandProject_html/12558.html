83             }

85                 @Override
86                 public Observable<GCVoteRating> call() {
87                     final GCVoteRating rating = GCVote.getRating(cache.getGuid(), geocode);
88                     return rating != null ? Observable.just(rating) : Observable.<GCVoteRating>empty();
89                 }
90             })).subscribe(new Action1<GCVoteRating>() {
91                 @Override
92                 public void call(final GCVoteRating rating) {
93                     cache.setRating(rating.getRating());
94                     cache.setVotes(rating.getVotes());
95                     DataStore.saveChangedCache(cache);

97                 }
