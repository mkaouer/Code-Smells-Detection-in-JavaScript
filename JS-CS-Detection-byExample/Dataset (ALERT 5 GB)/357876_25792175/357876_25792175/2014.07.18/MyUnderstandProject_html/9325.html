37         private void processAttributes(Context context, AttributeSet attrs, int defStyle) {
38             if (attrs == null) {
39                 return; // coward's retreat
40             }
41     
42             TypedArray types = context.obtainStyledAttributes(attrs, new int[] {
43                     R.attr.title, R.attr.text, R.attr.url, R.attr.urlButton },
44                     defStyle, 0);
45     
46             title = types.getString(0);
47             text = types.getString(1);
48             url = types.getString(2);
49             urlButton = types.getString(3);
50     
51             types.recycle();
52         }

55         protected View onCreateView(ViewGroup parent) {
56     
57             if (baseOnPrefChangeListener == null) {
58                 baseOnPrefChangeListener = getOnPreferenceChangeListener();

60     
61             // show dialog when checkbox enabled
62             setOnPreferenceChangeListener(new OnPreferenceChangeListener() {
63                 @Override
64                 public boolean onPreferenceChange(final Preference preference, Object newValue) {
65                     if (baseOnPrefChangeListener != null) {
66                         baseOnPrefChangeListener.onPreferenceChange(preference, newValue);
67                     }
68                     if (!(Boolean) newValue) {
69                         return true;
70                     }
71                     new UrlPopup(preference.getContext()).show(title, text, url, urlButton);
72                     return true;
73                 }
74             });
