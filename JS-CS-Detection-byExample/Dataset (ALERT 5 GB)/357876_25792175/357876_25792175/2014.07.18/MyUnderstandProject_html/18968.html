81         public void selectFilter(final Action1<IFilter> runAfterwards) {
82             final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
83             builder.setTitle(R.string.caches_filter_title);
84     
85             final ArrayAdapter<FactoryEntry> adapter = new ArrayAdapter<FactoryEntry>(activity, android.R.layout.select_dialog_item, registry);
86     
87             builder.setAdapter(adapter, new DialogInterface.OnClickListener() {
88                 @Override
89                 public void onClick(final DialogInterface dialog, final int itemIndex) {
90                     FactoryEntry entry = adapter.getItem(itemIndex);
91                     // reset?
92                     if (entry.filterFactory == null) {
93                         runAfterwards.call(null);
94                     }
95                     else {
96                         try {
97                             IFilterFactory factoryInstance = entry.filterFactory.newInstance();
98                             selectFromFactory(factoryInstance, entry.name, runAfterwards);
99                         } catch (Exception e) {
100                             Log.e("selectFilter", e);
101                         }
102                     }
103                 }
104             });
105     
106             builder.create().show();
107         }

109         private void selectFromFactory(final IFilterFactory factory, final String menuTitle, final Action1<IFilter> runAfterwards) {
110             final List<IFilter> filters = Collections.unmodifiableList(factory.getFilters());
111             if (filters.size() == 1) {
112                 runAfterwards.call(filters.get(0));
113                 return;
114             }
115     
116             final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
117             builder.setTitle(menuTitle);
118     
119             final ArrayAdapter<IFilter> adapter = new ArrayAdapter<IFilter>(activity, android.R.layout.select_dialog_item, filters);
120             builder.setAdapter(adapter, new DialogInterface.OnClickListener() {
121                 @Override
122                 public void onClick(final DialogInterface dialog, final int item) {
123                     runAfterwards.call(filters.get(item));
124                 }
125             });
126     
127             builder.create().show();
128         }
