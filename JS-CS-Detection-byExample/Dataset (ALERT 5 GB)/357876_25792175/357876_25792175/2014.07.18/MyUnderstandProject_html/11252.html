13         /**
14          * Insert an attribute-less tag with enclosed text in a XML serializer output.
15          *
16          * @param serializer an XML serializer
17          * @param prefix an XML prefix, see {@link XmlSerializer#startTag(String, String)}
18          * @param tag an XML tag
19          * @param text some text to insert, or <tt>null</tt> to omit completely this tag
20          * @throws IOException
21          */
22         public static void simpleText(final XmlSerializer serializer, final String prefix, final String tag, final String text) throws IOException {
23             if (text != null) {
24                 serializer.startTag(prefix, tag);
25                 serializer.text(text);
26                 serializer.endTag(prefix, tag);
27             }
28         }

30         /**
31          * Insert pairs of attribute-less tags and enclosed texts in a XML serializer output
32          *
33          * @param serializer an XML serializer
34          * @param prefix an XML prefix, see {@link XmlSerializer#startTag(String, String)} shared by all tags
35          * @param tagAndText an XML tag, the corresponding text, another XML tag, the corresponding text. <tt>null</tt> texts
36          *                   will be omitted along with their respective tag.
37          * @throws IOException
38          */
39         public static void multipleTexts(final XmlSerializer serializer, final String prefix, final String... tagAndText) throws IOException {
40             for (int i = 0; i < tagAndText.length; i += 2) {
41                 simpleText(serializer, prefix, tagAndText[i], tagAndText[i+1]);
42             }
43         }
