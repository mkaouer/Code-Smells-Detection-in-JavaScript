25         /**
26          * restore the database in a new thread, showing a progress window
27          *
28          * @param activity
29          *            calling activity
30          */
31         public static void restoreDatabase(final Activity activity) {
32             final Resources res = activity.getResources();
33             final ProgressDialog dialog = ProgressDialog.show(activity, res.getString(R.string.init_backup_restore), res.getString(R.string.init_restore_running), true, false);
34             final AtomicBoolean restoreSuccessful = new AtomicBoolean(false);
35             new Thread() {
36                 @Override
37                 public void run() {
38                     restoreSuccessful.set(DataStore.restoreDatabaseInternal());
39                     activity.runOnUiThread(new Runnable() {
40                         @Override
41                         public void run() {
42                             dialog.dismiss();
43                             boolean restored = restoreSuccessful.get();
44                             String message = restored ? res.getString(R.string.init_restore_success) : res.getString(R.string.init_restore_failed);
45                             Dialogs.message(activity, R.string.init_backup_restore, message);
46                             if (activity instanceof MainActivity) {
47                                 ((MainActivity) activity).updateCacheCounter();
48                             }
49                         }
50                     });
51                 }
52             }.start();
53         }

55         public static boolean createBackup(final Activity activity, final Runnable runAfterwards) {
56             final Context context = activity;
57             // avoid overwriting an existing backup with an empty database
58             // (can happen directly after reinstalling the app)
59             if (DataStore.getAllCachesCount() == 0) {
60                 Dialogs.message(activity, R.string.init_backup, R.string.init_backup_unnecessary);
61                 return false;
62             }
63     
64             final ProgressDialog dialog = ProgressDialog.show(context,
65                     context.getString(R.string.init_backup),
66                     context.getString(R.string.init_backup_running), true, false);
67             new Thread() {
68                 @Override
69                 public void run() {
70                     final String backupFileName = DataStore.backupDatabaseInternal();
71                     activity.runOnUiThread(new Runnable() {
72                         @Override
73                         public void run() {
74                             dialog.dismiss();
75                             Dialogs.message(activity,
76                                     R.string.init_backup_backup,
77                                     backupFileName != null
78                                             ? context.getString(R.string.init_backup_success)
79                                                     + "\n" + backupFileName
80                                             : context.getString(R.string.init_backup_failed));
81                             if (runAfterwards != null) {
82                                 runAfterwards.run();
83                             }
84                         }
85                     });
86                 }
87             }.start();
88             return true;
89         }

91         public static File getRestoreFile() {
92             final File fileSourceFile = DataStore.getBackupFileInternal();
93             return fileSourceFile.exists() && fileSourceFile.length() > 0 ? fileSourceFile : null;
94         }
95     
96         public static boolean hasBackup() {
97             return getRestoreFile() != null;
98         }
99     
100         public static String getBackupDateTime() {
101             final File restoreFile = getRestoreFile();
102             if (restoreFile == null) {
103                 return StringUtils.EMPTY;
104             }
105             return Formatter.formatShortDateTime(restoreFile.lastModified());
106         }
