47         void drawPosition(Canvas canvas, MapProjectionImpl projection) {
48             if (coordinates == null || location == null) {
49                 return;
50             }
51     
52             if (accuracyCircle == null) {
53                 accuracyCircle = new Paint();
54                 accuracyCircle.setAntiAlias(true);
55                 accuracyCircle.setStrokeWidth(1.0f);
56             }
57     
58             if (historyLine == null) {
59                 historyLine = new Paint();
60                 historyLine.setAntiAlias(true);
61                 historyLine.setStrokeWidth(3.0f);
62                 historyLine.setColor(0xFFFFFFFF);
63             }
64     
65             if (historyLineShadow == null) {
66                 historyLineShadow = new Paint();
67                 historyLineShadow.setAntiAlias(true);
68                 historyLineShadow.setStrokeWidth(7.0f);
69                 historyLineShadow.setColor(0x66000000);
70             }
71     
72             if (setfil == null) {
73                 setfil = new PaintFlagsDrawFilter(0, Paint.FILTER_BITMAP_FLAG);
74             }
75             if (remfil == null) {
76                 remfil = new PaintFlagsDrawFilter(Paint.FILTER_BITMAP_FLAG, 0);
77             }
78     
79             canvas.setDrawFilter(setfil);
80     
81             double latitude = coordinates.getLatitude();
82             double longitude = coordinates.getLongitude();
83             float accuracy = coordinates.getAccuracy();
84     
85             float[] result = new float[1];
86     
87             Location.distanceBetween(latitude, longitude, latitude, longitude + 1, result);
88             float longitudeLineDistance = result[0];
89     
90             final Geopoint leftCoords = new Geopoint(latitude, longitude - accuracy / longitudeLineDistance);
91             GeoPointImpl leftGeo = mapItemFactory.getGeoPointBase(leftCoords);
92             projection.toPixels(leftGeo, left);
93             projection.toPixels(location, center);
94             int radius = center.x - left.x;
95     
96             accuracyCircle.setColor(0x66000000);
97             accuracyCircle.setStyle(Style.STROKE);
98             canvas.drawCircle(center.x, center.y, radius, accuracyCircle);
99     
100             accuracyCircle.setColor(0x08000000);
101             accuracyCircle.setStyle(Style.FILL);
102             canvas.drawCircle(center.x, center.y, radius, accuracyCircle);
103     
104             positionHistory.rememberTrailPosition(coordinates);
105     
106             if (Settings.isMapTrail()) {
107                 // always add current position to drawn history to have a closed connection
108                 final ArrayList<Location> paintHistory = new ArrayList<Location>(positionHistory.getHistory());
109                 paintHistory.add(coordinates);
110     
111                 int size = paintHistory.size();
112                 if (size > 1) {
113                     int alphaCnt = size - 201;
114                     if (alphaCnt < 1) {
115                         alphaCnt = 1;
116                     }
117     
118                     Point pointNow = new Point();
119                     Point pointPrevious = new Point();
120                     Location prev = paintHistory.get(0);
121                     projection.toPixels(mapItemFactory.getGeoPointBase(new Geopoint(prev)), pointPrevious);
122     
123                     for (int cnt = 1; cnt < size; cnt++) {
124                         Location now = paintHistory.get(cnt);
125                         projection.toPixels(mapItemFactory.getGeoPointBase(new Geopoint(now)), pointNow);
126     
127                         int alpha;
128                         if ((alphaCnt - cnt) > 0) {
129                             alpha = 255 / (alphaCnt - cnt);
130                         }
131                         else {
132                             alpha = 255;
133                         }
134     
135                         historyLineShadow.setAlpha(alpha);
136                         historyLine.setAlpha(alpha);
137     
138                         canvas.drawLine(pointPrevious.x, pointPrevious.y, pointNow.x, pointNow.y, historyLineShadow);
139                         canvas.drawLine(pointPrevious.x, pointPrevious.y, pointNow.x, pointNow.y, historyLine);
140     
141                         pointPrevious.set(pointNow.x, pointNow.y);
142                     }
143                 }
144             }
145     
146             if (arrow == null) {
147                 arrow = BitmapFactory.decodeResource(activity.getResources(), R.drawable.my_location_chevron);
148                 widthArrowHalf = arrow.getWidth() / 2;
149                 heightArrowHalf = arrow.getHeight() / 2;
150             }
151     
152             int marginLeft = center.x - widthArrowHalf;
153             int marginTop = center.y - heightArrowHalf;
154     
155             Matrix matrix = new Matrix();
156             matrix.setRotate(heading, widthArrowHalf, heightArrowHalf);
157             matrix.postTranslate(marginLeft, marginTop);
158     
159             canvas.drawBitmap(arrow, matrix, null);
160     
161             canvas.setDrawFilter(remfil);
162         }
