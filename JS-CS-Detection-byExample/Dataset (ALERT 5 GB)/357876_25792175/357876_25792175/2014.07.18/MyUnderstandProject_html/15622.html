25         public static void addToCalendarWithIntent(final Activity activity, final Geocache cache) {
26             final Resources res = activity.getResources();
27             if (CalendarAddon.isAvailable()) {
28                 final Date hiddenDate = cache.getHiddenDate();
29                 final Parameters params = new Parameters(
30                         ICalendar.PARAM_NAME, cache.getName(),
31                         ICalendar.PARAM_NOTE, StringUtils.defaultString(cache.getPersonalNote()),
32                         ICalendar.PARAM_HIDDEN_DATE, hiddenDate != null ? String.valueOf(hiddenDate.getTime()) : StringUtils.EMPTY,
33                         ICalendar.PARAM_URL, StringUtils.defaultString(cache.getUrl()),
34                         ICalendar.PARAM_COORDS, cache.getCoords() == null ? "" : cache.getCoords().format(GeopointFormatter.Format.LAT_LON_DECMINUTE_RAW),
35                         ICalendar.PARAM_LOCATION, StringUtils.defaultString(cache.getLocation()),
36                         ICalendar.PARAM_SHORT_DESC, StringUtils.defaultString(cache.getShortDescription()),
37                         ICalendar.PARAM_START_TIME_MINUTES, StringUtils.defaultString(cache.guessEventTimeMinutes())
38                         );
39     
40                 activity.startActivity(new Intent(ICalendar.INTENT,
41                         Uri.parse(ICalendar.URI_SCHEME + "://" + ICalendar.URI_HOST + "?" + params.toString())));
42             } else {
43                 // Inform user the calendar add-on is not installed and let them get it from Google Play
44                 Dialogs.confirmYesNo(activity, R.string.addon_missing_title, new StringBuilder(res.getString(R.string.helper_calendar_missing))
45                         .append(' ')
46                         .append(res.getString(R.string.addon_download_prompt))
47                         .toString(), new DialogInterface.OnClickListener() {
48                     @Override
49                     public void onClick(DialogInterface dialog, int id) {
50                         final Intent intent = new Intent(Intent.ACTION_VIEW);
51                         intent.setData(Uri.parse(ICalendar.CALENDAR_ADDON_URI));
52                         activity.startActivity(intent);
53                     }
54                 });
55             }
56         }
