74         protected void fillViewHolder(final View convertView, LogViewHolder holder, final LogEntry log) {
75             if (log.date > 0) {
76                 holder.date.setText(Formatter.formatShortDateVerbally(log.date));
77                 holder.date.setVisibility(View.VISIBLE);
78             } else {
79                 holder.date.setVisibility(View.GONE);
80             }
81     
82             holder.type.setText(log.type.getL10n());
83             holder.author.setText(StringEscapeUtils.unescapeHtml4(log.author));
84     
85             fillCountOrLocation(holder, log);
86     
87             // log text, avoid parsing HTML if not necessary
88             String logText = log.log;
89             if (TextUtils.containsHtml(logText)) {
90                 logText = log.getDisplayText();
91                 // Fast preview: parse only HTML without loading any images
92                 final HtmlImageCounter imageCounter = new HtmlImageCounter();
93                 final UnknownTagsHandler unknownTagsHandler = new UnknownTagsHandler();
94                 holder.text.setText(Html.fromHtml(logText, imageCounter, unknownTagsHandler), TextView.BufferType.SPANNABLE);
95                 if (imageCounter.getImageCount() > 0) {
96                     // Complete view: parse again with loading images - if necessary ! If there are any images causing problems the user can see at least the preview
97                     final LogImageLoader loader = new LogImageLoader(holder);
98                     loader.execute(logText);
99                 }
100             }
101             else {
102                 holder.text.setText(logText, TextView.BufferType.SPANNABLE);
103             }
104     
105             // images
106             if (log.hasLogImages()) {
107                 holder.images.setText(log.getImageTitles());
108                 holder.images.setVisibility(View.VISIBLE);
109                 holder.images.setOnClickListener(new View.OnClickListener() {
110                     @Override
111                     public void onClick(View v) {
112                         ImagesActivity.startActivityLogImages(activity, getGeocode(), new ArrayList<Image>(log.getLogImages()));
113                     }
114                 });
115             } else {
116                 holder.images.setVisibility(View.GONE);
117             }
118     
119             // colored marker
120             final int marker = log.type.markerId;
121             if (marker != 0) {
122                 holder.marker.setVisibility(View.VISIBLE);
123                 holder.marker.setImageResource(marker);
124             }
125             else {
126                 holder.marker.setVisibility(View.GONE);
127             }
128     
129             if (null == convertView) {
130                 holder.author.setOnClickListener(createUserActionsListener());
131                 holder.text.setMovementMethod(AnchorAwareLinkMovementMethod.getInstance());
132                 holder.text.setOnClickListener(new DecryptTextClickListener(holder.text));
133                 activity.registerForContextMenu(holder.text);
134             }
135         }

149         /** Loads the Log Images outside the UI thread. */
150     
151         private class LogImageLoader extends AsyncTask<String, Progress, Spanned> {
152             final private LogViewHolder holder;
153             final private int position;
154     
155             public LogImageLoader(LogViewHolder holder) {
156                 this.holder = holder;
157                 this.position = holder.getPosition();
158             }
159     
160             @Override
161             protected Spanned doInBackground(String... logtext) {
162                 return Html.fromHtml(logtext[0], new HtmlImage(getGeocode(), false, StoredList.STANDARD_LIST_ID, false), null); //, TextView.BufferType.SPANNABLE)
163             }
164     
165             @Override
166             protected void onPostExecute(Spanned result) {
167                 // Ensure that this holder and its view still references the right item before updating the text.
168                 if (position == holder.getPosition()) {
169                     holder.text.setText(result);
170                 }
171             }
