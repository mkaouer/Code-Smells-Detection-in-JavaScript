124         /**
125          * Multipart POST HTTP request
126          *
127          * @param uri the URI to request
128          * @param params the parameters to add to the POST request
129          * @param fileFieldName the name of the file field name
130          * @param fileContentType the content-type of the file
131          * @param file the file to include in the request
132          * @return the HTTP response, or null in case of an encoding error param
133          */
134         @Nullable
135         public static HttpResponse postRequest(final String uri, final Parameters params,
136                 final String fileFieldName, final String fileContentType, final File file) {
137             final MultipartEntity entity = new MultipartEntity();
138             for (final NameValuePair param : params) {
139                 try {
140                     entity.addPart(param.getName(), new StringBody(param.getValue(), TextUtils.CHARSET_UTF8));
141                 } catch (final UnsupportedEncodingException e) {
142                     Log.e("Network.postRequest: unsupported encoding for parameter " + param.getName(), e);
143                     return null;
144                 }
145             }
146             entity.addPart(fileFieldName, new FileBody(file, fileContentType));
147     
148             final HttpPost request = new HttpPost(uri);
149             request.setEntity(entity);
150     
151             addHeaders(request, null, null);
152             return doLogRequest(request);
153         }

155         /**
156          * Make an HTTP request
157          *
158          * @param method
159          *            the HTTP method to use ("GET" or "POST")
160          * @param uri
161          *            the URI to request
162          * @param params
163          *            the parameters to add to the URI
164          * @param headers
165          *            the headers to add to the request
166          * @param cacheFile
167          *            the cache file used to cache this query
168          * @return the HTTP response, or null in case of an encoding error in a POST request arguments
169          */
170         @Nullable
171         private static HttpResponse request(final String method, final String uri,
172                                             @Nullable final Parameters params, @Nullable final Parameters headers, @Nullable final File cacheFile) {
173             HttpRequestBase request;
174             if (method.equals("GET")) {
175                 final String fullUri = params == null ? uri : Uri.parse(uri).buildUpon().encodedQuery(params.toString()).build().toString();
176                 request = new HttpGet(fullUri);
177             } else {
178                 request = new HttpPost(uri);
179                 if (params != null) {
180                     try {
181                         ((HttpPost) request).setEntity(new UrlEncodedFormEntity(params, CharEncoding.UTF_8));
182                     } catch (final UnsupportedEncodingException e) {
183                         Log.e("request", e);
184                         return null;
185                     }
186                 }
187             }
188     
189             addHeaders(request, headers, cacheFile);
190     
191             return doLogRequest(request);
192         }

194         /**
195          * Add headers to HTTP request.
196          * @param request
197          *            the request to add headers to
198          * @param headers
199          *            the headers to add (in addition to the standard headers), can be null
200          * @param cacheFile
201          *            if non-null, the file to take ETag and If-Modified-Since information from
202          */
203         private static void addHeaders(final HttpRequestBase request, @Nullable final Parameters headers, @Nullable final File cacheFile) {
204             for (final NameValuePair header : Parameters.extend(Parameters.merge(headers, cacheHeaders(cacheFile)),
205                     "Accept-Charset", "utf-8,iso-8859-1;q=0.8,utf-16;q=0.8,*;q=0.7",
206                     "Accept-Language", "en-US,*;q=0.9",
207                     "X-Requested-With", "XMLHttpRequest")) {
208                 request.setHeader(header.getName(), header.getValue());
209             }
210             request.getParams().setParameter(CoreProtocolPNames.USER_AGENT,
211                     Settings.getUseNativeUa() ? NATIVE_USER_AGENT : PC_USER_AGENT);
212         }

214         /**
215          * Perform an HTTP request and log it.
216          *
217          * @param request
218          *            the request to try
219          * @return
220          *            the response, or null if there has been a failure
221          */
222         @Nullable
223         private static HttpResponse doLogRequest(final HttpRequestBase request) {
224             final String reqLogStr = request.getMethod() + " " + hidePassword(request.getURI().toString());
225             Log.d(reqLogStr);
226     
227             final HttpClient client = getHttpClient();
228             final long before = System.currentTimeMillis();
229             try {
230                 final HttpResponse response = client.execute(request);
231                 int status = response.getStatusLine().getStatusCode();
232                 if (status == 200) {
233                     Log.d(status + formatTimeSpan(before) + reqLogStr);
234                 } else {
235                     Log.w(status + " [" + response.getStatusLine().getReasonPhrase() + "]" + formatTimeSpan(before) + reqLogStr);
236                 }
237                 return response;
238             } catch (final IOException e) {
239                 final String timeSpan = formatTimeSpan(before);
240                 Log.w("Failure" + timeSpan + reqLogStr + " (" + e.toString() + ")");
241             }
242     
243             return null;
244         }

246         @Nullable
247         private static Parameters cacheHeaders(@Nullable final File cacheFile) {
248             if (cacheFile == null || !cacheFile.exists()) {
249                 return null;
250             }
251     
252             final String etag = LocalStorage.getSavedHeader(cacheFile, LocalStorage.HEADER_ETAG);
253             if (etag != null) {
254                 // The ETag is a more robust check than a timestamp. If we have an ETag, it is enough
255                 // to identify the right version of the resource.
256                 return new Parameters("If-None-Match", etag);
257             }
258     
259             final String lastModified = LocalStorage.getSavedHeader(cacheFile, LocalStorage.HEADER_LAST_MODIFIED);
260             if (lastModified != null) {
261                 return new Parameters("If-Modified-Since", lastModified);
262             }
263     
264             return null;
265         }

340          * Get the result of a GET HTTP request returning a JSON body.
341          *
342          * @param uri the base URI of the GET HTTP request
343          * @param params the query parameters, or <code>null</code> if there are none
344          * @return a JSON object if the request was successful and the body could be decoded, <code>null</code> otherwise

346         @Nullable
347         public static JSONObject requestJSON(final String uri, @Nullable final Parameters params) {
348             final HttpResponse response = request("GET", uri, params, new Parameters("Accept", "application/json, text/javascript, */*; q=0.01"), null);
349             final String responseData = getResponseData(response, false);
350             if (responseData != null) {
351                 try {
352                     return new JSONObject(responseData);
353                 } catch (final JSONException e) {
354                     Log.w("Network.requestJSON", e);
355                 }
356             }
357     
358             return null;
359         }

361         /**
362          * Get the input stream corresponding to a HTTP response if it exists.
363          *
364          * @param response a HTTP response, which can be null
365          * @return the input stream if the HTTP request is successful, <code>null</code> otherwise
366          */
367         @Nullable
368         public static InputStream getResponseStream(@Nullable final HttpResponse response) {
369             if (!isSuccess(response)) {
370                 return null;
371             }
372             assert(response != null);
373             final HttpEntity entity = response.getEntity();
374             if (entity == null) {
375                 return null;
376             }
377             try {
378                 return entity.getContent();
379             } catch (final IOException e) {
380                 Log.e("Network.getResponseStream", e);
381                 return null;
382             }
383         }
