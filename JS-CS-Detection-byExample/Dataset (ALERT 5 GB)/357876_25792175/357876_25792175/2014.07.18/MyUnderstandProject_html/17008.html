141         @Override
142         public SearchResult searchByGeocode(final @Nullable String geocode, final @Nullable String guid, final CancellableHandler handler) {
143     
144             CancellableHandler.sendLoadProgressDetail(handler, R.string.cache_dialog_loading_details_status_loadpage);
145     
146             final String page = GCParser.requestHtmlPage(geocode, guid, "y", String.valueOf(GCConstants.NUMBER_OF_LOGS));
147     
148             if (StringUtils.isEmpty(page)) {
149                 final SearchResult search = new SearchResult();
150                 if (DataStore.isThere(geocode, guid, true, false)) {
151                     if (StringUtils.isBlank(geocode) && StringUtils.isNotBlank(guid)) {
152                         Log.i("Loading old cache from cache.");
153                         search.addGeocode(DataStore.getGeocodeForGuid(guid));
154                     } else {
155                         search.addGeocode(geocode);
156                     }
157                     search.setError(StatusCode.NO_ERROR);
158                     return search;
159                 }
160     
161                 Log.e("GCConnector.searchByGeocode: No data from server");
162                 search.setError(StatusCode.COMMUNICATION_ERROR);
163                 return search;
164             }
165             assert page != null;
166     
167             final SearchResult searchResult = GCParser.parseCache(page, handler);
168     
169             if (searchResult == null || CollectionUtils.isEmpty(searchResult.getGeocodes())) {
170                 Log.w("GCConnector.searchByGeocode: No cache parsed");
171                 return searchResult;
172             }
173     
174             // do not filter when searching for one specific cache
175             return searchResult;
176         }

295         @Override
296         public String getGeocodeFromUrl(String url) {
297             // coord.info URLs
298             String code = StringUtils.substringAfterLast(url, "coord.info/");
299             if (code != null && canHandle(code)) {
300                 return code;
301             }
302             // expanded geocaching.com URLs
303             code = StringUtils.substringBetween(url, "/geocache/", "_");
304             if (code != null && canHandle(code)) {
305                 return code;
306             }
307             return null;
308         }

323         @Override
324         public boolean login(Handler handler, Context fromActivity) {
325             // login
326             final StatusCode status = GCLogin.getInstance().login();
327     
328             if (status == StatusCode.NO_ERROR) {
329                 GCLogin.detectGcCustomDate();
330             }
331     
332             if (CgeoApplication.getInstance().showLoginToast && handler != null) {
333                 handler.sendMessage(handler.obtainMessage(0, status));
334                 CgeoApplication.getInstance().showLoginToast = false;
335     
336                 // invoke settings activity to insert login details
337                 if (status == StatusCode.NO_LOGIN_INFO_STORED && fromActivity != null) {
338                     SettingsActivity.openForScreen(R.string.preference_screen_gc, fromActivity);
339                 }
340             }
341             return status == StatusCode.NO_ERROR;
342         }

428         @Override
429         public boolean uploadFieldNotes(final File exportFile) {
430             if (!GCLogin.getInstance().isActualLoginStatus()) {
431                 // no need to upload (possibly large file) if we're not logged in
432                 final StatusCode loginState = GCLogin.getInstance().login();
433                 if (loginState != StatusCode.NO_ERROR) {
434                     Log.e("FieldnoteExport.ExportTask upload: Login failed");
435                 }
436             }
437     
438             final String uri = "http://www.geocaching.com/my/uploadfieldnotes.aspx";
439             final String page = GCLogin.getInstance().getRequestLogged(uri, null);
440     
441             if (StringUtils.isBlank(page)) {
442                 Log.e("FieldnoteExport.ExportTask get page: No data from server");
443                 return false;
444             }
445     
446             final String[] viewstates = GCLogin.getViewstates(page);
447     
448             final Parameters uploadParams = new Parameters(
449                     "__EVENTTARGET", "",
450                     "__EVENTARGUMENT", "",
451                     "ctl00$ContentBody$btnUpload", "Upload Field Note");
452     
453             GCLogin.putViewstates(uploadParams, viewstates);
454     
455             Network.getResponseData(Network.postRequest(uri, uploadParams, "ctl00$ContentBody$FieldNoteLoader", "text/plain", exportFile));
456     
457             if (StringUtils.isBlank(page)) {
458                 Log.e("FieldnoteExport.ExportTask upload: No data from server");
459                 return false;
460             }
461             return true;
462         }
