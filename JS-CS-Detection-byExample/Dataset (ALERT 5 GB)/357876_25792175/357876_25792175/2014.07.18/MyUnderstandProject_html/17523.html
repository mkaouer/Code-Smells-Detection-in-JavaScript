33         /**
34          * Creates a dialog so that the user can select an exporter.
35          *
36          * @param caches
37          *            The {@link List} of {@link cgeo.geocaching.Geocache} to be exported
38          * @param activity
39          *            The {@link Activity} in whose context the dialog should be shown
40          */
41         public static void showExportMenu(final List<Geocache> caches, final Activity activity) {
42             final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
43             builder.setTitle(R.string.export).setIcon(R.drawable.ic_menu_share);
44     
45             final ArrayList<Export> export = new ArrayList<Export>();
46             for (Class<? extends Export> exporterClass : exporterClasses) {
47                 try {
48                     export.add(exporterClass.newInstance());
49                 } catch (Exception ex) {
50                     Log.e("showExportMenu", ex);
51                 }
52             }
53     
54             final ArrayAdapter<Export> adapter = new ArrayAdapter<Export>(activity, android.R.layout.select_dialog_item, export);
55     
56             builder.setAdapter(adapter, new DialogInterface.OnClickListener() {
57                 @Override
58                 public void onClick(DialogInterface dialog, int item) {
59                     dialog.dismiss();
60                     final Export selectedExport = adapter.getItem(item);
61                     selectedExport.export(caches, activity);
62                 }
63             });
64     
65             builder.create().show();
66         }
