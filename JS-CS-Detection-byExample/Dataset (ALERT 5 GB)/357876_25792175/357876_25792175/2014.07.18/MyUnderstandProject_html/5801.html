87             assertThat(tb.getDetails()).startsWith("Auf der CD sind");
88             assertThat(tb.getImage()).isEqualTo("http://imgcdn.geocaching.com/track/display/38382780-87a7-4393-8393-78841678ee8c.jpg");
89             // Following data can change over time
90             assertThat(tb.getDistance()).isGreaterThanOrEqualTo(10617.8f);
91             assertThat(tb.getLogs().size()).isGreaterThanOrEqualTo(10);

94             // no assumption possible: assertThat(tb.getSpottedName()).isEqualTo("Nice place for a break cache");

96             // we can't check specifics in the log entries since they change, but we can verify data was parsed

341         @MediumTest
342         public static void testSearchByViewportNotLoggedIn() {
343             withMockedLoginDo(new Runnable() {
344     
345                 public void run() {
346                     final Strategy strategy = Settings.getLiveMapStrategy();
347                     final Strategy testStrategy = Strategy.FAST; // FASTEST, FAST or DETAILED for tests
348                     Settings.setLiveMapStrategy(testStrategy);
349                     final CacheType cacheType = Settings.getCacheType();
350     
351                     try {
352     
353                         // non premium cache
354                         MockedCache cache = new GC2CJPF();
355                         deleteCacheFromDBAndLogout(cache.getGeocode());
356                         Tile.cache.removeFromTileCache(cache);
357                         Settings.setCacheType(CacheType.ALL);
358     
359                         Viewport viewport = new Viewport(cache, 0.003, 0.003);
360                         SearchResult search = ConnectorFactory.searchByViewport(viewport, INVALID_TOKEN);
361     
362                         assertThat(search).isNotNull();
363                         assertThat(search.getGeocodes().contains(cache.getGeocode())).isTrue();
364                         // coords differ
365                         final Geocache cacheFromViewport = DataStore.loadCache(cache.getGeocode(), LoadFlags.LOAD_CACHE_OR_DB);
366                         Log.d("cgeoApplicationTest.testSearchByViewportNotLoggedIn: Coords expected = " + cache.getCoords());
367                         Log.d("cgeoApplicationTest.testSearchByViewportNotLoggedIn: Coords actual = " + cacheFromViewport.getCoords());
368                         assertThat(cache.getCoords().distanceTo(cacheFromViewport.getCoords()) <= 1e-3).isFalse();
369                         // depending on the chosen strategy the coords can be reliable or not
370                         assertThat(cacheFromViewport.isReliableLatLon()).isEqualTo(testStrategy == Strategy.DETAILED);
371     
372                         // premium cache
373                         cache = new GC2JVEH();
374                         deleteCacheFromDBAndLogout(cache.getGeocode());
375     
376                         viewport = new Viewport(cache, 0.003, 0.003);
377                         search = ConnectorFactory.searchByViewport(viewport, INVALID_TOKEN);
378     
379                         assertThat(search).isNotNull();
380                         // In the meantime, premium-member caches are also shown on map when not logged in
381                         assertThat(search.getGeocodes().contains(cache.getGeocode())).isTrue();
382     
383                     } finally {
384                         Settings.setLiveMapStrategy(strategy);
385                         Settings.setCacheType(cacheType);
386                     }
387                 }
388             });
