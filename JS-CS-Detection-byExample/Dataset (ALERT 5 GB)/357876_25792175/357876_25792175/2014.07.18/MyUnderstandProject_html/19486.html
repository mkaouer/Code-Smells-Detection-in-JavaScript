35         /**
36          * inverse lookup of waypoint IDs<br/>
37          * non public so that <code>null</code> handling can be handled centrally in the enum type itself
38          */
39         private static final Map<String, WaypointType> FIND_BY_ID;
40         public static final Set<WaypointType> ALL_TYPES_EXCEPT_OWN_AND_ORIGINAL = new HashSet<WaypointType>();
41         static {
42             final HashMap<String, WaypointType> mapping = new HashMap<String, WaypointType>();
43             for (WaypointType wt : values()) {
44                 mapping.put(wt.id, wt);
45                 if (wt != WaypointType.OWN && wt != WaypointType.ORIGINAL) {
46                     ALL_TYPES_EXCEPT_OWN_AND_ORIGINAL.add(wt);
47                 }
48             }
49             FIND_BY_ID = Collections.unmodifiableMap(mapping);
50         }
51     
52         /**
53          * inverse lookup of waypoint IDs<br/>
54          * here the <code>null</code> handling shall be done
55          */
56         public static WaypointType findById(final String id) {
57             if (null == id) {
58                 return WAYPOINT;
59             }
60             WaypointType waypointType = FIND_BY_ID.get(id);
61             if (null == waypointType) {
62                 return WAYPOINT;
63             }
64             return waypointType;
65         }
