9     /**
10      * Filter reader which can filter out invalid XML characters and character references.
11      *
12      */
13     public class InvalidXMLCharacterFilterReader extends FilterReader
14     {
15     
16         public InvalidXMLCharacterFilterReader(Reader in) {
17             super(in);
18         }
19     
20         /**
21          * Every overload of {@link Reader#read()} method delegates to this one so
22          * it is enough to override only this one. <br />
23          * To skip invalid characters this method shifts only valid chars to left
24          * and returns decreased value of the original read method. So after last
25          * valid character there will be some unused chars in the buffer.
26          *
27          * @return Number of read valid characters or <code>-1</code> if end of the
28          *         underling reader was reached.
29          */
30         @Override
31         public int read(char[] cbuf, int off, int len) throws IOException {
32             int read = super.read(cbuf, off, len);
33             // check for end
34             if (read == -1) {
35                 return -1;
36             }
37             // target position
38             int pos = off - 1;
39     
40             int entityStart = -1;
41             for (int readPos = off; readPos < off + read; readPos++) {
42                 boolean useChar = true;
43                 switch (cbuf[readPos]) {
44                     case '&':
45                         pos++;
46                         entityStart = readPos;
47                         break;
48                     case ';':
49                         pos++;
50                         if (entityStart >= 0) {
51                             int entityLength = readPos - entityStart + 1;
52                             if (entityLength <= 5) {
53                                 String entity = new String(cbuf, entityStart, entityLength);
54                                 if (StringUtils.startsWith(entity, "&#")) {
55                                     String numberString = StringUtils.substringBetween(entity, "&#", ";");
56                                     final int value;
57                                     if (StringUtils.startsWith(numberString, "x")) {
58                                         value = Integer.parseInt(numberString.substring(1), 16);
59                                     }
60                                     else {
61                                         value = Integer.parseInt(numberString);
62                                     }
63                                     if (!isValidXMLChar((char) value)) {
64                                         pos -= entityLength;
65                                         useChar = false;
66                                     }
67                                 }
68                             }
69                         }
70                         break;
71                     default:
72                         if (isValidXMLChar(cbuf[readPos])) {
73                             pos++;
74                         } else {
75                             continue;
76                         }
77                 }
78                 // copy, and skip unwanted characters
79                 if (pos < readPos && useChar) {
80                     cbuf[pos] = cbuf[readPos];
81                 }
82             }
83             return pos - off + 1;
84         }
85     
86         private static boolean isValidXMLChar(char c) {
87             if ((c == 0x9) ||
88                     (c == 0xA) ||
89                     (c == 0xD) ||
90                     ((c >= 0x20) && (c <= 0xD7FF)) ||
91                     ((c >= 0xE000) && (c <= 0xFFFD)) ||
92                     ((c >= 0x10000) && (c <= 0x10FFFF)))
93             {
94                 return true;
95             }
96             return false;
97         }
