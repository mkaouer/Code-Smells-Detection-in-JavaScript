79             public void promptForListSelection(final int titleId, @NonNull final Action1<Integer> runAfterwards, final boolean onlyConcreteLists, final int exceptListId, final String newListName) {
80                 final List<AbstractList> lists = new ArrayList<AbstractList>();
81                 lists.addAll(getSortedLists());
82     
83                 if (exceptListId > StoredList.TEMPORARY_LIST_ID) {
84                     StoredList exceptList = DataStore.getList(exceptListId);
85                     if (exceptList != null) {
86                         lists.remove(exceptList);
87                     }
88                 }
89     
90                 if (!onlyConcreteLists) {
91                     if (exceptListId != PseudoList.ALL_LIST.id) {
92                         lists.add(PseudoList.ALL_LIST);
93                     }
94                     if (exceptListId != PseudoList.HISTORY_LIST.id) {
95                         lists.add(PseudoList.HISTORY_LIST);
96                     }
97                 }
98                 lists.add(PseudoList.NEW_LIST);
99     
100                 final List<CharSequence> listsTitle = new ArrayList<CharSequence>();
101                 for (AbstractList list : lists) {
102                     listsTitle.add(list.getTitleAndCount());
103                 }
104     
105                 final CharSequence[] items = new CharSequence[listsTitle.size()];
106     
107                 AlertDialog.Builder builder = new AlertDialog.Builder(activity);
108                 builder.setTitle(res.getString(titleId));
109                 builder.setItems(listsTitle.toArray(items), new DialogInterface.OnClickListener() {
110                     @Override
111                     public void onClick(DialogInterface dialogInterface, int itemId) {
112                         final AbstractList list = lists.get(itemId);
113                         if (list == PseudoList.NEW_LIST) {
114                             // create new list on the fly
115                             promptForListCreation(runAfterwards, newListName);
116                         }
117                         else {
118                             runAfterwards.call(lists.get(itemId).id);
119                         }
120                     }
121                 });
122                 builder.create().show();
123             }

125             @NonNull
126             private static List<StoredList> getSortedLists() {
127                 final Collator collator = Collator.getInstance();
128                 final List<StoredList> lists = DataStore.getLists();
129                 Collections.sort(lists, new Comparator<StoredList>() {
130     
131                     @Override
132                     public int compare(StoredList lhs, StoredList rhs) {
133                         // have the standard list at the top
134                         if (lhs.id == STANDARD_LIST_ID) {
135                             return -1;
136                         }
137                         if (rhs.id == STANDARD_LIST_ID) {
138                             return 1;
139                         }
140                         // otherwise sort alphabetical
141                         return collator.compare(lhs.getTitle(), rhs.getTitle());
142                     }
143                 });
144                 return lists;
145             }

147             public void promptForListCreation(@NonNull final Action1<Integer> runAfterwards, String newListName) {
148                 handleListNameInput(newListName, R.string.list_dialog_create_title, R.string.list_dialog_create, new Action1<String>() {
149     
150                     // We need to update the list cache by creating a new StoredList object here.
151                     @SuppressWarnings("unused")
152                     @Override
153                     public void call(final String listName) {
154                         final int newId = DataStore.createList(listName);
155                         new StoredList(newId, listName, 0);
156     
157                         if (newId >= DataStore.customListIdOffset) {
158                             ActivityMixin.showToast(activity, res.getString(R.string.list_dialog_create_ok));
159                             runAfterwards.call(newId);
160                         } else {
161                             ActivityMixin.showToast(activity, res.getString(R.string.list_dialog_create_err));
162                         }
163                     }
164                 });
165             }

167             private void handleListNameInput(final String defaultValue, int dialogTitle, int buttonTitle, final Action1<String> runnable) {
168                 Dialogs.input(activity, dialogTitle, defaultValue, buttonTitle, new Action1<String>() {
169

172                         // remove whitespaces added by autocompletion of Android keyboard

174                         if (StringUtils.isNotBlank(listName)) {
175                             runnable.call(listName);

177                     }
178                 });
