20         private final static Pattern PATTERN_TB_CODE = Pattern.compile("(TB[0-9A-Z]+)|([0-9A-Z]{6})", Pattern.CASE_INSENSITIVE);
21     
22         @Override
23         public boolean canHandleTrackable(String geocode) {
24             return TravelBugConnector.PATTERN_TB_CODE.matcher(geocode).matches() && !StringUtils.startsWithIgnoreCase(geocode, "GC");
25         }
26     
27         @Override
28         public String getUrl(Trackable trackable) {
29             return "http://www.geocaching.com//track/details.aspx?tracker=" + trackable.getGeocode();
30         }

33         public boolean isLoggable() {
34             return true;
35         }

39             return GCParser.searchTrackable(geocode, guid, id);
40         }

45         private static class Holder {

57         @Override
58         public @Nullable
59         String getTrackableCodeFromUrl(@NonNull String url) {
60             // coord.info URLs
61             String code = StringUtils.substringAfterLast(url, "coord.info/");
62             if (code != null && canHandleTrackable(code)) {
63                 return code;
64             }
65             code = StringUtils.substringAfterLast(url, "?tracker=");
66             if (code != null && canHandleTrackable(code)) {
67                 return code;
68             }
69             return null;
70         }
