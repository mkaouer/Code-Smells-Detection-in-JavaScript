36         @Override
37         protected StatusCode login(boolean retry) {
38             final ImmutablePair<String, String> login = Settings.getCredentials(ECConnector.getInstance());
39     
40             if (StringUtils.isEmpty(login.left) || StringUtils.isEmpty(login.right)) {
41                 clearLoginInfo();
42                 Log.e("ECLogin.login: No login information stored");
43                 return StatusCode.NO_LOGIN_INFO_STORED;
44             }
45     
46             setActualStatus(app.getString(R.string.init_login_popup_working));
47     
48             final Parameters params = new Parameters("user", login.left, "pass", login.right);
49             HttpResponse loginResponse = Network.postRequest("https://extremcaching.com/exports/apilogin.php", params);
50     
51             String loginData = Network.getResponseData(loginResponse);
52     
53             if (StringUtils.isBlank(loginData)) {
54                 Log.e("ECLogin.login: Failed to retrieve login data");
55                 return StatusCode.CONNECTION_FAILED_EC; // no login page
56             }
57     
58             assert loginData != null;
59     
60             if (loginData.contains("Wrong username or password")) { // hardcoded in English
61                 Log.i("Failed to log in Extremcaching.com as " + login.left + " because of wrong username/password");
62                 return StatusCode.WRONG_LOGIN_DATA; // wrong login
63             }
64     
65             if (getLoginStatus(loginData)) {
66                 Log.i("Successfully logged in Extremcaching.com as " + login.left);
67     
68                 return StatusCode.NO_ERROR; // logged in
69             }
70     
71             Log.i("Failed to log in Extremcaching.com as " + login.left + " for some unknown reason");
72             if (retry) {
73                 return login(false);
74             }
75     
76             return StatusCode.UNKNOWN_ERROR; // can't login
77         }

80         /**
81          * Check if the user has been logged in when he retrieved the data.
82          *
83          * @param data
84          * @return <code>true</code> if user is logged in, <code>false</code> otherwise
85          */
86         public boolean getLoginStatus(@Nullable final String data) {
87             if (StringUtils.isBlank(data) || StringUtils.equals(data, "[]")) {
88                 Log.e("ECLogin.getLoginStatus: No or empty data given");
89                 return false;
90             }
91             assert data != null;
92     
93             setActualStatus(app.getString(R.string.init_login_popup_ok));
94     
95             try {
96                 final JSONObject json = new JSONObject(data);
97     
98                 final String sid = json.getString("sid");
99                 if (!StringUtils.isBlank(sid)) {
100                     sessionId = sid;
101                     setActualLoginStatus(true);
102                     setActualUserName(json.getString("username"));
103                     setActualCachesFound(json.getInt("found"));
104                     return true;
105                 }
106                 resetLoginStatus();
107             } catch (final JSONException e) {
108                 Log.e("ECLogin.getLoginStatus", e);
109             }
110     
111             setActualStatus(app.getString(R.string.init_login_popup_failed));
112             return false;
113         }

115         @Override
116         protected void resetLoginStatus() {
117             sessionId = null;
118             setActualLoginStatus(false);
119         }

121         public String getSessionId() {
122             if (!StringUtils.isBlank(sessionId) || login() == StatusCode.NO_ERROR) {
123                 return sessionId;

125             return StringUtils.EMPTY;
