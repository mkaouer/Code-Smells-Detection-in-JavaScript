38         public void testGetWaypointsFileNameForGpxFile() throws IOException {
39             final String[] gpxFiles = new String[] { "1234567.gpx", "1.gpx", "1234567.9.gpx",
40                     "1234567.GPX", "gpx.gpx.gpx", ".gpx",
41                     "1234567_query.gpx", "123-4.gpx", "123(5).gpx" };
42             final String[] wptsFiles = new String[] { "1234567-wpts.gpx", "1-wpts.gpx", "1234567.9-wpts.gpx",
43                     "1234567-wpts.GPX", "gpx.gpx-wpts.gpx", "-wpts.gpx",
44                     "1234567_query-wpts.gpx", "123-wpts-4.gpx", "123-wpts(5).gpx" };
45             for (int i = 0; i < gpxFiles.length; i++) {
46                 String gpxFileName = gpxFiles[i];
47                 String wptsFileName = wptsFiles[i];
48                 File gpx = new File(tempDir, gpxFileName);
49                 File wpts = new File(tempDir, wptsFileName);
50                 // the files need to exist - we create them
51                 assertThat(gpx.createNewFile()).isTrue();
52                 assertThat(wpts.createNewFile()).isTrue();
53                 // the "real" method check
54                 assertThat(GPXImporter.getWaypointsFileNameForGpxFile(gpx)).isEqualTo(wptsFileName);
55                 // they also need to be deleted, because of case sensitive tests that will not work correct on case insensitive file systems
56                 FileUtils.deleteQuietly(gpx);
57                 FileUtils.deleteQuietly(wpts);
58             }
59             final File gpx1 = new File(tempDir, "abc.gpx");
60             assertThat(GPXImporter.getWaypointsFileNameForGpxFile(gpx1)).isNull();
61         }

256         static class TestHandler extends CancellableHandler {
257             private final List<Message> messages = new ArrayList<Message>();
258             private long lastMessage = System.currentTimeMillis();
259     
260             @Override
261             public synchronized void handleRegularMessage(Message msg) {
262                 final Message msg1 = Message.obtain();
263                 msg1.copyFrom(msg);
264                 messages.add(msg1);
265                 lastMessage = System.currentTimeMillis();
266                 notifyAll();
267             }
268     
269             public synchronized void waitForCompletion(final long milliseconds, final int maxMessages) {
270                 try {
271                     while (System.currentTimeMillis() - lastMessage <= milliseconds && messages.size() <= maxMessages) {
272                         wait(milliseconds);
273                     }
274                 } catch (InterruptedException e) {
275                     // intentionally left blank
276                 }
277             }
278     
279             public void waitForCompletion() {
280                 // Use reasonable defaults
281                 waitForCompletion(1000, 10);
282             }
283         }
