1115         private class LoadDetailsThread extends Thread {
1116     
1117             final private CancellableHandler handler;
1118             final private int listIdLD;
1119             final private List<Geocache> caches;
1120     
1121             public LoadDetailsThread(CancellableHandler handler, List<Geocache> caches, int listId) {
1122                 this.handler = handler;
1123                 this.caches = caches;
1124     
1125                 // in case of online lists, set the list id to the standard list
1126                 this.listIdLD = Math.max(listId, StoredList.STANDARD_LIST_ID);
1127             }
1128     
1129             @Override
1130             public void run() {
1131                 // First refresh caches that do not yet have static maps to get them a chance to get a copy
1132                 // before the limit expires, unless we do not want to store offline maps.
1133                 final List<Geocache> allCaches = Settings.isStoreOfflineMaps() ?
1134                         ListUtils.union(ListUtils.selectRejected(caches, Geocache.hasStaticMap),
1135                                 ListUtils.select(caches, Geocache.hasStaticMap)) :
1136                         caches;
1137     
1138                 for (final Geocache cache : allCaches) {
1139                     if (!refreshCache(cache)) {
1140                         break;
1141                     }
1142                 }
1143     
1144                 handler.sendEmptyMessage(MSG_DONE);
1145             }
1146     
1147             /**
1148              * Refreshes the cache information.
1149              *
1150              * @param cache
1151              *            The cache to refresh
1152              * @return
1153              *         <code>false</code> if the storing was interrupted, <code>true</code> otherwise
1154              */
1155             private boolean refreshCache(Geocache cache) {
1156                 try {
1157                     if (handler.isCancelled()) {
1158                         throw new InterruptedException("Stopped storing process.");
1159                     }
1160                     detailProgress++;
1161                     cache.refreshSynchronous(listIdLD, null);
1162                     handler.sendEmptyMessage(cacheList.indexOf(cache));
1163                 } catch (final InterruptedException e) {
1164                     Log.i(e.getMessage());
1165                     return false;
1166                 } catch (final Exception e) {
1167                     Log.e("CacheListActivity.LoadDetailsThread", e);
1168                 }
1169     
1170                 return true;
1171             }
1172         }

1174         private static class LoadFromWebThread extends Thread {
1175     
1176             final private CancellableHandler handler;
1177             final private int listIdLFW;
1178     
1179             public LoadFromWebThread(CancellableHandler handler, int listId) {
1180                 this.handler = handler;
1181                 listIdLFW = StoredList.getConcreteList(listId);
1182             }
1183     
1184             @Override
1185             public void run() {
1186                 long baseTime = System.currentTimeMillis();
1187     
1188                 final String deviceCode = StringUtils.defaultString(Settings.getWebDeviceCode());
1189                 final Parameters params = new Parameters("code", deviceCode);
1190                 while (!handler.isCancelled() && System.currentTimeMillis() - baseTime < 3 * 60000) { // maximum: 3 minutes
1191                     // Download new code
1192                     final HttpResponse responseFromWeb = Network.getRequest("http://send2.cgeo.org/read.html", params);
1193     
1194                     if (responseFromWeb != null && responseFromWeb.getStatusLine().getStatusCode() == 200) {
1195                         final String response = Network.getResponseData(responseFromWeb);
1196                         if (response != null && response.length() > 2) {
1197                             handler.sendMessage(handler.obtainMessage(MSG_LOADING, response));
1198     
1199                             Geocache.storeCache(null, response, listIdLFW, false, null);
1200     
1201                             handler.sendMessage(handler.obtainMessage(MSG_LOADED, response));
1202                             baseTime = System.currentTimeMillis();
1203                         } else if ("RG".equals(response)) {
1204                             //Server returned RG (registration) and this device no longer registered.
1205                             Settings.setWebNameCode(null, null);
1206                             handler.sendEmptyMessage(MSG_NO_REGISTRATION);
1207                             handler.cancel();
1208                             break;
1209                         } else {
1210                             try {
1211                                 sleep(5000); // Wait for 5s if no cache found
1212                             } catch (final InterruptedException e) {
1213                             }
1214                             handler.sendEmptyMessage(MSG_WAITING);
1215                         }
1216                     } else {
1217                         handler.sendEmptyMessage(MSG_SERVER_FAIL);
1218                         handler.cancel();
1219                         break;
1220                     }
1221                 }
1222     
1223                 handler.sendEmptyMessage(MSG_DONE);
1224             }
1225         }

1535         @Override
1536         public Loader<SearchResult> onCreateLoader(int type, Bundle extras) {
1537             if (type >= CacheListLoaderType.values().length) {
1538                 throw new IllegalArgumentException("invalid loader type " + type);
1539             }
1540             final CacheListLoaderType enumType = CacheListLoaderType.values()[type];
1541             AbstractSearchLoader loader = null;
1542             switch (enumType) {
1543                 case OFFLINE:
1544                     // open either the requested or the last list
1545                     if (extras.containsKey(Intents.EXTRA_LIST_ID)) {
1546                         listId = extras.getInt(Intents.EXTRA_LIST_ID);
1547                     }
1548                     else {
1549                         listId = Settings.getLastList();
1550                     }
1551                     if (listId <= StoredList.TEMPORARY_LIST_ID) {
1552                         listId = StoredList.STANDARD_LIST_ID;
1553                         title = res.getString(R.string.stored_caches_button);
1554                     } else {
1555                         final StoredList list = DataStore.getList(listId);
1556                         // list.id may be different if listId was not valid
1557                         if (list.id != listId) {
1558                             showToast(getString(R.string.list_not_available));
1559                         }
1560                         listId = list.id;
1561                         title = list.title;
1562                     }
1563     
1564                     loader = new OfflineGeocacheListLoader(this.getBaseContext(), coords, listId);
1565     
1566                     break;
1567                 case HISTORY:
1568                     title = res.getString(R.string.caches_history);
1569                     loader = new HistoryGeocacheListLoader(app, coords);
1570                     break;
1571                 case NEAREST:
1572                     title = res.getString(R.string.caches_nearby);
1573                     loader = new CoordsGeocacheListLoader(app, coords);
1574                     break;
1575                 case COORDINATE:
1576                     title = coords.toString();
1577                     loader = new CoordsGeocacheListLoader(app, coords);
1578                     break;
1579                 case KEYWORD:
1580                     final String keyword = extras.getString(Intents.EXTRA_KEYWORD);
1581                     rememberTerm(keyword);
1582                     loader = new KeywordGeocacheListLoader(app, keyword);
1583                     break;
1584                 case ADDRESS:
1585                     final String address = extras.getString(Intents.EXTRA_ADDRESS);
1586                     if (StringUtils.isNotBlank(address)) {
1587                         rememberTerm(address);
1588                     } else {
1589                         title = coords.toString();
1590                     }
1591                     if (coords != null) {
1592                         loader = new CoordsGeocacheListLoader(app, coords);
1593                     }
1594                     else {
1595                         loader = new AddressGeocacheListLoader(app, address);
1596                     }
1597                     break;
1598                 case FINDER:
1599                     final String username = extras.getString(Intents.EXTRA_USERNAME);
1600                     rememberTerm(username);
1601                     loader = new FinderGeocacheListLoader(app, username);
1602                     break;
1603                 case OWNER:
1604                     final String ownerName = extras.getString(Intents.EXTRA_USERNAME);
1605                     rememberTerm(ownerName);
1606                     loader = new OwnerGeocacheListLoader(app, ownerName);
1607                     break;
1608                 case MAP:
1609                     //TODO Build Null loader
1610                     title = res.getString(R.string.map_map);
1611                     search = (SearchResult) extras.get(Intents.EXTRA_SEARCH);
1612                     replaceCacheListFromSearch();
1613                     loadCachesHandler.sendMessage(Message.obtain());
1614                     break;
1615                 case REMOVE_FROM_HISTORY:
1616                     title = res.getString(R.string.caches_history);
1617                     loader = new RemoveFromHistoryLoader(app, extras.getStringArray(Intents.EXTRA_CACHELIST), coords);
1618                     break;
1619                 case NEXT_PAGE:
1620                     loader = new NextPageGeocacheListLoader(app, search);
1621                     break;
1622                 case POCKET:
1623                     final String guid = extras.getString(Intents.EXTRA_POCKET_GUID);
1624                     title = extras.getString(Intents.EXTRA_NAME);
1625                     loader = new PocketGeocacheListLoader(app, guid);
1626                     break;
1627             }
1628             setTitle(title);
1629             showProgress(true);
1630             showFooterLoadingCaches();
1631     
1632             if (loader != null) {
1633                 loader.setRecaptchaHandler(new RecaptchaHandler(this, loader));
1634             }
1635             return loader;
1636         }

1647             // The database search was moved into the UI call intentionally. If this is done before the runOnUIThread,

1653                 search = searchIn;
1654                 adapter.reFilter();
1655                 adapter.checkEvents();
1656                 adapter.forceSort();
1657                 adapter.notifyDataSetChanged();
