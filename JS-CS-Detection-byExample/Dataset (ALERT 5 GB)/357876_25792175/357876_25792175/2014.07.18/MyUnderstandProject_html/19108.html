50             FIND_BY_AUTH_PREF_ID = new SparseArray<OCPreferenceKeys>(values().length);
51             Map<String, OCPreferenceKeys> byIsactiveKey = new HashMap<String, OCPreferenceKeys>();
52             for (OCPreferenceKeys key : values()) {
53                 FIND_BY_ISACTIVE_ID.put(key.isActivePrefId, key);

55                 byIsactiveKey.put(CgeoApplication.getInstance().getString(key.isActivePrefId), key);
56             }
57             FIND_BY_ISACTIVE_KEY = Collections.unmodifiableMap(byIsactiveKey);
58         }
