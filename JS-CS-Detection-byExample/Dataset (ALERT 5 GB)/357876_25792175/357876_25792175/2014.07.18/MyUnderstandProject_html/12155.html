13     /**
14      * {@link PageViewCreator} for {@link ListView}, which can save scroll state on purging a page from the
15      * {@link ViewPager}, and restore the state on re-recreation.
16      *
17      */
18     public abstract class AbstractCachingListViewPageViewCreator extends AbstractCachingPageViewCreator<ListView> {
19         private static final String STATE_POSITION_FROM_TOP = "positionFromTop";
20         private static final String STATE_POSITION = "position";
21     
22         /**
23          * Get the state of the current view
24          *
25          * @return the state encapsulated in a bundle
26          */
27         @Override
28         public @Nullable
29         Bundle getViewState() {
30             if (view == null) {
31                 return null;
32             }
33             int position = view.getFirstVisiblePosition();
34             View child = view.getChildAt(0);
35             int positionFromTop = (child == null) ? 0 : child.getTop();
36             Bundle state = new Bundle();
37             state.putInt(STATE_POSITION, position);
38             state.putInt(STATE_POSITION_FROM_TOP, positionFromTop);
39             return state;
40         }
41     
42         /**
43          * Restore a previously stored state of the view
44          *
45          */
46         @Override
47         public void setViewState(@NonNull Bundle state) {
48             if (view == null) {
49                 return;
50             }
51             int logViewPosition = state.getInt(STATE_POSITION);
52             int logViewPositionFromTop = state.getInt(STATE_POSITION_FROM_TOP);
53             view.setSelectionFromTop(logViewPosition, logViewPositionFromTop);
54             return;
55         }
