15     public class OAuthPreference extends AbstractClickablePreference {

17         private static final int NO_KEY = -1;
18     
19         private enum OAuthActivityMapping {
20             NONE(NO_KEY, null, null),
21             OCDE(R.string.pref_fakekey_ocde_authorization, OCAuthorizationActivity.class, OCAuthParams.OC_DE_AUTH_PARAMS),
22             OCPL(R.string.pref_fakekey_ocpl_authorization, OCAuthorizationActivity.class, OCAuthParams.OC_PL_AUTH_PARAMS),
23             OCNL(R.string.pref_fakekey_ocnl_authorization, OCAuthorizationActivity.class, OCAuthParams.OC_NL_AUTH_PARAMS),
24             OCUS(R.string.pref_fakekey_ocus_authorization, OCAuthorizationActivity.class, OCAuthParams.OC_US_AUTH_PARAMS),
25             OCRO(R.string.pref_fakekey_ocro_authorization, OCAuthorizationActivity.class, OCAuthParams.OC_RO_AUTH_PARAMS),
26             TWITTER(R.string.pref_fakekey_twitter_authorization, TwitterAuthorizationActivity.class, TwitterAuthorizationActivity.TWITTER_OAUTH_PARAMS);
27     
28             public final int prefKeyId;
29             public final Class<?> authActivity;
30             public final OAuthParameters authParams;
31     
32             OAuthActivityMapping(int prefKeyId, Class<?> authActivity, OAuthParameters authParams) {
33                 this.prefKeyId = prefKeyId;
34                 this.authActivity = authActivity;
35                 this.authParams = authParams;
36             }
37         }
38     
39         private final OAuthActivityMapping oAuthMapping;
40     
41         private OAuthActivityMapping getAuthorization() {
42             final String prefKey = getKey();
43             for (OAuthActivityMapping auth : OAuthActivityMapping.values()) {
44                 if (auth.prefKeyId != NO_KEY && prefKey.equals(CgeoApplication.getInstance().getString(auth.prefKeyId))) {
45                     return auth;
46                 }
47             }
48             return OAuthActivityMapping.NONE;
49         }
50     
51         public OAuthPreference(Context context, AttributeSet attrs) {
52             super(context, attrs);
53             this.oAuthMapping = getAuthorization();
54         }
55     
56         public OAuthPreference(Context context, AttributeSet attrs, int defStyle) {
57             super(context, attrs, defStyle);
58             this.oAuthMapping = getAuthorization();
59         }
60     
61         @Override
62         protected OnPreferenceClickListener getOnPreferenceClickListener(final SettingsActivity activity) {
63             activity.setAuthTitle(oAuthMapping.prefKeyId);
64             return new OnPreferenceClickListener() {
65                 @Override
66                 public boolean onPreferenceClick(Preference preference) {
67                     if (oAuthMapping.authActivity != null && oAuthMapping.authParams != null) {
68                         Intent authIntent = new Intent(preference.getContext(),
69                                 oAuthMapping.authActivity);
70                         oAuthMapping.authParams.setOAuthExtras(authIntent);
71                         activity.startActivityForResult(authIntent,
72                                 oAuthMapping.prefKeyId);
73                     }
74                     return false; // no shared preference has to be changed
75                 }
76             };
