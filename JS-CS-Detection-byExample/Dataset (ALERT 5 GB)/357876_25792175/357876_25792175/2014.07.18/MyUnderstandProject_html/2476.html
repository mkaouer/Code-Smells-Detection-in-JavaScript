32         public UncertainProperty<T> getMergedProperty(final UncertainProperty<T> other) {
33             if (null == other || null == other.value) {
34                 return this;

36             if (null == this.value) {
37                 return other;
38             }
39             if (other.certaintyLevel > certaintyLevel) {
40                 return other;
41             }

43             return this;
44         }
