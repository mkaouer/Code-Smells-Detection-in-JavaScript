54         /**
55          * This test verifies that a loop of import, export, import leads to the same cache information.
56          *
57          * @throws IOException
58          * @throws ParserException
59          */
60         public void testStableExportImportExport() throws IOException, ParserException {
61             final String geocode = "GC1BKP3";
62             final int cacheResource = R.raw.gc1bkp3_gpx101;
63             final Geocache cache = loadCacheFromResource(cacheResource);
64             assertThat(cache).isNotNull();
65     
66             final String gpxFirst = getGPXFromCache(geocode);
67     
68             assertThat(gpxFirst.length() > 0).isTrue();
69     
70             final GPX10Parser parser = new GPX10Parser(StoredList.TEMPORARY_LIST_ID);
71     
72             final InputStream stream = new ByteArrayInputStream(gpxFirst.getBytes(CharEncoding.UTF_8));
73             Collection<Geocache> caches = parser.parse(stream, null);
74             assertThat(caches).isNotNull();
75             assertThat(caches).hasSize(1);
76     
77             final String gpxSecond = getGPXFromCache(geocode);
78             assertThat(replaceLogIds(gpxSecond)).isEqualTo(replaceLogIds(gpxFirst));
79         }
80     
81         private static String replaceLogIds(String gpx) {

85         private static String getGPXFromCache(String geocode) throws IOException {
86             final StringWriter writer = new StringWriter();
87             new GpxSerializer().writeGPX(Collections.singletonList(geocode), writer, null);
88             return writer.toString();
