126         /**
127          * Specialized way to handle selection of navigation tool.<br />
128          * A dialog is created for tool selection and the selected tool is started afterwards.
129          *
130          * @param activity
131          * @param cache
132          *            may be <code>null</code>
133          * @param waypoint
134          *            may be <code>null</code>
135          * @param destination
136          *            may be <code>null</code>
137          * @param showInternalMap
138          *            should be <code>false</code> only when called from within the internal map
139          * @param showDefaultNavigation
140          *            should be <code>false</code> by default
141          *
142          * @see #showNavigationMenu(Activity, cgeo.geocaching.Geocache, cgeo.geocaching.Waypoint, Geopoint)
143          */
144         public static void showNavigationMenu(final Activity activity,
145                 final Geocache cache, final Waypoint waypoint, final Geopoint destination,
146                 final boolean showInternalMap, final boolean showDefaultNavigation) {
147             final List<NavigationAppsEnum> items = new ArrayList<NavigationAppFactory.NavigationAppsEnum>();
148             final int defaultNavigationTool = Settings.getDefaultNavigationTool();
149             for (final NavigationAppsEnum navApp : getInstalledNavigationApps()) {
150                 if ((showInternalMap || !(navApp.app instanceof InternalMap)) &&
151                         (showDefaultNavigation || defaultNavigationTool != navApp.id)) {
152                     if (Settings.isUseNavigationApp(navApp)) {
153                         boolean add = false;
154                         if (cache != null && navApp.app instanceof CacheNavigationApp && navApp.app.isEnabled(cache)) {
155                             add = true;
156                         }
157                         if (waypoint != null && navApp.app instanceof WaypointNavigationApp && ((WaypointNavigationApp) navApp.app).isEnabled(waypoint)) {
158                             add = true;
159                         }
160                         if (destination != null && navApp.app instanceof GeopointNavigationApp) {
161                             add = true;
162                         }
163                         if (add) {
164                             items.add(navApp);
165                         }
166                     }
167                 }
168             }
169     
170             if (items.size() == 1) {
171                 invokeNavigation(activity, cache, waypoint, destination, items.get(0).app);
172                 return;
173             }
174     
175             /*
176              * Using an ArrayAdapter with list of NavigationAppsEnum items avoids
177              * handling between mapping list positions allows us to do dynamic filtering of the list based on use case.
178              */
179             final ArrayAdapter<NavigationAppsEnum> adapter = new ArrayAdapter<NavigationAppsEnum>(activity, android.R.layout.select_dialog_item, items);
180     
181             final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
182             builder.setTitle(R.string.cache_menu_navigate);
183             builder.setAdapter(adapter, new DialogInterface.OnClickListener() {
184                 @Override
185                 public void onClick(DialogInterface dialog, int item) {
186                     final NavigationAppsEnum selectedItem = adapter.getItem(item);
187                     invokeNavigation(activity, cache, waypoint, destination, selectedItem.app);
188                 }
189             });
190             final AlertDialog alert = builder.create();
191             alert.show();
192         }

261         private static App getAppFromMenuItem(MenuItem item) {
262             final int id = item.getItemId();
263             for (final NavigationAppsEnum navApp : NavigationAppsEnum.values()) {

333         private static App getNavigationAppForId(final int navigationAppId) {
334             final List<NavigationAppsEnum> installedNavigationApps = getInstalledNavigationApps();
335     
336             for (final NavigationAppsEnum navigationApp : installedNavigationApps) {
337                 if (navigationApp.id == navigationAppId) {
338                     return navigationApp.app;
339                 }
340             }
341             // default navigation tool wasn't set already or couldn't be found (not installed any more for example)
342             return NavigationAppsEnum.COMPASS.app;
343         }

345         private static void invokeNavigation(final Activity activity, final Geocache cache, final Waypoint waypoint, final Geopoint destination, final App app) {
346             if (cache != null) {
347                 navigateCache(activity, cache, app);
348             }
349             else if (waypoint != null) {
350                 navigateWaypoint(activity, waypoint, app);
351             }
352             else {
353                 navigateGeopoint(activity, destination, app);
354             }
355         }
