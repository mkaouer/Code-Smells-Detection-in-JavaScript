10     public final class DistanceParser {
11     
12         private static final Pattern pattern = Pattern.compile("^([0-9.,]+)[ ]*(m|km|ft|yd|mi|)?$", Pattern.CASE_INSENSITIVE);
13     
14         /**
15          * Parse a distance string composed by a number and an optional suffix
16          * (such as "1.2km").
17          *
18          * @param distanceText
19          *            the string to analyze
20          * @return the distance in kilometers
21          *
22          * @throws NumberFormatException
23          *             if the given number is invalid
24          */
25         public static float parseDistance(String distanceText, final boolean metricUnit)
26                 throws NumberFormatException {
27             final MatcherWrapper matcher = new MatcherWrapper(pattern, distanceText);
28     
29             if (!matcher.find()) {
30                 throw new NumberFormatException(distanceText);
31             }
32     
33             final float value = Float.parseFloat(matcher.group(1).replace(',', '.'));
34             final String unit = matcher.group(2).toLowerCase(Locale.US);
35     
36             if (unit.equals("m") || (StringUtils.isEmpty(unit) && metricUnit)) {
37                 return value / 1000;
38             }
39             if (unit.equals("km")) {
40                 return value;
41             }
42             if (unit.equals("yd")) {
43                 return value * IConversion.YARDS_TO_KILOMETER;
44             }
45             if (unit.equals("mi")) {
46                 return value * IConversion.MILES_TO_KILOMETER;
47             }
48             return value * IConversion.FEET_TO_KILOMETER;
49         }
