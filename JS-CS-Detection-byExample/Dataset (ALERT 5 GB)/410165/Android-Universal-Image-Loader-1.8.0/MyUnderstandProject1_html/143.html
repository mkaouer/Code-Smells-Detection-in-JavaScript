16     package com.nostra13.universalimageloader.core.download;
17     
18     import java.io.IOException;
19     import java.io.InputStream;
20     import java.net.URI;
21     
22     import com.nostra13.universalimageloader.core.DisplayImageOptions;
23     
24     /**
25      * Provides retrieving of {@link InputStream} of image by URI.<br />
26      * Implementations have to be thread-safe.
27      * 
28      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
29      * @since 1.4.0
30      */
31     public interface ImageDownloader {
32     
33     	/// Supported URI schemes(protocols)
34     	String SCHEME_HTTP = "http";
35     	String SCHEME_HTTPS = "https";
36     	String SCHEME_FILE = "file";
37     	String SCHEME_CONTENT = "content";
38     	String SCHEME_ASSETS = "assets";
39     	String SCHEME_DRAWABLE = "drawable";
40     
41     	/**
42     	 * Retrieves {@link InputStream} of image by URI.
43     	 * 
44     	 * @param imageUri Image URI
45     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
46     	 *            DisplayImageOptions.extraForDownloader(Object)}; can be null
47     	 * @return {@link InputStream} of image
48     	 * @throws IOException if some I/O error occurs during getting image stream
49     	 * @throws UnsupportedOperationException if image URI has unsupported scheme(protocol)
50     	 */
51     	InputStream getStream(URI imageUri, Object extra) throws IOException;
52     }
