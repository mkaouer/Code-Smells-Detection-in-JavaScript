16     package com.nostra13.universalimageloader.core.display;
17     
18     import android.graphics.Bitmap;
19     import android.view.animation.AlphaAnimation;
20     import android.view.animation.DecelerateInterpolator;
21     import android.widget.ImageView;
22     
23     /**
24      * Displays image with "fade in" animation
25      * 
26      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
27      * @since 1.6.4
28      */
29     public class FadeInBitmapDisplayer implements BitmapDisplayer {
30     
31     	private final int durationMillis;
32     
33     	public FadeInBitmapDisplayer(int durationMillis) {
34     		this.durationMillis = durationMillis;
35     	}
36     
37     	@Override
38     	public Bitmap display(Bitmap bitmap, ImageView imageView) {
39     		imageView.setImageBitmap(bitmap);
40     
41     		animate(imageView, durationMillis);
42     
43     		return bitmap;
44     	}
45     
46     	/**
47     	 * Animates {@link ImageView} with "fade-in" effect
48     	 * 
49     	 * @param imageView {@link ImageView} which display image in
50     	 * @param durationMillis The length of the animation in milliseconds
51     	 */
52     	public static void animate(ImageView imageView, int durationMillis) {
53     		AlphaAnimation fadeImage = new AlphaAnimation(0, 1);
54     		fadeImage.setDuration(durationMillis);
55     		fadeImage.setInterpolator(new DecelerateInterpolator());
56     		imageView.startAnimation(fadeImage);
57     	}
58     
59     }
