39     /**
40      * Factory for providing of default options for {@linkplain ImageLoaderConfiguration configuration}
41      * 
42      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
43      * @since 1.5.6
44      */
45     public class DefaultConfigurationFactory {
46     
47     	/** Create {@linkplain HashCodeFileNameGenerator default implementation} of FileNameGenerator */
48     	public static FileNameGenerator createFileNameGenerator() {
49     		return new HashCodeFileNameGenerator();
50     	}
51     
52     	/** Create default implementation of {@link DisckCacheAware} depends on incoming parameters */
53     	public static DiscCacheAware createDiscCache(Context context, FileNameGenerator discCacheFileNameGenerator, int discCacheSize, int discCacheFileCount) {
54     		if (discCacheSize > 0) {
55     			File individualCacheDir = StorageUtils.getIndividualCacheDirectory(context);
56     			return new TotalSizeLimitedDiscCache(individualCacheDir, discCacheFileNameGenerator, discCacheSize);
57     		} else if (discCacheFileCount > 0) {
58     			File individualCacheDir = StorageUtils.getIndividualCacheDirectory(context);
59     			return new FileCountLimitedDiscCache(individualCacheDir, discCacheFileNameGenerator, discCacheFileCount);
60     		} else {
61     			File cacheDir = StorageUtils.getCacheDirectory(context);
62     			return new UnlimitedDiscCache(cacheDir, discCacheFileNameGenerator);
63     		}
64     	}
65     
66     	/** Create default implementation of {@link MemoryCacheAware} depends on incoming parameters */
67     	public static MemoryCacheAware<String, Bitmap> createMemoryCache(int memoryCacheSize, boolean denyCacheImageMultipleSizesInMemory) {
68     		MemoryCacheAware<String, Bitmap> memoryCache = new UsingFreqLimitedMemoryCache(memoryCacheSize);
69     		if (denyCacheImageMultipleSizesInMemory) {
70     			memoryCache = new FuzzyKeyMemoryCache<String, Bitmap>(memoryCache, MemoryCacheUtil.createFuzzyKeyComparator());
71     		}
72     		return memoryCache;
73     	}
74     
75     	/** Create default implementation of {@link ImageDownloader} - {@link BaseImageDownloader} */
76     	public static ImageDownloader createImageDownloader(Context context) {
77     		return new BaseImageDownloader(context);
78     	}
79     
80     	/** Create default implementation of {@link BitmapDisplayer} */
81     	public static BitmapDisplayer createBitmapDisplayer() {
82     		return new SimpleBitmapDisplayer();
83     	}
84     }
