16     package com.nostra13.universalimageloader.core;
17     
18     import static com.nostra13.universalimageloader.core.ImageLoader.LOG_DISPLAY_IMAGE_IN_IMAGEVIEW;
19     import static com.nostra13.universalimageloader.core.ImageLoader.LOG_TASK_CANCELLED;
20     import android.graphics.Bitmap;
21     import android.widget.ImageView;
22     
23     import com.nostra13.universalimageloader.core.assist.ImageLoadingListener;
24     import com.nostra13.universalimageloader.core.display.BitmapDisplayer;
25     import com.nostra13.universalimageloader.utils.L;
26     
27     /**
28      * Displays bitmap in {@link ImageView}. Must be called on UI thread.
29      * 
30      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
31      * @since 1.3.1
32      * @see ImageLoadingListener
33      * @see BitmapDisplayer
34      */
35     final class DisplayBitmapTask implements Runnable {
36     
37     	private final Bitmap bitmap;
38     	private final String imageUri;
39     	private final ImageView imageView;
40     	private final String memoryCacheKey;
41     	private final BitmapDisplayer displayer;
42     	private final ImageLoadingListener listener;
43     	private final ImageLoaderEngine engine;
44     
45     	private boolean loggingEnabled;
46     
47     	public DisplayBitmapTask(Bitmap bitmap, ImageLoadingInfo imageLoadingInfo, ImageLoaderEngine engine) {
48     		this.bitmap = bitmap;
49     		imageUri = imageLoadingInfo.uri;
50     		imageView = imageLoadingInfo.imageView;
51     		memoryCacheKey = imageLoadingInfo.memoryCacheKey;
52     		displayer = imageLoadingInfo.options.getDisplayer();
53     		listener = imageLoadingInfo.listener;
54     		this.engine = engine;
55     	}
56     
57     	public void run() {
58     		if (isViewWasReused()) {
59     			if (loggingEnabled) L.i(LOG_TASK_CANCELLED, memoryCacheKey);
60     			listener.onLoadingCancelled(imageUri, imageView);
61     		} else {
62     			if (loggingEnabled) L.i(LOG_DISPLAY_IMAGE_IN_IMAGEVIEW, memoryCacheKey);
63     			Bitmap displayedBitmap = displayer.display(bitmap, imageView);
64     			listener.onLoadingComplete(imageUri, imageView, displayedBitmap);
65     			engine.cancelDisplayTaskFor(imageView);
66     		}
67     	}
68     
69     	/** Checks whether memory cache key (image URI) for current ImageView is actual */
70     	private boolean isViewWasReused() {
71     		String currentCacheKey = engine.getLoadingUriForView(imageView);
72     		return !memoryCacheKey.equals(currentCacheKey);
73     	}
74     
75     	void setLoggingEnabled(boolean loggingEnabled) {
76     		this.loggingEnabled = loggingEnabled;
77     	}
78     }
