16     package com.nostra13.universalimageloader.cache.memory.impl;
17     
18     import java.lang.ref.Reference;
19     import java.lang.ref.WeakReference;
20     import java.util.Collections;
21     import java.util.HashMap;
22     import java.util.Map;
23     import java.util.Map.Entry;
24     import java.util.Set;
25     
26     import com.nostra13.universalimageloader.cache.memory.LimitedMemoryCache;
27     
28     import android.graphics.Bitmap;
29     
30     /**
31      * Limited {@link Bitmap bitmap} cache. Provides {@link Bitmap bitmaps} storing. Size of all stored bitmaps will not to
32      * exceed size limit. When cache reaches limit size then the bitmap which used the least frequently is deleted from
33      * cache.
34      * 
35      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
36      * @since 1.0.0
37      */
38     public class UsingFreqLimitedMemoryCache extends LimitedMemoryCache<String, Bitmap> {
39     	/**
40     	 * Contains strong references to stored objects (keys) and last object usage date (in milliseconds). If hard cache
41     	 * size will exceed limit then object with the least frequently usage is deleted (but it continue exist at
42     	 * {@link #softMap} and can be collected by GC at any time)
43     	 */
44     	private final Map<Bitmap, Integer> usingCounts = Collections.synchronizedMap(new HashMap<Bitmap, Integer>());
45     
46     	public UsingFreqLimitedMemoryCache(int sizeLimit) {
47     		super(sizeLimit);
48     	}
49     
50     	@Override
51     	public boolean put(String key, Bitmap value) {
52     		if (super.put(key, value)) {
53     			usingCounts.put(value, 0);
54     			return true;
55     		} else {
56     			return false;
57     		}
58     	}
59     
60     	@Override
61     	public Bitmap get(String key) {
62     		Bitmap value = super.get(key);
63     		// Increment usage count for value if value is contained in hardCahe
64     		if (value != null) {
65     			Integer usageCount = usingCounts.get(value);
66     			if (usageCount != null) {
67     				usingCounts.put(value, usageCount + 1);
68     			}
69     		}
70     		return value;
71     	}
72     
73     	@Override
74     	public void remove(String key) {
75     		Bitmap value = super.get(key);
76     		if (value != null) {
77     			usingCounts.remove(value);
78     		}
79     		super.remove(key);
80     	}
81     
82     	@Override
83     	public void clear() {
84     		usingCounts.clear();
85     		super.clear();
86     	}
87     
88     	@Override
89     	protected int getSize(Bitmap value) {
90     		return value.getRowBytes() * value.getHeight();
91     	}
92     
93     	@Override
94     	protected Bitmap removeNext() {
95     		Integer minUsageCount = null;
96     		Bitmap leastUsedValue = null;
97     		Set<Entry<Bitmap, Integer>> entries = usingCounts.entrySet();
98     		synchronized (usingCounts) {
99     			for (Entry<Bitmap, Integer> entry : entries) {
100     				if (leastUsedValue == null) {
101     					leastUsedValue = entry.getKey();
102     					minUsageCount = entry.getValue();
103     				} else {
104     					Integer lastValueUsage = entry.getValue();
105     					if (lastValueUsage < minUsageCount) {
106     						minUsageCount = lastValueUsage;
107     						leastUsedValue = entry.getKey();
108     					}
109     				}
110     			}
111     		}
112     		usingCounts.remove(leastUsedValue);
113     		return leastUsedValue;
114     	}
115     
116     	@Override
117     	protected Reference<Bitmap> createReference(Bitmap value) {
118     		return new WeakReference<Bitmap>(value);
119     	}
120     }
