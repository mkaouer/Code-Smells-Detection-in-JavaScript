16     package com.nostra13.example.universalimageloader;
17     
18     import android.app.ActivityManager;
19     import android.app.Application;
20     import android.content.Context;
21     import android.os.Build;
22     import android.os.StrictMode;
23     
24     import com.nostra13.example.universalimageloader.Constants.Config;
25     import com.nostra13.universalimageloader.cache.disc.naming.Md5FileNameGenerator;
26     import com.nostra13.universalimageloader.core.ImageLoader;
27     import com.nostra13.universalimageloader.core.ImageLoaderConfiguration;
28     import com.nostra13.universalimageloader.core.assist.QueueProcessingType;
29     
30     /**
31      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
32      */
33     public class UILApplication extends Application {
34     	@SuppressWarnings("unused")
35     	@Override
36     	public void onCreate() {
37     		if (Config.DEVELOPER_MODE && Build.VERSION.SDK_INT >= Build.VERSION_CODES.GINGERBREAD) {
38     			StrictMode.setThreadPolicy(new StrictMode.ThreadPolicy.Builder().detectAll().penaltyDialog().build());
39     			StrictMode.setVmPolicy(new StrictMode.VmPolicy.Builder().detectAll().penaltyDeath().build());
40     		}
41     
42     		super.onCreate();
43     
44     		initImageLoader(getApplicationContext());
45     	}
46     
47     	public static void initImageLoader(Context context) {
48     		int memoryCacheSize;
49     		if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.ECLAIR) {
50     			int memClass = ((ActivityManager) context.getSystemService(Context.ACTIVITY_SERVICE)).getMemoryClass();
51     			memoryCacheSize = (memClass / 8) * 1024 * 1024; // 1/8 of app memory limit 
52     		} else {
53     			memoryCacheSize = 2 * 1024 * 1024;
54     		}
55     
56     		// This configuration tuning is custom. You can tune every option, you may tune some of them, 
57     		// or you can create default configuration by
58     		//  ImageLoaderConfiguration.createDefault(this);
59     		// method.
60     		ImageLoaderConfiguration config = new ImageLoaderConfiguration.Builder(context)
61     			.threadPriority(Thread.NORM_PRIORITY - 2)
62     			.memoryCacheSize(memoryCacheSize)
63     			.denyCacheImageMultipleSizesInMemory()
64     			.discCacheFileNameGenerator(new Md5FileNameGenerator())
65     			.tasksProcessingOrder(QueueProcessingType.LIFO)
66     			.enableLogging() // Not necessary in common
67     			.build();
68     		// Initialize ImageLoader with configuration.
69     		ImageLoader.getInstance().init(config);
70     	}
71     }
