16     package com.nostra13.universalimageloader.core.download;
17     
18     import java.io.BufferedInputStream;
19     import java.io.ByteArrayInputStream;
20     import java.io.ByteArrayOutputStream;
21     import java.io.FileNotFoundException;
22     import java.io.IOException;
23     import java.io.InputStream;
24     import java.net.HttpURLConnection;
25     import java.net.URI;
26     import java.net.URLConnection;
27     
28     import android.content.ContentResolver;
29     import android.content.Context;
30     import android.graphics.Bitmap;
31     import android.graphics.Bitmap.CompressFormat;
32     import android.graphics.drawable.BitmapDrawable;
33     import android.net.Uri;
34     
35     import com.nostra13.universalimageloader.core.DisplayImageOptions;
36     import com.nostra13.universalimageloader.core.assist.FlushedInputStream;
37     
38     /**
39      * Provides retrieving of {@link InputStream} of image by URI from network or file system or app resources.<br />
40      * {@link URLConnection} is used to retrieve image stream from network.
41      * 
42      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
43      * 
44      * @see HttpClientImageDownloader
45      * @since 1.8.0
46      */
47     public class BaseImageDownloader implements ImageDownloader {
48     	/** {@value} */
49     	public static final int DEFAULT_HTTP_CONNECT_TIMEOUT = 5 * 1000; // milliseconds
50     	/** {@value} */
51     	public static final int DEFAULT_HTTP_READ_TIMEOUT = 20 * 1000; // milliseconds
52     
53     	protected static final int BUFFER_SIZE = 8 * 1024; // 8 Kb
54     
55     	private static final String ERROR_UNSUPPORTED_SCHEME = "UIL doesn't support scheme [%s] by default. You should implement this support byself";
56     
57     	protected static final String SCHEME_ASSETS_PREFIX = SCHEME_ASSETS + "://";
58     	protected static final String SCHEME_DRAWABLE_PREFIX = SCHEME_DRAWABLE + "://";
59     
60     	protected final Context context;
61     	protected final int connectTimeout;
62     	protected final int readTimeout;
63     
64     	public BaseImageDownloader(Context context) {
65     		this.context = context.getApplicationContext();
66     		this.connectTimeout = DEFAULT_HTTP_CONNECT_TIMEOUT;
67     		this.readTimeout = DEFAULT_HTTP_READ_TIMEOUT;
68     	}
69     
70     	public BaseImageDownloader(Context context, int connectTimeout, int readTimeout) {
71     		this.context = context.getApplicationContext();
72     		this.connectTimeout = connectTimeout;
73     		this.readTimeout = readTimeout;
74     	}
75     
76     	@Override
77     	public InputStream getStream(URI imageUri, Object extra) throws IOException {
78     		String scheme = imageUri.getScheme();
79     		if (SCHEME_HTTP.equals(scheme) || SCHEME_HTTPS.equals(scheme)) {
80     			return getStreamFromNetwork(imageUri, extra);
81     		} else if (SCHEME_FILE.equals(scheme)) {
82     			return getStreamFromFile(imageUri, extra);
83     		} else if (SCHEME_CONTENT.equals(scheme)) {
84     			return getStreamFromContent(imageUri, extra);
85     		} else if (SCHEME_ASSETS.equals(scheme)) {
86     			return getStreamFromAssets(imageUri, extra);
87     		} else if (SCHEME_DRAWABLE.equals(scheme)) {
88     			return getStreamFromDrawable(imageUri, extra);
89     		} else {
90     			return getStreamFromOtherSource(imageUri, extra);
91     		}
92     	}
93     
94     	/**
95     	 * Retrieves {@link InputStream} of image by URI (image is located in the network).
96     	 * 
97     	 * @param imageUri Image URI
98     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
99     	 *            DisplayImageOptions.extraForDownloader(Object)}; can be null
100     	 * @return {@link InputStream} of image
101     	 * @throws IOException if some I/O error occurs during network request or if no InputStream could be created for
102     	 *             URI.
103     	 */
104     	protected InputStream getStreamFromNetwork(URI imageUri, Object extra) throws IOException {
105     		HttpURLConnection conn = (HttpURLConnection) imageUri.toURL().openConnection();
106     		conn.setConnectTimeout(connectTimeout);
107     		conn.setReadTimeout(readTimeout);
108     		return new FlushedInputStream(conn.getInputStream(), BUFFER_SIZE);
109     	}
110     
111     	/**
112     	 * Retrieves {@link InputStream} of image by URI (image is located on the local file system or SD card).
113     	 * 
114     	 * @param imageUri Image URI
115     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
116     	 *            DisplayImageOptions.extraForDownloader(Object)}; can be null
117     	 * @return {@link InputStream} of image
118     	 * @throws IOException if some I/O error occurs reading from file system
119     	 */
120     	protected InputStream getStreamFromFile(URI imageUri, Object extra) throws IOException {
121     		return new BufferedInputStream(imageUri.toURL().openStream(), BUFFER_SIZE);
122     	}
123     
124     	/**
125     	 * Retrieves {@link InputStream} of image by URI (image is accessed using {@link ContentResolver}).
126     	 * 
127     	 * @param imageUri Image URI
128     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
129     	 *            DisplayImageOptions.extraForDownloader(Object)}; can be null
130     	 * @return {@link InputStream} of image
131     	 * @throws FileNotFoundException if the provided URI could not be opened
132     	 */
133     	protected InputStream getStreamFromContent(URI imageUri, Object extra) throws FileNotFoundException {
134     		ContentResolver res = context.getContentResolver();
135     		Uri uri = Uri.parse(imageUri.toString());
136     		return res.openInputStream(uri);
137     	}
138     
139     	/**
140     	 * Retrieves {@link InputStream} of image by URI (image is located in assets of application).
141     	 * 
142     	 * @param imageUri Image URI
143     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
144     	 *            DisplayImageOptions.extraForDownloader(Object)}; can be null
145     	 * @return {@link InputStream} of image
146     	 * @throws IOException if some I/O error occurs file reading
147     	 */
148     	protected InputStream getStreamFromAssets(URI imageUri, Object extra) throws IOException {
149     		String filePath = imageUri.toString().substring(SCHEME_ASSETS_PREFIX.length()); // Remove "assets://" prefix from image URI
150     		return context.getAssets().open(filePath);
151     	}
152     
153     	/**
154     	 * Retrieves {@link InputStream} of image by URI (image is located in drawable resources of application).
155     	 * 
156     	 * @param imageUri Image URI
157     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
158     	 *            DisplayImageOptions.extraForDownloader(Object)}; can be null
159     	 * @return {@link InputStream} of image
160     	 */
161     	protected InputStream getStreamFromDrawable(URI imageUri, Object extra) {
162     		String drawableIdString = imageUri.toString().substring(SCHEME_DRAWABLE_PREFIX.length()); // Remove "drawable://" prefix from image URI
163     		int drawableId = Integer.parseInt(drawableIdString);
164     		BitmapDrawable drawable = (BitmapDrawable) context.getResources().getDrawable(drawableId);
165     		Bitmap bitmap = drawable.getBitmap();
166     
167     		ByteArrayOutputStream os = new ByteArrayOutputStream();
168     		bitmap.compress(CompressFormat.PNG, 0, os);
169     		return new ByteArrayInputStream(os.toByteArray());
170     	}
171     
172     	/**
173     	 * Retrieves {@link InputStream} of image by URI from other source with unsupported scheme. Should be overriden by
174     	 * successors to implement image downloading from special sources.<br />
175     	 * This method is called only if image URI has unsupported scheme. Throws {@link UnsupportedOperationException} by
176     	 * default.
177     	 * 
178     	 * @param imageUri Image URI
179     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
180     	 *            DisplayImageOptions.extraForDownloader(Object)}; can be null
181     	 * @return {@link InputStream} of image
182     	 * @throws IOException if some I/O error occurs
183     	 * @throws UnsupportedOperationException @throws UnsupportedOperationException if image URI has unsupported
184     	 *             scheme(protocol)
185     	 */
186     	protected InputStream getStreamFromOtherSource(URI imageUri, Object extra) throws IOException {
187     		throw new UnsupportedOperationException(String.format(ERROR_UNSUPPORTED_SCHEME, imageUri.getScheme()));
188     	}
189     }
