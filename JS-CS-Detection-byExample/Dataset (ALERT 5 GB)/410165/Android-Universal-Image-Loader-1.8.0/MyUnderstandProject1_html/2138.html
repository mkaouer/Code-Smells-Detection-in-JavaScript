16     package com.nostra13.universalimageloader.utils;
17     
18     import java.io.File;
19     import java.io.IOException;
20     
21     import android.content.Context;
22     import android.os.Environment;
23     
24     /**
25      * Provides application storage paths
26      * 
27      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
28      * @since 1.0.0
29      */
30     public final class StorageUtils {
31     
32     	private static final String INDIVIDUAL_DIR_NAME = "uil-images";
33     
34     	private StorageUtils() {
35     	}
36     
37     	/**
38     	 * Returns application cache directory. Cache directory will be created on SD card
39     	 * <i>("/Android/data/[app_package_name]/cache")</i> if card is mounted. Else - Android defines cache directory on
40     	 * device's file system.
41     	 * 
42     	 * @param context Application context
43     	 * @return Cache {@link File directory}
44     	 */
45     	public static File getCacheDirectory(Context context) {
46     		File appCacheDir = null;
47     		if (Environment.getExternalStorageState().equals(android.os.Environment.MEDIA_MOUNTED)) {
48     			appCacheDir = getExternalCacheDir(context);
49     		}
50     		if (appCacheDir == null) {
51     			appCacheDir = context.getCacheDir();
52     		}
53     		return appCacheDir;
54     	}
55     
56     	/**
57     	 * Returns individual application cache directory (for only image caching from ImageLoader). Cache directory will be
58     	 * created on SD card <i>("/Android/data/[app_package_name]/cache/uil-images")</i> if card is mounted. Else -
59     	 * Android defines cache directory on device's file system.
60     	 * 
61     	 * @param context Application context
62     	 * @return Cache {@link File directory}
63     	 */
64     	public static File getIndividualCacheDirectory(Context context) {
65     		File cacheDir = getCacheDirectory(context);
66     		File individualCacheDir = new File(cacheDir, INDIVIDUAL_DIR_NAME);
67     		if (!individualCacheDir.exists()) {
68     			if (!individualCacheDir.mkdir()) {
69     				individualCacheDir = cacheDir;
70     			}
71     		}
72     		return individualCacheDir;
73     	}
74     
75     	/**
76     	 * Returns specified application cache directory. Cache directory will be created on SD card by defined path if card
77     	 * is mounted. Else - Android defines cache directory on device's file system.
78     	 * 
79     	 * @param context Application context
80     	 * @param cacheDir Cache directory path (e.g.: "AppCacheDir", "AppDir/cache/images")
81     	 * @return Cache {@link File directory}
82     	 */
83     	public static File getOwnCacheDirectory(Context context, String cacheDir) {
84     		File appCacheDir = null;
85     		if (Environment.getExternalStorageState().equals(android.os.Environment.MEDIA_MOUNTED)) {
86     			appCacheDir = new File(Environment.getExternalStorageDirectory(), cacheDir);
87     		}
88     		if (appCacheDir == null || (!appCacheDir.exists() && !appCacheDir.mkdirs())) {
89     			appCacheDir = context.getCacheDir();
90     		}
91     		return appCacheDir;
92     	}
93     
94     	private static File getExternalCacheDir(Context context) {
95     		File dataDir = new File(new File(Environment.getExternalStorageDirectory(), "Android"), "data");
96     		File appCacheDir = new File(new File(dataDir, context.getPackageName()), "cache");
97     		if (!appCacheDir.exists()) {
98     			if (!appCacheDir.mkdirs()) {
99     				L.w("Unable to create external cache directory");
100     				return null;
101     			}
102     			try {
103     				new File(appCacheDir, ".nomedia").createNewFile();
104     			} catch (IOException e) {
105     				L.i("Can't create \".nomedia\" file in application external cache directory");
106     			}
107     		}
108     		return appCacheDir;
109     	}
110     }
