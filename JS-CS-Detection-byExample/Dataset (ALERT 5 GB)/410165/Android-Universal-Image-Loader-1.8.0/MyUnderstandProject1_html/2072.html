37     /**
38      * Decodes images to {@link Bitmap}, scales them to needed size
39      * 
40      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
41      * @since 1.0.0
42      * @see ImageScaleType
43      * @see ViewScaleType
44      * @see ImageDownloader
45      * @see DisplayImageOptions
46      */
47     class ImageDecoder {
48     
49     	private final URI imageUri;
50     	private final ImageDownloader imageDownloader;
51     	private final DisplayImageOptions displayOptions;
52     
53     	private boolean loggingEnabled;
54     
55     	/**
56     	 * @param imageUri Image URI (<b>i.e.:</b> "http://site.com/image.png", "file:///mnt/sdcard/image.png")
57     	 * @param imageDownloader Image downloader
58     	 * 
59     	 */
60     	ImageDecoder(URI imageUri, ImageDownloader imageDownloader, DisplayImageOptions options) {
61     		this.imageUri = imageUri;
62     		this.imageDownloader = imageDownloader;
63     		this.displayOptions = options;
64     	}
65     
66     	/**
67     	 * Decodes image from URI into {@link Bitmap}. Image is scaled close to incoming {@link ImageSize image size} during
68     	 * decoding (depend on incoming image scale type).
69     	 * 
70     	 * @param targetSize Image size to scale to during decoding
71     	 * @param scaleType {@link ImageScaleType Image scale type}
72     	 * @param viewScaleType {@link ViewScaleType View scale type}
73     	 * 
74     	 * @return Decoded bitmap
75     	 * @throws IOException
76     	 * @throws UnsupportedOperationException
77     	 */
78     	public Bitmap decode(ImageSize targetSize, ImageScaleType scaleType, ViewScaleType viewScaleType) throws IOException {
79     		Options decodeOptions = getBitmapOptionsForImageDecoding(targetSize, scaleType, viewScaleType);
80     		InputStream imageStream = imageDownloader.getStream(imageUri, displayOptions.getExtraForDownloader());
81     		Bitmap subsampledBitmap;
82     		try {
83     			subsampledBitmap = BitmapFactory.decodeStream(imageStream, null, decodeOptions);
84     		} finally {
85     			IoUtils.closeSilently(imageStream);
86     		}
87     		if (subsampledBitmap == null) {
88     			log(LOG_CANT_DECODE_IMAGE, imageUri);
89     			return null;
90     		}
91     
92     		// Scale to exact size if need
93     		if (scaleType == ImageScaleType.EXACTLY || scaleType == ImageScaleType.EXACTLY_STRETCHED) {
94     			subsampledBitmap = scaleImageExactly(subsampledBitmap, targetSize, scaleType, viewScaleType);
95     		}
96     
97     		return subsampledBitmap;
98     	}
99     
100     	private Options getBitmapOptionsForImageDecoding(ImageSize targetSize, ImageScaleType scaleType, ViewScaleType viewScaleType) throws IOException {
101     		Options decodeOptions = new Options();
102     		decodeOptions.inSampleSize = scaleType == ImageScaleType.NONE ? 1 : computeImageScale(targetSize, scaleType, viewScaleType);
103     		decodeOptions.inPreferredConfig = displayOptions.getBitmapConfig();
104     		return decodeOptions;
105     	}
106     
107     	private int computeImageScale(ImageSize targetSize, ImageScaleType scaleType, ViewScaleType viewScaleType) throws IOException {
108     		int targetWidth = targetSize.getWidth();
109     		int targetHeight = targetSize.getHeight();
110     
111     		// decode image size
112     		Options options = new Options();
113     		options.inJustDecodeBounds = true;
114     		InputStream imageStream = imageDownloader.getStream(imageUri, displayOptions.getExtraForDownloader());
115     		try {
116     			BitmapFactory.decodeStream(imageStream, null, options);
117     		} finally {
118     			IoUtils.closeSilently(imageStream);
119     		}
120     
121     		int scale = 1;
122     		int imageWidth = options.outWidth;
123     		int imageHeight = options.outHeight;
124     		int widthScale = imageWidth / targetWidth;
125     		int heightScale = imageHeight / targetHeight;
126     
127     		if (viewScaleType == ViewScaleType.FIT_INSIDE) {
128     			if (scaleType == ImageScaleType.IN_SAMPLE_POWER_OF_2) {
129     				while (imageWidth / 2 >= targetWidth || imageHeight / 2 >= targetHeight) { // ||
130     					imageWidth /= 2;
131     					imageHeight /= 2;
132     					scale *= 2;
133     				}
134     			} else {
135     				scale = Math.max(widthScale, heightScale); // max
136     			}
137     		} else { // ViewScaleType.CROP
138     			if (scaleType == ImageScaleType.IN_SAMPLE_POWER_OF_2) {
139     				while (imageWidth / 2 >= targetWidth && imageHeight / 2 >= targetHeight) { // &&
140     					imageWidth /= 2;
141     					imageHeight /= 2;
142     					scale *= 2;
143     				}
144     			} else {
145     				scale = Math.min(widthScale, heightScale); // min
146     			}
147     		}
148     
149     		if (scale < 1) {
150     			scale = 1;
151     		}
152     
153     		log(LOG_IMAGE_SUBSAMPLING, imageWidth, imageHeight, targetWidth, targetHeight, scale);
154     		return scale;
155     	}
156     
157     	private Bitmap scaleImageExactly(Bitmap subsampledBitmap, ImageSize targetSize, ImageScaleType scaleType, ViewScaleType viewScaleType) {
158     		float srcWidth = subsampledBitmap.getWidth();
159     		float srcHeight = subsampledBitmap.getHeight();
160     
161     		float widthScale = srcWidth / targetSize.getWidth();
162     		float heightScale = srcHeight / targetSize.getHeight();
163     
164     		int destWidth;
165     		int destHeight;
166     		if ((viewScaleType == ViewScaleType.FIT_INSIDE && widthScale >= heightScale) || (viewScaleType == ViewScaleType.CROP && widthScale < heightScale)) {
167     			destWidth = targetSize.getWidth();
168     			destHeight = (int) (srcHeight / widthScale);
169     		} else {
170     			destWidth = (int) (srcWidth / heightScale);
171     			destHeight = targetSize.getHeight();
172     		}
173     
174     		Bitmap scaledBitmap;
175     		if ((scaleType == ImageScaleType.EXACTLY && destWidth < srcWidth && destHeight < srcHeight)
176     				|| (scaleType == ImageScaleType.EXACTLY_STRETCHED && destWidth != srcWidth && destHeight != srcHeight)) {
177     			scaledBitmap = Bitmap.createScaledBitmap(subsampledBitmap, destWidth, destHeight, true);
178     			if (scaledBitmap != subsampledBitmap) {
179     				subsampledBitmap.recycle();
180     			}
181     			log(LOG_IMAGE_SCALED, (int) srcWidth, (int) srcHeight, destWidth, destHeight);
182     		} else {
183     			scaledBitmap = subsampledBitmap;
184     		}
185     
186     		return scaledBitmap;
187     	}
188     
189     	void setLoggingEnabled(boolean loggingEnabled) {
190     		this.loggingEnabled = loggingEnabled;
191     	}
192     
193     	private void log(String message, Object... args) {
194     		if (loggingEnabled) L.i(message, args);
195     	}
196     }
