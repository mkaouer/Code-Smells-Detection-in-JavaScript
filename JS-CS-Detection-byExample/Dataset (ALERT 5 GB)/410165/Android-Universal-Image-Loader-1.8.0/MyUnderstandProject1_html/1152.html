23     import android.graphics.PorterDuffXfermode;
24     import android.graphics.Rect;
25     import android.graphics.RectF;
26     import android.widget.ImageView;

28     import com.nostra13.universalimageloader.utils.L;

30     /**
31      * Displays bitmap with rounded corners. <br />
32      * <b>NOTE:</b> It's strongly recommended your {@link ImageView} has defined width (<i>layout_width</i>) and height
33      * (<i>layout_height</i>) .<br />
34      * <b>NOTE:</b> New {@link Bitmap} object is created for displaying. So this class needs more memory and can cause
35      * {@link OutOfMemoryError}.
36      * 
37      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
38      * @since 1.5.6
39      */
40     public class RoundedBitmapDisplayer implements BitmapDisplayer {
41     
42     	private final int roundPixels;
43     
44     	public RoundedBitmapDisplayer(int roundPixels) {
45     		this.roundPixels = roundPixels;
46     	}
47     
48     	@Override
49     	public Bitmap display(Bitmap bitmap, ImageView imageView) {
50     		Bitmap roundedBitmap = roundCorners(bitmap, imageView, roundPixels);
51     		imageView.setImageBitmap(roundedBitmap);
52     		return roundedBitmap;
53     	}
54     
55     	/**
56     	 * Process incoming {@linkplain Bitmap} to make rounded corners according to target {@link ImageView}.<br />
57     	 * This method <b>doesn't display</b> result bitmap in {@link ImageView}
58     	 * 
59     	 * @param bitmap Incoming Bitmap to process
60     	 * @param imageView Target {@link ImageView} to display bitmap in
61     	 * @param roundPixels
62     	 * @return Result bitmap with rounded corners
63     	 */
64     	public static Bitmap roundCorners(Bitmap bitmap, ImageView imageView, int roundPixels) {
65     		Bitmap roundBitmap;
66     
67     		int bw = bitmap.getWidth();
68     		int bh = bitmap.getHeight();
69     		int vw = imageView.getWidth();
70     		int vh = imageView.getHeight();
71     		if (vw <= 0) vw = bw;
72     		if (vh <= 0) vh = bh;
73     
74     		int width, height;
75     		Rect srcRect;
76     		Rect destRect;
77     		switch (imageView.getScaleType()) {
78     			case CENTER_INSIDE:
79     				float vRation = (float) vw / vh;
80     				float bRation = (float) bw / bh;
81     				int destWidth;
82     				int destHeight;
83     				if (vRation > bRation) {
84     					destHeight = Math.min(vh, bh);
85     					destWidth = (int) (bw / ((float) bh / destHeight));
86     				} else {
87     					destWidth = Math.min(vw, bw);
88     					destHeight = (int) (bh / ((float) bw / destWidth));
89     				}
90     				int x = (vw - destWidth) / 2;
91     				int y = (vh - destHeight) / 2;
92     				srcRect = new Rect(0, 0, bw, bh);
93     				destRect = new Rect(x, y, x + destWidth, y + destHeight);
94     				width = vw;
95     				height = vh;
96     				break;
97     			case FIT_CENTER:
98     			case FIT_START:
99     			case FIT_END:
100     			default:
101     				vRation = (float) vw / vh;
102     				bRation = (float) bw / bh;
103     				if (vRation > bRation) {
104     					width = (int) (bw / ((float) bh / vh));
105     					height = vh;
106     				} else {
107     					width = vw;
108     					height = (int) (bh / ((float) bw / vw));
109     				}
110     				srcRect = new Rect(0, 0, bw, bh);
111     				destRect = new Rect(0, 0, width, height);
112     				break;
113     			case CENTER_CROP:
114     				vRation = (float) vw / vh;
115     				bRation = (float) bw / bh;
116     				int srcWidth;
117     				int srcHeight;
118     				if (vRation > bRation) {
119     					srcWidth = bw;
120     					srcHeight = (int) (vh * ((float) bw / vw));
121     					x = 0;
122     					y = (bh - srcHeight) / 2;
123     				} else {
124     					srcWidth = (int) (vw * ((float) bh / vh));
125     					srcHeight = bh;
126     					x = (bw - srcWidth) / 2;
127     					y = 0;
128     				}
129     				width = Math.min(vw, bw);
130     				height = Math.min(vh, bh);
131     				srcRect = new Rect(x, y, x + srcWidth, y + srcHeight);
132     				destRect = new Rect(0, 0, width, height);
133     				break;
134     			case FIT_XY:
135     				width = vw;
136     				height = vh;
137     				srcRect = new Rect(0, 0, bw, bh);
138     				destRect = new Rect(0, 0, width, height);
139     				break;
140     			case CENTER:
141     			case MATRIX:
142     				width = Math.min(vw, bw);
143     				height = Math.min(vh, bh);
144     				x = (bw - width) / 2;
145     				y = (bh - height) / 2;
146     				srcRect = new Rect(x, y, x + width, y + height);
147     				destRect = new Rect(0, 0, width, height);
148     				break;
149     		}
150     
151     		try {
152     			roundBitmap = getRoundedCornerBitmap(bitmap, roundPixels, srcRect, destRect, width, height);
153     		} catch (OutOfMemoryError e) {
154     			L.e(e, "Can't create bitmap with rounded corners. Not enough memory.");
155     			roundBitmap = bitmap;
156     		}
157     
158     		return roundBitmap;
159     	}
160     
161     	private static Bitmap getRoundedCornerBitmap(Bitmap bitmap, int roundPixels, Rect srcRect, Rect destRect, int width, int height) {
162     		Bitmap output = Bitmap.createBitmap(width, height, Config.ARGB_8888);
163     		Canvas canvas = new Canvas(output);
164     
165     		final Paint paint = new Paint();
166     		final RectF destRectF = new RectF(destRect);
167     
168     		paint.setAntiAlias(true);
169     		canvas.drawARGB(0, 0, 0, 0);
170     		paint.setColor(0xFF000000);
171     		canvas.drawRoundRect(destRectF, roundPixels, roundPixels, paint);
172     
173     		paint.setXfermode(new PorterDuffXfermode(PorterDuff.Mode.SRC_IN));
174     		canvas.drawBitmap(bitmap, srcRect, destRectF, paint);
175     
176     		return output;
177     	}
178     }
