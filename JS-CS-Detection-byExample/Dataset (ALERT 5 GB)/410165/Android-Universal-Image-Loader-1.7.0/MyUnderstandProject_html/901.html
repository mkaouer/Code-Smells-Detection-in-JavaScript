42     	 * @param cacheDir

47     	 * @param sizeLimit
48     	 *            Cache limit value. If cache exceeds this limit then file with the most oldest last usage date will be
49     	 *            deleted.

51     	public LimitedDiscCache(File cacheDir, FileNameGenerator fileNameGenerator, int sizeLimit) {
52     		super(cacheDir, fileNameGenerator);
53     		this.sizeLimit = sizeLimit;
54     		calculateCacheSizeAndFillUsageMap();
55     	}

57     	private void calculateCacheSizeAndFillUsageMap() {
58     		int size = 0;
59     		File[] cachedFiles = getCacheDir().listFiles();
60     		for (File cachedFile : cachedFiles) {
61     			size += getSize(cachedFile);
62     			lastUsageDates.put(cachedFile, cachedFile.lastModified());
63     		}
64     		cacheSize = size;
65     	}

67     	@Override
68     	public void put(String key, File file) {
69     		int valueSize = getSize(file);
70     		while (cacheSize + valueSize > sizeLimit) {
71     			int freedSize = removeNext();
72     			if (freedSize == 0) break; // cache is empty (have nothing to delete)
73     			cacheSize -= freedSize;
74     		}
75     		cacheSize += valueSize;
76     
77     		Long currentTime = System.currentTimeMillis();
78     		file.setLastModified(currentTime);
79     		lastUsageDates.put(file, currentTime);
80     	}

83     	public File get(String key) {
84     		File file = super.get(key);
85     
86     		Long currentTime = System.currentTimeMillis();
87     		file.setLastModified(currentTime);
88     		lastUsageDates.put(file, currentTime);
89     
90     		return file;
91     	}

94     	public void clear() {
95     		lastUsageDates.clear();
96     		cacheSize = 0;
97     		super.clear();
98     	}

100     	/** Remove next file and returns it's size */
101     	private int removeNext() {
102     		if (lastUsageDates.isEmpty()) {
103     			return 0;
104     		}
105     
106     		Long oldestUsage = null;
107     		File mostLongUsedFile = null;
108     		Set<Entry<File, Long>> entries = lastUsageDates.entrySet();
109     		synchronized (lastUsageDates) {
110     			for (Entry<File, Long> entry : entries) {
111     				if (mostLongUsedFile == null) {
112     					mostLongUsedFile = entry.getKey();
113     					oldestUsage = entry.getValue();
114     				} else {
115     					Long lastValueUsage = entry.getValue();
116     					if (lastValueUsage < oldestUsage) {
117     						oldestUsage = lastValueUsage;
118     						mostLongUsedFile = entry.getKey();
119     					}
120     				}
121     			}
122     		}
123     
124     		int fileSize = getSize(mostLongUsedFile);
125     		if (mostLongUsedFile.delete()) {
126     			lastUsageDates.remove(mostLongUsedFile);
127     		}
128     		return fileSize;
129     	}
