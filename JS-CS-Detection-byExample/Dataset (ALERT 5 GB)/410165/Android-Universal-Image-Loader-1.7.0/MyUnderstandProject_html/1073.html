14      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
15      * @see MemoryCacheAware

17     public class LimitedAgeMemoryCache<K, V> implements MemoryCacheAware<K, V> {
18     
19     	private final MemoryCacheAware<K, V> cache;
20     
21     	private final long maxAge;
22     	private final Map<K, Long> loadingDates = Collections.synchronizedMap(new HashMap<K, Long>());
23     
24     	/**
25     	 * @param cache
26     	 *            Wrapped memory cache
27     	 * @param maxAge
28     	 *            Max object age <b>(in seconds)</b>. If object age will exceed this value then it'll be removed from
29     	 *            cache on next treatment (and therefore be reloaded).
30     	 */
31     	public LimitedAgeMemoryCache(MemoryCacheAware<K, V> cache, long maxAge) {
32     		this.cache = cache;
33     		this.maxAge = maxAge * 1000; // to milliseconds
34     	}
35     
36     	@Override
37     	public boolean put(K key, V value) {
38     		boolean putSuccesfully = cache.put(key, value);
39     		if (putSuccesfully) {
40     			loadingDates.put(key, System.currentTimeMillis());
41     		}
42     		return putSuccesfully;
43     	}
44     
45     	@Override
46     	public V get(K key) {
47     		Long loadingDate = loadingDates.get(key);
48     		if (loadingDate != null && System.currentTimeMillis() - loadingDate > maxAge) {
49     			cache.remove(key);
50     			loadingDates.remove(key);
51     		}
52     
53     		return cache.get(key);
54     	}
55     
56     	@Override
57     	public void remove(K key) {
58     		cache.remove(key);
59     		loadingDates.remove(key);
60     	}
61     
62     	@Override
63     	public Collection<K> keys() {
64     		return cache.keys();
65     	}
66     
67     	@Override
68     	public void clear() {
69     		cache.clear();
