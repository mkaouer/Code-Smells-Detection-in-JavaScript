173     	public void displayImage(String uri, ImageView imageView, DisplayImageOptions options, ImageLoadingListener listener) {
174     		if (configuration == null) {
175     			throw new RuntimeException(ERROR_NOT_INIT);
176     		}
177     		if (imageView == null) {
178     			L.w(TAG, ERROR_WRONG_ARGUMENTS);
179     			return;
180     		}
181     		if (listener == null) {
182     			listener = emptyListener;
183     		}
184     		if (options == null) {
185     			options = configuration.defaultDisplayImageOptions;
186     		}
187     
188     		if (uri == null || uri.length() == 0) {
189     			cacheKeysForImageViews.remove(imageView.hashCode());
190     			listener.onLoadingStarted();
191     			if (options.isShowImageForEmptyUri()) {
192     				imageView.setImageResource(options.getImageForEmptyUri());
193     			} else {
194     				imageView.setImageBitmap(null);
195     			}
196     			listener.onLoadingComplete(null);
197     			return;
198     		}
199     
200     		ImageSize targetSize = getImageSizeScaleTo(imageView);
201     		String memoryCacheKey = MemoryCacheUtil.generateKey(uri, targetSize);
202     		cacheKeysForImageViews.put(imageView.hashCode(), memoryCacheKey);
203     
204     		Bitmap bmp = configuration.memoryCache.get(memoryCacheKey);
205     		if (bmp != null && !bmp.isRecycled()) {
206     			if (configuration.loggingEnabled) L.i(LOG_LOAD_IMAGE_FROM_MEMORY_CACHE, memoryCacheKey);
207     			listener.onLoadingStarted();
208     			options.getDisplayer().display(bmp, imageView);
209     			listener.onLoadingComplete(bmp);
210     		} else {
211     			listener.onLoadingStarted();
212     
213     			if (options.isShowStubImage()) {
214     				imageView.setImageResource(options.getStubImage());
215     			} else {
216     				if (options.isResetViewBeforeLoading()) {
217     					imageView.setImageBitmap(null);
218     				}
219     			}
220     
221     			initExecutorsIfNeed();
222     			ImageLoadingInfo imageLoadingInfo = new ImageLoadingInfo(uri, imageView, targetSize, options, listener, getLockForUri(uri));

225     			if (isImageCachedOnDisc) {
226     				cachedImageLoadingExecutor.submit(displayImageTask);
227     			} else {
228     				imageLoadingExecutor.submit(displayImageTask);
229     			}

320     		if (minImageSize == null) {
321     			minImageSize = new ImageSize(configuration.maxImageWidthForMemoryCache, configuration.maxImageHeightForMemoryCache);
322     		}
323     		if (options == null) {
324     			options = configuration.defaultDisplayImageOptions;
325     		}
326     
327     		DisplayImageOptions optionsWithFakeDisplayer;
328     		if (options.getDisplayer() instanceof FakeBitmapDisplayer) {
329     			optionsWithFakeDisplayer = options;
330     		} else {
331     			optionsWithFakeDisplayer = new DisplayImageOptions.Builder()
332     				.cloneFrom(options)
333     				.displayer(fakeBitmapDisplayer)
334     				.build();
335     		}
336     
337     		ImageView fakeImage = new ImageView(context);
338     		fakeImage.setLayoutParams(new LayoutParams(minImageSize.getWidth(), minImageSize.getHeight()));
339     		fakeImage.setScaleType(ScaleType.CENTER_CROP);
340     
341     		displayImage(uri, fakeImage, optionsWithFakeDisplayer, listener);
342     	}

431     	/**
432     	 * Defines image size for loading at memory (for memory economy) by {@link ImageView} parameters.<br />
433     	 * Size computing algorithm:<br />
434     	 * 1) Get <b>layout_width</b> and <b>layout_height</b>. If both of them haven't exact value then go to step #2.</br>
435     	 * 2) Get <b>maxWidth</b> and <b>maxHeight</b>. If both of them are not set then go to step #3.<br />
436     	 * 3) Get <b>maxImageWidthForMemoryCache</b> and <b>maxImageHeightForMemoryCache</b> from configuration. If both of
437     	 * them are not set then go to step #3.<br />
438     	 * 4) Get device screen dimensions.
439     	 */
440     	private ImageSize getImageSizeScaleTo(ImageView imageView) {
441     		DisplayMetrics displayMetrics = imageView.getContext().getResources().getDisplayMetrics();
442     
443     		LayoutParams params = imageView.getLayoutParams();
444     		int width = params.width; // Get layout width parameter
445     		if (width <= 0) width = getFieldValue(imageView, "mMaxWidth"); // Check maxWidth parameter
446     		if (width <= 0) width = configuration.maxImageWidthForMemoryCache;
447     		if (width <= 0) width = displayMetrics.widthPixels;
448     
449     		int height = params.height; // Get layout height parameter
450     		if (height <= 0) height = getFieldValue(imageView, "mMaxHeight"); // Check maxHeight parameter
451     		if (height <= 0) height = configuration.maxImageHeightForMemoryCache;
452     		if (height <= 0) height = displayMetrics.heightPixels;
453     
454     		return new ImageSize(width, height);
455     	}

457     	private int getFieldValue(Object object, String fieldName) {
458     		int value = 0;
459     		try {
460     			Field field = ImageView.class.getDeclaredField(fieldName);
461     			field.setAccessible(true);
462     			int fieldValue = (Integer) field.get(object);
463     			if (fieldValue > 0 && fieldValue < Integer.MAX_VALUE) {
464     				value = fieldValue;
465     			}
466     		} catch (Exception e) {
467     			L.e(e);
468     		}
469     		return value;
470     	}
