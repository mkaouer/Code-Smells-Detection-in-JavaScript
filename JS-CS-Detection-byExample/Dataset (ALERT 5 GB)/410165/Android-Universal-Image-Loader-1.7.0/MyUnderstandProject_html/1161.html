18

30     	public DisplayBitmapTask(Bitmap bitmap, ImageLoadingInfo imageLoadingInfo) {
31     		this.bitmap = bitmap;
32     		imageView = imageLoadingInfo.imageView;
33     		memoryCacheKey = imageLoadingInfo.memoryCacheKey;
34     		bitmapDisplayer = imageLoadingInfo.options.getDisplayer();
35     		listener = imageLoadingInfo.listener;
36     	}
37     
38     	public void run() {
39     		if (isViewWasReused()) {
40     			if (loggingEnabled) L.i(LOG_TASK_CANCELLED, memoryCacheKey);
41     			listener.onLoadingCancelled();
42     		} else {
43     			if (loggingEnabled) L.i(LOG_DISPLAY_IMAGE_IN_IMAGEVIEW, memoryCacheKey);
44     			Bitmap displayedBitmap = bitmapDisplayer.display(bitmap, imageView);
45     			listener.onLoadingComplete(displayedBitmap);
46     			ImageLoader.getInstance().cancelDisplayTask(imageView);
47     		}
48     	}
49     
50     	/** Checks whether memory cache key (image URI) for current ImageView is actual */
51     	private boolean isViewWasReused() {
52     		String currentCacheKey = ImageLoader.getInstance().getLoadingUriForView(imageView);
53     		return !memoryCacheKey.equals(currentCacheKey);
54     	}
55     
56     	void setLoggingEnabled(boolean loggingEnabled) {
