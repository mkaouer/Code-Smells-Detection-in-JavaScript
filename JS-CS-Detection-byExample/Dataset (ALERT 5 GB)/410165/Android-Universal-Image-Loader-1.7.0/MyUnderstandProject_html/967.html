30     			queue.add(value);
31     			return true;
32     		} else {
33     			return false;
34     		}
35     	}

37     	@Override
38     	public void remove(String key) {
39     		Bitmap value = super.get(key);
40     		if (value != null) {
41     			queue.remove(value);
42     		}
43     		super.remove(key);
44     	}
45     
46     	@Override

48     		queue.clear();
49     		super.clear();
50     	}
