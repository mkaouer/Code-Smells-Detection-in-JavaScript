21     public class LRULimitedMemoryCache extends LimitedMemoryCache<String, Bitmap> {

39     			return false;
40     		}
41     	}
42

44     	public Bitmap get(String key) {

46     		return super.get(key);
47     	}
48

50     	public void remove(String key) {
51     		lruCache.remove(key);
52     		super.remove(key);

59     	}

64     	}

68     		Bitmap mostLongUsedValue = null;
69     		synchronized (lruCache) {
70     			Iterator<Entry<String, Bitmap>> it = lruCache.entrySet().iterator();
71     			if (it.hasNext()) {
72     				Entry<String, Bitmap> entry = it.next();
73     				mostLongUsedValue = entry.getValue();
74     				it.remove();
