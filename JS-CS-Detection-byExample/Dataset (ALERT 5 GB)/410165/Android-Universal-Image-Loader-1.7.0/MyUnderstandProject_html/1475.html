9     /**
10      * Decorator for {@link MemoryCacheAware}. Provides special feature for cache: some different keys are considered as
11      * equals (using {@link Comparator comparator}). And when you try to put some value into cache by key so entries with
12      * "equals" keys will be removed from cache before.<br />
13      * <b>NOTE:</b> Used for internal needs. Normally you don't need to use this class.
14      * 
15      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
16      */
17     public class FuzzyKeyMemoryCache<K, V> implements MemoryCacheAware<K, V> {
18     
19     	private final MemoryCacheAware<K, V> cache;
20     	private final Comparator<K> keyComparator;
21     
22     	public FuzzyKeyMemoryCache(MemoryCacheAware<K, V> cache, Comparator<K> keyComparator) {
23     		this.cache = cache;
24     		this.keyComparator = keyComparator;
25     	}
26     
27     	@Override
28     	public synchronized boolean put(K key, V value) {
29     		// Search equal key and remove this entry
30     		K keyToRemove = null;
31     		for (Iterator<K> it = cache.keys().iterator(); it.hasNext();) {
32     			K cacheKey = it.next();
33     			if (keyComparator.compare(key, cacheKey) == 0) {
34     				keyToRemove = cacheKey;
35     			}
36     		}
37     		cache.remove(keyToRemove);
38     
39     		return cache.put(key, value);
40     	}
41     
42     	@Override
43     	public synchronized V get(K key) {
44     		return cache.get(key);
45     	}
46     
47     	@Override
48     	public synchronized void remove(K key) {
49     		cache.remove(key);
50     	}
51     
52     	@Override
53     	public synchronized void clear() {
54     		cache.clear();
55     	}
56     
57     	@Override
58     	public synchronized Collection<K> keys() {
59     		return cache.keys();
60     	}
