25     	/**
26     	 * Generates key for memory cache for incoming image (URI + size).<br />
27     	 * Pattern for cache key - {@value #MEMORY_CACHE_KEY_FORMAT}, where (1) - image URI, (2) - image width, (3) - image
28     	 * height.
29     	 */
30     	public static String generateKey(String imageUri, ImageSize targetSize) {
31     		return String.format(MEMORY_CACHE_KEY_FORMAT, imageUri, targetSize.getWidth(), targetSize.getHeight());
32     	}

34     	public static Comparator<String> createFuzzyKeyComparator() {
35     		return new Comparator<String>() {
36     			@Override
37     			public int compare(String key1, String key2) {
38     				String imageUri1 = key1.substring(0, key1.lastIndexOf(URI_AND_SIZE_SEPARATOR));
39     				String imageUri2 = key2.substring(0, key2.lastIndexOf(URI_AND_SIZE_SEPARATOR));
40     				return imageUri1.compareTo(imageUri2);
41     			}
42     		};
43     	}

47     	 * <b>Note:</b> Memory cache can contain multiple sizes of the same image if only you didn't set

49     	 * denyCacheImageMultipleSizesInMemory()} option in {@linkplain ImageLoaderConfiguration configuration}

51     	public static List<Bitmap> findCachedBitmapsForImageUri(String imageUri, MemoryCacheAware<String, Bitmap> memoryCache) {
