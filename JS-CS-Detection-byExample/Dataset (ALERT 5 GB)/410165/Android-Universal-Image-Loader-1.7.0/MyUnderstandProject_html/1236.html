81     	public Bitmap decode(ImageSize targetSize, ImageScaleType scaleType, ViewScaleType viewScaleType) throws IOException {
82     		Options decodeOptions = getBitmapOptionsForImageDecoding(targetSize, scaleType, viewScaleType);
83     		InputStream imageStream = imageDownloader.getStream(imageUri);
84     		Bitmap subsampledBitmap;
85     		try {
86     			subsampledBitmap = BitmapFactory.decodeStream(imageStream, null, decodeOptions);
87     		} finally {
88     			imageStream.close();
89     		}
90     		if (subsampledBitmap == null) {
91     			return null;
92     		}
93     
94     		// Scale to exact size if need
95     		if (scaleType == ImageScaleType.EXACTLY || scaleType == ImageScaleType.EXACTLY_STRETCHED) {
96     			subsampledBitmap = scaleImageExactly(subsampledBitmap, targetSize, scaleType, viewScaleType);
97     		}
98     
99     		return subsampledBitmap;
100     	}

109     	@SuppressWarnings("deprecation")
110     	private int computeImageScale(ImageSize targetSize, ImageScaleType scaleType, ViewScaleType viewScaleType) throws IOException {
111     		int targetWidth = targetSize.getWidth();
112     		int targetHeight = targetSize.getHeight();
113     
114     		// decode image size
115     		Options options = new Options();
116     		options.inJustDecodeBounds = true;
117     		InputStream imageStream = imageDownloader.getStream(imageUri);
118     		try {
119     			BitmapFactory.decodeStream(imageStream, null, options);
120     		} finally {
121     			imageStream.close();
122     		}
123     
124     		int scale = 1;
125     		int imageWidth = options.outWidth;
126     		int imageHeight = options.outHeight;
127     		int widthScale = imageWidth / targetWidth;
128     		int heightScale = imageHeight / targetHeight;
129     
130     		if (viewScaleType == ViewScaleType.FIT_INSIDE) {
131     			if (scaleType == ImageScaleType.IN_SAMPLE_POWER_OF_2 || scaleType == ImageScaleType.POWER_OF_2) {
132     				while (imageWidth / 2 >= targetWidth || imageHeight / 2 >= targetHeight) { // ||
133     					imageWidth /= 2;
134     					imageHeight /= 2;
135     					scale *= 2;
136     				}
137     			} else {
138     				scale = Math.max(widthScale, heightScale); // max
139     			}
140     		} else { // ViewScaleType.CROP
141     			if (scaleType == ImageScaleType.IN_SAMPLE_POWER_OF_2 || scaleType == ImageScaleType.POWER_OF_2) {
142     				while (imageWidth / 2 >= targetWidth && imageHeight / 2 >= targetHeight) { // &&
143     					imageWidth /= 2;
144     					imageHeight /= 2;
145     					scale *= 2;
146     				}
147     			} else {
148     				scale = Math.min(widthScale, heightScale); // min
149     			}
150     		}
151     
152     		if (scale < 1) {
153     			scale = 1;
154     		}
155     
156     		if (loggingEnabled) L.d(LOG_IMAGE_SUBSAMPLED, imageWidth, imageHeight, targetWidth, targetHeight, scale);
157     		return scale;
158     	}

160     	private Bitmap scaleImageExactly(Bitmap subsampledBitmap, ImageSize targetSize, ImageScaleType scaleType, ViewScaleType viewScaleType) {
161     		float srcWidth = subsampledBitmap.getWidth();
162     		float srcHeight = subsampledBitmap.getHeight();
163     
164     		float widthScale = srcWidth / targetSize.getWidth();
165     		float heightScale = srcHeight / targetSize.getHeight();
166     
167     		int destWidth;
168     		int destHeight;
169     		if ((viewScaleType == ViewScaleType.FIT_INSIDE && widthScale >= heightScale) || (viewScaleType == ViewScaleType.CROP && widthScale < heightScale)) {
170     			destWidth = targetSize.getWidth();
171     			destHeight = (int) (srcHeight / widthScale);
172     		} else {
173     			destWidth = (int) (srcWidth / heightScale);
174     			destHeight = targetSize.getHeight();
175     		}
176     
177     		Bitmap scaledBitmap;
178     		if ((scaleType == ImageScaleType.EXACTLY && destWidth < srcWidth && destHeight < srcHeight)
179     				|| (scaleType == ImageScaleType.EXACTLY_STRETCHED && destWidth != srcWidth && destHeight != srcHeight)) {
180     			scaledBitmap = Bitmap.createScaledBitmap(subsampledBitmap, destWidth, destHeight, true);
181     			if (scaledBitmap != subsampledBitmap) {
182     				subsampledBitmap.recycle();
183     			}
184     			if (loggingEnabled) L.d(LOG_IMAGE_SCALED, (int) srcWidth, (int) srcHeight, destWidth, destHeight);
185     		} else {
186     			scaledBitmap = subsampledBitmap;
187     		}
188     
189     		return scaledBitmap;
190     	}
