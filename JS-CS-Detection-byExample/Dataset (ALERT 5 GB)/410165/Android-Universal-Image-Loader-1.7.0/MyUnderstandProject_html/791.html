42     	 */
43     	public PauseOnScrollListener(boolean pauseOnScroll, boolean pauseOnFling, OnScrollListener customListener) {

47     	}

49     	@Override
50     	public void onScrollStateChanged(AbsListView view, int scrollState) {
51     		switch (scrollState) {
52     			case OnScrollListener.SCROLL_STATE_IDLE:
53     				ImageLoader.getInstance().resume();
54     				break;
55     			case OnScrollListener.SCROLL_STATE_TOUCH_SCROLL:
56     				if (pauseOnScroll) {
57     					ImageLoader.getInstance().pause();
58     				}
59     				break;
60     			case OnScrollListener.SCROLL_STATE_FLING:
61     				if (pauseOnFling) {
62     					ImageLoader.getInstance().pause();
63     				}
64     				break;
65     		}
66     		if (externalListener != null) {
67     			externalListener.onScrollStateChanged(view, scrollState);
68     		}
69     	}

72     	public void onScroll(AbsListView view, int firstVisibleItem, int visibleItemCount, int totalItemCount) {

75     		}
