16     package com.nostra13.universalimageloader.cache.disc;
17     
18     import android.graphics.Bitmap;
19     import com.nostra13.universalimageloader.utils.IoUtils;
20     
21     import java.io.File;
22     import java.io.IOException;
23     import java.io.InputStream;
24     
25     /**
26      * Interface for disk cache
27      *
28      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
29      * @since 1.0.0
30      * @deprecated Use {@link DiskCache DiskCache} instead
31      */
32     @Deprecated
33     public interface DiscCacheAware {
34     
35     	/**
36     	 * Returns root directory of disk cache
37     	 *
38     	 * @return Root directory of disk cache
39     	 */
40     	File getDirectory();
41     
42     	/**
43     	 * Returns file of cached image
44     	 *
45     	 * @param imageUri Original image URI
46     	 * @return File of cached image or <b>null</b> if image wasn't cached
47     	 */
48     	File get(String imageUri);
49     
50     	/**
51     	 * Saves image stream in disk cache.
52     	 *
53     	 * @param imageUri    Original image URI
54     	 * @param imageStream Input stream of image
55     	 * @param listener    Listener for saving progress, can be ignored if you don't use
56     	 *                    {@linkplain com.nostra13.universalimageloader.core.listener.ImageLoadingProgressListener
57     	 *                    progress listener} in ImageLoader calls
58     	 * @return <b>true</b> - if image was saved successfully; <b>false</b> - if image wasn't saved in disk cache.
59     	 * @throws IOException
60     	 */
61     	boolean save(String imageUri, InputStream imageStream, IoUtils.CopyListener listener) throws IOException;
62     
63     	/**
64     	 * Saves image bitmap in disk cache.
65     	 *
66     	 * @param imageUri Original image URI
67     	 * @param bitmap   Image bitmap
68     	 * @return <b>true</b> - if bitmap was saved successfully; <b>false</b> - if bitmap wasn't saved in disk cache.
69     	 * @throws IOException
70     	 */
71     	boolean save(String imageUri, Bitmap bitmap) throws IOException;
72     
73     	/**
74     	 * Removes image file associated with incoming URI
75     	 *
76     	 * @param imageUri Image URI
77     	 * @return <b>true</b> - if image file is deleted successfully; <b>false</b> - if image file doesn't exist for
78     	 * incoming URI or image file can't be deleted.
79     	 */
80     	boolean remove(String imageUri);
81     
82     	/** Closes disk cache, releases resources. */
83     	void close();
84     
85     	/** Clears disk cache. */
86     	void clear();
87     }
