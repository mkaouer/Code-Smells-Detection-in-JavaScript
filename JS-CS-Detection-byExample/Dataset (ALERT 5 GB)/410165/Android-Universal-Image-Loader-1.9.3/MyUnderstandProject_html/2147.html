16     package com.nostra13.example.universalimageloader.fragment;
17     
18     import android.content.Intent;
19     import android.view.Menu;
20     import android.view.MenuItem;
21     import android.widget.AbsListView;
22     import com.nostra13.example.universalimageloader.Constants;
23     import com.nostra13.example.universalimageloader.R;
24     import com.nostra13.example.universalimageloader.activity.SimpleImageActivity;
25     import com.nostra13.universalimageloader.core.ImageLoader;
26     import com.nostra13.universalimageloader.core.listener.PauseOnScrollListener;
27     
28     /**
29      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
30      */
31     public abstract class AbsListViewBaseFragment extends BaseFragment {
32     
33     	protected static final String STATE_PAUSE_ON_SCROLL = "STATE_PAUSE_ON_SCROLL";
34     	protected static final String STATE_PAUSE_ON_FLING = "STATE_PAUSE_ON_FLING";
35     
36     	protected AbsListView listView;
37     
38     	protected boolean pauseOnScroll = false;
39     	protected boolean pauseOnFling = true;
40     
41     	@Override
42     	public void onResume() {
43     		super.onResume();
44     		applyScrollListener();
45     	}
46     
47     	@Override
48     	public void onPrepareOptionsMenu(Menu menu) {
49     		MenuItem pauseOnScrollItem = menu.findItem(R.id.item_pause_on_scroll);
50     		pauseOnScrollItem.setVisible(true);
51     		pauseOnScrollItem.setChecked(pauseOnScroll);
52     
53     		MenuItem pauseOnFlingItem = menu.findItem(R.id.item_pause_on_fling);
54     		pauseOnFlingItem.setVisible(true);
55     		pauseOnFlingItem.setChecked(pauseOnFling);
56     	}
57     
58     	@Override
59     	public boolean onOptionsItemSelected(MenuItem item) {
60     		switch (item.getItemId()) {
61     			case R.id.item_pause_on_scroll:
62     				pauseOnScroll = !pauseOnScroll;
63     				item.setChecked(pauseOnScroll);
64     				applyScrollListener();
65     				return true;
66     			case R.id.item_pause_on_fling:
67     				pauseOnFling = !pauseOnFling;
68     				item.setChecked(pauseOnFling);
69     				applyScrollListener();
70     				return true;
71     			default:
72     				return super.onOptionsItemSelected(item);
73     		}
74     	}
75     
76     	protected void startImagePagerActivity(int position) {
77     		Intent intent = new Intent(getActivity(), SimpleImageActivity.class);
78     		intent.putExtra(Constants.Extra.FRAGMENT_INDEX, ImagePagerFragment.INDEX);
79     		intent.putExtra(Constants.Extra.IMAGE_POSITION, position);
80     		startActivity(intent);
81     	}
82     
83     	private void applyScrollListener() {
84     		listView.setOnScrollListener(new PauseOnScrollListener(ImageLoader.getInstance(), pauseOnScroll, pauseOnFling));
85     	}
86     }
