16     package com.nostra13.example.universalimageloader.fragment;

18     import android.graphics.Bitmap;
19     import android.os.Bundle;
20     import android.os.Parcelable;

22     import android.support.v4.view.ViewPager;
23     import android.view.LayoutInflater;
24     import android.view.View;
25     import android.view.ViewGroup;
26     import android.widget.ImageView;
27     import android.widget.ProgressBar;
28     import android.widget.Toast;
29     import com.nostra13.example.universalimageloader.Constants;
30     import com.nostra13.example.universalimageloader.R;
31     import com.nostra13.universalimageloader.core.DisplayImageOptions;
32     import com.nostra13.universalimageloader.core.ImageLoader;
33     import com.nostra13.universalimageloader.core.assist.FailReason;
34     import com.nostra13.universalimageloader.core.assist.ImageScaleType;
35     import com.nostra13.universalimageloader.core.display.FadeInBitmapDisplayer;
36     import com.nostra13.universalimageloader.core.listener.SimpleImageLoadingListener;
37     
38     /**
39      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
40      */
41     public class ImagePagerFragment extends BaseFragment {
42     
43     	public static final int INDEX = 2;
44     
45     	String[] imageUrls = Constants.IMAGES;
46     
47     	DisplayImageOptions options;
48     
49     	@Override
50     	public void onCreate(Bundle savedInstanceState) {
51     		super.onCreate(savedInstanceState);
52     
53     		options = new DisplayImageOptions.Builder()
54     				.showImageForEmptyUri(R.drawable.ic_empty)
55     				.showImageOnFail(R.drawable.ic_error)
56     				.resetViewBeforeLoading(true)
57     				.cacheOnDisk(true)
58     				.imageScaleType(ImageScaleType.EXACTLY)
59     				.bitmapConfig(Bitmap.Config.RGB_565)
60     				.considerExifParams(true)
61     				.displayer(new FadeInBitmapDisplayer(300))
62     				.build();
63     	}
64     
65     	@Override
66     	public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
67     		View rootView = inflater.inflate(R.layout.fr_image_pager, container, false);
68     		ViewPager pager = (ViewPager) rootView.findViewById(R.id.pager);
69     		pager.setAdapter(new ImageAdapter());
70     		pager.setCurrentItem(getArguments().getInt(Constants.Extra.IMAGE_POSITION, 0));
71     		return rootView;
72     	}
73     
74     	private class ImageAdapter extends PagerAdapter {
75     
76     		private LayoutInflater inflater;
77     
78     		ImageAdapter() {
79     			inflater = LayoutInflater.from(getActivity());
80     		}
81     
82     		@Override
83     		public void destroyItem(ViewGroup container, int position, Object object) {
84     			container.removeView((View) object);
85     		}
86     
87     		@Override
88     		public int getCount() {
89     			return imageUrls.length;
90     		}
91     
92     		@Override
93     		public Object instantiateItem(ViewGroup view, int position) {
94     			View imageLayout = inflater.inflate(R.layout.item_pager_image, view, false);
95     			assert imageLayout != null;
96     			ImageView imageView = (ImageView) imageLayout.findViewById(R.id.image);
97     			final ProgressBar spinner = (ProgressBar) imageLayout.findViewById(R.id.loading);
98     
99     			ImageLoader.getInstance().displayImage(imageUrls[position], imageView, options, new SimpleImageLoadingListener() {
100     				@Override
101     				public void onLoadingStarted(String imageUri, View view) {
102     					spinner.setVisibility(View.VISIBLE);
103     				}
104     
105     				@Override
106     				public void onLoadingFailed(String imageUri, View view, FailReason failReason) {
107     					String message = null;
108     					switch (failReason.getType()) {
109     						case IO_ERROR:
110     							message = "Input/Output error";
111     							break;
112     						case DECODING_ERROR:
113     							message = "Image can't be decoded";
114     							break;
115     						case NETWORK_DENIED:
116     							message = "Downloads are denied";
117     							break;
118     						case OUT_OF_MEMORY:
119     							message = "Out Of Memory error";
120     							break;
121     						case UNKNOWN:
122     							message = "Unknown error";
123     							break;
124     					}
125     					Toast.makeText(getActivity(), message, Toast.LENGTH_SHORT).show();
126     
127     					spinner.setVisibility(View.GONE);
128     				}
129     
130     				@Override
131     				public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
132     					spinner.setVisibility(View.GONE);
133     				}
134     			});
135     
136     			view.addView(imageLayout, 0);
137     			return imageLayout;
138     		}
139     
140     		@Override
141     		public boolean isViewFromObject(View view, Object object) {
142     			return view.equals(object);
143     		}
144     
145     		@Override
146     		public void restoreState(Parcelable state, ClassLoader loader) {
147     		}
148     
149     		@Override
150     		public Parcelable saveState() {
151     			return null;
152     		}
153     	}
154     }
