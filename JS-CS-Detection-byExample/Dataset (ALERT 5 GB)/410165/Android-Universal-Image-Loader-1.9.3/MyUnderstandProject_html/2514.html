16     package com.nostra13.example.universalimageloader.fragment;
17     
18     import android.content.Intent;
19     import android.graphics.Bitmap;
20     import android.os.Bundle;
21     import android.view.LayoutInflater;
22     import android.view.View;
23     import android.view.ViewGroup;
24     import android.widget.AdapterView;
25     import android.widget.AdapterView.OnItemClickListener;
26     import android.widget.BaseAdapter;
27     import android.widget.Gallery;
28     import android.widget.ImageView;
29     import com.nostra13.example.universalimageloader.Constants;
30     import com.nostra13.example.universalimageloader.R;
31     import com.nostra13.example.universalimageloader.activity.SimpleImageActivity;
32     import com.nostra13.universalimageloader.core.DisplayImageOptions;
33     import com.nostra13.universalimageloader.core.ImageLoader;
34     
35     /**
36      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
37      */
38     public class ImageGalleryFragment extends BaseFragment {
39     
40     	public static final int INDEX = 3;
41     
42     	String[] imageUrls = Constants.IMAGES;
43     
44     	DisplayImageOptions options;
45     
46     	@Override
47     	public void onCreate(Bundle savedInstanceState) {
48     		super.onCreate(savedInstanceState);
49     
50     		options = new DisplayImageOptions.Builder()
51     				.showImageOnLoading(R.drawable.ic_stub)
52     				.showImageForEmptyUri(R.drawable.ic_empty)
53     				.showImageOnFail(R.drawable.ic_error)
54     				.cacheInMemory(true)
55     				.cacheOnDisk(true)
56     				.considerExifParams(true)
57     				.bitmapConfig(Bitmap.Config.RGB_565)
58     				.build();
59     	}
60     
61     	@SuppressWarnings("deprecation")
62     	@Override
63     	public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
64     		View rootView = inflater.inflate(R.layout.fr_image_gallery, container, false);
65     		Gallery gallery = (Gallery) rootView.findViewById(R.id.gallery);
66     		gallery.setAdapter(new ImageAdapter());
67     		gallery.setOnItemClickListener(new OnItemClickListener() {
68     			@Override
69     			public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
70     				startImagePagerActivity(position);
71     			}
72     		});
73     		return rootView;
74     	}
75     
76     
77     	private void startImagePagerActivity(int position) {
78     		Intent intent = new Intent(getActivity(), SimpleImageActivity.class);
79     		intent.putExtra(Constants.Extra.FRAGMENT_INDEX, ImagePagerFragment.INDEX);
80     		intent.putExtra(Constants.Extra.IMAGE_POSITION, position);
81     		startActivity(intent);
82     	}
83     
84     	private class ImageAdapter extends BaseAdapter {
85     
86     		private LayoutInflater inflater;
87     
88     		ImageAdapter() {
89     			inflater = LayoutInflater.from(getActivity());
90     		}
91     
92     		@Override
93     		public int getCount() {
94     			return imageUrls.length;
95     		}
96     
97     		@Override
98     		public Object getItem(int position) {
99     			return position;
100     		}
101     
102     		@Override
103     		public long getItemId(int position) {
104     			return position;
105     		}
106     
107     		@Override
108     		public View getView(int position, View convertView, ViewGroup parent) {
109     			ImageView imageView = (ImageView) convertView;
110     			if (imageView == null) {
111     				imageView = (ImageView) inflater.inflate(R.layout.item_gallery_image, parent, false);
112     			}
113     			ImageLoader.getInstance().displayImage(imageUrls[position], imageView, options);
114     			return imageView;
115     		}
116     	}
117     }
