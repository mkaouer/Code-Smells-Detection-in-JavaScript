48     /**
49      * Factory for providing of default options for {@linkplain ImageLoaderConfiguration configuration}
50      *
51      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
52      * @since 1.5.6
53      */
54     public class DefaultConfigurationFactory {
55     
56     	/** Creates default implementation of task executor */
57     	public static Executor createExecutor(int threadPoolSize, int threadPriority,
58     			QueueProcessingType tasksProcessingType) {
59     		boolean lifo = tasksProcessingType == QueueProcessingType.LIFO;
60     		BlockingQueue<Runnable> taskQueue =
61     				lifo ? new LIFOLinkedBlockingDeque<Runnable>() : new LinkedBlockingQueue<Runnable>();
62     		return new ThreadPoolExecutor(threadPoolSize, threadPoolSize, 0L, TimeUnit.MILLISECONDS, taskQueue,
63     				createThreadFactory(threadPriority, "uil-pool-"));
64     	}
65     
66     	/** Creates default implementation of task distributor */
67     	public static Executor createTaskDistributor() {
68     		return Executors.newCachedThreadPool(createThreadFactory(Thread.NORM_PRIORITY, "uil-pool-d-"));
69     	}
70     
71     	/** Creates {@linkplain HashCodeFileNameGenerator default implementation} of FileNameGenerator */
72     	public static FileNameGenerator createFileNameGenerator() {
73     		return new HashCodeFileNameGenerator();
74     	}
75     
76     	/**
77     	 * Creates default implementation of {@link DiskCache} depends on incoming parameters
78     	 */
79     	public static DiskCache createDiskCache(Context context, FileNameGenerator diskCacheFileNameGenerator,
80     			long diskCacheSize, int diskCacheFileCount) {
81     		File reserveCacheDir = createReserveDiskCacheDir(context);
82     		if (diskCacheSize > 0 || diskCacheFileCount > 0) {
83     			File individualCacheDir = StorageUtils.getIndividualCacheDirectory(context);
84     			try {
85     				return new LruDiscCache(individualCacheDir, reserveCacheDir, diskCacheFileNameGenerator, diskCacheSize,
86     						diskCacheFileCount);
87     			} catch (IOException e) {
88     				L.e(e);
89     				// continue and create unlimited cache
90     			}
91     		}
92     		File cacheDir = StorageUtils.getCacheDirectory(context);
93     		return new UnlimitedDiscCache(cacheDir, reserveCacheDir, diskCacheFileNameGenerator);
94     	}
95     
96     	/** Creates reserve disk cache folder which will be used if primary disk cache folder becomes unavailable */
97     	private static File createReserveDiskCacheDir(Context context) {
98     		File cacheDir = StorageUtils.getCacheDirectory(context, false);
99     		File individualDir = new File(cacheDir, "uil-images");
100     		if (individualDir.exists() || individualDir.mkdir()) {
101     			cacheDir = individualDir;
102     		}
103     		return cacheDir;
104     	}
105     
106     	/**
107     	 * Creates default implementation of {@link MemoryCache} - {@link LruMemoryCache}<br />
108     	 * Default cache size = 1/8 of available app memory.
109     	 */
110     	public static MemoryCache createMemoryCache(int memoryCacheSize) {
111     		if (memoryCacheSize == 0) {
112     			memoryCacheSize = (int) (Runtime.getRuntime().maxMemory() / 8);
113     		}
114     		return new LruMemoryCache(memoryCacheSize);
115     	}
116     
117     	/** Creates default implementation of {@link ImageDownloader} - {@link BaseImageDownloader} */
118     	public static ImageDownloader createImageDownloader(Context context) {
119     		return new BaseImageDownloader(context);
120     	}
121     
122     	/** Creates default implementation of {@link ImageDecoder} - {@link BaseImageDecoder} */
123     	public static ImageDecoder createImageDecoder(boolean loggingEnabled) {
124     		return new BaseImageDecoder(loggingEnabled);
125     	}
126     
127     	/** Creates default implementation of {@link BitmapDisplayer} - {@link SimpleBitmapDisplayer} */
128     	public static BitmapDisplayer createBitmapDisplayer() {
129     		return new SimpleBitmapDisplayer();
130     	}
131     
132     	/** Creates default implementation of {@linkplain ThreadFactory thread factory} for task executor */
133     	private static ThreadFactory createThreadFactory(int threadPriority, String threadNamePrefix) {
134     		return new DefaultThreadFactory(threadPriority, threadNamePrefix);
135     	}
136     
137     	private static class DefaultThreadFactory implements ThreadFactory {
138     
139     		private static final AtomicInteger poolNumber = new AtomicInteger(1);
140     
141     		private final ThreadGroup group;
142     		private final AtomicInteger threadNumber = new AtomicInteger(1);
143     		private final String namePrefix;
144     		private final int threadPriority;
145     
146     		DefaultThreadFactory(int threadPriority, String threadNamePrefix) {
147     			this.threadPriority = threadPriority;
148     			group = Thread.currentThread().getThreadGroup();
149     			namePrefix = threadNamePrefix + poolNumber.getAndIncrement() + "-thread-";
150     		}
151     
152     		@Override
153     		public Thread newThread(Runnable r) {
154     			Thread t = new Thread(group, r, namePrefix + threadNumber.getAndIncrement(), 0);
155     			if (t.isDaemon()) t.setDaemon(false);
156     			t.setPriority(threadPriority);
157     			return t;
158     		}
159     	}
