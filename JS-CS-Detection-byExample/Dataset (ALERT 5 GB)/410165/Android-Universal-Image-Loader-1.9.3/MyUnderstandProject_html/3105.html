89     	private void copyTestImageToSdCard(final File testImageOnSdCard) {
90     		new Thread(new Runnable() {
91     			@Override
92     			public void run() {
93     				try {
94     					InputStream is = getAssets().open(TEST_FILE_NAME);
95     					FileOutputStream fos = new FileOutputStream(testImageOnSdCard);
96     					byte[] buffer = new byte[8192];
97     					int read;
98     					try {
99     						while ((read = is.read(buffer)) != -1) {
100     							fos.write(buffer, 0, read);
101     						}
102     					} finally {
103     						fos.flush();
104     						fos.close();
105     						is.close();
106     					}
107     				} catch (IOException e) {
108     					L.w("Can't copy test image onto SD card");
109     				}
110     			}
111     		}).start();
112     	}
