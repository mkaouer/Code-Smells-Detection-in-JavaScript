41     /**
42      * Provides retrieving of {@link InputStream} of image by URI from network or file system or app resources.<br />
43      * {@link URLConnection} is used to retrieve image stream from network.
44      *
45      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
46      * @since 1.8.0
47      */
48     public class BaseImageDownloader implements ImageDownloader {
49     	/** {@value} */
50     	public static final int DEFAULT_HTTP_CONNECT_TIMEOUT = 5 * 1000; // milliseconds
51     	/** {@value} */
52     	public static final int DEFAULT_HTTP_READ_TIMEOUT = 20 * 1000; // milliseconds
53     
54     	/** {@value} */
55     	protected static final int BUFFER_SIZE = 32 * 1024; // 32 Kb
56     	/** {@value} */
57     	protected static final String ALLOWED_URI_CHARS = "@#&=*+-_.,:!?()/~'%";
58     
59     	protected static final int MAX_REDIRECT_COUNT = 5;
60     
61     	protected static final String CONTENT_CONTACTS_URI_PREFIX = "content://com.android.contacts/";
62     
63     	private static final String ERROR_UNSUPPORTED_SCHEME = "UIL doesn't support scheme(protocol) by default [%s]. " + "You should implement this support yourself (BaseImageDownloader.getStreamFromOtherSource(...))";
64     
65     	protected final Context context;
66     	protected final int connectTimeout;
67     	protected final int readTimeout;
68     
69     	public BaseImageDownloader(Context context) {
70     		this.context = context.getApplicationContext();
71     		this.connectTimeout = DEFAULT_HTTP_CONNECT_TIMEOUT;
72     		this.readTimeout = DEFAULT_HTTP_READ_TIMEOUT;
73     	}
74     
75     	public BaseImageDownloader(Context context, int connectTimeout, int readTimeout) {
76     		this.context = context.getApplicationContext();
77     		this.connectTimeout = connectTimeout;
78     		this.readTimeout = readTimeout;
79     	}
80     
81     	@Override
82     	public InputStream getStream(String imageUri, Object extra) throws IOException {
83     		switch (Scheme.ofUri(imageUri)) {
84     			case HTTP:
85     			case HTTPS:
86     				return getStreamFromNetwork(imageUri, extra);
87     			case FILE:
88     				return getStreamFromFile(imageUri, extra);
89     			case CONTENT:
90     				return getStreamFromContent(imageUri, extra);
91     			case ASSETS:
92     				return getStreamFromAssets(imageUri, extra);
93     			case DRAWABLE:
94     				return getStreamFromDrawable(imageUri, extra);
95     			case UNKNOWN:
96     			default:
97     				return getStreamFromOtherSource(imageUri, extra);
98     		}
99     	}
100     
101     	/**
102     	 * Retrieves {@link InputStream} of image by URI (image is located in the network).
103     	 *
104     	 * @param imageUri Image URI
105     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
106     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
107     	 * @return {@link InputStream} of image
108     	 * @throws IOException if some I/O error occurs during network request or if no InputStream could be created for
109     	 *                     URL.
110     	 */
111     	protected InputStream getStreamFromNetwork(String imageUri, Object extra) throws IOException {
112     		HttpURLConnection conn = createConnection(imageUri, extra);
113     
114     		int redirectCount = 0;
115     		while (conn.getResponseCode() / 100 == 3 && redirectCount < MAX_REDIRECT_COUNT) {
116     			conn = createConnection(conn.getHeaderField("Location"), extra);
117     			redirectCount++;
118     		}
119     
120     		InputStream imageStream;
121     		try {
122     			imageStream = conn.getInputStream();
123     		} catch (IOException e) {
124     			// Read all data to allow reuse connection (http://bit.ly/1ad35PY)
125     			IoUtils.readAndCloseStream(conn.getErrorStream());
126     			throw e;
127     		}
128     		return new ContentLengthInputStream(new BufferedInputStream(imageStream, BUFFER_SIZE), conn.getContentLength());
129     	}
130     
131     	/**
132     	 * Create {@linkplain HttpURLConnection HTTP connection} for incoming URL
133     	 *
134     	 * @param url   URL to connect to
135     	 * @param extra Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
136     	 *              DisplayImageOptions.extraForDownloader(Object)}; can be null
137     	 * @return {@linkplain HttpURLConnection Connection} for incoming URL. Connection isn't established so it still configurable.
138     	 * @throws IOException if some I/O error occurs during network request or if no InputStream could be created for
139     	 *                     URL.
140     	 */
141     	protected HttpURLConnection createConnection(String url, Object extra) throws IOException {
142     		String encodedUrl = Uri.encode(url, ALLOWED_URI_CHARS);
143     		HttpURLConnection conn = (HttpURLConnection) new URL(encodedUrl).openConnection();
144     		conn.setConnectTimeout(connectTimeout);
145     		conn.setReadTimeout(readTimeout);
146     		return conn;
147     	}
148     
149     	/**
150     	 * Retrieves {@link InputStream} of image by URI (image is located on the local file system or SD card).
151     	 *
152     	 * @param imageUri Image URI
153     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
154     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
155     	 * @return {@link InputStream} of image
156     	 * @throws IOException if some I/O error occurs reading from file system
157     	 */
158     	protected InputStream getStreamFromFile(String imageUri, Object extra) throws IOException {
159     		String filePath = Scheme.FILE.crop(imageUri);
160     		return new ContentLengthInputStream(new BufferedInputStream(new FileInputStream(filePath), BUFFER_SIZE),
161     				(int) new File(filePath).length());
162     	}
163     
164     	/**
165     	 * Retrieves {@link InputStream} of image by URI (image is accessed using {@link ContentResolver}).
166     	 *
167     	 * @param imageUri Image URI
168     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
169     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
170     	 * @return {@link InputStream} of image
171     	 * @throws FileNotFoundException if the provided URI could not be opened
172     	 */
173     	protected InputStream getStreamFromContent(String imageUri, Object extra) throws FileNotFoundException {
174     		ContentResolver res = context.getContentResolver();
175     
176     		Uri uri = Uri.parse(imageUri);
177     		if (isVideoUri(uri)) { // video thumbnail
178     			Long origId = Long.valueOf(uri.getLastPathSegment());
179     			Bitmap bitmap = MediaStore.Video.Thumbnails
180     					.getThumbnail(res, origId, MediaStore.Images.Thumbnails.MINI_KIND, null);
181     			if (bitmap != null) {
182     				ByteArrayOutputStream bos = new ByteArrayOutputStream();
183     				bitmap.compress(CompressFormat.PNG, 0, bos);
184     				return new ByteArrayInputStream(bos.toByteArray());
185     			}
186     		} else if (imageUri.startsWith(CONTENT_CONTACTS_URI_PREFIX)) { // contacts photo
187     			return ContactsContract.Contacts.openContactPhotoInputStream(res, uri);
188     		}
189     
190     		return res.openInputStream(uri);
191     	}
192     
193     	/**
194     	 * Retrieves {@link InputStream} of image by URI (image is located in assets of application).
195     	 *
196     	 * @param imageUri Image URI
197     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
198     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
199     	 * @return {@link InputStream} of image
200     	 * @throws IOException if some I/O error occurs file reading
201     	 */
202     	protected InputStream getStreamFromAssets(String imageUri, Object extra) throws IOException {
203     		String filePath = Scheme.ASSETS.crop(imageUri);
204     		return context.getAssets().open(filePath);
205     	}
206     
207     	/**
208     	 * Retrieves {@link InputStream} of image by URI (image is located in drawable resources of application).
209     	 *
210     	 * @param imageUri Image URI
211     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
212     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
213     	 * @return {@link InputStream} of image
214     	 */
215     	protected InputStream getStreamFromDrawable(String imageUri, Object extra) {
216     		String drawableIdString = Scheme.DRAWABLE.crop(imageUri);
217     		int drawableId = Integer.parseInt(drawableIdString);
218     		return context.getResources().openRawResource(drawableId);
219     	}
220     
221     	/**
222     	 * Retrieves {@link InputStream} of image by URI from other source with unsupported scheme. Should be overriden by
223     	 * successors to implement image downloading from special sources.<br />
224     	 * This method is called only if image URI has unsupported scheme. Throws {@link UnsupportedOperationException} by
225     	 * default.
226     	 *
227     	 * @param imageUri Image URI
228     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
229     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
230     	 * @return {@link InputStream} of image
231     	 * @throws IOException                   if some I/O error occurs
232     	 * @throws UnsupportedOperationException if image URI has unsupported scheme(protocol)
233     	 */
234     	protected InputStream getStreamFromOtherSource(String imageUri, Object extra) throws IOException {
235     		throw new UnsupportedOperationException(String.format(ERROR_UNSUPPORTED_SCHEME, imageUri));
236     	}
237     
238     	private boolean isVideoUri(Uri uri) {
239     		String mimeType = context.getContentResolver().getType(uri);
240     
241     		if (mimeType == null) {
242     			return false;
243     		}
244     
245     		return mimeType.startsWith("video/");
246     	}
