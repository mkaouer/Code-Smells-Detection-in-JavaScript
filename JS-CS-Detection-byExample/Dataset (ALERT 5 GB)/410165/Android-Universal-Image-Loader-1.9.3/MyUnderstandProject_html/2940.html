29     import com.nostra13.example.universalimageloader.Constants;
30     import com.nostra13.example.universalimageloader.R;
31     import com.nostra13.universalimageloader.core.DisplayImageOptions;
32     import com.nostra13.universalimageloader.core.ImageLoader;
33     import com.nostra13.universalimageloader.core.assist.FailReason;
34     import com.nostra13.universalimageloader.core.listener.ImageLoadingProgressListener;
35     import com.nostra13.universalimageloader.core.listener.SimpleImageLoadingListener;
36     
37     /**
38      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
39      */
40     public class ImageGridFragment extends AbsListViewBaseFragment {
41     
42     	public static final int INDEX = 1;
43     
44     	String[] imageUrls = Constants.IMAGES;
45     
46     	DisplayImageOptions options;
47     
48     	@Override
49     	public void onCreate(Bundle savedInstanceState) {
50     		super.onCreate(savedInstanceState);
51     
52     		options = new DisplayImageOptions.Builder()
53     				.showImageOnLoading(R.drawable.ic_stub)
54     				.showImageForEmptyUri(R.drawable.ic_empty)
55     				.showImageOnFail(R.drawable.ic_error)
56     				.cacheInMemory(true)
57     				.cacheOnDisk(true)
58     				.considerExifParams(true)
59     				.bitmapConfig(Bitmap.Config.RGB_565)
60     				.build();
61     	}
62     
63     	@Override
64     	public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
65     		View rootView = inflater.inflate(R.layout.fr_image_grid, container, false);
66     		listView = (GridView) rootView.findViewById(R.id.grid);
67     		((GridView) listView).setAdapter(new ImageAdapter());
68     		listView.setOnItemClickListener(new OnItemClickListener() {
69     			@Override
70     			public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
71     				startImagePagerActivity(position);
72     			}
73     		});
74     		return rootView;
75     	}
76     
77     	public class ImageAdapter extends BaseAdapter {
78     
79     		private LayoutInflater inflater;
80     
81     		ImageAdapter() {
82     			inflater = LayoutInflater.from(getActivity());
83     		}
84     
85     		@Override
86     		public int getCount() {
87     			return imageUrls.length;
88     		}
89     
90     		@Override
91     		public Object getItem(int position) {
92     			return null;
93     		}
94     
95     		@Override
96     		public long getItemId(int position) {
97     			return position;
98     		}
99     
100     		@Override
101     		public View getView(int position, View convertView, ViewGroup parent) {
102     			final ViewHolder holder;
103     			View view = convertView;
104     			if (view == null) {
105     				view = inflater.inflate(R.layout.item_grid_image, parent, false);
106     				holder = new ViewHolder();
107     				assert view != null;
108     				holder.imageView = (ImageView) view.findViewById(R.id.image);
109     				holder.progressBar = (ProgressBar) view.findViewById(R.id.progress);
110     				view.setTag(holder);
111     			} else {
112     				holder = (ViewHolder) view.getTag();
113     			}
114     
115     			ImageLoader.getInstance()
116     					.displayImage(imageUrls[position], holder.imageView, options, new SimpleImageLoadingListener() {
117     						@Override
118     						public void onLoadingStarted(String imageUri, View view) {
119     							holder.progressBar.setProgress(0);
120     							holder.progressBar.setVisibility(View.VISIBLE);
121     						}
122     
123     						@Override
124     						public void onLoadingFailed(String imageUri, View view, FailReason failReason) {
125     							holder.progressBar.setVisibility(View.GONE);
126     						}
127     
128     						@Override
129     						public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
130     							holder.progressBar.setVisibility(View.GONE);
131     						}
132     					}, new ImageLoadingProgressListener() {
133     						@Override
134     						public void onProgressUpdate(String imageUri, View view, int current, int total) {
135     							holder.progressBar.setProgress(Math.round(100.0f * current / total));
136     						}
137     					});
138     
139     			return view;
140     		}
141     	}
142     
143     	static class ViewHolder {
144     		ImageView imageView;
145     		ProgressBar progressBar;
146     	}
147     }
