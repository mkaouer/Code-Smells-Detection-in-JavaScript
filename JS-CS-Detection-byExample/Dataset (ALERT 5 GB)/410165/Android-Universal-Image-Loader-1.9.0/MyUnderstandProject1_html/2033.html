16     package com.nostra13.example.universalimageloader;
17     
18     import android.graphics.Bitmap;
19     import android.os.Bundle;
20     import android.os.Parcelable;
21     import android.support.v4.view.PagerAdapter;
22     import android.support.v4.view.ViewPager;
23     import android.view.LayoutInflater;
24     import android.view.View;
25     import android.view.ViewGroup;
26     import android.widget.ImageView;
27     import android.widget.ProgressBar;
28     import android.widget.Toast;
29     
30     import com.nostra13.example.universalimageloader.Constants.Extra;
31     import com.nostra13.universalimageloader.core.DisplayImageOptions;
32     import com.nostra13.universalimageloader.core.assist.FailReason;
33     import com.nostra13.universalimageloader.core.assist.ImageScaleType;
34     import com.nostra13.universalimageloader.core.assist.SimpleImageLoadingListener;
35     import com.nostra13.universalimageloader.core.display.FadeInBitmapDisplayer;
36     
37     /**
38      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
39      */
40     public class ImagePagerActivity extends BaseActivity {
41     
42     	private static final String STATE_POSITION = "STATE_POSITION";
43     
44     	DisplayImageOptions options;
45     
46     	ViewPager pager;
47     
48     	public void onCreate(Bundle savedInstanceState) {
49     		super.onCreate(savedInstanceState);
50     		setContentView(R.layout.ac_image_pager);
51     
52     		Bundle bundle = getIntent().getExtras();
53     		assert bundle != null;
54     		String[] imageUrls = bundle.getStringArray(Extra.IMAGES);
55     		int pagerPosition = bundle.getInt(Extra.IMAGE_POSITION, 0);
56     
57     		if (savedInstanceState != null) {
58     			pagerPosition = savedInstanceState.getInt(STATE_POSITION);
59     		}
60     
61     		options = new DisplayImageOptions.Builder()
62     			.showImageForEmptyUri(R.drawable.ic_empty)
63     			.showImageOnFail(R.drawable.ic_error)
64     			.resetViewBeforeLoading(true)
65     			.cacheOnDisc(true)
66     			.imageScaleType(ImageScaleType.EXACTLY)
67     			.bitmapConfig(Bitmap.Config.RGB_565)
68     			.considerExifParams(true)
69     			.displayer(new FadeInBitmapDisplayer(300))
70     			.build();
71     
72     		pager = (ViewPager) findViewById(R.id.pager);
73     		pager.setAdapter(new ImagePagerAdapter(imageUrls));
74     		pager.setCurrentItem(pagerPosition);
75     	}
76     
77     	@Override
78     	public void onSaveInstanceState(Bundle outState) {
79     		outState.putInt(STATE_POSITION, pager.getCurrentItem());
80     	}
81     
82     	private class ImagePagerAdapter extends PagerAdapter {
83     
84     		private String[] images;
85     		private LayoutInflater inflater;
86     
87     		ImagePagerAdapter(String[] images) {
88     			this.images = images;
89     			inflater = getLayoutInflater();
90     		}
91     
92     		@Override
93     		public void destroyItem(ViewGroup container, int position, Object object) {
94     			container.removeView((View) object);
95     		}
96     
97     		@Override
98     		public int getCount() {
99     			return images.length;
100     		}
101     
102     		@Override
103     		public Object instantiateItem(ViewGroup view, int position) {
104     			View imageLayout = inflater.inflate(R.layout.item_pager_image, view, false);
105     			assert imageLayout != null;
106     			ImageView imageView = (ImageView) imageLayout.findViewById(R.id.image);
107     			final ProgressBar spinner = (ProgressBar) imageLayout.findViewById(R.id.loading);
108     
109     			imageLoader.displayImage(images[position], imageView, options, new SimpleImageLoadingListener() {
110     				@Override
111     				public void onLoadingStarted(String imageUri, View view) {
112     					spinner.setVisibility(View.VISIBLE);
113     				}
114     
115     				@Override
116     				public void onLoadingFailed(String imageUri, View view, FailReason failReason) {
117     					String message = null;
118     					switch (failReason.getType()) {
119     						case IO_ERROR:
120     							message = "Input/Output error";
121     							break;
122     						case DECODING_ERROR:
123     							message = "Image can't be decoded";
124     							break;
125     						case NETWORK_DENIED:
126     							message = "Downloads are denied";
127     							break;
128     						case OUT_OF_MEMORY:
129     							message = "Out Of Memory error";
130     							break;
131     						case UNKNOWN:
132     							message = "Unknown error";
133     							break;
134     					}
135     					Toast.makeText(ImagePagerActivity.this, message, Toast.LENGTH_SHORT).show();
136     
137     					spinner.setVisibility(View.GONE);
138     				}
139     
140     				@Override
141     				public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
142     					spinner.setVisibility(View.GONE);
143     				}
144     			});
145     
146     			view.addView(imageLayout, 0);
147     			return imageLayout;
148     		}
149     
150     		@Override
151     		public boolean isViewFromObject(View view, Object object) {
152     			return view.equals(object);
153     		}
154     
155     		@Override
156     		public void restoreState(Parcelable state, ClassLoader loader) {
157     		}
158     
159     		@Override
160     		public Parcelable saveState() {
161     			return null;
162     		}
163     	}
164     }
