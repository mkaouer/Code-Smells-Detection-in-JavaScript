16     package com.nostra13.example.universalimageloader;
17     
18     import android.content.Intent;
19     import android.graphics.Bitmap;
20     import android.os.Bundle;
21     import android.view.View;
22     import android.view.ViewGroup;
23     import android.widget.*;
24     import android.widget.AdapterView.OnItemClickListener;
25     import com.nostra13.example.universalimageloader.Constants.Extra;
26     import com.nostra13.universalimageloader.core.DisplayImageOptions;
27     import com.nostra13.universalimageloader.core.assist.ImageLoadingListener;
28     import com.nostra13.universalimageloader.core.assist.SimpleImageLoadingListener;
29     import com.nostra13.universalimageloader.core.display.FadeInBitmapDisplayer;
30     import com.nostra13.universalimageloader.core.display.RoundedBitmapDisplayer;
31     
32     import java.util.Collections;
33     import java.util.LinkedList;
34     import java.util.List;
35     
36     /**
37      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
38      */
39     public class ImageListActivity extends AbsListViewBaseActivity {
40     
41     	DisplayImageOptions options;
42     
43     	String[] imageUrls;
44     
45     	@Override
46     	public void onCreate(Bundle savedInstanceState) {
47     		super.onCreate(savedInstanceState);
48     		setContentView(R.layout.ac_image_list);
49     
50     		Bundle bundle = getIntent().getExtras();
51     		imageUrls = bundle.getStringArray(Extra.IMAGES);
52     
53     		options = new DisplayImageOptions.Builder()
54     			.showImageOnLoading(R.drawable.ic_stub)
55     			.showImageForEmptyUri(R.drawable.ic_empty)
56     			.showImageOnFail(R.drawable.ic_error)
57     			.cacheInMemory(true)
58     			.cacheOnDisc(true)
59     			.considerExifParams(true)
60     			.displayer(new RoundedBitmapDisplayer(20))
61     			.build();
62     
63     		listView = (ListView) findViewById(android.R.id.list);
64     		((ListView) listView).setAdapter(new ItemAdapter());
65     		listView.setOnItemClickListener(new OnItemClickListener() {
66     			@Override
67     			public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
68     				startImagePagerActivity(position);
69     			}
70     		});
71     	}
72     
73     	@Override
74     	public void onBackPressed() {
75     		AnimateFirstDisplayListener.displayedImages.clear();
76     		super.onBackPressed();
77     	}
78     
79     	private void startImagePagerActivity(int position) {
80     		Intent intent = new Intent(this, ImagePagerActivity.class);
81     		intent.putExtra(Extra.IMAGES, imageUrls);
82     		intent.putExtra(Extra.IMAGE_POSITION, position);
83     		startActivity(intent);
84     	}
85     
86     	class ItemAdapter extends BaseAdapter {
87     
88     		private ImageLoadingListener animateFirstListener = new AnimateFirstDisplayListener();
89     
90     		private class ViewHolder {
91     			public TextView text;
92     			public ImageView image;
93     		}
94     
95     		@Override
96     		public int getCount() {
97     			return imageUrls.length;
98     		}
99     
100     		@Override
101     		public Object getItem(int position) {
102     			return position;
103     		}
104     
105     		@Override
106     		public long getItemId(int position) {
107     			return position;
108     		}
109     
110     		@Override
111     		public View getView(final int position, View convertView, ViewGroup parent) {
112     			View view = convertView;
113     			final ViewHolder holder;
114     			if (convertView == null) {
115     				view = getLayoutInflater().inflate(R.layout.item_list_image, parent, false);
116     				holder = new ViewHolder();
117     				holder.text = (TextView) view.findViewById(R.id.text);
118     				holder.image = (ImageView) view.findViewById(R.id.image);
119     				view.setTag(holder);
120     			} else {
121     				holder = (ViewHolder) view.getTag();
122     			}
123     
124     			holder.text.setText("Item " + (position + 1));
125     
126     			imageLoader.displayImage(imageUrls[position], holder.image, options, animateFirstListener);
127     
128     			return view;
129     		}
130     	}
131     
132     	private static class AnimateFirstDisplayListener extends SimpleImageLoadingListener {
133     
134     		static final List<String> displayedImages = Collections.synchronizedList(new LinkedList<String>());
135     
136     		@Override
137     		public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
138     			if (loadedImage != null) {
139     				ImageView imageView = (ImageView) view;
140     				boolean firstDisplay = !displayedImages.contains(imageUri);
141     				if (firstDisplay) {
142     					FadeInBitmapDisplayer.animate(imageView, 500);
143     					displayedImages.add(imageUri);
144     				}
145     			}
146     		}
147     	}
148     }
