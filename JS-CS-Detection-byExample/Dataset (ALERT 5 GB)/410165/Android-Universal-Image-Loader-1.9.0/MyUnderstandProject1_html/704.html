16     package com.nostra13.universalimageloader.cache.disc;
17     
18     import com.nostra13.universalimageloader.cache.disc.naming.FileNameGenerator;
19     import com.nostra13.universalimageloader.core.DefaultConfigurationFactory;
20     
21     import java.io.File;
22     
23     /**
24      * Base disc cache. Implements common functionality for disc cache.
25      *
26      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
27      * @see DiscCacheAware
28      * @see FileNameGenerator
29      * @since 1.0.0
30      */
31     public abstract class BaseDiscCache implements DiscCacheAware {
32     
33     	private static final String ERROR_ARG_NULL = "\"%s\" argument must be not null";
34     
35     	protected File cacheDir;
36     
37     	private FileNameGenerator fileNameGenerator;
38     
39     	public BaseDiscCache(File cacheDir) {
40     		this(cacheDir, DefaultConfigurationFactory.createFileNameGenerator());
41     	}
42     
43     	public BaseDiscCache(File cacheDir, FileNameGenerator fileNameGenerator) {
44     		if (cacheDir == null) {
45     			throw new IllegalArgumentException(String.format(ERROR_ARG_NULL, "cacheDir"));
46     		}
47     		if (fileNameGenerator == null) {
48     			throw new IllegalArgumentException(String.format(ERROR_ARG_NULL, "fileNameGenerator"));
49     		}
50     
51     		this.cacheDir = cacheDir;
52     		this.fileNameGenerator = fileNameGenerator;
53     	}
54     
55     	@Override
56     	public File get(String key) {
57     		String fileName = fileNameGenerator.generate(key);
58     		return new File(cacheDir, fileName);
59     	}
60     
61     	@Override
62     	public void clear() {
63     		File[] files = cacheDir.listFiles();
64     		if (files != null) {
65     			for (File f : files) {
66     				f.delete();
67     			}
68     		}
69     	}
70     }
