16     package com.nostra13.universalimageloader.core.imageaware;
17     
18     import android.graphics.Bitmap;
19     import android.graphics.drawable.Drawable;
20     import android.view.View;
21     import com.nostra13.universalimageloader.core.assist.ImageSize;
22     import com.nostra13.universalimageloader.core.assist.ViewScaleType;
23     
24     /**
25      * ImageAware which provides needed info for processing of original image but do nothing for displaying image. It's
26      * used when user need just load and decode image and get it in {@linkplain
27      * com.nostra13.universalimageloader.core.assist.ImageLoadingListener#onLoadingComplete(String, android.view.View,
28      * android.graphics.Bitmap) callback}.
29      *
30      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
31      * @since 1.9.0
32      */
33     public class ImageNonViewAware implements ImageAware {
34     
35     	protected final ImageSize imageSize;
36     	protected final ViewScaleType scaleType;
37     
38     	public ImageNonViewAware(ImageSize imageSize, ViewScaleType scaleType) {
39     		this.imageSize = imageSize;
40     		this.scaleType = scaleType;
41     	}
42     
43     	@Override
44     	public int getWidth() {
45     		return imageSize.getWidth();
46     	}
47     
48     	@Override
49     	public int getHeight() {
50     		return imageSize.getHeight();
51     	}
52     
53     	@Override
54     	public ViewScaleType getScaleType() {
55     		return scaleType;
56     	}
57     
58     	@Override
59     	public View getWrappedView() {
60     		return null;
61     	}
62     
63     	@Override
64     	public boolean isCollected() {
65     		return false;
66     	}
67     
68     	@Override
69     	public int getId() {
70     		return super.hashCode();
71     	}
72     
73     	@Override
74     	public boolean setImageDrawable(Drawable drawable) { // Do nothing
75     		return true;
76     	}
77     
78     	@Override
79     	public boolean setImageBitmap(Bitmap bitmap) { // Do nothing
80     		return true;
81     	}
82     }
