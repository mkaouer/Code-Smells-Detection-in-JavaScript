142     	ImageSize getMaxImageSize() {
143     		DisplayMetrics displayMetrics = resources.getDisplayMetrics();
144     
145     		int width = maxImageWidthForMemoryCache;
146     		if (width <= 0) {
147     			width = displayMetrics.widthPixels;
148     		}
149     		int height = maxImageHeightForMemoryCache;
150     		if (height <= 0) {
151     			height = displayMetrics.heightPixels;
152     		}
153     		return new ImageSize(width, height);
154     	}

536     		 * Consider {@link com.nostra13.universalimageloader.utils.L#disableLogging()} to disable ImageLoader logging completely (even error logs)
537     		 */
538     		public Builder writeDebugLogs() {
539     			this.writeLogs = true;
540     			return this;
541     		}
542     
543     		/** Builds configured {@link ImageLoaderConfiguration} object */
544     		public ImageLoaderConfiguration build() {
545     			initEmptyFieldsWithDefaultValues();
546     			return new ImageLoaderConfiguration(this);
547     		}
548     
549     		private void initEmptyFieldsWithDefaultValues() {
550     			if (taskExecutor == null) {
551     				taskExecutor = DefaultConfigurationFactory
552     						.createExecutor(threadPoolSize, threadPriority, tasksProcessingType);
553     			} else {
554     				customExecutor = true;
555     			}
556     			if (taskExecutorForCachedImages == null) {
557     				taskExecutorForCachedImages = DefaultConfigurationFactory
558     						.createExecutor(threadPoolSize, threadPriority, tasksProcessingType);
559     			} else {
560     				customExecutorForCachedImages = true;
561     			}
562     			if (discCache == null) {
563     				if (discCacheFileNameGenerator == null) {
564     					discCacheFileNameGenerator = DefaultConfigurationFactory.createFileNameGenerator();
565     				}
566     				discCache = DefaultConfigurationFactory
567     						.createDiscCache(context, discCacheFileNameGenerator, discCacheSize, discCacheFileCount);
568     			}
569     			if (memoryCache == null) {
570     				memoryCache = DefaultConfigurationFactory.createMemoryCache(memoryCacheSize);
571     			}
572     			if (denyCacheImageMultipleSizesInMemory) {
573     				memoryCache = new FuzzyKeyMemoryCache<String, Bitmap>(memoryCache, MemoryCacheUtil
574     						.createFuzzyKeyComparator());
575     			}
576     			if (downloader == null) {
577     				downloader = DefaultConfigurationFactory.createImageDownloader(context);
578     			}
579     			if (decoder == null) {
580     				decoder = DefaultConfigurationFactory.createImageDecoder(writeLogs);
581     			}
582     			if (defaultDisplayImageOptions == null) {
583     				defaultDisplayImageOptions = DisplayImageOptions.createSimple();
584     			}
585     		}
586     	}
