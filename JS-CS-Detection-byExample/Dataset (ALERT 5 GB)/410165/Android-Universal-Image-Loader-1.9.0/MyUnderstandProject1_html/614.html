16     package com.nostra13.universalimageloader.core.download;
17     
18     import java.io.IOException;
19     import java.io.InputStream;
20     
21     /**
22      * Decorator. Prevents downloads from network (throws {@link IllegalStateException exception}).<br />
23      * In most cases this downloader shouldn't be used directly.
24      *
25      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
26      * @since 1.8.0
27      */
28     public class NetworkDeniedImageDownloader implements ImageDownloader {
29     
30     	private final ImageDownloader wrappedDownloader;
31     
32     	public NetworkDeniedImageDownloader(ImageDownloader wrappedDownloader) {
33     		this.wrappedDownloader = wrappedDownloader;
34     	}
35     
36     	@Override
37     	public InputStream getStream(String imageUri, Object extra) throws IOException {
38     		switch (Scheme.ofUri(imageUri)) {
39     			case HTTP:
40     			case HTTPS:
41     				throw new IllegalStateException();
42     			default:
43     				return wrappedDownloader.getStream(imageUri, extra);
44     		}
45     	}
46     }
