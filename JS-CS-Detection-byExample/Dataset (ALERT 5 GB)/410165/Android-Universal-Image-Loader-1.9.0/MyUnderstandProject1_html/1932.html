16     package com.nostra13.universalimageloader.utils;
17     
18     import java.io.Closeable;
19     import java.io.IOException;
20     import java.io.InputStream;
21     import java.io.OutputStream;
22     
23     /**
24      * Provides I/O operations
25      *
26      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
27      * @since 1.0.0
28      */
29     public final class IoUtils {
30     
31     	private static final int BUFFER_SIZE = 32 * 1024; // 32 KB
32     
33     	private IoUtils() {
34     	}
35     
36     	public static void copyStream(InputStream is, OutputStream os) throws IOException {
37     		byte[] bytes = new byte[BUFFER_SIZE];
38     		while (true) {
39     			int count = is.read(bytes, 0, BUFFER_SIZE);
40     			if (count == -1) {
41     				break;
42     			}
43     			os.write(bytes, 0, count);
44     		}
45     	}
46     
47     	public static void closeSilently(Closeable closeable) {
48     		try {
49     			closeable.close();
50     		} catch (Exception e) {
51     			// Do nothing
52     		}
53     	}
54     }
