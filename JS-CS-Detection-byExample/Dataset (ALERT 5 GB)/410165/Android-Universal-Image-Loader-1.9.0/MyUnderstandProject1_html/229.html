34     /**
35      * Decodes images to {@link Bitmap}, scales them to needed size
36      *
37      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
38      * @see ImageDecodingInfo
39      * @since 1.8.3
40      */
41     public class BaseImageDecoder implements ImageDecoder {
42     
43     	protected static final String LOG_SABSAMPLE_IMAGE = "Subsample original image (%1$s) to %2$s (scale = %3$d) [%4$s]";
44     	protected static final String LOG_SCALE_IMAGE = "Scale subsampled image (%1$s) to %2$s (scale = %3$.5f) [%4$s]";
45     	protected static final String LOG_ROTATE_IMAGE = "Rotate image on %1$d\u00B0 [%2$s]";
46     	protected static final String LOG_FLIP_IMAGE = "Flip image horizontally [%s]";
47     	protected static final String ERROR_CANT_DECODE_IMAGE = "Image can't be decoded [%s]";
48     
49     	protected final boolean loggingEnabled;
50     
51     	/**
52     	 * @param loggingEnabled Whether debug logs will be written to LogCat.
53     	 *                       Usually should match {@link com.nostra13.universalimageloader.core.ImageLoaderConfiguration.Builder#writeDebugLogs() ImageLoaderConfiguration.writeDebugLogs()}
54     	 */
55     	public BaseImageDecoder(boolean loggingEnabled) {
56     		this.loggingEnabled = loggingEnabled;
57     	}
58     
59     	/**
60     	 * Decodes image from URI into {@link Bitmap}. Image is scaled close to incoming {@linkplain ImageSize target size}
61     	 * during decoding (depend on incoming parameters).
62     	 *
63     	 * @param decodingInfo Needed data for decoding image
64     	 * @return Decoded bitmap
65     	 * @throws IOException                   if some I/O exception occurs during image reading
66     	 * @throws UnsupportedOperationException if image URI has unsupported scheme(protocol)
67     	 */
68     	public Bitmap decode(ImageDecodingInfo decodingInfo) throws IOException {
69     		InputStream imageStream = getImageStream(decodingInfo);
70     		ImageFileInfo imageInfo = defineImageSizeAndRotation(imageStream, decodingInfo);
71     		Options decodingOptions = prepareDecodingOptions(imageInfo.imageSize, decodingInfo);
72     		imageStream = resetStream(imageStream, decodingInfo);
73     		Bitmap decodedBitmap = decodeStream(imageStream, decodingOptions);
74     		if (decodedBitmap == null) {
75     			L.e(ERROR_CANT_DECODE_IMAGE, decodingInfo.getImageKey());
76     		} else {
77     			decodedBitmap = considerExactScaleAndOrientaiton(decodedBitmap, decodingInfo, imageInfo.exif.rotation, imageInfo.exif.flipHorizontal);
78     		}
79     		return decodedBitmap;
80     	}
81     
82     	protected InputStream getImageStream(ImageDecodingInfo decodingInfo) throws IOException {
83     		return decodingInfo.getDownloader().getStream(decodingInfo.getImageUri(), decodingInfo.getExtraForDownloader());
84     	}
85     
86     	protected ImageFileInfo defineImageSizeAndRotation(InputStream imageStream, ImageDecodingInfo decodingInfo)
87     			throws IOException {
88     		Options options = new Options();
89     		options.inJustDecodeBounds = true;
90     		BitmapFactory.decodeStream(imageStream, null, options);
91     
92     		ExifInfo exif;
93     		String imageUri = decodingInfo.getImageUri();
94     		if (decodingInfo.shouldConsiderExifParams() && canDefineExifParams(imageUri, options.outMimeType)) {
95     			exif = defineExifOrientation(imageUri);
96     		} else {
97     			exif = new ExifInfo();
98     		}
99     		return new ImageFileInfo(new ImageSize(options.outWidth, options.outHeight, exif.rotation), exif);
100     	}
101     
102     	private boolean canDefineExifParams(String imageUri, String mimeType) {
103     		return Build.VERSION.SDK_INT >= Build.VERSION_CODES.ECLAIR && "image/jpeg".equalsIgnoreCase(mimeType) && Scheme
104     				.ofUri(imageUri) == Scheme.FILE;
105     	}
106     
107     	protected ExifInfo defineExifOrientation(String imageUri) {
108     		int rotation = 0;
109     		boolean flip = false;
110     		try {
111     			ExifInterface exif = new ExifInterface(Scheme.FILE.crop(imageUri));
112     			int exifOrientation = exif.getAttributeInt(ExifInterface.TAG_ORIENTATION, ExifInterface.ORIENTATION_NORMAL);
113     			switch (exifOrientation) {
114     				case ExifInterface.ORIENTATION_FLIP_HORIZONTAL:
115     					flip = true;
116     				case ExifInterface.ORIENTATION_NORMAL:
117     					rotation = 0;
118     					break;
119     				case ExifInterface.ORIENTATION_TRANSVERSE:
120     					flip = true;
121     				case ExifInterface.ORIENTATION_ROTATE_90:
122     					rotation = 90;
123     					break;
124     				case ExifInterface.ORIENTATION_FLIP_VERTICAL:
125     					flip = true;
126     				case ExifInterface.ORIENTATION_ROTATE_180:
127     					rotation = 180;
128     					break;
129     				case ExifInterface.ORIENTATION_TRANSPOSE:
130     					flip = true;
131     				case ExifInterface.ORIENTATION_ROTATE_270:
132     					rotation = 270;
133     					break;
134     			}
135     		} catch (IOException e) {
136     			L.w("Can't read EXIF tags from file [%s]", imageUri);
137     		}
138     		return new ExifInfo(rotation, flip);
139     	}
140     
141     	protected Options prepareDecodingOptions(ImageSize imageSize, ImageDecodingInfo decodingInfo) {
142     		ImageScaleType scaleType = decodingInfo.getImageScaleType();
143     		ImageSize targetSize = decodingInfo.getTargetSize();
144     		int scale = 1;
145     		if (scaleType != ImageScaleType.NONE) {
146     			boolean powerOf2 = scaleType == ImageScaleType.IN_SAMPLE_POWER_OF_2;
147     			scale = ImageSizeUtils
148     					.computeImageSampleSize(imageSize, targetSize, decodingInfo.getViewScaleType(), powerOf2);
149     
150     			if (loggingEnabled) {
151     				L.d(LOG_SABSAMPLE_IMAGE, imageSize, imageSize.scaleDown(scale), scale, decodingInfo.getImageKey());
152     			}
153     		}
154     		Options decodingOptions = decodingInfo.getDecodingOptions();
155     		decodingOptions.inSampleSize = scale;
156     		return decodingOptions;
157     	}
158     
159     	protected InputStream resetStream(InputStream imageStream, ImageDecodingInfo decodingInfo) throws IOException {
160     		try {
161     			imageStream.reset();
162     		} catch (IOException e) {
163     			imageStream = getImageStream(decodingInfo);
164     		}
165     		return imageStream;
166     	}
167     
168     	protected Bitmap decodeStream(InputStream imageStream, Options decodingOptions) throws IOException {
169     		try {
170     			return BitmapFactory.decodeStream(imageStream, null, decodingOptions);
171     		} finally {
172     			IoUtils.closeSilently(imageStream);
173     		}
174     	}
175     
176     	protected Bitmap considerExactScaleAndOrientaiton(Bitmap subsampledBitmap, ImageDecodingInfo decodingInfo,
177     													  int rotation, boolean flipHorizontal) {
178     		Matrix m = new Matrix();
179     		// Scale to exact size if need
180     		ImageScaleType scaleType = decodingInfo.getImageScaleType();
181     		if (scaleType == ImageScaleType.EXACTLY || scaleType == ImageScaleType.EXACTLY_STRETCHED) {
182     			ImageSize srcSize = new ImageSize(subsampledBitmap.getWidth(), subsampledBitmap.getHeight(), rotation);
183     			float scale = ImageSizeUtils.computeImageScale(srcSize, decodingInfo.getTargetSize(), decodingInfo
184     					.getViewScaleType(), scaleType == ImageScaleType.EXACTLY_STRETCHED);
185     			if (Float.compare(scale, 1f) != 0) {
186     				m.setScale(scale, scale);
187     
188     				if (loggingEnabled) {
189     					L.d(LOG_SCALE_IMAGE, srcSize, srcSize.scale(scale), scale, decodingInfo.getImageKey());
190     				}
191     			}
192     		}
193     		// Flip bitmap if need
194     		if (flipHorizontal) {
195     			m.postScale(-1, 1);
196     
197     			if (loggingEnabled) L.d(LOG_FLIP_IMAGE, decodingInfo.getImageKey());
198     		}
199     		// Rotate bitmap if need
200     		if (rotation != 0) {
201     			m.postRotate(rotation);
202     
203     			if (loggingEnabled) L.d(LOG_ROTATE_IMAGE, rotation, decodingInfo.getImageKey());
204     		}
205     
206     		Bitmap finalBitmap = Bitmap.createBitmap(subsampledBitmap, 0, 0, subsampledBitmap.getWidth(), subsampledBitmap
207     				.getHeight(), m, true);
208     		if (finalBitmap != subsampledBitmap) {
209     			subsampledBitmap.recycle();
210     		}
211     		return finalBitmap;
212     	}

215     
216     		public final int rotation;
217     		public final boolean flipHorizontal;
218     
219     		protected ExifInfo() {
220     			this.rotation = 0;
221     			this.flipHorizontal = false;
222     		}
223     
224     		protected ExifInfo(int rotation, boolean flipHorizontal) {
225     			this.rotation = rotation;
226     			this.flipHorizontal = flipHorizontal;
227     		}
228     	}
229     
230     	protected static class ImageFileInfo {
231     
232     		public final ImageSize imageSize;
233     		public final ExifInfo exif;
234     
235     		protected ImageFileInfo(ImageSize imageSize, ExifInfo exif) {
236     			this.imageSize = imageSize;
237     			this.exif = exif;
238     		}
239     	}
