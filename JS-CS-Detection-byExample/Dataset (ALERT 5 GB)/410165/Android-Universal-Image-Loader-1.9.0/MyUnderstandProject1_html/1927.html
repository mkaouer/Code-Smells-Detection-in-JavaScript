20     import com.nostra13.universalimageloader.core.imageaware.ImageAware;

22     /**
23      * Just displays {@link Bitmap} in {@link com.nostra13.universalimageloader.core.imageaware.ImageAware}
24      *
25      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
26      * @since 1.5.6
27      */
28     public final class SimpleBitmapDisplayer implements BitmapDisplayer {
29     	@Override
30     	public Bitmap display(Bitmap bitmap, ImageAware imageAware, LoadedFrom loadedFrom) {
31     		imageAware.setImageBitmap(bitmap);
32     		return bitmap;
33     	}
34     }
