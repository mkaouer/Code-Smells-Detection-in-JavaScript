26     /**
27      * Displays bitmap with rounded corners. This implementation works only with ImageViews wrapped in ImageViewAware.<br />
28      * <b>NOTE:</b> It's strongly recommended your {@link ImageView} has defined width (<i>layout_width</i>) and height
29      * (<i>layout_height</i>) .<br />
30      * <b>NOTE:</b> New {@link Bitmap} object is created for displaying. So this class needs more memory and can cause
31      * {@link OutOfMemoryError}.
32      *
33      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
34      * @since 1.5.6
35      */
36     public class RoundedBitmapDisplayer implements BitmapDisplayer {
37     
38     	private final int roundPixels;
39     
40     	public RoundedBitmapDisplayer(int roundPixels) {
41     		this.roundPixels = roundPixels;
42     	}
43     
44     	@Override
45     	public Bitmap display(Bitmap bitmap, ImageAware imageAware, LoadedFrom loadedFrom) {
46     		if (!(imageAware instanceof ImageViewAware)) {
47     			throw new IllegalArgumentException("ImageAware should wrap ImageView. ImageViewAware is expected.");
48     		}
49     		Bitmap roundedBitmap = roundCorners(bitmap, (ImageViewAware) imageAware, roundPixels);
50     		imageAware.setImageBitmap(roundedBitmap);
51     		return roundedBitmap;
52     	}
53     
54     	/**
55     	 * Process incoming {@linkplain Bitmap} to make rounded corners according to target
56     	 * {@link com.nostra13.universalimageloader.core.imageaware.ImageViewAware}.<br />
57     	 * This method <b>doesn't display</b> result bitmap in {@link ImageView}
58     	 *
59     	 * @param bitmap      Incoming Bitmap to process
60     	 * @param imageAware  Target {@link com.nostra13.universalimageloader.core.imageaware.ImageAware ImageAware} to
61     	 *                    display bitmap in
62     	 * @param roundPixels Rounded pixels of corner
63     	 * @return Result bitmap with rounded corners
64     	 */
65     	public static Bitmap roundCorners(Bitmap bitmap, ImageViewAware imageAware, int roundPixels) {
66     		ImageView imageView = imageAware.getWrappedView();
67     		if (imageView == null) {
68     			L.w("View is collected probably. Can't round bitmap corners without view properties.");
69     			return bitmap;
70     		}
71     
72     		Bitmap roundBitmap;
73     
74     		int bw = bitmap.getWidth();
75     		int bh = bitmap.getHeight();
76     		int vw = imageAware.getWidth();
77     		int vh = imageAware.getHeight();
78     		if (vw <= 0) vw = bw;
79     		if (vh <= 0) vh = bh;
80     
81     		final ImageView.ScaleType scaleType = imageView.getScaleType();
82     		if (scaleType == null) {
83     			return bitmap;
84     		}
85     
86     		int width, height;
87     		Rect srcRect;
88     		Rect destRect;
89     		switch (scaleType) {
90     			case CENTER_INSIDE:
91     				float vRation = (float) vw / vh;
92     				float bRation = (float) bw / bh;
93     				int destWidth;
94     				int destHeight;
95     				if (vRation > bRation) {
96     					destHeight = Math.min(vh, bh);
97     					destWidth = (int) (bw / ((float) bh / destHeight));
98     				} else {
99     					destWidth = Math.min(vw, bw);
100     					destHeight = (int) (bh / ((float) bw / destWidth));
101     				}
102     				int x = (vw - destWidth) / 2;
103     				int y = (vh - destHeight) / 2;
104     				srcRect = new Rect(0, 0, bw, bh);
105     				destRect = new Rect(x, y, x + destWidth, y + destHeight);
106     				width = vw;
107     				height = vh;
108     				break;
109     			case FIT_CENTER:
110     			case FIT_START:
111     			case FIT_END:
112     			default:
113     				vRation = (float) vw / vh;
114     				bRation = (float) bw / bh;
115     				if (vRation > bRation) {
116     					width = (int) (bw / ((float) bh / vh));
117     					height = vh;
118     				} else {
119     					width = vw;
120     					height = (int) (bh / ((float) bw / vw));
121     				}
122     				srcRect = new Rect(0, 0, bw, bh);
123     				destRect = new Rect(0, 0, width, height);
124     				break;
125     			case CENTER_CROP:
126     				vRation = (float) vw / vh;
127     				bRation = (float) bw / bh;
128     				int srcWidth;
129     				int srcHeight;
130     				if (vRation > bRation) {
131     					srcWidth = bw;
132     					srcHeight = (int) (vh * ((float) bw / vw));
133     					x = 0;
134     					y = (bh - srcHeight) / 2;
135     				} else {
136     					srcWidth = (int) (vw * ((float) bh / vh));
137     					srcHeight = bh;
138     					x = (bw - srcWidth) / 2;
139     					y = 0;
140     				}
141     				width = srcWidth;// Math.min(vw, bw);
142     				height = srcHeight;//Math.min(vh, bh);
143     				srcRect = new Rect(x, y, x + srcWidth, y + srcHeight);
144     				destRect = new Rect(0, 0, width, height);
145     				break;
146     			case FIT_XY:
147     				width = vw;
148     				height = vh;
149     				srcRect = new Rect(0, 0, bw, bh);
150     				destRect = new Rect(0, 0, width, height);
151     				break;
152     			case CENTER:
153     			case MATRIX:
154     				width = Math.min(vw, bw);
155     				height = Math.min(vh, bh);
156     				x = (bw - width) / 2;
157     				y = (bh - height) / 2;
158     				srcRect = new Rect(x, y, x + width, y + height);
159     				destRect = new Rect(0, 0, width, height);
160     				break;
161     		}
162     
163     		try {
164     			roundBitmap = getRoundedCornerBitmap(bitmap, roundPixels, srcRect, destRect, width, height);
165     		} catch (OutOfMemoryError e) {
166     			L.e(e, "Can't create bitmap with rounded corners. Not enough memory.");
167     			roundBitmap = bitmap;
168     		}
169     
170     		return roundBitmap;
171     	}
172     
173     	private static Bitmap getRoundedCornerBitmap(Bitmap bitmap, int roundPixels, Rect srcRect, Rect destRect, int width,
174     												 int height) {
175     		Bitmap output = Bitmap.createBitmap(width, height, Config.ARGB_8888);
176     		Canvas canvas = new Canvas(output);
177     
178     		final Paint paint = new Paint();
179     		final RectF destRectF = new RectF(destRect);
180     
181     		paint.setAntiAlias(true);
182     		canvas.drawARGB(0, 0, 0, 0);
183     		paint.setColor(0xFF000000);
184     		canvas.drawRoundRect(destRectF, roundPixels, roundPixels, paint);
185     
186     		paint.setXfermode(new PorterDuffXfermode(PorterDuff.Mode.SRC_IN));
187     		canvas.drawBitmap(bitmap, srcRect, destRectF, paint);
188     
189     		return output;
190     	}
