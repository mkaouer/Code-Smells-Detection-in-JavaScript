16     package com.nostra13.universalimageloader.core.imageaware;
17     
18     import android.graphics.Bitmap;
19     import android.graphics.drawable.Drawable;
20     import android.view.View;
21     import com.nostra13.universalimageloader.core.assist.ViewScaleType;
22     
23     /**
24      * Represents image aware view which provides all needed properties and behavior for image processing and displaying
25      * through {@link com.nostra13.universalimageloader.core.ImageLoader ImageLoader}.
26      * It can wrap any Android {@link android.view.View View} which can be accessed by {@link #getWrappedView()}. Wrapped
27      * view is returned in {@link com.nostra13.universalimageloader.core.assist.ImageLoadingListener ImageLoadingListener}'s
28      * callbacks.
29      *
30      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
31      * @see ImageViewAware
32      * @see ImageNonViewAware
33      * @since 1.9.0
34      */
35     public interface ImageAware {
36     	/**
37     	 * Returns width of image aware view. This value is used to define scale size for original image.
38     	 * Can return 0 if width is undefined.<br />
39     	 * Called on UI thread.
40     	 */
41     	int getWidth();
42     
43     	/**
44     	 * Returns height of image aware view. This value is used to define scale size for original image.
45     	 * Can return 0 if height is undefined.<br />
46     	 * Called on UI thread.
47     	 */
48     	int getHeight();
49     
50     	/**
51     	 * Returns {@linkplain com.nostra13.universalimageloader.core.assist.ViewScaleType scale type} which is used for
52     	 * scaling image for this image aware view.
53     	 */
54     	ViewScaleType getScaleType();
55     
56     	/**
57     	 * Returns wrapped Android {@link android.view.View View}. Can return <b>null</b> if no view is wrapped.<br />
58     	 * Called on UI thread.
59     	 */
60     	View getWrappedView();
61     
62     	/**
63     	 * Returns a flag whether image aware view is collected by GC or whatsoever. If so then ImageLoader stop processing
64     	 * of task for this image aware view and fires
65     	 * {@link com.nostra13.universalimageloader.core.assist.ImageLoadingListener#onLoadingCancelled(String,
66     	 * android.view.View) ImageLoadingListener#onLoadingCancelled(String, View)} callback.<br />
67     	 * May be called on UI thread.
68     	 *
69     	 * @return <b>true</b> - if view is collected by GC and ImageLoader should stop processing this image aware view;
70     	 * <b>false</b> - otherwise
71     	 */
72     	boolean isCollected();
73     
74     	/**
75     	 * Returns ID of image aware view. Point of ID is similar to Object's hashCode. This ID should be unique for every
76     	 * image view instance and should be the same for same instances. This ID identifies processing task in ImageLoader
77     	 * so ImageLoader won't process two image aware views with the same ID in one time. When ImageLoader get new task
78     	 * it cancels old task with this ID (if any) and starts new task.
79     	 * <p/>
80     	 * It's reasonable to return hash code of wrapped view (if any) to prevent displaying non-actual images in view
81     	 * because of view re-using.
82     	 */
83     	int getId();
84     
85     	/**
86     	 * Sets image drawable into this image aware view.<br />
87     	 * Called on UI thread to display drawable in this image aware view
88     	 * {@linkplain com.nostra13.universalimageloader.core.DisplayImageOptions.Builder#showImageForEmptyUri(
89     	 *android.graphics.drawable.Drawable) for empty Uri},
90     	 * {@linkplain com.nostra13.universalimageloader.core.DisplayImageOptions.Builder#showImageOnLoading(
91     	 *android.graphics.drawable.Drawable) on loading} or
92     	 * {@linkplain com.nostra13.universalimageloader.core.DisplayImageOptions.Builder#showImageOnFail(
93     	 *android.graphics.drawable.Drawable) on loading fail}. These drawables can be specified in
94     	 * {@linkplain com.nostra13.universalimageloader.core.DisplayImageOptions display options}.
95     	 *
96     	 * @return <b>true</b> if drawable was set successfully; <b>false</b> - otherwise
97     	 */
98     	boolean setImageDrawable(Drawable drawable);
99     
100     	/**
101     	 * Sets image bitmap into this image aware view.<br />
102     	 * Called on UI thread to display loaded and decoded image {@link android.graphics.Bitmap} in this image view aware.
103     	 * Actually it's used only in
104     	 * {@link com.nostra13.universalimageloader.core.display.BitmapDisplayer BitmapDisplayer}.
105     	 *
106     	 * @return <b>true</b> if bitmap was set successfully; <b>false</b> - otherwise
107     	 */
108     	boolean setImageBitmap(Bitmap bitmap);
109     }
