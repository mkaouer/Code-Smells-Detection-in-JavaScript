16     package com.nostra13.universalimageloader.core;
17     
18     import android.view.View;
19     import com.nostra13.universalimageloader.core.assist.FailReason;
20     import com.nostra13.universalimageloader.core.assist.FlushedInputStream;
21     import com.nostra13.universalimageloader.core.assist.ImageLoadingListener;
22     import com.nostra13.universalimageloader.core.imageaware.ImageAware;
23     
24     import java.util.Collections;
25     import java.util.HashMap;
26     import java.util.Map;
27     import java.util.WeakHashMap;
28     import java.util.concurrent.Executor;
29     import java.util.concurrent.ExecutorService;
30     import java.util.concurrent.Executors;
31     import java.util.concurrent.atomic.AtomicBoolean;
32     import java.util.concurrent.locks.ReentrantLock;
33     
34     /**
35      * {@link ImageLoader} engine which responsible for {@linkplain LoadAndDisplayImageTask display task} execution.
36      *
37      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
38      * @since 1.7.1
39      */
40     class ImageLoaderEngine {
41     
42     	final ImageLoaderConfiguration configuration;
43     
44     	private Executor taskExecutor;
45     	private Executor taskExecutorForCachedImages;
46     	private ExecutorService taskDistributor;
47     
48     	private final Map<Integer, String> cacheKeysForImageAwares = Collections
49     			.synchronizedMap(new HashMap<Integer, String>());
50     	private final Map<String, ReentrantLock> uriLocks = new WeakHashMap<String, ReentrantLock>();
51     
52     	private final AtomicBoolean paused = new AtomicBoolean(false);
53     	private final AtomicBoolean networkDenied = new AtomicBoolean(false);
54     	private final AtomicBoolean slowNetwork = new AtomicBoolean(false);
55     
56     	ImageLoaderEngine(ImageLoaderConfiguration configuration) {
57     		this.configuration = configuration;
58     
59     		taskExecutor = configuration.taskExecutor;
60     		taskExecutorForCachedImages = configuration.taskExecutorForCachedImages;
61     
62     		taskDistributor = Executors.newCachedThreadPool();
63     	}
64     
65     	/** Submits task to execution pool */
66     	void submit(final LoadAndDisplayImageTask task) {
67     		taskDistributor.execute(new Runnable() {
68     			@Override
69     			public void run() {
70     				boolean isImageCachedOnDisc = configuration.discCache.get(task.getLoadingUri()).exists();
71     				initExecutorsIfNeed();
72     				if (isImageCachedOnDisc) {
73     					taskExecutorForCachedImages.execute(task);
74     				} else {
75     					taskExecutor.execute(task);
76     				}
77     			}
78     		});
79     	}
80     
81     	/** Submits task to execution pool */
82     	void submit(ProcessAndDisplayImageTask task) {
83     		initExecutorsIfNeed();
84     		taskExecutorForCachedImages.execute(task);
85     	}
86     
87     	private void initExecutorsIfNeed() {
88     		if (!configuration.customExecutor && ((ExecutorService) taskExecutor).isShutdown()) {
89     			taskExecutor = createTaskExecutor();
90     		}
91     		if (!configuration.customExecutorForCachedImages && ((ExecutorService) taskExecutorForCachedImages)
92     				.isShutdown()) {
93     			taskExecutorForCachedImages = createTaskExecutor();
94     		}
95     	}
96     
97     	private Executor createTaskExecutor() {
98     		return DefaultConfigurationFactory
99     				.createExecutor(configuration.threadPoolSize, configuration.threadPriority, configuration.tasksProcessingType);
100     	}
101     
102     	/**
103     	 * Returns URI of image which is loading at this moment into passed
104     	 * {@link com.nostra13.universalimageloader.core.imageaware.ImageAware}
105     	 */
106     	String getLoadingUriForView(ImageAware imageAware) {
107     		return cacheKeysForImageAwares.get(imageAware.getId());
108     	}
109     
110     	/**
111     	 * Associates <b>memoryCacheKey</b> with <b>imageAware</b>. Then it helps to define image URI is loaded into
112     	 * View at exact moment.
113     	 */
114     	void prepareDisplayTaskFor(ImageAware imageAware, String memoryCacheKey) {
115     		cacheKeysForImageAwares.put(imageAware.getId(), memoryCacheKey);
116     	}
117     
118     	/**
119     	 * Cancels the task of loading and displaying image for incoming <b>imageAware</b>.
120     	 *
121     	 * @param imageAware {@link com.nostra13.universalimageloader.core.imageaware.ImageAware} for which display task
122     	 *                   will be cancelled
123     	 */
124     	void cancelDisplayTaskFor(ImageAware imageAware) {
125     		cacheKeysForImageAwares.remove(imageAware.getId());
126     	}
127     
128     	/**
129     	 * Denies or allows engine to download images from the network.<br />
130     	 * <br />
131     	 * If downloads are denied and if image isn't cached then
132     	 * {@link ImageLoadingListener#onLoadingFailed(String, View, FailReason)} callback will be fired with
133     	 * {@link FailReason.FailType#NETWORK_DENIED}
134     	 *
135     	 * @param denyNetworkDownloads pass <b>true</b> - to deny engine to download images from the network; <b>false</b> -
136     	 *                             to allow engine to download images from network.
137     	 */
138     	void denyNetworkDownloads(boolean denyNetworkDownloads) {
139     		networkDenied.set(denyNetworkDownloads);
140     	}
141     
142     	/**
143     	 * Sets option whether ImageLoader will use {@link FlushedInputStream} for network downloads to handle <a
144     	 * href="http://code.google.com/p/android/issues/detail?id=6066">this known problem</a> or not.
145     	 *
146     	 * @param handleSlowNetwork pass <b>true</b> - to use {@link FlushedInputStream} for network downloads; <b>false</b>
147     	 *                          - otherwise.
148     	 */
149     	void handleSlowNetwork(boolean handleSlowNetwork) {
150     		slowNetwork.set(handleSlowNetwork);
151     	}
152     
153     	/**
154     	 * Pauses engine. All new "load&display" tasks won't be executed until ImageLoader is {@link #resume() resumed}.<br />
155     	 * Already running tasks are not paused.
156     	 */
157     	void pause() {
158     		paused.set(true);
159     	}
160     
161     	/** Resumes engine work. Paused "load&display" tasks will continue its work. */
162     	void resume() {
163     		synchronized (paused) {
164     			paused.set(false);
165     			paused.notifyAll();
166     		}
167     	}
168     
169     	/** Stops engine, cancels all running and scheduled display image tasks. Clears internal data. */
170     	void stop() {
171     		if (!configuration.customExecutor) {
172     			((ExecutorService) taskExecutor).shutdownNow();
173     		}
174     		if (!configuration.customExecutorForCachedImages) {
175     			((ExecutorService) taskExecutorForCachedImages).shutdownNow();
176     		}
177     
178     		cacheKeysForImageAwares.clear();
179     		uriLocks.clear();
180     	}
181     
182     	ReentrantLock getLockForUri(String uri) {
183     		ReentrantLock lock = uriLocks.get(uri);
184     		if (lock == null) {
185     			lock = new ReentrantLock();
186     			uriLocks.put(uri, lock);
187     		}
188     		return lock;
189     	}
190     
191     	AtomicBoolean getPause() {
192     		return paused;
193     	}
194     
195     	boolean isNetworkDenied() {
196     		return networkDenied.get();
197     	}
198     
199     	boolean isSlowNetwork() {
200     		return slowNetwork.get();
201     	}
202     }
