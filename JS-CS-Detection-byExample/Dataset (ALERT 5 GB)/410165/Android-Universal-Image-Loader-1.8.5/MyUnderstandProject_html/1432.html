41     /**
42      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
43      */
44     public class ImageListActivity extends AbsListViewBaseActivity {
45     
46     	DisplayImageOptions options;
47     
48     	String[] imageUrls;
49     
50     	@Override
51     	public void onCreate(Bundle savedInstanceState) {
52     		super.onCreate(savedInstanceState);
53     		setContentView(R.layout.ac_image_list);
54     
55     		Bundle bundle = getIntent().getExtras();
56     		imageUrls = bundle.getStringArray(Extra.IMAGES);
57     
58     		options = new DisplayImageOptions.Builder()
59     			.showStubImage(R.drawable.ic_stub)
60     			.showImageForEmptyUri(R.drawable.ic_empty)
61     			.showImageOnFail(R.drawable.ic_error)
62     			.cacheInMemory(true)
63     			.cacheOnDisc(true)
64     			.displayer(new RoundedBitmapDisplayer(20))
65     			.build();
66     
67     		listView = (ListView) findViewById(android.R.id.list);
68     		((ListView) listView).setAdapter(new ItemAdapter());
69     		listView.setOnItemClickListener(new OnItemClickListener() {
70     			@Override
71     			public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
72     				startImagePagerActivity(position);
73     			}
74     		});
75     	}
76     
77     	@Override
78     	public void onBackPressed() {
79     		AnimateFirstDisplayListener.displayedImages.clear();
80     		super.onBackPressed();
81     	}
82     
83     	private void startImagePagerActivity(int position) {
84     		Intent intent = new Intent(this, ImagePagerActivity.class);
85     		intent.putExtra(Extra.IMAGES, imageUrls);
86     		intent.putExtra(Extra.IMAGE_POSITION, position);
87     		startActivity(intent);
88     	}
89     
90     	class ItemAdapter extends BaseAdapter {
91     
92     		private ImageLoadingListener animateFirstListener = new AnimateFirstDisplayListener();
93     
94     		private class ViewHolder {
95     			public TextView text;
96     			public ImageView image;
97     		}
98     
99     		@Override
100     		public int getCount() {
101     			return imageUrls.length;
102     		}
103     
104     		@Override
105     		public Object getItem(int position) {
106     			return position;
107     		}
108     
109     		@Override
110     		public long getItemId(int position) {
111     			return position;
112     		}
113     
114     		@Override
115     		public View getView(final int position, View convertView, ViewGroup parent) {
116     			View view = convertView;
117     			final ViewHolder holder;
118     			if (convertView == null) {
119     				view = getLayoutInflater().inflate(R.layout.item_list_image, parent, false);
120     				holder = new ViewHolder();
121     				holder.text = (TextView) view.findViewById(R.id.text);
122     				holder.image = (ImageView) view.findViewById(R.id.image);
123     				view.setTag(holder);
124     			} else {
125     				holder = (ViewHolder) view.getTag();
126     			}
127     
128     			holder.text.setText("Item " + (position + 1));
129     
130     			imageLoader.displayImage(imageUrls[position], holder.image, options, animateFirstListener);
131     
132     			return view;
133     		}
134     	}
135     
136     	private static class AnimateFirstDisplayListener extends SimpleImageLoadingListener {
137     
138     		static final List<String> displayedImages = Collections.synchronizedList(new LinkedList<String>());
139     
140     		@Override
141     		public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
142     			if (loadedImage != null) {
143     				ImageView imageView = (ImageView) view;
144     				boolean firstDisplay = !displayedImages.contains(imageUri);
145     				if (firstDisplay) {
146     					FadeInBitmapDisplayer.animate(imageView, 500);
147     					displayedImages.add(imageUri);
148     				}
149     			}
150     		}
151     	}
152     }
