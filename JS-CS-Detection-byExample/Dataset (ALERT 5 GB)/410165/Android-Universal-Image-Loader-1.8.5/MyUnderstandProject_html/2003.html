35     public class HomeActivity extends BaseActivity {
36     
37     	private static final String TEST_FILE_NAME = "Universal Image Loader @#&=+-_.,!()~'%20.png";
38     
39     	@Override
40     	public void onCreate(Bundle savedInstanceState) {
41     		super.onCreate(savedInstanceState);
42     		setContentView(R.layout.ac_home);
43     
44     		File testImageOnSdCard = new File("/mnt/sdcard", TEST_FILE_NAME);
45     		if (!testImageOnSdCard.exists()) {
46     			copyTestImageToSdCard(testImageOnSdCard);
47     		}
48     	}
49

74     	@Override
75     	public void onBackPressed() {
76     		imageLoader.stop();
77     		super.onBackPressed();
78     	}
79     
80     	private void copyTestImageToSdCard(final File testImageOnSdCard) {
81     		new Thread(new Runnable() {
82     			@Override
83     			public void run() {
84     				try {
85     					InputStream is = getAssets().open(TEST_FILE_NAME);
86     					FileOutputStream fos = new FileOutputStream(testImageOnSdCard);
87     					byte[] buffer = new byte[8192];
88     					int read;
89     					try {
90     						while ((read = is.read(buffer)) != -1) {
91     							fos.write(buffer, 0, read);
92     						}
93     					} finally {
94     						fos.flush();
95     						fos.close();
96     						is.close();
97     					}
98     				} catch (IOException e) {
99     					L.w("Can't copy test image onto SD card");
100     				}
101     			}
102     		}).start();
103     	}
104     }
