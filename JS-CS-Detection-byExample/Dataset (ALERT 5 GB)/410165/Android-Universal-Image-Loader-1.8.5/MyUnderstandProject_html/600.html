16     package com.nostra13.example.universalimageloader.widget;
17     
18     import static com.nostra13.example.universalimageloader.Constants.IMAGES;
19     
20     import android.appwidget.AppWidgetManager;
21     import android.appwidget.AppWidgetProvider;
22     import android.content.Context;
23     import android.graphics.Bitmap;
24     import android.view.View;
25     import android.widget.RemoteViews;
26     
27     import com.nostra13.example.universalimageloader.R;
28     import com.nostra13.example.universalimageloader.UILApplication;
29     import com.nostra13.universalimageloader.core.DisplayImageOptions;
30     import com.nostra13.universalimageloader.core.ImageLoader;
31     import com.nostra13.universalimageloader.core.assist.ImageSize;
32     import com.nostra13.universalimageloader.core.assist.SimpleImageLoadingListener;
33     import com.nostra13.universalimageloader.core.display.FakeBitmapDisplayer;
34     
35     /**
36      * Example widget provider
37      * 
38      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
39      */
40     public class UILWidgetProvider extends AppWidgetProvider {
41     
42     	private static DisplayImageOptions optionsWithFakeDisplayer;
43     
44     	static {
45     		optionsWithFakeDisplayer = new DisplayImageOptions.Builder().displayer(new FakeBitmapDisplayer()).build();
46     	}
47     
48     	@Override
49     	public void onUpdate(Context context, AppWidgetManager appWidgetManager, int[] appWidgetIds) {
50     		UILApplication.initImageLoader(context);
51     
52     		final int widgetCount = appWidgetIds.length;
53     		for (int i = 0; i < widgetCount; i++) {
54     			int appWidgetId = appWidgetIds[i];
55     			updateAppWidget(context, appWidgetManager, appWidgetId);
56     		}
57     	}
58     
59     	static void updateAppWidget(Context context, final AppWidgetManager appWidgetManager, final int appWidgetId) {
60     		final RemoteViews views = new RemoteViews(context.getPackageName(), R.layout.widget);
61     
62     		ImageSize minImageSize = new ImageSize(70, 70); // 70 - approximate size of ImageView in widget
63     		ImageLoader.getInstance().loadImage(IMAGES[0], minImageSize, optionsWithFakeDisplayer, new SimpleImageLoadingListener() {
64     			@Override
65     			public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
66     				views.setImageViewBitmap(R.id.image_left, loadedImage);
67     				appWidgetManager.updateAppWidget(appWidgetId, views);
68     			}
69     		});
70     		ImageLoader.getInstance().loadImage(IMAGES[1], minImageSize, optionsWithFakeDisplayer, new SimpleImageLoadingListener() {
71     			@Override
72     			public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
73     				views.setImageViewBitmap(R.id.image_right, loadedImage);
74     				appWidgetManager.updateAppWidget(appWidgetId, views);
75     			}
76     		});
77     	}
78     }
