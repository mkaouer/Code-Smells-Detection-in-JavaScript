36      * Presents load'n'display image task. Used to load image from Internet or file system, decode it to {@link Bitmap}, and
37      * display it in {@link ImageView} using {@link DisplayBitmapTask}.
38      *
39      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
40      * @see ImageLoaderConfiguration
41      * @see ImageLoadingInfo
42      * @since 1.3.1

88     	public LoadAndDisplayImageTask(ImageLoaderEngine engine, ImageLoadingInfo imageLoadingInfo, Handler handler) {
89     		this.engine = engine;
90     		this.imageLoadingInfo = imageLoadingInfo;
91     		this.handler = handler;
92     
93     		configuration = engine.configuration;
94     		downloader = configuration.downloader;
95     		networkDeniedDownloader = configuration.networkDeniedDownloader;
96     		slowNetworkDownloader = configuration.slowNetworkDownloader;
97     		decoder = configuration.decoder;
98     		loggingEnabled = configuration.loggingEnabled;
99     		uri = imageLoadingInfo.uri;
100     		memoryCacheKey = imageLoadingInfo.memoryCacheKey;
101     		imageView = imageLoadingInfo.imageView;
102     		targetSize = imageLoadingInfo.targetSize;
103     		options = imageLoadingInfo.options;
104     		listener = imageLoadingInfo.listener;
105     	}

107     	@Override
108     	public void run() {
109     		if (waitIfPaused()) return;
110     		if (delayIfNeed()) return;
111     
112     		ReentrantLock loadFromUriLock = imageLoadingInfo.loadFromUriLock;
113     		log(LOG_START_DISPLAY_IMAGE_TASK);
114     		if (loadFromUriLock.isLocked()) {
115     			log(LOG_WAITING_FOR_IMAGE_LOADED);
116     		}
117     
118     		loadFromUriLock.lock();
119     		Bitmap bmp;
120     		try {
121     			if (checkTaskIsNotActual()) return;
122     
123     			bmp = configuration.memoryCache.get(memoryCacheKey);
124     			if (bmp == null) {
125     				bmp = tryLoadBitmap();
126     				if (bmp == null) return; // listener callback already was fired
127     
128     				if (checkTaskIsNotActual() || checkTaskIsInterrupted()) return;
129     
130     				if (options.shouldPreProcess()) {
131     					log(LOG_PREPROCESS_IMAGE);
132     					bmp = options.getPreProcessor().process(bmp);
133     					if (bmp == null) {
134     						L.e(ERROR_PRE_PROCESSOR_NULL);
135     					}
136     				}
137     
138     				if (bmp != null && options.isCacheInMemory()) {
139     					log(LOG_CACHE_IMAGE_IN_MEMORY);
140     					configuration.memoryCache.put(memoryCacheKey, bmp);
141     				}
142     			} else {
143     				loadedFrom = LoadedFrom.MEMORY_CACHE;
144     				log(LOG_GET_IMAGE_FROM_MEMORY_CACHE_AFTER_WAITING);
145     			}
146     
147     			if (bmp != null && options.shouldPostProcess()) {
148     				log(LOG_POSTPROCESS_IMAGE);
149     				bmp = options.getPostProcessor().process(bmp);
150     				if (bmp == null) {
151     					L.e(ERROR_POST_PROCESSOR_NULL, memoryCacheKey);
152     				}
153     			}
154     		} finally {
155     			loadFromUriLock.unlock();
156     		}
157     
158     		if (checkTaskIsNotActual() || checkTaskIsInterrupted()) return;
159     
160     		DisplayBitmapTask displayBitmapTask = new DisplayBitmapTask(bmp, imageLoadingInfo, engine, loadedFrom);
161     		displayBitmapTask.setLoggingEnabled(loggingEnabled);
162     		handler.post(displayBitmapTask);
163     	}

165     	/** @return true - if task should be interrupted; false - otherwise */
166     	private boolean waitIfPaused() {
167     		AtomicBoolean pause = engine.getPause();
168     		if (pause.get()) {
169     			synchronized (pause) {
170     				log(LOG_WAITING_FOR_RESUME);
171     				try {
172     					pause.wait();
173     				} catch (InterruptedException e) {
174     					L.e(LOG_TASK_INTERRUPTED, memoryCacheKey);
175     					return true;
176     				}
177     				log(LOG_RESUME_AFTER_PAUSE);
178     			}
179     		}
180     		return checkTaskIsNotActual();
181     	}
182     
183     	/** @return true - if task should be interrupted; false - otherwise */
184     	private boolean delayIfNeed() {
185     		if (options.shouldDelayBeforeLoading()) {
186     			log(LOG_DELAY_BEFORE_LOADING, options.getDelayBeforeLoading(), memoryCacheKey);
187     			try {
188     				Thread.sleep(options.getDelayBeforeLoading());
189     			} catch (InterruptedException e) {
190     				L.e(LOG_TASK_INTERRUPTED, memoryCacheKey);
191     				return true;
192     			}
193     			return checkTaskIsNotActual();
194     		}
195     		return false;
196     	}

198     	/**
199     	 * Check whether the image URI of this task matches to image URI which is actual for current ImageView at this
200     	 * moment and fire {@link ImageLoadingListener#onLoadingCancelled(String, android.view.View)}} event if it doesn't.
201     	 */
202     	private boolean checkTaskIsNotActual() {
203     		String currentCacheKey = engine.getLoadingUriForView(imageView);
204     		// Check whether memory cache key (image URI) for current ImageView is actual. 
205     		// If ImageView is reused for another task then current task should be cancelled.
206     		boolean imageViewWasReused = !memoryCacheKey.equals(currentCacheKey);
207     		if (imageViewWasReused) {
208     			handler.post(new Runnable() {
209     				@Override
210     				public void run() {
211     					listener.onLoadingCancelled(uri, imageView);
212     				}
213     			});
214     			log(LOG_TASK_CANCELLED);
215     		}
216     		return imageViewWasReused;
217     	}
218     
219     	/** Check whether the current task was interrupted */
220     	private boolean checkTaskIsInterrupted() {
221     		boolean interrupted = Thread.interrupted();
222     		if (interrupted) log(LOG_TASK_INTERRUPTED);
223     		return interrupted;
224     	}

226     	private Bitmap tryLoadBitmap() {
227     		File imageFile = getImageFileInDiscCache();
228     
229     		Bitmap bitmap = null;
230     		try {
231     			if (imageFile.exists()) {
232     				log(LOG_LOAD_IMAGE_FROM_DISC_CACHE);
233     
234     				loadedFrom = LoadedFrom.DISC_CACHE;
235     				bitmap = decodeImage(Scheme.FILE.wrap(imageFile.getAbsolutePath()));
236     			}
237     			if (bitmap == null || bitmap.getWidth() <= 0 || bitmap.getHeight() <= 0) {
238     				log(LOG_LOAD_IMAGE_FROM_NETWORK);
239     
240     				loadedFrom = LoadedFrom.NETWORK;
241     				String imageUriForDecoding = options.isCacheOnDisc() ? tryCacheImageOnDisc(imageFile) : uri;
242     				if (!checkTaskIsNotActual()) {
243     					bitmap = decodeImage(imageUriForDecoding);
244     					if (bitmap == null || bitmap.getWidth() <= 0 || bitmap.getHeight() <= 0) {
245     						fireImageLoadingFailedEvent(FailType.DECODING_ERROR, null);
246     					}
247     				}
248     			}
249     		} catch (IllegalStateException e) {
250     			fireImageLoadingFailedEvent(FailType.NETWORK_DENIED, null);
251     		} catch (IOException e) {
252     			L.e(e);
253     			fireImageLoadingFailedEvent(FailType.IO_ERROR, e);
254     			if (imageFile.exists()) {
255     				imageFile.delete();
256     			}
257     		} catch (OutOfMemoryError e) {
258     			L.e(e);
259     			fireImageLoadingFailedEvent(FailType.OUT_OF_MEMORY, e);
260     		} catch (Throwable e) {
261     			L.e(e);
262     			fireImageLoadingFailedEvent(FailType.UNKNOWN, e);
263     		}
264     		return bitmap;
265     	}

267     	private File getImageFileInDiscCache() {
268     		DiscCacheAware discCache = configuration.discCache;
269     		File imageFile = discCache.get(uri);
270     		File cacheDir = imageFile.getParentFile();
271     		if (cacheDir == null || (!cacheDir.exists() && !cacheDir.mkdirs())) {
272     			imageFile = configuration.reserveDiscCache.get(uri);
273     			cacheDir = imageFile.getParentFile();
274     			if (cacheDir != null && !cacheDir.exists()) {
275     				cacheDir.mkdirs();
276     			}
277     		}
278     		return imageFile;
279     	}

281     	private Bitmap decodeImage(String imageUri) throws IOException {
282     		ViewScaleType viewScaleType = ViewScaleType.fromImageView(imageView);
283     		ImageDecodingInfo decodingInfo = new ImageDecodingInfo(memoryCacheKey, imageUri, targetSize, viewScaleType, getDownloader(), options);
284     		return decoder.decode(decodingInfo);
285     	}

287     	/** @return Cached image URI; or original image URI if caching failed */
288     	private String tryCacheImageOnDisc(File targetFile) {
289     		log(LOG_CACHE_IMAGE_ON_DISC);
290     
291     		try {
292     			int width = configuration.maxImageWidthForDiscCache;
293     			int height = configuration.maxImageHeightForDiscCache;
294     			boolean saved = false;
295     			if (width > 0 || height > 0) {
296     				saved = downloadSizedImage(targetFile, width, height);
297     			}
298     			if (!saved) {
299     				downloadImage(targetFile);
300     			}
301     
302     			configuration.discCache.put(uri, targetFile);
303     			return Scheme.FILE.wrap(targetFile.getAbsolutePath());
304     		} catch (IOException e) {
305     			L.e(e);
306     			return uri;
307     		}
308     	}

310     	private boolean downloadSizedImage(File targetFile, int maxWidth, int maxHeight) throws IOException {
311     		// Download, decode, compress and save image
312     		ImageSize targetImageSize = new ImageSize(maxWidth, maxHeight);
313     		DisplayImageOptions specialOptions = new DisplayImageOptions.Builder().cloneFrom(options).imageScaleType(ImageScaleType.IN_SAMPLE_INT).build();
314     		ImageDecodingInfo decodingInfo = new ImageDecodingInfo(memoryCacheKey, uri, targetImageSize, ViewScaleType.FIT_INSIDE, getDownloader(), specialOptions);
315     		Bitmap bmp = decoder.decode(decodingInfo);
316     		if (bmp == null) return false;
317     
318     		if (configuration.processorForDiscCache != null) {
319     			log(LOG_PROCESS_IMAGE_BEFORE_CACHE_ON_DISC);
320     			bmp = configuration.processorForDiscCache.process(bmp);
321     			if (bmp == null) {
322     				L.e(ERROR_PROCESSOR_FOR_DISC_CACHE_NULL, memoryCacheKey);
323     				return false;
324     			}
325     		}
326     
327     		OutputStream os = new BufferedOutputStream(new FileOutputStream(targetFile), BUFFER_SIZE);
328     		boolean savedSuccessfully;
329     		try {
330     			savedSuccessfully = bmp.compress(configuration.imageCompressFormatForDiscCache, configuration.imageQualityForDiscCache, os);
331     		} finally {
332     			IoUtils.closeSilently(os);
333     		}
334     		bmp.recycle();
335     		return savedSuccessfully;
336     	}
337     
338     	private void downloadImage(File targetFile) throws IOException {
339     		InputStream is = getDownloader().getStream(uri, options.getExtraForDownloader());
340     		try {
341     			OutputStream os = new BufferedOutputStream(new FileOutputStream(targetFile), BUFFER_SIZE);
342     			try {
343     				IoUtils.copyStream(is, os);
344     			} finally {
345     				IoUtils.closeSilently(os);
346     			}
347     		} finally {
348     			IoUtils.closeSilently(is);
349     		}
350     	}

352     	private void fireImageLoadingFailedEvent(final FailType failType, final Throwable failCause) {
353     		if (!Thread.interrupted()) {
354     			handler.post(new Runnable() {
355     				@Override
356     				public void run() {
357     					if (options.shouldShowImageOnFail()) {
358     						imageView.setImageResource(options.getImageOnFail());
359     					}
360     					listener.onLoadingFailed(uri, imageView, new FailReason(failType, failCause));
361     				}
362     			});
363     		}
364     	}

366     	private ImageDownloader getDownloader() {
367     		ImageDownloader d;
368     		if (engine.isNetworkDenied()) {
369     			d = networkDeniedDownloader;
370     		} else if (engine.isSlowNetwork()) {
371     			d = slowNetworkDownloader;
372     		} else {
373     			d = downloader;
374     		}
375     		return d;
376     	}
