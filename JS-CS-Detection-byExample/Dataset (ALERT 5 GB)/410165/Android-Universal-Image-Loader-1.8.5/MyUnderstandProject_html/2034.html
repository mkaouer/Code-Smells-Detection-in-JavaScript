16     package com.nostra13.universalimageloader.core.download;
17     
18     import com.nostra13.universalimageloader.core.DisplayImageOptions;
19     
20     import java.io.IOException;
21     import java.io.InputStream;
22     
23     /**
24      * Provides retrieving of {@link InputStream} of image by URI.<br />
25      * Implementations have to be thread-safe.
26      *
27      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
28      * @since 1.4.0
29      */
30     public interface ImageDownloader {
31     	/**
32     	 * Retrieves {@link InputStream} of image by URI.
33     	 *
34     	 * @param imageUri Image URI
35     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
36     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
37     	 * @return {@link InputStream} of image
38     	 * @throws IOException                   if some I/O error occurs during getting image stream
39     	 * @throws UnsupportedOperationException if image URI has unsupported scheme(protocol)
40     	 */
41     	InputStream getStream(String imageUri, Object extra) throws IOException;
42     
43     	/** Represents supported schemes(protocols) of URI. Provides convenient methods for work with schemes and URIs. */
44     	public enum Scheme {
45     		HTTP("http"), HTTPS("https"), FILE("file"), CONTENT("content"), ASSETS("assets"), DRAWABLE("drawable"), UNKNOWN("");
46     
47     		private String scheme;
48     		private String uriPrefix;
49     
50     		Scheme(String scheme) {
51     			this.scheme = scheme;
52     			uriPrefix = scheme + "://";
53     		}
54     
55     		/**
56     		 * Defines scheme of incoming URI
57     		 *
58     		 * @param uri URI for scheme detection
59     		 * @return Scheme of incoming URI
60     		 */
61     		public static Scheme ofUri(String uri) {
62     			if (uri != null) {
63     				for (Scheme s : values()) {
64     					if (s.belongsTo(uri)) {
65     						return s;
66     					}
67     				}
68     			}
69     			return UNKNOWN;
70     		}
71     
72     		private boolean belongsTo(String uri) {
73     			return uri.startsWith(uriPrefix);
74     		}
75     
76     		/** Appends scheme to incoming path */
77     		public String wrap(String path) {
78     			return uriPrefix + path;
79     		}
80     
81     		/** Removed scheme part ("scheme://") from incoming URI */
82     		public String crop(String uri) {
83     			if (!belongsTo(uri)) {
84     				throw new IllegalArgumentException(String.format("URI [%1$s] doesn't have expected scheme [%2$s]", uri, scheme));
85     			}
86     			return uri.substring(uriPrefix.length());
87     		}
88     	}
89     }
