16     package com.nostra13.universalimageloader.core.assist;
17     
18     import android.graphics.Bitmap;
19     import com.nostra13.universalimageloader.cache.memory.MemoryCacheAware;
20     import com.nostra13.universalimageloader.core.ImageLoaderConfiguration;
21     
22     import java.util.ArrayList;
23     import java.util.Comparator;
24     import java.util.List;
25     
26     /**
27      * Utility for generating of keys for memory cache, key comparing and other work with memory cache
28      *
29      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
30      * @since 1.6.3
31      */
32     public final class MemoryCacheUtil {
33     
34     	private static final String URI_AND_SIZE_SEPARATOR = "_";
35     	private static final String WIDTH_AND_HEIGHT_SEPARATOR = "x";
36     
37     	private MemoryCacheUtil() {
38     	}
39     
40     	/**
41     	 * Generates key for memory cache for incoming image (URI + size).<br />
42     	 * Pattern for cache key - <b>[imageUri]_[width]x[height]</b>.
43     	 */
44     	public static String generateKey(String imageUri, ImageSize targetSize) {
45     		return new StringBuilder(imageUri).append(URI_AND_SIZE_SEPARATOR).append(targetSize.getWidth()).append(WIDTH_AND_HEIGHT_SEPARATOR).append(targetSize.getHeight()).toString();
46     	}
47     
48     	public static Comparator<String> createFuzzyKeyComparator() {
49     		return new Comparator<String>() {
50     			@Override
51     			public int compare(String key1, String key2) {
52     				String imageUri1 = key1.substring(0, key1.lastIndexOf(URI_AND_SIZE_SEPARATOR));
53     				String imageUri2 = key2.substring(0, key2.lastIndexOf(URI_AND_SIZE_SEPARATOR));
54     				return imageUri1.compareTo(imageUri2);
55     			}
56     		};
57     	}
58     
59     	/**
60     	 * Searches all bitmaps in memory cache which are corresponded to incoming URI.<br />
61     	 * <b>Note:</b> Memory cache can contain multiple sizes of the same image if only you didn't set
62     	 * {@link ImageLoaderConfiguration.Builder#denyCacheImageMultipleSizesInMemory()
63     	 * denyCacheImageMultipleSizesInMemory()} option in {@linkplain ImageLoaderConfiguration configuration}
64     	 */
65     	public static List<Bitmap> findCachedBitmapsForImageUri(String imageUri, MemoryCacheAware<String, Bitmap> memoryCache) {
66     		List<Bitmap> values = new ArrayList<Bitmap>();
67     		for (String key : memoryCache.keys()) {
68     			if (key.startsWith(imageUri)) {
69     				values.add(memoryCache.get(key));
70     			}
71     		}
72     		return values;
73     	}
74     
75     	/**
76     	 * Searches all keys in memory cache which are corresponded to incoming URI.<br />
77     	 * <b>Note:</b> Memory cache can contain multiple sizes of the same image if only you didn't set
78     	 * {@link ImageLoaderConfiguration.Builder#denyCacheImageMultipleSizesInMemory()
79     	 * denyCacheImageMultipleSizesInMemory()} option in {@linkplain ImageLoaderConfiguration configuration}
80     	 */
81     	public static List<String> findCacheKeysForImageUri(String imageUri, MemoryCacheAware<String, Bitmap> memoryCache) {
82     		List<String> values = new ArrayList<String>();
83     		for (String key : memoryCache.keys()) {
84     			if (key.startsWith(imageUri)) {
85     				values.add(key);
86     			}
87     		}
88     		return values;
89     	}
90     
91     	/**
92     	 * Removes from memory cache all images for incoming URI.<br />
93     	 * <b>Note:</b> Memory cache can contain multiple sizes of the same image if only you didn't set
94     	 * {@link ImageLoaderConfiguration.Builder#denyCacheImageMultipleSizesInMemory()
95     	 * denyCacheImageMultipleSizesInMemory()} option in {@linkplain ImageLoaderConfiguration configuration}
96     	 */
97     	public static void removeFromCache(String imageUri, MemoryCacheAware<String, Bitmap> memoryCache) {
98     		List<String> keysToRemove = new ArrayList<String>();
99     		for (String key : memoryCache.keys()) {
100     			if (key.startsWith(imageUri)) {
101     				keysToRemove.add(key);
102     			}
103     		}
104     		for (String keyToRemove : keysToRemove) {
105     			memoryCache.remove(keyToRemove);
106     		}
107     	}
108     }
