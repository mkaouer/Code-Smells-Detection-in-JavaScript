16     package com.nostra13.universalimageloader.cache.disc.impl;
17     
18     import com.nostra13.universalimageloader.cache.disc.BaseDiscCache;
19     import com.nostra13.universalimageloader.cache.disc.naming.FileNameGenerator;
20     import com.nostra13.universalimageloader.core.DefaultConfigurationFactory;
21     
22     import java.io.File;
23     import java.util.Collections;
24     import java.util.HashMap;
25     import java.util.Map;
26     
27     /**
28      * Cache which deletes files which were loaded more than defined time. Cache size is unlimited.
29      *
30      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
31      * @see BaseDiscCache
32      * @since 1.3.1
33      */
34     public class LimitedAgeDiscCache extends BaseDiscCache {
35     
36     	private final long maxFileAge;
37     
38     	private final Map<File, Long> loadingDates = Collections.synchronizedMap(new HashMap<File, Long>());
39     
40     	/**
41     	 * @param cacheDir Directory for file caching
42     	 * @param maxAge   Max file age (in seconds). If file age will exceed this value then it'll be removed on next
43     	 *                 treatment (and therefore be reloaded).
44     	 */
45     	public LimitedAgeDiscCache(File cacheDir, long maxAge) {
46     		this(cacheDir, DefaultConfigurationFactory.createFileNameGenerator(), maxAge);
47     	}
48     
49     	/**
50     	 * @param cacheDir          Directory for file caching
51     	 * @param fileNameGenerator Name generator for cached files
52     	 * @param maxAge            Max file age (in seconds). If file age will exceed this value then it'll be removed on next
53     	 *                          treatment (and therefore be reloaded).
54     	 */
55     	public LimitedAgeDiscCache(File cacheDir, FileNameGenerator fileNameGenerator, long maxAge) {
56     		super(cacheDir, fileNameGenerator);
57     		this.maxFileAge = maxAge * 1000; // to milliseconds
58     	}
59     
60     	@Override
61     	public void put(String key, File file) {
62     		long currentTime = System.currentTimeMillis();
63     		file.setLastModified(currentTime);
64     		loadingDates.put(file, currentTime);
65     	}
66     
67     	@Override
68     	public File get(String key) {
69     		File file = super.get(key);
70     		if (file.exists()) {
71     			boolean cached;
72     			Long loadingDate = loadingDates.get(file);
73     			if (loadingDate == null) {
74     				cached = false;
75     				loadingDate = file.lastModified();
76     			} else {
77     				cached = true;
78     			}
79     
80     			if (System.currentTimeMillis() - loadingDate > maxFileAge) {
81     				file.delete();
82     				loadingDates.remove(file);
83     			} else if (!cached) {
84     				loadingDates.put(file, loadingDate);
85     			}
86     		}
87     		return file;
88     	}
89     }
