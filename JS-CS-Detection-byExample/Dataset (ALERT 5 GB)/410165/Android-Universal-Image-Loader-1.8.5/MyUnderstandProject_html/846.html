16     package com.nostra13.universalimageloader.core.decode;
17     
18     import android.annotation.TargetApi;
19     import android.graphics.BitmapFactory.Options;
20     import android.os.Build;
21     import com.nostra13.universalimageloader.core.DisplayImageOptions;
22     import com.nostra13.universalimageloader.core.assist.ImageScaleType;
23     import com.nostra13.universalimageloader.core.assist.ImageSize;
24     import com.nostra13.universalimageloader.core.assist.MemoryCacheUtil;
25     import com.nostra13.universalimageloader.core.assist.ViewScaleType;
26     import com.nostra13.universalimageloader.core.download.ImageDownloader;
27     
28     /**
29      * Contains needed information for decoding image to Bitmap
30      *
31      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
32      * @since 1.8.3
33      */
34     public class ImageDecodingInfo {
35     
36     	private final String imageKey;
37     	private final String imageUri;
38     	private final ImageSize targetSize;
39     
40     	private final ImageScaleType imageScaleType;
41     	private final ViewScaleType viewScaleType;
42     
43     	private final ImageDownloader downloader;
44     	private final Object extraForDownloader;
45     
46     	private final Options decodingOptions;
47     
48     	public ImageDecodingInfo(String imageKey, String imageUri, ImageSize targetSize, ViewScaleType viewScaleType, ImageDownloader downloader, DisplayImageOptions displayOptions) {
49     		this.imageKey = imageKey;
50     		this.imageUri = imageUri;
51     		this.targetSize = targetSize;
52     
53     		this.imageScaleType = displayOptions.getImageScaleType();
54     		this.viewScaleType = viewScaleType;
55     
56     		this.downloader = downloader;
57     		this.extraForDownloader = displayOptions.getExtraForDownloader();
58     
59     		decodingOptions = new Options();
60     		copyOptions(displayOptions.getDecodingOptions(), decodingOptions);
61     	}
62     
63     	private void copyOptions(Options srcOptions, Options destOptions) {
64     		destOptions.inDensity = srcOptions.inDensity;
65     		destOptions.inDither = srcOptions.inDither;
66     		destOptions.inInputShareable = srcOptions.inInputShareable;
67     		destOptions.inJustDecodeBounds = srcOptions.inJustDecodeBounds;
68     		destOptions.inPreferredConfig = srcOptions.inPreferredConfig;
69     		destOptions.inPurgeable = srcOptions.inPurgeable;
70     		destOptions.inSampleSize = srcOptions.inSampleSize;
71     		destOptions.inScaled = srcOptions.inScaled;
72     		destOptions.inScreenDensity = srcOptions.inScreenDensity;
73     		destOptions.inTargetDensity = srcOptions.inTargetDensity;
74     		destOptions.inTempStorage = srcOptions.inTempStorage;
75     		if (Build.VERSION.SDK_INT >= 10) copyOptions10(srcOptions, destOptions);
76     		if (Build.VERSION.SDK_INT >= 11) copyOptions11(srcOptions, destOptions);
77     	}
78     
79     	@TargetApi(10)
80     	private void copyOptions10(Options srcOptions, Options destOptions) {
81     		destOptions.inPreferQualityOverSpeed = srcOptions.inPreferQualityOverSpeed;
82     	}
83     
84     	@TargetApi(11)
85     	private void copyOptions11(Options srcOptions, Options destOptions) {
86     		destOptions.inBitmap = srcOptions.inBitmap;
87     		destOptions.inMutable = srcOptions.inMutable;
88     	}
89     
90     	/** @return Original {@linkplain MemoryCacheUtil#generateKey(String, ImageSize) image key} (used in memory cache). */
91     	public String getImageKey() {
92     		return imageKey;
93     	}
94     
95     	/** @return Image URI for decoding (usually image from disc cache) */
96     	public String getImageUri() {
97     		return imageUri;
98     	}
99     
100     	/**
101     	 * @return Target size for image. Decoded bitmap should close to this size according to {@linkplain ImageScaleType
102     	 *         image scale type} and {@linkplain ViewScaleType view scale type}.
103     	 */
104     	public ImageSize getTargetSize() {
105     		return targetSize;
106     	}
107     
108     	/**
109     	 * @return {@linkplain ImageScaleType Scale type for image sampling and scaling}. This parameter affects result size
110     	 *         of decoded bitmap.
111     	 */
112     	public ImageScaleType getImageScaleType() {
113     		return imageScaleType;
114     	}
115     
116     	/** @return {@linkplain ViewScaleType View scale type}. This parameter affects result size of decoded bitmap. */
117     	public ViewScaleType getViewScaleType() {
118     		return viewScaleType;
119     	}
120     
121     	/** @return Downloader for image loading */
122     	public ImageDownloader getDownloader() {
123     		return downloader;
124     	}
125     
126     	/** @return Auxiliary object for downloader */
127     	public Object getExtraForDownloader() {
128     		return extraForDownloader;
129     	}
130     
131     	/** @return Decoding options */
132     	public Options getDecodingOptions() {
133     		return decodingOptions;
134     	}
135     }
