34     /**
35      * Decodes images to {@link Bitmap}, scales them to needed size
36      *
37      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
38      * @see ImageDecodingInfo
39      * @since 1.8.3
40      */
41     public class BaseImageDecoder implements ImageDecoder {
42     
43     	protected static final String LOG_SABSAMPLE_IMAGE = "Subsample original image (%1$s) to %2$s (scale = %3$d) [%4$s]";
44     	protected static final String LOG_SCALE_IMAGE = "Scale subsampled image (%1$s) to %2$s (scale = %3$.5f) [%4$s]";
45     	protected static final String LOG_ROTATE_IMAGE = "Rotate image on %1$d\u00B0 [%2$s]";
46     	protected static final String LOG_FLIP_IMAGE = "Flip image horizontally [%s]";
47     	protected static final String ERROR_CANT_DECODE_IMAGE = "Image can't be decoded [%s]";
48     
49     	protected boolean loggingEnabled;
50     
51     	public BaseImageDecoder() {
52     	}
53     
54     	public BaseImageDecoder(boolean loggingEnabled) {
55     		this.loggingEnabled = loggingEnabled;
56     	}

58     	/**
59     	 * Decodes image from URI into {@link Bitmap}. Image is scaled close to incoming {@linkplain ImageSize target size}
60     	 * during decoding (depend on incoming parameters).
61     	 *
62     	 * @param decodingInfo Needed data for decoding image
63     	 * @return Decoded bitmap
64     	 * @throws IOException                   if some I/O exception occurs during image reading
65     	 * @throws UnsupportedOperationException if image URI has unsupported scheme(protocol)
66     	 */
67     	public Bitmap decode(ImageDecodingInfo decodingInfo) throws IOException {
68     		InputStream imageStream = getImageStream(decodingInfo);
69     		ImageFileInfo imageInfo = defineImageSizeAndRotation(imageStream, decodingInfo.getImageUri());
70     		Options decodingOptions = prepareDecodingOptions(imageInfo.imageSize, decodingInfo);
71     		imageStream = getImageStream(decodingInfo);
72     		Bitmap decodedBitmap = decodeStream(imageStream, decodingOptions);
73     		if (decodedBitmap == null) {
74     			L.e(ERROR_CANT_DECODE_IMAGE, decodingInfo.getImageKey());
75     		} else {
76     			decodedBitmap = considerExactScaleAndOrientaiton(decodedBitmap, decodingInfo, imageInfo.exif.rotation, imageInfo.exif.flipHorizontal);
77     		}
78     		return decodedBitmap;
79     	}
80     
81     	protected InputStream getImageStream(ImageDecodingInfo decodingInfo) throws IOException {
82     		return decodingInfo.getDownloader().getStream(decodingInfo.getImageUri(), decodingInfo.getExtraForDownloader());
83     	}
84     
85     	protected ImageFileInfo defineImageSizeAndRotation(InputStream imageStream, String imageUri) throws IOException {
86     		Options options = new Options();
87     		options.inJustDecodeBounds = true;
88     		try {
89     			BitmapFactory.decodeStream(imageStream, null, options);
90     		} finally {
91     			IoUtils.closeSilently(imageStream);
92     		}
93     
94     		ExifInfo exif;
95     		if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.ECLAIR) {
96     			exif = defineExifOrientation(imageUri, options.outMimeType);
97     		} else {
98     			exif = new ExifInfo();
99     		}
100     		return new ImageFileInfo(new ImageSize(options.outWidth, options.outHeight, exif.rotation), exif);
101     	}
102     
103     	protected ExifInfo defineExifOrientation(String imageUri, String mimeType) {
104     		int rotation = 0;
105     		boolean flip = false;
106     		if ("image/jpeg".equalsIgnoreCase(mimeType) && Scheme.ofUri(imageUri) == Scheme.FILE) {
107     			try {
108     				ExifInterface exif = new ExifInterface(Scheme.FILE.crop(imageUri));
109     				int exifOrientation = exif.getAttributeInt(ExifInterface.TAG_ORIENTATION, ExifInterface.ORIENTATION_NORMAL);
110     				switch (exifOrientation) {
111     					case ExifInterface.ORIENTATION_FLIP_HORIZONTAL:
112     						flip = true;
113     					case ExifInterface.ORIENTATION_NORMAL:
114     						rotation = 0;
115     						break;
116     					case ExifInterface.ORIENTATION_TRANSVERSE:
117     						flip = true;
118     					case ExifInterface.ORIENTATION_ROTATE_90:
119     						rotation = 90;
120     						break;
121     					case ExifInterface.ORIENTATION_FLIP_VERTICAL:
122     						flip = true;
123     					case ExifInterface.ORIENTATION_ROTATE_180:
124     						rotation = 180;
125     						break;
126     					case ExifInterface.ORIENTATION_TRANSPOSE:
127     						flip = true;
128     					case ExifInterface.ORIENTATION_ROTATE_270:
129     						rotation = 270;
130     						break;
131     				}
132     			} catch (IOException e) {
133     				L.w("Can't read EXIF tags from file [%s]", imageUri);
134     			}
135     		}
136     		return new ExifInfo(rotation, flip);
137     	}

139     	protected Options prepareDecodingOptions(ImageSize imageSize, ImageDecodingInfo decodingInfo) {
140     		ImageScaleType scaleType = decodingInfo.getImageScaleType();
141     		ImageSize targetSize = decodingInfo.getTargetSize();
142     		int scale = 1;
143     		if (scaleType != ImageScaleType.NONE) {
144     			boolean powerOf2 = scaleType == ImageScaleType.IN_SAMPLE_POWER_OF_2;
145     			scale = ImageSizeUtils.computeImageSampleSize(imageSize, targetSize, decodingInfo.getViewScaleType(), powerOf2);
146     
147     			if (loggingEnabled) L.i(LOG_SABSAMPLE_IMAGE, imageSize, imageSize.scaleDown(scale), scale, decodingInfo.getImageKey());
148     		}
149     		Options decodingOptions = decodingInfo.getDecodingOptions();
150     		decodingOptions.inSampleSize = scale;
151     		return decodingOptions;
152     	}
153     
154     	protected Bitmap decodeStream(InputStream imageStream, Options decodingOptions) throws IOException {
155     		try {
156     			return BitmapFactory.decodeStream(imageStream, null, decodingOptions);
157     		} finally {
158     			IoUtils.closeSilently(imageStream);
159     		}
160     	}
161     
162     	protected Bitmap considerExactScaleAndOrientaiton(Bitmap subsampledBitmap, ImageDecodingInfo decodingInfo, int rotation, boolean flipHorizontal) {
163     		Matrix m = new Matrix();
164     		// Scale to exact size if need
165     		ImageScaleType scaleType = decodingInfo.getImageScaleType();
166     		if (scaleType == ImageScaleType.EXACTLY || scaleType == ImageScaleType.EXACTLY_STRETCHED) {
167     			ImageSize srcSize = new ImageSize(subsampledBitmap.getWidth(), subsampledBitmap.getHeight(), rotation);
168     			float scale = ImageSizeUtils.computeImageScale(srcSize, decodingInfo.getTargetSize(), decodingInfo.getViewScaleType(), scaleType == ImageScaleType.EXACTLY_STRETCHED);
169     			if (Float.compare(scale, 1f) != 0) {
170     				m.setScale(scale, scale);
171     
172     				if (loggingEnabled) L.i(LOG_SCALE_IMAGE, srcSize, srcSize.scale(scale), scale, decodingInfo.getImageKey());
173     			}
174     		}
175     		// Flip bitmap if need
176     		if (flipHorizontal) {
177     			m.postScale(-1, 1);
178     
179     			if (loggingEnabled) L.i(LOG_FLIP_IMAGE, decodingInfo.getImageKey());
180     		}
181     		// Rotate bitmap if need
182     		if (rotation != 0) {
183     			m.postRotate(rotation);
184     
185     			if (loggingEnabled) L.i(LOG_ROTATE_IMAGE, rotation, decodingInfo.getImageKey());
186     		}
187     
188     		Bitmap finalBitmap = Bitmap.createBitmap(subsampledBitmap, 0, 0, subsampledBitmap.getWidth(), subsampledBitmap.getHeight(), m, true);
189     		if (finalBitmap != subsampledBitmap) {
190     			subsampledBitmap.recycle();
191     		}
192     		return finalBitmap;
193     	}
