16     package com.nostra13.universalimageloader.core.download;
17     
18     import android.content.ContentResolver;
19     import android.content.Context;
20     import android.graphics.Bitmap;
21     import android.graphics.Bitmap.CompressFormat;
22     import android.graphics.drawable.BitmapDrawable;
23     import android.net.Uri;
24     import com.nostra13.universalimageloader.core.DisplayImageOptions;
25     
26     import java.io.*;
27     import java.net.HttpURLConnection;
28     import java.net.URL;
29     import java.net.URLConnection;
30     
31     /**
32      * Provides retrieving of {@link InputStream} of image by URI from network or file system or app resources.<br />
33      * {@link URLConnection} is used to retrieve image stream from network.
34      *
35      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
36      * @see HttpClientImageDownloader
37      * @since 1.8.0
38      */
39     public class BaseImageDownloader implements ImageDownloader {
40     	/** {@value} */
41     	public static final int DEFAULT_HTTP_CONNECT_TIMEOUT = 5 * 1000; // milliseconds
42     	/** {@value} */
43     	public static final int DEFAULT_HTTP_READ_TIMEOUT = 20 * 1000; // milliseconds
44     
45     	/** {@value} */
46     	protected static final int BUFFER_SIZE = 32 * 1024; // 32 Kb
47     	/** {@value} */
48     	protected static final String ALLOWED_URI_CHARS = "@#&=*+-_.,:!?()/~'%";
49     
50     	protected static final int MAX_REDIRECT_COUNT = 5;
51     
52     	private static final String ERROR_UNSUPPORTED_SCHEME = "UIL doesn't support scheme(protocol) by default [%s]. "
53     			+ "You should implement this support yourself (BaseImageDownloader.getStreamFromOtherSource(...))";
54     
55     	protected final Context context;
56     	protected final int connectTimeout;
57     	protected final int readTimeout;
58     
59     	public BaseImageDownloader(Context context) {
60     		this.context = context.getApplicationContext();
61     		this.connectTimeout = DEFAULT_HTTP_CONNECT_TIMEOUT;
62     		this.readTimeout = DEFAULT_HTTP_READ_TIMEOUT;
63     	}
64     
65     	public BaseImageDownloader(Context context, int connectTimeout, int readTimeout) {
66     		this.context = context.getApplicationContext();
67     		this.connectTimeout = connectTimeout;
68     		this.readTimeout = readTimeout;
69     	}
70     
71     	@Override
72     	public InputStream getStream(String imageUri, Object extra) throws IOException {
73     		switch (Scheme.ofUri(imageUri)) {
74     			case HTTP:
75     			case HTTPS:
76     				return getStreamFromNetwork(imageUri, extra);
77     			case FILE:
78     				return getStreamFromFile(imageUri, extra);
79     			case CONTENT:
80     				return getStreamFromContent(imageUri, extra);
81     			case ASSETS:
82     				return getStreamFromAssets(imageUri, extra);
83     			case DRAWABLE:
84     				return getStreamFromDrawable(imageUri, extra);
85     			case UNKNOWN:
86     			default:
87     				return getStreamFromOtherSource(imageUri, extra);
88     		}
89     	}
90     
91     	/**
92     	 * Retrieves {@link InputStream} of image by URI (image is located in the network).
93     	 *
94     	 * @param imageUri Image URI
95     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
96     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
97     	 * @return {@link InputStream} of image
98     	 * @throws IOException if some I/O error occurs during network request or if no InputStream could be created for
99     	 *                     URL.
100     	 */
101     	protected InputStream getStreamFromNetwork(String imageUri, Object extra) throws IOException {
102     		HttpURLConnection conn = createConnection(imageUri);
103     
104     		int redirectCount = 0;
105     		while (conn.getResponseCode() / 100 == 3 && redirectCount < MAX_REDIRECT_COUNT) {
106     			conn = createConnection(conn.getHeaderField("Location"));
107     			redirectCount++;
108     		}
109     
110     		return new BufferedInputStream(conn.getInputStream(), BUFFER_SIZE);
111     	}
112     
113     	/**
114     	 * Create {@linkplain HttpURLConnection HTTP connection} for incoming URL
115     	 *
116     	 * @param url URL to connect to
117     	 * @return {@linkplain HttpURLConnection Connection} for incoming URL. Connection isn't established so it still
118     	 *         configurable.
119     	 * @throws IOException if some I/O error occurs during network request or if no InputStream could be created for
120     	 *                     URL.
121     	 */
122     	protected HttpURLConnection createConnection(String url) throws IOException {
123     		String encodedUrl = Uri.encode(url, ALLOWED_URI_CHARS);
124     		HttpURLConnection conn = (HttpURLConnection) new URL(encodedUrl).openConnection();
125     		conn.setConnectTimeout(connectTimeout);
126     		conn.setReadTimeout(readTimeout);
127     		return conn;
128     	}
129     
130     	/**
131     	 * Retrieves {@link InputStream} of image by URI (image is located on the local file system or SD card).
132     	 *
133     	 * @param imageUri Image URI
134     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
135     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
136     	 * @return {@link InputStream} of image
137     	 * @throws IOException if some I/O error occurs reading from file system
138     	 */
139     	protected InputStream getStreamFromFile(String imageUri, Object extra) throws IOException {
140     		String filePath = Scheme.FILE.crop(imageUri);
141     		return new BufferedInputStream(new FileInputStream(filePath), BUFFER_SIZE);
142     	}
143     
144     	/**
145     	 * Retrieves {@link InputStream} of image by URI (image is accessed using {@link ContentResolver}).
146     	 *
147     	 * @param imageUri Image URI
148     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
149     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
150     	 * @return {@link InputStream} of image
151     	 * @throws FileNotFoundException if the provided URI could not be opened
152     	 */
153     	protected InputStream getStreamFromContent(String imageUri, Object extra) throws FileNotFoundException {
154     		ContentResolver res = context.getContentResolver();
155     		Uri uri = Uri.parse(imageUri);
156     		return res.openInputStream(uri);
157     	}
158     
159     	/**
160     	 * Retrieves {@link InputStream} of image by URI (image is located in assets of application).
161     	 *
162     	 * @param imageUri Image URI
163     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
164     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
165     	 * @return {@link InputStream} of image
166     	 * @throws IOException if some I/O error occurs file reading
167     	 */
168     	protected InputStream getStreamFromAssets(String imageUri, Object extra) throws IOException {
169     		String filePath = Scheme.ASSETS.crop(imageUri);
170     		return context.getAssets().open(filePath);
171     	}
172     
173     	/**
174     	 * Retrieves {@link InputStream} of image by URI (image is located in drawable resources of application).
175     	 *
176     	 * @param imageUri Image URI
177     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
178     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
179     	 * @return {@link InputStream} of image
180     	 */
181     	protected InputStream getStreamFromDrawable(String imageUri, Object extra) {
182     		String drawableIdString = Scheme.DRAWABLE.crop(imageUri);
183     		int drawableId = Integer.parseInt(drawableIdString);
184     		BitmapDrawable drawable = (BitmapDrawable) context.getResources().getDrawable(drawableId);
185     		Bitmap bitmap = drawable.getBitmap();
186     
187     		ByteArrayOutputStream os = new ByteArrayOutputStream();
188     		bitmap.compress(CompressFormat.PNG, 0, os);
189     		return new ByteArrayInputStream(os.toByteArray());
190     	}
191     
192     	/**
193     	 * Retrieves {@link InputStream} of image by URI from other source with unsupported scheme. Should be overriden by
194     	 * successors to implement image downloading from special sources.<br />
195     	 * This method is called only if image URI has unsupported scheme. Throws {@link UnsupportedOperationException} by
196     	 * default.
197     	 *
198     	 * @param imageUri Image URI
199     	 * @param extra    Auxiliary object which was passed to {@link DisplayImageOptions.Builder#extraForDownloader(Object)
200     	 *                 DisplayImageOptions.extraForDownloader(Object)}; can be null
201     	 * @return {@link InputStream} of image
202     	 * @throws IOException                   if some I/O error occurs
203     	 * @throws UnsupportedOperationException if image URI has unsupported scheme(protocol)
204     	 */
205     	protected InputStream getStreamFromOtherSource(String imageUri, Object extra) throws IOException {
206     		throw new UnsupportedOperationException(String.format(ERROR_UNSUPPORTED_SCHEME, imageUri));
207     	}
208     }
