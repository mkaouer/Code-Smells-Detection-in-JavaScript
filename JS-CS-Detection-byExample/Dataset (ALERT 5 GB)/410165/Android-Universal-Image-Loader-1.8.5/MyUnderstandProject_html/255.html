16     package com.nostra13.example.universalimageloader;
17     
18     import android.graphics.Bitmap;
19     import android.os.Bundle;
20     import android.os.Parcelable;
21     import android.support.v4.view.PagerAdapter;
22     import android.support.v4.view.ViewPager;
23     import android.view.LayoutInflater;
24     import android.view.View;
25     import android.view.ViewGroup;
26     import android.widget.ImageView;
27     import android.widget.ProgressBar;
28     import android.widget.Toast;
29     
30     import com.nostra13.example.universalimageloader.Constants.Extra;
31     import com.nostra13.universalimageloader.core.DisplayImageOptions;
32     import com.nostra13.universalimageloader.core.assist.FailReason;
33     import com.nostra13.universalimageloader.core.assist.ImageScaleType;
34     import com.nostra13.universalimageloader.core.assist.SimpleImageLoadingListener;
35     import com.nostra13.universalimageloader.core.display.FadeInBitmapDisplayer;
36     
37     /**
38      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
39      */
40     public class ImagePagerActivity extends BaseActivity {
41     
42     	private static final String STATE_POSITION = "STATE_POSITION";
43     
44     	DisplayImageOptions options;
45     
46     	ViewPager pager;
47     
48     	public void onCreate(Bundle savedInstanceState) {
49     		super.onCreate(savedInstanceState);
50     		setContentView(R.layout.ac_image_pager);
51     
52     		Bundle bundle = getIntent().getExtras();
53     		String[] imageUrls = bundle.getStringArray(Extra.IMAGES);
54     		int pagerPosition = bundle.getInt(Extra.IMAGE_POSITION, 0);
55     
56     		if (savedInstanceState != null) {
57     			pagerPosition = savedInstanceState.getInt(STATE_POSITION);
58     		}
59     
60     		options = new DisplayImageOptions.Builder()
61     			.showImageForEmptyUri(R.drawable.ic_empty)
62     			.showImageOnFail(R.drawable.ic_error)
63     			.resetViewBeforeLoading(true)
64     			.cacheOnDisc(true)
65     			.imageScaleType(ImageScaleType.EXACTLY)
66     			.bitmapConfig(Bitmap.Config.RGB_565)
67     			.displayer(new FadeInBitmapDisplayer(300))
68     			.build();
69     
70     		pager = (ViewPager) findViewById(R.id.pager);
71     		pager.setAdapter(new ImagePagerAdapter(imageUrls));
72     		pager.setCurrentItem(pagerPosition);
73     	}
74     
75     	@Override
76     	public void onSaveInstanceState(Bundle outState) {
77     		outState.putInt(STATE_POSITION, pager.getCurrentItem());
78     	}
79     
80     	private class ImagePagerAdapter extends PagerAdapter {
81     
82     		private String[] images;
83     		private LayoutInflater inflater;
84     
85     		ImagePagerAdapter(String[] images) {
86     			this.images = images;
87     			inflater = getLayoutInflater();
88     		}
89     
90     		@Override
91     		public void destroyItem(ViewGroup container, int position, Object object) {
92     			((ViewPager) container).removeView((View) object);
93     		}
94     
95     		@Override
96     		public void finishUpdate(View container) {
97     		}
98     
99     		@Override
100     		public int getCount() {
101     			return images.length;
102     		}
103     
104     		@Override
105     		public Object instantiateItem(ViewGroup view, int position) {
106     			View imageLayout = inflater.inflate(R.layout.item_pager_image, view, false);
107     			ImageView imageView = (ImageView) imageLayout.findViewById(R.id.image);
108     			final ProgressBar spinner = (ProgressBar) imageLayout.findViewById(R.id.loading);
109     
110     			imageLoader.displayImage(images[position], imageView, options, new SimpleImageLoadingListener() {
111     				@Override
112     				public void onLoadingStarted(String imageUri, View view) {
113     					spinner.setVisibility(View.VISIBLE);
114     				}
115     
116     				@Override
117     				public void onLoadingFailed(String imageUri, View view, FailReason failReason) {
118     					String message = null;
119     					switch (failReason.getType()) {
120     						case IO_ERROR:
121     							message = "Input/Output error";
122     							break;
123     						case DECODING_ERROR:
124     							message = "Image can't be decoded";
125     							break;
126     						case NETWORK_DENIED:
127     							message = "Downloads are denied";
128     							break;
129     						case OUT_OF_MEMORY:
130     							message = "Out Of Memory error";
131     							break;
132     						case UNKNOWN:
133     							message = "Unknown error";
134     							break;
135     					}
136     					Toast.makeText(ImagePagerActivity.this, message, Toast.LENGTH_SHORT).show();
137     
138     					spinner.setVisibility(View.GONE);
139     				}
140     
141     				@Override
142     				public void onLoadingComplete(String imageUri, View view, Bitmap loadedImage) {
143     					spinner.setVisibility(View.GONE);
144     				}
145     			});
146     
147     			((ViewPager) view).addView(imageLayout, 0);
148     			return imageLayout;
149     		}
150     
151     		@Override
152     		public boolean isViewFromObject(View view, Object object) {
153     			return view.equals(object);
154     		}
155     
156     		@Override
157     		public void restoreState(Parcelable state, ClassLoader loader) {
158     		}
159     
160     		@Override
161     		public Parcelable saveState() {
162     			return null;
163     		}
164     
165     		@Override
166     		public void startUpdate(View container) {
167     		}
168     	}
169     }
