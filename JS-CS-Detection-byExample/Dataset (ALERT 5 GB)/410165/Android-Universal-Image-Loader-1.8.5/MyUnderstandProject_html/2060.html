16     package com.nostra13.example.universalimageloader;
17     
18     import android.content.Intent;
19     import android.graphics.Bitmap;
20     import android.os.Bundle;
21     import android.view.View;
22     import android.view.ViewGroup;
23     import android.widget.AdapterView;
24     import android.widget.AdapterView.OnItemClickListener;
25     import android.widget.BaseAdapter;
26     import android.widget.Gallery;
27     import android.widget.ImageView;
28     
29     import com.nostra13.example.universalimageloader.Constants.Extra;
30     import com.nostra13.universalimageloader.core.DisplayImageOptions;
31     
32     /**
33      * @author Sergey Tarasevich (nostra13[at]gmail[dot]com)
34      */
35     @SuppressWarnings("deprecation")
36     public class ImageGalleryActivity extends BaseActivity {
37     
38     	String[] imageUrls;
39     
40     	DisplayImageOptions options;
41     
42     	public void onCreate(Bundle savedInstanceState) {
43     		super.onCreate(savedInstanceState);
44     		setContentView(R.layout.ac_image_gallery);
45     
46     		Bundle bundle = getIntent().getExtras();
47     		imageUrls = bundle.getStringArray(Extra.IMAGES);
48     
49     		options = new DisplayImageOptions.Builder()
50     			.showStubImage(R.drawable.ic_stub)
51     			.showImageForEmptyUri(R.drawable.ic_empty)
52     			.showImageOnFail(R.drawable.ic_error)
53     			.cacheInMemory(true)
54     			.cacheOnDisc(true)
55     			.bitmapConfig(Bitmap.Config.RGB_565)
56     			.build();
57     
58     		Gallery gallery = (Gallery) findViewById(R.id.gallery);
59     		gallery.setAdapter(new ImageGalleryAdapter());
60     		gallery.setOnItemClickListener(new OnItemClickListener() {
61     			@Override
62     			public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
63     				startImagePagerActivity(position);
64     			}
65     		});
66     	}
67     
68     	private void startImagePagerActivity(int position) {
69     		Intent intent = new Intent(this, ImagePagerActivity.class);
70     		intent.putExtra(Extra.IMAGES, imageUrls);
71     		intent.putExtra(Extra.IMAGE_POSITION, position);
72     		startActivity(intent);
73     	}
74     
75     	private class ImageGalleryAdapter extends BaseAdapter {
76     		@Override
77     		public int getCount() {
78     			return imageUrls.length;
79     		}
80     
81     		@Override
82     		public Object getItem(int position) {
83     			return position;
84     		}
85     
86     		@Override
87     		public long getItemId(int position) {
88     			return position;
89     		}
90     
91     		@Override
92     		public View getView(int position, View convertView, ViewGroup parent) {
93     			ImageView imageView = (ImageView) convertView;
94     			if (imageView == null) {
95     				imageView = (ImageView) getLayoutInflater().inflate(R.layout.item_gallery_image, parent, false);
96     			}
97     			imageLoader.displayImage(imageUrls[position], imageView, options);
98     			return imageView;
99     		}
100     	}
101     }
