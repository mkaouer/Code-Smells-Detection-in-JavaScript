13     namespace thrust_shell {
14     
15     class ThrustWindow;
16     class MenuController;
17     class ThrustMenuBinding;
18     
19     class ThrustMenu : public ui::SimpleMenuModel::Delegate {
20     public:
21       /****************************************************************************/
22       /* PUBLIC API */
23       /****************************************************************************/
24       // ### ThrustMenu
25       ThrustMenu(ThrustMenuBinding* binding);
26       // ### ~ThrustMenu
27       virtual ~ThrustMenu();
28     
29       void AddItem(int command_id, 
30                    const base::string16& label);
31       void AddSeparator();
32       void AddCheckItem(int command_id,
33                         const base::string16& label);
34       void AddRadioItem(int command_id,
35                         const base::string16& label,
36                         int group_id);
37       void AddSubMenu(int command_id,
38                       const base::string16& label,
39                       ThrustMenu* menu);
40     
41       void Clear();
42     
43       void SetChecked(int command_id, bool checked);
44       void SetEnabled(int command_id, bool enabled);
45       void SetVisible(int command_id, bool visible);
46       void SetAccelerator(int command_id, std::string accelerator);
47     
48       int GetIndexOfCommandId(int command_id);
49       int GetItemCount() const;
50       int GetCommandIdAt(int index) const;
51       base::string16 GetLabelAt(int index) const;
52     
53       bool IsItemCheckedAt(int index) const;
54       bool IsEnabledAt(int index) const;
55       bool IsVisibleAt(int index) const;
56     
57       virtual void AttachToWindow(ThrustWindow* window);
58       virtual void Popup(ThrustWindow* window) {
59         return PlatformPopup(window);
60       }
61     
62     #if defined(OS_MACOSX)
63       // Set the global menubar.
64       static void SetApplicationMenu(ThrustMenu* menu);
65       // Fake sending an action from the application menu.
66       static void SendActionToFirstResponder(const std::string& action);
67     #endif
68     
69       ui::SimpleMenuModel* model() const { return model_.get(); }
70     
71      protected:
72       /****************************************************************************/
73       /* SIMPLEMENUMODEL DELEGATE IMPLEMENTATION */
74       /****************************************************************************/
75       virtual bool IsCommandIdChecked(int command_id) const OVERRIDE;
76       virtual bool IsCommandIdEnabled(int command_id) const OVERRIDE;
77       virtual bool IsCommandIdVisible(int command_id) const OVERRIDE;
78     
79       virtual bool GetAcceleratorForCommandId(
80           int command_id,
81           ui::Accelerator* accelerator) OVERRIDE;
82     
83       virtual void ExecuteCommand(int command_id, int event_flags) OVERRIDE;
84       virtual void MenuWillShow(ui::SimpleMenuModel* source) OVERRIDE;
85       virtual void MenuClosed(ui::SimpleMenuModel* source) OVERRIDE;
86     
87      private:
88       /****************************************************************************/
89       /* PLATFORM INTERFACE */
90       /****************************************************************************/
91       void PlatformPopup(ThrustWindow* window);
92     
93       ThrustMenuBinding*                        binding_;
94     
95       scoped_ptr<ui::SimpleMenuModel>           model_;
96       ThrustMenu*                               parent_;
97     
98       std::map<int, bool>                       checked_;
99       std::map<int, bool>                       enabled_;
100       std::map<int, bool>                       visible_;
101       std::map<int, std::string>                accelerator_;
102     
103       DISALLOW_COPY_AND_ASSIGN(ThrustMenu);
104     };
105     
106     } // namespace thrust_shell
