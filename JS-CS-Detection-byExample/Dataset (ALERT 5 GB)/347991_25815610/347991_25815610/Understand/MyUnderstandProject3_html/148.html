25     Context::Context(v8::Handle<v8::Context> v8_context,
26                      blink::WebFrame* web_frame)
27         : v8_context_(v8_context),
28           web_frame_(web_frame),
29           safe_builtins_(this),
30           isolate_(v8_context->GetIsolate()) {
31       VLOG(1) << "Created context:\n"
32               << "  frame:        " << web_frame_;
33     }
34     
35     Context::~Context() {
36       VLOG(1) << "Destroyed context:\n"
37               << "  frame:        " << web_frame_;
38       Invalidate();
39     }
40     
41     void Context::Invalidate() {
42       if (!is_valid())
43         return;
44       if (module_system_)
45         module_system_->Invalidate();
46       web_frame_ = NULL;
47       v8_context_.reset();
48     }
49     
50     content::RenderView* Context::GetRenderView() const {
51       if (web_frame_ && web_frame_->view())
52         return content::RenderView::FromWebView(web_frame_->view());
53       else
54         return NULL;
55     }

57     GURL Context::GetURL() const {
58       /* TODO(spolu): Check OK                      */
59       /*              see: user_script_slave.cc:216 */
60       return web_frame_ ?

64     v8::Local<v8::Value> Context::CallFunction(
65         v8::Handle<v8::Function> function,
66         int argc,
67         v8::Handle<v8::Value> argv[]) const {
68       v8::EscapableHandleScope handle_scope(isolate());
69       v8::Context::Scope scope(v8_context());
70     
71       blink::WebScopedMicrotaskSuppression suppression;
72       if (!is_valid()) {
73         return handle_scope.Escape(
74             v8::Local<v8::Primitive>(v8::Undefined(isolate())));
75       }
76     
77       v8::Handle<v8::Object> global = v8_context()->Global();
78       if (!web_frame_)
79         return handle_scope.Escape(function->Call(global, argc, argv));
80       return handle_scope.Escape(
81           v8::Local<v8::Value>(web_frame_->callFunctionEvenIfScriptDisabled(
82               function, global, argc, argv)));
83     }
