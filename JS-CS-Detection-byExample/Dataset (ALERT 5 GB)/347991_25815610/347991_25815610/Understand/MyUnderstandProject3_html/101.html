61     // ### ThrustWindow
62     //
63     // This represents an ThrustShell window. The window opens on a `root_url` 
64     // provided at creation. The window exposes only one webcontents with support 
65     // for the <webview> tag.
66     //
67     // The ThrustWindow lives on the BrowserThread::UI thread
68     class ThrustWindow : public brightray::DefaultWebContentsDelegate,
69                          public brightray::InspectableWebContentsDelegate,
70                          public content::WebContentsObserver,
71     #if defined(USE_AURA)
72                      public views::WidgetDelegateView,
73                      public views::WidgetObserver,
74     #elif defined(OS_MACOSX)
75     #endif
76                      public content::NotificationObserver {
77     public:
78     
79       /****************************************************************************/
80       /* STATIC INTERFACE */
81       /****************************************************************************/
82       // ### CreateNew
83       //
84       // Creates a new ThrustWindow with the specified `root_url`
85       // ```
86       // @binding   {ThrustWindowBinding} the binding associated
87       // @session   {ThrustSession} the session to use
88       // @root_url  {GURL} the main document root url
89       // @size      {Size} the initial size of the window
90       // @title     {string} the title to use
91       // @icon_path {string} icon_path (no effect on OSX)
92       // @has_frame {boolean} has a frame
93       // ```
94       static ThrustWindow* CreateNew(
95           ThrustWindowBinding* binding,
96           ThrustSession* session,
97           const GURL& root_url,
98           const gfx::Size& size,
99           const std::string& title,
100           const std::string& icon_path,
101           const bool has_frame);
102     
103       // ### CreateNew
104       //
105       // Creates a new ThrustWindow out of an existing WebContents
106       // ```
107       // @binding      {ThrustWindowBinding} the binding associated
108       // @web_contents {WebContents} the web_contents to use
109       // @size         {Size} the initial size of the window
110       // @title        {string} the title to use
111       // @icon_path    {string} icon_path (no effect on OSX)
112       // @has_frame    {boolean} has a frame
113       // ```
114       static ThrustWindow* CreateNew(
115           ThrustWindowBinding* binding,
116           content::WebContents* web_contents,
117           const gfx::Size& size,
118           const std::string& title,
119           const std::string& icon_path,
120           const bool has_frame);
