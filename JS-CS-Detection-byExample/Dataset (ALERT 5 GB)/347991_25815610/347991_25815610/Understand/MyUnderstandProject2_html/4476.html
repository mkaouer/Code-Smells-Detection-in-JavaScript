45     // This observer ensures that the WebViewGuest destroys itself when its
46     // embedder goes away.
47     class WebViewGuest::EmbedderWebContentsObserver : public WebContentsObserver {
48      public:
49       explicit EmbedderWebContentsObserver(WebViewGuest* guest)
50           : WebContentsObserver(guest->embedder_web_contents()),
51             guest_(guest) {
52       }
53     
54       virtual ~EmbedderWebContentsObserver() {
55       }
56     
57       // WebContentsObserver implementation.
58       virtual void WebContentsDestroyed() OVERRIDE {
59         guest_->embedder_web_contents_ = NULL;
60         //guest_->EmbedderDestroyed();
61         guest_->Destroy();
62       }
63     
64      private:
65       WebViewGuest* guest_;
66     
67       DISALLOW_COPY_AND_ASSIGN(EmbedderWebContentsObserver);
68     };

89     WebViewGuest::WebViewGuest(
90         int guest_instance_id,
91         WebContents* guest_web_contents)
92     : WebContentsObserver(guest_web_contents),
93       guest_web_contents_(guest_web_contents),
94       embedder_web_contents_(NULL),
95       embedder_render_process_id_(0),
96       browser_context_(guest_web_contents->GetBrowserContext()),
97       guest_instance_id_(guest_instance_id),
98       view_instance_id_(webview::kInstanceIDNone),
99       weak_ptr_factory_(this) 
100     {
101       WebContentsObserver::Observe(guest_web_contents);
102       guest_web_contents->SetDelegate(this);
103       webcontents_webview_map.Get().insert(
104           std::make_pair(guest_web_contents, this));
105     
106       notification_registrar_.Add(
107           this, content::NOTIFICATION_LOAD_COMPLETED_MAIN_FRAME,
108           content::Source<WebContents>(guest_web_contents));
109     
110       notification_registrar_.Add(
111           this, content::NOTIFICATION_RESOURCE_RECEIVED_REDIRECT,
112           content::Source<WebContents>(guest_web_contents));
113     
114       LOG(INFO) << "WebViewGuest Constructor: " << this;
115       ThrustShellBrowserClient::Get()->ThrustSessionForBrowserContext(browser_context_)->
116         AddGuest(guest_instance_id_, guest_web_contents);
117     }

129     WebViewGuest* 
130     WebViewGuest::FromWebContents(
131         WebContents* web_contents) 
132     {
133       WebContentsWebViewGuestMap* webview_map = webcontents_webview_map.Pointer();
134       WebContentsWebViewGuestMap::iterator it = webview_map->find(web_contents);
135       return it == webview_map->end() ? NULL : it->second;
136     }

140     WebViewGuest::From(
141         int embedder_process_id, 
142         int guest_instance_id) 
143     {
144       EmbedderWebViewGuestMap* guest_map = embedder_webview_map.Pointer();
145       EmbedderWebViewGuestMap::iterator it = guest_map->find(
146           std::make_pair(embedder_process_id, guest_instance_id));
147       return it == guest_map->end() ? NULL : it->second;

162     // static
163     void 
164     WebViewGuest::ParsePartitionParam(
165         const base::DictionaryValue* extra_params,
166         std::string* storage_partition_id,
167         bool* persist_storage) 
168     {
169       std::string partition_str;
170       if (!extra_params->GetString(webview::kStoragePartitionId, &partition_str)) {
171         return;
172       }
173     
174       // Since the "persist:" prefix is in ASCII, StartsWith will work fine on
175       // UTF-8 encoded |partition_id|. If the prefix is a match, we can safely
176       // remove the prefix without splicing in the middle of a multi-byte codepoint.
177       // We can use the rest of the string as UTF-8 encoded one.
178       if (StartsWithASCII(partition_str, "persist:", true)) {
179         size_t index = partition_str.find(":");
180         CHECK(index != std::string::npos);
181         // It is safe to do index + 1, since we tested for the full prefix above.
182         *storage_partition_id = partition_str.substr(index + 1);
183     
184         if (storage_partition_id->empty()) {
185           // TODO(lazyboy): Better way to deal with this error.
186           return;
187         }
188         *persist_storage = true;
189       } else {
190         *storage_partition_id = partition_str;
191         *persist_storage = false;
192       }
193     }

219     void 
220     WebViewGuest::WillAttach(
221         content::WebContents* embedder_web_contents,
222         const base::DictionaryValue& extra_params) 
223     {
224       LOG(INFO) << "WILL ATTACH ***************";
225       LOG(INFO) << "WebViewGuest WillAttach: " << embedder_web_contents;
226       embedder_web_contents_ = embedder_web_contents;
227       embedder_web_contents_observer_.reset(
228           new EmbedderWebContentsObserver(this));
229       embedder_render_process_id_ =
230           embedder_web_contents->GetRenderProcessHost()->GetID();
231       extra_params.GetInteger(webview::kParameterInstanceId, &view_instance_id_);
232       extra_params_.reset(extra_params.DeepCopy());
233     
234       std::pair<int, int> key(embedder_render_process_id_, guest_instance_id_);
235       embedder_webview_map.Get().insert(std::make_pair(key, this));
236     }

239     WebViewGuest::~WebViewGuest() 
240     {
241       LOG(INFO) << "WebViewGuest Destructor: " << this;
242     
243       std::pair<int, int> key(embedder_render_process_id_, guest_instance_id_);
244       embedder_webview_map.Get().erase(key);
245       webcontents_webview_map.Get().erase(guest_web_contents());
246     
247       pending_events_.clear();
248     
249       ThrustShellBrowserClient::Get()->ThrustSessionForBrowserContext(browser_context_)->
250         RemoveGuest(guest_instance_id_);
251     }

258       switch (type) {
259         case content::NOTIFICATION_LOAD_COMPLETED_MAIN_FRAME: {
260           DCHECK_EQ(content::Source<WebContents>(source).ptr(),
261                     guest_web_contents());
262           if(content::Source<WebContents>(source).ptr() == guest_web_contents()) {
263             //LoadHandlerCalled();
264           }
265           break;
266         }
267         case content::NOTIFICATION_RESOURCE_RECEIVED_REDIRECT: {
268           DCHECK_EQ(content::Source<WebContents>(source).ptr(),
269                     guest_web_contents());
270           /*
271           content::ResourceRedirectDetails* resource_redirect_details =
272               content::Details<content::ResourceRedirectDetails>(details).ptr();
273           bool is_top_level =
274               resource_redirect_details->resource_type == ResourceType::MAIN_FRAME;
275           LoadRedirect(resource_redirect_details->url,
276                        resource_redirect_details->new_url,
277                        is_top_level);
278           */
279           break;
280         }
281         default:
282           NOTREACHED() << "Unexpected notification sent.";
283           break;
284       }

362     void 
363     WebViewGuest::SendQueuedEvents() 
364     {
365       if (!attached())
366         return;
367     
368       while (!pending_events_.empty()) {
369         linked_ptr<Event> event_ptr = pending_events_.front();
370         pending_events_.pop_front();
371         DispatchEvent(event_ptr.release());
372       }
373     }
