24     // Writes |message| to stack to show up in minidump, then crashes.
25     void CheckWithMinidump(const std::string& message) {
26       char minidump[1024];
27       base::debug::Alias(&minidump);
28       base::snprintf(minidump, arraysize(minidump),
29                      "e::console: %s", message.c_str());
30       CHECK(false) << message;
31     }

33     typedef void (*LogMethod)(v8::Handle<v8::Context> context,
34                               const std::string& message);
35     
36     void BoundLogMethodCallback(const v8::FunctionCallbackInfo<v8::Value>& info) {
37       LogMethod log_method = reinterpret_cast<LogMethod>(
38           info.Data().As<v8::External>()->Value());
39       std::string message;
40       for (int i = 0; i < info.Length(); ++i) {
41         if (i > 0)
42           message += " ";
43         message += *v8::String::Utf8Value(info[i]);
44       }
45       (*log_method)(info.GetIsolate()->GetCallingContext(), message);
46     }
47     
48     void BindLogMethod(v8::Isolate* isolate,
49                        v8::Local<v8::Object> target,
50                        const std::string& name,
51                        LogMethod log_method) {
52       v8::Local<v8::FunctionTemplate> tmpl = v8::FunctionTemplate::New(
53           isolate,
54           &BoundLogMethodCallback,
55           v8::External::New(isolate, reinterpret_cast<void*>(log_method)));
56       target->Set(v8::String::NewFromUtf8(isolate, name.c_str()),
57                   tmpl->GetFunction());
58     }

83     void AddMessage(content::RenderView* render_view,
84                     content::ConsoleMessageLevel level,
85                     const std::string& message) {
86       blink::WebView* web_view = render_view->GetWebView();
87       if (!web_view || !web_view->mainFrame())
88         return;
89       blink::WebConsoleMessage::Level target_level =
90           blink::WebConsoleMessage::LevelLog;
91       switch (level) {
92         case content::CONSOLE_MESSAGE_LEVEL_DEBUG:
93           target_level = blink::WebConsoleMessage::LevelDebug;
94           break;
95         case content::CONSOLE_MESSAGE_LEVEL_LOG:
96           target_level = blink::WebConsoleMessage::LevelLog;
97           break;
98         case content::CONSOLE_MESSAGE_LEVEL_WARNING:
99           target_level = blink::WebConsoleMessage::LevelWarning;
100           break;
101         case content::CONSOLE_MESSAGE_LEVEL_ERROR:
102           target_level = blink::WebConsoleMessage::LevelError;
103           break;
104       }
105       web_view->mainFrame()->addMessageToConsole(
106           blink::WebConsoleMessage(target_level, base::UTF8ToUTF16(message)));
107     }

130     void AddMessage(v8::Handle<v8::Context> context,
131                     content::ConsoleMessageLevel level,
132                     const std::string& message) {
133       if (context.IsEmpty()) {
134         LOG(WARNING) << "Could not log \"" << message << "\": no context given";
135         return;
136       }
137       /* TODO(spolu): Mechanism to retrieve RenderView by V8 Context */
138       content::RenderView* render_view = NULL; //ByContextFinder::Find(context);
139       if (!render_view) {
140         LOG(WARNING) << "Could not log \"" << message << "\": no render view found";
141         return;
142       }
143       AddMessage(render_view, level, message);
144     }

146     v8::Local<v8::Object> AsV8Object() {
147       v8::Isolate* isolate = v8::Isolate::GetCurrent();
148       v8::EscapableHandleScope handle_scope(isolate);
149       v8::Local<v8::Object> console_object = v8::Object::New(isolate);
150       BindLogMethod(isolate, console_object, "debug", &Debug);
151       BindLogMethod(isolate, console_object, "log", &Log);
152       BindLogMethod(isolate, console_object, "warn", &Warn);
153       BindLogMethod(isolate, console_object, "error", &Error);
154       return handle_scope.Escape(console_object);
155     }
