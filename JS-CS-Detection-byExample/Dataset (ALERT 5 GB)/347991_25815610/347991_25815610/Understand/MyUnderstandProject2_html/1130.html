28     void GlobalMenuBarRegistrarX11::OnWindowMapped(unsigned long xid) {
29       live_xids_.insert(xid);
30     
31       if (registrar_proxy_)
32         RegisterXID(xid);
33     }

35     void GlobalMenuBarRegistrarX11::OnWindowUnmapped(unsigned long xid) {
36       if (registrar_proxy_)
37         UnregisterXID(xid);
38     
39       live_xids_.erase(xid);
40     }

42     GlobalMenuBarRegistrarX11::GlobalMenuBarRegistrarX11()
43         : registrar_proxy_(NULL) {
44       // libdbusmenu uses the gio version of dbus; I tried using the code in dbus/,
45       // but it looks like that's isn't sharing the bus name with the gio version,
46       // even when |connection_type| is set to SHARED.
47       g_dbus_proxy_new_for_bus(
48           G_BUS_TYPE_SESSION,
49           static_cast<GDBusProxyFlags>(
50               G_DBUS_PROXY_FLAGS_DO_NOT_LOAD_PROPERTIES |
51               G_DBUS_PROXY_FLAGS_DO_NOT_CONNECT_SIGNALS |
52               G_DBUS_PROXY_FLAGS_DO_NOT_AUTO_START),
53           NULL,
54           kAppMenuRegistrarName,
55           kAppMenuRegistrarPath,
56           kAppMenuRegistrarName,
57           NULL,  // TODO: Probalby want a real cancelable.
58           static_cast<GAsyncReadyCallback>(OnProxyCreatedThunk),
59           this);
60     }

62     GlobalMenuBarRegistrarX11::~GlobalMenuBarRegistrarX11() {
63       if (registrar_proxy_) {
64         g_signal_handlers_disconnect_by_func(
65             registrar_proxy_,
66             reinterpret_cast<void*>(OnNameOwnerChangedThunk),
67             this);
68         g_object_unref(registrar_proxy_);
69       }
70     }

72     void GlobalMenuBarRegistrarX11::RegisterXID(unsigned long xid) {
73       DCHECK(registrar_proxy_);
74       std::string path = thrust_shell::GlobalMenuBarX11::GetPathForWindow(xid);
75     
76       ANNOTATE_SCOPED_MEMORY_LEAK; // http://crbug.com/314087
77       // TODO(erg): The mozilla implementation goes to a lot of callback trouble
78       // just to make sure that they react to make sure there's some sort of
79       // cancelable object; including making a whole callback just to handle the
80       // cancelable.
81       //
82       // I don't see any reason why we should care if "RegisterWindow" completes or
83       // not.
84       g_dbus_proxy_call(registrar_proxy_,
85                         "RegisterWindow",
86                         g_variant_new("(uo)", xid, path.c_str()),
87                         G_DBUS_CALL_FLAGS_NONE, -1,
88                         NULL,
89                         NULL,
90                         NULL);
91     }

94       DCHECK(registrar_proxy_);
95       std::string path = thrust_shell::GlobalMenuBarX11::GetPathForWindow(xid);
96     
97       ANNOTATE_SCOPED_MEMORY_LEAK; // http://crbug.com/314087
98       // TODO(erg): The mozilla implementation goes to a lot of callback trouble
99       // just to make sure that they react to make sure there's some sort of

114     void GlobalMenuBarRegistrarX11::OnProxyCreated(GObject* source,
115                                                    GAsyncResult* result) {
116       GError* error = NULL;
117       GDBusProxy* proxy = g_dbus_proxy_new_for_bus_finish(result, &error);
118       if (error) {
119         g_error_free(error);
120         return;
121       }
122     
123       // TODO(erg): Mozilla's implementation has a workaround for GDBus
124       // cancellation here. However, it's marked as fixed. If there's weird
125       // problems with cancelation, look at how they fixed their issues.
126     
127       registrar_proxy_ = proxy;
128     
129       g_signal_connect(registrar_proxy_, "notify::g-name-owner",
130                        G_CALLBACK(OnNameOwnerChangedThunk), this);
131     
132       OnNameOwnerChanged(NULL, NULL);
133     }
134     
135     void GlobalMenuBarRegistrarX11::OnNameOwnerChanged(GObject* /* ignored */,
136                                                        GParamSpec* /* ignored */) {
137       // If the name owner changed, we need to reregister all the live xids with
138       // the system.
139       for (std::set<unsigned long>::const_iterator it = live_xids_.begin();
140            it != live_xids_.end(); ++it) {
141         RegisterXID(*it);
142       }
143     }
