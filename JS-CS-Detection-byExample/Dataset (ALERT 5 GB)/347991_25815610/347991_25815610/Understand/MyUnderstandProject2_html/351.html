30     // ### ThrustSession
31     //
32     // The ThrustSession is BrowserContext passed to an ThrustShell to be used with all
33     // its associated control and frames. A session represents the contextual 
34     // parameters and data needed to render a web page:
35     // - Whether or not Local HTML5 Storage is allowed or in-memory
36     // - An interface to retrieve cookies
37     // - An interface to retrieve visited links
38     //
39     // These interfaces are exposed to Javascript. So that user profiles can be
40     // handled from there. HTML5 Storage is not exposed to Javascript for the sake
41     // of simplicity, but we do allow to force it into memory (use of public
42     // computer)
43     //
44     // TODO(spolu): This discrepancy between cookies and local storage may lead to 
45     // issues as some website may expect to find local storage given the presence of 
46     // a cookie. Exposing HTML5 Storage to JS is on a future "distant?" roadmap...
47     // see http://www.w3.org/TR/webstorage/
48     // - SessionStorage not impacted
49     // - LocalStorage can be expired / deleted by user so semantics are not too off
50     class ThrustSession : public brightray::BrowserContext,
51                           public content::BrowserPluginGuestManager {
52     public:
53       /****************************************************************************/
54       /* PUBLIC INTERFACE */
55       /****************************************************************************/
56       // ### ThrustSession
57       ThrustSession(ThrustSessionBinding* binding,
58                     const bool off_the_record,
59                     const std::string& path,
60                     bool dummy_cookie_store = false);
61       // ### ~ThrustSession
62       virtual ~ThrustSession();
63     
64       /****************************************************************************/
65       /* EXOFRAME / DEVTOOLS I/F */
66       /****************************************************************************/
67       ThrustShellDevToolsDelegate* devtools_delegate() {
68         return devtools_delegate_;
69       }
70     
71       // ### GetDevToolsURL
72       //
73       // Returns the DevTools URL for this session
74       GURL GetDevToolsURL();
75     
76       ThrustSessionCookieStore* GetCookieStore();
77       ThrustSessionVisitedLinkStore* GetVisitedLinkStore();
78     
79       /****************************************************************************/
80       /* REQUEST CONTEXT GETTER HELPERS */
81       /****************************************************************************/
82       net::URLRequestContextGetter* CreateRequestContext(
83           content::ProtocolHandlerMap* protocol_handlers,
84           content::URLRequestInterceptorScopedVector request_interceptors);
85       net::URLRequestContextGetter* CreateRequestContextForStoragePartition(
86           const base::FilePath& partition_path,
87           bool in_memory,
88           content::ProtocolHandlerMap* protocol_handlers,
89           content::URLRequestInterceptorScopedVector request_interceptors);
90     
91       /****************************************************************************/
92       /* BROWSER CONTEXT IMPLEMENTATION */
93       /****************************************************************************/
94       virtual base::FilePath GetPath() const OVERRIDE;
95       virtual bool IsOffTheRecord() const OVERRIDE;
96     
97       virtual content::DownloadManagerDelegate* 
98         GetDownloadManagerDelegate() OVERRIDE;
99     
100       virtual content::ResourceContext* GetResourceContext() OVERRIDE;
101       virtual content::BrowserPluginGuestManager* GetGuestManager() OVERRIDE;
102     
103       /****************************************************************************/
104       /* BROWSER_PLUGIN_GUEST_MANAGER_IMPLEMENTATION */
105       /****************************************************************************/
106       virtual content::WebContents* CreateGuest(
107           content::SiteInstance* embedder_site_instance,
108           int instance_id,
109           scoped_ptr<base::DictionaryValue> extra_params) OVERRIDE;
110       virtual int GetNextInstanceID() OVERRIDE;
111       virtual void MaybeGetGuestByInstanceIDOrKill(
112           int guest_instance_id,
113           int embedder_render_process_id,
114           const GuestByInstanceIDCallback& callback) OVERRIDE;
115       virtual bool ForEachGuest(content::WebContents* embedder_web_contents,
116                                 const GuestCallback& callback) OVERRIDE;
117     
118       content::WebContents* GetGuestByInstanceID(int guest_instance_id,
119                                                  int embedder_render_process_id);
120     
121     private:
122       class ExoResourceContext;
123     
124       /****************************************************************************/
125       /* PRIVATE INTERFACE                                                        */
126       /****************************************************************************/
127       virtual void AddGuest(int guest_instance_id,
128                             content::WebContents* guest_web_contents);
129     
130       void RemoveGuest(int guest_instance_id);
131     
132       /****************************************************************************/
133       /* MEMBERS                                                                   */
134       /****************************************************************************/
135       ThrustSessionBinding*                               binding_;
136     
137       bool                                                off_the_record_;
138       bool                                                ignore_certificate_errors_;
139       base::FilePath                                      path_;
140     
141       scoped_ptr<ExoResourceContext>                      resource_context_;

145       scoped_refptr<ThrustSessionVisitedLinkStore>        visitedlink_store_;
146     
147       ThrustShellDevToolsDelegate*                        devtools_delegate_;
148     
149       std::map<int, content::WebContents*>                guest_web_contents_;
150       int                                                 current_instance_id_;
151     
152       friend class ThrustSessionCookieStore;
153       friend class ThrustShellDevToolsDelegate;
154       friend class WebViewGuest;
155       friend class GuestWebContentsObserver;
156     
157       DISALLOW_COPY_AND_ASSIGN(ThrustSession);
158     };
