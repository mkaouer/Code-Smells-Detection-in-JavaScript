29     ThrustWindowBinding::ThrustWindowBinding(
30         const unsigned int id, 
31         scoped_ptr<base::DictionaryValue> args)
32       : APIBinding("window", id)
33     {
34       LOG(INFO) << "ThrustWindowBinding Constructor [" << this << "] " << id_;
35     
36       std::string root_url = "http://google.com";
37       args->GetString("root_url", &root_url);
38     
39       int width = 800;
40       int height = 600;
41     
42       args->GetInteger("size.width", &width);
43       args->GetInteger("size.height", &height);
44     
45       std::string title = "ThrustShell";
46       args->GetString("title", &title);
47     
48       std::string icon_path = "";
49       args->GetString("icon_path", &icon_path);
50     
51       ThrustSession* session = NULL;
52     
53       int session_id = -1;
54       args->GetInteger("session_id", &session_id);
55     
56       ThrustSessionBinding* sb = 
57         (ThrustSessionBinding*)(API::Get()->GetBinding(session_id));
58       if(sb != NULL) {
59         session = sb->GetSession();
60       }
61       else {
62         session = ThrustShellBrowserClient::Get()->system_session();
63       }
64     
65       window_.reset(ThrustWindow::CreateNew(
66             this,
67             session,
68             GURL(root_url), 
69             gfx::Size(width, height), 
70             title, 
71             icon_path, 
72             true));
73     }

75     ThrustWindowBinding::~ThrustWindowBinding()
76     {
77       LOG(INFO) << "ThrustWindowBinding Destructor [" << this << "] " << id_;
78       window_.reset();
79     }

82     void
83     ThrustWindowBinding::CallLocalMethod(
84         const std::string& method,
85         scoped_ptr<base::DictionaryValue> args,
86         const API::MethodCallback& callback)
87     {
88       std::string err = std::string("");
89       base::DictionaryValue* res = new base::DictionaryValue;
90     
91       LOG(INFO) << "ThrustWindow call [" << method << "]";
92       if(method.compare("show") == 0) {
93         window_->Show();
94       }
95       else if(method.compare("focus") == 0) {
96         bool focus = true;
97         args->GetBoolean("focus", &focus);
98         window_->Focus(focus);
99       }
100       else if(method.compare("maximize") == 0) {
101         window_->Maximize();
102       }
103       else if(method.compare("unmaximize") == 0) {
104         window_->UnMaximize();
105       }
106       else if(method.compare("minimize") == 0) {
107         window_->Minimize();
108       }
109       else if(method.compare("restore") == 0) {
110         window_->Restore();
111       }
112       else if(method.compare("set_title") == 0) {
113         std::string title = "";
114         args->GetString("title", &title);
115         window_->SetTitle(title);
116       }
117       else if(method.compare("move") == 0) {
118     	int x, y;
119     	args->GetInteger("x", &x);
120     	args->GetInteger("y", &y);
121     
122     	window_->Move(x, y);
123       }
124       else if(method.compare("resize") == 0) {
125     	int width, height;
126     	args->GetInteger("width", &width);
127     	args->GetInteger("height", &height);
128     
129     	LOG(INFO) << "calling window_->Resize(" << width << ", " << height << ")";
130     	window_->Resize(width, height);
131       }
132       else if(method.compare("close") == 0) {
133         window_->Close();
134       }
135       else if(method.compare("is_closed") == 0) {
136         res->SetBoolean("is_closed", window_->is_closed());
137       }
138       else if(method.compare("size") == 0) {
139         res->SetInteger("size.width", window_->size().width());
140         res->SetInteger("size.height", window_->size().height());
141       }
142       else if(method.compare("position") == 0) {
143         res->SetInteger("position.x", window_->position().x());
144         res->SetInteger("position.y", window_->position().y());
145       }
146       else {
147         err = "thrust_window_binding:method_not_found";
148       }
149     
150       callback.Run(err, scoped_ptr<base::DictionaryValue>(res).Pass());
151     }
