29     ScriptContext::ScriptContext(const v8::Handle<v8::Context>& v8_context,
30                                  blink::WebFrame* web_frame)
31         : v8_context_(v8_context),
32           web_frame_(web_frame),
33           safe_builtins_(this),
34           isolate_(v8_context->GetIsolate()) {
35       VLOG(1) << "Created context:\n"
36               << "  frame:        " << web_frame_;
37     }

39     ScriptContext::~ScriptContext() {
40       VLOG(1) << "Destroyed context for extension";
41       Invalidate();
42     }

44     void ScriptContext::Invalidate() {
45       if (!is_valid())
46         return;
47       if (module_system_)
48         module_system_->Invalidate();
49       web_frame_ = NULL;
50       v8_context_.reset();
51     }

53     content::RenderView* ScriptContext::GetRenderView() const {
54       if (web_frame_ && web_frame_->view())
55         return content::RenderView::FromWebView(web_frame_->view());
56       else
57         return NULL;
58     }

60     v8::Local<v8::Value> ScriptContext::CallFunction(
61         v8::Handle<v8::Function> function,
62         int argc,
63         v8::Handle<v8::Value> argv[]) const {
64       v8::EscapableHandleScope handle_scope(isolate());
65       v8::Context::Scope scope(v8_context());
66     
67       blink::WebScopedMicrotaskSuppression suppression;
68       if (!is_valid()) {
69         return handle_scope.Escape(
70             v8::Local<v8::Primitive>(v8::Undefined(isolate())));
71       }
72     
73       v8::Handle<v8::Object> global = v8_context()->Global();
74       if (!web_frame_)
75         return handle_scope.Escape(function->Call(global, argc, argv));
76       return handle_scope.Escape(
77           v8::Local<v8::Value>(web_frame_->callFunctionEvenIfScriptDisabled(
78               function, global, argc, argv)));
79     }

89     // static
90     GURL ScriptContext::GetDataSourceURLForFrame(const blink::WebFrame* frame) {
91       // Normally we would use frame->document().url() to determine the document's
92       // URL, but to decide whether to inject a content script, we use the URL from
93       // the data source. This "quirk" helps prevents content scripts from
94       // inadvertently adding DOM elements to the compose iframe in Gmail because
95       // the compose iframe's dataSource URL is about:blank, but the document URL
96       // changes to match the parent document after Gmail document.writes into
97       // it to create the editor.
98       // http://code.google.com/p/chromium/issues/detail?id=86742
99       blink::WebDataSource* data_source = frame->provisionalDataSource()
100                                               ? frame->provisionalDataSource()
101                                               : frame->dataSource();
102       CHECK(data_source);
103       return GURL(data_source->request().url());
104     }

106     // static
107     GURL ScriptContext::GetEffectiveDocumentURL(const blink::WebFrame* frame,
108                                                 const GURL& document_url,
109                                                 bool match_about_blank) {
110       // Common scenario. If |match_about_blank| is false (as is the case in most
111       // extensions), or if the frame is not an about:-page, just return
112       // |document_url| (supposedly the URL of the frame).
113       if (!match_about_blank || !document_url.SchemeIs(url::kAboutScheme))
114         return document_url;
115     
116       // Non-sandboxed about:blank and about:srcdoc pages inherit their security
117       // origin from their parent frame/window. So, traverse the frame/window
118       // hierarchy to find the closest non-about:-page and return its URL.
119       const blink::WebFrame* parent = frame;
120       do {
121         parent = parent->parent() ? parent->parent() : parent->opener();
122       } while (parent != NULL &&
123                GURL(parent->document().url()).SchemeIs(url::kAboutScheme));
124     
125       if (parent) {
126         // Only return the parent URL if the frame can access it.
127         const blink::WebDocument& parent_document = parent->document();
128         if (frame->document().securityOrigin().canAccess(
129                 parent_document.securityOrigin()))
130           return parent_document.url();
131       }
132       return document_url;
133     }
