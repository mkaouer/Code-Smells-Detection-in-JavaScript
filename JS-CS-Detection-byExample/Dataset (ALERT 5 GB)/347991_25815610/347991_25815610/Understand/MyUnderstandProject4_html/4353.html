43     int 
44     FramelessView::ResizingBorderHitTest(
45         const gfx::Point& point) 
46     {
47       // Check the frame first, as we allow a small area overlapping the contents
48       // to be used for resize handles.
49       bool can_ever_resize = frame_->widget_delegate() ?
50           frame_->widget_delegate()->CanResize() :
51           false;
52       // Don't allow overlapping resize handles when the window is maximized or
53       // fullscreen, as it can't be resized in those states.
54       int resize_border =
55           frame_->IsMaximized() || frame_->IsFullscreen() ? 0 :
56           kResizeInsideBoundsSize;
57       return GetHTComponentForFrame(point, resize_border, resize_border,
58           kResizeAreaCornerSize, kResizeAreaCornerSize, can_ever_resize);
59     }

67     gfx::Rect 
68     FramelessView::GetWindowBoundsForClientBounds(
69         const gfx::Rect& client_bounds) const 
70     {
71       gfx::Rect window_bounds = client_bounds;
72       // Enforce minimum size (1, 1) in case that client_bounds is passed with
73       // empty size. This could occur when the frameless window is being
74       // initialized.
75       if (window_bounds.IsEmpty()) {
76         window_bounds.set_width(1);
77         window_bounds.set_height(1);
78       }
79       return window_bounds;
80     }

82     int 
83     FramelessView::NonClientHitTest(
84         const gfx::Point& cursor) 
85     {
86       if (frame_->IsFullscreen())
87         return HTCLIENT;
88     
89       // Check for possible draggable region in the client area for the frameless
90       // window.
91       SkRegion* draggable_region = window_->GetDraggableRegion();
92       if (draggable_region && draggable_region->contains(cursor.x(), cursor.y()))
93         return HTCAPTION;
94     
95       // Support resizing frameless window by dragging the border.
96       int frame_component = ResizingBorderHitTest(cursor);
97       if (frame_component != HTNOWHERE)
98         return frame_component;
99     
100       return HTCLIENT;
101     }
