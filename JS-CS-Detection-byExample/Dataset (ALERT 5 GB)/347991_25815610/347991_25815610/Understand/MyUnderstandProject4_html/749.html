76     bool IsAltKey(const content::NativeWebKeyboardEvent& event) {
77     #if defined(USE_X11)
78       // 164 and 165 represent VK_LALT and VK_RALT.
79       return event.windowsKeyCode == 164 || event.windowsKeyCode == 165;
80     #else
81       return event.windowsKeyCode == ui::VKEY_MENU;
82     #endif
83     }
84     
85     bool IsAltModifier(const content::NativeWebKeyboardEvent& event) {
86       typedef content::NativeWebKeyboardEvent::Modifiers Modifiers;
87       return (event.modifiers == Modifiers::AltKey) ||
88              (event.modifiers == (Modifiers::AltKey | Modifiers::IsLeft)) ||
89              (event.modifiers == (Modifiers::AltKey | Modifiers::IsRight));
90     }

121     void 
122     ThrustWindow::PlatformCreateWindow(
123         const gfx::Size& size)
124     {
125       window_.reset(new views::Widget());
126     
127       LOG(INFO) << "Create Window: " << size.width() << "x" << size.height();
128     
129       gfx::Rect bounds(0, 0, size.width(), size.height());
130       window_->AddObserver(this);
131     
132       views::Widget::InitParams params;
133       params.ownership = views::Widget::InitParams::WIDGET_OWNS_NATIVE_WIDGET;
134       params.bounds = bounds;
135       params.delegate = this;
136       params.type = views::Widget::InitParams::TYPE_WINDOW;
137       params.remove_standard_frame = !has_frame_;
138     
139     #if defined(USE_X11)
140       // Set WM_WINDOW_ROLE.
141       params.wm_role_name = base::StringPrintf(
142           "%s/%s/%d", "Thrust", 
143           ThrustShellBrowserClient::Get()->GetAppName().c_str(),
144           ++kWindowsCreated);
145       // Set WM_CLASS.
146       params.wm_class_name = "thrust";
147       params.wm_class_class = "Thrust";
148     #endif
149     
150       window_->Init(params);
151     
152     #if defined(USE_X11)
153       // Set _GTK_THEME_VARIANT to dark if we have "dark-theme" option set.
154       bool use_dark_theme = false;
155       /* TODO(spolu): Add option */
156       /*
157       if (options.Get(switches::kDarkTheme, &use_dark_theme) && use_dark_theme) {
158         XDisplay* xdisplay = gfx::GetXDisplay();
159         XChangeProperty(xdisplay, GetAcceleratedWidget(),
160                         XInternAtom(xdisplay, "_GTK_THEME_VARIANT", False),
161                         XInternAtom(xdisplay, "UTF8_STRING", False),
162                         8, PropModeReplace,
163                         reinterpret_cast<const unsigned char*>("dark"),
164                         4);
165       }
166       */
167     
168       // Before the window is mapped the SetWMSpecState can not work, so we have
169       // to manually set the _NET_WM_STATE.
170       bool skip_taskbar = false;
171       /* TODO(spolu): Add option */
172       /*
173       if (options.Get(switches::kSkipTaskbar, &skip_taskbar) && skip_taskbar) {
174         std::vector<::Atom> state_atom_list;
175         state_atom_list.push_back(GetAtom("_NET_WM_STATE_SKIP_TASKBAR"));
176         ui::SetAtomArrayProperty(GetAcceleratedWidget(), "_NET_WM_STATE", "ATOM",
177                                  state_atom_list);
178       }
179       */
180     #endif
181     
182       // Add web view.
183       SetLayoutManager(new views::FillLayout());
184       set_background(views::Background::CreateStandardPanelBackground());
185       AddChildView(inspectable_web_contents()->GetView()->GetView());
186     
187       /* TODO(spolu): Add option */
188       /*
189       if(has_frame_ &&
190          options.Get(switches::kUseContentSize, &use_content_size_) &&
191          use_content_size_)
192         bounds = ContentBoundsToWindowBounds(bounds);
193       */
194     
195       window_->UpdateWindowIcon();
196       window_->CenterWindow(bounds.size());
197       Layout();
198     }

330     void 
331     ThrustWindow::PlatformSetContentSize(
332         int width, int height)
333     {
334       if (!has_frame_) {
335         PlatformResize(width, height);
336         return;
337       }
338     
339       gfx::Size size(width, height);
340       gfx::Rect bounds = window_->GetWindowBoundsInScreen();
341       gfx::Size new_size = 
342           ContentBoundsToWindowBounds(gfx::Rect(bounds.origin(), size)).size();
343       PlatformResize(size.width(), size.height());
344     }

374     void 
375     ThrustWindow::OnWidgetActivationChanged(
376         views::Widget* widget, 
377         bool active) 
378     {
379       if(widget != window_.get())
380         return;
381     
382       if(active) {
383         binding_->EmitFocus();
384       }
385       else {
386         binding_->EmitBlur();
387       }
388     
389       if(active && web_contents()) {
390         web_contents()->Focus();
391       }
392     }

528     views::NonClientFrameView* 
529     ThrustWindow::CreateNonClientFrameView(
530         views::Widget* widget) 
531     {
532     #if defined(OS_WIN)
533       WinFrameView* frame_view =  new WinFrameView;
534       frame_view->Init(this, widget);
535       return frame_view;
536     #elif defined(OS_LINUX)
537       if(has_frame_) {
538         return new views::NativeFrameView(widget);
539       } 
540       else {
541         FramelessView* frame_view =  new FramelessView();
542         frame_view->Init(this, widget);
543         return frame_view;
544       }
545     #else
546       return NULL;
547     #endif
548     }
