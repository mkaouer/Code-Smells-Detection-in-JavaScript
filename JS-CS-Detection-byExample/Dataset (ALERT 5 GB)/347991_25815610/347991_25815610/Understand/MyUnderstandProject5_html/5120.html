33     void SetGtkTransientForAura(GtkWidget* dialog, aura::Window* parent) {
34       if (!parent || !parent->GetHost())
35         return;
36     
37       gtk_widget_realize(dialog);
38       GdkWindow* gdk_window = gtk_widget_get_window(dialog);
39     
40       // TODO(erg): Check to make sure we're using X11 if wayland or some other
41       // display server ever happens. Otherwise, this will crash.
42       XSetTransientForHint(GDK_WINDOW_XDISPLAY(gdk_window),
43                            GDK_WINDOW_XID(gdk_window),
44                            parent->GetHost()->GetAcceleratedWidget());
45     
46       // We also set the |parent| as a property of |dialog|, so that we can unlink
47       // the two later.
48       g_object_set_data(G_OBJECT(dialog), kAuraTransientParent, parent);
49     }

51     // Makes sure that .jpg also shows .JPG.
52     gboolean FileFilterCaseInsensitive(const GtkFileFilterInfo* file_info,
53                                        std::string* file_extension) {
54       return EndsWith(file_info->filename, *file_extension, false);
55     }

57     // Deletes |data| when gtk_file_filter_add_custom() is done with it.
58     void OnFileFilterDataDestroyed(std::string* file_extension) {
59       delete file_extension;
60     }

62     class FileChooserDialog {
63      public:
64       FileChooserDialog(GtkFileChooserAction action,
65                         thrust_shell::ThrustWindow* parent_window,
66                         const std::string& title,
67                         const base::FilePath& default_path,
68                         const Filters& filters) {
69         const char* confirm_text = GTK_STOCK_OK;
70         if (action == GTK_FILE_CHOOSER_ACTION_SAVE)
71           confirm_text = GTK_STOCK_SAVE;
72         else if (action == GTK_FILE_CHOOSER_ACTION_OPEN)
73           confirm_text = GTK_STOCK_OPEN;
74     
75         dialog_ = gtk_file_chooser_dialog_new(
76             title.c_str(),
77             NULL,
78             action,
79             GTK_STOCK_CANCEL, GTK_RESPONSE_CANCEL,
80             confirm_text, GTK_RESPONSE_ACCEPT,
81             NULL);
82         if (parent_window) {
83           gfx::NativeWindow window = parent_window->GetNativeWindow();
84           SetGtkTransientForAura(dialog_, window);
85         }
86     
87         if (action == GTK_FILE_CHOOSER_ACTION_SAVE)
88           gtk_file_chooser_set_do_overwrite_confirmation(GTK_FILE_CHOOSER(dialog_),
89                                                          TRUE);
90         if (action != GTK_FILE_CHOOSER_ACTION_OPEN)
91           gtk_file_chooser_set_create_folders(GTK_FILE_CHOOSER(dialog_), TRUE);
92     
93         gtk_window_set_modal(GTK_WINDOW(dialog_), TRUE);
94     
95         if (!default_path.empty()) {
96           if (base::DirectoryExists(default_path))
97             gtk_file_chooser_set_current_folder(GTK_FILE_CHOOSER(dialog_),
98                                                 default_path.value().c_str());
99           else
100             gtk_file_chooser_set_filename(GTK_FILE_CHOOSER(dialog_),
101                                           default_path.value().c_str());
102         }
103     
104         if (!filters.empty())
105           AddFilters(filters);
106       }
107     
108       virtual ~FileChooserDialog() {
109         gtk_widget_destroy(dialog_);
110       }
111     
112       void RunAsynchronous() {
113         g_signal_connect(dialog_, "delete-event",
114                          G_CALLBACK(gtk_widget_hide_on_delete), NULL);
115         g_signal_connect(dialog_, "response",
116                          G_CALLBACK(OnFileDialogResponseThunk), this);
117         gtk_widget_show_all(dialog_);
118     
119         // We need to call gtk_window_present after making the widgets visible to
120         // make sure window gets correctly raised and gets focus.
121         int time = views::X11DesktopHandler::get()->wm_user_time_ms();
122         gtk_window_present_with_time(GTK_WINDOW(dialog_), time);
123       }
124     
125       void RunSaveAsynchronous(const SaveDialogCallback& callback) {
126         save_callback_ = callback;
127         RunAsynchronous();
128       }
129     
130       void RunOpenAsynchronous(const OpenDialogCallback& callback) {
131         open_callback_ = callback;
132         RunAsynchronous();
133       }
134     
135       base::FilePath GetFileName() const {
136         gchar* filename = gtk_file_chooser_get_filename(GTK_FILE_CHOOSER(dialog_));
137         base::FilePath path(filename);
138         g_free(filename);
139         return path;
140       }
141     
142       std::vector<base::FilePath> GetFileNames() const {
143         std::vector<base::FilePath> paths;
144         GSList* filenames = gtk_file_chooser_get_filenames(
145             GTK_FILE_CHOOSER(dialog_));
146         for (GSList* iter = filenames; iter != NULL; iter = g_slist_next(iter)) {
147           base::FilePath path(static_cast<char*>(iter->data));
148           g_free(iter->data);
149           paths.push_back(path);
150         }
151         g_slist_free(filenames);
152         return paths;
153       }
154     
155       CHROMEGTK_CALLBACK_1(FileChooserDialog, void, OnFileDialogResponse, int);
156     
157       GtkWidget* dialog() const { return dialog_; }
158     
159      private:
160       void AddFilters(const Filters& filters);
161     
162       GtkWidget* dialog_;
163     
164       SaveDialogCallback save_callback_;
165       OpenDialogCallback open_callback_;
166     
167       DISALLOW_COPY_AND_ASSIGN(FileChooserDialog);
168     };
169     
170     void FileChooserDialog::OnFileDialogResponse(GtkWidget* widget, int response) {
171       gtk_widget_hide_all(dialog_);
172     
173       if (!save_callback_.is_null()) {
174         if (response == GTK_RESPONSE_ACCEPT)
175           save_callback_.Run(true, GetFileName());
176         else
177           save_callback_.Run(false, base::FilePath());
178       } else if (!open_callback_.is_null()) {
179         if (response == GTK_RESPONSE_ACCEPT)
180           open_callback_.Run(true, GetFileNames());
181         else
182           open_callback_.Run(false, std::vector<base::FilePath>());
183       }
184       delete this;
185     }

187     void FileChooserDialog::AddFilters(const Filters& filters) {
188       for (size_t i = 0; i < filters.size(); ++i) {
189         const Filter& filter = filters[i];
190         GtkFileFilter* gtk_filter = gtk_file_filter_new();
191     
192         for (size_t j = 0; j < filter.second.size(); ++j) {
193           scoped_ptr<std::string> file_extension(
194               new std::string("." + filter.second[j]));
195           gtk_file_filter_add_custom(
196               gtk_filter,
197               GTK_FILE_FILTER_FILENAME,
198               reinterpret_cast<GtkFileFilterFunc>(FileFilterCaseInsensitive),
199               file_extension.release(),
200               reinterpret_cast<GDestroyNotify>(OnFileFilterDataDestroyed));
201         }
202     
203         gtk_file_filter_set_name(gtk_filter, filter.first.c_str());
204         gtk_file_chooser_add_filter(GTK_FILE_CHOOSER(dialog_), gtk_filter);
205       }
206     }

210     bool ShowOpenDialog(thrust_shell::ThrustWindow* parent_window,
211                         const std::string& title,
212                         const base::FilePath& default_path,
213                         const Filters& filters,
214                         int properties,
215                         std::vector<base::FilePath>* paths) {
216       GtkFileChooserAction action = GTK_FILE_CHOOSER_ACTION_OPEN;
217       if (properties & FILE_DIALOG_OPEN_DIRECTORY)
218         action = GTK_FILE_CHOOSER_ACTION_SELECT_FOLDER;
219       FileChooserDialog open_dialog(action, parent_window, title, default_path,
220                                     filters);
221       if (properties & FILE_DIALOG_MULTI_SELECTIONS)
222         gtk_file_chooser_set_select_multiple(GTK_FILE_CHOOSER(open_dialog.dialog()),
223                                              TRUE);
224     
225       gtk_widget_show_all(open_dialog.dialog());
226       int response = gtk_dialog_run(GTK_DIALOG(open_dialog.dialog()));
227       if (response == GTK_RESPONSE_ACCEPT) {
228         *paths = open_dialog.GetFileNames();
229         return true;
230       } else {
231         return false;
232       }
233     }

235     void ShowOpenDialog(thrust_shell::ThrustWindow* parent_window,
236                         const std::string& title,
237                         const base::FilePath& default_path,
238                         const Filters& filters,
239                         int properties,
240                         const OpenDialogCallback& callback) {
241       GtkFileChooserAction action = GTK_FILE_CHOOSER_ACTION_OPEN;
242       if (properties & FILE_DIALOG_OPEN_DIRECTORY)
243         action = GTK_FILE_CHOOSER_ACTION_SELECT_FOLDER;
244       FileChooserDialog* open_dialog = new FileChooserDialog(
245           action, parent_window, title, default_path, filters);
246       if (properties & FILE_DIALOG_MULTI_SELECTIONS)
247         gtk_file_chooser_set_select_multiple(
248             GTK_FILE_CHOOSER(open_dialog->dialog()), TRUE);
249     
250       open_dialog->RunOpenAsynchronous(callback);
251     }

253     bool ShowSaveDialog(thrust_shell::ThrustWindow* parent_window,
254                         const std::string& title,
255                         const base::FilePath& default_path,
256                         const Filters& filters,
257                         base::FilePath* path) {
258       FileChooserDialog save_dialog(GTK_FILE_CHOOSER_ACTION_SAVE, parent_window,
259                                     title, default_path, filters);
260       gtk_widget_show_all(save_dialog.dialog());
261       int response = gtk_dialog_run(GTK_DIALOG(save_dialog.dialog()));
262       if (response == GTK_RESPONSE_ACCEPT) {
263         *path = save_dialog.GetFileName();
264         return true;
265       } else {
266         return false;
267       }
268     }
