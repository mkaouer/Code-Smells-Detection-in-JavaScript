84     bool 
85     WebViewBindings::AttemptEmitEvent(
86         int guest_instance_id,
87         const std::string type,
88         const base::DictionaryValue& event)
89     {
90       if(guest_handlers_.find(guest_instance_id) != guest_handlers_.end()) {
91         v8::HandleScope handle_scope(context()->isolate());
92     
93         v8::Local<v8::String> type_arg = 
94           v8::String::NewFromUtf8(context()->isolate(), type.c_str());
95         scoped_ptr<V8ValueConverter> converter(V8ValueConverter::create());
96         v8::Handle<v8::Value> event_arg = converter->ToV8Value(&event, context()->v8_context());
97     
98         v8::Local<v8::Function> handler = 
99                 v8::Local<v8::Function>::New(context()->isolate(),
100                                              guest_handlers_[guest_instance_id]);
101     
102         v8::Local<v8::Value> argv[2] = { type_arg,
103                                          event_arg };
104         context()->CallFunction(handler, 2, argv);
105       }
106       return false;
107     }

109     void 
110     WebViewBindings::CreateGuest(
111         const v8::FunctionCallbackInfo<v8::Value>& args) 
112     {
113       if(args.Length() != 1 || !args[0]->IsObject()) {
114         NOTREACHED();
115         return;
116       }
117     
118       v8::Local<v8::Object> object = args[0]->ToObject();
119     
120       scoped_ptr<V8ValueConverter> converter(V8ValueConverter::create());
121       scoped_ptr<base::Value> value(
122           converter->FromV8Value(object, context()->v8_context()));
123     
124       if(!value) {
125         return;                                                             
126       }
127       if(!value->IsType(base::Value::TYPE_DICTIONARY)) {
128         return;
129       }
130     
131       scoped_ptr<base::DictionaryValue> params(
132           static_cast<base::DictionaryValue*>(value.release()));
133     
134       LOG(INFO) << "WEB_VIEW_BINDINGS: CreateGuest";
135     
136       int id = 0;
137       render_frame_observer_->Send(
138           new ThrustFrameHostMsg_CreateWebViewGuest(
139             render_frame_observer_->routing_id(), 
140             *params.get(), &id));
141     
142       args.GetReturnValue().Set(v8::Integer::New(context()->isolate(), id));
143     }

312     void 
313     WebViewBindings::Find(
314         const v8::FunctionCallbackInfo<v8::Value>& args) 
315     {
316       if(args.Length() != 4 || !args[0]->IsNumber() || 
317          !args[1]->IsNumber() || !args[2]->IsString() || !args[3]->IsObject()) {
318         NOTREACHED();
319         return;
320       }
321     
322       int guest_instance_id = args[0]->NumberValue();
323       int request_id = args[1]->NumberValue();
324       std::string search_text(*v8::String::Utf8Value(args[2]));
325       v8::Local<v8::Object> object = args[3]->ToObject();
326     
327       scoped_ptr<V8ValueConverter> converter(V8ValueConverter::create());
328       scoped_ptr<base::Value> value(
329           converter->FromV8Value(object, context()->v8_context()));
330     
331       if(!value) {
332         return;                                                             
333       }
334       if(!value->IsType(base::Value::TYPE_DICTIONARY)) {
335         return;
336       }
337     
338       scoped_ptr<base::DictionaryValue> options(
339           static_cast<base::DictionaryValue*>(value.release()));
340     
341       LOG(INFO) << "WEB_VIEW_BINDINGS: Find " << guest_instance_id << " " 
342                 << request_id << " " << search_text;
343     
344       render_frame_observer_->Send(
345           new ThrustFrameHostMsg_WebViewGuestFind(
346             render_frame_observer_->routing_id(), 
347             guest_instance_id, request_id, search_text, *options.get()));
348     }
