21     std::vector<ThrustShellRenderFrameObserver*> 
22       ThrustShellRenderFrameObserver::s_instances;
23     
24     ThrustShellRenderFrameObserver::ThrustShellRenderFrameObserver(
25         RenderFrame* render_frame)
26         : RenderFrameObserver(render_frame) 
27     {
28       LOG(INFO) << "************* RENDER FRAME CREATED " << render_frame;
29       s_instances.push_back(this);
30     }
31     
32     ThrustShellRenderFrameObserver::~ThrustShellRenderFrameObserver()
33     {
34       for(size_t i = 0; i < s_instances.size(); ++i) {
35         if(s_instances[i] == this) {
36           s_instances.erase(s_instances.begin() + i);
37           break;
38         }
39       }
40     }
41     
42     // static
43     ThrustShellRenderFrameObserver* 
44     ThrustShellRenderFrameObserver::FromRenderFrame(
45         RenderFrame* render_frame)
46     {
47       for(size_t i = 0; i < s_instances.size(); ++i) {
48         if(s_instances[i]->render_frame() == render_frame) {
49           return s_instances[i];
50         }
51       }
52       return NULL;
53     }
54     
55     bool 
56     ThrustShellRenderFrameObserver::OnMessageReceived(
57         const IPC::Message& message) 
58     {
59       bool handled = true;
60       IPC_BEGIN_MESSAGE_MAP(ThrustShellRenderFrameObserver, message)
61         IPC_MESSAGE_HANDLER(ThrustFrameMsg_WebViewEmit, 
62                             WebViewEmit)
63         IPC_MESSAGE_UNHANDLED(handled = false)
64       IPC_END_MESSAGE_MAP()
65     
66       return handled;
67     }
68     
69     void 
70     ThrustShellRenderFrameObserver::AddWebViewBindings(
71         extensions::WebViewBindings* bindings)
72     {
73       /* Does not own bindings just stores a pointer to it. */
74       bindings_.push_back(bindings);
75     }
76     
77     void 
78     ThrustShellRenderFrameObserver::RemoveWebViewBindings(
79         extensions::WebViewBindings* bindings)
80     {
81       for(size_t i = 0; i < bindings_.size(); ++i) {
82         if(bindings_[i] == bindings) {
83           bindings_.erase(bindings_.begin() + i);
84           break;
85         }
86       }
87     }
88     
89     void 
90     ThrustShellRenderFrameObserver::WebViewEmit(
91         int guest_instance_id,
92         const std::string type,
93         const base::DictionaryValue& event)
94     {
95       for(size_t i = 0; i < bindings_.size(); ++i) {
96         bindings_[i]->AttemptEmitEvent(guest_instance_id, type, event);
97       }
98     }
