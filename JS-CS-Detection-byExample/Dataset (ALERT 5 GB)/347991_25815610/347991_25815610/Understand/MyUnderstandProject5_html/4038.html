92     // This observer ensures that the WebViewGuest destroys itself when its
93     // embedder goes away.
94     class WebViewGuest::EmbedderWebContentsObserver : public WebContentsObserver {
95      public:
96       explicit EmbedderWebContentsObserver(WebViewGuest* guest)
97           : WebContentsObserver(guest->embedder_web_contents()),
98             destroyed_(false),
99             guest_(guest) {
100       }
101     
102       virtual ~EmbedderWebContentsObserver() {
103       }
104     
105       // WebContentsObserver implementation.
106       virtual void WebContentsDestroyed() OVERRIDE {
107         Destroy();
108       }
109     
110       virtual void RenderProcessGone(base::TerminationStatus status) OVERRIDE {
111         Destroy();
112       }
113     
114      private:
115       void Destroy() {
116         if(destroyed_) {
117           return;
118         }
119         destroyed_ = true;
120         guest_->embedder_web_contents_ = NULL;
121         //guest_->EmbedderDestroyed();
122         guest_->Destroy();
123       }
124     
125       bool          destroyed_;
126       WebViewGuest* guest_;
127     
128       DISALLOW_COPY_AND_ASSIGN(EmbedderWebContentsObserver);
129     };

145     void
146     WebViewGuest::Init(
147         WebContents* guest_web_contents)
148     {
149       WebContentsObserver::Observe(guest_web_contents);
150       guest_web_contents_.reset(
151           brightray::InspectableWebContents::Create(guest_web_contents));
152       guest_web_contents->SetDelegate(this);
153       browser_context_ = guest_web_contents->GetBrowserContext();
154     
155       webcontents_webview_map.Get().insert(
156           std::make_pair(guest_web_contents, this));
157     
158       notification_registrar_.Add(
159           this, content::NOTIFICATION_LOAD_COMPLETED_MAIN_FRAME,
160           content::Source<WebContents>(guest_web_contents));
161     
162       notification_registrar_.Add(
163           this, content::NOTIFICATION_RESOURCE_RECEIVED_REDIRECT,
164           content::Source<WebContents>(guest_web_contents));
165     
166       ThrustShellBrowserClient::Get()->ThrustSessionForBrowserContext(browser_context_)->
167         AddGuest(guest_instance_id_, guest_web_contents);
168     
169       LOG(INFO) << "WebViewGuest Init: " << this;
170     }

203     // static
204     void 
205     WebViewGuest::ParsePartitionParam(
206         const base::DictionaryValue* extra_params,
207         std::string* storage_partition_id,
208         bool* persist_storage) 
209     {
210       std::string partition_str;
211       if (!extra_params->GetString(webview::kStoragePartitionId, &partition_str)) {
212         return;
213       }
214     
215       // Since the "persist:" prefix is in ASCII, StartsWith will work fine on
216       // UTF-8 encoded |partition_id|. If the prefix is a match, we can safely
217       // remove the prefix without splicing in the middle of a multi-byte codepoint.
218       // We can use the rest of the string as UTF-8 encoded one.
219       if (StartsWithASCII(partition_str, "persist:", true)) {
220         size_t index = partition_str.find(":");
221         CHECK(index != std::string::npos);
222         // It is safe to do index + 1, since we tested for the full prefix above.
223         *storage_partition_id = partition_str.substr(index + 1);
224     
225         if (storage_partition_id->empty()) {
226           // TODO(lazyboy): Better way to deal with this error.
227           return;
228         }
229         *persist_storage = true;
230       } else {
231         *storage_partition_id = partition_str;
232         *persist_storage = false;
233       }
234     }

241     {
242       LOG(INFO) << "WebViewGuest Destroy: " << this;
243     
244       if(!destruction_callback_.is_null())
245         destruction_callback_.Run();
246     
247       webcontents_webview_map.Get().erase(guest_web_contents());
248     
249       ThrustShellBrowserClient::Get()->ThrustSessionForBrowserContext(browser_context_)->
250         RemoveGuest(guest_instance_id_);
251     
252       guest_web_contents_.reset();
253     }

327     void 
328     WebViewGuest::Observe(
329         int type,
330         const content::NotificationSource& source,
331         const content::NotificationDetails& details) 
332     {
333       switch (type) {
334         case content::NOTIFICATION_LOAD_COMPLETED_MAIN_FRAME: {
335           DCHECK_EQ(content::Source<WebContents>(source).ptr(),
336                     guest_web_contents());
337           if(content::Source<WebContents>(source).ptr() == guest_web_contents()) {
338           }
339           break;
340         }
341         case content::NOTIFICATION_RESOURCE_RECEIVED_REDIRECT: {
342           DCHECK_EQ(content::Source<WebContents>(source).ptr(),
343                     guest_web_contents());
344           break;
345         }
346         default:
347           NOTREACHED() << "Unexpected notification sent.";
348           break;
349       }
350     }

494     void 
495     WebViewGuest::SetAutoSize(
496         bool enabled,
497         const gfx::Size& min_size,
498         const gfx::Size& max_size) 
499     {
500       min_auto_size_ = min_size;
501       min_auto_size_.SetToMin(max_size);
502       max_auto_size_ = max_size;
503       max_auto_size_.SetToMax(min_size);
504     
505       enabled &= !min_auto_size_.IsEmpty() && !max_auto_size_.IsEmpty();
506       if (!enabled && !auto_size_enabled_)
507         return;
508     
509       auto_size_enabled_ = enabled;
510     
511       if(!attached())
512         return;
513     
514       content::RenderViewHost* rvh = guest_web_contents()->GetRenderViewHost();
515       if (auto_size_enabled_) {
516         rvh->EnableAutoResize(min_auto_size_, max_auto_size_);
517       } 
518       else {
519         rvh->DisableAutoResize(element_size_);
520         guest_size_ = element_size_;
521         //GuestSizeChangedDueToAutoSize(guest_size_, element_size_);
522       }
523     }

679           content::HostZoomMap::GetZoomLevel(guest_web_contents()));
680       if(current_zoom_factor_ != zoom_factor) {
681         base::DictionaryValue event;
682         event.SetDouble("old_zoom_factor", current_zoom_factor_);
683         event.SetDouble("new_zoom_factor", zoom_factor);
684     
685         GetThrustWindow()->WebViewEmit(
686             guest_instance_id_,
687             "zoom-changed",
688             event);
689         current_zoom_factor_ = zoom_factor;

728     void 
729     WebViewGuest::RenderProcessGone(
730         base::TerminationStatus status) 
731     {
732       // Cancel all find sessions in progress.
733       // find_helper_.CancelAllFindSessions();
734     
735       base::DictionaryValue event;
736       event.SetInteger("process_id",
737                        guest_web_contents()->GetRenderProcessHost()->GetID());
738       event.SetString("reason", TerminationStatusToString(status));
739     
740       GetThrustWindow()->WebViewEmit(
741           guest_instance_id_,

749     {
750       if(!attached()) {
751         return;
752       }
753       content::NavigationController& controller =
754           guest_web_contents()->GetController();
755       content::NavigationEntry* entry = controller.GetVisibleEntry();
756       if(!entry) {
757         return;
758       }

760       guest_web_contents()->GetController().Reload(false);
761     }

838     content::WebContents* 
839     WebViewGuest::OpenURLFromTab(
840         content::WebContents* source,
841         const content::OpenURLParams& params) 
842     {
843       if(params.disposition == CURRENT_TAB) {
844         content::NavigationController::LoadURLParams load_url_params(params.url);
845         load_url_params.referrer = params.referrer;
846         load_url_params.transition_type = params.transition;
847         load_url_params.extra_headers = params.extra_headers;
848         load_url_params.should_replace_current_entry =
849           params.should_replace_current_entry;
850         load_url_params.is_renderer_initiated = params.is_renderer_initiated;
851         load_url_params.transferred_global_request_id =
852           params.transferred_global_request_id;
853     
854         guest_web_contents()->GetController().LoadURLWithParams(load_url_params);
855         return guest_web_contents();
856       }
857       else {
858         base::DictionaryValue event;
859         event.SetString("target_url", params.url.spec());
860         event.SetString("disposition", 
861             WindowOpenDispositionToString(params.disposition));
862     
863         GetThrustWindow()->WebViewEmit(
864             guest_instance_id_,
865             "new-window",
866             event);
867         return NULL;
868       }
869     }
