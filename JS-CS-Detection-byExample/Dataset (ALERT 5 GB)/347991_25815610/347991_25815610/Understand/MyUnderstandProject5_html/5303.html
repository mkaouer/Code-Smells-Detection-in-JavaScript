17     // Filter out the "&" in menu label.
18     base::string16 FilterAccecelator(const base::string16& label) {
19       base::string16 out;
20       base::RemoveChars(label, base::ASCIIToUTF16("&").c_str(), &out);
21       return out;
22     }

26     SubmenuButton::SubmenuButton(
27         views::ButtonListener* listener,
28         const base::string16& title,
29         views::MenuButtonListener* menu_button_listener)
30       : views::MenuButton(listener, FilterAccecelator(title),
31         menu_button_listener, false),
32         accelerator_(0),
33         show_underline_(false),
34         underline_start_(-1),
35         underline_end_(-1),
36         text_width_(0),
37         text_height_(0),
38         underline_color_(SK_ColorBLACK) 
39     {
40     #if defined(OS_LINUX)
41       // Dont' use native style border.
42       SetBorder(CreateDefaultBorder().PassAs<views::Border>());
43     #endif
44     
45       if (GetUnderlinePosition(title, &accelerator_, &underline_start_,
46                                &underline_end_))
47         gfx::Canvas::SizeStringInt(GetText(), GetFontList(), &text_width_,
48                                    &text_height_, 0, 0);
49     }

53     }

55     void 
56     SubmenuButton::SetAcceleratorVisibility(
57         bool visible) 
58     {
59       if (visible == show_underline_)
60         return;
61     
62       show_underline_ = visible;
63       SchedulePaint();
64     }
65     
66     void 
67     SubmenuButton::SetUnderlineColor(
68         SkColor color) 
69     {
70       underline_color_ = color;
71     }
72     
73     void 
74     SubmenuButton::OnPaint(
75         gfx::Canvas* canvas) 
76     {
77       views::MenuButton::OnPaint(canvas);
78     
79       if (show_underline_ && (underline_start_ != underline_end_)) {
80         int padding = (width() - text_width_) / 2;
81         int underline_height = (height() + text_height_) / 2 - 2;
82         canvas->DrawLine(gfx::Point(underline_start_ + padding, underline_height),
83                          gfx::Point(underline_end_ + padding, underline_height),
84                          underline_color_);
85       }
86     }
87     
88     bool 
89     SubmenuButton::GetUnderlinePosition(
90         const base::string16& text,
91         base::char16* accelerator,
92         int* start, int* end) 
93     {
94       int pos, span;
95       base::string16 trimmed = gfx::RemoveAcceleratorChar(text, '&', &pos, &span);
96       if (pos > -1 && span != 0) {
97         *accelerator = base::ToUpperASCII(trimmed[pos]);
98         GetCharacterPosition(trimmed, pos, start);
99         GetCharacterPosition(trimmed, pos + span, end);
100         return true;
101       }
102     
103       return false;
104     }
105     
106     void 
107     SubmenuButton::GetCharacterPosition(
108         const base::string16& text, 
109         int index, 
110         int* pos) 
111     {
112       int height;
113       gfx::Canvas::SizeStringInt(text.substr(0, index), GetFontList(), pos, &height,
114                                  0, 0);
115     }
