36     GetMenuBarColor(
37           SkColor* enabled, 
38           SkColor* disabled, 
39           SkColor* highlight,
40           SkColor* hover, 
41           SkColor* background) 
42     {
43       libgtk2ui::OwnedWidgetGtk fake_menu_bar;

46       GtkStyle* style = gtk_rc_get_style(fake_menu_bar.get());

49       *highlight = libgtk2ui::GdkColorToSkColor(style->fg[GTK_STATE_SELECTED]);
50       *hover = libgtk2ui::GdkColorToSkColor(style->fg[GTK_STATE_PRELIGHT]);
51       *background = libgtk2ui::GdkColorToSkColor(style->bg[GTK_STATE_NORMAL]);
52     }

57     MenuBar::MenuBar()
58       : background_color_(kDefaultColor),
59         menu_model_(NULL) 
60     {
61     #if defined(OS_WIN)
62       background_color_ = color_utils::GetSysSkColor(COLOR_MENUBAR);
63     #elif defined(USE_X11)
64       GetMenuBarColor(&enabled_color_, &disabled_color_, &highlight_color_,
65                       &hover_color_, &background_color_);
66     #endif
67     
68       set_background(views::Background::CreateSolidBackground(background_color_));
69       SetLayoutManager(new views::BoxLayout(
70           views::BoxLayout::kHorizontal, 0, 0, 0));
71     }

77     void 
78     MenuBar::SetMenu(
79         ui::MenuModel* model) 
80     {
81       menu_model_ = model;
82       RemoveAllChildViews(true);
83     
84       for (int i = 0; i < model->GetItemCount(); ++i) {
85         SubmenuButton* button = new SubmenuButton(this, model->GetLabelAt(i), this);
86         button->set_tag(i);
87     
88     #if defined(USE_X11)
89         button->SetTextColor(views::Button::STATE_NORMAL, enabled_color_);
90         button->SetTextColor(views::Button::STATE_DISABLED, disabled_color_);
91         button->SetTextColor(views::Button::STATE_PRESSED, highlight_color_);
92         button->SetTextColor(views::Button::STATE_HOVERED, hover_color_);
93         button->SetUnderlineColor(enabled_color_);
94     #elif defined(OS_WIN)
95         button->SetUnderlineColor(color_utils::GetSysSkColor(COLOR_GRAYTEXT));
96     #endif
97     
98         AddChildView(button);
99       }
100     }

102     void 
103     MenuBar::SetAcceleratorVisibility(
104         bool visible) 
105     {
106       for (int i = 0; i < child_count(); ++i)
107         static_cast<SubmenuButton*>(child_at(i))->SetAcceleratorVisibility(visible);
108     }
109     
110     int 
111     MenuBar::GetAcceleratorIndex(
112         base::char16 key) 
113     {
114       for (int i = 0; i < child_count(); ++i) {
115         SubmenuButton* button = static_cast<SubmenuButton*>(child_at(i));
116         if (button->accelerator() == key)
117           return i;
118       }
119       return -1;
120     }
121     
122     void 
123     MenuBar::ActivateAccelerator(
124         base::char16 key) 
125     {
126       int i = GetAcceleratorIndex(key);
127       if (i != -1)
128         static_cast<SubmenuButton*>(child_at(i))->Activate();
129     }

137     bool 
138     MenuBar::GetMenuButtonFromScreenPoint(
139         const gfx::Point& point,
140         ui::MenuModel** menu_model,
141         views::MenuButton** button) 
142     {
143       gfx::Point location(point);
144       views::View::ConvertPointFromScreen(this, &location);
145     
146       if (location.x() < 0 || location.x() >= width() || location.y() < 0 ||
147           location.y() >= height())
148         return false;
149     
150       for (int i = 0; i < child_count(); ++i) {
151         views::View* view = child_at(i);
152         if (view->bounds().Contains(location) &&
153             (menu_model_->GetTypeAt(i) == ui::MenuModel::TYPE_SUBMENU)) {
154           *menu_model = menu_model_->GetSubmenuModelAt(i);
155           *button = static_cast<views::MenuButton*>(view);
156           return true;
157         }
158       }
159     
160       return false;
161     }

176     void 
177     MenuBar::OnMenuButtonClicked(
178         views::View* source,
179         const gfx::Point& point) 
180     {
181       // Hide the accelerator when a submenu is activated.
182       SetAcceleratorVisibility(false);
183     
184       if (!menu_model_)
185         return;
186     
187       views::MenuButton* button = static_cast<views::MenuButton*>(source);
188       int id = button->tag();
189       ui::MenuModel::ItemType type = menu_model_->GetTypeAt(id);
190       if (type != ui::MenuModel::TYPE_SUBMENU)
191         return;
192     
193       menu_delegate_.reset(new MenuDelegate(this));
194       menu_delegate_->RunMenu(menu_model_->GetSubmenuModelAt(id), button);
195     }
