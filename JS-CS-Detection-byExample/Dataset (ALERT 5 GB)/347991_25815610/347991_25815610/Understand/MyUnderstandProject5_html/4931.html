84     bool IsAltKey(const content::NativeWebKeyboardEvent& event) {
85     #if defined(USE_X11)
86       // 164 and 165 represent VK_LALT and VK_RALT.
87       return event.windowsKeyCode == 164 || event.windowsKeyCode == 165;
88     #else
89       return event.windowsKeyCode == ui::VKEY_MENU;
90     #endif
91     }
92     
93     bool IsAltModifier(const content::NativeWebKeyboardEvent& event) {
94       typedef content::NativeWebKeyboardEvent::Modifiers Modifiers;
95       return (event.modifiers == Modifiers::AltKey) ||
96              (event.modifiers == (Modifiers::AltKey | Modifiers::IsLeft)) ||
97              (event.modifiers == (Modifiers::AltKey | Modifiers::IsRight));
98     }

196     void 
197     ThrustWindow::PlatformCreateWindow(
198         const gfx::Size& size)
199     {
200       window_.reset(new views::Widget());
201     
202       LOG(INFO) << "Create Window: " << size.width() << "x" << size.height();
203     
204       gfx::Rect bounds(0, 0, size.width(), size.height());
205       window_->AddObserver(this);
206     
207       views::Widget::InitParams params;
208       params.ownership = views::Widget::InitParams::WIDGET_OWNS_NATIVE_WIDGET;
209       params.bounds = bounds;
210       params.delegate = this;
211       params.type = views::Widget::InitParams::TYPE_WINDOW;
212       params.remove_standard_frame = !has_frame_;
213     
214     #if defined(USE_X11)
215       // Set WM_WINDOW_ROLE.
216       params.wm_role_name = base::StringPrintf(
217           "%s/%s/%d", "Thrust", 
218           ThrustShellBrowserClient::Get()->GetAppName().c_str(),
219           ++kWindowsCreated);
220       // Set WM_CLASS.
221       params.wm_class_name = "thrust";
222       params.wm_class_class = "Thrust";
223     #endif
224     
225       window_->Init(params);
226     
227     #if defined(USE_X11)
228       // Set _GTK_THEME_VARIANT to dark if we have "dark-theme" option set.
229       bool use_dark_theme = false;
230       /* TODO(spolu): Add option */
231       /*
232       if (options.Get(switches::kDarkTheme, &use_dark_theme) && use_dark_theme) {
233         XDisplay* xdisplay = gfx::GetXDisplay();
234         XChangeProperty(xdisplay, GetAcceleratedWidget(),
235                         XInternAtom(xdisplay, "_GTK_THEME_VARIANT", False),
236                         XInternAtom(xdisplay, "UTF8_STRING", False),
237                         8, PropModeReplace,
238                         reinterpret_cast<const unsigned char*>("dark"),
239                         4);
240       }
241       */
242     
243       // Before the window is mapped the SetWMSpecState can not work, so we have
244       // to manually set the _NET_WM_STATE.
245       bool skip_taskbar = false;
246       /* TODO(spolu): Add option */
247       /*
248       if (options.Get(switches::kSkipTaskbar, &skip_taskbar) && skip_taskbar) {
249         std::vector<::Atom> state_atom_list;
250         state_atom_list.push_back(GetAtom("_NET_WM_STATE_SKIP_TASKBAR"));
251         ui::SetAtomArrayProperty(GetAcceleratedWidget(), "_NET_WM_STATE", "ATOM",
252                                  state_atom_list);
253       }
254       */
255     #endif
256     
257       // Add web view.
258       SetLayoutManager(new views::FillLayout());
259       set_background(views::Background::CreateStandardPanelBackground());
260       AddChildView(inspectable_web_contents()->GetView()->GetView());
261     
262     #if defined(OS_WIN)
263         if (!has_frame_) {
264           /* Set Window style so that we get a minimize and maximize animation */
265           /* when frameless. */
266           DWORD frame_style = WS_THICKFRAME | WS_MINIMIZEBOX | 
267                               WS_MAXIMIZEBOX | WS_CAPTION;
268           ::SetWindowLong(
269               GetNativeWindow()->GetHost()->GetAcceleratedWidget(),
270               GWL_STYLE, frame_style);
271         }
272     #endif
273     
274       window_->UpdateWindowIcon();
275       window_->CenterWindow(bounds.size());
276       Layout();
277     
278     #if defined(USE_X11)
279       if(ThrustMenu::GetApplicationMenu() != NULL) {
280         ThrustWindow::AttachMenu(ThrustMenu::GetApplicationMenu()->model_.get());
281       }
282     #endif
283     }

406     ThrustWindow::PlatformSetContentSize(
407         int width, int height)
408     {
409       if(!has_frame_) {
410         PlatformResize(width, height);
411         return;
412       }
413     
414       gfx::Size size(width, height);
415       gfx::Rect bounds = window_->GetWindowBoundsInScreen();

417           ContentBoundsToWindowBounds(gfx::Rect(bounds.origin(), size)).size();

449     void 
450     ThrustWindow::PlatformUpdateDraggableRegions(
451         const std::vector<DraggableRegion>& regions)
452     {
453       if(has_frame_) {
454         return;
455       }
456     
457       SkRegion* draggable_region = new SkRegion;
458     
459       // By default, the whole window is non-draggable. We need to explicitly
460       // include those draggable regions.
461       for (std::vector<DraggableRegion>::const_iterator iter = regions.begin();
462            iter != regions.end(); ++iter) {
463         const DraggableRegion& region = *iter;
464         draggable_region->op(
465             region.bounds.x(),
466             region.bounds.y(),
467             region.bounds.right(),
468             region.bounds.bottom(),
469             region.draggable ? SkRegion::kUnion_Op : SkRegion::kDifference_Op);
470       }
471     
472       draggable_region_.reset(draggable_region);
473     }

480     ThrustWindow::OnWidgetActivationChanged(
481         views::Widget* widget, 
482         bool active) 
483     {
484       if(widget != window_.get())
485         return;
486     
487       if(active) {
488         binding_->EmitFocus();
489       }
490       else {
491         binding_->EmitBlur();
492       }
493     
494       if(active && web_contents()) {
495         web_contents()->Focus();
496       }
497     }

569     bool 
570     ThrustWindow::ShouldDescendIntoChildForEventHandling(
571         gfx::NativeView child,
572         const gfx::Point& location) 
573     {
574       // App window should claim mouse events that fall within the draggable region.
575       if (draggable_region_ &&
576           draggable_region_->contains(location.x(), location.y()))
577         return false;
578     
579       // And the events on border for dragging resizable frameless window.
580       if(!has_frame_) {
581         FramelessView* frame = static_cast<FramelessView*>(
582             window_->non_client_view()->frame_view());
583         return frame->ResizingBorderHitTest(location) == HTNOWHERE;
584       }
585       return true;
586     }

595     views::NonClientFrameView* 
596     ThrustWindow::CreateNonClientFrameView(
597         views::Widget* widget) 
598     {
599     #if defined(OS_WIN)
600       if(ui::win::IsAeroGlassEnabled()) {
601         WinFrameView* frame_view =  new WinFrameView;
602         frame_view->Init(this, widget);
603         return frame_view;
604       }
605     #elif defined(OS_LINUX)
606       if(has_frame_) {
607         return new views::NativeFrameView(widget);
608       } 
609       else {
610         FramelessView* frame_view =  new FramelessView();
611         frame_view->Init(this, widget);
612         return frame_view;
613       }
614     #endif
615       return NULL;
616     }
