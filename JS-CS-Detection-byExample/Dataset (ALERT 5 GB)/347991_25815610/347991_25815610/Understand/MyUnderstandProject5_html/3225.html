81     void 
82     ThrustShellRendererClient::RenderThreadStarted() 
83     {
84       observer_.reset(new ThrustShellRenderProcessObserver());
85       visited_link_slave_.reset(new visitedlink::VisitedLinkSlave());
86     #if defined(OS_MACOSX)
87       // We need to call this once before the sandbox was initialized to cache the
88       // value.
89       base::debug::BeingDebugged();
90     #endif
91     
92       RenderThread* thread = RenderThread::Get();
93     
94       thread->RegisterExtension(SafeBuiltins::CreateV8Extension());
95     
96     #include "./extensions/resources/web_view.js.bin"
97       std::string web_view_src(
98           (char*)src_renderer_extensions_resources_web_view_js,
99           src_renderer_extensions_resources_web_view_js_len);
100       source_map_.RegisterSource("webview", web_view_src);
101     
102       thread->AddObserver(observer_.get());
103       thread->AddObserver(visited_link_slave_.get());
104     }

131     void 
132     ThrustShellRendererClient::DidCreateScriptContext(
133         blink::WebFrame* frame, 
134         v8::Handle<v8::Context> v8_context, 
135         int extension_group,
136         int world_id) 
137     {
138       ScriptContext* context = new ScriptContext(v8_context, frame);
139       {
140         scoped_ptr<ModuleSystem> module_system(new ModuleSystem(context,
141                                                                 &source_map_));
142         context->set_module_system(module_system.Pass());
143       }
144       ModuleSystem* module_system = context->module_system();
145     
146       // Enable natives in startup.
147       ModuleSystem::NativesEnabledScope natives_enabled_scope(module_system);
148     
149       /* NOTE: please use the naming convention "foo_natives" for these. */
150       module_system->RegisterNativeHandler("document_natives",
151           scoped_ptr<NativeHandler>(
152               new DocumentBindings(context)));
153       module_system->RegisterNativeHandler("webview_natives",
154           scoped_ptr<NativeHandler>(
155               new WebViewBindings(context)));
156     
157       module_system->Require("webview");
158       LOG(INFO) << "Module requires called!";
159     }
