32     void 
33     MenuDelegate::RunMenu(
34         ui::MenuModel* model, 
35         views::MenuButton* button) 
36     {
37       gfx::Point screen_loc;
38       views::View::ConvertPointToScreen(button, &screen_loc);
39       // Subtract 1 from the height to make the popup flush with the button border.
40       gfx::Rect bounds(screen_loc.x(), screen_loc.y(), button->width(),
41                        button->height() - 1);
42     
43       id_ = button->tag();
44       views::MenuItemView* item = BuildMenu(model);
45     
46       views::MenuRunner menu_runner(
47           item,
48           views::MenuRunner::CONTEXT_MENU | views::MenuRunner::HAS_MNEMONICS);
49       ignore_result(menu_runner.RunMenuAt(
50           button->GetWidget()->GetTopLevelWidget(),
51           button,
52           bounds,
53           views::MENU_ANCHOR_TOPRIGHT,
54           ui::MENU_SOURCE_MOUSE));
55     }
56     
57     views::MenuItemView* 
58     MenuDelegate::BuildMenu(
59         ui::MenuModel* model) 
60     {
61       DCHECK_GE(id_, 0);
62     
63       if (!items_[id_]) {
64         views::MenuModelAdapter* delegate = new views::MenuModelAdapter(model);
65         delegates_[id_] = delegate;
66     
67         views::MenuItemView* item = new views::MenuItemView(this);
68         delegate->BuildMenu(item);
69         items_[id_] = item;
70       }
71     
72       return items_[id_];
73     }

152     views::MenuItemView* 
153     MenuDelegate::GetSiblingMenu(
154         views::MenuItemView* menu,
155         const gfx::Point& screen_point,
156         views::MenuAnchorPosition* anchor,
157         bool* has_mnemonics,
158         views::MenuButton** button) 
159     {
160       ui::MenuModel* model;
161       if (!menu_bar_->GetMenuButtonFromScreenPoint(screen_point, &model, button))
162         return NULL;
163     
164       *anchor = views::MENU_ANCHOR_TOPLEFT;
165       *has_mnemonics = true;
166     
167       id_ = (*button)->tag();
168       return BuildMenu(model);
169     }
