45       //
46       // Override application version.
47       static void OverrideAppVersion(const std::string& name);

65           int child_process_id) OVERRIDE;

71       virtual void ResourceDispatcherHostCreated() OVERRIDE;

93       virtual bool IsHandledURL(const GURL& url) OVERRIDE;

107       // ### UnRegisterThrustSession
108       // ```
109       // @session {ThrustSession} the session to unregister
110       // ```
111       void UnRegisterThrustSession(ThrustSession* session);
112     
113       // ### ThrustSessionFroBrowserContext
114       // ```
115       // @content_browser_context {BrowserContext}
