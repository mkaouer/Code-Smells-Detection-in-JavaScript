51     void InstallProtocolHandlers(net::URLRequestJobFactoryImpl* job_factory,
52                                  ProtocolHandlerMap* protocol_handlers) {
53       for (ProtocolHandlerMap::iterator it =
54                protocol_handlers->begin();
55            it != protocol_handlers->end();
56            ++it) {
57         bool set_protocol = job_factory->SetProtocolHandler(
58             it->first, it->second.release());
59         DCHECK(set_protocol);
60       }
61       protocol_handlers->clear();
62     }

66     ThrustShellURLRequestContextGetter::ThrustShellURLRequestContextGetter(
67         ThrustSession* parent,
68         bool ignore_certificate_errors,
69         const base::FilePath& base_path,
70         base::MessageLoop* io_loop,
71         base::MessageLoop* file_loop,
72         ProtocolHandlerMap* protocol_handlers,
73         URLRequestInterceptorScopedVector request_interceptors,
74         net::NetLog* net_log)
75         : parent_(parent),
76           ignore_certificate_errors_(ignore_certificate_errors),
77           base_path_(base_path),
78           io_loop_(io_loop),
79           file_loop_(file_loop),
80           net_log_(net_log),
81           request_interceptors_(request_interceptors.Pass())
82     {
83       // Must first be created on the UI thread.
84       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::UI));
85       std::swap(protocol_handlers_, *protocol_handlers);
86       
87       // We must create the proxy config service on the UI loop on Linux because it
88       // must synchronously run on the glib message loop. This will be passed to
89       // the URLRequestContextStorage on the IO thread in GetURLRequestContext().
90       proxy_config_service_.reset(
91           net::ProxyService::CreateSystemProxyConfigService(
92             io_loop_->message_loop_proxy().get(), file_loop_));
93     }

99     net::URLRequestContext* 
100     ThrustShellURLRequestContextGetter::GetURLRequestContext() 
101     {
102       DCHECK(BrowserThread::CurrentlyOn(BrowserThread::IO));
103     
104       if (!url_request_context_) {
105         url_request_context_.reset(new net::URLRequestContext());
106         url_request_context_->set_net_log(net_log_);
107         network_delegate_.reset(new ThrustShellNetworkDelegate);
108         url_request_context_->set_network_delegate(network_delegate_.get());
109         storage_.reset(
110             new net::URLRequestContextStorage(url_request_context_.get()));
111     
112         scoped_refptr<net::CookieStore> cookie_store = NULL;
113         if(parent_) {
114           cookie_store = new net::CookieMonster(parent_->GetCookieStore(), NULL);
115         }
116         else {
117           cookie_store = new net::CookieMonster(NULL, NULL);
118         }
119         //cookie_store->GetCookieMonster()->SetPersistSessionCookies(false);
120         storage_->set_cookie_store(cookie_store);
121     
122         /*
123         const char* schemes[] = {"http", "https", "file", "app"};
124         cookie_store->GetCookieMonster()->SetCookieableSchemes(schemes, 4);
125         */
126               
127         storage_->set_server_bound_cert_service(new net::ServerBoundCertService(
128             new net::DefaultServerBoundCertStore(NULL),
129             base::WorkerPool::GetTaskRunner(true)));
130         storage_->set_http_user_agent_settings(
131             new net::StaticHttpUserAgentSettings("en-us,en", std::string()));
132     
133         scoped_ptr<net::HostResolver> host_resolver(
134             net::HostResolver::CreateDefaultResolver(
135                 url_request_context_->net_log()));
136     
137         storage_->set_cert_verifier(net::CertVerifier::CreateDefault());
138         storage_->set_transport_security_state(new net::TransportSecurityState);
139     
140         /* TODO(spolu): use v8 if possible, look at chrome code. */
141         storage_->set_proxy_service(
142             net::ProxyService::CreateUsingSystemProxyResolver(
143               proxy_config_service_.release(),
144               0,
145               url_request_context_->net_log()));
146     
147         storage_->set_ssl_config_service(new net::SSLConfigServiceDefaults);
148         storage_->set_http_auth_handler_factory(
149             net::HttpAuthHandlerFactory::CreateDefault(host_resolver.get()));
150         storage_->set_http_server_properties(
151             scoped_ptr<net::HttpServerProperties>(
152                 new net::HttpServerPropertiesImpl()));
153     
154     
155         net::HttpCache::BackendFactory* main_backend = NULL;
156         if(parent_->IsOffTheRecord()) {
157           main_backend = net::HttpCache::DefaultBackend::InMemory(0);
158         }
159         else {
160           base::FilePath cache_path = base_path_.Append(FILE_PATH_LITERAL("Cache"));
161           main_backend =
162             new net::HttpCache::DefaultBackend(
163                 net::DISK_CACHE,
164                 net::CACHE_BACKEND_DEFAULT,
165                 cache_path,
166                 0,
167                 BrowserThread::GetMessageLoopProxyForThread(BrowserThread::CACHE)
168                     .get());
169         }
170     
171         net::HttpNetworkSession::Params network_session_params;
172         network_session_params.cert_verifier =
173             url_request_context_->cert_verifier();
174         network_session_params.transport_security_state =
175             url_request_context_->transport_security_state();
176         network_session_params.server_bound_cert_service =
177             url_request_context_->server_bound_cert_service();
178         network_session_params.proxy_service =
179             url_request_context_->proxy_service();
180         network_session_params.ssl_config_service =
181             url_request_context_->ssl_config_service();
182         network_session_params.http_auth_handler_factory =
183             url_request_context_->http_auth_handler_factory();
184         network_session_params.network_delegate =
185             network_delegate_.get();
186         network_session_params.http_server_properties =
187             url_request_context_->http_server_properties();
188         network_session_params.net_log =
189             url_request_context_->net_log();
190         network_session_params.ignore_certificate_errors =
191             ignore_certificate_errors_;
192     
193         // Give |storage_| ownership at the end in case it's |mapped_host_resolver|.
194         storage_->set_host_resolver(host_resolver.Pass());
195         network_session_params.host_resolver =
196             url_request_context_->host_resolver();
197     
198         net::HttpCache* main_cache = new net::HttpCache(
199             network_session_params, main_backend);
200         storage_->set_http_transaction_factory(main_cache);
201     
202         scoped_ptr<net::URLRequestJobFactoryImpl> job_factory(
203             new net::URLRequestJobFactoryImpl());
204         // Keep ProtocolHandlers added in sync with
205         // ThrustShellContentBrowserClient::IsHandledURL().
206         InstallProtocolHandlers(job_factory.get(), &protocol_handlers_);
207         bool set_protocol = job_factory->SetProtocolHandler(
208             url::kDataScheme, new net::DataProtocolHandler);
209         DCHECK(set_protocol);
210         set_protocol = job_factory->SetProtocolHandler(
211             url::kFileScheme,
212             new net::FileProtocolHandler(
213                 content::BrowserThread::GetBlockingPool()->
214                     GetTaskRunnerWithShutdownBehavior(
215                         base::SequencedWorkerPool::SKIP_ON_SHUTDOWN)));
216         DCHECK(set_protocol);
217     
218         // Set up interceptors in the reverse order.
219         scoped_ptr<net::URLRequestJobFactory> top_job_factory =
220             job_factory.PassAs<net::URLRequestJobFactory>();
221         for (URLRequestInterceptorScopedVector::reverse_iterator i =
222                  request_interceptors_.rbegin();
223              i != request_interceptors_.rend();
224              ++i) {
225           top_job_factory.reset(new net::URLRequestInterceptingJobFactory(
226               top_job_factory.Pass(), make_scoped_ptr(*i)));
227         }
228         request_interceptors_.weak_clear();
229     
230         storage_->set_job_factory(top_job_factory.release());
231       }
232     
233       return url_request_context_.get();
234     }
