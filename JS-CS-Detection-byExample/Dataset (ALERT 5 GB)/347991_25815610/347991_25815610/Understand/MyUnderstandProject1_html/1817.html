30     // Chrome's wrapper for a v8 context.
31     class Context {
32      public:
33        Context(v8::Handle<v8::Context> context,
34                blink::WebFrame* frame);
35       virtual ~Context();
36     
37       // Clears the WebFrame for this contexts and invalidates the associated
38       // ModuleSystem.
39       void Invalidate();
40     
41       // Returns true if this context is still valid, false if it isn't.
42       // A context becomes invalid via Invalidate().
43       bool is_valid() const {
44         return !v8_context_.IsEmpty();
45       }
46     
47       v8::Handle<v8::Context> v8_context() const {
48         return v8_context_.NewHandle(v8::Isolate::GetCurrent());
49       }
50     
51       blink::WebFrame* web_frame() const {
52         return web_frame_;
53       }
54     
55       void set_module_system(scoped_ptr<ModuleSystem> module_system) {
56         module_system_ = module_system.Pass();
57       }
58     
59       ModuleSystem* module_system() { return module_system_.get(); }
60     
61       SafeBuiltins* safe_builtins() {
62         return &safe_builtins_;
63       }
64       const SafeBuiltins* safe_builtins() const {
65         return &safe_builtins_;
66       }
67     
68       // Returns the RenderView associated with this context. Can return NULL if the
69       // context is in the process of being destroyed.
70       content::RenderView* GetRenderView() const;
71     
72       // Get the URL of this context's web frame.
73       GURL GetURL() const;
74     
75       // Runs |function| with appropriate scopes. Doesn't catch exceptions, callers
76       // must do that if they want.
77       //
78       // USE THIS METHOD RATHER THAN v8::Function::Call WHEREVER POSSIBLE.
79       v8::Local<v8::Value> CallFunction(v8::Handle<v8::Function> function,
80                                         int argc,
81                                         v8::Handle<v8::Value> argv[]) const;
82     
83       v8::Isolate* isolate() const {
84         return isolate_;
85       }
86     
87      private:
88       // The v8 context the bindings are accessible to.
89       ScopedPersistent<v8::Context> v8_context_;
90     
91       // The WebFrame associated with this context. This can be NULL because this
92       // object can outlive is destroyed asynchronously.
93       blink::WebFrame* web_frame_;
94     
95       // Owns and structures the JS that is injected to set up extension bindings.
96       scoped_ptr<ModuleSystem> module_system_;
97     
98       // Contains safe copies of builtin objects like Function.prototype.
99       SafeBuiltins safe_builtins_;
100     
101       v8::Isolate* isolate_;
102     
103       DISALLOW_COPY_AND_ASSIGN(Context);
104     };
105     
106     }  // namespace extensions
