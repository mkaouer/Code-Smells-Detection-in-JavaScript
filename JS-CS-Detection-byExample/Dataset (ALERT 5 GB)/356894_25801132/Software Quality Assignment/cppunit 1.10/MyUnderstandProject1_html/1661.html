50     // Paint the background of the progress bar region
51     void 
52     ProgressBar::paintBackground( CDC &dc )
53     {
54       CBrush brshBackground;
55       CPen penGray( PS_SOLID, 1, RGB (128, 128, 128) );
56       CPen penWhite( PS_SOLID, 1, RGB (255, 255, 255) );
57     
58       VERIFY( brshBackground.CreateSolidBrush( ::GetSysColor (COLOR_BTNFACE) ) );
59     
60       dc.FillRect( m_bounds, &brshBackground );
61       
62       CPen *pOldPen = dc.SelectObject( &penGray );
63       int xRight = m_bounds.left + m_bounds.Width() -1;
64       int yBottom = m_bounds.top + m_bounds.Height() -1;
65       {
66         dc.MoveTo( m_bounds.left, m_bounds.top );
67         dc.LineTo( xRight, m_bounds.top );
68     
69         dc.MoveTo( m_bounds.left, m_bounds.top );
70         dc.LineTo( m_bounds.left, yBottom );
71       }
72     
73       dc.SelectObject( &penWhite );
74       {
75         dc.MoveTo( xRight, m_bounds.top );
76         dc.LineTo( xRight, yBottom );
77     
78         dc.MoveTo( m_bounds.left, yBottom );
79         dc.LineTo( xRight, yBottom );
80       }
81     
82       dc.SelectObject( pOldPen );
83     }

86     // Paint the actual status of the progress bar
87     void 
88     ProgressBar::paintStatus( CDC &dc )
89     {
90       if ( m_progress <= 0 )
91         return;
92     
93       CBrush brshStatus;
94       CRect rect( m_bounds.left, m_bounds.top, 
95                   m_bounds.left + m_progressX, m_bounds.bottom );
96     
97       COLORREF statusColor = getStatusColor();
98     
99       VERIFY( brshStatus.CreateSolidBrush( statusColor ) );
100     
101       rect.DeflateRect( 1, 1 );
102       dc.FillRect( rect, &brshStatus );
103     }

126     // Take one step, indicating whether it was a successful step
127     void 
128     ProgressBar::step( bool successful )
129     {
130       m_progress++;
131     
132       int x = m_progressX;
133     
134       m_progressX = scale (m_progress);
135     
136       if ( !m_error  &&  !successful )
137       {
138         m_error = true;
139         x = 1;
140       }
141     
142       paintStep( x, m_progressX );
143     }

146     // Map from steps to display units
147     int 
148     ProgressBar::scale( int value )
149     {
150       if ( m_total > 0 )
151           return max( 1, value * (m_bounds.Width() - 1) / m_total );
152     
153       return value;
154     }

157     // Reset the progress bar
158     void 
159     ProgressBar::reset()
160     {
161       m_progressX = 1;
162       m_progress = 0;
163       m_error = false;
164     
165       RedrawWindow( m_bounds );
166       UpdateWindow( );
167     }
