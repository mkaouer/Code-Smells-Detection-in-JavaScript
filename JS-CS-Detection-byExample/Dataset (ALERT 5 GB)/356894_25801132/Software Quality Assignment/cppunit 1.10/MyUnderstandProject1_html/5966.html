29     {

31     
32       std::string::const_iterator itStart = text.begin();
33       while ( !text.empty() )
34       {
35         std::string::const_iterator itSeparator = std::find( itStart, 
36                                                              text.end(), 
37                                                              separator );
38         splittedText.push_back( text.substr( itStart - text.begin(),
39                                              itSeparator - itStart ) );

43       }

50     StringTools::wrap( const std::string &text,
51                        int wrapColumn )
52     {
53       const char lineBreak = '\n';
54       Strings lines = split( text, lineBreak );
55     
56       std::string wrapped;
57       for ( Strings::const_iterator it = lines.begin(); it != lines.end(); ++it )
58       {
59         if ( it != lines.begin() )
60           wrapped += lineBreak;
61     
62         const std::string &line = *it;
63         unsigned int index =0;
64         while ( index < line.length() )
65         {
66           std::string lineSlice( line.substr( index, wrapColumn ) );
67           wrapped += lineSlice;
68           index += wrapColumn;
69           if ( index < line.length() )
70             wrapped += lineBreak;
71         }
72       }
73     
74       return wrapped;
