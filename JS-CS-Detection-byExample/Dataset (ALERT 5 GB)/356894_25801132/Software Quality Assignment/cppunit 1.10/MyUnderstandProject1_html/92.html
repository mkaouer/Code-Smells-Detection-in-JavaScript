16     QtTestRunner::QtTestRunner() :
17       _suite( new CPPUNIT_NS::TestSuite( "All Tests" ) ),
18       _tests( new Tests() )
19     {
20     }
21     
22     
23     QtTestRunner::~QtTestRunner()
24     {
25       delete _suite;
26     
27       Tests::iterator it = _tests->begin();
28       while ( it != _tests->end() )
29         delete *it++;
30     
31       delete _tests;
32     }
33     
34     
35     Test *
36     QtTestRunner::getRootTest()
37     {
38       if ( _tests->size() != 1 )
39       {
40         Tests::iterator it = _tests->begin();
41         while ( it != _tests->end() )
42           _suite->addTest( *it++ );
43         _tests->clear();
44         return _suite;
45       }
46       return (*_tests)[0];
47     }

50     void 
51     QtTestRunner::run( bool autoRun )
52     {
53       TestRunnerDlg *dlg = new TestRunnerDlg( qApp->mainWidget(), 
54                                               "QtTestRunner", 
55                                               TRUE );
56       dlg->setModel( new TestRunnerModel( getRootTest() ),
57                      autoRun );
58       dlg->exec();
59       delete dlg;
60     }

63     void 
64     QtTestRunner::addTest( CPPUNIT_NS::Test *test )
65     {
66       _tests->push_back( test );
67     }
