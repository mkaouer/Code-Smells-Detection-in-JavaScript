22     void 
23     MostRecentTests::setLastTestRun( CPPUNIT_NS::Test *test )
24     {
25       for ( TestRuns::iterator it = m_runs.begin(); it != m_runs.end(); ++it )
26       {
27         if ( it->second == test )
28         {
29           m_runs.erase( it );
30           break;
31         }
32       }
33       
34       if ( test != NULL )
35         m_runs.push_front( TestRun( test->getName(), test ) );
36     }
