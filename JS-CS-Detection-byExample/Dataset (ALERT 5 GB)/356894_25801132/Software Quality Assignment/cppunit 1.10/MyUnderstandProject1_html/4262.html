46     void 
47     TestBrowser::insertItemFor( CPPUNIT_NS::Test *test,
48                                 QListViewItem *parentItem )
49     {
50       QListViewItem *item = new TestListViewItem( test, parentItem );
51       QString testName = QString::fromLatin1( test->getName().c_str() );
52       item->setText( 0, testName );
53       if ( test->getChildTestCount() > 0  ||    // suite with test
54            test->countTestCases() == 0 )        // empty suite
55       {
56         for ( int index =0; index < test->getChildTestCount(); ++index )
57           insertItemFor( test->getChildTestAt( index ), item );
58       }
59     }
