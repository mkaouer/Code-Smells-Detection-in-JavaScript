40     // Spawn a thread to a test
41     void 
42     ActiveTest::run( CPPUNIT_NS::TestResult *result )
43     {
44       CWinThread *thread;
45       
46       setTestResult( result );
47       m_runCompleted.ResetEvent();
48     
49       thread = ::AfxBeginThread( threadFunction, 
50                                  this, 
51                                  THREAD_PRIORITY_NORMAL, 
52                                  0, 
53                                  CREATE_SUSPENDED);
54       
55       ::DuplicateHandle( GetCurrentProcess(), 
56                          thread->m_hThread,
57                          GetCurrentProcess(), 
58                          &m_threadHandle, 
59                          0, 
60                          FALSE, 
61                          DUPLICATE_SAME_ACCESS );
62     
63       thread->ResumeThread ();
64     }

67     // Simple execution thread.  Assuming that an ActiveTest instance
68     // only creates one of these at a time.
69     UINT 
70     ActiveTest::threadFunction( LPVOID thisInstance )
71     {
72       ActiveTest *test = (ActiveTest *)thisInstance;
73     
74       test->run ();
75     
76       ::CloseHandle( test->m_threadHandle );
77       test->m_threadHandle = INVALID_HANDLE_VALUE;
78     
79       test->m_runCompleted.SetEvent();
80     
81       return 0;
82     }
