69     void 
70     TestPlugIn::reloadDll()
71     {
72       m_manager.unload( m_copyFileName );
73       makeDllCopy();
74       loadDll();
75     }

77     
78     void 
79     TestPlugIn::makeDllCopy()
80     {
81       if ( ::CopyFile( m_fileName.c_str(), m_copyFileName.c_str(), FALSE ) == FALSE )
82       {
83         throw TestPlugInException( "Failed to copy DLL" + m_fileName +
84             " to " + m_copyFileName, TestPlugInException::failedToCopyDll );
85       }
86     }
87     
88     
89     void 
90     TestPlugIn::loadDll()
91     {
92       try
93       {
94         m_manager.load( m_copyFileName );
95       }
96       catch ( CPPUNIT_NS::DynamicLibraryManagerException &e )
97       {
98         throw TestPlugInException( e.what(), 
99                                    TestPlugInException::failedToLoadDll );
100       }
101     }
