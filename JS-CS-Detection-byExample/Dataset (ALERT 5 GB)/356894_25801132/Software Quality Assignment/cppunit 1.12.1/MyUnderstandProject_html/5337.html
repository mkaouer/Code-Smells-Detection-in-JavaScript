46     // Spawn a thread to a test
47     void 
48     ActiveTest::run( CPPUNIT_NS::TestResult *result )
49     {
50       CWinThread *thread;
51       
52       setTestResult( result );
53       m_runCompleted.ResetEvent();
54     
55       thread = ::AfxBeginThread( threadFunction, 
56                                  this, 
57                                  THREAD_PRIORITY_NORMAL, 
58                                  0, 
59                                  CREATE_SUSPENDED);
60       
61       ::DuplicateHandle( GetCurrentProcess(), 
62                          thread->m_hThread,
63                          GetCurrentProcess(), 
64                          &m_threadHandle, 
65                          0, 
66                          FALSE, 
67                          DUPLICATE_SAME_ACCESS );
68     
69       thread->ResumeThread ();
70     }

73     // Simple execution thread.  Assuming that an ActiveTest instance
74     // only creates one of these at a time.
75     UINT 
76     ActiveTest::threadFunction( LPVOID thisInstance )
77     {
78       ActiveTest *test = (ActiveTest *)thisInstance;
79     
80       test->run ();
81     
82       ::CloseHandle( test->m_threadHandle );
83       test->m_threadHandle = INVALID_HANDLE_VALUE;
84     
85       test->m_runCompleted.SetEvent();
86     
87       return 0;
88     }
