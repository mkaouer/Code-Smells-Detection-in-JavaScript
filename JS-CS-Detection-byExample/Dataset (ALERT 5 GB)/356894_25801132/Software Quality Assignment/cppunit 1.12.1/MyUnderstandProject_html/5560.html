26     /*! Runs the specified tests located in the root suite.
27      * \param parser Command line parser.
28      * \return \c true if the run succeed, \c false if a test failed or if a test
29      *         path was not resolved.
30      */
31     bool 
32     runTests( const CommandLineParser &parser )
33     {
34       bool wasSuccessful = false;
35       CPPUNIT_NS::PlugInManager plugInManager;
36     
37       // The following scope is used to explicitely free all memory allocated before
38       // unload the test plug-ins (uppon plugInManager destruction).
39       {
40         CPPUNIT_NS::TestResult controller;
41         CPPUNIT_NS::TestResultCollector result;
42         controller.addListener( &result );        
43     
44         // Set up outputters
45         CPPUNIT_NS::OStream *stream = &CPPUNIT_NS::stdCErr();
46         if ( parser.useCoutStream() )
47           stream = &CPPUNIT_NS::stdCOut();
48     
49         CPPUNIT_NS::OStream *xmlStream = stream;
50         if ( !parser.getXmlFileName().empty() )
51           xmlStream = new CPPUNIT_NS::OFileStream( parser.getXmlFileName().c_str() );
52     
53         CPPUNIT_NS::XmlOutputter xmlOutputter( &result, *xmlStream, parser.getEncoding() );
54         xmlOutputter.setStyleSheet( parser.getXmlStyleSheet() );
55         CPPUNIT_NS::TextOutputter textOutputter( &result, *stream );
56         CPPUNIT_NS::CompilerOutputter compilerOutputter( &result, *stream );
57     
58         // Set up test listeners
59         CPPUNIT_NS::BriefTestProgressListener briefListener;
60         CPPUNIT_NS::TextTestProgressListener dotListener;
61         if ( parser.useBriefTestProgress() )
62           controller.addListener( &briefListener );
63         else if ( !parser.noTestProgress() )
64           controller.addListener( &dotListener );
65     
66         // Set up plug-ins
67         for ( int index =0; index < parser.getPlugInCount(); ++index )
68         {
69           CommandLinePlugInInfo plugIn = parser.getPlugInAt( index );
70           plugInManager.load( plugIn.m_fileName, plugIn.m_parameters );
71         }
72     
73         // Registers plug-in specific TestListener (global setUp/tearDown, custom TestListener...)
74         plugInManager.addListener( &controller );
75     
76         // Adds the default registry suite
77         CPPUNIT_NS::TestRunner runner;
78         runner.addTest( CPPUNIT_NS::TestFactoryRegistry::getRegistry().makeTest() );
79     
80         // Runs the specified test
81         try
82         {
83           runner.run( controller, parser.getTestPath() );
84           wasSuccessful = result.wasSuccessful();
85         }
86         catch ( std::invalid_argument & )
87         {
88           CPPUNIT_NS::stdCOut()  <<  "Failed to resolve test path: "  
89                                  <<  parser.getTestPath() 
90                                  <<  "\n";
91         }
92     
93         // Removes plug-in specific TestListener (not really needed but...)
94         plugInManager.removeListener( &controller );
95     
96         // write using outputters
97         if ( parser.useCompilerOutputter() )
98           compilerOutputter.write();
99     
100         if ( parser.useTextOutputter() )
101           textOutputter.write();
102     
103         if ( parser.useXmlOutputter() )
104         {
105           plugInManager.addXmlOutputterHooks( &xmlOutputter );
106           xmlOutputter.write();
107           plugInManager.removeXmlOutputterHooks();
108         }
109     
110         if ( !parser.getXmlFileName().empty() )
111           delete xmlStream;
112       }
113     
114       return wasSuccessful;
115     }

178     /*! Main
179      * 
180      * Usage: 
181      *
182      * DllPlugInTester.exe dll-filename1 [dll-filename2 [dll-filename3 ...]] [:testpath]
183      *
184      * <em>dll-filename</em> must be the name of the DLL. If the DLL use some other DLL, they
185      * should be in the path or in the same directory as the DLL. The DLL must export
186      * a function named "GetTestPlugInInterface" with the signature
187      * GetTestPlugInInterfaceFunction. Both are defined in:
188      * \code
189      * #include <msvc6/testrunner/TestPlugInInterface.h>
190      * \endcode.
191      *
192      * See examples/msvc6/TestPlugIn for an example of post-build testing.
193      *
194      * If no test path is specified, they all the test of the suite returned by the DLL
195      * are run. If a test path is specified, then only the specified test is run. The test
196      * path must be prefixed by ':'.
197      *
198      * Test paths are resolved using Test::resolveTestPath() on the suite returned by
199      * TestFactoryRegistry::getRegistry().makeTest();
200      *
201      * If all test succeed and no error happen then the application exit with code 0.
202      * If any error occurs (failed to load dll, failed to resolve test paths) or a 
203      * test fail, the application exit with code 1. If the application failed to
204      * parse the command line, it exits with code 2.
205      */
206     int 
207     main( int argc, 
208           const char *argv[] )
209     {
210       const int successReturnCode = 0;
211       const int failureReturnCode = 1;
212       const int badCommadLineReturnCode = 2;
213     
214       // check command line
215       std::string applicationName( argv[0] );
216       if ( argc < 2 )
217       {
218         printUsage( applicationName );
219         return badCommadLineReturnCode;
220       }
221     
222       CommandLineParser parser( argc, argv );
223       try
224       {
225         parser.parse();
226       }
227       catch ( CommandLineParserException &e )
228       {
229         CPPUNIT_NS::stdCOut()  <<  "Error while parsing command line: "  <<  e.what()  
230                                << "\n\n";
231         printShortUsage( applicationName );
232         return badCommadLineReturnCode;
233       }
234     
235       bool wasSuccessful = false;
236       try
237       {
238         wasSuccessful = runTests( parser );
239       }
240       catch ( CPPUNIT_NS::DynamicLibraryManagerException &e )
241       {
242         CPPUNIT_NS::stdCOut()  << "Failed to load test plug-in:\n"
243                                << e.what() << "\n";
244       }
245     
246     #if !defined( CPPUNIT_NO_STREAM )
247       if ( parser.waitBeforeExit() )
248       {
249         CPPUNIT_NS::stdCOut() << "Please press <RETURN> to exit\n";
250         CPPUNIT_NS::stdCOut().flush();
251         std::cin.get();
252       }
253     #endif
254     
255       return wasSuccessful ? successReturnCode : failureReturnCode;
256     }
