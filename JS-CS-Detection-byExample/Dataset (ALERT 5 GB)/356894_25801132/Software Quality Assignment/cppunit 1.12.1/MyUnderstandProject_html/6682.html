83        static void goToLineInSourceCode( CString fileName, int line )
84        {
85           USES_CONVERSION;
86     
87           CComPtr< IRunningObjectTable > pIRunningObjectTable;
88           HRESULT hr = ::GetRunningObjectTable( 0, &pIRunningObjectTable );
89     
90           CComPtr< IEnumMoniker > pIEnumMoniker;
91           hr = pIRunningObjectTable->EnumRunning( &pIEnumMoniker );
92     
93           CComPtr< EnvDTE::_DTE > pIEnvDTE;
94           while ( true )
95           {
96              ULONG celtFetched;
97              CComPtr< IMoniker > pIMoniker;
98              if ( S_OK != pIEnumMoniker->Next( 1, &pIMoniker, &celtFetched ) )
99                 break;
100     
101              CComPtr< IBindCtx > pIBindCtx; 
102              hr = ::CreateBindCtx( NULL, &pIBindCtx ); 
103     
104              LPOLESTR pszDisplayName;
105              pIMoniker->GetDisplayName( pIBindCtx, NULL, &pszDisplayName );
106     
107              TRACE( "Moniker %s\n", W2A( pszDisplayName ) );
108     
109              CString strDisplayName( pszDisplayName );
110              CComPtr< IMalloc > pIMalloc;
111              ::CoGetMalloc( 1, &pIMalloc );
112              pIMalloc->Free( pszDisplayName );
113     
114              if ( strDisplayName.Right( 4 ) == _T(".sln") 
115                   || strDisplayName.Find( _T("VisualStudio.DTE") ) >= 0 )
116              {
117                 CComPtr< IUnknown > pIUnknown;
118                 pIRunningObjectTable->GetObject( pIMoniker, &pIUnknown );
119                 pIUnknown->QueryInterface( &pIEnvDTE );
120                 if( pIEnvDTE.p )
121                    break;
122              }
123           }
124     
125           if ( pIEnvDTE.p )
126           {
127              CComPtr< EnvDTE::Documents > pIDocuments;
128              HRESULT result = pIEnvDTE->get_Documents( &pIDocuments );
129              if ( !SUCCEEDED( result ) )
130                 return;
131              
132              assert( pIDocuments.p );
133              
134              CComPtr< EnvDTE::Document > pIDocument;
135              CComBSTR bstrFileName( fileName );
136              CComVariant type=_T("Text");
137              CComVariant read=_T("False");
138              result = pIDocuments->Open( bstrFileName, 
139                                          type.bstrVal,
140                                          read.bVal, 
141                                          &pIDocument );
142              if ( !SUCCEEDED( result ) )
143                 return;
144              
145              assert( pIDocument.p );
146              
147              CComPtr< IDispatch > pIDispatch;
148              result = pIDocument->get_Selection( &pIDispatch );
149              if ( !SUCCEEDED( result ) )
150                 return;
151              
152              CComPtr< EnvDTE::TextSelection > pITextSelection;
153              pIDispatch->QueryInterface( &pITextSelection );
154              
155              assert( pITextSelection.p );
156              
157              result = pITextSelection->GotoLine( line, TRUE );
158              if ( !SUCCEEDED( result ) )
159                 return;
160           }
161        }

222        int hotItem = GetNextSelectedItem(pos);
