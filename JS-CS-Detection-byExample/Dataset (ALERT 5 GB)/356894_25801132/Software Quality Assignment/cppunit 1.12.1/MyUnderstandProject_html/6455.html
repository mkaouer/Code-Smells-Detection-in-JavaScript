52                        int wrapColumn )
53     {
54       const char lineBreak = '\n';
55       Strings lines = split( text, lineBreak );
56     
57       std::string wrapped;
58       for ( Strings::const_iterator it = lines.begin(); it != lines.end(); ++it )
59       {
60         if ( it != lines.begin() )
61           wrapped += lineBreak;
62     
63         const std::string &line = *it;
64         unsigned int index =0;
65         while ( index < line.length() )
66         {
67           std::string lineSlice( line.substr( index, wrapColumn ) );
68           wrapped += lineSlice;
69           index += wrapColumn;

72         }
73       }
