10     DumperListener::DumperListener( bool flatten )
11         : m_flatten( flatten )
12         , m_suiteCount( 0 )
13         , m_testCount( 0 )
14         , m_suiteWithTestCount( 0 )
15     {
16     }

20     {
21     }

24     void 
25     DumperListener::startTest( CPPUNIT_NS::Test *test )
26     {
27       printPath( test, false );
28       ++m_testCount;
29     }
30     
31     
32     void 
33     DumperListener::endTest( CPPUNIT_NS::Test *test )
34     {
35       m_path.up();
36       if ( !m_suiteHasTest.empty() )
37       {
38         m_suiteHasTest.pop();
39         m_suiteHasTest.push( true );
40       }
41     }
42     
43     
44     void 
45     DumperListener::startSuite( CPPUNIT_NS::Test *suite )
46     {
47       printPath( suite, true );
48       ++m_suiteCount;
49       m_suiteHasTest.push( false );
50     }
51     
52     
53     void 
54     DumperListener::endSuite( CPPUNIT_NS::Test *suite )
55     {
56       m_path.up();
57       if ( m_suiteHasTest.top() )
58         ++m_suiteWithTestCount;
59       m_suiteHasTest.pop();
60     }
61     
62     
63     void 
64     DumperListener::endTestRun( CPPUNIT_NS::Test *test, 
65                                 CPPUNIT_NS::TestResult *eventManager )
66     {
67       double average = 0;
68       if ( m_suiteWithTestCount > 0 )
69         average = double(m_testCount) / m_suiteWithTestCount;
70     
71       CPPUNIT_NS::stdCOut() 
72                 << "Statistics: "  <<  m_testCount  <<  " test cases, "
73                 << m_suiteCount << " suites, "
74                 << average << " test cases / suite with test cases"
75                 << "\n";
76     }

79     void 
80     DumperListener::printPath( CPPUNIT_NS::Test *test, 
81                                bool isSuite )
82     {
83       m_path.add( test );
84     
85       if ( m_flatten )
86         printFlattenedPath( isSuite );
87       else
88         printIndentedPathChild();
89     }
90     
91     
92     void 
93     DumperListener::printFlattenedPath( bool isSuite )
94     {
95       std::string path = m_path.toString();
96       if ( isSuite )
97         path += "/";
98       CPPUNIT_NS::stdCOut()  <<  path  <<  "\n";
99     }

101     
102     void 
103     DumperListener::printIndentedPathChild()
104     {
105       std::string indent = makeIndentString( m_path.getTestCount() -1 );
106       CPPUNIT_NS::stdCOut()  <<  indent  <<  m_path.getChildTest()->getName()  <<  "\n";
107     }
108     
109     
110     std::string 
111     DumperListener::makeIndentString( int indentLevel )
112     {
113       std::string indent;
114       for ( int parentIndent =0; parentIndent < indentLevel-1; ++parentIndent )
115         indent += "|  ";
116       
117       if ( indentLevel > 0 )
118         indent += "+--";
119     
120       return indent;
121     }
