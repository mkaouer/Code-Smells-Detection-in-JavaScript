10     /*! Returns time spent in the thread.
11      * @param rquadTime Receive the time spent in the thread (user+kernel time) 
12      *                  in unit of 100 nano-seconds.
13      *                  In pratice, the effective resolution is 10ms !!!
14      *
15      * @return \c true if sucess, \c false otherwise.
16      */
17     static bool 
18     GetThreadSpentTime( LONGLONG &rquadTime )
19     {
20       FILETIME timeCreation;
21       FILETIME timeExit;
22       FILETIME timeKernel;
23       FILETIME timeUser;
24       if ( !::GetThreadTimes( ::GetCurrentThread(),
25                               &timeCreation,
26                               &timeExit,
27                               &timeKernel,
28                               &timeUser) )
29       {
30         rquadTime = 0;
31         return false;
32       }
33     
34       LARGE_INTEGER lintKernel;
35       lintKernel.LowPart = timeKernel.dwLowDateTime;
36       lintKernel.HighPart = timeKernel.dwHighDateTime;
37     
38       LARGE_INTEGER lintUser;
39       lintUser.LowPart = timeUser.dwLowDateTime;
40       lintUser.HighPart = timeUser.dwHighDateTime;
41     
42       rquadTime = lintKernel.QuadPart + lintUser.QuadPart;
43     
44       return true;
45     }
46

48     
49     void 
50     WinNtTimer::start()
51     {
52       m_isValid = GetThreadSpentTime( m_beginTime );
53     
54     }
55     
56     
57     void 
58     WinNtTimer::finish()
59     {
60       LONGLONG quadTimeEnd;
61       LONGLONG quadProcessedElapse;
62       m_isValid = m_isValid  && GetThreadSpentTime( quadTimeEnd );
63       if ( m_isValid )
64       {
65         quadProcessedElapse = quadTimeEnd - m_beginTime;
66         m_elapsedTime = double(quadProcessedElapse) / 10000000;
67       }
68       else
69         m_elapsedTime = -1;
70     }
71     
72     
73     double 
74     WinNtTimer::elapsedTime() const
75     {
76       return m_elapsedTime;
77     }
