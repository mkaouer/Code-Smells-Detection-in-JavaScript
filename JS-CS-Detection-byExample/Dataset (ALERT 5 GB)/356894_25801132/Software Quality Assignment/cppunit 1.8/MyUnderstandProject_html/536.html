13     /* A Microsoft-specific active test
14      *
15      * An active test manages its own
16      * thread of execution.  This one 
17      * is very simple and only sufficient
18      * for the limited use we put it through
19      * in the TestRunner.  It spawns a thread
20      * on run (TestResult *) and signals
21      * completion of the test.
22      *
23      * We assume that only one thread 
24      * will be active at once for each
25      * instance.
26      *
27      */
28     
29     class ActiveTest : public CppUnit::TestDecorator
30     {
31     public:
32                         ActiveTest (CppUnit::Test *test);
33                         ~ActiveTest ();
34     
35         void            run (CppUnit::TestResult *result);
36     
37     protected:
38         HANDLE          m_threadHandle;
39         CEvent          m_runCompleted;
40         CppUnit::TestResult      *m_currentTestResult;
41     
42         void            run ();
43         void            setTestResult (CppUnit::TestResult *result);
44         static UINT     threadFunction (LPVOID thisInstance);
45     
46     
47     };
48     
49     
50     // Construct the active test
51     inline ActiveTest::ActiveTest (CppUnit::Test *test)
52     : TestDecorator (test), m_runCompleted () 
53     { m_currentTestResult = NULL; m_threadHandle = INVALID_HANDLE_VALUE; }
54     
55     
56     // Pend until the test has completed
57     inline ActiveTest::~ActiveTest ()
58     { CSingleLock (&m_runCompleted, TRUE); }
59     
60     
61     // Set the test result that we are to run
62     inline void ActiveTest::setTestResult (CppUnit::TestResult *result)
63     { m_currentTestResult = result; }
64     
65     // Run our test result
66     inline void ActiveTest::run ()
67     { TestDecorator::run (m_currentTestResult); }
