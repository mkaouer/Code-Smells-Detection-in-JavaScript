41     void MsDevCallerListCtrl::OnDblclk(NMHDR* pNMHDR, LRESULT* pResult) 
42     {
43       HRESULT hr = S_OK;
44       int hotItem = 0;
45     
46       CComPtr< ITestRunnerDSPlugin> pIDSPlugin;
47      
48       hr = CoCreateInstance( CLSID_DSAddIn, NULL, CLSCTX_LOCAL_SERVER, IID_ITestRunnerDSPlugin, 
49                               reinterpret_cast< void**>(&pIDSPlugin));
50       
51       if ( SUCCEEDED( hr))
52       {
53         CPoint pt;
54         UINT flags = 0;
55         CString lineNumber, fileName;
56     
57         GetCursorPos( &pt);
58         ScreenToClient( &pt);
59     
60         // some dirty hack to get some selection
61         // should get the border-width + 1, but WINDOWINFO
62         // is not supported in Win95
63         pt.x = 5;
64     
65         hotItem = HitTest( pt, &flags);
66     
67         lineNumber = GetItemText( hotItem, s_lineNumberSubItem);
68         fileName = GetItemText( hotItem, s_fileNameSubItem);
69         
70         pIDSPlugin->goToLineInSourceCode( CComBSTR( fileName), _ttoi( lineNumber));
71     
72       }
73     
74     	*pResult = 0;
75     }
