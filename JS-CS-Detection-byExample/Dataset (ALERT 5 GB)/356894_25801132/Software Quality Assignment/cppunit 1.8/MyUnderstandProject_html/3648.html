147     TestAssertTest::checkMessageContains( CppUnit::Exception *e,
148                                           std::string expected )
149     {
150       std::string actual = e->what();
151       CPPUNIT_ASSERT_MESSAGE( "Expected message not found: " + expected +
152                               ", was: " + actual,

154                        expected.begin(), expected.end() ) != actual.end() );
