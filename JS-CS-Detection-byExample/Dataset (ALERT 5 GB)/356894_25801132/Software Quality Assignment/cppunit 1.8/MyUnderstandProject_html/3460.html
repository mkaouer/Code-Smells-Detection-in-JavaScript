27     // Paint the background of the progress bar region
28     void 
29     ProgressBar::paintBackground( CDC &dc )
30     {
31       CBrush brshBackground;
32       CPen penGray( PS_SOLID, 1, RGB (128, 128, 128) );
33       CPen penWhite( PS_SOLID, 1, RGB (255, 255, 255) );
34     
35       VERIFY( brshBackground.CreateSolidBrush( ::GetSysColor (COLOR_BTNFACE) ) );
36     
37       dc.FillRect( m_bounds, &brshBackground );
38       
39       CPen *pOldPen = dc.SelectObject( &penGray );
40       int xRight = m_bounds.left + m_bounds.Width() -1;
41       int yBottom = m_bounds.top + m_bounds.Height() -1;
42       {
43         dc.MoveTo( m_bounds.left, m_bounds.top );
44         dc.LineTo( xRight, m_bounds.top );
45     
46         dc.MoveTo( m_bounds.left, m_bounds.top );
47         dc.LineTo( m_bounds.left, yBottom );
48       }
49     
50       dc.SelectObject( &penWhite );
51       {
52         dc.MoveTo( xRight, m_bounds.top );
53         dc.LineTo( xRight, yBottom );
54     
55         dc.MoveTo( m_bounds.left, yBottom );
56         dc.LineTo( xRight, yBottom );
57       }
58     
59       dc.SelectObject( pOldPen );
60     }

63     // Paint the actual status of the progress bar
64     void 
65     ProgressBar::paintStatus( CDC &dc )
66     {
67       if ( m_progress <= 0 )
68         return;
69     
70       CBrush brshStatus;
71       CRect rect( m_bounds.left, m_bounds.top, 
72                   m_bounds.left + m_progressX, m_bounds.bottom );
73     
74       COLORREF statusColor = getStatusColor();
75     
76       VERIFY( brshStatus.CreateSolidBrush( statusColor ) );
77     
78       rect.DeflateRect( 1, 1 );
79       dc.FillRect( rect, &brshStatus );
80     }

102     // Set the width of the progress bar, scale the  and invalidate the base window
103     void 
104     ProgressBar::setWidth( int width )
105     {
106       m_baseWindow->InvalidateRect( m_bounds ); // invalidate before
107       m_bounds.right = m_bounds.left + width;
108       m_progressX = scale( m_progress );
109       m_baseWindow->InvalidateRect( m_bounds ); // and invalidate after
110     }

113     // Take one step, indicating whether it was a successful step
114     void 
115     ProgressBar::step( bool successful )
116     {
117       m_progress++;
118     
119       int x = m_progressX;
120     
121       m_progressX = scale (m_progress);
122     
123       if ( !m_error  &&  !successful )
124       {
125         m_error = true;
126         x = 1;
127       }
128     
129       paintStep( x, m_progressX );
130     }

133     // Map from steps to display units
134     int 
135     ProgressBar::scale (int value)
136     {
137       if ( m_total > 0 )
138           return max( 1, value * (m_bounds.Width() - 1) / m_total );
139     
140       return value;
141     }

145     void
