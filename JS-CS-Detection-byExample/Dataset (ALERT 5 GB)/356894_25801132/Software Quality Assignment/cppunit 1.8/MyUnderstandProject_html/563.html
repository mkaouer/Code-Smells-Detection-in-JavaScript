13     TestPlugIn::TestPlugIn( const std::string fileName ) :

29     void 
30     TestPlugIn::releaseDll()
31     {
32       if ( m_dllHandle != NULL )
33         ::FreeLibrary( m_dllHandle );
34       m_dllHandle = NULL;
35       m_interfaceFunction = NULL;
36     }

64     CppUnit::Test *
65     TestPlugIn::makeTest()
66     {
67       reloadDll();
68       TestPlugInInterface *theInterface = (*m_interfaceFunction)();
69     
70       try
71       {
72         return theInterface->makeTest();
73       }
74       catch ( ... )
75       {
76         throw TestPlugInException( "Failed to make test using GetTestPlugInInterface", 
77                                    TestPlugInException::failedToMakeTest );
78       }

82     void 
83     TestPlugIn::reloadDll()
84     {
85       releaseDll();
86       makeDllCopy();
87       loadDll();
88       getDllInterface();
89     }

92     void 
93     TestPlugIn::makeDllCopy()
94     {
95       if ( ::CopyFile( m_fileName.c_str(), m_copyFileName.c_str(), FALSE ) == FALSE )
96       {
97         throw TestPlugInException( "Failed to copy DLL" + m_fileName +
98             " to " + m_copyFileName, TestPlugInException::failedToCopyDll );
99       }
100     }
101     
102     
103     void 
104     TestPlugIn::loadDll()
105     {
106       m_dllHandle = ::LoadLibrary( m_copyFileName.c_str() );
107       if ( m_dllHandle == NULL )
108         throw TestPlugInException( "Failed to load DLL " + m_copyFileName, 
109                                    TestPlugInException::failedToLoadDll );
110     }
111     
112     
113     void
114     TestPlugIn::getDllInterface()
115     {
116       m_interfaceFunction = (GetTestPlugInInterfaceFunction )
117           ::GetProcAddress( m_dllHandle, "GetTestPlugInInterface" );
118       if ( m_interfaceFunction == NULL )
119         throw TestPlugInException( "Failed to locate function GetTestPlugInInterface "
120                                    " in DLL " + m_fileName, 
121                                    TestPlugInException::failedToGetInterfaceFunction );
122     }
