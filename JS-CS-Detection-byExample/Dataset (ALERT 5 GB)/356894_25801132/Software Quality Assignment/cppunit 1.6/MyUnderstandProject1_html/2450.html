77     HTREEITEM
78     TreeHierarchyDlg::addTest( CppUnit::Test *test, 
79                                HTREEITEM hParent )
80     {
81       int testType = isTestSuite( test ) ? imgSuite : imgUnitTest;
82       HTREEITEM hItem = m_treeTests.InsertItem( test->getName().c_str(),
83                                                 testType,
84                                                 testType,
85                                                 hParent );
86       if ( hItem != NULL )
87       {
88         VERIFY( m_treeTests.SetItemData( hItem, (DWORD)test ) );
89         if ( isTestSuite( test ) )
90           addTestSuiteChildrenTo( static_cast<CppUnit::TestSuite *>(test),
91                                   hItem );
92       }
93       return hItem;
94     }

97     void 
98     TreeHierarchyDlg::addTestSuiteChildrenTo( CppUnit::TestSuite *suite,
99                                               HTREEITEM hItemSuite )
100     {
101       Tests tests( suite->getTests() );
102       sortByName( tests );
103     
104       for ( Tests::const_iterator it = tests.begin(); it != tests.end(); ++it )
105       {
106         addTest( *it, hItemSuite );
107       }
108     }

115       return suite != NULL;

119     struct PredSortTest
120     {
121       bool operator()( CppUnit::Test *test1, CppUnit::Test *test2 ) const
122       {
123         bool isTest1Suite = TreeHierarchyDlg::isTestSuite( test1 );
124         bool isTest2Suite = TreeHierarchyDlg::isTestSuite( test2 );
125         if ( isTest1Suite  &&  !isTest2Suite )
126           return true;
127         if ( isTest1Suite  &&  isTest2Suite )
128           return test1->getName() < test2->getName();
129         return false;
130       }
131     };
132     
133     void 
134     TreeHierarchyDlg::sortByName( Tests &tests ) const
135     {
136       std::stable_sort( tests.begin(), tests.end(), PredSortTest() );
137     }

141     TreeHierarchyDlg::OnOK() 
142     {
143       CppUnit::Test *test = findSelectedTest();
144       if ( test == NULL )

146         AfxMessageBox( IDS_ERROR_SELECT_TEST, MB_OK );
147         return;

155     CppUnit::Test *
156     TreeHierarchyDlg::findSelectedTest()
157     {
158       HTREEITEM hItem = m_treeTests.GetSelectedItem();
159       if ( hItem != NULL )
160       {
161         DWORD data;
162         VERIFY( data = m_treeTests.GetItemData( hItem ) );
163         return reinterpret_cast<CppUnit::Test *>( data );
164       }
165       return NULL;
166     }
