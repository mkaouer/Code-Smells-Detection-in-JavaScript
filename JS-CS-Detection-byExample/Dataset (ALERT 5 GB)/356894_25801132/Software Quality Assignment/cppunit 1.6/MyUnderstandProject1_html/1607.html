7     // Spawn a thread to a test
8     void ActiveTest::run (CppUnit::TestResult *result)
9     {
10         CWinThread *thread;
11         
12         setTestResult (result);
13         m_runCompleted.ResetEvent ();
14     
15         thread = AfxBeginThread (threadFunction, 
16             this, 
17             THREAD_PRIORITY_NORMAL, 
18             0, 
19             CREATE_SUSPENDED);
20         
21         DuplicateHandle (GetCurrentProcess (), 
22             thread->m_hThread,
23             GetCurrentProcess (), 
24             &m_threadHandle, 
25             0, 
26             FALSE, 
27             DUPLICATE_SAME_ACCESS);
28     
29         thread->ResumeThread ();
30     
31     }

34     // Simple execution thread.  Assuming that an ActiveTest instance
35     // only creates one of these at a time.
36     UINT ActiveTest::threadFunction (LPVOID thisInstance)
37     {
38         ActiveTest *test = (ActiveTest *)thisInstance;
39     
40         test->run ();
41         test->m_runCompleted.SetEvent ();
42     
43         return 0;
44     }
