8     /*
9      * Orthodox performs a simple set of tests on an arbitary
10      * class to make sure that it supports at least the
11      * following operations:
12      *
13      *      default construction    - constructor
14      *      equality/inequality     - operator== && operator!=
15      *      assignment              - operator=
16      *      negation                - operator!
17      *      safe passage            - copy construction
18      *
19      * If operations for each of these are not declared
20      * the template will not instantiate.  If it does 
21      * instantiate, tests are performed to make sure
22      * that the operations have correct semantics.
23      *      
24      * Adding an orthodox test to a suite is very 
25      * easy: 
26      * 
27      * public: Test *suite ()  {
28      *     TestSuite *suiteOfTests = new TestSuite;
29      *     suiteOfTests->addTest (new ComplexNumberTest ("testAdd");
30      *     suiteOfTests->addTest (new TestCaller<Orthodox<Complex> > ());
31      *     return suiteOfTests;
32      *  }
33      *
34      * Templated test cases be very useful when you are want to
35      * make sure that a group of classes have the same form.
36      *
37      * see TestSuite
38      */
39     
40     
41     template <typename ClassUnderTest> class Orthodox : public TestCase
42     {
43     public:
44                         Orthodox () : TestCase ("Orthodox") {}
45     
46     protected:
47         ClassUnderTest  call (ClassUnderTest object);
48         void            runTest ();
49     
50     
51     };
52

54     // Run an orthodoxy test
55     template <typename ClassUnderTest> void Orthodox<ClassUnderTest>::runTest ()
56     {
57         // make sure we have a default constructor
58         ClassUnderTest   a, b, c;
59     
60         // make sure we have an equality operator
61         CPPUNIT_ASSERT (a == b);
62     
63         // check the inverse
64         b.operator= (a.operator! ());
65         CPPUNIT_ASSERT (a != b);
66     
67         // double inversion
68         b = !!a;
69         CPPUNIT_ASSERT (a == b);
70     
71         // invert again
72         b = !a;
73     
74         // check calls
75         c = a;
76         CPPUNIT_ASSERT (c == call (a));
77     
78         c = b;
79         CPPUNIT_ASSERT (c == call (b));
80     
81     }
