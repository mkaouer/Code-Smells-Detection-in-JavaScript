3     // from a DLL simpler. All files within this DLL are compiled with the TESTPLUGIN_EXPORTS
4     // symbol defined on the command line. this symbol should not be defined on any project
5     // that uses this DLL. This way any other project whose source files include this file see 
6     // TESTPLUGIN_API functions as being imported from a DLL, wheras this DLL sees symbols
7     // defined with this macro as being exported.
8     #ifdef TESTPLUGIN_EXPORTS
9     #define TESTPLUGIN_API __declspec(dllexport)
10     #else
