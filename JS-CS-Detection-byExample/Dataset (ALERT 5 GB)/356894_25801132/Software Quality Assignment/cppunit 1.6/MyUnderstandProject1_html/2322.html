143     void TestRunnerDlg::OnRun() 
144     {
145       if ( m_bIsRunning )
146         return;
147     
148       m_selectedTest = m_model->selectedTest();
149     
150       if (m_selectedTest == 0)
151     	  return;
152     
153       freeState       (); 
154       reset           ();
155     
156       beRunning       ();
157     
158       int numberOfTests = m_selectedTest->countTestCases ();
159     
160       m_testsProgress->start (numberOfTests);
161     
162       m_result            = new GUITestResult ((TestRunnerDlg *)this);
163       m_activeTest        = new ActiveTest (m_selectedTest);
164     
165       m_testStartTime     = timeGetTime ();
166     
167       m_activeTest->run (m_result);
168     
169       m_testEndTime       = timeGetTime ();
170     }

265     void TestRunnerDlg::endTest (CppUnit::TestResult *result, CppUnit::Test *test)
266     {
267     	if (m_selectedTest == 0)
268     		return;
269     
270         m_testsRun++;
271         updateCountsDisplay ();
272         m_testsProgress->step (m_failures == 0 && m_errors == 0);
273     
274         m_testEndTime   = timeGetTime ();
275     
276         updateCountsDisplay ();
277     
278         if (m_testsRun >= m_selectedTest->countTestCases ())
279             beIdle ();
280     }

347         CStatic *statFailures   = (CStatic *)GetDlgItem (IDC_STATIC_FAILURES);

384     void 
385     TestRunnerDlg::OnSelchangeComboTest() 
386     {
387       int currentSelection = getHistoryCombo()->GetCurSel ();
388     
389       if ( currentSelection >= 0  &&
390            currentSelection < m_model->history().size() )
391       {
392         CppUnit::Test *selectedTest = m_model->history()[currentSelection];
393         m_model->selectHistoryTest( selectedTest );
394         updateHistoryCombo();
395         beIdle();
396       }
397       else
398         beRunDisabled();
399     }

402     void
403     TestRunnerDlg::updateHistoryCombo()
404     {
405       getHistoryCombo()->LockWindowUpdate();
406     
407       getHistoryCombo()->ResetContent();
408     
409       const TestRunnerModel::History &history = m_model->history();
410       for ( TestRunnerModel::History::const_iterator it = history.begin(); 
411             it != history.end(); 
412             ++it )
413       {
414         CppUnit::Test *test = *it;
415         getHistoryCombo()->AddString( test->getName().c_str() );
416       }
417     
418       if ( history.size() > 0 )
419       {
420         getHistoryCombo()->SetCurSel( 0 );
421         beIdle();
422       }
423       else
424         beRunDisabled();
425     
426       getHistoryCombo()->UnlockWindowUpdate();
427     }
