11       template <class T>
12       struct assertion_traits 
13       {  
14           static bool equal( const T& x, const T& y )
15           {
16               return x == y;
17           }
18     
19           static std::string toString( const T& x )
20           {
21               OStringStream ost;
22               ost << x;
23               return ost.str();
24           }
25       };

41         template <class T>
42         void assertEquals( const T& expected,
43                            const T& actual,
44                            long lineNumber = Exception::UNKNOWNLINENUMBER,
45                            std::string fileName = Exception::UNKNOWNFILENAME )
46         {
47           if ( !assertion_traits<T>::equal(expected,actual) ) // lazy toString conversion...
48           {
49             assertNotEqualImplementation( assertion_traits<T>::toString(expected),
50                                           assertion_traits<T>::toString(actual),
51                                           lineNumber, 
52                                           fileName );
53           }
54         }

57                            double actual, 
58                            double delta, 
59                            long lineNumber = Exception::UNKNOWNLINENUMBER,
60                            std::string fileName = Exception::UNKNOWNFILENAME);

95     /// Generalized macro for primitive value comparisons
96     /** Equality and string representation can be defined with
97      * an appropriate assertion_traits class.
98      * A diagnostic is printed if actual and expected values disagree.
99      */
100     #define CPPUNIT_ASSERT_EQUAL(expected,actual)\
101       (CppUnit::TestAssert::assertEquals ((expected),\
102         (actual),__LINE__,__FILE__))
103     
104     /// Macro for primitive value comparisons
105     #define CPPUNIT_ASSERT_DOUBLES_EQUAL(expected,actual,delta)\
