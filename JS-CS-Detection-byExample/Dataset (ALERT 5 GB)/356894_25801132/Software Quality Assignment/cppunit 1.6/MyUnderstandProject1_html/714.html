53     /*! \brief Generate a test case from a fixture method.
54      *
55      * A test caller provides access to a test case method 
56      * on a test case class.  Test callers are useful when 
57      * you want to run an individual test or add it to a 
58      * suite.
59      * Test Callers invoke only one Test (i.e. test method) on one 
60      * Fixture of a TestCase.
61      * 
62      * Here is an example:
63      * \code
64      * class MathTest : public CppUnit::TestCase {
65      *         ...
66      *     public:
67      *         void         setUp ();
68      *         void         tearDown ();
69      *
70      *         void         testAdd ();
71      *         void         testSubtract ();
72      * };
73      *
74      * CppUnit::Test *MathTest::suite () {
75      *     CppUnit::TestSuite *suite = new CppUnit::TestSuite;
76      *
77      *     suite->addTest (new CppUnit::TestCaller<MathTest> ("testAdd", testAdd));
78      *     return suite;
79      * }
80      * \endcode
81      *
82      * You can use a TestCaller to bind any test method on a TestCase
83      * class, as long as it accepts void and returns void.
84      * 
85      * \see TestCase
86      */
87     
88     template <typename Fixture,  
89     	  typename ExpectedException = NoExceptionExpected>
90     class TestCaller : public TestCase
91     { 
92         typedef void (Fixture::*TestMethod)();
93         
94     public:
95         /**
96          * Constructor for TestCaller. This constructor builds a new Fixture
97          * instance owned by the TestCaller.
98          * \param name name of this TestCaller
99          * \param test the method this TestCaller calls in runTest()
100          **/
101         TestCaller (std::string name, TestMethod test) :
102     	TestCase (name), 
103     	m_ownFixture(true),
104     	m_fixture (new Fixture ()),
105     	m_test (test)
106         {}
107     
108         /**
109          * Constructor for TestCaller. 
110          * This constructor does not create a new Fixture instance but accepts
111          * an existing one as parameter. The TestCaller will not own the
112          * Fixture object.
113          * \param name name of this TestCaller
114          * \param test the method this TestCaller calls in runTest()
115          * \param fixture the Fixture to invoke the test method on.
116          **/
117         TestCaller(std::string name, TestMethod test, Fixture& fixture) :
118     	TestCase (name), 
119     	m_ownFixture(false),
120     	m_fixture (&fixture),
121     	m_test (test)
122         {}
123         
124         /**
125          * Constructor for TestCaller. 
126          * This constructor does not create a new Fixture instance but accepts
127          * an existing one as parameter. The TestCaller will own the
128          * Fixture object and delete it in its destructor.
129          * \param name name of this TestCaller
130          * \param test the method this TestCaller calls in runTest()
131          * \param fixture the Fixture to invoke the test method on.
132          **/
133         TestCaller(std::string name, TestMethod test, Fixture* fixture) :
134     	TestCase (name), 
135     	m_ownFixture(true),
136     	m_fixture (fixture),
137     	m_test (test)
138         {}
139         
140         ~TestCaller() {
141     	if (m_ownFixture) {
142     	    if (m_fixture) {
143     		delete m_fixture;
144     		m_fixture = NULL;
145     	    }
146     	}
147         }
148     
149     protected:
150         void runTest () 
151         { 
152     	try {
153     	    (m_fixture->*m_test)();
154     	}
155     	catch ( ExpectedException & ) {
156     	    return;
157     	}
158     
159     	ExpectedExceptionTraits<ExpectedException>::expectedException();
160         }  
161     
162         void setUp ()
163         { 
164     	m_fixture->setUp (); 
165         }
166     
167         void tearDown ()
168         { 
169     	m_fixture->tearDown (); 
170         }
171     
172         std::string toString () const
173         { 
174     	return "TestCaller " + getName(); 
175         }
176     
177     private: 
178         TestCaller (const TestCaller& other); 
179         TestCaller& operator= (const TestCaller& other); 
180     
181     private:
182         bool       m_ownFixture;
183         Fixture*   m_fixture;
184         TestMethod m_test;
185     
186     };
