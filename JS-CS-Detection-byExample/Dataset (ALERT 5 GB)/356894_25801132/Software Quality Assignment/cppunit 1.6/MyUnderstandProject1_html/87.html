14       /**
15        * A TestResult collects the results of executing a test case. It is an 
16        * instance of the Collecting Parameter pattern.
17        *
18        * The test framework distinguishes between failures and errors.
19        * A failure is anticipated and checked for with assertions. Errors are
20        * unanticipated problems signified by exceptions that are not generated
21        * by the framework.
22        *
23        * TestResult supplies a template method 'setSynchronizationObject ()'
24        * so that subclasses can provide mutual exclusion in the face of multiple
25        * threads.  This can be useful when tests execute in one thread and
26        * they fill a subclass of TestResult which effects change in another 
27        * thread.  To have mutual exclusion, override setSynchronizationObject ()
28        * and make sure that you create an instance of ExclusiveZone at the 
29        * beginning of each method.
30        *
31        * \see Test
32        */
33     
34       class TestResult
35       {
36     
37         public:
38           TestResult  ();
39           virtual                             ~TestResult ();
40     
41           virtual void                        addError       (Test *test, Exception *e);
42           virtual void                        addFailure     (Test *test, Exception *e);
43           virtual void                        startTest      (Test *test);
44           virtual void                        endTest        (Test *test);
45           virtual int                         runTests       ();
46           virtual int                         testErrors     ();
47           virtual int                         testFailures   ();
48           virtual bool                        wasSuccessful  ();
49           virtual bool                        shouldStop     ();
50           virtual void                        stop           ();
51     
52           virtual std::vector<TestFailure *>& errors         ();
53           virtual std::vector<TestFailure *>& failures       ();
54     
55           virtual void                        addListener    ( TestListener *listener );
56           virtual void                        removeListener ( TestListener *listener );

61             public:

64
