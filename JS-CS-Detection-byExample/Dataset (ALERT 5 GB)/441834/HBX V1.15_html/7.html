311         private void saveMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_saveMouseClicked
312             //compile the BinauralEnvelope and save it on a file
313             if (Main.buttonsDisabled) {
314                 return;
315             }
316             try {
317                 if (env.getText().trim().isEmpty()) {
318                     throw new Exception(nothing);
319                 }
320                 baseF.commitEdit();
321                 String toCompile = ((Double) baseF.getValue()) + "\n" + env.getText();
322                 BinauralEnvelope be = BinauralEnvelope.fromString(toCompile);
323                 if (be.getLength() < 1) {
324                     throw new Exception(tooShort);
325                 }
326                 Object[] o = FileChooser.show(new FileFilter[]{Main.ffX, Main.ffL, Main.ffS}, false);
327                 File x = (File) o[0];
328                 FileFilter ff = (FileFilter) o[1];
329                 if (x == null) {
330                     return;
331                 }
332                 if (ff.equals(Main.ffX)) {
333                     if (!x.getName().toLowerCase().endsWith(".hbx")) {
334                         x = new File(x.getAbsolutePath() + ".hbx");
335                     }
336                     FileOutputStream fos = new FileOutputStream(x);
337                     fos.write(Main.HBX_HEADER);
338                     GZIPOutputStream gos = new GZIPOutputStream(fos);
339                     ObjectOutputStream oos = new ObjectOutputStream(gos);
340                     oos.writeObject(be);
341                     oos.flush();
342                     oos.close();
343                     gos.flush();
344                     gos.close();
345                     fos.flush();
346                     fos.close();
347                 }
348                 if (ff.equals(Main.ffS)) {
349                     if (!x.getName().toLowerCase().endsWith(".hbs")) {
350                         x = new File(x.getAbsolutePath() + ".hbs");
351                     }
352                     FileOutputStream fos = new FileOutputStream(x);
353                     fos.write(Main.HBS_HEADER);
354                     fos.write(be.toHES());
355                     fos.flush();
356                     fos.close();
357                 }
358                 if (ff.equals(Main.ffL)) {
359                     if (!x.getName().toLowerCase().endsWith(".hbl")) {
360                         x = new File(x.getAbsolutePath() + ".hbl");
361                     }
362                     FileWriter fw=new FileWriter(x);
363                     fw.write(be.toXML());
364                     fw.close();
365                 }
366                 String e = be.toString();
367                 e = e.substring(e.indexOf('\n') + 1, e.length());
368                 env.setText(e);
369                 MessageBox.info(saved, savedTitle);
370             } catch (Exception e) {
371                 MessageBox.error(err + e.getMessage(), errTitle);
372             }
373         }//GEN-LAST:event_saveMouseClicked

417         private void exportMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_exportMouseClicked
418             //compile and save to wav file
419             if (Main.buttonsDisabled) {
420                 return;
421             }
422             try {
423                 if (env.getText().trim().isEmpty()) {
424                     throw new Exception(nothing);
425                 }
426                 baseF.commitEdit();
427                 String toCompile = ((Double) baseF.getValue()) + "\n" + env.getText();
428                 BinauralEnvelope be = BinauralEnvelope.fromString(toCompile);
429                 if (be.getLength() < 1) {
430                     throw new Exception(tooShort);
431                 }
432                 File x = (File) FileChooser.show(new FileFilter[]{new FileFilter() {
433                         @Override
434                         public boolean accept(File f) {
435                             if (f.isDirectory() || f.getName().toLowerCase().endsWith(".wav")) {
436                                 return true;
437                             } else {
438                                 return false;
439                             }
440                         }
441     
442                         @Override
443                         public String getDescription() {
444                             return ".wav File";
445                         }
446                     }}, false)[0];
447                 if (x == null) {
448                     return;
449                 }
450                 if (!x.getName().toLowerCase().endsWith(".wav")) {
451                     x = new File(x.getAbsolutePath() + ".wav");
452                 }
453                 RenderWindow.render(be, new FileOutputStream(x));
454             } catch (Exception e) {
455                 MessageBox.error(err + e.getMessage(), errTitle);
456             }
457         }//GEN-LAST:event_exportMouseClicked
