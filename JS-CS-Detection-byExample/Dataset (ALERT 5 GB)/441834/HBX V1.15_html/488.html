63          * "interface" with the BinauralEnvelopePlayer

65         private class BEPThread extends Thread {
66     
67             private BinauralEnvelopePlayer beplayer;
68             private BinauralEnvelope be;
69             private boolean killASAP = false;
70     
71             public BEPThread(BinauralEnvelope be) {
72                 try {
73                     this.be = be;
74                     playing = false;
75                     playPause.setIcon(play);
76                     createBEPlayer(be);
77                 } catch (LineUnavailableException ex) {
78                     playerError();
79                 }
80             }
81     
82             public void pause() {
83                 beplayer.paused = true;
84             }
85     
86             public void unPause() {
87                 beplayer.paused = false;
88             }
89     
90             @Override
91             public void run() {
92                 if (inactive) {
93                     return;
94                 }
95                 for (;;) {
96                     while (beplayer.getPosition() <= 1) {
97                         if (killASAP) {
98                             return;
99                         }
100                         try {
101                             Thread.sleep(10);
102                         } catch (InterruptedException ex) {
103                         }
104                         if (killASAP) {
105                             return;
106                         }
107                         prog.setValue((int) (100 * beplayer.getPosition()));
108                         int len = (int) beplayer.getLength();
109                         int pos = (int) (beplayer.getPosition() * beplayer.getLength());
110                         int lenH = len / 3600;
111                         len %= 3600;
112                         int lenM = len / 60;
113                         len %= 60;
114                         int lenS = len;
115                         int posH = pos / 3600;
116                         pos %= 3600;
117                         int posM = pos / 60;
118                         pos %= 60;
119                         int posS = pos;
120                         time.setText((posH < 10 ? "0" + posH : posH) + ":" + (posM < 10 ? "0" + posM : posM) + ":" + (posS < 10 ? "0" + posS : posS) + " / " + (lenH < 10 ? "0" + lenH : lenH) + ":" + (lenM < 10 ? "0" + lenM : lenM) + ":" + (lenS < 10 ? "0" + lenS : lenS));
121                     }
122                     beplayer.stopPlaying();
123                     playPause.setIcon(play);
124                     playing = false;
125                     try {
126                         createBEPlayer(be);
127                     } catch (LineUnavailableException ex) {
128                         playerError();
129                     }
130                 }
131             }
132     
133             private void createBEPlayer(BinauralEnvelope be) throws LineUnavailableException {
134                 beplayer = new BinauralEnvelopePlayer(be);
135                 beplayer.paused = true;
136                 beplayer.start();
137             }
138     
139             private void playerError() {
140                 prog.setStringPainted(true);
141                 prog.setString(scErr);
142                 prog.setIndeterminate(true);
143                 deactivate();
144             }
145     
146             public void stopPlaying() {
147                 beplayer.stopPlaying();
148                 killASAP = true;
149             }

152                 beplayer.setPosition(p);
153             }
154     
155             private void setVolume(double d) {
156                 beplayer.setVolume(d);
157             }
158         }

276         private void progMouseDragged(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_progMouseDragged
277             if (Main.buttonsDisabled) {
278                 return;
279             }
280             if (inactive) {
281                 return;
282             }
283             double p = 1 - ((double) (prog.getWidth() - evt.getX()) / (double) prog.getWidth());
284             bep.setPosition(p < 0 ? 0 : p > 1 ? 1 : p);
285         }//GEN-LAST:event_progMouseDragged

287         private void playPauseMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_playPauseMouseClicked
288             if (Main.buttonsDisabled) {

295                 bep.pause();
296                 playPause.setIcon(play);
297                 playing = false;
298             } else {
299                 bep.unPause();

301                 playing = true;
302             }

313         private void volMouseDragged(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_volMouseDragged
314             if (Main.buttonsDisabled) {
315                 return;
316             }
317             if (inactive) {
318                 return;
319             }
320             double p = 1 - ((double) (vol.getWidth() - evt.getX()) / (double) vol.getWidth());
321             p = p < 0 ? 0 : p > 1 ? 1 : p;
322             bep.setVolume(p);
323             vol.setValue((int) (100 * p));
324         }//GEN-LAST:event_volMouseDragged
