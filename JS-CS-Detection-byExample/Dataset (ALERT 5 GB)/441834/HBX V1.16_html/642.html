30         /**
31          * headers (magic numbers actually) of HBX and HBS files are calculated from
32          * 2 simple strings (HBX and HBS respectively)
33          *
34          */
35         public static final byte[] HBX_HEADER, HBS_HEADER;
36     
37         static {
38             String h = "HBX";
39             HBX_HEADER = new byte[h.length()];
40             char[] arr = h.toCharArray();
41             for (int i = 0; i < arr.length; i++) {
42                 HBX_HEADER[i] = (byte) arr[i];
43             }
44             h = "HBS";
45             HBS_HEADER = new byte[h.length()];
46             arr = h.toCharArray();
47             for (int i = 0; i < arr.length; i++) {
48                 HBS_HEADER[i] = (byte) arr[i];
49             }
50         }

52         /**
53          * loads a preset from a file
54          *
55          * @param x input file
56          * @return the BinauralEnvelope stored in the file
57          * @throws FileNotFoundException if an IO error occurs
58          * @throws IOException if an IO error occurs
59          * @throws Exception if an IO error occurs
60          */
61         public static BinauralEnvelope loadPreset(File x) throws FileNotFoundException, IOException, Exception {
62             if (x == null) {
63                 throw new Exception("Invalid file");
64             }
65             if (x.getName().toLowerCase().endsWith(".hbl")) {
66                 BufferedReader xmlFile = new BufferedReader(new FileReader(x));
67                 String xml = "";
68                 for (;;) {
69                     try {
70                         String line = xmlFile.readLine();
71                         if (line == null) {
72                             break;
73                         } else {
74                             xml += line + "\n";
75                         }
76                     } catch (IOException e) {
77                         break;
78                     }
79                 }
80                 xmlFile.close();
81                 return BinauralEnvelope.fromXML(xml);
82             } else {
83                 FileInputStream fis = new FileInputStream(x);
84                 boolean hbx = true, hbs = true;
85                 byte[] headerX = new byte[HBX_HEADER.length];
86                 byte[] headerS = new byte[HBS_HEADER.length];
87                 fis.read(headerX);
88                 fis.close();
89                 fis = new FileInputStream(x);
90                 fis.read(headerS);
91                 for (int i = 0; i < headerX.length; i++) {
92                     if (headerX[i] != HBX_HEADER[i]) {
93                         hbx = false;
94                     }
95                 }
96                 for (int i = 0; i < headerS.length; i++) {
97                     if (headerS[i] != HBS_HEADER[i]) {
98                         hbs = false;
99                     }
100                 }
101                 fis.close();
102                 fis = new FileInputStream(x);
103                 if (!hbx && !hbs) {
104                     throw new Exception("Invalid file");
105                 }
106                 if (hbx) {
107                     fis.read(headerX);
108                     ObjectInputStream ois = new ObjectInputStream(new GZIPInputStream(fis));
109                     BinauralEnvelope be = (BinauralEnvelope) (ois.readObject());
110                     ois.close();
111                     return be;
112                 }
113                 if (hbs) {
114                     fis.read(headerS);
115                     ByteArrayOutputStream baos = new ByteArrayOutputStream();
116                     for (;;) {
117                         int in = fis.read();
118                         if (in == -1) {
119                             break;
120                         } else {
121                             baos.write(in);
122                         }
123                     }
124                     BinauralEnvelope be = BinauralEnvelope.fromHES(baos.toByteArray());
125                     fis.close();
126                     return be;
127                 }
128             }
129             throw new Exception("Invalid file");
130         }

132         /**
133          * saves a BinauralEnvelope to a file (HBX format)
134          *
135          * @param be The BinauralEnvelope to save
136          * @param f destination file
137          * @throws Exception if some IO error occurs
138          */
139         public static void toHBX(BinauralEnvelope be, File f) throws Exception {
140             FileOutputStream fos = new FileOutputStream(f);
141             fos.write(HBX_HEADER);
142             GZIPOutputStream gos = new GZIPOutputStream(fos);
143             ObjectOutputStream oos = new ObjectOutputStream(gos);
144             oos.writeObject(be);
145             oos.flush();
146             oos.close();
147             gos.flush();
148             gos.close();
149             fos.flush();
150             fos.close();
151         }

153         /**
154          * saves a BinauralEnvelope to a file (HBS format)
155          *
156          * @param be The BinauralEnvelope to save
157          * @param f destination file
158          * @throws Exception if some IO error occurs
159          */
160         public static void toHBS(BinauralEnvelope be, File f) throws Exception {
161             FileOutputStream fos = new FileOutputStream(f);
162             fos.write(HBS_HEADER);
163             fos.write(be.toHES());
164             fos.flush();
165             fos.close();
166         }

182         private static void writeS(OutputStream out, String s) {
183             for (int i = 0; i < s.length(); i++) {
184                 try {
185                     out.write(s.charAt(i));
186                 } catch (IOException ex) {
187                 }
188             }
189         }

192         private static void writeInt(OutputStream out, int val) {
193             try {
194                 out.write(val);
195                 out.write(val >> 8);
196                 out.write(val >> 16);
197                 out.write(val >> 24);
198             } catch (IOException ex) {
199             }

201         }

204         private static void writeShort(OutputStream out, short val) {
205             try {
206                 out.write(val);
207                 out.write(val >> 8);
208             } catch (IOException ex) {
209             }
210         }

212         /**
213          * renders a BinauralEnvelope to a wav file. (note: this method is
214          * non-blocking!)
215          *
216          * @param be The BinauralEnvelope to render
217          * @param f the OutputStream where the data will be written to
218          * @return an instance of BinauralEnvelopePlayer. This can be used to check
219          * progress. when the render is complete, it can be stopped with
220          * .stopPlaying() at any time. .join() can be called when the render is
221          * complete (or aborted) to stop the thread and then file can be closed
222          * @throws Exception if the output file is invalid or some other IO error
223          * occurs
224          */
225         public static BinauralEnvelopePlayer toWAV(BinauralEnvelope be, OutputStream os) throws Exception {
226             //wav header stuff
227             writeS(os, "RIFF");
228             writeInt(os, (int) (Math.ceil(36 + be.getLength() * 44100 * 4)));
229             writeS(os, "WAVE");
230             writeS(os, "fmt ");
231             writeInt(os, 16);
232             writeShort(os, (short) 1);
233             writeShort(os, (short) 2);
234             writeInt(os, 44100);
235             writeInt(os, 2 * 44100 * 16 / 8);
236             writeShort(os, (short) (2 * 16 / 8));
237             writeShort(os, (short) 16);
238             writeS(os, "data");
239             writeInt(os, (int) (Math.ceil(be.getLength() * 44100 * 4)));
240             //start rendering
241             BinauralEnvelopePlayer bep = new BinauralEnvelopePlayer(be, os);
242             bep.start();
243             return bep;
244         }
