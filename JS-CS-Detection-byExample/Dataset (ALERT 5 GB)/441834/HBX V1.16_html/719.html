173         private void jLabel2MouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_jLabel2MouseClicked
174             if (Main.DISABLE_TRANSLUCENCY) {
175                 dispose();
176                 return;
177             }
178             setOpacity(1);
179             for (float i = 1; i >= 0; i -= 0.1) {
180                 setOpacity(i);
181                 try {
182                     Thread.sleep(20);
183                 } catch (InterruptedException ex) {
184                 }
185             }
186             setOpacity(0);
187             dispose();
188         }//GEN-LAST:event_jLabel2MouseClicked

207         private void formWindowOpened(java.awt.event.WindowEvent evt) {//GEN-FIRST:event_formWindowOpened
208             if (Main.DISABLE_TRANSLUCENCY) {
209                 return;
210             }

212                 public void run() {
213                     for (float i = 0; i <= 1; i += 0.1) {
214                         setOpacity(i);
215                         try {
216                             Thread.sleep(20);

218                         }
219                     }
220                     setOpacity(1);
221                 }
222             }.start();
