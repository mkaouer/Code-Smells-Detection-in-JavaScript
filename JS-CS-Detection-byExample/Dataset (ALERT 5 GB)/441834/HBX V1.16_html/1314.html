162         private void jFileChooser1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jFileChooser1ActionPerformed
163             if (evt.getActionCommand().equals(JFileChooser.APPROVE_SELECTION)) {
164                 selectedFile = jFileChooser1.getSelectedFile();
165                 if (jFileChooser1.getDialogType() == JFileChooser.SAVE_DIALOG && selectedFile != null && selectedFile.exists()) {
166                     setAlwaysOnTop(false);
167                     int sel = YesNoCancelDialog.confirm(overwriteConfirm, title.getText(), true);
168                     setAlwaysOnTop(true);
169                     if (sel == YesNoCancelDialog.NO) {
170                         selectedFile = null;
171                         return;
172                     }
173                     if (sel == YesNoCancelDialog.CANCEL) {
174                         selectedFile = null;
175                     }
176                 }
177                 if (selectedFile != null) {
178                     lastPath = selectedFile.getParent();
179                 }
180                 selectedFF = jFileChooser1.getFileFilter();
181             }
182             done = true;
183             if (Main.DISABLE_TRANSLUCENCY) {
184                 dispose();
185                 return;
186             }
187             setOpacity(1);
188             for (float i = 1; i >= 0; i -= 0.1) {
189                 setOpacity(i);
190                 try {
191                     Thread.sleep(20);
192                 } catch (InterruptedException ex) {
193                 }
194             }
195             setOpacity(0);
196             dispose();
197         }//GEN-LAST:event_jFileChooser1ActionPerformed

226         public static Object[] show(FileFilter[] ff, boolean openDialog) {
227             FileChooser dialog = new FileChooser(new javax.swing.JFrame(), true, ff, new Point(Toolkit.getDefaultToolkit().getScreenSize().width / 2, Toolkit.getDefaultToolkit().getScreenSize().height / 2), openDialog);
228             dialog.setVisible(true);
229             while (!dialog.done) {
230                 try {
231                     Thread.sleep(100);
232                 } catch (InterruptedException ex) {
233                     Logger.getLogger(FileChooser.class.getName()).log(Level.SEVERE, null, ex);
234                 }
235             }
236             return new Object[]{dialog.selectedFile, dialog.selectedFF};
237         }
