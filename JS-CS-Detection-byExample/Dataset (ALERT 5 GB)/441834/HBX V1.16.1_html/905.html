38         public RenderWindow(java.awt.Frame parent, boolean modal, final BinauralEnvelope be, final OutputStream os, Point location) {
39             super(parent, modal);
40             initComponents();
41             setLocation(location.x - getWidth() / 2, location.y - getHeight() / 2);
42             if (!Main.DISABLE_TRANSLUCENCY) {
43                 setOpacity(0);
44             }
45             new Thread() {
46                 public void run() {
47                     try {
48                         BinauralEnvelopePlayer bep=PresetCodec.toWAV(be, os);
49                         //"interface" with actual rendering thread to check progress
50                         while (bep.getPosition() < 1) {
51                             jProgressBar1.setValue((int) (bep.getPosition() * 100));
52                             try {
53                                 Thread.sleep(100);
54                             } catch (InterruptedException ex) {
55                                 Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
56                             }
57                             if (stopRequired) {
58                                 bep.stopPlaying();
59                                 try {
60                                     bep.join();
61                                 } catch (InterruptedException ex) {
62                                     Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
63                                 }
64                                 break;
65                             }
66                         }
67                         //render complete, kill BinauralEnvelopePlayer and close file
68                         try {
69                             bep.join();
70                         } catch (InterruptedException ex) {
71                             Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
72                         }
73                         try {
74                             os.close();
75                         } catch (IOException ex) {
76                         }
77                         jProgressBar1.setValue(100);
78                         cancel.setText(close);
79                         message.setText(sFinished);
80                         finished = true;
81                     } catch (Exception ex) {
82                         Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
83                     }
84                 }
85             }.start();
86         }

229         private void cancelMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_cancelMouseClicked
230             if (!finished) {
231                 stopRequired = true;
232             }
233             if (Main.DISABLE_TRANSLUCENCY) {
234                 dispose();
235                 return;
236             }
237             setOpacity(1);
238             for (float i = 1; i >= 0; i -= 0.1) {
239                 setOpacity(i);
240                 try {
241                     Thread.sleep(20);
242                 } catch (InterruptedException ex) {
243                 }
244             }
245             setOpacity(0);
246             dispose();
247         }//GEN-LAST:event_cancelMouseClicked
