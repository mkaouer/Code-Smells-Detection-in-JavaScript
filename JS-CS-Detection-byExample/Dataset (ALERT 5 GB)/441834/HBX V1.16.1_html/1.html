311         //THIS THREAD DOES ALL THE ANIMATIONS THAT INVOLVE RESIZING THE WINDOW
312         private class AnimationThread extends Thread {
313     
314             public double desiredW = defW, desiredH = defH; //desired width and height (can be set to any value)
315             private long f = 0; //current frame
316     
317             @Override
318             public void run() {
319                 setPriority(MIN_PRIORITY);
320                 double currentW = getWidth(), currentH = getHeight();
321                 for (;;) {
322                     long t = System.nanoTime();
323                     currentW = currentW * 0.8 + desiredW * 0.2;
324                     currentH = currentH * 0.8 + desiredH * 0.2;
325                     Toolkit.getDefaultToolkit().sync();
326                     setSize((int) currentW, (int) currentH);
327                     f++;
328                     //MAX SPEED 50 FPS
329                     try {
330                         long diff = (System.nanoTime() - t) / 1000000L;
331                         if (diff < 20) {
332                             Thread.sleep(20 - diff);
333                         }
334                     } catch (InterruptedException ex) {
335                     }
336                 }
337             }
338             //waits for n frames
339     
340             public void waitFrames(int n) {
341                 buttonsDisabled = true;
342                 long startF = f;
343                 while (f - startF < n) {
344                     try {
345                         sleep(1);
346                     } catch (InterruptedException ex) {
347                     }
348                 }
349                 buttonsDisabled = false;
350             }
351         }

353         private void fadeIn() {
354             buttonsDisabled = true;
355             for (float i = 0; i <= 1; i += 0.1) {
356                 long t = System.nanoTime();
357                 setOpacity(i);
358                 try {
359                     long diff = (System.nanoTime() - t) / 1000000L;
360                     if (diff < 20) {
361                         Thread.sleep(20 - diff);
362                     }
363                 } catch (InterruptedException ex) {
364                 }
365             }
366             setOpacity(1);
367             buttonsDisabled = false;
368         }
369     
370         private void fadeOut() {
371             buttonsDisabled = true;
372             for (float i = 1; i >= 0; i -= 0.1) {
373                 long t = System.nanoTime();
374                 setOpacity(i);
375                 try {
376                     long diff = (System.nanoTime() - t) / 1000000L;
377                     if (diff < 20) {
378                         Thread.sleep(20 - diff);
379                     }
380                 } catch (InterruptedException ex) {
381                 }
382             }
383             setOpacity(0);
384             buttonsDisabled = false;
385         }
386     
387         private void minimize() {
388             buttonsDisabled = true;
389             Point locationBeforeMinimizing = getLocation();
390             for (float i = 1; i >= 0; i -= 0.05) {
391                 long t = System.nanoTime();
392                 setOpacity(i);
393                 setLocation((int) (getLocation().x * 0.9), (int) (getLocation().y * 0.9 + Toolkit.getDefaultToolkit().getScreenSize().height * 0.1));
394                 try {
395                     long diff = (System.nanoTime() - t) / 1000000L;
396                     if (diff < 10) {
397                         Thread.sleep(10 - diff);
398                     }
399                 } catch (InterruptedException ex) {
400                 }
401             }
402             setOpacity(0);
403             setLocation(locationBeforeMinimizing);
404             buttonsDisabled = false;
405         }
406     
407         private void restore() {
408             buttonsDisabled = true;
409             Point newLocation = getLocation();
410             setLocation(0, Toolkit.getDefaultToolkit().getScreenSize().height);
411             setOpacity(0);
412             for (float i = 0; i <= 1; i += 0.05) {
413                 long t = System.nanoTime();
414                 setOpacity(i);
415                 setLocation((int) (getLocation().x * 0.75 + newLocation.x * 0.25), (int) (getLocation().y * 0.75 + newLocation.y * 0.25));
416                 try {
417                     long diff = (System.nanoTime() - t) / 1000000L;
418                     if (diff < 10) {
419                         Thread.sleep(10 - diff);
420                     }
421                 } catch (InterruptedException ex) {
422                 }
423             }
424             setOpacity(1);
425             buttonsDisabled = false;
426         }

430             for (Component c : x.getComponents()) {
431                 if (c instanceof IDeactivable) {
432                     ((IDeactivable) c).deactivate();
433                     remove(c);
434                 } else if (c instanceof Container) {
435                     destroyExpansionPanels((Container) c);
436                 }
437             }

1017         //returns true if there is a working sound card, false otherwise
1018         private static boolean checkSoundCard() {
1019             try {
1020                 AudioFormat af = new AudioFormat(44100f, 16, 2, true, false);
1021                 DataLine.Info info = new DataLine.Info(SourceDataLine.class, af);
1022                 SourceDataLine speaker = (SourceDataLine) AudioSystem.getLine(info);
1023                 speaker.open(af, 8192);
1024                 speaker.start();
1025                 byte[] testTone = new byte[32768];
1026                 double t = 0;
1027                 for (int i = 0; i < testTone.length / 4; i++) {
1028                     short s = (short) (Short.MAX_VALUE * 0.5 * Math.sin(Math.PI * 2 * t * 50));
1029                     testTone[4 * i] = (byte) s;
1030                     testTone[4 * i + 1] = (byte) (s >> 8);
1031                     testTone[4 * i + 2] = (byte) s;
1032                     testTone[4 * i + 3] = (byte) (s >> 8);
1033                     t += 1.0 / 44100.0;
1034                 }
1035                 speaker.write(testTone, 0, testTone.length);
1036                 speaker.flush();
1037                 speaker.close();
1038                 return true;
1039             } catch (Throwable t) {
1040                 return false;
1041             }
1042         }

1053         /**
1054          * @param args the command line arguments
1055          */
1056         public static void main(String args[]) {
1057             try {
1058                 if (args.length == 0) {
1059                     //START NORMALLY
1060                     if (checkSoundCard()) {
1061                         if (checkScreen()) {
1062                             new Main();
1063                         } else {
1064                             MessageBox.error(resErr, errTitle);
1065                             System.exit(10);
1066                         }
1067                     } else {
1068                         MessageBox.error(scErr, errTitle);
1069                         System.exit(11);
1070                     }
1071                 } else if (args.length == 1) {
1072                     if (args[0].equalsIgnoreCase("--help")) {
1073                         showHelp();
1074                         System.exit(0);
1075                     }
1076                     //PLAY PRESET PASSED FROM COMMAND LINE
1077                     if (checkSoundCard()) {
1078                         if (checkScreen()) {
1079                             new Main(args[0]);
1080                         } else {
1081                             MessageBox.error(resErr, errTitle);
1082                             System.exit(10);
1083                         }
1084                     } else {
1085                         MessageBox.error(scErr, errTitle);
1086                         System.exit(11);
1087                     }
1088                 } else {
1089                     if ((args.length == 3 || args.length == 4) && args[0].equalsIgnoreCase("--convert")) {
1090                         //CONVERSION MODE (COMMAND LINE)
1091                         String dstFormat = args[1];
1092                         if (!(dstFormat.equalsIgnoreCase("HBX") || dstFormat.equalsIgnoreCase("HBS") || dstFormat.equalsIgnoreCase("HBL") || dstFormat.equalsIgnoreCase("WAV"))) {
1093                             System.out.println("Unknown destination format " + dstFormat);
1094                             System.exit(1);
1095                         }
1096                         String sourceFile = args[2];
1097                         File s = new File(sourceFile);
1098                         String destFile = args.length == 4 ? args[3] : (args[2] + "." + dstFormat);
1099                         File d = new File(destFile);
1100                         try {
1101                             BinauralEnvelope source = PresetCodec.loadPreset(s);
1102                             FileOutputStream os = null;
1103                             try {
1104                                 os = new FileOutputStream(d);
1105                             } catch (Throwable t) {
1106                                 System.out.println("Can't write destination file");
1107                                 System.exit(2);
1108                             }
1109                             try {
1110                                 //<editor-fold defaultstate="collapsed" desc="HBX Conversion">
1111                                 if (dstFormat.equalsIgnoreCase("HBX")) {
1112                                     PresetCodec.toHBX(source, d);
1113                                 }
1114                                 //</editor-fold>
1115                                 //<editor-fold defaultstate="collapsed" desc="HBS Conversion">
1116                                 if (dstFormat.equalsIgnoreCase("HBS")) {
1117                                     PresetCodec.toHBS(source, d);
1118                                 }
1119                                 //</editor-fold>
1120                                 //<editor-fold defaultstate="collapsed" desc="HBL Conversion">
1121                                 if (dstFormat.equalsIgnoreCase("HBL")) {
1122                                     PresetCodec.toHBL(source, d);
1123                                 }
1124                                 //</editor-fold>
1125                                 //<editor-fold defaultstate="collapsed" desc="WAV Conversion">
1126                                 if (dstFormat.equalsIgnoreCase("WAV")) {
1127                                     try {
1128                                         BinauralEnvelopePlayer bep = PresetCodec.toWAV(source, os);
1129                                         //"interface" with actual rendering thread to check progress
1130                                         while (bep.getPosition() < 1) {
1131                                             System.out.println("Rendering... " + bep.getPosition() * 100 + "%");
1132                                             try {
1133                                                 Thread.sleep(500);
1134                                             } catch (InterruptedException ex) {
1135                                                 Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);
1136                                             }
1137                                         }
1138                                         //render complete, kill BinauralEnvelopePlayer and close file
1139                                         try {
1140                                             bep.join();
1141                                         } catch (InterruptedException ex) {
1142                                             Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);
1143                                         }
1144                                         try {
1145                                             os.close();
1146                                         } catch (IOException ex) {
1147                                         }
1148                                     } catch (Exception ex) {
1149                                         Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);
1150                                     }
1151                                 }
1152                                 //</editor-fold>
1153                                 System.out.println("Operation completed");
1154                                 System.exit(0);
1155                             } catch (Throwable t) {
1156                                 System.out.println("Conversion error");
1157                                 System.exit(4);
1158                             }
1159                         } catch (Throwable t) {
1160                             System.out.println("Invalid source file");
1161                             System.exit(3);
1162                         }
1163                     } else {
1164                         showHelp();
1165                         System.exit(5);
1166                     }
1167                 }
1168             } catch (Throwable e) {
1169                 JOptionPane.showMessageDialog(new JOptionPane(), "Couldn't start application.\nPlease make sure that you have Java 7 or newer installed and that this .jar file is corrupt\n\nPress OK to exit.", "HBX", JOptionPane.ERROR_MESSAGE);
1170                 System.exit(12);
1171             }
