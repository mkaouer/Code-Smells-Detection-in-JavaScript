159         private void jFileChooser1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jFileChooser1ActionPerformed
160             selectedFile = jFileChooser1.getSelectedFile();
161             selectedFF = jFileChooser1.getFileFilter();
162             done = true;
163             if (Main.DISABLE_TRANSLUCENCY) {
164                 dispose();
165                 return;
166             }
167             setOpacity(1);
168             for (float i = 1; i >= 0; i -= 0.1) {
169                 setOpacity(i);
170                 try {
171                     Thread.sleep(20);
172                 } catch (InterruptedException ex) {
173                 }
174             }
175             setOpacity(0);
176             dispose();
177         }//GEN-LAST:event_jFileChooser1ActionPerformed

206         public static Object[] show(FileFilter[] ff, boolean openDialog) {
207             FileChooser dialog = new FileChooser(new javax.swing.JFrame(), true, ff, new Point(Toolkit.getDefaultToolkit().getScreenSize().width / 2, Toolkit.getDefaultToolkit().getScreenSize().height / 2), openDialog);
208             dialog.setVisible(true);
209             while (!dialog.done) {
210                 try {
211                     Thread.sleep(100);
212                 } catch (InterruptedException ex) {
213                     Logger.getLogger(FileChooser.class.getName()).log(Level.SEVERE, null, ex);
214                 }
215             }
216             return new Object[]{dialog.selectedFile, dialog.selectedFF};
217         }
