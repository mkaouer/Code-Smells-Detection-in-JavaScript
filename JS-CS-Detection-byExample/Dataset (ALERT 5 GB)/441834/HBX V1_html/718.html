233         //THIS THREAD DOES ALL THE ANIMATIONS THAT INVOLVE RESIZING THE WINDOW
234         private class AnimationThread extends Thread {
235     
236             public double desiredW = defW, desiredH = defH; //desired width and height (can be set to any value)
237             private long f = 0; //current frame
238     
239             @Override
240             public void run() {
241                 setPriority(MIN_PRIORITY);
242                 double currentW = getWidth(), currentH = getHeight();
243                 for (;;) {
244                     long t = System.nanoTime();
245                     currentW = currentW * 0.8 + desiredW * 0.2;
246                     currentH = currentH * 0.8 + desiredH * 0.2;
247                     Toolkit.getDefaultToolkit().sync();
248                     setSize((int) currentW, (int) currentH);
249                     f++;
250                     //MAX SPEED 50 FPS
251                     try {
252                         long diff = (System.nanoTime() - t) / 1000000L;
253                         if (diff < 20) {
254                             Thread.sleep(20 - diff);
255                         }
256                     } catch (InterruptedException ex) {
257                     }
258                 }
259             }
260             //waits for n frames
261     
262             public void waitFrames(int n) {
263                 buttonsDisabled = true;
264                 long startF = f;
265                 while (f - startF < n) {
266                     try {
267                         sleep(1);
268                     } catch (InterruptedException ex) {
269                     }
270                 }
271                 buttonsDisabled = false;
272             }
273         }

275         private void fadeIn() {
276             buttonsDisabled = true;
277             for (float i = 0; i <= 1; i += 0.1) {
278                 long t = System.nanoTime();
279                 setOpacity(i);
280                 try {
281                     long diff = (System.nanoTime() - t) / 1000000L;
282                     if (diff < 20) {
283                         Thread.sleep(20 - diff);
284                     }
285                 } catch (InterruptedException ex) {
286                 }
287             }
288             setOpacity(1);
289             buttonsDisabled = false;
290         }
291     
292         private void fadeOut() {
293             buttonsDisabled = true;
294             for (float i = 1; i >= 0; i -= 0.1) {
295                 long t = System.nanoTime();
296                 setOpacity(i);
297                 try {
298                     long diff = (System.nanoTime() - t) / 1000000L;
299                     if (diff < 20) {
300                         Thread.sleep(20 - diff);
301                     }
302                 } catch (InterruptedException ex) {
303                 }
304             }
305             setOpacity(0);
306             buttonsDisabled = false;
307         }
308     
309         private void minimize() {
310             buttonsDisabled = true;
311             Point locationBeforeMinimizing = getLocation();
312             for (float i = 1; i >= 0; i -= 0.05) {
313                 long t = System.nanoTime();
314                 setOpacity(i);
315                 setLocation((int) (getLocation().x * 0.9), (int) (getLocation().y * 0.9 + Toolkit.getDefaultToolkit().getScreenSize().height * 0.1));
316                 try {
317                     long diff = (System.nanoTime() - t) / 1000000L;
318                     if (diff < 10) {
319                         Thread.sleep(10 - diff);
320                     }
321                 } catch (InterruptedException ex) {
322                 }
323             }
324             setOpacity(0);
325             setLocation(locationBeforeMinimizing);
326             buttonsDisabled = false;
327         }
328     
329         private void restore() {
330             buttonsDisabled = true;
331             Point newLocation = getLocation();
332             setLocation(0, Toolkit.getDefaultToolkit().getScreenSize().height);
333             setOpacity(0);
334             for (float i = 0; i <= 1; i += 0.05) {
335                 long t = System.nanoTime();
336                 setOpacity(i);
337                 setLocation((int) (getLocation().x * 0.75 + newLocation.x * 0.25), (int) (getLocation().y * 0.75 + newLocation.y * 0.25));
338                 try {
339                     long diff = (System.nanoTime() - t) / 1000000L;
340                     if (diff < 10) {
341                         Thread.sleep(10 - diff);
342                     }
343                 } catch (InterruptedException ex) {
344                 }
345             }
346             setOpacity(1);
347             buttonsDisabled = false;
348         }

351         private void destroyExpansionPanels(Container x) {
352             for (Component c : x.getComponents()) {
353                 if (c instanceof IDeactivable) {
354                     ((IDeactivable) c).deactivate();
355                     remove(c);
356                 } else if (c instanceof Container) {
357                     destroyExpansionPanels((Container) c);
358                 }
359             }
360         }

362         private void createPlayerPanel(BinauralEnvelope be) {
363             destroyExpansionPanels(rootPane);
364             PlayerPanel pp = new PlayerPanel(be) {
365                 @Override
366                 public void onDeactivation() {
367                     animo.desiredH = defH;
368                     animo.waitFrames(30);
369                 }
370             };
371             pp.setBounds(0, 78, 650, 130);
372             add(pp);
373             animo.desiredH = defH + pp.getHeight();
374             animo.waitFrames(30);
375         }

377         private void createEditorPanel(BinauralEnvelope be) {
378             inEditor = true;
379             destroyExpansionPanels(rootPane);
380             EditorPanel pp = new EditorPanel(be) {
381                 @Override
382                 public void onDeactivation() {
383                     animo.desiredH = defH;
384                     animo.waitFrames(30);
385                     inEditor = false;
386                     if (isTestRequired()) {
387                         lastOpened = getTestEnvelope();
388                         new Thread() {
389                             public void run() {
390                                 createPlayerPanel(lastOpened);
391                             }
392                         }.start();
393                     }
394                 }
395             };
396             pp.setBounds(0, 78, 650, 613);
397             add(pp);
398             animo.desiredH = defH + pp.getHeight();
399             animo.waitFrames(30);
400         }

757         private void loadInPlayer(File x) throws FileNotFoundException, IOException, Exception {
758             if (x == null) {
759                 return;
760             }
761             if (x.getName().toLowerCase().endsWith(".hbl")) {
762                 BufferedReader xmlFile=new BufferedReader(new FileReader(x));
763                 String xml="";
764                 for(;;){
765                     try {
766                         String line=xmlFile.readLine();
767                         if(line==null) break; else xml += line + "\n";
768                     } catch (IOException e) {
769                         break;
770                     }
771                 }
772                 xmlFile.close();
773                 BinauralEnvelope be=BinauralEnvelope.fromXML(xml);
774                 lastOpened=be;
775                 createPlayerPanel(be);
776             } else {
777                 FileInputStream fis = new FileInputStream(x);
778                 boolean hbx = true, hbs = true;
779                 byte[] headerX = new byte[HBX_HEADER.length];
780                 byte[] headerS = new byte[HBS_HEADER.length];
781                 fis.read(headerX);
782                 fis.close();
783                 fis = new FileInputStream(x);
784                 fis.read(headerS);
785                 for (int i = 0; i < headerX.length; i++) {
786                     if (headerX[i] != HBX_HEADER[i]) {
787                         hbx = false;
788                     }
789                 }
790                 for (int i = 0; i < headerS.length; i++) {
791                     if (headerS[i] != HBS_HEADER[i]) {
792                         hbs = false;
793                     }
794                 }
795                 fis.close();
796                 fis = new FileInputStream(x);
797                 if (!hbx && !hbs) {
798                     throw new Exception("Invalid file");
799                 }
800                 if (hbx) {
801                     fis.read(headerX);
802                     ObjectInputStream ois = new ObjectInputStream(new GZIPInputStream(fis));
803                     BinauralEnvelope be = (BinauralEnvelope) (ois.readObject());
804                     lastOpened = be;
805                     createPlayerPanel(be);
806                     ois.close();
807                 }
808                 if (hbs) {
809                     fis.read(headerS);
810                     ByteArrayOutputStream baos = new ByteArrayOutputStream();
811                     for (;;) {
812                         int in = fis.read();
813                         if (in == -1) {
814                             break;
815                         } else {
816                             baos.write(in);
817                         }
818                     }
819                     BinauralEnvelope be = SmallSave.fromHES(baos.toByteArray());
820                     lastOpened = be;
821                     createPlayerPanel(be);
822                     fis.close();
823                 }
824             }
825         }

954         //headers (magic numbers actually) of HBX and HBS files are calculated from 2 simple strings (HBX and HBS respectively)
955         public static final byte[] HBX_HEADER, HBS_HEADER;
956     
957         static {
958             String h = "HBX";
959             HBX_HEADER = new byte[h.length()];
960             char[] arr = h.toCharArray();
961             for (int i = 0; i < arr.length; i++) {
962                 HBX_HEADER[i] = (byte) arr[i];
963             }
964             h = "HBS";
965             HBS_HEADER = new byte[h.length()];
966             arr = h.toCharArray();
967             for (int i = 0; i < arr.length; i++) {
968                 HBS_HEADER[i] = (byte) arr[i];
969             }
970         }
971         //returns true if there is a working sound card, false otherwise
972     
973         private static boolean checkSoundCard() {
974             try {
975                 AudioFormat af = new AudioFormat(44100f, 16, 2, true, false);
976                 DataLine.Info info = new DataLine.Info(SourceDataLine.class, af);
977                 SourceDataLine speaker = (SourceDataLine) AudioSystem.getLine(info);
978                 speaker.open(af, 8192);
979                 speaker.start();
980                 byte[] testTone = new byte[32768];
981                 double t = 0;
982                 for (int i = 0; i < testTone.length / 4; i++) {
983                     short s = (short) (Short.MAX_VALUE * 0.5 * Math.sin(Math.PI * 2 * t * 50));
984                     testTone[4 * i] = (byte) s;
985                     testTone[4 * i + 1] = (byte) (s >> 8);
986                     testTone[4 * i + 2] = (byte) s;
987                     testTone[4 * i + 3] = (byte) (s >> 8);
988                     t += 1.0 / 44100.0;
989                 }
990                 speaker.write(testTone, 0, testTone.length);
991                 speaker.flush();
992                 speaker.close();
993                 return true;
994             } catch (Throwable t) {
995                 return false;
996             }
997         }

1000             Dimension res = Toolkit.getDefaultToolkit().getScreenSize();
