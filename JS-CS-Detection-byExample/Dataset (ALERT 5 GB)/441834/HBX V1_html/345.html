34         private static void writeS(OutputStream out, String s) {
35             for (int i = 0; i < s.length(); i++) {
36                 try {
37                     out.write(s.charAt(i));
38                 } catch (IOException ex) {
39                     Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
40                 }
41             }
42         }
43     //WAV HEADER STUFF
44         private static void writeInt(OutputStream out, int val) {
45             try {
46                 out.write(val);
47                 out.write(val >> 8);
48                 out.write(val >> 16);
49                 out.write(val >> 24);
50             } catch (IOException ex) {
51                 Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
52             }
53     
54         }
55     //WAV HEADER STUFF
56         private static void writeShort(OutputStream out, short val) {
57             try {
58                 out.write(val);
59                 out.write(val >> 8);
60             } catch (IOException ex) {
61                 Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
62             }
63         }

68         public RenderWindow(java.awt.Frame parent, boolean modal, final BinauralEnvelope be, final OutputStream os, Point location) {
69             super(parent, modal);
70             initComponents();
71             setLocation(location.x - getWidth() / 2, location.y - getHeight() / 2);
72             if (!Main.DISABLE_TRANSLUCENCY) {
73                 setOpacity(0);
74             }
75             new Thread() {
76                 public void run() {
77                     //starts the BinauralEnvelopePlayer with the file as OutputStream
78                     BinauralEnvelopePlayer bep = new BinauralEnvelopePlayer(be, os);
79                     bep.start();
80                     jProgressBar1.setValue(0);
81                     //wav header stuff
82                     writeS(os, "RIFF");
83                     writeInt(os, (int) (Math.ceil(36 + be.getLength() * 44100 * 4)));
84                     writeS(os, "WAVE");
85                     writeS(os, "fmt ");
86                     writeInt(os, 16);
87                     writeShort(os, (short) 1);
88                     writeShort(os, (short) 2);
89                     writeInt(os, 44100);
90                     writeInt(os, 2 * 44100 * 16 / 8);
91                     writeShort(os, (short) (2 * 16 / 8));
92                     writeShort(os, (short) 16);
93                     writeS(os, "data");
94                     writeInt(os, (int) (Math.ceil(be.getLength() * 44100 * 4)));
95                     //"interface" with actual rendering thread to check progress
96                     while (bep.getPosition() < 1) {
97                         jProgressBar1.setValue((int) (bep.getPosition() * 100));
98                         try {
99                             Thread.sleep(100);
100                         } catch (InterruptedException ex) {
101                             Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
102                         }
103                         if (stopRequired) {
104                             bep.stopPlaying();
105                             try {
106                                 bep.join();
107                             } catch (InterruptedException ex) {
108                                 Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
109                             }
110                             break;
111                         }
112                     }
113                     //render complete, kill BinauralEnvelopePlayer and close file
114                     try {
115                         bep.join();
116                     } catch (InterruptedException ex) {
117                         Logger.getLogger(RenderWindow.class.getName()).log(Level.SEVERE, null, ex);
118                     }
119                     try {
120                         os.close();
121                     } catch (IOException ex) {
122                     }
123                     jProgressBar1.setValue(100);
124                     cancel.setText(close);
125                     message.setText(sFinished);
126                     finished = true;
127                 }
128             }.start();
129         }

272         private void cancelMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_cancelMouseClicked
273             if (!finished) {
274                 stopRequired = true;
275             }
276             if (Main.DISABLE_TRANSLUCENCY) {
277                 dispose();
278                 return;
279             }
280             setOpacity(1);
281             for (float i = 1; i >= 0; i -= 0.1) {
282                 setOpacity(i);
283                 try {
284                     Thread.sleep(20);
285                 } catch (InterruptedException ex) {
286                 }
287             }
288             setOpacity(0);
289             dispose();
290         }//GEN-LAST:event_cancelMouseClicked
